42   object LoaderStatus {
43 
44     private val loading = Collections.newSetFromMap(new ConcurrentHashMap[AnyRef, java.lang.Boolean]())
45     private val firstLoadStartTime = new AtomicLong(0L)
46 
47     def startLoad(loader: AnyRef): Boolean = synchronized {
48       if (!loading.add(loader)) {
49         logger.warn(s"Called startLoad for a loader that was already registered and has not yet completed: $loader")
50       }
51       firstLoadStartTime.compareAndSet(0L, System.currentTimeMillis())
52     }
53 
54     def completedLoad(loader: AnyRef): Unit = synchronized {
55       if (!loading.remove(loader)) {
56         logger.warn(s"Called completedLoad for a loader that was not registered or already deregistered: $loader")
57       } else if (loading.isEmpty) {
58         logger.info(s"Last active initial load completed.  " +
59           s"Initial loads took ${System.currentTimeMillis() - firstLoadStartTime.get} milliseconds.")
60         firstLoadStartTime.set(0L)
61       }
62     }
63 
64     def allLoaded(): Boolean = loading.isEmpty
65   }

72   class KafkaCacheLoaderImpl(
73       sft: SimpleFeatureType,
74       override val cache: KafkaFeatureCache,
75       consumers: Seq[Consumer[Array[Byte], Array[Byte]]],
76       topic: String,
77       frequency: Long,
78       serializer: GeoMessageSerializer,
79       doInitialLoad: Boolean,
80       initialLoadConfig: ExpiryTimeConfig
81     ) extends ThreadedConsumer(consumers, Duration.ofMillis(frequency)) with KafkaCacheLoader {
82     try { classOf[ConsumerRecord[Any, Any]].getMethod("timestamp") } catch {
83       case _: NoSuchMethodException => logger.warn("This version of Kafka doesn't support timestamps, using system time")
84     }
85 
86     private val initialLoader =
87       if (doInitialLoad) {
88         // for the initial load, don't bother spatially indexing until we have the final state
89         Some(new InitialLoader(sft, consumers, topic, frequency, serializer, initialLoadConfig, this))
90       } else {
91         None
92       }
93 
94     def start(): Unit = {
95       initialLoader match {
96         case None => startConsumers()
97         case Some(loader) => loader.start()
98       }
99     }
100 
101     override def close(): Unit = {
102       try { super.close() } finally {
103         CloseWithLogging(initialLoader)
104         cache.close()
105       }
106     }
107 
108     override protected [KafkaCacheLoader] def consume(record: ConsumerRecord[Array[Byte], Array[Byte]]): Unit = {
109       val headers = RecordVersions.getHeaders(record)
110       val timestamp = RecordVersions.getTimestamp(record)
111       val message = serializer.deserialize(record.key(), record.value(), headers, timestamp)
112       logger.trace(s"Consumed message [$topic:${record.partition}:${record.offset}] $message")
113       message match {
114         case m: Change => cache.fireChange(timestamp, m.feature); cache.put(m.feature)
115         case m: Delete => cache.fireDelete(timestamp, m.id, cache.query(m.id).orNull); cache.remove(m.id)
116         case _: Clear  => cache.fireClear(timestamp); cache.clear()
117         case m => throw new IllegalArgumentException(s"Unknown message: $m")
118       }
119     }
120   }

122   /**
123     * Handles initial loaded 'from-beginning' without indexing features in the spatial index. Will still
124     * trigger message events.
125     *
126     * @param consumers consumers, won't be closed even on call to 'close()'
127     * @param topic kafka topic
128     * @param frequency polling frequency in milliseconds
129     * @param serializer message serializer
130     * @param toLoad main cache loader, used for callback when bulk loading is done
131     */
132   private class InitialLoader(
133       sft: SimpleFeatureType,
134       consumers: Seq[Consumer[Array[Byte], Array[Byte]]],
135       topic: String,
136       frequency: Long,
137       serializer: GeoMessageSerializer,
138       ordering: ExpiryTimeConfig,
139       toLoad: KafkaCacheLoaderImpl
140     ) extends ThreadedConsumer(consumers, Duration.ofMillis(frequency), false) with Runnable {
141 
142     import scala.collection.JavaConverters._
143 
144     private val cache = KafkaFeatureCache.nonIndexing(sft, ordering)
145 
146     // track the offsets that we want to read to
147     private val offsets = new ConcurrentHashMap[Int, Long]()
148     private val done = new AtomicBoolean(false)
149     private var latch: CountDownLatch = _
150     @volatile
151     private var submission: Future[_] = _
152 
153     override protected def createConsumerRunnable(
154         id: String,
155         consumer: Consumer[Array[Byte], Array[Byte]],
156         handler: ConsumerErrorHandler): Runnable = {
157       new InitialLoaderConsumerRunnable(id, consumer, handler)
158     }
159 
160     override protected def consume(record: ConsumerRecord[Array[Byte], Array[Byte]]): Unit = {
161       if (done.get) { toLoad.consume(record) } else {
162         val headers = RecordVersions.getHeaders(record)
163         val timestamp = RecordVersions.getTimestamp(record)
164         val message = serializer.deserialize(record.key, record.value, headers, timestamp)
165         logger.trace(s"Consumed message [$topic:${record.partition}:${record.offset}] $message")
166         message match {
167           case m: Change => toLoad.cache.fireChange(timestamp, m.feature); cache.put(m.feature)
168           case m: Delete => toLoad.cache.fireDelete(timestamp, m.id, cache.query(m.id).orNull); cache.remove(m.id)
169           case _: Clear  => toLoad.cache.fireClear(timestamp); cache.clear()
170           case m => throw new IllegalArgumentException(s"Unknown message: $m")
171         }
172         if (record.offset > 0 && record.offset % 1048576 == 0) { // magic number 2^20
173           logger.info(s"Initial load: consumed [$topic:${record.partition}:${record.offset}]")
174         }
175       }
176     }
177 
178     def start(): Unit = {
179       LoaderStatus.startLoad(this)
180       try {
181         val partitions = consumers.head.partitionsFor(topic).asScala.map(_.partition)
182         val doInitialLoad =
183           try {
184             // note: these methods are not available in kafka 0.9, which will cause it to fall back to normal loading
185             val beginningOffsets = KafkaConsumerVersions.beginningOffsets(consumers.head, topic, partitions.toSeq)
186             val endOffsets = KafkaConsumerVersions.endOffsets(consumers.head, topic, partitions.toSeq)
187             partitions.exists(p => beginningOffsets.getOrElse(p, 0L) < endOffsets.getOrElse(p, 0L))
188           } catch {
189             case e: NoSuchMethodException =>
190               logger.warn(s"Can't support initial bulk loading for current Kafka version: $e")
191               false
192           }
193         if (doInitialLoad) {
194           logger.info(s"Starting initial load for [$topic] with ${partitions.size} partitions")
195           latch = new CountDownLatch(partitions.size)
196           startConsumers() // kick off the asynchronous consumer threads
197           submission = CachedThreadPool.submit(this)
198         } else {
199           // don't bother spinning up the consumer threads if we don't need to actually bulk load anything
200           startNormalLoad()
201           LoaderStatus.completedLoad(this)
202         }
203       } catch {
204         case NonFatal(e) =>
205           LoaderStatus.completedLoad(this)
206           throw e
207       }
208     }
209 
210     override def run(): Unit = {
211       try {
212         try { latch.await() } finally {
213           // stop the consumer threads, but won't close the consumers due to `closeConsumers`
214           // note: don't call this.close() as it would interrupt this thread
215           super.close()
216         }
217         // set a flag just in case the consumer threads haven't finished spinning down, so that we will
218         // pass any additional messages back to the main loader
219         done.set(true)
220         logger.info(s"Finished initial load, transferring to indexed cache for [$topic]")
221         cache.query(Filter.INCLUDE).foreach(toLoad.cache.put)
222         logger.info(s"Finished transfer for [$topic]")
223         startNormalLoad()
224       } finally {
225         LoaderStatus.completedLoad(this)
226       }
227     }
228 
229     override def close(): Unit = {
230       try { super.close() } finally {
231         if (submission != null && !submission.isDone) {
232           submission.cancel(true)
233         }
234       }
235     }
236     // start the normal loading
237     private def startNormalLoad(): Unit = {
238       logger.info(s"Starting normal load for [$topic]")
239       toLoad.startConsumers()
240     }
241 
242     /**
243      * Consumer runnable for the initial load. It's not possible to check assignments until after poll() has been called,
244      * and checking endOffsets is not reliable as a high water mark, due to transaction markers or other issues that
245      * may advance the endOffset past where the consumer can actually reach (unless new messages are written).
246      *
247      * @param id id
248      * @param consumer consumer
249      * @param handler error handler
250      */
251     private class InitialLoaderConsumerRunnable(id: String, consumer: Consumer[Array[Byte], Array[Byte]], handler: ConsumerErrorHandler)
252         extends ConsumerRunnable(id, consumer, handler) {
253 
254       private var checkOffsets = true
255 
256       override protected def processPoll(result: ConsumerRecords[Array[Byte], Array[Byte]]): Unit = {
257         if (checkOffsets) {
258           consumer.assignment().asScala.foreach { tp =>
259             // the only reliable way we've found to check max offset is to seek to the end and check the position there
260             val position = consumer.position(tp)
261             consumer.seekToEnd(Collections.singleton(tp))
262             val end = consumer.position(tp)
263             // return to the original position
264             consumer.seek(tp, position)
265             logger.debug(s"Setting max offset to [${tp.topic}:${tp.partition}:${end - 1}]")
266             offsets.put(tp.partition(), end - 1)
267           }
268           checkOffsets = false
269         }
270         try {
271           super.processPoll(result)
272         } finally {
273           result.partitions().asScala.foreach { tp =>
274             val position = consumer.position(tp)
275             // once we've hit the max offset for the partition, remove from the offset map so we don't double count it
276             if (position >= offsets.getOrDefault(tp.partition(), Long.MaxValue)) {
277               offsets.remove(tp.partition)
278               latch.countDown()
279               logger.info(s"Initial load: consumed [$topic:${tp.partition}:${position - 1}]")
280               logger.info(s"Initial load completed for [$topic:${tp.partition}], ${latch.getCount} partitions remaining")
281             }
282           }
283         }
284       }
285     }
286   }
