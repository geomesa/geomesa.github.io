47     private val firstLoadStartTime = new AtomicLong(0L)

49     def startLoad(loader: AnyRef): Boolean = synchronized {
50       if (!loading.add(loader)) {
51         logger.warn(s"Called startLoad for a loader that was already registered and has not yet completed: $loader")
52       }
53       firstLoadStartTime.compareAndSet(0L, System.currentTimeMillis())
54     }
55 
56     def completedLoad(loader: AnyRef): Unit = synchronized {
57       if (!loading.remove(loader)) {
58         logger.warn(s"Called completedLoad for a loader that was not registered or already deregistered: $loader")
59       } else if (loading.isEmpty) {
60         logger.info(s"Last active initial load completed.  " +
61           s"Initial loads took ${System.currentTimeMillis() - firstLoadStartTime.get} milliseconds.")
62         firstLoadStartTime.set(0L)
63       }
64     }
65 
66     def allLoaded(): Boolean = loading.isEmpty
67   }

112     override protected [KafkaCacheLoader] def consume(record: ConsumerRecord[Array[Byte], Array[Byte]]): Unit = {
113       val headers = RecordVersions.getHeaders(record)
114       val timestamp = RecordVersions.getTimestamp(record)
115       val message = serializer.deserialize(record.key(), record.value(), headers, timestamp)
116       logger.trace(s"Consumed message [$topic:${record.partition}:${record.offset}] $message")
117       message match {
118         case m: Change => cache.fireChange(timestamp, m.feature); cache.put(m.feature)
119         case m: Delete => cache.fireDelete(timestamp, m.id, cache.query(m.id).orNull); cache.remove(m.id)
120         case _: Clear  => cache.fireClear(timestamp); cache.clear()
121         case m => throw new IllegalArgumentException(s"Unknown message: $m")
122       }
123     }

126   /**
127    * Handles initial loaded 'from-beginning' without indexing features in the spatial index. Will still
128    * trigger message events.
129    *
130    * @param sft simple feature type
131    * @param consumers consumers, won't be closed even on call to 'close()'
132    * @param topic kafka topic
133    * @param frequency polling frequency
134    * @param serializer message serializer
135    * @param readBack initial load read back
136    * @param ordering feature ordering
137    * @param toLoad main cache loader, used for callback when bulk loading is done

139   private class InitialLoader(
140       sft: SimpleFeatureType,
141       consumers: Seq[Consumer[Array[Byte], Array[Byte]]],
142       topic: String,
143       frequency: Duration,
144       serializer: GeoMessageSerializer,
145       readBack: scala.concurrent.duration.Duration,
146       ordering: ExpiryTimeConfig,
147       toLoad: KafkaCacheLoaderImpl
148     ) extends ThreadedConsumer(consumers, frequency, false) with Runnable with ConsumerRebalanceListener {
149 
150     import scala.collection.JavaConverters._
151 
152     private val cache = KafkaFeatureCache.nonIndexing(sft, ordering)
153 
154     // track the offsets that we want to read to
155     private val offsets = new ConcurrentHashMap[Int, Long]()
156     private val assignment = Collections.newSetFromMap(new ConcurrentHashMap[Int, java.lang.Boolean]())
157     @volatile
158     private var done: Boolean = false
159     private var latch: CountDownLatch = _
160     @volatile
161     private var submission: Future[_] = _
162 
163     override def onPartitionsRevoked(topicPartitions: java.util.Collection[TopicPartition]): Unit = {}
164 
165     override def onPartitionsAssigned(topicPartitions: java.util.Collection[TopicPartition]): Unit = {
166       logger.debug(s"Partitions assigned: ${topicPartitions.asScala.mkString(", ")}")
167       topicPartitions.asScala.foreach { tp =>
168         if (assignment.add(tp.partition())) {
169           val consumer = consumers.find(_.assignment().contains(tp)).orNull
170           if (consumer == null) {
171             logger.warn("Partition assigned but no consumer contains the assignment")
172           } else {
173             KafkaConsumerVersions.pause(consumer, tp)
174             try {
175               logger.debug(s"Checking offsets for [${tp.topic()}:${tp.partition()}]")
176               // the only reliable way we've found to check max offset is to seek to the end and check the position there
177               consumer.seekToEnd(Collections.singleton(tp))
178               val end = consumer.position(tp)
179               logger.debug(s"Setting max offset to [${tp.topic}:${tp.partition}:${end - 1}]")
180               offsets.put(tp.partition(), end - 1)
181               if (!readBack.isFinite) {
182                 KafkaConsumerVersions.seekToBeginning(consumer, tp)
183               } else {
184                 val offset = Try {
185                   val time = System.currentTimeMillis() - readBack.toMillis
186                   KafkaConsumerVersions.offsetsForTimes(consumer, tp.topic, Seq(tp.partition), time).get(tp.partition)
187                 }
188                 offset match {
189                   case Success(Some(o)) =>
190                     logger.debug(s"Seeking to offset $o for read-back $readBack on [${tp.topic}:${tp.partition}]")
191                     consumer.seek(tp, o)
192 
193                   case Success(None) =>
194                     logger.debug(s"No prior offset found for read-back $readBack on [${tp.topic}:${tp.partition}], " +
195                       "reading from head of queue")
196 
197                   case Failure(e) =>
198                     logger.warn(s"Error finding initial offset: [${tp.topic}:${tp.partition}], seeking to beginning", e)
199                     KafkaConsumerVersions.seekToBeginning(consumer, tp)
200                 }
201               }
202             } finally {
203               KafkaConsumerVersions.resume(consumer, tp)
204             }
205           }
206         }
207       }
208     }

211         id: String,

213         handler: ConsumerErrorHandler): Runnable = {

217     override protected def consume(record: ConsumerRecord[Array[Byte], Array[Byte]]): Unit = {
218       if (done) { toLoad.consume(record) } else {
219         val headers = RecordVersions.getHeaders(record)
220         val timestamp = RecordVersions.getTimestamp(record)
221         val message = serializer.deserialize(record.key, record.value, headers, timestamp)
222         logger.trace(s"Consumed message [$topic:${record.partition}:${record.offset}] $message")
223         message match {
224           case m: Change => toLoad.cache.fireChange(timestamp, m.feature); cache.put(m.feature)
225           case m: Delete => toLoad.cache.fireDelete(timestamp, m.id, cache.query(m.id).orNull); cache.remove(m.id)
226           case _: Clear  => toLoad.cache.fireClear(timestamp); cache.clear()
227           case m => throw new IllegalArgumentException(s"Unknown message: $m")
228         }
229         if (record.offset > 0 && record.offset % 1048576 == 0) { // magic number 2^20
230           logger.info(s"Initial load: consumed [$topic:${record.partition}:${record.offset}]")
231         }
232       }
233     }
234 
235     def start(): Unit = {
236       LoaderStatus.startLoad(this)
237       try {
238         val partitions = consumers.head.partitionsFor(topic).asScala.map(_.partition)
239         logger.info(s"Starting initial load for [$topic] with ${partitions.size} partitions")
240         latch = new CountDownLatch(partitions.size)
241         startConsumers() // kick off the asynchronous consumer threads
242         submission = CachedThreadPool.submit(this)
243       } catch {
244         case NonFatal(e) =>
245           LoaderStatus.completedLoad(this)
246           throw e
247       }
248     }
249 
250     override def run(): Unit = {
251       try {
252         try { latch.await() } finally {
253           // stop the consumer threads, but won't close the consumers due to `closeConsumers`
254           // note: don't call this.close() as it would interrupt this thread
255           super.close()
256         }
257         // set a flag just in case the consumer threads haven't finished spinning down, so that we will
258         // pass any additional messages back to the main loader
259         done = true
260         logger.info(s"Finished initial load, transferring to indexed cache for [$topic]")
261         cache.query(Filter.INCLUDE).foreach(toLoad.cache.put)
262         logger.info(s"Finished transfer for [$topic], starting normal load")
263         toLoad.startConsumers()
264       } finally {
265         LoaderStatus.completedLoad(this)
266       }
267     }
268 
269     override def close(): Unit = {
270       try { super.close() } finally {
271         if (submission != null && !submission.isDone) {
272           submission.cancel(true)
273         }
274       }
275     }

286 
287       override protected def processPoll(result: ConsumerRecords[Array[Byte], Array[Byte]]): Unit = {
288         try {
289           super.processPoll(result)
290         } finally {
291           result.partitions().asScala.foreach { tp =>
292             val position = consumer.position(tp)
293             // once we've hit the max offset for the partition, remove from the offset map so we don't double count it
294             if (position >= offsets.getOrDefault(tp.partition(), Long.MaxValue)) {
295               offsets.remove(tp.partition)
296               latch.countDown()
297               logger.info(s"Initial load: consumed [$topic:${tp.partition}:${position - 1}]")
298               logger.info(s"Initial load completed for [$topic:${tp.partition}], ${latch.getCount} partitions remaining")
299             }
300           }
301         }
302       }
303     }
