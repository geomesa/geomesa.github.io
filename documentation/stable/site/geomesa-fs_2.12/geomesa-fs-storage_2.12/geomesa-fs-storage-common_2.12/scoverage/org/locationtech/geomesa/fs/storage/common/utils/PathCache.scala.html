22 /**
23   * Caches file statuses to avoid repeated file system operations. Status expires after a
24   * configurable period, by default 10 minutes.
25   */
26 object PathCache extends LazyLogging {
27 
28   val CacheDurationProperty: SystemProperty = SystemProperty("geomesa.fs.file.cache.duration", "15 minutes")
29 
30   private val duration = CacheDurationProperty.toDuration.get.toMillis
31 
32   // cache for checking existence of files
33   private val pathCache: LoadingCache[(FileSystem, Path), lang.Boolean] =
34     Caffeine.newBuilder().expireAfterWrite(duration, TimeUnit.MILLISECONDS).build(
35       new CacheLoader[(FileSystem, Path), java.lang.Boolean]() {
36         override def load(key: (FileSystem, Path)): java.lang.Boolean = key._1.exists(key._2)
37       }
38     )
39 
40   // cache for checking directory contents
41   private val listCache: LoadingCache[(FileSystem, Path), Stream[FileStatus]] =
42     Caffeine.newBuilder().expireAfterWrite(duration, TimeUnit.MILLISECONDS).build(
43       new CacheLoader[(FileSystem, Path), Stream[FileStatus]]() {
44         override def load(key: (FileSystem, Path)): Stream[FileStatus] =
45           RemoteIterator(key._1.listStatusIterator(key._2)).toStream
46       }
47     )
48 
49   // cache for individual file status
50   private val statusCache: LoadingCache[(FileSystem, Path), FileStatus] =
51     Caffeine.newBuilder().expireAfterWrite(duration, TimeUnit.MILLISECONDS).build(
52       new CacheLoader[(FileSystem, Path), FileStatus]() {
53         override def load(key: (FileSystem, Path)): FileStatus = {
54           try { key._1.getFileStatus(key._2) } catch {
55               case _: FileNotFoundException => null
56           }
57         }
58 
59         override def asyncLoad(key: (FileSystem, Path), executor: Executor): CompletableFuture[FileStatus] = {
60           super.asyncLoad(key, executor)
61             .whenCompleteAsync(new ListCacheRefresh(key), executor)
62             .asInstanceOf[CompletableFuture[FileStatus]]
63         }
64 
65         override def asyncReload(
66             key: (FileSystem, Path),
67             oldValue: FileStatus,
68             executor: Executor): CompletableFuture[FileStatus] = {
69           super.asyncReload(key, oldValue, executor)
70             .whenCompleteAsync(new ListCacheRefresh(key), executor)
71             .asInstanceOf[CompletableFuture[FileStatus]]
72         }
73       }
74     )
75 
76   // we use reflection to get around Caffeine 2.x/3.x API differences in some environments
77   private val refresh: Method = classOf[LoadingCache[_, _]].getMethods.find(_.getName == "refresh").getOrElse {
78     logger.warn("Could not get refresh cache method - cache operations will be less efficient")
79     null
80   }
81 
82   /**
83     * Register a path as existing
84     *
85     * @param fs file system
86     * @param path path
87     */
88   def register(fs: FileSystem, path: Path): Unit = {
89     pathCache.put((fs, path), java.lang.Boolean.TRUE)
90     if (refresh != null) {
91       refresh.invoke(statusCache, (fs, path)) // also triggers listCache update
92     }
93   }
94 
95   /**
96     * Check to see if a path exists
97     *
98     * @param fs file system
99     * @param path path
100     * @param reload reload the file status from the underlying file system before checking
101     * @return
102     */
103   def exists(fs: FileSystem, path: Path, reload: Boolean = false): Boolean = {
104     if (reload) {
105       invalidate(fs, path)
106     }
107     pathCache.get((fs, path)).booleanValue()
108   }
109 
110   /**
111     * Gets the file status for a path. Path must exist.
112     *
113     * @param fs file system
114     * @param path path
115     * @return
116     */
117   def status(fs: FileSystem, path: Path, reload: Boolean = false): FileStatus = {
118     if (reload) {
119       invalidate(fs, path)
120     }
121     statusCache.get((fs, path))
122   }
123 
124   /**
125     * List the children of a path
126     *
127     * @param fs file system
128     * @param dir directory path
129     * @return
130     */
131   def list(fs: FileSystem, dir: Path, reload: Boolean = false): Iterator[FileStatus] = {
132     if (reload) {
133       invalidate(fs, dir)
134     }
135     listCache.get((fs, dir)).iterator
136   }
137 
138   /**
139     * Invalidate any cached values for the path - they will be re-loaded on next access
140     *
141     * @param fs file system
142     * @param path path
143     */
144   def invalidate(fs: FileSystem, path: Path): Unit = {
145     Seq(pathCache, statusCache, listCache).foreach(_.invalidate((fs, path)))
146     if (path.getParent != null) {
147       listCache.invalidate((fs, path.getParent))
148     }
149   }
150 
151   object RemoteIterator {
152     def apply[T](iter: RemoteIterator[T]): Iterator[T] = new Iterator[T] {
153       override def hasNext: Boolean = iter.hasNext
154       override def next(): T = iter.next
155     }
156   }
157 
158   private class ListCacheRefresh(key: (FileSystem, Path)) extends BiConsumer[FileStatus, Throwable] {
159     override def accept(status: FileStatus, u: Throwable): Unit = {
160       if (status != null) { // could be null if load fails
161         val (fs, path) = key
162         val parent = path.getParent
163         if (parent != null) {
164           listCache.asMap().computeIfPresent((fs, parent), load(status)_)
165         }
166       }
167     }
168 
169     // noinspection ScalaUnusedSymbol
170     private def load(status: FileStatus)(ignored: (FileSystem, Path), list: Stream[FileStatus]): Stream[FileStatus] =
171       list.filterNot(f => f.getPath == status.getPath) :+ status
172   }
