18 /**
19  * Abstract baseclass for writing s3 object tags
20  *
21  * @param s3 s3 client
22  * @param path file path
23  */
24 abstract class S3ObjectTagObserver(s3: AmazonS3, path: Path) extends FileSystemObserver {
25 
26   private val (bucket, key) = {
27     val uri = path.toUri
28     val uriPath = uri.getPath
29     val key = if (uriPath.startsWith("/")) { uriPath.substring(1) } else { uriPath }
30     (uri.getHost, key)
31   }
32 
33   /**
34    * Return the tags to set on this file
35    *
36    * @return
37    */
38   protected def tags(): Iterable[(String, String)]
39 
40   override def flush(): Unit = {}
41 
42   override def close(): Unit = {
43     val iter = tags()
44     if (iter.nonEmpty) {
45       try {
46         val list = new java.util.ArrayList[Tag]()
47         iter.foreach { case (k, v) => list.add(new Tag(k, v)) }
48         s3.setObjectTagging(new SetObjectTaggingRequest(bucket, key, new ObjectTagging(list)))
49       } catch {
50         case e: Exception => throw new IOException("Error tagging object", e)
51       }
52     }
53   }
54 }
