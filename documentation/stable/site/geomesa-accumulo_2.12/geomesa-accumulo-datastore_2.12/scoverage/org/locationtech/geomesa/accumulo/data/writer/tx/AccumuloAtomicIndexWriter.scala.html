77   override protected def update(
78       feature: WritableFeature,
79       values: Array[RowKeyValue[_]],
80       previous: WritableFeature,
81       previousValues: Array[RowKeyValue[_]]): Unit = {
82     val mutations = Array.ofDim[Seq[MutationBuilder]](values.length)
83     // note: these are temporary conditions - we update them below
84     val updates = buildMutations(values, AppendBuilder.apply)
85     val deletes = buildMutations(previousValues, DeleteBuilder.apply)
86     var i = 0
87     while (i < values.length) {
88       val dels = ArrayBuffer(deletes(i): _*)
89       val puts = updates(i).map { mutations =>
90         // find any previous values that will be updated
91         val d = dels.indexWhere(p => java.util.Arrays.equals(p.row, mutations.row))
92         if (d == -1) { mutations } else {
93           // note: side-effect
94           val toUpdate = dels.remove(d)
95           // any previous values that are overwritten are added as conditions on the put
96           // any previous values that aren't overwritten need to be deleted
97           val remaining = toUpdate.kvs.filterNot(kv => mutations.kvs.exists(_.equalKey(kv)))
98           if (remaining.nonEmpty) {
99             dels.append(toUpdate.copy(kvs = remaining))
100           }
101           UpdateBuilder(mutations.row, mutations.kvs, toUpdate.kvs)
102         }
103       }
104       // note: order here is important, we need puts to operate before deletes in order for conditions to succeed
105       mutations(i) = puts ++ dels
106       i += 1
107     }
108     mutate(feature.feature.getID, mutations)
109   }

111   /**
112    * Apply the mutations
113    *
114    * @param id feature id
115    * @param mutations mutations to apply
116    */
117   @throws[ConditionalWriteException]
118   private def mutate(id: String, mutations: Array[Seq[MutationBuilder]]): Unit = {
119     val errors = ArrayBuffer.empty[ConditionalWriteStatus]
120     val successes = ArrayBuffer.empty[(Int, MutationBuilder)]
121     var i = 0
122     while (i < mutations.length) {
123       // note: mutations need to be applied in order so that conditions are correct
124       mutations(i).foreach { m =>
125         val status = writers(i).write(m.apply()).getStatus
126         if (status == ConditionalWriter.Status.ACCEPTED ||
127             (status == ConditionalWriter.Status.UNKNOWN && verifyWrite(tables(i), m))) {
128           successes += i -> m
129         } else {
130           errors += ConditionalWriteStatus(indices(i), m.name, status)
131         }
132       }
133       i += 1
134     }
135     if (errors.nonEmpty) {
136       // revert any writes we made
137       successes.foreach { case (i, m) =>
138         // note: if these are rejected, some other update has come through and we don't want to overwrite it
139         writers(i).write(m.invert())
140       }
141       throw ConditionalWriteException(id, errors.toSeq)
142     }
143   }

145   /**
146    * Verify if a mutation was successfully applied or not
147    *
148    * @param table table being mutated
149    * @param mutation mutation being applied
150    * @return true if the current state of the table reflects the mutation's intended result
151    */
152   private def verifyWrite(table: String, mutation: MutationBuilder): Boolean = {
153     WithClose(new IsolatedScanner(ds.connector.createScanner(table, ds.auths))) { scanner =>
154       val key = new Key(mutation.row)
155       scanner.setRange(new org.apache.accumulo.core.data.Range(key, true, key.followingKey(PartialKey.ROW), false))
156       val found = scanner.iterator().asScala.toList
157       mutation.apply().getUpdates.asScala.forall { update =>
158         if (update.isDeleted) {
159           found.forall { entry =>
160             !compare(entry.getKey.getColumnFamily, update.getColumnFamily) ||
161                 !compare(entry.getKey.getColumnQualifier, update.getColumnQualifier) ||
162                 !compare(entry.getKey.getColumnVisibility, update.getColumnVisibility) ||
163                 entry.getKey.isDeleted
164           }
165         } else {
166           found.exists { entry =>
167             compare(entry.getKey.getColumnFamily, update.getColumnFamily) &&
168                 compare(entry.getKey.getColumnQualifier, update.getColumnQualifier) &&
169                 compare(entry.getKey.getColumnVisibility, update.getColumnVisibility) &&
170                 entry.getValue.compareTo(update.getValue) == 0
171           }
172         }
173       }
174     }
175   }

177   private def buildMutations[T <: MutationBuilder](
178       values: Array[RowKeyValue[_]],
179       builderFactory: (Array[Byte], Seq[MutationValue]) => T): Array[Seq[T]] = {
180     val mutations = Array.ofDim[Seq[T]](values.length)
181     var i = 0
182     while (i < values.length) {
183       mutations(i) = values(i) match {
184         case kv: SingleRowKeyValue[_] =>
185           val values = kv.values.map { v =>
186             MutationValue(colFamilyMappings(i)(v.cf), v.cq, visCache(v.vis), v.value)
187           }
188           Seq(builderFactory(kv.row, values))
189 
190         case mkv: MultiRowKeyValue[_] =>
191           mkv.rows.map { row =>
192             val values = mkv.values.map { v =>
193               MutationValue(colFamilyMappings(i)(v.cf), v.cq, visCache(v.vis), v.value)
194             }
195             builderFactory(row, values)
196           }
197       }
198       i += 1
199     }
200     mutations
201   }
