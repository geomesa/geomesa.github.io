64   /**
65     * Version 3 encoding. @see `org.locationtech.geomesa.features.kryo.KryoFeatureSerializer` for details on
66     * the serialization scheme
67     *
68     * @param keys keys for this row
69     * @param values values for this row
70     * @return
71     */
72   private def encodeV3(keys: java.util.List[Key], values: java.util.List[Value]): Value = {
73     // Calculate length of serialised attributes, excluding attribute values themselves
74     var length = 1 +            // version
75       2 +                       // attribute count
76       1 +                       // offset size
77       count * 4 +               // attribute offsets (will use 4 bytes each) <--- offsets relative to first byte here
78       1 * 4 +                   // user data offset (will use 4 bytes)
79       IntBitSet.size(count) * 4 // null bit set, written in units of ints
80 
81     // Tracks our output position for copying attribute value bytes
82     // Begins immediately after the above
83     var valueCursor = length
84 
85     var i = 0
86     while (i < keys.size) {
87       val bytes = values.get(i).get
88       val input = KryoFeatureDeserialization.getInput(bytes, 1, bytes.length - 1) // skip the version byte
89       val metadata = Metadata(input) // read count, size, etc
90       // Column qualifiers tell us which attributes are intended to be populated given visibilities
91       val key = keys.get(i)
92       key.getColumnQualifier.getBytes.foreach { unsigned =>
93         val index = java.lang.Byte.toUnsignedInt(unsigned)
94         if(!metadata.nulls.contains(index)) {
95           val pos = metadata.setPosition(index)
96           val len = metadata.setPosition(index + 1) - pos
97           attributes(index) = (bytes, 4 + pos, len) // pos is relative to first byte of attribute offsets, see length calc
98           attributeVis(index) = key.getColumnVisibility.toString
99           length += len
100         }
101       }
102       i += 1
103     }
104 
105     // generate the serialized vis separately, as we can't reliably calculate it's size up front
106     val vis = KryoFeatureSerialization.getOutput(null)
107     KryoUserDataSerialization.serialize(vis, Collections.singletonMap(FEATURE_VISIBILITY, attributeVis.mkString(",")))
108 
109     length += vis.position()
110     val value = Array.ofDim[Byte](length)
111     val output = new Output(value)
112     output.writeByte(KryoFeatureSerializer.Version3)
113     output.writeShort(count)
114     output.write(4) // size of each offset, we always use 4 (vs 2) so we are sure to have enough space
115 
116     val nulls = IntBitSet(count)
117 
118     i = 0
119     while (i < count) {
120       // Write attribute offset relative to (version + count + size) = 4 bytes
121       output.writeInt(valueCursor - 4)
122       val attribute = attributes(i)
123       if (attribute == null /* Attribute value not provided by any values */ ||
124         attribute._3 == 0 /* Attribute value is provided but is null (size==0) */ ) {
125         // Attribute value must be null, no data to copy in
126         nulls.add(i)
127       } else {
128         // Attribute has a value so copy it in
129         val (bytes, offset, len) = attribute
130         System.arraycopy(bytes, offset, value, valueCursor, len)
131         valueCursor += len
132       }
133       // reset for next time through with new keys/values
134       attributes(i) = null
135       attributeVis(i) = ""
136       i += 1
137     }
138     output.writeInt(valueCursor - 4) // user-data offset. Note no user data has actually been copied in.
139 
140     // write nulls - we should already be in the right position
141     nulls.serialize(output)
142 
143     // copy in user data
144     System.arraycopy(vis.getBuffer, 0, value, valueCursor, vis.position())
145 
146     new Value(value)
147   }

149   /**
150     * Version 2 encoding. @see `org.locationtech.geomesa.features.kryo.impl.KryoFeatureDeserializationV2` for
151     * details on the serialization scheme
152     *
153     * @param keys keys for this row
154     * @param values values for this row
155     * @return
156     */
157   private def encodeV2(keys: java.util.List[Key], values: java.util.List[Value]): Value = {
158     setupV2()
159     val allValues = Array.ofDim[Array[Byte]](sft.getAttributeCount)
160     var i = 0
161     while (i < keys.size) {
162       val cq = keys.get(i).getColumnQualifier
163       val comma = cq.find(",")
164       // convert unsigned bytes to int
165       val indices = if (comma == -1) { cq.getBytes.map(java.lang.Byte.toUnsignedInt) } else { cq.getBytes.drop(comma + 1).map(java.lang.Byte.toUnsignedInt) }
166 
167       val bytes = values.get(i).get
168 
169       val input = KryoFeatureDeserialization.getInput(bytes, 0, bytes.length)
170       // reset our offsets
171       input.setPosition(1) // skip version
172       offsetStart = input.readInt()
173       input.setPosition(offsetStart) // set to offsets start
174       var j = 0
175       while (j < offsetsV2.length) {
176         offsetsV2(j) = if (input.position < input.limit) { input.readInt(true) } else { -1 }
177         j += 1
178       }
179 
180       // set the non-null values
181       indices.foreach { index =>
182         val endIndex = offsetsV2.indexWhere(_ != -1, index + 1)
183         val end = if (endIndex == -1) offsetStart else offsetsV2(endIndex)
184         val length = end - offsetsV2(index)
185         val values = Array.ofDim[Byte](length)
186         System.arraycopy(bytes, offsetsV2(index), values, 0, length)
187         allValues(index) = values
188       }
189 
190       i += 1
191     }
192 
193     i = 0
194     while (i < allValues.length) {
195       if (allValues(i) == null) {
196         allValues(i) = nullBytesV2(i)
197       }
198       i += 1
199     }
200 
201     KryoVisibilityRowEncoder.encodeV2(allValues, output, offsetsV2)
202   }
203 
204   private def setupV2(): Unit = {
205     if (nullBytesV2 == null) {
206       if (offsetsV2 == null || offsetsV2.length != sft.getAttributeCount) {
207         offsetsV2 = Array.ofDim[Int](sft.getAttributeCount)
208       }
209       nullBytesV2 = Array.tabulate(sft.getAttributeCount) { i =>
210         val descriptor = sft.getDescriptor(i)
211         val bindings = ObjectType.selectType(descriptor)
212         output.clear()
213         bindings.head match {
214           case ObjectType.STRING if bindings.last != ObjectType.JSON => output.writeString(null) // write string supports nulls
215           case ObjectType.LIST | ObjectType.MAP | ObjectType.BYTES   => output.writeInt(-1, true)
216           case _ => output.write(KryoFeatureSerializer.NullByte)
217         }
218         output.toBytes
219       }
220     }
221   }

250   private def encodeV2(values: Array[Array[Byte]], output: Output, offsets: Array[Int]): Value = {
251     output.clear()
252     output.writeInt(KryoFeatureSerializer.Version2, true)
253     output.setPosition(5) // leave 4 bytes to write the offsets
254     // note: we don't write ID - tables are assumed to be using serialization without IDs
255     // write attributes and keep track off offset into byte array
256     var i = 0
257     while (i < values.length) {
258       offsets(i) = output.position()
259       output.write(values(i))
260       i += 1
261     }
262     // write the offsets - variable width
263     i = 0
264     val offsetStart = output.position()
265     while (i < values.length) {
266       output.writeInt(offsets(i), true)
267       i += 1
268     }
269     // got back and write the start position for the offsets
270     val total = output.position()
271     output.setPosition(1)
272     output.writeInt(offsetStart)
273     output.setPosition(total) // set back to the end so that we get all the bytes
274 
275     new Value(output.toBytes)
276   }
