44   /**
45     * Index a polygon by it's bounding box
46     *
47     * @param xmin min x value in xBounds
48     * @param ymin min y value in yBounds
49     * @param xmax max x value in xBounds, must be >= xmin
50     * @param ymax max y value in yBounds, must be >= ymin
51     * @param lenient standardize boundaries to valid values, or raise an exception
52     * @return z value for the bounding box
53     */
54   def index(xmin: Double, ymin: Double, xmax: Double, ymax: Double, lenient: Boolean = false): Long = {
55     // normalize inputs to [0,1]
56     val (nxmin, nymin, nxmax, nymax) = normalize(xmin, ymin, xmax, ymax, lenient)
57 
58     // calculate the length of the sequence code (section 4.1 of XZ-Ordering paper)
59 
60     val maxDim = math.max(nxmax - nxmin, nymax - nymin)
61 
62     // l1 (el-one) is a bit confusing to read, but corresponds with the paper's definitions
63     val l1 = math.floor(math.log(maxDim) / XZSFC.LogPointFive).toInt
64 
65     // the length will either be (l1) or (l1 + 1)
66     val length = if (l1 >= g) { g } else {
67       val w2 = math.pow(0.5, l1 + 1) // width of an element at resolution l2 (l1 + 1)
68 
69       // predicate for checking how many axis the polygon intersects
70       // math.floor(min / w2) * w2 == start of cell containing min
71       def predicate(min: Double, max: Double): Boolean = max <= (math.floor(min / w2) * w2) + (2 * w2)
72 
73       if (predicate(nxmin, nxmax) && predicate(nymin, nymax)) l1 + 1 else l1
74     }
75 
76     sequenceCode(nxmin, nymin, length)
77   }

139   /**
140     * Determine XZ-curve ranges that will cover a given query window
141     *
142     * @param query a sequence of OR'd windows to cover, normalized to [0,1]
143     * @param rangeStop a rough max value for the number of ranges to return
144     * @return
145     */
146   private def ranges(query: Array[QueryWindow], rangeStop: Int): Seq[IndexRange] = {
147 
148     import XZ2SFC.{LevelOneElements, LevelTerminator}
149 
150     // stores our results - initial size of 100 in general saves us some re-allocation
151     val ranges = new java.util.ArrayList[IndexRange](100)
152 
153     // values remaining to process - initial size of 100 in general saves us some re-allocation
154     val remaining = new java.util.ArrayDeque[XElement](100)
155 
156     // checks if a quad is contained in the search space
157     def isContained(quad: XElement): Boolean = {
158       var i = 0
159       while (i < query.length) {
160         if (quad.isContained(query(i))) {
161           return true
162         }
163         i += 1
164       }
165       false
166     }
167 
168     // checks if a quad overlaps the search space
169     def isOverlapped(quad: XElement): Boolean = {
170       var i = 0
171       while (i < query.length) {
172         if (quad.overlaps(query(i))) {
173           return true
174         }
175         i += 1
176       }
177       false
178     }
179 
180     // checks a single value and either:
181     //   eliminates it as out of bounds
182     //   adds it to our results as fully matching, or
183     //   adds it to our results as partial matching and queues up it's children for further processing
184     def checkValue(quad: XElement, level: Short): Unit = {
185       if (isContained(quad)) {
186         // whole range matches, happy day
187         val (min, max) = sequenceInterval(quad.xmin, quad.ymin, level, partial = false)
188         ranges.add(IndexRange(min, max, contained = true))
189       } else if (isOverlapped(quad)) {
190         // some portion of this range is excluded
191         // add the partial match and queue up each sub-range for processing
192         val (min, max) = sequenceInterval(quad.xmin, quad.ymin, level, partial = true)
193         ranges.add(IndexRange(min, max, contained = false))
194         quad.children.foreach(remaining.add)
195       }
196     }
197 
198     // initial level
199     LevelOneElements.foreach(remaining.add)
200     remaining.add(LevelTerminator)
201 
202     // level of recursion
203     var level: Short = 1
204 
205     while (level < g && !remaining.isEmpty && ranges.size < rangeStop) {
206       val next = remaining.poll
207       if (next.eq(LevelTerminator)) {
208         // we've fully processed a level, increment our state
209         if (!remaining.isEmpty) {
210           level = (level + 1).toShort
211           remaining.add(LevelTerminator)
212         }
213       } else {
214         checkValue(next, level)
215       }
216     }
217 
218     // bottom out and get all the ranges that partially overlapped but we didn't fully process
219     while (!remaining.isEmpty) {
220       val quad = remaining.poll
221       if (quad.eq(LevelTerminator)) {
222         level = (level + 1).toShort
223       } else {
224         val (min, max) = sequenceInterval(quad.xmin, quad.ymin, level, partial = false)
225         ranges.add(IndexRange(min, max, contained = false))
226       }
227     }
228 
229     // we've got all our ranges - now reduce them down by merging overlapping values
230     // note: we don't bother reducing the ranges as in the XZ paper, as accumulo handles lots of ranges fairly well
231     ranges.sort(IndexRange.IndexRangeIsOrdered)
232 
233     var current = ranges.get(0) // note: should always be at least one range
234     val result = ArrayBuffer.empty[IndexRange]
235     var i = 1
236     while (i < ranges.size()) {
237       val range = ranges.get(i)
238       if (range.lower <= current.upper + 1) {
239         // merge the two ranges
240         current = IndexRange(current.lower, math.max(current.upper, range.upper), current.contained && range.contained)
241       } else {
242         // append the last range and set the current range for future merging
243         result.append(current)
244         current = range
245       }
246       i += 1
247     }
248     // append the last range - there will always be one left that wasn't added
249     result.append(current)
250 
251     result.toSeq
252   }

254   /**
255     * Computes the sequence code for a given point - for polygons this is the lower-left corner.
256     *
257     * Based on Definition 2 from the XZ-Ordering paper
258     *
259     * @param x normalized x value [0,1]
260     * @param y normalized y value [0,1]
261     * @param length length of the sequence code that will be generated
262     * @return
263     */
264   private def sequenceCode(x: Double, y: Double, length: Int): Long = {
265     var xmin = 0.0
266     var ymin = 0.0
267     var xmax = 1.0
268     var ymax = 1.0
269 
270     var cs = 0L
271 
272     var i = 0
273     while (i < length) {
274       val xCenter = (xmin + xmax) / 2.0
275       val yCenter = (ymin + ymax) / 2.0
276       (x < xCenter, y < yCenter) match {
277         case (true,  true)  => cs += 1L                                             ; xmax = xCenter; ymax = yCenter
278         case (false, true)  => cs += 1L + 1L * (math.pow(4, g - i).toLong - 1L) / 3L; xmin = xCenter; ymax = yCenter
279         case (true,  false) => cs += 1L + 2L * (math.pow(4, g - i).toLong - 1L) / 3L; xmax = xCenter; ymin = yCenter
280         case (false, false) => cs += 1L + 3L * (math.pow(4, g - i).toLong - 1L) / 3L; xmin = xCenter; ymin = yCenter
281       }
282       i += 1
283     }
284 
285     cs
286   }

288   /**
289     * Computes an interval of sequence codes for a given point - for polygons this is the lower-left corner.
290     *
291     * @param x normalized x value [0,1]
292     * @param y normalized y value [0,1]
293     * @param length length of the sequence code that will used as the basis for this interval
294     * @param partial true if the element partially intersects the query window, false if it is fully contained
295     * @return
296     */
297   private def sequenceInterval(x: Double, y: Double, length: Short, partial: Boolean): (Long, Long) = {
298     val min = sequenceCode(x, y, length)
299     // if a partial match, we just use the single sequence code as an interval
300     // if a full match, we have to match all sequence codes starting with the single sequence code
301     val max = if (partial) { min } else {
302       // from lemma 3 in the XZ-Ordering paper
303       min + (math.pow(4, g - length + 1).toLong - 1L) / 3L
304     }
305     (min, max)
306   }

308   /**
309     * Normalize user space values to [0,1]
310     *
311     * @param xmin min x value in user space
312     * @param ymin min y value in user space
313     * @param xmax max x value in user space, must be >= xmin
314     * @param ymax max y value in user space, must be >= ymin
315     * @param lenient standardize boundaries to valid values, or raise an exception
316     * @return
317     */
318   private def normalize(xmin: Double,
319                         ymin: Double,
320                         xmax: Double,
321                         ymax: Double,
322                         lenient: Boolean): (Double, Double, Double, Double) = {
323     require(xmin <= xmax && ymin <= ymax, s"Bounds must be ordered: [$xmin $xmax] [$ymin $ymax]")
324 
325     try {
326       require(xmin >= xLo && xmax <= xHi && ymin >= yLo && ymax <= yHi,
327         s"Values out of bounds ([$xLo $xHi] [$yLo $yHi]): [$xmin $xmax] [$ymin $ymax]")
328 
329       val nxmin = (xmin - xLo) / xSize
330       val nymin = (ymin - yLo) / ySize
331       val nxmax = (xmax - xLo) / xSize
332       val nymax = (ymax - yLo) / ySize
333 
334       (nxmin, nymin, nxmax, nymax)
335     } catch {
336       case _: IllegalArgumentException if lenient =>
337 
338         val bxmin = if (xmin < xLo) { xLo } else if (xmin > xHi) { xHi } else { xmin }
339         val bymin = if (ymin < yLo) { yLo } else if (ymin > yHi) { yHi } else { ymin }
340         val bxmax = if (xmax < xLo) { xLo } else if (xmax > xHi) { xHi } else { xmax }
341         val bymax = if (ymax < yLo) { yLo } else if (ymax > yHi) { yHi } else { ymax }
342 
343         val nxmin = (bxmin - xLo) / xSize
344         val nymin = (bymin - yLo) / ySize
345         val nxmax = (bxmax - xLo) / xSize
346         val nymax = (bymax - yLo) / ySize
347 
348         (nxmin, nymin, nxmax, nymax)
349     }
350   }

382   /**
383     * An extended Z curve element. Bounds refer to the non-extended z element for simplicity of calculation.
384     *
385     * An extended Z element refers to a normal Z curve element that has it's upper bounds expanded by double it's
386     * width/height. By convention, an element is always square.
387     *
388     * @param xmin x lower bound in [0-1]
389     * @param ymin y lower bound in [0-1]
390     * @param xmax x upper bound in [0-1], must be >= xmin
391     * @param ymax y upper bound in [0-1], must be >= ymin
392     * @param length length of the non-extended side (note: by convention width should be equal to height)
393     */
394   private case class XElement(xmin: Double, ymin: Double, xmax: Double, ymax: Double, length: Double) {
395 
396     // extended x and y bounds
397     lazy val xext = xmax + length
398     lazy val yext = ymax + length
399 
400     def isContained(window: QueryWindow): Boolean =
401       window.xmin <= xmin && window.ymin <= ymin && window.xmax >= xext && window.ymax >= yext
402 
403     def overlaps(window: QueryWindow): Boolean =
404       window.xmax >= xmin && window.ymax >= ymin && window.xmin <= xext && window.ymin <= yext
405 
406     def children: Seq[XElement] = {
407       val xCenter = (xmin + xmax) / 2.0
408       val yCenter = (ymin + ymax) / 2.0
409       val len = length / 2.0
410       val c0 = copy(xmax = xCenter, ymax = yCenter, length = len)
411       val c1 = copy(xmin = xCenter, ymax = yCenter, length = len)
412       val c2 = copy(xmax = xCenter, ymin = yCenter, length = len)
413       val c3 = copy(xmin = xCenter, ymin = yCenter, length = len)
414       Seq(c0, c1, c2, c3)
415     }
416   }
