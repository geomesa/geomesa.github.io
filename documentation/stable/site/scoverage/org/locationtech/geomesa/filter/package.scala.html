43   /**
44    * This function rewrites a org.geotools.api.filter.Filter in terms of a top-level OR with children filters which
45    * 1) do not contain further ORs, (i.e., ORs bubble up)
46    * 2) only contain at most one AND which is at the top of their 'tree'
47    *
48    * Note that this further implies that NOTs have been 'pushed down' and do have not have ANDs nor ORs as children.
49    *
50    * In boolean logic, this form is called disjunctive normal form (DNF).
51    *
52    * @param filter An arbitrary filter.
53    * @return       A filter in DNF (described above).
54    */
55   def rewriteFilterInDNF(filter: Filter)(implicit ff: FilterFactory): Filter = {
56     val ll = logicDistributionDNF(filter)
57     if (ll.size == 1) {
58       if (ll.head.size == 1) {
59         ll.head.head
60       } else {
61         ff.and(ll.head.asJava)
62       }
63     } else {
64       val children = ll.map { l =>
65         l.size match {
66           case 1 => l.head
67           case _ => ff.and(l.asJava)
68         }
69       }
70       ff.or(children.asJava)
71     }
72   }

75    *
76    * @param x: An arbitrary @org.geotools.api.filter.Filter
77    * @return   A List[ List[Filter] ] where the inner List of Filters are to be joined by
78    *           Ands and the outer list combined by Ors.

80   private[filter] def logicDistributionDNF(x: Filter): List[List[Filter]] = x match {
81     case or: Or  => or.getChildren.asScala.toList.flatMap(logicDistributionDNF)
82 
83     case and: And => and.getChildren.asScala.foldRight (List(List.empty[Filter])) {
84       (f, dnf) => for {
85         a <- logicDistributionDNF (f)
86         b <- dnf
87       } yield a ++ b
88     }
89 
90     case not: Not =>
91       not.getFilter match {
92         case and: And => logicDistributionDNF(deMorgan(and))
93         case or:  Or => logicDistributionDNF(deMorgan(or))
94         case _: Filter => List(List(not))
95       }
96 
97     case f: Filter => List(List(f))
98   }

100   /**
101    * This function rewrites a org.geotools.api.filter.Filter in terms of a top-level AND with children filters which
102    * 1) do not contain further ANDs, (i.e., ANDs bubble up)
103    * 2) only contain at most one OR which is at the top of their 'tree'
104    *
105    * Note that this further implies that NOTs have been 'pushed down' and do have not have ANDs nor ORs as children.
106    *
107    * In boolean logic, this form is called conjunctive normal form (CNF).
108    *
109    * The main use case for this function is to aid in splitting filters between a combination of a
110    * GeoMesa data store and some other data store. This is done with the AndSplittingFilter class.
111    * In the examples below, anything with "XAttr" is assumed to be a filter that CANNOT be answered
112    * through GeoMesa. In having a filter split on the AND, the portion of the filter that GeoMesa
113    * CAN answer will be applied in GeoMesa, returning a result set, and then the portion that GeoMesa CANNOT
114    * answer will be applied on that result set.
115    *
116    * Examples:
117    *  1. (
118    *       (GmAttr ILIKE 'test')
119    *       OR
120    *       (date BETWEEN '2014-01-01T10:30:00.000Z' AND '2014-01-02T10:30:00.000Z')
121    *     )
122    *      AND
123    *     (XAttr ILIKE = 'example')
124    *
125    *     Converting to CNF will allow easily splitting the filter on the AND into two children
126    *      - one child is the "GmAttr" and "date" filters that can be answered with GeoMesa
127    *      - one child is the "XAttr" filter that cannot be answered by GeoMesa
128    *
129    *      In this case, the GeoMesa child filter will be processed first, and then the "XAttr" filter will
130    *    be processed on the GeoMesa result set to return a subset of the GeoMesa results.
131    *
132    *  2. (GmAttr ILIKE 'test')
133    *      AND
134    *          (
135    *            (date BETWEEN '2014-01-01T10:30:00.000Z' AND '2014-01-02T10:30:00.000Z')
136    *             OR
137    *            (XAttr1 ILIKE = 'example1')
138    *          )
139    *      AND
140    *     (XAttr2 ILIKE = 'example2')
141    *
142    *     Converting to CNF still allows easily splitting the filter on the AND into three children
143    *      - one child is the "GmAttr" filter
144    *      - one child is the "date" OR "XAttr1" filter
145    *      - one child is the "XAttr2" filter
146    *
147    *      In this case, the "GmAttr" child will be processed first, returning a result set from GeoMesa
148    *    called RS1. Then, RS1 will be further filtered with the "date" predicate that can be handled
149    *    by GeoMesa, returning a subset of RS1 called SS1. The additional filter which cannot be answered
150    *    by GeoMesa, "XAttr1," will be applied to RS1 and return subset SS2. Finally, the final child,
151    *    the "XAttr2" filter, which cannot be answered by GeoMesa, will be applied to both SS1 and SS2 to
152    *    return SS3, a JOIN of SS1+SS2 filtered with "XAttr2."
153    *
154    *  3. (GmAttr ILIKE 'test')
155    *      OR
156    *     (XAttr ILIKE = 'example')
157    *
158    *     This is the worst-case-scenario for a query that is answered through two data stores, both
159    *     GeoMesa and some other store.
160    *
161    *     CNF converts this to:
162    *      - one child of "GmAttr" OR "XAttr"
163    *
164    *      In this case, the "GmAttr" will return a result set, RS1. The reason this is the
165    *    worst-case-scenario is because, to answer the "XAttr" portion of the query (which cannot be
166    *    answered by GeoMesa), a "Filter.INCLUDE" A.K.A a full table scan (on Accumulo) A.K.A. every
167    *    record in GeoMesa is necessary to find the results that satisfy the "XAttr" portion of the
168    *    query. This will product result set RS2. The returned results will be a JOIN of RS1+RS2.
169    *
170    *
171    * @param filter An arbitrary filter.
172    * @return       A filter in CNF (described above).
173    */
174   def rewriteFilterInCNF(filter: Filter)(implicit ff: FilterFactory): Filter = {
175     val ll = logicDistributionCNF(FilterHelper.simplify(filter))
176     if (ll.size == 1) {
177       if (ll.head.size == 1) ll.head.head else ff.or(ll.head.asJava)
178     } else {
179       val children = ll.map { l =>
180         l.size match {
181           case 1 => l.head
182           case _ => ff.or(l.asJava)
183         }
184       }
185       ff.and(children.asJava)
186     }
187   }

216    *  The input is a filter which had a Not applied to it.
217    *  This function uses deMorgan's law to 'push the Not down'
218    *   as well as cancel adjacent Nots.
219    */
220   private[filter] def deMorgan(f: Filter)(implicit ff: FilterFactory): Filter = f match {
221     case and: And => ff.or(and.getChildren.asScala.map(a => ff.not(a).asInstanceOf[Filter]).asJava)
222     case or:  Or  => ff.and(or.getChildren.asScala.map(a => ff.not(a).asInstanceOf[Filter]).asJava)
223     case not: Not => not.getFilter

289   def getAttributeProperty(f: Filter): Option[String] = {
290     f match {
291       // equals checks
292       case f: PropertyIsEqualTo => checkOrder(f.getExpression1, f.getExpression2).map(_.name)
293       case f: TEquals           => checkOrder(f.getExpression1, f.getExpression2).map(_.name)
294 
295       // like checks
296       case f: PropertyIsLike =>
297         if (likeEligible(f)) {
298           val prop = f.getExpression.asInstanceOf[PropertyName].getPropertyName
299           Some(prop)
300         } else {
301           None
302         }
303 
304       // range checks
305       case f: PropertyIsGreaterThan          => checkOrder(f.getExpression1, f.getExpression2).map(_.name)
306       case f: PropertyIsGreaterThanOrEqualTo => checkOrder(f.getExpression1, f.getExpression2).map(_.name)
307       case f: PropertyIsLessThan             => checkOrder(f.getExpression1, f.getExpression2).map(_.name)
308       case f: PropertyIsLessThanOrEqualTo    => checkOrder(f.getExpression1, f.getExpression2).map(_.name)
309       case f: PropertyIsBetween =>
310         val prop = f.getExpression.asInstanceOf[PropertyName].getPropertyName
311         val (left, right) = (f.getLowerBoundary, f.getUpperBoundary) match {
312           case (l: Literal, r: Literal) => (l, r)
313           case _ => (null, null)
314         }
315         if (left != null && right != null) {
316           Some(prop)
317         } else {
318           None
319         }
320 
321       // date range checks
322       case f: Before => checkOrder(f.getExpression1, f.getExpression2).map(_.name)
323       case f: After  => checkOrder(f.getExpression1, f.getExpression2).map(_.name)
324       case f: During => checkOrder(f.getExpression1, f.getExpression2).map(_.name)
325 
326       // not check - we only support 'not null' for an indexed attribute
327       case n: Not =>
328         Option(n.getFilter).collect { case f: PropertyIsNull => f }.map { f =>
329           f.getExpression.asInstanceOf[PropertyName].getPropertyName
330         }
331 
332       case _ => None
333     }
334   }

396   /**
397    * Checks the order of properties and literals in the expression
398    *
399    * @param one first expression
400    * @param two second expression
401    * @return (prop, literal, whether the order was flipped)
402    */
403   def checkOrder(one: Expression, two: Expression): Option[AttributeExpression] = {
404     (one, two) match {
405       case (p: PropertyName, l: Literal) => Some(PropertyLiteral(p.getPropertyName, l, flipped = false))
406       case (l: Literal, p: PropertyName) => Some(PropertyLiteral(p.getPropertyName, l, flipped = true))
407       case (f: Function, l: Literal)     => attribute(f).map(FunctionLiteral(_, f, l, flipped = false))
408       case (l: Literal, f: Function)     => attribute(f).map(FunctionLiteral(_, f, l, flipped = true))
409 
410       case (p: PropertyName, f: Function) if attribute(f).isEmpty =>
411         Some(PropertyLiteral(p.getPropertyName, ff.literal(f.evaluate(null)), flipped = false))
412 
413       case (f: Function, p: PropertyName) if attribute(f).isEmpty =>
414         Some(PropertyLiteral(p.getPropertyName, ff.literal(f.evaluate(null)), flipped = true))
415 
416       case (f1: Function, f2: Function) =>
417         (attribute(f1), attribute(f2)) match {
418           case (Some(a), None) => Some(FunctionLiteral(a, f1, ff.literal(f2.evaluate(null)), flipped = false))
419           case (None, Some(a)) => Some(FunctionLiteral(a, f2, ff.literal(f1.evaluate(null)), flipped = true))
420           case _ => None
421         }
422 
423       case _ => None
424     }
425   }
