19 /**
20   * Extracts filters for a given attribute
21   */
22 object FilterExtractingVisitor {
23 
24   /**
25     * Extract filters on a given attribute. If a schema is available,
26     * prefer `apply(Filter, String, SimpleFeatureType, Predicate)` as that will handle
27     * things like default geometry bboxes
28     *
29     * @param filter filter to evaluate
30     * @param attribute attribute to extract
31     * @param predicate additional predicate on the filters matching the attribute
32     * @return (filter on the attribute, filter on everything else)
33     */
34   def apply(filter: Filter,
35             attribute: String,
36             predicate: Filter => Boolean): (Option[Filter], Option[Filter]) =
37     apply(filter, attribute, null, predicate)
38 
39   /**
40     * Extract filters on a given attribute
41     *
42     * @param filter filter to evaluate
43     * @param attribute attribute to extract
44     * @param sft simple feature type
45     * @param predicate additional predicate on the filters matching the attribute
46     * @return (filter on the attribute, filter on everything else)
47     */
48   def apply(filter: Filter,
49             attribute: String,
50             sft: SimpleFeatureType,
51             predicate: Filter => Boolean = _ => true): (Option[Filter], Option[Filter]) = {
52     val visitor = new FilterExtractingVisitor(attribute, sft, predicate)
53     val (yes, no) = filter.accept(visitor, null).asInstanceOf[(Filter, Filter)]
54     (Option(yes), Option(no))
55   }
56 }

58 class FilterExtractingVisitor(attribute: String, sft: SimpleFeatureType, predicate: Filter => Boolean)
59     extends FilterVisitor {
60 
61   import org.locationtech.geomesa.filter.{andOption, ff, orOption}
62 
63   /**
64     * Should we keep the filter or discard it
65     *
66     * @param f filter
67     * @return true if it contains the attribute we want
68     */
69   private def keep(f: Filter): Boolean = FilterHelper.propertyNames(f, sft).contains(attribute) && predicate(f)
70 
71   override def visit(f: Or, data: AnyRef): AnyRef = {
72     val children = f.getChildren.asScala.map(_.accept(this, data).asInstanceOf[(Filter, Filter)])
73     val yes = children.map(_._1).filter(_ != null).toSeq
74     val no  = children.map(_._2).filter(_ != null).toSeq
75     // only return ORs where both sides satisfy the predicate
76     // we handle the returned predicates as implicit ANDs, if we split an OR we break that
77     if (no.isEmpty) {
78       (orOption(yes).orNull, null)
79     } else {
80       (null, orOption(yes ++ no).orNull)
81     }
82   }
83 
84   override def visit(f: And, data: AnyRef): AnyRef = {
85     val children = f.getChildren.asScala.map(_.accept(this, data).asInstanceOf[(Filter, Filter)])
86     val yes = children.map(_._1).filter(_ != null).toSeq
87     val no  = children.map(_._2).filter(_ != null).toSeq
88     (andOption(yes).orNull, andOption(no).orNull)
89   }
90 
91   override def visit(f: Not, data: AnyRef): AnyRef = {
92     if (predicate(f)) {
93       val (yes, no) = f.getFilter.accept(this, data).asInstanceOf[(Filter, Filter)]
94       (if (yes == null) null else ff.not(yes), if (no == null) null else ff.not(no))
95     } else {
96       (null, f)
97     }
98   }
