50   private def toLatLonString(point: Point): String = {
51     val coordinate = point.getCoordinate
52     s"${latLonFormat(coordinate.y, lat = true)} ${latLonFormat(coordinate.x, lat = false)}"

54 
55   private def latLonFormat(value: Double, lat: Boolean): String = {
56     val degrees = value.floor
57     val decimal = value - degrees
58     val minutes = (decimal * 60).floor
59     val seconds = (decimal * 60 - minutes) * 60
60     if (lat)
61       f"${degrees.abs}%1.0f\u00B0$minutes%1.0f" +"\'" + f"$seconds%1.3f" + "\"" + s"${if (degrees < 0) "S" else "N"}"
62     else
63       f"${degrees.abs}%1.0f\u00B0$minutes%1.0f" +"\'" + f"$seconds%1.3f" + "\"" + s"${if (degrees < 0) "W" else "E"}"
64   }
