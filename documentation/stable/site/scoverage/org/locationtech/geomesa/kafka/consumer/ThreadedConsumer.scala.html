40   class ConsumerRunnable(id: String, consumer: Consumer[Array[Byte], Array[Byte]], handler: ConsumerErrorHandler)
41       extends Runnable {
42 
43     private var errorCount = 0
44 
45     lazy private val topics = consumer.subscription().asScala.mkString(", ")
46 
47     override def run(): Unit = {
48       try {
49         var interrupted = false
50         while (isOpen && !interrupted) {
51           try {
52             val result = KafkaConsumerVersions.poll(consumer, frequency)
53             lazy val topics = result.partitions.asScala.map(tp => s"[${tp.topic}:${tp.partition}]").mkString(",")
54             logger.debug(s"Consumer [$id] poll received ${result.count()} records for $topics")
55             if (!result.isEmpty) {
56               val records = result.iterator()
57               while (records.hasNext) {
58                 consume(records.next())
59               }
60               logger.trace(s"Consumer [$id] finished processing ${result.count()} records from topic $topics")
61               consumer.commitAsync(callback)
62               errorCount = 0 // reset error count
63             }
64           } catch {
65             case _: WakeupException | _: InterruptException | _: InterruptedException => interrupted = true
66             case NonFatal(e) =>
67               errorCount += 1
68               logger.warn(s"Consumer [$id] error receiving message from topic $topics:", e)
69               if (errorCount <= 300 || handler.handle(id, e)) {
70                 Thread.sleep(1000)
71               } else {
72                 logger.error(s"Consumer [$id] shutting down due to too many errors from topic $topics:", e)
73                 throw e
74               }
75           }
76         }
77       } finally {
78         if (closeConsumers) {
79           try { consumer.close() } catch {
80             case NonFatal(e) => logger.warn(s"Error calling close on consumer: ", e)
81           }
82         }
83       }
84     }
85   }
86 }
