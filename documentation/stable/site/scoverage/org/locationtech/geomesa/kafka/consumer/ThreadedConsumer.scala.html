40   class ConsumerRunnable(id: String, consumer: Consumer[Array[Byte], Array[Byte]], handler: ConsumerErrorHandler)
41       extends Runnable {
42 
43     private var errorCount = 0
44 
45     lazy private val topics = consumer.subscription().asScala.mkString(", ")
46 
47     override def run(): Unit = {
48       try {
49         var interrupted = false
50         while (isOpen && !interrupted) {
51           try {
52             processPoll(KafkaConsumerVersions.poll(consumer, frequency))
53           } catch {
54             case _: WakeupException | _: InterruptException | _: InterruptedException => interrupted = true
55             case NonFatal(e) =>
56               errorCount += 1
57               logger.warn(s"Consumer [$id] error receiving message from topic $topics:", e)
58               if (errorCount <= 300 || handler.handle(id, e)) {
59                 Thread.sleep(1000)
60               } else {
61                 logger.error(s"Consumer [$id] shutting down due to too many errors from topic $topics:", e)
62                 throw e
63               }
64           }
65         }
66       } finally {
67         if (closeConsumers) {
68           try { consumer.close() } catch {
69             case NonFatal(e) => logger.warn(s"Error calling close on consumer: ", e)
70           }
71         }
72       }
73     }
74 
75     /**
76      * Process the results of a call to poll()
77      *
78      * @param result records
79      */
80     protected def processPoll(result: ConsumerRecords[Array[Byte], Array[Byte]]): Unit = {
81       lazy val topics = result.partitions.asScala.map(tp => s"[${tp.topic}:${tp.partition}]").mkString(",")
82       logger.debug(s"Consumer [$id] poll received ${result.count()} records for $topics")
83       if (!result.isEmpty) {
84         val records = result.iterator()
85         while (records.hasNext) {
86           consume(records.next())
87         }
88         logger.trace(s"Consumer [$id] finished processing ${result.count()} records from topic $topics")
89         consumer.commitAsync(callback)
90         errorCount = 0 // reset error count
91       }
92     }
93   }
94 }
