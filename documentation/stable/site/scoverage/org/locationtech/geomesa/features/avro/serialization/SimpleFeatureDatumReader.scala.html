18 /**
19  * Avro reader for simple features. Due to avro lifecycles, the following methods must be
20  * called in order before use:
21  *
22  * 1) setSchema (if available)
23  * 2) setFeatureType
24  */
25 class SimpleFeatureDatumReader extends DatumReader[SimpleFeature] {
26 
27   import AvroField._
28 
29   import scala.collection.JavaConverters._
30 
31   private val versionedFields = Array.ofDim[VersionedFields](SerializationVersions.MaxVersion)
32 
33   private var schema: Schema = _
34   private var sft: SimpleFeatureType = _
35   private var fid: Option[AvroField[String]] = None
36   private var userData: Option[AvroField[java.util.Map[AnyRef, AnyRef]]] = None
37   private var fields: Seq[AvroField[AnyRef]] = Seq.empty
38 
39   override def setSchema(schema: Schema): Unit = {
40     require(this.schema == null || this.schema == schema, "setSchema can only be called once")
41     require(
42       schema.getFields.size > 0 && schema.getFields.get(0).name() == VersionField.name,
43       s"Unexpected schema: $schema")
44     this.schema = schema
45   }
46 
47   /**
48    * Set the feature type. This must be called before the reader can be used
49    *
50    * @param sft simple feature type
51    */
52   def setFeatureType(sft: SimpleFeatureType): Unit = {
53     // ideally, we could re-construct the feature type from the schema... going forward, that should be possible,
54     // but to support older data files we still need this
55     require(schema != null, "setSchema must be called before setFeatureType")
56     this.sft = sft
57     this.fid = if (schema.getField(FidField.name) != null) { Some(FidField) } else { None }
58     this.userData = if (schema.getField(UserDataField.name) != null) { Some(UserDataField) } else { None }
59     this.fields = sft.getAttributeDescriptors.asScala.map(AvroField.apply).toSeq
60   }
61 
62   override def read(reuse: SimpleFeature, in: Decoder): SimpleFeature = {
63     val version = VersionField.read(in)
64     var fields = versionedFields(version - 1)
65     if (fields == null) {
66       val fid = this.fid.map(_.withVersion(version)).orNull
67       val userData = this.userData.map(_.withVersion(version)).orNull
68       fields = VersionedFields(fid, userData, this.fields.map(_.withVersion(version)).toArray)
69       versionedFields(version - 1) = fields
70     }
71     val id = if (fields.fid == null) { "" } else { fields.fid.read(in) }
72     val sf = reuse match {
73       case f: ScalaSimpleFeature if sft.eq(f.getFeatureType) => f.setId(id); f.getUserData.clear(); f
74       case _ => new ScalaSimpleFeature(sft, id)
75     }
76     var i = 0
77     while (i < sft.getAttributeCount) {
78       sf.setAttributeNoConvert(i, fields.fields(i).read(in))
79       i += 1
80     }
81     if (fields.userData != null) {
82       sf.getUserData.putAll(fields.userData.read(in))
83     }
84     sf
85   }
86 }
