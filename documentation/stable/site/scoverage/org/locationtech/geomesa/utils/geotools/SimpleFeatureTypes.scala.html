339   /**
340     * Creates an immutable copy of the simple feature type
341     *
342     * Note that immutable types will never be `equals` to a mutable type, due to specific class checking
343     * in the geotools implementation
344     *
345     * Note that some parts of the feature type may still be mutable - in particular AttributeType,
346     * GeometryType and SuperType are not used by geomesa so we don't bother with them. In addition,
347     * user data keys and values may be mutable objects, so while the user data map will not change,
348     * the values inside may
349     *
350     * @param sft simple feature type
351     * @param extraData additional user data to add to the simple feature type
352     * @return immutable copy of the simple feature type
353     */
354   def immutable(
355       sft: SimpleFeatureType,
356       extraData: java.util.Map[_ <: AnyRef, _ <: AnyRef] = null): SimpleFeatureType = {
357     sft match {
358       case immutable: ImmutableSimpleFeatureType if extraData == null || extraData.isEmpty => immutable
359       case _ =>
360         val schema = new java.util.ArrayList[AttributeDescriptor](sft.getAttributeCount)
361         var geom: GeometryDescriptor = null
362         var i = 0
363         while (i < sft.getAttributeCount) {
364           val descriptor = sft.getDescriptor(i)
365           val im = immutable(descriptor)
366           if (descriptor == sft.getGeometryDescriptor) {
367             geom = im.asInstanceOf[GeometryDescriptor]
368           }
369           schema.add(im)
370           i += 1
371         }
372         val userData = Option(extraData).filterNot(_.isEmpty).map { data =>
373           val map = new java.util.HashMap[AnyRef, AnyRef](data.size() + sft.getUserData.size)
374           map.putAll(sft.getUserData)
375           map.putAll(data)
376           map
377         }
378 
379         new ImmutableSimpleFeatureType(sft.getName, schema, geom, sft.isAbstract, sft.getRestrictions, sft.getSuper,
380           sft.getDescription, userData.getOrElse(sft.getUserData))
381     }
382   }

461   /**
462     * Compare two feature types. This method compares the the schemas of each feature type, i.e. the
463     * names and type bindings of the attributes. The result will be:
464     *
465     * <ul>
466     *   <li>a positive int if type A is a sub type or reorder of type B</li>
467     *   <li>zero if type A has the same schema as type B</li>
468     *   <li>a negative int if none of the above</li>
469     * </ul>
470     *
471     * Note that this shouldn't be used for sorting.
472     *
473     * Note: GeoTools DataUtilities has a similar method that does not work correctly
474     *
475     * @param a schema A
476     * @param b schema B
477     * @return comparison
478     */
479   def compare(a: SimpleFeatureType, b: SimpleFeatureType): Int = {
480     if (a.getAttributeCount > b.getAttributeCount) {
481       return -1 // not equals, subtype or reorder
482     }
483 
484     var exact = a.getAttributeCount == b.getAttributeCount
485 
486     var i = 0
487     while (i < a.getAttributeCount) {
488       val ad = a.getDescriptor(i)
489       val bd = b.getDescriptor(i)
490       if (ad.getLocalName == bd.getLocalName) {
491         if (ad.getType.getBinding != bd.getType.getBinding) {
492           if (bd.getType.getBinding.isAssignableFrom(ad.getType.getBinding)) {
493             exact = false // we've seen a subtyping
494           } else {
495             return -1 // not a subtype/reorder
496           }
497         }
498       } else {
499         val reorder = b.getDescriptor(ad.getLocalName)
500         if (reorder == null || !reorder.getType.getBinding.isAssignableFrom(ad.getType.getBinding)) {
501           return -1 // not a subtype/reorder
502         } else {
503           exact = false // we've seen a reordering
504         }
505       }
506       i += 1
507     }
508 
509     if (exact) { 0 } else {
510       1  // we haven't hit an absolute inequality, but we've found a reordering or subtype
511     }
512   }

515    * Compares that index tables are the same format (key and value) between the two feature types
516    *
517    * @param a first type
518    * @param b second type
519    * @return 0 if indices are the same, non-zero otherwise

521   def compareIndexConfigs(a: SimpleFeatureType, b: SimpleFeatureType): Int = {
522     def compareAttributes(i: Int): Boolean = {
523       val (ad, bd) = (a.getDescriptor(i), b.getDescriptor(i))
524       ad.getType.getBinding == bd.getType.getBinding && AttributeChecks.forall(c => c(ad) == c(bd))
525     }
526     if (a.getAttributeCount == b.getAttributeCount &&
527         IndexChecks.forall(c => c(a) == c(b)) &&
528         Range(0, a.getAttributeCount).forall(compareAttributes)) {
529       0
530     } else {
531       1
532     }
533   }

535   private def createFeatureType(namespace: String,
536                                 name: String,
537                                 spec: SimpleFeatureSpec,
538                                 factory: Option[FeatureTypeFactory] = None): SimpleFeatureType = {
539     import AttributeOptions.OptDefault
540 
541     val defaultGeom = {
542       val geomAttributes = spec.attributes.collect { case g: GeomAttributeSpec => g }
543       geomAttributes.find(_.options.get(OptDefault).exists(_.toBoolean))
544           .orElse(geomAttributes.headOption)
545           .map(_.name)
546     }
547     val defaultDate = if (spec.options.get(IndexIgnoreDtg).exists(toBoolean)) { None } else {
548       val dateAttributes = spec.attributes.filter(a => classOf[Date].isAssignableFrom(a.clazz))
549       spec.options.get(DefaultDtgField).flatMap(dtg => dateAttributes.find(_.name == dtg))
550           .orElse(dateAttributes.find(_.options.get(OptDefault).exists(_.toBoolean)))
551           .orElse(dateAttributes.headOption)
552           .map(_.name)
553     }
554 
555     val b = factory.map(new SimpleFeatureTypeBuilder(_)).getOrElse(new SimpleFeatureTypeBuilder())
556     b.setNamespaceURI(namespace)
557     b.setName(name)
558     b.addAll(spec.attributes.map(_.toDescriptor).asJava)
559     defaultGeom.foreach(b.setDefaultGeometry)
560 
561     val sft = b.buildFeatureType()
562     sft.getUserData.putAll(spec.options.asJava)
563     defaultDate.foreach(sft.setDtgField)
564     sft
565   }
