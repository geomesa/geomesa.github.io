23 // A CloseableIterator is one which involves some kind of close function which should be called at the end of use.
24 object CloseableIterator {

54   private def wrap[A](t: GenTraversableOnce[A]): CloseableIterator[A] = t match {
55     case c: CloseableIterator[A] => c

59   class CloseableIteratorImpl[A](iter: Iterator[A], closeIter: => Unit) extends CloseableIterator[A] {

85   private final class CloseableSingleIterator[A](elem: => A, closeIter: => Unit) extends CloseableIterator[A] {
86     private var result = true
87     override def hasNext: Boolean = result
88     override def next(): A = if (result) { result = false; elem } else { empty.next() }
89     override def close(): Unit = closeIter
90   }

92   /**
93     * Based on scala's ++ iterator implementation
94     *
95     * Avoid stack overflows when applying ++ to lots of iterators by flattening the unevaluated
96     * iterators out into a vector of closures.
97     */
98   private final class ConcatCloseableIterator[+A](queue: scala.collection.mutable.Queue[() => CloseableIterator[A]])
99       extends CloseableIterator[A] {
100 
101     private [this] var current: CloseableIterator[A] = queue.dequeue()()
102 
103     // Advance current to the next non-empty iterator
104     // current is set to empty when all iterators are exhausted
105     @tailrec
106     private [this] def advance(): Boolean = {
107       current.close()
108       if (queue.isEmpty) {
109         current = CloseableIterator.empty
110         false
111       } else {
112         current = queue.dequeue()()
113         current.hasNext || advance()
114       }
115     }
116 
117     override def hasNext: Boolean = current.hasNext || advance()
118     override def next(): A = current.next
119 
120     override def close(): Unit = {
121       current.close()
122       queue.foreach(_.apply().close())
123       queue.clear()
124     }
125   }
126 
127   private final class FlatMapCloseableIterator[A, B](source: CloseableIterator[A], f: A => GenTraversableOnce[B])
128       extends CloseableIterator[B] {
129 
130     private var cur: CloseableIterator[B] = empty
131 
132     @tailrec
133     override def hasNext: Boolean = cur.hasNext || {
134       cur.close()
135       if (source.hasNext) {
136         cur = CloseableIterator.wrap(f(source.next()))
137         hasNext
138       } else {
139         cur = empty
140         false
141       }
142     }
143 
144     override def next(): B = if (hasNext) { cur.next() } else { empty.next() }
145 
146     override def close(): Unit = CloseQuietly(cur, source).foreach(f => throw f)
147   }
148 }

158 
159   override def take(n: Int): CloseableIterator[A] = new CloseableIteratorImpl(super.take(n), close())
160 
161   override def takeWhile(p: A => Boolean): CloseableIterator[A] =
162     new CloseableIteratorImpl(super.takeWhile(p), close())
163 
164   override def collect[B](pf: PartialFunction[A, B]): CloseableIterator[B] =
165     new CloseableIteratorImpl(super.collect(pf), close())
166 
167   def concat[B >: A](xs: => GenTraversableOnce[B]): CloseableIterator[B] = {
168     lazy val applied = CloseableIterator.wrap(xs)
169     val queue = new scala.collection.mutable.Queue[() => CloseableIterator[B]]
170     queue.enqueue(() => this)
171     queue.enqueue(() => applied)
172     new ConcatCloseableIterator[B](queue)
173   }
174 
175   // in scala 2.13 this method is final, and can cause resource leaks due to not returning a closeable iterator
176   override def ++[B >: A](that: => GenTraversableOnce[B]): CloseableIterator[B] =
177     throw new NotImplementedError("Not safe for cross-scala usage")
178 
179   override def flatMap[B](f: A => GenTraversableOnce[B]): CloseableIterator[B] =
180     new FlatMapCloseableIterator(this, f)
