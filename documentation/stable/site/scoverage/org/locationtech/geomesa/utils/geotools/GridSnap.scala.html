16 /**
17   * Snaps points to cells of a defined grid
18   *
19   * @param env bounding envelope
20   * @param xSize number of x cells in the grid
21   * @param ySize number of y cells in the grid
22   */
23 class GridSnap(env: Envelope, xSize: Int, ySize: Int) {
24 
25   lazy val envelope: Geometry = GeometryUtils.geoFactory.toGeometry(env)
26 
27   private val xMin = env.getMinX
28   private val xMax = env.getMaxX
29   private val yMin = env.getMinY
30   private val yMax = env.getMaxY
31 
32   private val dx = (xMax - xMin) / xSize
33   private val dy = (yMax - yMin) / ySize
34 
35   private val xOffset = xMin + dx / 2 // offset by half dx to get to center of grid cell
36   private val yOffset = yMin + dy / 2 // offset by half dy to get to center of grid cell
37 
38   /**
39    * Computes the X ordinate of the i'th grid column.
40    *
41    * @param i the index of a grid column
42    * @return the X ordinate of the column
43    */
44   def x(i: Int): Double = xOffset + dx * i
45 
46   /**
47    * Computes the Y ordinate of the i'th grid row.
48    *
49    * @param j the index of a grid row
50    * @return the Y ordinate of the row
51    */
52   def y(j: Int): Double = yOffset + dy * j
53 
54   /**
55    * Computes the column index of an X ordinate.
56    *
57    * @param x the X ordinate
58    * @return the column index
59    */
60   def i(x: Double): Int = {
61     if (x < xMin || x > xMax) { -1 } else {
62       val i = math.floor((x - xMin) / dx).toInt
63       // i >= size check catches the upper bound
64       if (i >= xSize) { xSize - 1 } else { i }
65     }
66   }
67 
68   /**
69    * Computes the column index of an Y ordinate.
70    *
71    * @param y the Y ordinate
72    * @return the column index
73    */
74   def j(y: Double): Int = {
75     if (y < yMin || y > yMax) { -1 } else {
76       val i = math.floor((y - yMin) / dy).toInt
77       // i >= size check catches the upper bound
78       if (i >= ySize) { ySize - 1 } else { i }
79     }
80   }
81 
82   def snap(x: Double, y: Double): (Double, Double) = (this.x(i(x)), this.y(j(y)))
83 
84   /**
85     * Generate a sequence of snapped points between two given snapped coordinates using Bresenham's Line Algorithm.
86     * Will not return any duplicate points, and will always include the start and end points
87     *
88     * @param x0 x0
89     * @param y0 y0
90     * @param x1 x1
91     * @param y1 y1
92     * @return
93     */
94   def bresenhamLine(x0: Int, y0: Int, x1: Int, y1: Int): Iterator[(Int, Int)] = {
95     val deltaX = abs(x1 - x0)
96     val deltaY = abs(y1 - y0)
97     if (deltaX == 0 && deltaY == 0) { Iterator.single((x0, y0)) } else {
98       val stepX = if (x0 < x1) { 1 } else { -1 }
99       val stepY = if (y0 < y1) { 1 } else { -1 }
100       if (deltaX > deltaY) {
101         val deltaError = deltaY.toDouble / deltaX
102         var error = 0d
103         Iterator.iterate((x0, y0)) { case (x, y) =>
104           error += deltaError
105           if (error >= 0.5) {
106             error -= 1d
107             (x + stepX, y + stepY)
108           } else {
109             (x + stepX, y)
110           }
111         }.take(deltaX)
112       } else {
113         val deltaError = deltaX.toDouble / deltaY
114         var error = 0d
115         Iterator.iterate((x0, y0)) { case (x, y) =>
116           error += deltaError
117           if (error >= 0.5) {
118             error -= 1d
119             (x + stepX, y + stepY)
120           } else {
121             (x, y + stepY)
122           }
123         }.take(deltaY)
124       }
125     }
126   }
127 }
