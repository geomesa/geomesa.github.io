36 /**
37   * HBase filter for CQL predicates and transformations
38   *
39   * The internal processing logic is kept in a delegate filter that will do either filtering, transforming, or both.
40   *
41   * @param delegate delegate filter
42   */
43 class CqlTransformFilter(delegate: DelegateFilter) extends FilterBase {
44 
45   /**
46     * From the Filter javadocs:
47     *
48     * A filter can expect the following call sequence:
49     *
50     * <ul>
51     * <li> `reset()` : reset the filter state before filtering a new row. </li>
52     * <li> `filterAllRemaining()`: true means row scan is over; false means keep going. </li>
53     * <li> `filterRowKey(byte[],int,int)`: true means drop this row; false means include.</li>
54     * <li> `filterKeyValue(Cell)`: decides whether to include or exclude this KeyValue. See `ReturnCode`. </li>
55     * <li> `transform(KeyValue)`: if the KeyValue is included, let the filter transform the KeyValue. </li>
56     * <li> `filterRowCells(List)`: allows direct modification of the final list to be submitted
57     * <li> `filterRow()`: last chance to drop entire row based on the sequence of
58     *   filter calls. Eg: filter a row if it doesn't contain a specified column. </li>
59     * </ul>
60     */
61 
62   override def filterKeyValue(v: Cell): ReturnCode = delegate.filterKeyValue(v)
63   override def transformCell(v: Cell): Cell = delegate.transformCell(v)
64   override def toByteArray: Array[Byte] = CqlTransformFilter.serialize(delegate)
65   override def toString: String = delegate.toString
66 }

84     * Create a new filter. Typically, filters created by this method will just be serialized to bytes and sent
85     * as part of an HBase scan
86     *
87     * Note: calling this without at least one of the cql filter or transform would introduce extra processing
88     * but not actually do anything, and thus is not allowed
89     *
90     * @param sft simple feature type
91     * @param filter cql filter, if any
92     * @param transform transform, if any
93     * @param hints is used to get sampling options
94     * @return

96   def apply(
97       sft: SimpleFeatureType,
98       index: GeoMesaFeatureIndex[_, _],
99       filter: Option[Filter],
100       transform: Option[(String, SimpleFeatureType)],
101       hints: Hints): CqlTransformFilter = {
102     if (filter.isEmpty && transform.isEmpty && hints.getSampling.isEmpty) {
103       throw new IllegalArgumentException("The filter must have a predicate and/or transform")
104     }
105 
106     val feature = KryoFeatureSerializer(sft, SerializationOptions.withoutId).getReusableFeature
107     feature.setIdParser(index.getIdFromRow(_, _, _, null))
108     transform.foreach { case (tdefs, tsft) => feature.setTransforms(tdefs, tsft) }
109 
110     val samplingOpts: Option[(Float, Option[String])] = hints.getSampling
111 
112     val delegate = (filter, transform, samplingOpts) match {
113       case (None, None, Some(_))    => new FilterDelegate(sft, index, feature, Filter.INCLUDE, samplingOpts)
114       case (None, Some(_), Some(_)) => new FilterTransformDelegate(sft, index, feature, Filter.INCLUDE, samplingOpts)
115       case (Some(f), None , _)      => new FilterDelegate(sft, index, feature, f, samplingOpts)
116       case (Some(f), Some(_), _)    => new FilterTransformDelegate(sft, index, feature, f, samplingOpts)
117       case (None, Some(_), _)       => new TransformDelegate(sft, index, feature, samplingOpts)
118     }
119 
120     // noinspection ScalaDeprecation
121     hints.getFilterCompatibility match {
122       case None => new CqlTransformFilter(delegate)
123       case Some(FilterCompatibility.`2.3`) => new JSimpleFeatureFilter(delegate)
124       case Some(c) => throw new NotImplementedError(s"Filter compatibility $c is not implemented for this query")
125     }

288   /**
289     * Deserialize a new delegate instance
290     *
291     * @param bytes serialized bytes
292     * @throws org.apache.hadoop.hbase.exceptions.DeserializationException if anything goes wrong
293     * @return
294     */
295   @throws(classOf[DeserializationException])
296   private def deserialize(bytes: Array[Byte]): DelegateFilter = {
297     try {
298       var offset = 0
299       val sftLength = ByteArrays.readInt(bytes, offset)
300       offset += 4
301       val spec = new String(bytes, offset, sftLength,StandardCharsets.UTF_8)
302       offset += sftLength
303 
304       val sft = IteratorCache.sft(spec)
305       val feature = IteratorCache.serializer(spec, SerializationOptions.withoutId).getReusableFeature
306 
307       val cqlLength = ByteArrays.readInt(bytes, offset)
308       offset += 4
309       val cql = if (cqlLength == 0) { null } else {
310         IteratorCache.filter(sft, spec, new String(bytes, offset, cqlLength,StandardCharsets.UTF_8))
311       }
312       offset += cqlLength
313 
314       val tdefsLength = ByteArrays.readInt(bytes, offset)
315 
316       if (tdefsLength == -1) {
317         if (cql == null) {
318           throw new DeserializationException("No filter or transform defined")
319         } else {
320           val (index,newOffset) = deserializeIndex(sft, spec, bytes, offset + 4)
321           offset = newOffset
322           feature.setIdParser(index.getIdFromRow(_, _, _, null))
323 
324           val samplingOptions = deserializeSamplingOptions(bytes,offset)
325 
326           new FilterDelegate(sft, index, feature, cql,samplingOptions)
327         }
328       } else {
329         offset += 4
330         val tdefs = new String(bytes, offset, tdefsLength,StandardCharsets.UTF_8)
331         offset += tdefsLength
332 
333         val tsftLength = ByteArrays.readInt(bytes, offset)
334         offset += 4
335         val tsft = IteratorCache.sft(new String(bytes, offset, tsftLength,StandardCharsets.UTF_8))
336 
337         feature.setTransforms(tdefs, tsft)
338 
339         val (index,newOffset) = deserializeIndex(sft, spec, bytes, offset + tsftLength)
340         offset = newOffset
341         feature.setIdParser(index.getIdFromRow(_, _, _, null))
342 
343         val samplingOptions = deserializeSamplingOptions(bytes,offset)
344 
345         if (cql == null) {
346           new TransformDelegate(sft, index, feature,samplingOptions)
347         } else {
348           new FilterTransformDelegate(sft, index, feature, cql,samplingOptions)
349         }
350       }
351     } catch {
352       case e: DeserializationException => throw e
353       case NonFatal(e) => throw new DeserializationException("Error deserializing filter", e)
354     }
355   }

365                                 bytes: Array[Byte],
366                                 start: Int) : Option[(Float,Option[String])]={
367 
368     var samplingOption: Option[(Float,Option[String])]= Option.empty
369 
370     var offset = start
371     val factorLength = ByteArrays.readInt(bytes, offset)
372     offset += 4
373     if (factorLength != 0) {
374       val samplingFactor = ByteBuffer.wrap(bytes,offset,factorLength).getFloat
375       offset+=factorLength
376       val fieldNameLength = ByteArrays.readInt(bytes, offset)
377       offset+=4
378       if (fieldNameLength!=0){
379         val fieldName = new String(bytes, offset, fieldNameLength,StandardCharsets.UTF_8)
380         samplingOption = Option.apply((samplingFactor,Option.apply(fieldName)))
381       }
382       else {

384       }
385     }
