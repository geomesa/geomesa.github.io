24 /**
25   * Age-off a feature based on an attribute time
26   */
27 trait DtgAgeOffFilter extends AgeOffFilter with LazyLogging {
28 
29   protected var sft: SimpleFeatureType = _
30   protected var index: GeoMesaFeatureIndex[_, _] = _
31 
32   protected var reusableSf: KryoBufferSimpleFeature = _
33   protected var dtgIndex: Int = -1
34 
35   override def init(options: Map[String, String]): Unit = {
36     import DtgAgeOffFilter.Configuration.{DtgOpt, IndexOpt, IndexSftOpt, SftOpt}
37 
38     super.init(options)
39 
40     val spec = options(SftOpt)
41     sft = IteratorCache.sft(spec)
42     index = options.get(IndexSftOpt) match {
43       case None => IteratorCache.index(sft, spec, options(IndexOpt))
44       case Some(ispec) => IteratorCache.index(IteratorCache.sft(ispec), ispec, options(IndexOpt))
45     }
46 
47     // noinspection ScalaDeprecation
48     val withId = if (index.serializedWithId) { SerializationOptions.none } else { SerializationOptions.withoutId }
49     reusableSf = IteratorCache.serializer(spec, withId).getReusableFeature
50     dtgIndex = options(DtgOpt).toInt // note: keep this last, for back-compatibility with DtgAgeOffIterator
51   }
52 
53   override def accept(row: Array[Byte],
54                       rowOffset: Int,
55                       rowLength: Int,
56                       value: Array[Byte],
57                       valueOffset: Int,
58                       valueLength: Int,
59                       timestamp: Long): Boolean = {
60     try {
61       reusableSf.setBuffer(value, valueOffset, valueLength)
62       reusableSf.getDateAsLong(dtgIndex) > expiry
63     } catch {
64       case NonFatal(e) =>
65         logger.error(s"Error checking age-off for " +
66             Option(value).getOrElse(Array.empty).mkString("[", ",", s"], offset: $valueOffset, length: $valueLength"))
67         false
68     }
69   }
70 }

73

75   object Configuration {

77     val IndexOpt    = "index"
78     val IndexSftOpt = "index-sft"
79     val DtgOpt      = "dtg"
80   }

82   def configure(sft: SimpleFeatureType,
83                 index: GeoMesaFeatureIndex[_, _],
84                 expiry: scala.concurrent.duration.Duration,
85                 dtgField: Option[String]): Map[String, String] = {
86     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
87 
88     require(!sft.isTableSharing || SystemProperty("geomesa.age-off.override").option.exists(_.toBoolean),
89       "DtgAgeOff filter should only be applied to features that don't use table sharing. You may override this check" +
90           "by setting the system property 'geomesa.age-off.override=true', however please note that age-off" +
91           "will affect all shared feature types in the same catalog, and may not work correctly with multiple feature" +
92           "types")
93 
94     val dtgIndex = dtgField match {
95       case None =>
96         sft.getDtgIndex.getOrElse {
97           throw new IllegalArgumentException("Simple feature type does not have a valid date field")
98         }
99       case Some(dtg) =>
100         val i = sft.indexOf(dtg)
101         if (i == -1 || !classOf[Date].isAssignableFrom(sft.getDescriptor(i).getType.getBinding)) {
102           throw new IllegalArgumentException(s"Simple feature type does not have a valid date field '$dtg'")
103         }
104         i
105     }
106 
107     val indexSftOpt = Some(index.sft).collect {
108       case s if s != sft => SimpleFeatureTypes.encodeType(s, includeUserData = true)
109     }
110 
111     AgeOffFilter.configure(sft, expiry) ++ Map (
112       Configuration.SftOpt   -> SimpleFeatureTypes.encodeType(sft),
113       Configuration.IndexOpt -> index.identifier,
114       Configuration.DtgOpt   -> dtgIndex.toString
115     ) ++ optionalMap(Configuration.IndexSftOpt -> indexSftOpt)
116   }
