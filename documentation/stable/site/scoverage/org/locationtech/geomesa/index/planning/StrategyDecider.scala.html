56   /**
57     * Selects a strategy for executing a given query.
58     *
59     * If a particular strategy has been requested, that strategy will be used (note - this is only
60     * partially supported, and should be used with care.)
61     *
62     * Otherwise, the query will be examined for strategies that could be used to execute it. The cost of
63     * executing each available strategy will be calculated, and the least expensive strategy will be used.
64     *
65     * @param ds data store
66     * @param sft simple feature type
67     * @param filter filter to execute
68     * @param transform return transformation
69     * @param requested requested index
70     * @param explain for trace logging
71     * @return
72     */
73   def getFilterPlan[DS <: GeoMesaDataStore[DS]](ds: DS,
74                                                 sft: SimpleFeatureType,
75                                                 filter: Filter,
76                                                 transform: Option[SimpleFeatureType],
77                                                 evaluation: CostEvaluation,
78                                                 requested: Option[String],
79                                                 explain: Explainer = ExplainNull): Seq[FilterStrategy] = {
80 
81     def complete(op: String, time: Long, count: Int): Unit = explain(s"$op took ${time}ms for $count options")
82 
83     val indices = ds.manager.indices(sft, mode = IndexMode.Read)
84 
85     // get the various options that we could potentially use
86     val options = profile((o: Seq[FilterPlan], t: Long) => complete("Query processing", t, o.length)) {
87       new FilterSplitter(sft, indices).getQueryOptions(filter, transform, requested)
88     }
89 
90     val selected = profile(t => complete("Strategy selection", t, options.length)) {
91       if (requested.isDefined) {
92         val forced = matchRequested(requested.get, indices, options, filter)
93         explain(s"Filter plan forced to $forced")
94         forced
95       } else if (options.isEmpty) {
96         // corresponds to filter.exclude
97         // we still need to return something so that we can handle reduce steps, if needed
98         explain("No filter plans found - creating empty plan")
99         FilterPlan(Seq(FilterStrategy(new EmptyIndex(ds, sft), Some(Filter.EXCLUDE), None, temporal = false, 1f)))
100       } else if (options.lengthCompare(1) == 0) {
101         // only a single option, so don't bother with cost
102         explain(s"Filter plan: ${options.head}")
103         options.head
104       } else {
105         // choose the best option based on cost
106         val stats = evaluation match {
107           case CostEvaluation.Stats => Some(ds.stats)
108           case CostEvaluation.Index => None
109         }
110         decider.selectFilterPlan(sft, options, stats, explain)
111       }
112     }
113 
114     selected.strategies
115   }
116 
117   private def matchRequested(id: String,
118                              indices: Seq[GeoMesaFeatureIndex[_, _]],
119                              options: Seq[FilterPlan],
120                              filter: Filter): FilterPlan = {
121     // see if one of the normal plans matches the requested type - if not, force it
122     def byId: Option[FilterPlan] = options.find(_.strategies.forall(_.index.identifier.equalsIgnoreCase(id)))
123     def byName: Option[FilterPlan] = options.find(_.strategies.forall(_.index.name.equalsIgnoreCase(id)))
124     // back-compatibility for attr vs join index name
125     def byJoin: Option[FilterPlan] = if (!AttributeIndex.name.equalsIgnoreCase(id)) { None } else {
126       options.find(_.strategies.forall(_.index.name == AttributeIndex.JoinIndexName))
127     }
128 
129     def fallback: FilterPlan = {
130       val index = indices.find(_.identifier.equalsIgnoreCase(id))
131           .orElse(indices.find(_.name.equalsIgnoreCase(id)))
132           .getOrElse {
133             throw new IllegalArgumentException(s"Invalid index strategy: $id. Valid values are " +
134                 indices.map(i => s"${i.name}, ${i.identifier}").mkString(", "))
135           }
136       val secondary = if (filter == Filter.INCLUDE) { None } else { Some(filter) }
137       FilterPlan(Seq(FilterStrategy(index, None, secondary, temporal = false, Float.PositiveInfinity)))
138     }
139 
140     byId.orElse(byName).orElse(byJoin).getOrElse(fallback)
141   }

143   class CostBasedStrategyDecider extends StrategyDecider with MethodProfiling {
144 
145     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
146 
147     override def selectFilterPlan(
148         sft: SimpleFeatureType,
149         options: Seq[FilterPlan],
150         stats: Option[GeoMesaStats],
151         explain: Explainer): FilterPlan = {
152 
153       def cost(option: FilterPlan): FilterPlanCost = {
154         profile((fpc: FilterPlanCost, time: Long) => fpc.copy(time = time)) {
155           var cost = BigInt(0)
156           option.strategies.foreach { strategy =>
157             val filter = strategy.primary.getOrElse(Filter.INCLUDE)
158             val count = stats.flatMap(_.getCount(sft, filter, exact = false)).getOrElse(100L)
159             cost = cost + BigInt((count * strategy.costMultiplier).toLong)
160           }
161           FilterPlanCost(option, if (cost.isValidLong) { cost.longValue } else { Long.MaxValue }, 0L)
162         }
163       }
164 
165       val costs = options.map(cost).sorted
166 
167       val temporal = if (!sft.isTemporalPriority) { None } else {
168         costs.find(c => c.plan.strategies.nonEmpty && c.plan.strategies.forall(_.temporal))
169       }
170       val selected = temporal.getOrElse(costs.head)
171       explain(s"Filter plan selected: $selected")
172       explain(s"Filter plans not selected: ${costs.filterNot(_.eq(selected)).mkString(", ")}")
173       selected.plan
174     }
175 
176     private case class FilterPlanCost(plan: FilterPlan, cost: Long, time: Long) extends Comparable[FilterPlanCost] {
177       override def compareTo(o: FilterPlanCost): Int = java.lang.Long.compare(cost, o.cost)
178       override def toString: String = s"$plan (Cost $cost in ${time}ms)"
179     }
180   }
