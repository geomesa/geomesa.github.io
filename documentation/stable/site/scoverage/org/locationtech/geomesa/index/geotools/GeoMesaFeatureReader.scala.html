27 class GeoMesaFeatureReader(result: QueryResult) {
28 
29   def schema: SimpleFeatureType = result.schema
30   def hints: Hints = result.hints
31 
32   def reader(): SimpleFeatureReader = new ResultReader()
33 
34   private class ResultReader extends SimpleFeatureReader {
35 
36     private val iter = result.iterator()
37 
38     override def getFeatureType: SimpleFeatureType = result.schema
39     override def hasNext: Boolean = iter.hasNext
40     override def next(): SimpleFeature = iter.next()
41     override def close(): Unit = iter.close()
42   }
43 }

45 object GeoMesaFeatureReader extends MethodProfiling {
46 
47   def apply(
48       sft: SimpleFeatureType,
49       query: Query,
50       qp: QueryRunner,
51       audit: Option[(AuditWriter, AuditProvider, String)]): GeoMesaFeatureReader = {
52     audit match {
53       case None => new GeoMesaFeatureReader(qp.runQuery(sft, query))
54       case Some((w, p, t)) =>
55         val timings = new TimingsImpl()
56         val result = profile(time => timings.occurrence("planning", time))(qp.runQuery(sft, query))
57         new GeoMesaFeatureReaderWithAudit(result, timings, w, p, t, sft.getTypeName, query.getFilter)
58     }
59   }

78   class GeoMesaFeatureReaderWithAudit(
79       result: QueryResult,
80       timings: Timings,
81       auditWriter: AuditWriter,
82       auditProvider: AuditProvider,
83       storeType: String,
84       typeName: String,
85       filter: Filter) extends GeoMesaFeatureReader(result) {
86 
87     override def reader(): SimpleFeatureReader = new ResultReaderWithAudit()
88 
89     private class ResultReaderWithAudit extends SimpleFeatureReader with MethodProfiling {
90 
91       private val closed = new AtomicBoolean(false)
92       private val count = new AtomicLong(0L)
93       private val iter = profile(time => timings.occurrence("planning", time)) {
94         val base = result.iterator()
95         // note: has to be done after running the query so hints are set
96         if (result.schema == BinaryOutputEncoder.BinEncodedSft) {
97           // bin queries pack multiple records into each feature
98           // to count the records, we have to count the total bytes coming back, instead of the number of features
99           val bytesPerHit = if (result.hints.getBinLabelField.isDefined) { 24 } else { 16 }
100           base.map { sf => count.addAndGet(sf.getAttribute(0).asInstanceOf[Array[Byte]].length / bytesPerHit); sf }
101         } else {
102           base.map { sf => count.incrementAndGet(); sf }
103         }
104       }
105 
106       override def getFeatureType: SimpleFeatureType = result.schema
107 
108       override def hasNext: Boolean = profile(time => timings.occurrence("hasNext", time))(iter.hasNext)
109       override def next(): SimpleFeature = profile(time => timings.occurrence("next", time))(iter.next())
110 
111       override def close(): Unit = {
112         if (closed.compareAndSet(false, true)) {
113           try { iter.close() } finally {
114             val stat = QueryEvent(
115               storeType,
116               typeName,
117               System.currentTimeMillis(),
118               auditProvider.getCurrentUserId,
119               filterToString(filter),
120               ViewParams.getReadableHints(result.hints),
121               timings.time("planning"),
122               timings.time("next") + timings.time("hasNext"),
123               count.get
124             )
125             auditWriter.writeEvent(stat) // note: implementations should be asynchronous
126           }
127         }
128       }
129     }
130   }
