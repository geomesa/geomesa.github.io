42   /**
43     * Plan the query, but don't execute it - used for m/r jobs and explain query
44     *
45     * @param sft simple feature type
46     * @param query query to plan
47     * @param index override index to use for executing the query
48     * @param output planning explanation output
49     * @return
50     */
51   def planQuery(
52       sft: SimpleFeatureType,
53       query: Query,
54       index: Option[String] = None,
55       output: Explainer = new ExplainLogging): Seq[QueryPlan[DS]] =
56     getQueryPlans(sft, configureQuery(sft, query), query.getFilter, index, output).toList // toList forces evaluation of entire iterator

58   override def runQuery(sft: SimpleFeatureType, original: Query, explain: Explainer): QueryResult = {
59     val query = configureQuery(sft, original)
60     val plans = getQueryPlans(sft, query, original.getFilter, None, explain)
61     QueryResult(query.getHints.getReturnSft, query.getHints, run(query, plans))
62   }

64   private def run(query: Query, plans: Seq[QueryPlan[DS]])(): CloseableIterator[SimpleFeature] = {
65     var iterator = SelfClosingIterator(plans.iterator).flatMap(p => p.scan(ds).map(p.resultsToFeatures.apply))
66 
67     if (!query.getHints.isSkipReduce) {
68       plans.headOption.flatMap(_.reducer).foreach { reducer =>
69         require(plans.tail.forall(_.reducer.contains(reducer)), "Reduce must be the same in all query plans")
70         iterator = reducer.apply(iterator)
71       }
72     }
73 
74     plans.headOption.flatMap(_.sort).foreach { sort =>
75       require(plans.tail.forall(_.sort.contains(sort)), "Sort must be the same in all query plans")
76       iterator = new SortingSimpleFeatureIterator(iterator, sort)
77     }
78 
79     plans.headOption.flatMap(_.maxFeatures).foreach { maxFeatures =>
80       require(plans.tail.forall(_.maxFeatures.contains(maxFeatures)),
81         "Max features must be the same in all query plans")
82       if (query.getHints.getReturnSft == BinaryOutputEncoder.BinEncodedSft) {
83         // bin queries pack multiple records into each feature
84         // to count the records, we have to count the total bytes coming back, instead of the number of features
85         val label = query.getHints.getBinLabelField.isDefined
86         iterator = new BinaryOutputEncoder.FeatureLimitingIterator(iterator, maxFeatures, label)
87       } else {
88         iterator = iterator.take(maxFeatures)
89       }
90     }
91 
92     plans.headOption.flatMap(_.projection).foreach { projection =>
93       require(plans.tail.forall(_.projection.contains(projection)), "Projection must be the same in all query plans")
94       val project = Reprojection(query.getHints.getReturnSft, projection)
95       iterator = iterator.map(project.apply)
96     }
97 
98     // wrap in an exceptional iterator to prevent geoserver from suppressing any exceptions
99     ExceptionalIterator(iterator)
100   }

102   /**
103     * Set up the query plans and strategies used to execute them
104     *
105     * @param sft simple feature type
106     * @param query query to plan - must have been run through `configureQuery` to set expected hints, etc
107     * @param requested override index to use for executing the query
108     * @param output planning explanation output
109     * @return
110     */
111   private def getQueryPlans(
112       sft: SimpleFeatureType,
113       query: Query,
114       original: Filter,
115       requested: Option[String],
116       output: Explainer): Seq[QueryPlan[DS]] = {
117     import org.locationtech.geomesa.filter.filterToString
118 
119     profile(time => output(s"Query planning took ${time}ms")) {
120       val hints = query.getHints
121 
122       output.pushLevel(s"Planning '${query.getTypeName}' ${filterToString(query.getFilter)}")
123       output(s"Original filter: ${filterToString(original)}")
124       output(s"Hints: bin[${hints.isBinQuery}] arrow[${hints.isArrowQuery}] density[${hints.isDensityQuery}] " +
125           s"stats[${hints.isStatsQuery}] " +
126           s"sampling[${hints.getSampling.map { case (s, f) => s"$s${f.map(":" + _).getOrElse("")}"}.getOrElse("none")}]")
127       output(s"Sort: ${query.getHints.getSortFields.map(QueryHints.sortReadableString).getOrElse("none")}")
128       output(s"Transforms: ${query.getHints.getTransformDefinition.map(t => if (t.isEmpty) { "empty" } else { t }).getOrElse("none")}")
129       output(s"Max features: ${query.getHints.getMaxFeatures.getOrElse("none")}")
130       hints.getFilterCompatibility.foreach(c => output(s"Filter compatibility: $c"))
131 
132       output.pushLevel("Strategy selection:")
133       val requestedIndex = requested.orElse(hints.getRequestedIndex)
134       val transform = query.getHints.getTransformSchema
135       val evaluation = query.getHints.getCostEvaluation
136       val strategies =
137         StrategyDecider.getFilterPlan(ds, sft, query.getFilter, transform, evaluation, requestedIndex, output)
138       output.popLevel()
139 
140       var strategyCount = 1
141       strategies.map { strategy =>
142         def complete(plan: QueryPlan[DS], time: Long): Unit = {
143           plan.explain(output)
144           output(s"Plan creation took ${time}ms").popLevel()
145         }
146 
147         output.pushLevel(s"Strategy $strategyCount of ${strategies.length}: ${strategy.index}")
148         strategyCount += 1
149         output(s"Strategy filter: $strategy")
150         profile(complete _) {
151           val qs = strategy.getQueryStrategy(hints, output)
152           // query guard hook
153           interceptors(sft).foreach(_.guard(qs).foreach(e => throw e))
154           ds.adapter.createQueryPlan(qs)
155         }
156       }
157     }
158   }

194   /**
195    * Extract and parse transforms from the query
196    *
197    * @param sft simple feature type
198    * @param query query
199    * @return
200    */
201   def extractQueryTransforms(
202       sft: SimpleFeatureType,
203       query: Query): Option[(SimpleFeatureType, Seq[Transform], String)] = {
204     // even if a transform is not specified, some queries only use a subset of attributes
205     // specify them here so that it's easier to pick the best column group later
206     def fromQueryType: Option[Seq[String]] = {
207       val hints = query.getHints
208       if (hints.isBinQuery) {
209         Some(BinAggregatingScan.propertyNames(hints, sft))
210       } else if (hints.isDensityQuery) {
211         Some(DensityScan.propertyNames(hints, sft))
212       } else if (hints.isStatsQuery) {
213         Some(StatParser.propertyNames(sft, hints.getStatsQuery))
214       } else {
215         None
216       }
217     }
218 
219     // since we do sorting on the client, just add any sort-by attributes to the transform
220     // TODO GEOMESA-2655 we should sort and then transform back to the requested props, but it's complicated...
221     def withSort(props: Array[String]): Seq[String] = {
222       val names = props.map(p => if (p.contains('=')) { p.substring(0, p.indexOf('=')) } else { p })
223       props ++ Option(query.getSortBy).toSeq.flatMap { sort =>
224         sort.flatMap(s => Option(s.getPropertyName).flatMap(p => Option(p.getPropertyName).filterNot(names.contains)))
225       }
226     }
227 
228     // ignore transforms that don't actually do anything
229     def noop(props: Seq[String]): Boolean = props == sft.getAttributeDescriptors.asScala.map(_.getLocalName)
230 
231     Option(query.getPropertyNames).map(withSort).filterNot(noop).orElse(fromQueryType).map { props =>
232       val transforms = Transforms(sft, props)
233       val schema = Transforms.schema(sft, transforms)
234       val definition = props.mkString(Transform.DefinitionDelimiter)
235       (schema, transforms, definition)
236     }
237   }

239   /**
240     * Sets query hints for sorting
241     *
242     * @param sft sft
243     * @param query query
244     */
245   def setQuerySort(sft: SimpleFeatureType, query: Query): Unit = {
246     val sortBy = query.getSortBy
247     if (sortBy != null && sortBy.nonEmpty) {
248       val hints = query.getHints
249       if (hints.isArrowQuery) {
250         if (sortBy.lengthCompare(1) > 0) {
251           throw new IllegalArgumentException("Arrow queries only support sort by a single field: " +
252               sortBy.mkString(", "))
253         } else if (sortBy.head.getPropertyName == null) {
254           throw new IllegalArgumentException("Arrow queries only support sort by properties: " +
255               sortBy.mkString(", "))
256         }
257         val field = sortBy.head.getPropertyName.getPropertyName
258         val reverse = sortBy.head.getSortOrder == SortOrder.DESCENDING
259         hints.getArrowSort.foreach { case (f, r) =>
260           if (f != field || r != reverse) {
261             throw new IllegalArgumentException(s"Query sort does not match Arrow hints sort: " +
262                 s"${sortBy.mkString(", ")} != $f:${if (r) "DESC" else "ASC"}")
263           }
264         }
265         hints.put(QueryHints.ARROW_SORT_FIELD, field)
266         hints.put(QueryHints.ARROW_SORT_REVERSE, reverse)
267       } else if (hints.isBinQuery) {
268         val dtg = hints.getBinDtgField.orElse(sft.getDtgField).orNull
269         if (dtg == null ||
270             sortBy.map(s => Option(s.getPropertyName).map(_.getPropertyName).orNull).toSeq != Seq(dtg)) {
271           throw new IllegalArgumentException("BIN queries only support sort by a date-type field: " +
272               sortBy.mkString(", "))
273         }
274         if (sortBy.head.getSortOrder == SortOrder.DESCENDING) {
275           throw new IllegalArgumentException("BIN queries only support sort in ASCENDING order: " +
276               sortBy.mkString(", "))
277         }
278         if (hints.get(QueryHints.BIN_SORT) != null && !hints.isBinSorting) {
279           throw new IllegalArgumentException("Query sort order contradicts BIN sorting hint: " +
280               sortBy.mkString(", "))
281         }
282         hints.put(QueryHints.BIN_SORT, java.lang.Boolean.TRUE)
283       } else {
284         hints.put(QueryHints.Internal.SORT_FIELDS, QueryHints.Internal.toSortHint(sortBy))
285       }
286     }
287   }
