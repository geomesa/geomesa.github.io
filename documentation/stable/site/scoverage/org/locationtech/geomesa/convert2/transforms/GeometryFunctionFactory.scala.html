119   private val projectFromParserFn: TransformerFunction = new NamedTransformerFunction(Seq("projectFrom"), pure = true) {
120 
121     import scala.collection.JavaConverters._
122 
123     private val cache = new ConcurrentHashMap[String, MathTransform].asScala
124 
125     override def apply(args: Array[AnyRef]): AnyRef = {
126       import org.locationtech.geomesa.utils.geotools.CRS_EPSG_4326
127 
128       val geom = args(1).asInstanceOf[Geometry]
129       if (geom == null) { null } else {
130         val epsg = args(0).asInstanceOf[String]
131         val lenient = if (args.length > 2) { java.lang.Boolean.parseBoolean(args(2).toString) } else { true }
132         // transforms should be thread safe according to https://sourceforge.net/p/geotools/mailman/message/32123017/
133         val transform = cache.getOrElseUpdate(s"$epsg:$lenient",
134           CRS.findMathTransform(CRS.decode(epsg), CRS_EPSG_4326, lenient))
135         JTS.transform(geom, transform)
136       }
137     }
138   }
