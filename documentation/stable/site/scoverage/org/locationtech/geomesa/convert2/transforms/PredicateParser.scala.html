21   @throws(classOf[ParsingException])
22   def parse(predicate: String, report: Boolean = true): Predicate = {
23     logger.trace(s"Parsing predicate: $predicate")
24     if (predicate == null) {
25       throw new IllegalArgumentException("Invalid predicate: null")
26     }
27     val runner = if (report) { ReportingParseRunner(Parser.predicate) } else { BasicParseRunner(Parser.predicate) }
28     val parsing = runner.run(predicate)
29     parsing.result.getOrElse(throw new ParsingException(s"Invalid predicate: ${ErrorUtils.printParseErrors(parsing)}"))
30   }
