17 /**
18  * Wrapper for SFT and Config parsing that throws ParameterExceptions for JCommander
19  */
20 object CLArgResolver {
21 
22   /**
23    * @throws ParameterException if the SFT cannot be parsed
24    * @return the SFT parsed from the Args
25    */
26   @throws[ParameterException]
27   def getSft(specArg: String, featureName: String = null): SimpleFeatureType = {
28       SftArgResolver.getArg(SftArgs(specArg, featureName)) match {
29         case Right(sft) => sft
30         case Left(e)    => throw new ParameterException(e)
31       }
32     }
33 
34   /**
35    * @throws ParameterException if the config cannot be parsed
36    * @return the converter config parsed from the args
37    */
38   @throws[ParameterException]
39   def getConfig(configArg: String): Config =
40     ConverterConfigResolver.getArg(ConfArgs(configArg)) match {
41       case Right(config) => config
42       case Left(e)       => throw new ParameterException(e)
43     }
44 }
