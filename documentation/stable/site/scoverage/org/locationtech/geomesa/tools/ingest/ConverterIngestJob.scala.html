27 /**
28  * Distributed job that uses converters to process input files
29  *
30  * @param dsParams data store connection
31  * @param sft simple feature type
32  * @param converterConfig converter definition
33  * @param paths input files
34  * @param libjarsFiles lib jars
35  * @param libjarsPaths lib jars paths
36  */
37 class ConverterIngestJob(
38     dsParams: Map[String, String],
39     sft: SimpleFeatureType,
40     converterConfig: Config,
41     paths: Seq[String],
42     libjarsFiles: Seq[String],
43     libjarsPaths: Iterator[() => Seq[File]]
44   ) extends Awaitable with JobWithLibJars {
45 
46   import ConverterInputFormat.ConverterCounters
47   import GeoMesaOutputFormat.OutputCounters
48 
49   private val failCounters =
50     Seq((ConverterCounters.Group, ConverterCounters.Failed), (OutputCounters.Group, OutputCounters.Failed))
51 
52   private val job = {
53     val job = Job.getInstance(new Configuration, "GeoMesa Tools Ingest")
54     setLibJars(job, libjarsFiles, libjarsPaths)
55     configureJob(job)
56     JobRunner.submit(job)
57     job
58   }
59 
60   protected def mapCounters(job: Job): Seq[(String, Long)] = {
61     Seq(
62       (IngestCounters.Ingested, job.getCounters.findCounter(OutputCounters.Group, OutputCounters.Written).getValue),
63       (IngestCounters.Failed, failCounters.map(c => job.getCounters.findCounter(c._1, c._2).getValue).sum)
64     )
65   }
66 
67   protected def reduceCounters(job: Job): Seq[(String, Long)] = Seq.empty
68 
69   override def await(reporter: StatusCallback): JobResult =
70     JobRunner.monitor(job, reporter, mapCounters(job), reduceCounters(job))
71 
72   def configureJob(job: Job): Unit = {
73     job.setJarByClass(getClass)
74     job.setMapperClass(classOf[ConverterIngestJob.IngestMapper])
75     job.setInputFormatClass(classOf[ConverterInputFormat])
76     job.setOutputFormatClass(classOf[GeoMesaOutputFormat])
77     job.setMapOutputKeyClass(classOf[Text])
78     job.setOutputValueClass(classOf[ScalaSimpleFeature])
79     job.setNumReduceTasks(0)
80     job.getConfiguration.set("mapred.map.tasks.speculative.execution", "false")
81     job.getConfiguration.set("mapred.reduce.tasks.speculative.execution", "false")
82     job.getConfiguration.set("mapreduce.job.user.classpath.first", "true")
83 
84     FileInputFormat.setInputPaths(job, paths.mkString(","))
85     ConverterInputFormat.setConverterConfig(job, converterConfig.root().render(ConfigRenderOptions.concise()))
86     ConverterInputFormat.setSft(job, sft)
87 
88     GeoMesaOutputFormat.setOutput(job.getConfiguration, dsParams, sft)
89   }
90 }

94   /**
95    * Takes the input and writes it to the output - all our main work is done in the input format
96    */
97   class IngestMapper extends Mapper[LongWritable, SimpleFeature, Text, SimpleFeature] with LazyLogging {
98 
99     type Context = Mapper[LongWritable, SimpleFeature, Text, SimpleFeature]#Context
100 
101     private val text: Text = new Text
102 
103     override def map(key: LongWritable, sf: SimpleFeature, context: Context): Unit = {
104       logger.debug(s"map key ${key.toString}, map value ${DataUtilities.encodeFeature(sf)}")
105       context.write(text, sf)
106     }
107   }
