49   /**
50     * Gets splits for a job.
51     *
52     * Our delegated AccumuloInputFormat creates a split for each range - because we set a lot of ranges in
53     * geomesa, that creates too many mappers. Instead, we try to group the ranges by tservers. We use the
54     * location assignment of the tablets to tservers to determine the number of splits returned.
55     */
56   override def getSplits(context: JobContext): java.util.List[InputSplit] = {
57     val accumuloSplits = delegate.getSplits(context)
58     // Get the appropriate number of mapper splits using the following priority
59     // 1. Get splits from AccumuloMapperProperties.DESIRED_ABSOLUTE_SPLITS (geomesa.mapreduce.splits.max)
60     // 2. Get splits from #tserver locations * AccumuloMapperProperties.DESIRED_SPLITS_PER_TSERVER (geomesa.mapreduce.splits.tserver.max)
61     // 3. Get splits from AccumuloInputFormat.getSplits(context)
62     def positive(prop: SystemProperty): Option[Int] = {
63       val int = prop.toInt
64       if (int.exists(_ < 1)) {
65         throw new IllegalArgumentException(s"${prop.property} contains an invalid int: ${prop.get}")
66       }
67       int
68     }
69 
70     val grpSplitsMax = positive(AccumuloMapperProperties.DESIRED_ABSOLUTE_SPLITS)
71 
72     lazy val grpSplitsPerTServer = positive(AccumuloMapperProperties.DESIRED_SPLITS_PER_TSERVER).flatMap { perTS =>
73       val numLocations = accumuloSplits.asScala.flatMap(_.getLocations).distinct.length
74       if (numLocations < 1) { None } else { Some(numLocations * perTS) }
75     }
76 
77     grpSplitsMax.orElse(grpSplitsPerTServer) match {
78       case Some(numberOfSplits) =>
79         logger.debug(s"Using desired splits with result of $numberOfSplits splits")
80         val splitSize: Int = math.ceil(accumuloSplits.size().toDouble / numberOfSplits).toInt
81         accumuloSplits.asScala.groupBy(_.getLocations.head).flatMap { case (location, splits) =>
82           splits.grouped(splitSize).map { group =>
83             val split = new GroupedSplit()
84             split.location = location
85             split.splits.append(group.map(f => f.asInstanceOf[RangeInputSplit]).toSeq: _*)
86             split.asInstanceOf[InputSplit]
87           }
88         }.toList.asJava
89 
90       case None =>
91         logger.debug(s"Using default Accumulo Splits with ${accumuloSplits.size} splits")
92         accumuloSplits
93     }
94   }

208    * This takes any jars that have been loaded by spark in the context classloader and makes them
209    * available to the general classloader. This is required as not all classes (even spark ones) check
210    * the context classloader.

212   def ensureSparkClasspath(): Unit = {
213     val sysLoader = ClassLoader.getSystemClassLoader
214     val ccl = Thread.currentThread().getContextClassLoader
215     if (ccl == null || !ccl.getClass.getCanonicalName.startsWith("org.apache.spark.")) {
216       logger.debug("No spark context classloader found")
217     } else if (!ccl.isInstanceOf[URLClassLoader]) {
218       logger.warn(s"Found context classloader, but can't handle type ${ccl.getClass.getCanonicalName}")
219     } else if (!sysLoader.isInstanceOf[URLClassLoader]) {
220       logger.warn(s"Found context classloader, but can't add to type ${sysLoader.getClass.getCanonicalName}")
221     } else {
222       // hack to get around protected visibility of addURL
223       // this might fail if there is a security manager present
224       val addUrl = classOf[URLClassLoader].getDeclaredMethod("addURL", classOf[URL])
225       addUrl.setAccessible(true)
226       val sysUrls = sysLoader.asInstanceOf[URLClassLoader].getURLs.map(_.toString).toSet
227       val (dupeUrls, newUrls) = ccl.asInstanceOf[URLClassLoader].getURLs.filterNot(_.toString.contains("__app__.jar")).partition(url => sysUrls.contains(url.toString))
228       newUrls.foreach(addUrl.invoke(sysLoader, _))
229       logger.debug(s"Loaded ${newUrls.length} urls from context classloader into system classloader " +
230           s"and ignored ${dupeUrls.length} that are already loaded")
231     }
232   }

270   /**
271     * Input split that groups a series of RangeInputSplits. Has to implement Hadoop Writable, thus the vars and
272     * mutable state.
273     */
274   class GroupedSplit extends InputSplit with Writable {
275 
276     // if we're running in spark, we need to load the context classpath before anything else,
277     // otherwise we get classloading and serialization issues
278     if (sys.env.get(GeoMesaAccumuloInputFormat.SYS_PROP_SPARK_LOAD_CP).exists(_.toBoolean)) {
279       GeoMesaAccumuloInputFormat.ensureSparkClasspath()
280     }
281 
282     private [mapreduce] var location: String = _
283     private [mapreduce] val splits: ArrayBuffer[RangeInputSplit] = ArrayBuffer.empty
284 
285     override def getLength: Long = splits.foldLeft(0L)((l: Long, r: RangeInputSplit) => l + r.getLength)
286 
287     override def getLocations: Array[String] = if (location == null) { Array.empty } else { Array(location) }
288 
289     override def write(out: DataOutput): Unit = {
290       out.writeUTF(location)
291       out.writeInt(splits.length)
292       splits.foreach(_.write(out))
293     }
294 
295     override def readFields(in: DataInput): Unit = {
296       location = in.readUTF()
297       splits.clear()
298       var i = 0
299       val size = in.readInt()
300       while (i < size) {
301         val split = new RangeInputSplit()
302         split.readFields(in)
303         splits.append(split)
304         i = i + 1
305       }
306     }
307 
308     override def toString = s"mapreduce.GroupedSplit[$location](${splits.length})"
309   }
