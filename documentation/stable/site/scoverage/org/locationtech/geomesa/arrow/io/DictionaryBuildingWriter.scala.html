31 /**
32  * Writes an arrow file of simple features. Dictionaries will be built up as features are observed.
33  * Dictionaries are encoded based on `maxSize`. Values will not be correctly encoded if
34  * more than `maxSize` distinct values are encountered
35  *
36  * @param sft simple feature type
37  * @param dictionaries dictionary fields
38  * @param encoding vector encoding
39  * @param maxSize max size for dictionaries, will be used to pick the encoding for dictionary values
40  */
41 class DictionaryBuildingWriter(
42     sft: SimpleFeatureType,
43     dictionaries: Seq[String],
44     encoding: SimpleFeatureEncoding,
45     ipcOpts: IpcOption,
46     maxSize: Int = Short.MaxValue
47   ) extends Closeable {
48 
49   import org.locationtech.geomesa.utils.geotools.RichAttributeDescriptors.RichAttributeDescriptor
50 
51   private val allocator = ArrowAllocator(s"dictionary-builder:${sft.getTypeName}")
52 
53   private val underlying = {
54     val struct = StructVector.empty(sft.getTypeName, allocator)
55     struct.allocateNew()
56     struct
57   }
58 
59   private val arrowWriter = underlying.getWriter
60 
61   private val idWriter = ArrowAttributeWriter.id(sft, encoding, underlying)
62   private val attributeWriters = DictionaryBuildingWriter.writers(sft, underlying, dictionaries, encoding, maxSize)
63 
64   private val root = {
65     val fields = Collections.singletonList[Field](underlying.getField)
66     new VectorSchemaRoot(fields, Collections.singletonList[FieldVector](underlying), 0)
67   }
68 
69   private var index = 0
70 
71   def size: Int = index
72 
73   def add(feature: SimpleFeature): Unit = {
74     arrowWriter.setPosition(index)
75     arrowWriter.start()
76     idWriter.apply(index, feature)
77     var i = 0
78     while (i < attributeWriters.length) {
79       attributeWriters(i).apply(index, feature.getAttribute(i))
80       i += 1
81     }
82     arrowWriter.end()
83     index += 1
84   }
85 
86   /**
87     * Clear any simple features currently stored in the vector
88     */
89   def clear(): Unit = {
90     underlying.setValueCount(0)
91     index = 0
92     attributeWriters.foreach {
93       case w: ArrowDictionaryWriter => w.dictionary.asInstanceOf[ArrowDictionaryBuilder].clear()
94       case _ => // no-op
95     }
96   }
97 
98   /**
99     * Writes out as a stand-alone streaming arrow file, including current dictionary values
100     *
101     * @param os output stream to write to
102     */
103   def encode(os: OutputStream): Unit = {
104     arrowWriter.setValueCount(index)
105     idWriter.setValueCount(index)
106     attributeWriters.foreach(_.setValueCount(index))
107     root.setRowCount(index)
108 
109     val dictionaries = attributeWriters.collect { case w: ArrowDictionaryWriter =>
110       val name = s"dict-${w.dictionary.encoding.getId}"
111       val original = sft.getDescriptor(w.vector.getField.getName)
112       val descriptor = if (original.isList) {
113         SimpleFeatureTypes.createDescriptor(s"$name:${original.getListType().getSimpleName}")
114       } else {
115         SimpleFeatureTypes.renameDescriptor(original, name)
116       }
117       val writer = ArrowAttributeWriter(sft, descriptor, None, encoding, allocator)
118 
119       var i = 0
120       w.dictionary.foreach { value =>
121         writer.apply(i, value)
122         i += 1
123       }
124       writer.setValueCount(w.dictionary.length)
125 
126       new Dictionary(writer.vector, w.dictionary.encoding)
127     }
128 
129     val provider = new MapDictionaryProvider(dictionaries: _*)
130     WithClose(new ArrowStreamWriter(root, provider, Channels.newChannel(os), ipcOpts)) { writer =>
131       writer.start()
132       writer.writeBatch()
133     }
134 
135     dictionaries.foreach(d => CloseWithLogging(d.getVector))
136   }
137 
138   override def close(): Unit = CloseWithLogging.raise(Seq(arrowWriter, underlying, allocator))
139 }

145   /**
146     * Gets an attribute writer or a dictionary building writer, as appropriate
147     */
148   private def writers(
149       sft: SimpleFeatureType,
150       vector: StructVector,
151       dictionaries: Seq[String],
152       encoding: SimpleFeatureEncoding,
153       maxSize: Int): Array[ArrowAttributeWriter] = {
154     var dictionaryId = -1L
155     val writers = sft.getAttributeDescriptors.asScala.map { descriptor =>
156       val dictionary = dictionaries.collectFirst {
157         case name if name == descriptor.getLocalName =>
158           dictionaryId += 1
159           val bits = if (maxSize <= Byte.MaxValue) { 8 } else if (maxSize <= Short.MaxValue) { 16 } else { 32 }
160           new ArrowDictionaryBuilder(new DictionaryEncoding(dictionaryId, false, new ArrowType.Int(bits, true)))
161       }
162       ArrowAttributeWriter(sft, descriptor, dictionary, encoding, vector)
163     }
164     writers.toArray
165   }
