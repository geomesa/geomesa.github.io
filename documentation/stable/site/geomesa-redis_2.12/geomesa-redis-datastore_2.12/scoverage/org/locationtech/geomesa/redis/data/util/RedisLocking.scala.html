19 /**
20   * Implements the basic single-node locking scheme from https://redis.io/topics/distlock
21   *
22   * The lock is only considered valid for a duration of `geomesa.distributed.lock.timeout` (default 2 minutes).
23   *
24   * Note: exclusivity when holding the lock is not 100% guaranteed
25   */
26 trait RedisLocking extends DistributedLocking {
27 
28   private val id = UUID.randomUUID().toString
29 
30   private val params = {
31     val timeout = DistributedLockTimeout.toDuration.getOrElse {
32       // note: should always be a valid fallback value so this exception should never be triggered
33       throw new IllegalArgumentException(s"Couldn't convert '${DistributedLockTimeout.get}' to a duration")
34     }
35     new SetParams().nx().px(timeout.toMillis)
36   }
37 
38   def connection: JedisPool
39 
40   override protected def acquireDistributedLock(key: String): Releasable =
41     acquireDistributedLock(key, Long.MaxValue).orNull
42 
43   override protected def acquireDistributedLock(key: String, timeOut: Long): Option[Releasable] = {
44     val start = System.currentTimeMillis()
45     var lock: Releasable = null
46 
47     while (lock == null && System.currentTimeMillis() - start < timeOut) {
48       if (WithClose(connection.getResource)(_.set(key, id, params)) != null) {
49         lock = new JedisReleasable(key)
50       }
51     }
52 
53     Option(lock)
54   }
55 
56   private class JedisReleasable(key: String) extends Releasable {
57     override def release(): Unit = {
58       WithClose(connection.getResource) { jedis =>
59         if (jedis.get(key) == id) {
60           jedis.del(key)
61         }
62       }
63     }
64   }
65 }
