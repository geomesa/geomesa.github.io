42   /**
43    * Turns a SimpleFeatureType attribute class binding into an enumeration.
44    *
45    * The first element in the result will be the primary binding. For geometries, lists and maps,
46    * the result will also contain secondary types.
47    *
48    * Lists will contain the type of the list elements.
49    * Maps will contain the type of the map keys, then the type of the map values.
50    * Geometries will contain the specific geometry type.
51    *
52    * Note: geometries will always return GEOMETRY as the primary type to allow for generic matching.
53    *
54    * @param clazz class, must be valid for a SimpleFeatureType attribute
55    * @param metadata attribute metadata (user data)
56    * @return binding
57    */
58   def selectType(clazz: Class[_], metadata: jMap[_, _] = jCollections.emptyMap()): Seq[ObjectType] = {
59     clazz match {
60       case c if classOf[java.lang.String].isAssignableFrom(c) =>
61         if (metadata.get(OptJson) == "true") { Seq(STRING, JSON) } else { Seq(STRING) }
62       case c if classOf[java.lang.Integer].isAssignableFrom(c) => Seq(INT)
63       case c if classOf[java.lang.Long].isAssignableFrom(c) => Seq(LONG)
64       case c if classOf[java.lang.Float].isAssignableFrom(c) => Seq(FLOAT)
65       case c if classOf[java.lang.Double].isAssignableFrom(c) => Seq(DOUBLE)
66       case c if classOf[java.lang.Boolean].isAssignableFrom(c) => Seq(BOOLEAN)
67       case c if classOf[java.util.Date].isAssignableFrom(c) => Seq(DATE)
68       case c if classOf[UUID].isAssignableFrom(c) => Seq(UUID)
69       case c if classOf[Geometry].isAssignableFrom(c) => geometryType(c.asInstanceOf[Class[_ <: Geometry]])
70       case c if classOf[Array[Byte]].isAssignableFrom(c) => Seq(BYTES)
71       case c if classOf[jList[_]].isAssignableFrom(c) => listType(metadata)
72       case c if classOf[jMap[_, _]].isAssignableFrom(c) => mapType(metadata)
73 
74       case _ => throw new IllegalArgumentException(s"Class $clazz can't be serialized")
75     }
76   }
77 
78   private def geometryType(clazz: Class[_ <: Geometry]): Seq[ObjectType] = {
79     val subtype = clazz match {
80       case c if c == classOf[Point]              => POINT
81       case c if c == classOf[LineString]         => LINESTRING
82       case c if c == classOf[Polygon]            => POLYGON
83       case c if c == classOf[MultiLineString]    => MULTILINESTRING
84       case c if c == classOf[MultiPolygon]       => MULTIPOLYGON
85       case c if c == classOf[MultiPoint]         => MULTIPOINT
86       case c if c == classOf[GeometryCollection] => GEOMETRY_COLLECTION
87       case _                                     => GEOMETRY
88     }
89     Seq(GEOMETRY, subtype)
90   }
91 
92   private def listType(metadata: jMap[_, _]): Seq[ObjectType] = {
93     val clazz = Class.forName(metadata.get(UserDataListType).asInstanceOf[String])
94     selectType(clazz) match {
95       case Seq(binding) => Seq(LIST, binding)
96       case _ => throw new IllegalArgumentException(s"Can't serialize list sub-type of ${clazz.getName}")
97     }
98   }
99 
100   private def mapType(metadata: jMap[_, _]): Seq[ObjectType] = {
101     val keyClass = Class.forName(metadata.get(UserDataMapKeyType).asInstanceOf[String])
102     val keyType = selectType(keyClass) match {
103       case Seq(binding) => binding
104       case _ => throw new IllegalArgumentException(s"Can't serialize map key type of ${keyClass.getName}")
105     }
106     val valueClass = Class.forName(metadata.get(UserDataMapValueType).asInstanceOf[String])
107     val valueType = selectType(valueClass) match {
108       case Seq(binding) => binding
109       case _ => throw new IllegalArgumentException(s"Can't serialize map value type of ${valueClass.getName}")
110     }
111     Seq(MAP, keyType, valueType)
112   }
