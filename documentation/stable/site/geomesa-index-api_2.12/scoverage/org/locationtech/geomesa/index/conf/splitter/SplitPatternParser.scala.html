19 /**
20   * Parses patterns into splits
21   */
22 object SplitPatternParser {
23 
24   private val Parser = new SplitPatternParser()
25 
26   @throws(classOf[ParsingException])
27   def parse(pattern: String): SplitPattern = parse(pattern, report = true)
28 
29   @throws(classOf[ParsingException])
30   def parse(pattern: String, report: Boolean): SplitPattern = {
31     if (pattern == null) {
32       throw new IllegalArgumentException("Invalid split pattern: null")
33     }
34 
35     val runner = if (report) { ReportingParseRunner(Parser.patterns) } else { BasicParseRunner(Parser.patterns) }
36     val parsing = runner.run(pattern.stripMargin('|').replaceAll("\\s*", ""))
37 
38     parsing.result.getOrElse {
39       throw new ParsingException(s"Invalid split pattern: ${ErrorUtils.printParseErrors(parsing)}")
40     }
41   }
42 
43   sealed trait SplitPattern {
44     def range: Seq[String]
45     private [SplitPatternParser] def reverse: Seq[String] = range.reverse
46   }
47 
48   case class AlphaPattern(start: Char, end: Char) extends SplitPattern {
49     override def range: Seq[String] =
50       Seq.range(math.min(start, end), math.max(start, end) + 1).map(_.toChar.toString)
51   }
52 
53   case class NumericPattern(start: Byte, end: Byte) extends SplitPattern {
54     override def range: Seq[String] =
55       Seq.range(math.min(start, end), math.max(start, end) + 1).map(_.toString)
56   }
57 
58   case class CompositePattern(patterns: Seq[SplitPattern]) extends SplitPattern {
59     override def range: Seq[String] = patterns.flatMap(_.range)
60     override private [SplitPatternParser] def reverse: Seq[String] = patterns.flatMap(_.reverse)
61   }
62 
63   case class TieredPatterns(tiers: Seq[SplitPattern]) extends SplitPattern {
64     override def range: Seq[String] =
65       tiers.map(_.range).reduceLeft { (left, right) => for (a <- left; b <- right) yield { a + b } }
66   }
67 
68   case class NegativePatterns(tiers: Seq[SplitPattern]) extends SplitPattern {
69     override def range: Seq[String] =
70       tiers.map(_.reverse).foldLeft(Seq("-")) { (left, right) => for (a <- left; b <- right) yield { a + b } }
71   }
72 
73   case class DatePattern(from: Date, to: Date, ranges: Int) extends SplitPattern {
74     override def range: Seq[String] = {
75       val interval = (to.getTime - from.getTime) / ranges
76       Seq.tabulate(ranges) { i =>
77         AttributeIndexKey.encodeForQuery(new Date(from.getTime + interval * i), classOf[Date])
78       }
79     }
80   }
81 }

83 private class SplitPatternParser extends BasicParser {
84 
85   // Valid patterns look like the following:
86 
87   // [a-z][0-9]
88   // [0-35-8][a-z]
89   // [f][0-9]
90 
91   import org.parboiled.scala._
92 
93   def patterns: Rule1[SplitPattern] = rule {
94     (mixedPatterns | negativePatterns | dateRange) ~ EOI
95   }
96 
97   private def mixedPatterns: Rule1[SplitPattern] = rule {
98     oneOrMore(mixedTier) ~~> {
99       case tiers if tiers.lengthCompare(1) == 0 => tiers.head
100       case tiers => TieredPatterns(tiers)
101     }
102   }
103 
104   private def negativePatterns: Rule1[SplitPattern] = rule {
105     "[-]" ~ oneOrMore(numericTier) ~~> { tiers => NegativePatterns(tiers) }
106   }
107 
108   private def mixedTier: Rule1[SplitPattern] = rule {
109     "[" ~ { oneOrMore(alpha | numeric) ~~> { p => if (p.length == 1) { p.head } else { CompositePattern(p) } } } ~ "]"
110   }
111 
112   private def numericTier: Rule1[SplitPattern] = rule {
113     "[" ~ { oneOrMore(numeric) ~~> { p => if (p.length == 1) { p.head } else { CompositePattern(p) } } } ~ "]"
114   }
115 
116   private def dateRange: Rule1[SplitPattern] = rule {
117     (date ~ "/" ~ date ~ "/" ~ int) ~~> { (from, to, splits) => DatePattern(from, to, splits) }
118   }
119 
120   private def alpha: Rule1[AlphaPattern] = rule {
121     alphaRange | alphaSingle
122   }
123 
124   private def numeric: Rule1[NumericPattern] = rule {
125     numericRange | numericSingle
126   }
127 
128   private def alphaSingle: Rule1[AlphaPattern] = rule {
129     alphaEndpoint ~~> { e => AlphaPattern(e, e) }
130   }
131 
132   private def numericSingle: Rule1[NumericPattern] = rule {
133     numericEndpoint ~~> { e => NumericPattern(e, e) }
134   }
135 
136   private def alphaRange: Rule1[AlphaPattern] = rule {
137     (alphaEndpoint ~ "-" ~ alphaEndpoint) ~~> { (from, to) => AlphaPattern(from, to) }
138   }
139 
140   private def numericRange: Rule1[NumericPattern] = rule {
141     (numericEndpoint ~ "-" ~ numericEndpoint) ~~> { (from, to) => NumericPattern(from, to) }
142   }
143 
144   private def alphaEndpoint: Rule1[Char] = rule { ("a" - "z" | "A" - "Z") ~> { c => c.charAt(0) } }
145 
146   private def numericEndpoint: Rule1[Byte] = rule { ("0" - "9") ~> { c => c.toByte } }
147 
148   private def date: Rule1[Date] = rule {
149     group(year ~ "-" ~ month ~ "-" ~ day) ~> { date => DateParsing.parseDate(date) }
150   }
151 
152   private def year: Rule0 = rule { nTimes(4, "0" - "9")  }
153 
154   private def month: Rule0 = rule { ("0" ~ ("1" - "9")) | ("1" ~ ("0" - "2")) }
155 
156   private def day: Rule0 = rule { (("0" - "2") ~ ("0" - "9")) | ("3" ~ ("0" - "1")) }
157 }
