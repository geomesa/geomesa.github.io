14 trait DistributedLocking {
15 
16   /**
17     * Gets and acquires a distributed lock based on the key.
18     * Make sure that you 'release' the lock in a finally block.
19     *
20     * @param key key to lock on - equivalent to a path in zookeeper
21     * @return the lock
22     */
23   protected def acquireDistributedLock(key: String): Releasable
24 
25   /**
26     * Gets and acquires a distributed lock based on the key.
27     * Make sure that you 'release' the lock in a finally block.
28     *
29     * @param key key to lock on - equivalent to a path in zookeeper
30     * @param timeOut how long to wait to acquire the lock, in millis
31     * @return the lock, if obtained
32     */
33   protected def acquireDistributedLock(key: String, timeOut: Long): Option[Releasable]
34 
35   /**
36     * Execute a function wrapped in a lock
37     *
38     * @param key key to lock on
39     * @param fn function to run with the lock
40     * @tparam T result type
41     * @return
42     */
43   protected def withLock[T](key: String, fn: => T): T = {
44     val lock = acquireDistributedLock(key)
45     try { fn } finally { lock.release() }
46   }
47 
48   /**
49     * Execute a function wrapped in a lock
50     *
51     * @param key key to lock on
52     * @param timeOut how long to wait to acquire the lock, in millis
53     * @param fn function to run with the lock
54     * @param fallback function to run if the lock could not be acquired
55     * @tparam T result type
56     * @return
57     */
58   protected def withLock[T](key: String, timeOut: Long, fn: => T, fallback: => T): T = {
59     acquireDistributedLock(key, timeOut) match {
60       case None => fallback
61       case Some(lock) => try { fn } finally { lock.release() }
62     }
63   }
64 }

66 trait LocalLocking extends DistributedLocking {
67 
68   import LocalLocking.locks
69 
70   override protected def acquireDistributedLock(key: String): Releasable = {
71     val lock = locks.synchronized(locks.getOrElseUpdate(key, new ReentrantLock()))
72     lock.lock()
73     Releasable(lock)
74   }
75 
76   override protected def acquireDistributedLock(key: String, timeOut: Long): Option[Releasable] = {
77     val lock = locks.synchronized(locks.getOrElseUpdate(key, new ReentrantLock()))
78     if (lock.tryLock(timeOut, TimeUnit.MILLISECONDS)) { Some(Releasable(lock)) } else { None }
79   }
80 }
