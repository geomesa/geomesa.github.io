31   /**
32     * Splits the query up into different filter plans to be evaluated. Each filter plan will consist of one or
33     * more query plans. Each query plan will have a primary part (that would be used for query planning)
34     * and an optional secondary part (that would be applied as a secondary filter).
35     *
36     * Examples:
37     *
38     * bbox(geom) AND attr1 = ? =>
39     *
40     * Seq(FilterPlan(Seq(QueryFilter(ST,Some([ geom bbox ]),Some([ attr1 = ? ])))))
41     *
42     * bbox(geom) OR attr1 = ? =>
43     *
44     * Seq(FilterPlan(Seq(QueryFilter(ST,Some([ geom bbox ]),None), QueryFilter(ATTRIBUTE,Some([ attr1 = ? ]),None))))
45     *
46     * bbox(geom) AND dtg DURING ? AND attr1 = ? =>
47     *
48     * Seq(FilterPlan(Seq(QueryFilter(Z3,Some([ geom bbox AND dtg during ? ]),Some([ attr1 = ? ])))),
49     *     FilterPlan(Seq(QueryFilter(ATTRIBUTE,Some([ attr1 = ? ]),Some([ geom bbox AND dtg during ? ])))))
50     *
51     * note: spatial and temporal filters are combined.
52     *
53     * (bbox(geom) OR geom INTERSECT) AND attr1 = ? =>
54     *
55     * Seq(FilterPlan(Seq(QueryFilter(ST,Some([ geom bbox OR geom intersect ]),Some([ attr1 = ? ]))))
56     *     FilterPlan(Seq(QueryFilter(ATTRIBUTE,Seq([ attr1 = ? ]),Some([ geom bbox OR geom intersect ])))))
57     *
58     * note: ORs will not be split if they operate on a single attribute
59     *
60     */
61   def getQueryOptions(
62       filter: Filter,
63       transform: Option[SimpleFeatureType] = None,
64       requestedIndex: Option[String] = None): Seq[FilterPlan] = {
65     // cnf gives us a top level AND with ORs as first children
66     rewriteFilterInCNF(filter) match {
67       case a: And =>
68         // look for ORs across attributes, e.g. bbox OR dtg
69         val complex = ArrayBuffer.empty[Or]
70         val simple = ArrayBuffer.empty[Filter]
71         var permutations = 1
72 
73         a.getChildren.asScala.foreach {
74           case or: Or =>
75             val attributes = attributeAndIdCount(or, sft)
76             if (attributes > 1) {
77               complex += or
78               permutations *= attributes
79             } else {
80               simple += or
81             }
82 
83           case f =>
84             simple += f
85         }
86 
87         if (complex.isEmpty) {
88           // no cross-attribute ORs
89           getSimpleQueryOptions(a, transform, requestedIndex).map(fs => FilterPlan(Seq(fs)))
90         } else {
91           // we don't generally consider all the permutations available, as that takes exponential time
92           // instead, we consider the cross-attribute-ors and the rest of the query separately
93 
94           // the cross-attribute or plans
95           // each option should just result in 1 filter plan since there are no ANDs
96           val complexOptions = complex.map(getQueryOptions(_, transform, requestedIndex))
97 
98           // the non-cross-attribute ors, with the ors tacked on as secondary filters at the end
99           lazy val simpleOptions = if (simple.isEmpty) { Seq.empty } else {
100             getSimpleQueryOptions(andFilters(simple.toSeq), transform, requestedIndex)
101                 .map(s => FilterPlan(Seq(addSecondaryPredicates(s, complex.toSeq))))
102           }
103           lazy val expandReduceOptions =
104             expandReduceOrOptions(rewriteFilterInDNF(filter).asInstanceOf[Or], transform, requestedIndex)
105                 .map(fp => FilterPlan(makeDisjoint(fp.strategies)))
106 
107           if (complexOptions.forall(o => o.lengthCompare(1) == 0 && o.head.strategies.forall(_.isPreferredScan))) {
108             // if each of the complex options has a good scan available, return those plus the simple options
109             complexOptions.map(o => FilterPlan(o.head.strategies.map(addSecondaryPredicates(_, simple.toSeq)))).toSeq ++
110                 simpleOptions.toSeq
111           } else if (ExpandReduceThreshold.toInt.exists(_ > permutations)) {
112             logger.debug(s"Using ${getExpandReduceLog(filter, permutations)}")
113             expandReduceOptions
114           } else if (simpleOptions.nonEmpty) {
115             // if there's not a clear option for the complex scans, ignore them and just return the simple ones
116             logger.warn("Not considering complex OR predicates in query planning: " +
117                 s"${complex.map(filterToString).mkString("(", ") AND (", ")")}")
118             simpleOptions
119           } else {
120             // if there aren't any simple plans or clear options for the complex plans,
121             // fall back to the expand-reduce logic
122             logger.warn(s"Falling back to ${getExpandReduceLog(filter, permutations)}")
123             expandReduceOptions
124           }
125         }
126 
127       case o: Or =>
128         // there are no ands - just ors between fields
129         // this implies that each child has only a single property or ID
130         def getGroup(f: Filter): (Seq[String], Boolean) =
131           (FilterHelper.propertyNames(f, sft), FilterHelper.hasIdFilter(f))
132 
133         // group and then recombine the OR'd filters by the attribute they operate on
134         val groups = o.getChildren.asScala.groupBy(getGroup).values.map(g => ff.or(g.asJava)).toSeq
135         val perAttributeOptions = groups.flatMap { g =>
136           val options = getSimpleQueryOptions(g, transform, requestedIndex)
137           require(options.length < 2, s"Expected only a single option for ${filterToString(g)} but got $options")
138           options.headOption
139         }
140         if (perAttributeOptions.exists(_.isFullTableScan)) {
141           // we have to do a full table scan for part of the query, just append everything to that
142           Seq(FilterPlan(Seq(fullTableScanOption(o, transform))))
143         } else {
144           Seq(FilterPlan(makeDisjoint(perAttributeOptions)))
145         }
146 
147       case f =>
148         getSimpleQueryOptions(f, transform, requestedIndex).map(fs => FilterPlan(Seq(fs)))
149     }
150   }

152   /**
153    * Gets options for a 'simple' filter, where each OR is on a single attribute, e.g.
154    *   (bbox1 OR bbox2) AND dtg
155    *   bbox AND dtg AND (attr1 = foo OR attr = bar)
156    * not:
157    *   bbox OR dtg
158    *
159    * Because the inputs are simple, each one can be satisfied with a single query filter.
160    * The returned values will each satisfy the query, using a different strategy.
161    *
162    * @param filter input filter
163    * @param transform query transform, if any
164    * @param requestedIndex user requested index, if nay
165    * @return sequence of options, any of which can satisfy the query
166    */
167   private def getSimpleQueryOptions(
168       filter: Filter,
169       transform: Option[SimpleFeatureType],
170       requestedIndex: Option[String]): Seq[FilterStrategy] = {
171     if (filter == Filter.EXCLUDE) { Seq.empty } else {
172       val requestedPlan = requestedIndex.flatMap { requested =>
173         // check by id and by name
174         val index = indices.find(i => i.identifier == requested || i.name.equalsIgnoreCase(requested)).orElse {
175           // back-compatibility for attr vs join index name
176           if (!AttributeIndex.name.equalsIgnoreCase(requested)) { None } else {
177             indices.find(_.name == AttributeIndex.JoinIndexName)
178           }
179         }
180         index.flatMap(_.getFilterStrategy(filter, transform)).map(Seq(_))
181       }
182       requestedPlan.getOrElse {
183         val preferredScans  = ArrayBuffer.empty[FilterStrategy]
184         val acceptableScans = ArrayBuffer.empty[FilterStrategy]
185         val fullTableScans  = ArrayBuffer.empty[FilterStrategy]
186 
187         indices.foreach { i =>
188           i.getFilterStrategy(filter, transform).foreach {
189             case f if f.isFullTableScan => fullTableScans += f
190             case f if f.isPreferredScan => preferredScans += f
191             case f                      => acceptableScans += f
192           }
193         }
194 
195         if (preferredScans.nonEmpty) {
196           preferredScans.toSeq
197         } else if (acceptableScans.nonEmpty) {
198           acceptableScans.toSeq
199         } else {
200           fullTableScans.take(1).toSeq
201         }
202       }
203     }
204   }

206   /**
207     * Calculates all possible options for each part of the filter, then determines all permutations of
208     * the options. This can end up being expensive (O(2&#94;n)), so is only used as a fall-back.
209     */
210   private def expandReduceOrOptions(
211       filter: Or,
212       transform: Option[SimpleFeatureType],
213       requestedIndex: Option[String]): Seq[FilterPlan] = {
214 
215     // for each child of the or, get the query options
216     // each filter plan should only have a single query filter
217     def getChildOptions: Seq[Seq[FilterPlan]] =
218       filter.getChildren.asScala.map(getSimpleQueryOptions(_, transform, requestedIndex).map(fs => FilterPlan(Seq(fs))).toSeq).toSeq
219 
220     // combine the filter plans so that each plan has multiple query filters
221     // use the permutations of the different options for each child
222     // TODO GEOMESA-941 Fix algorithmically dangerous (2^N exponential runtime)
223     def reduceChildOptions(childOptions: Seq[Seq[FilterPlan]]): Seq[FilterPlan] =
224       childOptions.reduce { (left, right) =>
225         left.flatMap(l => right.map(r => FilterPlan(l.strategies ++ r.strategies)))
226       }
227 
228     // try to combine query filters in each filter plan if they have the same primary filter
229     // this avoids scanning the same ranges twice with different secondary predicates
230     def combineSecondaryFilters(options: Seq[FilterPlan]): Seq[FilterPlan] = options.map { r =>
231       // build up the result array instead of using a group by to preserve filter order
232       val groups = ArrayBuffer.empty[FilterStrategy]
233       r.strategies.distinct.foreach { f =>
234         val i = groups.indexWhere(g => g.index == f.index && g.primary == f.primary)
235         if (i == -1) {
236           groups.append(f)
237         } else {
238           val current = groups(i).secondary match {
239             case Some(o: Or) => o.getChildren.asScala
240             case Some(n) => Seq(n)
241             case None => Seq.empty
242           }
243           val secondary = orOption((current ++ f.secondary).toSeq)
244           val temporal = f.temporal && groups(i).temporal
245           groups.update(i, FilterStrategy(f.index, f.primary, secondary, temporal, f.costMultiplier))
246         }
247       }
248       FilterPlan(groups.toSeq)
249     }
250 
251     // if a filter plan has any query filters that scan a subset of the range of a different query filter,
252     // then we can combine them, as we have to scan the larger range anyway
253     def mergeOverlappedFilters(options: Seq[FilterPlan]): Seq[FilterPlan] = options.map { filterPlan =>
254       val filters = ArrayBuffer[FilterStrategy](filterPlan.strategies: _*)
255       var merged: FilterStrategy = null
256       var i = 0
257       while (i < filters.length) {
258         val toMerge = filters(i)
259         if (toMerge != null) {
260           var j = 0
261           while (j < filters.length && merged == null) {
262             if (i != j) {
263               val mergeTo = filters(j)
264               if (mergeTo != null) {
265                 merged = tryMerge(toMerge, mergeTo)
266               }
267             }
268             j += 1
269           }
270           if (merged != null) {
271             // remove the merged query filter and replace the one merged into
272             filters.update(i, null)
273             filters.update(j - 1, merged)
274             merged = null
275           }
276         }
277         i += 1
278       }
279 
280       // if we have replaced anything, recreate the filter plan
281       val overlapped = filters.filter(_ != null)
282       if (overlapped.length < filterPlan.strategies.length) {
283         FilterPlan(overlapped.toSeq)
284       } else {
285         filterPlan
286       }
287     }
288 
289     val childOpts   = getChildOptions
290     val reducedOpts = reduceChildOptions(childOpts)
291     val combinedSec = combineSecondaryFilters(reducedOpts)
292     val merged      = mergeOverlappedFilters(combinedSec)
293 
294     if (merged.nonEmpty) {
295       merged
296     } else {
297       Seq(FilterPlan(Seq(fullTableScanOption(filter, transform))))
298     }
299   }

301   /**
302     * Will perform a full table scan - used when we don't have anything better. Currently z3, z2 and record
303     * tables support full table scans.
304     */
305   private def fullTableScanOption(filter: Filter, transform: Option[SimpleFeatureType]): FilterStrategy = {
306     val secondary = if (filter == Filter.INCLUDE) { None } else { Some(filter) }
307     // note: early return after we find a matching strategy
308     val iter = indices.iterator
309     while (iter.hasNext) {
310       iter.next().getFilterStrategy(Filter.INCLUDE, transform) match {
311         case None => // no-op
312         case Some(i) => return FilterStrategy(i.index, i.primary, secondary, i.temporal, i.costMultiplier)
313       }
314     }
315     throw new UnsupportedOperationException(s"Configured indices do not support the query ${filterToString(filter)}")
316   }
317 
318   private def addSecondaryPredicates(filter: FilterStrategy, predicates: Seq[Filter]): FilterStrategy = {
319     val secondary = andOption(filter.secondary.toSeq ++ predicates)
320     FilterStrategy(filter.index, filter.primary, secondary, filter.temporal, filter.costMultiplier)
321   }

340   /**
341    * Try to merge the two query filters. Return the merged query filter if successful, else null.
342    */
343   def tryMerge(toMerge: FilterStrategy, mergeTo: FilterStrategy): FilterStrategy = {
344     if (mergeTo.primary.forall(_ == Filter.INCLUDE)) {
345       // this is a full table scan, we can just append the OR to the secondary filter
346       val secondary = orOption(mergeTo.secondary.toSeq ++ toMerge.filter)
347       FilterStrategy(mergeTo.index, mergeTo.primary, secondary, mergeTo.temporal, mergeTo.costMultiplier)
348     } else if (toMerge.index.name == AttributeIndex.name && mergeTo.index.name == AttributeIndex.name) {
349       // TODO extract this out into the API?
350       tryMergeAttrStrategy(toMerge, mergeTo)
351     } else {
352       // overlapping geoms, date ranges, attribute ranges, etc will be handled when extracting bounds
353       null
354     }
355   }
356 
357   /**
358     * Tries to merge the two filters that are OR'd together into a single filter that can be queried in one pass.
359     * Will return the merged filter, or null if they can't be merged.
360     *
361     * We can merge filters if they have the same secondary filter AND:
362     *   1. One of them does not have a primary filter
363     *   2. They both have a primary filter on the same attribute
364     *
365     * @param toMerge first filter
366     * @param mergeTo second filter
367     * @return merged filter that satisfies both inputs, or null if that isn't possible
368     */
369   def tryMergeAttrStrategy(toMerge: FilterStrategy, mergeTo: FilterStrategy): FilterStrategy = {
370     // TODO this will be incorrect for multi-valued properties where we have an AND in the primary filter
371     val leftAttributes = toMerge.primary.map(FilterHelper.propertyNames(_, null))
372     val rightAttributes = mergeTo.primary.map(FilterHelper.propertyNames(_, null))
373 
374     val canMergePrimary = (leftAttributes, rightAttributes) match {
375       case (Some(left), Some(right)) => left.length == 1 && right.length == 1 && left.head == right.head
376       case _ => true
377     }
378 
379     if (canMergePrimary && toMerge.secondary == mergeTo.secondary) {
380       val primary = orOption(toMerge.primary.toSeq ++ mergeTo.primary)
381       FilterStrategy(mergeTo.index, primary, mergeTo.secondary, mergeTo.temporal, mergeTo.costMultiplier)
382     } else {
383       null
384     }
385   }
386 
387   /**
388     * Make a filter plan disjoint ORs - this way we don't have to deduplicate results
389     *
390     * @param strategies strategies
391     * @return same strategies with disjoint ORs
392     */
393   private def makeDisjoint(strategies: Seq[FilterStrategy]): Seq[FilterStrategy] = {
394     if (strategies.lengthCompare(2) < 0) { strategies } else {
395       // A OR B OR C becomes... A OR (B NOT A) OR (C NOT A and NOT B)
396       // keep track of our current disjoint clause
397       val nots = ArrayBuffer.empty[Filter]
398       strategies.map { filter =>
399         val res = if (nots.isEmpty) { filter } else {
400           val secondary = Some(andFilters((nots ++ filter.secondary).toSeq))
401           FilterStrategy(filter.index, filter.primary, secondary, filter.temporal, filter.costMultiplier)
402         }
403         nots ++= filter.filter.map(ff.not) // note - side effect
404         res
405       }
406     }
407   }
