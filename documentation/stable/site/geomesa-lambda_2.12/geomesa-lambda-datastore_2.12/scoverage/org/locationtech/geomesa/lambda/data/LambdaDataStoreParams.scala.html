115   def parse(params: java.util.Map[String, _], namespace: String): LambdaConfig = {
116     val brokers = BrokersParam.lookup(params)
117     val expiry = if (!PersistParam.lookup(params).booleanValue) { None } else {
118       ExpiryParam.lookupOpt(params).collect { case d: FiniteDuration => d}
119     }
120 
121     val partitions = PartitionsParam.lookup(params).intValue
122     val consumers = ConsumersParam.lookup(params).intValue
123     val batchSize = BatchSizeParam.lookupOpt(params).map(_.intValue())
124 
125     val bootstrap = Map("bootstrap.servers" -> brokers)
126     val consumerConfig = ConsumerOptsParam.lookupOpt(params).map(_.asScala.toMap).getOrElse(Map.empty) ++ bootstrap
127     val producerConfig = ProducerOptsParam.lookupOpt(params).map(_.asScala.toMap).getOrElse(Map.empty) ++ bootstrap
128 
129     val zk = ZookeepersParam.lookup(params)
130     val zkNamespace = s"gm_lambda_$namespace"
131 
132     val offsetCommitInterval = ConsumerOffsetCommitInterval.lookup(params)
133 
134     LambdaConfig(zk, zkNamespace, producerConfig, consumerConfig, partitions, consumers, expiry, batchSize, offsetCommitInterval)
135   }
