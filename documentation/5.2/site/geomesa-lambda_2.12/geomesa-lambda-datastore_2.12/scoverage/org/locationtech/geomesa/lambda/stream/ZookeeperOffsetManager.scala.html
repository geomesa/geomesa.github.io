25 class ZookeeperOffsetManager(zookeepers: String, namespace: String = "geomesa") extends OffsetManager {
26 
27   import ZookeeperOffsetManager.offsetsPath
28 
29   private val client = CuratorHelper.client(zookeepers).namespace(namespace).build()
30   client.start()
31 
32   private val listeners = scala.collection.mutable.Map.empty[String, CuratorOffsetListener]
33 
34   override def getOffset(topic: String, partition: Int): Long = {
35     val path = ZookeeperOffsetManager.offsetsPath(topic, partition)
36     if (client.checkExists().forPath(path) == null) { -1L } else {
37       ZookeeperOffsetManager.deserializeOffsets(client.getData.forPath(path))
38     }
39   }
40 
41   override def setOffset(topic: String, partition: Int, offset: Long): Unit = {
42     val path = ZookeeperOffsetManager.offsetsPath(topic, partition)
43     if (client.checkExists().forPath(path) == null) {
44       client.create().creatingParentsIfNeeded().forPath(path)
45     }
46     client.setData().forPath(path, ZookeeperOffsetManager.serializeOffset(offset))
47   }
48 
49   override def deleteOffsets(topic: String): Unit = {
50     val path = ZookeeperOffsetManager.offsetsPath(topic)
51     if (client.checkExists().forPath(path) != null) {
52       client.delete().deletingChildrenIfNeeded().forPath(path)
53     }
54   }
55 
56   override def addOffsetListener(topic: String, listener: OffsetListener): Unit = synchronized {
57     listeners.getOrElseUpdate(topic, new CuratorOffsetListener(client, offsetsPath(topic))).addListener(listener)
58   }
59 
60   override def removeOffsetListener(topic: String, listener: OffsetListener): Unit = synchronized {
61     listeners.get(topic).foreach(_.removeListener(listener))
62   }
63 
64   override def close(): Unit = synchronized {
65     CloseWithLogging(listeners.values)
66     listeners.clear()
67     CloseWithLogging(client)
68   }
69 
70   override protected def acquireDistributedLock(path: String): Closeable =
71     acquireLock(path, lock => { lock.acquire(); true })
72 
73   override protected def acquireDistributedLock(path: String, timeOut: Long): Option[Closeable] =
74     Option(acquireLock(path, lock => lock.acquire(timeOut, TimeUnit.MILLISECONDS)))
75 
76   private def acquireLock(path: String, acquire: InterProcessSemaphoreMutex => Boolean): Closeable = {
77     val lock = new InterProcessSemaphoreMutex(client, s"/$path/locks")
78     if (acquire(lock)) {
79       () => lock.release()
80     } else {
81       null
82     }
83   }

88   def serializeOffset(offset: Long): Array[Byte] = offset.toString.getBytes(StandardCharsets.UTF_8)
89   def deserializeOffsets(bytes: Array[Byte]): Long = new String(bytes, StandardCharsets.UTF_8).toLong

91   private def offsetsPath(topic: String): String = s"/$topic/offsets"
92   private def offsetsPath(topic: String, partition: Int): String = s"${offsetsPath(topic)}/$partition"
93   private def partitionFromPath(path: String): Int = path.substring(path.lastIndexOf("/") + 1).toInt
94 
95   private class CuratorOffsetListener(client: CuratorFramework, path: String)
96       extends PathChildrenCacheListener with Closeable with LazyLogging {
97 
98     private val executor = Executors.newCachedThreadPool()
99 
100     private val listeners = scala.collection.mutable.Set.empty[OffsetListener]
101 
102     private var cache: PathChildrenCache = _
103 
104     def addListener(listener: OffsetListener): Unit = {
105       closeCache()
106       listeners += listener
107       cache = new PathChildrenCache(client, path, true) {
108         // override ensure path to avoid using containers, which aren't supported in zk 3.4
109         override protected def ensurePath(): Unit = {
110           if (client.checkExists().forPath(path) == null) {
111             client.create().creatingParentsIfNeeded().forPath(path)
112           }
113         }
114       }
115       cache.getListenable.addListener(this)
116       cache.start()
117     }
118 
119     def removeListener(listener: OffsetListener): Unit = {
120       listeners -= listener
121       if (listeners.isEmpty) {
122         closeCache()
123       }
124     }
125 
126     override def childEvent(client: CuratorFramework, event: PathChildrenCacheEvent): Unit = {
127       import PathChildrenCacheEvent.Type.{CHILD_ADDED, CHILD_UPDATED}
128       try {
129         val eventPath = Option(event.getData).map(_.getPath).getOrElse("")
130         if ((event.getType == CHILD_ADDED || event.getType == CHILD_UPDATED) && eventPath.startsWith(path)) {
131           logger.trace(s"ZK event triggered for: $eventPath")
132           val partition = partitionFromPath(eventPath)
133           val offset = ZookeeperOffsetManager.deserializeOffsets(event.getData.getData)
134           listeners.foreach { listener =>
135             executor.execute(new Runnable {
136               override def run(): Unit = {
137                 try { listener.offsetChanged(partition, offset) } catch {
138                   case NonFatal(e) => logger.warn("Error calling offset listener", e)
139                 }
140               }
141             })
142           }
143         }
144       } catch {
145         case NonFatal(e) => logger.warn("Error handling ZK event", e)
146       }
147     }
148 
149     override def close(): Unit = {
150       executor.shutdown()
151       closeCache()
152     }
153 
154     private def closeCache(): Unit = synchronized {
155       if (cache != null) {
156         cache.getListenable.removeListener(this)
157         CloseWithLogging(cache)
158         cache = null
159       }
160     }
161   }
