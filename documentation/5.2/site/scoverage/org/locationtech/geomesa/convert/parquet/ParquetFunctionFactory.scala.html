35   /**
36     * Base class for handling parquet-encoded geometries
37     *
38     * @param name function name
39     * @tparam T geometry type
40     * @tparam U column type
41     */
42   abstract class ParquetGeometryFn[T <: Geometry, U](name: String, path: AvroPath)
43       extends NamedTransformerFunction(Seq(name), pure = true) {
44 
45     import SimpleFeatureParquetSchema.{GeometryColumnX, GeometryColumnY}
46 
47     override def apply(args: Array[AnyRef]): AnyRef = {
48       path.eval(args(0).asInstanceOf[GenericRecord]).collect {
49         case r: GenericRecord => eval(r.get(GeometryColumnX).asInstanceOf[U], r.get(GeometryColumnY).asInstanceOf[U])
50       }.orNull
51     }
52 
53     protected def getPath(args: List[Expression]): AvroPath = {
54       args match {
55         case _ :: LiteralString(s) :: _ => AvroPath(s)
56         case _ => throw new IllegalArgumentException(s"Expected Avro path but got: ${args.headOption.orNull}")
57       }
58     }
59 
60     protected def eval(x: U, y: U): T
61   }
