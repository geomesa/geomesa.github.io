47   /**
48    * Infer a configuration and simple feature type from an input stream, if possible
49    *
50    * Available hints:
51    *  - `featurePath` - json path expression pointing to the feature element
52    *
53    * @param is input
54    * @param sft simple feature type, if known ahead of time
55    * @param hints implementation specific hints about the input
56    * @return
57    */
58   override def infer(
59       is: InputStream,
60       sft: Option[SimpleFeatureType],
61       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = {
62 
63     val tryElements = Try {
64       WithClose(new JsonReader(new InputStreamReader(is, StandardCharsets.UTF_8))) { reader =>
65         reader.setLenient(true)
66         val iter: Iterator[JsonElement] = new Iterator[JsonElement] {
67           override def hasNext: Boolean = reader.peek() != JsonToken.END_DOCUMENT
68           override def next(): JsonElement = JsonParser.parseReader(reader)
69         }
70         iter.take(AbstractConverterFactory.inferSampleSize).toList
71       }
72     }
73     tryElements.flatMap { elements =>
74       lazy val featurePath = hints.get(JsonConverterFactory.FeaturePathKey).map(_.toString)
75       if (elements.isEmpty) {
76         Failure(new RuntimeException("Could not parse the input as JSON"))
77       } else if (JsonConverter.isFeature(elements.head)) {
78         Try(elements.map(JsonConverter.parseFeature))
79             .flatMap(inferGeoJson(_, None, sft))
80       } else if (JsonConverter.isFeatureCollection(elements.head)) {
81         Try(elements.flatMap(JsonConverter.parseFeatureCollection))
82             .flatMap(inferGeoJson(_, Some("$.features[*]"), sft))
83       } else if (elements.head.isJsonObject) {
84         inferJson(elements, featurePath, sft)
85       } else if (elements.head.isJsonArray) {
86         inferJson(elements, featurePath.orElse(Some("$.[*]")), sft)
87       } else {
88         Failure(new RuntimeException("Could not parse the input as a JSON object or array"))
89       }
90     }
91   }

93   private def inferGeoJson(
94       features: Seq[GeoJsonFeature],
95       featurePath: Option[String],
96       sft: Option[SimpleFeatureType]): Try[(SimpleFeatureType, Config)] = Try {
97     // track the 'properties', geometry type and 'id' in each feature
98     // use linkedHashMap to retain insertion order
99     val props = scala.collection.mutable.LinkedHashMap.empty[String, ListBuffer[Any]]
100     val geoms = ListBuffer.empty[Any]
101     var hasId = true
102 
103     features.take(AbstractConverterFactory.inferSampleSize).foreach { feature =>
104       feature.properties.foreach { case (k, v) => props.getOrElseUpdate(k, ListBuffer.empty) += v }
105       geoms += feature.geom
106       hasId = hasId && feature.id.isDefined
107     }
108 
109     val pathsAndValues =
110       props.toSeq.map { case (path, values) => PathWithValues(path, values) } :+
111           PathWithValues(JsonConverterFactory.GeoJsonGeometryPath, geoms)
112     val inferredTypes = TypeInference.infer(pathsAndValues, sft.toRight("inferred-json"), new PropNamer())
113 
114     val idJsonField = if (hasId) { Some(StringJsonField("id", "$.id", pathIsRoot = false, None)) } else { None }
115     val idField = idJsonField match {
116       case None    => Some(Expression("md5(stringToBytes(toString($0)))"))
117       case Some(f) => Some(Expression(s"$$${f.name}"))
118     }
119     val fieldConfig = idJsonField.toSeq ++ inferredTypes.types.map(createFieldConfig)
120 
121     val jsonConfig = JsonConfig(typeToProcess, featurePath, idField, Map.empty, Map.empty)
122 
123     val config = configConvert.to(jsonConfig)
124         .withFallback(fieldConvert.to(fieldConfig))
125         .withFallback(optsConvert.to(BasicOptions.default))
126         .toConfig
127 
128     (inferredTypes.sft, config)
129   }

131   private def inferJson(
132       elements: Seq[JsonElement],
133       featurePath: Option[String],
134       sft: Option[SimpleFeatureType]): Try[(SimpleFeatureType, Config)] = {
135     val tryFeatures = Try {
136       val features = featurePath match {
137         case None => elements
138         case Some(p) =>
139           val path = JsonPath.compile(p)
140           elements.flatMap { el =>
141             val res = try { path.read[JsonElement](el, JsonConverter.JsonConfiguration) } catch {
142               case _: PathNotFoundException => JsonNull.INSTANCE
143             }
144             if (res.isJsonArray) {
145               res.getAsJsonArray.asList().asScala
146             } else {
147               Seq(res)
148             }
149           }
150       }
151       features.collect { case r if r.isJsonObject => r.getAsJsonObject }
152     }
153 
154     tryFeatures.flatMap { features =>
155       if (features.isEmpty) {
156         Failure(new RuntimeException("Could not parse input as JSON"))
157       } else {
158         Try {
159           // track the properties in each feature
160           // use linkedHashMap to retain insertion order
161           val props = scala.collection.mutable.LinkedHashMap.empty[String, ListBuffer[Any]]
162 
163           features.take(AbstractConverterFactory.inferSampleSize).foreach { feature =>
164             GeoJsonParsing.parseElement(feature, "").foreach { case (k, v) =>
165               props.getOrElseUpdate(k, ListBuffer.empty) += v
166             }
167           }
168 
169           val pathsAndValues = props.toSeq.map { case (path, values) => PathWithValues(path, values) }
170           val inferredTypes = TypeInference.infer(pathsAndValues, sft.toRight("inferred-json"))
171 
172           val idField = Some(Expression("md5(stringToBytes(toString($0)))"))
173           val fieldConfig = inferredTypes.types.map(createFieldConfig)
174 
175           val jsonConfig = JsonConfig(typeToProcess, featurePath, idField, Map.empty, Map.empty)
176 
177           val config =
178             configConvert.to(jsonConfig)
179                 .withFallback(fieldConvert.to(fieldConfig))
180                 .withFallback(optsConvert.to(BasicOptions.default))
181                 .toConfig
182 
183           (inferredTypes.sft, config)
184         }
185       }
186     }
187   }

189   private def createFieldConfig(typed: TypeWithPath): JsonField = {
190     val TypeWithPath(path, inferredType) = typed
191     val transform = inferredType.transform match {
192       case IdentityTransform => None
193       // account for optional nodes by wrapping transform with a try/null
194       case t => Some(Expression(s"try(${t.apply(0)},null)"))
195     }
196     if (path.isEmpty) {
197       DerivedField(inferredType.name, transform)
198     } else {
199       inferredType.typed match {
200         case ObjectType.BOOLEAN =>
201           BooleanJsonField(inferredType.name, path, pathIsRoot = false, transform)
202         case ObjectType.LIST =>
203           // if type is list, that means the transform is 'identity', but we need to replace it with jsonList.
204           // this is due to GeoJsonParsing decoding the json array for us, above
205           ArrayJsonField(inferredType.name, path, pathIsRoot = false, Some(Expression("try(jsonList('string',$0),null)")))
206         case t if transform.isEmpty && (t == ObjectType.GEOMETRY || ObjectType.GeometrySubtypes.contains(t)) =>
207           GeometryJsonField(inferredType.name, path, pathIsRoot = false, None)
208         case _ =>
209           // all other types will be parsed as strings with appropriate transforms
210           StringJsonField(inferredType.name, path, pathIsRoot = false, transform)
211       }
212     }
213   }

242   object JsonFieldConvert extends FieldConvert[JsonField] with OptionConvert {
243     override protected def decodeField(cur: ConfigObjectCursor,
244                                        name: String,
245                                        transform: Option[Expression]): Either[ConfigReaderFailures, JsonField] = {
246       val jsonTypeCur =  cur.atKeyOrUndefined("json-type")
247       val jsonType = if (jsonTypeCur.isUndefined) { Right(None) } else { jsonTypeCur.asString.right.map(Option.apply) }
248 
249       val config = for {
250         jType    <- jsonType.right
251         path     <- optional(cur, "path").right
252         rootPath <- optional(cur, "root-path").right
253       } yield {
254         (jType, path, rootPath)
255       }
256       config.right.flatMap { case (jType, path, rootPath) =>
257         if (path.isDefined && rootPath.isDefined) {
258           cur.failed(CannotConvert(cur.toString, "JsonField", "Json fields must define only one of 'path' or 'root-path'"))
259         } else if (jType.isDefined && path.isEmpty && rootPath.isEmpty) {
260           cur.failed(CannotConvert(cur.toString, "JsonField", "Json fields must define a 'path' or 'root-path'"))
261         } else if (jType.isEmpty) {
262           Right(DerivedField(name, transform))
263         } else {
264           val (jsonPath, pathIsRoot) = (path, rootPath) match {
265             case (Some(p), None) => (p, false)
266             case (None, Some(p)) => (p, true)
267           }
268           jType.get.toLowerCase(Locale.US) match {
269             case "string"           => Right(StringJsonField(name, jsonPath, pathIsRoot, transform))
270             case "float"            => Right(FloatJsonField(name, jsonPath, pathIsRoot, transform))
271             case "double"           => Right(DoubleJsonField(name, jsonPath, pathIsRoot, transform))
272             case "integer" | "int"  => Right(IntJsonField(name, jsonPath, pathIsRoot, transform))
273             case "boolean" | "bool" => Right(BooleanJsonField(name, jsonPath, pathIsRoot, transform))
274             case "long"             => Right(LongJsonField(name, jsonPath, pathIsRoot, transform))
275             case "geometry"         => Right(GeometryJsonField(name, jsonPath, pathIsRoot, transform))
276             case "array" | "list"   => Right(ArrayJsonField(name, jsonPath, pathIsRoot, transform))
277             case "object" | "map"   => Right(ObjectJsonField(name, jsonPath, pathIsRoot, transform))
278             case t => cur.failed(CannotConvert(cur.toString, "JsonField", s"Invalid json-type '$t'"))
279           }
280         }
281       }
282     }
283 
284     override protected def encodeField(field: JsonField, base: java.util.Map[String, AnyRef]): Unit = {
285       field match {
286         case f: TypedJsonField =>
287           base.put("json-type", f.jsonType)
288           base.put(if (f.pathIsRoot) { "root-path" } else { "path" }, f.path)
289 
290         case _ => // no-op
291       }
292     }
