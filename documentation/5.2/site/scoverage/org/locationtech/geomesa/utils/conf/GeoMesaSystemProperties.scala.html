19   case class SystemProperty(property: String, default: String = null) {
20 
21     val threadLocalValue = new InheritableThreadLocal[String]()
22 
23     def set(value: String): Unit = System.setProperty(property, value)
24 
25     def clear(): Unit = System.clearProperty(property)
26 
27     def get: String = ConfigLoader.Config.get(property) match {
28       case Some((value, true))  => value // final value - can't be overridden
29       case Some((value, false)) => fromSysProps.getOrElse(value)
30       case None => fromSysProps.getOrElse(default)
31     }
32 
33     def option: Option[String] = Option(get)
34 
35     def toDuration: Option[Duration] = option.flatMap { value =>
36       Try(DurationParsing.caseInsensitive(value)) match {
37         case Success(v) if v.isFinite && v > Duration.Zero => Some(v)
38         case _ =>
39           logger.warn(s"Invalid duration for property $property: $value")
40           Option(default).map(Duration.apply)
41       }
42     }
43 
44     def toUnboundedDuration: Option[Duration] = option.flatMap { value =>
45       Try(DurationParsing.caseInsensitive(value)) match {
46         case Success(v) => Some(v)
47         case _ =>
48           logger.warn(s"Invalid duration for property $property: $value")
49           Option(default).map(Duration.apply)
50       }
51     }
52 
53     def toBytes: Option[Long] = option.flatMap { value =>
54       Suffixes.Memory.bytes(value) match {
55         case Success(b) => Some(b)
56         case Failure(e) =>
57           logger.warn(s"Invalid bytes for property $property=$value: ${e.toString}")
58           Option(default).flatMap(Suffixes.Memory.bytes(_).toOption)
59       }
60     }
61 
62     def toInt: Option[Int] = option.flatMap { value =>
63       Try(value.toInt) match {
64         case Success(v) => Some(v)

68       }
69     }
70

77       }
78     }
79 
80     def toFloat: Option[Float] = option.flatMap { value =>
81       Try(value.toFloat) match {
82         case Success(v) => Some(v)

87     }

89     def toDouble: Option[Double] = option.flatMap { value =>
90       Try(value.toDouble) match {

99       Try(value.toBoolean) match {
100         case Success(v) => Some(v)
101         case Failure(_) =>
102           logger.warn(s"Invalid Boolean for property $property: $value")
103           Option(default).map(java.lang.Boolean.parseBoolean)
104       }
105     }
106 
107     private def fromSysProps: Option[String] =
108       Option(threadLocalValue.get).orElse(sys.props.get(property).filterNot(_.isEmpty))
109   }
