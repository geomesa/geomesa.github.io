69   // converts a range to a printable string - only includes the row
70   private def rangeToString(r: org.apache.accumulo.core.data.Range): String = {
71     val a = if (r.isStartKeyInclusive) "[" else "("
72     val z = if (r.isEndKeyInclusive) "]" else ")"
73     val start = if (r.isInfiniteStartKey) "-inf" else keyToString(r.getStartKey)
74     val stop = if (r.isInfiniteStopKey) "+inf" else keyToString(r.getEndKey)
75     s"$a$start::$stop$z"
76   }

97   case class BatchScanPlan(
98       filter: FilterStrategy,
99       tables: Seq[String],
100       ranges: Seq[org.apache.accumulo.core.data.Range],
101       iterators: Seq[IteratorSetting],
102       columnFamily: Option[Text],
103       resultsToFeatures: ResultsToFeatures[Entry[Key, Value]],
104       reducer: Option[FeatureReducer],
105       sort: Option[Seq[(String, Boolean)]],
106       maxFeatures: Option[Int],
107       projection: Option[QueryReferenceSystems],
108       numThreads: Int
109     ) extends AccumuloQueryPlan {
110 
111     override def scan(ds: AccumuloDataStore): CloseableIterator[Entry[Key, Value]] = {
112       // convert the relative timeout to an absolute timeout up front
113       val timeout = ds.config.queries.timeout.map(Timeout.apply)
114       // note: calculate authorizations up front so that multi-threading doesn't mess up auth providers
115       scan(ds.connector, ds.auths, ds.config.queries.parallelPartitionScans, timeout)
116     }
117 
118     /**
119      * Scan with pre-computed auths
120      *
121      * @param connector connector
122      * @param auths auths
123      * @param partitionParallelScans parallelize scans across multiple partitions (vs execute them serially)
124      * @param timeout absolute stop time, as sys time
125      * @return
126      */
127     def scan(
128         connector: AccumuloClient,
129         auths: Authorizations,
130         partitionParallelScans: Boolean,
131         timeout: Option[Timeout]): CloseableIterator[Entry[Key, Value]] = {
132       if (partitionParallelScans) {
133         // kick off all the scans at once
134         tables.map(scanner(connector, _, auths, timeout)).foldLeft(CloseableIterator.empty[Entry[Key, Value]])(_ concat _)
135       } else {
136         // kick off the scans sequentially as they finish
137         SelfClosingIterator(tables.iterator).flatMap(scanner(connector, _, auths, timeout))
138       }
139     }
140 
141     /**
142      *
143      * @param connector connector
144      * @param table table
145      * @param auths auths
146      * @param timeout absolute stop time, as sys time
147      * @return
148      */
149     private def scanner(
150         connector: AccumuloClient,
151         table: String,
152         auths: Authorizations,
153         timeout: Option[Timeout]): CloseableIterator[Entry[Key, Value]] = {
154       val scanner = connector.createBatchScanner(table, auths, numThreads)
155       scanner.setRanges(ranges.asJava)
156       configure(scanner)
157       timeout match {
158         case None => new ScanIterator(scanner)
159         case Some(t) => new ManagedScanIterator(t, new AccumuloScanner(scanner), this)
160       }
161     }
162   }

164   // join on multiple tables - requires multiple scans
165   case class JoinPlan(
166       filter: FilterStrategy,
167       tables: Seq[String],
168       ranges: Seq[org.apache.accumulo.core.data.Range],
169       iterators: Seq[IteratorSetting],
170       columnFamily: Option[Text],
171       numThreads: Int,
172       joinFunction: JoinFunction,
173       joinQuery: BatchScanPlan
174     ) extends AccumuloQueryPlan {
175 
176     override val join: Some[(JoinFunction, BatchScanPlan)] = Some((joinFunction, joinQuery))
177     override def resultsToFeatures: ResultsToFeatures[Entry[Key, Value]] = joinQuery.resultsToFeatures
178     override def reducer: Option[FeatureReducer] = joinQuery.reducer
179     override def sort: Option[Seq[(String, Boolean)]] = joinQuery.sort
180     override def maxFeatures: Option[Int] = joinQuery.maxFeatures
181     override def projection: Option[QueryReferenceSystems] = joinQuery.projection
182 
183     override def scan(ds: AccumuloDataStore): CloseableIterator[Entry[Key, Value]] = {
184       // convert the relative timeout to an absolute timeout up front
185       val timeout = ds.config.queries.timeout.map(Timeout.apply)
186       // calculate authorizations up front so that multi-threading doesn't mess up auth providers
187       val auths = ds.auths
188       val joinTables = joinQuery.tables.iterator
189       if (ds.config.queries.parallelPartitionScans) {
190         // kick off all the scans at once
191         tables.map(scanner(ds.connector, _, joinTables.next, auths, partitionParallelScans = true, timeout))
192             .foldLeft(CloseableIterator.empty[Entry[Key, Value]])(_ concat _)
193       } else {
194         // kick off the scans sequentially as they finish
195         SelfClosingIterator(tables.iterator)
196             .flatMap(scanner(ds.connector, _, joinTables.next, auths, partitionParallelScans = false, timeout))
197       }
198     }
199 
200     private def scanner(
201         connector: AccumuloClient,
202         table: String,
203         joinTable: String,
204         auths: Authorizations,
205         partitionParallelScans: Boolean,
206         timeout: Option[Timeout]): CloseableIterator[Entry[Key, Value]] = {
207       val primary = if (ranges.lengthCompare(1) == 0) {
208         val scanner = connector.createScanner(table, auths)
209         scanner.setRange(ranges.head)
210         scanner
211       } else {
212         val scanner = connector.createBatchScanner(table, auths, numThreads)
213         scanner.setRanges(ranges.asJava)
214         scanner
215       }
216       configure(primary)
217       val join = joinQuery.copy(tables = Seq(joinTable))
218       new BatchMultiScanner(connector, primary, join, joinFunction, auths, partitionParallelScans, timeout)
219     }
220   }
