51   private def isOperationGeomWholeWorld[Op <: BinarySpatialOperator]
52       (op: Op, order: SpatialOpOrder.SpatialOpOrder = SpatialOpOrder.AnyOrder): Boolean = {
53     val prop = checkOrder(op.getExpression1, op.getExpression2)
54     // validate that property and literal are in the specified order
55     prop.exists { p =>
56       val ordered = order match {
57         case SpatialOpOrder.AnyOrder      => true
58         case SpatialOpOrder.PropertyFirst => !p.flipped
59         case SpatialOpOrder.LiteralFirst  => p.flipped
60       }
61       ordered && Option(FastConverter.evaluate(p.literal, classOf[Geometry])).exists(isWholeWorld)
62     }

103     * Extract geometries from a filter without validating boundaries.

106     * @param attribute attribute to consider
107     * @param intersect intersect AND'd geometries or return them all
108     * @return geometry bounds from spatial filters

110   private def extractUnclippedGeometries(filter: Filter, attribute: String, intersect: Boolean): FilterValues[Geometry] = {
111     filter match {
112       case o: Or  =>
113         val all = o.getChildren.asScala.map(extractUnclippedGeometries(_, attribute, intersect))
114         val join = FilterValues.or[Geometry]((l, r) => l ++ r) _
115         all.reduceLeftOption[FilterValues[Geometry]](join).getOrElse(FilterValues.empty)
116 
117       case a: And =>
118         val all = a.getChildren.asScala.map(extractUnclippedGeometries(_, attribute, intersect)).filter(_.nonEmpty)
119         if (intersect) {
120           val intersect = FilterValues.and[Geometry]((l, r) => Option(l.intersection(r)).filterNot(_.isEmpty)) _
121           all.reduceLeftOption[FilterValues[Geometry]](intersect).getOrElse(FilterValues.empty)
122         } else {
123           FilterValues(all.toSeq.flatMap(_.values))
124         }
125 
126       // Note: although not technically required, all known spatial predicates are also binary spatial operators
127       case f: BinarySpatialOperator if isSpatialFilter(f) =>
128         FilterValues(GeometryProcessing.extract(f, attribute))
129 
130       case _ =>
131         FilterValues.empty
132     }

135   /**
136    * Extracts intervals from a filter. Intervals will be merged where possible - the resulting sequence
137    * is considered to be a union (i.e. OR)
138    *
139    * @param filter filter to evaluate
140    * @param attribute attribute to consider
141    * @param handleExclusiveBounds will make all exclusive bounds inclusive instead, by rounding the value slightly
142    * @return a sequence of intervals, if any. disjoint intervals will result in Seq((null, null))
143    */
144   def extractIntervals(
145       filter: Filter,
146       attribute: String,
147       handleExclusiveBounds: Boolean = false): FilterValues[Bounds[ZonedDateTime]] = {
148     extractAttributeBounds(filter, attribute, classOf[Date]).map { bounds =>
149       var lower, upper: Bound[ZonedDateTime] = null
150       // this if check determines if rounding will be used and if we need to account for narrow ranges
151       if (!handleExclusiveBounds || bounds.lower.value.isEmpty || bounds.upper.value.isEmpty ||
152           (bounds.lower.inclusive && bounds.upper.inclusive)) {
153         lower = createDateTime(bounds.lower, roundSecondsUp, handleExclusiveBounds)
154         upper = createDateTime(bounds.upper, roundSecondsDown, handleExclusiveBounds)
155       } else {
156         // check for extremely narrow filters where our rounding makes the result out-of-order
157         // note: both upper and lower are known to be defined based on hitting this else branch
158         val margin = if (bounds.lower.inclusive || bounds.upper.inclusive) { 1000 } else { 2000 }
159         val round = bounds.upper.value.get.getTime - bounds.lower.value.get.getTime > margin
160         lower = createDateTime(bounds.lower, roundSecondsUp, round)
161         upper = createDateTime(bounds.upper, roundSecondsDown, round)
162       }
163       Bounds(lower, upper)
164     }
165   }

167   private def createDateTime(bound: Bound[Date],
168                              round: ZonedDateTime => ZonedDateTime,
169                              roundExclusive: Boolean): Bound[ZonedDateTime] = {
170     if (bound.value.isEmpty) { Bound.unbounded } else {
171       val dt = bound.value.map(d => ZonedDateTime.ofInstant(d.toInstant, ZoneOffset.UTC))
172       if (roundExclusive && !bound.inclusive) {
173         Bound(dt.map(round), inclusive = true)
174       } else {
175         Bound(dt, bound.inclusive)
176       }
177     }
178   }
179 
180   private def roundSecondsUp(dt: ZonedDateTime): ZonedDateTime = dt.plusSeconds(1).withNano(0)
181 
182   private def roundSecondsDown(dt: ZonedDateTime): ZonedDateTime = {
183     val nanos = dt.getNano
184     if (nanos == 0) { dt.minusSeconds(1) } else { dt.withNano(0) }
185   }

187   /**
188     * Extracts bounds from filters that pertain to a given attribute. Bounds will be merged where
189     * possible.
190     *
191     * @param filter filter to evaluate
192     * @param attribute attribute name to consider
193     * @param binding attribute type
194     * @return a sequence of bounds, if any
195     */
196   def extractAttributeBounds[T](filter: Filter, attribute: String, binding: Class[T]): FilterValues[Bounds[T]] = {
197     filter match {
198       case o: Or =>
199         val union = FilterValues.or[Bounds[T]](Bounds.union[T]) _
200         o.getChildren.asScala.map(f =>
201           extractAttributeBounds(f, attribute, binding)
202         ).reduceLeft[FilterValues[Bounds[T]]]((acc, child) => {
203           if (acc.isEmpty || child.isEmpty) {
204             FilterValues.empty
205           } else {
206             union(acc, child)
207           }
208         })
209 
210       case a: And =>
211         val all = a.getChildren.asScala.flatMap { f =>
212           val child = extractAttributeBounds(f, attribute, binding)
213           if (child.isEmpty) { Seq.empty } else { Seq(child) }
214         }
215         val intersection = FilterValues.and[Bounds[T]](Bounds.intersection[T]) _
216         all.reduceLeftOption[FilterValues[Bounds[T]]](intersection).getOrElse(FilterValues.empty)
217 
218       case f: PropertyIsEqualTo =>
219         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
220           case e: PropertyLiteral =>
221             Option(FastConverter.evaluate(e.literal, binding)).map { lit =>
222               val bound = Bound(Some(lit), inclusive = true)
223               FilterValues(Seq(Bounds(bound, bound)))
224             }
225 
226           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = true, binding)
227         }.getOrElse(FilterValues.empty)
228 
229       case f: PropertyIsBetween =>
230         try {
231           val prop = f.getExpression.asInstanceOf[PropertyName].getPropertyName
232           if (prop != attribute) { FilterValues.empty } else {
233             // note that between is inclusive
234             val lower = Bound(Option(FastConverter.evaluate(f.getLowerBoundary, binding)), inclusive = true)
235             val upper = Bound(Option(FastConverter.evaluate(f.getUpperBoundary, binding)), inclusive = true)
236             FilterValues(Seq(Bounds(lower, upper)))
237           }
238         } catch {
239           case e: Exception =>
240             FilterHelperLogger.log.warn(s"Unable to extract bounds from filter '${filterToString(f)}'", e)
241             FilterValues.empty
242         }
243 
244       case f: During if classOf[Date].isAssignableFrom(binding) =>
245         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
246           case e: PropertyLiteral =>
247             Option(FastConverter.evaluate(e.literal, classOf[Period])).map { p =>
248               // note that during is exclusive
249               val lower = Bound(Option(p.getBeginning.getPosition.getDate.asInstanceOf[T]), inclusive = false)
250               val upper = Bound(Option(p.getEnding.getPosition.getDate.asInstanceOf[T]), inclusive = false)
251               FilterValues(Seq(Bounds(lower, upper)))
252             }
253 
254           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = false, binding)
255         }.getOrElse(FilterValues.empty)
256 
257       case f: PropertyIsGreaterThan =>
258         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
259           case e: PropertyLiteral =>
260             Option(FastConverter.evaluate(e.literal, binding)).map { lit =>
261               val bound = Bound(Some(lit), inclusive = false)
262               val (lower, upper) = if (e.flipped) { (Bound.unbounded[T], bound) } else { (bound, Bound.unbounded[T]) }
263               FilterValues(Seq(Bounds(lower, upper)))
264             }
265 
266           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = false, binding)
267         }.getOrElse(FilterValues.empty)
268 
269       case f: PropertyIsGreaterThanOrEqualTo =>
270         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
271           case e: PropertyLiteral =>
272             Option(FastConverter.evaluate(e.literal, binding)).map { lit =>
273               val bound = Bound(Some(lit), inclusive = true)
274               val (lower, upper) = if (e.flipped) { (Bound.unbounded[T], bound) } else { (bound, Bound.unbounded[T]) }
275               FilterValues(Seq(Bounds(lower, upper)))
276             }
277 
278           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = true, binding)
279         }.getOrElse(FilterValues.empty)
280 
281       case f: PropertyIsLessThan =>
282         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
283           case e: PropertyLiteral =>
284             Option(FastConverter.evaluate(e.literal, binding)).map { lit =>
285               val bound = Bound(Some(lit), inclusive = false)
286               val (lower, upper) = if (e.flipped) { (bound, Bound.unbounded[T]) } else { (Bound.unbounded[T], bound) }
287               FilterValues(Seq(Bounds(lower, upper)))
288             }
289 
290           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = false, binding)
291         }.getOrElse(FilterValues.empty)
292 
293       case f: PropertyIsLessThanOrEqualTo =>
294         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
295           case e: PropertyLiteral =>
296             Option(FastConverter.evaluate(e.literal, binding)).map { lit =>
297               val bound = Bound(Some(lit), inclusive = true)
298               val (lower, upper) = if (e.flipped) { (bound, Bound.unbounded[T]) } else { (Bound.unbounded[T], bound) }
299               FilterValues(Seq(Bounds(lower, upper)))
300             }
301 
302           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = true, binding)
303         }.getOrElse(FilterValues.empty)
304 
305       case f: Before =>
306         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
307           case e: PropertyLiteral =>
308             Option(FastConverter.evaluate(e.literal, binding)).map { lit =>
309               // note that before is exclusive
310               val bound = Bound(Some(lit), inclusive = false)
311               val (lower, upper) = if (e.flipped) { (bound, Bound.unbounded[T]) } else { (Bound.unbounded[T], bound) }
312               FilterValues(Seq(Bounds(lower, upper)))
313             }
314 
315           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = false, binding)
316         }.getOrElse(FilterValues.empty)
317 
318       case f: After =>
319         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
320           case e: PropertyLiteral =>
321             Option(FastConverter.evaluate(e.literal, binding)).map { lit =>
322               // note that after is exclusive
323               val bound = Bound(Some(lit), inclusive = false)
324               val (lower, upper) = if (e.flipped) { (Bound.unbounded[T], bound) } else { (bound, Bound.unbounded[T]) }
325               FilterValues(Seq(Bounds(lower, upper)))
326             }
327 
328           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = false, binding)
329         }.getOrElse(FilterValues.empty)
330 
331       case f: PropertyIsLike if binding == classOf[String] =>
332         try {
333           val prop = f.getExpression.asInstanceOf[PropertyName].getPropertyName
334           if (prop != attribute) { FilterValues.empty } else {
335             // find the first wildcard and create a range prefix
336             val literal = f.getLiteral
337             var i = literal.indexWhere(Wildcards.contains)
338             // check for escaped wildcards
339             while (i > 1 && literal.charAt(i - 1) == '\\' && literal.charAt(i - 2) == '\\') {
340               i = literal.indexWhere(Wildcards.contains, i + 1)
341             }
342             if (i == -1) {
343               val literals = if (f.isMatchingCase) { Seq(literal) } else { casePermutations(literal) }
344               val bounds = literals.map { lit =>
345                 val bound = Bound(Some(lit), inclusive = true)
346                 Bounds(bound, bound)
347               }
348               FilterValues(bounds.asInstanceOf[Seq[Bounds[T]]], precise = true)
349             } else {
350               val prefix = literal.substring(0, i)
351               val prefixes = if (f.isMatchingCase) { Seq(prefix) } else { casePermutations(prefix) }
352               val bounds = prefixes.map { p =>
353                 Bounds(Bound(Some(p), inclusive = true), Bound(Some(p + WildcardSuffix), inclusive = true))
354               }
355               // our ranges fully capture the filter if there's a single trailing multi-char wildcard
356               val exact = i == literal.length - 1 && literal.charAt(i) == WildcardMultiChar
357               FilterValues(bounds.asInstanceOf[Seq[Bounds[T]]], precise = exact)
358             }
359           }
360         } catch {
361           case e: Exception =>
362             FilterHelperLogger.log.warn(s"Unable to extract bounds from filter '${filterToString(f)}'", e)
363             FilterValues.empty
364         }
365 
366       case f: Not if f.getFilter.isInstanceOf[PropertyIsNull] =>
367         try {
368           val isNull = f.getFilter.asInstanceOf[PropertyIsNull]
369           val prop = isNull.getExpression.asInstanceOf[PropertyName].getPropertyName
370           if (prop != attribute) { FilterValues.empty } else {
371             FilterValues(Seq(Bounds.everything[T]))
372           }
373         } catch {
374           case e: Exception =>
375             FilterHelperLogger.log.warn(s"Unable to extract bounds from filter '${filterToString(f)}'", e)
376             FilterValues.empty
377         }
378 
379       case f: Not =>
380         // we extract the sub-filter bounds, then invert them
381         val inverted = extractAttributeBounds(f.getFilter, attribute, binding)
382         if (inverted.isEmpty) {
383           inverted
384         } else if (inverted.disjoint) {
385           FilterValues(Seq(Bounds.everything[T])) // equivalent to not null
386         } else if (!inverted.precise) {
387           FilterHelperLogger.log.warn(s"Falling back to full table scan for inverted query: '${filterToString(f)}'")
388           FilterValues(Seq(Bounds.everything[T]), precise = false)
389         } else {
390           // NOT(A OR B) turns into NOT(A) AND NOT(B)
391           val uninverted = inverted.values.map { bounds =>
392             // NOT the single bound
393             val not = bounds.bounds match {
394               case (None, None) => Seq.empty
395               case (Some(lo), None) => Seq(Bounds(Bound.unbounded, Bound(Some(lo), !bounds.lower.inclusive)))
396               case (None, Some(hi)) => Seq(Bounds(Bound(Some(hi), !bounds.upper.inclusive), Bound.unbounded))
397               case (Some(lo), Some(hi)) => Seq(
398                   Bounds(Bound.unbounded, Bound(Some(lo), !bounds.lower.inclusive)),
399                   Bounds(Bound(Some(hi), !bounds.upper.inclusive), Bound.unbounded)
400                 )
401             }
402             FilterValues(not)
403           }
404           // AND together
405           val intersect = FilterValues.and[Bounds[T]](Bounds.intersection[T]) _
406           uninverted.reduceLeft[FilterValues[Bounds[T]]](intersect)
407         }
408 
409       case f: TEquals =>
410         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap { prop =>
411           Option(FastConverter.evaluate(prop.literal, binding)).map { lit =>
412             val bound = Bound(Some(lit), inclusive = true)
413             FilterValues(Seq(Bounds(bound, bound)))
414           }
415         }.getOrElse(FilterValues.empty)
416 
417       case _ => FilterValues.empty
418     }
419   }

428   /**
429    * Calculates all the different case permutations of a string.
430    *
431    * For example, "foo" -> Seq("foo", "Foo", "fOo", "foO", "fOO", "FoO", "FOo", "FOO")
432    *
433    * @param string input string
434    * @return
435    */
436   private def casePermutations(string: String): Seq[String] = {
437     val max = FilterProperties.CaseInsensitiveLimit.toInt.getOrElse {
438       // has a valid default value so should never return a none
439       throw new IllegalStateException(
440         s"Error getting default value for ${FilterProperties.CaseInsensitiveLimit.property}")
441     }
442 
443     val lower = string.toLowerCase(Locale.US)
444     val upper = string.toUpperCase(Locale.US)
445     // account for chars without upper/lower cases, which we don't need to permute
446     val count = (0 until lower.length).count(i => lower(i) != upper(i))
447 
448     if (count > max) {
449       FilterHelperLogger.log.warn(s"Not expanding case-insensitive prefix due to length: $string")
450       Seq.empty
451     } else {
452       // there will be 2^n different permutations, accounting for chars that don't have an upper/lower case
453       val permutations = Array.fill(math.pow(2, count).toInt)(Array(lower: _*))
454       var i = 0 // track the index of the current char
455       var c = 0 // track the index of the bit check, which skips chars that don't have an upper/lower case
456       while (i < string.length) {
457         val upperChar = upper.charAt(i)
458         if (lower.charAt(i) != upperChar) {
459           var j = 0
460           while (j < permutations.length) {
461             // set upper/lower based on the bit
462             if (((j >> c) & 1) != 0) {
463               permutations(j)(i) = upperChar
464             }
465             j += 1
466           }
467           c += 1
468         }
469         i += 1
470       }
471 
472       permutations.map(new String(_))
473     }
474   }

506   /**
507     * Simplifies filters to make them easier to process.
508     *
509     * Current simplifications:
510     *
511     *   1) Extracts out common parts in an OR clause to simplify further processing.
512     *
513     *      Example: OR(AND(1, 2), AND(1, 3), AND(1, 4)) -> AND(1, OR(2, 3, 4))
514     *
515     *   2) N/A - add more simplifications here as needed
516     *
517     * @param filter filter
518     * @return
519     */
520   def simplify(filter: Filter): Filter = {
521     def deduplicateOrs(f: Filter): Filter = f match {
522       case and: And => ff.and(and.getChildren.asScala.map(deduplicateOrs).asJava)
523 
524       case or: Or =>
525         // OR(AND(1,2,3), AND(1,2,4)) -> Seq(Seq(1,2,3), Seq(1,2,4))
526         val decomposed = or.getChildren.asScala.map(decomposeAnd)
527         val clauses = decomposed.head // Seq(1,2,3)
528         val duplicates = clauses.filter(c => decomposed.tail.forall(_.contains(c))) // Seq(1,2)
529         if (duplicates.isEmpty) { or } else {
530           val simplified = decomposed.flatMap(d => andOption(d.filterNot(duplicates.contains)))
531           if (simplified.length < decomposed.length) {
532             // the duplicated filters are an entire clause, so we can ignore the rest of the clauses
533             andFilters(duplicates)
534           } else {
535             andFilters(orOption(simplified.toSeq).toSeq ++ duplicates)
536           }
537         }
538 
539       case _ => f
540     }
541     // TODO GEOMESA-1533 simplify ANDs of ORs for DNF
542     flatten(deduplicateOrs(flatten(filter)))
543   }

557       case f: Filter => f
558     }

561   private [filter] def flattenAnd(filters: Seq[Filter]): ListBuffer[Filter] = {
562     val remaining = ListBuffer.empty[Filter] ++ filters
563     val result = ListBuffer.empty[Filter]
564     while (remaining.nonEmpty) {
565       remaining.remove(0) match {
566         case f: And => remaining.appendAll(f.getChildren.asScala)
567         case f      => result.append(flatten(f))
568       }
