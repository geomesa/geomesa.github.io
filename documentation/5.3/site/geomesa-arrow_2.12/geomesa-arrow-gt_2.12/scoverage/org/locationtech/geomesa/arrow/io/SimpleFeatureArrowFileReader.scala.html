29 /**
30   * For reading simple features from an arrow file written by SimpleFeatureArrowFileWriter.
31   *
32   * Expects arrow streaming format (no footer). Can handle multiple 'files' in a single input stream
33   */
34 trait SimpleFeatureArrowFileReader extends Closeable {
35 
36   /**
37     * The simple feature type for the file. Note: this may change as features are read,
38     * if there are multiple logical 'files' in the input stream. By convention, we keep
39     * a single file with a single sft, but that is not enforced.
40     *
41     * @return current simple feature type
42     */
43   def sft: SimpleFeatureType
44 
45   /**
46     * Dictionaries from the file. Note: this may change as features are read, if there are
47     * multiple logical 'files' in the input stream. This method is exposed for completeness,
48     * but generally would not be needed since dictionary values are automatically decoded
49     * into the returned simple features.
50     *
51     * @return current dictionaries, keyed by attribute
52     */
53   def dictionaries: Map[String, ArrowDictionary]
54 
55   def vectors: Seq[SimpleFeatureVector]
56 
57   /**
58     * Reads features from the underlying arrow file
59     *
60     * @param filter filter to apply
61     * @return
62     */
63   def features(filter: Filter = Filter.INCLUDE): CloseableIterator[ArrowSimpleFeature]
64 }

74   /**

76     * the input stream. Returned features will be valid until `close()` is called

86     *
87     * @param is creates a new input stream for reading
88     * @return
89     */

92   /**
93    * A reader that streams results. Note that `features()` can only be called one time, as the input stream will be exhausted.

95    *
96    * @param is input stream
97    * @return

99   def streaming(is: InputStream): SimpleFeatureArrowFileReader = new StreamingSimpleFeatureArrowFileReader(is)

102    * Create a reader from a byte array. Returned features may not be valid after a call to `next()`, as the

105    * @param bytes file bytes
106    * @return
107    */
108   def streaming(bytes: Array[Byte]): SimpleFeatureArrowFileReader = streaming(() => new ByteArrayInputStream(bytes))

110   /**

128   /**
129     *
130     * @param fields dictionary encoded fields
131     * @param provider dictionary provider
132     * @return
133     */
134   private [io] def loadDictionaries(
135       fields: Seq[Field],
136       provider: DictionaryProvider,
137       precision: SimpleFeatureEncoding): Map[String, ArrowDictionary] = {
138     fields.flatMap { field =>
139       // check top-level dictionaries plus nested (i.e. for list-type attributes)
140       val encodings = Seq(field.getDictionary) ++ field.getChildren.asScala.map(_.getDictionary)
141       encodings.collect { case encoding if encoding != null =>
142         val descriptor = SimpleFeatureTypes.createDescriptor(field.getMetadata.get(DescriptorKey))
143         val bindings = {
144           val main = ObjectType.selectType(descriptor)
145           // for list types, get the list item binding (which is the tail of the bindings)
146           if (descriptor.isList) { main.tail } else { main }
147         }
148         val vector = provider.lookup(encoding.getId).getVector
149         field.getName -> ArrowDictionary.create(encoding, vector, bindings, precision)
150       }
151     }.toMap
152   }

154   /**
155     * Reads features from simple feature vectors based on a filter
156     *
157     * @param sft simple feature type
158     * @param filter filter
159     * @param skip indicator that we should skip any further batches
160     * @param sort sort for the file being read, if any
161     * @param dictionaries dictionaries
162     * @return
163     */
164   private [io] def features(sft: SimpleFeatureType,
165                             filter: Filter,
166                             skip: SkipIndicator,
167                             sort: Option[(String, Boolean)],
168                             dictionaries: Map[String, ArrowDictionary]): VectorToIterator = {
169     val optimized = ArrowFilterOptimizer.rewrite(filter, sft, dictionaries)
170     sort match {
171       case None => features(_, optimized)
172       case Some((field, reverse)) =>
173         val i = sft.indexOf(field)
174         val binding = sft.getDescriptor(i).getType.getBinding
175         val bounds = FilterHelper.extractAttributeBounds(filter, field, binding).values
176         if (bounds.isEmpty) {
177           features(_, optimized)
178         } else {
179           sortedFeatures(_, optimized, skip, bounds.asInstanceOf[Seq[Bounds[Comparable[Any]]]], i, reverse)
180         }
181     }
182   }
183 
184   /**
185     * Reads features from a simple feature vector
186     *
187     * @param vector simple feature vector
188     * @param filter filter
189     * @return
190     */
191   private def features(vector: SimpleFeatureVector, filter: Filter): Iterator[ArrowSimpleFeature] = {
192     val total = vector.reader.getValueCount
193     if (total == 0) { Iterator.empty } else {
194       // re-use the same feature object
195       val feature = vector.reader.feature
196       val all = Iterator.range(0, total).map { i => vector.reader.load(i); feature }
197       if (filter == Filter.INCLUDE) { all } else {
198         all.filter(filter.evaluate)
199       }
200     }
201   }
202 
203   /**
204     * Reads features from a simple feature vector. The underlying features are assumed to be sorted
205     *
206     * @param vector simple feature vector
207     * @param filter filter
208     * @param skip will be toggled if no further vectors need to be queried due to sort order and filter bounds
209     * @param filterBounds bounds for the sort field, extracted from the filter
210     * @param sortField field that the features are sorted by
211     * @param reverse if the sort order is reversed or not
212     * @return
213     */
214   private def sortedFeatures(vector: SimpleFeatureVector,
215                              filter: Filter,
216                              skip: SkipIndicator,
217                              filterBounds: Seq[Bounds[Comparable[Any]]],
218                              sortField: Int,
219                              reverse: Boolean): Iterator[ArrowSimpleFeature] = {
220     val total = vector.reader.getValueCount
221 
222     if (total == 0 || skip.skip) { Iterator.empty } else {
223       // re-use the same feature object
224       val feature = vector.reader.feature
225 
226       // bounds for the current batch
227       val currentBatchBounds = {
228         vector.reader.load(0)
229         val lo = Bound(Option(feature.getAttribute(sortField).asInstanceOf[Comparable[Any]]), inclusive = true)
230         vector.reader.load(total - 1)
231         val hi = Bound(Option(feature.getAttribute(sortField).asInstanceOf[Comparable[Any]]), inclusive = true)
232         if (reverse) { Bounds(hi, lo) } else { Bounds(lo, hi) }
233       }
234 
235       if (filterBounds.exists(Bounds.intersection(_, currentBatchBounds).isDefined)) {
236         // we have a match in this batch
237         val all = Iterator.range(0, vector.reader.getValueCount).map { i => vector.reader.load(i); feature }
238         all.filter(filter.evaluate)
239       } else {
240         // nothing from this batch matches, check to see if any further batches could match
241         val hasMore = if (reverse) {
242           filterBounds.exists(fb => Bounds.smallerLowerBound(fb.lower, currentBatchBounds.lower).eq(fb.lower))
243         } else {
244           filterBounds.exists(fb => Bounds.largerUpperBound(fb.upper, currentBatchBounds.upper).eq(fb.upper))
245         }
246         // toggle the skip indicator if there are no further batches that could match
247         skip.skip = !hasMore
248         Iterator.empty
249       }
250     }
251   }
252 
253   // holder for a skip indicator - this will be toggled if we ever determine there can be no more results
254   private [io] class SkipIndicator(var skip: Boolean = false)
