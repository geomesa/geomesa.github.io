99   /**
100    * Create a transfer pair between two vectors. This handles geometry vectors correctly, which the underlying
101    * arrow transfer pairs do not.
102    *
103    * @param from from vector
104    * @param to to vector
105    * @return transfer(fromIndex, toIndex)
106    */
107   def createTransferPair(sft: SimpleFeatureType, from: FieldVector, to: FieldVector): (Int, Int) => Unit = {
108     val i = sft.indexOf(from.getField.getName)
109     lazy val binding = sft.getDescriptor(i).getType.getBinding
110     if (i != -1 && classOf[Geometry].isAssignableFrom(binding)) {
111       // geometry vectors use FixedSizeList vectors, for which transfer pairs aren't implemented
112       val fromGeom = GeometryFields.wrap(from, binding).asInstanceOf[GeometryVector[Geometry, FieldVector]]
113       val toGeom = GeometryFields.wrap(to, binding).asInstanceOf[GeometryVector[Geometry, FieldVector]]
114       (fromIndex: Int, toIndex: Int) => fromGeom.transfer(fromIndex, toIndex, toGeom)
115     } else {
116       val transfer = from.makeTransferPair(to)
117       (fromIndex: Int, toIndex: Int) => transfer.copyValueSafe(fromIndex, toIndex)
118     }
119   }

121   /**
122    * Write out the header, dictionaries, and first batch of an arrow streaming file
123    *
124    * @param result vector loaded with first batch
125    * @param dictionaries dictionaries
126    * @param sort sort
127    * @param count number of records in first batch
128    * @return
129    */
130   def writeHeaderAndFirstBatch(
131       result: SimpleFeatureVector,
132       dictionaries: Map[String, ArrowDictionary],
133       ipcOpts: IpcOption,
134       sort: Option[(String, Boolean)],
135       count: Int): Array[Byte] = {
136     val metadata = sort match {
137       case None => null
138       case Some((sortBy, reverse)) => getSortAsMetadata(sortBy, reverse)
139     }
140     // note: don't close root as it will close the underlying vector
141     val root = createRoot(result.underlying, metadata)
142     root.setRowCount(count)
143     val os = new ByteArrayOutputStream()
144     WithClose(SimpleFeatureArrowFileWriter.provider(dictionaries, result.encoding)) { provider =>
145       WithClose(new ArrowStreamWriter(root, provider, Channels.newChannel(os), ipcOpts)) { writer =>
146         writer.writeBatch()
147         os.toByteArray
148       }
149     }
150   }

179   private class ArrowFileIterator(
180       sft: SimpleFeatureType,
181       dictionaries: Map[String, ArrowDictionary],
182       encoding: SimpleFeatureEncoding,
183       sort: Option[(String, Boolean)],
184       ipcOpts: IpcOption,
185       batches: CloseableIterator[Array[Byte]],
186       firstBatchHasHeader: Boolean,
187       footer: Array[Byte]
188     ) extends CloseableIterator[Array[Byte]] {
189 
190     private var seenBatch = false
191     private var seenFooter = false
192 
193     override def hasNext: Boolean = batches.hasNext || !seenBatch || !seenFooter
194 
195     override def next(): Array[Byte] = {
196       if (seenBatch) {
197         if (batches.hasNext) {
198           batches.next()
199         } else if (seenFooter) {
200           throw new NoSuchElementException("Next on an empty iterator")
201         } else {
202           seenFooter = true
203           footer
204         }
205       } else {
206         seenBatch = true
207         if (batches.hasNext) {
208           if (firstBatchHasHeader) { batches.next } else {
209             // add the file header and dictionaries
210             WithClose(SimpleFeatureVector.create(sft, dictionaries, encoding)) { vector =>
211               new RecordBatchLoader(vector.underlying).load(batches.next)
212               writeHeaderAndFirstBatch(vector, dictionaries, ipcOpts, sort, vector.reader.getValueCount)
213             }
214           }
215         } else {
216           // write out an empty batch so that we get the header and dictionaries
217           WithClose(SimpleFeatureVector.create(sft, dictionaries, encoding, 0)) { vector =>
218             writeHeaderAndFirstBatch(vector, dictionaries, ipcOpts, sort, 0)
219           }
220         }
221       }
222     }
223 
224     override def close(): Unit = batches.close()
225   }
