33   /**
34     * After filter comparing an expression (e.g. property or function) to a literal
35     *
36     * @param exp1 expression
37     * @param exp2 literal
38     */
39   private final class FastAfterLiteral(exp1: Expression, exp2: Literal)
40       extends FastTemporalOperator(exp1, exp2, After.NAME) with After {
41 
42     private val lit = FastConverter.convert(exp2.evaluate(null), classOf[Date])
43 
44     override def evaluate(obj: AnyRef): Boolean = {
45       val date = exp1.evaluate(obj).asInstanceOf[Date]
46       date != null && date.after(lit)
47     }
48 
49     override def accept(visitor: FilterVisitor, extraData: AnyRef): AnyRef = visitor.visit(this, extraData)
50   }
51

110   /**
111     * During filter comparing an expression (e.g. property or function) to a literal
112     *
113     * @param exp1 expression
114     * @param exp2 literal
115     */
116   private final class FastDuring(exp1: Expression, exp2: Literal)
117       extends FastTemporalOperator(exp1, exp2, During.NAME) with During {
118 
119     private val lit = FastConverter.convert(exp2.evaluate(null), classOf[Period])
120     private val beg = lit.getBeginning.getPosition.getDate
121     private val end = lit.getEnding.getPosition.getDate
122 
123     override def evaluate(obj: AnyRef): Boolean = {
124       val date = exp1.evaluate(obj).asInstanceOf[Date]
125       date != null && date.after(beg) && date.before(end)
126     }
127 
128     override def accept(visitor: FilterVisitor, extraData: AnyRef): AnyRef = visitor.visit(this, extraData)
129   }

132 sealed private abstract class FastTemporalOperator(exp1: Expression, exp2: Expression, private val op: String)
133     extends BinaryTemporalOperator {
134 
135   override def getExpression1: Expression = exp1
136 
137   override def getExpression2: Expression = exp2
138 
139   override def getMatchAction: MatchAction = MatchAction.ANY
140 
141   override def toString: String = s"[ $exp1 $op $exp2 ]"
142 
143   def canEqual(other: Any): Boolean = other.isInstanceOf[FastTemporalOperator]
144 
145   override def equals(other: Any): Boolean = other match {
146     case that: FastTemporalOperator =>
147       (that canEqual this) && exp1 == that.getExpression1 && exp2 == that.getExpression2 && op == that.op
148     case _ => false
149   }
150 
151   override def hashCode(): Int =
152     Seq(exp1, exp2, op).map(_.hashCode()).foldLeft(0)((a, b) => 31 * a + b)
