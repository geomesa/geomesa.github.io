93   /**
94     * Calculates ranges in index space that match any of the input bounds. Uses breadth-first searching to
95     * allow a limit on the number of ranges returned.
96     *
97     * To improve performance, the following decisions have been made:
98     *   uses loops instead of foreach/maps
99     *   uses java queues instead of scala queues
100     *   allocates initial sequences of decent size
101     *   sorts once at the end before merging
102     *
103     * @param zbounds search space
104     * @param precision precision to consider, in bits (max 64)
105     * @param maxRanges loose cap on the number of ranges to return. A higher number of ranges will have less
106     *                  false positives, but require more processing.
107     * @param maxRecurse max levels of recursion to apply before stopping
108     * @return ranges covering the search space
109     */
110   def zranges(zbounds: Array[ZRange],
111               precision: Int = 64,
112               maxRanges: Option[Int] = None,
113               maxRecurse: Option[Int] = Some(ZN.DefaultRecurse)): Seq[IndexRange] = {
114 
115     import ZN.LevelTerminator
116 
117     // stores our results - initial size of 100 in general saves us some re-allocation
118     val ranges = new java.util.ArrayList[IndexRange](100)
119 
120     // values remaining to process - initial size of 100 in general saves us some re-allocation
121     val remaining = new java.util.ArrayDeque[(Long, Long)](100)
122 
123     // calculate the common prefix in the z-values - we start processing with the first diff
124     val ZPrefix(commonPrefix, commonBits) = longestCommonPrefix(zbounds.flatMap(b => Seq(b.min, b.max)): _*)
125 
126     var offset = 64 - commonBits
127 
128     // checks if a range is contained in the search space
129     def isContained(range: ZRange): Boolean = {
130       var i = 0
131       while (i < zbounds.length) {
132         if (contains(zbounds(i), range)) {
133           return true
134         }
135         i += 1
136       }
137       false
138     }
139 
140     // checks if a range overlaps the search space
141     def isOverlapped(range: ZRange): Boolean = {
142       var i = 0
143       while (i < zbounds.length) {
144         if (overlaps(zbounds(i), range)) {
145           return true
146         }
147         i += 1
148       }
149       false
150     }
151 
152     // checks a single value and either:
153     //   eliminates it as out of bounds
154     //   adds it to our results as fully matching, or
155     //   queues up it's children for further processing
156     def checkValue(prefix: Long, quadrant: Long): Unit = {
157       val min: Long = prefix | (quadrant << offset) // QR + 000...
158       val max: Long = min | (1L << offset) - 1 // QR + 111...
159       val quadrantRange = ZRange(min, max)
160 
161       if (isContained(quadrantRange) || offset < 64 - precision) {
162         // whole range matches, happy day
163         ranges.add(IndexRange(min, max, contained = true))
164       } else if (isOverlapped(quadrantRange)) {
165         // some portion of this range is excluded
166         // queue up each sub-range for processing
167         remaining.add((min, max))
168       }
169     }
170 
171     // bottom out and get all the ranges that partially overlapped but we didn't fully process
172     // note: this method is only called when we know there are items remaining in the queue
173     def bottomOut(): Unit = {
174       while ({{
175         val minMax = remaining.poll
176         if (!minMax.eq(LevelTerminator)) {
177           ranges.add(IndexRange(minMax._1, minMax._2, contained = false))
178         }
179       }; !remaining.isEmpty })()
180     }
181 
182     // initial level - we just check the single quadrant
183     checkValue(commonPrefix, 0)
184     remaining.add(LevelTerminator)
185     offset -= Dimensions
186 
187     // level of recursion
188     var level = 0
189 
190     val rangeStop = maxRanges.getOrElse(Int.MaxValue)
191     val recurseStop = maxRecurse.getOrElse(ZN.DefaultRecurse)
192 
193     while ({{
194       val next = remaining.poll
195       if (next.eq(LevelTerminator)) {
196         // we've fully processed a level, increment our state
197         if (!remaining.isEmpty) {
198           level += 1
199           offset -= Dimensions
200           if (level >= recurseStop || offset < 0) {
201             bottomOut()
202           } else {
203             remaining.add(LevelTerminator)
204           }
205         }
206       } else {
207         val prefix = next._1
208         var quadrant = 0L
209         while (quadrant < Quadrants) {
210           checkValue(prefix, quadrant)
211           quadrant += 1
212         }
213         // subtract one from remaining.size to account for the LevelTerminator
214         if (ranges.size + remaining.size - 1 >= rangeStop) {
215           bottomOut()
216         }
217       }
218     }; !remaining.isEmpty })()
219 
220     // we've got all our ranges - now reduce them down by merging overlapping values
221     ranges.sort(IndexRange.IndexRangeIsOrdered)
222 
223     var current = ranges.get(0) // note: should always be at least one range
224     val result = ArrayBuffer.empty[IndexRange]
225     var i = 1
226     while (i < ranges.size()) {
227       val range = ranges.get(i)
228       if (range.lower <= current.upper + 1) {
229         // merge the two ranges
230         current = IndexRange(current.lower, math.max(current.upper, range.upper), current.contained && range.contained)
231       } else {
232         // append the last range and set the current range for future merging
233         result.append(current)
234         current = range
235       }
236       i += 1
237     }
238     // append the last range - there will always be one left that wasn't added
239     result.append(current)
240 
241     result.toSeq
242   }

244   /**
245    * Cuts Z-Range in two and trims based on user space, can be used to perform augmented binary search
246    *
247    * @param xd: division point
248    * @param inRange: is xd in query range
249    */
250   def cut(r: ZRange, xd: Long, inRange: Boolean): List[ZRange] = {
251     if (r.min == r.max) {
252       Nil
253     } else if (inRange) {
254       if (xd == r.min) { // degenerate case, two nodes min has already been counted
255         ZRange(r.max, r.max) :: Nil
256       } else if (xd == r.max) { // degenerate case, two nodes max has already been counted
257         ZRange(r.min, r.min) :: Nil
258       } else {
259         ZRange(r.min, xd - 1) :: ZRange(xd + 1, r.max) :: Nil
260       }
261     } else {
262       val (litmax, bigmin) = zdivide(xd, r.min, r.max)
263       ZRange(r.min, litmax) :: ZRange(bigmin, r.max) :: Nil
264     }
265   }

267   /**
268     * Calculates the longest common binary prefix between two z longs
269     *
270     * @return (common prefix, number of bits in common)
271     */
272   def longestCommonPrefix(values: Long*): ZPrefix = {
273     var bitShift = TotalBits - Dimensions
274     var head = values.head >>> bitShift
275     while (values.tail.forall(v => (v >>> bitShift) == head) && bitShift > -1) {
276       bitShift -= Dimensions
277       head = values.head >>> bitShift
278     }
279     bitShift += Dimensions // increment back to the last valid value
280     ZPrefix(values.head & (Long.MaxValue << bitShift), 64 - bitShift)
281   }

283   /** Loads either 1000... or 0111... into starting at given bit index of a given dimension */
284   private def load(target: Long, p: Long, bits: Int, dim: Int): Long = {
285     val mask = ~(split(MaxMask >> (BitsPerDimension - bits)) << dim)
286     val wiped = target & mask
287     wiped | (split(p) << dim)
288   }

298   /**
299     * Implements the the algorithm defined in: Tropf paper to find:
300     * LITMAX: maximum z-index in query range smaller than current point, xd
301     * BIGMIN: minimum z-index in query range greater than current point, xd
302     *
303     * @param load: function that knows how to load bits into appropraite dimension of a z-index
304     * @param xd: z-index that is outside of the query range
305     * @param rmin: minimum z-index of the query range, inclusive
306     * @param rmax: maximum z-index of the query range, inclusive
307     * @return (LITMAX, BIGMIN)
308     */
309   private [zorder] def zdiv(load: (Long, Long, Int, Int) => Long, dims: Int)
310                            (xd: Long, rmin: Long, rmax: Long): (Long, Long) = {
311     require(rmin < rmax, "min ($rmin) must be less than max $(rmax)")
312     var zmin: Long = rmin
313     var zmax: Long = rmax
314     var bigmin: Long = 0L
315     var litmax: Long = 0L
316 
317     def bit(x: Long, idx: Int) = {
318       ((x & (1L << idx)) >> idx).toInt
319     }
320     def over(bits: Long)  = 1L << (bits - 1)
321     def under(bits: Long) = (1L << (bits - 1)) - 1
322 
323     var i = 64
324     while (i > 0) {
325       i -= 1
326 
327       val bits = i/dims+1
328       val dim  = i%dims
329 
330       ( bit(xd, i), bit(zmin, i), bit(zmax, i) ) match {
331         case (0, 0, 0) =>
332         // continue
333 
334         case (0, 0, 1) =>
335           zmax   = load(zmax, under(bits), bits, dim)
336           bigmin = load(zmin, over(bits), bits, dim)
337 
338         case (0, 1, 0) =>
339         // sys.error(s"Not possible, MIN <= MAX, (0, 1, 0)  at index $i")
340 
341         case (0, 1, 1) =>
342           bigmin = zmin
343           return (litmax, bigmin)
344 
345         case (1, 0, 0) =>
346           litmax = zmax
347           return (litmax, bigmin)
348 
349         case (1, 0, 1) =>
350           litmax = load(zmax, under(bits), bits, dim)
351           zmin = load(zmin, over(bits), bits, dim)
352 
353         case (1, 1, 0) =>
354         // sys.error(s"Not possible, MIN <= MAX, (1, 1, 0) at index $i")
355 
356         case (1, 1, 1) =>
357         // continue
358       }
359     }
360     (litmax, bigmin)
361   }
