17   /**
18     * Precision for serialized geometries
19     *
20     * Precision is the number of base-10 decimal places stored. A positive precision implies retaining
21     * information to the right of the decimal place, while a negative precision implies rounding up to
22     * the left of the decimal place
23     *
24     * See https://github.com/TWKB/Specification/blob/master/twkb.md#type--precision
25     *
26     * If precision exceeds the max precision for TWKB serialization, full double serialization will
27     * be used instead
28     *
29     * @param xy 4-bit signed integer precision for the x and y dimensions
30     *           for lat/lon, a precision of 6 is about 10cm
31     * @param z 3-bit unsigned integer precision for the z dimension (if used)
32     *          for meters, precision of 1 is 10cm
33     * @param m 3-bit unsigned integer precision for the z dimension (if used)
34     *          for milliseconds, precision of 0 is 1ms
35     */
36   case class TwkbPrecision(xy: Byte = 6, z: Byte = 1, m: Byte = 0) extends GeometryPrecision {
37     require(xy < 8 && xy > -8 && z < 8 && z > -1 && m < 8 && m > -1, s"Invalid TWKB precision: $xy,$z,$m")
38   }
39 }
