22 abstract class ThreadedConsumer(

43   class ConsumerRunnable(id: String, consumer: Consumer[Array[Byte], Array[Byte]], handler: ConsumerErrorHandler)
44       extends Runnable {
45 
46     private var errorCount = 0
47 
48     lazy private val topics = consumer.subscription().asScala.mkString(", ")
49 
50     private var lastOffsetCommit = System.currentTimeMillis()
51 
52     override def run(): Unit = {
53       try {
54         var interrupted = false
55         while (isOpen && !interrupted) {
56           try {
57             processPoll(KafkaConsumerVersions.poll(consumer, frequency))
58           } catch {
59             case _: WakeupException | _: InterruptException | _: InterruptedException => interrupted = true
60             case NonFatal(e) =>
61               errorCount += 1
62               logger.warn(s"Consumer [$id] error receiving message from topic $topics:", e)
63               if (errorCount <= 300 || handler.handle(id, e)) {
64                 Thread.sleep(1000)
65               } else {
66                 logger.error(s"Consumer [$id] shutting down due to too many errors from topic $topics:", e)
67                 throw e
68               }
69           }
70         }
71       } finally {
72         if (closeConsumers) {
73           try { consumer.close() } catch {
74             case NonFatal(e) => logger.warn(s"Error calling close on consumer: ", e)
75           }
76         }
77       }
78     }
79 
80     /**
81      * Process the results of a call to poll()
82      *
83      * @param result records
84      */
85     protected def processPoll(result: ConsumerRecords[Array[Byte], Array[Byte]]): Unit = {
86       if (result.isEmpty) {
87         logger.trace(s"Consumer [$id] poll received 0 records")
88       } else {
89         lazy val topics = result.partitions.asScala.map(tp => s"[${tp.topic}:${tp.partition}]").mkString(",")
90         logger.debug(s"Consumer [$id] poll received ${result.count()} records for $topics")
91         val records = result.iterator()
92         while (records.hasNext) {
93           consume(records.next())
94         }
95         logger.trace(s"Consumer [$id] finished processing ${result.count()} records from topic $topics")
96         if (System.currentTimeMillis() - lastOffsetCommit > offsetCommitIntervalMillis) {
97           logger.trace(s"Consumer [$id] committing offsets")
98           consumer.commitAsync(logOffsetCallback)
99           lastOffsetCommit = System.currentTimeMillis()
100         }
101         errorCount = 0 // reset error count
102       }
103     }
104   }
105 }
