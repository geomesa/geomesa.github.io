19 /**
20   * Feature cache implementation
21   *
22   * @param sft simple feature type
23   * @param config index config
24   */
25 class KafkaFeatureCacheImpl(sft: SimpleFeatureType, config: IndexConfig, layerViews: Seq[LayerView] = Seq.empty)
26     extends KafkaFeatureCache with FeatureExpiration {
27 
28   import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
29 
30   // keeps location and expiry keyed by feature ID (we need a way to retrieve a feature based on ID for
31   // update/delete operations). to reduce contention, we never iterate over this map
32   private val state = new ConcurrentHashMap[String, FeatureState]
33 
34   private val support = createSupport(sft)
35 
36   private val factory = FeatureStateFactory(sft, support.index, config.expiry, this, config.executor)
37 
38   override val views: Seq[KafkaFeatureCacheView] =
39     layerViews.map(view => KafkaFeatureCacheView(view, createSupport(view.viewSft)))
40 
41   logger.debug(s"Initialized KafkaFeatureCache with factory $factory and support $support")
42 
43   /**
44     * Note: this method is not thread-safe. The `state` and `index` can get out of sync if the same feature
45     * is updated simultaneously from two different threads
46     *
47     * In our usage, this isn't a problem, as a given feature ID is always operated on by a single thread
48     * due to kafka consumer partitioning
49     */
50   override def put(feature: SimpleFeature): Unit = {
51     if (feature.getDefaultGeometry == null) {
52       logger.warn(s"Null geometry detected for feature ${feature.getID}. Skipping loading into cache.")
53       return
54     }
55     val featureState = factory.createState(feature)
56     logger.trace(s"${featureState.id} adding feature $featureState")
57     val old = state.put(featureState.id, featureState)
58     if (old == null) {
59       featureState.insertIntoIndex()
60       views.foreach(_.put(feature))
61     } else if (old.time <= featureState.time) {
62       logger.trace(s"${featureState.id} removing old feature")
63       old.removeFromIndex()
64       featureState.insertIntoIndex()
65       views.foreach { view =>
66         view.remove(featureState.id)
67         view.put(feature)
68       }
69     } else {
70       logger.trace(s"${featureState.id} ignoring out of sequence feature")
71       if (!state.replace(featureState.id, featureState, old)) {
72         logger.warn(s"${featureState.id} detected inconsistent state... spatial index may be incorrect")
73         old.removeFromIndex()
74         views.foreach(_.remove(featureState.id))
75       }
76     }
77     logger.trace(s"Current index size: ${state.size()}/${support.index.size()}")
78   }
79 
80   /**
81     * Note: this method is not thread-safe. The `state` and `index` can get out of sync if the same feature
82     * is updated simultaneously from two different threads
83     *
84     * In our usage, this isn't a problem, as a given feature ID is always operated on by a single thread
85     * due to kafka consumer partitioning
86     */
87   override def remove(id: String): Unit = {
88     logger.trace(s"$id removing feature")
89     val old = state.remove(id)
90     if (old != null) {
91       old.removeFromIndex()
92       views.foreach(_.remove(id))
93     }
94     logger.trace(s"Current index size: ${state.size()}/${support.index.size()}")
95   }
96 
97   override def expire(featureState: FeatureState): Unit = {
98     logger.trace(s"${featureState.id} expiring from index")
99     if (state.remove(featureState.id, featureState)) {
100       featureState.removeFromIndex()
101       views.foreach(_.remove(featureState.id))
102     }
103     logger.trace(s"Current index size: ${state.size()}/${support.index.size()}")
104   }
105 
106   override def clear(): Unit = {
107     logger.trace("Clearing index")
108     state.clear()
109     support.index.clear()
110     views.foreach(_.clear())
111   }
112 
113   override def size(): Int = state.size()
114 
115   // optimized for filter.include
116   override def size(f: Filter): Int = if (f == Filter.INCLUDE) { size() } else { query(f).length }
117 
118   override def query(id: String): Option[SimpleFeature] =
119     Option(state.get(id)).flatMap(f => Option(f.retrieveFromIndex()))
120 
121   override def query(filter: Filter): Iterator[SimpleFeature] = support.query(filter)
122 
123   override def close(): Unit = factory.close()
124 
125   private def createSupport(sft: SimpleFeatureType): SpatialIndexSupport = {
126     if (config.cqAttributes.nonEmpty) {
127       // note: CQEngine handles points vs non-points internally
128       KafkaFeatureCache.cqIndexSupport(sft, config)
129     } else if (sft.isPoints) {
130       BucketIndexSupport(sft, config.resolution.x, config.resolution.y)
131     } else {
132       SizeSeparatedBucketIndexSupport(sft, config.ssiTiers, config.resolution.x / 360d, config.resolution.y / 180d)
133     }
134   }
