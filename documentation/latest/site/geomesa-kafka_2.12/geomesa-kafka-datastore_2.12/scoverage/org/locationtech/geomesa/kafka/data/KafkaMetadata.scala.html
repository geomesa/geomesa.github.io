108   /**
109    * Ensure that the topic has a cleanup policy of 'compact', so that feature types aren't aged off
110    *
111    * @param adminClient admin client
112    */
113   private def ensureCompactionPolicy(adminClient: AdminClient): Unit = {
114     val catalogResource = new ConfigResource(ConfigResource.Type.TOPIC, config.catalog)
115     val catalogConfigs = adminClient.describeConfigs(Collections.singleton(catalogResource)).all().get().values()
116     val isCompact = catalogConfigs.asScala.exists { config =>
117       config.get(CleanupPolicyConfig) != null && config.get(CleanupPolicyConfig).value() == CompactCleanupPolicy
118     }
119     if (!isCompact) {
120       val catalogConfigEntry = new ConfigEntry(CleanupPolicyConfig, CompactCleanupPolicy)
121       val alterOps = Collections.singleton(new AlterConfigOp(catalogConfigEntry, AlterConfigOp.OpType.SET))
122       adminClient.incrementalAlterConfigs(Collections.singletonMap(catalogResource, alterOps)).all().get()
123     }
124   }

126   private def adminClientOp[V](fn: AdminClient => V): V = {
127     val props = new Properties()
128     props.put(AdminClientConfig.BOOTSTRAP_SERVERS_CONFIG, config.brokers)
129     config.consumers.properties.foreach { case (k, v) => props.put(k, v) }
130     config.producers.properties.foreach { case (k, v) => props.put(k, v) }

134   /**
135    * Models the topic as a map of key-value pairs
136    */
137   private class TopicMap extends Runnable with Closeable {
138 
139     private val groupId = UUID.randomUUID().toString
140     private val poll = Duration.of(100, ChronoUnit.MILLIS)
141 
142     private val state = new ConcurrentHashMap[KeyBytes, Array[Byte]]()
143     private val complete = new CountDownLatch(1)
144     private val closed = new AtomicBoolean(false)
145 
146     private val consumer =
147       KafkaDataStore.consumer(config.brokers,
148         config.consumers.properties ++ Map(GROUP_ID_CONFIG -> groupId, AUTO_OFFSET_RESET_CONFIG -> "earliest"))
149 
150     private var future: Future[_] = _
151 
152     KafkaConsumerVersions.subscribe(consumer, config.catalog)
153     doInitialLoad()
154 
155     override def run(): Unit = {
156       try {
157         var interrupted = Thread.currentThread().isInterrupted
158         while (!interrupted) {
159           try {
160             val result = KafkaConsumerVersions.poll(consumer, poll)
161             if (!result.isEmpty) {
162               val records = result.iterator()
163               while (records.hasNext) {
164                 val r = records.next()
165                 val v = r.value()
166                 if (v == null) {
167                   state.remove(KeyBytes(r.key()))
168                 } else {
169                   state.put(KeyBytes(r.key()), v)
170                 }
171               }
172               consumer.commitAsync()
173             }
174           } catch {
175             case _: WakeupException | _: InterruptException | _: InterruptedException => interrupted = true
176             case NonFatal(e) =>
177               logger.warn(s"Consumer [$groupId] error receiving message from topic:", e)
178               Thread.sleep(1000)
179           }
180         }
181       } finally {
182         complete.countDown()
183       }
184     }
185 
186     private def doInitialLoad(): Unit = {
187       try {
188         val offsets = scala.collection.mutable.Map.empty[Int, Long]
189         // noinspection RedundantCollectionConversion
190         val partitions = consumer.partitionsFor(config.catalog).asScala.map(_.partition).toSeq
191         // note: end offsets are the *next* offset that will be returned, so subtract one to track the last offset
192         // we will actually consume
193         offsets ++=
194             KafkaConsumerVersions.endOffsets(consumer, config.catalog, partitions)
195                 .collect { case (p, o) if o > 0 => (p, o - 1) }
196         while (offsets.nonEmpty) {
197           val result = KafkaConsumerVersions.poll(consumer, poll)
198           if (!result.isEmpty) {
199             val records = result.iterator()
200             while (records.hasNext) {
201               val r = records.next()
202               val v = r.value()
203               if (v == null) {
204                 state.remove(KeyBytes(r.key()))
205               } else {
206                 state.put(KeyBytes(r.key()), v)
207               }
208               if (offsets.get(r.partition()).exists(o => r.offset() >= o)) {
209                 offsets.remove(r.partition())
210               }
211             }
212             consumer.commitAsync()
213           }
214         }
215         lazy val stateStrings =
216           state.asScala.map { case (k, v) => new String(k.bytes, StandardCharsets.UTF_8) -> new String(v, StandardCharsets.UTF_8)}
217         logger.debug(s"Completed initial load of catalog '${config.catalog}': \n  ${stateStrings.mkString("\n  ")}")
218 
219         future = CachedThreadPool.submit(this)
220         sys.addShutdownHook(future.cancel(true)) // prevent consumer from hanging if ds isn't disposed properly
221       } catch {
222         case NonFatal(e) => complete.countDown(); throw e
223       }
224     }
225 
226     def get(key: Array[Byte]): Option[Array[Byte]] = Option(state.get(KeyBytes(key)))
227 
228     def all(): CloseableIterator[(Array[Byte], Array[Byte])] =
229       CloseableIterator(state.asScala.iterator.map { case (k, v) => k.bytes -> v })
230 
231     def prefix(prefix: Array[Byte]): CloseableIterator[(Array[Byte], Array[Byte])] =
232       all().filter { case (k, _) => k.startsWith(prefix) }
233 
234     override def close(): Unit = {
235       try {
236         if (future != null) {
237           future.cancel(true)
238         }
239         complete.await(10, TimeUnit.SECONDS)
240       } finally {
241         // avoid checking consumer assignment if consumer is already closed, which will throw an error
242         if (closed.compareAndSet(false, true)) {
243           cleanupConsumer()
244         }
245       }
246     }
247 
248     private def cleanupConsumer(): Unit = {
249       try {
250         val topics = consumer.assignment()
251         consumer.unsubscribe()
252         if (!topics.isEmpty) {
253           Try(adminClientOp(KafkaAdminVersions.deleteConsumerGroupOffsets(_, groupId, topics))).failed.foreach { e =>
254             logger.warn("Error deleting consumer group offsets:", e)
255           }
256         }
257       } finally {
258         consumer.close()
259       }
260     }
261   }

269     override def hashCode(): Int = java.util.Arrays.hashCode(bytes)
270     override def equals(obj: Any): Boolean = {
271       obj match {
272         case KeyBytes(other) => java.util.Arrays.equals(bytes, other)
273         case _ => false
274       }
275     }
