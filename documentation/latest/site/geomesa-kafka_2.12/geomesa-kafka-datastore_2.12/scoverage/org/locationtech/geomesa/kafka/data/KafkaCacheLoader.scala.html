56     /**
57      * Register a loader starting
58      *
59      * @param loader loader instance
60      * @return
61      */
62     def startLoad(loader: AnyRef): Boolean = synchronized {
63       if (!loading.add(loader)) {
64         logger.warn(s"Called startLoad for a loader that was already registered and has not yet completed: $loader")
65       }
66       firstLoadStartTime.compareAndSet(0L, System.currentTimeMillis())
67     }
68 
69     /**
70      * De-register a loader as having completed
71      *
72      * @param loader loader instance
73      */
74     def completedLoad(loader: AnyRef): Unit = synchronized {
75       if (!loading.remove(loader)) {
76         logger.warn(s"Called completedLoad for a loader that was not registered or already deregistered: $loader")
77       } else if (loading.isEmpty) {
78         logger.info(s"Last active initial load completed.  " +
79           s"Initial loads took ${System.currentTimeMillis() - firstLoadStartTime.get} milliseconds.")
80         firstLoadStartTime.set(0L)
81       }
82     }
83 
84     /**
85      * Indicates if any active loads are ongoing
86      *
87      * @return
88      */
89     def allLoaded(): Boolean = loading.isEmpty

154     override protected def consume(record: ConsumerRecord[Array[Byte], Array[Byte]]): Unit = {
155       val headers = RecordVersions.getHeaders(record)
156       val timestamp = RecordVersions.getTimestamp(record)
157       val message = serializer.deserialize(record.key(), record.value(), headers, timestamp)
158       logger.trace(s"Consumed message [$topic:${record.partition}:${record.offset}] $message")
159       message match {
160         case m: Change =>
161           cache.fireChange(timestamp, m.feature)
162           cache.put(m.feature)
163           updates.increment()
164           dtgMetrics.foreach { case DateMetrics(index, dtg, latency) =>
165             val date = m.feature.getAttribute(index).asInstanceOf[Date]
166             if (date != null) {
167               dtg.updateAndGet(prev => Math.max(prev, date.getTime))
168               latency.record(System.currentTimeMillis() - date.getTime)
169             }
170           }
171 
172         case m: Delete =>
173           cache.fireDelete(timestamp, m.id, cache.query(m.id).orNull)
174           cache.remove(m.id)
175           deletes.increment()
176 
177         case _: Clear =>
178           cache.fireClear(timestamp)
179           cache.clear()
180           clears.increment()
181 
182         case m => throw new IllegalArgumentException(s"Unknown message: $m")
183       }
184     }

186     /**
187      * Handles initial loaded 'from-beginning' without indexing features in the spatial index. Will still
188      * trigger message events.
189      *
190      * Since the entire consumer group is in this process, we shouldn't usually get rebalance events after the initial assignment.
191      * However, it can still occur if the number of partitions is administratively altered. Note that we don't try to
192      * re-read the topic if a partition is revoked and then reassigned.
193      *
194      * @param readBack initial load read back
195      */
196     private class InitialLoader(readBack: scala.concurrent.duration.Duration)
197         extends ThreadedConsumer(consumers, frequency, offsetCommitInterval, false) with Runnable with ConsumerRebalanceListener {
198 
199       import scala.collection.JavaConverters._
200 
201       private val cache = KafkaFeatureCache.nonIndexing(sft, initialLoadConfig)
202       private val toLoad = KafkaCacheLoaderImpl.this
203 
204       private val updates = Metrics.counter(s"$MetricsPrefix.readback.updates", tags)
205       private val deletes = Metrics.counter(s"$MetricsPrefix.readback.deletes", tags)
206       private val clears = Metrics.counter(s"$MetricsPrefix.readback.clears", tags)
207 
208       // track the offsets that we want to read to
209       private val offsets = new ConcurrentHashMap[Int, java.lang.Long]()
210       private val assignment = Collections.newSetFromMap(new ConcurrentHashMap[Int, java.lang.Boolean]())
211       @volatile
212       private var done: Boolean = false
213       private var latch: CountDownLatch = _
214       @volatile
215       private var submission: Future[_] = _

217       override def onPartitionsAssigned(topicPartitions: java.util.Collection[TopicPartition]): Unit = {
218         logger.debug(s"Partitions assigned: ${topicPartitions.asScala.mkString(", ")}")
219         topicPartitions.asScala.foreach { tp =>
220           if (assignment.add(tp.partition())) {
221             val consumer = consumers.find(_.assignment().contains(tp)).orNull
222             if (consumer == null) {
223               logger.warn("Partition assigned but no consumer contains the assignment")
224             } else {
225               KafkaConsumerVersions.pause(consumer, tp)
226               try {
227                 logger.debug(s"Checking offsets for [${tp.topic()}:${tp.partition()}]")
228                 // the only reliable way we've found to check max offset is to seek to the end and check the position there
229                 consumer.seekToEnd(Collections.singleton(tp))
230                 val end = consumer.position(tp)
231                 logger.debug(s"Setting max offset to [${tp.topic}:${tp.partition}:${end - 1}]")
232                 offsets.put(tp.partition(), end - 1)
233                 try {
234                   if (!readBack.isFinite) {
235                     KafkaConsumerVersions.seekToBeginning(consumer, tp)
236                   } else {
237                     val offset = Try {
238                       val time = System.currentTimeMillis() - readBack.toMillis
239                       KafkaConsumerVersions.offsetsForTimes(consumer, tp.topic, Seq(tp.partition), time).get(tp.partition)
240                     }
241                     offset match {
242                       case Success(Some(o)) =>
243                         logger.debug(s"Seeking to offset $o for read-back $readBack on [${tp.topic}:${tp.partition}]")
244                         consumer.seek(tp, o)
245 
246                       case Success(None) =>
247                         logger.debug(s"No prior offset found for read-back $readBack on [${tp.topic}:${tp.partition}], " +
248                           "reading from head of queue")
249 
250                       case Failure(e) =>
251                         logger.warn(s"Error finding initial offset: [${tp.topic}:${tp.partition}], seeking to beginning", e)
252                         KafkaConsumerVersions.seekToBeginning(consumer, tp)
253                     }
254                   }
255                 } finally {
256                   checkComplete(consumer, tp)
257                 }
258               } finally {
259                 KafkaConsumerVersions.resume(consumer, tp)
260               }
261             }
262           }
263         }
264       }

266       override def onPartitionsRevoked(topicPartitions: java.util.Collection[TopicPartition]): Unit = {
267         logger.debug(s"Partitions revoked: ${topicPartitions.asScala.mkString(", ")}")
268         topicPartitions.asScala.foreach { tp =>
269           if (offsets.remove(tp.partition) != null) {
270             latch.countDown()
271             logger.info(
272               s"Stopping initial load due to revocation of assignment for [$topic:${tp.partition}], " +
273                 s"${latch.getCount} partitions remaining")
274           }
275         }
276       }

285       override protected def consume(record: ConsumerRecord[Array[Byte], Array[Byte]]): Unit = {
286         if (done) { toLoad.consume(record) } else {
287           val headers = RecordVersions.getHeaders(record)
288           val timestamp = RecordVersions.getTimestamp(record)
289           val message = serializer.deserialize(record.key, record.value, headers, timestamp)
290           logger.trace(s"Consumed message [$topic:${record.partition}:${record.offset}] $message")
291           message match {
292             case m: Change =>
293               toLoad.cache.fireChange(timestamp, m.feature)
294               cache.put(m.feature)
295               updates.increment()
296 
297             case m: Delete =>
298               toLoad.cache.fireDelete(timestamp, m.id, cache.query(m.id).orNull)
299               cache.remove(m.id)
300               deletes.increment()
301 
302             case _: Clear =>
303               toLoad.cache.fireClear(timestamp)
304               cache.clear()
305               clears.increment()
306 
307             case m => throw new IllegalArgumentException(s"Unknown message: $m")
308           }
309           if (record.offset > 0 && record.offset % 1048576 == 0) { // magic number 2^20
310             logger.info(s"Initial load: consumed [$topic:${record.partition}:${record.offset}]")
311           }
312         }
313       }

315       def start(): Unit = {
316         LoaderStatus.startLoad(this)
317         try {
318           val partitions = consumers.head.partitionsFor(topic).asScala.map(_.partition)
319           logger.info(s"Starting initial load for [$topic] with ${partitions.size} partitions")
320           latch = new CountDownLatch(partitions.size)
321           startConsumers() // kick off the asynchronous consumer threads
322           submission = CachedThreadPool.submit(this)
323         } catch {
324           case NonFatal(e) =>
325             LoaderStatus.completedLoad(this)
326             throw e
327         }
328       }
329 
330       override def run(): Unit = {
331         try {
332           try { latch.await() } finally {
333             // stop the consumer threads, but won't close the consumers due to `closeConsumers`
334             // note: don't call this.close() as it would interrupt this thread
335             super.close()
336           }
337           // set a flag just in case the consumer threads haven't finished spinning down, so that we will
338           // pass any additional messages back to the main loader
339           done = true
340           logger.info(s"Finished initial load, transferring to indexed cache for [$topic]")
341           cache.query(Filter.INCLUDE).foreach(toLoad.cache.put)
342           logger.info(s"Finished transfer for [$topic], starting normal load")
343           toLoad.startConsumers()
344         } finally {
345           LoaderStatus.completedLoad(this)
346         }
347       }
348 
349       override def close(): Unit = {
350         try { super.close() } finally {
351           if (submission != null && !submission.isDone) {
352             submission.cancel(true)
353           }
354         }
355       }
356 
357       /**
358        * Checks the current position of the consumer, and notifies through the countdown latch when the initial load
359        * is complete
360        *
361        * @param consumer consumer
362        * @param tp topic and partition to check
363        */
364       private def checkComplete(consumer: Consumer[Array[Byte], Array[Byte]], tp: TopicPartition): Unit = {
365         val position = consumer.position(tp)
366         // once we've hit the max offset for the partition, remove from the offset map so we don't double count it
367         if (position > offsets.getOrDefault(tp.partition(), Long.MaxValue) && offsets.remove(tp.partition) != null) {
368           latch.countDown()
369           logger.info(s"Initial load completed for [$topic:${tp.partition}], ${latch.getCount} partitions remaining")
370         }
371       }
372 
373       /**
374        * Consumer runnable that tracks when we have completed the initial load
375        *
376        * @param id id
377        * @param consumer consumer
378        * @param handler error handler
379        */
380       private class InitialLoaderConsumerRunnable(id: String, consumer: Consumer[Array[Byte], Array[Byte]], handler: ConsumerErrorHandler)
381           extends ConsumerRunnable(id, consumer, handler) {
382         override protected def processPoll(result: ConsumerRecords[Array[Byte], Array[Byte]]): Unit = {
383           try { super.processPoll(result) } finally {
384             result.partitions().asScala.foreach(checkComplete(consumer, _))
385           }
386         }
387       }
388     }
