19 object TruncateTopic extends LazyLogging {
20 
21   /**
22    * Truncation of a Kafka topic
23    *
24    * @param admin admin
25    * @param topic the topic name to truncate
26    */
27   def apply(admin: Admin, topic: String): Unit = {
28     val deleteCleanupPolicy = hasDeleteCleanupPolicy(admin, topic)
29     val latestRecords = getRecordsToDelete(admin, topic)
30     try {
31       if (!deleteCleanupPolicy) {
32         logger.debug(s"Adding 'delete' cleanup policy to topic $topic so it can be truncated")
33         alterDeleteCleanupPolicy(admin, topic, AlterConfigOp.OpType.APPEND)
34       }
35       logger.debug(s"Starting truncate of topic $topic")
36       admin.deleteRecords(latestRecords.asJava).all().get()
37       logger.debug(s"Completed truncate of topic $topic")
38     } finally {
39       if (!deleteCleanupPolicy) {
40         logger.debug(s"Removing 'delete' cleanup policy to topic $topic")
41         alterDeleteCleanupPolicy(admin, topic, AlterConfigOp.OpType.SUBTRACT)
42       }
43     }
44   }
45 
46   /**
47    * Get the latest offsets for a topic, by partition
48    */
49   private def getRecordsToDelete(admin: Admin, topic: String): Map[TopicPartition, RecordsToDelete] = {
50     val topicInfo = admin.describeTopics(Collections.singleton(topic)).topicNameValues().get(topic).get()
51     val partitions = topicInfo.partitions().asScala.map(info => new TopicPartition(topic, info.partition()) -> OffsetSpec.latest())
52     val offsets = admin.listOffsets(partitions.toMap.asJava).all().get().asScala.toMap
53     offsets.map { case (tp, info) => tp -> RecordsToDelete.beforeOffset(info.offset()) }
54   }
55 
56   /**
57    * Check if the topic has the 'delete' cleanup policy.
58    */
59   private def hasDeleteCleanupPolicy(admin: Admin, topicName: String): Boolean = {
60     val configResource = new ConfigResource(ConfigResource.Type.TOPIC, topicName)
61     val configsResult = admin.describeConfigs(Collections.singleton(configResource))
62     val config = configsResult.values().get(configResource).get()
63     config.get(TopicConfig.CLEANUP_POLICY_CONFIG).value().contains(TopicConfig.CLEANUP_POLICY_DELETE)
64   }
65 
66   /**
67    * Alter the 'delete' cleanup policy in the topic's 'cleanup.policy' config
68    */
69   private def alterDeleteCleanupPolicy(admin: Admin, topic: String, op: AlterConfigOp.OpType): Unit = {
70     val configResource = new ConfigResource(ConfigResource.Type.TOPIC, topic)
71     val alterConfigOp = new AlterConfigOp(new ConfigEntry(TopicConfig.CLEANUP_POLICY_CONFIG, TopicConfig.CLEANUP_POLICY_DELETE), op)
72     admin.incrementalAlterConfigs(java.util.Map.of(configResource, Collections.singleton(alterConfigOp))).all().get()
73   }
