291   /**
292    * Parse the dynamic expiry param value into a seq of pairs
293    *
294    * @param params data store params
295    * @return
296    */
297   private[data] def parseDynamicExpiry(params: java.util.Map[String, _]): Seq[(String, Duration)] = {
298     lazy val key = s"Invalid property for parameter '${KafkaDataStoreParams.DynamicCacheExpiry.key}'"
299     val expiry = KafkaDataStoreParams.DynamicCacheExpiry.lookupOpt(params).toSeq.flatMap { value =>
300       ConfigFactory.parseString(value).resolve().root().unwrapped().asScala.toSeq.map {
301         case (filter, exp: String) =>
302           // validate the filter, but leave it as a string so we can optimize it based on the sft later
303           try { ECQL.toFilter(filter) } catch {
304             case NonFatal(e) => throw new IOException(s"$key, expected a CQL filter but got: $filter", e)
305           }
306           val duration = try { Duration(exp) } catch {
307             case NonFatal(e) => throw new IOException(s"$key, expected a duration for key '$filter' but got: $exp", e)
308           }
309           filter -> duration
310 
311         case (filter, exp) =>
312           throw new IOException(s"$key, expected a JSON string for key '$filter' but got: $exp")
313       }
314     }
315     if (expiry.dropRight(1).exists(_._1.equalsIgnoreCase("INCLUDE"))) {
316       throw new IOException(s"$key, defined a filter after Filter.INCLUDE (which would never be invoked)")
317     }
318     expiry
319   }

321   /**
322    * Parse the typesafe config for a layer view. Views take the form:
323    *
324    * {
325    *   foo-sft = [
326    *     {
327    *       type-name = foo-sft-enhanced
328    *       filter = "foo = bar"
329    *       transform = [ "foo", "bar", "baz", "blu" ]
330    *     },
331    *     {
332    *       type-name = foo-sft-reduced
333    *       filter = "foo = baz"
334    *       transform = [ "foo", "bar", "baz" ]
335    *     }
336    *   ]
337    * }
338    *
339    * @param params params
340    * @return
341    */
342   private[kafka] def parseLayerViewConfig(params: java.util.Map[String, _]): Map[String, Seq[LayerViewConfig]] = {
343     def asConfigObject(o: AnyRef) = o match {
344       case c: ConfigObject => c
345       case _ => throw new IllegalArgumentException(s"Invalid layer view, expected a config object but got: $o")
346     }
347 
348     KafkaDataStoreParams.LayerViews.lookupOpt(params) match {
349       case None => Map.empty[String, Seq[LayerViewConfig]]
350       case Some(conf) =>
351         val config = ConfigFactory.parseString(conf).resolve()
352         val entries = config.entrySet().asScala.map { e =>
353           val views = e.getValue match {
354             case c: ConfigList => c.asScala.map(asConfigObject)
355             case c => Seq(asConfigObject(c))
356           }
357           e.getKey -> views.map { c =>
358             ConfigSource.fromConfig(c.toConfig).loadOrThrow[LayerViewConfig](LayerViewClassTag, LayerViewReader)
359           }
360         }
361         val configs = entries.map(f => (f._1, f._2.toSeq))
362         val typeNames = configs.toSeq.flatMap(_._2.map(_.typeName))
363         if (typeNames != typeNames.distinct) {
364           throw new IllegalArgumentException(
365             s"Detected duplicate type name in layer view config: ${config.root().render(ConfigRenderOptions.concise)}")
366         }
367         configs.toMap
368     }
369   }

377   private def readLayerViewConfig(cur: ConfigCursor): Either[ConfigReaderFailures, LayerViewConfig] = {
378     val config = for {
379       obj       <- cur.asObjectCursor.right
380       typeName  <- obj.atKey("type-name").right.flatMap(_.asString).right
381       filter    <- readFilter(obj.atKeyOrUndefined("filter")).right
382       transform <- readTransform(obj.atKeyOrUndefined("transform")).right
383     } yield {
384       LayerViewConfig(typeName, filter, transform)
385     }
386     config.right.flatMap { c =>
387       if (c.filter.isEmpty && c.transform.isEmpty) {
388         val err = "LayerViews must define at least one of 'filter' or 'transform'"
389         cur.failed(new FailureReason { override def description: String = err })
390       } else {
391         Right(c)
392       }
393     }
394   }

396   private def readFilter(cur: ConfigCursor): Either[ConfigReaderFailures, Option[Filter]] = {
397     if (cur.isUndefined) { Right(None) } else {
398       cur.asString.right.flatMap { ecql =>
399         try { Right(Some(ECQL.toFilter(ecql)).filter(_ != Filter.INCLUDE)) } catch {
400           case NonFatal(e) => cur.failed(CannotConvert(ecql, "Filter", e.toString))
401         }
402       }
403     }
404   }

406   private def readTransform(cur: ConfigCursor): Either[ConfigReaderFailures, Option[Seq[String]]] = {
407     if (cur.isUndefined) { Right(None) } else {
408       val transforms = cur.asList.right.flatMap { list =>
409         list.foldLeft[Either[ConfigReaderFailures, Seq[String]]](Right(Seq.empty)) { case (res, elem) =>
410           res.right.flatMap(r => elem.asString.right.map(r :+ _))
411         }
412       }
413       transforms.right.map(t => if (t.isEmpty) { None } else { Some(t) })
414     }
415   }

454   private def parseDeprecatedMetrics(conf: Config): Option[MetricsConfig] = {
455     val reporters = if (conf.hasPath("reporters")) { conf.getConfigList("reporters").asScala } else { Seq(conf) }
456     val supported = reporters.flatMap { reporter =>
457       if (reporter.hasPath("type")) {
458         Option(reporter.getString("type")).flatMap {
459           case t if t.equalsIgnoreCase("prometheus") =>
460             // note: 'port' is the same config key used by the old geomesa-metrics
461             Some(MetricsConfig(PrometheusFactory, Some(reporter.withOnlyPath("port"))))
462 
463           case t if t.equalsIgnoreCase("prometheus-pushgateway") =>
464             val config =
465               ConfigFactory.empty()
466                 .withValue("push-gateway.host", reporter.getValue("gateway"))
467                 .withValue("push-gateway.job", reporter.getValue("job-name"))
468             Some(MetricsConfig(PrometheusFactory, Some(config)))
469 
470           case t if t.equalsIgnoreCase("cloudwatch") =>
471             // note: 'namespace' is the same config key used by the old geomesa-metrics
472             Some(MetricsConfig(CloudwatchFactory, Some(reporter.withOnlyPath("namespace"))))
473 
474           case t =>
475             logger.warn(s"Ignoring unsupported metrics reporter of type $t")
476             None
477         }
478       } else {
479         logger.warn("Ignoring metrics reporter without 'type' field")
480         None
481       }
482     }
483     if (supported.lengthCompare(1) > 0) {
484       logger.warn(
485         s"Only one metric registry is supported at once, using ${supported.head.registry.name} " +
486           s"and ignoring ${supported.tail.map(_.registry.name).mkString(", ")}")
487     }
488     supported.headOption
489   }
