296   /**
297    * Parse the dynamic expiry param value into a seq of pairs
298    *
299    * @param params data store params
300    * @return
301    */
302   private[data] def parseDynamicExpiry(params: java.util.Map[String, _]): Seq[(String, Duration)] = {
303     lazy val key = s"Invalid property for parameter '${KafkaDataStoreParams.DynamicCacheExpiry.key}'"
304     val expiry = KafkaDataStoreParams.DynamicCacheExpiry.lookupOpt(params).toSeq.flatMap { value =>
305       ConfigFactory.parseString(value).resolve().root().unwrapped().asScala.toSeq.map {
306         case (filter, exp: String) =>
307           // validate the filter, but leave it as a string so we can optimize it based on the sft later
308           try { ECQL.toFilter(filter) } catch {
309             case NonFatal(e) => throw new IOException(s"$key, expected a CQL filter but got: $filter", e)
310           }
311           val duration = try { Duration(exp) } catch {
312             case NonFatal(e) => throw new IOException(s"$key, expected a duration for key '$filter' but got: $exp", e)
313           }
314           filter -> duration
315 
316         case (filter, exp) =>
317           throw new IOException(s"$key, expected a JSON string for key '$filter' but got: $exp")
318       }
319     }
320     if (expiry.dropRight(1).exists(_._1.equalsIgnoreCase("INCLUDE"))) {
321       throw new IOException(s"$key, defined a filter after Filter.INCLUDE (which would never be invoked)")
322     }
323     expiry
324   }

326   /**
327    * Parse the typesafe config for a layer view. Views take the form:
328    *
329    * {
330    *   foo-sft = [
331    *     {
332    *       type-name = foo-sft-enhanced
333    *       filter = "foo = bar"
334    *       transform = [ "foo", "bar", "baz", "blu" ]
335    *     },
336    *     {
337    *       type-name = foo-sft-reduced
338    *       filter = "foo = baz"
339    *       transform = [ "foo", "bar", "baz" ]
340    *     }
341    *   ]
342    * }
343    *
344    * @param params params
345    * @return
346    */
347   private[kafka] def parseLayerViewConfig(params: java.util.Map[String, _]): Map[String, Seq[LayerViewConfig]] = {
348     def asConfigObject(o: AnyRef) = o match {
349       case c: ConfigObject => c
350       case _ => throw new IllegalArgumentException(s"Invalid layer view, expected a config object but got: $o")
351     }
352 
353     KafkaDataStoreParams.LayerViews.lookupOpt(params) match {
354       case None => Map.empty[String, Seq[LayerViewConfig]]
355       case Some(conf) =>
356         val config = ConfigFactory.parseString(conf).resolve()
357         val entries = config.entrySet().asScala.map { e =>
358           val views = e.getValue match {
359             case c: ConfigList => c.asScala.map(asConfigObject)
360             case c => Seq(asConfigObject(c))
361           }
362           e.getKey -> views.map { c =>
363             ConfigSource.fromConfig(c.toConfig).loadOrThrow[LayerViewConfig](LayerViewClassTag, LayerViewReader)
364           }
365         }
366         val configs = entries.map(f => (f._1, f._2.toSeq))
367         val typeNames = configs.toSeq.flatMap(_._2.map(_.typeName))
368         if (typeNames != typeNames.distinct) {
369           throw new IllegalArgumentException(
370             s"Detected duplicate type name in layer view config: ${config.root().render(ConfigRenderOptions.concise)}")
371         }
372         configs.toMap
373     }
374   }

377    * Parse a single layer view config
378    *
379    * @param cur cursor
380    * @return

382   private def readLayerViewConfig(cur: ConfigCursor): Either[ConfigReaderFailures, LayerViewConfig] = {
383     val config = for {
384       obj       <- cur.asObjectCursor.right
385       typeName  <- obj.atKey("type-name").right.flatMap(_.asString).right
386       filter    <- readFilter(obj.atKeyOrUndefined("filter")).right
387       transform <- readTransform(obj.atKeyOrUndefined("transform")).right
388     } yield {
389       LayerViewConfig(typeName, filter, transform)
390     }
391     config.right.flatMap { c =>
392       if (c.filter.isEmpty && c.transform.isEmpty) {
393         val err = "LayerViews must define at least one of 'filter' or 'transform'"
394         cur.failed(new FailureReason { override def description: String = err })
395       } else {
396         Right(c)
397       }
398     }
399   }

401   private def readFilter(cur: ConfigCursor): Either[ConfigReaderFailures, Option[Filter]] = {
402     if (cur.isUndefined) { Right(None) } else {
403       cur.asString.right.flatMap { ecql =>
404         try { Right(Some(ECQL.toFilter(ecql)).filter(_ != Filter.INCLUDE)) } catch {
405           case NonFatal(e) => cur.failed(CannotConvert(ecql, "Filter", e.toString))
406         }
407       }
408     }
409   }
410 
411   private def readTransform(cur: ConfigCursor): Either[ConfigReaderFailures, Option[Seq[String]]] = {
412     if (cur.isUndefined) { Right(None) } else {
413       val transforms = cur.asList.right.flatMap { list =>
414         list.foldLeft[Either[ConfigReaderFailures, Seq[String]]](Right(Seq.empty)) { case (res, elem) =>
415           res.right.flatMap(r => elem.asString.right.map(r :+ _))
416         }
417       }
418       transforms.right.map(t => if (t.isEmpty) { None } else { Some(t) })
419     }
420   }
