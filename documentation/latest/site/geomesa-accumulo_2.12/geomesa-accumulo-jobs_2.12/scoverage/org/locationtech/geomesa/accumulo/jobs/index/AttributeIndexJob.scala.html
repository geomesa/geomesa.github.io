86   class AttributeMapper extends Mapper[Text, SimpleFeature, Text, Mutation] {
87 
88     type Context = Mapper[Text, SimpleFeature, Text, Mutation]#Context
89 
90     private var counter: Counter = _
91 
92     // TODO GEOMESA-2545 have a standardized writer that returns mutations instead of using a batch writer
93 
94     private var wrapper: FeatureWrapper[WritableFeature] = _
95     private var converters: Seq[(WriteConverter[_], Int)] = _
96     private var colFamilyMappings: IndexedSeq[Array[Byte] => Array[Byte]] = _
97     private var defaultVis: ColumnVisibility = _
98     private var tables: SimpleFeature => IndexedSeq[Text] = _
99 
100     override protected def setup(context: Context): Unit = {
101       counter = context.getCounter("org.locationtech.geomesa", "attributes-written")
102       WithStore[AccumuloDataStore](GeoMesaConfigurator.getDataStoreOutParams(context.getConfiguration)) { ds =>
103         val sft = SimpleFeatureTypes.mutable(ds.getSchema(getTypeName(context.getConfiguration)))
104         val attributes = getAttributes(context.getConfiguration).toSet
105         val indices = ds.manager.indices(sft, IndexMode.Write).filter { i =>
106           (i.name == AttributeIndex.name || i.name == JoinIndex.name) &&
107               i.attributes.headOption.exists(attributes.contains)
108         }
109         wrapper = WritableFeature.wrapper(sft, ds.adapter.groups)
110         converters = indices.map(_.createConverter()).zipWithIndex
111         colFamilyMappings = indices.map(ColumnFamilyMapper.apply).toIndexedSeq
112         defaultVis = new ColumnVisibility()
113         tables = TablePartition(ds, sft) match {
114           case Some(tp) =>
115             val tables = scala.collection.mutable.Map.empty[String, IndexedSeq[Text]]
116             f => {
117               // create the new partition table if needed, which also writes the metadata for it
118               val partition = tp.partition(f)
119               tables.getOrElseUpdate(partition,
120                 indices.map(i => new Text(i.configureTableName(Some(partition)))).toIndexedSeq)
121             }
122 
123           case None =>
124             val names = indices.map(i => new Text(i.getTableName())).toIndexedSeq
125             _ => names
126         }
127       }
128     }
129 
130     override protected def cleanup(context: Context): Unit = {}
131 
132     override def map(key: Text, value: SimpleFeature, context: Context): Unit = {
133       val writable = wrapper.wrap(value)
134       val out = tables(value)
135       converters.foreach { case (converter, i) =>
136         converter.convert(writable) match {
137           case SingleRowKeyValue(row, _, _, _, _, _, vals) =>
138             val mutation = new Mutation(row)
139             vals.foreach { v =>
140               val vis = if (v.vis.isEmpty) { defaultVis } else { new ColumnVisibility(v.vis) }
141               mutation.put(colFamilyMappings(i)(v.cf), v.cq, vis, v.value)
142             }
143             context.write(out(i), mutation)
144 
145           case MultiRowKeyValue(rows, _, _, _, _, _, vals) =>
146             rows.foreach { row =>
147               val mutation = new Mutation(row)
148               vals.foreach { v =>
149                 val vis = if (v.vis.isEmpty) { defaultVis } else { new ColumnVisibility(v.vis) }
150                 mutation.put(colFamilyMappings(i)(v.cf), v.cq, vis, v.value)
151               }
152               context.write(out(i), mutation)
153             }
154         }
155       }
156       counter.increment(converters.length)
157     }
158   }

175     val coverage = Option(parsedArgs.coverage).map { c =>
176       try { IndexCoverage.withName(c) } catch {
177         case NonFatal(_) => throw new IllegalArgumentException(s"Invalid coverage value $c")
178       }
179     }.getOrElse(IndexCoverage.FULL)

181     // validation and initialization - ensure the types exist before launching distributed job
182     WithStore[AccumuloDataStore](dsInParams) { ds =>
183       require(ds != null, "The specified input data store could not be created - check your job parameters")

193 
194       // update the schema - this will create the table if it doesn't exist, and add splits
195       ds.updateSchema(sft.getTypeName, sft)

197       // re-load the sft now that we've updated it
198       sft = ds.getSchema(sft.getTypeName)
199 
200       val job = Job.getInstance(conf,

202 
203       AccumuloJobUtils.setLibJars(job.getConfiguration)

208       job.setOutputFormatClass(classOf[AccumuloOutputFormat])
209       job.setMapOutputKeyClass(classOf[Text])
210       job.setMapOutputValueClass(classOf[Mutation])
211       job.setNumReduceTasks(0)
212

214       GeoMesaAccumuloInputFormat.configure(job.getConfiguration, dsInParams.asJava, plan)
215       GeoMesaConfigurator.setDataStoreOutParams(job.getConfiguration, dsInParams)
216       AttributeIndexJob.setAttributes(job.getConfiguration, attributes.toSeq)
217       AttributeIndexJob.setTypeName(job.getConfiguration, sft.getTypeName)
218 
219       val props = AccumuloDataStoreFactory.buildAccumuloClientConfig(
220         java.util.Map.of(

222           AccumuloDataStoreParams.ZookeepersParam.key, parsedArgs.inZookeepers,
223           AccumuloDataStoreParams.UserParam.key, parsedArgs.inUser,
224           AccumuloDataStoreParams.PasswordParam.key, parsedArgs.inPassword
225         )
226       )

229

231

233     }
