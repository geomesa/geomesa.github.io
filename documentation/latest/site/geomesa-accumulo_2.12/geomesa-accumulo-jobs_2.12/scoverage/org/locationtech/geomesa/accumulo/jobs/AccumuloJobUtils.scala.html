65   /**
66     * Get a sequence of one or more query plans, which is guaranteed not to contain
67     * a JoinPlan (return a fallback in this case). If we get multiple scan plans,
68     * that's groovy.
69     */
70   @deprecated("Deprecated with no replacement")
71   def getMultipleQueryPlan(ds: AccumuloDataStore, query: Query): Seq[AccumuloQueryPlan] = {
72     // disable join plans
73     JoinIndex.AllowJoinPlans.set(false)
74 
75     try {
76       lazy val fallbackIndex = {
77         val schema = ds.getSchema(query.getTypeName)
78         ds.manager.indices(schema, IndexMode.Read).headOption.getOrElse {
79           throw new IllegalStateException(s"Schema '${schema.getTypeName}' does not have any readable indices")
80         }
81       }
82 
83       val queryPlans = ds.getQueryPlan(query)
84       if (queryPlans.isEmpty) {
85         Seq(EmptyPlan(FilterStrategy(fallbackIndex, None, Some(Filter.EXCLUDE), temporal = false, Float.PositiveInfinity)))
86       } else {
87         queryPlans
88       }
89     } finally {
90       // make sure we reset the thread locals
91       JoinIndex.AllowJoinPlans.remove()
92     }
93   }
