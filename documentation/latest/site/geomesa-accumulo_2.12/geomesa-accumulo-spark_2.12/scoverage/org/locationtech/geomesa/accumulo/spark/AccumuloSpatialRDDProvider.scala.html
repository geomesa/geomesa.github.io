36   override def rdd(conf: Configuration,
37                    sc: SparkContext,
38                    params: Map[String, String],
39                    query: Query): SpatialRDD = {
40     val paramsAsJava = params.asJava
41     val ds = DataStoreFinder.getDataStore(paramsAsJava).asInstanceOf[AccumuloDataStore]
42 
43     lazy val transform = query.getHints.getTransformSchema
44 
45     def queryPlanToRDD(sft: SimpleFeatureType, qp: AccumuloQueryPlan): RDD[SimpleFeature] = {
46       if (ds == null || sft == null || qp.isInstanceOf[EmptyPlan]) {
47         sc.emptyRDD[SimpleFeature]
48       } else {
49         val config = new JobConf(conf)
50         GeoMesaAccumuloInputFormat.configure(config, paramsAsJava, qp, Some(ds.auths))
51         sc.newAPIHadoopRDD(config, classOf[GeoMesaAccumuloInputFormat], classOf[Text], classOf[SimpleFeature]).map(_._2)
52       }
53     }
54 
55     try {
56       // get the query plan to set up the iterators, ranges, etc
57       // getMultipleQueryPlan will return the fallback if any
58       // element of the plan is a JoinPlan
59       val sft = ds.getSchema(query.getTypeName)
60       // disable join plans
61       JoinIndex.AllowJoinPlans.set(false)
62       val qps = try { ds.getQueryPlan(query) } finally {
63         // make sure we reset the thread locals
64         JoinIndex.AllowJoinPlans.remove()
65       }
66 
67       // can return a union of the RDDs because the query planner *should*
68       // be rewriting ORs to make them logically disjoint
69       // e.g. "A OR B OR C" -> "A OR (B NOT A) OR ((C NOT A) NOT B)"
70       val sfrdd = if (qps.lengthCompare(1) == 0 && qps.head.tables.lengthCompare(1) == 0) {
71         queryPlanToRDD(sft, qps.head) // no union needed for single query plan
72       } else {
73         // flatten and duplicate the query plans so each one only has a single table
74         val expanded = qps.flatMap {
75           case qp: BatchScanPlan => qp.tables.map(t => qp.copy(tables = Seq(t)))
76           case _: EmptyPlan => Seq.empty
77           case qp => throw new UnsupportedOperationException(s"Unexpected query plan type: $qp")
78         }
79         if (expanded.isEmpty) {
80           sc.emptyRDD[SimpleFeature]
81         } else {
82           sc.union(expanded.map(queryPlanToRDD(sft, _)))
83         }
84       }
85       SpatialRDD(sfrdd, transform.getOrElse(sft))
86     } finally {
87       if (ds != null) {
88         ds.dispose()
89       }
90     }
91   }
92 
93   /**
94     * Writes this RDD to a GeoMesa table.
95     * The type must exist in the data store, and all of the features in the RDD must be of this type.
96     *
97     * @param rdd rdd
98     * @param params params
99     * @param typeName type name
100     */
101   def save(rdd: RDD[SimpleFeature], params: Map[String, String], typeName: String): Unit = {
102     WithStore[AccumuloDataStore](params) { ds =>
103       require(ds.getSchema(typeName) != null,
104         "Feature type must exist before calling save. Call createSchema on the DataStore first.")
105     }
106 
107     rdd.foreachPartition { iter =>
108       WithStore[AccumuloDataStore](params) { ds =>
109         WithClose(ds.getFeatureWriterAppend(typeName, Transaction.AUTO_COMMIT)) { writer =>
110           val helper = FeatureWriterHelper(writer, useProvidedFids = true)
111           iter.foreach(helper.write)
112         }
113       }
114     }
115   }
