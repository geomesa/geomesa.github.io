48   /**
49     * Create a query plan against a join index - if possible, will use the reduced index-values to scan
50     * the single table, otherwise will require a join against the id index
51     *
52     * @param filter filter strategy
53     * @param tables tables to scan
54     * @param ranges ranges to scan
55     * @param colFamily column family to scan, optional
56     * @param schema simple feature schema being scanned
57     * @param ecql secondary push-down predicates
58     * @param hints query hints
59     * @param numThreads query threads
60     * @return
61     */
62   def createQueryPlan(
63       ds: AccumuloDataStore,
64       index: AttributeJoinIndex,
65       strategy: QueryStrategy,
66       tables: Seq[String],
67       ranges: Seq[org.apache.accumulo.core.data.Range],
68       colFamily: Option[Text],
69       schema: SimpleFeatureType,
70       ecql: Option[Filter],
71       hints: Hints,
72       numThreads: Int): AccumuloQueryPlan = {
73 
74     // TODO seems like this should be using 'schema' here, which may be a reduced version of the indexSft due to col groups
75     val indexSft = index.indexSft
76     val transform = hints.getTransformSchema
77 
78     // for queries that don't require a join, creates a regular batch scan plan
79     def plan(
80         iters: Seq[IteratorSetting],
81         kvsToFeatures: ResultsToFeatures[Entry[Key, Value]],
82         reduce: Option[FeatureReducer]): BatchScanPlan = {
83       // add the attribute-level vis iterator if necessary
84       val iterators = visibilityIter(index) ++ iters
85       val sort = hints.getSortFields
86       val max = hints.getMaxFeatures
87       val project = hints.getProjection
88       BatchScanPlan(strategy, tables, ranges, iterators, colFamily, kvsToFeatures, reduce, sort, max, project, numThreads)
89     }
90 
91     // used when remote processing is disabled
92     def localPlan(overrides: Option[Seq[IteratorSetting]] = None): BatchScanPlan = {
93       val returnSchema = transform.getOrElse(indexSft)
94       if (hints.isSkipReduce) {
95         // override the return sft to reflect what we're actually returning,
96         // since the bin sft is only created in the local reduce step
97         hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
98       }
99       val iters = overrides.getOrElse(FilterTransformIterator.configure(indexSft, index, ecql, hints).toSeq)
100       val localReducer = Some(new LocalTransformReducer(returnSchema, None, None, None, hints))
101       plan(iters, AccumuloResultsToFeatures(index, returnSchema), localReducer)
102     }
103 
104     val qp = if (hints.isBinQuery) {
105       // check to see if we can execute against the index values
106       if (indexSft.indexOf(hints.getBinTrackIdField) != -1 &&
107           hints.getBinGeomField.forall(indexSft.indexOf(_) != -1) &&
108           hints.getBinLabelField.forall(indexSft.indexOf(_) != -1) &&
109           index.supportsFilter(ecql)) {
110         if (ds.config.remote.bin) {
111           val binIter = BinAggregatingIterator.configure(indexSft, index, ecql, hints)
112           plan(Seq(binIter), new AccumuloBinResultsToFeatures(), None)
113         } else {
114           localPlan()
115         }
116       } else {
117         // have to do a join against the record table
118         createJoinPlan(ds, index, strategy, tables, ranges, colFamily, ecql, hints)
119       }
120     } else if (hints.isArrowQuery) {
121       // check to see if we can execute against the index values
122       if (index.canUseIndexSchema(ecql, transform)) {
123         if (ds.config.remote.arrow) {
124           val (arrowIter, reduce) = ArrowIterator.configure(indexSft, index, ds.stats, strategy.filter.filter, ecql, hints)
125           plan(Seq(arrowIter), new AccumuloArrowResultsToFeatures(), Some(reduce))
126         } else {
127           localPlan()
128         }
129       } else if (index.canUseIndexSchemaPlusKey(ecql, transform)) {
130         val transformSft = transform.getOrElse {
131           throw new IllegalStateException("Must have a transform for attribute key plus value scan")
132         }
133         // first filter and apply the transform
134         val filterTransformIter = FilterTransformIterator.configure(indexSft, index, ecql, hints, 23).get
135         // next add the attribute value from the row key
136         val rowValueIter = AttributeKeyValueIterator.configure(index.asInstanceOf[AttributeIndex], transformSft, 24)
137         if (ds.config.remote.arrow) {
138           // clear the transforms as we've already accounted for them
139           val newHints = new Hints(hints)
140           newHints.clearTransforms()
141           // finally apply the arrow iterator on the resulting features
142           val (arrowIter, reduce) = ArrowIterator.configure(transformSft, index, ds.stats, None, None, newHints)
143           val iters = Seq(filterTransformIter, rowValueIter, arrowIter)
144           plan(iters, new AccumuloArrowResultsToFeatures(), Some(reduce))
145         } else {
146           localPlan(Some(Seq(filterTransformIter, rowValueIter)))
147         }
148       } else {
149         // have to do a join against the record table
150         createJoinPlan(ds, index, strategy, tables, ranges, colFamily, ecql, hints)
151       }
152     } else if (hints.isDensityQuery) {
153       // check to see if we can execute against the index values
154       val weightIsAttribute = hints.getDensityWeight.contains(index.attributes.head)
155       if (index.supportsFilter(ecql) && (weightIsAttribute || hints.getDensityWeight.forall(indexSft.indexOf(_) != -1))) {
156         if (ds.config.remote.density) {
157           val iters = if (weightIsAttribute) {
158             // create a transform sft with the attribute added
159             val transform = {
160               val builder = new SimpleFeatureTypeBuilder()
161               builder.setNamespaceURI(null: String)
162               builder.setName(indexSft.getTypeName + "--attr")
163               builder.setAttributes(indexSft.getAttributeDescriptors)
164               builder.add(index.sft.getDescriptor(index.attributes.head))
165               if (indexSft.getGeometryDescriptor != null) {
166                 builder.setDefaultGeometry(indexSft.getGeometryDescriptor.getLocalName)
167               }
168               builder.setCRS(indexSft.getCoordinateReferenceSystem)
169               val tmp = builder.buildFeatureType()
170               tmp.getUserData.putAll(indexSft.getUserData)
171               tmp
172             }
173             // priority needs to be between vis iter (21) and density iter (25)
174             val keyValueIter = AttributeKeyValueIterator.configure(index.asInstanceOf[AttributeIndex], transform, 23)
175             val densityIter = DensityIterator.configure(transform, index, ecql, hints)
176             Seq(keyValueIter, densityIter)
177           } else {
178             Seq(DensityIterator.configure(indexSft, index, ecql, hints))
179           }
180           plan(iters, new AccumuloDensityResultsToFeatures(), None)
181         } else {
182           localPlan()
183         }
184       } else {
185         // have to do a join against the record table
186         createJoinPlan(ds, index, strategy, tables, ranges, colFamily, ecql, hints)
187       }
188     } else if (hints.isStatsQuery) {
189       // check to see if we can execute against the index values
190       if (Try(Stat(indexSft, hints.getStatsQuery)).isSuccess && index.supportsFilter(ecql)) {
191         if (ds.config.remote.stats) {
192           val statsIter = StatsIterator.configure(indexSft, index, ecql, hints)
193           val reduce = Some(StatsScan.StatsReducer(indexSft, hints))
194           plan(Seq(statsIter), new AccumuloStatsResultsToFeatures(), reduce)
195         } else {
196           localPlan()
197         }
198       } else {
199         // have to do a join against the record table
200         createJoinPlan(ds, index, strategy, tables, ranges, colFamily, ecql, hints)
201       }
202     } else if (index.canUseIndexSchema(ecql, transform)) {
203       // we can use the index value
204       // transform has to be non-empty to get here and can only include items
205       // in the index value (not the index keys aka the attribute indexed)
206       val transformSft = transform.getOrElse {
207         throw new IllegalStateException("Must have a transform for attribute value scan")
208       }
209       val fti = FilterTransformIterator.configure(indexSft, index, ecql, hints.getTransform, hints.getSampling)
210       // need to use transform to convert key/values
211       val toFeatures = AccumuloResultsToFeatures(index, transformSft)
212       plan(fti.toSeq, toFeatures, None)
213     } else if (index.canUseIndexSchemaPlusKey(ecql, transform)) {
214       // we can use the index PLUS the value
215       val transformSft = transform.getOrElse {
216         throw new IllegalStateException("Must have a transform for attribute key plus value scan")
217       }
218       val fti = FilterTransformIterator.configure(indexSft, index, ecql, hints.getTransform, hints.getSampling)
219       val iters = fti.toSeq :+ AttributeKeyValueIterator.configure(index.asInstanceOf[AttributeIndex], transformSft)
220       // need to use transform to convert key/values
221       val toFeatures = AccumuloResultsToFeatures(index, transformSft)
222       plan(iters, toFeatures, None)
223     } else {
224       // have to do a join against the record table
225       createJoinPlan(ds, index, strategy, tables, ranges, colFamily, ecql, hints)
226     }
227 
228     if (ranges.nonEmpty) { qp } else { EmptyPlan(strategy, qp.reducer) }
229   }

231   /**
232     * Gets a query plan comprised of a join against the record table. This is the slowest way to
233     * execute a query, so we avoid it if possible.
234     */
235   private def createJoinPlan(
236       ds: AccumuloDataStore,
237       index: AttributeJoinIndex,
238       strategy: QueryStrategy,
239       tables: Seq[String],
240       ranges: Seq[org.apache.accumulo.core.data.Range],
241       colFamily: Option[Text],
242       ecql: Option[Filter],
243       hints: Hints): AccumuloQueryPlan = {
244     import org.locationtech.geomesa.filter.ff
245     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
246     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
247 
248     // apply any secondary filters or transforms against the record table
249     val recordIndex = ds.manager.indices(index.sft, IndexMode.Read).find(_.name == IdIndex.name).getOrElse {
250       throw new RuntimeException("Id index does not exist for join query: " +
251           ds.manager.indices(index.sft, IndexMode.Read).map(_.identifier).mkString(", "))
252     }
253 
254     // break out the st filter to evaluate against the attribute table
255     val (stFilter, ecqlFilter) = ecql.map { f =>
256       val (geomFilters, otherFilters) = partitionPrimarySpatials(f, index.sft)
257       val (temporalFilters, nonSTFilters) = partitionPrimaryTemporals(otherFilters, index.sft)
258       (andOption(geomFilters ++ temporalFilters), andOption(nonSTFilters))
259     }.getOrElse((None, None))
260 
261     val (recordColFamily, recordSchema) = {
262       val (cf, s) = ds.adapter.groups.group(index.sft, hints.getTransformDefinition, ecqlFilter)
263       (Some(new Text(ColumnFamilyMapper(recordIndex)(cf))), s)
264     }
265 
266     // since each range is a single row, it wouldn't be very efficient to do any aggregating scans
267     // instead, handle them with the local query runner
268     val resultSft = hints.getTransformSchema.getOrElse(index.sft)
269     val recordIterators = {
270       val recordIter = FilterTransformIterator.configure(recordSchema, recordIndex, ecqlFilter, hints).toSeq
271       if (index.sft.getVisibilityLevel != VisibilityLevel.Attribute) { recordIter } else {
272         Seq(KryoVisibilityRowEncoder.configure(recordSchema)) ++ recordIter
273       }
274     }
275     val toFeatures = AccumuloResultsToFeatures(recordIndex, resultSft)
276     val reducer = new LocalTransformReducer(resultSft, None, None, None, hints)
277     if (hints.isSkipReduce) {
278       // override the return sft to reflect what we're actually returning,
279       // since the arrow sft is only created in the local reduce step
280       hints.put(QueryHints.Internal.RETURN_SFT, resultSft)
281     }
282 
283     val recordTables = recordIndex.getTablesForQuery(strategy.filter.filter)
284     val recordThreads = ds.config.queries.recordThreads
285 
286     // function to join the attribute index scan results to the record table
287     // have to pull the feature id from the row
288     val joinFunction: JoinFunction = {
289       val prefix = index.sft.getTableSharingBytes
290       val idToBytes = GeoMesaFeatureIndex.idToBytes(index.sft)
291       kv => {
292         val row = kv.getKey.getRow
293         new Range(new Text(ByteArrays.concat(prefix, idToBytes(index.getIdFromRow(row.getBytes, 0, row.getLength, null)))))
294       }
295     }
296 
297     val joinQuery = BatchScanPlan(strategy, recordTables, Seq.empty, recordIterators, recordColFamily, toFeatures,
298       Some(reducer), hints.getSortFields, hints.getMaxFeatures, hints.getProjection, recordThreads)
299 
300     val attributeIters = visibilityIter(index) ++
301         FilterTransformIterator.configure(index.indexSft, index, stFilter, None, hints.getSampling).toSeq
302 
303     JoinPlan(strategy, tables, ranges, attributeIters, colFamily, recordThreads, joinFunction, joinQuery)
304   }
