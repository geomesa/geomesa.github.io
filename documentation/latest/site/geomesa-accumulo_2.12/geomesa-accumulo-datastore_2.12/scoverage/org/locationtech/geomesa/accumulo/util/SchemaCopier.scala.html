37 /**
38  * Copies a schema from one cluster (or catalog) to another, using bulk file operations
39  *

85   // sft should be shareable/the same from both datastores
86   lazy private val sft: SimpleFeatureType = {
87     val sft = from.ds.getSchema(typeName)
88     if (sft == null) {
89       throw new IllegalArgumentException(s"Schema '$typeName' does not exist in the source store")
90     } else {
91       val toSft = to.ds.getSchema(typeName)
92       if (toSft == null) {
93         throw new IllegalArgumentException(s"Schema '$typeName' does not exist in the destination store")
94       } else if (SimpleFeatureTypes.compare(sft, toSft) != 0) {
95         throw new IllegalArgumentException(s"Schema '$typeName' is not the same in the source and destination store")
96       } else if (SimpleFeatureTypes.compareIndexConfigs(sft, toSft) != 0) {
97         throw new IllegalArgumentException(s"Schema '$typeName' does not have compatible indices in the source and destination store")
98       }
99     }
100     sft
101   }

103   lazy private val fromIndices = {
104     val all = from.ds.manager.indices(sft)
105     if (indices.isEmpty) { all } else {
106       val builder = Seq.newBuilder[GeoMesaFeatureIndex[_, _]]
107       indices.foreach { ident =>
108         val filtered = all.filter(_.identifier.contains(ident))
109         if (filtered.isEmpty) {
110           throw new IllegalArgumentException(
111             s"Index '$ident' does not exist in the schema. Available indices: ${all.map(_.identifier).mkString(", ")}")
112         }
113         logger.debug(s"Mapped identifier $ident to ${filtered.map(_.identifier).mkString(", ")}")
114         builder ++= filtered
115       }
116       builder.result.distinct
117     }
118   }

121   // while for non-partitioned schemas it must always be Seq(None)
122   lazy private val fromPartitions: Seq[Option[String]] = {
123     if (sft.isPartitioned) {
124       val builder = ListBuffer.empty[String]
125       lazy val partitioning = TablePartition(from.ds, sft).get
126       lazy val sf = new ScalaSimpleFeature(sft, "")
127       builder ++= partitions.map {
128         case PartitionName(name) => name
129         case PartitionValue(value) =>
130           sf.setAttribute(sft.getDtgIndex.get, value)
131           val partition = partitioning.partition(sf)
132           logger.debug(s"Generated partition $partition from value $value")
133           partition
134       }
135       if (builder.isEmpty) {
136         logger.debug("No partitions specified - loading all partitions from store")
137         builder ++= fromIndices.flatMap(_.getPartitions)
138       }
139       builder.result.distinct.sorted.map(Option.apply)
140     } else {
141       if (partitions.nonEmpty) {
142         throw new IllegalArgumentException("partitions are not applicable for a non-partitioned schema")
143       }
144       Seq(None)
145     }

158   def run(resume: Boolean): Unit = {
159     CachedThreadPool.executor(options.tableConcurrency) { executor =>
160       fromPartitions.foreach { partition =>
161         fromIndices.foreach { fromIndex =>
162           val toIndex = to.ds.manager.index(sft, fromIndex.identifier)
163           val partitionLogId = s"${partition.fold(s"index")(p => s"partition $p")} ${fromIndex.identifier}"
164           val runnable: Runnable = () => {
165             try {
166               logger.info(s"Copying $partitionLogId")
167               copy(fromIndex, toIndex, partition, resume, partitionLogId)
168               logger.info(s"Bulk copy complete for $partitionLogId")
169             } catch {
170               // catch Throwable so NoClassDefFound still gets logged
171               case e: Throwable =>
172                 logger.error(s"Error copying $partitionLogId: ${e.getMessage}")
173                 logger.debug(s"Error copying $partitionLogId", e)
174             }
175           }
176           executor.submit(runnable)
177         }
178       }
179     }

182   /**
183    * Copy a single index + partition
184    *
185    * @param fromIndex from index
186    * @param toIndex to index
187    * @param partition partition name - must be Some if schema is partitioned
188    * @param resume use any partial results from a previous run, if present
189    * @param partitionLogId identifier for log messages
190    */
191   private def copy(
192       fromIndex: GeoMesaFeatureIndex[_, _],
193       toIndex: GeoMesaFeatureIndex[_, _],
194       partition: Option[String],
195       resume: Boolean,
196       partitionLogId: String): Unit = {
197 
198     require(sft.isPartitioned == partition.isDefined) // sanity check - this should always be true due to our setup
199 
200     val fromTable = try { fromIndex.getTableName(partition) } catch {
201       case NonFatal(e) => throw new RuntimeException("Could not get source table", e)
202     }
203 
204     val completeMarker = new Path(exportPath, s"$fromTable.complete")
205     if (exportFs.exists(completeMarker)) {
206       if (resume) {
207         logger.debug("Skipping already completed copy")
208         return
209       } else {
210         exportFs.delete(completeMarker, false)
211       }
212     }
213 
214     val tableExportPath = new Path(exportPath, fromTable)
215     val distcpPath = new Path(tableExportPath, "distcp.txt")
216     val copyToDir = new Path(tableExportPath, "files")
217     val cloneTable = s"${fromTable}_bc_tmp"
218 
219     logger.debug(s"Source table $fromTable (${from.tableOps.tableIdMap().get(fromTable)})")
220     logger.debug(s"Export path $tableExportPath")
221 
222     if (resume && from.tableOps.exists(cloneTable)) {
223       logger.debug(s"Using existing cloned table $cloneTable - ensuring table is offline")
224       from.tableOps.offline(cloneTable, true)
225     } else {
226       // clone the table as we have to take it offline in order to export it
227       // note that cloning is just a metadata op as it shares the underlying data files (until they change)
228       logger.debug(s"Checking for existence and deleting any existing cloned table $cloneTable")
229       from.ds.adapter.deleteTable(cloneTable) // no-op if table doesn't exist
230       logger.debug(s"Cloning $fromTable to $cloneTable")
231       from.tableOps.clone(fromTable, cloneTable, false, Collections.emptyMap(), Collections.emptySet()) // use 2.0 method for compatibility
232       logger.debug(s"Taking $cloneTable offline")
233       from.tableOps.offline(cloneTable, true)
234     }
235 
236     if (resume && exportFs.exists(distcpPath) && exportFs.getFileStatus(distcpPath).getLen > 0) {
237       logger.debug(s"Using existing export results $distcpPath")
238     } else {
239       if (exportFs.exists(tableExportPath)) {
240         logger.debug(s"Deleting existing export directory $tableExportPath")
241         exportFs.delete(tableExportPath, true)
242       }
243       logger.debug(s"Exporting table to $tableExportPath")
244       from.tableOps.exportTable(cloneTable, tableExportPath.toString)
245 
246       if (!exportFs.exists(distcpPath) || exportFs.getFileStatus(distcpPath).getLen == 0) {
247         throw new RuntimeException(s"Could not read table export results at $distcpPath")
248       }
249     }
250 
251     // ensures the destination table exists
252     logger.debug(s"Checking destination for table $fromTable")
253     to.ds.adapter.createTable(toIndex, partition, Seq.empty)
254     val toTable = try { toIndex.getTableName(partition) } catch {
255       case NonFatal(e) => throw new RuntimeException("Could not get destination table", e)
256     }
257     logger.debug(s"Destination table $toTable (${to.tableOps.tableIdMap().get(toTable)})")
258 
259     // create splits, do this separately in case the table already exists
260     val splits = new java.util.TreeSet(from.tableOps.listSplits(cloneTable))
261     val existingSplits = to.tableOps.listSplits(toTable)
262     splits.removeAll(existingSplits)
263     if (!splits.isEmpty) {
264       if (!existingSplits.isEmpty) {
265         logger.warn(s"Detected split mismatch between source ($fromTable) and destination ($toTable) for $partitionLogId")
266       }
267       logger.debug(s"Adding splits to destination table $toTable")
268       to.tableOps.addSplits(toTable, splits)
269     }
270 
271     val hadCopyError = new AtomicBoolean(false)
272     // read the distcp.txt file produced by the table export
273     // consumer: (src, dest) => Unit
274     def distCpConsumer(threads: Int)(consumer: (Path, Path) => Unit): Unit = {
275       logger.debug(s"Reading $distcpPath")
276       WithClose(IOUtils.lineIterator(exportFs.open(distcpPath), StandardCharsets.UTF_8)) { files =>
277         CachedThreadPool.executor(threads) { executor =>
278           files.asScala.foreach { file =>
279             val runnable: Runnable = () => {
280               val path = new Path(file)
281               val copy = new Path(copyToDir, path.getName)
282               try {
283                 if (resume && exportFs.exists(copy) &&
284                   path.getFileSystem(from.conf).getFileStatus(path).getLen == exportFs.getFileStatus(copy).getLen) {
285                   logger.debug(s"Using existing copy of $path at $copy")
286                 } else {
287                   consumer(path, copy)
288                 }
289               } catch {
290                 // catch Throwable so NoClassDefFound still gets logged
291                 case e: Throwable =>
292                   hadCopyError.set(true)
293                   logger.error(s"Failed to copy $path to $copy: ${e.getMessage}")
294                   logger.debug(s"Failed to copy $path to $copy", e)
295               }
296             }
297             executor.submit(runnable)
298           }
299         }
300       }
301     }
302 
303     if (options.distCp) {
304       var inputPath = distcpPath
305       if (resume) {
306         logger.debug(s"Checking copy status of files in $distcpPath")
307         inputPath = new Path(tableExportPath, "distcp-remaining.txt")
308         WithClose(exportFs.create(inputPath, true)) { out =>
309           distCpConsumer(1) { (path, _) =>
310             logger.debug(s"Adding $path to distcp")
311             out.writeUTF(s"$path\n")
312           }
313         }
314       }
315       val job = new DistCp(from.conf, DistributedCopyOptions(inputPath, copyToDir)).execute()
316       logger.info(s"Tracking available at ${job.getStatus.getTrackingUrl}")
317       while (!job.isComplete) {
318         Thread.sleep(500)
319       }
320       if (job.isSuccessful) {
321         logger.info(s"Successfully copied data to $copyToDir")
322       } else {
323         hadCopyError.set(true)
324         logger.error(s"Job failed with state ${job.getStatus.getState} due to: ${job.getStatus.getFailureInfo}")
325       }
326     } else {
327       distCpConsumer(options.fileConcurrency) { (path, copy) =>
328         logger.debug(s"Copying $path to $copy")
329         val fs = path.getFileSystem(from.conf)
330         if (!FileUtil.copy(fs, path, exportFs, copy, false, true, to.conf)) {
331           // consolidate error handling in the catch block
332           throw new IOException(s"Failed to copy $path to $copy, copy returned false")
333         }
334       }
335     }
336     if (hadCopyError.get) {
337       throw new RuntimeException("Error copying data files")
338     }
339 
340     logger.debug(s"Loading rfiles from $copyToDir to $toTable")
341     val importDir = to.tableOps.importDirectory(copyToDir.toString).to(toTable)
342     try { importDir.ignoreEmptyDir(true) } catch {
343       case _: NoSuchMethodError => // accumulo 2.0, ignore
344     }
345     try { importDir.load() } catch {
346       case e: IllegalArgumentException => logger.trace("Error importing directory:", e) // should mean empty dir
347     }
348 
349     // create marker indicating this copy was successful
350     logger.debug(s"Creating completion marker $completeMarker")
351     exportFs.create(completeMarker).close()
352 
353     // cleanup
354     logger.debug(s"Deleting export path $tableExportPath")
355     exportFs.delete(tableExportPath, true)
356     logger.debug(s"Deleting clone table $cloneTable")
357     from.tableOps.delete(cloneTable)
358   }

362     CloseWithLogging(tryTo.toOption)
363   }
