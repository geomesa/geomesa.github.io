100   override protected def transitionIndices(sft: SimpleFeatureType): Unit = {
101     // note: versions already correspond to accumulo index versions
102     val dtg = sft.getDtgField.toSeq
103     val geom = Option(sft.getGeomField).toSeq
104     val indices = sft.getIndices.flatMap {
105       case id if id.name == IdIndex.name  => Seq(id) // no update needed
106       case id if id.name == "records"     => Seq(id.copy(name = IdIndex.name))
107       case id if id.name == Z3Index.name  => Seq(id.copy(attributes = geom ++ dtg))
108       case id if id.name == XZ3Index.name => Seq(id.copy(attributes = geom ++ dtg))
109       case id if id.name == Z2Index.name  => Seq(id.copy(attributes = geom))
110       case id if id.name == XZ2Index.name => Seq(id.copy(attributes = geom))
111       case id if id.name == AttributeIndex.name =>
112         lazy val fields = if (id.version < 4) { dtg } else { geom ++ dtg }
113         sft.getAttributeDescriptors.asScala.flatMap { d =>
114           val index = d.getUserData.remove(AttributeOptions.OptIndex).asInstanceOf[String]
115           if (index == null || index.equalsIgnoreCase(IndexCoverage.NONE.toString) || index.equalsIgnoreCase("false")) {
116             Seq.empty
117           } else if (index.equalsIgnoreCase(IndexCoverage.FULL.toString)) {
118             Seq(id.copy(name = AttributeIndex.name, attributes = Seq(d.getLocalName) ++ fields))
119           } else if (index.equalsIgnoreCase(IndexCoverage.JOIN.toString) || java.lang.Boolean.valueOf(index)) {
120             Seq(id.copy(name = JoinIndex.name, attributes = Seq(d.getLocalName) ++ fields))
121           } else {
122             throw new IllegalStateException(s"Expected an index coverage or boolean but got: $index")
123           }
124         }
125     }
126     sft.setIndices(indices)
127   }

150   override protected def preSchemaCreate(sft: SimpleFeatureType): Unit = {
151     import org.locationtech.geomesa.index.conf.SchemaProperties.ValidateDistributedClasspath
152 
153     // call super first so that user data keys are updated
154     super.preSchemaCreate(sft)
155 
156     def getNamespace(prefix: String): String = prefix.indexOf('.') match {
157       case -1 => ""
158       case i  => prefix.substring(0, i)
159     }
160 
161     val prefixes = Seq(config.catalog) ++ sft.getIndices.flatMap(i => sft.getTablePrefix(i.name))
162     prefixes.map(getNamespace).distinct.foreach { namespace =>
163       if (namespace.nonEmpty) {
164         adapter.ensureNamespaceExists(namespace)
165       }
166       // validate that the accumulo runtime is available
167       val canLoad = client.namespaceOperations().testClassLoad(namespace,
168         classOf[ProjectVersionIterator].getName, classOf[SortedKeyValueIterator[_, _]].getName)
169 
170       if (!canLoad) {
171         val msg = s"Could not load GeoMesa distributed code from the Accumulo classpath"
172         logger.error(s"$msg for catalog ${config.catalog}")
173         if (ValidateDistributedClasspath.toBoolean.contains(true)) {
174           val nsMsg = if (namespace.isEmpty) { "" } else { s" for the namespace '$namespace'" }
175           throw new RuntimeException(s"$msg. You may override this check by setting the system property " +
176             s"'${ValidateDistributedClasspath.property}=false'. Otherwise, please verify that the appropriate " +
177             s"JARs are installed$nsMsg - see https://www.geomesa.org/documentation/stable/user/accumulo/install.html" +
178             "#installing-the-accumulo-distributed-runtime-library")
179         }
180       }
181     }
182 
183     if (sft.getVisibilityLevel == VisibilityLevel.Attribute && sft.getAttributeCount > 255) {
184       throw new IllegalArgumentException("Attribute level visibility only supports up to 255 attributes")
185     }
186 
187     sft.getDtgField.foreach { dtg =>
188       if (sft.getIndices.exists(i => i.name == JoinIndex.name && i.attributes.headOption.contains(dtg))) {
189         if (!GeoMesaSchemaValidator.declared(sft, OverrideDtgJoin)) {
190           throw new IllegalArgumentException("Trying to create a schema with a partial (join) attribute index " +
191               s"on the default date field '$dtg'. This may cause whole-world queries with time bounds to be much " +
192               "slower. If this is intentional, you may override this check by putting Boolean.TRUE into the " +
193               s"SimpleFeatureType user data under the key '$OverrideDtgJoin' before calling createSchema, or by " +
194               s"setting the system property '$OverrideDtgJoin' to 'true'. Otherwise, please either specify a " +
195               "full attribute index or remove it entirely.")
196         }
197       }
198     }
199   }

201   override protected def onSchemaCreated(sft: SimpleFeatureType): Unit = {
202     super.onSchemaCreated(sft)
203     if (sft.statsEnabled) {
204       // configure the stats combining iterator on the table for this sft
205       adapter.ensureTableExists(stats.metadata.table)
206       stats.configureStatCombiner(client, sft)
207     }
208     sft.getFeatureExpiration.foreach {
209       case IngestTimeExpiration(ttl) =>
210         val tableOps = client.tableOperations()
211         getAllIndexTableNames(sft.getTypeName).filter(tableOps.exists).foreach { table =>
212           AgeOffIterator.set(tableOps, table, sft, ttl)
213         }
214 
215       case FeatureTimeExpiration(dtg, _, ttl) =>
216         val tableOps = client.tableOperations()
217         manager.indices(sft).foreach { index =>
218           val indexSft = index match {
219             case joinIndex: AttributeJoinIndex => joinIndex.indexSft
220             case _ => sft
221           }
222           DtgAgeOffIterator.set(tableOps, indexSft, index, ttl, dtg)
223         }
224 
225       case e =>
226         throw new IllegalArgumentException(s"Unexpected feature expiration: $e")
227     }
228   }

231   override protected def preSchemaUpdate(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
232     if (sft.getVisibilityLevel == VisibilityLevel.Attribute && sft.getAttributeCount > 255) {
233       throw new IllegalArgumentException("Attribute level visibility only supports up to 255 attributes")
234     }
235 
236     // check for attributes flagged 'index=join' and convert them to sft-level user data
237     sft.getAttributeDescriptors.asScala.foreach { d =>
238       val index = d.getUserData.get(AttributeOptions.OptIndex).asInstanceOf[String]
239       if (index != null && index.equalsIgnoreCase(IndexCoverage.JOIN.toString)) {
240         d.getUserData.remove(AttributeOptions.OptIndex) // remove it so it's not processed again
241         val fields = Seq(d.getLocalName) ++ Option(sft.getGeomField) ++ sft.getDtgField.filter(_ != d.getLocalName)
242         val attribute = IndexId(JoinIndex.name, JoinIndex.version, fields, IndexMode.ReadWrite)
243         val existing = sft.getIndices.map(GeoMesaFeatureIndex.identifier)
244         if (!existing.contains(GeoMesaFeatureIndex.identifier(attribute))) {
245           sft.setIndices(sft.getIndices :+ attribute)
246         }
247       }
248     }
249 
250     // check any previous age-off - previously age-off wasn't tied to the sft metadata
251     if (!sft.isFeatureExpirationEnabled && !previous.isFeatureExpirationEnabled) {
252       // explicitly set age-off in the feature type if found
253       val tableOps = client.tableOperations()
254       val tables = getAllIndexTableNames(previous.getTypeName).filter(tableOps.exists)
255       val ageOff = tables.foldLeft[Option[FeatureExpiration]](None) { (res, table) =>
256         res.orElse(AgeOffIterator.expiry(tableOps, table))
257       }
258       val configured = ageOff.orElse {
259         tables.foldLeft[Option[FeatureExpiration]](None) { (res, table) =>
260           res.orElse(DtgAgeOffIterator.expiry(tableOps, previous, table))
261         }
262       }
263       configured.foreach(sft.setFeatureExpiration)
264     }
265 
266     super.preSchemaUpdate(sft, previous)
267   }

269   override protected def onSchemaUpdated(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
270     super.onSchemaUpdated(sft, previous)
271 
272     if (previous.statsEnabled) {
273       stats.removeStatCombiner(client, previous)
274     }
275     if (sft.statsEnabled) {
276       adapter.ensureTableExists(stats.metadata.table)
277       stats.configureStatCombiner(client, sft)
278     }
279 
280     val tableOps = client.tableOperations()
281     val previousTables = getAllIndexTableNames(previous.getTypeName).filter(tableOps.exists)
282     val tables = getAllIndexTableNames(sft.getTypeName).filter(tableOps.exists)
283 
284     if (previous.isVisibilityRequired != sft.isVisibilityRequired) {
285       previousTables.foreach(VisibilityIterator.clear(tableOps, _))
286       if (sft.isVisibilityRequired) {
287         tables.foreach(VisibilityIterator.set(tableOps, _))
288       }
289     }
290 
291     previousTables.foreach { table =>
292       AgeOffIterator.clear(tableOps, table)
293       DtgAgeOffIterator.clear(tableOps, table)
294     }
295 
296     sft.getFeatureExpiration.foreach {
297       case IngestTimeExpiration(ttl) =>
298         tables.foreach(AgeOffIterator.set(tableOps, _, sft, ttl))
299 
300       case FeatureTimeExpiration(dtg, _, ttl) =>
301         manager.indices(sft).foreach { index =>
302           val indexSft = index match {
303             case joinIndex: AttributeJoinIndex => joinIndex.indexSft
304             case _ => sft
305           }
306           DtgAgeOffIterator.set(tableOps, indexSft, index, ttl, dtg)
307         }
308 
309       case e => throw new IllegalArgumentException(s"Unexpected feature expiration: $e")
310     }
311   }

318   /**
319    * Gets a query plan that can be satisfied via AccumuloInputFormat - e.g. only 1 table and configuration.
320    *
321    * @param query query
322    * @return
323    */
324   def getSingleQueryPlan(query: Query): AccumuloQueryPlan = {
325     // disable join plans as those have multiple tables
326     JoinIndex.AllowJoinPlans.set(false)
327 
328     try {
329       lazy val fallbackIndex = {
330         val schema = getSchema(query.getTypeName)
331         manager.indices(schema, IndexMode.Read).headOption.getOrElse {
332           throw new IllegalStateException(s"Schema '${schema.getTypeName}' does not have any readable indices")
333         }
334       }
335 
336       val queryPlans = getQueryPlan(query)
337 
338       if (queryPlans.isEmpty) {
339         val filter =
340           FilterStrategy(fallbackIndex, None, Some(Filter.EXCLUDE), temporal = false, Float.PositiveInfinity, query.getHints)
341         EmptyPlan(QueryStrategy(filter, Seq.empty, Seq.empty, Seq.empty, filter.filter, None))
342       } else {
343         val qps =
344           if (queryPlans.lengthCompare(1) == 0) { queryPlans } else {
345             // this query requires multiple scans, which we can't execute from some input formats
346             // instead, fall back to a full table scan
347             logger.warn("Desired query plan requires multiple scans - falling back to full table scan")
348             getQueryPlan(query, Some(fallbackIndex.identifier))
349           }
350         if (qps.lengthCompare(1) > 0 || qps.exists(_.tables.lengthCompare(1) > 0)) {
351           logger.error("The query being executed requires multiple scans, which is not currently " +
352             "supported by GeoMesa. Your result set will be partially incomplete. " +
353             s"Query: ${FilterHelper.toString(query.getFilter)}")
354         }
355         qps.head
356       }
357     } finally {
358       // make sure we reset the thread locals
359       JoinIndex.AllowJoinPlans.remove()
360     }
361   }
