101   override protected def transitionIndices(sft: SimpleFeatureType): Unit = {
102     // note: versions already correspond to accumulo index versions
103     val dtg = sft.getDtgField.toSeq
104     val geom = Option(sft.getGeomField).toSeq
105     val indices = sft.getIndices.flatMap {
106       case id if id.name == IdIndex.name  => Seq(id) // no update needed
107       case id if id.name == "records"     => Seq(id.copy(name = IdIndex.name))
108       case id if id.name == Z3Index.name  => Seq(id.copy(attributes = geom ++ dtg))
109       case id if id.name == XZ3Index.name => Seq(id.copy(attributes = geom ++ dtg))
110       case id if id.name == Z2Index.name  => Seq(id.copy(attributes = geom))
111       case id if id.name == XZ2Index.name => Seq(id.copy(attributes = geom))
112       case id if id.name == AttributeIndex.name =>
113         lazy val fields = if (id.version < 4) { dtg } else { geom ++ dtg }
114         sft.getAttributeDescriptors.asScala.flatMap { d =>
115           val index = d.getUserData.remove(AttributeOptions.OptIndex).asInstanceOf[String]
116           if (index == null || index.equalsIgnoreCase(IndexCoverage.NONE.toString) || index.equalsIgnoreCase("false")) {
117             Seq.empty
118           } else if (index.equalsIgnoreCase(IndexCoverage.FULL.toString)) {
119             Seq(id.copy(name = AttributeIndex.name, attributes = Seq(d.getLocalName) ++ fields))
120           } else if (index.equalsIgnoreCase(IndexCoverage.JOIN.toString) || java.lang.Boolean.valueOf(index)) {
121             Seq(id.copy(name = JoinIndex.name, attributes = Seq(d.getLocalName) ++ fields))
122           } else {
123             throw new IllegalStateException(s"Expected an index coverage or boolean but got: $index")
124           }
125         }
126     }
127     sft.setIndices(indices)
128   }

199   override protected def onSchemaCreated(sft: SimpleFeatureType): Unit = {
200     super.onSchemaCreated(sft)
201     if (sft.statsEnabled) {
202       // configure the stats combining iterator on the table for this sft
203       adapter.ensureTableExists(stats.metadata.table)
204       stats.configureStatCombiner(connector, sft)
205     }
206     sft.getFeatureExpiration.foreach {
207       case IngestTimeExpiration(ttl) =>
208         val tableOps = connector.tableOperations()
209         getAllIndexTableNames(sft.getTypeName).filter(tableOps.exists).foreach { table =>
210           AgeOffIterator.set(tableOps, table, sft, ttl)
211         }
212 
213       case FeatureTimeExpiration(dtg, _, ttl) =>
214         val tableOps = connector.tableOperations()
215         manager.indices(sft).foreach { index =>
216           val indexSft = index match {
217             case joinIndex: AttributeJoinIndex => joinIndex.indexSft
218             case _ => sft
219           }
220           DtgAgeOffIterator.set(tableOps, indexSft, index, ttl, dtg)
221         }
222 
223       case e =>
224         throw new IllegalArgumentException(s"Unexpected feature expiration: $e")
225     }

229   override protected def preSchemaUpdate(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
230     if (sft.getVisibilityLevel == VisibilityLevel.Attribute && sft.getAttributeCount > 255) {
231       throw new IllegalArgumentException("Attribute level visibility only supports up to 255 attributes")
232     }
233 
234     // check for attributes flagged 'index=join' and convert them to sft-level user data
235     sft.getAttributeDescriptors.asScala.foreach { d =>
236       val index = d.getUserData.get(AttributeOptions.OptIndex).asInstanceOf[String]
237       if (index != null && index.equalsIgnoreCase(IndexCoverage.JOIN.toString)) {
238         d.getUserData.remove(AttributeOptions.OptIndex) // remove it so it's not processed again
239         val fields = Seq(d.getLocalName) ++ Option(sft.getGeomField) ++ sft.getDtgField.filter(_ != d.getLocalName)
240         val attribute = IndexId(JoinIndex.name, JoinIndex.version, fields, IndexMode.ReadWrite)
241         val existing = sft.getIndices.map(GeoMesaFeatureIndex.identifier)
242         if (!existing.contains(GeoMesaFeatureIndex.identifier(attribute))) {
243           sft.setIndices(sft.getIndices :+ attribute)
244         }
245       }
246     }
247 
248     // check any previous age-off - previously age-off wasn't tied to the sft metadata
249     if (!sft.isFeatureExpirationEnabled && !previous.isFeatureExpirationEnabled) {
250       // explicitly set age-off in the feature type if found
251       val tableOps = connector.tableOperations()
252       val tables = getAllIndexTableNames(previous.getTypeName).filter(tableOps.exists)
253       val ageOff = tables.foldLeft[Option[FeatureExpiration]](None) { (res, table) =>
254         res.orElse(AgeOffIterator.expiry(tableOps, table))
255       }
256       val configured = ageOff.orElse {
257         tables.foldLeft[Option[FeatureExpiration]](None) { (res, table) =>
258           res.orElse(DtgAgeOffIterator.expiry(tableOps, previous, table))
259         }
260       }
261       configured.foreach(sft.setFeatureExpiration)
262     }
263 
264     super.preSchemaUpdate(sft, previous)
265   }

284       if (sft.isVisibilityRequired) {
285         tables.foreach(VisibilityIterator.set(tableOps, _))
286       }
287     }
288 
289     previousTables.foreach { table =>
290       AgeOffIterator.clear(tableOps, table)
291       DtgAgeOffIterator.clear(tableOps, table)
292     }
293 
294     sft.getFeatureExpiration.foreach {
295       case IngestTimeExpiration(ttl) =>
296         tables.foreach(AgeOffIterator.set(tableOps, _, sft, ttl))
297 
298       case FeatureTimeExpiration(dtg, _, ttl) =>
299         manager.indices(sft).foreach { index =>
300           val indexSft = index match {
301             case joinIndex: AttributeJoinIndex => joinIndex.indexSft
302             case _ => sft
303           }
304           DtgAgeOffIterator.set(tableOps, indexSft, index, ttl, dtg)
305         }
306 
307       case e => throw new IllegalArgumentException(s"Unexpected feature expiration: $e")
308     }

319   override def getSchema(typeName: String): SimpleFeatureType = {
320     var sft = super.getSchema(typeName)
321     if (sft == null) {
322       // check for old-style metadata and re-write it if necessary
323       if (oldMetadata.getFeatureTypes.contains(typeName)) {
324         val lock = acquireCatalogLock()
325         try {
326           if (oldMetadata.getFeatureTypes.contains(typeName)) {
327             oldMetadata.migrate(typeName)
328             new SingleRowAccumuloMetadata[Stat](stats.metadata).migrate(typeName)
329           }
330         } finally {
331           lock.close()
332         }
333         sft = super.getSchema(typeName)
334       }
335     }
336     if (sft != null) {
337       // back compatible check for index versions
338       if (sft.getIndices.isEmpty) {
339         // make the sft temporarily mutable so we can update the keys
340         sft = SimpleFeatureTypes.mutable(sft)
341         // back compatible check if user data wasn't encoded with the sft
342         if (!sft.getUserData.containsKey(AccumuloDataStore.DeprecatedSchemaVersionKey)) {
343           metadata.read(typeName, "dtgfield").foreach(sft.setDtgField)
344           sft.getUserData.put(AccumuloDataStore.DeprecatedSchemaVersionKey,
345             metadata.readRequired(typeName, GeoMesaMetadata.VersionKey))
346 
347           // if no data is written, we default to 'false' in order to support old tables
348           if (metadata.read(typeName, "tables.sharing").exists(_.toBoolean)) {
349             sft.getUserData.put(SimpleFeatureTypes.Configs.TableSharing, "true")
350             // use schema id if available or fall back to old type name for backwards compatibility
351             val prefix = metadata.read(typeName, "id").getOrElse(s"${sft.getTypeName}~")
352             sft.getUserData.put(SimpleFeatureTypes.InternalConfigs.TableSharingPrefix, prefix)
353           }
354           // noinspection deprecation
355           SimpleFeatureTypes.Configs.ENABLED_INDEX_OPTS.foreach { i =>
356             metadata.read(typeName, i).foreach(e => sft.getUserData.put(SimpleFeatureTypes.Configs.EnabledIndices, e))
357           }
358         }
359 
360         // set the enabled indices
361         sft.setIndices(AccumuloDataStore.translateSchemaVersion(sft))
362 
363         // store the metadata and reload the sft again to validate indices
364         val encoded = SimpleFeatureTypes.encodeType(sft, includeUserData = true)
365         metadata.insert(typeName, GeoMesaMetadata.AttributesKey, encoded)
366         sft = super.getSchema(typeName)
367       }
368 
369       // back compatibility check for stat configuration
370       if (sft.statsEnabled && metadata.read(typeName, GeoMesaMetadata.StatsGenerationKey).isEmpty) {
371         // configure the stats combining iterator - we only use this key for older data stores
372         val configuredKey = "stats-configured"
373         if (!metadata.read(typeName, configuredKey).contains("true")) {
374           val lock = acquireCatalogLock()
375           try {
376             if (!metadata.read(typeName, configuredKey, cache = false).contains("true")) {
377               adapter.ensureTableExists(stats.metadata.table)
378               stats.configureStatCombiner(connector, sft)
379               metadata.insert(typeName, configuredKey, "true")
380             }
381           } finally {
382             lock.close()
383           }
384         }
385         // kick off asynchronous stats run for the existing data - this will set the stat date
386         // this may get triggered more than once, but should only run one time
387         val statsRunner = new StatsRunner(this)
388         statsRunner.submit(sft)
389         statsRunner.close()
390       }
391     }
392 
393     sft
394   }

411   /**
412     * Converts the old 'index schema' into the appropriate index identifiers
413     *
414     * @param sft simple feature type
415     * @return
416     */
417   private def translateSchemaVersion(sft: SimpleFeatureType): Seq[IndexId] = {
418     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
419     lazy val docs =
420       "http://www.geomesa.org/documentation/user/jobs.html#updating-existing-data-to-the-latest-index-format"
421 
422     val geom = Option(sft.getGeomField)
423     val dtg = sft.getDtgField
424 
425     val id = IndexId(IdIndex.name, -1, Seq.empty, IndexMode.ReadWrite)
426     val z3 = for { g <- geom; d <- dtg } yield { IndexId(Z3Index.name, -1, Seq(g, d), IndexMode.ReadWrite) }
427     val xz3 = for { g <- geom; d <- dtg } yield { IndexId(XZ3Index.name, 1, Seq(g, d), IndexMode.ReadWrite) }
428     val z2 = geom.map(g => IndexId(Z2Index.name, -1, Seq(g), IndexMode.ReadWrite))
429     val xz2 = geom.map(g => IndexId(XZ2Index.name, 1, Seq(g), IndexMode.ReadWrite))
430     val attributes = sft.getAttributeDescriptors.asScala.flatMap { d =>
431       val index = d.getUserData.remove(AttributeOptions.OptIndex).asInstanceOf[String]
432       if (index == null || index.equalsIgnoreCase(IndexCoverage.NONE.toString) || index.equalsIgnoreCase("false")) {
433         Seq.empty
434       } else if (index.equalsIgnoreCase(IndexCoverage.FULL.toString)) {
435         Seq(IndexId(AttributeIndex.name, -1, Seq(d.getLocalName) ++ dtg, IndexMode.ReadWrite))
436       } else if (index.equalsIgnoreCase(IndexCoverage.JOIN.toString) || java.lang.Boolean.valueOf(index)) {
437         Seq(IndexId(JoinIndex.name, -1, Seq(d.getLocalName) ++ dtg, IndexMode.ReadWrite))
438       } else {
439         throw new IllegalStateException(s"Expected an index coverage or boolean but got: $index")
440       }
441     }
442 
443     // note: 10 was the last valid value for CURRENT_SCHEMA_VERSION, which is no longer used except
444     // to transition old schemas from the 1.2.5 era
445     val version = {
446       // noinspection ScalaDeprecation
447       val string = sft.getUserData.remove(DeprecatedSchemaVersionKey).asInstanceOf[String]
448       if (string != null) { string.toInt } else { 10 }
449     }
450     val indices: Seq[IndexId] = if (version > 8) {
451       // note: version 9 was never in a release
452       val zs = if (sft.isPoints) { z3.map(_.copy(version = 3)) ++ z2.map(_.copy(version = 2)) } else { xz3 ++ xz2 }
453       zs.toSeq ++ Seq(id.copy(version = 2)) ++ attributes.map(_.copy(version = 3))
454     } else if (version == 8) {
455       z3.map(_.copy(version = 2)).toSeq ++ z2.map(_.copy(version = 1)) ++
456           Seq(id.copy(version = 1)) ++ attributes.map(_.copy(version = 2))
457     } else if (version > 5) {
458       logger.warn("The GeoHash index is no longer supported. Some queries may take longer than normal. To " +
459           s"update your data to a newer format, see $docs")
460       val z = if (version == 7) { z3.map(_.copy(version = 2)) } else { z3.map(_.copy(version = 1)) }
461       z.toSeq ++ Seq(id.copy(version = 1)) ++ attributes.map(_.copy(version = 2))
462     } else {
463       throw new NotImplementedError("This schema format is no longer supported. Please use " +
464           s"GeoMesa 1.2.6 to update you data to a newer format. For more information, see $docs")
465     }
466 
467     // noinspection deprecation
468     SimpleFeatureTypes.Configs.ENABLED_INDEX_OPTS.map(sft.getUserData.get).find(_ != null) match {
469       case None => indices
470       case Some(enabled) =>
471         // check for old index names
472         val e = enabled.toString.toLowerCase(Locale.US).split(",").map(_.trim).filterNot(_.isEmpty).map {
473           case "attr_idx" => AttributeIndex.name
474           case "records" => IdIndex.name
475           case i => i
476         }
477         indices.filter(i => e.contains(i.name))
478     }
479   }
