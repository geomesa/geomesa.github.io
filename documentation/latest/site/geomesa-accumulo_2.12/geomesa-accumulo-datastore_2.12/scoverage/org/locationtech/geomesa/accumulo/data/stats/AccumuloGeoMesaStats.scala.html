37   private val compactionScheduled = new AtomicBoolean(false)
38   private val lastCompaction = new AtomicLong(0L)

40   private val running = new AtomicBoolean(true)
41   private var scheduledCompaction: ScheduledFuture[_] = _

43   private val compactor = new Runnable() {
44     override def run(): Unit = {
45       import org.locationtech.geomesa.accumulo.AccumuloProperties.StatsProperties.STAT_COMPACTION_INTERVAL
46       val compactInterval = STAT_COMPACTION_INTERVAL.toDuration.get.toMillis
47       if (lastCompaction.get < System.currentTimeMillis() - compactInterval &&
48           compactionScheduled.compareAndSet(true, false) ) {
49         compact()
50       }
51       if (running.get) {
52         synchronized{scheduledCompaction = executor.schedule(this, compactInterval, TimeUnit.MILLISECONDS)}
53       }
54     }
55   }
56 
57   compactor.run() // schedule initial compaction

59   override val writer: GeoMesaStatWriter = new AccumuloMetadataStatWriter()
60 
61   override def close(): Unit = {
62     super.close()
63     running.set(false)
64     synchronized{scheduledCompaction.cancel(false)}
65   }

67   /**
68     * Configures the stat combiner to sum stats dynamically.
69     *
70     * Note: should be called with a distributed lock on the stats table
71     *
72     * @param connector accumulo connector
73     * @param sft simple feature type
74     */
75   def configureStatCombiner(connector: AccumuloClient, sft: SimpleFeatureType): Unit = {
76     import MetadataBackedStats._
77 
78     StatsCombiner.configure(sft, connector, metadata.table, metadata.typeNameSeparator.toString)
79 
80     val keys = Seq(CountKey, BoundsKeyPrefix, TopKKeyPrefix, FrequencyKeyPrefix, HistogramKeyPrefix)
81     val splits = new java.util.TreeSet[Text](keys.map(k => new Text(metadata.encodeRow(sft.getTypeName, k))).asJava)
82     connector.tableOperations().addSplits(metadata.table, splits)
83   }

89     *

91     * @param sft simple feature type
92     */
93   def removeStatCombiner(connector: AccumuloClient, sft: SimpleFeatureType): Unit =
94     StatsCombiner.remove(sft, connector, metadata.table, metadata.typeNameSeparator.toString)
95 
96   override protected def write(typeName: String, stats: Seq[WritableStat]): Unit = {
97     val (merge, overwrite) = stats.partition(_.merge)
98     metadata.insert(typeName, merge.map(s => s.key -> s.stat).toMap)
99     // invalidate the cache as we would need to reload from accumulo for the combiner to take effect
100     merge.foreach(s => metadata.invalidateCache(typeName, s.key))
101     if (overwrite.nonEmpty) {
102       // due to accumulo issues with combiners, deletes and compactions, we have to:
103       // 1) delete the existing data; 2) compact the table; 3) insert the new value
104       // see: https://issues.apache.org/jira/browse/ACCUMULO-2232
105       metadata.remove(typeName, overwrite.map(_.key))
106       compact()
107       metadata.insert(typeName, overwrite.map(s => s.key -> s.stat).toMap)
108     }
109   }
110 
111   /**
112     * Performs a synchronous compaction of the stats table
113     */
114   private [accumulo] def compact(wait: Boolean = true): Unit = {
115     compactionScheduled.set(false)
116     ds.client.tableOperations().compact(metadata.table, null, null, true, wait)
117     lastCompaction.set(System.currentTimeMillis())
118   }

120   class AccumuloMetadataStatWriter extends MetadataStatWriter {
121 
122     override def rename(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
123       // the stat combiner should still be configured for the old sft
124       // the call to super() will read the old rows and write new rows, but it doesn't read any
125       // new rows after writing them, so the combiner does not need to be correct yet
126       super.rename(sft, previous)
127       // now remove the old sft and configure the new one
128       removeStatCombiner(ds.client, previous)
129       configureStatCombiner(ds.client, sft)
130     }
131 
132     override def updater(sft: SimpleFeatureType): StatUpdater =
133       if (sft.statsEnabled) { new AccumuloStatUpdater(sft) } else { NoopStatUpdater }
134   }

139     * @param sft SimpleFeatureType
140     */
141   class AccumuloStatUpdater(sft: SimpleFeatureType) extends MetadataStatUpdater(sft) {
142     override def close(): Unit = {
143       super.close()
144       // schedule a compaction so our metadata doesn't stack up too much
145       compactionScheduled.set(true)
146     }
147   }
