20 class AccumuloBackedMetadata[T](val connector: AccumuloClient, val table: String, val serializer: MetadataSerializer[T])
21     extends KeyValueStoreMetadata[T] {
22 
23   import scala.collection.JavaConverters._
24 
25   private val config = GeoMesaBatchWriterConfig().setMaxMemory(10000L).setMaxWriteThreads(2)
26 
27   private val empty = new Text()
28 
29   private var closed = false
30 
31   private var writer: BatchWriter = _
32 
33   override protected def checkIfTableExists: Boolean = connector.tableOperations().exists(table)
34 
35   override protected def createTable(): Unit = new TableManager(connector).ensureTableExists(table)
36 
37   override protected def createEmptyBackup(timestamp: String): AccumuloBackedMetadata[T] =
38     new AccumuloBackedMetadata(connector, s"${table}_${timestamp}_bak", serializer)
39 
40   override protected def write(rows: Seq[(Array[Byte], Array[Byte])]): Unit = {
41     synchronized {
42       if (writer == null) {
43         if (closed) {
44           throw new IllegalStateException("Trying to write using a closed instance")
45         }
46         writer = connector.createBatchWriter(table, config)
47       }
48     }
49     rows.foreach { case (k, v) =>
50       val m = new Mutation(k)
51       m.put(empty, empty, new Value(v))
52       writer.addMutation(m)
53     }
54     writer.flush()
55   }
56 
57   override protected def delete(rows: Seq[Array[Byte]]): Unit = {
58     val ranges = rows.map(r => Range.exact(new Text(r))).asJava
59     val deleter = connector.createBatchDeleter(table, Authorizations.EMPTY, 1, config)
60     deleter.setRanges(ranges)
61     deleter.delete()
62     deleter.close()
63   }
64 
65   override protected def scanValue(row: Array[Byte]): Option[Array[Byte]] = {
66     val scanner = connector.createScanner(table, Authorizations.EMPTY)
67     scanner.setRange(Range.exact(new Text(row)))
68     try {
69       val iter = scanner.iterator
70       if (iter.hasNext) {
71         Some(iter.next.getValue.get)
72       } else {
73         None
74       }
75     } finally {
76       scanner.close()
77     }
78   }
79 
80   override protected def scanRows(prefix: Option[Array[Byte]]): CloseableIterator[(Array[Byte], Array[Byte])] = {
81     // ensure we don't scan any single-row encoded values
82     val range = prefix.map(p => Range.prefix(new Text(p))).getOrElse(new Range("", "~"))
83     val scanner = connector.createScanner(table, Authorizations.EMPTY)
84     scanner.setRange(range)
85     CloseableIterator(scanner.iterator.asScala.map(r => (r.getKey.getRow.copyBytes, r.getValue.get)), scanner.close())
86   }
87 
88   override def close(): Unit = synchronized {
89     if (!closed) {
90       closed = true
91       if (writer != null) {
92         CloseWithLogging(writer)
93       }
94     }
95   }
96 }

103   class SingleRowAccumuloMetadata[T](metadata: AccumuloBackedMetadata[T]) {
104 
105     import scala.collection.JavaConverters._
106 
107     // if the table doesn't exist, we assume that we don't ever need to check it for old-encoded rows
108     private val tableExists = metadata.connector.tableOperations().exists(metadata.table)
109 
110     def getFeatureTypes: Array[String] = {
111       if (!tableExists) { Array.empty } else {
112         val scanner = metadata.connector.createScanner(metadata.table, Authorizations.EMPTY)
113         // restrict to just one cf so we only get 1 hit per feature
114         // use attributes as it's the only thing that's been there through all geomesa versions
115         scanner.fetchColumnFamily(new Text(GeoMesaMetadata.AttributesKey))
116         try {
117           scanner.iterator.asScala.map(e => SingleRowAccumuloMetadata.getTypeName(e.getKey.getRow)).toArray
118         } finally {
119           scanner.close()
120         }
121       }
122     }
123 
124     /**
125       * Migrate a table from the old single-row metadata to the new
126       *
127       * @param typeName simple feature type name
128       */
129     def migrate(typeName: String): Unit = {
130       if (tableExists) {
131         val scanner = metadata.connector.createScanner(metadata.table, Authorizations.EMPTY)
132         val writer = metadata.connector.createBatchWriter(metadata.table, GeoMesaBatchWriterConfig())
133         try {
134           scanner.setRange(SingleRowAccumuloMetadata.getRange(typeName))
135           scanner.iterator.asScala.foreach { entry =>
136             val key = entry.getKey.getColumnFamily.toString
137             metadata.insert(typeName, key, metadata.serializer.deserialize(typeName, entry.getValue.get))
138             // delete for the old entry
139             val delete = new Mutation(entry.getKey.getRow)
140             delete.putDelete(entry.getKey.getColumnFamily, entry.getKey.getColumnQualifier)
141             writer.addMutation(delete)
142           }
143         } finally {
144           CloseQuietly(scanner)
145           if (writer != null) {
146             CloseQuietly(writer)
147           }
148         }
149       }
150     }
151   }
152 
153   object SingleRowAccumuloMetadata {
154 
155     private val MetadataTag = "~METADATA"
156     private val MetadataRowKeyRegex = (MetadataTag + """_(.*)""").r
157 
158     def getRange(typeName: String): Range = new Range(s"${MetadataTag}_$typeName")
159 
160     def getTypeName(row: Text): String = {
161       val MetadataRowKeyRegex(typeName) = row.toString
162       typeName
163     }
164   }
