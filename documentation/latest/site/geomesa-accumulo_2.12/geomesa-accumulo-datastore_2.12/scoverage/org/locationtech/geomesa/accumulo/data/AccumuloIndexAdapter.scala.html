70       index: GeoMesaFeatureIndex[_, _],
71       partition: Option[String],
72       splits: => Seq[Array[Byte]]): Unit = {
73     val table = index.configureTableName(partition) // writes table name to metadata
74     // create table if it doesn't exist
75     val created = ensureTableExists(table, index.sft.isLogicalTime)
76 
77     def addSplitsAndGroups(): Unit = {
78       // create splits
79       val splitsToAdd = splits.map(new Text(_)).toSet -- tableOps.listSplits(table).asScala.toSet
80       if (splitsToAdd.nonEmpty) {
81         tableOps.addSplits(table, new java.util.TreeSet(splitsToAdd.asJava))
82       }
83 
84       // create locality groups
85       val existingGroups = tableOps.getLocalityGroups(table)
86       val localityGroups = new java.util.HashMap[String, java.util.Set[Text]](existingGroups)
87 
88       def addGroup(cf: Text): Unit = {
89         val key = cf.toString
90         if (localityGroups.containsKey(key)) {
91           val update = new java.util.HashSet[Text](localityGroups.get(key))
92           update.add(cf)
93           localityGroups.put(key, update)
94         } else {
95           localityGroups.put(key, Collections.singleton(cf))
96         }
97       }
98 
99       groups.apply(index.sft).foreach { case (k, _) => addGroup(new Text(k)) }
100 
101       if (localityGroups != existingGroups) {
102         tableOps.setLocalityGroups(table, localityGroups)
103       }
104     }
105 
106     if (created) {
107       import org.apache.accumulo.core.conf.Property.{TABLE_BLOCKCACHE_ENABLED, TABLE_BLOOM_ENABLED, TABLE_BLOOM_KEY_FUNCTOR}
108 
109       addSplitsAndGroups()
110 
111       // block cache config
112       val enableBlockCache = {
113         val key = if (index.sft.isPartitioned) { InternalConfigs.PartitionTableCache } else { Configs.TableCacheEnabled }
114         val config = index.sft.getUserData.get(key).asInstanceOf[String]
115         if (config == null) { true } else {
116           val enabled = config.split(',').exists { hint =>
117             hint.equalsIgnoreCase(index.name) || hint.equalsIgnoreCase((Seq(index.name) ++ index.attributes).mkString(":")) ||
118               hint.equalsIgnoreCase(index.identifier)
119           }
120           logger.debug(s"Setting ${TABLE_BLOCKCACHE_ENABLED.getKey}=$enabled for index ${index.identifier} based on user config: $config")
121           enabled
122         }
123       }
124       if (enableBlockCache) {
125         tableOps.setProperty(table, TABLE_BLOCKCACHE_ENABLED.getKey, "true")
126       }
127 
128       if (index.name == IdIndex.name) {
129         // enable the row functor as the feature ID is stored in the Row ID
130         tableOps.setProperty(table, TABLE_BLOOM_KEY_FUNCTOR.getKey, classOf[RowFunctor].getCanonicalName)
131         tableOps.setProperty(table, TABLE_BLOOM_ENABLED.getKey, "true")
132       }
133 
134       if (index.sft.isVisibilityRequired) {
135         VisibilityIterator.set(tableOps, table)
136       }
137     } else if (index.keySpace.sharing.nonEmpty) {
138       // even if the table existed, we still need to check the splits and locality groups if it's shared
139       addSplitsAndGroups()
140     }
141   }

221           }
222         } else if (hints.isDensityQuery) {

224             val iter = DensityIterator.configure(schema, index, ecql, hints)

226           } else {
227             semiLocalPlan()
228           }
229         } else if (hints.isStatsQuery) {

232             val reduce = Some(StatsScan.StatsReducer(schema, hints))
233             scanPlan(Some(iter), new AccumuloStatsResultsToFeatures(), reduce)
234           } else {
235             semiLocalPlan()
236           }
237         } else {
238           scanPlan(fti, AccumuloResultsToFeatures(index, returnSchema), None)
239         }
240 
241         if (tables.isEmpty || ranges.isEmpty) { EmptyPlan(strategy, plan.reducer) } else { plan }
242     }
243   }

259   override def getStrategyCost(strategy: FilterStrategy, explain: Explainer): Option[Long] = {
260     explain.pushLevel(s"Calculating cost for ${strategy.index.identifier}")
261     val start = System.currentTimeMillis()
262     try {
263       val tables = strategy.index.getTablesForQuery(strategy.filter)
264       if (tables.isEmpty) {
265         return Some(0L)
266       }
267       val ranges = strategy.getQueryStrategy(explain).ranges.map(toAccumuloRange).asJava
268       val cost =
269         tables.foldLeft(0d) { case (sum, table) =>
270           val numTablets = tableSizeCache.get(table)
271           val tabletsScanned = tableOps.locate(table, ranges).groupByTablet().size()
272           explain(s"Strategy hits $tabletsScanned/$numTablets tablets for table $table")
273           val cost = 100 * (tabletsScanned.toDouble / numTablets)
274           sum + cost
275         }
276       Some(cost.toLong)
277     } finally {
278       explain(s"Cost calculations took ${System.currentTimeMillis() - start}ms").popLevel()
279     }

288    * Converts a generic index-api range into an Accumulo range
289    *
290    * @param range range
291    * @return
292    */
293   private def toAccumuloRange(range: ByteRange): Range = range match {
294     case BoundedByteRange(lower, upper) =>
295       // index api defines empty start/end for open-ended range - in accumulo, it's indicated with null
296       val start = if (lower.length == 0) { null } else { new Key(new Text(lower)) }
297       val end = if (upper.length == 0) { null } else { new Key(new Text(upper)) }
298       // index api defines start row inclusive, end row exclusive
299       new Range(start, true, end, false)
300 
301     case SingleRowByteRange(row) =>
302       new Range(new Text(row))
303 
304     case _ =>
305       throw new IllegalArgumentException(s"Unexpected range type $range")
306   }
