73       index: GeoMesaFeatureIndex[_, _],
74       partition: Option[String],
75       splits: => Seq[Array[Byte]]): Unit = {
76     val table = index.configureTableName(partition) // writes table name to metadata
77     // create table if it doesn't exist
78     val created = ensureTableExists(table, index.sft.isLogicalTime)
79 
80     def addSplitsAndGroups(): Unit = {
81       // create splits
82       val splitsToAdd = splits.map(new Text(_)).toSet -- tableOps.listSplits(table).asScala.toSet
83       if (splitsToAdd.nonEmpty) {
84         tableOps.addSplits(table, new java.util.TreeSet(splitsToAdd.asJava))
85       }
86 
87       // create locality groups
88       val existingGroups = tableOps.getLocalityGroups(table)
89       val localityGroups = new java.util.HashMap[String, java.util.Set[Text]](existingGroups)
90 
91       def addGroup(cf: Text): Unit = {
92         val key = cf.toString
93         if (localityGroups.containsKey(key)) {
94           val update = new java.util.HashSet[Text](localityGroups.get(key))
95           update.add(cf)
96           localityGroups.put(key, update)
97         } else {
98           localityGroups.put(key, Collections.singleton(cf))
99         }
100       }
101 
102       groups.apply(index.sft).foreach { case (k, _) => addGroup(new Text(k)) }
103 
104       if (localityGroups != existingGroups) {
105         tableOps.setLocalityGroups(table, localityGroups)
106       }
107     }
108 
109     if (created) {
110       import org.apache.accumulo.core.conf.Property.{TABLE_BLOCKCACHE_ENABLED, TABLE_BLOOM_ENABLED, TABLE_BLOOM_KEY_FUNCTOR}
111 
112       addSplitsAndGroups()
113 
114       // block cache config
115       val enableBlockCache = {
116         val key = if (index.sft.isPartitioned) { InternalConfigs.PartitionTableCache } else { Configs.TableCacheEnabled }
117         val config = index.sft.getUserData.get(key).asInstanceOf[String]
118         if (config == null) { true } else {
119           val enabled = config.split(',').exists { hint =>
120             hint.equalsIgnoreCase(index.name) || hint.equalsIgnoreCase((Seq(index.name) ++ index.attributes).mkString(":")) ||
121               hint.equalsIgnoreCase(index.identifier)
122           }
123           logger.debug(s"Setting ${TABLE_BLOCKCACHE_ENABLED.getKey}=$enabled for index ${index.identifier} based on user config: $config")

125         }
126       }
127       if (enableBlockCache) {
128         tableOps.setProperty(table, TABLE_BLOCKCACHE_ENABLED.getKey, "true")
129       }
130 
131       if (index.name == IdIndex.name) {
132         // enable the row functor as the feature ID is stored in the Row ID
133         tableOps.setProperty(table, TABLE_BLOOM_KEY_FUNCTOR.getKey, classOf[RowFunctor].getCanonicalName)
134         tableOps.setProperty(table, TABLE_BLOOM_ENABLED.getKey, "true")
135       }
136 
137       if (index.sft.isVisibilityRequired) {
138         VisibilityIterator.set(tableOps, table)
139       }
140     } else if (index.keySpace.sharing.nonEmpty) {
141       // even if the table existed, we still need to check the splits and locality groups if it's shared
142       addSplitsAndGroups()
143     }
144   }

177     lazy val returnSchema = hints.getTransformSchema.getOrElse(schema)

205               // override the return sft to reflect what we're actually returning,
206               // since the arrow sft is only created in the local reduce step
207               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
208             }
209             (fti, resultsToFeatures, localReducer)
210           }
211         } else if (hints.isDensityQuery) {
212           if (ds.config.remote.density) {
213             val iter = DensityIterator.configure(schema, index, ecql, hints)
214             (Seq(iter), new AccumuloDensityResultsToFeatures(), None)
215           } else {
216             if (hints.isSkipReduce) {
217               // override the return sft to reflect what we're actually returning,

222           }

225             val iter = StatsIterator.configure(schema, index, ecql, hints)
226             val reduce = Some(StatsScan.StatsReducer(schema, hints))

228           } else {
229             if (hints.isSkipReduce) {
230               // override the return sft to reflect what we're actually returning,

232               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
233             }

235           }
236         } else {
237           (fti, resultsToFeatures, None)
238         }
239 
240         if (ranges.isEmpty) { EmptyPlan(strategy, reduce) } else {
241           // configure additional iterators based on the index
242           // TODO pull this out to be SPI loaded so that new indices can be added seamlessly
243           val indexIter = if (index.name == Z3Index.name) {

248           } else if (index.name == Z2Index.name) {

250               Z2Iterator.configure(v, index.keySpace.sharding.length + index.keySpace.sharing.length, ZIterPriority)
251             }
252           } else if (index.name == S3Index.name) {
253             strategy.values.toSeq.map { case v: S3IndexValues =>
254               S3Iterator.configure(v, index.keySpace.sharding.length, ZIterPriority)
255             }
256           } else if (index.name == S2Index.name) {
257             strategy.values.toSeq.map { case v: S2IndexValues =>
258               S2Iterator.configure(v, index.keySpace.sharding.length, ZIterPriority)
259             }
260           } else {
261             Seq.empty
262           }
263 
264           // add the attribute-level vis iterator if necessary
265           val visIter = index.sft.getVisibilityLevel match {
266             case VisibilityLevel.Attribute => Seq(KryoVisibilityRowEncoder.configure(schema))
267             case _ => Seq.empty
268           }
269 
270           val iters = iter ++ indexIter ++ visIter
271 
272           val sort = hints.getSortFields
273           val max = hints.getMaxFeatures
274           val project = hints.getProjection
275 
276           BatchScanPlan(strategy, tables, ranges, iters, colFamily, eToF, reduce, sort, max, project, numThreads)
277         }
278     }

295   override def getStrategyCost(strategy: FilterStrategy, explain: Explainer): Option[Long] = {
296     explain.pushLevel(s"Calculating cost for ${strategy.index.identifier}")
297     val start = System.currentTimeMillis()
298     try {
299       val tables = strategy.index.getTablesForQuery(strategy.filter)
300       if (tables.isEmpty) {
301         return Some(0L)
302       }
303       val ranges = strategy.getQueryStrategy(explain).ranges.map(toAccumuloRange).asJava
304       val cost =
305         tables.foldLeft(0d) { case (sum, table) =>
306           val numTablets = tableSizeCache.get(table)
307           val tabletsScanned = tableOps.locate(table, ranges).groupByTablet().size()
308           explain(s"Strategy hits $tabletsScanned/$numTablets tablets for table $table")
309           val cost = 100 * (tabletsScanned.toDouble / numTablets)
310           sum + cost
311         }
312       Some(cost.toLong)
313     } finally {
314       explain(s"Cost calculations took ${System.currentTimeMillis() - start}ms").popLevel()
315     }
316   }

323   /**
324    * Converts a generic index-api range into an Accumulo range
325    *
326    * @param range range
327    * @return
328    */
329   private def toAccumuloRange(range: ByteRange): Range = range match {
330     case BoundedByteRange(lower, upper) =>
331       // index api defines empty start/end for open-ended range - in accumulo, it's indicated with null
332       val start = if (lower.length == 0) { null } else { new Key(new Text(lower)) }
333       val end = if (upper.length == 0) { null } else { new Key(new Text(upper)) }
334       // index api defines start row inclusive, end row exclusive
335       new Range(start, true, end, false)
336 
337     case SingleRowByteRange(row) =>
338       new Range(new Text(row))
339 
340     case _ =>
341       throw new IllegalArgumentException(s"Unexpected range type $range")
342   }
