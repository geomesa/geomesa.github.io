131   /**
132    * Build an Accumulo client configuration, based on passed in parameters and any accumulo-client.properties
133    * file on the classpath
134    *
135    * @param params data store params
136    * @return
137    */
138   def buildAccumuloClientConfig(params: java.util.Map[String, _]): Properties = {
139     val config = AccumuloClientConfig.load()
140 
141     def setRequired(param: GeoMesaParam[String], key: ClientProperty): String = {
142       param.lookupOpt(params) match {
143         case Some(v) => config.put(key.getKey, v); v
144         case None =>
145           Option(config.getProperty(key.getKey)).getOrElse {
146             throw new IOException(s"Parameter ${param.key} is required: ${param.description}")
147           }
148       }
149     }
150 
151     def setOptional(param: GeoMesaParam[String], key: ClientProperty): Unit =
152       param.lookupOpt(params).foreach(config.put(key.getKey, _))
153 
154     def getRequired(param: GeoMesaParam[String], key: ClientProperty): String = {
155       param.lookupOpt(params).orElse(Option(config.getProperty(key.getKey))).getOrElse {
156         throw new IOException(s"Parameter ${param.key} is required: ${param.description}")
157       }
158     }
159 
160     setRequired(InstanceNameParam, ClientProperty.INSTANCE_NAME)
161     setRequired(ZookeepersParam, ClientProperty.INSTANCE_ZOOKEEPERS)
162     setOptional(ZookeeperTimeoutParam, ClientProperty.INSTANCE_ZOOKEEPERS_TIMEOUT)
163 
164     val user = setRequired(UserParam, ClientProperty.AUTH_PRINCIPAL)
165 
166     if (PasswordParam.exists(params) && KeytabPathParam.exists(params)) {
167       throw new IllegalArgumentException(
168         s"'${PasswordParam.key}' and '${KeytabPathParam.key}' are mutually exclusive, but are both set")
169     }
170 
171     val authType =
172       if (PasswordParam.exists(params)) {
173         AccumuloClientConfig.PasswordAuthType
174       } else if (KeytabPathParam.exists(params)) {
175         AccumuloClientConfig.KerberosAuthType
176       } else {
177         ClientProperty.AUTH_TYPE.getValue(config).toLowerCase(Locale.US)
178       }
179 
180     val token =
181       if (authType == AccumuloClientConfig.PasswordAuthType) {
182         new PasswordToken(getRequired(PasswordParam, ClientProperty.AUTH_TOKEN).getBytes(StandardCharsets.UTF_8))
183       } else if (authType == AccumuloClientConfig.KerberosAuthType) {
184         val file = new java.io.File(getRequired(KeytabPathParam, ClientProperty.AUTH_TOKEN))
185         // mimic behavior from accumulo 1.9 and earlier:
186         // `public KerberosToken(String principal, File keytab, boolean replaceCurrentUser)`
187         UserGroupInformation.loginUserFromKeytab(user, file.getAbsolutePath)
188         new KerberosToken(user, file)
189       } else {
190          throw new IllegalArgumentException(
191            s"Unsupported auth type: $authType - supported values are " +
192              s"${AccumuloClientConfig.PasswordAuthType}, ${AccumuloClientConfig.KerberosAuthType}")
193       }
194     ClientProperty.setAuthenticationToken(config, token)
195 
196     if (WriteThreadsParam.exists(params)) {
197       config.put(ClientProperty.BATCH_WRITER_THREADS_MAX.getKey, WriteThreadsParam.lookup(params).toString)
198     } else if (ClientProperty.BATCH_WRITER_THREADS_MAX.isEmpty(config)) {
199       BatchWriterProperties.WRITER_THREADS.option.foreach { threads =>
200         config.put(ClientProperty.BATCH_WRITER_THREADS_MAX.getKey, String.valueOf(threads))
201       }
202     }
203     if (ClientProperty.BATCH_WRITER_MEMORY_MAX.isEmpty(config)) {
204       BatchWriterProperties.WRITER_MEMORY_BYTES.toBytes.foreach { memory =>
205         config.put(ClientProperty.BATCH_WRITER_MEMORY_MAX.getKey, String.valueOf(memory))
206       }
207     }
208     if (ClientProperty.BATCH_WRITER_LATENCY_MAX.isEmpty(config)) {
209       BatchWriterProperties.WRITER_LATENCY.toDuration.foreach { duration =>
210         config.put(ClientProperty.BATCH_WRITER_LATENCY_MAX.getKey, s"${duration.toMillis}ms")
211       }
212     }
213     if (ClientProperty.BATCH_WRITER_TIMEOUT_MAX.isEmpty(config)) {
214       BatchWriterProperties.WRITE_TIMEOUT.toDuration.foreach { duration =>
215         config.put(ClientProperty.BATCH_WRITER_TIMEOUT_MAX.getKey, s"${duration.toMillis}ms")
216       }
217     }
218 
219     config
220   }

235       new AccumuloAuditWriter(client, s"${catalog}_queries", auditProvider, AuditQueriesParam.lookup(params).booleanValue())
236     val metrics = MetricsRegistryParam.lookupRegistry(params)
237 
238     val queries = AccumuloQueryConfig(
239       threads = QueryThreadsParam.lookup(params),
240       recordThreads = RecordThreadsParam.lookup(params),

242       looseBBox = LooseBBoxParam.lookup(params),
243       parallelPartitionScans = PartitionParallelScansParam.lookup(params),
244       consistency = QueryConsistencyParam.lookupOpt(params).map(n => ConsistencyLevel.valueOf(n.toUpperCase(Locale.US)))
245     )

247     val remote = RemoteScansEnabled(

249       bin = RemoteBinParam.lookup(params),
250       density = RemoteDensityParam.lookup(params),
251       stats = RemoteStatsParam.lookup(params)

254     AccumuloDataStoreConfig(
255       catalog = catalog,
256       generateStats = GenerateStatsParam.lookup(params),
257       authProvider = authProvider,
258       auditWriter = auditWriter,
259       metrics = metrics,
260       queries = queries,

265   }

275   def buildAuthsProvider(client: AccumuloClient, params: java.util.Map[String, _]): AuthorizationsProvider = {
276     // convert the connector authorizations into a string array - this is the maximum auths this connector can support
277     val securityOps = client.securityOperations
278     val masterAuths = securityOps.getUserAuthorizations(client.whoami).asScala.toSeq.map(b => new String(b))
279 
280     // get the auth params passed in as a comma-delimited string
281     val configuredAuths = AuthsParam.lookupOpt(params).getOrElse("").split(",").filterNot(_.isEmpty).toSeq
282 
283     // verify that the configured auths are valid for the connector we are using (fail-fast)
284     val invalidAuths = configuredAuths.filterNot(masterAuths.contains)
285     if (invalidAuths.nonEmpty) {
286       throw new IllegalArgumentException(s"The authorizations '${invalidAuths.mkString(",")}' " +
287         "are not valid for the Accumulo connection being used")
288     }
289 
290     val forceEmptyAuths = ForceEmptyAuthsParam.lookup(params)
291 
292     // if the caller provided any non-null string for authorizations, use it;
293     // otherwise, grab all authorizations to which the Accumulo user is entitled
294     if (configuredAuths.nonEmpty && forceEmptyAuths) {
295       throw new IllegalArgumentException("Forcing empty auths is checked, but explicit auths are provided")
296     }
297     val auths = if (forceEmptyAuths || configuredAuths.nonEmpty) { configuredAuths } else { masterAuths }
298 
299     AuthUtils.getProvider(params, auths)
300   }
