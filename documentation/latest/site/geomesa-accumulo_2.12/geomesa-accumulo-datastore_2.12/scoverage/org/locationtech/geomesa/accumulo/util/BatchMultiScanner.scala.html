23 /**
24  * Runs a join scan against two tables
25  *
26  * @param connector connector
27  * @param in input scan
28  * @param join join scan
29  * @param joinFunction maps results of input scan to ranges for join scan
30  * @param auths scan authorizations
31  * @param partitionParallelScans parallelize scans against partitioned tables (vs execute sequentially)
32  * @param timeout query timeout
33  * @param numThreads threads
34  * @param batchSize batch size
35  */
36 class BatchMultiScanner(
37     connector: AccumuloClient,
38     in: ScannerBase,
39     join: BatchScanPlan,
40     joinFunction: JoinFunction,
41     auths: Authorizations,
42     partitionParallelScans: Boolean,
43     timeout: Option[Timeout],

46   ) extends CloseableIterator[java.util.Map.Entry[Key, Value]] with LazyLogging {

50   require(batchSize > 0, f"Illegal batchSize ($batchSize%d). Value must be > 0")
51   require(numThreads > 0, f"Illegal numThreads ($numThreads%d). Value must be > 0")

54   private val executor = Executors.newFixedThreadPool(numThreads)
55 
56   private val inQ  = new LinkedBlockingQueue[Entry[Key, Value]](batchSize)
57   private val outQ = new LinkedBlockingQueue[Entry[Key, Value]](batchSize)
58 
59   private val inDone  = new AtomicBoolean(false)
60   private val outDone = new AtomicBoolean(false)
61 
62   private var prefetch: Entry[Key, Value] = _
63 
64   private def prefetchIfNull(): Unit = {
65     // loop while we might have another and we haven't set prefetch
66     while (prefetch == null && (!outDone.get || outQ.size > 0)) {
67       prefetch = outQ.poll(5, TimeUnit.MILLISECONDS)
68     }
69   }
70 
71   // must attempt a prefetch since we don't know whether or not the outQ
72   // will actually be filled with an item (filters may not match and the
73   // in scanner may never return a range)
74   override def hasNext(): Boolean = {
75     prefetchIfNull()
76     prefetch != null
77   }
78 
79   override def next(): Entry[Key, Value] = {
80     prefetchIfNull()
81     val ret = prefetch
82     prefetch = null
83     ret
84   }
85 
86   executor.submit(new Runnable {
87     override def run(): Unit = {
88       try {
89         in.asScala.foreach(inQ.put)
90       } finally {
91         inDone.set(true)
92       }
93     }
94   })
95 
96   executor.submit(new Runnable {
97     override def run(): Unit = {
98       try {
99         val tasks = collection.mutable.ListBuffer.empty[Future[_]]
100         while (!inDone.get || inQ.size() > 0) {
101           val entry = inQ.poll(5, TimeUnit.MILLISECONDS)
102           if (entry != null) {
103             val entries = collection.mutable.ListBuffer(entry)
104             inQ.drainTo(entries.asJava)
105             val task = executor.submit(new Runnable {
106               override def run(): Unit = {
107                 val iterator = join.copy(ranges = entries.map(joinFunction).toSeq).scan(connector, auths, partitionParallelScans, timeout)
108                 try {
109                   iterator.foreach(outQ.put)
110                 } finally {
111                   iterator.close()
112                 }
113               }
114             })
115             tasks.append(task)
116           }
117         }
118         tasks.foreach(_.get)
119       } catch {
120         case _: InterruptedException =>
121       } finally {
122         executor.shutdown()
123         outDone.set(true)
124       }
125     }
126   })
127 
128   override def close(): Unit = {
129     if (!executor.isShutdown) {
130       executor.shutdownNow()
131     }
132     in.close()
133   }
