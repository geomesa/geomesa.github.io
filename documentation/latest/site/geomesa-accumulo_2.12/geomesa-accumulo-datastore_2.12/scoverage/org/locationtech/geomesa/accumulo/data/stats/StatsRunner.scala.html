25 /**
26   * Update stats for a data store in a background thread. Uses distributed locking to ensure
27   * that work isn't being duplicated.
28   *
29   * Note: currently we do not schedule stats to automatically run. Instead, we can use the
30   * command line tools and OS scheduling (e.g. cron).
31   *
32   * @param ds data store to collect stats for
33   */
34 class StatsRunner(ds: AccumuloDataStore) extends Runnable with Closeable {
35 
36   private val es = Executors.newSingleThreadScheduledExecutor()
37   private val scheduled = new AtomicBoolean(false)
38   private val shutdown  = new AtomicBoolean(false)
39 
40   /**
41     * Runs updates asynchronously. Will continue scheduling itself until 'close' is called.
42     *
43     * @param initialDelay initial delay, in minutes
44     */
45   def scheduleRepeating(initialDelay: Int = 0): Unit = {
46     scheduled.set(true)
47     if (initialDelay > 0) {
48       es.schedule(this, initialDelay, TimeUnit.MINUTES)
49     } else {
50       es.submit(this)
51     }
52   }
53 
54   /**
55     * Submits a stat run for the given sft
56     *
57     * @param sft simple feature type
58     * @param delay delay, in minutes before executing
59     * @return
60     */
61   def submit(sft: SimpleFeatureType, delay: Int = 0): Future[Instant] = {
62     val runner = new StatRunner(ds, sft)
63     if (delay > 0) {
64       es.schedule(runner, delay, TimeUnit.MINUTES)
65     } else {
66       es.submit(runner)
67     }
68   }
69 
70   /**
71     * Checks for the last time stats were run, and runs if needed.
72     * Updates metadata accordingly.
73     */
74   override def run(): Unit = {
75     // convert to iterator so we check shutdown before each update
76     val sfts = ds.getTypeNames.map(ds.getSchema).iterator.filter(_ => !shutdown.get())
77     // try to get an exclusive lock on the sft - if not, don't wait just move along
78     val lockTimeout = Some(1000L)
79     // force execution of iterator
80     val minUpdate = if (sfts.isEmpty) { None } else {
81       Some(sfts.map(new StatRunner(ds, _, lockTimeout).call()).map(_.toEpochMilli).min)
82     }
83     // wait at least one minute before running again
84     val minWait = 60000L
85     val nextRun = minUpdate.map(_ - Instant.now(Clock.systemUTC()).toEpochMilli).filter(_ > minWait).getOrElse(minWait)
86 
87     if (scheduled.get() && !shutdown.get()) {
88       es.schedule(this, nextRun, TimeUnit.MILLISECONDS)
89     }
90   }
91 
92   override def close(): Unit = {
93     shutdown.getAndSet(true)
94     es.shutdown()
95   }
96 }

98 /**
99   * Callable to check and update if necessary stats for a single sft
100   *
101   * @param ds accumulo data store
102   * @param sft simple feature type to check stats for
103   * @param lockTimeout timeout for how long to wait for distributed lock, in millis.
104   *                    If none, will wait indefinitely
105   */
106 class StatRunner(ds: AccumuloDataStore, sft: SimpleFeatureType, lockTimeout: Option[Long] = None)
107     extends Callable[Instant] with ZookeeperLocking {
108 
109   override protected def zookeepers: String =
110     ds.client.properties().getProperty(ClientProperty.INSTANCE_ZOOKEEPERS.getKey)
111 
112   /**
113     * Runs stats for the simple feature type
114     *
115     * @return time of the next scheduled update
116     */
117   override def call(): Instant = {
118     // do a quick check on the last time stats were updated
119     val updateInterval = getUpdateInterval
120     val unsafeUpdate = getLastUpdate.plus(updateInterval, ChronoUnit.MINUTES)
121 
122     if (unsafeUpdate.isAfter(Instant.now(Clock.systemUTC()))) {
123       unsafeUpdate
124     } else {
125       // get the lock and re-check, in case stats have been updated by another thread
126       val lockOption = lockTimeout match {
127         case None => Some(acquireDistributedLock(lockKey))
128         case Some(timeout) => acquireDistributedLock(lockKey, timeout)
129       }
130       lockOption match {
131         case None => Instant.now(Clock.systemUTC()).plus(5, ChronoUnit.MINUTES) // default to check again in 5 minutes
132         case Some(lock) =>
133           try {
134             // reload next update now that we have the lock
135             val nextUpdate = getLastUpdate.plus(updateInterval, ChronoUnit.MINUTES)
136             if (nextUpdate.isAfter(Instant.now(Clock.systemUTC()))) {
137               nextUpdate
138             } else {
139               // run the update - this updates the last update time too
140               ds.stats.writer.analyze(sft)
141               Instant.now(Clock.systemUTC()).plus(updateInterval, ChronoUnit.MINUTES)
142             }
143           } finally {
144             lock.close()
145           }
146       }
147     }
148   }
149 
150   /**
151     * Reads the time of the last update
152     *
153     * @return last update
154     */
155   private def getLastUpdate: Instant = {
156     ds.metadata.read(sft.getTypeName, GeoMesaMetadata.StatsGenerationKey, cache = false) match {
157       case Some(dt) => Instant.from(GeoToolsDateFormat.parse(dt))
158       case None     => Instant.ofEpochSecond(0)
159     }
160   }
161 
162   /**
163     * Reads the update interval.
164     *
165     * @return update interval, in minutes
166     */
167   private def getUpdateInterval: Long =
168     // note: default is 1440 minutes (one day)
169     ds.metadata.read(sft.getTypeName, GeoMesaMetadata.StatsIntervalKey).map(_.toLong).getOrElse(1440)
170 
171   private def lockKey: String = {
172     val ca = StringSerialization.alphaNumericSafeString(ds.config.catalog)
173     val tn = StringSerialization.alphaNumericSafeString(sft.getTypeName)
174     s"/org.locationtech.geomesa/accumulo/stats/$ca/$tn"
175   }
