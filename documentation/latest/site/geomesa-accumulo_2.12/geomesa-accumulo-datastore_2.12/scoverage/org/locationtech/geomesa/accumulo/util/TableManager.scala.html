27 /**
28  * Manages table creation/deletion
29  *
30  * @param client accumulo client - note, this client must be cleaned up externally (as it's usually shared)
31  */
32 class TableManager(client: AccumuloClient) {
33 
34   private val delegate: TableLock = TableSynchronization(TableCreationSync.get) match {
35     case TableSynchronization.ZooKeeper => new TableLockZk(client)
36     case TableSynchronization.Local     => new TableLockLocal(client)
37     case TableSynchronization.None      => new TableLockNone(client)
38   }
39 
40   /**
41    * Create table if it does not exist.
42    *
43    * Note: to avoid having to lock and check namespaces, the namespace for the table must already exist. Generally
44    * all tables for a data store share a namespace, and the namespace is created up front, so it will always exist.
45    *
46    * @param table table name
47    * @param useLogicalTime use logical table time
48    * @return true if table was created, false if table already exists
49    */
50   def ensureTableExists(table: String, useLogicalTime: Boolean = true): Boolean =
51     delegate.ensureTableExists(table, if (useLogicalTime) { TimeType.LOGICAL } else { TimeType.MILLIS })
52 
53   /**
54    * Create namespace if it does not exist
55    *
56    * @param namespace namespace
57    */
58   def ensureNamespaceExists(namespace: String): Unit = delegate.ensureNamespaceExists(namespace)
59 
60   /**
61    * Delete a table
62    *
63    * @param table table to delete
64    */
65   def deleteTable(table: String): Unit = delegate.deleteTable(table)
66 
67   // note: shadows IndexAdapter.renameTable
68   // noinspection ScalaUnusedSymbol
69   def renameTable(from: String, to: String): Unit = delegate.renameTable(from, to)
70 }

129   /**
130    * Table utility class
131    *
132    * @param client accumulo client
133    */
134   private abstract class TableLock(client: AccumuloClient) extends DistributedLocking {
135 
136     private val timeoutMillis = DistributedLockTimeout.toDuration.map(_.toMillis).getOrElse {
137       // note: this property has a valid default value so this exception should never be triggered
138       throw new IllegalArgumentException(s"Couldn't convert '${DistributedLockTimeout.get}' to a duration")
139     }
140 
141     // note: value is not used here, we're treating this as a set
142     private val tableCache: Cache[String, java.lang.Boolean] =
143       Caffeine.newBuilder().expireAfterWrite(TableCacheExpiry.toDuration.get.toMillis, TimeUnit.MILLISECONDS).build()
144 
145     private val nsCache: Cache[String, java.lang.Boolean] =
146       Caffeine.newBuilder().expireAfterWrite(TableCacheExpiry.toDuration.get.toMillis, TimeUnit.MILLISECONDS).build()
147 
148     /**
149      * Create the table if it doesn't exist
150      *
151      * @param table table name
152      * @param timeType table time type
153      * @return true if table was created, false if it already exists
154      */
155     def ensureTableExists(table: String, timeType: TimeType): Boolean = {
156       var created = false
157       tableCache.get(table, _ => {
158         withLock(tablePath(table), timeoutMillis, {
159           val tableOps = client.tableOperations()
160           while (!tableOps.exists(table)) {
161             try {
162               tableOps.create(table, new NewTableConfiguration().setTimeType(timeType))
163               created = true
164             } catch {
165               case _: TableExistsException => onTableExists(table)
166             }
167           }
168         })
169         java.lang.Boolean.FALSE
170       })
171       created
172     }
173 
174     /**
175      * Creates the namespace if it doesn't exist
176      *
177      * @param namespace namespace (or table name with namespace included)
178      * @return true if namespace was created, false if it already existed
179      */
180     def ensureNamespaceExists(namespace: String): Unit = {
181       val ns = namespace.takeWhile(_ != '.')
182       nsCache.get(ns, _ => {
183         withLock(nsPath(ns), timeoutMillis, {
184           val nsOps = client.namespaceOperations
185           while (!nsOps.exists(ns)) {
186             try { nsOps.create(ns) } catch {
187               case _: NamespaceExistsException => onNamespaceExists(ns)
188             }
189           }
190         })
191         java.lang.Boolean.FALSE
192       })
193     }
194 
195     /**
196      * Rename a table
197      *
198      * @param from current name
199      * @param to new name
200      */
201     def renameTable(from: String, to: String): Unit = {
202       withLock(tablePath(from), timeoutMillis, {
203         val tableOps = client.tableOperations()
204         if (tableOps.exists(from)) {
205           withLock(tablePath(to), timeoutMillis, {
206             tableOps.rename(from, to)
207             tableCache.put(to, java.lang.Boolean.FALSE)
208           })
209         }
210         tableCache.invalidate(from)
211       })
212     }
213 
214     /**
215      * Delete a table
216      *
217      * @param table table to delete
218      */
219     def deleteTable(table: String): Unit = {
220       withLock(tablePath(table), timeoutMillis, {
221         val tableOps = client.tableOperations()
222         if (tableOps.exists(table)) {
223           tableOps.delete(table)
224         }
225         tableCache.invalidate(table)
226       })
227     }
228 
229     // can happen sometimes with multiple threads but usually not a problem
230     protected def onTableExists(table: String): Unit = {}
231     // can happen sometimes with multiple threads but usually not a problem
232     protected def onNamespaceExists(namespace: String): Unit = {}
233 
234     /**
235      * ZK path for acquiring a table lock
236      *
237      * @param table table name
238      * @return
239      */
240     private def tablePath(table: String): String =
241       s"/org.locationtech.geomesa/table-locks/${StringSerialization.alphaNumericSafeString(table)}"
242 
243     /**
244      * ZK path for acquiring a namespace lock
245      *
246      * @param namespace namespace
247      * @return
248      */
249     private def nsPath(namespace: String): String =
250       s"/org.locationtech.geomesa/ns-locks/${StringSerialization.alphaNumericSafeString(namespace)}"
251   }
