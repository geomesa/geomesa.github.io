68   // converts a range to a printable string - only includes the row
69   private def rangeToString(r: org.apache.accumulo.core.data.Range): String = {
70     val a = if (r.isStartKeyInclusive) "[" else "("
71     val z = if (r.isEndKeyInclusive) "]" else ")"
72     val start = if (r.isInfiniteStartKey) "-inf" else keyToString(r.getStartKey)
73     val stop = if (r.isInfiniteStopKey) "+inf" else keyToString(r.getEndKey)
74     s"$a$start::$stop$z"
75   }

96   case class BatchScanPlan(
97       strategy: QueryStrategy,
98       tables: Seq[String],
99       ranges: Seq[org.apache.accumulo.core.data.Range],
100       iterators: Seq[IteratorSetting],
101       columnFamily: Option[Text],
102       resultsToFeatures: ResultsToFeatures[Entry[Key, Value]],
103       reducer: Option[FeatureReducer],
104       sort: Option[Seq[(String, Boolean)]],
105       maxFeatures: Option[Int],
106       projection: Option[QueryReferenceSystems],
107       numThreads: Int
108     ) extends AccumuloQueryPlan {
109 
110     override def scan(ds: AccumuloDataStore): CloseableIterator[Entry[Key, Value]] = {
111       // query guard hook - also handles full table scan checks
112       strategy.runGuards(ds)
113       // convert the relative timeout to an absolute timeout up front
114       val timeout = ds.config.queries.timeout.map(Timeout.apply)
115       // note: calculate authorizations up front so that multi-threading doesn't mess up auth providers
116       scan(ds.connector, ds.auths, ds.config.queries.parallelPartitionScans, timeout)
117     }
118 
119     /**
120      * Scan with pre-computed auths
121      *
122      * @param connector connector
123      * @param auths auths
124      * @param partitionParallelScans parallelize scans across multiple partitions (vs execute them serially)
125      * @param timeout absolute stop time, as sys time
126      * @return
127      */
128     private[accumulo] def scan(
129         connector: AccumuloClient,
130         auths: Authorizations,
131         partitionParallelScans: Boolean,
132         timeout: Option[Timeout]): CloseableIterator[Entry[Key, Value]] = {
133       if (partitionParallelScans) {
134         // kick off all the scans at once
135         tables.map(scanner(connector, _, auths, timeout)).foldLeft(CloseableIterator.empty[Entry[Key, Value]])(_ concat _)
136       } else {
137         // kick off the scans sequentially as they finish
138         SelfClosingIterator(tables.iterator).flatMap(scanner(connector, _, auths, timeout))
139       }
140     }
141 
142     /**
143      *
144      * @param connector connector
145      * @param table table
146      * @param auths auths
147      * @param timeout absolute stop time, as sys time
148      * @return
149      */
150     private def scanner(
151         connector: AccumuloClient,
152         table: String,
153         auths: Authorizations,
154         timeout: Option[Timeout]): CloseableIterator[Entry[Key, Value]] = {
155       val scanner = connector.createBatchScanner(table, auths, numThreads)
156       scanner.setRanges(ranges.asJava)
157       configure(scanner)
158       timeout match {
159         case None => new ScanIterator(scanner)
160         case Some(t) => new ManagedScan(new AccumuloScanner(scanner), t, this)
161       }
162     }
163   }

165   // join on multiple tables - requires multiple scans
166   case class JoinPlan(
167       strategy: QueryStrategy,
168       tables: Seq[String],
169       ranges: Seq[org.apache.accumulo.core.data.Range],
170       iterators: Seq[IteratorSetting],
171       columnFamily: Option[Text],
172       numThreads: Int,
173       joinFunction: JoinFunction,
174       joinQuery: BatchScanPlan
175     ) extends AccumuloQueryPlan {
176 
177     override val join: Some[(JoinFunction, BatchScanPlan)] = Some((joinFunction, joinQuery))
178     override def resultsToFeatures: ResultsToFeatures[Entry[Key, Value]] = joinQuery.resultsToFeatures
179     override def reducer: Option[FeatureReducer] = joinQuery.reducer
180     override def sort: Option[Seq[(String, Boolean)]] = joinQuery.sort
181     override def maxFeatures: Option[Int] = joinQuery.maxFeatures
182     override def projection: Option[QueryReferenceSystems] = joinQuery.projection
183 
184     override def scan(ds: AccumuloDataStore): CloseableIterator[Entry[Key, Value]] = {
185       // query guard hook - also handles full table scan checks
186       strategy.runGuards(ds)
187       // convert the relative timeout to an absolute timeout up front
188       val timeout = ds.config.queries.timeout.map(Timeout.apply)
189       // calculate authorizations up front so that multi-threading doesn't mess up auth providers
190       val auths = ds.auths
191       val joinTables = joinQuery.tables.iterator
192       if (ds.config.queries.parallelPartitionScans) {
193         // kick off all the scans at once
194         tables.map(scanner(ds.connector, _, joinTables.next, auths, partitionParallelScans = true, timeout))
195             .foldLeft(CloseableIterator.empty[Entry[Key, Value]])(_ concat _)
196       } else {
197         // kick off the scans sequentially as they finish
198         SelfClosingIterator(tables.iterator)
199             .flatMap(scanner(ds.connector, _, joinTables.next, auths, partitionParallelScans = false, timeout))
200       }
201     }
202 
203     private def scanner(
204         connector: AccumuloClient,
205         table: String,
206         joinTable: String,
207         auths: Authorizations,
208         partitionParallelScans: Boolean,
209         timeout: Option[Timeout]): CloseableIterator[Entry[Key, Value]] = {
210       val primary = if (ranges.lengthCompare(1) == 0) {
211         val scanner = connector.createScanner(table, auths)
212         scanner.setRange(ranges.head)
213         scanner
214       } else {
215         val scanner = connector.createBatchScanner(table, auths, numThreads)
216         scanner.setRanges(ranges.asJava)
217         scanner
218       }
219       configure(primary)
220       val join = joinQuery.copy(tables = Seq(joinTable))
221       new BatchMultiScanner(connector, primary, join, joinFunction, auths, partitionParallelScans, timeout)
222     }
223   }
