64   // converts a range to a printable string - only includes the row
65   private def rangeToString(r: org.apache.accumulo.core.data.Range): String = {
66     val a = if (r.isStartKeyInclusive) "[" else "("
67     val z = if (r.isEndKeyInclusive) "]" else ")"
68     val start = if (r.isInfiniteStartKey) "-inf" else keyToString(r.getStartKey)
69     val stop = if (r.isInfiniteStopKey) "+inf" else keyToString(r.getEndKey)
70     s"$a$start::$stop$z"
71   }

94   abstract class AbstractBatchScanPlan(
95       tables: Seq[String],
96       ranges: Seq[org.apache.accumulo.core.data.Range],
97       iterators: Seq[IteratorSetting],
98       columnFamily: Option[Text],
99       numThreads: Int
100     ) extends AccumuloQueryPlan {
101 
102     /**
103      * Scan with pre-computed auths
104      *
105      * @param helper scan helper
106      * @return
107      */
108     private[AccumuloQueryPlan] def scan(helper: ScanHelper): CloseableIterator[Entry[Key, Value]] = {
109       if (helper.parallel) {
110         // kick off all the scans at once
111         tables.map(scanner(_, helper)).foldLeft(CloseableIterator.empty[Entry[Key, Value]])(_ concat _)
112       } else {
113         // kick off the scans sequentially as they finish
114         CloseableIterator(tables.iterator).flatMap(scanner(_, helper))
115       }
116     }
117 
118     /**
119      * Scan a table
120      *
121      * @param table table
122      * @param helper scan helper
123      * @return
124      */
125     private def scanner(table: String, helper: ScanHelper): CloseableIterator[Entry[Key, Value]] = {
126       val scanner = helper.client.createBatchScanner(table, helper.auths, numThreads)
127       scanner.setRanges(ranges.asJava)
128       iterators.foreach(scanner.addScanIterator)
129       columnFamily.foreach(scanner.fetchColumnFamily)
130       helper.consistency.foreach(scanner.setConsistencyLevel)
131       helper.timeout match {
132         case None => new ScanIterator(scanner)
133         case Some(t) => new ManagedScan(new AccumuloScanner(scanner), t, this)
134       }
135     }
136   }

187   case class JoinPlan(
188       ds: AccumuloDataStore,
189       strategy: QueryStrategy,
190       tables: Seq[String],
191       ranges: Seq[org.apache.accumulo.core.data.Range],
192       iterators: Seq[IteratorSetting],
193       columnFamily: Option[Text],
194       numThreads: Int,
195       joinFunction: JoinFunction,
196       joinQuery: BatchScanPlan,
197       processor: LocalProcessor,
198       projection: Option[QueryReferenceSystems],
199     ) extends AccumuloQueryPlan with LocalProcessorPlan {
200 
201     override val join: Some[(JoinFunction, BatchScanPlan)] = Some((joinFunction, joinQuery))
202 
203     override def scan(): CloseableIterator[SimpleFeature] = {
204       // query guard hook - also handles full table scan checks
205       ds.interceptors.runGuards(strategy)
206       // calculate auths and convert the relative timeout to an absolute timeout up front
207       val helper = ScanHelper(ds)
208       val joinTables = joinQuery.tables.iterator
209       val entries =
210         if (helper.parallel) {
211           // kick off all the scans at once
212           tables.map(scanner( _, joinTables.next, helper)).foldLeft(CloseableIterator.empty[Entry[Key, Value]])(_ concat _)
213         } else {
214           // kick off the scans sequentially as they finish
215           CloseableIterator(tables.iterator).flatMap(scanner(_, joinTables.next, helper))
216         }
217       processor(entries.map(joinQuery.resultsToFeatures.apply))
218     }
219 
220     private def scanner(table: String, joinTable: String, helper: ScanHelper): CloseableIterator[Entry[Key, Value]] = {
221       val primary = if (ranges.lengthCompare(1) == 0) {
222         val scanner = helper.client.createScanner(table, helper.auths)
223         scanner.setRange(ranges.head)
224         scanner
225       } else {
226         val scanner = helper.client.createBatchScanner(table, helper.auths, numThreads)
227         scanner.setRanges(ranges.asJava)
228         scanner
229       }
230       iterators.foreach(primary.addScanIterator)
231       columnFamily.foreach(primary.fetchColumnFamily)
232       helper.consistency.foreach(primary.setConsistencyLevel)
233       val join: Seq[Entry[Key, Value]] => CloseableIterator[Entry[Key, Value]] =
234         entries => joinQuery.copy(tables = Seq(joinTable), ranges = entries.map(joinFunction)).scan(helper)
235       new BatchMultiScanner(primary, join)
236     }
237   }

242   private object ScanHelper {
243     def apply(ds: AccumuloDataStore): ScanHelper = {
244       // convert the relative timeout to an absolute timeout up front
245       val timeout = ds.config.queries.timeout.map(Timeout.apply)
246       // calculate authorizations up front so that multi-threading doesn't mess up auth providers
247       ScanHelper(ds.client, ds.auths, timeout, ds.config.queries.parallelPartitionScans, ds.config.queries.consistency)
248     }
249   }
