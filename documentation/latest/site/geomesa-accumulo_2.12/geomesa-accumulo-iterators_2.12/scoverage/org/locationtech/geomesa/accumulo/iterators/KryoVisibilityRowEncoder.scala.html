63   /**
64     * Version 3 encoding. @see `org.locationtech.geomesa.features.kryo.KryoFeatureSerializer` for details on
65     * the serialization scheme
66     *
67     * @param keys keys for this row
68     * @param values values for this row
69     * @return
70     */
71   private def encodeV3(keys: java.util.List[Key], values: java.util.List[Value]): Value = {
72     // Calculate length of serialised attributes, excluding attribute values themselves
73     var length = 1 +            // version
74       2 +                       // attribute count
75       1 +                       // offset size
76       count * 4 +               // attribute offsets (will use 4 bytes each) <--- offsets relative to first byte here
77       1 * 4 +                   // user data offset (will use 4 bytes)
78       IntBitSet.size(count) * 4 // null bit set, written in units of ints
79 
80     // Tracks our output position for copying attribute value bytes
81     // Begins immediately after the above
82     var valueCursor = length
83 
84     var i = 0
85     while (i < keys.size) {
86       val bytes = values.get(i).get
87       val input = KryoFeatureDeserialization.getInput(bytes, 1, bytes.length - 1) // skip the version byte
88       val metadata = Metadata(input) // read count, size, etc
89       // Column qualifiers tell us which attributes are intended to be populated given visibilities
90       val key = keys.get(i)
91       key.getColumnQualifier.getBytes.foreach { unsigned =>
92         val index = java.lang.Byte.toUnsignedInt(unsigned)
93         if(!metadata.nulls.contains(index)) {
94           val pos = metadata.setPosition(index)
95           val len = metadata.setPosition(index + 1) - pos
96           attributes(index) = (bytes, 4 + pos, len) // pos is relative to first byte of attribute offsets, see length calc
97           attributeVis(index) = key.getColumnVisibility.toString
98           length += len
99         }
100       }
101       i += 1
102     }
103 
104     // generate the serialized vis separately, as we can't reliably calculate it's size up front
105     val vis = KryoFeatureSerialization.getOutput(null)
106     KryoUserDataSerialization.serialize(vis, Collections.singletonMap(FEATURE_VISIBILITY, attributeVis.mkString(",")))
107 
108     length += vis.position()
109     val value = Array.ofDim[Byte](length)
110     val output = new Output(value)
111     output.writeByte(KryoFeatureSerializer.Version3)
112     output.writeShort(count)
113     output.write(4) // size of each offset, we always use 4 (vs 2) so we are sure to have enough space
114 
115     val nulls = IntBitSet(count)
116 
117     i = 0
118     while (i < count) {
119       // Write attribute offset relative to (version + count + size) = 4 bytes
120       output.writeInt(valueCursor - 4)
121       val attribute = attributes(i)
122       if (attribute == null /* Attribute value not provided by any values */ ||
123         attribute._3 == 0 /* Attribute value is provided but is null (size==0) */ ) {
124         // Attribute value must be null, no data to copy in
125         nulls.add(i)
126       } else {
127         // Attribute has a value so copy it in
128         val (bytes, offset, len) = attribute
129         System.arraycopy(bytes, offset, value, valueCursor, len)
130         valueCursor += len
131       }
132       // reset for next time through with new keys/values
133       attributes(i) = null
134       attributeVis(i) = ""
135       i += 1
136     }
137     output.writeInt(valueCursor - 4) // user-data offset. Note no user data has actually been copied in.
138 
139     // write nulls - we should already be in the right position
140     nulls.serialize(output)
141 
142     // copy in user data
143     System.arraycopy(vis.getBuffer, 0, value, valueCursor, vis.position())
144 
145     new Value(value)
146   }

148   /**
149     * Version 2 encoding. @see `org.locationtech.geomesa.features.kryo.impl.KryoFeatureDeserializationV2` for
150     * details on the serialization scheme
151     *
152     * @param keys keys for this row
153     * @param values values for this row
154     * @return
155     */
156   private def encodeV2(keys: java.util.List[Key], values: java.util.List[Value]): Value = {
157     setupV2()
158     val allValues = Array.ofDim[Array[Byte]](sft.getAttributeCount)
159     var i = 0
160     while (i < keys.size) {
161       val cq = keys.get(i).getColumnQualifier
162       val comma = cq.find(",")
163       // convert unsigned bytes to int
164       val indices = if (comma == -1) { cq.getBytes.map(java.lang.Byte.toUnsignedInt) } else { cq.getBytes.drop(comma + 1).map(java.lang.Byte.toUnsignedInt) }
165 
166       val bytes = values.get(i).get
167 
168       val input = KryoFeatureDeserialization.getInput(bytes, 0, bytes.length)
169       // reset our offsets
170       input.setPosition(1) // skip version
171       offsetStart = input.readInt()
172       input.setPosition(offsetStart) // set to offsets start
173       var j = 0
174       while (j < offsetsV2.length) {
175         offsetsV2(j) = if (input.position < input.limit) { input.readInt(true) } else { -1 }
176         j += 1
177       }
178 
179       // set the non-null values
180       indices.foreach { index =>
181         val endIndex = offsetsV2.indexWhere(_ != -1, index + 1)
182         val end = if (endIndex == -1) offsetStart else offsetsV2(endIndex)
183         val length = end - offsetsV2(index)
184         val values = Array.ofDim[Byte](length)
185         System.arraycopy(bytes, offsetsV2(index), values, 0, length)
186         allValues(index) = values
187       }
188 
189       i += 1
190     }
191 
192     i = 0
193     while (i < allValues.length) {
194       if (allValues(i) == null) {
195         allValues(i) = nullBytesV2(i)
196       }
197       i += 1
198     }
199 
200     KryoVisibilityRowEncoder.encodeV2(allValues, output, offsetsV2)
201   }
202 
203   private def setupV2(): Unit = {
204     if (nullBytesV2 == null) {
205       if (offsetsV2 == null || offsetsV2.length != sft.getAttributeCount) {
206         offsetsV2 = Array.ofDim[Int](sft.getAttributeCount)
207       }
208       nullBytesV2 = Array.tabulate(sft.getAttributeCount) { i =>
209         val descriptor = sft.getDescriptor(i)
210         val bindings = ObjectType.selectType(descriptor)
211         output.clear()
212         bindings.head match {
213           case ObjectType.STRING if bindings.last != ObjectType.JSON => output.writeString(null) // write string supports nulls
214           case ObjectType.LIST | ObjectType.MAP | ObjectType.BYTES   => output.writeInt(-1, true)
215           case _ => output.write(KryoFeatureSerializer.NullByte)
216         }
217         output.toBytes
218       }
219     }
220   }

249   private def encodeV2(values: Array[Array[Byte]], output: Output, offsets: Array[Int]): Value = {
250     output.clear()
251     output.writeInt(KryoFeatureSerializer.Version2, true)
252     output.setPosition(5) // leave 4 bytes to write the offsets
253     // note: we don't write ID - tables are assumed to be using serialization without IDs
254     // write attributes and keep track off offset into byte array
255     var i = 0
256     while (i < values.length) {
257       offsets(i) = output.position()
258       output.write(values(i))
259       i += 1
260     }
261     // write the offsets - variable width
262     i = 0
263     val offsetStart = output.position()
264     while (i < values.length) {
265       output.writeInt(offsets(i), true)
266       i += 1
267     }
268     // got back and write the start position for the offsets
269     val total = output.position()
270     output.setPosition(1)
271     output.writeInt(offsetStart)
272     output.setPosition(total) // set back to the end so that we get all the bytes
273 
274     new Value(output.toBytes)
275   }
