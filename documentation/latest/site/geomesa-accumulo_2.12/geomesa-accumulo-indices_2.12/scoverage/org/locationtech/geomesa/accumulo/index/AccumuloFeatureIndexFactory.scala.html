22 /**
23   * Feature index factory that provides attribute join indices
24   */
25 class AccumuloFeatureIndexFactory extends DefaultFeatureIndexFactory {
26 
27   import scala.collection.JavaConverters._
28 
29   override val all: Seq[ConfiguredIndex] = Seq(JoinIndex)
30 
31   override def fromFeatureFlag(sft: SimpleFeatureType, flag: String): Seq[IndexId] = {
32     if (flag.equalsIgnoreCase(AttributeIndex.name)) {
33       sft.getAttributeDescriptors.asScala.toSeq.flatMap { d =>
34         if (Option(d.getUserData.get(AttributeOptions.OptIndex)).exists(_.toString.equalsIgnoreCase(IndexCoverage.JOIN.toString))) {
35           Some(joinIndex(sft, d))
36         } else {
37           None
38         }
39       }
40     } else {
41       // will handle "join"
42       super.fromFeatureFlag(sft, flag)
43     }
44   }
45 
46   override def fromAttributeFlag(sft: SimpleFeatureType, descriptor: AttributeDescriptor, flag: String): Seq[IndexId] = {
47     if (flag.equalsIgnoreCase(IndexCoverage.JOIN.toString)) {
48       Seq(joinIndex(sft, descriptor))
49     } else {
50       // will handle "join:<attributes>"
51       super.fromAttributeFlag(sft, descriptor, flag)
52     }
53   }
54 
55   private def joinIndex(sft: SimpleFeatureType, descriptor: AttributeDescriptor): IndexId =
56     JoinIndex.indexFor(sft, descriptor).getOrElse {
57       throw new IllegalArgumentException(
58         s"Attribute '${descriptor.getLocalName}' is configured for join indexing but it is not a supported type: " +
59           descriptor.getType.getBinding.getName)
60     }
61 
62   override def create[T, U](ds: GeoMesaDataStore[_], sft: SimpleFeatureType, index: IndexId): Option[GeoMesaFeatureIndex[T, U]] = {
63     val idx = if (index.name == JoinIndex.name) {
64       val Seq(attribute, secondary @ _*) = index.attributes
65       index.version match {
66         case 8  => Some(new JoinIndex(ds, sft, attribute, secondary, index.mode))
67         case 7  => Some(new JoinIndexV7(ds, sft, attribute, secondary, index.mode))
68         case 6  => Some(new JoinIndexV6(ds, sft, attribute, secondary, index.mode))
69         case 5  => Some(new JoinIndexV5(ds, sft, attribute, secondary, index.mode))
70         case 4  => Some(new JoinIndexV4(ds, sft, attribute, secondary, index.mode))
71         case 3  => Some(new JoinIndexV3(ds, sft, attribute, secondary.headOption, index.mode))
72         case 2  => Some(new JoinIndexV2(ds, sft, attribute, secondary.headOption, index.mode))
73         case _ => None
74       }
75     } else {
76       None
77     }
78     idx.asInstanceOf[Option[GeoMesaFeatureIndex[T, U]]]
79   }
80 }
