31   val indexSft: SimpleFeatureType = IndexValueSerializer.getIndexSft(sft)

33   override val name: String = JoinIndex.name

36   abstract override def getFilterStrategy(
37       filter: Filter,
38       transform: Option[SimpleFeatureType]): Option[FilterStrategy] = {
39     super.getFilterStrategy(filter, transform).flatMap { strategy =>
40       // verify that it's ok to return join plans, and filter them out if not
41       if (!requiresJoin(strategy.secondary, transform)) {
42         Some(strategy)
43       } else if (!JoinIndex.AllowJoinPlans.get) {
44         None
45       } else {
46         val primary = strategy.primary.getOrElse(Filter.INCLUDE)
47         val bounds = FilterHelper.extractAttributeBounds(primary, attribute, binding)
48         val joinMultiplier = 9f + bounds.values.length // 10 plus 1 per additional range being scanned
49         val multiplier = strategy.costMultiplier * joinMultiplier
50         Some(FilterStrategy(strategy.index, strategy.primary, strategy.secondary, strategy.temporal, multiplier))
51       }
52     }
53   }
54 
55   /**
56     * Does the query require a join against the record table, or can it be satisfied
57     * in a single scan
58     *
59     * @param filter non-attribute filter being evaluated, if any
60     * @param transform transform being applied, if any
61     * @return
62     */
63   private def requiresJoin(filter: Option[Filter], transform: Option[SimpleFeatureType]): Boolean =
64     !canUseIndexSchema(filter, transform) && !canUseIndexSchemaPlusKey(filter, transform)
65 
66   /**
67     * Determines if the given filter and transform can operate on index encoded values.
68     */
69   def canUseIndexSchema(filter: Option[Filter], transform: Option[SimpleFeatureType]): Boolean = {
70     // verify that transform *does* exist and only contains fields in the index sft,
71     // and that filter *does not* exist or can be fulfilled by the index sft
72     supportsTransform(transform) && supportsFilter(filter)
73   }
74 
75   /**
76     * Determines if the given filter and transform can operate on index encoded values
77     * in addition to the values actually encoded in the attribute index keys
78     */
79   def canUseIndexSchemaPlusKey(filter: Option[Filter], transform: Option[SimpleFeatureType]): Boolean = {
80     transform.exists { t =>
81       val attributes = t.getAttributeDescriptors.asScala.map(_.getLocalName)
82       attributes.forall(a => a == attribute || indexSft.indexOf(a) != -1) && supportsFilter(filter)
83     }
84   }
85 
86   def supportsTransform(transform: Option[SimpleFeatureType]): Boolean =
87     transform.exists(_.getAttributeDescriptors.asScala.map(_.getLocalName).forall(indexSft.indexOf(_) != -1))
88 
89   def supportsFilter(filter: Option[Filter]): Boolean =
90     filter.forall(FilterHelper.propertyNames(_, sft).forall(indexSft.indexOf(_) != -1))
91 }
