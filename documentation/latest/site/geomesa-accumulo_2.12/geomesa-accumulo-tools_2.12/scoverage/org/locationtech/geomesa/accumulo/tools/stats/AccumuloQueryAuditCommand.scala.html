36 class AccumuloQueryAuditCommand extends AccumuloDataStoreCommand {
37 
38   override val name = "query-audit-logs"
39   override val params = new AccumuloQueryAuditParams()
40 
41   override def execute(): Unit = withDataStore(exec)
42 
43   private def exec(ds: AccumuloDataStore): Unit = {
44     if (ds.getSchema(params.featureName) == null) {
45       throw new ParameterException(s"SimpleFeatureType '${params.featureName}' does not exist in the data store")
46     }
47 
48     val cql = Option(params.cqlFilter).filter(_ != Filter.INCLUDE)
49     val dateRanges = {
50       import FilterHelper.ff
51       val startFilter = Option(params.start).map(d => ff.greaterOrEqual(ff.property("date"), ff.literal(d)))
52       val endFilter = Option(params.end).map(d => ff.less(ff.property("date"), ff.literal(d)))
53       val dateFilter = andOption(cql.toSeq ++ startFilter ++ endFilter)
54       val intervals = dateFilter.map { f =>
55         val intervals = FilterHelper.extractIntervals(f, "date")
56         if (intervals.disjoint) {
57           throw new ParameterException(s"Invalid disjoint date filter: ${ECQL.toCQL(f)}")
58         }
59         intervals.values
60       }
61       val bounds = intervals.toSeq.flatten
62       if (bounds.isEmpty) {
63         Seq((Instant.EPOCH.atZone(ZoneOffset.UTC), Instant.now().atZone(ZoneOffset.UTC)))
64       } else {
65         val now = Instant.now().atZone(ZoneOffset.UTC)
66         bounds.map { bound =>
67           val start = bound.lower.value.map(d => if (bound.lower.inclusive) { d } else { d.plusNanos(1000000) })
68           val end = bound.upper.value.map(d => if (bound.upper.exclusive) { d } else { d.plusNanos(1000000) })
69           (start.getOrElse(Instant.EPOCH.atZone(ZoneOffset.UTC)), end.getOrElse(now))
70         }
71       }
72     }
73 
74     WithClose(new AccumuloAuditReader(ds)) { reader=>
75       val writer = if("json".equalsIgnoreCase(params.outputFormat)) { new JsonWriter() } else { new CsvWriter() }
76       writer.header().foreach(Command.output.info)
77       dateRanges.foreach { case (start, end) =>
78         WithClose(reader.getQueryEvents(params.featureName, (start, end))) { events =>
79           val out = cql match {
80             case None => events
81             case Some(f) => events.filter(e => f.evaluate(AccumuloQueryAuditCommand.toFeature(e)))
82           }
83           out.foreach { event =>
84             Command.output.info(writer.output(event))
85           }
86         }
87       }
88     }
89   }
90 }

94   private final val CqlDescription =
95     "CQL predicate to filter log entries. Schema is: " +

98   // note: the cql description needs to be a final string constant to work with annotations, so we extract the spec
99   // from there instead of building it up as we might normally do
100   private val Spec = CqlDescription.substring(CqlDescription.indexOf(": ") + 2)
101   private val Sft = SimpleFeatureTypes.createType("audit-logs", Spec)
102 
103   /**
104    * Convert an audit event to a simple feature
105    *
106    * @param event event
107    * @return
108    */
109   private def toFeature(event: QueryEvent): SimpleFeature = {
110     val date = new Date(event.date)
111     val planTime = Long.box(event.planTime)
112     val scanTime = Long.box(event.scanTime)
113     val hits = Long.box(event.hits)
114     new ScalaSimpleFeature(Sft, "", Array(date, event.user, event.filter, event.hints, planTime, scanTime, hits))
115   }

150   private trait OutputWriter {
151     def header(): Option[String]
152     def output(event: QueryEvent): String
153   }
154 
155   private class CsvWriter extends OutputWriter {
156     private val out = new java.lang.StringBuilder()
157     // note: don't need to close the printer, as it just closes the underlying stream (i.e. our StringBuilder)
158     private val printer = new CSVPrinter(out, CSVFormat.DEFAULT)
159 
160     override def header(): Option[String] = Some(Spec.replaceAll(":\\w+", ""))
161     override def output(event: QueryEvent): String = {
162       printer.println() // reset new record
163       out.setLength(0)
164       printer.print(DateParsing.formatMillis(event.date))
165       printer.print(event.user)
166       printer.print(event.filter)
167       printer.print(event.hints)
168       printer.print(Long.box(event.planTime))
169       printer.print(Long.box(event.scanTime))
170       printer.print(Long.box(event.hits))
171       // note: we don't want to end the record as that inserts a newline, which our logger output is already doing
172       out.toString
173     }
174   }
175 
176   private class JsonWriter extends OutputWriter {
177     private val gson = new GsonBuilder().disableHtmlEscaping().serializeNulls().create()
178     override def header(): Option[String] = None
179     override def output(event: QueryEvent): String = {
180       val model = java.util.Map.of(
181         "date", DateParsing.formatMillis(event.date),
182         "user", event.user,
183         "filter", event.filter,
184         "hints", event.hints,
185         "planTimeMillis", event.planTime,
186         "scanTimeMillis", event.scanTime,
187         "hits", event.hits
188       )
189       gson.toJson(model)
190     }
191   }
