93     lazy private val indices = {
94       val all = from.ds.manager.indices(sft)
95       if (params.indices == null || params.indices.isEmpty) { all } else {
96         val builder = Seq.newBuilder[GeoMesaFeatureIndex[_, _]]
97         params.indices.asScala.foreach { ident =>
98           val filtered = all.filter(_.identifier.contains(ident))
99           if (filtered.isEmpty) {
100             throw new ParameterException(
101               s"Index '$ident' does not exist in the schema. Available indices: ${all.map(_.identifier).mkString(", ")}")
102           }
103           logger.debug(s"Mapped identifier $ident to ${filtered.map(_.identifier).mkString(", ")}")
104           builder ++= filtered
105         }
106         builder.result.distinct
107       }
108     }

110     // these get passed into our index method calls - for partitioned schemas, it must be a Seq[Some[_]],
111     // while for non-partitioned schemas it must always be Seq(None)
112     lazy private val partitions: Seq[Option[String]] = {
113       if (sft.isPartitioned) {
114         val builder = ListBuffer.empty[String]
115         if (params.partitions != null && !params.partitions.isEmpty) {
116           builder ++= params.partitions.asScala
117         }
118         if (params.partitionValues != null && !params.partitionValues.isEmpty) {
119           val partitioning = TablePartition(from.ds, sft).get
120           val sf = new ScalaSimpleFeature(sft, "")
121           builder ++= params.partitionValues.asScala.map { value =>
122             sf.setAttribute(sft.getDtgIndex.get, value)
123             val partition = partitioning.partition(sf)
124             logger.debug(s"Generated partition $partition from value $value")
125             partition
126           }
127         }
128         if (builder.isEmpty) {
129           logger.debug("No partitions specified - loading all partitions from store")
130           indices.foreach { index =>
131             builder ++= index.getPartitions
132           }
133         }
134         builder.result.distinct.sorted.map(Option.apply)
135       } else {
136         if ((params.partitions != null && !params.partitions.isEmpty) ||
137           (params.partitionValues != null && !params.partitionValues.isEmpty)) {
138           throw new ParameterException("--partition and/or --partition-value are not applicable for a non-partitioned schema")
139         }
140         Seq(None)
141       }
142     }

145     override def run(): Unit = {
146       // validate our params/setup
147       if (exportPath.toUri.getScheme == "file") {
148         throw new RuntimeException("Could not read defaultFS - this may be caused by a missing Hadoop core-site.xml file")
149       }
150       if (sft == null) {
151         throw new ParameterException(s"Schema '${params.featureName}' does not exist in the source store")
152       } else {
153         val toSft = to.ds.getSchema(params.featureName)
154         if (toSft == null) {
155           throw new ParameterException(s"Schema '${params.featureName}' does not exist in the destination store")
156         } else if (SimpleFeatureTypes.compare(sft, toSft) != 0) {
157           throw new ParameterException(s"Schema '${params.featureName}' is not the same in the source and destination store")
158         } else if (SimpleFeatureTypes.compareIndexConfigs(sft, toSft) != 0) {
159           throw new ParameterException(
160             s"Schema '${params.featureName}' does not have compatible indices in the source and destination store")
161         }
162       }
163 
164       // now execute the copy
165       CachedThreadPool.executor(params.tableThreads) { executor =>
166         partitions.foreach { partition =>
167           indices.map { fromIndex =>
168             val toIndex = to.ds.manager.index(sft, fromIndex.identifier)
169             val partitionLogId = s"${partition.fold(s"index")(p => s"partition $p")} ${fromIndex.identifier}"
170             val runnable: Runnable = () => {
171               try {
172                 Command.user.info(s"Copying $partitionLogId")
173                 copy(fromIndex, toIndex, partition, partitionLogId)
174                 Command.user.info(s"Bulk copy complete for $partitionLogId")
175               } catch {
176                 // catch Throwable so NoClassDefFound still gets logged
177                 case e: Throwable =>
178                   Command.user.error(s"Error copying $partitionLogId: ${e.getMessage}")
179                   logger.error(s"Error copying $partitionLogId", e)
180               }
181             }
182             executor.submit(runnable)
183           }
184         }
185       }
186     }

188     /**
189      * Copy a single index + partition
190      *
191      * @param fromIndex from index
192      * @param toIndex to index
193      * @param partition partition name - must be Some if schema is partitioned
194      * @param partitionLogId identifier for log messages
195      */
196     private def copy(
197         fromIndex: GeoMesaFeatureIndex[_, _],
198         toIndex: GeoMesaFeatureIndex[_, _],
199         partition: Option[String],
200         partitionLogId: String): Unit = {
201       require(sft.isPartitioned == partition.isDefined) // sanity check - this should always be true due to our setup
202 
203       val fromTable = try { fromIndex.getTableName(partition) } catch {
204         case NonFatal(e) => throw new RuntimeException("Could not get source table", e)
205       }
206 
207       val completeMarker = new Path(exportPath, s"$fromTable.complete")
208       if (exportFs.exists(completeMarker)) {
209         if (params.resume) {
210           logger.debug("Skipping already completed copy")
211           return
212         } else {
213           exportFs.delete(completeMarker, false)
214         }
215       }
216 
217       val tableExportPath = new Path(exportPath, fromTable)
218       val distcpPath = new Path(tableExportPath, "distcp.txt")
219       val copyToDir = new Path(tableExportPath, "files")
220       val cloneTable = s"${fromTable}_bc_tmp"
221 
222       logger.debug(s"Source table $fromTable (${from.tableOps.tableIdMap().get(fromTable)})")
223       logger.debug(s"Export path $tableExportPath")
224 
225       if (params.resume && from.tableOps.exists(cloneTable)) {
226         logger.debug(s"Using existing cloned table $cloneTable - ensuring table is offline")
227         from.tableOps.offline(cloneTable, true)
228       } else {
229         // clone the table as we have to take it offline in order to export it
230         // note that cloning is just a metadata op as it shares the underlying data files (until they change)
231         logger.debug(s"Checking for existence and deleting any existing cloned table $cloneTable")
232         from.ds.adapter.deleteTable(cloneTable) // no-op if table doesn't exist
233         logger.debug(s"Cloning $fromTable to $cloneTable")
234         from.tableOps.clone(fromTable, cloneTable, false, Collections.emptyMap(), Collections.emptySet()) // use 2.0 method for compatibility
235         logger.debug(s"Taking $cloneTable offline")
236         from.tableOps.offline(cloneTable, true)
237       }
238 
239       if (params.resume && exportFs.exists(distcpPath) && exportFs.getFileStatus(distcpPath).getLen > 0) {
240         logger.debug(s"Using existing export results $distcpPath")
241       } else {
242         if (exportFs.exists(tableExportPath)) {
243           logger.debug(s"Deleting existing export directory $tableExportPath")
244           exportFs.delete(tableExportPath, true)
245         }
246         logger.debug(s"Exporting table to $tableExportPath")
247         from.tableOps.exportTable(cloneTable, tableExportPath.toString)
248 
249         if (!exportFs.exists(distcpPath) || exportFs.getFileStatus(distcpPath).getLen == 0) {
250           throw new RuntimeException(s"Could not read table export results at $distcpPath")
251         }
252       }
253 
254       // ensures the destination table exists
255       logger.debug(s"Checking destination for table $fromTable")
256       to.ds.adapter.createTable(toIndex, partition, Seq.empty)
257       val toTable = try { toIndex.getTableName(partition) } catch {
258         case NonFatal(e) => throw new RuntimeException("Could not get destination table", e)
259       }
260       logger.debug(s"Destination table $toTable (${to.tableOps.tableIdMap().get(toTable)})")
261 
262       // create splits, do this separately in case the table already exists
263       val splits = new java.util.TreeSet(from.tableOps.listSplits(cloneTable))
264       val existingSplits = to.tableOps.listSplits(toTable)
265       splits.removeAll(existingSplits)
266       if (!splits.isEmpty) {
267         if (!existingSplits.isEmpty) {
268           val warning = s"Detected split mismatch between source ($fromTable) and destination ($toTable) for $partitionLogId"
269           Command.user.warn(warning)
270           logger.warn(warning)
271         }
272         logger.debug(s"Adding splits to destination table $toTable")
273         to.tableOps.addSplits(toTable, splits)
274       }
275 
276       val hadCopyError = new AtomicBoolean(false)
277       // read the distcp.txt file produced by the table export
278       // consumer: (src, dest) => Unit
279       def distCpConsumer(threads: Int)(consumer: (Path, Path) => Unit): Unit = {
280         logger.debug(s"Reading $distcpPath")
281         WithClose(IOUtils.lineIterator(exportFs.open(distcpPath), StandardCharsets.UTF_8)) { files =>
282           CachedThreadPool.executor(threads) { executor =>
283             files.asScala.foreach { file =>
284               val runnable: Runnable = () => {
285                 val path = new Path(file)
286                 val copy = new Path(copyToDir, path.getName)
287                 try {
288                   if (params.resume && exportFs.exists(copy) &&
289                       path.getFileSystem(from.conf).getFileStatus(path).getLen == exportFs.getFileStatus(copy).getLen) {
290                     logger.debug(s"Using existing copy of $path at $copy")
291                   } else {
292                     consumer(path, copy)
293                   }
294                 } catch {
295                   // catch Throwable so NoClassDefFound still gets logged
296                   case e: Throwable =>
297                     hadCopyError.set(true)
298                     Command.user.error(s"Failed to copy $path to $copy")
299                     logger.error(s"Failed to copy $path to $copy", e)
300                 }
301               }
302               executor.submit(runnable)
303             }
304           }
305         }
306       }
307 
308       if (params.distCp) {
309         var inputPath = distcpPath
310         if (params.resume) {
311           logger.debug(s"Checking copy status of files in $distcpPath")
312           inputPath = new Path(tableExportPath, "distcp-remaining.txt")
313           WithClose(exportFs.create(inputPath, true)) { out =>
314             distCpConsumer(1) { (path, _) =>
315               logger.debug(s"Adding $path to distcp")
316               out.writeUTF(s"$path\n")
317             }
318           }
319         }
320         new DistributedCopy(from.conf).copy(inputPath, copyToDir, TerminalCallback()) match {
321           case JobSuccess(message, counts) =>
322             Command.user.info(message)
323             logger.debug(s"Distributed copy counters: ${counts.mkString("\n  ", "\n  ", "")}")
324 
325           case JobFailure(message) =>
326             hadCopyError.set(true)
327             Command.user.error(message)
328             logger.error(message)
329         }
330       } else {
331         distCpConsumer(params.fileThreads) { (path, copy) =>
332           logger.debug(s"Copying $path to $copy")
333           val fs = path.getFileSystem(from.conf)
334           if (!FileUtil.copy(fs, path, exportFs, copy, false, true, to.conf)) {
335             // consolidate error handling in the catch block
336             throw new IOException(s"Failed to copy $path to $copy, copy returned false")
337           }
338         }
339       }
340       if (hadCopyError.get) {
341         throw new RuntimeException("Error copying data files")
342       }
343 
344       logger.debug(s"Loading rfiles from $copyToDir to $toTable")
345       val importDir = to.tableOps.importDirectory(copyToDir.toString).to(toTable)
346       try { importDir.ignoreEmptyDir(true) } catch {
347         case _: NoSuchMethodError => // accumulo 2.0, ignore
348       }
349       try { importDir.load() } catch {
350         case e: IllegalArgumentException => logger.trace("Error importing directory:", e) // should mean empty dir
351       }
352 
353       // create marker indicating this copy was successful
354       logger.debug(s"Creating completion marker $completeMarker")
355       exportFs.create(completeMarker).close()
356 
357       // cleanup
358       logger.debug(s"Deleting export path $tableExportPath")
359       exportFs.delete(tableExportPath, true)
360       logger.debug(s"Deleting clone table $cloneTable")
361       from.tableOps.delete(cloneTable)
362     }
