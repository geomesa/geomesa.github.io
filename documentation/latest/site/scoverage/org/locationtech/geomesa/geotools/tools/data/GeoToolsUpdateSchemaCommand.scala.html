27 class GeoToolsUpdateSchemaCommand extends DataStoreCommand[DataStore] with GeoToolsDataStoreCommand {
28 
29   import scala.collection.JavaConverters._
30 
31   override val name = "update-schema"
32   override val params = new GeoToolsUpdateSchemaParams()
33 
34   override def execute(): Unit = withDataStore(update)
35 
36   protected def update(ds: DataStore): Unit = {
37 
38     // ensure we have an operation
39     params.validate().foreach(e => throw e)
40 
41     val sft = try { ds.getSchema(params.featureName) } catch { case _: IOException => null }
42     if (sft == null) {
43       throw new ParameterException(s"Schema '${params.featureName}' does not exist in the data store")
44     }
45 
46     var n = 0
47     // numbering for our prompts
48     def number: Int = { n += 1; n }
49     val prompts = new StringBuilder()
50 
51     val builder = new SimpleFeatureTypeBuilder()
52     builder.init(sft)
53 
54     val updated = builder.buildFeatureType()
55     updated.getUserData.putAll(sft.getUserData)
56 
57     if (!params.userData.isEmpty) {
58       params.userData.asScala.foreach { ud =>
59         ud.split(":", 2) match {
60           case Array(k, v) =>
61             updated.getUserData.put(k, v) match {
62               case null => prompts.append(s"\n  $number: Adding user data: '$k=$v'")
63               case old  => prompts.append(s"\n  $number: Updating user data: '$k=$v' (was '$old')")
64             }
65           case _ => throw new ParameterException(s"Invalid user data entry - expected 'key:value': $ud")
66         }
67       }
68     }
69 
70     Command.user.info(s"Preparing to update schema '${sft.getTypeName}':$prompts")
71     if (params.force || Prompt.confirm("Continue (y/n)? ")) {
72       Command.user.info("Updating, please wait...")
73       ds match {
74         case jdbc: JDBCDataStore =>
75           // update schema is not implemented for JDBC stores
76           val partitioning =
77             Option(jdbc.dialect).collect {
78               case d: PartitionedPostgisDialect => d
79               case d: PartitionedPostgisPsDialect => d
80             }
81 
82           val dialect = partitioning.getOrElse {
83             throw new RuntimeException(
84               "JDBCDataStore does not support schema updates unless using 'dbtype=postgis-partitioned'")
85           }
86 
87           WithClose(new DefaultTransaction()) { tx =>
88             WithClose(jdbc.getConnection(tx)) { cx =>
89               dialect.postCreateTable(jdbc.getDatabaseSchema, updated, cx)
90             }
91           }
92 
93         case _ =>
94           try { ds.updateSchema(sft.getTypeName, updated) } catch {
95             case _: UnsupportedOperationException =>
96               throw new RuntimeException(s"${ds.getClass.getSimpleName} does not support schema updates")
97           }
98       }
99 
100       Command.user.info("Update complete")
101     }
102   }
103 }
