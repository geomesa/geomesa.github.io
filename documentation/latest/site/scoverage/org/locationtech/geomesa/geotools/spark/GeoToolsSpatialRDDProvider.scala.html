24 /**
25   * Spatial RDD provider for arbitrary geotools data stores. If available, prefer a GeoMesa-specific
26   * implementation (e.g. HBaseSpatialRDDProvider), as it will be better optimized.
27   *
28   * To avoid being picked up inadvertently, the data store params must include "geotools" -> "true"
29   */
30 class GeoToolsSpatialRDDProvider extends SpatialRDDProvider with LazyLogging {
31 
32   import scala.collection.JavaConverters._
33 
34   override def canProcess(params: java.util.Map[String, _ <: java.io.Serializable]): Boolean = {
35     val parameters = params.asInstanceOf[java.util.Map[String, java.io.Serializable]]
36     Option(params.get("geotools")).exists(FastConverter.convert(_, classOf[java.lang.Boolean])) &&
37         DataStoreFinder.getAllDataStores.asScala.exists(_.canProcess(parameters))
38   }
39 
40   override def rdd(
41       conf: Configuration,
42       sc: SparkContext,
43       params: Map[String, String],
44       query: Query): SpatialRDD = {
45     WithStore[DataStore](params) { ds =>
46       val (sft, features) = WithClose(ds.getFeatureReader(query, Transaction.AUTO_COMMIT)) { reader =>
47         (reader.getFeatureType, CloseableIterator(reader).toList)
48       }
49       SpatialRDD(sc.parallelize(features), sft)
50     }
51   }
52 
53   /**
54     * Writes this RDD to a GeoMesa data store. The type must exist in the data store, and all of the features
55     * in the RDD must be of this type.
56     *
57     * @param rdd features to write
58     * @param params data store params
59     * @param typeName simple feature type name
60     */
61   override def save(rdd: RDD[SimpleFeature], params: Map[String, String], typeName: String): Unit = {
62     if (GeoToolsSpatialRDDProvider.StoreCheck.toBoolean.get) {
63       WithStore[DataStore](params) { ds =>
64         require(ds != null, "Could not load data store with the provided parameters")
65         require(ds.getSchema(typeName) != null, "Schema must exist before calling save - use `DataStore.createSchema`")
66       }
67     }
68 
69     rdd.foreachPartition { iter =>
70       WithStore[DataStore](params) { ds =>
71         WithClose(ds.getFeatureWriterAppend(typeName, Transaction.AUTO_COMMIT)) { writer =>
72           iter.foreach(FeatureUtils.write(writer, _, useProvidedFid = true))
73         }
74       }
75     }
76   }
77 }
