23   @transient private lazy val spatialContext = JtsSpatialContext.GEO
24   @transient private lazy val shapeFactory   = spatialContext.getShapeFactory
25   @transient private lazy val geometryFactory = new GeometryFactory()
26   @transient private val geometricShapeFactory =
27     new ThreadLocal[GeometricShapeFactory] {
28       override def initialValue(): GeometricShapeFactory = {
29         new GeometricShapeFactory(geometryFactory)
30       }
31     }
32 
33   private def fastCircleToGeom(circle: Circle): Geometry = {
34     val gsf = geometricShapeFactory.get()
35     gsf.setSize(circle.getBoundingBox.getWidth)
36     gsf.setNumPoints(4*25) //multiple of 4 is best
37     gsf.setCentre(new Coordinate(circle.getCenter.getX, circle.getCenter.getY))
38     ST_antimeridianSafeGeom(gsf.createCircle())
39   }
40 
41   val ST_antimeridianSafeGeom: Geometry => Geometry = nullableUDF(geom => {
42     def degreesToTranslate(x: Double): Double = (((x + 180) / 360.0).floor * -360).toInt
43 
44     val geomCopy = geometryFactory.createGeometry(geom)
45     if (geomCopy.getEnvelopeInternal.getMinX < -180 || geomCopy.getEnvelopeInternal.getMaxX > 180) {
46       geomCopy.apply(new CoordinateSequenceFilter() {
47         override def filter(seq: CoordinateSequence, i: Int): Unit = {
48           seq.setOrdinate(i, CoordinateSequence.X, seq.getX(i) + degreesToTranslate(seq.getX(i)))
49         }
50         override def isDone: Boolean = false
51         override def isGeometryChanged: Boolean = true
52       })
53     }
54 
55     val datelineSafeShape = shapeFactory.makeShapeFromGeometry(geomCopy)
56     shapeFactory.getGeometryFrom(datelineSafeShape)
57   })
58 
59   val ST_BufferPoint: (Point, Double) => Geometry = (p, d) => {
60     val degrees = DistanceUtils.dist2Degrees(d/1000.0, DistanceUtils.EARTH_MEAN_RADIUS_KM)
61     fastCircleToGeom(new JtsPoint(p, spatialContext).getBuffered(degrees, spatialContext))
62   }
63 
64   val ST_MakeValid: Geometry => Geometry = nullableUDF(geom => {
65     val fix = new util.GeometryFixer(geom)
66     fix.getResult
67   })
68 
69   private[geomesa] val processingNames = Map(
70     ST_antimeridianSafeGeom -> "st_antimeridianSafeGeom",
71     ST_BufferPoint -> "st_bufferPoint",
72     ST_MakeValid -> "st_makeValid"
73   )
74 
75 
76   private[jts] def registerFunctions(sqlContext: SQLContext): Unit = {
77     sqlContext.udf.register(processingNames(ST_antimeridianSafeGeom), ST_antimeridianSafeGeom)
78     sqlContext.udf.register("st_idlSafeGeom", ST_antimeridianSafeGeom)
79     sqlContext.udf.register(processingNames(ST_BufferPoint), ST_BufferPoint)
80     sqlContext.udf.register(processingNames(ST_MakeValid), ST_MakeValid)
81   }
