41   def registerSimpleFeatureClasses(kryo: Kryo): Unit = {
42     val serializer = new com.esotericsoftware.kryo.Serializer[SimpleFeature]() {
43       val cache = new ConcurrentHashMap[Int, SimpleFeatureSerializer]()
44 
45       override def write(kryo: Kryo, out: Output, feature: SimpleFeature): Unit = {
46         val id = GeoMesaSparkKryoRegistrator.putType(feature.getFeatureType)
47         var serializer = cache.get(id)
48         if (serializer == null) {
49           serializer = new SimpleFeatureSerializer(feature.getFeatureType)
50           cache.put(id, serializer)
51         }
52         out.writeInt(id, true)
53         serializer.write(kryo, out, feature)
54       }
55 
56       override def read(kryo: Kryo, in: Input, clazz: Class[SimpleFeature]): SimpleFeature = {
57         val id = in.readInt(true)
58         var serializer = cache.get(id)
59         if (serializer == null) {
60           serializer = new SimpleFeatureSerializer(GeoMesaSparkKryoRegistrator.getType(id))
61           cache.put(id, serializer)
62         }
63         serializer.read(kryo, in, clazz)
64       }
65     }
66     kryo.setReferences(false)
67     GeoMesaSparkKryoRegistrator.SimpleFeatureImpls.foreach(kryo.register(_, serializer, kryo.getNextRegistrationId))
68   }
69 
70   def registerSedonaClasses(kryo: Kryo): Unit = {
71     val registratorClass = Try(Class.forName("org.apache.sedona.viz.core.Serde.SedonaVizKryoRegistrator")).getOrElse(
72       Class.forName("org.apache.sedona.core.serde.SedonaKryoRegistrator"))
73     logger.debug(s"found sedona kryo registrator class ${registratorClass.getCanonicalName}")
74     val sedonaRegistrator = registratorClass.newInstance().asInstanceOf[KryoRegistrator]
75     sedonaRegistrator.registerClasses(kryo)
76   }
