33 /**
34   * Takes files as inputs and runs them through a SimpleFeatureConverter
35   *
36   * Parameters:
37   *   ``geomesa.converter`` - converter definition as typesafe config string
38   *   ``geomesa.converter.inputs`` - input file paths, comma-delimited
39   *   ``geomesa.sft`` - simple feature type, as spec string, config string, or environment lookup
40   *   ``geomesa.sft.name`` - (optional) simple feature type name
41   *   ``geomesa.ingest.type`` Alternative to giving the geomesa.converter and geomesa.sft values:
42   *      This option requires that the Converter and SFT can be looked up on this provider's classpath.
43   */
44 class ConverterSpatialRDDProvider extends SpatialRDDProvider with LazyLogging {
45 
46   import ConverterSpatialRDDProvider._
47 
48   override def canProcess(params: java.util.Map[String, _ <: java.io.Serializable]): Boolean =
49     ((params.containsKey(ConverterKey) && params.containsKey(SftKey))
50       || params.containsKey(IngestTypeKey)) && params.containsKey(InputFilesKey)
51 
52   override def rdd(conf: Configuration,
53                    sc: SparkContext,
54                    params: Map[String, String],
55                    query: Query): SpatialRDD = {
56     val sft = loadSft(params, query.getTypeName)
57     val converterConf = loadConverter(sft, params)
58 
59     ConverterInputFormat.setConverterConfig(conf, converterConf)
60     ConverterInputFormat.setSft(conf, sft)
61     // note: file input format requires a job object, but conf gets copied in job object creation,
62     // so we have to copy the file paths back out
63     val job = Job.getInstance(conf)
64     FileInputFormat.setInputPaths(job, params(InputFilesKey))
65     conf.set(FileInputFormat.INPUT_DIR, job.getConfiguration.get(FileInputFormat.INPUT_DIR))
66     val queryProperties = query.getPropertyNames
67     val sftProperties = sft.getAttributeDescriptors.asScala.map{_.getLocalName}
68     if (queryProperties != null && queryProperties.nonEmpty && sftProperties != queryProperties.toSeq) {
69       logger.debug("Query transform retyping results")
70       val modifiedSft = SimpleFeatureTypeBuilder.retype(sft, query.getPropertyNames: _*)
71       ConverterInputFormat.setRetypeSft(conf, modifiedSft)
72     }
73 
74     if (query.getFilter != null && query.getFilter != Filter.INCLUDE) {
75       ConverterInputFormat.setFilter(conf, ECQL.toCQL(query.getFilter))
76     }
77 
78     val rdd = sc.newAPIHadoopRDD(conf, classOf[ConverterInputFormat], classOf[LongWritable], classOf[SimpleFeature])
79     SpatialRDD(rdd.map(_._2), sft)
80   }

86     throw new NotImplementedError("Converter provider is read-only")

88   private def loadSft(params: Map[String, String], typeName: String): SimpleFeatureType = {
89     params.get(IngestTypeKey) match {
90       case Some(sftName) =>
91         // NB: Here we assume that the SFT name and Converter name match. (And there is no option to rename the SFT.)
92         //  Further, it is assumed that they can loaded from the classpath.
93         SimpleFeatureTypeLoader.sftForName(sftName).getOrElse {
94           throw new IllegalArgumentException(s"Could not resolve Simple Feature Type by name for $sftName.")
95         }
96 
97       case None =>
98         // This is the general case where the SFT and Converter config strings are given.
99         val sftName = params.get(FeatureNameKey).orElse(Option(typeName)).orNull
100 
101         SftArgResolver.getArg(SftArgs(params(SftKey), sftName)) match {
102           case Right(s) => s
103           case Left(e) => throw new IllegalArgumentException("Could not resolve simple feature type", e)
104         }
105     }
106   }
107 
108   // TODO:  Move the logic here and in the next function to utils (aka somewhere more general)
109   //  GEOMESA-1644 Move logic to retrieve SFT/Converter config from classpath to utils
110   private def loadConverter(sft: SimpleFeatureType, params: Map[String, String]): String = {
111     params.get(IngestTypeKey) match {
112       case Some(sftName) =>
113         // NB: Here we assume that the SFT name and Converter name match.  (And there is no option to rename the SFT.)
114         //  Further, it is assumed that they can loaded from the classpath.
115         val config = ConverterConfigLoader.confs.getOrElse(sftName,
116           throw new Exception(s"Could not resolve Converter by name for $sftName."))
117 
118         // Verify the config before returning.
119         try { SimpleFeatureConverter(sft, config).close() } catch {
120           case NonFatal(e) => throw new IllegalArgumentException("Could not resolve converter", e)
121         }
122 
123         config.root().render(ConfigRenderOptions.concise())
124 
125       case None =>
126         val config = params(ConverterKey)
127         // Verify the config before returning.
128         try { SimpleFeatureConverter(sft, ConfigFactory.parseString(config)).close() } catch {
129           case NonFatal(e) => throw new IllegalArgumentException("Could not resolve converter", e)
130         }
131         config
132     }
133   }
134 }
