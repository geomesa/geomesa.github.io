94   def sparkFilterToGTFilter(expr: Expression): Option[GTFilter] = {
95     expr match {
96       case udf: ScalaUDF => scalaUDFtoGTFilter(udf)
97       case binaryComp@BinaryComparison(left, right) =>
98         val leftExpr = sparkExprToGTExpr(left)
99         val rightExpr = sparkExprToGTExpr(right)
100         if (leftExpr.isEmpty || rightExpr.isEmpty) {
101           None
102         } else {
103           binaryComp match {
104             case _: EqualTo            => Some(ff.equals(leftExpr.get, rightExpr.get))
105             case _: LessThan           => Some(ff.less(leftExpr.get, rightExpr.get))
106             case _: LessThanOrEqual    => Some(ff.lessOrEqual(leftExpr.get, rightExpr.get))
107             case _: GreaterThan        => Some(ff.greater(leftExpr.get, rightExpr.get))
108             case _: GreaterThanOrEqual => Some(ff.greaterOrEqual(leftExpr.get, rightExpr.get))
109             case _ => None
110           }
111         }
112       case unary: UnaryExpression =>
113         val sparkExpr = unary.child
114         val gtExpr = sparkExprToGTExpr(sparkExpr)
115         if (gtExpr.isEmpty)
116           None
117         else {
118           unary match {
119             case _: IsNotNull => Some(ff.not(ff.isNull(gtExpr.get)))
120             case _: IsNull => Some(ff.isNull(gtExpr.get))
121             case _ => None
122           }
123         }
124       case _ =>
125         if (haveSedona && expr.isInstanceOf[Sedona_ST_Predicate]) {
126           sedonaExprToGTFilter(expr.asInstanceOf[Sedona_ST_Predicate])
127         } else {
128           logger.debug(s"Got expr: $expr.  Don't know how to turn this into a GeoTools Expression.")
129           None
130         }
131     }
132   }

134   def sparkExprToGTExpr(expression: Expression): Option[GTExpression] = expression match {
135     case g: GeometryLiteral =>
136       Some(ff.literal(g.geom))
137 
138     case a: AttributeReference if a.name != "__fid__" =>
139       Some(ff.property(a.name))
140 
141     case c: Cast =>
142       lazy val zone = c.timeZoneId.map(ZoneId.of).orNull
143       sparkExprToGTExpr(c.child).map {
144         case lit: GTLiteral if lit.getValue.isInstanceOf[Date] && zone != null =>
145           val date = LocalDateTime.ofInstant(lit.getValue.asInstanceOf[Date].toInstant, zone)
146           ff.literal(new Date(date.atZone(ZoneOffset.UTC).toInstant.toEpochMilli))
147         case e => e
148       }
149 
150     case lit: Literal if lit.dataType == DataTypes.StringType =>
151       // the actual class is org.apache.spark.unsafe.types.UTF8String, we need to make it
152       // a normal string so that geotools can handle it
153       Some(ff.literal(Option(lit.value).map(_.toString).orNull))
154 
155     case lit: Literal if lit.dataType == DataTypes.TimestampType =>
156       // timestamps are defined as microseconds
157       Some(ff.literal(new Date(lit.value.asInstanceOf[Long] / 1000)))
158 
159     case lit: Literal if haveSedona && lit.dataType.isInstanceOf[Sedona_GeometryUDT] =>
160       Some(ff.literal(lit.dataType.asInstanceOf[Sedona_GeometryUDT].deserialize(lit.value)))
161 
162     case lit: Literal =>
163       Some(ff.literal(lit.value))
164 
165     case _ =>
166       logger.debug(s"Can't turn expression into geotools: $expression")
167       None
168   }

184     // Replace the relation in a join with a GeoMesaJoin Relation
185     private def alterJoin(join: Join): LogicalPlan = {
186       val isSpatialUDF = join.condition.exists {
187         case u: ScalaUDF if u.function.isInstanceOf[(Geometry, Geometry) => java.lang.Boolean] =>
188           u.children.head.isInstanceOf[AttributeReference] && u.children(1).isInstanceOf[AttributeReference]
189         case _ => false
190       }
191 
192       (join.left, join.right) match {
193         case (left: LogicalRelation, right: LogicalRelation) if isSpatialUDF =>
194           (left.relation, right.relation) match {
195             case (leftRel: GeoMesaRelation, rightRel: GeoMesaRelation) =>
196               leftRel.join(rightRel, join.condition.get) match {
197                 case None => join
198                 case Some(joinRelation) =>
199                   val newLogicalRelLeft = SparkVersions.copy(left)(output = left.output ++ right.output, relation = joinRelation)
200                   SparkVersions.copy(join)(left = newLogicalRelLeft)
201               }
202 
203             case _ => join
204           }
205 
206         case (leftProject @ Project(leftProjectList, left: LogicalRelation),
207             Project(rightProjectList, right: LogicalRelation)) if isSpatialUDF =>
208           (left.relation, right.relation) match {
209             case (leftRel: GeoMesaRelation, rightRel: GeoMesaRelation) =>
210               leftRel.join(rightRel, join.condition.get) match {
211                 case None => join
212                 case Some(joinRelation) =>
213                   val newLogicalRelLeft = SparkVersions.copy(left)(output = left.output ++ right.output, relation = joinRelation)
214                   val newProjectLeft = leftProject.copy(projectList = leftProjectList ++ rightProjectList, child = newLogicalRelLeft)
215                   SparkVersions.copy(join)(left = newProjectLeft)
216               }
217 
218             case _ => join
219           }
220 
221         case _ => join
222       }
223     }

225     override def apply(plan: LogicalPlan): LogicalPlan = {
226       logger.debug(s"Optimizer sees $plan")
227 
228       // NOTE: The number of arguments in Aggregate constructor is 3 on community spark and 4 on DataBricks
229       // so we cannot use pattern matching to unapply the constructor also
230       // need to use reflection to safely create new Aggregate instance
231       val optimizeAggregate: PartialFunction[LogicalPlan, LogicalPlan] =
232         Function.unlift { plan: LogicalPlan =>
233           plan match {
234             case agg: Aggregate =>
235               agg.child match {
236                 case Project(projectList, join: Join) =>
237                   val alteredJoin = SpatialOptimizationsRule.alterJoin(join)
238                   Some(Aggregates.instance(agg.groupingExpressions, agg.aggregateExpressions, Project(projectList, alteredJoin), None))
239                 case join: Join =>
240                   val alteredJoin = SpatialOptimizationsRule.alterJoin(join)
241                   Some(Aggregates.instance(agg.groupingExpressions, agg.aggregateExpressions, alteredJoin, None))
242                 case _ => None
243               }
244             case _ => None
245           }
246         }
247 
248       val optimizeRest: PartialFunction[LogicalPlan, LogicalPlan] = {
249         case join: Join =>
250           alterJoin(join)
251         case sort @ Sort(_, _, _) => sort    // No-op.  Just realizing what we can do:)
252         case filt @ Filter(f, lr: LogicalRelation) if lr.relation.isInstanceOf[GeoMesaRelation] =>
253           // TODO: deal with `or`
254 
255           val gmRel = lr.relation.asInstanceOf[GeoMesaRelation]
256           // split up conjunctive predicates and extract the st_contains variable
257           val sparkFilters: Seq[Expression] =  splitConjunctivePredicates(f)
258 
259           val (gtFilters: Seq[GTFilter], sFilters: Seq[Expression]) = sparkFilters.foldLeft((Seq[GTFilter](), Seq[Expression]())) {
260             case ((gts: Seq[GTFilter], sfilters), expression: Expression) =>
261               sparkFilterToGTFilter(expression) match {
262                 case Some(gtf) => (gts.+:(gtf), sfilters)
263                 case None      => (gts,         sfilters.+:(expression))
264               }
265           }
266 
267           if (gtFilters.nonEmpty) {
268             // if we have a partitioned cache, exclude partitions that don't match the query filter
269             val partitioned = gmRel.cached.map {
270               case c: PartitionedIndexedRDD =>
271                 val hints = sparkFilters.flatMap(extractGridId(c.envelopes, _)).flatten
272                 if (hints.isEmpty) { c } else {
273                   c.copy(rdd = c.rdd.filter { case (key, _) => hints.contains(key) })
274                 }
275 
276               case c => c
277             }
278             val filt = FilterHelper.filterListAsAnd(gmRel.filter.toSeq ++ gtFilters)
279             val relation = gmRel.copy(filter = filt, cached = partitioned)
280             val newrel = SparkVersions.copy(lr)(output = lr.output, relation = relation)
281             if (sFilters.nonEmpty) {
282               // Keep filters that couldn't be transformed at the top level
283               Filter(sFilters.reduce(And), newrel)
284             } else {
285               // if all filters could be transformed to GeoTools filters, just return the new relation
286               newrel
287             }
288           } else {
289             filt
290           }
291       }
292       plan.transform(optimizeAggregate orElse optimizeRest)
293     }
294 
295   }

304       logicalPlan.left match {
