131   def closestPoint(g1: Geometry, g2: Geometry): Point = {
132     val op = new DistanceOp(g1, g2)
133     val coord = op.nearestPoints()
134     geomFactory.createPoint(coord(0))
135   }

137   def fastDistance(c1: Coordinate, c2: Coordinate): Double = {
138     val calc = geoCalcs.get()
139     val startPoint = spatialContext.getShapeFactory.pointXY(c1.x, c1.y)
140     DistanceUtils.DEG_TO_KM * calc.distance(startPoint, c2.x, c2.y) * 1000
141   }
142 
143   def translate(g: Geometry, deltax: Double, deltay: Double): Geometry = {
144     val affineTransform = new AffineTransformation()
145     affineTransform.setToTranslation(deltax, deltay)
146     affineTransform.transform(g)
147   }
