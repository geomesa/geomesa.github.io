58   /**
59     * Attempts to do an optimized join between two relations.
60     *
61     * Currently this method uses grid partitioning on both relations so that the join comparisons
62     * only need to be applied on each pair of partitions, instead of globally. This only works
63     * if both relations have already been grid partitioned.
64     *
65     * @param other relation to join
66     * @param condition join condition
67     * @return an optimized join, if possible to do so
68     */
69   def join(other: GeoMesaRelation, condition: Expression): Option[GeoMesaJoinRelation] = {
70     val opt = for { p <- partitioned; o <- other.partitioned } yield {
71       val toJoin = if (p.envelopes == o.envelopes) {
72         Some(other)
73       } else if (p.cover) {
74         val repartitioned: SpatialRDD = p.partitions match {
75           case None => p.raw
76           case Some(partitions) => SpatialRDD(p.raw.repartition(partitions), p.raw.schema)
77         }
78         val parallelism = p.partitions.getOrElse(sqlContext.sparkContext.defaultParallelism)
79         val rdd = RelationUtils.grid(repartitioned, p.envelopes, parallelism)
80         val partitioned = Some(PartitionedRDD(rdd, p.raw, p.envelopes, p.partitions, p.cover))
81         Some(other.copy(partitioned = partitioned))
82       } else {
83         logger.warn("Joining across two relations that are not partitioned by the same scheme - unable to optimize")
84         None
85       }
86       toJoin.map { rel =>
87         GeoMesaJoinRelation(sqlContext, this, rel, StructType(schema.fields ++ rel.schema.fields), condition)
88       }
89     }
90     opt.flatten
91   }

93   override def buildScan(
94       requiredColumns: Array[String],
95       filters: Array[org.apache.spark.sql.sources.Filter]): RDD[Row] = {
96     lazy val debug =
97       s"filt = $filter, filters = ${filters.mkString(",")}, requiredColumns = ${requiredColumns.mkString(",")}"
98 
99     val filt = {
100       val sum = Seq.newBuilder[org.geotools.api.filter.Filter]
101       filter.foreach(sum += _)
102       filters.foreach(f => SparkUtils.sparkFilterToCQLFilter(f).foreach(sum += _))
103       FilterHelper.filterListAsAnd(sum.result).getOrElse(org.geotools.api.filter.Filter.INCLUDE)
104     }
105     val requiredAttributes = requiredColumns.filterNot(_ == "__fid__")
106 
107     // avoid closures on complex objects
108     val schema = this.schema // note: referencing case class members evidently serializes the whole class??
109     val typeName = sft.getTypeName
110 
111     val result: RDD[SimpleFeature] = cached match {
112       case None =>
113         logger.debug(s"Building scan, $debug")
114         val conf = new Configuration(sqlContext.sparkContext.hadoopConfiguration)
115         val query = new Query(typeName, filt, requiredAttributes: _*)
116         GeoMesaSpark(params.asJava).rdd(conf, sqlContext.sparkContext, params, query)
117 
118       case Some(IndexedRDD(rdd)) =>
119         logger.debug(s"Building in-memory scan, $debug")
120         val cql = ECQL.toCQL(filt)
121         rdd.flatMap { engine =>
122           val query = new Query(typeName, ECQL.toFilter(cql), requiredAttributes: _*)
123           SelfClosingIterator(engine.getFeatureReader(query, Transaction.AUTO_COMMIT))
124         }
125 
126       case Some(PartitionedIndexedRDD(rdd, _)) =>
127         logger.debug(s"Building partitioned in-memory scan, $debug")
128         val cql = ECQL.toCQL(filt)
129         rdd.flatMap { case (_, engine) =>
130           val query = new Query(typeName, ECQL.toFilter(cql), requiredAttributes: _*)
131           SelfClosingIterator(engine.getFeatureReader(query, Transaction.AUTO_COMMIT))
132         }
133     }
134 
135     val extractors = SparkUtils.getExtractors(requiredColumns, schema)
136 
137     result.map(SparkUtils.sf2row(schema, _, extractors))
138   }

213       val query = new Query(sft.getTypeName, ECQL.toFilter(params.getOrElse("query", "INCLUDE")))
214       GeoMesaSpark(params.asJava).rdd(new Configuration(), sqlContext.sparkContext, params, query)

257

261             val engine = new GeoCQEngineDataStore(indexGeom)

265           }

275             Iterator.single(engine)
276           }
277           rdd.persist(StorageLevel.MEMORY_ONLY)

279       }
280     }

282    val filter = Option(ECQL.toFilter(params.getOrElse("query", "INCLUDE")))
283

285   }
