59   /**
60     * Attempts to do an optimized join between two relations.
61     *
62     * Currently this method uses grid partitioning on both relations so that the join comparisons
63     * only need to be applied on each pair of partitions, instead of globally. This only works
64     * if both relations have already been grid partitioned.
65     *
66     * @param other relation to join
67     * @param condition join condition
68     * @return an optimized join, if possible to do so
69     */
70   def join(other: GeoMesaRelation, condition: Expression): Option[GeoMesaJoinRelation] = {
71     val opt = for { p <- partitioned; o <- other.partitioned } yield {
72       val toJoin = if (p.envelopes == o.envelopes) {
73         Some(other)
74       } else if (p.cover) {
75         val repartitioned: SpatialRDD = p.partitions match {
76           case None => p.raw
77           case Some(partitions) => SpatialRDD(p.raw.repartition(partitions), p.raw.schema)
78         }
79         val parallelism = p.partitions.getOrElse(sqlContext.sparkContext.defaultParallelism)
80         val rdd = RelationUtils.grid(repartitioned, p.envelopes, parallelism)
81         val partitioned = Some(PartitionedRDD(rdd, p.raw, p.envelopes, p.partitions, p.cover))
82         Some(other.copy(partitioned = partitioned))
83       } else {
84         logger.warn("Joining across two relations that are not partitioned by the same scheme - unable to optimize")
85         None
86       }
87       toJoin.map { rel =>
88         GeoMesaJoinRelation(sqlContext, this, rel, StructType(schema.fields ++ rel.schema.fields), condition)
89       }
90     }
91     opt.flatten
92   }

94   override def buildScan(
95       requiredColumns: Array[String],
96       filters: Array[org.apache.spark.sql.sources.Filter]): RDD[Row] = {
97     lazy val debug =
98       s"filt = $filter, filters = ${filters.mkString(",")}, requiredColumns = ${requiredColumns.mkString(",")}"
99 
100     val filt = {
101       val sum = Seq.newBuilder[org.geotools.api.filter.Filter]
102       filter.foreach(sum += _)
103       filters.foreach(f => SparkUtils.sparkFilterToCQLFilter(f).foreach(sum += _))
104       FilterHelper.filterListAsAnd(sum.result).getOrElse(org.geotools.api.filter.Filter.INCLUDE)
105     }
106     val requiredAttributes = requiredColumns.filterNot(_ == "__fid__")
107 
108     // avoid closures on complex objects
109     val schema = this.schema // note: referencing case class members evidently serializes the whole class??
110     val typeName = sft.getTypeName
111 
112     val result: RDD[SimpleFeature] = cached match {
113       case None =>
114         logger.debug(s"Building scan, $debug")
115         val conf = new Configuration(sqlContext.sparkContext.hadoopConfiguration)
116         val query = new Query(typeName, filt, requiredAttributes: _*)
117         GeoMesaSpark(params.asJava).rdd(conf, sqlContext.sparkContext, params, query)
118 
119       case Some(IndexedRDD(rdd)) =>
120         logger.debug(s"Building in-memory scan, $debug")
121         val cql = ECQL.toCQL(filt)
122         rdd.flatMap { engine =>
123           val query = new Query(typeName, ECQL.toFilter(cql), requiredAttributes: _*)
124           SelfClosingIterator(engine.getFeatureReader(query, Transaction.AUTO_COMMIT))
125         }
126 
127       case Some(PartitionedIndexedRDD(rdd, _)) =>
128         logger.debug(s"Building partitioned in-memory scan, $debug")
129         val cql = ECQL.toCQL(filt)
130         rdd.flatMap { case (_, engine) =>
131           val query = new Query(typeName, ECQL.toFilter(cql), requiredAttributes: _*)
132           SelfClosingIterator(engine.getFeatureReader(query, Transaction.AUTO_COMMIT))
133         }
134     }
135 
136     val extractors = SparkUtils.getExtractors(requiredColumns, schema)
137 
138     result.map(SparkUtils.sf2row(schema, _, extractors))
139   }

203     def get[T](key: String, transform: String => T, default: => T): T = {
204       params.get(key) match {
205         case None => default
206         case Some(v) =>
207           try { transform(v) } catch {
208             case NonFatal(e) => logger.error(s"Error evaluating param '$key' with value '$v':", e); default
209           }
210       }
211     }
212 
213     def rawRDD: SpatialRDD = {
214       val query = new Query(sft.getTypeName, ECQL.toFilter(params.getOrElse("query", "INCLUDE")))
215       GeoMesaSpark(params.asJava).rdd(new Configuration(), sqlContext.sparkContext, params, query)
216     }
217 
218     val partitioned = if (!get[Boolean]("spatial", _.toBoolean, false)) { None } else {
219       val raw = rawRDD
220       val bounds: Envelope = params.get("bounds") match {
221         case None => RelationUtils.getBound(raw)
222         case Some(b) =>

226       }
227

233 
234       val envelopes = params.getOrElse("strategy", "equal").toLowerCase(Locale.US) match {

241 
242       val rdd = RelationUtils.grid(raw, envelopes, parallelism)
243       rdd.persist(StorageLevel.MEMORY_ONLY)
244       Some(PartitionedRDD(rdd, raw, envelopes, partitions, get[Boolean]("cover", _.toBoolean, false)))

246 
247     val cached = if (!get[Boolean]("cache", _.toBoolean, false)) { None } else {

252 
253       // avoid closure on full sft
254       val typeName = sft.getTypeName
255       val encodedSft = SimpleFeatureTypes.encodeType(sft, includeUserData = true)
256 
257       val indexGeom = get[Boolean]("indexGeom", _.toBoolean, false)
258 
259       partitioned match {
260         case Some(p) =>
261           val rdd = p.rdd.mapValues { iter =>
262             val engine = new GeoCQEngineDataStore(indexGeom)
263             engine.createSchema(SimpleFeatureTypes.createType(typeName, encodedSft))
264             WithClose(engine.getFeatureWriterAppend(typeName, Transaction.AUTO_COMMIT)) { writer =>
265               iter.foreach(FeatureUtils.write(writer, _, useProvidedFid = true))
266             }
267             engine
268           }
269           p.rdd.unpersist() // make this call blocking?
270           rdd.persist(StorageLevel.MEMORY_ONLY)
271           Some(PartitionedIndexedRDD(rdd, p.envelopes))
272 
273         case None =>
274           val rdd = rawRDD.mapPartitions { iter =>
275             val engine = new GeoCQEngineDataStore(indexGeom)
276             engine.createSchema(SimpleFeatureTypes.createType(typeName, encodedSft))
277             WithClose(engine.getFeatureWriterAppend(typeName, Transaction.AUTO_COMMIT)) { writer =>
278               iter.foreach(FeatureUtils.write(writer, _, useProvidedFid = true))
279             }
280             Iterator.single(engine)
281           }
282           rdd.persist(StorageLevel.MEMORY_ONLY)
283           Some(IndexedRDD(rdd))
284       }
285     }
286    
287    val filter = Option(ECQL.toFilter(params.getOrElse("query", "INCLUDE")))
288 
289     GeoMesaRelation(sqlContext, sft, schema, params, filter, cached, partitioned)
290   }
