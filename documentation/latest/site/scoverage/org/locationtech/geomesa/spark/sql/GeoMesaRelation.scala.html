60   /**
61     * Attempts to do an optimized join between two relations.
62     *
63     * Currently this method uses grid partitioning on both relations so that the join comparisons
64     * only need to be applied on each pair of partitions, instead of globally. This only works
65     * if both relations have already been grid partitioned.
66     *
67     * @param other relation to join
68     * @param condition join condition
69     * @return an optimized join, if possible to do so
70     */
71   def join(other: GeoMesaRelation, condition: Expression): Option[GeoMesaJoinRelation] = {
72     val opt = for { p <- partitioned; o <- other.partitioned } yield {
73       val toJoin = if (p.envelopes == o.envelopes) {
74         Some(other)
75       } else if (p.cover) {
76         val repartitioned: SpatialRDD = p.partitions match {
77           case None => p.raw
78           case Some(partitions) => SpatialRDD(p.raw.repartition(partitions), p.raw.schema)
79         }
80         val parallelism = p.partitions.getOrElse(sqlContext.sparkContext.defaultParallelism)
81         val rdd = RelationUtils.grid(repartitioned, p.envelopes, parallelism)
82         val partitioned = Some(PartitionedRDD(rdd, p.raw, p.envelopes, p.partitions, p.cover))
83         Some(other.copy(partitioned = partitioned))
84       } else {
85         logger.warn("Joining across two relations that are not partitioned by the same scheme - unable to optimize")
86         None
87       }
88       toJoin.map { rel =>
89         GeoMesaJoinRelation(sqlContext, this, rel, StructType(schema.fields ++ rel.schema.fields), condition)
90       }
91     }
92     opt.flatten
93   }

95   override def buildScan(
96       requiredColumns: Array[String],
97       filters: Array[org.apache.spark.sql.sources.Filter]): RDD[Row] = {
98     lazy val debug =
99       s"filt = $filter, filters = ${filters.mkString(",")}, requiredColumns = ${requiredColumns.mkString(",")}"
100 
101     val filt = {
102       val sum = Seq.newBuilder[org.geotools.api.filter.Filter]
103       filter.foreach(sum += _)
104       filters.foreach(f => SparkUtils.sparkFilterToCQLFilter(f).foreach(sum += _))
105       FilterHelper.filterListAsAnd(sum.result).getOrElse(org.geotools.api.filter.Filter.INCLUDE)
106     }
107     val requiredAttributes = requiredColumns.filterNot(_ == "__fid__")
108 
109     // avoid closures on complex objects
110     val schema = this.schema // note: referencing case class members evidently serializes the whole class??
111     val typeName = sft.getTypeName
112 
113     val result: RDD[SimpleFeature] = cached match {
114       case None =>
115         logger.debug(s"Building scan, $debug")
116         val conf = new Configuration(sqlContext.sparkContext.hadoopConfiguration)
117         val query = new Query(typeName, filt, requiredAttributes: _*)
118         GeoMesaSpark(params.asJava).rdd(conf, sqlContext.sparkContext, params, query)
119 
120       case Some(IndexedRDD(rdd)) =>
121         logger.debug(s"Building in-memory scan, $debug")
122         val cql = ECQL.toCQL(filt)
123         rdd.flatMap { engine =>
124           val query = new Query(typeName, ECQL.toFilter(cql), requiredAttributes: _*)
125           // note: this isn't getting closed, but since it's cq-engine it doesn't need to be
126           CloseableIterator(engine.getFeatureReader(query, Transaction.AUTO_COMMIT))
127         }
128 
129       case Some(PartitionedIndexedRDD(rdd, _)) =>
130         logger.debug(s"Building partitioned in-memory scan, $debug")
131         val cql = ECQL.toCQL(filt)
132         rdd.flatMap { case (_, engine) =>
133           val query = new Query(typeName, ECQL.toFilter(cql), requiredAttributes: _*)
134           // note: this isn't getting closed, but since it's cq-engine it doesn't need to be
135           CloseableIterator(engine.getFeatureReader(query, Transaction.AUTO_COMMIT))
136         }
137     }
138 
139     val extractors = SparkUtils.getExtractors(requiredColumns, schema)
140 
141     result.map(SparkUtils.sf2row(schema, _, extractors))
142   }

206     def get[T](key: String, transform: String => T, default: => T): T = {
207       params.get(key) match {
208         case None => default
209         case Some(v) =>

212           }
213       }
214     }

216     def rawRDD: SpatialRDD = {
217       val query = new Query(sft.getTypeName, ECQL.toFilter(params.getOrElse("query", "INCLUDE")))
218       GeoMesaSpark(params.asJava).rdd(new Configuration(), sqlContext.sparkContext, params, query)
219     }
220 
221     val partitioned = if (!get[Boolean]("spatial", _.toBoolean, false)) { None } else {
222       val raw = rawRDD
223       val bounds: Envelope = params.get("bounds") match {
224         case None => RelationUtils.getBound(raw)

229       }
230

236 
237       val envelopes = params.getOrElse("strategy", "equal").toLowerCase(Locale.US) match {

243       }
244 
245       val rdd = RelationUtils.grid(raw, envelopes, parallelism)
246       rdd.persist(StorageLevel.MEMORY_ONLY)

249 
250     val cached = if (!get[Boolean]("cache", _.toBoolean, false)) { None } else {

255 
256       // avoid closure on full sft
257       val typeName = sft.getTypeName
258       val encodedSft = SimpleFeatureTypes.encodeType(sft, includeUserData = true)
259 
260       val indexGeom = get[Boolean]("indexGeom", _.toBoolean, false)
261 
262       partitioned match {
263         case Some(p) =>
264           val rdd = p.rdd.mapValues { iter =>
265             val engine = new GeoCQEngineDataStore(indexGeom)
266             engine.createSchema(SimpleFeatureTypes.createType(typeName, encodedSft))

268               iter.foreach(FeatureUtils.write(writer, _, useProvidedFid = true))
269             }
270             engine
271           }
272           p.rdd.unpersist() // make this call blocking?
273           rdd.persist(StorageLevel.MEMORY_ONLY)
274           Some(PartitionedIndexedRDD(rdd, p.envelopes))
275 
276         case None =>
277           val rdd = rawRDD.mapPartitions { iter =>
278             val engine = new GeoCQEngineDataStore(indexGeom)
279             engine.createSchema(SimpleFeatureTypes.createType(typeName, encodedSft))
280             WithClose(engine.getFeatureWriterAppend(typeName, Transaction.AUTO_COMMIT)) { writer =>
281               iter.foreach(FeatureUtils.write(writer, _, useProvidedFid = true))
282             }

287       }
288     }

290    val filter = Option(ECQL.toFilter(params.getOrElse("query", "INCLUDE")))
291 
292     GeoMesaRelation(sqlContext, sft, schema, params, filter, cached, partitioned)
293   }
