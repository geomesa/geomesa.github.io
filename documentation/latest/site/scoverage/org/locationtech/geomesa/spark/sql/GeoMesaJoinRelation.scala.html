20 // A special case relation that is built when a join happens across two identically partitioned relations
21 // Uses the sweepline algorithm to lower the complexity of the join
22 case class GeoMesaJoinRelation(
23     sqlContext: SQLContext,
24     leftRel: GeoMesaRelation,
25     rightRel: GeoMesaRelation,
26     schema: StructType,
27     condition: Expression,
28     filt: org.geotools.api.filter.Filter = org.geotools.api.filter.Filter.INCLUDE,
29     props: Option[Seq[String]] = None
30   ) extends BaseRelation with PrunedFilteredScan {
31 
32   import RelationUtils.CoordinateOrdering
33 
34   private val leftPartitioning = leftRel.partitioned.getOrElse {
35     throw new IllegalArgumentException("Trying to join un-partitioned relations")
36   }
37   private val rightPartitioning = rightRel.partitioned.getOrElse {
38     throw new IllegalArgumentException("Trying to join un-partitioned relations")
39   }
40 
41   private def sweeplineJoin(overlapAction: OverlapAction): RDD[(Int, (SimpleFeature, SimpleFeature))] = {
42     leftPartitioning.rdd.join(rightPartitioning.rdd).flatMap { case (key, (left, right)) =>
43       val sweeplineIndex = new SweepLineIndex()
44       left.foreach {feature =>
45         val coords = feature.getDefaultGeometry.asInstanceOf[Geometry].getCoordinates
46         sweeplineIndex.add(new SweepLineInterval(coords.min.x, coords.max.x, (0, feature)))
47       }
48       right.foreach {feature =>
49         val coords = feature.getDefaultGeometry.asInstanceOf[Geometry].getCoordinates
50         sweeplineIndex.add(new SweepLineInterval(coords.min.x, coords.max.x, (1, feature)))
51       }
52       sweeplineIndex.computeOverlaps(overlapAction)
53       overlapAction.joinList.map(key -> _)
54     }
55   }
56 
57   override def buildScan(
58       requiredColumns: Array[String],
59       filters: Array[org.apache.spark.sql.sources.Filter]): RDD[Row] = {
60 
61     val leftSchema = leftRel.schema
62     val rightSchema = rightRel.schema
63     val leftExtractors = SparkUtils.getExtractors(leftSchema.fieldNames, leftSchema)
64     val rightExtractors = SparkUtils.getExtractors(rightSchema.fieldNames, rightSchema)
65     val joinedSchema = StructType(leftSchema.fields ++ rightSchema.fields)
66     val joinedExtractors = leftExtractors ++ rightExtractors
67 
68     // Extract geometry indexes and spatial function from condition expression and relation SFTs
69     val (leftIndex, rightIndex, conditionFunction) = {
70       val scalaUdf = condition.asInstanceOf[ScalaUDF]
71       val function = scalaUdf.function.asInstanceOf[(Geometry, Geometry) => Boolean]
72       val children = scalaUdf.children.asInstanceOf[Seq[AttributeReference]]
73       // Because the predicate may not have parameters in the right order, we must check both
74       val leftAttr = children.head.name
75       val rightAttr = children(1).name
76       val leftIndex = leftRel.sft.indexOf(leftAttr)
77       if (leftIndex == -1) {
78         (leftRel.sft.indexOf(rightAttr), rightRel.sft.indexOf(leftAttr), function)
79       } else {
80         (leftIndex, rightRel.sft.indexOf(rightAttr), function)
81       }
82     }
83 
84     // Perform the sweepline join and build rows containing matching features
85     val overlapAction = new OverlapAction(leftIndex, rightIndex, conditionFunction)
86     sweeplineJoin(overlapAction).mapPartitions { iter =>
87       iter.map { case (_, (leftFeature, rightFeature)) =>
88         SparkUtils.joinedSf2row(joinedSchema, leftFeature, rightFeature, joinedExtractors)
89       }
90     }
91   }
92 
93   override def unhandledFilters(filters: Array[Filter]): Array[Filter] = {
94     filters.filter {
95       case t @ (_:IsNotNull | _:IsNull) => true
96       case _ => false
97     }
98   }
