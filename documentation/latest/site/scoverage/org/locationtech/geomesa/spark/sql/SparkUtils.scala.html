98   def sparkFilterToCQLFilter(filt: org.apache.spark.sql.sources.Filter): Option[org.geotools.api.filter.Filter] = filt match {
99     case GreaterThanOrEqual(attribute, v) => Some(ff.greaterOrEqual(ff.property(attribute), ff.literal(v)))
100     case GreaterThan(attr, v)             => Some(ff.greater(ff.property(attr), ff.literal(v)))
101     case LessThanOrEqual(attr, v)         => Some(ff.lessOrEqual(ff.property(attr), ff.literal(v)))
102     case LessThan(attr, v)                => Some(ff.less(ff.property(attr), ff.literal(v)))
103     case EqualTo(attr, v) if attr == "__fid__" => Some(ff.id(ff.featureId(v.toString)))
104     case EqualTo(attr, v)                      => Some(ff.equals(ff.property(attr), ff.literal(v)))
105     case In(attr, values) if attr == "__fid__" => Some(ff.id(values.map(v => ff.featureId(v.toString)).toSet.asJava))
106     case In(attr, values)                      =>
107       Some(values.map(v => ff.equals(ff.property(attr), ff.literal(v))).reduce[org.geotools.api.filter.Filter]( (l,r) => ff.or(l,r)))
108     case And(left, right)                 => Some(ff.and(sparkFilterToCQLFilter(left).get, sparkFilterToCQLFilter(right).get)) // TODO: can these be null
109     case Or(left, right)                  => Some(ff.or(sparkFilterToCQLFilter(left).get, sparkFilterToCQLFilter(right).get))
110     case Not(f)                           => Some(ff.not(sparkFilterToCQLFilter(f).get))
111     case StringStartsWith(a, v)           => Some(ff.like(ff.property(a), s"$v%"))
112     case StringEndsWith(a, v)             => Some(ff.like(ff.property(a), s"%$v"))
113     case StringContains(a, v)             => Some(ff.like(ff.property(a), s"%$v%"))
114     case IsNull(attr)                     => None
115     case IsNotNull(attr)                  => None
116   }

122     def basicTypeBinding(dataType: DataType): Class[_] = dataType match {
123       case DataTypes.StringType              => classOf[java.lang.String]
124       case DataTypes.DateType                => classOf[java.util.Date]
125       case DataTypes.TimestampType           => classOf[java.util.Date]
126       case DataTypes.IntegerType             => classOf[java.lang.Integer]
127       case DataTypes.LongType                => classOf[java.lang.Long]
128       case DataTypes.FloatType               => classOf[java.lang.Float]
129       case DataTypes.DoubleType              => classOf[java.lang.Double]
130       case DataTypes.BooleanType             => classOf[java.lang.Boolean]
131       case DataTypes.BinaryType              => classOf[Array[Byte]]
132       case _                                 => null
133     }
134 
135     def geomTypeBinding(dataType: DataType): Class[_] = dataType match {
136       case JTSTypes.PointTypeInstance        => classOf[org.locationtech.jts.geom.Point]
137       case JTSTypes.MultiPointTypeInstance   => classOf[org.locationtech.jts.geom.MultiPoint]
138       case JTSTypes.LineStringTypeInstance   => classOf[org.locationtech.jts.geom.LineString]
139       case JTSTypes.MultiLineStringTypeInstance => classOf[org.locationtech.jts.geom.MultiLineString]
140       case JTSTypes.PolygonTypeInstance      => classOf[org.locationtech.jts.geom.Polygon]
141       case JTSTypes.MultipolygonTypeInstance => classOf[org.locationtech.jts.geom.MultiPolygon]
142       case JTSTypes.GeometryCollectionTypeInstance => classOf[org.locationtech.jts.geom.GeometryCollection]
143       case JTSTypes.GeometryTypeInstance     => classOf[org.locationtech.jts.geom.Geometry]
144       case _                                 => null
145     }
146 
147     struct.fields.filter(_.name != "__fid__").foreach { field =>
148       field.dataType match {
149         case ArrayType(elementType, _) =>
150           val elementBinding = basicTypeBinding(elementType)
151           if (elementBinding == null) {
152             throw new IllegalArgumentException(
153               s"list element in field ${field.name} is not basic type: ${elementType.typeName}")
154           }
155           val attributeSpec = ListAttributeSpec(field.name, elementBinding, Map.empty)
156           builder.add(attributeSpec.toDescriptor)
157 
158         case MapType(keyType, valueType, _) =>
159           val keyBinding = basicTypeBinding(keyType)
160           if (keyBinding == null) {
161             throw new IllegalArgumentException(
162               s"map key in field ${field.name} is not basic type: ${keyType.typeName}")
163           }
164           val valueBinding = basicTypeBinding(valueType)
165           if (valueBinding == null) {
166             throw new IllegalArgumentException(
167               s"map value in field ${field.name} is not basic type: ${valueType.typeName}")
168           }
169           val attributeSpec = MapAttributeSpec(field.name, keyBinding, valueBinding, Map.empty)
170           builder.add(attributeSpec.toDescriptor)
171 
172         case _ =>
173           var binding = basicTypeBinding(field.dataType)
174           if (binding == null) binding = geomTypeBinding(field.dataType)
175           if (binding == null) {
176             throw new IllegalArgumentException(
177               s"Unexpected data type for field ${field.name}: ${field.dataType.typeName}")
178           }
179           builder.add(field.name, binding)
180       }
181     }
182 
183     builder.buildFeatureType()
184   }

192     def basicTypeToSQLType(bindings: Seq[ObjectType.ObjectType]): Option[DataType] = {
193       bindings.head match {

200         case ObjectType.DATE     => Some(DataTypes.TimestampType)

209         case ObjectType.POINT               => JTSTypes.PointTypeInstance
210         case ObjectType.LINESTRING          => JTSTypes.LineStringTypeInstance
211         case ObjectType.POLYGON             => JTSTypes.PolygonTypeInstance
212         case ObjectType.MULTIPOINT          => JTSTypes.MultiPointTypeInstance
213         case ObjectType.MULTILINESTRING     => JTSTypes.MultiLineStringTypeInstance
214         case ObjectType.MULTIPOLYGON        => JTSTypes.MultipolygonTypeInstance
215         case ObjectType.GEOMETRY_COLLECTION => JTSTypes.GeometryCollectionTypeInstance
216         case _                              => JTSTypes.GeometryTypeInstance
217       }
218     }

220     def listTypeToSQLType(bindings: Seq[ObjectType.ObjectType]): Option[DataType] =
221       basicTypeToSQLType(bindings.tail).map(ArrayType(_))
222 
223     def mapTypeToSQLType(bindings: Seq[ObjectType.ObjectType]): Option[DataType] = {
224       (basicTypeToSQLType(bindings.tail), basicTypeToSQLType(bindings.tail.tail)) match {
225         case (Some(keyType), Some(valueType)) => Some(MapType(keyType, valueType))
226         case _ => None
227       }
228     }

230     val bindings = ObjectType.selectType(ad)
231     val dt = bindings.head match {

241       case ObjectType.LIST     => listTypeToSQLType(bindings)
242       case ObjectType.MAP      => mapTypeToSQLType(bindings)
243       case ObjectType.GEOMETRY => Some(geomTypeToSQLType(bindings))
244       case _ => logger.warn(s"Unexpected bindings for descriptor $ad: ${bindings.mkString(", ")}"); None
245     }
246     dt.map(StructField(ad.getLocalName, _))

260   def rowsToFeatures(sft: SimpleFeatureType, schema: StructType): SimpleFeatureRowMapping = {
261     val mappings = Seq.tabulate(sft.getAttributeCount) { i =>
262       val descriptor = sft.getDescriptor(i)
263       val binding = descriptor.getType.getBinding
264       val needConversion = (classOf[java.util.List[_]].isAssignableFrom(binding) ||
265         classOf[java.util.Map[_, _]].isAssignableFrom(binding))
266       (i, schema.fieldIndex(descriptor.getLocalName), needConversion)
267     }
268     val fid: Row => String = schema.fields.indexWhere(_.name == "__fid__") match {
269       case -1 => _ => TimeSortedUuidGenerator.createUuid().toString
270       case i  => r => r.getString(i)
271     }
272     SimpleFeatureRowMapping(sft, mappings, fid)
273   }
