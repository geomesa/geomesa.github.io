40 case class Bounds(low: Double,
41                   high: Double) {
42   lazy val mid = (low+high)/2.0

45 // This object is composed of minimal pieces from org.locationtech.geomesa.utils.geohash.GeoHash
46 // to support geohash UDFs while avoiding depending on that module.
47 private[util] object GeoHash {
48 
49   val MAX_PRECISION = 63 // our bitset operations assume all bits fit in one Long
50 
51   private val latBounds = Bounds(-90.0,90.0)
52   private val lonBounds = Bounds(-180.0,180.0)
53   private lazy val latRange: Double = latBounds.high - latBounds.low
54   private lazy val lonRange: Double = lonBounds.high - lonBounds.low
55 
56   private lazy val powersOf2Map: Map[Int, Long] =
57     (0 to   MAX_PRECISION).map(i => (i, 1L << i)).toMap // 1L << i == math.pow(2,i).toLong
58   private lazy val latDeltaMap: Map[Int, Double]  =
59     (0 to MAX_PRECISION).map(i => (i, latRange / powersOf2Map(i / 2))).toMap
60   private lazy val lonDeltaMap: Map[Int, Double] =
61     (0 to MAX_PRECISION).map(i => (i, lonRange / powersOf2Map(i / 2 + i % 2))).toMap
62 
63   def latitudeDeltaForPrecision(prec: Int): Double = {
64     checkPrecision(prec)
65     latDeltaMap(prec)
66   }
67 
68   def longitudeDeltaForPrecision(prec: Int): Double = {
69     checkPrecision(prec)
70     lonDeltaMap(prec)
71   }
72 
73   private val lonMax = 360 - math.pow(0.5, 32)
74   private val latMax = 180 - math.pow(0.5, 32)
75 
76   private val bits = Array(16,8,4,2,1)
77   protected[util] val base32 = "0123456789bcdefghjkmnpqrstuvwxyz"
78   private val characterMap: Map[Char, BitSet] =
79     base32.zipWithIndex.map { case (c, i) => c -> bitSetFromBase32Character(i) }.toMap
80 
81   lazy val latLonGeoFactory = new GeometryFactory(new PrecisionModel(PrecisionModel.FLOATING), 4326)
82 
83   private def envFromIndiciesDeltas(lonIndex: Long,
84                                     latIndex: Long,
85                                     lonDelta: Double,
86                                     latDelta: Double): Envelope =
87     new Envelope(lonBounds.low+lonDelta*lonIndex, lonBounds.low+lonDelta*(lonIndex+1),
88       latBounds.low+latDelta*latIndex, latBounds.low+latDelta*(latIndex+1))
89 
90   private def interleaveReverseBits(first: Long, second: Long, numBits: Int): Long = {
91     /* We start with the first value of the interleaved long, coming from first if
92        numBits is odd or from second if numBits is even */
93     val even = (numBits & 0x01) == 0
94     val (actualFirst, actualSecond) = if(even) (second, first) else (first, second)
95     val numPairs = numBits >> 1
96     var result = 0L
97     (0 until numPairs).foreach{ pairNum =>
98       result = (result << 1) | ((actualFirst >> pairNum) & 1L)
99       result = (result << 1) | ((actualSecond >> pairNum) & 1L)
100     }
101     if (!even) (result << 1) | ((actualFirst >> numPairs) & 1L) else result
102   }
103 
104   private def extractReverseBits(value: Long, numBits: Int): (Long, Long) = {
105     val numPairs = numBits >> 1
106     var first = 0L
107     var second = 0L
108     (0 until numPairs).foreach{ pairNum =>
109       first = (first << 1) | ((value >> pairNum * 2) & 1L)
110       second = (second << 1) | ((value >> (pairNum * 2 + 1)) & 1L)
111     }
112     ((if((numBits & 0x01) == 1) ((first << 1) | ((value >> (numBits - 1)) & 1L)) else first), second)
113   }
114 
115   private def shift(n: Int, bs: BitSet): BitSet = bs.map(_ + n)
116 
117   private def bitSetFromBase32Character(charIndex: Long): BitSet =
118     BitSet(toPaddedBinaryString(charIndex, 5).zipWithIndex.filter { case (c,idx) => c == '1' }.map(_._2): _*)
119 
120   private def toPaddedBinaryString(i: Long, length: Int): String =
121     String.format("%" + length + "s", i.toBinaryString).replace(' ', '0')
122 
123   private def toBase32(bitset: BitSet, prec: Int): String = {
124     // compute the precision padded to the next 5-bit boundary
125     val numLeftoverBits = prec % 5
126     val precision = prec + (numLeftoverBits match {
127       case 0 => 0
128       case _ => (5-numLeftoverBits)
129     })
130 
131     // take the bit positions in groups of 5, and map each set to a character
132     // (based on the current bit-set); this works for off-5 precisions, because
133     // the additional bits will simply not be there (assumed to be zero)
134     (0 until precision).grouped(5).map(i=>ch(i, bitset)).mkString
135   }
136 
137   private def ch(v: IndexedSeq[Int], bitset: BitSet) =
138     base32(v.foldLeft(0)((cur,i) => cur + (if (bitset(i)) bits(i%bits.length) else 0)))
139 
140   def checkPrecision(precision: Int) =
141     require(precision <= MAX_PRECISION,
142       s"GeoHash precision of $precision requested, but precisions above $MAX_PRECISION are not supported")
143 
144   def encode(lon: Double, lat: Double, prec: Int = 25): String = {
145     require(lon >= -180.0 && lon <= 180.0)
146     require(lat >= -90.0 && lat <= 90.0)
147     checkPrecision(prec)
148 
149     val lonDelta = lonDeltaMap(prec)
150     val latDelta = latDeltaMap(prec)
151     val lonIndex = if (lon == 180.0) (lonMax / lonDelta).toLong else ((lon - lonBounds.low) / lonDelta).toLong
152     val latIndex = if (lat == 90.0) (latMax / latDelta).toLong else ((lat - latBounds.low) / latDelta).toLong
153 
154     val bitSet = IBitSet.fromBitMaskNoCopy(Array(interleaveReverseBits(lonIndex, latIndex, prec)))
155     toBase32(bitSet, prec)
156   }
157 
158   def decode(string: String, precisionOption: Option[Int] = None): Geometry = {
159     // figure out what precision we should use
160     val prec = precisionOption.getOrElse(5*string.length)
161     checkPrecision(prec)
162 
163     // compute bit-sets for both the full and partial characters
164     val bitsets = string.zipWithIndex.map { case (c: Char, i: Int) => shift(i*5, characterMap(c)) }
165 
166     // OR all of these bit-sets together
167     val finalBitset: BitSet = bitsets.size match {
168       case 0 => BitSet()
169       case 1 => bitsets(0)
170       case _ => bitsets.reduce(_|_)
171     }
172 
173     val (lonIndex, latIndex) = extractReverseBits(finalBitset.toBitMask.head, prec)
174     val (lonDelta, latDelta) = (lonDeltaMap(prec), latDeltaMap(prec))
175 
176     val envelope = envFromIndiciesDeltas(lonIndex, latIndex, lonDelta, latDelta)
177 
178     latLonGeoFactory.toGeometry(envelope)
179   }
