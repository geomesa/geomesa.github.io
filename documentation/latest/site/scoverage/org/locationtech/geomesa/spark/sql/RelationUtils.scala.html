85   def equalPartitioning(bound: Envelope, numPartitions: Int): List[Envelope] = {
86     // Compute bounds of each partition
87     val partitionsPerDim = Math.sqrt(numPartitions).toInt
88     val partitionWidth = bound.getWidth / partitionsPerDim
89     val partitionHeight = bound.getHeight / partitionsPerDim
90     val minX = bound.getMinX
91     val minY = bound.getMinY
92     val partitionEnvelopes: ListBuffer[Envelope] = ListBuffer()
93 
94     // Build partitions
95     for (xIndex <- 0 until partitionsPerDim) {
96       val xPartitionStart = minX + (xIndex * partitionWidth)
97       val xPartitionEnd = xPartitionStart + partitionWidth
98       for (yIndex <- 0 until partitionsPerDim) {
99         val yPartitionStart = minY + (yIndex * partitionHeight)
100         val yPartitionEnd = yPartitionStart+ partitionHeight
101         partitionEnvelopes += new Envelope(xPartitionStart, xPartitionEnd, yPartitionStart, yPartitionEnd)
102       }
103     }
104     partitionEnvelopes.toList
105   }

107   def weightedPartitioning(rawRDD: RDD[SimpleFeature], bound: Envelope, numPartitions: Int, sampleSize: Int): List[Envelope] = {
108     val width: Int = Math.sqrt(numPartitions).toInt
109     val sample = rawRDD.takeSample(withReplacement = false, sampleSize)
110     val binSize = sample.length / width
111     if (binSize > 0)  {
112       val xSample = sample.map{f => f.getDefaultGeometry.asInstanceOf[Geometry].getCoordinates.min.x}
113       val ySample = sample.map{f => f.getDefaultGeometry.asInstanceOf[Geometry].getCoordinates.min.y}
114       val xSorted = xSample.sorted
115       val ySorted = ySample.sorted
116       val partitionEnvelopes: ListBuffer[Envelope] = ListBuffer()
117       for (xBin <- 0 until width) {
118         val minX = xSorted(xBin * binSize)
119         val maxX = xSorted(math.min((xBin + 1) * binSize, xSorted.length - 1))
120         for (yBin <- 0 until width) {
121           val minY = ySorted(yBin * binSize)
122           val maxY = ySorted(math.min((yBin + 1) * binSize, ySorted.length - 1))
123           partitionEnvelopes += new Envelope(minX, maxX, minY, maxY)
124         }
125       }
126       partitionEnvelopes.toList
127     } else List(bound)
128   }

134   // Constructs an RTree based on a sample of the data and returns its bounds as envelopes
135   // returns one less envelope than requested to account for the catch-all envelope
136   def rtreePartitioning(
137       rawRDD: RDD[SimpleFeature],
138       numPartitions: Int,
139       sampleSize: Int,
140       thresholdMultiplier: Double): List[Envelope] = {
141     val sample = rawRDD.takeSample(withReplacement = false, sampleSize)
142     val rtree = new STRtree()
143 
144     sample.foreach{ sf =>
145       rtree.insert(sf.getDefaultGeometry.asInstanceOf[Geometry].getEnvelopeInternal, sf)
146     }
147     val envelopes: java.util.List[Envelope] = new java.util.ArrayList[Envelope]()
148 
149     // get rtree envelopes, limited to those containing reasonable size
150     val reasonableSize = sampleSize / numPartitions
151     val threshold = (reasonableSize * thresholdMultiplier).toInt
152     val minSize = reasonableSize - threshold
153     val maxSize = reasonableSize + threshold
154     rtree.build()
155     queryBoundary(rtree.getRoot, envelopes, minSize, maxSize)
156     envelopes.asScala.take(numPartitions - 1).toList
157   }

159   // Helper method to get the envelopes of an RTree
160   def queryBoundary(node: AbstractNode, boundaries: java.util.List[Envelope], minSize: Int, maxSize: Int): Int =  {
161     // get node's immediate children
162     val childBoundables: java.util.List[_] = node.getChildBoundables
163 
164     // True if current node is leaf
165     var flagLeafnode = true
166     var i = 0
167     while (i < childBoundables.size && flagLeafnode) {
168       val childBoundable = childBoundables.get(i).asInstanceOf[Boundable]
169       if (childBoundable.isInstanceOf[AbstractNode]) {
170         flagLeafnode = false
171       }
172       i += 1
173     }
174 
175     if (flagLeafnode) {
176       childBoundables.size
177     } else {
178       var nodeCount = 0
179       for ( i <- 0 until childBoundables.size ) {
180         val childBoundable = childBoundables.get(i).asInstanceOf[Boundable]
181         childBoundable match {
182           case child: AbstractNode =>
183             val childSize = queryBoundary(child, boundaries, minSize, maxSize)
184             // check boundary for size and existence in chosen boundaries
185             if (childSize < maxSize && childSize > minSize) {
186               var alreadyAdded = false
187               if (node.getLevel != 1) {
188                 child.getChildBoundables.asInstanceOf[java.util.List[AbstractNode]].asScala.foreach { c =>
189                   alreadyAdded = alreadyAdded || boundaries.contains(c.getBounds.asInstanceOf[Envelope])
190                 }
191               }
192               if (!alreadyAdded) {
193                 boundaries.add(child.getBounds.asInstanceOf[Envelope])
194               }
195             }
196             nodeCount += childSize
197 
198           case _ => nodeCount += 1 // negligible difference but accurate
199         }
200       }
201       nodeCount
202     }
203   }
