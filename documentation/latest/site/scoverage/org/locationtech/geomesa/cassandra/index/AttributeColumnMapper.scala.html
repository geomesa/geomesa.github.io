46   override def bind(value: SingleRowKeyValue[_]): Seq[AnyRef] = {

63   override def select(range: ScanRange[_], tieredKeyRanges: Seq[ByteRange]): Seq[RowSelect] = {
64     val primary = range.asInstanceOf[ScanRange[AttributeIndexKey]] match {
65       case SingleRowRange(row)   => Seq(ColumnSelect(Value, row.value, row.value, startInclusive = true, endInclusive = true))
66       case BoundedRange(lo, hi)  => Seq(ColumnSelect(Value, lo.value, hi.value, lo.inclusive, hi.inclusive))
67       case LowerBoundedRange(lo) => Seq(ColumnSelect(Value, lo.value, null, lo.inclusive, endInclusive = false))
68       case UpperBoundedRange(hi) => Seq(ColumnSelect(Value, null, hi.value, startInclusive = false, hi.inclusive))
69       case PrefixRange(prefix)   => Seq(ColumnSelect(Value, prefix.value, prefix.value + "zzzz", prefix.inclusive, endInclusive = false)) // TODO ?
70       case UnboundedRange(empty) => Seq.empty
71       case _ => throw new IllegalArgumentException(s"Unexpected range type $range")
72     }
73     val clause = if (tieredKeyRanges.isEmpty) { primary } else {
74       val minTier = ByteRange.min(tieredKeyRanges)
75       val maxTier = ByteRange.max(tieredKeyRanges)
76       primary :+ ColumnSelect(Secondary, ByteBuffer.wrap(minTier), ByteBuffer.wrap(maxTier), startInclusive = true, endInclusive = true)
77     }
78     if (clause.isEmpty) { Seq(RowSelect(clause)) } else {
79       shards.map(s => RowSelect(clause.+:(s)))
80     }
81   }
