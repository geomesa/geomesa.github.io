21 class CassandraBackedMetadata[T](val session: Session, val catalog: String, val serializer: MetadataSerializer[T])
22     extends TableBasedMetadata[T] {
23 
24   override protected def checkIfTableExists: Boolean = {
25     val m = session.getCluster.getMetadata
26     val km = m.getKeyspace(session.getLoggedKeyspace)
27     km.getTable(catalog) != null
28   }
29 
30   override protected def createTable(): Unit =
31     session.execute(s"CREATE TABLE IF NOT EXISTS $catalog (sft text, key text, value text, PRIMARY KEY ((sft), key))")
32 
33   override protected def createEmptyBackup(timestamp: String): CassandraBackedMetadata[T] = {
34     val table = {
35       val full = s"${catalog}_${timestamp}_bak"
36       if (full.lengthCompare(CassandraIndexAdapter.TableNameLimit) <= 0) { full } else {
37         IndexAdapter.truncateTableName(full, CassandraIndexAdapter.TableNameLimit)
38       }
39     }
40     new CassandraBackedMetadata(session, table, serializer)
41   }
42 
43   override protected def write(typeName: String, rows: Seq[(String, Array[Byte])]): Unit = {
44     rows.foreach { case (key, value) =>
45       session.execute(s"INSERT INTO $catalog (sft, key, value) VALUES (?, ?, ?)",
46         typeName, key, new String(value, StandardCharsets.UTF_8))
47     }
48   }
49 
50   override protected def delete(typeName: String, keys: Seq[String]): Unit = {
51     keys.foreach { key =>
52       val query = QueryBuilder.delete().from(catalog).where(QueryBuilder.eq("sft", typeName)).and(QueryBuilder.eq("key", key))
53       session.execute(query)
54     }
55   }
56 
57   override protected def scanValue(typeName: String, key: String): Option[Array[Byte]] = {
58     val query = QueryBuilder.select("value").from(catalog)
59     query.where(QueryBuilder.eq("sft", typeName)).and(QueryBuilder.eq("key", key))
60     val rows = session.execute(query).all().asScala
61     if (rows.length < 1) { None } else {
62       Some(rows.head.getString("value").getBytes(StandardCharsets.UTF_8))
63     }
64   }
65 
66   override protected def scanValues(typeName: String, prefix: String): CloseableIterator[(String, Array[Byte])] = {
67     val select = QueryBuilder.select("key", "value").from(catalog).where(QueryBuilder.eq("sft", typeName))
68     val iter = session.execute(select).all().iterator.asScala.map { row =>
69       (row.getString("key"), row.getString("value").getBytes(StandardCharsets.UTF_8))
70     }
71     if (prefix == null || prefix.isEmpty) {
72       CloseableIterator(iter)
73     } else {
74       CloseableIterator(iter.filter { case (k, _) => k.startsWith(prefix) })
75     }
76   }
77 
78   override protected def scanKeys(): CloseableIterator[(String, String)] = {
79     val select = QueryBuilder.select("sft", "key").from(catalog)
80     val values = session.execute(select).all().iterator.asScala.map(row => (row.getString("sft"), row.getString("key")))
81     CloseableIterator(values)
82   }
83 
84   override def close(): Unit = {} // session gets closed by datastore dispose
85 }
