21 class CassandraBackedMetadata[T](val session: Session, val catalog: String, val serializer: MetadataSerializer[T])
22     extends TableBasedMetadata[T] {
23 
24   // note: session gets closed by datastore dispose
25 
26   override protected def checkIfTableExists: Boolean = {
27     val m = session.getCluster.getMetadata
28     val km = m.getKeyspace(session.getLoggedKeyspace)
29     km.getTable(catalog) != null
30   }
31 
32   override protected def createTable(): Unit =
33     session.execute(s"CREATE TABLE IF NOT EXISTS $catalog (sft text, key text, value text, PRIMARY KEY ((sft), key))")
34 
35   override protected def createEmptyBackup(timestamp: String): CassandraBackedMetadata[T] = {
36     val table = {
37       val full = s"${catalog}_${timestamp}_bak"
38       if (full.lengthCompare(CassandraIndexAdapter.TableNameLimit) <= 0) { full } else {
39         IndexAdapter.truncateTableName(full, CassandraIndexAdapter.TableNameLimit)
40       }
41     }
42     new CassandraBackedMetadata(session, table, serializer)
43   }
44 
45   override protected def write(typeName: String, rows: Seq[(String, Array[Byte])]): Unit = {
46     rows.foreach { case (key, value) =>
47       session.execute(s"INSERT INTO $catalog (sft, key, value) VALUES (?, ?, ?)",
48         typeName, key, new String(value, StandardCharsets.UTF_8))
49     }
50   }
51 
52   override protected def delete(typeName: String, keys: Seq[String]): Unit = {
53     keys.foreach { key =>
54       val query = QueryBuilder.delete().from(catalog).where(QueryBuilder.eq("sft", typeName)).and(QueryBuilder.eq("key", key))
55       session.execute(query)
56     }
57   }
58 
59   override protected def scanValue(typeName: String, key: String): Option[Array[Byte]] = {
60     val query = QueryBuilder.select("value").from(catalog)
61     query.where(QueryBuilder.eq("sft", typeName)).and(QueryBuilder.eq("key", key))
62     val rows = session.execute(query).all().asScala
63     if (rows.length < 1) { None } else {
64       Some(rows.head.getString("value").getBytes(StandardCharsets.UTF_8))
65     }
66   }
67 
68   override protected def scanValues(typeName: String, prefix: String): CloseableIterator[(String, Array[Byte])] = {
69     val select = QueryBuilder.select("key", "value").from(catalog).where(QueryBuilder.eq("sft", typeName))
70     val iter = session.execute(select).all().iterator.asScala.map { row =>
71       (row.getString("key"), row.getString("value").getBytes(StandardCharsets.UTF_8))
72     }
73     if (prefix == null || prefix.isEmpty) {
74       CloseableIterator(iter)
75     } else {
76       CloseableIterator(iter.filter { case (k, _) => k.startsWith(prefix) })
77     }
78   }
79 
80   override protected def scanKeys(): CloseableIterator[(String, String)] = {
81     val select = QueryBuilder.select("sft", "key").from(catalog)
82     val values = session.execute(select).all().iterator.asScala.map(row => (row.getString("sft"), row.getString("key")))
83     CloseableIterator(values)
84   }
85 }
