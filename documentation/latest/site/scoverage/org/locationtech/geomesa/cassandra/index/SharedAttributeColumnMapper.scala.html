44   override def select(range: ScanRange[_], tieredKeyRanges: Seq[ByteRange]): Seq[RowSelect] = {
45     val primary = range.asInstanceOf[ScanRange[AttributeIndexKey]] match {
46       case SingleRowRange(row) =>
47         val i = Short.box(row.i)
48         val indexSelect = ColumnSelect(Index, i, i, startInclusive = true, endInclusive = true)

52       case BoundedRange(lo, hi) =>

68         Seq(indexSelect, valueSelect)

71         val i = Short.box(prefix.i)

79 
80       case _ => throw new IllegalArgumentException(s"Unexpected range type $range")
81     }
82     val clause = if (tieredKeyRanges.isEmpty) { primary } else {
83       val minTier = ByteRange.min(tieredKeyRanges)
84       val maxTier = ByteRange.max(tieredKeyRanges)
85       primary :+ ColumnSelect(Secondary, ByteBuffer.wrap(minTier), ByteBuffer.wrap(maxTier), startInclusive = true, endInclusive = true)
86     }
87     Seq(RowSelect(clause))
88   }
