46   override def bind(value: SingleRowKeyValue[_]): Seq[AnyRef] = {
47     val shard = Byte.box(if (value.shard.isEmpty) { 0 } else { value.shard.head })

61   override def select(range: ScanRange[_], tieredKeyRanges: Seq[ByteRange]): Seq[RowSelect] = {
62     val clause = range.asInstanceOf[ScanRange[Z3IndexKey]] match {
63       case BoundedRange(lo, hi) =>
64         val binSelect = ColumnSelect(Period, lo.bin, hi.bin, startInclusive = true, endInclusive = true)
65         val zSelect = ColumnSelect(ZValue, lo.z, hi.z, startInclusive = true, endInclusive = true)
66         Seq(binSelect, zSelect)

68       case UnboundedRange(_) =>

70 
71       case SingleRowRange(row) =>
72         val binSelect = ColumnSelect(Period, row.bin, row.bin, startInclusive = true, endInclusive = true)
73         val zSelect = ColumnSelect(ZValue, row.z, row.z, startInclusive = true, endInclusive = true)
74         Seq(binSelect, zSelect)
75 
76       case LowerBoundedRange(lo) =>
77         val binSelect = ColumnSelect(Period, lo.bin, null, startInclusive = true, endInclusive = false)
78         val zSelect = ColumnSelect(ZValue, lo.z, null, startInclusive = true, endInclusive = false)
79         Seq(binSelect, zSelect)
80 
81       case UpperBoundedRange(hi) =>
82         val binSelect = ColumnSelect(Period, null, hi.bin, startInclusive = false, endInclusive = true)
83         val zSelect = ColumnSelect(ZValue, null, hi.z, startInclusive = false, endInclusive = true)
84         Seq(binSelect, zSelect)
85 
86       case PrefixRange(_) =>
87         Seq.empty // not supported
88 
89       case _ => throw new IllegalArgumentException(s"Unexpected range type $range")
90     }
91     if (clause.isEmpty) { Seq(RowSelect(clause)) } else {
92       shards.map(s => RowSelect(clause.+:(s)))
93     }
94   }
