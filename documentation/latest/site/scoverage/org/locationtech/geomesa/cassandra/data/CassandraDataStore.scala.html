31 class CassandraDataStore(val session: Session, config: CassandraDataStoreConfig)
32     extends GeoMesaDataStore[CassandraDataStore](config) with LocalLocking {

38   override val metadata: GeoMesaMetadata[String] =

41   override val adapter: CassandraIndexAdapter = new CassandraIndexAdapter(this)
42 
43   override val stats: GeoMesaStats = new RunnableStats(this)

45   override def getQueryPlan(query: Query, index: Option[String], explainer: Explainer): Seq[CassandraQueryPlan] =
46     super.getQueryPlan(query, index, explainer).asInstanceOf[Seq[CassandraQueryPlan]]
47 
48   override def delete(): Unit = {
49     val tables = getTypeNames.flatMap(getAllIndexTableNames).distinct :+ config.catalog
50     tables.toList.map(t => CachedThreadPool.submit(() => session.execute(s"drop table $t"))).foreach(_.get)
51   }
52 
53   override protected def preSchemaUpdate(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
54     val rename = sft.getUserData.remove(SimpleFeatureTypes.Configs.UpdateRenameTables)
55     if (FastConverter.convert(rename, classOf[java.lang.Boolean]) == java.lang.Boolean.TRUE) {
56       logger.warn("Cassandra does not support renaming tables - disabling rename option")
57     }
58     sft.getUserData.put(SimpleFeatureTypes.Configs.UpdateRenameTables, java.lang.Boolean.FALSE)
59     super.preSchemaUpdate(sft, previous)
60   }
61 
62   override protected def transitionIndices(sft: SimpleFeatureType): Unit = {
63     val dtg = sft.getDtgField.toSeq
64     val geom = Option(sft.getGeomField).toSeq
65 
66     val indices = Seq.newBuilder[IndexId]
67     val tableNameKeys = Seq.newBuilder[(String, String)]
68 
69     sft.getIndices.foreach {
70       case id if id.name == IdIndex.name =>
71         require(id.version == 1, s"Expected index version of 1 but got: $id")
72         indices += id.copy(version = 3)
73         tableNameKeys += ((s"table.${IdIndex.name}.v1", s"table.${IdIndex.name}.v3"))
74 
75       case id if id.name == Z3Index.name =>
76         require(id.version <= 2, s"Expected index version of 1 or 2 but got: $id")
77         indices += id.copy(attributes = geom ++ dtg, version = id.version + 3)
78         tableNameKeys += ((s"table.${Z3Index.name}.v${id.version}", s"table.${Z3Index.name}.v${id.version + 3}"))
79 
80       case id if id.name == XZ3Index.name =>
81         require(id.version == 1, s"Expected index version of 1 but got: $id")
82         indices += id.copy(attributes = geom ++ dtg)
83 
84       case id if id.name == Z2Index.name =>
85         require(id.version <= 2, s"Expected index version of 1 or 2 but got: $id")
86         indices += id.copy(attributes = geom, version = id.version + 2)
87         tableNameKeys += ((s"table.${Z2Index.name}.v${id.version}", s"table.${Z2Index.name}.v${id.version + 2}"))
88 
89       case id if id.name == XZ2Index.name =>
90         require(id.version == 1, s"Expected index version of 1 but got: $id")
91         indices += id.copy(attributes = geom)
92 
93       case id if id.name == AttributeIndex.name =>
94         require(id.version <= 2, s"Expected index version of 1 or 2 but got: $id")
95         val fields = geom ++ dtg
96         sft.getAttributeDescriptors.asScala.foreach { d =>
97           val index = d.getUserData.remove(AttributeOptions.OptIndex).asInstanceOf[String]
98           if (index == null || index.equalsIgnoreCase("false") || index.equalsIgnoreCase(IndexCoverage.NONE.toString)) {
99             // no-op
100           } else if (java.lang.Boolean.valueOf(index) || index.equalsIgnoreCase(IndexCoverage.FULL.toString) ||
101               index.equalsIgnoreCase(IndexCoverage.JOIN.toString)) {
102             indices += id.copy(attributes = Seq(d.getLocalName) ++ fields, version = id.version + 4)
103           } else {
104             throw new IllegalStateException(s"Expected an index coverage or boolean but got: $index")
105           }
106         }
107         tableNameKeys ++=
108             Seq(s"table.${AttributeIndex.name}.v${id.version}", "tables.idx.attr.name")
109                 .map((_, s"table.${AttributeIndex.name}.v${id.version + 4}"))
110     }
111 
112     sft.setIndices(indices.result)
113 
114     // update metadata keys for tables
115     tableNameKeys.result.foreach { case (from, to) =>
116       metadata.scan(sft.getTypeName, from, cache = false).foreach { case (key, name) =>
117         metadata.insert(sft.getTypeName, to + key.substring(from.length), name)
118         metadata.remove(sft.getTypeName, key)
119       }
120     }
121   }
