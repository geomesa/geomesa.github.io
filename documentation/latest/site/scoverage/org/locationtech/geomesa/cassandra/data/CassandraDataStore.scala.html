32 class CassandraDataStore(val session: Session, config: CassandraDataStoreConfig)
33     extends GeoMesaDataStore[CassandraDataStore](config) with LocalLocking {

39   override val metadata: GeoMesaMetadata[String] =
40     new CassandraBackedMetadata(session, config.catalog, MetadataStringSerializer)
41 
42   override val adapter: CassandraIndexAdapter = new CassandraIndexAdapter(this)
43 
44   override val stats: GeoMesaStats = new RunnableStats(this)

46   override def getQueryPlan(query: Query, index: Option[String], explainer: Explainer): Seq[CassandraQueryPlan] =
47     super.getQueryPlan(query, index, explainer).asInstanceOf[Seq[CassandraQueryPlan]]
48 
49   override def delete(): Unit = {
50     val tables = getTypeNames.flatMap(getAllIndexTableNames).distinct :+ config.catalog
51     tables.toList.map(t => CachedThreadPool.submit(() => session.execute(s"drop table $t"))).foreach(_.get)
52   }
53 
54   override protected def preSchemaUpdate(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
55     val rename = sft.getUserData.remove(SimpleFeatureTypes.Configs.UpdateRenameTables)
56     if (FastConverter.convert(rename, classOf[java.lang.Boolean]) == java.lang.Boolean.TRUE) {
57       logger.warn("Cassandra does not support renaming tables - disabling rename option")
58     }
59     sft.getUserData.put(SimpleFeatureTypes.Configs.UpdateRenameTables, java.lang.Boolean.FALSE)
60     super.preSchemaUpdate(sft, previous)
61   }
62 
63   override protected def transitionIndices(sft: SimpleFeatureType): Unit = {
64     val dtg = sft.getDtgField.toSeq
65     val geom = Option(sft.getGeomField).toSeq
66 
67     val indices = Seq.newBuilder[IndexId]
68     val tableNameKeys = Seq.newBuilder[(String, String)]
69 
70     sft.getIndices.foreach {
71       case id if id.name == IdIndex.name =>
72         require(id.version == 1, s"Expected index version of 1 but got: $id")
73         indices += id.copy(version = 3)
74         tableNameKeys += ((s"table.${IdIndex.name}.v1", s"table.${IdIndex.name}.v3"))
75 
76       case id if id.name == Z3Index.name =>
77         require(id.version <= 2, s"Expected index version of 1 or 2 but got: $id")
78         indices += id.copy(attributes = geom ++ dtg, version = id.version + 3)
79         tableNameKeys += ((s"table.${Z3Index.name}.v${id.version}", s"table.${Z3Index.name}.v${id.version + 3}"))
80 
81       case id if id.name == XZ3Index.name =>
82         require(id.version == 1, s"Expected index version of 1 but got: $id")
83         indices += id.copy(attributes = geom ++ dtg)
84 
85       case id if id.name == Z2Index.name =>
86         require(id.version <= 2, s"Expected index version of 1 or 2 but got: $id")
87         indices += id.copy(attributes = geom, version = id.version + 2)
88         tableNameKeys += ((s"table.${Z2Index.name}.v${id.version}", s"table.${Z2Index.name}.v${id.version + 2}"))
89 
90       case id if id.name == XZ2Index.name =>
91         require(id.version == 1, s"Expected index version of 1 but got: $id")
92         indices += id.copy(attributes = geom)
93 
94       case id if id.name == AttributeIndex.name =>
95         require(id.version <= 2, s"Expected index version of 1 or 2 but got: $id")
96         val fields = geom ++ dtg
97         sft.getAttributeDescriptors.asScala.foreach { d =>
98           val index = d.getUserData.remove(AttributeOptions.OptIndex).asInstanceOf[String]
99           if (index == null || index.equalsIgnoreCase("false") || index.equalsIgnoreCase(IndexCoverage.NONE.toString)) {
100             // no-op
101           } else if (java.lang.Boolean.valueOf(index) || index.equalsIgnoreCase(IndexCoverage.FULL.toString) ||
102               index.equalsIgnoreCase(IndexCoverage.JOIN.toString)) {
103             indices += id.copy(attributes = Seq(d.getLocalName) ++ fields, version = id.version + 4)
104           } else {
105             throw new IllegalStateException(s"Expected an index coverage or boolean but got: $index")
106           }
107         }
108         tableNameKeys ++=
109             Seq(s"table.${AttributeIndex.name}.v${id.version}", "tables.idx.attr.name")
110                 .map((_, s"table.${AttributeIndex.name}.v${id.version + 4}"))
111     }
112 
113     sft.setIndices(indices.result)
114 
115     // update metadata keys for tables
116     tableNameKeys.result.foreach { case (from, to) =>
117       metadata.scan(sft.getTypeName, from, cache = false).foreach { case (key, name) =>
118         metadata.insert(sft.getTypeName, to + key.substring(from.length), name)
119         metadata.remove(sft.getTypeName, key)
120       }
121     }
122   }
