184       ds: CassandraDataStore,
185       indices: Seq[GeoMesaFeatureIndex[_, _]],
186       wrapper: FeatureWrapper[WritableFeature],
187       partition: Option[String]
188     ) extends BaseIndexWriter(indices, wrapper) with StrictLogging {
189 
190     private val mappers = indices.toArray.map { index =>
191       val mapper = CassandraColumnMapper(index)
192       val table = index.getTableNames(partition) match {
193         case Seq(t) => t // should always be writing to a single table here
194         case tables => throw new IllegalStateException(s"Expected a single table but got: ${tables.mkString(", ")}")
195       }
196       val insert = mapper.insert(ds.session, table)
197       val delete = mapper.delete(ds.session, table)
198       (mapper, insert, delete)
199     }
200 
201     private var i = 0
202 
203     override protected def append(feature: WritableFeature, values: Array[RowKeyValue[_]]): Unit = {
204       i = 0
205       while (i < values.length) {
206         val (mapper, statement, _) = mappers(i)
207         values(i) match {
208           case kv: SingleRowKeyValue[_] =>
209             val bindings = mapper.bind(kv)
210             logger.trace(s"${statement.getQueryString} : ${debug(bindings)}")
211             ds.session.execute(statement.bind(bindings: _*))
212 
213           case mkv: MultiRowKeyValue[_] =>
214             mkv.split.foreach { kv =>
215               val bindings = mapper.bind(kv)
216               logger.trace(s"${statement.getQueryString} : ${debug(bindings)}")
217               ds.session.execute(statement.bind(bindings: _*))
218             }
219         }
220         i += 1
221       }
222     }

224     override protected def update(

226         values: Array[RowKeyValue[_]],
227         previous: WritableFeature,
228         previousValues: Array[RowKeyValue[_]]): Unit = {
