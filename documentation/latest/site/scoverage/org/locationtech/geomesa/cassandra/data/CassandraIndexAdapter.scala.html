182   class CassandraIndexWriter(
183       ds: CassandraDataStore,
184       indices: Seq[GeoMesaFeatureIndex[_, _]],
185       wrapper: FeatureWrapper[WritableFeature],
186       partition: Option[String]
187     ) extends BaseIndexWriter(indices, wrapper) with StrictLogging {
188 
189     private val mappers = indices.toArray.map { index =>
190       val mapper = CassandraColumnMapper(index)
191       // should always be writing to a single table here
192       val table = index.getTableName(partition)
193       val insert = mapper.insert(ds.session, table)
194       val delete = mapper.delete(ds.session, table)
195       (mapper, insert, delete)
196     }
197 
198     private var i = 0
199 
200     override protected def append(feature: WritableFeature, values: Array[RowKeyValue[_]]): Unit = {
201       i = 0
202       while (i < values.length) {
203         val (mapper, statement, _) = mappers(i)
204         values(i) match {
205           case kv: SingleRowKeyValue[_] =>
206             val bindings = mapper.bind(kv)
207             logger.trace(s"${statement.getQueryString} : ${debug(bindings)}")
208             ds.session.execute(statement.bind(bindings: _*))
209 
210           case mkv: MultiRowKeyValue[_] =>
211             mkv.split.foreach { kv =>
212               val bindings = mapper.bind(kv)
213               logger.trace(s"${statement.getQueryString} : ${debug(bindings)}")
214               ds.session.execute(statement.bind(bindings: _*))
215             }
216         }
217         i += 1
218       }
219     }
