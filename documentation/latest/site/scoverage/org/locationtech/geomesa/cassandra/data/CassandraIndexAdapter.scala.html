84   override def createQueryPlan(strategy: QueryStrategy): CassandraQueryPlan = {
85     import org.locationtech.geomesa.index.conf.QueryHints.RichHints

90     if (strategy.keyRanges.isEmpty) { EmptyPlan(strategy, reducer) } else {
91       val mapper = CassandraColumnMapper(strategy.index)

93       val tables = strategy.index.getTablesForQuery(strategy.filter.filter)

96       val rowsToFeatures = new CassandraResultsToFeatures(strategy.index, strategy.index.sft)
97       val threads = ds.config.queries.threads

99       val max = hints.getMaxFeatures
100       val project = hints.getProjection
101       StatementPlan(strategy, tables, statements, threads, strategy.ecql, rowsToFeatures, reducer, sort, max, project)

187       ds: CassandraDataStore,
188       indices: Seq[GeoMesaFeatureIndex[_, _]],
189       wrapper: FeatureWrapper[WritableFeature],
190       partition: Option[String]
191     ) extends BaseIndexWriter(indices, wrapper) with StrictLogging {
192 
193     private val mappers = indices.toArray.map { index =>
194       val mapper = CassandraColumnMapper(index)
195       // should always be writing to a single table here
196       val table = index.getTableName(partition)
197       val insert = mapper.insert(ds.session, table)
198       val delete = mapper.delete(ds.session, table)
199       (mapper, insert, delete)
200     }
201 
202     private var i = 0
203 
204     override protected def append(feature: WritableFeature, values: Array[RowKeyValue[_]]): Unit = {
205       i = 0
206       while (i < values.length) {
207         val (mapper, statement, _) = mappers(i)
208         values(i) match {
209           case kv: SingleRowKeyValue[_] =>
210             val bindings = mapper.bind(kv)
211             logger.trace(s"${statement.getQueryString} : ${debug(bindings)}")
212             ds.session.execute(statement.bind(bindings: _*))
213 
214           case mkv: MultiRowKeyValue[_] =>
215             mkv.split.foreach { kv =>
216               val bindings = mapper.bind(kv)
217               logger.trace(s"${statement.getQueryString} : ${debug(bindings)}")
218               ds.session.execute(statement.bind(bindings: _*))
219             }
220         }
221         i += 1
222       }
223     }
