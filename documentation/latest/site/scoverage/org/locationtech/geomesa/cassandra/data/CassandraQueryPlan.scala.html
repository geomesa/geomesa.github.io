60   case class StatementPlan(
61       ds: CassandraDataStore,
62       strategy: QueryStrategy,
63       tables: Seq[String],
64       ranges: Seq[Statement],
65       numThreads: Int,
66       localFilter: Option[Filter],
67       processor: LocalProcessor,
68       projection: Option[QueryReferenceSystems]
69     ) extends CassandraQueryPlan with LocalProcessorPlan {
70 
71     override def scan(): CloseableIterator[SimpleFeature] = {
72       ds.interceptors.runGuards(strategy) // query guard hook - also handles full table scan checks
73       val timeout = ds.config.queries.timeout.map(Timeout.apply)
74       val toFeatures = new CassandraResultsToFeatures(strategy.index, strategy.index.sft)
75       val scanner = CassandraBatchScan(this, ds.session, ranges, numThreads, timeout).map(toFeatures.apply)
76       val features = localFilter.fold(scanner)(f => scanner.filter(f.evaluate))
77       processor(features)
78     }
79 
80     override protected def moreExplaining(explainer: Explainer): Unit = {
81       explainer(s"Client-side filter: ${localFilter.fold("none")(FilterHelper.toString)}")
82       processor.explain(explainer)
83     }
84   }
