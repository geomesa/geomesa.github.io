33   override def createDataStore(params: util.Map[String, _]): DataStore = {
34     import org.locationtech.geomesa.cassandra.CassandraSystemProperties.{ConnectionTimeoutMillis, ReadTimeoutMillis}
35 
36     val (cp, portString) = ContactPointParam.lookup(params).split(":") match {
37       case Array(one, two) => (one, two)
38       case parts => throw new IllegalArgumentException(s"Invalid parameter '${ContactPointParam.key}', " +
39           s"expected '<host>:<port>' but got '${parts.mkString(":")}'")
40     }
41     val port = try { portString.toInt } catch {
42       case NonFatal(_) => throw new IllegalArgumentException(s"Invalid parameter '${ContactPointParam.key}', " +
43           s"expected '<host>:<port>' but port is not a number: '$cp:$portString'")
44     }
45     val ks = KeySpaceParam.lookup(params)
46     val generateStats = GenerateStatsParam.lookup(params)
47     val audit = if (AuditQueriesParam.lookup(params)) {
48       Some(AuditLogger, Option(AuditProvider.Loader.load(params)).getOrElse(NoOpAuditProvider), "cassandra")
49     } else {
50       None
51     }
52 
53     val clusterBuilder =
54       Cluster.builder()
55         .addContactPoint(cp)
56         .withPort(port)
57         .withQueryOptions(new QueryOptions().setConsistencyLevel(ConsistencyLevel.ONE))
58         .withRetryPolicy(DefaultRetryPolicy.INSTANCE)
59         .withLoadBalancingPolicy(new TokenAwarePolicy(DCAwareRoundRobinPolicy.builder().build()))
60 
61     val socketOptions = {
62       var options: SocketOptions = null
63       def ensureOptions(): SocketOptions = {
64         if (options == null) {
65           options = new SocketOptions()
66         }
67         options
68       }
69       ReadTimeoutMillis.toDuration.foreach(timeout => ensureOptions().setReadTimeoutMillis(timeout.toMillis.toInt))
70       ConnectionTimeoutMillis.toDuration.foreach(timeout => ensureOptions().setConnectTimeoutMillis(timeout.toMillis.toInt))
71       Option(options)
72     }
73     socketOptions.foreach(clusterBuilder.withSocketOptions)
74 
75     val user = UserNameParam.lookup(params)
76     val password = PasswordParam.lookup(params)
77     if (user != null && password != null) {
78       clusterBuilder.withCredentials(user, password)
79     }
80 
81     val cluster = clusterBuilder.build()
82     val session = cluster.connect(ks)
83     val catalog = CatalogParam.lookup(params)
84 
85     val queries = CassandraQueryConfig(
86       threads = QueryThreadsParam.lookup(params),
87       timeout = QueryTimeoutParam.lookupOpt(params).map(_.toMillis),
88       looseBBox = LooseBBoxParam.lookup(params),
89       parallelPartitionScans = PartitionParallelScansParam.lookup(params)
90     )
91 
92     val authProvider = new DefaultAuthorizationsProvider()
93 
94     val ns = Option(NamespaceParam.lookUp(params).asInstanceOf[String])
95 
96     val cfg = CassandraDataStoreConfig(catalog, generateStats, authProvider, audit, queries, ns)
97 
98     new CassandraDataStore(session, cfg)
99   }
