35   override def createDataStore(params: util.Map[String, _]): DataStore = {
36     import org.locationtech.geomesa.cassandra.CassandraSystemProperties.{ConnectionTimeoutMillis, ReadTimeoutMillis}
37 
38     val (cp, portString) = ContactPointParam.lookup(params).split(":") match {
39       case Array(one, two) => (one, two)
40       case parts => throw new IllegalArgumentException(s"Invalid parameter '${ContactPointParam.key}', " +
41           s"expected '<host>:<port>' but got '${parts.mkString(":")}'")
42     }
43     val port = try { portString.toInt } catch {
44       case NonFatal(_) => throw new IllegalArgumentException(s"Invalid parameter '${ContactPointParam.key}', " +
45           s"expected '<host>:<port>' but port is not a number: '$cp:$portString'")
46     }
47     val ks = KeySpaceParam.lookup(params)
48     val generateStats = GenerateStatsParam.lookup(params)
49     val audit = if (AuditQueriesParam.lookup(params)) {
50       Some(new AuditLogger("cassandra", AuditProvider.Loader.loadOrNone(params)))
51     } else {
52       None
53     }
54 
55     val clusterBuilder =
56       Cluster.builder()
57         .addContactPoint(cp)
58         .withPort(port)
59         .withQueryOptions(new QueryOptions().setConsistencyLevel(ConsistencyLevel.ONE))
60         .withRetryPolicy(DefaultRetryPolicy.INSTANCE)
61         .withLoadBalancingPolicy(new TokenAwarePolicy(DCAwareRoundRobinPolicy.builder().build()))
62 
63     val socketOptions = {
64       var options: SocketOptions = null
65       def ensureOptions(): SocketOptions = {
66         if (options == null) {
67           options = new SocketOptions()
68         }
69         options
70       }
71       ReadTimeoutMillis.toDuration.foreach(timeout => ensureOptions().setReadTimeoutMillis(timeout.toMillis.toInt))
72       ConnectionTimeoutMillis.toDuration.foreach(timeout => ensureOptions().setConnectTimeoutMillis(timeout.toMillis.toInt))
73       Option(options)
74     }
75     socketOptions.foreach(clusterBuilder.withSocketOptions)
76 
77     val user = UserNameParam.lookup(params)
78     val password = PasswordParam.lookup(params)
79     if (user != null && password != null) {
80       clusterBuilder.withCredentials(user, password)
81     }
82 
83     val cluster = clusterBuilder.build()
84     val session = cluster.connect(ks)
85     val catalog = CatalogParam.lookup(params)
86 
87     val queries = CassandraQueryConfig(
88       threads = QueryThreadsParam.lookup(params),
89       timeout = QueryTimeoutParam.lookupOpt(params).map(_.toMillis),
90       looseBBox = LooseBBoxParam.lookup(params),
91       parallelPartitionScans = PartitionParallelScansParam.lookup(params)
92     )
93 
94     val authProvider = new DefaultAuthorizationsProvider(Seq.empty)
95 
96     val ns = Option(NamespaceParam.lookUp(params).asInstanceOf[String])
97 
98     val cfg = CassandraDataStoreConfig(catalog, generateStats, authProvider, audit, queries, ns)
99 
100     new CassandraDataStore(session, cfg)
101   }
