35   override def createDataStore(params: util.Map[String, _]): DataStore = {
36     import org.locationtech.geomesa.cassandra.CassandraSystemProperties.{ConnectionTimeoutMillis, ReadTimeoutMillis}
37 
38     val (cp, portString) = ContactPointParam.lookup(params).split(":") match {
39       case Array(one, two) => (one, two)
40       case parts => throw new IllegalArgumentException(s"Invalid parameter '${ContactPointParam.key}', " +
41           s"expected '<host>:<port>' but got '${parts.mkString(":")}'")
42     }
43     val port = try { portString.toInt } catch {
44       case NonFatal(_) => throw new IllegalArgumentException(s"Invalid parameter '${ContactPointParam.key}', " +
45           s"expected '<host>:<port>' but port is not a number: '$cp:$portString'")
46     }
47     val ks = KeySpaceParam.lookup(params)
48     val generateStats = GenerateStatsParam.lookup(params)
49     val audit = if (AuditQueriesParam.lookup(params)) {
50       Some(new AuditLogger("cassandra", AuditProvider.Loader.loadOrNone(params)))
51     } else {
52       None
53     }
54     val metrics = MetricsRegistryParam.lookupRegistry(params)
55 
56     val clusterBuilder =
57       Cluster.builder()
58         .addContactPoint(cp)
59         .withPort(port)
60         .withQueryOptions(new QueryOptions().setConsistencyLevel(ConsistencyLevel.ONE))
61         .withRetryPolicy(DefaultRetryPolicy.INSTANCE)
62         .withLoadBalancingPolicy(new TokenAwarePolicy(DCAwareRoundRobinPolicy.builder().build()))
63 
64     val socketOptions = {
65       var options: SocketOptions = null
66       def ensureOptions(): SocketOptions = {
67         if (options == null) {
68           options = new SocketOptions()
69         }
70         options
71       }
72       ReadTimeoutMillis.toDuration.foreach(timeout => ensureOptions().setReadTimeoutMillis(timeout.toMillis.toInt))
73       ConnectionTimeoutMillis.toDuration.foreach(timeout => ensureOptions().setConnectTimeoutMillis(timeout.toMillis.toInt))
74       Option(options)
75     }
76     socketOptions.foreach(clusterBuilder.withSocketOptions)
77 
78     val user = UserNameParam.lookup(params)
79     val password = PasswordParam.lookup(params)
80     if (user != null && password != null) {
81       clusterBuilder.withCredentials(user, password)
82     }
83 
84     val cluster = clusterBuilder.build()
85     val session = cluster.connect(ks)
86     val catalog = CatalogParam.lookup(params)
87 
88     val queries = CassandraQueryConfig(
89       threads = QueryThreadsParam.lookup(params),
90       timeout = QueryTimeoutParam.lookupOpt(params).map(_.toMillis),
91       looseBBox = LooseBBoxParam.lookup(params),
92       parallelPartitionScans = PartitionParallelScansParam.lookup(params)
93     )
94 
95     val authProvider = new DefaultAuthorizationsProvider(Seq.empty)
96 
97     val ns = Option(NamespaceParam.lookUp(params).asInstanceOf[String])
98 
99     val cfg = CassandraDataStoreConfig(catalog, generateStats, authProvider, audit, metrics, queries, ns)
100 
101     new CassandraDataStore(session, cfg)
102   }
