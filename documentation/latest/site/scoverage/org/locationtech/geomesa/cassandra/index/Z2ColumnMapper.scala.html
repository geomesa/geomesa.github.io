59     val clause = range match {
60       case BoundedRange(lo, hi)  => Seq(ColumnSelect(ZValue, lo, hi, startInclusive = true, endInclusive = true))
61       case UnboundedRange(_)     => Seq.empty
62       case SingleRowRange(row)   => Seq(ColumnSelect(ZValue, row, row, startInclusive = true, endInclusive = true))

65       case PrefixRange(_)        => Seq.empty // not supported
66       case _ => throw new IllegalArgumentException(s"Unexpected range type $range")
67     }
68     if (clause.isEmpty) { Seq(RowSelect(clause)) } else {
69       shards.map(s => RowSelect(clause.+:(s)))
70     }
