88   /**
89     * Configure serialization for a simple feature type
90     *
91     * @param conf conf
92     * @param sft simple feature type
93     */
94   def setSerialization(conf: Configuration, sft: SimpleFeatureType): Unit = {
95     // register the feature serializer
96     conf.get(Keys.Serializers) match {
97       case null => conf.set(Keys.Serializers, SerializationString)
98       case existing =>
99         if (!existing.contains(SimpleFeatureSerialization)) {
100           conf.set(Keys.Serializers, Seq(existing, SimpleFeatureSerialization).mkString(","))
101         }
102     }
103 
104     // set the schema in the config
105     val spec = SimpleFeatureTypes.encodeType(sft, includeUserData = true)
106     // note: use the spec hash code to ensure that duplicate type names don't conflict
107     val key = s"${(sft.getTypeName + spec).hashCode}:${sft.getTypeName}"
108     // store the type name and hash under a common key for all types
109     conf.get(Keys.SftNames) match {
110       case null => conf.set(Keys.SftNames, key)
111       case encoded =>
112         val existing = StringSerialization.decodeSeq(encoded)
113         if (!existing.contains(key)) {
114           conf.set(Keys.SftNames, StringSerialization.encodeSeq(existing :+ key))
115         }
116     }
117     // store the spec under the unique key
118     conf.set(s"${Keys.SftNames}.$key", spec)
119   }
120 
121   /**
122     * Get the simple feature types configured for serialization, keyed by hash code
123     *
124     * @param conf conf
125     * @return (unique string key for the type, unique hash code for the type, type)
126     */
127   def getSerialization(conf: Configuration): Seq[(String, Int, SimpleFeatureType)] = {
128     Option(conf.get(Keys.SftNames)).map(StringSerialization.decodeSeq).getOrElse(Seq.empty).map { key =>
129       val sep = key.indexOf(':')
130       val hash = key.substring(0, sep).toInt
131       val typeName = key.substring(sep + 1)
132       val spec = conf.get(s"${Keys.SftNames}.$key")
133       (key, hash, SimpleFeatureTypes.createType(typeName, spec))
134     }
135   }
