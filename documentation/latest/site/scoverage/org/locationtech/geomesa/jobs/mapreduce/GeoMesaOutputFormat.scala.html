28 /**
29   * Output format that writes simple features using GeoMesaDataStore's FeatureWriterAppend. Can write only
30   * specific indices if desired
31   */
32 class GeoMesaOutputFormat extends OutputFormat[Text, SimpleFeature] {
33 
34   import scala.collection.JavaConverters._
35 
36   override def getRecordWriter(context: TaskAttemptContext): RecordWriter[Text, SimpleFeature] = {
37     val params = GeoMesaConfigurator.getDataStoreOutParams(context.getConfiguration)
38     val indices = GeoMesaConfigurator.getIndicesOut(context.getConfiguration)
39     new GeoMesaRecordWriter(params, indices, context)
40   }
41 
42   override def checkOutputSpecs(context: JobContext): Unit = {
43     val params =
44       GeoMesaConfigurator.getDataStoreOutParams(context.getConfiguration)
45         .asJava.asInstanceOf[java.util.Map[String, java.io.Serializable]]
46     if (!DataStoreFinder.getAvailableDataStores.asScala.exists(_.canProcess(params))) {
47       throw new IOException("Data store connection parameters are not set")
48     }
49   }
50 
51   override def getOutputCommitter(context: TaskAttemptContext): OutputCommitter =
52     new NullOutputFormat[Text, SimpleFeature]().getOutputCommitter(context)
53 }

55 object GeoMesaOutputFormat {
56 
57   import scala.collection.JavaConverters._
58 
59   object OutputCounters {
60     val Group   = "org.locationtech.geomesa.jobs.output"
61     val Written = "written"
62     val Failed  = "failed"
63   }
64 
65   /**
66     * Configure the data store you will be writing to
67     *
68     * @param conf conf
69     * @param params data store parameters
70     * @param sft simple feature type to write, must exist already in the store
71     * @param indices indices to write, or all indices
72     */
73   def setOutput(
74       conf: Configuration,
75       params: Map[String, String],
76       sft: SimpleFeatureType,
77       indices: Option[Seq[String]] = None): Unit = {
78     GeoMesaConfigurator.setDataStoreOutParams(conf, params)
79     GeoMesaConfigurator.setSerialization(conf, sft)
80     indices.foreach(GeoMesaConfigurator.setIndicesOut(conf, _))
81   }
82 
83   /**
84    * Helper for java interop
85    *
86    * @param conf conf
87    * @param params data store parameters
88    * @param sft simple feature type to write, must exist already in the store
89    */
90   def setOutputJava(
91       conf: Configuration,
92       params: java.util.Map[String, String],
93       sft: SimpleFeatureType): Unit = {
94     setOutput(conf, params.asScala.toMap, sft)
95   }
96 
97   /**
98     * Record writer for GeoMesa datastores.
99     *
100     * All feature types must exist already in the datastore. The input key is ignored.
101     */
102   class GeoMesaRecordWriter(params: Map[String, String], indices: Option[Seq[String]], context: TaskAttemptContext)
103       extends RecordWriter[Text, SimpleFeature] with LazyLogging {
104 
105     private val ds = DataStoreFinder.getDataStore(params.asJava)
106 
107     private val writers = scala.collection.mutable.Map.empty[String, FeatureWriter[SimpleFeatureType, SimpleFeature]]
108 
109     private val written = context.getCounter(OutputCounters.Group, OutputCounters.Written)
110     private val failed = context.getCounter(OutputCounters.Group, OutputCounters.Failed)
111 
112     override def write(key: Text, value: SimpleFeature): Unit = {
113       try {
114         val sftName = value.getFeatureType.getTypeName
115         val writer = writers.getOrElseUpdate(sftName, createWriter(sftName))
116         FeatureUtils.write(writer, value)
117         written.increment(1)
118       } catch {
119         case e: Exception =>
120           logger.error(s"Error writing feature '${DataUtilities.encodeFeature(value)}'", e)
121           failed.increment(1)
122       }
123     }
124 
125     private def createWriter(typeName: String): FeatureWriter[SimpleFeatureType, SimpleFeature] = {
126       ds match {
127         case gm: GeoMesaDataStore[_] =>
128           val sft = gm.getSchema(typeName)
129           val i = indices match {
130             case Some(names) => names.map(gm.manager.index(sft, _, IndexMode.Write))
131             case None => gm.manager.indices(sft, mode = IndexMode.Write)
132           }
133           gm.getIndexWriterAppend(typeName, i)
134 
135         case _ =>
136           indices.foreach { i =>
137             logger.warn(s"Ignoring index config '${i.mkString(",")}' for non-geomesa data store $ds")
138           }
139           ds.getFeatureWriterAppend(typeName, Transaction.AUTO_COMMIT)
140       }
141     }
142 
143     override def close(context: TaskAttemptContext): Unit = {
144       writers.values.foreach(v => CloseQuietly(v))
145       ds.dispose()
146     }
147   }
