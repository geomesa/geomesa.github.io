126       if (codec != null) {

129       } else {
130         fs.open(path)
131       }
132     iter = createIterator(stream, path, context)
133     logger.debug(s"Initialized record reader on split ${path.toString}")
134   }
135 
136   override def getCurrentKey: LongWritable = curKey
137 
138   override def close(): Unit = {
