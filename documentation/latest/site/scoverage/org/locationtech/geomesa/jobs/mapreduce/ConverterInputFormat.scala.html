88 class ConverterRecordReader extends FileStreamRecordReader with LazyLogging {
89 
90   import scala.collection.JavaConverters._
91 
92   override def createIterator(
93       stream: InputStream with Seekable,
94       filePath: Path,
95       context: TaskAttemptContext): Iterator[SimpleFeature] with Closeable = {
96 
97     val confStr   = context.getConfiguration.get(ConverterKey)
98     val conf      = ConfigFactory.parseString(confStr)
99     val sft       = FileStreamInputFormat.getSft(context.getConfiguration)
100     val converter = SimpleFeatureConverter(sft, conf)
101     val filter    = GeoMesaConfigurator.getFilter(context.getConfiguration).map(ECQL.toFilter)
102     val retypedSpec = context.getConfiguration.get(RetypeKey)
103 
104     def ec(path: String): EvaluationContext = {
105       // global success/failure counters for the entire job
106       val success = context.getCounter(ConverterCounters.Group, ConverterCounters.Converted)
107       val failure = context.getCounter(ConverterCounters.Group, ConverterCounters.Failed)
108       converter.createEvaluationContext(EvaluationContext.inputFileParam(path)).withListener(new MapReduceListener(success, failure))
109     }
110 
111     lazy val defaultEc = ec(filePath.toString)
112 
113     val streams: CloseableIterator[(Option[String], InputStream)] =
114       PathUtils.getUncompressedExtension(filePath.getName).toLowerCase(Locale.US) match {
115         case ArchiveStreamFactory.TAR =>
116           val archive: ArchiveInputStream[_ <: ArchiveEntry] =
117             new ArchiveStreamFactory().createArchiveInputStream(ArchiveStreamFactory.TAR, stream)
118           new ArchiveFileIterator(archive, filePath.toString)
119 
120         case ArchiveStreamFactory.ZIP | ArchiveStreamFactory.JAR =>
121           // we have to read the bytes into memory to get random access reads
122           // note: stream is closed in super class
123           val bytes = new SeekableInMemoryByteChannel(IOUtils.toByteArray(stream))
124           new ZipFileIterator(ZipFile.builder.setSeekableByteChannel(bytes).get(), filePath.toString)
125 
126         case _ =>
127           CloseableIterator.single(None -> stream, stream.close())
128       }
129 
130     val all = streams.flatMap { case (name, is) =>
131       converter.process(is, name.map(ec).getOrElse(defaultEc))
132     }
133     val iter = filter match {
134       case Some(f) => all.filter(f.evaluate)
135       case None    => all
136     }
137 
138     val featureReader = if (retypedSpec != null) {
139       val retypedSft = SimpleFeatureTypes.createType(sft.getTypeName, retypedSpec)
140       val reader = new DelegateSimpleFeatureReader(sft, new DelegateSimpleFeatureIterator(iter.asJava))
141       new ReTypeFeatureReader(reader, retypedSft)
142     } else {
143       new DelegateSimpleFeatureReader(sft, new DelegateSimpleFeatureIterator(iter.asJava))
144     }
145 
146     logger.debug(s"Initialized record reader on split ${filePath.toString} with " +
147       s"type name ${sft.getTypeName} and convert conf $confStr")
148 
149     new Iterator[SimpleFeature] with Closeable {
150       override def hasNext: Boolean = featureReader.hasNext
151       override def next(): SimpleFeature = featureReader.next
152       override def close(): Unit = {
153         CloseWithLogging(featureReader)
154         CloseWithLogging(iter)
155         CloseWithLogging(converter)
156       }
157     }
158   }
159 
160   private class MapReduceListener(success: Counter, failure: Counter) extends ContextListener {
161     override def onSuccess(i: Int): Unit = success.increment(i)
162     override def onFailure(i: Int): Unit = failure.increment(i)
163   }
