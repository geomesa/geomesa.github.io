87 class ConverterRecordReader extends FileStreamRecordReader with LazyLogging {
88 
89   import scala.collection.JavaConverters._
90 
91   override def createIterator(
92       stream: InputStream with Seekable,
93       filePath: Path,
94       context: TaskAttemptContext): Iterator[SimpleFeature] with Closeable = {
95 
96     val confStr   = context.getConfiguration.get(ConverterKey)
97     val conf      = ConfigFactory.parseString(confStr)
98     val sft       = FileStreamInputFormat.getSft(context.getConfiguration)
99     val converter = SimpleFeatureConverter(sft, conf)
100     val filter    = GeoMesaConfigurator.getFilter(context.getConfiguration).map(ECQL.toFilter)
101     val retypedSpec = context.getConfiguration.get(RetypeKey)
102 
103     def ec(path: String): EvaluationContext = {
104       // global success/failure counters for the entire job
105       val success = new MapReduceCounter(context.getCounter(ConverterCounters.Group, ConverterCounters.Converted))
106       val failure = new MapReduceCounter(context.getCounter(ConverterCounters.Group, ConverterCounters.Failed))
107       converter.createEvaluationContext(EvaluationContext.inputFileParam(path), success, failure)
108     }
109 
110     lazy val defaultEc = ec(filePath.toString)
111 
112     val streams: CloseableIterator[(Option[String], InputStream)] =
113       PathUtils.getUncompressedExtension(filePath.getName).toLowerCase(Locale.US) match {
114         case ArchiveStreamFactory.TAR =>
115           val archive: ArchiveInputStream[_ <: ArchiveEntry] =
116             new ArchiveStreamFactory().createArchiveInputStream(ArchiveStreamFactory.TAR, stream)
117           new ArchiveFileIterator(archive, filePath.toString)
118 
119         case ArchiveStreamFactory.ZIP | ArchiveStreamFactory.JAR =>
120           // we have to read the bytes into memory to get random access reads
121           // note: stream is closed in super class
122           val bytes = new SeekableInMemoryByteChannel(IOUtils.toByteArray(stream))
123           new ZipFileIterator(new ZipFile(bytes), filePath.toString)
124 
125         case _ =>
126           CloseableIterator.single(None -> stream, stream.close())
127       }
128 
129     val all = streams.flatMap { case (name, is) =>
130       converter.process(is, name.map(ec).getOrElse(defaultEc))
131     }
132     val iter = filter match {
133       case Some(f) => all.filter(f.evaluate)
134       case None    => all
135     }
136 
137     val featureReader = if (retypedSpec != null) {
138       val retypedSft = SimpleFeatureTypes.createType(sft.getTypeName, retypedSpec)
139       val reader = new DelegateSimpleFeatureReader(sft, new DelegateSimpleFeatureIterator(iter.asJava))
140       new ReTypeFeatureReader(reader, retypedSft)
141     } else {
142       new DelegateSimpleFeatureReader(sft, new DelegateSimpleFeatureIterator(iter.asJava))
143     }
144 
145     logger.debug(s"Initialized record reader on split ${filePath.toString} with " +
146       s"type name ${sft.getTypeName} and convert conf $confStr")
147 
148     new Iterator[SimpleFeature] with Closeable {
149       override def hasNext: Boolean = featureReader.hasNext
150       override def next(): SimpleFeature = featureReader.next
151       override def close(): Unit = {
152         CloseWithLogging(featureReader)
153         CloseWithLogging(iter)
154         CloseWithLogging(converter)
155       }
156     }
157   }
158 
159   class MapReduceCounter(counter: Counter) extends com.codahale.metrics.Counter {
160     override def inc(): Unit = counter.increment(1)
161     override def inc(n: Long): Unit = counter.increment(n)
162     override def dec(): Unit = counter.increment(-1)
163     override def dec(n: Long): Unit = counter.increment(-1 * n)
164     override def getCount: Long = counter.getValue
165   }
