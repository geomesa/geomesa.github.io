24 @DescribeProcess(
25   title = "Geomesa-enabled Proximity Search",
26   description = "Performs a proximity search on a Geomesa feature collection using another feature collection as input"
27 )
28 class ProximitySearchProcess extends GeoMesaProcess with LazyLogging {
29 
30   @DescribeResult(description = "Output feature collection")
31   def execute(
32                @DescribeParameter(
33                  name = "inputFeatures",
34                  description = "Input feature collection that defines the proximity search")
35                inputFeatures: SimpleFeatureCollection,
36 
37                @DescribeParameter(
38                  name = "dataFeatures",
39                  description = "The data set to query for matching features")
40                dataFeatures: SimpleFeatureCollection,
41 
42                @DescribeParameter(
43                  name = "bufferDistance",
44                  description = "Buffer size in meters")
45                bufferDistance: java.lang.Double
46 
47                ): SimpleFeatureCollection = {
48 
49     logger.debug(s"Attempting Geomesa Proximity Search on collection type ${dataFeatures.getClass.getName}")
50 
51     val visitor = new ProximityVisitor(inputFeatures, dataFeatures, bufferDistance.doubleValue())
52     GeoMesaFeatureCollection.visit(dataFeatures, visitor)
53     visitor.getResult.results
54   }
55 }
56 
57 class ProximityVisitor(inputFeatures: SimpleFeatureCollection,
58                        dataFeatures: SimpleFeatureCollection,
59                        bufferInMeters: Double) extends GeoMesaProcessVisitor with LazyLogging {
60 
61   import org.locationtech.geomesa.filter.{ff, mergeFilters, orFilters}
62 
63   private val dwithin = {
64     val geomProperty = ff.property(dataFeatures.getSchema.getGeometryDescriptor.getName)
65     val geomFilters = SelfClosingIterator(inputFeatures.features).map { sf =>
66       ff.dwithin(geomProperty, ff.literal(sf.getDefaultGeometry), bufferInMeters, "meters")
67     }
68     orFilters(geomFilters.toSeq)
69   }
70 
71   // normally handled in our query planner, but we are going to use the filter directly here
72   private lazy val manualFilter = FastFilterFactory.optimize(dataFeatures.getSchema, dwithin)
73   private val manualVisitResults = new ListFeatureCollection(dataFeatures.getSchema)
74 
75   private var resultCalc = FeatureResult(manualVisitResults)
76 
77   // non-optimized visit
78   // here we use degrees for our filters since we are manually evaluating them.
79   override def visit(feature: Feature): Unit = {
80     val sf = feature.asInstanceOf[SimpleFeature]
81     if (manualFilter.evaluate(sf)) {
82       manualVisitResults.add(sf)
83     }
84   }
85 
86   override def getResult: FeatureResult = resultCalc
87 
88   override def execute(source: SimpleFeatureSource, query: Query): Unit = {
89     logger.debug(s"Running Geomesa Proximity Search on source type ${source.getClass.getName}")
90     val combinedFilter = mergeFilters(query.getFilter, dwithin)
91     resultCalc = FeatureResult(source.getFeatures(combinedFilter))
92   }
93 }
