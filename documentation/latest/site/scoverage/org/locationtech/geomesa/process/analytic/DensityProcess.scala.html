42   @throws(classOf[ProcessException])
43   @DescribeResult(name = "result", description = "Output raster")
44   def execute(
45       @DescribeParameter(name = "data", description = "Input features")
46       obsFeatures: SimpleFeatureCollection,
47       @DescribeParameter(name = "radiusPixels", description = "Radius of the density kernel in pixels")
48       argRadiusPixels: Integer,
49       @DescribeParameter(name = "geomAttr", description = "Name of the geometry attribute to render", min = 0, max = 1)
50       argGeomAttr: String,
51       @DescribeParameter(name = "weightAttr", description = "Name of the attribute to use for data point weight", min = 0, max = 1)
52       argWeightAttr: String,
53       @DescribeParameter(name = "outputBBOX", description = "Bounding box of the output")
54       argOutputEnv: ReferencedEnvelope,
55       @DescribeParameter(name = "outputWidth", description = "Width of output raster in pixels")
56       argOutputWidth: Integer,
57       @DescribeParameter(name = "outputHeight", description = "Height of output raster in pixels")
58       argOutputHeight: Integer,
59       monitor: ProgressListener): GridCoverage2D = {
60 
61     val pixels = Option(argRadiusPixels).map(_.intValue).getOrElse(DefaultRadiusPixels)
62 
63     // buffer our calculations based on the pixel radius to avoid edge artifacts
64     val outputWidth  = argOutputWidth + 2 * pixels
65     val outputHeight = argOutputHeight + 2 * pixels
66     val bufferWidth  = pixels * argOutputEnv.getWidth / argOutputWidth
67     val bufferHeight = pixels * argOutputEnv.getHeight / argOutputHeight
68     val envelope = new ReferencedEnvelope(argOutputEnv)
69     envelope.expandBy(bufferWidth, bufferHeight)
70 
71     val decode = DensityScan.decodeResult(envelope, outputWidth, outputHeight)
72 
73     val heatMap = new HeatmapSurface(pixels, envelope, outputWidth, outputHeight)
74 
75     try {
76       WithClose(obsFeatures.features()) { features =>
77         while (features.hasNext) {
78           val pts = decode(features.next())
79           while (pts.hasNext) {
80             val (x, y, weight) = pts.next()
81             heatMap.addPoint(x, y, weight)
82           }
83         }
84       }
85     } catch {
86       case e: Exception => throw new ProcessException("Error processing heatmap", e)
87     }
88 
89     val heatMapGrid = DensityProcess.flipXY(heatMap.computeSurface)
90 
91     // create the raster from our unbuffered envelope and discard the buffered pixels in our final image
92     val raster = RasterFactory.createBandedRaster(DataBuffer.TYPE_FLOAT, argOutputWidth, argOutputHeight, 1, null)
93 
94     var i, j = pixels
95     while (j < heatMapGrid.length - pixels) {
96       val row = heatMapGrid(j)
97       while (i < row.length - pixels) {
98         raster.setSample(i - pixels, j - pixels, 0, row(i))
99         i += 1
100       }
101       j += 1
102       i = pixels
103     }
104 
105     val gcf = CoverageFactoryFinder.getGridCoverageFactory(GeoTools.getDefaultHints)
106     gcf.create("Process Results", raster, argOutputEnv)
107   }

109   /**

111    * input data of the process involved in rendering. In this process this method is used to:
112    * <ul>
113    * <li>determine the extent & CRS of the output grid
114    * <li>expand the query envelope to ensure stable surface generation
115    * <li>modify the query hints to ensure point features are returned
116    * </ul>

118    * parameter list of the `execute` method, even if they are not used there.
119    *

121    * @param targetQuery the query used against the data source
122    * @param targetGridGeometry the grid geometry of the destination image
123    * @return The transformed query

125   @throws(classOf[ProcessException])
126   def invertQuery(
127       @DescribeParameter(name = "radiusPixels", description = "Radius to use for the kernel", min = 0, max = 1)
128       argRadiusPixels: Integer,
129       @DescribeParameter(name = "geomAttr", description = "Name of the geometry attribute to render", min = 0, max = 1)
130       argGeomAttr: String,

132       argWeightAttr: String,

134       argOutputEnv: ReferencedEnvelope,
135       @DescribeParameter(name = "outputWidth", description = "Width of the output raster")
136       argOutputWidth: Integer,
137       @DescribeParameter(name = "outputHeight", description = "Height of the output raster")
138       argOutputHeight: Integer,
139       targetQuery: Query,
140       targetGridGeometry: GridGeometry): Query = {
141     if (argOutputWidth == null || argOutputHeight == null) {
142       throw new IllegalArgumentException("outputWidth and/or outputHeight not specified")
143     } else if (argOutputWidth < 0 || argOutputHeight < 0) {
144       throw new IllegalArgumentException("outputWidth and outputHeight must both be positive")
145     }
146 
147     val pixels = Option(argRadiusPixels).map(_.intValue).getOrElse(DefaultRadiusPixels)
148 
149     // buffer our calculations based on the pixel radius to avoid edge artifacts
150     val outputWidth  = argOutputWidth + 2 * pixels
151     val outputHeight = argOutputHeight + 2 * pixels
152 
153     val bufferWidth  = pixels * argOutputEnv.getWidth / argOutputWidth
154     val bufferHeight = pixels * argOutputEnv.getHeight / argOutputHeight
155 
156     val envelope = new ReferencedEnvelope(argOutputEnv)
157     envelope.expandBy(bufferWidth, bufferHeight)
158 
159     val filter = {
160       val buf = math.max(bufferWidth, bufferHeight)
161       targetQuery.getFilter.accept(new BBOXExpandingFilterVisitor(buf, buf, buf, buf), null).asInstanceOf[Filter]
162     }
163 
164     val invertedQuery = new Query(targetQuery)
165     invertedQuery.setFilter(filter)
166     invertedQuery.setProperties(null)
167     invertedQuery.getHints.put(QueryHints.DENSITY_BBOX, envelope)
168     invertedQuery.getHints.put(QueryHints.DENSITY_WIDTH, outputWidth)
169     invertedQuery.getHints.put(QueryHints.DENSITY_HEIGHT, outputHeight)
170     if (argGeomAttr != null) {
171       invertedQuery.getHints.put(QueryHints.DENSITY_GEOM, argGeomAttr)
172     }
173     if (argWeightAttr != null) {
174       invertedQuery.getHints.put(QueryHints.DENSITY_WEIGHT, argWeightAttr)
175     }
176     invertedQuery
177   }

184   /**
185    * Flips an XY matrix along the X=Y axis, and inverts the Y axis. Used to convert from
186    * "map orientation" into the "image orientation" used by GridCoverageFactory. The surface
187    * interpolation is done on an XY grid, with Y=0 being the bottom of the space. GridCoverages
188    * are stored in an image format, in a YX grid with Y=0 being the top.
189    *
190    * @param grid the grid to flip
191    * @return the flipped grid
192    */
193   def flipXY(grid: Array[Array[Float]]): Array[Array[Float]] = {
194     val length_x = grid.length
195     val length_y = grid(0).length
196 
197     val res = Array.fill(length_y,length_x)(0f)
198 
199     for ( x <- 0 until length_x ; y <- 0 until length_y ) {
200       val x1 = length_y - 1 - y
201       val y1 = x
202       res(x1)(y1) = grid(x)(y)
203     }
204 
205     res
206   }
