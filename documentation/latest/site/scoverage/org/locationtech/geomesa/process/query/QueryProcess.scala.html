27 @DescribeProcess(
28   title = "Geomesa Query",
29   description = "Performs a Geomesa optimized query using spatiotemporal indexes"
30 )
31 class QueryProcess extends GeoMesaProcess with LazyLogging {
32 
33   @DescribeResult(description = "Output feature collection")
34   def execute(
35                @DescribeParameter(
36                  name = "features",
37                  description = "The feature set on which to query")
38                features: SimpleFeatureCollection,
39 
40                @DescribeParameter(
41                  name = "filter",
42                  min = 0,
43                  description = "The filter to apply to the feature collection")
44                filter: Filter,
45 
46                @DescribeParameter(
47                  name = "properties",
48                  min = 0,
49                  max = 128,
50                  collectionType = classOf[String],
51                  description = "The properties/transforms to apply to the feature collection")
52                properties: java.util.List[String] = null
53 
54              ): SimpleFeatureCollection = {
55 
56     logger.debug("Attempting Geomesa query on type " + features.getClass.getName)
57 
58     val propsArray = Option(properties).map(_.toArray(Array.empty[String])).filter(_.length > 0).orNull
59 
60     val visitor = new QueryVisitor(features, Option(filter).getOrElse(Filter.INCLUDE), propsArray)
61     GeoMesaFeatureCollection.visit(features, visitor)
62     visitor.getResult.results
63   }
64 }
65 
66 class QueryVisitor(features: SimpleFeatureCollection, filter: Filter, properties: Array[String])
67     extends GeoMesaProcessVisitor with LazyLogging {
68 
69   import QueryHints.RichHints
70 
71   private val (sft, transformFeature) = if (properties == null) { (features.getSchema, null) } else {
72     val original = features.getSchema
73     val query = new Query(original.getTypeName, Filter.INCLUDE)
74     if (properties != null) {
75       query.setPropertyNames(properties: _*)
76     }
77     QueryRunner.configureQuery(original, query).getHints.getTransform match {
78       case None => (original, null)
79       case Some((tdefs, tsft)) => (tsft, TransformSimpleFeature(original, tsft, tdefs))
80     }
81   }
82 
83   private val retype: SimpleFeature => SimpleFeature =
84     if (transformFeature == null) {
85       sf => sf
86     } else {
87       sf => {
88         transformFeature.setFeature(sf)
89         ScalaSimpleFeature.copy(transformFeature)
90       }
91     }
92 
93   // normally handled in our query planner, but we are going to use the filter directly here
94   private lazy val manualFilter = FastFilterFactory.optimize(features.getSchema, filter)
95   private val manualVisitResults = new ListFeatureCollection(sft)
96   private var resultCalc = FeatureResult(manualVisitResults)
97 
98   // non-optimized visit
99   override def visit(feature: Feature): Unit = {
100     val sf = feature.asInstanceOf[SimpleFeature]
101     if (manualFilter.evaluate(sf)) {
102       manualVisitResults.add(retype(sf))
103     }
104   }
105 
106   override def getResult: FeatureResult = resultCalc
107 
108   override def execute(source: SimpleFeatureSource, query: Query): Unit = {
109     logger.debug(s"Running Geomesa query on source type ${source.getClass.getName}")
110     query.setFilter(org.locationtech.geomesa.filter.mergeFilters(query.getFilter, filter))
111     if (properties != null && properties.length > 0) {
112       if (query.getProperties != Query.ALL_PROPERTIES) {
113         logger.warn(s"Overriding inner query's properties (${query.getProperties}) " +
114             s"with properties/transforms ${properties.mkString(",")}.")
115       }
116       query.setPropertyNames(properties: _*)
117     }
118     resultCalc = FeatureResult(source.getFeatures(query))
119   }
120 }
