36   description = "Performs a search based on a route"

40   /**
41     * Finds features around a route that are heading along the route and not just crossing over it
42     *
43     * @param features input features to query
44     * @param routes routes to match
45     * @param bufferSize buffer around the routes to search, in meters
46     * @param headingThreshold threshold to match the heading of the route, in degrees
47     * @param routeGeomField geometry attribute in the route features that contains the route, optional
48     *                       will use default geometry if not provided
49     * @param geomField geometry attribute in input features to match with the route, optional
50     *                  will use default geometry if not provided
51     * @param bidirectional consider the direction of the route or just the path of the route
52     * @param headingField heading attribute in input features, required unless input geometries are linestrings
53     * @return
54     */
55   @DescribeResult(description = "Output feature collection")
56   def execute(
57               @DescribeParameter(name = "features", description = "Input feature collection to query ")
58               features: SimpleFeatureCollection,
59               @DescribeParameter(name = "routes", description = "Routes to search along. Features must have a geometry of LineString")
60               routes: SimpleFeatureCollection,
61               @DescribeParameter(name = "bufferSize", description = "Buffer size (in meters) to search around the route")
62               bufferSize: java.lang.Double,
63               @DescribeParameter(name = "headingThreshold", description = "Threshold for comparing headings, in degrees")
64               headingThreshold: java.lang.Double,
65               @DescribeParameter(name = "routeGeomField", description = "Attribute that will be examined for routes to match. Must be a LineString", min = 0)
66               routeGeomField: String,
67               @DescribeParameter(name = "geomField", description = "Attribute that will be examined for route matching. Must be a LineString", min = 0)
68               geomField: String,
69               @DescribeParameter(name = "bidirectional", description = "Match the route direction or match just the route path", min = 0)
70               bidirectional: java.lang.Boolean,
71               @DescribeParameter(name = "headingField", description = "Attribute that will be examined for heading in the input features. If not provided, input features geometries must be LineStrings", min = 0)
72               headingField: String
73              ): SimpleFeatureCollection = {
74 
75     logger.debug(s"Route searching on collection type ${features.getClass.getName}")
76 
77     val bi = Option(bidirectional).getOrElse(java.lang.Boolean.FALSE)
78 
79     val sft = features.getSchema
80 
81     // pull out attributes and validate inputs
82 
83     val (geomAttribute, isPoints) = {
84       val name = Option(geomField).getOrElse(sft.getGeometryDescriptor.getLocalName)
85       val descriptor = sft.getDescriptor(name)
86       if (descriptor == null) {
87         throw new IllegalArgumentException(s"Geometry field '$name' does not exist in input feature collection")
88       }
89       val binding = descriptor.getType.getBinding
90       val isPoints = classOf[Point].isAssignableFrom(binding)
91       if (!isPoints && !classOf[LineString].isAssignableFrom(binding)) {
92         throw new IllegalArgumentException(s"Geometry field '$name' must be a Point or LineString")
93       }
94       (name, isPoints)
95     }
96 
97     if (headingField == null && isPoints) {
98       throw new IllegalArgumentException("Heading must be specified unless input feature collection geometry is a LineString")
99     } else if (headingField != null && sft.indexOf(headingField) == -1) {
100       throw new IllegalArgumentException(s"Heading field '$headingField' does not exist in input feature collection")
101     }
102 
103     // extract the route geometries
104     val routeGeoms = {
105       val sft = routes.getSchema
106       val name = Option(routeGeomField).getOrElse(sft.getGeometryDescriptor.getLocalName)
107       val index = sft.indexOf(name)
108       if (index == -1) {
109         throw new IllegalArgumentException(s"Geometry field '$name' does not exist in route feature collection")
110       }
111       if (!classOf[LineString].isAssignableFrom(sft.getDescriptor(index).getType.getBinding)) {
112         throw new IllegalArgumentException(s"Route geometry field '$name' must be a LineString")
113       }
114       SelfClosingIterator(routes).map(_.getAttribute(index).asInstanceOf[LineString]).toSeq
115     }
116 
117     val visitor = new RouteVisitor(sft, routeGeoms, bufferSize, headingThreshold, bi,
118       geomAttribute, isPoints, Option(headingField))
119     GeoMesaFeatureCollection.visit(features, visitor)
120     visitor.getResult.results
121   }
122 }

124 class RouteVisitor(sft: SimpleFeatureType,
125                    routes: Seq[LineString],
126                    routeBuffer: Double,
127                    threshold: Double,
128                    bidirectional: Boolean,
129                    geomAttribute: String,
130                    isPoints: Boolean,
131                    headingAttribute: Option[String]) extends GeoMesaProcessVisitor with LazyLogging {
132 
133   private var resultCalc: FeatureResult = _
134 
135   private val routeFilter =
136     orFilters(routes.map(ls => ff.dwithin(ff.property(geomAttribute), ff.literal(ls), routeBuffer, "meters")))
137 
138   // for manual check, rewrite the filter to handle meters
139   // normally handled in our query planner, but we are going to use the filter directly here
140   private lazy val manualRouteFilter = FastFilterFactory.optimize(sft, routeFilter)
141 
142   // for collecting results manually
143   private var manualCollection: ListFeatureCollection = _
144 
145   override def getResult: FeatureResult = resultCalc
146 
147   // manually called for non-accumulo feature collections
148   override def visit(feature: Feature): Unit = {
149     val sf = feature.asInstanceOf[SimpleFeature]
150 
151     if (manualCollection == null) {
152       manualCollection = new ListFeatureCollection(sf.getFeatureType)
153       resultCalc = FeatureResult(matchRoutes(manualCollection))
154     }
155 
156     if (manualRouteFilter.evaluate(sf)) {
157       manualCollection.add(sf)
158     }
159   }
160 
161   override def execute(source: SimpleFeatureSource, query: Query): Unit = {
162     logger.debug(s"Visiting source type: ${source.getClass.getName}")
163     val result = if (routes.isEmpty) { source.getFeatures(Filter.EXCLUDE) } else {
164       import org.locationtech.geomesa.filter._
165       val filter = if (query == null || query.getFilter == Filter.INCLUDE) { routeFilter } else {
166         andFilters(Seq(query.getFilter, routeFilter))
167       }
168       matchRoutes(source.getFeatures(filter))
169     }
170     resultCalc = FeatureResult(result)
171   }
172 
173   /**
174     * Filters a feature collection by comparing routes
175     *
176     * @param input input feature collection
177     * @return
178     */
179   private def matchRoutes(input: SimpleFeatureCollection): SimpleFeatureCollection = {
180     val sft = input.getSchema
181     val geomIndex = sft.indexOf(geomAttribute)
182     val headingIndex = headingAttribute.map(sft.indexOf)
183 
184     logger.debug(s"Searching routes: ${routes.map(WKTUtils.write).mkString(", ")}")
185     logger.debug(s"Buffer (meters): $routeBuffer")
186     logger.debug(s"Geometry attribute: $geomAttribute")
187     logger.debug(s"Heading attribute: ${headingAttribute.getOrElse("none")}")
188 
189     RouteVisitor.matchRoutes(input, routes, geomIndex, isPoints, headingIndex, threshold, bidirectional)
190   }
191 }

193 object RouteVisitor {
194 
195   /**
196     * Match features to routes.
197     *
198     * @param input input features
199     * @param routes routes to match against
200     * @param geomIndex geometry attribute index of the input features
201     * @param isPoints are the input feature geometries points (or linestrings)
202     * @param headingIndex attribute index of the heading in the input features, not required if they are linestrings
203     * @param threshold threshold to consider when matching route heading, in degrees
204     * @param bidirectional match the route direction or just the path
205     * @return
206     */
207   def matchRoutes(input: SimpleFeatureCollection,
208                   routes: Seq[LineString],
209                   geomIndex: Int,
210                   isPoints: Boolean,
211                   headingIndex: Option[Int],
212                   threshold: Double,
213                   bidirectional: Boolean): SimpleFeatureCollection = {
214 
215     // just in case the features are operated on in parallel...
216     val calculator = new ThreadLocal[GeodeticCalculator] {
217       override def initialValue(): GeodeticCalculator = new GeodeticCalculator
218     }
219     val headingCache = {
220       import scala.collection.JavaConverters._
221       new ConcurrentHashMap[(LineString, Int), Double].asScala
222     }
223 
224     // gets the heading for an input feature
225     val getFeatureHeading: (SimpleFeature) => Double = headingIndex match {
226       case Some(index) => (sf) => FastConverter.convert(sf.getAttribute(index), classOf[Double])
227       case None =>
228       (sf) => {
229         val geom = sf.getAttribute(geomIndex).asInstanceOf[LineString]
230         // use the last two points in the line to find it's current heading - we don't care about history
231         getRouteHeading(geom, geom.getNumPoints - 2, calculator.get)
232       }
233     }
234 
235     // gets the point from a geometry that we want to match against routes
236     val getComparisonPoint: (Geometry) => Point = if (isPoints) {
237       (g) => g.asInstanceOf[Point]
238     } else {
239       // match the most recent point
240       (g) => g.asInstanceOf[LineString].getEndPoint
241     }
242 
243     // matches a feature against the routes
244     def matchRoute(sf: SimpleFeature): Boolean = {
245       val geom = sf.getAttribute(geomIndex).asInstanceOf[Geometry]
246       val (route, closestLocation) = getClosestRoute(getComparisonPoint(geom), routes)
247       val routeHeading = headingCache.getOrElseUpdate((route, closestLocation),
248         getRouteHeading(route, closestLocation, calculator.get))
249       val featureHeading = getFeatureHeading(sf)
250 
251       // compare the headings
252       var diff = math.abs(routeHeading - featureHeading)
253       // compass problem - correct for headings on either side of 0/360
254       if (diff > 180.0) {
255         diff = math.abs(diff - 360.0)
256       }
257       diff <= threshold || (bidirectional && math.abs(diff - 180.0) <= threshold)
258     }
259 
260     // delegate feature collection that filters the results based on route matching
261     new DecoratingSimpleFeatureCollection(input) {
262       override def features(): SimpleFeatureIterator = new SimpleFeatureIterator() {
263         private val delegate = CloseableIterator(input.features()).filter(matchRoute)
264         override def next(): SimpleFeature = delegate.next()
265         override def hasNext: Boolean = delegate.hasNext
266         override def close(): Unit = delegate.close()
267       }
268     }
269   }
270 
271   /**
272     * Gets the closest route to the input point
273     *
274     * @param geom input point
275     * @param routes routes to check
276     * @return route and index of start of line segment in the route that is closest to the point
277     */
278   private def getClosestRoute(geom: Point, routes: Seq[LineString]): (LineString, Int) = {
279     def closestPoint(ls: LineString): (Double, LineString, Int) = {
280       val op = new DistanceOp(ls, geom)
281       (op.distance(), ls, op.nearestLocations()(0).getSegmentIndex)
282     }
283     val (_, route, location) = routes.map(closestPoint).minBy(_._1)
284     (route, location)
285   }
286 
287   /**
288     * Gets the heading of a route at a particular line segment
289     *
290     * @param route route
291     * @param index index of the start of the line segment we want to consider
292     * @param calculator geodetic calculator instance
293     * @return heading between 0-360
294     */
295   private def getRouteHeading(route: LineString, index: Int, calculator: GeodeticCalculator): Double = {
296     val coords = route.getCoordinates
297     // the closest point falls somewhere along this line segment - calculate the heading for the segment
298     val segment0 = coords(index)
299     val segment1 = coords(index + 1)
300     calculator.setStartingGeographicPoint(segment0.x, segment0.y)
301     calculator.setDestinationGeographicPoint(segment1.x, segment1.y)
302     val azimuth = calculator.getAzimuth
303     // azimuth is between -180 and +180, where 0 corresponds to north - convert to degrees 0-360
304     if (azimuth < 0.0) {
305       360.0 + azimuth
306     } else {
307       azimuth
308     }
309   }
