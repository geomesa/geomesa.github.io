36   title = "Route Search",
37   description = "Performs a search based on a route"

39 class RouteSearchProcess extends GeoMesaProcess with LazyLogging {
40 
41   /**
42     * Finds features around a route that are heading along the route and not just crossing over it
43     *
44     * @param features input features to query
45     * @param routes routes to match
46     * @param bufferSize buffer around the routes to search, in meters
47     * @param headingThreshold threshold to match the heading of the route, in degrees
48     * @param routeGeomField geometry attribute in the route features that contains the route, optional
49     *                       will use default geometry if not provided
50     * @param geomField geometry attribute in input features to match with the route, optional
51     *                  will use default geometry if not provided
52     * @param bidirectional consider the direction of the route or just the path of the route
53     * @param headingField heading attribute in input features, required unless input geometries are linestrings
54     * @return
55     */
56   @DescribeResult(description = "Output feature collection")
57   def execute(
58               @DescribeParameter(name = "features", description = "Input feature collection to query ")
59               features: SimpleFeatureCollection,
60               @DescribeParameter(name = "routes", description = "Routes to search along. Features must have a geometry of LineString")
61               routes: SimpleFeatureCollection,
62               @DescribeParameter(name = "bufferSize", description = "Buffer size (in meters) to search around the route")
63               bufferSize: java.lang.Double,
64               @DescribeParameter(name = "headingThreshold", description = "Threshold for comparing headings, in degrees")
65               headingThreshold: java.lang.Double,
66               @DescribeParameter(name = "routeGeomField", description = "Attribute that will be examined for routes to match. Must be a LineString", min = 0)
67               routeGeomField: String,
68               @DescribeParameter(name = "geomField", description = "Attribute that will be examined for route matching. Must be a LineString", min = 0)
69               geomField: String,
70               @DescribeParameter(name = "bidirectional", description = "Match the route direction or match just the route path", min = 0)
71               bidirectional: java.lang.Boolean,
72               @DescribeParameter(name = "headingField", description = "Attribute that will be examined for heading in the input features. If not provided, input features geometries must be LineStrings", min = 0)
73               headingField: String
74              ): SimpleFeatureCollection = {
75 
76     logger.debug(s"Route searching on collection type ${features.getClass.getName}")
77 
78     val bi = Option(bidirectional).getOrElse(java.lang.Boolean.FALSE)
79 
80     val sft = features.getSchema
81 
82     // pull out attributes and validate inputs
83 
84     val (geomAttribute, isPoints) = {
85       val name = Option(geomField).getOrElse(sft.getGeometryDescriptor.getLocalName)
86       val descriptor = sft.getDescriptor(name)
87       if (descriptor == null) {
88         throw new IllegalArgumentException(s"Geometry field '$name' does not exist in input feature collection")
89       }
90       val binding = descriptor.getType.getBinding
91       val isPoints = classOf[Point].isAssignableFrom(binding)
92       if (!isPoints && !classOf[LineString].isAssignableFrom(binding)) {
93         throw new IllegalArgumentException(s"Geometry field '$name' must be a Point or LineString")
94       }
95       (name, isPoints)
96     }
97 
98     if (headingField == null && isPoints) {
99       throw new IllegalArgumentException("Heading must be specified unless input feature collection geometry is a LineString")
100     } else if (headingField != null && sft.indexOf(headingField) == -1) {
101       throw new IllegalArgumentException(s"Heading field '$headingField' does not exist in input feature collection")
102     }
103 
104     // extract the route geometries
105     val routeGeoms = {
106       val sft = routes.getSchema
107       val name = Option(routeGeomField).getOrElse(sft.getGeometryDescriptor.getLocalName)
108       val index = sft.indexOf(name)
109       if (index == -1) {
110         throw new IllegalArgumentException(s"Geometry field '$name' does not exist in route feature collection")
111       }
112       if (!classOf[LineString].isAssignableFrom(sft.getDescriptor(index).getType.getBinding)) {
113         throw new IllegalArgumentException(s"Route geometry field '$name' must be a LineString")
114       }
115       CloseableIterator(routes.features()).map(_.getAttribute(index).asInstanceOf[LineString]).toList
116     }
117 
118     val visitor = new RouteVisitor(sft, routeGeoms, bufferSize, headingThreshold, bi,
119       geomAttribute, isPoints, Option(headingField))
120     GeoMesaFeatureCollection.visit(features, visitor)
121     visitor.getResult.results
122   }
123 }

125 class RouteVisitor(sft: SimpleFeatureType,
126                    routes: Seq[LineString],
127                    routeBuffer: Double,
128                    threshold: Double,
129                    bidirectional: Boolean,
130                    geomAttribute: String,
131                    isPoints: Boolean,
132                    headingAttribute: Option[String]) extends GeoMesaProcessVisitor with LazyLogging {
133 
134   private var resultCalc: FeatureResult = _
135 
136   private val routeFilter =
137     orFilters(routes.map(ls => ff.dwithin(ff.property(geomAttribute), ff.literal(ls), routeBuffer, "meters")))
138 
139   // for manual check, rewrite the filter to handle meters
140   // normally handled in our query planner, but we are going to use the filter directly here
141   private lazy val manualRouteFilter = FastFilterFactory.optimize(sft, routeFilter)
142 
143   // for collecting results manually
144   private var manualCollection: ListFeatureCollection = _
145 
146   override def getResult: FeatureResult = resultCalc
147 
148   // manually called for non-accumulo feature collections
149   override def visit(feature: Feature): Unit = {
150     val sf = feature.asInstanceOf[SimpleFeature]
151 
152     if (manualCollection == null) {
153       manualCollection = new ListFeatureCollection(sf.getFeatureType)
154       resultCalc = FeatureResult(matchRoutes(manualCollection))
155     }
156 
157     if (manualRouteFilter.evaluate(sf)) {
158       manualCollection.add(sf)
159     }
160   }
161 
162   override def execute(source: SimpleFeatureSource, query: Query): Unit = {
163     logger.debug(s"Visiting source type: ${source.getClass.getName}")
164     val result = if (routes.isEmpty) { source.getFeatures(Filter.EXCLUDE) } else {
165       import org.locationtech.geomesa.filter._
166       val filter = if (query == null || query.getFilter == Filter.INCLUDE) { routeFilter } else {
167         andFilters(Seq(query.getFilter, routeFilter))
168       }
169       matchRoutes(source.getFeatures(filter))
170     }
171     resultCalc = FeatureResult(result)
172   }
173 
174   /**
175     * Filters a feature collection by comparing routes
176     *
177     * @param input input feature collection
178     * @return
179     */
180   private def matchRoutes(input: SimpleFeatureCollection): SimpleFeatureCollection = {
181     val sft = input.getSchema
182     val geomIndex = sft.indexOf(geomAttribute)
183     val headingIndex = headingAttribute.map(sft.indexOf)
184 
185     logger.debug(s"Searching routes: ${routes.map(WKTUtils.write).mkString(", ")}")
186     logger.debug(s"Buffer (meters): $routeBuffer")
187     logger.debug(s"Geometry attribute: $geomAttribute")
188     logger.debug(s"Heading attribute: ${headingAttribute.getOrElse("none")}")
189 
190     RouteVisitor.matchRoutes(input, routes, geomIndex, isPoints, headingIndex, threshold, bidirectional)
191   }

194 object RouteVisitor {
195 
196   /**
197     * Match features to routes.
198     *
199     * @param input input features
200     * @param routes routes to match against
201     * @param geomIndex geometry attribute index of the input features
202     * @param isPoints are the input feature geometries points (or linestrings)
203     * @param headingIndex attribute index of the heading in the input features, not required if they are linestrings
204     * @param threshold threshold to consider when matching route heading, in degrees
205     * @param bidirectional match the route direction or just the path
206     * @return
207     */
208   def matchRoutes(input: SimpleFeatureCollection,
209                   routes: Seq[LineString],
210                   geomIndex: Int,
211                   isPoints: Boolean,
212                   headingIndex: Option[Int],
213                   threshold: Double,
214                   bidirectional: Boolean): SimpleFeatureCollection = {
215 
216     // just in case the features are operated on in parallel...
217     val calculator = new ThreadLocal[GeodeticCalculator] {
218       override def initialValue(): GeodeticCalculator = new GeodeticCalculator
219     }
220     val headingCache = {
221       import scala.collection.JavaConverters._
222       new ConcurrentHashMap[(LineString, Int), Double].asScala
223     }
224 
225     // gets the heading for an input feature
226     val getFeatureHeading: (SimpleFeature) => Double = headingIndex match {
227       case Some(index) => (sf) => FastConverter.convert(sf.getAttribute(index), classOf[Double])
228       case None =>
229       (sf) => {
230         val geom = sf.getAttribute(geomIndex).asInstanceOf[LineString]
231         // use the last two points in the line to find it's current heading - we don't care about history
232         getRouteHeading(geom, geom.getNumPoints - 2, calculator.get)
233       }
234     }
235 
236     // gets the point from a geometry that we want to match against routes
237     val getComparisonPoint: (Geometry) => Point = if (isPoints) {
238       (g) => g.asInstanceOf[Point]
239     } else {
240       // match the most recent point
241       (g) => g.asInstanceOf[LineString].getEndPoint
242     }
243 
244     // matches a feature against the routes
245     def matchRoute(sf: SimpleFeature): Boolean = {
246       val geom = sf.getAttribute(geomIndex).asInstanceOf[Geometry]
247       val (route, closestLocation) = getClosestRoute(getComparisonPoint(geom), routes)
248       val routeHeading = headingCache.getOrElseUpdate((route, closestLocation),
249         getRouteHeading(route, closestLocation, calculator.get))
250       val featureHeading = getFeatureHeading(sf)
251 
252       // compare the headings
253       var diff = math.abs(routeHeading - featureHeading)
254       // compass problem - correct for headings on either side of 0/360
255       if (diff > 180.0) {
256         diff = math.abs(diff - 360.0)
257       }
258       diff <= threshold || (bidirectional && math.abs(diff - 180.0) <= threshold)
259     }
260 
261     // delegate feature collection that filters the results based on route matching
262     new DecoratingSimpleFeatureCollection(input) {
263       override def features(): SimpleFeatureIterator = new SimpleFeatureIterator() {
264         private val delegate = CloseableIterator(input.features()).filter(matchRoute)
265         override def next(): SimpleFeature = delegate.next()
266         override def hasNext: Boolean = delegate.hasNext
267         override def close(): Unit = delegate.close()
268       }
269     }
270   }
271 
272   /**
273     * Gets the closest route to the input point
274     *
275     * @param geom input point
276     * @param routes routes to check
277     * @return route and index of start of line segment in the route that is closest to the point
278     */
279   private def getClosestRoute(geom: Point, routes: Seq[LineString]): (LineString, Int) = {
280     def closestPoint(ls: LineString): (Double, LineString, Int) = {
281       val op = new DistanceOp(ls, geom)
282       (op.distance(), ls, op.nearestLocations()(0).getSegmentIndex)
283     }
284     val (_, route, location) = routes.map(closestPoint).minBy(_._1)
285     (route, location)
286   }
287 
288   /**
289     * Gets the heading of a route at a particular line segment
290     *
291     * @param route route
292     * @param index index of the start of the line segment we want to consider
293     * @param calculator geodetic calculator instance
294     * @return heading between 0-360
295     */
296   private def getRouteHeading(route: LineString, index: Int, calculator: GeodeticCalculator): Double = {
297     val coords = route.getCoordinates
298     // the closest point falls somewhere along this line segment - calculate the heading for the segment
299     val segment0 = coords(index)
300     val segment1 = coords(index + 1)
301     calculator.setStartingGeographicPoint(segment0.x, segment0.y)
302     calculator.setDestinationGeographicPoint(segment1.x, segment1.y)
303     val azimuth = calculator.getAzimuth
304     // azimuth is between -180 and +180, where 0 corresponds to north - convert to degrees 0-360
305     if (azimuth < 0.0) {
306       360.0 + azimuth
307     } else {
308       azimuth
309     }
310   }
