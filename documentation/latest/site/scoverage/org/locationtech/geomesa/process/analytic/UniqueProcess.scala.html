27 @DescribeProcess(title = "Geomesa Unique",
28   description = "Finds unique attributes values, optimized for GeoMesa")
29 class UniqueProcess extends GeoMesaProcess with LazyLogging {
30 
31   @DescribeResult(name = "result",
32     description = "Feature collection with an attribute containing the unique values")
33   def execute(
34     @DescribeParameter(name = "features", description = "Input feature collection")
35     features: SimpleFeatureCollection,
36     @DescribeParameter(name = "attribute", description = "Attribute whose unique values are extracted")
37     attribute: String,
38     @DescribeParameter(name = "filter", min = 0, description = "The filter to apply to the feature collection")
39     filter: Filter,
40     @DescribeParameter(name = "histogram", min = 0, description = "Create a histogram of attribute values")
41     histogram: java.lang.Boolean,
42     @DescribeParameter(name = "sort", min = 0, description = "Sort results - allowed to be ASC or DESC")
43     sort: String,
44     @DescribeParameter(name = "sortByCount", min = 0, description = "Sort by histogram counts instead of attribute values")
45     sortByCount: java.lang.Boolean,
46     progressListener: ProgressListener): SimpleFeatureCollection = {
47 
48     val attributeDescriptor = features
49         .getSchema
50         .getAttributeDescriptors
51         .asScala
52         .find(_.getLocalName == attribute)
53         .getOrElse(throw new IllegalArgumentException(s"Attribute $attribute does not exist in feature schema."))
54 
55     val hist = Option(histogram).exists(_.booleanValue)
56     val sortBy = Option(sortByCount).exists(_.booleanValue)
57 
58     val visitor = new AttributeValuesVisitor(features, attributeDescriptor, Option(filter).filter(_ != Filter.INCLUDE), hist)
59     GeoMesaFeatureCollection.visit(features, visitor, progressListener)
60     val uniqueValues = visitor.getResult.attributes
61 
62     val binding = attributeDescriptor.getType.getBinding
63     UniqueProcess.createReturnCollection(uniqueValues, binding, hist, Option(sort), sortBy)
64   }
65 }
66 
67 object UniqueProcess {
68 
69   val SftName = "UniqueValue"
70   val AttributeValue = "value"
71   val AttributeCount = "count"
72 
73   /**
74     * Duplicates output format from geotools UniqueProcess
75     *
76     * @param uniqueValues values
77     * @param binding value binding
78     * @param histogram include counts or just values
79     * @param sort sort
80     * @param sortByCount sort by count or by value
81     * @return
82     */
83   private def createReturnCollection(
84       uniqueValues: Map[AnyRef, Long],
85       binding: Class[_],
86       histogram: Boolean,
87       sort: Option[String],
88       sortByCount: Boolean): SimpleFeatureCollection = {
89 
90     val ft = createUniqueSft(binding, histogram)
91 
92     val result = new ListFeatureCollection(ft)
93 
94     // if sorting was requested do it here, otherwise return results in iterator order
95     val sorted = sort.map { s =>
96       if (sortByCount) {
97         val ordering = if (s.equalsIgnoreCase("desc")) Ordering[Long].reverse else Ordering[Long]
98         uniqueValues.iterator.toList.sortBy(_._2)(ordering)
99       } else {
100         val ordering = if (s.equalsIgnoreCase("desc")) Ordering[String].reverse else Ordering[String]
101         uniqueValues.iterator.toList.sortBy(_._1.toString)(ordering)
102       }
103     }.getOrElse(uniqueValues.iterator)
104 
105     // histogram includes extra 'count' attribute
106     val toSf = if (histogram) (key: AnyRef, value: Long) => {
107       val sf = new ScalaSimpleFeature(ft, UUID.randomUUID().toString)
108       sf.setAttributeNoConvert(0, key)
109       sf.setAttributeNoConvert(1, Long.box(value))
110       sf
111     } else (key: AnyRef, _: Long) => {
112       val sf = new ScalaSimpleFeature(ft, UUID.randomUUID().toString)
113       sf.setAttributeNoConvert(0, key)
114       sf
115     }
116 
117     sorted.foreach { case (key, value) => result.add(toSf(key, value)) }
118 
119     result
120   }
121 
122   /**
123     * Based on geotools UniqueProcess simple feature type
124     *
125     * @param binding class of attribute
126     * @param histogram return counts or not
127     * @return
128     */
129   private def createUniqueSft(binding: Class[_], histogram: Boolean): SimpleFeatureType = {
130     val sftb = new SimpleFeatureTypeBuilder
131     sftb.add(AttributeValue, binding)
132     if (histogram) {
133       // histogram includes extra 'count' attribute
134       sftb.add(AttributeCount, classOf[java.lang.Long])
135     }
136 
137     sftb.setName(SftName)
138     sftb.buildFeatureType
139   }
