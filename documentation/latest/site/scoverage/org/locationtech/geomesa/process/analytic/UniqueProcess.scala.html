39 @DescribeProcess(title = "Geomesa Unique",
40   description = "Finds unique attributes values, optimized for GeoMesa")
41 class UniqueProcess extends GeoMesaProcess with LazyLogging {
42 
43   @DescribeResult(name = "result",
44     description = "Feature collection with an attribute containing the unique values")
45   def execute(
46     @DescribeParameter(name = "features", description = "Input feature collection")
47     features: SimpleFeatureCollection,
48     @DescribeParameter(name = "attribute", description = "Attribute whose unique values are extracted")
49     attribute: String,
50     @DescribeParameter(name = "filter", min = 0, description = "The filter to apply to the feature collection")
51     filter: Filter,
52     @DescribeParameter(name = "histogram", min = 0, description = "Create a histogram of attribute values")
53     histogram: java.lang.Boolean,
54     @DescribeParameter(name = "sort", min = 0, description = "Sort results - allowed to be ASC or DESC")
55     sort: String,
56     @DescribeParameter(name = "sortByCount", min = 0, description = "Sort by histogram counts instead of attribute values")
57     sortByCount: java.lang.Boolean,
58     progressListener: ProgressListener): SimpleFeatureCollection = {
59 
60     val attributeDescriptor = features
61         .getSchema
62         .getAttributeDescriptors
63         .asScala
64         .find(_.getLocalName == attribute)
65         .getOrElse(throw new IllegalArgumentException(s"Attribute $attribute does not exist in feature schema."))
66 
67     val hist = Option(histogram).exists(_.booleanValue)
68     val sortBy = Option(sortByCount).exists(_.booleanValue)
69 
70     val visitor = new AttributeVisitor(features, attributeDescriptor, Option(filter).filter(_ != Filter.INCLUDE), hist)
71     GeoMesaFeatureCollection.visit(features, visitor, progressListener)
72     val uniqueValues = visitor.getResult.attributes
73 
74     val binding = attributeDescriptor.getType.getBinding
75     UniqueProcess.createReturnCollection(uniqueValues, binding, hist, Option(sort), sortBy)
76   }
77 }

79 object UniqueProcess {
80 
81   val SftName = "UniqueValue"
82   val AttributeValue = "value"
83   val AttributeCount = "count"
84 
85   /**
86     * Duplicates output format from geotools UniqueProcess
87     *
88     * @param uniqueValues values
89     * @param binding value binding
90     * @param histogram include counts or just values
91     * @param sort sort
92     * @param sortByCount sort by count or by value
93     * @return
94     */
95   def createReturnCollection(uniqueValues: Map[AnyRef, Long],
96                              binding: Class[_],
97                              histogram: Boolean,
98                              sort: Option[String],
99                              sortByCount: Boolean): SimpleFeatureCollection = {
100 
101     val ft = createUniqueSft(binding, histogram)
102 
103     val result = new ListFeatureCollection(ft)
104 
105     // if sorting was requested do it here, otherwise return results in iterator order
106     val sorted = sort.map { s =>
107       if (sortByCount) {
108         val ordering = if (s.equalsIgnoreCase("desc")) Ordering[Long].reverse else Ordering[Long]
109         uniqueValues.iterator.toList.sortBy(_._2)(ordering)
110       } else {
111         val ordering = if (s.equalsIgnoreCase("desc")) Ordering[String].reverse else Ordering[String]
112         uniqueValues.iterator.toList.sortBy(_._1.toString)(ordering)
113       }
114     }.getOrElse(uniqueValues.iterator)
115 
116     // histogram includes extra 'count' attribute
117     val toSf = if (histogram) (key: AnyRef, value: Long) => {
118       val sf = new ScalaSimpleFeature(ft, UUID.randomUUID().toString)
119       sf.setAttributeNoConvert(0, key)
120       sf.setAttributeNoConvert(1, Long.box(value))
121       sf
122     } else (key: AnyRef, _: Long) => {
123       val sf = new ScalaSimpleFeature(ft, UUID.randomUUID().toString)
124       sf.setAttributeNoConvert(0, key)
125       sf
126     }
127 
128     sorted.foreach { case (key, value) => result.add(toSf(key, value)) }
129 
130     result
131   }
132 
133   /**
134     * Based on geotools UniqueProcess simple feature type
135     *
136     * @param binding class of attribute
137     * @param histogram return counts or not
138     * @return
139     */
140   def createUniqueSft(binding: Class[_], histogram: Boolean): SimpleFeatureType = {
141     val sftb = new SimpleFeatureTypeBuilder
142     sftb.add(AttributeValue, binding)
143     if (histogram) {
144       // histogram includes extra 'count' attribute
145       sftb.add(AttributeCount, classOf[java.lang.Long])
146     }
147 
148     sftb.setName(SftName)
149     sftb.buildFeatureType
150   }
151 }

153 /**
154  * Visitor that tracks unique attribute values and counts
155  *
156  * @param features features to evaluate
157  * @param attributeDescriptor attribute to evaluate
158  * @param filter optional filter to apply to features before evaluating
159  * @param histogram return counts or not
160  */
161 class AttributeVisitor(val features: SimpleFeatureCollection,
162                        val attributeDescriptor: AttributeDescriptor,
163                        val filter: Option[Filter],
164                        histogram: Boolean) extends GeoMesaProcessVisitor with LazyLogging {
165 
166   import scala.collection.JavaConverters._
167 
168   private val attribute    = attributeDescriptor.getLocalName
169   private val uniqueValues = mutable.Map.empty[AnyRef, Long].withDefaultValue(0)
170 
171   private var attributeIdx: Int = -1
172 
173   // normally handled in our query planner, but we are going to use the filter directly here
174   private lazy val manualFilter = filter.map(FastFilterFactory.optimize(features.getSchema, _))
175 
176   private def getAttribute[T](f: SimpleFeature): T = {
177     if (attributeIdx == -1) {
178       attributeIdx = f.getType.indexOf(attribute)
179     }
180     f.getAttribute(attributeIdx).asInstanceOf[T]
181   }
182 
183   private def addSingularValue(f: SimpleFeature): Unit = {
184     val value = getAttribute[AnyRef](f)
185     if (value != null) {
186       uniqueValues(value) += 1
187     }
188   }
189 
190   private def addMultiValue(f: SimpleFeature): Unit = {
191     val values = getAttribute[java.util.Collection[AnyRef]](f)
192     if (values != null) {
193       values.asScala.foreach(uniqueValues(_) += 1)
194     }
195   }
196 
197   private val addValue: SimpleFeature => Unit =
198     if (attributeDescriptor.isList) { addMultiValue } else { addSingularValue }
199 
200   // non-optimized visit
201   override def visit(feature: Feature): Unit = {
202     val f = feature.asInstanceOf[SimpleFeature]
203     if (manualFilter.forall(_.evaluate(f))) {
204       addValue(f)
205     }
206   }
207 
208   override def getResult: AttributeResult = new AttributeResult(uniqueValues.toMap)
209 
210   override def execute(source: SimpleFeatureSource, query: Query): Unit = {
211 
212     import org.locationtech.geomesa.filter.mergeFilters
213 
214     logger.debug(s"Running Geomesa histogram process on source type ${source.getClass.getName}")
215 
216     // combine filters from this process and any input collection
217     filter.foreach(f => query.setFilter(mergeFilters(query.getFilter, f)))
218 
219     val sft = source.getSchema
220 
221     val enumerated = if (attributeDescriptor.isMultiValued) {
222       // stats don't support list types
223       uniqueV5(source, query)
224     } else {
225       // TODO if !histogram, we could write a new unique skipping iterator
226       query.getHints.put(QueryHints.STATS_STRING, Stat.Enumeration(attribute))
227       query.getHints.put(QueryHints.ENCODE_STATS, java.lang.Boolean.TRUE)
228 
229       // execute the query
230       val reader = source.getFeatures(query).features()
231 
232       val enumeration = try {
233         // stats should always return exactly one result, even if there are no features in the table
234         val encoded = reader.next.getAttribute(0).asInstanceOf[String]
235         StatsScan.decodeStat(sft)(encoded).asInstanceOf[EnumerationStat[AnyRef]]
236       } finally {
237         reader.close()
238       }
239 
240       enumeration.frequencies
241     }
242 
243     uniqueValues.clear()
244     enumerated.foreach { case (k, v) => uniqueValues.put(k, v) }
245   }
246 
247   private def uniqueV5(source: SimpleFeatureSource, query: Query): Iterable[(AnyRef, Long)] = {
248     // only return the attribute we are interested in to reduce bandwidth
249     query.setPropertyNames(Seq(attribute).asJava)
250 
251     // if there is no filter, try to force an attribute scan - should be fastest query
252     if (query.getFilter == Filter.INCLUDE && AttributeIndex.indexed(features.getSchema, attribute)) {
253       query.setFilter(AttributeVisitor.getIncludeAttributeFilter(attribute))
254     }
255 
256     // execute the query
257     SelfClosingIterator(source.getFeatures(query).features()).foreach(addValue)
258     uniqueValues.toMap
259   }
260 }

276 /**
277  * Result class to hold the attribute histogram
278  *
279  * @param attributes result
280  */
281 class AttributeResult(val attributes: Map[AnyRef, Long]) extends AbstractCalcResult {
282 
283   override def getValue: java.util.Map[AnyRef, Long] = attributes.asJava
284 
285   override def isCompatible(targetResults: CalcResult): Boolean =
286     targetResults.isInstanceOf[AttributeResult] || targetResults == CalcResult.NULL_RESULT
287 
288   override def merge(resultsToAdd: CalcResult): CalcResult = {
289     if (!isCompatible(resultsToAdd)) {
290       throw new IllegalArgumentException("Parameter is not a compatible type")
291     } else if (resultsToAdd == CalcResult.NULL_RESULT) {
292       this
293     } else if (resultsToAdd.isInstanceOf[AttributeResult]) {
294       val toAdd = resultsToAdd.getValue.asInstanceOf[Map[AnyRef, Long]]
295       // note ++ on maps will get all keys with second maps values if exists, if not first map values
296       val merged = attributes ++ toAdd.map {
297         case (attr, count) => attr -> (count + attributes.getOrElse(attr, 0L))
298       }
299       new AttributeResult(merged)
300     } else {
301       throw new IllegalArgumentException(
302         "The CalcResults claim to be compatible, but the appropriate merge method has not been implemented.")
303     }
304   }
