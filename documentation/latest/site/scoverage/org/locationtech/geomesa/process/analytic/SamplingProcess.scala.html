20 /**
21   * Returns a reduced set of features using statistical sampling.
22   */
23 @DescribeProcess(
24   title = "Sampling Process",
25   description = "Uses statistical sampling to reduces the features returned by a query"
26 )
27 class SamplingProcess extends GeoMesaProcess with LazyLogging {
28 
29   /**
30     * Reduces the feature collection by statistically sampling the features.
31     *
32     * @param data input feature collection
33     * @param samplePercent percent of features to return, in the range (0, 1)
34     * @param threadBy threading field, used to group sampling of features
35     * @param monitor listener to monitor progress
36     * @throws org.geotools.process.ProcessException if something goes wrong
37     * @return
38     */
39   @throws(classOf[ProcessException])
40   @DescribeResult(name = "result", description = "Output features")
41   def execute(@DescribeParameter(name = "data", description = "Input features")
42               data: SimpleFeatureCollection,
43               @DescribeParameter(name = "samplePercent", description = "Percent of features to return, between 0 and 1", minValue = 0, maxValue = 1)
44               samplePercent: Float,
45               @DescribeParameter(name = "threadBy", description = "Attribute field to link associated features for sampling", min = 0)
46               threadBy: String,
47               monitor: ProgressListener): SimpleFeatureCollection = {
48 
49     logger.trace(s"Attempting sampling on ${data.getClass.getName}")
50 
51     val visitor = new SamplingVisitor(data, samplePercent, Option(threadBy))
52     GeoMesaFeatureCollection.visit(data, visitor, monitor)
53     visitor.getResult.results
54   }
55 }
