32 @DescribeProcess(
33   title = "Tube Select",
34   description = "Performs a tube select on a Geomesa feature collection based on another feature collection"
35 )
36 class TubeSelectProcess extends GeoMesaProcess with LazyLogging {
37 
38   @DescribeResult(description = "Output feature collection")
39   def execute(
40                @DescribeParameter(
41                  name = "tubeFeatures",
42                  description = "Input feature collection (must have geometry and datetime)")
43                tubeFeatures: SimpleFeatureCollection,
44 
45                @DescribeParameter(
46                  name = "featureCollection",
47                  description = "The data set to query for matching features")
48                featureCollection: SimpleFeatureCollection,
49 
50                @DescribeParameter(
51                  name = "filter",
52                  min = 0,
53                  description = "The filter to apply to the featureCollection")
54                filter: Filter,
55 
56                @DescribeParameter(
57                  name = "maxSpeed",
58                  min = 0,
59                  description = "Max speed of the object in m/s for nofill & line gapfill methods")
60                maxSpeed: java.lang.Long,
61 
62                @DescribeParameter(
63                  name = "maxTime",
64                  min = 0,
65                  description = "Time as seconds for nofill & line gapfill methods")
66                maxTime: java.lang.Long,
67 
68                @DescribeParameter(
69                  name = "bufferSize",
70                  min = 0,
71                  description = "Buffer size in meters to use instead of maxSpeed/maxTime calculation")
72                bufferSize: java.lang.Double,
73 
74                @DescribeParameter(
75                  name = "maxBins",
76                  min = 0,
77                  description = "Number of bins to use for breaking up query into individual queries")
78                maxBins: java.lang.Integer,
79 
80                @DescribeParameter(
81                  name = "gapFill",
82                  min = 0,
83                  description = "Method of filling gap (nofill, line)")
84                gapFill: String
85 
86                ): SimpleFeatureCollection = {
87 
88     logger.debug("Tube selecting on collection type "+featureCollection.getClass.getName)
89 
90     // assume for now that firstFeatures is a singleton collection
91     val tubeVisitor = new TubeVisitor(
92                                       tubeFeatures,
93                                       featureCollection,
94                                       Option(filter).getOrElse(Filter.INCLUDE),
95                                       Option(maxSpeed).getOrElse(0L).asInstanceOf[Long],
96                                       Option(maxTime).getOrElse(0L).asInstanceOf[Long],
97                                       Option(bufferSize).getOrElse(0.0).asInstanceOf[Double],
98                                       Option(maxBins).getOrElse(0).asInstanceOf[Int],
99                                       Option(gapFill).map(GapFill.withName).getOrElse(GapFill.NOFILL))
100 
101     GeoMesaFeatureCollection.visit(featureCollection, tubeVisitor)
102 
103     tubeVisitor.getResult.asInstanceOf[TubeResult].results
104   }
105 }

107 object GapFill extends Enumeration{
108   type GapFill = Value
109   val NOFILL: Value       = Value("nofill")
110   val LINE: Value         = Value("line")
111   val INTERPOLATED: Value = Value("interpolated")
112 }

114 class TubeVisitor(val tubeFeatures: SimpleFeatureCollection,
115                   val featureCollection: SimpleFeatureCollection,
116                   val filter: Filter = Filter.INCLUDE,
117                   val maxSpeed: Long,
118                   val maxTime: Long,
119                   val bufferSize: Double,
120                   val maxBins: Int,
121                   val gapFill: GapFill.GapFill = GapFill.NOFILL)
122     extends GeoMesaProcessVisitor with LazyLogging {
123 
124   var resultCalc: TubeResult = TubeResult(new EmptyFeatureCollection(featureCollection.getSchema))
125 
126   def visit(feature: Feature): Unit = {}
127 
128   override def getResult: CalcResult = resultCalc
129 
130   private val bufferDistance = if (bufferSize > 0) { bufferSize } else { maxSpeed * maxTime }
131 
132   override def execute(source: SimpleFeatureSource, query: Query): Unit = {
133 
134     import org.locationtech.geomesa.filter.ff
135 
136     logger.debug("Visiting source type: "+source.getClass.getName)
137 
138     val geomProperty = ff.property(source.getSchema.getGeometryDescriptor.getName)
139     val dateProperty = ff.property(source.getSchema.getUserData.get(SimpleFeatureTypes.Configs.DefaultDtgField).asInstanceOf[String])
140 
141     logger.debug("Querying with date property: "+dateProperty)
142     logger.debug("Querying with geometry property: "+geomProperty)
143 
144     // Create a time binned set of tube features with no gap filling
145 
146     val tubeBuilder = gapFill match {
147       case GapFill.LINE => new LineGapFill(tubeFeatures, bufferDistance, maxBins)
148       case GapFill.INTERPOLATED => new InterpolatedGapFill(tubeFeatures, bufferDistance, maxBins)
149       case _ => new NoGapFill(tubeFeatures, bufferDistance, maxBins)
150     }
151 
152     val tube = tubeBuilder.createTube
153 
154     val queryResults = CloseableIterator(tube).flatMap { sf =>
155       val sfMin = tubeBuilder.getStartTime(sf).getTime
156       val minDate = new Date(sfMin - maxTime*1000)
157 
158       val sfMax = tubeBuilder.getEndTime(sf).getTime
159       val maxDate = new Date(sfMax + maxTime*1000)
160 
161       val dtg1 = ff.greater(dateProperty, ff.literal(minDate))
162       val dtg2 = ff.less(dateProperty, ff.literal(maxDate))
163 
164       val geom = sf.getDefaultGeometry.asInstanceOf[Geometry]
165 
166       // Eventually these can be combined into OR queries and the QueryPlanner can create multiple Accumulo Ranges
167       // Buf for now we issue multiple queries
168       val geoms = (0 until geom.getNumGeometries).toIterator.map(geom.getGeometryN)
169       SelfClosingIterator(geoms).flatMap { g =>
170         val geomFilter = ff.intersects(geomProperty, ff.literal(g))
171         val combinedFilter = ff.and(List(query.getFilter, geomFilter, dtg1, dtg2, filter).asJava)
172         SelfClosingIterator(source.getFeatures(combinedFilter).features)
173       }
174     }
175 
176     // Time slices may not be disjoint so we have to buffer results and dedupe for now
177     val collection = new ListFeatureCollection(source.getSchema)
178     WithClose(new DeduplicatingSimpleFeatureIterator(queryResults))(_.foreach(collection.add))
179 
180     resultCalc = TubeResult(collection)
181   }
182 }
