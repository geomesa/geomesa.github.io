27 /**
28   * Returns features from a feature type based on a join against a second feature type.
29   */
30 @DescribeProcess(
31   title = "Join Process",
32   description = "Queries a feature type based on attributes from a second feature type"
33 )
34 class JoinProcess extends GeoMesaProcess with LazyLogging {
35 
36   /**
37     *
38     * @param primary main feature collection to query
39     * @param secondary secondary feature collection to query with results from first feature collection
40     * @param joinAttribute attribute to join on
41     * @param joinFilter additional filter to apply to joined features
42     * @param attributes attributes to return, from both collections, qualified by schema name
43     * @param monitor listener to monitor progress
44     * @throws org.geotools.process.ProcessException if something goes wrong
45     * @return
46     */
47   @throws(classOf[ProcessException])
48   @DescribeResult(name = "result", description = "Output features")
49   def execute(@DescribeParameter(name = "primary", description = "Primary feature collection being queried", min = 1)
50               primary: SimpleFeatureCollection,
51               @DescribeParameter(name = "secondary", description = "Secondary feature collection to be joined", min = 1)
52               secondary: SimpleFeatureCollection,
53               @DescribeParameter(name = "joinAttribute", description = "Attribute field to join on", min = 1)
54               joinAttribute: String,
55               @DescribeParameter(name = "joinFilter", description = "Additional filter to apply to joined features", min = 0)
56               joinFilter: Filter,
57               @DescribeParameter(name = "attributes", description = "Attributes to return. Attribute names should be qualified with the schema name, e.g. foo.bar", min = 0, max = 128, collectionType = classOf[String])
58               attributes: java.util.List[String],
59               monitor: ProgressListener): SimpleFeatureCollection = {
60 
61     import org.locationtech.geomesa.filter.ff
62 
63     import scala.collection.JavaConverters._
64 
65     logger.trace(s"Attempting join query on ${joinAttribute.getClass.getName}")
66 
67     require(primary.getSchema.getDescriptor(joinAttribute) != null,
68       s"Attribute '$joinAttribute' does not exist in the primary feature collection")
69 
70     val joinDescriptor = secondary.getSchema.getDescriptor(joinAttribute)
71     require(joinDescriptor != null, s"Attribute '$joinAttribute' does not exist in the joined feature collection")
72 
73     // create the return sft based on the input attributes, or by combining the qualified names from each schema
74     val returnSft = if (attributes != null && attributes.asScala.nonEmpty) {
75       getCombinedSft(primary.getSchema, secondary.getSchema, attributes.asScala.toSeq, joinAttribute)
76     } else {
77       def toAttributes(schema: SimpleFeatureType): Seq[String] = {
78         val names = schema.getAttributeDescriptors.asScala.map(_.getLocalName)
79         names.filter(_ != joinAttribute).map(d => s"${schema.getTypeName}.$d").toSeq
80       }
81       val primaryAttributes = toAttributes(primary.getSchema)
82       val secondaryAttributes = toAttributes(secondary.getSchema)
83       val attributes = Seq(joinAttribute) ++ primaryAttributes ++ secondaryAttributes
84       getCombinedSft(primary.getSchema, secondary.getSchema, attributes, joinAttribute)
85     }
86 
87     // check for too many features coming back - limit is somewhat arbitrary, but this
88     // class is mainly intended for a single feature lookup
89     val primaryFeatures = SelfClosingIterator(primary).toList
90     require(primaryFeatures.length < 129,
91       s"Too many features returned from primary query - got ${primaryFeatures.length}, max 128")
92 
93     val joinProperty = ff.property(joinAttribute)
94     val joinFilters = {
95       val values = primaryFeatures.map(_.getAttribute(joinAttribute)).distinct
96       values.map(p => ff.equals(joinProperty, ff.literal(p)).asInstanceOf[Filter])
97     }
98 
99     val result = if (joinFilters.isEmpty) {
100       new ListFeatureCollection(returnSft)
101     } else {
102       val or = ff.or(joinFilters.toList.asJava)
103       val filter = if (joinFilter != null && joinFilter != Filter.INCLUDE) { ff.and(or, joinFilter) } else { or }
104       val visitor = new QueryVisitor(secondary, filter, null)
105       GeoMesaFeatureCollection.visit(secondary, visitor)
106       val results = visitor.getResult.results
107 
108       // mappings from the secondary feature result to the return schema
109       // (return sft index, result sft index, is from primary result (or secondary result))
110       val attributeMappings: Seq[(Int, Int, Boolean)] = returnSft.getAttributeDescriptors.asScala.map { d =>
111         val toAttribute = d.getLocalName
112         val dot = toAttribute.indexOf('.')
113         if (dot == -1) {
114           val fromPrimary = secondary.getSchema.getAttributeDescriptors.asScala.exists(_.getLocalName == toAttribute)
115           val from = if (fromPrimary) {
116             primary.getSchema.indexOf(toAttribute)
117           } else {
118             secondary.getSchema.indexOf(toAttribute)
119           }
120           (returnSft.indexOf(toAttribute), from, fromPrimary)
121         } else {
122           val fromPrimary = primary.getSchema.getTypeName == toAttribute.substring(0, dot)
123           val from = if (fromPrimary) {
124             primary.getSchema.indexOf(toAttribute.substring(dot + 1))
125           } else {
126             secondary.getSchema.indexOf(toAttribute.substring(dot + 1))
127           }
128           (returnSft.indexOf(toAttribute), from, fromPrimary)
129         }
130       }.toSeq
131 
132       new DecoratingSimpleFeatureCollection(results) {
133         override def getSchema: SimpleFeatureType = returnSft
134         override def features(): SimpleFeatureIterator = new SimpleFeatureIterator {
135           private val delegate = results.features
136           override def next(): SimpleFeature = {
137             val secondarySf = delegate.next()
138             val toJoin = secondarySf.getAttribute(joinAttribute)
139             val primarySf = primaryFeatures.find(_.getAttribute(joinAttribute) == toJoin).getOrElse {
140               throw new RuntimeException("No feature joined from attribute query")
141             }
142             val sf = new ScalaSimpleFeature(returnSft, s"${primarySf.getID}-${secondarySf.getID}")
143             attributeMappings.foreach { case (to, from, fromPrimary) =>
144               val a = if (fromPrimary) { primarySf.getAttribute(from) } else { secondarySf.getAttribute(from) }
145               sf.setAttribute(to, a)
146             }
147             sf
148           }
149           override def hasNext: Boolean = delegate.hasNext
150           override def close(): Unit = delegate.close()
151         }
152       }
153     }
154 
155     result
156   }
157 
158   /**
159     * Builds a combined sft
160     *
161     * @param primary primary sft being joined
162     * @param secondary secondary sft being joined
163     * @param attributes attributes to include - must be sft qualified if ambiguous
164     * @param join attribute being joined on - this will always be unqualified and should exist in both sfts
165     * @return
166     */
167   private def getCombinedSft(primary: SimpleFeatureType,
168                              secondary: SimpleFeatureType,
169                              attributes: Seq[String],
170                              join: String): SimpleFeatureType = {
171     val builder = new SimpleFeatureTypeBuilder()
172     builder.setName(s"${primary.getTypeName}_join_${secondary.getTypeName}")
173 
174     val descriptorBuilder = new AttributeTypeBuilder()
175 
176     attributes.foreach { attribute =>
177       val dot = attribute.indexOf('.')
178       val descriptor = if (dot == -1) {
179         val primaryDescriptor = primary.getDescriptor(attribute)
180         val secondaryDescriptor = secondary.getDescriptor(attribute)
181         if (primaryDescriptor == null || attribute == join) {
182           secondaryDescriptor
183         } else if (secondaryDescriptor == null) {
184           primaryDescriptor
185         } else {
186           throw new IllegalArgumentException(s"Ambiguous property requested: $attribute exists in both schemas")
187         }
188       } else {
189         val typeName = attribute.substring(0, dot)
190         if (typeName == primary.getTypeName) {
191           primary.getDescriptor(attribute.substring(dot + 1))
192         } else if (typeName == secondary.getTypeName) {
193           secondary.getDescriptor(attribute.substring(dot + 1))
194         } else {
195           null
196         }
197       }
198 
199       if (descriptor != null) {
200         descriptorBuilder.init(descriptor)
201         val toAdd = descriptorBuilder.buildDescriptor(attribute)
202         builder.add(toAdd)
203         if (descriptor == secondary.getGeometryDescriptor) {
204           builder.setDefaultGeometry(toAdd.getLocalName)
205         }
206       }
207     }
208 
209     builder.buildFeatureType()
210   }
