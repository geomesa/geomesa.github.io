28 @DescribeProcess(
29   title = "Stats Process",
30   description = "Gathers statistics for a data set"
31 )
32 class StatsProcess extends GeoMesaProcess with LazyLogging {
33 
34   @DescribeResult(description = "Output feature collection")
35   def execute(
36                  @DescribeParameter(
37                    name = "features",
38                    description = "The feature set on which to query")
39                  features: SimpleFeatureCollection,
40 
41                  @DescribeParameter(
42                    name = "statString",
43                    description = "The string indicating what stats to instantiate")
44                  statString: String,
45 
46                  @DescribeParameter(
47                    name = "encode",
48                    min = 0,
49                    description = "Return the values encoded or as json")
50                  encode: java.lang.Boolean = null,
51 
52                  @DescribeParameter(
53                    name = "properties",
54                    min = 0,
55                    max = 128,
56                    collectionType = classOf[String],
57                    description = "The properties / transforms to apply before gathering stats")
58                  properties: java.util.List[String] = null
59 
60              ): SimpleFeatureCollection = {
61 
62     logger.debug(s"Attempting stats iterator process on type ${features.getClass.getName}")
63 
64     val propsArray = Option(properties).map(_.toArray(Array.empty[String])).filter(_.length > 0).orNull
65 
66     val visitor = new StatsVisitor(features, statString, Option(encode).exists(_.booleanValue()), propsArray)
67     GeoMesaFeatureCollection.visit(features, visitor)
68     visitor.getResult.results
69   }
70 }

72 class StatsVisitor(features: SimpleFeatureCollection, statString: String, encode: Boolean, properties: Array[String])
73     extends GeoMesaProcessVisitor with LazyLogging {
74 
75   private val origSft = features.getSchema
76 
77   private lazy val transformDefinitions = Transforms(origSft, properties)
78   private lazy val transformSft = Transforms.schema(origSft, transformDefinitions)
79   private lazy val transformSf = TransformSimpleFeature(transformSft, transformDefinitions)
80 
81   private lazy val statSft = if (properties == null) { origSft } else { transformSft }
82 
83   private lazy val stat: Stat = Stat(statSft, statString)
84 
85   private var resultCalc: FeatureResult = _
86 
87   // non-optimized visit
88   override def visit(feature: Feature): Unit = {
89     val sf = feature.asInstanceOf[SimpleFeature]
90     if (properties != null) {
91       // There are transforms!
92       transformSf.setFeature(sf)
93       stat.observe(transformSf)
94     } else {
95       stat.observe(sf)
96     }
97   }
98 
99   override def getResult: FeatureResult = {
100     if (resultCalc != null) {
101       resultCalc
102     } else {
103       val stats = if (encode) {
104         StatsScan.encodeStat(statSft)(stat)
105       } else {
106         stat.toJson
107       }
108 
109       val sf = new ScalaSimpleFeature(StatsScan.StatsSft, "", Array(stats, GeometryUtils.zeroPoint))
110       val manualVisitResults = new ListFeatureCollection(StatsScan.StatsSft)
111       manualVisitResults.add(sf)
112       FeatureResult(manualVisitResults)
113     }
114   }
115 
116   override def execute(source: SimpleFeatureSource, query: Query): Unit = {
117     logger.debug(s"Running Geomesa stats iterator process on source type ${source.getClass.getName}")
118 
119     if (properties != null) {
120       if (query.getProperties != Query.ALL_PROPERTIES) {
121         logger.warn(s"Overriding inner query's properties (${query.getProperties}) " +
122             s"with properties/transforms ${properties.mkString(",")}.")
123       }
124       query.setPropertyNames(properties: _*)
125     }
126     query.getHints.put(QueryHints.STATS_STRING, statString)
127     query.getHints.put(QueryHints.ENCODE_STATS, new java.lang.Boolean(encode))
128     resultCalc = FeatureResult(source.getFeatures(query))
129   }
130 }
