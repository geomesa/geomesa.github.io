23 @DescribeProcess(
24   title = "Binary Conversion",
25   description = "Converts a feature collection to binary format"
26 )
27 class BinConversionProcess extends GeoMesaProcess with LazyLogging {
28 
29   /**
30     * Converts an input feature collection to BIN format
31     *
32     * @param features input features
33     * @param geom geometry attribute to encode in the bin format, optional
34     *             will use default geometry if not provided
35     * @param dtg date attribute to encode in the bin format, optional
36     *            will use default date if not provided
37     * @param track track id attribute to encode in the bin format, optional
38     *              will use feature id if not provided
39     * @param label label attribute to encode in the bin format, optional
40     * @param axisOrder axis order to encode points, optional
41     * @return
42     */
43   @DescribeResult(description = "Encoded feature collection")
44   def execute(
45               @DescribeParameter(name = "features", description = "Input feature collection to query ")
46               features: SimpleFeatureCollection,
47               @DescribeParameter(name = "track", description = "Track field to use for BIN records", min = 0)
48               track: String,
49               @DescribeParameter(name = "geom", description = "Geometry field to use for BIN records", min = 0)
50               geom: String,
51               @DescribeParameter(name = "dtg", description = "Date field to use for BIN records", min = 0)
52               dtg: String,
53               @DescribeParameter(name = "label", description = "Label field to use for BIN records", min = 0)
54               label: String,
55               @DescribeParameter(name = "axisOrder", description = "Axis order - either latlon or lonlat", min = 0)
56               axisOrder: String
57              ): java.util.Iterator[Array[Byte]] = {
58 
59     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
60 
61     logger.debug(s"Running BIN encoding for ${features.getClass.getName}")
62 
63     val sft = features.getSchema
64 
65     def indexOf(attribute: String): Int = {
66       val i = sft.indexOf(attribute)
67       if (i == -1) {
68         throw new IllegalArgumentException(s"Attribute $attribute doesn't exist in ${sft.getTypeName} " +
69             SimpleFeatureTypes.encodeType(sft))
70       }
71       i
72     }
73 
74     val geomField  = Option(geom).map(indexOf)
75     val dtgField   = Option(dtg).map(indexOf).orElse(sft.getDtgIndex)
76     val trackField = Option(track).filter(_ != "id").map(indexOf)
77     val labelField = Option(label).map(indexOf)
78 
79     val axis = Option(axisOrder).map {
80       case o if o.toLowerCase(Locale.US) == "latlon" => AxisOrder.LatLon
81       case o if o.toLowerCase(Locale.US) == "lonlat" => AxisOrder.LonLat
82       case o => throw new IllegalArgumentException(s"Invalid axis order '$o'. Valid values are 'latlon' and 'lonlat'")
83     }
84 
85     val visitor = new BinVisitor(sft, EncodingOptions(geomField, dtgField, trackField, labelField, axis))
86     GeoMesaFeatureCollection.visit(features, visitor)
87     visitor.getResult.results
88   }
