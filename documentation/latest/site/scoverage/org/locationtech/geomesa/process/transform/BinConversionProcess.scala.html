29 @DescribeProcess(
30   title = "Binary Conversion",
31   description = "Converts a feature collection to binary format"
32 )
33 class BinConversionProcess extends GeoMesaProcess with LazyLogging {
34 
35   /**
36     * Converts an input feature collection to BIN format
37     *
38     * @param features input features
39     * @param geom geometry attribute to encode in the bin format, optional
40     *             will use default geometry if not provided
41     * @param dtg date attribute to encode in the bin format, optional
42     *            will use default date if not provided
43     * @param track track id attribute to encode in the bin format, optional
44     *              will use feature id if not provided
45     * @param label label attribute to encode in the bin format, optional
46     * @param axisOrder axis order to encode points, optional
47     * @return
48     */
49   @DescribeResult(description = "Encoded feature collection")
50   def execute(
51               @DescribeParameter(name = "features", description = "Input feature collection to query ")
52               features: SimpleFeatureCollection,
53               @DescribeParameter(name = "track", description = "Track field to use for BIN records", min = 0)
54               track: String,
55               @DescribeParameter(name = "geom", description = "Geometry field to use for BIN records", min = 0)
56               geom: String,
57               @DescribeParameter(name = "dtg", description = "Date field to use for BIN records", min = 0)
58               dtg: String,
59               @DescribeParameter(name = "label", description = "Label field to use for BIN records", min = 0)
60               label: String,
61               @DescribeParameter(name = "axisOrder", description = "Axis order - either latlon or lonlat", min = 0)
62               axisOrder: String
63              ): java.util.Iterator[Array[Byte]] = {
64 
65     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
66 
67     logger.debug(s"Running BIN encoding for ${features.getClass.getName}")
68 
69     val sft = features.getSchema
70 
71     def indexOf(attribute: String): Int = {
72       val i = sft.indexOf(attribute)
73       if (i == -1) {
74         throw new IllegalArgumentException(s"Attribute $attribute doesn't exist in ${sft.getTypeName} " +
75             SimpleFeatureTypes.encodeType(sft))
76       }
77       i
78     }
79 
80     val geomField  = Option(geom).map(indexOf)
81     val dtgField   = Option(dtg).map(indexOf).orElse(sft.getDtgIndex)
82     val trackField = Option(track).filter(_ != "id").map(indexOf)
83     val labelField = Option(label).map(indexOf)
84 
85     val axis = Option(axisOrder).map {
86       case o if o.toLowerCase(Locale.US) == "latlon" => AxisOrder.LatLon
87       case o if o.toLowerCase(Locale.US) == "lonlat" => AxisOrder.LonLat
88       case o => throw new IllegalArgumentException(s"Invalid axis order '$o'. Valid values are 'latlon' and 'lonlat'")
89     }
90 
91     val visitor = new BinVisitor(sft, EncodingOptions(geomField, dtgField, trackField, labelField, axis))
92     GeoMesaFeatureCollection.visit(features, visitor)
93     visitor.getResult.results
94   }
95 }

97 class BinVisitor(sft: SimpleFeatureType, options: EncodingOptions)
98     extends GeoMesaProcessVisitor with LazyLogging {
99 
100   import scala.collection.JavaConverters._
101 
102   // for collecting results manually
103   private val manualResults = scala.collection.mutable.Queue.empty[Array[Byte]]
104   private val manualConversion = BinaryOutputEncoder(sft, options)
105 
106   private var result = new Iterator[Array[Byte]] {
107     override def next(): Array[Byte] = manualResults.dequeue()
108     override def hasNext: Boolean = manualResults.nonEmpty
109   }
110 
111   override def getResult: BinResult = BinResult(result.asJava)
112 
113   // manually called for non-accumulo feature collections
114   override def visit(feature: Feature): Unit =
115     manualResults += manualConversion.encode(feature.asInstanceOf[SimpleFeature])
116 
117   override def execute(source: SimpleFeatureSource, query: Query): Unit = {
118     logger.debug(s"Visiting source type: ${source.getClass.getName}")
119 
120     query.getHints.put(QueryHints.BIN_TRACK, options.trackIdField.map(sft.getDescriptor(_).getLocalName).getOrElse("id"))
121     options.geomField.foreach(i => query.getHints.put(QueryHints.BIN_GEOM, sft.getDescriptor(i).getLocalName))
122     options.dtgField.foreach(i => query.getHints.put(QueryHints.BIN_DTG, sft.getDescriptor(i).getLocalName))
123     options.labelField.foreach(i => query.getHints.put(QueryHints.BIN_LABEL, sft.getDescriptor(i).getLocalName))
124 
125     val features = SelfClosingIterator(source.getFeatures(query))
126     result ++= features.map(_.getAttribute(BIN_ATTRIBUTE_INDEX).asInstanceOf[Array[Byte]])
127   }
128 }
