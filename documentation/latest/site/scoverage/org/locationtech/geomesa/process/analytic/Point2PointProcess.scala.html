26 @DescribeProcess(title = "Point2PointProcess", description = "Aggregates a collection of points into a collection of line segments")
27 class Point2PointProcess extends GeoMesaProcess {
28 
29   private val baseType = SchemaBuilder.builder().addLineString("geom", default = true).build("point2point")
30 
31   private val gf = JTSFactoryFinder.getGeometryFactory
32 
33   @DescribeResult(name = "result", description = "Aggregated feature collection")
34   def execute(
35 
36                @DescribeParameter(name = "data", description = "Input feature collection")
37                data: SimpleFeatureCollection,
38 
39                @DescribeParameter(name = "groupingField", description = "Field on which to group")
40                groupingField: String,
41 
42                @DescribeParameter(name = "sortField", description = "Field on which to sort (must be Date type)")
43                sortField: String,
44 
45                @DescribeParameter(name = "minimumNumberOfPoints", description = "Minimum number of points")
46                minPoints: Int,
47 
48                @DescribeParameter(name = "breakOnDay", description = "Break connections on day marks")
49                breakOnDay: Boolean,
50 
51                @DescribeParameter(name = "filterSingularPoints", description = "Filter out segments that fall on the same point", defaultValue = "true")
52                filterSingularPoints: Boolean
53 
54                ): SimpleFeatureCollection = {
55 
56     import org.locationtech.geomesa.utils.geotools.Conversions._
57 
58     val queryType = data.getSchema
59     val sftBuilder = new SimpleFeatureTypeBuilder()
60     sftBuilder.init(baseType)
61     val groupingFieldIndex = data.getSchema.indexOf(groupingField)
62     sftBuilder.add(queryType.getAttributeDescriptors.get(groupingFieldIndex))
63     val sortFieldIndex = data.getSchema.indexOf(sortField)
64     val sortDesc = queryType.getAttributeDescriptors.get(sortFieldIndex)
65     val sortAttrName = sortDesc.getLocalName
66     val sortType = sortDesc.getType.getBinding
67     sftBuilder.add(s"${sortAttrName}_start", sortType)
68     sftBuilder.add(s"${sortAttrName}_end", sortType)
69 
70     val sft = sftBuilder.buildFeatureType()
71 
72     val lineFeatures: Iterable[SimpleFeature] =
73       SelfClosingIterator(data.features()).toList
74         .groupBy(f => String.valueOf(f.getAttribute(groupingFieldIndex)))
75         .filter { case (_, coll) => coll.lengthCompare(minPoints) > 0 }
76         .flatMap { case (_, coll) =>
77 
78           val globalSorted = coll.sortBy(_.get[java.util.Date](sortFieldIndex))
79 
80           val groups = if (!breakOnDay) { Array(globalSorted) } else {
81             globalSorted
82               .groupBy { f => getDayOfYear(sortFieldIndex, f) }
83               .filter { case (_, g) => g.lengthCompare(2) >= 0 }  // need at least two points in a day to create a
84               .map { case (_, g) => g }.toArray
85           }
86 
87           val results = groups.flatMap { sorted =>
88             sorted.sliding(2).zipWithIndex.map { case (ptLst, idx) =>
89               import org.locationtech.geomesa.utils.geotools.Conversions.RichSimpleFeature
90               val pts = ptLst.map(_.point.getCoordinate)
91               val length = JTS.orthodromicDistance(pts.head, pts.last, DefaultGeographicCRS.WGS84)
92 
93               val group    = ptLst.head.getAttribute(groupingFieldIndex)
94               val startDtg = ptLst.head.getAttribute(sortAttrName)
95               val endDtg   = ptLst.last.getAttribute(sortAttrName)
96               val attrs    = Array[AnyRef](gf.createLineString(pts.toArray), group, startDtg, endDtg)
97               val sf = new ScalaSimpleFeature(sft, s"$group-$idx", attrs)
98               (length, sf)
99             }
100           }
101 
102           if (filterSingularPoints) {
103             results.collect { case (length, sf) if length > 0.0 => sf }
104           } else {
105             results.map { case (_, sf) => sf }
106           }
107         }
108 
109     import scala.collection.JavaConverters._
110 
111     new ListFeatureCollection(sft, lineFeatures.toList.asJava)
112   }
113 
114   def getDayOfYear(sortFieldIndex: Int, f: SimpleFeature): Int =
115     ZonedDateTime.ofInstant(f.getAttribute(sortFieldIndex).asInstanceOf[Date].toInstant, ZoneOffset.UTC).getDayOfYear
