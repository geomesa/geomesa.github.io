27 @DescribeProcess(title = "Point2PointProcess", description = "Aggregates a collection of points into a collection of line segments")
28 class Point2PointProcess extends GeoMesaProcess {
29 
30   private val baseType = SchemaBuilder.builder().addLineString("geom", default = true).build("point2point")
31 
32   private val gf = JTSFactoryFinder.getGeometryFactory
33 
34   @DescribeResult(name = "result", description = "Aggregated feature collection")
35   def execute(
36 
37                @DescribeParameter(name = "data", description = "Input feature collection")
38                data: SimpleFeatureCollection,
39 
40                @DescribeParameter(name = "groupingField", description = "Field on which to group")
41                groupingField: String,
42 
43                @DescribeParameter(name = "sortField", description = "Field on which to sort (must be Date type)")
44                sortField: String,
45 
46                @DescribeParameter(name = "minimumNumberOfPoints", description = "Minimum number of points")
47                minPoints: Int,
48 
49                @DescribeParameter(name = "breakOnDay", description = "Break connections on day marks")
50                breakOnDay: Boolean,
51 
52                @DescribeParameter(name = "filterSingularPoints", description = "Filter out segments that fall on the same point", defaultValue = "true")
53                filterSingularPoints: Boolean
54 
55                ): SimpleFeatureCollection = {
56 
57     val queryType = data.getSchema
58     val sftBuilder = new SimpleFeatureTypeBuilder()
59     sftBuilder.init(baseType)
60     val groupingFieldIndex = data.getSchema.indexOf(groupingField)
61     sftBuilder.add(queryType.getAttributeDescriptors.get(groupingFieldIndex))
62     val sortFieldIndex = data.getSchema.indexOf(sortField)
63     val sortDesc = queryType.getAttributeDescriptors.get(sortFieldIndex)
64     val sortAttrName = sortDesc.getLocalName
65     val sortType = sortDesc.getType.getBinding
66     sftBuilder.add(s"${sortAttrName}_start", sortType)
67     sftBuilder.add(s"${sortAttrName}_end", sortType)
68 
69     val sft = sftBuilder.buildFeatureType()
70 
71     val lineFeatures: Iterable[SimpleFeature] =
72       CloseableIterator(data.features()).toList
73         .groupBy(f => String.valueOf(f.getAttribute(groupingFieldIndex)))
74         .filter { case (_, coll) => coll.lengthCompare(minPoints) > 0 }
75         .flatMap { case (_, coll) =>
76 
77           val globalSorted = coll.sortBy(_.getAttribute(sortFieldIndex).asInstanceOf[java.util.Date])
78 
79           val groups = if (!breakOnDay) { Array(globalSorted) } else {
80             globalSorted
81               .groupBy { f => getDayOfYear(sortFieldIndex, f) }
82               .filter { case (_, g) => g.lengthCompare(2) >= 0 }  // need at least two points in a day to create a
83               .map { case (_, g) => g }.toArray
84           }
85 
86           val results = groups.flatMap { sorted =>
87             sorted.sliding(2).zipWithIndex.map { case (ptLst, idx) =>
88               val pts = ptLst.map(_.getDefaultGeometry.asInstanceOf[Point].getCoordinate)
89               val length = JTS.orthodromicDistance(pts.head, pts.last, DefaultGeographicCRS.WGS84)
90               val group    = ptLst.head.getAttribute(groupingFieldIndex)
91               val startDtg = ptLst.head.getAttribute(sortAttrName)
92               val endDtg   = ptLst.last.getAttribute(sortAttrName)
93               val attrs    = Array[AnyRef](gf.createLineString(pts.toArray), group, startDtg, endDtg)
94               val sf = new ScalaSimpleFeature(sft, s"$group-$idx", attrs)
95               (length, sf)
96             }
97           }
98 
99           if (filterSingularPoints) {
100             results.collect { case (length, sf) if length > 0.0 => sf }
101           } else {
102             results.map { case (_, sf) => sf }
103           }
104         }
105 
106     import scala.collection.JavaConverters._
107 
108     new ListFeatureCollection(sft, lineFeatures.toList.asJava)
109   }
110 
111   def getDayOfYear(sortFieldIndex: Int, f: SimpleFeature): Int =
112     ZonedDateTime.ofInstant(f.getAttribute(sortFieldIndex).asInstanceOf[Date].toInstant, ZoneOffset.UTC).getDayOfYear
