27 @DescribeProcess(title = "Point2PointProcess", description = "Aggregates a collection of points into a collection of line segments")
28 class Point2PointProcess extends GeoMesaProcess {
29 
30   private val baseType = SchemaBuilder.builder().addLineString("geom", default = true).build("point2point")
31 
32   private val gf = JTSFactoryFinder.getGeometryFactory
33 
34   @DescribeResult(name = "result", description = "Aggregated feature collection")
35   def execute(
36 
37                @DescribeParameter(name = "data", description = "Input feature collection")
38                data: SimpleFeatureCollection,
39 
40                @DescribeParameter(name = "groupingField", description = "Field on which to group")
41                groupingField: String,
42 
43                @DescribeParameter(name = "sortField", description = "Field on which to sort (must be Date type)")
44                sortField: String,
45 
46                @DescribeParameter(name = "minimumNumberOfPoints", description = "Minimum number of points")
47                minPoints: Int,
48 
49                @DescribeParameter(name = "breakOnDay", description = "Break connections on day marks")
50                breakOnDay: Boolean,
51 
52                @DescribeParameter(name = "filterSingularPoints", description = "Filter out segments that fall on the same point", defaultValue = "true")
53                filterSingularPoints: Boolean
54 
55                ): SimpleFeatureCollection = {
56 
57     import org.locationtech.geomesa.utils.geotools.Conversions._
58 
59     val queryType = data.getSchema
60     val sftBuilder = new SimpleFeatureTypeBuilder()
61     sftBuilder.init(baseType)
62     val groupingFieldIndex = data.getSchema.indexOf(groupingField)
63     sftBuilder.add(queryType.getAttributeDescriptors.get(groupingFieldIndex))
64     val sortFieldIndex = data.getSchema.indexOf(sortField)
65     val sortDesc = queryType.getAttributeDescriptors.get(sortFieldIndex)
66     val sortAttrName = sortDesc.getLocalName
67     val sortType = sortDesc.getType.getBinding
68     sftBuilder.add(s"${sortAttrName}_start", sortType)
69     sftBuilder.add(s"${sortAttrName}_end", sortType)
70 
71     val sft = sftBuilder.buildFeatureType()
72 
73     val lineFeatures: Iterable[SimpleFeature] =
74       SelfClosingIterator(data.features()).toList
75         .groupBy(f => String.valueOf(f.getAttribute(groupingFieldIndex)))
76         .filter { case (_, coll) => coll.lengthCompare(minPoints) > 0 }
77         .flatMap { case (_, coll) =>
78 
79           val globalSorted = coll.sortBy(_.get[java.util.Date](sortFieldIndex))
80 
81           val groups = if (!breakOnDay) { Array(globalSorted) } else {
82             globalSorted
83               .groupBy { f => getDayOfYear(sortFieldIndex, f) }
84               .filter { case (_, g) => g.lengthCompare(2) >= 0 }  // need at least two points in a day to create a
85               .map { case (_, g) => g }.toArray
86           }
87 
88           val results = groups.flatMap { sorted =>
89             sorted.sliding(2).zipWithIndex.map { case (ptLst, idx) =>
90               import org.locationtech.geomesa.utils.geotools.Conversions.RichSimpleFeature
91               val pts = ptLst.map(_.point.getCoordinate)
92               val length = JTS.orthodromicDistance(pts.head, pts.last, DefaultGeographicCRS.WGS84)
93 
94               val group    = ptLst.head.getAttribute(groupingFieldIndex)
95               val startDtg = ptLst.head.getAttribute(sortAttrName)
96               val endDtg   = ptLst.last.getAttribute(sortAttrName)
97               val attrs    = Array[AnyRef](gf.createLineString(pts.toArray), group, startDtg, endDtg)
98               val sf = new ScalaSimpleFeature(sft, s"$group-$idx", attrs)
99               (length, sf)
100             }
101           }
102 
103           if (filterSingularPoints) {
104             results.collect { case (length, sf) if length > 0.0 => sf }
105           } else {
106             results.map { case (_, sf) => sf }
107           }
108         }
109 
110     import scala.collection.JavaConverters._
111 
112     new ListFeatureCollection(sft, lineFeatures.toList.asJava)
113   }
114 
115   def getDayOfYear(sortFieldIndex: Int, f: SimpleFeature): Int =
116     ZonedDateTime.ofInstant(toInstant(f.getAttribute(sortFieldIndex).asInstanceOf[Date]), ZoneOffset.UTC).getDayOfYear
117 }
