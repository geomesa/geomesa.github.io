33 /**
34  * Build a tube for input to a TubeSelect by buffering and binning the input
35  * tubeFeatures into SimpleFeatures that can be used as inputs to Geomesa queries
36  */
37 abstract class TubeBuilder(val tubeFeatures: SimpleFeatureCollection,
38                            val bufferDistance: Double,
39                            val maxBins: Int) extends LazyLogging {
40 
41   protected val calc = new GeodeticCalculator()
42   protected val dtgField: String = tubeFeatures.getSchema.getDtgField.getOrElse(TubeBuilder.DefaultDtgField)
43   protected val geoFac = new GeometryFactory
44 
45   private val tubeType: SimpleFeatureType = SimpleFeatureTypes.createType("tubeType", "geom:Geometry:srid=4326,start:Date,end:Date")
46   protected val builder: SimpleFeatureBuilder = ScalaSimpleFeatureFactory.featureBuilder(tubeType)
47 
48   def getGeom(sf: SimpleFeature): Geometry = sf.getAttribute(0).asInstanceOf[Geometry]
49   def getStartTime(sf: SimpleFeature): Date = sf.getAttribute(1).asInstanceOf[Date]
50   def getEndTime(sf: SimpleFeature): Date = sf.getAttribute(2).asInstanceOf[Date]
51 
52   private def bufferGeom(geom: Geometry, meters: Double): Geometry = {
53     import org.locationtech.geomesa.utils.geotools.Conversions.RichGeometry
54     geom.buffer(metersToDegrees(meters, geom.safeCentroid()))
55   }
56 
57   // visible for testing
58   private[geomesa] def metersToDegrees(meters: Double, point: Point): Double = {
59     logger.debug("Buffering: "+meters.toString + " "+WKTUtils.write(point))
60 
61     calc.setStartingGeographicPoint(point.getX, point.getY)
62     calc.setDirection(0, meters)
63     val dest2D = calc.getDestinationGeographicPoint
64     val destPoint = geoFac.createPoint(new Coordinate(dest2D.getX, dest2D.getY))
65     point.distance(destPoint)
66   }
67 
68   protected def buffer(sf: SimpleFeature, meters: Double): SimpleFeature = {
69     val bufferedGeom = bufferGeom(getGeom(sf), meters)
70     builder.reset()
71     builder.init(sf)
72     builder.set(0, bufferedGeom)
73     builder.buildFeature(sf.getID)
74   }
75 
76   // transform the input tubeFeatures into the intermediate SF used by the
77   // tubing code consisting of three attributes (geom, startTime, endTime)
78   // handle date parsing from input -> TODO revisit date parsing...
79   protected def transform(tubeFeatures: SimpleFeatureCollection, dtgField: String): List[SimpleFeature] = {
80     val features = CloseableIterator(tubeFeatures.features).map { sf =>
81       val date = FastConverter.convert(sf.getAttribute(dtgField), classOf[Date])
82 
83       if (date == null) {
84         logger.error("Unable to retrieve date field from input tubeFeatures...ensure there a field named " + dtgField)
85         throw new IllegalArgumentException("Unable to retrieve date field from input tubeFeatures...ensure there a field named \"" + dtgField + "\"")
86       }
87 
88       builder.reset()
89       builder.buildFeature(sf.getID, sf.getDefaultGeometry, date, null)
90     }
91     features.toList
92   }
93 
94   /**
95     * Return an Array containing either 1 or 2 LineStrings that straddle but
96     * do not cross the IDL.
97     * @param input1 The first point in the segment
98     * @param input2 The second point in the segment
99     * @return an array of LineString containing either 1 or 2 LineStrings that do not
100     *         span the IDL.
101     */
102   protected def makeIDLSafeLineString(input1:Coordinate, input2:Coordinate): Geometry = {
103     //If the points cross the IDL we must generate two line segments
104     if (GeometryUtils.crossesIDL(input1, input2)) {
105       //Find the latitude where the segment intercepts the IDL
106       val latIntercept = GeometryUtils.calcIDLIntercept(input1, input2)
107       val p1 = new Coordinate(-180, latIntercept)
108       val p2 = new Coordinate(180, latIntercept)
109       //This orders the points so that point1 is always the east-most point
110       val (point1, point2) = if (input1.x > 0) (input1, input2) else (input2, input1)
111       val westLine = new LineString(new CoordinateArraySequence(Array(p1, point2)), geoFac)
112       val eastLine = new LineString(new CoordinateArraySequence(Array(point1, p2)), geoFac)
113       new MultiLineString(Array[LineString](westLine,eastLine), geoFac)
114     } else {
115       new LineString(new CoordinateArraySequence(Array(input1, input2)), geoFac)
116     }
117   }

120 }

122 /**
123  * Build a tube with no gap filling - only buffering and binning
124  */
125 class NoGapFill(tubeFeatures: SimpleFeatureCollection,
126                 bufferDistance: Double,
127                 maxBins: Int) extends TubeBuilder(tubeFeatures, bufferDistance, maxBins) with LazyLogging {
128 
129   // Bin ordered features into maxBins that retain order by date then union by geometry
130   private def timeBinAndUnion(features: Iterable[SimpleFeature], maxBins: Int): Iterator[SimpleFeature] = {
131     val numFeatures = features.size
132 
133     if (numFeatures == 0) { Iterator.empty } else {
134       //If 0 is passed in then don't bin the features, if 1 then make one bin, otherwise calculate number
135       //of bins based on numFeatures and maxBins
136       val binSize = maxBins match {
137         case 0 => 1
138         case 1 => numFeatures
139         case _ => numFeatures / maxBins + (if (numFeatures % maxBins == 0 ) 0 else 1)
140       }
141       features.grouped(binSize).zipWithIndex.map { case(bin, idx) => unionFeatures(bin.toSeq, idx.toString) }
142     }
143   }
144 
145   // Union features to create a single geometry and single combined time range
146   private def unionFeatures(orderedFeatures: Seq[SimpleFeature], id: String): SimpleFeature = {
147     import scala.collection.JavaConverters._
148     val geoms = orderedFeatures.map { sf => getGeom(sf) }
149     val unionGeom = geoFac.buildGeometry(geoms.asJava).union
150     val min = getStartTime(orderedFeatures.head)
151     val max = getStartTime(orderedFeatures.last)
152 
153     builder.reset()
154     builder.buildFeature(id, unionGeom, min, max)
155   }
156 
157   override def createTube(): Iterator[SimpleFeature] = {
158     logger.debug("Creating tube with no gap filling")
159 
160     val transformed = transform(tubeFeatures, dtgField)
161     val buffered = transformed.map(buffer(_, bufferDistance))
162     val sortedTube = buffered.sortBy(sf => getStartTime(sf).getTime)
163 
164     logger.debug(s"sorted tube size: ${sortedTube.size}")
165     timeBinAndUnion(sortedTube, maxBins)
166   }
167 }

172 /**
173  * Build a tube with gap filling that draws a line between time-ordered features
174  * from the given tubeFeatures
175  */
176 class LineGapFill(tubeFeatures: SimpleFeatureCollection,
177                   bufferDistance: Double,
178                   maxBins: Int) extends TubeBuilder(tubeFeatures, bufferDistance, maxBins) with LazyLogging {
179 
180   private val id = new AtomicInteger(0)
181 
182   private def nextId: String = id.getAndIncrement.toString
183 
184   override def createTube(): Iterator[SimpleFeature] = {
185     import org.locationtech.geomesa.utils.geotools.Conversions.RichGeometry
186 
187     logger.debug("Creating tube with line gap fill")
188 
189     val transformed = transform(tubeFeatures, dtgField)
190     val sortedTube = transformed.sortBy(sf => getStartTime(sf).getTime)
191     val pointsAndTimes = sortedTube.map(sf => (getGeom(sf).safeCentroid(), getStartTime(sf)))
192     val lineFeatures = if (pointsAndTimes.lengthCompare(1) == 0) {
193       val (p1, t1) = pointsAndTimes.head
194       logger.debug("Only a single result - can't create a line")
195       Iterator(builder.buildFeature(nextId, p1, t1, t1))
196     } else {
197       pointsAndTimes.sliding(2).map { case Seq((p1, t1), (p2, t2)) =>
198         val geo = if (p1.equals(p2)) p1 else makeIDLSafeLineString(p1.getCoordinate,p2.getCoordinate)
199         logger.debug(s"Created Line-filled Geometry: ${WKTUtils.write(geo)} From ${WKTUtils.write(p1)} and ${WKTUtils.write(p2)}")
200         builder.buildFeature(nextId, geo, t1, t2)
201       }
202     }
203     lineFeatures.map(buffer(_, bufferDistance))
204   }
205 }

216

220 
221   override def createTube(): Iterator[SimpleFeature] = {
222     import org.locationtech.geomesa.utils.geotools.Conversions.RichGeometry
223 
224     logger.debug("Creating tube with line interpolated line gap fill")
225 
226     val transformed = transform(tubeFeatures, dtgField)
227     val sortedTube = transformed.sortBy(sf => getStartTime(sf).getTime)
228     val pointsAndTimes = sortedTube.map(sf => (getGeom(sf).safeCentroid(), getStartTime(sf)))
229     val lineFeatures = if (pointsAndTimes.lengthCompare(1) == 0) {
230       val (p1, t1) = pointsAndTimes.head
231       logger.debug("Only a single result - can't create a line")
232       Iterator(builder.buildFeature(nextId, p1, t1, t1))
233     } else {
234       pointsAndTimes.sliding(2).flatMap { case Seq((p1, t1), (p2, t2)) =>
235         calc.setStartingGeographicPoint(p1.getX, p1.getY)
236         calc.setDestinationGeographicPoint(p2.getX, p2.getY)
237         val dist = calc.getOrthodromicDistance
238         //If the distance between points is greater than the buffer distance, segment the line
239         //So that no segment is larger than the buffer. This ensures that each segment has an
240         //times and distance. Also ensure that features do not share a time value.
241         val timeDiffMillis = t2.getTime - t1.getTime
242         val segCount = (dist / bufferDistance).toInt
243         val segDuration = timeDiffMillis / segCount
244         val timeSteps = NumericRange.inclusive(t1.getTime, t2.getTime, segDuration)
245         if (dist > bufferDistance && timeSteps.lengthCompare(1) > 0) {
246           val heading = calc.getAzimuth
247           var segStep = new Coordinate(p1.getX, p1.getY, 0)
248           timeSteps.sliding(2).map { case Seq(time0, time1) =>
249             val segP1 = segStep
250             calc.setStartingGeographicPoint(segP1.x, segP1.y)
251             calc.setDirection(heading, bufferDistance)
252             val destPoint = calc.getDestinationGeographicPoint
253             segStep = new Coordinate(destPoint.getX, destPoint.getY, 0)
254             val geo = makeIDLSafeLineString(segP1, segStep)
255             builder.buildFeature(nextId, geo, new Date(time0), new Date(time1))
256           }
257         } else {
258           val geo = if (p1.equals(p2)) { p1 } else { makeIDLSafeLineString(p1.getCoordinate, p2.getCoordinate) }
259           logger.debug(s"Created line-filled geometry: ${WKTUtils.write(geo)} " +
260               s"from ${WKTUtils.write(p1)} and ${WKTUtils.write(p2)}")
261           Seq(builder.buildFeature(nextId, geo, t1, t2))
262         }
263       }
264     }
265     lineFeatures.map(buffer(_, bufferDistance))
266   }
