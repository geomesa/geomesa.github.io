38 @DescribeProcess(
39   title = "Arrow Conversion",

44   /**
45     * Converts an input feature collection to arrow format
46     *
47     * @param features input features
48     * @param dictionaryFields attributes to dictionary encode, optional
49     * @return
50     */
51   @DescribeResult(description = "Encoded feature collection")
52   def execute(
53               @DescribeParameter(name = "features", description = "Input feature collection to encode")
54               features: SimpleFeatureCollection,
55               @DescribeParameter(name = "includeFids", description = "Include feature IDs in arrow file", min = 0)
56               includeFids: java.lang.Boolean,
57               @DescribeParameter(name = "proxyFids", description = "Proxy feature IDs to ints instead of strings", min = 0)
58               proxyFids: java.lang.Boolean,
59               @DescribeParameter(name = "formatVersion", description = "Arrow IPC format version", min = 0)
60               formatVersion: String,
61               @DescribeParameter(name = "dictionaryFields", description = "Attributes to dictionary encode", min = 0, max = 128, collectionType = classOf[String])
62               dictionaryFields: java.util.List[String],
63               @DescribeParameter(name = "sortField", description = "Attribute to sort by", min = 0)
64               sortField: String,
65               @DescribeParameter(name = "sortReverse", description = "Reverse the default sort order", min = 0)
66               sortReverse: java.lang.Boolean,
67               @DescribeParameter(name = "batchSize", description = "Number of features to include in each record batch", min = 0)
68               batchSize: java.lang.Integer,
69               @DescribeParameter(name = "flattenStruct", description = "Removes the outer SFT struct yielding top level feature access", min = 0)
70               flattenStruct: java.lang.Boolean
71              ): java.util.Iterator[Array[Byte]] = {
72 
73     import scala.collection.JavaConverters._
74 
75     logger.debug(s"Running arrow encoding for ${features.getClass.getName}")
76 
77     val sft = features.getSchema
78 
79     // validate inputs
80     val toEncode: Seq[String] = Option(dictionaryFields).map(_.asScala.toSeq).getOrElse(Seq.empty)
81     toEncode.foreach { attribute =>
82       if (sft.indexOf(attribute) == -1) {
83         throw new IllegalArgumentException(s"Attribute $attribute doesn't exist in $sft")
84       }
85     }
86 
87     val encoding = SimpleFeatureEncoding.min(includeFids == null || includeFids, proxyFids != null && proxyFids)
88     val ipcVersion = Option(formatVersion).getOrElse(FormatVersion.ArrowFormatVersion.get)
89     val reverse = Option(sortReverse).map(_.booleanValue())
90     val batch = Option(batchSize).map(_.intValue).getOrElse(ArrowProperties.BatchSize.get.toInt)
91 
92     val visitor =
93       new ArrowVisitor(sft, encoding, ipcVersion, toEncode, Option(sortField), reverse, false, batch, flattenStruct)
94     GeoMesaFeatureCollection.visit(features, visitor)
95     visitor.getResult.results
96   }
97 }

99 object ArrowConversionProcess {
100 
101   class ArrowVisitor(
102       sft: SimpleFeatureType,
103       encoding: SimpleFeatureEncoding,
104       ipcVersion: String,
105       dictionaryFields: Seq[String],
106       sortField: Option[String],
107       sortReverse: Option[Boolean],
108       preSorted: Boolean,
109       batchSize: Int,
110       flattenStruct: Boolean
111     ) extends GeoMesaProcessVisitor with LazyLogging {
112 
113     import scala.collection.JavaConverters._
114 
115     // for collecting results manually
116     private lazy val manualVisitor: ArrowManualVisitor = {
117       val sort = sortField.map(s => (s, sortReverse.getOrElse(false)))
118       val ipcOpts = FormatVersion.options(ipcVersion)
119       if (dictionaryFields.isEmpty && (sortField.isEmpty || preSorted)) {
120         new SimpleArrowManualVisitor(sft, encoding, ipcOpts, sort, batchSize, flattenStruct)
121       } else {
122         new ComplexArrowManualVisitor(sft, encoding, ipcOpts, dictionaryFields, sort, preSorted, batchSize, flattenStruct)
123       }
124     }
125 
126     private var result: Iterator[Array[Byte]] = _
127 
128     override def getResult: ArrowResult = {
129       if (result != null) {
130         ArrowResult(result.asJava)
131       } else {
132         ArrowResult(manualVisitor.results.asJava)
133       }
134     }
135 
136     // manually called for non-accumulo feature collections
137     override def visit(feature: Feature): Unit = manualVisitor.visit(feature.asInstanceOf[SimpleFeature])
138 
139     /**
140       * Optimized method to run distributed query. Sets the result, available from `getResult`
141       *
142       * @param source simple feature source
143       * @param query may contain additional filters to apply
144       */
145     override def execute(source: SimpleFeatureSource, query: Query): Unit = {
146       logger.debug(s"Visiting source type: ${source.getClass.getName}")
147 
148       query.getHints.put(QueryHints.ARROW_ENCODE, true)
149       query.getHints.put(QueryHints.ARROW_DICTIONARY_FIELDS, dictionaryFields.mkString(","))
150       query.getHints.put(QueryHints.ARROW_INCLUDE_FID, encoding.fids.isDefined)
151       query.getHints.put(QueryHints.ARROW_PROXY_FID, encoding.fids.contains(Encoding.Min))
152       query.getHints.put(QueryHints.ARROW_BATCH_SIZE, batchSize)
153       query.getHints.put(QueryHints.ARROW_FORMAT_VERSION, ipcVersion)
154       query.getHints.put(QueryHints.ARROW_FLATTEN_STRUCT, flattenStruct)
155       sortField.foreach(query.getHints.put(QueryHints.ARROW_SORT_FIELD, _))
156       sortReverse.foreach(query.getHints.put(QueryHints.ARROW_SORT_REVERSE, _))
157 
158       val features = SelfClosingIterator(source.getFeatures(query))
159       result = features.map(_.getAttribute(0).asInstanceOf[Array[Byte]])
160     }
161   }
162 
163   trait ArrowManualVisitor {
164     def visit(feature: SimpleFeature): Unit
165     def results: Iterator[Array[Byte]]
166   }
167 
168   /**
169     * Writes out features in batches, without sorting or dictionaries
170     *
171     * @param sft simple feature type
172     * @param encoding arrow encoding
173     * @param sort sort field, only used for metadata - no sorting will be done
174     * @param batchSize batch size
175     */
176   private class SimpleArrowManualVisitor(
177       sft: SimpleFeatureType,
178       encoding: SimpleFeatureEncoding,
179       ipcOpts: IpcOption,
180       sort: Option[(String, Boolean)],
181       batchSize: Int,
182       flattenStruct: Boolean
183     ) extends ArrowManualVisitor {
184 
185     private val out = new ByteArrayOutputStream()
186     private val bytes = ListBuffer.empty[Array[Byte]]
187     private var count = 0L
188 
189     private val writer =
190       SimpleFeatureArrowFileWriter(out, sft, Map.empty[String, ArrowDictionary], encoding, ipcOpts, sort, flattenStruct)
191 
192     override def visit(feature: SimpleFeature): Unit = {
193       writer.add(feature)
194       count += 1
195       if (count % batchSize == 0) {
196         writer.flush()
197         bytes.append(out.toByteArray)
198         out.reset()
199       }
200     }
201 
202     override def results: Iterator[Array[Byte]] = {
203       CloseWithLogging(writer)
204       bytes.append(out.toByteArray)
205       bytes.iterator
206     }
207   }
208 
209   /**
210     * Caches features locally in order to compute dictionaries and/or sorting
211     *
212     * @param sft simple feature type
213     * @param encoding arrow encoding
214     * @param dictionaryFields dictionary fields
215     * @param sort sort
216     * @param batchSize batch size
217     */
218   private class ComplexArrowManualVisitor(
219       sft: SimpleFeatureType,
220       encoding: SimpleFeatureEncoding,
221       ipcOpts: IpcOption,
222       dictionaryFields: Seq[String],
223       sort: Option[(String, Boolean)],
224       preSorted: Boolean,
225       batchSize: Int,
226       flattenStruct: Boolean
227     ) extends ArrowManualVisitor {
228 
229     private val features = ArrayBuffer.empty[SimpleFeature]
230 
231     override def visit(feature: SimpleFeature): Unit = {
232       // copy the feature in case it is being re-used in the iterator
233       features.append(ScalaSimpleFeature.copy(feature))
234     }
235 
236     override def results: Iterator[Array[Byte]] = {
237       val dictionaries = if (dictionaryFields.isEmpty) { Map.empty[String, ArrowDictionary] } else {
238         val indicesAndValues = dictionaryFields.map { field =>
239           (field, sft.indexOf(field), scala.collection.mutable.HashSet.empty[AnyRef])
240         }
241         features.foreach { f =>
242           indicesAndValues.foreach { case (_, i, v) => v.add(f.getAttribute(i)) }
243         }
244         indicesAndValues.map { case (n, i, v) =>
245           n -> ArrowDictionary.create(sft.getTypeName, i, v.toArray)(ClassTag[AnyRef](sft.getDescriptor(i).getType.getBinding))
246         }.toMap
247       }
248 
249       val ordering = if (preSorted) { None } else {
250         sort.map { case (field, reverse) => SimpleFeatureOrdering(sft, field, reverse) }
251       }
252       val sorted = ordering match {
253         case None    => features.iterator
254         case Some(o) => features.sorted(o).iterator
255       }
256 
257       buildResult(dictionaries, sorted)
258     }
259 
260     private def buildResult(dictionaries: Map[String, ArrowDictionary],
261                             sorted: Iterator[SimpleFeature]): Iterator[Array[Byte]] = {
262       val out = new ByteArrayOutputStream()
263       val bytes = ListBuffer.empty[Array[Byte]]
264 
265       WithClose(SimpleFeatureArrowFileWriter(out, sft, dictionaries, encoding, ipcOpts, sort, flattenStruct)) { writer =>
266         while (sorted.hasNext) { // send batches
267           var i = 0
268           while (i < batchSize && sorted.hasNext) {
269             writer.add(sorted.next)
270             i += 1
271           }
272           writer.flush()
273           bytes.append(out.toByteArray)
274           out.reset()
275         }
276       }
277       bytes.append(out.toByteArray)
278       bytes.iterator
279     }
280   }
281 
282   case class ArrowResult(results: java.util.Iterator[Array[Byte]]) extends AbstractCalcResult {
283     override def getValue: AnyRef = results
284   }
