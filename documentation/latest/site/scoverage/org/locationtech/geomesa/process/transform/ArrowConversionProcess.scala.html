38 @DescribeProcess(
39   title = "Arrow Conversion",

44   /**
45     * Converts an input feature collection to arrow format
46     *
47     * @param features input features
48     * @param dictionaryFields attributes to dictionary encode, optional
49     * @return
50     */
51   @DescribeResult(description = "Encoded feature collection")
52   def execute(
53               @DescribeParameter(name = "features", description = "Input feature collection to encode")
54               features: SimpleFeatureCollection,
55               @DescribeParameter(name = "includeFids", description = "Include feature IDs in arrow file", min = 0)
56               includeFids: java.lang.Boolean,
57               @DescribeParameter(name = "proxyFids", description = "Proxy feature IDs to ints instead of strings", min = 0)
58               proxyFids: java.lang.Boolean,
59               @DescribeParameter(name = "formatVersion", description = "Arrow IPC format version", min = 0)
60               formatVersion: String,
61               @DescribeParameter(name = "dictionaryFields", description = "Attributes to dictionary encode", min = 0, max = 128, collectionType = classOf[String])
62               dictionaryFields: java.util.List[String],
63               @DescribeParameter(name = "sortField", description = "Attribute to sort by", min = 0)
64               sortField: String,
65               @DescribeParameter(name = "sortReverse", description = "Reverse the default sort order", min = 0)
66               sortReverse: java.lang.Boolean,
67               @DescribeParameter(name = "batchSize", description = "Number of features to include in each record batch", min = 0)
68               batchSize: java.lang.Integer,
69               @DescribeParameter(name = "flattenStruct", description = "Removes the outer SFT struct yielding top level feature access", min = 0)
70               flattenStruct: java.lang.Boolean,
71               @DescribeParameter(name = "flipAxisOrder", description = "Flip the axis order of returned coordinates from latitude/longitude (default) to longitude/latitude", min = 0, defaultValue = "false")
72               flipAxisOrder: java.lang.Boolean = false
73              ): java.util.Iterator[Array[Byte]] = {
74 
75     import scala.collection.JavaConverters._
76 
77     logger.debug(s"Running arrow encoding for ${features.getClass.getName}")
78 
79     val sft = features.getSchema
80 
81     // validate inputs
82     val toEncode: Seq[String] = Option(dictionaryFields).map(_.asScala.toSeq).getOrElse(Seq.empty)
83     toEncode.foreach { attribute =>
84       if (sft.indexOf(attribute) == -1) {
85         throw new IllegalArgumentException(s"Attribute $attribute doesn't exist in $sft")
86       }
87     }
88 
89     val encoding = SimpleFeatureEncoding.min(includeFids == null || includeFids, proxyFids != null && proxyFids, Option(flipAxisOrder).exists(_.booleanValue))
90     val ipcVersion = Option(formatVersion).getOrElse(FormatVersion.ArrowFormatVersion.get)
91     val reverse = Option(sortReverse).map(_.booleanValue())
92     val batch = Option(batchSize).map(_.intValue).getOrElse(ArrowProperties.BatchSize.get.toInt)
93 
94     val visitor =
95       new ArrowVisitor(sft, encoding, ipcVersion, toEncode, Option(sortField), reverse, false, batch, flattenStruct)
96     GeoMesaFeatureCollection.visit(features, visitor)
97     visitor.getResult.results
98   }
99 }

101 object ArrowConversionProcess {
102 
103   class ArrowVisitor(
104       sft: SimpleFeatureType,
105       encoding: SimpleFeatureEncoding,
106       ipcVersion: String,
107       dictionaryFields: Seq[String],
108       sortField: Option[String],
109       sortReverse: Option[Boolean],
110       preSorted: Boolean,
111       batchSize: Int,
112       flattenStruct: Boolean
113     ) extends GeoMesaProcessVisitor with LazyLogging {
114 
115     import scala.collection.JavaConverters._
116 
117     // for collecting results manually
118     private lazy val manualVisitor: ArrowManualVisitor = {
119       val sort = sortField.map(s => (s, sortReverse.getOrElse(false)))
120       val ipcOpts = FormatVersion.options(ipcVersion)
121       if (dictionaryFields.isEmpty && (sortField.isEmpty || preSorted)) {
122         new SimpleArrowManualVisitor(sft, encoding, ipcOpts, sort, batchSize, flattenStruct)
123       } else {
124         new ComplexArrowManualVisitor(sft, encoding, ipcOpts, dictionaryFields, sort, preSorted, batchSize, flattenStruct)
125       }
126     }
127 
128     private var result: Iterator[Array[Byte]] = _
129 
130     override def getResult: ArrowResult = {
131       if (result != null) {
132         ArrowResult(result.asJava)
133       } else {
134         ArrowResult(manualVisitor.results.asJava)
135       }
136     }
137 
138     // manually called for non-accumulo feature collections
139     override def visit(feature: Feature): Unit = manualVisitor.visit(feature.asInstanceOf[SimpleFeature])
140 
141     /**
142       * Optimized method to run distributed query. Sets the result, available from `getResult`
143       *
144       * @param source simple feature source
145       * @param query may contain additional filters to apply
146       */
147     override def execute(source: SimpleFeatureSource, query: Query): Unit = {
148       logger.debug(s"Visiting source type: ${source.getClass.getName}")
149 
150       query.getHints.put(QueryHints.ARROW_ENCODE, true)
151       query.getHints.put(QueryHints.ARROW_DICTIONARY_FIELDS, dictionaryFields.mkString(","))
152       query.getHints.put(QueryHints.ARROW_INCLUDE_FID, encoding.fids.isDefined)
153       query.getHints.put(QueryHints.ARROW_PROXY_FID, encoding.fids.contains(Encoding.Min))
154       query.getHints.put(QueryHints.ARROW_BATCH_SIZE, batchSize)
155       query.getHints.put(QueryHints.ARROW_FORMAT_VERSION, ipcVersion)
156       query.getHints.put(QueryHints.ARROW_FLATTEN_STRUCT, flattenStruct)
157       query.getHints.put(QueryHints.FLIP_AXIS_ORDER, encoding.flipAxisOrder)
158       sortField.foreach(query.getHints.put(QueryHints.ARROW_SORT_FIELD, _))
159       sortReverse.foreach(query.getHints.put(QueryHints.ARROW_SORT_REVERSE, _))
160 
161       val features = SelfClosingIterator(source.getFeatures(query))
162       result = features.map(_.getAttribute(0).asInstanceOf[Array[Byte]])
163     }
164   }
165 
166   trait ArrowManualVisitor {
167     def visit(feature: SimpleFeature): Unit
168     def results: Iterator[Array[Byte]]
169   }
170 
171   /**
172     * Writes out features in batches, without sorting or dictionaries
173     *
174     * @param sft simple feature type
175     * @param encoding arrow encoding
176     * @param sort sort field, only used for metadata - no sorting will be done
177     * @param batchSize batch size
178     */
179   private class SimpleArrowManualVisitor(
180       sft: SimpleFeatureType,
181       encoding: SimpleFeatureEncoding,
182       ipcOpts: IpcOption,
183       sort: Option[(String, Boolean)],
184       batchSize: Int,
185       flattenStruct: Boolean
186     ) extends ArrowManualVisitor {
187 
188     private val out = new ByteArrayOutputStream()
189     private val bytes = ListBuffer.empty[Array[Byte]]
190     private var count = 0L
191 
192     private val writer =
193       SimpleFeatureArrowFileWriter(out, sft, Map.empty[String, ArrowDictionary], encoding, ipcOpts, sort, flattenStruct)
194 
195     override def visit(feature: SimpleFeature): Unit = {
196       writer.add(feature)
197       count += 1
198       if (count % batchSize == 0) {
199         writer.flush()
200         bytes.append(out.toByteArray)
201         out.reset()
202       }
203     }
204 
205     override def results: Iterator[Array[Byte]] = {
206       CloseWithLogging(writer)
207       bytes.append(out.toByteArray)
208       bytes.iterator
209     }
210   }
211 
212   /**
213     * Caches features locally in order to compute dictionaries and/or sorting
214     *
215     * @param sft simple feature type
216     * @param encoding arrow encoding
217     * @param dictionaryFields dictionary fields
218     * @param sort sort
219     * @param batchSize batch size
220     */
221   private class ComplexArrowManualVisitor(
222       sft: SimpleFeatureType,
223       encoding: SimpleFeatureEncoding,
224       ipcOpts: IpcOption,
225       dictionaryFields: Seq[String],
226       sort: Option[(String, Boolean)],
227       preSorted: Boolean,
228       batchSize: Int,
229       flattenStruct: Boolean
230     ) extends ArrowManualVisitor {
231 
232     import org.locationtech.geomesa.utils.geotools.RichAttributeDescriptors.RichAttributeDescriptor
233 
234     import scala.collection.JavaConverters._
235 
236     private val features = ArrayBuffer.empty[SimpleFeature]
237 
238     override def visit(feature: SimpleFeature): Unit = {
239       // copy the feature in case it is being re-used in the iterator
240       features.append(ScalaSimpleFeature.copy(feature))
241     }
242 
243     override def results: Iterator[Array[Byte]] = {
244       val dictionaries = dictionaryFields.map { field =>
245         val i = sft.indexOf(field)
246         val descriptor = sft.getDescriptor(i)
247         val isList = descriptor.isList
248         val values = scala.collection.mutable.HashSet.empty[AnyRef]
249         features.foreach { f =>
250           val value = f.getAttribute(i)
251           if (value != null) {
252             if (isList) {
253               // for list types encode the list elements instead of the list itself
254               value.asInstanceOf[java.util.List[AnyRef]].asScala.foreach(values.add)
255             } else {
256               values.add(value)
257             }
258           }
259         }
260         val binding = if (isList) { descriptor.getListType() } else { descriptor.getType.getBinding }
261         field -> ArrowDictionary.create(sft.getTypeName, i, values.toArray)(ClassTag[AnyRef](binding))
262       }
263 
264       val sorted = sort match {
265         case Some((field, reverse)) if !preSorted => features.sorted(SimpleFeatureOrdering(sft, field, reverse)).iterator
266         case _ => features.iterator
267       }
268 
269       val out = new ByteArrayOutputStream()
270       val bytes = ListBuffer.empty[Array[Byte]]
271 
272       WithClose(SimpleFeatureArrowFileWriter(out, sft, dictionaries.toMap, encoding, ipcOpts, sort, flattenStruct)) { writer =>
273         while (sorted.hasNext) { // send batches
274           var i = 0
275           while (i < batchSize && sorted.hasNext) {
276             writer.add(sorted.next)
277             i += 1
278           }
279           writer.flush()
280           bytes.append(out.toByteArray)
281           out.reset()
282         }
283       }
284       bytes.append(out.toByteArray)
285       bytes.iterator
286     }
287   }

290     override def getValue: AnyRef = results
291   }
