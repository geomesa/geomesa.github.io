44   /**
45     * Converts an input feature collection to arrow format
46     *
47     * @param features input features
48     * @param dictionaryFields attributes to dictionary encode, optional
49     * @return
50     */
51   @DescribeResult(description = "Encoded feature collection")
52   def execute(
53               @DescribeParameter(name = "features", description = "Input feature collection to encode")
54               features: SimpleFeatureCollection,
55               @DescribeParameter(name = "includeFids", description = "Include feature IDs in arrow file", min = 0)
56               includeFids: java.lang.Boolean,
57               @DescribeParameter(name = "proxyFids", description = "Proxy feature IDs to ints instead of strings", min = 0)
58               proxyFids: java.lang.Boolean,
59               @DescribeParameter(name = "formatVersion", description = "Arrow IPC format version", min = 0)
60               formatVersion: String,
61               @DescribeParameter(name = "dictionaryFields", description = "Attributes to dictionary encode", min = 0, max = 128, collectionType = classOf[String])
62               dictionaryFields: java.util.List[String],
63               @DescribeParameter(name = "sortField", description = "Attribute to sort by", min = 0)
64               sortField: String,
65               @DescribeParameter(name = "sortReverse", description = "Reverse the default sort order", min = 0)
66               sortReverse: java.lang.Boolean,
67               @DescribeParameter(name = "batchSize", description = "Number of features to include in each record batch", min = 0)
68               batchSize: java.lang.Integer
69              ): java.util.Iterator[Array[Byte]] = {
70 
71     import scala.collection.JavaConverters._
72 
73     logger.debug(s"Running arrow encoding for ${features.getClass.getName}")
74 
75     val sft = features.getSchema
76 
77     // validate inputs
78     val toEncode: Seq[String] = Option(dictionaryFields).map(_.asScala.toSeq).getOrElse(Seq.empty)
79     toEncode.foreach { attribute =>
80       if (sft.indexOf(attribute) == -1) {
81         throw new IllegalArgumentException(s"Attribute $attribute doesn't exist in $sft")
82       }
83     }
84 
85     val encoding = SimpleFeatureEncoding.min(includeFids == null || includeFids, proxyFids != null && proxyFids)
86     val ipcVersion = Option(formatVersion).getOrElse(FormatVersion.ArrowFormatVersion.get)
87     val reverse = Option(sortReverse).map(_.booleanValue())
88     val batch = Option(batchSize).map(_.intValue).getOrElse(ArrowProperties.BatchSize.get.toInt)
89 
90     val visitor =
91       new ArrowVisitor(sft, encoding, ipcVersion, toEncode, Option(sortField), reverse, false, batch)
92     GeoMesaFeatureCollection.visit(features, visitor)
93     visitor.getResult.results
94   }
95 }

99   class ArrowVisitor(
100       sft: SimpleFeatureType,
101       encoding: SimpleFeatureEncoding,
102       ipcVersion: String,
103       dictionaryFields: Seq[String],
104       sortField: Option[String],
105       sortReverse: Option[Boolean],
106       preSorted: Boolean,
107       batchSize: Int
108     ) extends GeoMesaProcessVisitor with LazyLogging {
109 
110     import scala.collection.JavaConverters._
111 
112     // for collecting results manually
113     private lazy val manualVisitor: ArrowManualVisitor = {
114       val sort = sortField.map(s => (s, sortReverse.getOrElse(false)))
115       val ipcOpts = FormatVersion.options(ipcVersion)
116       if (dictionaryFields.isEmpty && (sortField.isEmpty || preSorted)) {
117         new SimpleArrowManualVisitor(sft, encoding, ipcOpts, sort, batchSize)
118       } else {
119         new ComplexArrowManualVisitor(sft, encoding, ipcOpts, dictionaryFields, sort, preSorted, batchSize)
120       }
121     }
122 
123     private var result: Iterator[Array[Byte]] = _
124 
125     override def getResult: ArrowResult = {
126       if (result != null) {
127         ArrowResult(result.asJava)
128       } else {
129         ArrowResult(manualVisitor.results.asJava)
130       }
131     }
132 
133     // manually called for non-accumulo feature collections
134     override def visit(feature: Feature): Unit = manualVisitor.visit(feature.asInstanceOf[SimpleFeature])
135 
136     /**
137       * Optimized method to run distributed query. Sets the result, available from `getResult`
138       *
139       * @param source simple feature source
140       * @param query may contain additional filters to apply
141       */
142     override def execute(source: SimpleFeatureSource, query: Query): Unit = {
143       logger.debug(s"Visiting source type: ${source.getClass.getName}")
144 
145       query.getHints.put(QueryHints.ARROW_ENCODE, true)
146       query.getHints.put(QueryHints.ARROW_DICTIONARY_FIELDS, dictionaryFields.mkString(","))
147       query.getHints.put(QueryHints.ARROW_INCLUDE_FID, encoding.fids.isDefined)
148       query.getHints.put(QueryHints.ARROW_PROXY_FID, encoding.fids.contains(Encoding.Min))
149       query.getHints.put(QueryHints.ARROW_BATCH_SIZE, batchSize)
150       query.getHints.put(QueryHints.ARROW_FORMAT_VERSION, ipcVersion)
151       sortField.foreach(query.getHints.put(QueryHints.ARROW_SORT_FIELD, _))
152       sortReverse.foreach(query.getHints.put(QueryHints.ARROW_SORT_REVERSE, _))
153 
154       val features = SelfClosingIterator(source.getFeatures(query))
155       result = features.map(_.getAttribute(0).asInstanceOf[Array[Byte]])
156     }
157   }
158 
159   trait ArrowManualVisitor {
160     def visit(feature: SimpleFeature): Unit
161     def results: Iterator[Array[Byte]]
162   }
163 
164   /**
165     * Writes out features in batches, without sorting or dictionaries
166     *
167     * @param sft simple feature type
168     * @param encoding arrow encoding
169     * @param sort sort field, only used for metadata - no sorting will be done
170     * @param batchSize batch size
171     */
172   private class SimpleArrowManualVisitor(
173       sft: SimpleFeatureType,
174       encoding: SimpleFeatureEncoding,
175       ipcOpts: IpcOption,
176       sort: Option[(String, Boolean)],
177       batchSize: Int
178     ) extends ArrowManualVisitor {
179 
180     private val out = new ByteArrayOutputStream()
181     private val bytes = ListBuffer.empty[Array[Byte]]
182     private var count = 0L
183 
184     private val writer =
185       SimpleFeatureArrowFileWriter(out, sft, Map.empty[String, ArrowDictionary], encoding, ipcOpts, sort)
186 
187     override def visit(feature: SimpleFeature): Unit = {
188       writer.add(feature)
189       count += 1
190       if (count % batchSize == 0) {
191         writer.flush()
192         bytes.append(out.toByteArray)
193         out.reset()
194       }
195     }
196 
197     override def results: Iterator[Array[Byte]] = {
198       CloseWithLogging(writer)
199       bytes.append(out.toByteArray)
200       bytes.iterator
201     }
202   }
203 
204   /**
205     * Caches features locally in order to compute dictionaries and/or sorting
206     *
207     * @param sft simple feature type
208     * @param encoding arrow encoding
209     * @param dictionaryFields dictionary fields
210     * @param sort sort
211     * @param batchSize batch size
212     */
213   private class ComplexArrowManualVisitor(
214       sft: SimpleFeatureType,
215       encoding: SimpleFeatureEncoding,
216       ipcOpts: IpcOption,
217       dictionaryFields: Seq[String],
218       sort: Option[(String, Boolean)],
219       preSorted: Boolean,
220       batchSize: Int
221     ) extends ArrowManualVisitor {
222 
223     private val features = ArrayBuffer.empty[SimpleFeature]
224 
225     override def visit(feature: SimpleFeature): Unit = {
226       // copy the feature in case it is being re-used in the iterator
227       features.append(ScalaSimpleFeature.copy(feature))
228     }
229 
230     override def results: Iterator[Array[Byte]] = {
231       val dictionaries = if (dictionaryFields.isEmpty) { Map.empty[String, ArrowDictionary] } else {
232         val indicesAndValues = dictionaryFields.map { field =>
233           (field, sft.indexOf(field), scala.collection.mutable.HashSet.empty[AnyRef])
234         }
235         features.foreach { f =>
236           indicesAndValues.foreach { case (_, i, v) => v.add(f.getAttribute(i)) }
237         }
238         indicesAndValues.map { case (n, i, v) =>
239           n -> ArrowDictionary.create(sft.getTypeName, i, v.toArray)(ClassTag[AnyRef](sft.getDescriptor(i).getType.getBinding))
240         }.toMap
241       }
242 
243       val ordering = if (preSorted) { None } else {
244         sort.map { case (field, reverse) => SimpleFeatureOrdering(sft, field, reverse) }
245       }
246       val sorted = ordering match {
247         case None    => features.iterator
248         case Some(o) => features.sorted(o).iterator
249       }
250 
251       buildResult(dictionaries, sorted)
252     }
253 
254     private def buildResult(dictionaries: Map[String, ArrowDictionary],
255                             sorted: Iterator[SimpleFeature]): Iterator[Array[Byte]] = {
256       val out = new ByteArrayOutputStream()
257       val bytes = ListBuffer.empty[Array[Byte]]
258 
259       WithClose(SimpleFeatureArrowFileWriter(out, sft, dictionaries, encoding, ipcOpts, sort)) { writer =>
260         while (sorted.hasNext) { // send batches
261           var i = 0
262           while (i < batchSize && sorted.hasNext) {
263             writer.add(sorted.next)
264             i += 1
265           }
266           writer.flush()
267           bytes.append(out.toByteArray)
268           out.reset()
269         }
270       }
271       bytes.append(out.toByteArray)
272       bytes.iterator
273     }
274   }
