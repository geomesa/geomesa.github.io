21 @DescribeProcess(
22   title = "Arrow Conversion",
23   description = "Converts a feature collection to arrow format"
24 )
25 class ArrowConversionProcess extends GeoMesaProcess with LazyLogging {
26 
27   /**
28     * Converts an input feature collection to arrow format
29     *
30     * @param features input features
31     * @param dictionaryFields attributes to dictionary encode, optional
32     * @return
33     */
34   @DescribeResult(description = "Encoded feature collection")
35   def execute(
36               @DescribeParameter(name = "features", description = "Input feature collection to encode")
37               features: SimpleFeatureCollection,
38               @DescribeParameter(name = "includeFids", description = "Include feature IDs in arrow file", min = 0)
39               includeFids: java.lang.Boolean,
40               @DescribeParameter(name = "proxyFids", description = "Proxy feature IDs to ints instead of strings", min = 0)
41               proxyFids: java.lang.Boolean,
42               @DescribeParameter(name = "formatVersion", description = "Arrow IPC format version", min = 0)
43               formatVersion: String,
44               @DescribeParameter(name = "dictionaryFields", description = "Attributes to dictionary encode", min = 0, max = 128, collectionType = classOf[String])
45               dictionaryFields: java.util.List[String],
46               @DescribeParameter(name = "sortField", description = "Attribute to sort by", min = 0)
47               sortField: String,
48               @DescribeParameter(name = "sortReverse", description = "Reverse the default sort order", min = 0)
49               sortReverse: java.lang.Boolean,
50               @DescribeParameter(name = "batchSize", description = "Number of features to include in each record batch", min = 0)
51               batchSize: java.lang.Integer,
52               @DescribeParameter(name = "flattenStruct", description = "Removes the outer SFT struct yielding top level feature access", min = 0)
53               flattenStruct: java.lang.Boolean,
54               @DescribeParameter(name = "flipAxisOrder", description = "Flip the axis order of returned coordinates from latitude/longitude (default) to longitude/latitude", min = 0, defaultValue = "false")
55               flipAxisOrder: java.lang.Boolean = false
56              ): java.util.Iterator[Array[Byte]] = {
57 
58     import scala.collection.JavaConverters._
59 
60     logger.debug(s"Running arrow encoding for ${features.getClass.getName}")
61 
62     val sft = features.getSchema
63 
64     // validate inputs
65     val toEncode: Seq[String] = Option(dictionaryFields).map(_.asScala.toSeq).getOrElse(Seq.empty)
66     toEncode.foreach { attribute =>
67       if (sft.indexOf(attribute) == -1) {
68         throw new IllegalArgumentException(s"Attribute $attribute doesn't exist in $sft")
69       }
70     }
71 
72     val encoding = SimpleFeatureEncoding.min(includeFids == null || includeFids, proxyFids != null && proxyFids, Option(flipAxisOrder).exists(_.booleanValue))
73     val ipcVersion = Option(formatVersion).getOrElse(FormatVersion.ArrowFormatVersion.get)
74     val reverse = Option(sortReverse).map(_.booleanValue())
75     val batch = Option(batchSize).map(_.intValue).getOrElse(ArrowProperties.BatchSize.get.toInt)
76 
77     val visitor =
78       new ArrowVisitor(sft, encoding, ipcVersion, toEncode, Option(sortField), reverse, false, batch, flattenStruct)
79     GeoMesaFeatureCollection.visit(features, visitor)
80     visitor.getResult.results
81   }
82 }
