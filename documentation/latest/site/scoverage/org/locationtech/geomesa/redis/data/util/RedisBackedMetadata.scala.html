19 /**
20   * Redis-backed metadata implementation. Metadata is stored as a redis hashset
21   *
22   * @param connection connection pool
23   * @param table metadata table name
24   * @param serializer serializer
25   * @tparam T type param
26   */
27 class RedisBackedMetadata[T](connection: Pool[_ <: CloseableJedisCommands], table: String, val serializer: MetadataSerializer[T])
28     extends KeyValueStoreMetadata[T] {
29 
30   import scala.collection.JavaConverters._
31 
32   // note: connection pool gets closed by the owning data store
33 
34   val key: Array[Byte] = table.getBytes(StandardCharsets.UTF_8)
35 
36   override protected def write(rows: Seq[(Array[Byte], Array[Byte])]): Unit = {
37     if (rows.lengthCompare(1) == 0) {
38       val (k, v) = rows.head
39       WithClose(connection.getResource)(_.hset(key, k, v))
40     } else {
41       val map = new java.util.HashMap[Array[Byte], Array[Byte]](rows.size)
42       rows.foreach { case (k, v) => map.put(k, v) }
43       WithClose(connection.getResource)(_.hset(key, map))
44     }
45   }
46 
47   override protected def delete(rows: Seq[Array[Byte]]): Unit =
48     WithClose(connection.getResource)(_.hdel(key, rows: _*))
49 
50   override protected def scanValue(row: Array[Byte]): Option[Array[Byte]] =
51     Option(WithClose(connection.getResource)(_.hget(key, row)))
52 
53   override protected def scanRows(prefix: Option[Array[Byte]]): CloseableIterator[(Array[Byte], Array[Byte])] = {
54     val all = WithClose(connection.getResource)(_.hgetAll(key)).asScala.iterator
55     prefix match {
56       case None    => CloseableIterator(all)
57       case Some(p) => CloseableIterator(all.filter { case (k, _) => k.startsWith(p) })
58     }
59   }
60 
61   override protected def createEmptyBackup(timestamp: String): RedisBackedMetadata[T] =
62     new RedisBackedMetadata(connection, s"${table}_${timestamp}_bak", serializer)
63 
64   override protected def checkIfTableExists: Boolean = true
65 
66   override protected def createTable(): Unit = {}
67 }
