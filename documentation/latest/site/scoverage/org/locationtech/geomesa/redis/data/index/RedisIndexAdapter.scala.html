113   val ExclusiveRangePrefix: Byte = "(".getBytes(StandardCharsets.UTF_8).head

115   /**
116     * Convert a byte range to a redis zlex range
117     *
118     * @param byteRange geomesa range
119     * @return
120     */
121   private def toRedisRange(byteRange: ByteRange): BoundedByteRange = byteRange match {
122     case BoundedByteRange(start, end) =>
123       val rangeStart = if (start.isEmpty) { RedisIndexAdapter.MinRange } else {
124         val range = Array.ofDim[Byte](start.length + 1)
125         System.arraycopy(start, 0, range, 1, start.length)
126         range(0) = RedisIndexAdapter.InclusiveRangePrefix
127         range
128       }
129       val rangeEnd = if (end.isEmpty) { RedisIndexAdapter.MaxRange } else {
130         val range = Array.ofDim[Byte](end.length + 1)
131         System.arraycopy(end, 0, range, 1, end.length)
132         range(0) = RedisIndexAdapter.ExclusiveRangePrefix
133         range
134       }
135       BoundedByteRange(rangeStart, rangeEnd)
136 
137     case SingleRowByteRange(row) =>
138       val rangeStart = Array.ofDim[Byte](row.length + 1)
139       System.arraycopy(row, 0, rangeStart, 1, row.length)
140       rangeStart(0) = RedisIndexAdapter.InclusiveRangePrefix
141       // since the value is appended to the row, we have to add a suffix
142       val rangeEnd = Array.ofDim[Byte](row.length + 1 + ByteRange.UnboundedUpperRange.length)
143       System.arraycopy(row, 0, rangeEnd, 1, row.length)
144       System.arraycopy(ByteRange.UnboundedUpperRange, 0, rangeEnd, row.length + 1, ByteRange.UnboundedUpperRange.length)
145       rangeEnd(0) = RedisIndexAdapter.ExclusiveRangePrefix
146       BoundedByteRange(rangeStart, rangeEnd)
147   }
148 
149   /**
150     * Convert a byte range into a zlex range, specifically for the id index. Since we store the length
151     * of the id in the row key, we have to prepend that to our ranges
152     *
153     * @param byteRange geomesa range
154     * @return
155     */
156   private def toRedisIdRange(byteRange: ByteRange): BoundedByteRange = byteRange match {
157     case BoundedByteRange(start, end) =>
158       val rangeStart = if (start.isEmpty) { RedisIndexAdapter.MinRange } else {
159         // add the two byte length prefix
160         val range = Array.ofDim[Byte](start.length + 3)
161         System.arraycopy(start, 0, range, 3, start.length)
162         ByteArrays.writeShort(start.length.toShort, range, 1)
163         range(0) = RedisIndexAdapter.InclusiveRangePrefix
164         range
165       }
166       val rangeEnd = if (end.isEmpty) { RedisIndexAdapter.MaxRange } else {
167         // add the two byte length prefix
168         val range = Array.ofDim[Byte](end.length + 3)
169         System.arraycopy(end, 0, range, 3, end.length)
170         ByteArrays.writeShort(end.length.toShort, range, 1)
171         range(0) = RedisIndexAdapter.ExclusiveRangePrefix
172         range
173       }
174       BoundedByteRange(rangeStart, rangeEnd)
175 
176     case SingleRowByteRange(row) =>
177       // add the two byte length prefix
178       val rangeStart = Array.ofDim[Byte](row.length + 3)
179       System.arraycopy(row, 0, rangeStart, 3, row.length)
180       ByteArrays.writeShort(row.length.toShort, rangeStart, 1)
181       rangeStart(0) = RedisIndexAdapter.InclusiveRangePrefix
182       // since the value is appended to the row, we have to add a suffix
183       val rangeEnd = Array.ofDim[Byte](row.length + 3 + ByteRange.UnboundedUpperRange.length)
184       System.arraycopy(row, 0, rangeEnd, 3, row.length)
185       System.arraycopy(ByteRange.UnboundedUpperRange, 0, rangeEnd, row.length + 3, ByteRange.UnboundedUpperRange.length)
186       ByteArrays.writeShort(row.length.toShort, rangeEnd, 1)
187       rangeEnd(0) = RedisIndexAdapter.ExclusiveRangePrefix
188       BoundedByteRange(rangeStart, rangeEnd)
189   }

194     private var idSerializer: (Array[Byte], Int, Int, SimpleFeature) => String = _

196     override def apply(result: Array[Byte]): SimpleFeature = {
197       // parse out the feature id and the serialized value from the concatenated row + value
198       val idStart = index.getIdOffset(result, 0, result.length)
199       val idLength = ByteArrays.readShort(result, idStart)
200       val id = idSerializer(result, idStart + 2, idLength, null)
201       val valueStart = idStart + idLength + 2
202       serializer.deserialize(id, result, valueStart, result.length - valueStart)
203     }

206       idSerializer = GeoMesaFeatureIndex.idFromBytes(index.sft)

211   /**
212     * Writer for redis
213     *
214     * @param jedis connection
215     * @param indices indices to write to
216     * @param partition partition to write to
217     * @param aging age-off writer
218     * @param wrapper feature wrapper
219     */
220   class RedisIndexWriter(
221       jedis: Pool[_ <: CloseableJedisCommands],
222       indices: Seq[GeoMesaFeatureIndex[_, _]],
223       partition: Option[String],
224       aging: Option[AgeOffWriter],
225       wrapper: FeatureWrapper[RedisWritableFeature]
226     ) extends BaseIndexWriter[RedisWritableFeature](indices, wrapper) {
227 
228     private val batchSize = RedisSystemProperties.WriteBatchSize.toInt match {
229       case Some(s) if s > 0 => s - 1
230       case _ =>
231         throw new IllegalStateException(s"Value '${RedisSystemProperties.WriteBatchSize.get}' for " +
232             s"'${RedisSystemProperties.WriteBatchSize.property}' is not a positive int")
233     }
234 
235     private val tables = indices.toArray.map { index =>
236       // should always be writing to a single table here
237       index.getTableName(partition).getBytes(StandardCharsets.UTF_8)
238     }
239 
240     private val inserts = Array.fill[java.util.Map[Array[Byte], java.lang.Double]](tables.length)(new java.util.HashMap[Array[Byte], java.lang.Double]())
241     private val deletes = Array.fill[ArrayBuffer[Array[Byte]]](tables.length)(ArrayBuffer.empty[Array[Byte]])
242 
243     private var i = 0
244     private var batch = 0
245 
246     private val errors = ArrayBuffer.empty[Throwable]
247 
248     override protected def append(feature: RedisWritableFeature, values: Array[RowKeyValue[_]]): Unit = {
249       i = 0
250       while (i < values.length) {
251         val insert = inserts(i)
252         values(i) match {
253           case kv: SingleRowKeyValue[_] =>
254             kv.values.foreach(v => insert.put(ByteArrays.concat(kv.row, v.value), 0d))
255           case kv: MultiRowKeyValue[_] =>
256             kv.rows.foreach(row => kv.values.foreach(v => insert.put(ByteArrays.concat(row, v.value), 0d)))
257         }
258         i += 1
259       }
260 
261       aging.foreach(_.write(feature))
262 
263       if (batch < batchSize) {
264         batch += 1
265       } else {
266         flush()
267       }
268     }
269 
270     override protected def update(
271         feature: RedisWritableFeature,
272         values: Array[RowKeyValue[_]],
273         previous: RedisWritableFeature,
274         previousValues: Array[RowKeyValue[_]]): Unit = {
275       delete(previous, previousValues)
276       append(feature, values)
277     }
278 
279     override protected def delete(feature: RedisWritableFeature, values: Array[RowKeyValue[_]]): Unit = {
280       i = 0
281       while (i < values.length) {
282         val buffer = deletes(i)
283         values(i) match {
284           case kv: SingleRowKeyValue[_] =>
285             kv.values.foreach(v => buffer.append(ByteArrays.concat(kv.row, v.value)))
286           case kv: MultiRowKeyValue[_] =>
287             kv.rows.foreach(row => kv.values.foreach(v => buffer.append(ByteArrays.concat(row, v.value))))
288         }
289         i += 1
290       }
291 
292       aging.foreach(_.delete(feature))
293 
294       if (batch < batchSize) {
295         batch += 1
296       } else {
297         flush()
298       }
299     }
300 
301     override def flush(): Unit = {
302       i = 0
303       while (i < tables.length) {
304         if (deletes(i).nonEmpty) {
305           try { WithClose(jedis.getResource)(_.zrem(tables(i), deletes(i).toSeq: _*)) } catch {
306             case NonFatal(e) => errors.append(e)
307           }
308           deletes(i).clear()
309         }
310         if (!inserts(i).isEmpty) {
311           try { WithClose(jedis.getResource)(_.zadd(tables(i), inserts(i))) } catch {
312             case NonFatal(e) => errors.append(e)
313           }
314           inserts(i).clear()
315         }
316         i += 1
317       }
318       batch = 0
319 
320       try { aging.foreach(_.flush()) } catch {
321         case NonFatal(e) => errors.append(e)
322       }
323 
324       throwErrors()
325     }
326 
327     override def close(): Unit = {
328       try { flush() } catch {
329         case NonFatal(e) => errors.append(e)
330       }
331       try { aging.foreach(_.close()) } catch {
332         case NonFatal(e) => errors.append(e)
333       }
334       throwErrors()
335     }
336 
337     private def throwErrors(): Unit = {
338       if (errors.nonEmpty) {
339         val error = errors.head
340         errors.tail.foreach(error.addSuppressed)
341         errors.clear()
342         throw error
343       }
344     }
345   }
