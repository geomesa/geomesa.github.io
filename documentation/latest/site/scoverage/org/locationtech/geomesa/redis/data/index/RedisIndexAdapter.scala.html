106   /**
107     * Convert a byte range to a redis zlex range
108     *
109     * @param byteRange geomesa range
110     * @return
111     */
112   private def toRedisRange(byteRange: ByteRange): BoundedByteRange = byteRange match {
113     case BoundedByteRange(start, end) =>
114       val rangeStart = if (start.isEmpty) { RedisIndexAdapter.MinRange } else {
115         val range = Array.ofDim[Byte](start.length + 1)
116         System.arraycopy(start, 0, range, 1, start.length)
117         range(0) = RedisIndexAdapter.InclusiveRangePrefix
118         range
119       }
120       val rangeEnd = if (end.isEmpty) { RedisIndexAdapter.MaxRange } else {
121         val range = Array.ofDim[Byte](end.length + 1)
122         System.arraycopy(end, 0, range, 1, end.length)
123         range(0) = RedisIndexAdapter.ExclusiveRangePrefix
124         range
125       }
126       BoundedByteRange(rangeStart, rangeEnd)
127 
128     case SingleRowByteRange(row) =>
129       val rangeStart = Array.ofDim[Byte](row.length + 1)
130       System.arraycopy(row, 0, rangeStart, 1, row.length)
131       rangeStart(0) = RedisIndexAdapter.InclusiveRangePrefix
132       // since the value is appended to the row, we have to add a suffix
133       val rangeEnd = Array.ofDim[Byte](row.length + 1 + ByteRange.UnboundedUpperRange.length)
134       System.arraycopy(row, 0, rangeEnd, 1, row.length)
135       System.arraycopy(ByteRange.UnboundedUpperRange, 0, rangeEnd, row.length + 1, ByteRange.UnboundedUpperRange.length)
136       rangeEnd(0) = RedisIndexAdapter.ExclusiveRangePrefix
137       BoundedByteRange(rangeStart, rangeEnd)
138   }
139 
140   /**
141     * Convert a byte range into a zlex range, specifically for the id index. Since we store the length
142     * of the id in the row key, we have to prepend that to our ranges
143     *
144     * @param byteRange geomesa range
145     * @return
146     */
147   private def toRedisIdRange(byteRange: ByteRange): BoundedByteRange = byteRange match {
148     case BoundedByteRange(start, end) =>
149       val rangeStart = if (start.isEmpty) { RedisIndexAdapter.MinRange } else {
150         // add the two byte length prefix
151         val range = Array.ofDim[Byte](start.length + 3)
152         System.arraycopy(start, 0, range, 3, start.length)
153         ByteArrays.writeShort(start.length.toShort, range, 1)
154         range(0) = RedisIndexAdapter.InclusiveRangePrefix
155         range
156       }
157       val rangeEnd = if (end.isEmpty) { RedisIndexAdapter.MaxRange } else {
158         // add the two byte length prefix
159         val range = Array.ofDim[Byte](end.length + 3)
160         System.arraycopy(end, 0, range, 3, end.length)
161         ByteArrays.writeShort(end.length.toShort, range, 1)
162         range(0) = RedisIndexAdapter.ExclusiveRangePrefix
163         range
164       }
165       BoundedByteRange(rangeStart, rangeEnd)
166 
167     case SingleRowByteRange(row) =>
168       // add the two byte length prefix
169       val rangeStart = Array.ofDim[Byte](row.length + 3)
170       System.arraycopy(row, 0, rangeStart, 3, row.length)
171       ByteArrays.writeShort(row.length.toShort, rangeStart, 1)
172       rangeStart(0) = RedisIndexAdapter.InclusiveRangePrefix
173       // since the value is appended to the row, we have to add a suffix
174       val rangeEnd = Array.ofDim[Byte](row.length + 3 + ByteRange.UnboundedUpperRange.length)
175       System.arraycopy(row, 0, rangeEnd, 3, row.length)
176       System.arraycopy(ByteRange.UnboundedUpperRange, 0, rangeEnd, row.length + 3, ByteRange.UnboundedUpperRange.length)
177       ByteArrays.writeShort(row.length.toShort, rangeEnd, 1)
178       rangeEnd(0) = RedisIndexAdapter.ExclusiveRangePrefix
179       BoundedByteRange(rangeStart, rangeEnd)
180   }

202   /**
203     * Writer for redis
204     *
205     * @param jedis connection
206     * @param indices indices to write to
207     * @param partition partition to write to
208     * @param aging age-off writer
209     * @param wrapper feature wrapper
210     */
211   class RedisIndexWriter(
212       jedis: Pool[_ <: CloseableJedisCommands],
213       indices: Seq[GeoMesaFeatureIndex[_, _]],
214       partition: Option[String],
215       aging: Option[AgeOffWriter],
216       wrapper: FeatureWrapper[RedisWritableFeature]
217     ) extends BaseIndexWriter[RedisWritableFeature](indices, wrapper) {
218 
219     private val batchSize = RedisSystemProperties.WriteBatchSize.toInt match {
220       case Some(s) if s > 0 => s - 1
221       case _ =>
222         throw new IllegalStateException(s"Value '${RedisSystemProperties.WriteBatchSize.get}' for " +
223             s"'${RedisSystemProperties.WriteBatchSize.property}' is not a positive int")
224     }
225 
226     private val tables = indices.toArray.map { index =>
227       // should always be writing to a single table here
228       index.getTableName(partition).getBytes(StandardCharsets.UTF_8)
229     }
230 
231     private val inserts = Array.fill[java.util.Map[Array[Byte], java.lang.Double]](tables.length)(new java.util.HashMap[Array[Byte], java.lang.Double]())
232     private val deletes = Array.fill[ArrayBuffer[Array[Byte]]](tables.length)(ArrayBuffer.empty[Array[Byte]])
233 
234     private var i = 0
235     private var batch = 0
236 
237     private val errors = ArrayBuffer.empty[Throwable]
238 
239     override protected def append(feature: RedisWritableFeature, values: Array[RowKeyValue[_]]): Unit = {
240       i = 0
241       while (i < values.length) {
242         val insert = inserts(i)
243         values(i) match {
244           case kv: SingleRowKeyValue[_] =>
245             kv.values.foreach(v => insert.put(ByteArrays.concat(kv.row, v.value), 0d))
246           case kv: MultiRowKeyValue[_] =>
247             kv.rows.foreach(row => kv.values.foreach(v => insert.put(ByteArrays.concat(row, v.value), 0d)))
248         }
249         i += 1
250       }
251 
252       aging.foreach(_.write(feature))
253 
254       if (batch < batchSize) {
255         batch += 1
256       } else {
257         flush()
258       }
259     }
260 
261     override protected def update(
262         feature: RedisWritableFeature,
263         values: Array[RowKeyValue[_]],
264         previous: RedisWritableFeature,
265         previousValues: Array[RowKeyValue[_]]): Unit = {
266       delete(previous, previousValues)
267       append(feature, values)
268     }
269 
270     override protected def delete(feature: RedisWritableFeature, values: Array[RowKeyValue[_]]): Unit = {
271       i = 0
272       while (i < values.length) {
273         val buffer = deletes(i)
274         values(i) match {
275           case kv: SingleRowKeyValue[_] =>
276             kv.values.foreach(v => buffer.append(ByteArrays.concat(kv.row, v.value)))
277           case kv: MultiRowKeyValue[_] =>
278             kv.rows.foreach(row => kv.values.foreach(v => buffer.append(ByteArrays.concat(row, v.value))))
279         }
280         i += 1
281       }
282 
283       aging.foreach(_.delete(feature))
284 
285       if (batch < batchSize) {
286         batch += 1
287       } else {
288         flush()
289       }
290     }
291 
292     override def flush(): Unit = {
293       i = 0
294       while (i < tables.length) {
295         if (deletes(i).nonEmpty) {
296           try { WithClose(jedis.getResource)(_.zrem(tables(i), deletes(i).toSeq: _*)) } catch {
297             case NonFatal(e) => errors.append(e)
298           }
299           deletes(i).clear()
300         }
301         if (!inserts(i).isEmpty) {
302           try { WithClose(jedis.getResource)(_.zadd(tables(i), inserts(i))) } catch {
303             case NonFatal(e) => errors.append(e)
304           }
305           inserts(i).clear()
306         }
307         i += 1
308       }
309       batch = 0
310 
311       try { aging.foreach(_.flush()) } catch {
312         case NonFatal(e) => errors.append(e)
313       }
314 
315       throwErrors()
316     }
317 
318     override def close(): Unit = {
319       try { flush() } catch {
320         case NonFatal(e) => errors.append(e)
321       }
322       try { aging.foreach(_.close()) } catch {
323         case NonFatal(e) => errors.append(e)
324       }
325       throwErrors()
326     }
327 
328     private def throwErrors(): Unit = {
329       if (errors.nonEmpty) {
330         val error = errors.head
331         errors.tail.foreach(error.addSuppressed)
332         errors.clear()
333         throw error
334       }
335     }
336   }
