105   val MinRange: Array[Byte] = "-".getBytes(StandardCharsets.UTF_8)

111   /**
112     * Convert a byte range to a redis zlex range
113     *
114     * @param byteRange geomesa range
115     * @return
116     */
117   private def toRedisRange(byteRange: ByteRange): BoundedByteRange = byteRange match {
118     case BoundedByteRange(start, end) =>
119       val rangeStart = if (start.isEmpty) { RedisIndexAdapter.MinRange } else {
120         val range = Array.ofDim[Byte](start.length + 1)
121         System.arraycopy(start, 0, range, 1, start.length)
122         range(0) = RedisIndexAdapter.InclusiveRangePrefix
123         range
124       }
125       val rangeEnd = if (end.isEmpty) { RedisIndexAdapter.MaxRange } else {
126         val range = Array.ofDim[Byte](end.length + 1)
127         System.arraycopy(end, 0, range, 1, end.length)
128         range(0) = RedisIndexAdapter.ExclusiveRangePrefix
129         range
130       }
131       BoundedByteRange(rangeStart, rangeEnd)
132 
133     case SingleRowByteRange(row) =>
134       val rangeStart = Array.ofDim[Byte](row.length + 1)
135       System.arraycopy(row, 0, rangeStart, 1, row.length)
136       rangeStart(0) = RedisIndexAdapter.InclusiveRangePrefix
137       // since the value is appended to the row, we have to add a suffix
138       val rangeEnd = Array.ofDim[Byte](row.length + 1 + ByteRange.UnboundedUpperRange.length)
139       System.arraycopy(row, 0, rangeEnd, 1, row.length)
140       System.arraycopy(ByteRange.UnboundedUpperRange, 0, rangeEnd, row.length + 1, ByteRange.UnboundedUpperRange.length)
141       rangeEnd(0) = RedisIndexAdapter.ExclusiveRangePrefix
142       BoundedByteRange(rangeStart, rangeEnd)
143   }
144 
145   /**
146     * Convert a byte range into a zlex range, specifically for the id index. Since we store the length
147     * of the id in the row key, we have to prepend that to our ranges
148     *
149     * @param byteRange geomesa range
150     * @return
151     */
152   private def toRedisIdRange(byteRange: ByteRange): BoundedByteRange = byteRange match {
153     case BoundedByteRange(start, end) =>
154       val rangeStart = if (start.isEmpty) { RedisIndexAdapter.MinRange } else {
155         // add the two byte length prefix
156         val range = Array.ofDim[Byte](start.length + 3)
157         System.arraycopy(start, 0, range, 3, start.length)
158         ByteArrays.writeShort(start.length.toShort, range, 1)
159         range(0) = RedisIndexAdapter.InclusiveRangePrefix
160         range
161       }
162       val rangeEnd = if (end.isEmpty) { RedisIndexAdapter.MaxRange } else {
163         // add the two byte length prefix
164         val range = Array.ofDim[Byte](end.length + 3)
165         System.arraycopy(end, 0, range, 3, end.length)
166         ByteArrays.writeShort(end.length.toShort, range, 1)
167         range(0) = RedisIndexAdapter.ExclusiveRangePrefix
168         range
169       }
170       BoundedByteRange(rangeStart, rangeEnd)
171 
172     case SingleRowByteRange(row) =>
173       // add the two byte length prefix
174       val rangeStart = Array.ofDim[Byte](row.length + 3)
175       System.arraycopy(row, 0, rangeStart, 3, row.length)
176       ByteArrays.writeShort(row.length.toShort, rangeStart, 1)
177       rangeStart(0) = RedisIndexAdapter.InclusiveRangePrefix
178       // since the value is appended to the row, we have to add a suffix
179       val rangeEnd = Array.ofDim[Byte](row.length + 3 + ByteRange.UnboundedUpperRange.length)
180       System.arraycopy(row, 0, rangeEnd, 3, row.length)
181       System.arraycopy(ByteRange.UnboundedUpperRange, 0, rangeEnd, row.length + 3, ByteRange.UnboundedUpperRange.length)
182       ByteArrays.writeShort(row.length.toShort, rangeEnd, 1)
183       rangeEnd(0) = RedisIndexAdapter.ExclusiveRangePrefix
184       BoundedByteRange(rangeStart, rangeEnd)
185   }

190     private var idSerializer: (Array[Byte], Int, Int, SimpleFeature) => String = _

192     override def apply(result: Array[Byte]): SimpleFeature = {
193       // parse out the feature id and the serialized value from the concatenated row + value
194       val idStart = index.getIdOffset(result, 0, result.length)
195       val idLength = ByteArrays.readShort(result, idStart)
196       val id = idSerializer(result, idStart + 2, idLength, null)
197       val valueStart = idStart + idLength + 2
198       serializer.deserialize(id, result, valueStart, result.length - valueStart)
199     }

207   /**
208     * Writer for redis
209     *
210     * @param jedis connection
211     * @param indices indices to write to
212     * @param partition partition to write to
213     * @param aging age-off writer
214     * @param wrapper feature wrapper
215     */
216   class RedisIndexWriter(
217       jedis: JedisPool,
218       indices: Seq[GeoMesaFeatureIndex[_, _]],
219       partition: Option[String],
220       aging: Option[AgeOffWriter],
221       wrapper: FeatureWrapper[RedisWritableFeature]
222     ) extends BaseIndexWriter[RedisWritableFeature](indices, wrapper) {
223 
224     private val batchSize = RedisSystemProperties.WriteBatchSize.toInt match {
225       case Some(s) if s > 0 => s - 1
226       case _ =>
227         throw new IllegalStateException(s"Value '${RedisSystemProperties.WriteBatchSize.get}' for " +
228             s"'${RedisSystemProperties.WriteBatchSize.property}' is not a positive int")
229     }
230 
231     private val tables = indices.toArray.map { index =>
232       // should always be writing to a single table here
233       index.getTableName(partition).getBytes(StandardCharsets.UTF_8)
234     }
235 
236     private val inserts = Array.fill[java.util.Map[Array[Byte], java.lang.Double]](tables.length)(new java.util.HashMap[Array[Byte], java.lang.Double]())
237     private val deletes = Array.fill[ArrayBuffer[Array[Byte]]](tables.length)(ArrayBuffer.empty[Array[Byte]])
238 
239     private var i = 0
240     private var batch = 0
241 
242     private val errors = ArrayBuffer.empty[Throwable]
243 
244     override protected def append(feature: RedisWritableFeature, values: Array[RowKeyValue[_]]): Unit = {
245       i = 0
246       while (i < values.length) {
247         val insert = inserts(i)
248         values(i) match {
249           case kv: SingleRowKeyValue[_] =>
250             kv.values.foreach(v => insert.put(ByteArrays.concat(kv.row, v.value), 0d))
251           case kv: MultiRowKeyValue[_] =>
252             kv.rows.foreach(row => kv.values.foreach(v => insert.put(ByteArrays.concat(row, v.value), 0d)))
253         }
254         i += 1
255       }
256 
257       aging.foreach(_.write(feature))
258 
259       if (batch < batchSize) {
260         batch += 1
261       } else {
262         flush()
263       }
264     }
265 
266     override protected def update(
267         feature: RedisWritableFeature,
268         values: Array[RowKeyValue[_]],
269         previous: RedisWritableFeature,
270         previousValues: Array[RowKeyValue[_]]): Unit = {
271       delete(previous, previousValues)
272       append(feature, values)
273     }
274 
275     override protected def delete(feature: RedisWritableFeature, values: Array[RowKeyValue[_]]): Unit = {
276       i = 0
277       while (i < values.length) {
278         val buffer = deletes(i)
279         values(i) match {
280           case kv: SingleRowKeyValue[_] =>
281             kv.values.foreach(v => buffer.append(ByteArrays.concat(kv.row, v.value)))
282           case kv: MultiRowKeyValue[_] =>
283             kv.rows.foreach(row => kv.values.foreach(v => buffer.append(ByteArrays.concat(row, v.value))))
284         }
285         i += 1
286       }
287 
288       aging.foreach(_.delete(feature))
289 
290       if (batch < batchSize) {
291         batch += 1
292       } else {
293         flush()
294       }
295     }
296 
297     override def flush(): Unit = {
298       i = 0
299       while (i < tables.length) {
300         if (deletes(i).nonEmpty) {
301           try { WithClose(jedis.getResource)(_.zrem(tables(i), deletes(i).toSeq: _*)) } catch {
302             case NonFatal(e) => errors.append(e)
303           }
304           deletes(i).clear()
305         }
306         if (!inserts(i).isEmpty) {
307           try { WithClose(jedis.getResource)(_.zadd(tables(i), inserts(i))) } catch {
308             case NonFatal(e) => errors.append(e)
309           }
310           inserts(i).clear()
311         }
312         i += 1
313       }
314       batch = 0
315 
316       try { aging.foreach(_.flush()) } catch {
317         case NonFatal(e) => errors.append(e)
318       }
319 
320       throwErrors()
321     }
322 
323     override def close(): Unit = {
324       try { flush() } catch {
325         case NonFatal(e) => errors.append(e)
326       }
327       try { aging.foreach(_.close()) } catch {
328         case NonFatal(e) => errors.append(e)
329       }
330       throwErrors()
331     }
332 
333     private def throwErrors(): Unit = {
334       if (errors.nonEmpty) {
335         val error = errors.head
336         errors.tail.foreach(error.addSuppressed)
337         errors.clear()
338         throw error
339       }
340     }
341   }
