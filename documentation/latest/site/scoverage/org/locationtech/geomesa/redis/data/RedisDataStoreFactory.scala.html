87   /**
88     * Builds a redis connection from the data store parameters
89     *
90     * @param params params
91     * @return
92     */
93   def buildConnection(params: java.util.Map[String, _]): JedisPool = {
94     ConnectionPoolParam.lookupOpt(params).getOrElse {
95       val uri = {
96         val url = RedisUrlParam.lookup(params)
97         // if there is no protocol/port, or the url is a valid redis url, use as is
98         // else use the redis:// protocol to support databases, etc
99         val parsed =
100           if (url.indexOf(":") == -1) {
101             Try(new URI(url))
102           } else {
103             parse(url).orElse(parse(s"redis://$url"))
104           }
105         parsed match {
106           case Success(uri) => uri
107           case Failure(e) =>
108             throw new IllegalArgumentException(s"Could not create valid Redis connection URI from: $url", e)
109         }
110       }
111 
112       val config = new GenericObjectPoolConfig[Jedis]()
113       PoolSizeParam.lookupOpt(params).foreach(s => config.setMaxTotal(s.intValue()))
114       config.setTestOnBorrow(TestConnectionParam.lookup(params))
115       val timeout = SocketTimeoutParam.lookup(params).toMillis.toInt
116 
117       new JedisPool(config, uri, timeout)
118     }
119   }
120 
121   /**
122     * Builds configuration from data store parameters
123     *
124     * @param params params
125     * @return
126     */
127   def buildConfig(params: java.util.Map[String, _]): RedisDataStoreConfig = {
128     val catalog = RedisCatalogParam.lookup(params)
129     val generateStats = GenerateStatsParam.lookup(params)
130     val pipeline = PipelineParam.lookup(params)
131 
132     val audit = if (!AuditQueriesParam.lookup(params)) { None } else {
133       Some(new AuditLogger("redis", AuditProvider.Loader.loadOrNone(params)))
134     }
135     // get the auth params passed in as a comma-delimited string
136     val authProvider = AuthUtils.getProvider(params,
137       AuthsParam.lookupOpt(params).map(_.split(",").toSeq.filterNot(_.isEmpty)).getOrElse(Seq.empty))
138 
139     val queries = RedisQueryConfig(
140       threads = QueryThreadsParam.lookup(params),
141       timeout = QueryTimeoutParam.lookupOpt(params).map(_.toMillis),
142       looseBBox = LooseBBoxParam.lookup(params).booleanValue(),
143       parallelPartitionScans = PartitionParallelScansParam.lookup(params)
144     )
145 
146     val ns = Option(NamespaceParam.lookUp(params).asInstanceOf[String])
147 
148     RedisDataStoreConfig(catalog, generateStats, audit, authProvider, queries, pipeline, ns)
149   }
