85   /**
86     * Builds a redis connection from the data store parameters
87     *
88     * @param params params
89     * @return
90     */
91   def buildConnection(params: java.util.Map[String, _]): JedisPool = {
92     ConnectionPoolParam.lookupOpt(params).getOrElse {
93       val uri = {
94         val url = RedisUrlParam.lookup(params)
95         // if there is no protocol/port, or the url is a valid redis url, use as is
96         // else use the redis:// protocol to support databases, etc
97         val parsed =
98           if (url.indexOf(":") == -1) {
99             Try(new URI(url))
100           } else {
101             parse(url).orElse(parse(s"redis://$url"))
102           }
103         parsed match {
104           case Success(uri) => uri
105           case Failure(e) =>
106             throw new IllegalArgumentException(s"Could not create valid Redis connection URI from: $url", e)
107         }
108       }
109 
110       val config = new GenericObjectPoolConfig[Jedis]()
111       PoolSizeParam.lookupOpt(params).foreach(s => config.setMaxTotal(s.intValue()))
112       config.setTestOnBorrow(TestConnectionParam.lookup(params))
113       val timeout = SocketTimeoutParam.lookup(params).toMillis.toInt
114 
115       new JedisPool(config, uri, timeout)
116     }
117   }
118 
119   /**
120     * Builds configuration from data store parameters
121     *
122     * @param params params
123     * @return
124     */
125   def buildConfig(params: java.util.Map[String, _]): RedisDataStoreConfig = {
126     val catalog = RedisCatalogParam.lookup(params)
127     val generateStats = GenerateStatsParam.lookup(params)
128     val pipeline = PipelineParam.lookup(params)
129 
130     val audit = if (!AuditQueriesParam.lookup(params)) { None } else {
131       Some((AuditLogger, Option(AuditProvider.Loader.load(params)).getOrElse(NoOpAuditProvider), "redis"))
132     }
133     // get the auth params passed in as a comma-delimited string
134     val authProvider = AuthUtils.getProvider(params,
135       AuthsParam.lookupOpt(params).map(_.split(",").toSeq.filterNot(_.isEmpty)).getOrElse(Seq.empty))
136 
137     val queries = RedisQueryConfig(
138       threads = QueryThreadsParam.lookup(params),
139       timeout = QueryTimeoutParam.lookupOpt(params).map(_.toMillis),
140       looseBBox = LooseBBoxParam.lookup(params).booleanValue(),
141       parallelPartitionScans = PartitionParallelScansParam.lookup(params)
142     )
143 
144     val ns = Option(NamespaceParam.lookUp(params).asInstanceOf[String])
145 
146     RedisDataStoreConfig(catalog, generateStats, audit, authProvider, queries, pipeline, ns)
147   }
