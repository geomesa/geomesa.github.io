91   /**
92     * Builds a redis connection from the data store parameters
93     *
94     * @param params params
95     * @return
96     */
97   def buildConnection(params: java.util.Map[String, _]): Pool[_ <: CloseableJedisCommands] = {
98     ConnectionPoolParam.lookupOpt(params).getOrElse {
99       RedisClusterParam.lookup(params).booleanValue() match {
100         case false =>
101           val uri = {
102             val urls = RedisUrlParam.lookup(params).split(",")
103             if (urls.length > 1) {
104               throw new IllegalArgumentException(
105                 "Multiple Redis URLs can only be used in cluster mode. Set 'redis.clusterMode' to true.")
106             }
107             val url = urls.headOption.getOrElse {
108               throw new IllegalArgumentException("No Redis URL provided. Please set 'redis.url'")
109             }
110             // if there is no protocol/port, or the url is a valid redis url, use as is
111             // else use the redis:// protocol to support databases, etc
112             val parsed =
113               if (url.indexOf(":") == -1) {
114                 Try(new URI(url))
115               } else {
116                 parse(url).orElse(parse(s"redis://$url"))
117               }
118             parsed match {
119               case Success(uri) => uri
120               case Failure(e) =>
121                 throw new IllegalArgumentException(s"Could not create valid Redis connection URI from: $url", e)
122             }
123           }
124           val config = new GenericObjectPoolConfig[Jedis]()
125           PoolSizeParam.lookupOpt(params).foreach(s => config.setMaxTotal(s.intValue()))
126           config.setTestOnBorrow(TestConnectionParam.lookup(params))
127           val timeout = SocketTimeoutParam.lookup(params).toMillis.toInt
128 
129           new JedisPool(config, uri, timeout)
130         case true =>
131           val urls = RedisUrlParam.lookup(params).split(",")
132           if (urls.isEmpty) {
133             throw new IllegalArgumentException("No Redis URLs provided. Please set 'redis.url'")
134           }
135           val uri = URI.create(urls.head)
136           val objectPoolConfig = new GenericObjectPoolConfig[Connection]()
137           PoolSizeParam.lookupOpt(params).foreach(s => objectPoolConfig.setMaxTotal(s.intValue()))
138           objectPoolConfig.setTestOnBorrow(TestConnectionParam.lookup(params))
139           val timeout = SocketTimeoutParam.lookup(params).toMillis.toInt
140 
141           val clusterNodes: java.util.Set[HostAndPort] =
142             urls.map { url =>
143               val parsed = parse(url).getOrElse {
144                 throw new IllegalArgumentException(s"Could not create valid Redis connection URI from: $url")
145               }
146               JedisURIHelper.getHostAndPort(parsed)
147             }.toSet.asJava
148           val jedisClientConfig: JedisClientConfig =
149             DefaultJedisClientConfig.builder.connectionTimeoutMillis(timeout)
150               .socketTimeoutMillis(timeout)
151               .blockingSocketTimeoutMillis(0)
152               .user(JedisURIHelper.getUser(uri))
153               .password(JedisURIHelper.getPassword(uri))
154               .database(JedisURIHelper.getDBIndex(uri))
155               .clientName(null)
156               .build
157           new SingletonJedisClusterPool(clusterNodes, jedisClientConfig, objectPoolConfig)
158       }
159     }
160   }

162   /**
163     * Builds configuration from data store parameters
164     *
165     * @param params params
166     * @return
167     */
168   def buildConfig(params: java.util.Map[String, _]): RedisDataStoreConfig = {
169     val catalog = RedisCatalogParam.lookup(params)
170     val generateStats = GenerateStatsParam.lookup(params)
171     val pipeline = PipelineParam.lookup(params)
172 
173     val audit = if (!AuditQueriesParam.lookup(params)) { None } else {
174       Some(new AuditLogger("redis", AuditProvider.Loader.loadOrNone(params)))
175     }
176     val metrics = MetricsRegistryParam.lookupRegistry(params)
177     // get the auth params passed in as a comma-delimited string
178     val authProvider = AuthUtils.getProvider(params,
179       AuthsParam.lookupOpt(params).map(_.split(",").toSeq.filterNot(_.isEmpty)).getOrElse(Seq.empty))
180 
181     val queries = RedisQueryConfig(
182       threads = QueryThreadsParam.lookup(params),
183       timeout = QueryTimeoutParam.lookupOpt(params).map(_.toMillis),
184       looseBBox = LooseBBoxParam.lookup(params).booleanValue(),
185       parallelPartitionScans = PartitionParallelScansParam.lookup(params)
186     )
187 
188     val ns = Option(NamespaceParam.lookUp(params).asInstanceOf[String])
189 
190     RedisDataStoreConfig(catalog, generateStats, authProvider, audit, metrics, queries, pipeline, ns)
191   }
