20 /**
21   * Writable feature values cache for Redis
22   *
23   * @param feature feature being written
24   * @param serializer serializer
25   * @param idSerializer id serializer
26   */
27 class RedisWritableFeature(
28     val feature: SimpleFeature,
29     serializer: SimpleFeatureSerializer,
30     idSerializer: String => Array[Byte]
31   ) extends WritableFeature {
32 
33   import RedisWritableFeature.EmptyBytes
34 
35   // id value without the 2 byte prefix indicating the id length
36   lazy val rawId: Array[Byte] = id.drop(2)
37 
38   // calculate the age-off as a expiration timestamp, which we'll store as the sorted set score
39   // redis/java doubles should be able to exactly represent every Long value up to 9007199254740992L,
40   // which corresponds to the year 287396
41   lazy val ttl: Double = feature.getUserData.get(RedisAgeOff.TtlUserDataKey).asInstanceOf[java.lang.Long].toDouble
42 
43   // we don't use column families, column qualifiers or visibilities in the the rows
44   override lazy val values: Seq[KeyValue] =
45     Seq(KeyValue(EmptyBytes, EmptyBytes, EmptyBytes, serializer.serialize(feature)))
46 
47   override lazy val id: Array[Byte] = idSerializer(feature.getID)
48 }

50 object RedisWritableFeature {
51 
52   import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
53 
54   private val EmptyBytes = Array.empty[Byte]
55 
56   /**
57     * Create a writable feature factory
58     *
59     * @param sft simple feature type
60     * @return
61     */
62   def wrapper(sft: SimpleFeatureType): FeatureWrapper[RedisWritableFeature] = {
63     val id: String => Array[Byte] = GeoMesaFeatureIndex.idToBytes(sft)
64     // add the length of the feature id into the byte array so that we can decode it after the value is concatenated
65     val idSerializer: String => Array[Byte] = fid => {
66       val bytes = id(fid)
67       val result = Array.ofDim[Byte](bytes.length + 2)
68       System.arraycopy(bytes, 0, result, 2, bytes.length)
69       // note: if the feature id is longer than 32k characters this will cause problems...
70       ByteArrays.writeShort(bytes.length.toShort, result)
71       result
72     }
73     // we serialize with user data to store visibilities
74     val serializer = KryoFeatureSerializer.builder(sft).withUserData.withoutId.build()
75     sft.getFeatureExpiration match {
76       case None => new RedisFeatureWrapper(serializer, idSerializer)
77       case Some(aging) => new RedisExpiringFeatureWrapper(serializer, idSerializer, aging)
78     }
79   }
80 
81   class RedisFeatureWrapper(serializer: SimpleFeatureSerializer, idSerializer: String => Array[Byte])
82       extends FeatureWrapper[RedisWritableFeature] {
83     override def wrap(feature: SimpleFeature, delete: Boolean): RedisWritableFeature = {
84       // remove all user data except visibilities
85       // we need to keep visibilities for filtering, but don't want to store anything else,
86       // as generally we don't store user data
87       // note: at this point provided_fid has already been handled
88       val visibility = feature.getUserData.get(SecurityUtils.FEATURE_VISIBILITY)
89       feature.getUserData.clear()
90       if (visibility != null) {
91         feature.getUserData.put(SecurityUtils.FEATURE_VISIBILITY, visibility)
92       }
93       new RedisWritableFeature(feature, serializer, idSerializer)
94     }
95   }
96 
97   class RedisExpiringFeatureWrapper(
98       serializer: SimpleFeatureSerializer,
99       idSerializer: String => Array[Byte],
100       aging: FeatureExpiration
101   ) extends RedisFeatureWrapper(serializer, idSerializer) {
102     override def wrap(feature: SimpleFeature, delete: Boolean): RedisWritableFeature = {
103       val ttl = if (delete) {
104         // use the existing ttl so our serialized key matches
105         feature.getUserData.get(RedisAgeOff.TtlUserDataKey)
106       } else {
107         Long.box(aging.expires(feature))
108       }
109       val wrapped = super.wrap(feature, delete)
110       // store ttl in the user data
111       feature.getUserData.put(RedisAgeOff.TtlUserDataKey, ttl)
112       wrapped
113     }
114   }
115 }
