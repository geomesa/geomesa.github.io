60   def explain(plan: RedisQueryPlan, explainer: Explainer, prefix: String): Unit = {
61     explainer.pushLevel(s"${prefix}Plan: ${plan.getClass.getSimpleName}")
62     explainer(s"Tables: ${plan.tables.mkString(", ")}")
63     explainer(s"ECQL: ${plan.ecql.map(filterToString).getOrElse("none")}")
64     explainer(s"Ranges (${plan.ranges.size}): ${plan.ranges.take(5).map(rangeToString).mkString(", ")}")
65     plan.explain(explainer)
66     explainer(s"Reduce: ${plan.reducer.getOrElse("none")}")
67     explainer.popLevel()
68   }

70   private [data] def rangeToString(range: BoundedByteRange): String = {
71     // based on accumulo's byte representation
72     def printable(b: Byte): String = {
73       val c = 0xff & b
74       if (c >= 32 && c <= 126) { c.toChar.toString } else { f"%%$c%02x;" }
75     }
76     s"[${range.lower.map(printable).mkString("")}::${range.upper.map(printable).mkString("")}]"
77   }

107     override def scan(ds: RedisDataStore): CloseableIterator[Array[Byte]] = {
108       val iter = tables.iterator.map(_.getBytes(StandardCharsets.UTF_8))
109       val scans = iter.map(singleTableScan(ds, _))
110       if (ds.config.queries.parallelPartitionScans) {
111         // kick off all the scans at once
112         scans.foldLeft(CloseableIterator.empty[Array[Byte]])(_ concat _)
113       } else {
114         // kick off the scans sequentially as they finish
115         SelfClosingIterator(scans).flatMap(s => s)
116       }
117     }
118 
119     override protected def explain(explainer: Explainer): Unit =
120       explainer(s"Pipelining: ${if (pipeline) { "enabled" } else { "disabled" }}")
121 
122     private def singleTableScan(ds: RedisDataStore, table: Array[Byte]): CloseableIterator[Array[Byte]] = {
123       if (pipeline) {
124         val result = Seq.newBuilder[Response[java.util.List[Array[Byte]]]]
125         result.sizeHint(ranges.length)
126         WithClose(ds.connection.getResource) { jedis =>
127           WithClose(jedis.pipelined()) { pipe =>
128             // note: use a foreach here to ensure the calls are all executing inside our close block
129             ranges.foreach(range => result += pipe.zrangeByLex(table, range.lower, range.upper))
130             pipe.sync()
131           }
132         }
133         result.result.iterator.flatMap(_.get.iterator().asScala)
134       } else {
135         RedisBatchScan(ds.connection, table, ranges, ds.config.queries.threads)
136       }
137     }
138   }
