69   private [data] def rangeToString(range: BoundedByteRange): String = {
70     // based on accumulo's byte representation
71     def printable(b: Byte): String = {
72       val c = 0xff & b
73       if (c >= 32 && c <= 126) { c.toChar.toString } else { f"%%$c%02x;" }
74     }
75     s"[${range.lower.map(printable).mkString("")}::${range.upper.map(printable).mkString("")}]"
76   }

106     override def scan(ds: RedisDataStore): CloseableIterator[Array[Byte]] = {
107       // query guard hook - also handles full table scan checks
108       strategy.runGuards(ds)
109       val iter = tables.iterator.map(_.getBytes(StandardCharsets.UTF_8))
110       val scans = iter.map(singleTableScan(ds, _))
111       if (ds.config.queries.parallelPartitionScans) {
112         // kick off all the scans at once
113         scans.foldLeft(CloseableIterator.empty[Array[Byte]])(_ concat _)
114       } else {
115         // kick off the scans sequentially as they finish
116         SelfClosingIterator(scans).flatMap(s => s)
117       }
118     }
119 
120     override protected def explain(explainer: Explainer): Unit =
121       explainer(s"Pipelining: ${if (pipeline) { "enabled" } else { "disabled" }}")
122 
123     private def singleTableScan(ds: RedisDataStore, table: Array[Byte]): CloseableIterator[Array[Byte]] = {
124       if (pipeline) {
125         val result = Seq.newBuilder[Response[java.util.List[Array[Byte]]]]
126         result.sizeHint(ranges.length)
127         WithClose(ds.connection.getResource) {
128           case jedis: Jedis =>
129             WithClose(jedis.pipelined()) { pipe =>
130               // note: use a foreach here to ensure the calls are all executing inside our close block
131               ranges.foreach(range => result += pipe.zrangeByLex(table, range.lower, range.upper))
132               pipe.sync()
133             }
134           case jedis: UnifiedJedis =>
135             WithClose(jedis.pipelined()) { pipe =>
136               // note: use a foreach here to ensure the calls are all executing inside our close block
137               ranges.foreach(range => result += pipe.zrangeByLex(table, range.lower, range.upper))
138               pipe.sync()
139             }
140         }
141         result.result.iterator.flatMap(_.get.iterator().asScala)
142       } else {
143         RedisBatchScan(ds.connection, table, ranges, ds.config.queries.threads)
144       }
145     }
146   }
