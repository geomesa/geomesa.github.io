62   private [data] def rangeToString(range: BoundedByteRange): String = {
63     // based on accumulo's byte representation
64     def printable(b: Byte): String = {
65       val c = 0xff & b
66       if (c >= 32 && c <= 126) { c.toChar.toString } else { f"%%$c%02x;" }
67     }
68     s"[${range.lower.map(printable).mkString("")}::${range.upper.map(printable).mkString("")}]"
69   }

94     import scala.collection.JavaConverters._

96     override def scan(): CloseableIterator[SimpleFeature] = {
97       // query guard hook - also handles full table scan checks
98       ds.interceptors.runGuards(strategy)
99       val toFeatures = new RedisResultsToFeatures(strategy.index, strategy.index.sft)
100       val iter = tables.iterator.map(_.getBytes(StandardCharsets.UTF_8))
101       val scans = iter.map(singleTableScan(ds, _))
102       val scanner =
103         if (ds.config.queries.parallelPartitionScans) {
104           // kick off all the scans at once
105           scans.foldLeft(CloseableIterator.empty[Array[Byte]])(_ concat _).map(toFeatures.apply)
106         } else {
107           // kick off the scans sequentially as they finish
108           CloseableIterator(scans).flatMap(s => s.map(toFeatures.apply))
109         }
110       val features = localFilter.fold(scanner)(f => scanner.filter(f.evaluate))
111       processor(features)
112     }
113 
114     override def moreExplaining(explainer: Explainer): Unit = {
115       explainer(s"Pipelining: ${if (pipeline) { "enabled" } else { "disabled" }}")
116       explainer(s"Client-side filter: ${localFilter.fold("none")(FilterHelper.toString)}")
117       processor.explain(explainer)
118     }
119 
120     private def singleTableScan(ds: RedisDataStore, table: Array[Byte]): CloseableIterator[Array[Byte]] = {
121       if (pipeline) {
122         val result = Seq.newBuilder[Response[java.util.List[Array[Byte]]]]
123         result.sizeHint(ranges.length)
124         WithClose(ds.connection.getResource) {
125           case jedis: Jedis =>
126             WithClose(jedis.pipelined()) { pipe =>
127               // note: use a foreach here to ensure the calls are all executing inside our close block
128               ranges.foreach(range => result += pipe.zrangeByLex(table, range.lower, range.upper))
129               pipe.sync()
130             }
131           case jedis: UnifiedJedis =>
132             WithClose(jedis.pipelined()) { pipe =>
133               // note: use a foreach here to ensure the calls are all executing inside our close block
134               ranges.foreach(range => result += pipe.zrangeByLex(table, range.lower, range.upper))
135               pipe.sync()
136             }
137         }
138         CloseableIterator(result.result.iterator.flatMap(_.get.iterator().asScala))
139       } else {
140         RedisBatchScan(ds.connection, table, ranges, ds.config.queries.threads)
141       }
142     }
143   }
