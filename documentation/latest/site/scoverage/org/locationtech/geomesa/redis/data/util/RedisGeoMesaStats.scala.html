29 class RedisGeoMesaStats(ds: RedisDataStore, metadata: RedisBackedMetadata[Stat])
30     extends MetadataBackedStats(ds, metadata) {
31 
32   @volatile
33   private var run = true
34   private var errors = 0
35 
36   // note: the default values ensure that these will always be Somes
37   private val retries = RedisSystemProperties.TransactionRetries.toInt.get
38   private val pause = RedisSystemProperties.TransactionPause.toDuration.get.toMillis
39   private val rand = new SecureRandom()
40 
41   private val queue = new LinkedBlockingQueue[RedisStat]()
42   private val es = java.util.concurrent.Executors.newScheduledThreadPool(2)
43 
44   es.submit(new Runnable() { override def run(): Unit = writeQueued() })
45 
46   override protected def write(typeName: String, stats: Seq[WritableStat]): Unit = {
47     val queued = stats.forall { stat =>
48       val keyBytes = metadata.encodeRow(typeName, stat.key)
49       val serialized = metadata.serializer.serialize(typeName, stat.stat)
50       queue.offer(RedisStat(typeName, stat.key, keyBytes, stat.stat, serialized, stat.merge, 0))
51     }
52     if (!queued) {
53       logger.error(s"Could not queue stat for writing - queue size: ${queue.size()}")
54     }
55   }
56 
57   private def writeQueued(): Unit = {
58     while (run) {
59       try {
60         val stat = queue.poll(1000L, TimeUnit.MILLISECONDS)
61         if (stat != null) {
62           val success = WithClose(ds.connection.getResource) { jedis =>
63             // Use Lua script to atomically check, merge if needed, and write the stat
64             val script = """
65               local existing = redis.call('hget', KEYS[1], ARGV[1])
66               local write
67               if ARGV[3] == '1' then
68                 if existing then
69                   return { existing, 0 } -- need to merge on client side
70                 else
71                   write = ARGV[2]
72                 end
73               else
74                 write = ARGV[2]
75               end
76               if write then
77                 redis.call('hset', KEYS[1], ARGV[1], write)
78               end
79               return { existing or false, write and 1 or 0 }
80             """
81             val merge = if (stat.merge) "1" else "0"
82             val result = jedis.eval(
83               script.getBytes(StandardCharsets.UTF_8),
84               1,
85               metadata.key,
86               stat.keyBytes, stat.statBytes, merge.getBytes(StandardCharsets.UTF_8)
87             )
88 
89             result match {
90               case list: java.util.List[_] if !list.isEmpty =>
91                 val existing = Option(list.get(0)).map {
92                   case bytes: Array[Byte] => bytes
93                   case _ => null
94                 }
95                 val success = list.get(1) match {
96                   case i: java.lang.Long => i == 1L
97                   case i: java.lang.Integer => i == 1
98                   case s: String => s.toInt == 1
99                   case _ => false
100                 }
101 
102                 if (success) {
103                   true // write was successful
104                 } else if (existing.isDefined && stat.merge) {
105                   // need to merge and retry
106                   val merged = metadata.serializer.serialize(stat.typeName,
107                     metadata.serializer.deserialize(stat.typeName, existing.get) + stat.stat)
108                   val retryScript = """
109                     redis.call('hset', KEYS[1], ARGV[1], ARGV[2])
110                     return 1
111                   """
112                   val retryResult = jedis.eval(
113                     retryScript.getBytes(StandardCharsets.UTF_8),
114                     1,
115                     metadata.key,
116                     stat.keyBytes, merged
117                   )
118                   retryResult match {
119                     case i: java.lang.Long => i == 1L
120                     case i: java.lang.Integer => i == 1
121                     case s: String => s.toInt == 1
122                     case _ => false
123                   }
124                 } else {
125                   false
126                 }
127 
128               case _ => false
129             }
130           }
131 
132           if (success) {
133             // since we didn't write through the metadata instance, we have to invalidate the cache
134             metadata.invalidateCache(stat.typeName, stat.key)
135             errors = 0 // reset error count on success
136           } else if (stat.attempt < retries) {
137             // we had a conflict, so backoff and retry
138             val delay = TransactionBackoffs.applyOrElse(stat.attempt, (_: Int) => TransactionBackoffs.last) * pause
139             val runnable = new BackoffScheduler(stat.copy(attempt = stat.attempt + 1))
140             es.schedule(runnable, delay + rand.nextInt(10), TimeUnit.MILLISECONDS)
141           } else {
142             throw new RuntimeException(
143               s"Could not write stat for ${stat.typeName}:${stat.key} after ${stat.attempt + 1} attempts")
144           }
145         }
146       } catch {
147         case NonFatal(e) =>
148           logger.error("Error in stat writing thread:", e)
149           if (errors > 20) {
150             logger.error("Terminating stat writing thread due to previous errors")
151             run = false
152             throw e
153           } else {
154             errors += 1
155           }
156       }
157     }
158   }
159 
160   override def close(): Unit = {
161     run = false
162     es.shutdown()
163     super.close()
164   }
165 
166   private class BackoffScheduler(stat: RedisStat) extends Runnable {
167     override def run(): Unit = {
168       if (!queue.offer(stat)) {
169         logger.error(s"Could not queue stat for writing - queue size: ${queue.size()}")
170       }
171     }
172   }
173 }
