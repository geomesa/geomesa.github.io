27 class RedisGeoMesaStats(ds: RedisDataStore, metadata: RedisBackedMetadata[Stat])
28     extends MetadataBackedStats(ds, metadata) {

30   @volatile
31   private var run = true
32   private var errors = 0
33 
34   // note: the default values ensure that these will always be Somes
35   private val retries = RedisSystemProperties.TransactionRetries.toInt.get
36   private val pause = RedisSystemProperties.TransactionPause.toDuration.get.toMillis
37   private val rand = new SecureRandom()
38 
39   private val queue = new LinkedBlockingQueue[RedisStat]()
40   private val es = java.util.concurrent.Executors.newScheduledThreadPool(2)
41 
42   es.submit(new Runnable() { override def run(): Unit = writeQueued() })
43 
44   override protected def write(typeName: String, stats: Seq[WritableStat]): Unit = {
45     val queued = stats.forall { stat =>
46       val keyBytes = metadata.encodeRow(typeName, stat.key)
47       val serialized = metadata.serializer.serialize(typeName, stat.stat)
48       queue.offer(RedisStat(typeName, stat.key, keyBytes, stat.stat, serialized, stat.merge, 0))
49     }
50     if (!queued) {
51       logger.error(s"Could not queue stat for writing - queue size: ${queue.size()}")
52     }
53   }
54 
55   private def writeQueued(): Unit = {
56     while (run) {
57       try {
58         val stat = queue.poll(1000L, TimeUnit.MILLISECONDS)
59         if (stat != null) {
60           val success = WithClose(ds.connection.getResource) { jedis =>
61             // Use Lua script to atomically check, merge if needed, and write the stat
62             val script = """
63               local existing = redis.call('hget', KEYS[1], ARGV[1])
64               local write
65               if ARGV[3] == '1' then
66                 if existing then
67                   return { existing, 0 } -- need to merge on client side
68                 else
69                   write = ARGV[2]
70                 end
71               else
72                 write = ARGV[2]
73               end
74               if write then
75                 redis.call('hset', KEYS[1], ARGV[1], write)
76               end
77               return { existing or false, write and 1 or 0 }
78             """
79             val merge = if (stat.merge) "1" else "0"
80             val result = jedis.eval(
81               script.getBytes(StandardCharsets.UTF_8),
82               1,
83               metadata.key,
84               stat.keyBytes, stat.statBytes, merge.getBytes(StandardCharsets.UTF_8)
85             )
86 
87             result match {
88               case list: java.util.List[_] if !list.isEmpty =>
89                 val existing = Option(list.get(0)).map {
90                   case bytes: Array[Byte] => bytes
91                   case _ => null
92                 }
93                 val success = list.get(1) match {
94                   case i: java.lang.Long => i.longValue() == 1L
95                   case i: java.lang.Integer => i.intValue() == 1
96                   case s: String => s.toInt == 1
97                   case _ => false
98                 }
99 
100                 if (success) {
101                   true // write was successful
102                 } else if (existing.isDefined && stat.merge) {
103                   // need to merge and retry
104                   val merged = metadata.serializer.serialize(stat.typeName,
105                     metadata.serializer.deserialize(stat.typeName, existing.get) + stat.stat)
106                   val retryScript = """
107                     redis.call('hset', KEYS[1], ARGV[1], ARGV[2])
108                     return 1
109                   """
110                   val retryResult = jedis.eval(
111                     retryScript.getBytes(StandardCharsets.UTF_8),
112                     1,
113                     metadata.key,
114                     stat.keyBytes, merged
115                   )
116                   retryResult match {
117                     case i: java.lang.Long => i == 1L
118                     case i: java.lang.Integer => i == 1
119                     case s: String => s.toInt == 1
120                     case _ => false
121                   }
122                 } else {
123                   false
124                 }
125 
126               case _ => false
127             }
128           }
129 
130           if (success) {
131             // since we didn't write through the metadata instance, we have to invalidate the cache
132             metadata.invalidateCache(stat.typeName, stat.key)
133             errors = 0 // reset error count on success
134           } else if (stat.attempt < retries) {
135             // we had a conflict, so backoff and retry
136             val delay = TransactionBackoffs.applyOrElse(stat.attempt, (_: Int) => TransactionBackoffs.last) * pause
137             val runnable = new BackoffScheduler(stat.copy(attempt = stat.attempt + 1))
138             es.schedule(runnable, delay + rand.nextInt(10), TimeUnit.MILLISECONDS)
139           } else {
140             throw new RuntimeException(
141               s"Could not write stat for ${stat.typeName}:${stat.key} after ${stat.attempt + 1} attempts")
142           }
143         }
144       } catch {
145         case NonFatal(e) =>
146           logger.error("Error in stat writing thread:", e)
147           if (errors > 20) {
148             logger.error("Terminating stat writing thread due to previous errors")
149             run = false
150             throw e
151           } else {
152             errors += 1
153           }
154       }
155     }
156   }
157 
158   override def close(): Unit = {
159     run = false
160     es.shutdown()
161     super.close()
162   }
163 
164   private class BackoffScheduler(stat: RedisStat) extends Runnable {
165     override def run(): Unit = {
166       if (!queue.offer(stat)) {
167         logger.error(s"Could not queue stat for writing - queue size: ${queue.size()}")
168       }
169     }
170   }
171 }
