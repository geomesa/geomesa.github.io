27 class RedisGeoMesaStats(ds: RedisDataStore, metadata: RedisBackedMetadata[Stat])
28     extends MetadataBackedStats(ds, metadata) {
29 
30   @volatile
31   private var run = true
32   private var errors = 0
33 
34   // note: the default values ensure that these will always be Somes
35   private val retries = RedisSystemProperties.TransactionRetries.toInt.get
36   private val pause = RedisSystemProperties.TransactionPause.toDuration.get.toMillis
37   private val rand = new SecureRandom()
38 
39   private val queue = new LinkedBlockingQueue[RedisStat]()
40   private val es = java.util.concurrent.Executors.newScheduledThreadPool(2)
41 
42   es.submit(new Runnable() { override def run(): Unit = writeQueued() })
43 
44   override protected def write(typeName: String, stats: Seq[WritableStat]): Unit = {
45     val queued = stats.forall { stat =>
46       val keyBytes = metadata.encodeRow(typeName, stat.key)
47       val serialized = metadata.serializer.serialize(typeName, stat.stat)
48       queue.offer(RedisStat(typeName, stat.key, keyBytes, stat.stat, serialized, stat.merge, 0))
49     }
50     if (!queued) {
51       logger.error(s"Could not queue stat for writing - queue size: ${queue.size()}")
52     }
53   }
54 
55   private def writeQueued(): Unit = {
56     while (run) {
57       try {
58         val stat = queue.poll(1000L, TimeUnit.MILLISECONDS)
59         if (stat != null) {
60           // use a redis watch to ensure that we aren't overwriting some other change
61           val success = WithClose(ds.connection.getResource) { jedis =>
62             jedis.watch(metadata.key)
63             val write = if (stat.merge) {
64               val existing = jedis.hget(metadata.key, stat.keyBytes)
65               if (existing == null) { stat.statBytes } else {
66                 metadata.serializer.serialize(stat.typeName,
67                   metadata.serializer.deserialize(stat.typeName, existing) + stat.stat)
68               }
69             } else {
70               stat.statBytes
71             }
72             // with a watch, we have to use a transaction to ensure the value hasn't changed
73             val tx = jedis.multi()
74             tx.hset(metadata.key, stat.keyBytes, write)
75             tx.exec() != null // null means invalid
76           }
77           if (success) {
78             // since we didn't write through the metadata instance, we have to invalidate the cache
79             metadata.invalidateCache(stat.typeName, stat.key)
80             errors = 0 // reset error count on success
81           } else if (stat.attempt < retries) {
82             // we had a conflict, so backoff and retry
83             val delay = TransactionBackoffs.applyOrElse(stat.attempt, (_: Int) => TransactionBackoffs.last) * pause
84             val runnable = new BackoffScheduler(stat.copy(attempt = stat.attempt + 1))
85             es.schedule(runnable, delay + rand.nextInt(10), TimeUnit.MILLISECONDS)
86           } else {
87             throw new RuntimeException(
88               s"Could not write stat for ${stat.typeName}:${stat.key} after ${stat.attempt + 1} attempts")
89           }
90         }
91       } catch {
92         case NonFatal(e) =>
93           logger.error("Error in stat writing thread:", e)
94           if (errors > 20) {
95             logger.error("Terminating stat writing thread due to previous errors")
96             run = false
97             throw e
98           } else {
99             errors += 1
100           }
101       }
102     }
103   }
104 
105   override def close(): Unit = {
106     run = false
107     es.shutdown()
108     super.close()
109   }
110 
111   private class BackoffScheduler(stat: RedisStat) extends Runnable {
112     override def run(): Unit = {
113       if (!queue.offer(stat)) {
114         logger.error(s"Could not queue stat for writing - queue size: ${queue.size()}")
115       }
116     }
117   }
118 }
