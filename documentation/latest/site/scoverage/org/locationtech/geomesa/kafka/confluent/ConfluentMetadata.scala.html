24 class ConfluentMetadata(schemaRegistry: SchemaRegistryClient, sftOverrides: Map[String, SimpleFeatureType] = Map.empty)
25   extends GeoMesaMetadata[String] with LazyLogging {
26 
27   private val topicSftCache: LoadingCache[String, String] = {
28     Caffeine.newBuilder().expireAfterWrite(10, TimeUnit.MINUTES).build(
29       new CacheLoader[String, String] {
30         override def load(topic: String): String = {
31           try {
32             // use the overridden sft for the topic if it exists, else look it up in the registry
33             val sft = sftOverrides.getOrElse(topic, {
34               val subject = topic + SubjectPostfix
35               val schemaId = schemaRegistry.getLatestSchemaMetadata(subject).getId
36               val sft = SchemaParser.schemaToSft(schemaRegistry.getById(schemaId))
37               // store the schema id to access the schema when creating the feature serializer
38               sft.getUserData.put(SchemaIdKey, schemaId.toString)
39               sft
40             })
41             KafkaDataStore.setTopic(sft, topic)
42             SimpleFeatureTypes.encodeType(sft, includeUserData = true)
43           } catch {
44             case NonFatal(e) => logger.error("Error retrieving schema from confluent registry: ", e); null
45           }
46         }
47       }
48     )
49   }
50 
51   override def getFeatureTypes: Array[String] = {
52     schemaRegistry.getAllSubjects.asScala.collect {
53       case s: String if s.endsWith(SubjectPostfix) => s.substring(0, s.lastIndexOf(SubjectPostfix))
54     }.toArray
55   }
56 
57   override def read(typeName: String, key: String, cache: Boolean): Option[String] = {
58     if (key == GeoMesaMetadata.AttributesKey) {
59       if (!cache) {
60         topicSftCache.invalidate(typeName)
61       }
62       Option(topicSftCache.get(typeName))
63     } else if (typeName == "migration" && key == "check") {
64       // skip metadata migration check since there's nothing to migrate
65       Some("true")
66     } else {
67       logger.warn(
68         s"Requested read on ConfluentMetadata with unsupported key $key. " +
69             s"ConfluentMetadata only supports ${GeoMesaMetadata.AttributesKey}")
70       None
71     }
72   }
73 
74   override def invalidateCache(typeName: String, key: String): Unit = {
75     if (key != GeoMesaMetadata.AttributesKey) {
76       logger.warn(s"Requested invalidate cache on ConfluentMetadata with unsupported key $key. " +
77         s"ConfluentMetadata only supports ${GeoMesaMetadata.AttributesKey}")
78     } else {
79       topicSftCache.invalidate(typeName)
80     }
81   }
82 
83   override def close(): Unit = schemaRegistry.close()
84 
85   override def scan(typeName: String, prefix: String, cache: Boolean): Seq[(String, String)] =
86     throw new NotImplementedError(s"ConfluentMetadata only supports ${GeoMesaMetadata.AttributesKey}")
87 
88   override def insert(typeName: String, key: String, value: String): Unit = {}
89   override def insert(typeName: String, kvPairs: Map[String, String]): Unit = {}
90   override def remove(typeName: String, key: String): Unit = {}
91   override def remove(typeName: String, keys: Seq[String]): Unit = {}
92   override def delete(typeName: String): Unit = {}
93   override def backup(typeName: String): Unit = {}
94   override def resetCache(): Unit = {}
95 }
