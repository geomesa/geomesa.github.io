23 /**
24  * Wrapper for a kafka streams builder that will configure serialization based on a GeoMesa Kafka feature store
25  *
26  * @param wrapped streams builder
27  * @param serde serialization for geomesa messages
28  * @param timestampExtractor timestamp extractor
29  * @param resetPolicy reset policy
30  */
31 class GeoMesaStreamsBuilder(
32     val wrapped: StreamsBuilder,
33     serde: GeoMesaSerde,
34     timestampExtractor: TimestampExtractor,
35     resetPolicy: Option[AutoOffsetReset]) {
36 
37   import org.apache.kafka.streams.scala.Serdes.String
38 
39   private implicit val s: GeoMesaSerde = serde
40 
41   implicit val consumed: Consumed[String, GeoMesaMessage] = resetPolicy match {
42     case None => Consumed.`with`(timestampExtractor)
43     case Some(p) => Consumed.`with`(timestampExtractor, p)
44   }
45 
46   /**
47    * Gets a serde for the given feature type
48    *
49    * @param typeName feature type name
50    * @return
51    */
52   def serde(typeName: String): Serde[GeoMesaMessage] = serde.forType(typeName)
53 
54   /**
55    * Create a stream of updates for a given feature type
56    *
57    * @param typeName feature type name
58    * @return
59    */
60   def stream(typeName: String): KStream[String, GeoMesaMessage] = wrapped.stream(serde.topic(typeName))
61 
62   /**
63    * Create a table for a given feature type
64    *
65    * @param typeName feature type name
66    * @return
67    */
68   def table(typeName: String): KTable[String, GeoMesaMessage] = wrapped.table(serde.topic(typeName))
69 
70   /**
71    * Create a table for a given feature type
72    *
73    * @param typeName feature type name
74    * @param materialized materialized
75    * @return
76    */
77   def table(
78       typeName: String,
79       materialized: Materialized[String, GeoMesaMessage, ByteArrayKeyValueStore]): KTable[String, GeoMesaMessage] =
80     wrapped.table(serde.topic(typeName), materialized)
81 
82   /**
83    * Create a global table for a given feature type
84    *
85    * @param typeName feature type name
86    * @return
87    */
88   def globalTable(typeName: String): GlobalKTable[String, GeoMesaMessage] =
89     wrapped.globalTable(serde.topic(typeName))
90 
91   /**
92    * Create a global table for a given feature type
93    *
94    * @param typeName feature type name
95    * @param materialized materialized
96    * @return
97    */
98   def globalTable(
99       typeName: String,
100       materialized: Materialized[String, GeoMesaMessage, ByteArrayKeyValueStore]): GlobalKTable[String, GeoMesaMessage] =
101     wrapped.globalTable(serde.topic(typeName), materialized)
102 
103   /**
104    * Write the stream to the given feature type, which must already exist. The messages
105    * must conform to the feature type schema
106    *
107    * @param typeName feature type name
108    * @param stream stream to persist
109    */
110   def to(typeName: String, stream: KStream[String, GeoMesaMessage]): Unit = {
111     implicit val produced: Produced[String, GeoMesaMessage] =
112       if (serde.usesDefaultPartitioning(typeName)) {
113         Produced.`with`
114       } else {
115         Produced.`with`(new GeoMessageStreamPartitioner())
116       }
117     stream.to(serde.topic(typeName))
118   }
119 
120   /**
121    * Convenience method to build the underlying topology
122    *
123    * @return
124    */
125   def build(): Topology = wrapped.build()
126 }

235   /**
236    * Create a streams builder
237    *
238    * @param params data store parameters
239    * @param timestampExtractor timestamp extractor for message stream
240    * @param resetPolicy auto offset reset for reading existing topics
241    * @param streamsBuilder underlying streams builder to use
242    * @return
243    */
244   def apply(
245       params: java.util.Map[String, String],
246       timestampExtractor: TimestampExtractor,
247       resetPolicy: AutoOffsetReset,
248       streamsBuilder: StreamsBuilder): GeoMesaStreamsBuilder = {
249     val serde = new GeoMesaSerde()
250     serde.configure(params, isKey = false)
251     val builder = Option(streamsBuilder).getOrElse(new StreamsBuilder())
252     val timestamps = Option(timestampExtractor).getOrElse(GeoMesaTimestampExtractor(params))
253     val reset = Option(resetPolicy).orElse(resetConfig(params))
254     new GeoMesaStreamsBuilder(builder, serde, timestamps, reset)
255   }
256 
257   private def resetConfig(params: java.util.Map[String, String]): Option[AutoOffsetReset] = {
258     val config = KafkaDataStoreFactory.buildConfig(params.asInstanceOf[java.util.Map[String, java.io.Serializable]])
259     config.consumers.properties.get(ConsumerConfig.AUTO_OFFSET_RESET_CONFIG).map {
260       case r if r.equalsIgnoreCase(AutoOffsetReset.EARLIEST.name()) => AutoOffsetReset.EARLIEST
261       case r if r.equalsIgnoreCase(AutoOffsetReset.LATEST.name()) => AutoOffsetReset.LATEST
262       case r => throw new IllegalArgumentException(s"Invalid ${ConsumerConfig.AUTO_OFFSET_RESET_CONFIG}: $r")
263     }.orElse {
264       config.consumers.readBack.collect { case Duration.Inf => AutoOffsetReset.EARLIEST }
265     }
266   }
