89       KafkaDataStoreFactory.ParameterInfo.filterNot(UnusedParams.contains)
90 
91   override def canProcess(params: java.util.Map[String, _]): Boolean =
92     KafkaDataStoreParams.Brokers.exists(params) && SchemaRegistryUrl.exists(params)
93 
94   private[confluent] def parseSchemaOverrides(config: Option[String]): Map[String, (SimpleFeatureType, Schema)] = {
95     try {
96       config.map {
97         ConfigFactory.parseString(_).resolve().getObject("schemas").asScala.toMap.map {
98           case (topic, schemaConfig) =>
99             try {
100               val schema = new Schema.Parser().parse(schemaConfig.render(ConfigRenderOptions.concise()))
101               val sft = SchemaParser.schemaToSft(schema, Some(topic))
102               topic -> (sft, schema)
103             } catch {
104               case NonFatal(ex) =>
105                 throw new IllegalArgumentException(s"Schema override for topic '$topic' is invalid: ${ex.getMessage}")
106             }
107         }
108       }.getOrElse(Map.empty)
109     } catch {
110       case NonFatal(ex) => throw new IllegalArgumentException(s"Failed to parse schema overrides: ${ex.getMessage}")
111     }
112   }
