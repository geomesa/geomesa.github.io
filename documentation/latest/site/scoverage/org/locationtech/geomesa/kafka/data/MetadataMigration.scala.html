17 /**
18   * Migrates kafka data store metadata from geomesa 1.3.x to 1.4.x
19   *
20   * @param ds datastore
21   * @param zkPath zk path
22   * @param zookeepers zookeepers
23   */
24 class MetadataMigration(ds: KafkaDataStore, zkPath: String, zookeepers: String) extends Runnable {
25 
26   override def run(): Unit = {
27     import scala.collection.JavaConverters._
28 
29     val client =
30       CuratorFrameworkFactory.builder()
31         .connectString(zookeepers)
32         .retryPolicy(new ExponentialBackoffRetry(1000, 3))
33         .dontUseContainerParents()
34         .namespace(zkPath)
35         .build()
36 
37     try {
38       client.start()
39       client.blockUntilConnected()
40 
41       if (client.checkExists().forPath("/") != null) {
42         client.getChildren.forPath("/").asScala.foreach { name =>
43           if (name != KafkaDataStore.MetadataPath && client.checkExists().forPath(s"/$name/Topic") != null) {
44             if (name.indexOf("-REPLAY-") == -1) {
45               val schema = new String(client.getData.forPath(s"/$name"), StandardCharsets.UTF_8)
46               val sft = SimpleFeatureTypes.createType(name, schema)
47               KafkaDataStore.setTopic(sft, new String(client.getData.forPath(s"/$name/Topic"), StandardCharsets.UTF_8))
48               sft.getUserData.put(SimpleFeatureTypes.Configs.OverrideReservedWords, "true")
49               ds.createSchema(sft)
50             }
51             client.delete().deletingChildrenIfNeeded().forPath(s"/$name")
52           }
53         }
54       }
55     } finally {
56       client.close()
57     }
58   }
59 }
