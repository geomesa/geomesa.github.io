19 class ZookeeperMetadata[T](val namespace: String, val zookeepers: String, val serializer: MetadataSerializer[T])
20     extends KeyValueStoreMetadata[T] {
21 
22   import org.locationtech.geomesa.kafka.data.ZookeeperMetadata.Root
23 
24   private val client =
25     CuratorFrameworkFactory.builder()
26       .connectString(zookeepers)
27       .retryPolicy(new ExponentialBackoffRetry(1000, 3))
28       .dontUseContainerParents()
29       .namespace(namespace)
30       .build()
31   client.start()
32 
33   override protected def checkIfTableExists: Boolean = true
34 
35   override protected def createTable(): Unit = {}
36 
37   override protected def createEmptyBackup(timestamp: String): ZookeeperMetadata[T] =
38     new ZookeeperMetadata(s"${namespace}_${timestamp}_bak", zookeepers, serializer)
39 
40   override protected def write(rows: Seq[(Array[Byte], Array[Byte])]): Unit = {
41     rows.foreach { case (row, value) =>
42       val path = toPath(row)
43       if (client.checkExists().forPath(path) == null) {
44         client.create().creatingParentsIfNeeded().forPath(path)
45       }
46       client.setData().forPath(path, value)
47     }
48   }
49 
50   override protected def delete(rows: Seq[Array[Byte]]): Unit = {
51     rows.foreach { row =>
52       val path = toPath(row)
53       if (client.checkExists().forPath(path) != null) {
54         client.delete().deletingChildrenIfNeeded().forPath(path)
55       }
56     }
57   }
58 
59   override protected def scanValue(row: Array[Byte]): Option[Array[Byte]] = {
60     val path = toPath(row)
61     if (client.checkExists().forPath(path) == null) { None } else {
62       Option(client.getData.forPath(path))
63     }
64   }
65 
66   override protected def scanRows(prefix: Option[Array[Byte]]): CloseableIterator[(Array[Byte], Array[Byte])] = {
67     val path = prefix.map(toPath(_, withSlash = false))
68     if (client.checkExists().forPath(Root) == null) { CloseableIterator.empty } else {
69       val all = CloseableIterator(client.getChildren.forPath(Root).iterator)
70       val filtered = path match {
71         case None => all
72         case Some(p) => all.filter(_.startsWith(p))
73       }
74       filtered.map { f =>
75         val bytes = f.getBytes(StandardCharsets.UTF_8)
76         (bytes, client.getData.forPath(toPath(bytes)))
77       }
78     }
79   }
80 
81   override def close(): Unit = {
82     CloseWithLogging(client)
83     super.close()
84   }
85 
86   private def toPath(row: Array[Byte], withSlash: Boolean = true): String = {
87     val string = new String(row, StandardCharsets.UTF_8) // TODO escape?
88     if (withSlash) { "/" + string } else { string }
89   }
90 }
