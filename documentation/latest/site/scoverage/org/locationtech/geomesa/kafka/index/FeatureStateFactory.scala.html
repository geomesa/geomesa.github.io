41   def apply(
42       sft: SimpleFeatureType,
43       index: SpatialIndex[SimpleFeature],
44       expiry: ExpiryTimeConfig,
45       expiration: FeatureExpiration,
46       executor: Option[(ScheduledExecutorService, Ticker)]): FeatureStateFactory = {
47 
48     val geom = sft.getGeomIndex
49 
50     lazy val (es, ticker) = executor.getOrElse {
51       val es = new ScheduledThreadPoolExecutor(2)
52       // don't keep running scheduled tasks after shutdown
53       es.setExecuteExistingDelayedTasksAfterShutdownPolicy(false)
54       // remove tasks when canceled, otherwise they will only be removed from the task queue
55       // when they would be executed. we expect frequent cancellations due to feature updates
56       es.setRemoveOnCancelPolicy(true)
57       (es, CurrentTimeTicker)
58     }
59 
60     expiry match {
61       case NeverExpireConfig =>
62         new BasicFactory(index, geom)
63 
64       case IngestTimeConfig(ex) =>
65         new ExpiryFactory(index, geom, expiration, es, ex.toMillis)
66 
67       case EventTimeConfig(ex, time, ordering) =>
68         val expression = FastFilterFactory.toExpression(sft, time)
69         if (!ex.isFinite && ordering) {
70           new EventTimeFactory(index, geom, expression)
71         } else if (ordering) {
72           new EventTimeOrderedExpiryFactory(index, geom, expression, expiration, es, ticker, ex.toMillis)
73         } else {
74           new EventTimeExpiryFactory(index, geom, expression, expiration, es, ticker, ex.toMillis)
75         }
76 
77       case FilteredExpiryConfig(ex) =>
78         val delegates = ex.map { case (ecql, e) =>
79           FastFilterFactory.toFilter(sft, ecql) -> apply(sft, index, e, expiration, Some(es -> ticker))
80         }
81         new FilteredExpiryFactory(delegates)
82 
83       case ImmediatelyExpireConfig =>
84         throw new IllegalStateException("Can't use feature state with immediate expiration")
85     }
86   }

88   def time(expression: Expression, feature: SimpleFeature): Long = {
89     try {
90       expression.evaluate(feature) match {
91         case d: Date   => d.getTime
92         case d: Number => d.longValue()
93         case d => Option(FastConverter.convert(d, classOf[Date])).map(_.getTime).getOrElse(0L)
94       }
95     } catch {
96       case NonFatal(e) => logger.error(s"Error evaluating event time for $feature", e); 0L
97     }
98   }

120   /**

123     * @param feature feature

126     * @param time feature time

128   class BasicState(feature: SimpleFeature, index: SpatialIndex[SimpleFeature], geom: Int, val time: Long)
129       extends FeatureState {
130 
131     private val g = feature.getAttribute(geom).asInstanceOf[Geometry]
132 
133     override val id: String = feature.getID
134     override def insertIntoIndex(): Unit = index.insert(g, id, feature)
135     override def removeFromIndex(): SimpleFeature = index.remove(g, id)
136     override def retrieveFromIndex(): SimpleFeature = index.get(g, id)

138     override def toString: String = s"FeatureState($feature)"
139   }

141   /**
142     * Handles time-based expiration
143     *
144     * @param feature feature
145     * @param index spatial index
146     * @param geom geometry attribute index
147     * @param time feature time
148     * @param expiration callback for expiration
149     * @param executor executor used to schedule expiration callback
150     * @param expiry expiry in milliseconds

152   class ExpiryState(feature: SimpleFeature,
153                     index: SpatialIndex[SimpleFeature],
154                     geom: Int,
155                     time: Long,
156                     expiration: FeatureExpiration,
157                     executor: ScheduledExecutorService,
158                     expiry: Long) extends BasicState(feature, index, geom, time) with Runnable {
159 
160     private var future: ScheduledFuture[_] = _
161 
162     override def run(): Unit = expiration.expire(this)
163 
164     override def insertIntoIndex(): Unit = {
165       super.insertIntoIndex()
166       future = executor.schedule(this, expiry, TimeUnit.MILLISECONDS)
167     }
168 
169     override def removeFromIndex(): SimpleFeature = {
170       future.cancel(false)
171       super.removeFromIndex()
172     }
173 
174     override def toString: String = s"ExpiryState($feature)"
175   }

246   /**
247     * Feature state factory with event time expiration but message time ordering
248     *
249     * @param index spatial index
250     * @param geom geometry attribute
251     * @param eventTime event time expression
252     * @param expiration expiration callback
253     * @param executor executor
254     * @param ticker scheduler
255     * @param expiry expiry in millis
256     */
257   class EventTimeExpiryFactory(index: SpatialIndex[SimpleFeature],
258                                geom: Int,
259                                eventTime: Expression,
260                                expiration: FeatureExpiration,
261                                executor: ScheduledExecutorService,
262                                ticker: Ticker,
263                                expiry: Long) extends FeatureStateFactory {
264 
265     override def createState(feature: SimpleFeature): FeatureState = {
266       val expiry = FeatureStateFactory.time(eventTime, feature) + this.expiry - (ticker.read() / 1000000L)
267       if (expiry < 1L) {
268         new ExpiredState(feature, 0L, expiration)
269       } else {
270         new ExpiryState(feature, index, geom, 0L, expiration, executor, expiry)
271       }
272     }
273 
274     override def close(): Unit = executor.shutdownNow()
275 
276     override def toString: String =
277       s"EventTimeExpiryFactory[geom:$geom,eventTime:${ECQL.toCQL(eventTime)},expiry:$expiry]"
278   }

302       if (expiry < 1L) {

304       } else {

306       }
307     }

315   class FilteredExpiryFactory(delegates: Seq[(Filter, FeatureStateFactory)]) extends FeatureStateFactory {
316 
317     require(delegates.last._1 == Filter.INCLUDE,
318       "Filter feature state factory requires a fall back Filter.INCLUDE entry")
319 
320     override def createState(feature: SimpleFeature): FeatureState = {
321       val opt = delegates.collectFirst {
322         case (f, factory) if f.evaluate(feature) => factory.createState(feature)
323       }
324       opt.get // should always get a result due to the Filter.INCLUDE
325     }
326 
327     override def close(): Unit = CloseWithLogging(delegates.map(_._2))
328 
329     override def toString: String =
330       s"FilteredExpiryFactory[delegates:${delegates.map { case (f, d) => s"${ECQL.toCQL(f)}->$d"}.mkString(",")}]"
331   }
