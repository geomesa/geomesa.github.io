41   class ConsumerRunnable(id: String, consumer: Consumer[Array[Byte], Array[Byte]], handler: ConsumerErrorHandler)
42       extends Runnable {
43 
44     private var errorCount = 0
45 
46     lazy private val topics = consumer.subscription().asScala.mkString(", ")
47 
48     private var lastOffsetCommitMs = System.currentTimeMillis()
49 
50     override def run(): Unit = {
51       try {
52         var interrupted = false
53         while (isOpen && !interrupted) {
54           try {
55             val result = KafkaConsumerVersions.poll(consumer, frequency)
56             lazy val topics = result.partitions.asScala.map(tp => s"[${tp.topic}:${tp.partition}]").mkString(",")
57             if (result.isEmpty) {
58               logger.trace(s"Consumer [$id] poll received 0 records")
59             } else {
60               logger.debug(s"Consumer [$id] poll received ${result.count()} records for $topics")
61               val records = result.iterator()
62               while (records.hasNext) {
63                 consume(records.next())
64               }
65               logger.trace(s"Consumer [$id] finished processing ${result.count()} records from topic $topics")
66               if (System.currentTimeMillis() - lastOffsetCommitMs > offsetCommitIntervalMs) {
67                 logger.trace(s"Consumer [$id] committing offsets")
68                 consumer.commitAsync()
69                 lastOffsetCommitMs = System.currentTimeMillis()
70               }
71               errorCount = 0 // reset error count
72             }
73           } catch {
74             case _: WakeupException | _: InterruptException | _: InterruptedException => interrupted = true
75             case NonFatal(e) =>
76               errorCount += 1
77               logger.warn(s"Consumer [$id] error receiving message from topic $topics:", e)
78               if (errorCount <= 300 || handler.handle(id, e)) {
79                 Thread.sleep(1000)
80               } else {
81                 logger.error(s"Consumer [$id] shutting down due to too many errors from topic $topics:", e)
82                 throw e
83               }
84           }
85         }
86       } finally {
87         if (closeConsumers) {
88           try { consumer.close() } catch {
89             case NonFatal(e) => logger.warn(s"Error calling close on consumer: ", e)
90           }
91         }
92       }
93     }
94   }
95 }
