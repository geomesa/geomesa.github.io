41   class ConsumerRunnable(id: String, consumer: Consumer[Array[Byte], Array[Byte]], handler: ConsumerErrorHandler)
42       extends Runnable {
43 
44     private var errorCount = 0
45 
46     lazy private val topics = consumer.subscription().asScala.mkString(", ")
47 
48     private var lastOffsetCommitMs = System.currentTimeMillis()
49 
50     override def run(): Unit = {
51       try {
52         var interrupted = false
53         while (isOpen && !interrupted) {
54           try {
55             processPoll(KafkaConsumerVersions.poll(consumer, frequency))
56           } catch {
57             case _: WakeupException | _: InterruptException | _: InterruptedException => interrupted = true
58             case NonFatal(e) =>
59               errorCount += 1
60               logger.warn(s"Consumer [$id] error receiving message from topic $topics:", e)
61               if (errorCount <= 300 || handler.handle(id, e)) {
62                 Thread.sleep(1000)
63               } else {
64                 logger.error(s"Consumer [$id] shutting down due to too many errors from topic $topics:", e)
65                 throw e
66               }
67           }
68         }
69       } finally {
70         if (closeConsumers) {
71           try { consumer.close() } catch {
72             case NonFatal(e) => logger.warn(s"Error calling close on consumer: ", e)
73           }
74         }
75       }
76     }
77 
78     /**
79      * Process the results of a call to poll()
80      *
81      * @param result records
82      */
83     protected def processPoll(result: ConsumerRecords[Array[Byte], Array[Byte]]): Unit = {
84       if (result.isEmpty) {
85         logger.trace(s"Consumer [$id] poll received 0 records")
86       } else {
87         lazy val topics = result.partitions.asScala.map(tp => s"[${tp.topic}:${tp.partition}]").mkString(",")
88         logger.debug(s"Consumer [$id] poll received ${result.count()} records for $topics")
89         val records = result.iterator()
90         while (records.hasNext) {
91           consume(records.next())
92         }
93         logger.trace(s"Consumer [$id] finished processing ${result.count()} records from topic $topics")
94         if (System.currentTimeMillis() - lastOffsetCommitMs > offsetCommitIntervalMs) {
95           logger.trace(s"Consumer [$id] committing offsets")
96           consumer.commitAsync()
97           lastOffsetCommitMs = System.currentTimeMillis()
98         }
99         errorCount = 0 // reset error count
100       }
101     }
102   }
103 }
