23   /**
24    * Get a timestamp extractor based on the data store config
25    *
26    * @param params data store params
27    * @return
28    */
29   def apply(params: java.util.Map[String, _]): TimestampExtractor = {
30     val paramsWithLazyFeatures = new java.util.HashMap[String, Any](params)
31     // set lazy features so we don't deserialize the whole message just to get the timestamp
32     paramsWithLazyFeatures.put(KafkaDataStoreParams.LazyFeatures.key, "true")
33     // disable consumers if not already done
34     paramsWithLazyFeatures.put(KafkaDataStoreParams.ConsumerCount.key, 0)
35 
36     // check for event time config
37     val eventTime = if (!KafkaDataStoreParams.EventTimeOrdering.lookup(params)) { None } else {
38       KafkaDataStoreParams.EventTime.lookupOpt(params)
39     }
40     eventTime match {
41       case None => new DefaultDateExtractor(paramsWithLazyFeatures)
42       case Some(e) => new EventTimestampExtractor(e, paramsWithLazyFeatures)
43     }
44   }
45 
46   /**

49    * @param expression expression for the timestamp
50    * @param params data store params

71   /**
72    * Base class for extracting timestamps from simple features
73    *
74    * @param params data store params
75    */
76   private[streams] abstract class FeatureTimestampExtractor(params: java.util.Map[String, _])
77       extends TimestampExtractor {
78 
79     private val cache = new SerializerCache(params.asInstanceOf[java.util.Map[String, Any]])
80     private val expressions = new ConcurrentHashMap[String, Option[Expression]]
81 
82     private val expressionLoader = new java.util.function.Function[String, Option[Expression]]() {
83       override def apply(topic: String): Option[Expression] =
84         loadExpression(cache.serializer(topic).sft)
85     }
86 
87     // track last-used serializer so we don't have to look them up by hash each
88     // time if we're just reading/writing to one topic (which is the standard use-case)
89     @volatile
90     private var last: (String, Option[Expression]) = ("", null)
91 
92     protected def loadExpression(sft: SimpleFeatureType): Option[Expression]
93 
94     override def extract(record: ConsumerRecord[AnyRef, AnyRef], previousTimestamp: Long): Long = {
95       record.value() match {
96         case null =>
97           record.timestamp()
98 
99         case m: GeoMesaMessage if m.action == MessageAction.Upsert =>
100           val topic = record.topic()
101           val (lastTopic, lastExpression) = last
102           val expression = if (lastTopic == topic) { lastExpression } else {
103             val expression = expressions.computeIfAbsent(topic, expressionLoader)
104             // should be thread-safe due to volatile
105             last = (topic, expression)
106             expression
107           }
108           expression match {
109             case None => record.timestamp()
110             case Some(e) => FeatureStateFactory.time(e, cache.serializer(topic).wrap(m))
111           }
112 
113         case m: GeoMesaMessage if m.action == MessageAction.Delete =>
114           record.timestamp()
115 
116         case _: GeoMesaMessage =>
117           throw new NotImplementedError() // if we forget to handle a message action
118 
119         case v =>
120           throw new IllegalArgumentException(s"Expected a GeoMesaMessage but got: ${v.getClass.getName} $v")
121       }
122     }
123   }
