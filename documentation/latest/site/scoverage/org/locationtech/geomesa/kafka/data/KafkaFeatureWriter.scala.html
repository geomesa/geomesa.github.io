41   class AppendKafkaFeatureWriter(
42       sft: SimpleFeatureType,
43       producer: KafkaFeatureProducer,
44       protected val serializer: GeoMessageSerializer
45     ) extends KafkaFeatureWriter with LazyLogging {
46 
47     protected val topic: String = KafkaDataStore.topic(sft)
48 
49     protected val feature = new ScalaSimpleFeature(sft, "-1")
50 
51     override def getFeatureType: SimpleFeatureType = sft
52 
53     override def hasNext: Boolean = false
54 
55     override def next(): SimpleFeature = {
56       reset(featureIds.getAndIncrement().toString)
57       feature
58     }
59 
60     override def write(): Unit = {
61       val sf = GeoMesaFeatureWriter.featureWithFid(feature)
62       // we've handled the fid hints, remove them so that we don't serialize them
63       FeatureIdHints.foreach(sf.getUserData.remove)
64       logger.debug(s"Writing update to $topic: $sf")
65       val (key, value, headers) = serializer.serialize(GeoMessage.change(sf))
66       val record = new ProducerRecord(topic, key, value)
67       headers.foreach { case (k, v) => RecordVersions.setHeader(record, k, v) }
68       producer.send(record)
69     }
70 
71     override def remove(): Unit = throw new NotImplementedError()
72 
73     override def flush(): Unit = producer.flush()
74 
75     override def close(): Unit = producer.flush() // note: the producer is shared, so don't close it
76 
77     override def clear(): Unit = {
78       logger.debug(s"Writing clear to $topic")
79       val (key, value, headers) = serializer.serialize(GeoMessage.clear())
80       val record = new ProducerRecord(topic, key, value)
81       headers.foreach { case (k, v) => RecordVersions.setHeader(record, k, v) }
82       producer.send(record)
83     }
84 
85     protected def reset(id: String): Unit = {
86       feature.setId(id)
87       var i = 0
88       while (i < sft.getAttributeCount) {
89         feature.setAttributeNoConvert(i, null)
90         i += 1
91       }
92       feature.getUserData.clear()
93     }
94   }

146   case class KafkaTransactionState(producer: Producer[Array[Byte], Array[Byte]])
147       extends Transaction.State with KafkaFeatureProducer {
148 
149     private val tx = new AtomicReference[Transaction](null)
150     private var inTransaction = false
151 
152     override def send(record: ProducerRecord[Array[Byte], Array[Byte]]): Unit = {
153       if (!inTransaction) {
154         inTransaction = true
155         producer.beginTransaction()
156       }
157       producer.send(record)
158     }
159 
160     /**
161      * Sends offsets to the consumer coordinator, as part of a consume/transform/produce operation.
162      * See `org.apache.kafka.clients.producer.KafkaProducer#sendOffsetsToTransaction(Map,String)`
163      *
164      * @param offsets offsets to send
165      * @param consumerGroupId consumer group id
166      */
167     def sendOffsets(offsets: java.util.Map[TopicPartition, OffsetAndMetadata], consumerGroupId: String): Unit = {
168       if (inTransaction) {
169         producer.sendOffsetsToTransaction(offsets, consumerGroupId)
170       }
171     }
172 
173     override def flush(): Unit = producer.flush()
174 
175     override def setTransaction(transaction: Transaction): Unit = {
176       if (transaction == null) {
177         // null transaction indicates the transaction has been closed
178         try {
179           if (inTransaction) {
180             inTransaction = false
181             producer.commitTransaction()
182           }
183         } finally {
184           producer.close()
185         }
186       } else if (tx.compareAndSet(null, transaction)) {
187         producer.initTransactions()
188       } else {
189         throw new IllegalStateException(
190           s"State is already associated with transaction ${tx.get} and can't be associated with $transaction")
191       }
192     }
193 
194     override def commit(): Unit = {
195       if (inTransaction) {
196         inTransaction = false
197         producer.commitTransaction()
198       }
199     }
200 
201     override def rollback(): Unit = {
202       if (inTransaction) {
203         inTransaction = false
204         producer.abortTransaction()
205       }
206     }
207 
208     override def addAuthorization(authID: String): Unit = {}
209   }
