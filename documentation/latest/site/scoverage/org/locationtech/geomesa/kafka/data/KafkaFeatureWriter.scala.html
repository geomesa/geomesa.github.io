39   private val featureIds = new AtomicLong(0)
40   private val FeatureIdHints = Seq(Hints.USE_PROVIDED_FID, Hints.PROVIDED_FID)
41   private val MetricsPrefix = s"${KafkaDataStore.MetricsPrefix}.producer"

43   class AppendKafkaFeatureWriter(
44       sft: SimpleFeatureType,
45       producer: KafkaFeatureProducer,
46       protected val serializer: GeoMessageSerializer,
47       tags: Tags
48     ) extends KafkaFeatureWriter with LazyLogging {
49 
50     protected val topic: String = KafkaDataStore.topic(sft)
51 
52     protected val feature = new ScalaSimpleFeature(sft, "-1")
53 
54     private val writeCounter = Metrics.counter(s"$MetricsPrefix.produced", tags.and("op", "update"))
55     private val clearCounter = Metrics.counter(s"$MetricsPrefix.produced", tags.and("op", "clear"))
56 
57     override def getFeatureType: SimpleFeatureType = sft
58 
59     override def hasNext: Boolean = false
60 
61     override def next(): FastSettableFeature = {
62       reset(featureIds.getAndIncrement().toString)
63       feature
64     }
65 
66     override def write(): Unit = {
67       val sf = GeoMesaFeatureWriter.featureWithFid(feature)
68       // we've handled the fid hints, remove them so that we don't serialize them
69       FeatureIdHints.foreach(sf.getUserData.remove)
70       logger.debug(s"Writing update to $topic: $sf")
71       val (key, value, headers) = serializer.serialize(GeoMessage.change(sf))
72       val record = new ProducerRecord(topic, key, value)
73       headers.foreach { case (k, v) => RecordVersions.setHeader(record, k, v) }
74       producer.send(record)
75       writeCounter.increment()
76     }
77 
78     override def remove(): Unit = throw new UnsupportedOperationException()
79 
80     override def flush(): Unit = producer.flush()
81 
82     override def close(): Unit = producer.flush() // note: the producer is shared, so don't close it
83 
84     override def clear(): Unit = {
85       logger.debug(s"Writing clear to $topic")
86       val (key, value, headers) = serializer.serialize(GeoMessage.clear())
87       val record = new ProducerRecord(topic, key, value)
88       headers.foreach { case (k, v) => RecordVersions.setHeader(record, k, v) }
89       producer.send(record)
90       clearCounter.increment()
91     }
92 
93     protected def reset(id: String): Unit = {
94       feature.setId(id)
95       var i = 0
96       while (i < sft.getAttributeCount) {
97         feature.setAttributeNoConvert(i, null)
98         i += 1
99       }
100       feature.getUserData.clear()
101     }
102   }

104   class ModifyKafkaFeatureWriter(
105       sft: SimpleFeatureType,
106       producer: KafkaFeatureProducer,
107       serializer: GeoMessageSerializer,
108       tags: Tags,
109       filter: Filter
110     ) extends AppendKafkaFeatureWriter(sft, producer, serializer, tags) {

112     import scala.collection.JavaConverters._
113 
114     private val ids: Iterator[String] = filter match {
115       case ids: Id => ids.getIDs.iterator.asScala.map(_.toString)
116       case _ => throw new UnsupportedOperationException("Only modify by ID is supported")
117     }
118 
119     private val deleteCounter = Metrics.counter(s"$MetricsPrefix.produced", tags.and("op", "delete"))
120 
121     override def hasNext: Boolean = ids.hasNext
122 
123     override def next(): FastSettableFeature = {
124       reset(if (ids.hasNext) { ids.next() } else { featureIds.getAndIncrement().toString })
125       // default to using the provided fid
126       feature.getUserData.put(Hints.USE_PROVIDED_FID, java.lang.Boolean.TRUE)
127       feature
128     }
129 
130     override def remove(): Unit = {
131       val id = GeoMesaFeatureWriter.featureWithFid(feature).getID
132       logger.debug(s"Writing delete to $topic: $id")
133       val (key, value, headers) = serializer.serialize(GeoMessage.delete(id))
134       val record = new ProducerRecord(topic, key, value)
135       headers.foreach { case (k, v) => RecordVersions.setHeader(record, k, v) }
136       producer.send(record)
137       deleteCounter.increment()
138     }
139   }

157   case class KafkaTransactionState(producer: Producer[Array[Byte], Array[Byte]])
158       extends Transaction.State with KafkaFeatureProducer {
159 
160     private val tx = new AtomicReference[Transaction](null)
161     private var inTransaction = false
162 
163     override def send(record: ProducerRecord[Array[Byte], Array[Byte]]): Unit = {
164       if (!inTransaction) {
165         inTransaction = true
166         producer.beginTransaction()
167       }
168       producer.send(record)
169     }
170 
171     /**
172      * Sends offsets to the consumer coordinator, as part of a consume/transform/produce operation.
173      * See `org.apache.kafka.clients.producer.KafkaProducer#sendOffsetsToTransaction(Map,String)`
174      *
175      * @param offsets offsets to send
176      * @param consumerGroupId consumer group id
177      */
178     def sendOffsets(offsets: java.util.Map[TopicPartition, OffsetAndMetadata], consumerGroupId: String): Unit = {
179       if (inTransaction) {
180         producer.sendOffsetsToTransaction(offsets, consumerGroupId)
181       }
182     }
183 
184     override def flush(): Unit = producer.flush()
185 
186     override def setTransaction(transaction: Transaction): Unit = {
187       if (transaction == null) {
188         // null transaction indicates the transaction has been closed
189         try {
190           if (inTransaction) {
191             inTransaction = false
192             producer.commitTransaction()
193           }
194         } finally {
195           producer.close()
196         }
197       } else if (tx.compareAndSet(null, transaction)) {
198         producer.initTransactions()
199       } else {
200         throw new IllegalStateException(
201           s"State is already associated with transaction ${tx.get} and can't be associated with $transaction")
202       }
203     }
204 
205     override def commit(): Unit = {
206       if (inTransaction) {
207         inTransaction = false
208         producer.commitTransaction()
209       }
210     }
211 
212     override def rollback(): Unit = {
213       if (inTransaction) {
214         inTransaction = false
215         producer.abortTransaction()
216       }
217     }
218 
219     override def addAuthorization(authID: String): Unit = {}
220   }
