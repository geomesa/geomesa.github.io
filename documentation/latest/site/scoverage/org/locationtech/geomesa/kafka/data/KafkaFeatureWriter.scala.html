43   class AppendKafkaFeatureWriter(
44       sft: SimpleFeatureType,
45       producer: KafkaFeatureProducer,
46       protected val serializer: GeoMessageSerializer,
47       tags: Tags
48     ) extends KafkaFeatureWriter with LazyLogging {
49 
50     protected val topic: String = KafkaDataStore.topic(sft)
51 
52     protected val feature = new ScalaSimpleFeature(sft, "-1")
53 
54     private val writeCounter =
55       Counter.builder(s"$MetricsPrefix.writes")
56         .description("Count of feature writes")
57         .tags(tags)
58         .register(Metrics.globalRegistry)
59 
60     private val clearCounter =
61       Counter.builder(s"$MetricsPrefix.clears")
62         .description("Count of feature clears")
63         .tags(tags)
64         .register(Metrics.globalRegistry)
65 
66     override def getFeatureType: SimpleFeatureType = sft
67 
68     override def hasNext: Boolean = false
69 
70     override def next(): FastSettableFeature = {
71       reset(featureIds.getAndIncrement().toString)
72       feature
73     }
74 
75     override def write(): Unit = {
76       val sf = GeoMesaFeatureWriter.featureWithFid(feature)
77       // we've handled the fid hints, remove them so that we don't serialize them
78       FeatureIdHints.foreach(sf.getUserData.remove)
79       logger.debug(s"Writing update to $topic: $sf")
80       val (key, value, headers) = serializer.serialize(GeoMessage.change(sf))
81       val record = new ProducerRecord(topic, key, value)
82       headers.foreach { case (k, v) => RecordVersions.setHeader(record, k, v) }
83       producer.send(record)
84       writeCounter.increment()
85     }
86 
87     override def remove(): Unit = throw new UnsupportedOperationException()
88 
89     override def flush(): Unit = producer.flush()
90 
91     override def close(): Unit = producer.flush() // note: the producer is shared, so don't close it
92 
93     override def clear(): Unit = {
94       logger.debug(s"Writing clear to $topic")
95       val (key, value, headers) = serializer.serialize(GeoMessage.clear())
96       val record = new ProducerRecord(topic, key, value)
97       headers.foreach { case (k, v) => RecordVersions.setHeader(record, k, v) }
98       producer.send(record)
99       clearCounter.increment()
100     }
101 
102     protected def reset(id: String): Unit = {
103       feature.setId(id)
104       var i = 0
105       while (i < sft.getAttributeCount) {
106         feature.setAttributeNoConvert(i, null)
107         i += 1
108       }
109       feature.getUserData.clear()
110     }
111   }

171   case class KafkaTransactionState(producer: Producer[Array[Byte], Array[Byte]])
172       extends Transaction.State with KafkaFeatureProducer {
173 
174     private val tx = new AtomicReference[Transaction](null)
175     private var inTransaction = false
176 
177     override def send(record: ProducerRecord[Array[Byte], Array[Byte]]): Unit = {
178       if (!inTransaction) {
179         inTransaction = true
180         producer.beginTransaction()
181       }
182       producer.send(record)
183     }
184 
185     /**
186      * Sends offsets to the consumer coordinator, as part of a consume/transform/produce operation.
187      * See `org.apache.kafka.clients.producer.KafkaProducer#sendOffsetsToTransaction(Map,String)`
188      *
189      * @param offsets offsets to send
190      * @param consumerGroupId consumer group id
191      */
192     def sendOffsets(offsets: java.util.Map[TopicPartition, OffsetAndMetadata], consumerGroupId: String): Unit = {
193       if (inTransaction) {
194         producer.sendOffsetsToTransaction(offsets, consumerGroupId)
195       }
196     }
197 
198     override def flush(): Unit = producer.flush()
199 
200     override def setTransaction(transaction: Transaction): Unit = {
201       if (transaction == null) {
202         // null transaction indicates the transaction has been closed
203         try {
204           if (inTransaction) {
205             inTransaction = false
206             producer.commitTransaction()
207           }
208         } finally {
209           producer.close()
210         }
211       } else if (tx.compareAndSet(null, transaction)) {
212         producer.initTransactions()
213       } else {
214         throw new IllegalStateException(
215           s"State is already associated with transaction ${tx.get} and can't be associated with $transaction")
216       }
217     }
218 
219     override def commit(): Unit = {
220       if (inTransaction) {
221         inTransaction = false
222         producer.commitTransaction()
223       }
224     }
225 
226     override def rollback(): Unit = {
227       if (inTransaction) {
228         inTransaction = false
229         producer.abortTransaction()
230       }
231     }
232 
233     override def addAuthorization(authID: String): Unit = {}
234   }
