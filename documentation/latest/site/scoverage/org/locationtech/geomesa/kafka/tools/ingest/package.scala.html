19   /**
20    * Query over a time frame and return the features in sorted order, delayed based on the date of each feature
21    * to simulate the original ingestion stream
22    *
23    * Requires the iterator to be sorted by time
24    *
25    * @param dtg date attribute to sort by
26    * @param transforms query transforms, if any
27    * @param rate multiplier for the rate of returning features, applied to the original delay between features
28    * @param live project dates to current time
29    */
30   class SimplePlaybackIterator(
31       iterator: CloseableIterator[SimpleFeature],
32       sft: SimpleFeatureType,
33       dtg: Option[String] = None,
34       transforms: Array[String] = null,
35       rate: Float = 10f,
36       live: Boolean = false
37     ) extends CloseableIterator[SimpleFeature] with StrictLogging {
38 
39     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
40 
41     //  require(interval._2.after(interval._1), s"Interval is not ordered correctly: ${interval._1}/${interval._2}")
42 
43     private val dtgName = dtg.orElse(sft.getDtgField).getOrElse {
44       throw new IllegalArgumentException("Schema does not have a default date field")
45     }
46     private val tdefs = transforms match {
47       case null => null
48       case t if t.indexOf(dtgName) == -1 => t :+ dtgName
49       case t => t
50     }
51     private val dtgIndex = tdefs match {
52       case null => sft.indexOf(dtgName)
53       case t => t.indexOf(dtgName)
54     }
55     require(dtgIndex != -1, "Invalid date field")
56     private var start: Long = -1
57     private var eventStart: Long = -1
58 
59     override def hasNext: Boolean = iterator.hasNext
60 
61     override def next(): SimpleFeature = {
62       val feature = iterator.next();
63       val featureTime = feature.getAttribute(dtgIndex).asInstanceOf[Date].getTime
64       if (start == -1L) {
65         // emit the first feature as soon as it's available, and set the clock to start timing from here
66         start = System.currentTimeMillis()
67         logger.info("Starting replay clock at: {}", start)
68         eventStart = featureTime
69       }
70       val featureRelativeTime = start + ((featureTime - eventStart) / rate).toLong
71       val sleep = featureRelativeTime - System.currentTimeMillis()
72       if (sleep > 0) {
73         Thread.sleep(sleep)
74       }
75       if (live) {
76         feature.setAttribute(dtgIndex, new Date(featureRelativeTime))
77       }
78       feature
79     }
80 
81     override def close(): Unit = iterator.close()
82   }
