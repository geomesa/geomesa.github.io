42   /**
43    * Create a standard feature cache
44    *
45    * @param sft simple feature type
46    * @param config cache config
47    * @param views layer view config
48    * @param metrics optional metrics hook
49    * @return
50    */
51   def apply(
52     sft: SimpleFeatureType,
53     config: IndexConfig,
54     views: Seq[LayerView] = Seq.empty,
55     metrics: Option[GeoMesaMetrics] = None): KafkaFeatureCache = {
56     if (config.expiry == ImmediatelyExpireConfig) {
57       new NoOpFeatureCache(views.map(v => KafkaFeatureCacheView.empty(v.viewSft)))
58     } else {
59       metrics match {
60         case None => new KafkaFeatureCacheImpl(sft, config, views)
61         case Some(m) => new KafkaFeatureCacheWithMetrics(sft, config, views, m)
62       }
63     }
64   }

74   /**
75     * Cache that won't spatially index the features
76     *
77     * @param sft simple feature type
78     * @param ordering feature ordering
79     * @return
80     */
81   def nonIndexing(sft: SimpleFeatureType, ordering: ExpiryTimeConfig = NeverExpireConfig): KafkaFeatureCache = {
82     val event: PartialFunction[ExpiryTimeConfig, String] = {
83       case EventTimeConfig(_, exp, true) => exp
84     }
85 
86     val ord = ordering match {
87       case o if event.isDefinedAt(o) => Some(event.apply(o))
88       // all filters use the same event time ordering
89       case FilteredExpiryConfig(filters) if event.isDefinedAt(filters.head._2) => Some(event.apply(filters.head._2))
90       case _ => None
91     }
92 
93     ord.map(FastFilterFactory.toExpression(sft, _)) match {
94       case None => new NonIndexingFeatureCache()
95       case Some(exp) => new NonIndexingEventTimeFeatureCache(exp)
96     }
97   }

99   /**
100     * Create a CQEngine index support. Note that CQEngine handles points vs non-points internally
101     *
102     * @param sft simple feature type
103     * @param config index config
104     * @return
105     */
106   private [index] def cqIndexSupport(sft: SimpleFeatureType, config: IndexConfig): SpatialIndexSupport = {
107     val attributes = if (config.cqAttributes == Seq(KafkaDataStore.CqIndexFlag)) {
108       // deprecated boolean config to enable indices based on the stored simple feature type
109       CQIndexType.getDefinedAttributes(sft) ++ Option(sft.getGeomField).map((_, CQIndexType.GEOMETRY))
110     } else {
111       config.cqAttributes
112     }
113     GeoCQIndexSupport(sft, attributes, config.resolution.x, config.resolution.y)
114   }

148   /**
149     * Non-indexing feature cache that just tracks the most recent feature, based on event time
150     *
151     * @param time event time expression
152     */
153   class NonIndexingEventTimeFeatureCache(time: Expression) extends KafkaFeatureCache {
154 
155     private val state = new ConcurrentHashMap[String, (SimpleFeature, Long)]
156 
157     /**
158       * Note: this method is not thread-safe. The `state` and can fail to replace the correct values
159       * if the same feature is updated simultaneously from two different threads
160       *
161       * In our usage, this isn't a problem, as a given feature ID is always operated on by a single thread
162       * due to kafka consumer partitioning
163       */
164     override def put(feature: SimpleFeature): Unit = {
165       val tuple = (feature, FeatureStateFactory.time(time, feature))
166       val old = state.put(feature.getID, tuple)
167       if (old != null && old._2 > tuple._2) {
168         state.replace(feature.getID, tuple, old)
169       }
170     }
171 
172     override def remove(id: String): Unit = state.remove(id)
173 
174     override def clear(): Unit = state.clear()
175 
176     override def close(): Unit = {}
177 
178     override def size(): Int = state.size()
179 
180     override def size(filter: Filter): Int = query(filter).length
181 
182     override def query(id: String): Option[SimpleFeature] = Option(state.get(id)).map(_._1)
183 
184     override def query(filter: Filter): Iterator[SimpleFeature] = {
185       import scala.collection.JavaConverters._
186       val features = state.asScala.valuesIterator.map(_._1)
187       if (filter == Filter.INCLUDE) { features } else {
188         features.filter(filter.evaluate)
189       }
190     }
191 
192     override def views: Seq[KafkaFeatureCacheView] = Seq.empty
193   }
