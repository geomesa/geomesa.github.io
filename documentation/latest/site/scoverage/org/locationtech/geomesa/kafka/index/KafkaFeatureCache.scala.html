42   /**
43    * Create a standard feature cache
44    *
45    * @param sft simple feature type
46    * @param config cache config
47    * @param views layer view config

50   def apply(sft: SimpleFeatureType, config: IndexConfig, views: Seq[LayerView] = Seq.empty, tags: Tags = Tags.empty()): KafkaFeatureCache = {
51     if (config.expiry == ImmediatelyExpireConfig) {
52       new NoOpFeatureCache(views.map(v => KafkaFeatureCacheView.empty(v.viewSft)))

66   /**
67     * Cache that won't spatially index the features
68     *
69     * @param sft simple feature type
70     * @param ordering feature ordering
71     * @return
72     */
73   def nonIndexing(sft: SimpleFeatureType, ordering: ExpiryTimeConfig = NeverExpireConfig): KafkaFeatureCache = {
74     val event: PartialFunction[ExpiryTimeConfig, String] = {
75       case EventTimeConfig(_, exp, true) => exp
76     }
77 
78     val ord = ordering match {
79       case o if event.isDefinedAt(o) => Some(event.apply(o))
80       // all filters use the same event time ordering
81       case FilteredExpiryConfig(filters) if event.isDefinedAt(filters.head._2) => Some(event.apply(filters.head._2))
82       case _ => None
83     }
84 
85     ord.map(FastFilterFactory.toExpression(sft, _)) match {
86       case None => new NonIndexingFeatureCache()
87       case Some(exp) => new NonIndexingEventTimeFeatureCache(exp)
88     }
89   }

91   /**
92     * Create a CQEngine index support. Note that CQEngine handles points vs non-points internally
93     *
94     * @param sft simple feature type
95     * @param config index config
96     * @return
97     */
98   private [index] def cqIndexSupport(sft: SimpleFeatureType, config: IndexConfig): SpatialIndexSupport = {
99     val attributes = if (config.cqAttributes == Seq(KafkaDataStore.CqIndexFlag)) {
100       // deprecated boolean config to enable indices based on the stored simple feature type
101       CQIndexType.getDefinedAttributes(sft) ++ Option(sft.getGeomField).map((_, CQIndexType.GEOMETRY))
102     } else {
103       config.cqAttributes
104     }
105     GeoCQIndexSupport(sft, attributes, config.resolution.x, config.resolution.y)
106   }

140   /**
141     * Non-indexing feature cache that just tracks the most recent feature, based on event time
142     *
143     * @param time event time expression
144     */
145   class NonIndexingEventTimeFeatureCache(time: Expression) extends KafkaFeatureCache {
146 
147     private val state = new ConcurrentHashMap[String, (SimpleFeature, Long)]
148 
149     /**
150       * Note: this method is not thread-safe. The `state` and can fail to replace the correct values
151       * if the same feature is updated simultaneously from two different threads
152       *
153       * In our usage, this isn't a problem, as a given feature ID is always operated on by a single thread
154       * due to kafka consumer partitioning
155       */
156     override def put(feature: SimpleFeature): Unit = {
157       val tuple = (feature, FeatureStateFactory.time(time, feature))
158       val old = state.put(feature.getID, tuple)
159       if (old != null && old._2 > tuple._2) {
160         state.replace(feature.getID, tuple, old)
161       }
162     }
163 
164     override def remove(id: String): Unit = state.remove(id)
165 
166     override def clear(): Unit = state.clear()
167 
168     override def close(): Unit = {}
169 
170     override def size(): Int = state.size()
171 
172     override def size(filter: Filter): Int = query(filter).length
173 
174     override def query(id: String): Option[SimpleFeature] = Option(state.get(id)).map(_._1)
175 
176     override def query(filter: Filter): Iterator[SimpleFeature] = {
177       import scala.collection.JavaConverters._
178       val features = state.asScala.valuesIterator.map(_._1)
179       if (filter == Filter.INCLUDE) { features } else {
180         features.filter(filter.evaluate)
181       }
182     }
183 
184     override def views: Seq[KafkaFeatureCacheView] = Seq.empty
185   }
