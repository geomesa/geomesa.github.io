41   /**
42    * Create a standard feature cache
43    *
44    * @param sft simple feature type
45    * @param config cache config
46    * @param views layer view config
47    * @return
48    */
49   def apply(sft: SimpleFeatureType, config: IndexConfig, views: Seq[LayerView] = Seq.empty): KafkaFeatureCache = {
50     if (config.expiry == ImmediatelyExpireConfig) {
51       new NoOpFeatureCache(views.map(v => KafkaFeatureCacheView.empty(v.viewSft)))
52     } else {
53       new KafkaFeatureCacheImpl(sft, config, views)
54     }

65   /**
66     * Cache that won't spatially index the features
67     *
68     * @param sft simple feature type
69     * @param ordering feature ordering
70     * @return
71     */
72   def nonIndexing(sft: SimpleFeatureType, ordering: ExpiryTimeConfig = NeverExpireConfig): KafkaFeatureCache = {
73     val event: PartialFunction[ExpiryTimeConfig, String] = {
74       case EventTimeConfig(_, exp, true) => exp
75     }
76 
77     val ord = ordering match {
78       case o if event.isDefinedAt(o) => Some(event.apply(o))
79       // all filters use the same event time ordering
80       case FilteredExpiryConfig(filters) if event.isDefinedAt(filters.head._2) => Some(event.apply(filters.head._2))
81       case _ => None
82     }
83 
84     ord.map(FastFilterFactory.toExpression(sft, _)) match {
85       case None => new NonIndexingFeatureCache()
86       case Some(exp) => new NonIndexingEventTimeFeatureCache(exp)
87     }
88   }

90   /**
91     * Create a CQEngine index support. Note that CQEngine handles points vs non-points internally
92     *
93     * @param sft simple feature type
94     * @param config index config
95     * @return
96     */
97   private [index] def cqIndexSupport(sft: SimpleFeatureType, config: IndexConfig): SpatialIndexSupport = {
98     val attributes = if (config.cqAttributes == Seq(KafkaDataStore.CqIndexFlag)) {
99       // deprecated boolean config to enable indices based on the stored simple feature type
100       CQIndexType.getDefinedAttributes(sft) ++ Option(sft.getGeomField).map((_, CQIndexType.GEOMETRY))
101     } else {
102       config.cqAttributes
103     }
104     GeoCQIndexSupport(sft, attributes, config.resolution.x, config.resolution.y)
105   }

139   /**
140     * Non-indexing feature cache that just tracks the most recent feature, based on event time
141     *
142     * @param time event time expression
143     */
144   class NonIndexingEventTimeFeatureCache(time: Expression) extends KafkaFeatureCache {
145 
146     private val state = new ConcurrentHashMap[String, (SimpleFeature, Long)]
147 
148     /**
149       * Note: this method is not thread-safe. The `state` and can fail to replace the correct values
150       * if the same feature is updated simultaneously from two different threads
151       *
152       * In our usage, this isn't a problem, as a given feature ID is always operated on by a single thread
153       * due to kafka consumer partitioning
154       */
155     override def put(feature: SimpleFeature): Unit = {
156       val tuple = (feature, FeatureStateFactory.time(time, feature))
157       val old = state.put(feature.getID, tuple)
158       if (old != null && old._2 > tuple._2) {
159         state.replace(feature.getID, tuple, old)
160       }
161     }
162 
163     override def remove(id: String): Unit = state.remove(id)
164 
165     override def clear(): Unit = state.clear()
166 
167     override def close(): Unit = {}
168 
169     override def size(): Int = state.size()
170 
171     override def size(filter: Filter): Int = query(filter).length
172 
173     override def query(id: String): Option[SimpleFeature] = Option(state.get(id)).map(_._1)
174 
175     override def query(filter: Filter): Iterator[SimpleFeature] = {
176       import scala.collection.JavaConverters._
177       val features = state.asScala.valuesIterator.map(_._1)
178       if (filter == Filter.INCLUDE) { features } else {
179         features.filter(filter.evaluate)
180       }
181     }
182 
183     override def views: Seq[KafkaFeatureCacheView] = Seq.empty
184   }
