23 trait KafkaFeatureCacheView extends KafkaFeatureCache {
24   def sft: SimpleFeatureType
25   override val views: Seq[KafkaFeatureCacheView] = Seq.empty
26 }
27 
28 object KafkaFeatureCacheView {
29 
30   /**
31    * Create a new layer view
32    *
33    * @param view view config
34    * @param index spatial index to use
35    * @return
36    */
37   def apply(view: LayerView, index: SimpleFeatureSpatialIndex): KafkaFeatureCacheView = {
38     (view.filter, view.transform) match {
39       case (Some(f), None)    => new KafkaFeatureCacheFilterView(view.viewSft, index, f)
40       case (None, Some(t))    => new KafkaFeatureCacheTransformView(view.viewSft, index, t.toArray)
41       case (Some(f), Some(t)) => new KafkaFeatureCacheFilterTransformView(view.viewSft, index, f, t.toArray)
42       case _ => throw new IllegalArgumentException("LayerView must define at least one of filter or transform")
43     }
44   }
45 
46   /**
47    * Create an empty view
48    *
49    * @param sft view feature type
50    * @return
51    */
52   def empty(sft: SimpleFeatureType): KafkaFeatureCacheView = new KafkaFeatureCacheEmptyView(sft)
53 
54   /**
55    * Empty view class
56    *
57    * @param sft view feature type
58    */
59   private class KafkaFeatureCacheEmptyView(val sft: SimpleFeatureType)
60     extends EmptyFeatureCache(Seq.empty) with KafkaFeatureCacheView
61 
62   /**
63    * View that filters input records based on a CQL predicate
64    *
65    * @param sft view feature type
66    * @param spatialIndex spatial index
67    * @param filter filter
68    */
69   private class KafkaFeatureCacheFilterView(
70       sft: SimpleFeatureType,
71       spatialIndex: SimpleFeatureSpatialIndex,
72       filter: Filter
73     ) extends BaseFeatureCacheView(sft, spatialIndex) {
74     override def put(feature: SimpleFeature): Unit = {
75       if (filter.evaluate(feature)) {
76         super.put(new WrappedSimpleFeature(sft, feature))
77       }
78     }
79   }
80 
81   /**
82    * Simple re-typing of a simple feature, with all attributes the same
83    *
84    * @param sft simple feature type
85    * @param wrapped wrapped feature
86    */
87   private class WrappedSimpleFeature(sft: SimpleFeatureType, wrapped: SimpleFeature)
88       extends AbstractImmutableSimpleFeature(sft) {
89     this.id = wrapped.getID // set id in constructor
90     override def getAttribute(i: Int): AnyRef = wrapped.getAttribute(i)
91     override def getUserData: java.util.Map[AnyRef, AnyRef] = wrapped.getUserData
92   }
93 
94   /**
95    * View that transforms input records to a new feature type
96    *
97    * @param sft view feature type
98    * @param spatialIndex spatial index
99    * @param transforms transform definitions
100    */
101   private class KafkaFeatureCacheTransformView(
102       sft: SimpleFeatureType,
103       spatialIndex: SimpleFeatureSpatialIndex,
104       transforms: Array[Transform]
105     ) extends BaseFeatureCacheView(sft, spatialIndex) {
106     override def put(feature: SimpleFeature): Unit = super.put(new TransformSimpleFeature(sft, transforms, feature))
107   }
108 
109   /**
110    * View that both filters and transforms input records
111    *
112    * @param sft view feature type
113    * @param spatialIndex spatial index
114    * @param filter filter
115    * @param transforms transform definitions
116    */
117   private class KafkaFeatureCacheFilterTransformView(
118       sft: SimpleFeatureType,
119       spatialIndex: SimpleFeatureSpatialIndex,
120       filter: Filter,
121       transforms: Array[Transform]
122     ) extends BaseFeatureCacheView(sft, spatialIndex) {
123     override def put(feature: SimpleFeature): Unit = {
124       if (filter.evaluate(feature)) {
125         super.put(new TransformSimpleFeature(sft, transforms, feature))
126       }
127     }
128   }
129 
130   /**
131    * Base view implementation
132    *
133    * @param sft view feature type
134    * @param spatialIndex spatial index
135    */
136   private abstract class BaseFeatureCacheView(val sft: SimpleFeatureType, spatialIndex: SimpleFeatureSpatialIndex)
137     extends KafkaFeatureCacheView {
138 
139     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
140 
141     private val geomIndex = sft.getGeomIndex
142     private val idMap = new ConcurrentHashMap[String, SimpleFeature]()
143 
144     override def put(feature: SimpleFeature): Unit = {
145       idMap.put(feature.getID, feature)
146       spatialIndex.insert(feature.getAttribute(geomIndex).asInstanceOf[Geometry], feature.getID, feature)
147     }
148 
149     override def remove(id: String): Unit = {
150       val feature = idMap.remove(id)
151       if (feature != null) {
152         spatialIndex.remove(feature.getAttribute(geomIndex).asInstanceOf[Geometry], id)
153       }
154     }
155 
156     override def clear(): Unit = {
157       idMap.clear()
158       spatialIndex.clear()
159     }
160 
161     override def size(): Int = idMap.size()
162 
163     override def size(filter: Filter): Int =
164       if (filter == Filter.INCLUDE) { idMap.size() } else { query(filter).length }
165 
166     override def query(id: String): Option[SimpleFeature] = Option(idMap.get(id))
167 
168     override def query(filter: Filter): Iterator[SimpleFeature] = spatialIndex.query(filter)
169 
170     override def close(): Unit = {}
171   }
172 }
