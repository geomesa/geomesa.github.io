22 trait KafkaFeatureCacheView extends KafkaFeatureCache {
23   def sft: SimpleFeatureType
24   override val views: Seq[KafkaFeatureCacheView] = Seq.empty
25 }
26 
27 object KafkaFeatureCacheView {
28 
29   /**
30    * Create a new layer view
31    *
32    * @param view view config
33    * @param index spatial index to use
34    * @return
35    */
36   def apply(view: LayerView, index: SpatialIndexSupport): KafkaFeatureCacheView = {
37     (view.filter, view.transform) match {
38       case (Some(f), None)    => new KafkaFeatureCacheFilterView(view.viewSft, index, f)
39       case (None, Some(t))    => new KafkaFeatureCacheTransformView(view.viewSft, index, t.toArray)
40       case (Some(f), Some(t)) => new KafkaFeatureCacheFilterTransformView(view.viewSft, index, f, t.toArray)
41       case _ => throw new IllegalArgumentException("LayerView must define at least one of filter or transform")
42     }
43   }
44 
45   /**
46    * Create an empty view
47    *
48    * @param sft view feature type
49    * @return
50    */
51   def empty(sft: SimpleFeatureType): KafkaFeatureCacheView = new KafkaFeatureCacheEmptyView(sft)
52 
53   /**
54    * Empty view class
55    *
56    * @param sft view feature type
57    */
58   class KafkaFeatureCacheEmptyView(val sft: SimpleFeatureType)
59     extends EmptyFeatureCache(Seq.empty) with KafkaFeatureCacheView
60 
61   /**
62    * View that filters input records based on a CQL predicate
63    *
64    * @param sft view feature type
65    * @param support spatial index
66    * @param filter filter
67    */
68   class KafkaFeatureCacheFilterView(
69       sft: SimpleFeatureType,
70       support: SpatialIndexSupport,
71       filter: Filter
72     ) extends BaseFeatureCacheView(sft, support) {
73     override def put(feature: SimpleFeature): Unit = {
74       if (filter.evaluate(feature)) {
75         super.put(ScalaSimpleFeature.wrap(sft, feature))
76       }
77     }
78   }
79 
80   /**
81    * View that transforms input records to a new feature type
82    *
83    * @param sft view feature type
84    * @param support spatial index
85    * @param transforms transform definitions
86    */
87   class KafkaFeatureCacheTransformView(
88       sft: SimpleFeatureType,
89       support: SpatialIndexSupport,
90       transforms: Array[Transform]
91     ) extends BaseFeatureCacheView(sft, support) {
92     override def put(feature: SimpleFeature): Unit = super.put(new TransformSimpleFeature(sft, transforms, feature))
93   }
94 
95   /**
96    * View that both filters and transforms input records
97    *
98    * @param sft view feature type
99    * @param support spatial index
100    * @param filter filter
101    * @param transforms transform definitions
102    */
103   class KafkaFeatureCacheFilterTransformView(
104       sft: SimpleFeatureType,
105       support: SpatialIndexSupport,
106       filter: Filter,
107       transforms: Array[Transform]
108     ) extends BaseFeatureCacheView(sft, support) {
109     override def put(feature: SimpleFeature): Unit = {
110       if (filter.evaluate(feature)) {
111         super.put(new TransformSimpleFeature(sft, transforms, feature))
112       }
113     }
114   }
115 
116   /**
117    * Base view implementation
118    *
119    * @param sft view feature type
120    * @param support spatial index
121    */
122   abstract class BaseFeatureCacheView(val sft: SimpleFeatureType, support: SpatialIndexSupport)
123     extends KafkaFeatureCacheView {
124 
125     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
126 
127     private val geomIndex = sft.getGeomIndex
128     private val idMap = new ConcurrentHashMap[String, SimpleFeature]()
129 
130     override def put(feature: SimpleFeature): Unit = {
131       idMap.put(feature.getID, feature)
132       support.index.insert(feature.getAttribute(geomIndex).asInstanceOf[Geometry], feature.getID, feature)
133     }
134 
135     override def remove(id: String): Unit = {
136       val feature = idMap.remove(id)
137       if (feature != null) {
138         support.index.remove(feature.getAttribute(geomIndex).asInstanceOf[Geometry], id)
139       }
140     }
141 
142     override def clear(): Unit = {
143       idMap.clear()
144       support.index.clear()
145     }
146 
147     override def size(): Int = idMap.size()
148 
149     override def size(filter: Filter): Int =
150       if (filter == Filter.INCLUDE) { idMap.size() } else { query(filter).length }
151 
152     override def query(id: String): Option[SimpleFeature] = Option(idMap.get(id))
153 
154     override def query(filter: Filter): Iterator[SimpleFeature] = support.query(filter)
155 
156     override def close(): Unit = {}
157   }
158 }
