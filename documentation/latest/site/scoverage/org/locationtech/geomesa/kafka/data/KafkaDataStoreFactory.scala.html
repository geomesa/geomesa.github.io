292   /**
293    * Parse the dynamic expiry param value into a seq of pairs
294    *
295    * @param params data store params
296    * @return
297    */
298   private[data] def parseDynamicExpiry(params: java.util.Map[String, _]): Seq[(String, Duration)] = {
299     lazy val key = s"Invalid property for parameter '${KafkaDataStoreParams.DynamicCacheExpiry.key}'"
300     val expiry = KafkaDataStoreParams.DynamicCacheExpiry.lookupOpt(params).toSeq.flatMap { value =>
301       ConfigFactory.parseString(value).resolve().root().unwrapped().asScala.toSeq.map {
302         case (filter, exp: String) =>
303           // validate the filter, but leave it as a string so we can optimize it based on the sft later
304           try { ECQL.toFilter(filter) } catch {
305             case NonFatal(e) => throw new IOException(s"$key, expected a CQL filter but got: $filter", e)
306           }
307           val duration = try { Duration(exp) } catch {
308             case NonFatal(e) => throw new IOException(s"$key, expected a duration for key '$filter' but got: $exp", e)
309           }
310           filter -> duration
311 
312         case (filter, exp) =>
313           throw new IOException(s"$key, expected a JSON string for key '$filter' but got: $exp")
314       }
315     }
316     if (expiry.dropRight(1).exists(_._1.equalsIgnoreCase("INCLUDE"))) {
317       throw new IOException(s"$key, defined a filter after Filter.INCLUDE (which would never be invoked)")
318     }
319     expiry
320   }

322   /**
323    * Parse the typesafe config for a layer view. Views take the form:
324    *
325    * {
326    *   foo-sft = [
327    *     {
328    *       type-name = foo-sft-enhanced
329    *       filter = "foo = bar"
330    *       transform = [ "foo", "bar", "baz", "blu" ]
331    *     },
332    *     {
333    *       type-name = foo-sft-reduced
334    *       filter = "foo = baz"
335    *       transform = [ "foo", "bar", "baz" ]
336    *     }
337    *   ]
338    * }
339    *
340    * @param params params
341    * @return
342    */
343   private[kafka] def parseLayerViewConfig(params: java.util.Map[String, _]): Map[String, Seq[LayerViewConfig]] = {
344     def asConfigObject(o: AnyRef) = o match {
345       case c: ConfigObject => c
346       case _ => throw new IllegalArgumentException(s"Invalid layer view, expected a config object but got: $o")
347     }
348 
349     KafkaDataStoreParams.LayerViews.lookupOpt(params) match {
350       case None => Map.empty[String, Seq[LayerViewConfig]]
351       case Some(conf) =>
352         val config = ConfigFactory.parseString(conf).resolve()
353         val entries = config.entrySet().asScala.map { e =>
354           val views = e.getValue match {
355             case c: ConfigList => c.asScala.map(asConfigObject)
356             case c => Seq(asConfigObject(c))
357           }
358           e.getKey -> views.map { c =>
359             ConfigSource.fromConfig(c.toConfig).loadOrThrow[LayerViewConfig](LayerViewClassTag, LayerViewReader)
360           }
361         }
362         val configs = entries.map(f => (f._1, f._2.toSeq))
363         val typeNames = configs.toSeq.flatMap(_._2.map(_.typeName))
364         if (typeNames != typeNames.distinct) {
365           throw new IllegalArgumentException(
366             s"Detected duplicate type name in layer view config: ${config.root().render(ConfigRenderOptions.concise)}")
367         }
368         configs.toMap
369     }
370   }

378   private def readLayerViewConfig(cur: ConfigCursor): Either[ConfigReaderFailures, LayerViewConfig] = {
379     val config = for {
380       obj       <- cur.asObjectCursor.right
381       typeName  <- obj.atKey("type-name").right.flatMap(_.asString).right
382       filter    <- readFilter(obj.atKeyOrUndefined("filter")).right
383       transform <- readTransform(obj.atKeyOrUndefined("transform")).right
384     } yield {
385       LayerViewConfig(typeName, filter, transform)
386     }
387     config.right.flatMap { c =>
388       if (c.filter.isEmpty && c.transform.isEmpty) {
389         val err = "LayerViews must define at least one of 'filter' or 'transform'"
390         cur.failed(new FailureReason { override def description: String = err })
391       } else {
392         Right(c)
393       }
394     }
395   }

397   private def readFilter(cur: ConfigCursor): Either[ConfigReaderFailures, Option[Filter]] = {
398     if (cur.isUndefined) { Right(None) } else {
399       cur.asString.right.flatMap { ecql =>
400         try { Right(Some(ECQL.toFilter(ecql)).filter(_ != Filter.INCLUDE)) } catch {
401           case NonFatal(e) => cur.failed(CannotConvert(ecql, "Filter", e.toString))
402         }
403       }
404     }
405   }

407   private def readTransform(cur: ConfigCursor): Either[ConfigReaderFailures, Option[Seq[String]]] = {
408     if (cur.isUndefined) { Right(None) } else {
409       val transforms = cur.asList.right.flatMap { list =>
410         list.foldLeft[Either[ConfigReaderFailures, Seq[String]]](Right(Seq.empty)) { case (res, elem) =>
411           res.right.flatMap(r => elem.asString.right.map(r :+ _))
412         }
413       }
414       transforms.right.map(t => if (t.isEmpty) { None } else { Some(t) })
415     }
416   }

455   private def parseDeprecatedMetrics(conf: Config): Option[MetricsConfig] = {
456     val reporters = if (conf.hasPath("reporters")) { conf.getConfigList("reporters").asScala } else { Seq(conf) }
457     val supported = reporters.flatMap { reporter =>
458       if (reporter.hasPath("type")) {
459         Option(reporter.getString("type")).flatMap {
460           case t if t.equalsIgnoreCase("prometheus") =>
461             // note: 'port' is the same config key used by the old geomesa-metrics
462             Some(MetricsConfig(PrometheusFactory, Some(reporter.withOnlyPath("port"))))
463 
464           case t if t.equalsIgnoreCase("prometheus-pushgateway") =>
465             val config =
466               ConfigFactory.empty()
467                 .withValue("push-gateway.host", reporter.getValue("gateway"))
468                 .withValue("push-gateway.job", reporter.getValue("job-name"))
469             Some(MetricsConfig(PrometheusFactory, Some(config)))
470 
471           case t if t.equalsIgnoreCase("cloudwatch") =>
472             // note: 'namespace' is the same config key used by the old geomesa-metrics
473             Some(MetricsConfig(CloudwatchFactory, Some(reporter.withOnlyPath("namespace"))))
474 
475           case t =>
476             logger.warn(s"Ignoring unsupported metrics reporter of type $t")
477             None
478         }
479       } else {
480         logger.warn("Ignoring metrics reporter without 'type' field")
481         None
482       }
483     }
484     if (supported.lengthCompare(1) > 0) {
485       logger.warn(
486         s"Only one metric registry is supported at once, using ${supported.head.registry.name} " +
487           s"and ignoring ${supported.tail.map(_.registry.name).mkString(", ")}")
488     }
489     supported.headOption
490   }
