288   /**
289    * Parse the dynamic expiry param value into a seq of pairs
290    *
291    * @param params data store params
292    * @return
293    */
294   private[data] def parseDynamicExpiry(params: java.util.Map[String, _]): Seq[(String, Duration)] = {
295     lazy val key = s"Invalid property for parameter '${KafkaDataStoreParams.DynamicCacheExpiry.key}'"
296     val expiry = KafkaDataStoreParams.DynamicCacheExpiry.lookupOpt(params).toSeq.flatMap { value =>
297       ConfigFactory.parseString(value).resolve().root().unwrapped().asScala.toSeq.map {
298         case (filter, exp: String) =>
299           // validate the filter, but leave it as a string so we can optimize it based on the sft later
300           try { ECQL.toFilter(filter) } catch {
301             case NonFatal(e) => throw new IOException(s"$key, expected a CQL filter but got: $filter", e)
302           }
303           val duration = try { Duration(exp) } catch {
304             case NonFatal(e) => throw new IOException(s"$key, expected a duration for key '$filter' but got: $exp", e)
305           }
306           filter -> duration
307 
308         case (filter, exp) =>
309           throw new IOException(s"$key, expected a JSON string for key '$filter' but got: $exp")
310       }
311     }
312     if (expiry.dropRight(1).exists(_._1.equalsIgnoreCase("INCLUDE"))) {
313       throw new IOException(s"$key, defined a filter after Filter.INCLUDE (which would never be invoked)")
314     }
315     expiry
316   }

318   /**
319    * Parse the typesafe config for a layer view. Views take the form:
320    *
321    * {
322    *   foo-sft = [
323    *     {
324    *       type-name = foo-sft-enhanced
325    *       filter = "foo = bar"
326    *       transform = [ "foo", "bar", "baz", "blu" ]
327    *     },
328    *     {
329    *       type-name = foo-sft-reduced
330    *       filter = "foo = baz"
331    *       transform = [ "foo", "bar", "baz" ]
332    *     }
333    *   ]
334    * }
335    *
336    * @param params params
337    * @return
338    */
339   private[kafka] def parseLayerViewConfig(params: java.util.Map[String, _]): Map[String, Seq[LayerViewConfig]] = {
340     def asConfigObject(o: AnyRef) = o match {
341       case c: ConfigObject => c
342       case _ => throw new IllegalArgumentException(s"Invalid layer view, expected a config object but got: $o")
343     }
344 
345     KafkaDataStoreParams.LayerViews.lookupOpt(params) match {
346       case None => Map.empty[String, Seq[LayerViewConfig]]
347       case Some(conf) =>
348         val config = ConfigFactory.parseString(conf).resolve()
349         val entries = config.entrySet().asScala.map { e =>
350           val views = e.getValue match {
351             case c: ConfigList => c.asScala.map(asConfigObject)
352             case c => Seq(asConfigObject(c))
353           }
354           e.getKey -> views.map { c =>
355             ConfigSource.fromConfig(c.toConfig).loadOrThrow[LayerViewConfig](LayerViewClassTag, LayerViewReader)
356           }
357         }
358         val configs = entries.map(f => (f._1, f._2.toSeq))
359         val typeNames = configs.toSeq.flatMap(_._2.map(_.typeName))
360         if (typeNames != typeNames.distinct) {
361           throw new IllegalArgumentException(
362             s"Detected duplicate type name in layer view config: ${config.root().render(ConfigRenderOptions.concise)}")
363         }
364         configs.toMap
365     }
366   }

369    * Parse a single layer view config
370    *
371    * @param cur cursor
372    * @return
373    */
374   private def readLayerViewConfig(cur: ConfigCursor): Either[ConfigReaderFailures, LayerViewConfig] = {
375     val config = for {
376       obj       <- cur.asObjectCursor.right
377       typeName  <- obj.atKey("type-name").right.flatMap(_.asString).right
378       filter    <- readFilter(obj.atKeyOrUndefined("filter")).right
379       transform <- readTransform(obj.atKeyOrUndefined("transform")).right
380     } yield {
381       LayerViewConfig(typeName, filter, transform)
382     }
383     config.right.flatMap { c =>
384       if (c.filter.isEmpty && c.transform.isEmpty) {
385         val err = "LayerViews must define at least one of 'filter' or 'transform'"
386         cur.failed(new FailureReason { override def description: String = err })
387       } else {
388         Right(c)
389       }
390     }
391   }

393   private def readFilter(cur: ConfigCursor): Either[ConfigReaderFailures, Option[Filter]] = {
394     if (cur.isUndefined) { Right(None) } else {
395       cur.asString.right.flatMap { ecql =>
396         try { Right(Some(ECQL.toFilter(ecql)).filter(_ != Filter.INCLUDE)) } catch {
397           case NonFatal(e) => cur.failed(CannotConvert(ecql, "Filter", e.toString))
398         }
399       }
400     }
401   }
402 
403   private def readTransform(cur: ConfigCursor): Either[ConfigReaderFailures, Option[Seq[String]]] = {
404     if (cur.isUndefined) { Right(None) } else {
405       val transforms = cur.asList.right.flatMap { list =>
406         list.foldLeft[Either[ConfigReaderFailures, Seq[String]]](Right(Seq.empty)) { case (res, elem) =>
407           res.right.flatMap(r => elem.asString.right.map(r :+ _))
408         }
409       }
410       transforms.right.map(t => if (t.isEmpty) { None } else { Some(t) })
411     }
412   }
