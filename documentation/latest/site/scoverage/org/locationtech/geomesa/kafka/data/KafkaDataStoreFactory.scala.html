286   /**
287    * Parse the dynamic expiry param value into a seq of pairs
288    *
289    * @param params data store params
290    * @return
291    */
292   private[data] def parseDynamicExpiry(params: java.util.Map[String, _]): Seq[(String, Duration)] = {
293     lazy val key = s"Invalid property for parameter '${KafkaDataStoreParams.DynamicCacheExpiry.key}'"
294     val expiry = KafkaDataStoreParams.DynamicCacheExpiry.lookupOpt(params).toSeq.flatMap { value =>
295       ConfigFactory.parseString(value).resolve().root().unwrapped().asScala.toSeq.map {
296         case (filter, exp: String) =>
297           // validate the filter, but leave it as a string so we can optimize it based on the sft later
298           try { ECQL.toFilter(filter) } catch {
299             case NonFatal(e) => throw new IOException(s"$key, expected a CQL filter but got: $filter", e)
300           }
301           val duration = try { Duration(exp) } catch {
302             case NonFatal(e) => throw new IOException(s"$key, expected a duration for key '$filter' but got: $exp", e)
303           }
304           filter -> duration
305 
306         case (filter, exp) =>
307           throw new IOException(s"$key, expected a JSON string for key '$filter' but got: $exp")
308       }
309     }
310     if (expiry.dropRight(1).exists(_._1.equalsIgnoreCase("INCLUDE"))) {
311       throw new IOException(s"$key, defined a filter after Filter.INCLUDE (which would never be invoked)")
312     }
313     expiry
314   }

316   /**
317    * Parse the typesafe config for a layer view. Views take the form:
318    *
319    * {
320    *   foo-sft = [
321    *     {
322    *       type-name = foo-sft-enhanced
323    *       filter = "foo = bar"
324    *       transform = [ "foo", "bar", "baz", "blu" ]
325    *     },
326    *     {
327    *       type-name = foo-sft-reduced
328    *       filter = "foo = baz"
329    *       transform = [ "foo", "bar", "baz" ]
330    *     }
331    *   ]
332    * }
333    *
334    * @param params params
335    * @return
336    */
337   private[kafka] def parseLayerViewConfig(params: java.util.Map[String, _]): Map[String, Seq[LayerViewConfig]] = {
338     def asConfigObject(o: AnyRef) = o match {
339       case c: ConfigObject => c
340       case _ => throw new IllegalArgumentException(s"Invalid layer view, expected a config object but got: $o")
341     }
342 
343     KafkaDataStoreParams.LayerViews.lookupOpt(params) match {
344       case None => Map.empty[String, Seq[LayerViewConfig]]
345       case Some(conf) =>
346         val config = ConfigFactory.parseString(conf).resolve()
347         val entries = config.entrySet().asScala.map { e =>
348           val views = e.getValue match {
349             case c: ConfigList => c.asScala.map(asConfigObject)
350             case c => Seq(asConfigObject(c))
351           }
352           e.getKey -> views.map { c =>
353             ConfigSource.fromConfig(c.toConfig).loadOrThrow[LayerViewConfig](LayerViewClassTag, LayerViewReader)
354           }
355         }
356         val configs = entries.map(f => (f._1, f._2.toSeq))
357         val typeNames = configs.toSeq.flatMap(_._2.map(_.typeName))
358         if (typeNames != typeNames.distinct) {
359           throw new IllegalArgumentException(
360             s"Detected duplicate type name in layer view config: ${config.root().render(ConfigRenderOptions.concise)}")
361         }
362         configs.toMap
363     }
364   }

367    * Parse a single layer view config
368    *
369    * @param cur cursor
370    * @return
371    */
372   private def readLayerViewConfig(cur: ConfigCursor): Either[ConfigReaderFailures, LayerViewConfig] = {
373     val config = for {
374       obj       <- cur.asObjectCursor.right
375       typeName  <- obj.atKey("type-name").right.flatMap(_.asString).right
376       filter    <- readFilter(obj.atKeyOrUndefined("filter")).right
377       transform <- readTransform(obj.atKeyOrUndefined("transform")).right
378     } yield {
379       LayerViewConfig(typeName, filter, transform)
380     }
381     config.right.flatMap { c =>
382       if (c.filter.isEmpty && c.transform.isEmpty) {
383         val err = "LayerViews must define at least one of 'filter' or 'transform'"
384         cur.failed(new FailureReason { override def description: String = err })
385       } else {
386         Right(c)
387       }
388     }
389   }

391   private def readFilter(cur: ConfigCursor): Either[ConfigReaderFailures, Option[Filter]] = {
392     if (cur.isUndefined) { Right(None) } else {
393       cur.asString.right.flatMap { ecql =>
394         try { Right(Some(ECQL.toFilter(ecql)).filter(_ != Filter.INCLUDE)) } catch {
395           case NonFatal(e) => cur.failed(CannotConvert(ecql, "Filter", e.toString))
396         }
397       }
398     }
399   }
400 
401   private def readTransform(cur: ConfigCursor): Either[ConfigReaderFailures, Option[Seq[String]]] = {
402     if (cur.isUndefined) { Right(None) } else {
403       val transforms = cur.asList.right.flatMap { list =>
404         list.foldLeft[Either[ConfigReaderFailures, Seq[String]]](Right(Seq.empty)) { case (res, elem) =>
405           res.right.flatMap(r => elem.asString.right.map(r :+ _))
406         }
407       }
408       transforms.right.map(t => if (t.isEmpty) { None } else { Some(t) })
409     }
410   }
