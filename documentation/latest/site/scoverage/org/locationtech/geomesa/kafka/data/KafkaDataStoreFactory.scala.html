308   /**
309    * Parse the dynamic expiry param value into a seq of pairs
310    *
311    * @param params data store params
312    * @return
313    */
314   private[data] def parseDynamicExpiry(params: java.util.Map[String, _]): Seq[(String, Duration)] = {
315     lazy val key = s"Invalid property for parameter '${KafkaDataStoreParams.DynamicCacheExpiry.key}'"
316     val expiry = KafkaDataStoreParams.DynamicCacheExpiry.lookupOpt(params).toSeq.flatMap { value =>
317       ConfigFactory.parseString(value).resolve().root().unwrapped().asScala.toSeq.map {
318         case (filter, exp: String) =>
319           // validate the filter, but leave it as a string so we can optimize it based on the sft later
320           try { ECQL.toFilter(filter) } catch {
321             case NonFatal(e) => throw new IOException(s"$key, expected a CQL filter but got: $filter", e)
322           }
323           val duration = try { Duration(exp) } catch {
324             case NonFatal(e) => throw new IOException(s"$key, expected a duration for key '$filter' but got: $exp", e)
325           }
326           filter -> duration
327 
328         case (filter, exp) =>
329           throw new IOException(s"$key, expected a JSON string for key '$filter' but got: $exp")
330       }
331     }
332     if (expiry.dropRight(1).exists(_._1.equalsIgnoreCase("INCLUDE"))) {
333       throw new IOException(s"$key, defined a filter after Filter.INCLUDE (which would never be invoked)")
334     }
335     expiry
336   }

338   /**
339    * Parse the typesafe config for a layer view. Views take the form:
340    *
341    * {
342    *   foo-sft = [
343    *     {
344    *       type-name = foo-sft-enhanced
345    *       filter = "foo = bar"
346    *       transform = [ "foo", "bar", "baz", "blu" ]
347    *     },
348    *     {
349    *       type-name = foo-sft-reduced
350    *       filter = "foo = baz"
351    *       transform = [ "foo", "bar", "baz" ]
352    *     }
353    *   ]
354    * }
355    *
356    * @param params params
357    * @return
358    */
359   private[kafka] def parseLayerViewConfig(params: java.util.Map[String, _]): Map[String, Seq[LayerViewConfig]] = {
360     def asConfigObject(o: AnyRef) = o match {
361       case c: ConfigObject => c
362       case _ => throw new IllegalArgumentException(s"Invalid layer view, expected a config object but got: $o")
363     }
364 
365     KafkaDataStoreParams.LayerViews.lookupOpt(params) match {
366       case None => Map.empty[String, Seq[LayerViewConfig]]
367       case Some(conf) =>
368         val config = ConfigFactory.parseString(conf).resolve()
369         val entries = config.entrySet().asScala.map { e =>
370           val views = e.getValue match {
371             case c: ConfigList => c.asScala.map(asConfigObject)
372             case c => Seq(asConfigObject(c))
373           }
374           e.getKey -> views.map { c =>
375             ConfigSource.fromConfig(c.toConfig).loadOrThrow[LayerViewConfig](LayerViewClassTag, LayerViewReader)
376           }
377         }
378         val configs = entries.map(f => (f._1, f._2.toSeq))
379         val typeNames = configs.toSeq.flatMap(_._2.map(_.typeName))
380         if (typeNames != typeNames.distinct) {
381           throw new IllegalArgumentException(
382             s"Detected duplicate type name in layer view config: ${config.root().render(ConfigRenderOptions.concise)}")
383         }
384         configs.toMap
385     }
386   }

390    *

394   private def readLayerViewConfig(cur: ConfigCursor): Either[ConfigReaderFailures, LayerViewConfig] = {
395     val config = for {
396       obj       <- cur.asObjectCursor.right
397       typeName  <- obj.atKey("type-name").right.flatMap(_.asString).right
398       filter    <- readFilter(obj.atKeyOrUndefined("filter")).right
399       transform <- readTransform(obj.atKeyOrUndefined("transform")).right
400     } yield {
401       LayerViewConfig(typeName, filter, transform)
402     }
403     config.right.flatMap { c =>
404       if (c.filter.isEmpty && c.transform.isEmpty) {
405         val err = "LayerViews must define at least one of 'filter' or 'transform'"
406         cur.failed(new FailureReason { override def description: String = err })
407       } else {
408         Right(c)
409       }
410     }
411   }

413   private def readFilter(cur: ConfigCursor): Either[ConfigReaderFailures, Option[Filter]] = {
414     if (cur.isUndefined) { Right(None) } else {
415       cur.asString.right.flatMap { ecql =>
416         try { Right(Some(ECQL.toFilter(ecql)).filter(_ != Filter.INCLUDE)) } catch {
417           case NonFatal(e) => cur.failed(CannotConvert(ecql, "Filter", e.toString))
418         }
419       }
420     }
421   }
422 
423   private def readTransform(cur: ConfigCursor): Either[ConfigReaderFailures, Option[Seq[String]]] = {
424     if (cur.isUndefined) { Right(None) } else {
425       val transforms = cur.asList.right.flatMap { list =>
426         list.foldLeft[Either[ConfigReaderFailures, Seq[String]]](Right(Seq.empty)) { case (res, elem) =>
427           res.right.flatMap(r => elem.asString.right.map(r :+ _))
428         }
429       }
430       transforms.right.map(t => if (t.isEmpty) { None } else { Some(t) })
431     }
432   }
