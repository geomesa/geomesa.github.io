105   /**
106    * Ensure that the topic has a cleanup policy of 'compact', so that feature types aren't aged off
107    *
108    * @param adminClient admin client
109    */
110   private def ensureCompactionPolicy(adminClient: AdminClient): Unit = {
111     val catalogResource = new ConfigResource(ConfigResource.Type.TOPIC, config.catalog)
112     val catalogConfigs = adminClient.describeConfigs(Collections.singleton(catalogResource)).all().get().values()
113     val isCompact = catalogConfigs.asScala.exists { config =>
114       config.get(CleanupPolicyConfig) != null && config.get(CleanupPolicyConfig).value() == CompactCleanupPolicy
115     }
116     if (!isCompact) {
117       val catalogConfigEntry = new ConfigEntry(CleanupPolicyConfig, CompactCleanupPolicy)
118       val alterOps = Collections.singleton(new AlterConfigOp(catalogConfigEntry, AlterConfigOp.OpType.SET))
119       adminClient.incrementalAlterConfigs(Collections.singletonMap(catalogResource, alterOps)).all().get()
120     }
121   }

123   private def adminClientOp[V](fn: AdminClient => V): V = {
124     val props = new Properties()
125     props.put(AdminClientConfig.BOOTSTRAP_SERVERS_CONFIG, config.brokers)
126     config.consumers.properties.foreach { case (k, v) => props.put(k, v) }
127     config.producers.properties.foreach { case (k, v) => props.put(k, v) }

131   /**
132    * Models the topic as a map of key-value pairs
133    */
134   private class TopicMap extends Runnable with Closeable {
135 
136     private val groupId = UUID.randomUUID().toString
137     private val poll = Duration.of(100, ChronoUnit.MILLIS)
138 
139     private val state = new ConcurrentHashMap[KeyBytes, Array[Byte]]()
140     private val complete = new CountDownLatch(1)
141     private val closed = new AtomicBoolean(false)
142 
143     private val consumer =
144       KafkaDataStore.consumer(config.brokers,
145         config.consumers.properties ++ Map(GROUP_ID_CONFIG -> groupId, AUTO_OFFSET_RESET_CONFIG -> "earliest"))
146 
147     private var future: Future[_] = _
148 
149     KafkaConsumerVersions.subscribe(consumer, config.catalog)
150     doInitialLoad()
151 
152     override def run(): Unit = {
153       try {
154         var interrupted = Thread.currentThread().isInterrupted
155         while (!interrupted) {
156           try {
157             val result = KafkaConsumerVersions.poll(consumer, poll)
158             if (!result.isEmpty) {
159               val records = result.iterator()
160               while (records.hasNext) {
161                 val r = records.next()
162                 val v = r.value()
163                 if (v == null) {
164                   state.remove(KeyBytes(r.key()))
165                 } else {
166                   state.put(KeyBytes(r.key()), v)
167                 }
168               }
169               consumer.commitAsync()
170             }
171           } catch {
172             case _: WakeupException | _: InterruptException | _: InterruptedException => interrupted = true
173             case NonFatal(e) =>
174               logger.warn(s"Consumer [$groupId] error receiving message from topic:", e)
175               Thread.sleep(1000)
176           }
177         }
178       } finally {
179         complete.countDown()
180       }
181     }
182 
183     private def doInitialLoad(): Unit = {
184       try {
185         val offsets = scala.collection.mutable.Map.empty[Int, Long]
186         // noinspection RedundantCollectionConversion
187         val partitions = consumer.partitionsFor(config.catalog).asScala.map(_.partition).toSeq
188         // note: end offsets are the *next* offset that will be returned, so subtract one to track the last offset
189         // we will actually consume
190         offsets ++=
191             KafkaConsumerVersions.endOffsets(consumer, config.catalog, partitions)
192                 .collect { case (p, o) if o > 0 => (p, o - 1) }
193         while (offsets.nonEmpty) {
194           val result = KafkaConsumerVersions.poll(consumer, poll)
195           if (!result.isEmpty) {
196             val records = result.iterator()
197             while (records.hasNext) {
198               val r = records.next()
199               val v = r.value()
200               if (v == null) {
201                 state.remove(KeyBytes(r.key()))
202               } else {
203                 state.put(KeyBytes(r.key()), v)
204               }
205               if (offsets.get(r.partition()).exists(o => r.offset() >= o)) {
206                 offsets.remove(r.partition())
207               }
208             }
209             consumer.commitAsync()
210           }
211         }
212         lazy val stateStrings =
213           state.asScala.map { case (k, v) => new String(k.bytes, StandardCharsets.UTF_8) -> new String(v, StandardCharsets.UTF_8)}
214         logger.debug(s"Completed initial load of catalog '${config.catalog}': \n  ${stateStrings.mkString("\n  ")}")
215 
216         future = CachedThreadPool.submit(this)
217         sys.addShutdownHook(future.cancel(true)) // prevent consumer from hanging if ds isn't disposed properly
218       } catch {
219         case NonFatal(e) => complete.countDown(); throw e
220       }
221     }
222 
223     def get(key: Array[Byte]): Option[Array[Byte]] = Option(state.get(KeyBytes(key)))
224 
225     def all(): CloseableIterator[(Array[Byte], Array[Byte])] =
226       CloseableIterator(state.asScala.iterator.map { case (k, v) => k.bytes -> v })
227 
228     def prefix(prefix: Array[Byte]): CloseableIterator[(Array[Byte], Array[Byte])] =
229       all().filter { case (k, _) => k.startsWith(prefix) }
230 
231     override def close(): Unit = {
232       try {
233         if (future != null) {
234           future.cancel(true)
235         }
236         complete.await(10, TimeUnit.SECONDS)
237       } finally {
238         // avoid checking consumer assignment if consumer is already closed, which will throw an error
239         if (closed.compareAndSet(false, true)) {
240           cleanupConsumer()
241         }
242       }
243     }
244 
245     private def cleanupConsumer(): Unit = {
246       try {
247         val topics = consumer.assignment()
248         consumer.unsubscribe()
249         if (!topics.isEmpty) {
250           Try(adminClientOp(KafkaAdminVersions.deleteConsumerGroupOffsets(_, groupId, topics))).failed.foreach { e =>
251             logger.warn("Error deleting consumer group offsets:", e)
252           }
253         }
254       } finally {
255         consumer.close()
256       }
257     }
258   }

266     override def hashCode(): Int = java.util.Arrays.hashCode(bytes)
267     override def equals(obj: Any): Boolean = {
268       obj match {
269         case KeyBytes(other) => java.util.Arrays.equals(bytes, other)
270         case _ => false
271       }
272     }
