108   /**
109    * Ensure that the topic has a cleanup policy of 'compact', so that feature types aren't aged off
110    *
111    * @param adminClient admin client
112    */
113   private def ensureCompactionPolicy(adminClient: AdminClient): Unit = {
114     val catalogResource = new ConfigResource(ConfigResource.Type.TOPIC, config.catalog)
115     val catalogConfigs = adminClient.describeConfigs(Collections.singleton(catalogResource)).all().get().values()
116     val isCompact = catalogConfigs.asScala.exists { config =>
117       config.get(CleanupPolicyConfig) != null && config.get(CleanupPolicyConfig).value() == CompactCleanupPolicy
118     }
119     if (!isCompact) {
120       val catalogConfigEntry = new ConfigEntry(CleanupPolicyConfig, CompactCleanupPolicy)
121       val alterOps = Collections.singleton(new AlterConfigOp(catalogConfigEntry, AlterConfigOp.OpType.SET))
122       adminClient.incrementalAlterConfigs(Collections.singletonMap(catalogResource, alterOps)).all().get()
123     }
124   }

126   /**
127    * Models the topic as a map of key-value pairs
128    */
129   private class TopicMap extends Runnable with Closeable {
130 
131     private val groupId = UUID.randomUUID().toString
132     private val poll = Duration.of(100, ChronoUnit.MILLIS)
133 
134     private val state = new ConcurrentHashMap[KeyBytes, Array[Byte]]()
135     private val complete = new CountDownLatch(1)
136     private val closed = new AtomicBoolean(false)
137 
138     private val consumer =
139       KafkaDataStore.consumer(config.brokers,
140         config.consumers.properties ++ Map(GROUP_ID_CONFIG -> groupId, AUTO_OFFSET_RESET_CONFIG -> "earliest"))
141 
142     private var future: Future[_] = _
143 
144     KafkaConsumerVersions.subscribe(consumer, config.catalog)
145     doInitialLoad()
146 
147     override def run(): Unit = {
148       try {
149         var interrupted = Thread.currentThread().isInterrupted
150         while (!interrupted) {
151           try {
152             val result = KafkaConsumerVersions.poll(consumer, poll)
153             if (!result.isEmpty) {
154               val records = result.iterator()
155               while (records.hasNext) {
156                 val r = records.next()
157                 val v = r.value()
158                 if (v == null) {
159                   state.remove(KeyBytes(r.key()))
160                 } else {
161                   state.put(KeyBytes(r.key()), v)
162                 }
163               }
164               consumer.commitAsync()
165             }
166           } catch {
167             case _: WakeupException | _: InterruptException | _: InterruptedException => interrupted = true
168             case NonFatal(e) =>
169               logger.warn(s"Consumer [$groupId] error receiving message from topic:", e)
170               Thread.sleep(1000)
171           }
172         }
173       } finally {
174         complete.countDown()
175       }
176     }
177 
178     private def doInitialLoad(): Unit = {
179       try {
180         val offsets = scala.collection.mutable.Map.empty[Int, Long]
181         // noinspection RedundantCollectionConversion
182         val partitions = consumer.partitionsFor(config.catalog).asScala.map(_.partition).toSeq
183         // note: end offsets are the *next* offset that will be returned, so subtract one to track the last offset
184         // we will actually consume
185         offsets ++=
186             KafkaConsumerVersions.endOffsets(consumer, config.catalog, partitions)
187                 .collect { case (p, o) if o > 0 => (p, o - 1) }
188         while (offsets.nonEmpty) {
189           val result = KafkaConsumerVersions.poll(consumer, poll)
190           if (!result.isEmpty) {
191             val records = result.iterator()
192             while (records.hasNext) {
193               val r = records.next()
194               val v = r.value()
195               if (v == null) {
196                 state.remove(KeyBytes(r.key()))
197               } else {
198                 state.put(KeyBytes(r.key()), v)
199               }
200               if (offsets.get(r.partition()).exists(o => r.offset() >= o)) {
201                 offsets.remove(r.partition())
202               }
203             }
204             consumer.commitAsync()
205           }
206         }
207         lazy val stateStrings =
208           state.asScala.map { case (k, v) => new String(k.bytes, StandardCharsets.UTF_8) -> new String(v, StandardCharsets.UTF_8)}
209         logger.debug(s"Completed initial load of catalog '${config.catalog}': \n  ${stateStrings.mkString("\n  ")}")
210 
211         future = CachedThreadPool.submit(this)
212         sys.addShutdownHook(future.cancel(true)) // prevent consumer from hanging if ds isn't disposed properly
213       } catch {
214         case NonFatal(e) => complete.countDown(); throw e
215       }
216     }
217 
218     def get(key: Array[Byte]): Option[Array[Byte]] = Option(state.get(KeyBytes(key)))
219 
220     def all(): CloseableIterator[(Array[Byte], Array[Byte])] =
221       CloseableIterator(state.asScala.iterator.map { case (k, v) => k.bytes -> v })
222 
223     def prefix(prefix: Array[Byte]): CloseableIterator[(Array[Byte], Array[Byte])] =
224       all().filter { case (k, _) => k.startsWith(prefix) }
225 
226     override def close(): Unit = {
227       try {
228         if (future != null) {
229           future.cancel(true)
230         }
231         complete.await(10, TimeUnit.SECONDS)
232       } finally {
233         // avoid checking consumer assignment if consumer is already closed, which will throw an error
234         if (closed.compareAndSet(false, true)) {
235           cleanupConsumer()
236         }
237       }
238     }
239 
240     private def cleanupConsumer(): Unit = {
241       try {
242         val topics = consumer.assignment()
243         consumer.unsubscribe()
244         if (!topics.isEmpty) {
245           Try(adminClientOp(config)(KafkaAdminVersions.deleteConsumerGroupOffsets(_, groupId, topics))).failed.foreach { e =>
246             logger.warn("Error deleting consumer group offsets:", e)
247           }
248         }
249       } finally {
250         consumer.close()
251       }
252     }
253   }

255   /**
256    * Supports using a byte array as a map key
257    *
258    * @param bytes bytes
259    */
260   private case class KeyBytes(bytes: Array[Byte]) {
261     override def hashCode(): Int = java.util.Arrays.hashCode(bytes)
262     override def equals(obj: Any): Boolean = {
263       obj match {
264         case KeyBytes(other) => java.util.Arrays.equals(bytes, other)
265         case _ => false
266       }
267     }
268   }
