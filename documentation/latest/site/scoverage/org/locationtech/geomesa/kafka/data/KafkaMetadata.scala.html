119   /**
120    * Models the topic as a map of key-value pairs
121    */
122   private class TopicMap extends Runnable with Closeable {
123 
124     private val groupId = UUID.randomUUID().toString
125     private val poll = Duration.of(100, ChronoUnit.MILLIS)
126 
127     private val state = new ConcurrentHashMap[KeyBytes, Array[Byte]]()
128     private val complete = new CountDownLatch(1)
129     private val closed = new AtomicBoolean(false)
130 
131     private val consumer =
132       KafkaDataStore.consumer(config.brokers,
133         config.consumers.properties ++ Map(GROUP_ID_CONFIG -> groupId, AUTO_OFFSET_RESET_CONFIG -> "earliest"))
134 
135     private var future: Future[_] = _
136 
137     KafkaConsumerVersions.subscribe(consumer, config.catalog)
138     doInitialLoad()
139 
140     override def run(): Unit = {
141       try {
142         var interrupted = Thread.currentThread().isInterrupted
143         while (!interrupted) {
144           try {
145             val result = KafkaConsumerVersions.poll(consumer, poll)
146             if (!result.isEmpty) {
147               val records = result.iterator()
148               while (records.hasNext) {
149                 val r = records.next()
150                 val v = r.value()
151                 if (v == null) {
152                   state.remove(KeyBytes(r.key()))
153                 } else {
154                   state.put(KeyBytes(r.key()), v)
155                 }
156               }
157               consumer.commitAsync()
158             }
159           } catch {
160             case _: WakeupException | _: InterruptException | _: InterruptedException => interrupted = true
161             case NonFatal(e) =>
162               logger.warn(s"Consumer [$groupId] error receiving message from topic:", e)
163               Thread.sleep(1000)
164           }
165         }
166       } finally {
167         complete.countDown()
168       }
169     }
170 
171     private def doInitialLoad(): Unit = {
172       try {
173         val offsets = scala.collection.mutable.Map.empty[Int, Long]
174         // noinspection RedundantCollectionConversion
175         val partitions = consumer.partitionsFor(config.catalog).asScala.map(_.partition).toSeq
176         // note: end offsets are the *next* offset that will be returned, so subtract one to track the last offset
177         // we will actually consume
178         offsets ++=
179             KafkaConsumerVersions.endOffsets(consumer, config.catalog, partitions)
180                 .collect { case (p, o) if o > 0 => (p, o - 1) }
181         while (offsets.nonEmpty) {
182           val result = KafkaConsumerVersions.poll(consumer, poll)
183           if (!result.isEmpty) {
184             val records = result.iterator()
185             while (records.hasNext) {
186               val r = records.next()
187               val v = r.value()
188               if (v == null) {
189                 state.remove(KeyBytes(r.key()))
190               } else {
191                 state.put(KeyBytes(r.key()), v)
192               }
193               if (offsets.get(r.partition()).exists(o => r.offset() >= o)) {
194                 offsets.remove(r.partition())
195               }
196             }
197             consumer.commitAsync()
198           }
199         }
200         lazy val stateStrings =
201           state.asScala.map { case (k, v) => new String(k.bytes, StandardCharsets.UTF_8) -> new String(v, StandardCharsets.UTF_8)}
202         logger.debug(s"Completed initial load of catalog '${config.catalog}': \n  ${stateStrings.mkString("\n  ")}")
203 
204         future = CachedThreadPool.submit(this)
205         sys.addShutdownHook(future.cancel(true)) // prevent consumer from hanging if ds isn't disposed properly
206       } catch {
207         case NonFatal(e) => complete.countDown(); throw e
208       }
209     }
210 
211     def get(key: Array[Byte]): Option[Array[Byte]] = Option(state.get(KeyBytes(key)))
212 
213     def all(): CloseableIterator[(Array[Byte], Array[Byte])] =
214       CloseableIterator(state.asScala.iterator.map { case (k, v) => k.bytes -> v })
215 
216     def prefix(prefix: Array[Byte]): CloseableIterator[(Array[Byte], Array[Byte])] =
217       all().filter { case (k, _) => k.startsWith(prefix) }
218 
219     override def close(): Unit = {
220       try {
221         if (future != null) {
222           future.cancel(true)
223         }
224         complete.await(10, TimeUnit.SECONDS)
225       } finally {
226         // avoid checking consumer assignment if consumer is already closed, which will throw an error
227         if (closed.compareAndSet(false, true)) {
228           cleanupConsumer()
229         }
230       }
231     }
232 
233     private def cleanupConsumer(): Unit = {
234       try {
235         val topics = consumer.assignment()
236         consumer.unsubscribe()
237         if (!topics.isEmpty) {
238           Try(adminClientOp(KafkaAdminVersions.deleteConsumerGroupOffsets(_, groupId, topics))).failed.foreach { e =>
239             logger.warn("Error deleting consumer group offsets:", e)
240           }
241         }
242       } finally {
243         consumer.close()
244       }
245     }
246   }

248   /**
249    * Supports using a byte array as a map key
250    *
251    * @param bytes bytes
252    */
253   private case class KeyBytes(bytes: Array[Byte]) {
254     override def hashCode(): Int = java.util.Arrays.hashCode(bytes)
255     override def equals(obj: Any): Boolean = {
256       obj match {
257         case KeyBytes(other) => java.util.Arrays.equals(bytes, other)
258         case _ => false
259       }
260     }
261   }
