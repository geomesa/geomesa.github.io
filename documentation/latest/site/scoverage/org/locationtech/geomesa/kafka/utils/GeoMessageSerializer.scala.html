28 /**
29   * Serialized `GeoMessage`s
30   *
31   * Current encoding (version 2/3), designed to work with kafka log compaction.
32   * See https://kafka.apache.org/10/documentation.html#design_compactionbasics
33   *
34   * Version 3 uses avro for serialized features, version 2 uses kryo
35   *
36   * change:
37   *   key: n bytes feature id
38   *   value: n bytes for serialized feature (without id)
39   *   headers: "v" -> serialization version
40   *
41   * delete:
42   *   key: n bytes feature id
43   *   value: null - this allows for log compaction to delete the feature out
44   *   headers: "v" -> serialization version
45   *
46   * clear:
47   *   key: empty
48   *   value: empty
49   *   headers: "v" -> serialization version
50   *
51   * Version 1 legacy encoding:
52   *
53   * change:
54   *   key: 1 byte message version, 1 byte for message type ('C'), 8 byte long for epoch millis
55   *   value: n bytes for serialized feature (with id)
56   *
57   * delete:
58   *   key: 1 byte message version, 1 byte for message type ('D'), 8 byte long for epoch millis
59   *   value: n bytes for feature id
60   *
61   * clear:
62   *   key: 1 byte message version, 1 byte for message type ('X'), 8 byte long for epoch millis
63   *   value: empty
64   */

77   /**
78    * Create a message serializer
79    *
80    * @param sft simple feature type
81    * @param serialization serialization type (avro or kryo)
82    * @param opts extra serialization options
83    * @return
84    */
85   def apply(
86       sft: SimpleFeatureType,
87       serialization: SerializationType = SerializationType.KRYO,
88       opts: Set[SerializationOption] = Set.empty): GeoMessageSerializer = {
89     val options = DefaultOpts ++ opts
90     val kryoSerializer = KryoFeatureSerializer.builder(sft).opts(options).build()
91     // note: native vs non-native deserialization is handled automatically by the avro serializer, regardless of opts
92     val avroSerializer = AvroFeatureSerializer.builder(sft).opts(options).build()
93     val (serializer, version) = serialization match {
94       case SerializationType.KRYO => (kryoSerializer, KryoVersion)
95       case SerializationType.AVRO => (avroSerializer, AvroVersion)
96       case _ => throw new UnsupportedOperationException(s"Unhandled serialization type '$serialization'")
97     }
98     new GeoMessageSerializer(sft, serializer, kryoSerializer, avroSerializer, version)
99   }

101   def partition(numPartitions: Int, toKey: => Array[Byte]): Int = {
102     // use the feature id if available, otherwise (for clear) use random shard
103     if (numPartitions < 2) { 0 } else {
104       val key = toKey
105       if (key != null && key.length > 0) {
106         Math.abs(MurmurHash3.bytesHash(key, MurmurHash3.arraySeed)) % numPartitions
107       } else {
108         Random.nextInt(numPartitions)
109       }
110     }
111   }

144 /**
145   * Serializes `GeoMessage`s
146   *
147   * @param sft simple feature type being serialized
148   * @param serializer serializer used for reading or writing messages
149   * @param kryo kryo serializer used for deserializing kryo messages
150   * @param avro avro serializer used for deserializing avro messages
151   * @param version version byte corresponding to the serializer type
152   */
153 class GeoMessageSerializer(sft: SimpleFeatureType,
154                            val serializer: SimpleFeatureSerializer,
155                            kryo: KryoFeatureSerializer,
156                            avro: AvroFeatureSerializer,
157                            version: Byte) extends LazyLogging {
158 
159   private val headers = Map(GeoMessageSerializer.VersionHeader -> Array(version))
160 
161   private lazy val serializerV1 = KryoFeatureSerializer.builder(sft).withUserData.immutable.build()
162 
163   /**
164     * Serializes a message
165     *
166     * @param msg message
167     * @return (serialized key, serialized value)
168     */
169   def serialize(msg: GeoMessage): (Array[Byte], Array[Byte], Map[String, Array[Byte]]) = {
170     msg match {
171       case m: Change => serialize(m)
172       case m: Delete => serialize(m)
173       case m: Clear  => serialize(m)
174       case _ => throw new IllegalArgumentException(s"Invalid message: '$msg'")
175     }
176   }
177 
178   /**
179     * Deserializes a serialized `GeoMessage`
180     *
181     * @param key the serialized message key
182     * @param value the serialized message body
183     * @param timestamp the kafka message timestamp
184     * @return the deserialized message
185     */
186   def deserialize(
187       key: Array[Byte],
188       value: Array[Byte],
189       headers: Map[String, Array[Byte]] = Map.empty,
190       timestamp: Long = System.currentTimeMillis()): GeoMessage = {
191     try {
192       headers.get(GeoMessageSerializer.VersionHeader) match {
193         case Some(h) if h.length == 1 && h(0) == GeoMessageSerializer.KryoVersion => deserialize(key, value, kryo)
194         case Some(h) if h.length == 1 && h(0) == GeoMessageSerializer.AvroVersion => deserialize(key, value, avro)
195         case _ => tryDeserializeVersions(key, value)
196       }
197     } catch {
198       case NonFatal(e) =>
199         throw new IllegalArgumentException(
200           s"Unexpected message format: ${Option(key).map(ByteArrays.toHex).getOrElse("")} " +
201               s"${Option(value).map(ByteArrays.toHex).getOrElse("")}", e)
202     }
203   }
204 
205   /**
206     * Serializes a change message
207     *
208     * key: n bytes feature id
209     * value: n bytes for serialized feature (without id)
210     * headers: "v" -> 1 byte message version
211     *
212     * @param msg msg
213     * @return (serialized key, serialized value, headers)
214     */
215   private def serialize(msg: Change): (Array[Byte], Array[Byte], Map[String, Array[Byte]]) =
216     (msg.feature.getID.getBytes(StandardCharsets.UTF_8), serializer.serialize(msg.feature), headers)
217 
218   /**
219     * Serializes a delete message
220     *
221     * key: n bytes feature id
222     * value: null
223     * headers: "v" -> 1 byte message version
224     *
225     * @param msg msg
226     * @return (serialized key, serialized value, headers)
227     */
228   private def serialize(msg: Delete): (Array[Byte], Array[Byte], Map[String, Array[Byte]]) =
229     (msg.id.getBytes(StandardCharsets.UTF_8), null, headers)
230 
231   /**
232     * Serializes a clear message
233     *
234     * key: empty
235     * value: empty
236     * headers: "v" -> 1 byte message version
237     *
238     * @param msg msg
239     * @return (serialized key, serialized value, headers)
240     */
241   private def serialize(msg: Clear): (Array[Byte], Array[Byte], Map[String, Array[Byte]]) =
242     (GeoMessageSerializer.Empty, GeoMessageSerializer.Empty, headers)
243 
244   /**
245     * Deserialize a message using the appropriate serializer
246     *
247     * @param key message key
248     * @param value message value
249     * @param deserializer deserializer appropriate for the message encoding
250     * @return
251     */
252   protected def deserialize(key: Array[Byte], value: Array[Byte], deserializer: SimpleFeatureSerializer): GeoMessage = {
253     if (key.isEmpty && (value == null || value.isEmpty)) { Clear } else {
254       val id = new String(key, StandardCharsets.UTF_8)
255       if (value == null) { Delete(id) } else { Change(deserializer.deserialize(id, value)) }
256     }
257   }
258 
259   /**
260     * Used to deserialize messages without headers, which may be caused by:
261     *
262     * a) version 1 messages
263     * b) an older kafka version that doesn't support message headers
264     *
265     * @param key message key
266     * @param value message value
267     * @return
268     */
269   private def tryDeserializeVersions(key: Array[Byte], value: Array[Byte]): GeoMessage = {
270     if (key.length == 10 && key(0) == 1 && Seq('C', 'D', 'X').contains(key(1).toChar)) {
271       try { deserializeV1(key, value) } catch {
272         case NonFatal(e) =>
273           try { tryDeserializeTypes(key, value) } catch {
274             case NonFatal(suppressed) => e.addSuppressed(suppressed); throw e
275           }
276       }
277     } else {
278       tryDeserializeTypes(key, value)
279     }
280   }
281 
282   /**
283     * Try to deserialize using both kryo and avro
284     *
285     * @param key message key
286     * @param value message value
287     * @return
288     */
289   private def tryDeserializeTypes(key: Array[Byte], value: Array[Byte]): GeoMessage = {
290     try { deserialize(key, value, serializer) } catch {
291       case NonFatal(e) =>
292         try { deserialize(key, value, if (serializer.eq(kryo)) { avro } else { kryo }) } catch {
293           case NonFatal(suppressed) => e.addSuppressed(suppressed); throw e
294         }
295     }
296   }
297 
298   /**
299     * Deserialize version 1 messages
300     *
301     * @param key message key
302     * @param value message value
303     * @return
304     */
305   private def deserializeV1(key: Array[Byte], value: Array[Byte]): GeoMessage = {
306     key(1).toChar match {
307       case 'C' => Change(serializerV1.deserialize(value))
308       case 'D' => Delete(new String(value, StandardCharsets.UTF_8))
309       case 'X' => Clear
310       case m   => throw new IllegalArgumentException(s"Unknown message type: $m" )
311     }
312   }
