28 /**
29   * Serialized `GeoMessage`s
30   *
31   * Current encoding (version 2/3), designed to work with kafka log compaction.
32   * See https://kafka.apache.org/10/documentation.html#design_compactionbasics
33   *
34   * Version 3 uses avro for serialized features, version 2 uses kryo
35   *
36   * change:
37   *   key: n bytes feature id
38   *   value: n bytes for serialized feature (without id)
39   *   headers: "v" -> serialization version
40   *
41   * delete:
42   *   key: n bytes feature id
43   *   value: null - this allows for log compaction to delete the feature out
44   *   headers: "v" -> serialization version
45   *
46   * clear:
47   *   key: empty
48   *   value: empty
49   *   headers: "v" -> serialization version
50   *
51   * Version 1 legacy encoding:
52   *
53   * change:
54   *   key: 1 byte message version, 1 byte for message type ('C'), 8 byte long for epoch millis
55   *   value: n bytes for serialized feature (with id)
56   *
57   * delete:
58   *   key: 1 byte message version, 1 byte for message type ('D'), 8 byte long for epoch millis
59   *   value: n bytes for feature id
60   *
61   * clear:
62   *   key: 1 byte message version, 1 byte for message type ('X'), 8 byte long for epoch millis
63   *   value: empty
64   */

90   /**
91    * Create a message serializer
92    *
93    * @param sft simple feature type
94    * @param serialization serialization type (avro or kryo)
95    * @param opts extra serialization options
96    * @return
97    */
98   def apply(
99       sft: SimpleFeatureType,
100       serialization: SerializationType,
101       opts: Set[SerializationOption]): GeoMessageSerializer = {
102     val options = SerializationOptions.builder.withoutId.withUserData.immutable.build ++ opts
103     val kryoSerializer = KryoFeatureSerializer.builder(sft).opts(options).build()
104     val avroSerializer = AvroFeatureSerializer.builder(sft).opts(options).build()
105     val (serializer, version) = serialization match {
106       case SerializationType.KRYO => (kryoSerializer, KryoVersion)
107       case SerializationType.AVRO => (avroSerializer, AvroVersion)
108       case _ => throw new UnsupportedOperationException(s"Unhandled serialization type '$serialization'")
109     }
110     new GeoMessageSerializer(sft, serializer, kryoSerializer, avroSerializer, version)
111   }

113   def partition(numPartitions: Int, toKey: => Array[Byte]): Int = {
114     // use the feature id if available, otherwise (for clear) use random shard
115     if (numPartitions < 2) { 0 } else {
116       val key = toKey
117       if (key != null && key.length > 0) {
118         Math.abs(MurmurHash3.bytesHash(key, MurmurHash3.arraySeed)) % numPartitions
119       } else {
120         Random.nextInt(numPartitions)
121       }
122     }
123   }

156 /**
157   * Serializes `GeoMessage`s
158   *
159   * @param sft simple feature type being serialized
160   * @param serializer serializer used for reading or writing messages
161   * @param kryo kryo serializer used for deserializing kryo messages
162   * @param avro avro serializer used for deserializing avro messages
163   * @param version version byte corresponding to the serializer type
164   */
165 class GeoMessageSerializer(sft: SimpleFeatureType,
166                            val serializer: SimpleFeatureSerializer,
167                            kryo: KryoFeatureSerializer,
168                            avro: AvroFeatureSerializer,
169                            version: Byte) extends LazyLogging {
170 
171   private val headers = Map(GeoMessageSerializer.VersionHeader -> Array(version))
172 
173   private lazy val serializerV1 = KryoFeatureSerializer.builder(sft).withUserData.immutable.build()
174 
175   /**
176     * Serializes a message
177     *
178     * @param msg message
179     * @return (serialized key, serialized value)
180     */
181   def serialize(msg: GeoMessage): (Array[Byte], Array[Byte], Map[String, Array[Byte]]) = {
182     msg match {
183       case m: Change => serialize(m)
184       case m: Delete => serialize(m)
185       case m: Clear  => serialize(m)
186       case _ => throw new IllegalArgumentException(s"Invalid message: '$msg'")
187     }
188   }
189 
190   /**
191     * Deserializes a serialized `GeoMessage`
192     *
193     * @param key the serialized message key
194     * @param value the serialized message body
195     * @param timestamp the kafka message timestamp
196     * @return the deserialized message
197     */
198   def deserialize(
199       key: Array[Byte],
200       value: Array[Byte],
201       headers: Map[String, Array[Byte]] = Map.empty,
202       timestamp: Long = System.currentTimeMillis()): GeoMessage = {
203     try {
204       headers.get(GeoMessageSerializer.VersionHeader) match {
205         case Some(h) if h.length == 1 && h(0) == GeoMessageSerializer.KryoVersion => deserialize(key, value, kryo)
206         case Some(h) if h.length == 1 && h(0) == GeoMessageSerializer.AvroVersion => deserialize(key, value, avro)
207         case _ => tryDeserializeVersions(key, value)
208       }
209     } catch {
210       case NonFatal(e) =>
211         throw new IllegalArgumentException(
212           s"Unexpected message format: ${Option(key).map(ByteArrays.toHex).getOrElse("")} " +
213               s"${Option(value).map(ByteArrays.toHex).getOrElse("")}", e)
214     }
215   }
216 
217   /**
218     * Serializes a change message
219     *
220     * key: n bytes feature id
221     * value: n bytes for serialized feature (without id)
222     * headers: "v" -> 1 byte message version
223     *
224     * @param msg msg
225     * @return (serialized key, serialized value, headers)
226     */
227   private def serialize(msg: Change): (Array[Byte], Array[Byte], Map[String, Array[Byte]]) =
228     (msg.feature.getID.getBytes(StandardCharsets.UTF_8), serializer.serialize(msg.feature), headers)
229 
230   /**
231     * Serializes a delete message
232     *
233     * key: n bytes feature id
234     * value: null
235     * headers: "v" -> 1 byte message version
236     *
237     * @param msg msg
238     * @return (serialized key, serialized value, headers)
239     */
240   private def serialize(msg: Delete): (Array[Byte], Array[Byte], Map[String, Array[Byte]]) =
241     (msg.id.getBytes(StandardCharsets.UTF_8), null, headers)
242 
243   /**
244     * Serializes a clear message
245     *
246     * key: empty
247     * value: empty
248     * headers: "v" -> 1 byte message version
249     *
250     * @param msg msg
251     * @return (serialized key, serialized value, headers)
252     */
253   private def serialize(msg: Clear): (Array[Byte], Array[Byte], Map[String, Array[Byte]]) =
254     (GeoMessageSerializer.Empty, GeoMessageSerializer.Empty, headers)
255 
256   /**
257     * Deserialize a message using the appropriate serializer
258     *
259     * @param key message key
260     * @param value message value
261     * @param deserializer deserializer appropriate for the message encoding
262     * @return
263     */
264   protected def deserialize(key: Array[Byte], value: Array[Byte], deserializer: SimpleFeatureSerializer): GeoMessage = {
265     if (key.isEmpty && (value == null || value.isEmpty)) { Clear } else {
266       val id = new String(key, StandardCharsets.UTF_8)
267       if (value == null) { Delete(id) } else { Change(deserializer.deserialize(id, value)) }
268     }
269   }
270 
271   /**
272     * Used to deserialize messages without headers, which may be caused by:
273     *
274     * a) version 1 messages
275     * b) an older kafka version that doesn't support message headers
276     *
277     * @param key message key
278     * @param value message value
279     * @return
280     */
281   private def tryDeserializeVersions(key: Array[Byte], value: Array[Byte]): GeoMessage = {
282     if (key.length == 10 && key(0) == 1 && Seq('C', 'D', 'X').contains(key(1).toChar)) {
283       try { deserializeV1(key, value) } catch {
284         case NonFatal(e) =>
285           try { tryDeserializeTypes(key, value) } catch {
286             case NonFatal(suppressed) => e.addSuppressed(suppressed); throw e
287           }
288       }
289     } else {
290       tryDeserializeTypes(key, value)
291     }
292   }
293 
294   /**
295     * Try to deserialize using both kryo and avro
296     *
297     * @param key message key
298     * @param value message value
299     * @return
300     */
301   private def tryDeserializeTypes(key: Array[Byte], value: Array[Byte]): GeoMessage = {
302     try { deserialize(key, value, serializer) } catch {
303       case NonFatal(e) =>
304         try { deserialize(key, value, if (serializer.eq(kryo)) { avro } else { kryo }) } catch {
305           case NonFatal(suppressed) => e.addSuppressed(suppressed); throw e
306         }
307     }
308   }
309 
310   /**
311     * Deserialize version 1 messages
312     *
313     * @param key message key
314     * @param value message value
315     * @return
316     */
317   private def deserializeV1(key: Array[Byte], value: Array[Byte]): GeoMessage = {
318     key(1).toChar match {
319       case 'C' => Change(serializerV1.deserialize(value))
320       case 'D' => Delete(new String(value, StandardCharsets.UTF_8))
321       case 'X' => Clear
322       case m   => throw new IllegalArgumentException(s"Unknown message type: $m" )
323     }
324   }
