28 /**
29   * Serialized `GeoMessage`s
30   *
31   * Current encoding (version 2/3), designed to work with kafka log compaction.
32   * See https://kafka.apache.org/10/documentation.html#design_compactionbasics
33   *
34   * Version 3 uses avro for serialized features, version 2 uses kryo
35   *
36   * change:
37   *   key: n bytes feature id
38   *   value: n bytes for serialized feature (without id)
39   *   headers: "v" -> serialization version
40   *
41   * delete:
42   *   key: n bytes feature id
43   *   value: null - this allows for log compaction to delete the feature out
44   *   headers: "v" -> serialization version
45   *
46   * clear:
47   *   key: empty
48   *   value: empty
49   *   headers: "v" -> serialization version
50   *
51   * Version 1 legacy encoding:
52   *
53   * change:
54   *   key: 1 byte message version, 1 byte for message type ('C'), 8 byte long for epoch millis
55   *   value: n bytes for serialized feature (with id)
56   *
57   * delete:
58   *   key: 1 byte message version, 1 byte for message type ('D'), 8 byte long for epoch millis
59   *   value: n bytes for feature id
60   *
61   * clear:
62   *   key: 1 byte message version, 1 byte for message type ('X'), 8 byte long for epoch millis
63   *   value: empty
64   */

74   /**
75    * Create a message serializer
76    *
77    * @param sft simple feature type
78    * @param serialization serialization type (avro or kryo)
79    * @param opts extra serialization options
80    * @return
81    */
82   def apply(
83       sft: SimpleFeatureType,
84       serialization: SerializationType = SerializationType.KRYO,
85       opts: Set[SerializationOption] = Set.empty): GeoMessageSerializer = {
86     val options = SerializationOption.builder.withoutId.withUserData.immutable.build() ++ opts
87     val kryoSerializer = KryoFeatureSerializer.builder(sft).opts(options).build()
88     val avroSerializer = AvroFeatureSerializer.builder(sft).opts(options).build()
89     val (serializer, version) = serialization match {
90       case SerializationType.KRYO => (kryoSerializer, KryoVersion)
91       case SerializationType.AVRO => (avroSerializer, AvroVersion)
92       case _ => throw new UnsupportedOperationException(s"Unhandled serialization type '$serialization'")
93     }
94     new GeoMessageSerializer(sft, serializer, kryoSerializer, avroSerializer, version)
95   }
96 
97   def partition(numPartitions: Int, toKey: => Array[Byte]): Int = {
98     // use the feature id if available, otherwise (for clear) use random shard
99     if (numPartitions < 2) { 0 } else {
100       val key = toKey
101       if (key != null && key.length > 0) {
102         Math.abs(MurmurHash3.bytesHash(key, MurmurHash3.arraySeed)) % numPartitions
103       } else {
104         Random.nextInt(numPartitions)
105       }
106     }
107   }

140 /**
141   * Serializes `GeoMessage`s
142   *
143   * @param sft simple feature type being serialized
144   * @param serializer serializer used for reading or writing messages
145   * @param kryo kryo serializer used for deserializing kryo messages
146   * @param avro avro serializer used for deserializing avro messages
147   * @param version version byte corresponding to the serializer type
148   */
149 class GeoMessageSerializer(sft: SimpleFeatureType,
150                            val serializer: SimpleFeatureSerializer,
151                            kryo: KryoFeatureSerializer,
152                            avro: AvroFeatureSerializer,
153                            version: Byte) extends LazyLogging {
154 
155   private val headers = Map(GeoMessageSerializer.VersionHeader -> Array(version))
156 
157   private lazy val serializerV1 = KryoFeatureSerializer.builder(sft).withUserData.immutable.build()
158 
159   /**
160     * Serializes a message
161     *
162     * @param msg message
163     * @return (serialized key, serialized value)
164     */
165   def serialize(msg: GeoMessage): (Array[Byte], Array[Byte], Map[String, Array[Byte]]) = {
166     msg match {
167       case m: Change => serialize(m)
168       case m: Delete => serialize(m)
169       case m: Clear  => serialize(m)
170       case _ => throw new IllegalArgumentException(s"Invalid message: '$msg'")
171     }
172   }
173 
174   /**
175     * Deserializes a serialized `GeoMessage`
176     *
177     * @param key the serialized message key
178     * @param value the serialized message body
179     * @param timestamp the kafka message timestamp
180     * @return the deserialized message
181     */
182   def deserialize(
183       key: Array[Byte],
184       value: Array[Byte],
185       headers: Map[String, Array[Byte]] = Map.empty,
186       timestamp: Long = System.currentTimeMillis()): GeoMessage = {
187     try {
188       headers.get(GeoMessageSerializer.VersionHeader) match {
189         case Some(h) if h.length == 1 && h(0) == GeoMessageSerializer.KryoVersion => deserialize(key, value, kryo)
190         case Some(h) if h.length == 1 && h(0) == GeoMessageSerializer.AvroVersion => deserialize(key, value, avro)
191         case _ => tryDeserializeVersions(key, value)
192       }
193     } catch {
194       case NonFatal(e) =>
195         throw new IllegalArgumentException(
196           s"Unexpected message format: ${Option(key).map(ByteArrays.toHex).getOrElse("")} " +
197               s"${Option(value).map(ByteArrays.toHex).getOrElse("")}", e)
198     }
199   }
200 
201   /**
202     * Serializes a change message
203     *
204     * key: n bytes feature id
205     * value: n bytes for serialized feature (without id)
206     * headers: "v" -> 1 byte message version
207     *
208     * @param msg msg
209     * @return (serialized key, serialized value, headers)
210     */
211   private def serialize(msg: Change): (Array[Byte], Array[Byte], Map[String, Array[Byte]]) =
212     (msg.feature.getID.getBytes(StandardCharsets.UTF_8), serializer.serialize(msg.feature), headers)
213 
214   /**
215     * Serializes a delete message
216     *
217     * key: n bytes feature id
218     * value: null
219     * headers: "v" -> 1 byte message version
220     *
221     * @param msg msg
222     * @return (serialized key, serialized value, headers)
223     */
224   private def serialize(msg: Delete): (Array[Byte], Array[Byte], Map[String, Array[Byte]]) =
225     (msg.id.getBytes(StandardCharsets.UTF_8), null, headers)
226 
227   /**
228     * Serializes a clear message
229     *
230     * key: empty
231     * value: empty
232     * headers: "v" -> 1 byte message version
233     *
234     * @param msg msg
235     * @return (serialized key, serialized value, headers)
236     */
237   private def serialize(msg: Clear): (Array[Byte], Array[Byte], Map[String, Array[Byte]]) =
238     (GeoMessageSerializer.Empty, GeoMessageSerializer.Empty, headers)
239 
240   /**
241     * Deserialize a message using the appropriate serializer
242     *
243     * @param key message key
244     * @param value message value
245     * @param deserializer deserializer appropriate for the message encoding
246     * @return
247     */
248   protected def deserialize(key: Array[Byte], value: Array[Byte], deserializer: SimpleFeatureSerializer): GeoMessage = {
249     if (key.isEmpty && (value == null || value.isEmpty)) { Clear } else {
250       val id = new String(key, StandardCharsets.UTF_8)
251       if (value == null) { Delete(id) } else { Change(deserializer.deserialize(id, value)) }
252     }
253   }
254 
255   /**
256     * Used to deserialize messages without headers, which may be caused by:
257     *
258     * a) version 1 messages
259     * b) an older kafka version that doesn't support message headers
260     *
261     * @param key message key
262     * @param value message value
263     * @return
264     */
265   private def tryDeserializeVersions(key: Array[Byte], value: Array[Byte]): GeoMessage = {
266     if (key.length == 10 && key(0) == 1 && Seq('C', 'D', 'X').contains(key(1).toChar)) {
267       try { deserializeV1(key, value) } catch {
268         case NonFatal(e) =>
269           try { tryDeserializeTypes(key, value) } catch {
270             case NonFatal(suppressed) => e.addSuppressed(suppressed); throw e
271           }
272       }
273     } else {
274       tryDeserializeTypes(key, value)
275     }
276   }
277 
278   /**
279     * Try to deserialize using both kryo and avro
280     *
281     * @param key message key
282     * @param value message value
283     * @return
284     */
285   private def tryDeserializeTypes(key: Array[Byte], value: Array[Byte]): GeoMessage = {
286     try { deserialize(key, value, serializer) } catch {
287       case NonFatal(e) =>
288         try { deserialize(key, value, if (serializer.eq(kryo)) { avro } else { kryo }) } catch {
289           case NonFatal(suppressed) => e.addSuppressed(suppressed); throw e
290         }
291     }
292   }
293 
294   /**
295     * Deserialize version 1 messages
296     *
297     * @param key message key
298     * @param value message value
299     * @return
300     */
301   private def deserializeV1(key: Array[Byte], value: Array[Byte]): GeoMessage = {
302     key(1).toChar match {
303       case 'C' => Change(serializerV1.deserialize(value))
304       case 'D' => Delete(new String(value, StandardCharsets.UTF_8))
305       case 'X' => Clear
306       case m   => throw new IllegalArgumentException(s"Unknown message type: $m" )
307     }
308   }
