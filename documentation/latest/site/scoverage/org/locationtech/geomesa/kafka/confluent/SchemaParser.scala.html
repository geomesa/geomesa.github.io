38    * Convert an Avro [[Schema]] into a [[SimpleFeatureType]].
39    */
40   def schemaToSft(schema: Schema, name: Option[String] = None): SimpleFeatureType = {
41     val builder = new SchemaBuilder()
42 
43     var defaultGeomField: Option[String] = None
44     var visibilityField: Option[String] = None
45 
46     schema.getFields.asScala.foreach { field =>
47       val metadata = parseMetadata(field)
48 
49       metadata.field match {
50         case GeometryField(geomType, default) if !metadata.exclude =>
51           if (default) {
52             defaultGeomField.foreach { name =>
53               throw new IllegalArgumentException("There may be only one default geometry field in a schema: " +
54                   s"'$name' was already declared as the default")
55             }
56             defaultGeomField = Some(field.name)
57           }
58 
59           addGeometryToBuilder(builder, field.name, geomType, default, metadata.extraProps)
60 
61         case DateField(_) if !metadata.exclude =>
62           builder.addDate(field.name).withOptions(metadata.extraProps.toSeq: _*)
63 
64         case VisibilityField =>
65           visibilityField.foreach { name =>
66             throw new IllegalArgumentException(s"There may be only one visibility field in a schema: " +
67                 s"'$name' was already declared as the visibility")
68           }
69           visibilityField = Some(field.name)
70 
71           if (!metadata.exclude) {
72             addFieldToBuilder(builder, field, userData = metadata.extraProps)
73           }
74           builder.userData(GeoMesaAvroVisibilityField.KEY, field.name)
75 
76         case StandardField if !metadata.exclude =>
77           addFieldToBuilder(builder, field, userData = metadata.extraProps)
78 
79         case _ =>
80       }
81     }
82 
83     // any extra props on the schema go in the SFT user data
84     builder.userData.userData(schema.getObjectProps.asScala.map { case (k, v) => k -> v.toString }.toMap)
85     builder.build(name.getOrElse(schema.getName))
86   }

88   @tailrec
89   private def addFieldToBuilder(
90       builder: SchemaBuilder,
91       field: Schema.Field,
92       typeOverride: Option[Schema.Type] = None,
93       userData: Map[String, String] = Map.empty): Unit = {
94     lazy val logicalType = if (typeOverride.isDefined) { None } else { Option(field.schema().getLogicalType) }
95     typeOverride.getOrElse(field.schema.getType) match {
96       case Schema.Type.STRING  => builder.addString(field.name).withOptions(userData.toSeq: _*)
97       case Schema.Type.BOOLEAN => builder.addBoolean(field.name).withOptions(userData.toSeq: _*)
98       case Schema.Type.INT     => builder.addInt(field.name).withOptions(userData.toSeq: _*)
99       case Schema.Type.DOUBLE  => builder.addDouble(field.name).withOptions(userData.toSeq: _*)
100       case Schema.Type.LONG    =>
101         logicalType match {
102           case Some(_: TimestampMillis) => builder.addDate(field.name).withOptions(userData.toSeq: _*)
103           case Some(_: TimestampMicros) => builder.addDate(field.name).withOptions(userData.toSeq: _*)
104           case _ => builder.addLong(field.name).withOptions(userData.toSeq: _*)
105         }
106       case Schema.Type.FLOAT   => builder.addFloat(field.name).withOptions(userData.toSeq: _*)
107       case Schema.Type.BYTES   => builder.addBytes(field.name).withOptions(userData.toSeq: _*)
108       case Schema.Type.UNION   =>
109         // if a union has more than one non-null type, it is not supported
110         val types = field.schema.getTypes.asScala.map(_.getType).filter(_ != Schema.Type.NULL).toSet
111         if (types.size != 1) {
112           throw UnsupportedAvroTypeException(types.mkString("[", ", ", "]"))
113         } else {
114           addFieldToBuilder(builder, field, Option(types.head), userData)
115         }
116       case Schema.Type.ENUM    => builder.addString(field.name).withOptions(userData.toSeq: _*)
117       case Schema.Type.MAP     => throw UnsupportedAvroTypeException(Schema.Type.MAP.getName)
118       case Schema.Type.RECORD  => throw UnsupportedAvroTypeException(Schema.Type.RECORD.getName)
119       case Schema.Type.ARRAY   => throw UnsupportedAvroTypeException(Schema.Type.ARRAY.getName)
120       case Schema.Type.FIXED   => throw UnsupportedAvroTypeException(Schema.Type.FIXED.getName)
121       case Schema.Type.NULL    => throw UnsupportedAvroTypeException(Schema.Type.NULL.getName)
122       case _                   => throw UnsupportedAvroTypeException("unknown")
123     }
124   }

126   private def addGeometryToBuilder(
127       builder: SchemaBuilder,
128       name: String,
129       geometry: Class[_],
130       default: Boolean,
131       userData: Map[String, String] = Map.empty): Unit = {
132     geometry match {
133       case t if t == classOf[Point] => builder.addPoint(name, default).withOptions(userData.toSeq: _*)
134       case t if t == classOf[LineString] => builder.addLineString(name, default).withOptions(userData.toSeq: _*)
135       case t if t == classOf[Polygon] => builder.addPolygon(name, default).withOptions(userData.toSeq: _*)
136       case t if t == classOf[MultiPoint] => builder.addMultiPoint(name, default).withOptions(userData.toSeq: _*)
137       case t if t == classOf[MultiLineString] => builder.addMultiLineString(name, default).withOptions(userData.toSeq: _*)
138       case t if t == classOf[MultiPolygon] => builder.addMultiPolygon(name, default).withOptions(userData.toSeq: _*)
139       case t if t == classOf[GeometryCollection] => builder.addGeometryCollection(name, default).withOptions(userData.toSeq: _*)
140       case t if t == classOf[Geometry] => builder.addMixedGeometry(name, default).withOptions(userData.toSeq: _*)
141       case _ => throw new IllegalArgumentException(s"Unknown geometry type: $geometry")
142     }
143   }

145   private def parseMetadata(field: Schema.Field): GeoMesaAvroMetadata = {
146     lazy val geomType = GeoMesaAvroGeomType.parse(field)
147     lazy val geomDefault = GeoMesaAvroGeomDefault.parse(field).getOrElse(false)
148     lazy val dateFormat = GeoMesaAvroDateFormat.parse(field)
149     lazy val visibility = GeoMesaAvroVisibilityField.parse(field).getOrElse(false)
150 
151     val metadata =
152       if (geomType.isDefined) {
153         GeometryField(geomType.get, geomDefault)
154       } else if (dateFormat.isDefined) {
155         DateField(dateFormat.get)
156       } else if (visibility) {
157         VisibilityField
158       } else {
159         StandardField
160       }
161 
162     // any field properties that are not one of the defined geomesa avro properties will go in the attribute user data
163     val extraProps = field.getObjectProps.asScala.collect {
164       case (k, v: String) if !SkippedPropertyKeys.contains(k) => k -> v
165     }
166 
167     val exclude = GeoMesaAvroExcludeField.parse(field).getOrElse(false)
168 
169     GeoMesaAvroMetadata(metadata, extraProps.toMap, exclude)
170   }

183   /**
184    * A property in an Avro [[Schema.Field]] to provide additional information when generating a [[SimpleFeatureType]].
185    *
186    * @tparam T the type of the value to be parsed from this property
187    */
188   trait GeoMesaAvroProperty[T] {
189     /**
190      * The key in the [[Schema.Field]] for this property.
191      */
192     val KEY: String
193 
194     /**
195      * Parse the value from the [[Schema.Field]] properties at this property's `KEY`.
196      *
197      * @return `None` if the `KEY` does not exist, else the value at the `KEY`
198      */
199     def parse(field: Schema.Field): Option[T]
200 
201     protected final def assertFieldType(field: Schema.Field, typ: Schema.Type): Unit = {
202       field.schema.getType match {
203         case Schema.Type.UNION =>
204           // if a union has more than one non-null type, it should not be converted to an SFT
205           val unionTypes = field.schema.getTypes.asScala.map(_.getType).filter(_ != Schema.Type.NULL).toSet
206           if (unionTypes.size != 1 || typ != unionTypes.head) {
207             throw GeoMesaAvroProperty.InvalidPropertyTypeException(typ.getName, KEY)
208           }
209         case fieldType: Schema.Type =>
210           if (typ != fieldType) {
211             throw GeoMesaAvroProperty.InvalidPropertyTypeException(typ.getName, KEY)
212           }
213       }
214     }
215   }

317   object GeoMesaAvroGeomFormat extends GeoMesaAvroDeserializableEnumProperty[String, Geometry] {
318 
319     override val KEY: String = "geomesa.geom.format"
320 
321     /**
322      * Well-Known Text representation as a [[String]]
323      */
324     val WKT: String = "wkt"
325     /**
326      * Well-Known Bytes representation as an [[Array]] of [[Byte]]s
327      */
328     val WKB: String = "wkb"
329 
330     override protected val valueMatcher: PartialFunction[(String, Schema.Field), String] = {
331       case (WKT, field) => assertFieldType(field, Schema.Type.STRING); WKT
332       case (WKB, field) => assertFieldType(field, Schema.Type.BYTES); WKB
333     }
334 
335     override protected val fieldReaderMatcher: PartialFunction[String, AnyRef => Geometry] = {
336       case WKT => data => WKTUtils.read(data.toString)
337       case WKB => data => WKBUtils.read(unwrap(data.asInstanceOf[ByteBuffer]))
338     }
339 
340     override protected val fieldWriterMatcher: PartialFunction[String, Geometry => AnyRef] = {
341       case WKT => geom => WKTUtils.write(geom)
342       case WKB => geom => ByteBuffer.wrap(WKBUtils.write(geom))
343     }
344 
345     private def unwrap(buf: ByteBuffer): Array[Byte] = {
346       if (buf.hasArray && buf.arrayOffset() == 0 && buf.limit() == buf.array().length) {
347         buf.array()
348       } else {
349         val array = Array.ofDim[Byte](buf.limit())
350         buf.get(array)
351         array
352       }
353     }
354   }
