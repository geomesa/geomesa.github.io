137   def createConsumer(
138       typeName: String,
139       groupId: String,
140       processor: GeoMessageProcessor,
141       errorHandler: Option[ConsumerErrorHandler]): Closeable = {
142     val sft = getSchema(typeName)
143     if (sft == null) {
144       throw new IllegalArgumentException(s"Schema '$typeName' does not exist; call `createSchema` first")
145     }
146     val topic = KafkaDataStore.topic(sft)
147     val consumers = {
148       // add group id and
149       // disable read-back so we don't trigger a re-balance listener that messes with group offset tracking
150       val props = config.consumers.properties + (GROUP_ID_CONFIG -> groupId)
151       val conf = config.consumers.copy(properties = props, readBack = None)
152       KafkaDataStore.consumers(config.brokers, topic, conf)
153     }
154     val frequency = java.time.Duration.ofMillis(KafkaDataStore.LoadIntervalProperty.toDuration.get.toMillis)
155     val serializer = serialization.apply(sft)
156     val consumer = new GeoMessageConsumer(consumers, frequency, serializer, processor)
157     consumer.startConsumers(errorHandler)
158     consumer
159   }

161   override def getSchema(typeName: String): SimpleFeatureType = {
162     layerViewLookup.get(typeName) match {
163       case None => super.getSchema(typeName)
164       case Some(orig) =>
165         val parent = super.getSchema(orig)

172             throw new IllegalStateException("Inconsistent layer view config")

176     }

192   override protected def preSchemaCreate(sft: SimpleFeatureType): Unit = {
193     // note: kafka doesn't allow slashes in topic names
194     KafkaDataStore.topic(sft) match {
195       case null  => KafkaDataStore.setTopic(sft, s"${config.catalog}-${sft.getTypeName}".replaceAll("/", "-"))
196       case topic if topic.contains("/") => throw new IllegalArgumentException(s"Topic cannot contain '/': $topic")
197       case topic => logger.debug(s"Using user-defined topic [$topic]")
198     }
199     // disable our custom partitioner by default, as it messes with Kafka streams co-partition joining
200     // and it's not required since we switched our keys to be feature ids
201     if (!sft.getUserData.containsKey(KafkaDataStore.PartitioningKey)) {
202       sft.getUserData.put(KafkaDataStore.PartitioningKey, KafkaDataStore.PartitioningDefault)
203     }
204     // remove table sharing as it's not relevant
205     sft.getUserData.remove(TableSharing)
206     sft.getUserData.remove(TableSharingPrefix)
207   }

209   @throws(classOf[IllegalArgumentException])
210   override protected def preSchemaUpdate(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
211     if (layerViewLookup.contains(sft.getTypeName)) {
212       throw new IllegalArgumentException(
213         s"Schema '${sft.getTypeName}' is a read-only view of '${layerViewLookup(sft.getTypeName)}'")
214     }
215     val topic = KafkaDataStore.topic(sft)
216     if (topic == null) {
217       throw new IllegalArgumentException(s"Topic must be defined in user data under '$TopicKey'")
218     } else if (topic != KafkaDataStore.topic(previous)) {
219       if (topic.contains("/")) {
220         throw new IllegalArgumentException(s"Topic cannot contain '/': $topic")
221       }
222       onSchemaDeleted(previous)
223       onSchemaCreated(sft)
224     }
225   }

244         admin.createTopics(Collections.singletonList(newTopic)).all().get
245       }
246     }

304   override private[geomesa] def getFeatureWriter(
305       sft: SimpleFeatureType,
306       transaction: Transaction,
307       filter: Option[Filter]): FlushableFeatureWriter = {
308     if (layerViewLookup.contains(sft.getTypeName)) {
309       throw new IllegalArgumentException(
310         s"Schema '${sft.getTypeName}' is a read-only view of '${layerViewLookup(sft.getTypeName)}'")
311     }
312     val producer = getTransactionalProducer(sft, transaction)
313     val vis = sft.isVisibilityRequired
314     val serializer = serialization.apply(sft)
315     val writer = filter match {
316       case None if vis    => new AppendKafkaFeatureWriter(sft, producer, serializer) with RequiredVisibilityWriter
317       case None           => new AppendKafkaFeatureWriter(sft, producer, serializer)
318       case Some(f) if vis => new ModifyKafkaFeatureWriter(sft, producer, serializer, f) with RequiredVisibilityWriter
319       case Some(f)        => new ModifyKafkaFeatureWriter(sft, producer, serializer, f)
320     }
321     if (config.clearOnStart && cleared.add(sft.getTypeName)) {
322       writer.clear()
323     }
324     writer
325   }

336   private def getTransactionalProducer(sft: SimpleFeatureType, transaction: Transaction): KafkaFeatureProducer = {
337     val useDefaultPartitioning = KafkaDataStore.usesDefaultPartitioning(sft)
338 
339     if (transaction == null || transaction == Transaction.AUTO_COMMIT) {
340       val producer = if (useDefaultPartitioning) { defaultProducer.instance } else { partitionedProducer.instance }
341       return AutoCommitProducer(producer)
342     }
343 
344     val state = transaction.getState(KafkaDataStore.TransactionStateKey)
345     if (state == null) {
346       val partitioner = if (useDefaultPartitioning) { Map.empty } else {
347         Map(PARTITIONER_CLASS_CONFIG -> classOf[GeoMessagePartitioner].getName)
348       }
349       // add kafka transactional id if it's not set, but force acks to "all" as required by kafka
350       val props =
351         Map(TRANSACTIONAL_ID_CONFIG -> UUID.randomUUID().toString) ++
352             partitioner ++
353             config.producers.properties ++
354             Map(ACKS_CONFIG -> "all")
355       val producer = KafkaTransactionState(KafkaDataStore.producer(config.brokers, props))
356       transaction.putState(KafkaDataStore.TransactionStateKey, producer)
357       producer
358     } else {
359       state match {
360         case p: KafkaTransactionState => p
361         case _ => throw new IllegalArgumentException(s"Found non-kafka state in transaction: $state")
362       }
363     }
364   }

503   /**
504     * Rebalance listener that seeks the consumer to the an offset based on a read-back duration
505     *
506     * @param consumer consumer
507     * @param partitions shared partition map, to ensure we only read-back once per partition. For subsequent
508     *                   rebalances, we should have committed offsets that will be used
509     * @param readBack duration to read back, or Duration.Inf to go to the beginning
510     */
511   private [kafka] class ReadBackRebalanceListener(consumer: Consumer[Array[Byte], Array[Byte]],
512                                                   partitions: java.util.Set[Int],
513                                                   readBack: Duration)
514       extends ConsumerRebalanceListener with LazyLogging {
515 
516     import scala.collection.JavaConverters._
517 
518     override def onPartitionsRevoked(topicPartitions: java.util.Collection[TopicPartition]): Unit = {}
519 
520     override def onPartitionsAssigned(topicPartitions: java.util.Collection[TopicPartition]): Unit = {
521       topicPartitions.asScala.foreach { tp =>
522         if (partitions.add(tp.partition())) {
523           KafkaConsumerVersions.pause(consumer, tp)
524           try {
525             if (readBack.isFinite) {
526               val offset = Try {
527                 val time = System.currentTimeMillis() - readBack.toMillis
528                 KafkaConsumerVersions.offsetsForTimes(consumer, tp.topic, Seq(tp.partition), time).get(tp.partition)
529               }
530               offset match {
531                 case Success(Some(o)) =>
532                   logger.debug(s"Seeking to offset $o for read-back $readBack on [${tp.topic}:${tp.partition}]")
533                   consumer.seek(tp, o)
