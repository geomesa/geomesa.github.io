85   private val caches = Caffeine.newBuilder().build[String, KafkaCacheLoader](new CacheLoader[String, KafkaCacheLoader] {
86     override def load(key: String): KafkaCacheLoader = {
87       if (config.consumers.count < 1) {
88         logger.info("Kafka consumers disabled for this data store instance")
89         KafkaCacheLoader.NoOpLoader
90       } else {
91         val sft = KafkaDataStore.super.getSchema(key)
92         val views = config.layerViewsConfig.getOrElse(key, Seq.empty).map(KafkaDataStore.createLayerView(sft, _))
93         val tags = KafkaDataStore.this.tags(sft.getTypeName)
94         // if the expiry is zero, this will return a NoOpFeatureCache
95         val cache = KafkaFeatureCache(sft, config.indices, views, tags)
96         val topic = KafkaDataStore.topic(sft)
97         val consumers = KafkaDataStore.consumers(config.brokers, topic, config.consumers)
98         val frequency = java.time.Duration.ofMillis(KafkaDataStore.LoadIntervalProperty.toDuration.get.toMillis)
99         val serializer = serialization.apply(sft)
100         val readBack = config.consumers.readBack
101         val expiry = config.indices.expiry
102         val offsetCommitInterval = config.consumers.offsetCommitInterval
103         val loader = new KafkaCacheLoaderImpl(sft, cache, consumers, topic, frequency, offsetCommitInterval, serializer, readBack, expiry, tags)
104         try { loader.start() } catch {
105           case NonFatal(e) =>
106             CloseWithLogging(loader)
107             throw e
108         }
109         loader
110       }
111     }

144   def createConsumer(
145       typeName: String,
146       groupId: String,
147       processor: GeoMessageProcessor,
148       errorHandler: Option[ConsumerErrorHandler]): Closeable = {
149     val sft = getSchema(typeName)
150     if (sft == null) {
151       throw new IllegalArgumentException(s"Schema '$typeName' does not exist; call `createSchema` first")
152     }
153     val topic = KafkaDataStore.topic(sft)
154     val consumers =
155       KafkaDataStore.consumers(config.brokers, topic,
156         config.consumers.copy(properties = config.consumers.properties + (GROUP_ID_CONFIG -> groupId))) // add group id
157     consumers.foreach(KafkaConsumerVersions.subscribe(_, topic))
158     val frequency = java.time.Duration.ofMillis(KafkaDataStore.LoadIntervalProperty.toDuration.get.toMillis)
159     val serializer = serialization.apply(sft)
160     val consumer = new GeoMessageConsumer(consumers, frequency, serializer, processor)
161     consumer.startConsumers(errorHandler)
162     consumer
163   }

165   override def getSchema(typeName: String): SimpleFeatureType = {
166     layerViewLookup.get(typeName) match {
167       case None => super.getSchema(typeName)
168       case Some(orig) =>
169         val parent = super.getSchema(orig)
170         if (parent == null) {
171           logger.warn(s"Backing schema '$orig' for configured layer view '$typeName' does not exist")
172           null
173         } else {
174           val view = config.layerViewsConfig.get(orig).flatMap(_.find(_.typeName == typeName)).getOrElse {
175             // this should be impossible since we created the lookup from the view config
176             throw new IllegalStateException("Inconsistent layer view config")
177           }
178           KafkaDataStore.createLayerView(parent, view).viewSft
179         }
180     }

195   @throws(classOf[IllegalArgumentException])
196   override protected def preSchemaCreate(sft: SimpleFeatureType): Unit = {
197     // note: kafka doesn't allow slashes in topic names
198     KafkaDataStore.topic(sft) match {
199       case null  => KafkaDataStore.setTopic(sft, s"${config.catalog}-${sft.getTypeName}".replaceAll("/", "-"))
200       case topic if topic.contains("/") => throw new IllegalArgumentException(s"Topic cannot contain '/': $topic")
201       case topic => logger.debug(s"Using user-defined topic [$topic]")
202     }
203     // disable our custom partitioner by default, as it messes with Kafka streams co-partition joining
204     // and it's not required since we switched our keys to be feature ids
205     if (!sft.getUserData.containsKey(KafkaDataStore.PartitioningKey)) {
206       sft.getUserData.put(KafkaDataStore.PartitioningKey, KafkaDataStore.PartitioningDefault)
207     }
208     // remove table sharing as it's not relevant
209     sft.getUserData.remove(TableSharing)
210     sft.getUserData.remove(TableSharingPrefix)
211   }

213   @throws(classOf[IllegalArgumentException])
214   override protected def preSchemaUpdate(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
215     if (layerViewLookup.contains(sft.getTypeName)) {
216       throw new IllegalArgumentException(
217         s"Schema '${sft.getTypeName}' is a read-only view of '${layerViewLookup(sft.getTypeName)}'")
218     }
219     val topic = KafkaDataStore.topic(sft)
220     if (topic == null) {
221       throw new IllegalArgumentException(s"Topic must be defined in user data under '$TopicKey'")
222     } else if (topic != KafkaDataStore.topic(previous)) {
223       if (topic.contains("/")) {
224         throw new IllegalArgumentException(s"Topic cannot contain '/': $topic")
225       }
226       onSchemaDeleted(previous)
227       onSchemaCreated(sft)
228     }
229   }

231   // create kafka topic
232   override protected def onSchemaCreated(sft: SimpleFeatureType): Unit = {
233     val topic = KafkaDataStore.topic(sft)
234     val props = new Properties()
235     props.put(AdminClientConfig.BOOTSTRAP_SERVERS_CONFIG, config.brokers)
236     config.consumers.properties.foreach { case (k, v) => props.put(k, v) }
237     config.producers.properties.foreach { case (k, v) => props.put(k, v) }
238 
239     WithClose(AdminClient.create(props)) { admin =>
240       if (admin.listTopics().names().get.contains(topic)) {
241         logger.warn(
242           s"Topic [$topic] already exists - it may contain invalid data and/or not " +
243               "match the expected topic configuration")
244       } else {
245         val newTopic =
246           new NewTopic(topic, config.topics.partitions, config.topics.replication.toShort)
247               .configs(KafkaDataStore.topicConfig(sft))
248         admin.createTopics(Collections.singletonList(newTopic)).all().get
249       }
250     }
251   }

261   // stop consumers and delete kafka topic
262   override protected def onSchemaDeleted(sft: SimpleFeatureType): Unit = {
263     if (layerViewLookup.contains(sft.getTypeName)) {
264       throw new IllegalArgumentException(
265         s"Schema '${sft.getTypeName}' is a read-only view of '${layerViewLookup(sft.getTypeName)}'")
266     }
267     Option(caches.getIfPresent(sft.getTypeName)).foreach { cache =>
268       cache.close()
269       caches.invalidate(sft.getTypeName)
270     }
271     val topic = KafkaDataStore.topic(sft)
272     val props = new Properties()
273     props.put(AdminClientConfig.BOOTSTRAP_SERVERS_CONFIG, config.brokers)
274     config.consumers.properties.foreach { case (k, v) => props.put(k, v) }
275     config.producers.properties.foreach { case (k, v) => props.put(k, v) }
276 
277     WithClose(AdminClient.create(props)) { admin =>
278       if (admin.listTopics().names().get.contains(topic)) {
279         if (config.onSchemaDeleteTruncate) {
280           logger.info(s"Truncating topic $topic")
281           KafkaTruncateTopic(admin).truncate(topic)
282         } else {
283           logger.info(s"Deleting topic $topic")
284           admin.deleteTopics(Collections.singletonList(topic)).all().get
285         }
286       } else {
287         logger.warn(s"Topic [$topic] does not exist, can't delete it")
288       }
289     }
290   }

314   override private[geomesa] def getFeatureWriter(
315       sft: SimpleFeatureType,
316       transaction: Transaction,
317       filter: Option[Filter]): FlushableFeatureWriter = {
318     if (layerViewLookup.contains(sft.getTypeName)) {
319       throw new IllegalArgumentException(
320         s"Schema '${sft.getTypeName}' is a read-only view of '${layerViewLookup(sft.getTypeName)}'")
321     }
322     val producer = getTransactionalProducer(sft, transaction)
323     val vis = sft.isVisibilityRequired
324     val serializer = serialization.apply(sft)
325     val tags = this.tags(sft.getTypeName)
326     val writer = filter match {
327       case None if vis    => new AppendKafkaFeatureWriter(sft, producer, serializer, tags) with RequiredVisibilityWriter
328       case None           => new AppendKafkaFeatureWriter(sft, producer, serializer, tags)
329       case Some(f) if vis => new ModifyKafkaFeatureWriter(sft, producer, serializer, tags, f) with RequiredVisibilityWriter
330       case Some(f)        => new ModifyKafkaFeatureWriter(sft, producer, serializer, tags, f)
331     }
332     if (config.clearOnStart && cleared.add(sft.getTypeName)) {
333       writer.clear()
334     }
335     writer
336   }

347   private def getTransactionalProducer(sft: SimpleFeatureType, transaction: Transaction): KafkaFeatureProducer = {
348     val useDefaultPartitioning = KafkaDataStore.usesDefaultPartitioning(sft)
349 
350     if (transaction == null || transaction == Transaction.AUTO_COMMIT) {
351       val producer = if (useDefaultPartitioning) { defaultProducer.instance } else { partitionedProducer.instance }
352       return AutoCommitProducer(producer)
353     }
354 
355     val state = transaction.getState(KafkaDataStore.TransactionStateKey)
356     if (state == null) {
357       val partitioner = if (useDefaultPartitioning) { Map.empty } else {
358         Map(PARTITIONER_CLASS_CONFIG -> classOf[GeoMessagePartitioner].getName)
359       }
360       // add kafka transactional id if it's not set, but force acks to "all" as required by kafka
361       val props =
362         Map(TRANSACTIONAL_ID_CONFIG -> UUID.randomUUID().toString) ++
363             partitioner ++
364             config.producers.properties ++
365             Map(ACKS_CONFIG -> "all")
366       val producer = KafkaTransactionState(KafkaDataStore.producer(config.brokers, props))
367       transaction.putState(KafkaDataStore.TransactionStateKey, producer)
368       producer
369     } else {
370       state match {
371         case p: KafkaTransactionState => p
372         case _ => throw new IllegalArgumentException(s"Found non-kafka state in transaction: $state")
373       }
374     }
375   }
