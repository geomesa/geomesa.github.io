115   /**
116    * Create a message consumer for the given feature type. This can be used for guaranteed at-least-once
117    * message processing
118    *
119    * @param typeName type name
120    * @param groupId consumer group id
121    * @param processor message processor
122    * @return

124   def createConsumer(
125       typeName: String,
126       groupId: String,
127       processor: GeoMessageProcessor,
128       errorHandler: Option[ConsumerErrorHandler] = None): Closeable = {
129     val sft = getSchema(typeName)
130     if (sft == null) {
131       throw new IllegalArgumentException(s"Schema '$typeName' does not exist; call `createSchema` first")
132     }
133     val topic = KafkaDataStore.topic(sft)
134     val consumers = {
135       // add group id and
136       // disable read-back so we don't trigger a re-balance listener that messes with group offset tracking
137       val props = config.consumers.properties + (GROUP_ID_CONFIG -> groupId)
138       val conf = config.consumers.copy(properties = props, readBack = None)
139       KafkaDataStore.consumers(config.brokers, topic, conf)
140     }
141     val frequency = java.time.Duration.ofMillis(KafkaDataStore.LoadIntervalProperty.toDuration.get.toMillis)
142     val serializer = serialization.apply(sft)
143     val consumer = new GeoMessageConsumer(consumers, frequency, serializer, processor)
144     consumer.startConsumers(errorHandler)
145     consumer
146   }

148   override def getSchema(typeName: String): SimpleFeatureType = {
149     layerViewLookup.get(typeName) match {
150       case None => super.getSchema(typeName)
151       case Some(orig) =>
152         val parent = super.getSchema(orig)
153         if (parent == null) {
154           logger.warn(s"Backing schema '$orig' for configured layer view '$typeName' does not exist")
155           null
156         } else {
157           val view = config.layerViewsConfig.get(orig).flatMap(_.find(_.typeName == typeName)).getOrElse {
158             // this should be impossible since we created the lookup from the view config
159             throw new IllegalStateException("Inconsistent layer view config")
160           }
161           KafkaDataStore.createLayerView(parent, view).viewSft
162         }
163     }
164   }

178   @throws(classOf[IllegalArgumentException])
179   override protected def preSchemaCreate(sft: SimpleFeatureType): Unit = {
180     // note: kafka doesn't allow slashes in topic names
181     KafkaDataStore.topic(sft) match {
182       case null  => KafkaDataStore.setTopic(sft, s"${config.catalog}-${sft.getTypeName}".replaceAll("/", "-"))
183       case topic if topic.contains("/") => throw new IllegalArgumentException(s"Topic cannot contain '/': $topic")
184       case topic => logger.debug(s"Using user-defined topic [$topic]")
185     }
186     // disable our custom partitioner by default, as it messes with Kafka streams co-partition joining
187     // and it's not required since we switched our keys to be feature ids
188     if (!sft.getUserData.containsKey(KafkaDataStore.PartitioningKey)) {
189       sft.getUserData.put(KafkaDataStore.PartitioningKey, KafkaDataStore.PartitioningDefault)
190     }
191     // remove table sharing as it's not relevant
192     sft.getUserData.remove(TableSharing)
193     sft.getUserData.remove(TableSharingPrefix)
194   }

196   @throws(classOf[IllegalArgumentException])
197   override protected def preSchemaUpdate(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
198     if (layerViewLookup.contains(sft.getTypeName)) {
199       throw new IllegalArgumentException(
200         s"Schema '${sft.getTypeName}' is a read-only view of '${layerViewLookup(sft.getTypeName)}'")
201     }
202     val topic = KafkaDataStore.topic(sft)
203     if (topic == null) {
204       throw new IllegalArgumentException(s"Topic must be defined in user data under '$TopicKey'")
205     } else if (topic != KafkaDataStore.topic(previous)) {
206       if (topic.contains("/")) {
207         throw new IllegalArgumentException(s"Topic cannot contain '/': $topic")
208       }
209       onSchemaDeleted(previous)
210       onSchemaCreated(sft)
211     }
212   }

232     }

289   override private[geomesa] def getFeatureWriter(
290       sft: SimpleFeatureType,
291       transaction: Transaction,
292       filter: Option[Filter]): FlushableFeatureWriter = {
293     if (layerViewLookup.contains(sft.getTypeName)) {
294       throw new IllegalArgumentException(
295         s"Schema '${sft.getTypeName}' is a read-only view of '${layerViewLookup(sft.getTypeName)}'")
296     }
297     val producer = getTransactionalProducer(sft, transaction)
298     val vis = sft.isVisibilityRequired
299     val serializer = serialization.apply(sft)
300     val writer = filter match {
301       case None if vis    => new AppendKafkaFeatureWriter(sft, producer, serializer) with RequiredVisibilityWriter
302       case None           => new AppendKafkaFeatureWriter(sft, producer, serializer)
303       case Some(f) if vis => new ModifyKafkaFeatureWriter(sft, producer, serializer, f) with RequiredVisibilityWriter
304       case Some(f)        => new ModifyKafkaFeatureWriter(sft, producer, serializer, f)
305     }
306     if (config.clearOnStart && cleared.add(sft.getTypeName)) {
307       writer.clear()
308     }
309     writer
310   }

321   private def getTransactionalProducer(sft: SimpleFeatureType, transaction: Transaction): KafkaFeatureProducer = {
322     val useDefaultPartitioning = KafkaDataStore.usesDefaultPartitioning(sft)
323 
324     if (transaction == null || transaction == Transaction.AUTO_COMMIT) {
325       val producer = if (useDefaultPartitioning) { defaultProducer.instance } else { partitionedProducer.instance }
326       return AutoCommitProducer(producer)
327     }
328 
329     val state = transaction.getState(KafkaDataStore.TransactionStateKey)
330     if (state == null) {
331       val partitioner = if (useDefaultPartitioning) { Map.empty } else {
332         Map(PARTITIONER_CLASS_CONFIG -> classOf[GeoMessagePartitioner].getName)
333       }
334       // add kafka transactional id if it's not set, but force acks to "all" as required by kafka
335       val props =
336         Map(TRANSACTIONAL_ID_CONFIG -> UUID.randomUUID().toString) ++
337             partitioner ++
338             config.producers.properties ++
339             Map(ACKS_CONFIG -> "all")
340       val producer = KafkaTransactionState(KafkaDataStore.producer(config.brokers, props))
341       transaction.putState(KafkaDataStore.TransactionStateKey, producer)
342       producer
343     } else {
344       state match {
345         case p: KafkaTransactionState => p
346         case _ => throw new IllegalArgumentException(s"Found non-kafka state in transaction: $state")
347       }
348     }
349   }

451   // creates a consumer and sets to the latest offsets
452   private[kafka] def consumers(
453       brokers: String,
454       topic: String,
455       config: ConsumerConfig): Seq[Consumer[Array[Byte], Array[Byte]]] = {
456     require(config.count > 0, "Number of consumers must be greater than 0")
457 
458     val props = Map(GROUP_ID_CONFIG -> s"${config.groupPrefix}${UUID.randomUUID()}") ++ config.properties
459     lazy val partitions = Collections.newSetFromMap(new ConcurrentHashMap[Int, java.lang.Boolean])
460 
461     logger.debug(s"Creating ${config.count} consumers for topic [$topic] with group-id [${props(GROUP_ID_CONFIG)}]")
462 
463     Seq.fill(config.count) {
464       val consumer = KafkaDataStore.consumer(brokers, props)
465       config.readBack match {
466         case None    => KafkaConsumerVersions.subscribe(consumer, topic)
467         case Some(d) => KafkaConsumerVersions.subscribe(consumer, topic, new ReadBackRebalanceListener(consumer, partitions, d))
468       }
469       consumer
470     }

488   /**
489     * Rebalance listener that seeks the consumer to the an offset based on a read-back duration
490     *
491     * @param consumer consumer
492     * @param partitions shared partition map, to ensure we only read-back once per partition. For subsequent
493     *                   rebalances, we should have committed offsets that will be used
494     * @param readBack duration to read back, or Duration.Inf to go to the beginning
495     */
496   private [kafka] class ReadBackRebalanceListener(consumer: Consumer[Array[Byte], Array[Byte]],
497                                                   partitions: java.util.Set[Int],
498                                                   readBack: Duration)
499       extends ConsumerRebalanceListener with LazyLogging {
500 
501     import scala.collection.JavaConverters._
502 
503     override def onPartitionsRevoked(topicPartitions: java.util.Collection[TopicPartition]): Unit = {}
504 
505     override def onPartitionsAssigned(topicPartitions: java.util.Collection[TopicPartition]): Unit = {
506       topicPartitions.asScala.foreach { tp =>
507         if (partitions.add(tp.partition())) {
508           KafkaConsumerVersions.pause(consumer, tp)
509           try {
510             if (readBack.isFinite) {
511               val offset = Try {
512                 val time = System.currentTimeMillis() - readBack.toMillis
513                 KafkaConsumerVersions.offsetsForTimes(consumer, tp.topic, Seq(tp.partition), time).get(tp.partition)
514               }
515               offset match {
516                 case Success(Some(o)) =>
517                   logger.debug(s"Seeking to offset $o for read-back $readBack on [${tp.topic}:${tp.partition}]")
518                   consumer.seek(tp, o)
519 
520                 case Success(None) =>
521                   logger.debug(s"No prior offset found for read-back $readBack on [${tp.topic}:${tp.partition}], " +
522                       "reading from head of queue")
523 
524                 case Failure(e) =>
525                   logger.warn(s"Error finding initial offset: [${tp.topic}:${tp.partition}], seeking to beginning", e)
526                   KafkaConsumerVersions.seekToBeginning(consumer, tp)
527               }
528             } else {
529               KafkaConsumerVersions.seekToBeginning(consumer, tp)
530             }
531           } finally {
532             KafkaConsumerVersions.resume(consumer, tp)
533           }
534         }
535       }
