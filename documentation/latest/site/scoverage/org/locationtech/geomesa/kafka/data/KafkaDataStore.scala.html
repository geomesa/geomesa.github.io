87   private val caches = Caffeine.newBuilder().build[String, KafkaCacheLoader](new CacheLoader[String, KafkaCacheLoader] {
88     override def load(key: String): KafkaCacheLoader = {
89       if (config.consumers.count < 1) {
90         logger.info("Kafka consumers disabled for this data store instance")
91         KafkaCacheLoader.NoOpLoader
92       } else {
93         val sft = KafkaDataStore.super.getSchema(key)
94         val views = config.layerViewsConfig.getOrElse(key, Seq.empty).map(KafkaDataStore.createLayerView(sft, _))
95         val tags = KafkaDataStore.this.tags(sft.getTypeName)
96         // if the expiry is zero, this will return a NoOpFeatureCache
97         val cache = KafkaFeatureCache(sft, config.indices, views, tags)
98         val topic = KafkaDataStore.topic(sft)
99         val consumers = KafkaDataStore.consumers(config.brokers, topic, config.consumers)
100         val frequency = java.time.Duration.ofMillis(KafkaDataStore.LoadIntervalProperty.toDuration.get.toMillis)
101         val serializer = serialization.apply(sft)
102         val readBack = config.consumers.readBack
103         val expiry = config.indices.expiry
104         val offsetCommitInterval = config.consumers.offsetCommitInterval
105         val loader = new KafkaCacheLoaderImpl(sft, cache, consumers, topic, frequency, offsetCommitInterval, serializer, readBack, expiry, tags)
106         try { loader.start() } catch {
107           case NonFatal(e) =>
108             CloseWithLogging(loader)
109             throw e
110         }
111         loader
112       }
113     }
114   })

146   def createConsumer(
147       typeName: String,
148       groupId: String,
149       processor: GeoMessageProcessor,
150       errorHandler: Option[ConsumerErrorHandler]): Closeable = {
151     val sft = getSchema(typeName)
152     if (sft == null) {
153       throw new IllegalArgumentException(s"Schema '$typeName' does not exist; call `createSchema` first")
154     }
155     val topic = KafkaDataStore.topic(sft)
156     val consumers =
157       KafkaDataStore.consumers(config.brokers, topic,
158         config.consumers.copy(properties = config.consumers.properties + (GROUP_ID_CONFIG -> groupId))) // add group id
159     consumers.foreach(KafkaConsumerVersions.subscribe(_, topic))
160     val frequency = java.time.Duration.ofMillis(KafkaDataStore.LoadIntervalProperty.toDuration.get.toMillis)
161     val serializer = serialization.apply(sft)
162     val consumer = new GeoMessageConsumer(consumers, frequency, serializer, processor)
163     consumer.startConsumers(errorHandler)
164     consumer
165   }

167   override def getSchema(typeName: String): SimpleFeatureType = {
168     layerViewLookup.get(typeName) match {
169       case None => super.getSchema(typeName)
170       case Some(orig) =>
171         val parent = super.getSchema(orig)
172         if (parent == null) {
173           logger.warn(s"Backing schema '$orig' for configured layer view '$typeName' does not exist")
174           null
175         } else {
176           val view = config.layerViewsConfig.get(orig).flatMap(_.find(_.typeName == typeName)).getOrElse {
177             // this should be impossible since we created the lookup from the view config
178             throw new IllegalStateException("Inconsistent layer view config")
179           }
180           KafkaDataStore.createLayerView(parent, view).viewSft
181         }
182     }
183   }

197   @throws(classOf[IllegalArgumentException])
198   override protected def preSchemaCreate(sft: SimpleFeatureType): Unit = {
199     // note: kafka doesn't allow slashes in topic names
200     KafkaDataStore.topic(sft) match {
201       case null  => KafkaDataStore.setTopic(sft, s"${config.catalog}-${sft.getTypeName}".replaceAll("/", "-"))
202       case topic if topic.contains("/") => throw new IllegalArgumentException(s"Topic cannot contain '/': $topic")
203       case topic => logger.debug(s"Using user-defined topic [$topic]")
204     }
205     // disable our custom partitioner by default, as it messes with Kafka streams co-partition joining
206     // and it's not required since we switched our keys to be feature ids
207     if (!sft.getUserData.containsKey(KafkaDataStore.PartitioningKey)) {
208       sft.getUserData.put(KafkaDataStore.PartitioningKey, KafkaDataStore.PartitioningDefault)
209     }
210     // remove table sharing as it's not relevant
211     sft.getUserData.remove(TableSharing)
212     sft.getUserData.remove(TableSharingPrefix)
213   }

215   // create kafka topic
216   override protected def onSchemaCreated(sft: SimpleFeatureType): Unit = {
217     val topic = KafkaDataStore.topic(sft)
218     adminClientOp(config) { admin =>
219       if (admin.listTopics().names().get.contains(topic)) {
220         logger.warn(
221           s"Topic [$topic] already exists - it may contain invalid data and/or not " +
222               "match the expected topic configuration")
223       } else {
224         val newTopic =
225           new NewTopic(topic, config.topics.partitions, config.topics.replication.toShort)
226               .configs(KafkaDataStore.topicConfig(sft))
227         admin.createTopics(Collections.singletonList(newTopic)).all().get
228       }
229     }
230     metadata.insert(sft.getTypeName,
231       Map("avro-schema-native" -> Set(SerializationOption.NativeCollections), "avro-schema" -> Set.empty).map { case (k, opts) =>
232         k -> AvroSerialization(sft, GeoMessageSerializer.DefaultOpts ++ opts).schema.toString()
233       }
234     )
235   }

237   @throws(classOf[IllegalArgumentException])
238   override protected def preSchemaUpdate(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
239     requireNotLayerView(sft.getTypeName)
240     val topic = KafkaDataStore.topic(sft)
241     if (topic == null) {
242       throw new IllegalArgumentException(s"Topic must be defined in user data under '$TopicKey'")
243     } else if (topic != KafkaDataStore.topic(previous)) {
244       if (topic.contains("/")) {
245         throw new IllegalArgumentException(s"Topic cannot contain '/': $topic")
246       }
247       onSchemaDeleted(previous)
248       onSchemaCreated(sft)
249     }
250   }

256   // stop consumers and delete kafka topic
257   override protected def onSchemaDeleted(sft: SimpleFeatureType): Unit = {
258     requireNotLayerView(sft.getTypeName)
259     closeCache(sft.getTypeName)
260     val topic = KafkaDataStore.topic(sft)
261     adminClientOp(config) { admin =>
262       if (admin.listTopics().names().get.contains(topic)) {
263         if (config.onSchemaDeleteTruncate) {
264           logger.debug(s"Truncating topic $topic")
265           TruncateTopic.apply(admin, topic)
266         } else {
267           logger.debug(s"Deleting topic $topic")
268           admin.deleteTopics(Collections.singletonList(topic)).all().get
269         }
270       } else {
271         logger.warn(s"Topic [$topic] does not exist, can't delete it")
272       }
273     }
274   }

298   override private[geomesa] def getFeatureWriter(
299       sft: SimpleFeatureType,
300       transaction: Transaction,
301       filter: Option[Filter]): FlushableFeatureWriter = {
302     requireNotLayerView(sft.getTypeName)
303     val producer = getTransactionalProducer(sft, transaction)
304     val vis = sft.isVisibilityRequired
305     val serializer = serialization.apply(sft)
306     val tags = this.tags(sft.getTypeName)
307     val writer = filter match {
308       case None if vis    => new AppendKafkaFeatureWriter(sft, producer, serializer, tags) with RequiredVisibilityWriter
309       case None           => new AppendKafkaFeatureWriter(sft, producer, serializer, tags)
310       case Some(f) if vis => new ModifyKafkaFeatureWriter(sft, producer, serializer, tags, f) with RequiredVisibilityWriter
311       case Some(f)        => new ModifyKafkaFeatureWriter(sft, producer, serializer, tags, f)
312     }
313     if (config.clearOnStart && cleared.add(sft.getTypeName)) {
314       writer.clear()
315     }
316     writer
317   }

328   private def getTransactionalProducer(sft: SimpleFeatureType, transaction: Transaction): KafkaFeatureProducer = {
329     val useDefaultPartitioning = KafkaDataStore.usesDefaultPartitioning(sft)
330 
331     if (transaction == null || transaction == Transaction.AUTO_COMMIT) {
332       val producer = if (useDefaultPartitioning) { defaultProducer.instance } else { partitionedProducer.instance }
333       return AutoCommitProducer(producer)
334     }
335 
336     val state = transaction.getState(KafkaDataStore.TransactionStateKey)
337     if (state == null) {
338       val partitioner = if (useDefaultPartitioning) { Map.empty } else {
339         Map(PARTITIONER_CLASS_CONFIG -> classOf[GeoMessagePartitioner].getName)
340       }
341       // add kafka transactional id if it's not set, but force acks to "all" as required by kafka
342       val props =
343         Map(TRANSACTIONAL_ID_CONFIG -> UUID.randomUUID().toString) ++
344             partitioner ++
345             config.producers.properties ++
346             Map(ACKS_CONFIG -> "all")
347       val producer = KafkaTransactionState(KafkaDataStore.producer(config.brokers, props))
348       transaction.putState(KafkaDataStore.TransactionStateKey, producer)
349       producer
350     } else {
351       state match {
352         case p: KafkaTransactionState => p
353         case _ => throw new IllegalArgumentException(s"Found non-kafka state in transaction: $state")
354       }
355     }
356   }

360    *
361    * @param typeName type name
362    * @return

364   private def cache(typeName: String): KafkaFeatureCache = {
365     layerViewLookup.get(typeName) match {
366       case None => caches.get(typeName).cache
367       case Some(orig) =>
368         caches.get(orig).cache.views.find(_.sft.getTypeName == typeName).getOrElse {
369           throw new IllegalStateException(
370             s"Could not find layer view for typeName '$typeName' in cache ${caches.get(orig)}")
371         }
372     }
373   }
