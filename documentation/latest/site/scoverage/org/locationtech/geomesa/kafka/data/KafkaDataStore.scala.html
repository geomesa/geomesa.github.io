87   private val caches = Caffeine.newBuilder().build[String, KafkaCacheLoader](new CacheLoader[String, KafkaCacheLoader] {
88     override def load(key: String): KafkaCacheLoader = {
89       if (config.consumers.count < 1) {
90         logger.info("Kafka consumers disabled for this data store instance")
91         KafkaCacheLoader.NoOpLoader
92       } else {
93         val sft = KafkaDataStore.super.getSchema(key)
94         val views = config.layerViewsConfig.getOrElse(key, Seq.empty).map(KafkaDataStore.createLayerView(sft, _))
95         val tags = KafkaDataStore.this.tags(sft.getTypeName)
96         // if the expiry is zero, this will return a NoOpFeatureCache
97         val cache = KafkaFeatureCache(sft, config.indices, views, tags)
98         val topic = KafkaDataStore.topic(sft)
99         val consumers = KafkaDataStore.consumers(config.brokers, topic, config.consumers)
100         val frequency = java.time.Duration.ofMillis(KafkaDataStore.LoadIntervalProperty.toDuration.get.toMillis)
101         val serializer = serialization.apply(sft)
102         val readBack = config.consumers.readBack
103         val expiry = config.indices.expiry
104         val offsetCommitInterval = config.consumers.offsetCommitInterval
105         val loader = new KafkaCacheLoaderImpl(sft, cache, consumers, topic, frequency, offsetCommitInterval, serializer, readBack, expiry, tags)
106         try { loader.start() } catch {
107           case NonFatal(e) =>
108             CloseWithLogging(loader)
109             throw e
110         }
111         loader
112       }
113     }

146   def createConsumer(
147       typeName: String,
148       groupId: String,
149       processor: GeoMessageProcessor,
150       errorHandler: Option[ConsumerErrorHandler]): Closeable = {
151     val sft = getSchema(typeName)
152     if (sft == null) {
153       throw new IllegalArgumentException(s"Schema '$typeName' does not exist; call `createSchema` first")
154     }
155     val topic = KafkaDataStore.topic(sft)
156     val consumers =
157       KafkaDataStore.consumers(config.brokers, topic,
158         config.consumers.copy(properties = config.consumers.properties + (GROUP_ID_CONFIG -> groupId))) // add group id
159     consumers.foreach(KafkaConsumerVersions.subscribe(_, topic))
160     val frequency = java.time.Duration.ofMillis(KafkaDataStore.LoadIntervalProperty.toDuration.get.toMillis)
161     val serializer = serialization.apply(sft)
162     val consumer = new GeoMessageConsumer(consumers, frequency, serializer, processor)
163     consumer.startConsumers(errorHandler)
164     consumer
165   }

167   override def getSchema(typeName: String): SimpleFeatureType = {
168     layerViewLookup.get(typeName) match {
169       case None => super.getSchema(typeName)
170       case Some(orig) =>
171         val parent = super.getSchema(orig)
172         if (parent == null) {
173           logger.warn(s"Backing schema '$orig' for configured layer view '$typeName' does not exist")
174           null
175         } else {
176           val view = config.layerViewsConfig.get(orig).flatMap(_.find(_.typeName == typeName)).getOrElse {
177             // this should be impossible since we created the lookup from the view config
178             throw new IllegalStateException("Inconsistent layer view config")
179           }
180           KafkaDataStore.createLayerView(parent, view).viewSft
181         }
182     }
183   }

197   @throws(classOf[IllegalArgumentException])
198   override protected def preSchemaCreate(sft: SimpleFeatureType): Unit = {
199     // note: kafka doesn't allow slashes in topic names
200     KafkaDataStore.topic(sft) match {
201       case null  => KafkaDataStore.setTopic(sft, s"${config.catalog}-${sft.getTypeName}".replaceAll("/", "-"))
202       case topic if topic.contains("/") => throw new IllegalArgumentException(s"Topic cannot contain '/': $topic")
203       case topic => logger.debug(s"Using user-defined topic [$topic]")
204     }
205     // disable our custom partitioner by default, as it messes with Kafka streams co-partition joining
206     // and it's not required since we switched our keys to be feature ids
207     if (!sft.getUserData.containsKey(KafkaDataStore.PartitioningKey)) {
208       sft.getUserData.put(KafkaDataStore.PartitioningKey, KafkaDataStore.PartitioningDefault)
209     }
210     // remove table sharing as it's not relevant
211     sft.getUserData.remove(TableSharing)
212     sft.getUserData.remove(TableSharingPrefix)
213   }

215   @throws(classOf[IllegalArgumentException])
216   override protected def preSchemaUpdate(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
217     if (layerViewLookup.contains(sft.getTypeName)) {
218       throw new IllegalArgumentException(
219         s"Schema '${sft.getTypeName}' is a read-only view of '${layerViewLookup(sft.getTypeName)}'")
220     }
221     val topic = KafkaDataStore.topic(sft)
222     if (topic == null) {
223       throw new IllegalArgumentException(s"Topic must be defined in user data under '$TopicKey'")
224     } else if (topic != KafkaDataStore.topic(previous)) {
225       if (topic.contains("/")) {
226         throw new IllegalArgumentException(s"Topic cannot contain '/': $topic")
227       }
228       onSchemaDeleted(previous)
229       onSchemaCreated(sft)
230     }
231   }

233   // create kafka topic
234   override protected def onSchemaCreated(sft: SimpleFeatureType): Unit = {
235     val topic = KafkaDataStore.topic(sft)
236     val props = new Properties()
237     props.put(AdminClientConfig.BOOTSTRAP_SERVERS_CONFIG, config.brokers)
238     config.consumers.properties.foreach { case (k, v) => props.put(k, v) }
239     config.producers.properties.foreach { case (k, v) => props.put(k, v) }
240 
241     WithClose(AdminClient.create(props)) { admin =>
242       if (admin.listTopics().names().get.contains(topic)) {
243         logger.warn(
244           s"Topic [$topic] already exists - it may contain invalid data and/or not " +
245               "match the expected topic configuration")
246       } else {
247         val newTopic =
248           new NewTopic(topic, config.topics.partitions, config.topics.replication.toShort)
249               .configs(KafkaDataStore.topicConfig(sft))
250         admin.createTopics(Collections.singletonList(newTopic)).all().get
251       }
252     }
253   }

263   // stop consumers and delete kafka topic
264   override protected def onSchemaDeleted(sft: SimpleFeatureType): Unit = {
265     if (layerViewLookup.contains(sft.getTypeName)) {
266       throw new IllegalArgumentException(
267         s"Schema '${sft.getTypeName}' is a read-only view of '${layerViewLookup(sft.getTypeName)}'")
268     }
269     Option(caches.getIfPresent(sft.getTypeName)).foreach { cache =>
270       cache.close()
271       caches.invalidate(sft.getTypeName)
272     }
273     val topic = KafkaDataStore.topic(sft)
274     val props = new Properties()
275     props.put(AdminClientConfig.BOOTSTRAP_SERVERS_CONFIG, config.brokers)
276     config.consumers.properties.foreach { case (k, v) => props.put(k, v) }
277     config.producers.properties.foreach { case (k, v) => props.put(k, v) }
278 
279     WithClose(AdminClient.create(props)) { admin =>
280       if (admin.listTopics().names().get.contains(topic)) {
281         admin.deleteTopics(Collections.singletonList(topic)).all().get
282       } else {
283         logger.warn(s"Topic [$topic] does not exist, can't delete it")
284       }
285     }

310   override private[geomesa] def getFeatureWriter(
311       sft: SimpleFeatureType,
312       transaction: Transaction,
313       filter: Option[Filter]): FlushableFeatureWriter = {
314     if (layerViewLookup.contains(sft.getTypeName)) {
315       throw new IllegalArgumentException(
316         s"Schema '${sft.getTypeName}' is a read-only view of '${layerViewLookup(sft.getTypeName)}'")
317     }
318     val producer = getTransactionalProducer(sft, transaction)
319     val vis = sft.isVisibilityRequired
320     val serializer = serialization.apply(sft)
321     val tags = this.tags(sft.getTypeName)
322     val writer = filter match {
323       case None if vis    => new AppendKafkaFeatureWriter(sft, producer, serializer, tags) with RequiredVisibilityWriter
324       case None           => new AppendKafkaFeatureWriter(sft, producer, serializer, tags)
325       case Some(f) if vis => new ModifyKafkaFeatureWriter(sft, producer, serializer, tags, f) with RequiredVisibilityWriter
326       case Some(f)        => new ModifyKafkaFeatureWriter(sft, producer, serializer, tags, f)
327     }
328     if (config.clearOnStart && cleared.add(sft.getTypeName)) {
329       writer.clear()
330     }
331     writer
332   }

344   private def getTransactionalProducer(sft: SimpleFeatureType, transaction: Transaction): KafkaFeatureProducer = {
345     val useDefaultPartitioning = KafkaDataStore.usesDefaultPartitioning(sft)
346 
347     if (transaction == null || transaction == Transaction.AUTO_COMMIT) {
348       val producer = if (useDefaultPartitioning) { defaultProducer.instance } else { partitionedProducer.instance }
349       return AutoCommitProducer(producer)
350     }
351 
352     val state = transaction.getState(KafkaDataStore.TransactionStateKey)
353     if (state == null) {
354       val partitioner = if (useDefaultPartitioning) { Map.empty } else {
355         Map(PARTITIONER_CLASS_CONFIG -> classOf[GeoMessagePartitioner].getName)
356       }
357       // add kafka transactional id if it's not set, but force acks to "all" as required by kafka
358       val props =
359         Map(TRANSACTIONAL_ID_CONFIG -> UUID.randomUUID().toString) ++
360             partitioner ++
361             config.producers.properties ++
362             Map(ACKS_CONFIG -> "all")
363       val producer = KafkaTransactionState(KafkaDataStore.producer(config.brokers, props))
364       transaction.putState(KafkaDataStore.TransactionStateKey, producer)
365       producer
366     } else {
367       state match {
368         case p: KafkaTransactionState => p
369         case _ => throw new IllegalArgumentException(s"Found non-kafka state in transaction: $state")
370       }
371     }
372   }

376    *
377    * @param typeName type name
