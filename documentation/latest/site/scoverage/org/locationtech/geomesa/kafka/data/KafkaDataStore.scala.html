137   def createConsumer(
138       typeName: String,
139       groupId: String,
140       processor: GeoMessageProcessor,
141       errorHandler: Option[ConsumerErrorHandler]): Closeable = {
142     val sft = getSchema(typeName)
143     if (sft == null) {
144       throw new IllegalArgumentException(s"Schema '$typeName' does not exist; call `createSchema` first")
145     }
146     val topic = KafkaDataStore.topic(sft)
147     val consumers = {
148       // add group id and
149       // disable read-back so we don't trigger a re-balance listener that messes with group offset tracking
150       val props = config.consumers.properties + (GROUP_ID_CONFIG -> groupId)
151       val conf = config.consumers.copy(properties = props, readBack = None)
152       KafkaDataStore.consumers(config.brokers, topic, conf)
153     }
154     val frequency = java.time.Duration.ofMillis(KafkaDataStore.LoadIntervalProperty.toDuration.get.toMillis)
155     val serializer = serialization.apply(sft)
156     val consumer = new GeoMessageConsumer(consumers, frequency, serializer, processor)
157     consumer.startConsumers(errorHandler)
158     consumer
159   }

162     layerViewLookup.get(typeName) match {
163       case None => super.getSchema(typeName)

165         val parent = super.getSchema(orig)

192   override protected def preSchemaCreate(sft: SimpleFeatureType): Unit = {
193     // note: kafka doesn't allow slashes in topic names
194     KafkaDataStore.topic(sft) match {
195       case null  => KafkaDataStore.setTopic(sft, s"${config.catalog}-${sft.getTypeName}".replaceAll("/", "-"))
196       case topic if topic.contains("/") => throw new IllegalArgumentException(s"Topic cannot contain '/': $topic")
197       case topic => logger.debug(s"Using user-defined topic [$topic]")
198     }
199     // disable our custom partitioner by default, as it messes with Kafka streams co-partition joining
200     // and it's not required since we switched our keys to be feature ids
201     if (!sft.getUserData.containsKey(KafkaDataStore.PartitioningKey)) {
202       sft.getUserData.put(KafkaDataStore.PartitioningKey, KafkaDataStore.PartitioningDefault)
203     }
204     // remove table sharing as it's not relevant
205     sft.getUserData.remove(TableSharing)
206     sft.getUserData.remove(TableSharingPrefix)
207   }

209   @throws(classOf[IllegalArgumentException])
210   override protected def preSchemaUpdate(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
211     if (layerViewLookup.contains(sft.getTypeName)) {
212       throw new IllegalArgumentException(
213         s"Schema '${sft.getTypeName}' is a read-only view of '${layerViewLookup(sft.getTypeName)}'")
214     }
215     val topic = KafkaDataStore.topic(sft)
216     if (topic == null) {
217       throw new IllegalArgumentException(s"Topic must be defined in user data under '$TopicKey'")
218     } else if (topic != KafkaDataStore.topic(previous)) {
219       if (topic.contains("/")) {
220         throw new IllegalArgumentException(s"Topic cannot contain '/': $topic")
221       }
222       onSchemaDeleted(previous)
223       onSchemaCreated(sft)
224     }
225   }

302   override private[geomesa] def getFeatureWriter(
303       sft: SimpleFeatureType,
304       transaction: Transaction,
305       filter: Option[Filter]): FlushableFeatureWriter = {
306     if (layerViewLookup.contains(sft.getTypeName)) {
307       throw new IllegalArgumentException(
308         s"Schema '${sft.getTypeName}' is a read-only view of '${layerViewLookup(sft.getTypeName)}'")
309     }
310     val producer = getTransactionalProducer(sft, transaction)
311     val vis = sft.isVisibilityRequired
312     val serializer = serialization.apply(sft)
313     val writer = filter match {
314       case None if vis    => new AppendKafkaFeatureWriter(sft, producer, serializer) with RequiredVisibilityWriter
315       case None           => new AppendKafkaFeatureWriter(sft, producer, serializer)
316       case Some(f) if vis => new ModifyKafkaFeatureWriter(sft, producer, serializer, f) with RequiredVisibilityWriter
317       case Some(f)        => new ModifyKafkaFeatureWriter(sft, producer, serializer, f)
318     }
319     if (config.clearOnStart && cleared.add(sft.getTypeName)) {
320       writer.clear()
321     }
322     writer
323   }

334   private def getTransactionalProducer(sft: SimpleFeatureType, transaction: Transaction): KafkaFeatureProducer = {
335     val useDefaultPartitioning = KafkaDataStore.usesDefaultPartitioning(sft)
336 
337     if (transaction == null || transaction == Transaction.AUTO_COMMIT) {
338       val producer = if (useDefaultPartitioning) { defaultProducer.instance } else { partitionedProducer.instance }
339       return AutoCommitProducer(producer)
340     }
341 
342     val state = transaction.getState(KafkaDataStore.TransactionStateKey)
343     if (state == null) {
344       val partitioner = if (useDefaultPartitioning) { Map.empty } else {
345         Map(PARTITIONER_CLASS_CONFIG -> classOf[GeoMessagePartitioner].getName)
346       }
347       // add kafka transactional id if it's not set, but force acks to "all" as required by kafka
348       val props =
349         Map(TRANSACTIONAL_ID_CONFIG -> UUID.randomUUID().toString) ++
350             partitioner ++
351             config.producers.properties ++
352             Map(ACKS_CONFIG -> "all")
353       val producer = KafkaTransactionState(KafkaDataStore.producer(config.brokers, props))
354       transaction.putState(KafkaDataStore.TransactionStateKey, producer)
355       producer
356     } else {
357       state match {
358         case p: KafkaTransactionState => p
359         case _ => throw new IllegalArgumentException(s"Found non-kafka state in transaction: $state")
360       }
361     }
362   }

466       brokers: String,

468       config: ConsumerConfig): Seq[Consumer[Array[Byte], Array[Byte]]] = {
469     require(config.count > 0, "Number of consumers must be greater than 0")

479         case None    => KafkaConsumerVersions.subscribe(consumer, topic)

481       }

501   /**
502     * Rebalance listener that seeks the consumer to the an offset based on a read-back duration
503     *
504     * @param consumer consumer
505     * @param partitions shared partition map, to ensure we only read-back once per partition. For subsequent
506     *                   rebalances, we should have committed offsets that will be used
507     * @param readBack duration to read back, or Duration.Inf to go to the beginning
508     */
509   private [kafka] class ReadBackRebalanceListener(consumer: Consumer[Array[Byte], Array[Byte]],
510                                                   partitions: java.util.Set[Int],
511                                                   readBack: Duration)
512       extends ConsumerRebalanceListener with LazyLogging {
513 
514     import scala.collection.JavaConverters._
515 
516     override def onPartitionsRevoked(topicPartitions: java.util.Collection[TopicPartition]): Unit = {}
517 
518     override def onPartitionsAssigned(topicPartitions: java.util.Collection[TopicPartition]): Unit = {
519       topicPartitions.asScala.foreach { tp =>
520         if (partitions.add(tp.partition())) {
521           KafkaConsumerVersions.pause(consumer, tp)
522           try {
523             if (readBack.isFinite) {
524               val offset = Try {
525                 val time = System.currentTimeMillis() - readBack.toMillis
526                 KafkaConsumerVersions.offsetsForTimes(consumer, tp.topic, Seq(tp.partition), time).get(tp.partition)
527               }
528               offset match {
529                 case Success(Some(o)) =>
530                   logger.debug(s"Seeking to offset $o for read-back $readBack on [${tp.topic}:${tp.partition}]")
531                   consumer.seek(tp, o)
532 
533                 case Success(None) =>
534                   logger.debug(s"No prior offset found for read-back $readBack on [${tp.topic}:${tp.partition}], " +
