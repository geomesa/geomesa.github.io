85   private val caches = Caffeine.newBuilder().build[String, KafkaCacheLoader](new CacheLoader[String, KafkaCacheLoader] {
86     override def load(key: String): KafkaCacheLoader = {
87       if (config.consumers.count < 1) {
88         logger.info("Kafka consumers disabled for this data store instance")
89         KafkaCacheLoader.NoOpLoader
90       } else {
91         val sft = KafkaDataStore.super.getSchema(key)
92         val views = config.layerViewsConfig.getOrElse(key, Seq.empty).map(KafkaDataStore.createLayerView(sft, _))
93         // if the expiry is zero, this will return a NoOpFeatureCache
94         val cache = KafkaFeatureCache(sft, config.indices, views, config.metrics)
95         val topic = KafkaDataStore.topic(sft)
96         val consumers = KafkaDataStore.consumers(config.brokers, topic, config.consumers)
97         val frequency = java.time.Duration.ofMillis(KafkaDataStore.LoadIntervalProperty.toDuration.get.toMillis)
98         val serializer = serialization.apply(sft)
99         val readBack = config.consumers.readBack
100         val expiry = config.indices.expiry
101         val offsetCommitInterval = config.consumers.offsetCommitInterval
102         val loader = new KafkaCacheLoaderImpl(sft, cache, consumers, topic, frequency, offsetCommitInterval, serializer, readBack, expiry)
103         try { loader.start() } catch {
104           case NonFatal(e) =>
105             CloseWithLogging(loader)
106             throw e
107         }
108         loader
109       }
110     }

143   def createConsumer(
144       typeName: String,
145       groupId: String,
146       processor: GeoMessageProcessor,
147       errorHandler: Option[ConsumerErrorHandler]): Closeable = {
148     val sft = getSchema(typeName)
149     if (sft == null) {
150       throw new IllegalArgumentException(s"Schema '$typeName' does not exist; call `createSchema` first")
151     }
152     val topic = KafkaDataStore.topic(sft)
153     val consumers =
154       KafkaDataStore.consumers(config.brokers, topic,
155         config.consumers.copy(properties = config.consumers.properties + (GROUP_ID_CONFIG -> groupId))) // add group id
156     consumers.foreach(KafkaConsumerVersions.subscribe(_, topic))
157     val frequency = java.time.Duration.ofMillis(KafkaDataStore.LoadIntervalProperty.toDuration.get.toMillis)
158     val serializer = serialization.apply(sft)
159     val consumer = new GeoMessageConsumer(consumers, frequency, serializer, processor)
160     consumer.startConsumers(errorHandler)
161     consumer
162   }

164   override def getSchema(typeName: String): SimpleFeatureType = {
165     layerViewLookup.get(typeName) match {
166       case None => super.getSchema(typeName)
167       case Some(orig) =>
168         val parent = super.getSchema(orig)
169         if (parent == null) {
170           logger.warn(s"Backing schema '$orig' for configured layer view '$typeName' does not exist")
171           null
172         } else {
173           val view = config.layerViewsConfig.get(orig).flatMap(_.find(_.typeName == typeName)).getOrElse {
174             // this should be impossible since we created the lookup from the view config
175             throw new IllegalStateException("Inconsistent layer view config")
176           }
177           KafkaDataStore.createLayerView(parent, view).viewSft
178         }
179     }
180   }

194   @throws(classOf[IllegalArgumentException])
195   override protected def preSchemaCreate(sft: SimpleFeatureType): Unit = {
196     // note: kafka doesn't allow slashes in topic names
197     KafkaDataStore.topic(sft) match {
198       case null  => KafkaDataStore.setTopic(sft, s"${config.catalog}-${sft.getTypeName}".replaceAll("/", "-"))
199       case topic if topic.contains("/") => throw new IllegalArgumentException(s"Topic cannot contain '/': $topic")
200       case topic => logger.debug(s"Using user-defined topic [$topic]")
201     }
202     // disable our custom partitioner by default, as it messes with Kafka streams co-partition joining
203     // and it's not required since we switched our keys to be feature ids
204     if (!sft.getUserData.containsKey(KafkaDataStore.PartitioningKey)) {
205       sft.getUserData.put(KafkaDataStore.PartitioningKey, KafkaDataStore.PartitioningDefault)
206     }
207     // remove table sharing as it's not relevant
208     sft.getUserData.remove(TableSharing)
209     sft.getUserData.remove(TableSharingPrefix)
210   }

212   @throws(classOf[IllegalArgumentException])
213   override protected def preSchemaUpdate(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
214     if (layerViewLookup.contains(sft.getTypeName)) {
215       throw new IllegalArgumentException(
216         s"Schema '${sft.getTypeName}' is a read-only view of '${layerViewLookup(sft.getTypeName)}'")
217     }
218     val topic = KafkaDataStore.topic(sft)
219     if (topic == null) {
220       throw new IllegalArgumentException(s"Topic must be defined in user data under '$TopicKey'")
221     } else if (topic != KafkaDataStore.topic(previous)) {
222       if (topic.contains("/")) {
223         throw new IllegalArgumentException(s"Topic cannot contain '/': $topic")
224       }
225       onSchemaDeleted(previous)
226       onSchemaCreated(sft)
227     }
228   }

230   // create kafka topic
231   override protected def onSchemaCreated(sft: SimpleFeatureType): Unit = {
232     val topic = KafkaDataStore.topic(sft)
233     val props = new Properties()
234     props.put(AdminClientConfig.BOOTSTRAP_SERVERS_CONFIG, config.brokers)
235     config.consumers.properties.foreach { case (k, v) => props.put(k, v) }
236     config.producers.properties.foreach { case (k, v) => props.put(k, v) }
237 
238     WithClose(AdminClient.create(props)) { admin =>
239       if (admin.listTopics().names().get.contains(topic)) {
240         logger.warn(
241           s"Topic [$topic] already exists - it may contain invalid data and/or not " +
242               "match the expected topic configuration")
243       } else {
244         val newTopic =
245           new NewTopic(topic, config.topics.partitions, config.topics.replication.toShort)
246               .configs(KafkaDataStore.topicConfig(sft))
247         admin.createTopics(Collections.singletonList(newTopic)).all().get
248       }
249     }
250   }

260   // stop consumers and delete kafka topic
261   override protected def onSchemaDeleted(sft: SimpleFeatureType): Unit = {
262     if (layerViewLookup.contains(sft.getTypeName)) {
263       throw new IllegalArgumentException(
264         s"Schema '${sft.getTypeName}' is a read-only view of '${layerViewLookup(sft.getTypeName)}'")
265     }
266     Option(caches.getIfPresent(sft.getTypeName)).foreach { cache =>
267       cache.close()
268       caches.invalidate(sft.getTypeName)
269     }
270     val topic = KafkaDataStore.topic(sft)
271     val props = new Properties()
272     props.put(AdminClientConfig.BOOTSTRAP_SERVERS_CONFIG, config.brokers)
273     config.consumers.properties.foreach { case (k, v) => props.put(k, v) }
274     config.producers.properties.foreach { case (k, v) => props.put(k, v) }
275 
276     WithClose(AdminClient.create(props)) { admin =>
277       if (admin.listTopics().names().get.contains(topic)) {
278         admin.deleteTopics(Collections.singletonList(topic)).all().get
279       } else {
280         logger.warn(s"Topic [$topic] does not exist, can't delete it")
281       }
282     }

307   override private[geomesa] def getFeatureWriter(
308       sft: SimpleFeatureType,
309       transaction: Transaction,
310       filter: Option[Filter]): FlushableFeatureWriter = {
311     if (layerViewLookup.contains(sft.getTypeName)) {
312       throw new IllegalArgumentException(
313         s"Schema '${sft.getTypeName}' is a read-only view of '${layerViewLookup(sft.getTypeName)}'")
314     }
315     val producer = getTransactionalProducer(sft, transaction)
316     val vis = sft.isVisibilityRequired
317     val serializer = serialization.apply(sft)
318     val writer = filter match {
319       case None if vis    => new AppendKafkaFeatureWriter(sft, producer, serializer) with RequiredVisibilityWriter
320       case None           => new AppendKafkaFeatureWriter(sft, producer, serializer)
321       case Some(f) if vis => new ModifyKafkaFeatureWriter(sft, producer, serializer, f) with RequiredVisibilityWriter
322       case Some(f)        => new ModifyKafkaFeatureWriter(sft, producer, serializer, f)
323     }
324     if (config.clearOnStart && cleared.add(sft.getTypeName)) {
325       writer.clear()
326     }
327     writer
328   }

339   private def getTransactionalProducer(sft: SimpleFeatureType, transaction: Transaction): KafkaFeatureProducer = {
340     val useDefaultPartitioning = KafkaDataStore.usesDefaultPartitioning(sft)
341 
342     if (transaction == null || transaction == Transaction.AUTO_COMMIT) {
343       val producer = if (useDefaultPartitioning) { defaultProducer.instance } else { partitionedProducer.instance }
344       return AutoCommitProducer(producer)
345     }
346 
347     val state = transaction.getState(KafkaDataStore.TransactionStateKey)
348     if (state == null) {
349       val partitioner = if (useDefaultPartitioning) { Map.empty } else {
350         Map(PARTITIONER_CLASS_CONFIG -> classOf[GeoMessagePartitioner].getName)
351       }
352       // add kafka transactional id if it's not set, but force acks to "all" as required by kafka
353       val props =
354         Map(TRANSACTIONAL_ID_CONFIG -> UUID.randomUUID().toString) ++
355             partitioner ++
356             config.producers.properties ++
357             Map(ACKS_CONFIG -> "all")
358       val producer = KafkaTransactionState(KafkaDataStore.producer(config.brokers, props))
359       transaction.putState(KafkaDataStore.TransactionStateKey, producer)
360       producer
361     } else {
362       state match {
363         case p: KafkaTransactionState => p
364         case _ => throw new IllegalArgumentException(s"Found non-kafka state in transaction: $state")
365       }
366     }
367   }
