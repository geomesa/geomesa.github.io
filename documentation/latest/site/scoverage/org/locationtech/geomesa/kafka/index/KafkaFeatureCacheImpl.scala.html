25 /**
26  * Feature cache implementation
27  *
28  * @param sft simple feature type
29  * @param config index config
30  * @param layerViews layer views
31  */
32 class KafkaFeatureCacheImpl(sft: SimpleFeatureType, config: IndexConfig, layerViews: Seq[LayerView] = Seq.empty, tags: Tags = Tags.empty())
33     extends KafkaFeatureCache with FeatureExpiration {
34 
35   import org.locationtech.geomesa.kafka.index.KafkaFeatureCacheImpl.MetricsPrefix
36   import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
37 
38   // keeps location and expiry keyed by feature ID (we need a way to retrieve a feature based on ID for
39   // update/delete operations). to reduce contention, we never iterate over this map
40   private val state = GaugeUtils.mapSizeGauge(s"$MetricsPrefix.size", tags, new ConcurrentHashMap[String, FeatureState]())
41 
42   private val spatialIndex = createIndex(sft)
43 
44   private val factory = FeatureStateFactory(sft, spatialIndex, config.expiry, this, config.executor)
45 
46   override val views: Seq[KafkaFeatureCacheView] =
47     layerViews.map(view => KafkaFeatureCacheView(view, createIndex(view.viewSft)))
48 
49   private val expirations = Metrics.counter(s"$MetricsPrefix.expirations", tags)
50 
51   logger.debug(s"Initialized KafkaFeatureCache with factory $factory and support $spatialIndex")
52 
53   /**
54     * Note: this method is not thread-safe. The `state` and `index` can get out of sync if the same feature
55     * is updated simultaneously from two different threads
56     *
57     * In our usage, this isn't a problem, as a given feature ID is always operated on by a single thread
58     * due to kafka consumer partitioning
59     */
60   override def put(feature: SimpleFeature): Unit = {
61     if (feature.getDefaultGeometry == null) {
62       logger.warn(s"Null geometry detected for feature ${feature.getID}. Skipping loading into cache.")
63       return
64     }
65     val featureState = factory.createState(feature)
66     logger.trace(s"${featureState.id} adding feature $featureState")
67     val old = state.put(featureState.id, featureState)
68     if (old == null) {
69       featureState.insertIntoIndex()
70       views.foreach(_.put(feature))
71     } else if (old.time <= featureState.time) {
72       logger.trace(s"${featureState.id} removing old feature")
73       old.removeFromIndex()
74       featureState.insertIntoIndex()
75       views.foreach { view =>
76         view.remove(featureState.id)
77         view.put(feature)
78       }
79     } else {
80       logger.trace(s"${featureState.id} ignoring out of sequence feature")
81       if (!state.replace(featureState.id, featureState, old)) {
82         logger.warn(s"${featureState.id} detected inconsistent state... spatial index may be incorrect")
83         old.removeFromIndex()
84         views.foreach(_.remove(featureState.id))
85       }
86     }
87     logger.trace(s"Current index size: ${state.size()}/${spatialIndex.size()}")
88   }
89 
90   /**
91     * Note: this method is not thread-safe. The `state` and `index` can get out of sync if the same feature
92     * is updated simultaneously from two different threads
93     *
94     * In our usage, this isn't a problem, as a given feature ID is always operated on by a single thread
95     * due to kafka consumer partitioning
96     */
97   override def remove(id: String): Unit = {
98     logger.trace(s"$id removing feature")
99     val old = state.remove(id)
100     if (old != null) {
101       old.removeFromIndex()
102       views.foreach(_.remove(id))
103     }
104     logger.trace(s"Current index size: ${state.size()}/${spatialIndex.size()}")
105   }
106 
107   override def expire(featureState: FeatureState): Unit = {
108     logger.trace(s"${featureState.id} expiring from index")
109     if (state.remove(featureState.id, featureState)) {
110       featureState.removeFromIndex()
111       views.foreach(_.remove(featureState.id))
112       expirations.increment()
113     }
114     logger.trace(s"Current index size: ${state.size()}/${spatialIndex.size()}")
115   }
116 
117   override def clear(): Unit = {
118     logger.trace("Clearing index")
119     state.clear()
120     spatialIndex.clear()
121     views.foreach(_.clear())
122   }
123 
124   override def size(): Int = state.size()
125 
126   // optimized for filter.include
127   override def size(f: Filter): Int = if (f == Filter.INCLUDE) { size() } else { query(f).length }
128 
129   override def query(id: String): Option[SimpleFeature] =
130     Option(state.get(id)).flatMap(f => Option(f.retrieveFromIndex()))
131 
132   override def query(filter: Filter): Iterator[SimpleFeature] = spatialIndex.query(filter)
133 
134   override def close(): Unit = factory.close()
135 
136   private def createIndex(sft: SimpleFeatureType): SimpleFeatureSpatialIndex = {
137     if (config.cqAttributes.nonEmpty) {
138       val attributes =
139         if (config.cqAttributes == Seq(KafkaDataStore.CqIndexFlag)) {
140           // deprecated boolean config to enable indices based on the stored simple feature type
141           CQIndexType.getDefinedAttributes(sft) ++ Option(sft.getGeomField).map((_, CQIndexType.GEOMETRY))
142         } else {
143           config.cqAttributes
144         }
145       // note: CQEngine handles points vs non-points internally
146       new GeoCQIndex(sft, attributes, config.resolution.x, config.resolution.y)
147     } else if (sft.isPoints) {
148       BucketIndex(sft, config.resolution.x, config.resolution.y)
149     } else {
150       SizeSeparatedBucketIndex(sft, config.ssiTiers, config.resolution.x / 360d, config.resolution.y / 180d)
151     }
152   }
153 }
