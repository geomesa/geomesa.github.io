21 /**
22  * Feature cache implementation
23  *
24  * @param sft simple feature type
25  * @param config index config
26  * @param layerViews layer views
27  */
28 class KafkaFeatureCacheImpl(sft: SimpleFeatureType, config: IndexConfig, layerViews: Seq[LayerView] = Seq.empty, tags: Tags = Tags.empty())
29     extends KafkaFeatureCache with FeatureExpiration {
30 
31   import org.locationtech.geomesa.kafka.index.KafkaFeatureCacheImpl.MetricsPrefix
32   import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
33 
34   // keeps location and expiry keyed by feature ID (we need a way to retrieve a feature based on ID for
35   // update/delete operations). to reduce contention, we never iterate over this map
36   private val state = Metrics.gaugeMapSize(s"$MetricsPrefix.size", tags, new ConcurrentHashMap[String, FeatureState]())
37 
38   private val support = createSupport(sft)
39 
40   private val factory = FeatureStateFactory(sft, support.index, config.expiry, this, config.executor)
41 
42   override val views: Seq[KafkaFeatureCacheView] =
43     layerViews.map(view => KafkaFeatureCacheView(view, createSupport(view.viewSft)))
44 
45   private val expirations = Metrics.counter(s"$MetricsPrefix.expirations", tags)
46 
47   logger.debug(s"Initialized KafkaFeatureCache with factory $factory and support $support")
48 
49   /**
50     * Note: this method is not thread-safe. The `state` and `index` can get out of sync if the same feature
51     * is updated simultaneously from two different threads
52     *
53     * In our usage, this isn't a problem, as a given feature ID is always operated on by a single thread
54     * due to kafka consumer partitioning
55     */
56   override def put(feature: SimpleFeature): Unit = {
57     if (feature.getDefaultGeometry == null) {
58       logger.warn(s"Null geometry detected for feature ${feature.getID}. Skipping loading into cache.")
59       return
60     }
61     val featureState = factory.createState(feature)
62     logger.trace(s"${featureState.id} adding feature $featureState")
63     val old = state.put(featureState.id, featureState)
64     if (old == null) {
65       featureState.insertIntoIndex()
66       views.foreach(_.put(feature))
67     } else if (old.time <= featureState.time) {
68       logger.trace(s"${featureState.id} removing old feature")
69       old.removeFromIndex()
70       featureState.insertIntoIndex()
71       views.foreach { view =>
72         view.remove(featureState.id)
73         view.put(feature)
74       }
75     } else {
76       logger.trace(s"${featureState.id} ignoring out of sequence feature")
77       if (!state.replace(featureState.id, featureState, old)) {
78         logger.warn(s"${featureState.id} detected inconsistent state... spatial index may be incorrect")
79         old.removeFromIndex()
80         views.foreach(_.remove(featureState.id))
81       }
82     }
83     logger.trace(s"Current index size: ${state.size()}/${support.index.size()}")
84   }
85 
86   /**
87     * Note: this method is not thread-safe. The `state` and `index` can get out of sync if the same feature
88     * is updated simultaneously from two different threads
89     *
90     * In our usage, this isn't a problem, as a given feature ID is always operated on by a single thread
91     * due to kafka consumer partitioning
92     */
93   override def remove(id: String): Unit = {
94     logger.trace(s"$id removing feature")
95     val old = state.remove(id)
96     if (old != null) {
97       old.removeFromIndex()
98       views.foreach(_.remove(id))
99     }
100     logger.trace(s"Current index size: ${state.size()}/${support.index.size()}")
101   }
102 
103   override def expire(featureState: FeatureState): Unit = {
104     logger.trace(s"${featureState.id} expiring from index")
105     if (state.remove(featureState.id, featureState)) {
106       featureState.removeFromIndex()
107       views.foreach(_.remove(featureState.id))
108       expirations.increment()
109     }
110     logger.trace(s"Current index size: ${state.size()}/${support.index.size()}")
111   }
112 
113   override def clear(): Unit = {
114     logger.trace("Clearing index")
115     state.clear()
116     support.index.clear()
117     views.foreach(_.clear())
118   }
119 
120   override def size(): Int = state.size()
121 
122   // optimized for filter.include
123   override def size(f: Filter): Int = if (f == Filter.INCLUDE) { size() } else { query(f).length }
124 
125   override def query(id: String): Option[SimpleFeature] =
126     Option(state.get(id)).flatMap(f => Option(f.retrieveFromIndex()))
127 
128   override def query(filter: Filter): Iterator[SimpleFeature] = support.query(filter)
129 
130   override def close(): Unit = factory.close()
131 
132   private def createSupport(sft: SimpleFeatureType): SpatialIndexSupport = {
133     if (config.cqAttributes.nonEmpty) {
134       // note: CQEngine handles points vs non-points internally
135       KafkaFeatureCache.cqIndexSupport(sft, config)
136     } else if (sft.isPoints) {
137       BucketIndexSupport(sft, config.resolution.x, config.resolution.y)
138     } else {
139       SizeSeparatedBucketIndexSupport(sft, config.ssiTiers, config.resolution.x / 360d, config.resolution.y / 180d)
140     }
141   }
142 }
