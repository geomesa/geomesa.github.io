21 /**
22  * Feature cache implementation
23  *
24  * @param sft simple feature type
25  * @param config index config
26  * @param layerViews layer views
27  */
28 class KafkaFeatureCacheImpl(sft: SimpleFeatureType, config: IndexConfig, layerViews: Seq[LayerView] = Seq.empty)
29     extends KafkaFeatureCache with FeatureExpiration {
30 
31   import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
32 
33   private val tags = KafkaDataStore.tags(sft)
34 
35   // keeps location and expiry keyed by feature ID (we need a way to retrieve a feature based on ID for
36   // update/delete operations). to reduce contention, we never iterate over this map
37   private val state = Metrics.gaugeMapSize(s"$MetricsPrefix.index.size", tags, new ConcurrentHashMap[String, FeatureState]())
38 
39   private val support = createSupport(sft)
40 
41   private val factory = FeatureStateFactory(sft, support.index, config.expiry, this, config.executor)
42 
43   override val views: Seq[KafkaFeatureCacheView] =
44     layerViews.map(view => KafkaFeatureCacheView(view, createSupport(view.viewSft)))
45 
46   private val expirations = Metrics.counter(s"$MetricsPrefix.index.expirations", tags)
47 
48   logger.debug(s"Initialized KafkaFeatureCache with factory $factory and support $support")
49 
50   /**
51     * Note: this method is not thread-safe. The `state` and `index` can get out of sync if the same feature
52     * is updated simultaneously from two different threads
53     *
54     * In our usage, this isn't a problem, as a given feature ID is always operated on by a single thread
55     * due to kafka consumer partitioning
56     */
57   override def put(feature: SimpleFeature): Unit = {
58     if (feature.getDefaultGeometry == null) {
59       logger.warn(s"Null geometry detected for feature ${feature.getID}. Skipping loading into cache.")
60       return
61     }
62     val featureState = factory.createState(feature)
63     logger.trace(s"${featureState.id} adding feature $featureState")
64     val old = state.put(featureState.id, featureState)
65     if (old == null) {
66       featureState.insertIntoIndex()
67       views.foreach(_.put(feature))
68     } else if (old.time <= featureState.time) {
69       logger.trace(s"${featureState.id} removing old feature")
70       old.removeFromIndex()
71       featureState.insertIntoIndex()
72       views.foreach { view =>
73         view.remove(featureState.id)
74         view.put(feature)
75       }
76     } else {
77       logger.trace(s"${featureState.id} ignoring out of sequence feature")
78       if (!state.replace(featureState.id, featureState, old)) {
79         logger.warn(s"${featureState.id} detected inconsistent state... spatial index may be incorrect")
80         old.removeFromIndex()
81         views.foreach(_.remove(featureState.id))
82       }
83     }
84     logger.trace(s"Current index size: ${state.size()}/${support.index.size()}")
85   }
86 
87   /**
88     * Note: this method is not thread-safe. The `state` and `index` can get out of sync if the same feature
89     * is updated simultaneously from two different threads
90     *
91     * In our usage, this isn't a problem, as a given feature ID is always operated on by a single thread
92     * due to kafka consumer partitioning
93     */
94   override def remove(id: String): Unit = {
95     logger.trace(s"$id removing feature")
96     val old = state.remove(id)
97     if (old != null) {
98       old.removeFromIndex()
99       views.foreach(_.remove(id))
100     }
101     logger.trace(s"Current index size: ${state.size()}/${support.index.size()}")
102   }
103 
104   override def expire(featureState: FeatureState): Unit = {
105     logger.trace(s"${featureState.id} expiring from index")
106     if (state.remove(featureState.id, featureState)) {
107       featureState.removeFromIndex()
108       views.foreach(_.remove(featureState.id))
109       expirations.increment()
110     }
111     logger.trace(s"Current index size: ${state.size()}/${support.index.size()}")
112   }
113 
114   override def clear(): Unit = {
115     logger.trace("Clearing index")
116     state.clear()
117     support.index.clear()
118     views.foreach(_.clear())
119   }
120 
121   override def size(): Int = state.size()
122 
123   // optimized for filter.include
124   override def size(f: Filter): Int = if (f == Filter.INCLUDE) { size() } else { query(f).length }
125 
126   override def query(id: String): Option[SimpleFeature] =
127     Option(state.get(id)).flatMap(f => Option(f.retrieveFromIndex()))
128 
129   override def query(filter: Filter): Iterator[SimpleFeature] = support.query(filter)
130 
131   override def close(): Unit = factory.close()
132 
133   private def createSupport(sft: SimpleFeatureType): SpatialIndexSupport = {
134     if (config.cqAttributes.nonEmpty) {
135       // note: CQEngine handles points vs non-points internally
136       KafkaFeatureCache.cqIndexSupport(sft, config)
137     } else if (sft.isPoints) {
138       BucketIndexSupport(sft, config.resolution.x, config.resolution.y)
139     } else {
140       SizeSeparatedBucketIndexSupport(sft, config.ssiTiers, config.resolution.x / 360d, config.resolution.y / 180d)
141     }
142   }
