19 /**
20   * Feature cache implementation
21   *
22   * @param sft simple feature type
23   * @param config index config
24   */
25 class KafkaFeatureCacheImpl(sft: SimpleFeatureType, config: IndexConfig, layerViews: Seq[LayerView] = Seq.empty)
26     extends KafkaFeatureCache with FeatureExpiration {
27 
28   import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
29 
30   // keeps location and expiry keyed by feature ID (we need a way to retrieve a feature based on ID for
31   // update/delete operations). to reduce contention, we never iterate over this map
32   private val state = new ConcurrentHashMap[String, FeatureState]
33 
34   private val support = createSupport(sft)
35 
36   private val factory = FeatureStateFactory(sft, support.index, config.expiry, this, config.executor)
37 
38   override val views: Seq[KafkaFeatureCacheView] =
39     layerViews.map(view => KafkaFeatureCacheView(view, createSupport(view.viewSft)))
40 
41   logger.debug(s"Initialized KafkaFeatureCache with factory $factory and support $support")
42 
43   /**
44     * Note: this method is not thread-safe. The `state` and `index` can get out of sync if the same feature
45     * is updated simultaneously from two different threads
46     *
47     * In our usage, this isn't a problem, as a given feature ID is always operated on by a single thread
48     * due to kafka consumer partitioning
49     */
50   override def put(feature: SimpleFeature): Unit = {
51     val featureState = factory.createState(feature)
52     logger.trace(s"${featureState.id} adding feature $featureState")
53     val old = state.put(featureState.id, featureState)
54     if (old == null) {
55       featureState.insertIntoIndex()
56       views.foreach(_.put(feature))
57     } else if (old.time <= featureState.time) {
58       logger.trace(s"${featureState.id} removing old feature")
59       old.removeFromIndex()
60       featureState.insertIntoIndex()
61       views.foreach { view =>
62         view.remove(featureState.id)
63         view.put(feature)
64       }
65     } else {
66       logger.trace(s"${featureState.id} ignoring out of sequence feature")
67       if (!state.replace(featureState.id, featureState, old)) {
68         logger.warn(s"${featureState.id} detected inconsistent state... spatial index may be incorrect")
69         old.removeFromIndex()
70         views.foreach(_.remove(featureState.id))
71       }
72     }
73     logger.trace(s"Current index size: ${state.size()}/${support.index.size()}")
74   }
75 
76   /**
77     * Note: this method is not thread-safe. The `state` and `index` can get out of sync if the same feature
78     * is updated simultaneously from two different threads
79     *
80     * In our usage, this isn't a problem, as a given feature ID is always operated on by a single thread
81     * due to kafka consumer partitioning
82     */
83   override def remove(id: String): Unit = {
84     logger.trace(s"$id removing feature")
85     val old = state.remove(id)
86     if (old != null) {
87       old.removeFromIndex()
88       views.foreach(_.remove(id))
89     }
90     logger.trace(s"Current index size: ${state.size()}/${support.index.size()}")
91   }
92 
93   override def expire(featureState: FeatureState): Unit = {
94     logger.trace(s"${featureState.id} expiring from index")
95     if (state.remove(featureState.id, featureState)) {
96       featureState.removeFromIndex()
97       views.foreach(_.remove(featureState.id))
98     }
99     logger.trace(s"Current index size: ${state.size()}/${support.index.size()}")
100   }
101 
102   override def clear(): Unit = {
103     logger.trace("Clearing index")
104     state.clear()
105     support.index.clear()
106     views.foreach(_.clear())
107   }
108 
109   override def size(): Int = state.size()
110 
111   // optimized for filter.include
112   override def size(f: Filter): Int = if (f == Filter.INCLUDE) { size() } else { query(f).length }
113 
114   override def query(id: String): Option[SimpleFeature] =
115     Option(state.get(id)).flatMap(f => Option(f.retrieveFromIndex()))
116 
117   override def query(filter: Filter): Iterator[SimpleFeature] = support.query(filter)
118 
119   override def close(): Unit = factory.close()
120 
121   private def createSupport(sft: SimpleFeatureType): SpatialIndexSupport = {
122     if (config.cqAttributes.nonEmpty) {
123       // note: CQEngine handles points vs non-points internally
124       KafkaFeatureCache.cqIndexSupport(sft, config)
125     } else if (sft.isPoints) {
126       BucketIndexSupport(sft, config.resolution.x, config.resolution.y)
127     } else {
128       SizeSeparatedBucketIndexSupport(sft, config.ssiTiers, config.resolution.x / 360d, config.resolution.y / 180d)
129     }
130   }
131 }
