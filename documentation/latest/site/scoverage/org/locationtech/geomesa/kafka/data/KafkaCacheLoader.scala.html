47     private val loading = Collections.newSetFromMap(new ConcurrentHashMap[AnyRef, java.lang.Boolean]())
48     private val firstLoadStartTime = new AtomicLong(0L)

50     def startLoad(loader: AnyRef): Boolean = synchronized {
51       if (!loading.add(loader)) {
52         logger.warn(s"Called startLoad for a loader that was already registered and has not yet completed: $loader")
53       }
54       firstLoadStartTime.compareAndSet(0L, System.currentTimeMillis())
55     }
56 
57     def completedLoad(loader: AnyRef): Unit = synchronized {
58       if (!loading.remove(loader)) {
59         logger.warn(s"Called completedLoad for a loader that was not registered or already deregistered: $loader")
60       } else if (loading.isEmpty) {
61         logger.info(s"Last active initial load completed.  " +
62           s"Initial loads took ${System.currentTimeMillis() - firstLoadStartTime.get} milliseconds.")
63         firstLoadStartTime.set(0L)
64       }
65     }
66 
67     def allLoaded(): Boolean = loading.isEmpty
68   }

114     override protected [KafkaCacheLoader] def consume(record: ConsumerRecord[Array[Byte], Array[Byte]]): Unit = {
115       val headers = RecordVersions.getHeaders(record)
116       val timestamp = RecordVersions.getTimestamp(record)
117       val message = serializer.deserialize(record.key(), record.value(), headers, timestamp)
118       logger.trace(s"Consumed message [$topic:${record.partition}:${record.offset}] $message")
119       message match {
120         case m: Change => cache.fireChange(timestamp, m.feature); cache.put(m.feature)
121         case m: Delete => cache.fireDelete(timestamp, m.id, cache.query(m.id).orNull); cache.remove(m.id)
122         case _: Clear  => cache.fireClear(timestamp); cache.clear()
123         case m => throw new IllegalArgumentException(s"Unknown message: $m")
124       }

128   /**
129    * Handles initial loaded 'from-beginning' without indexing features in the spatial index. Will still
130    * trigger message events.
131    *
132    * @param sft simple feature type
133    * @param consumers consumers, won't be closed even on call to 'close()'
134    * @param topic kafka topic
135    * @param frequency polling frequency
136    * @param offsetCommitInterval how often to commit offsets
137    * @param serializer message serializer
138    * @param readBack initial load read back
139    * @param ordering feature ordering
140    * @param toLoad main cache loader, used for callback when bulk loading is done

142   private class InitialLoader(
143       sft: SimpleFeatureType,
144       consumers: Seq[Consumer[Array[Byte], Array[Byte]]],
145       topic: String,
146       frequency: Duration,
147       offsetCommitInterval: FiniteDuration,
148       serializer: GeoMessageSerializer,
149       readBack: scala.concurrent.duration.Duration,
150       ordering: ExpiryTimeConfig,
151       toLoad: KafkaCacheLoaderImpl
152     ) extends ThreadedConsumer(consumers, frequency, offsetCommitInterval, false) with Runnable with ConsumerRebalanceListener {
153 
154     import scala.collection.JavaConverters._
155 
156     private val cache = KafkaFeatureCache.nonIndexing(sft, ordering)
157 
158     // track the offsets that we want to read to
159     private val offsets = new ConcurrentHashMap[Int, Long]()
160     private val assignment = Collections.newSetFromMap(new ConcurrentHashMap[Int, java.lang.Boolean]())
161     @volatile
162     private var done: Boolean = false
163     private var latch: CountDownLatch = _
164     @volatile
165     private var submission: Future[_] = _
166 
167     override def onPartitionsRevoked(topicPartitions: java.util.Collection[TopicPartition]): Unit = {}
168 
169     override def onPartitionsAssigned(topicPartitions: java.util.Collection[TopicPartition]): Unit = {
170       logger.debug(s"Partitions assigned: ${topicPartitions.asScala.mkString(", ")}")
171       topicPartitions.asScala.foreach { tp =>
172         if (assignment.add(tp.partition())) {
173           val consumer = consumers.find(_.assignment().contains(tp)).orNull
174           if (consumer == null) {
175             logger.warn("Partition assigned but no consumer contains the assignment")
176           } else {
177             KafkaConsumerVersions.pause(consumer, tp)
178             try {
179               logger.debug(s"Checking offsets for [${tp.topic()}:${tp.partition()}]")
180               // the only reliable way we've found to check max offset is to seek to the end and check the position there
181               consumer.seekToEnd(Collections.singleton(tp))
182               val end = consumer.position(tp)
183               logger.debug(s"Setting max offset to [${tp.topic}:${tp.partition}:${end - 1}]")
184               offsets.put(tp.partition(), end - 1)
185               if (!readBack.isFinite) {
186                 KafkaConsumerVersions.seekToBeginning(consumer, tp)
187               } else {
188                 val offset = Try {
189                   val time = System.currentTimeMillis() - readBack.toMillis
190                   KafkaConsumerVersions.offsetsForTimes(consumer, tp.topic, Seq(tp.partition), time).get(tp.partition)
191                 }
192                 offset match {
193                   case Success(Some(o)) =>
194                     logger.debug(s"Seeking to offset $o for read-back $readBack on [${tp.topic}:${tp.partition}]")
195                     consumer.seek(tp, o)
196 
197                   case Success(None) =>
198                     logger.debug(s"No prior offset found for read-back $readBack on [${tp.topic}:${tp.partition}], " +
199                       "reading from head of queue")
200 
201                   case Failure(e) =>
202                     logger.warn(s"Error finding initial offset: [${tp.topic}:${tp.partition}], seeking to beginning", e)
203                     KafkaConsumerVersions.seekToBeginning(consumer, tp)
204                 }
205               }
206             } finally {
207               KafkaConsumerVersions.resume(consumer, tp)
208             }
209           }
210         }
211       }
212     }

221     override protected def consume(record: ConsumerRecord[Array[Byte], Array[Byte]]): Unit = {
222       if (done) { toLoad.consume(record) } else {
223         val headers = RecordVersions.getHeaders(record)
224         val timestamp = RecordVersions.getTimestamp(record)
225         val message = serializer.deserialize(record.key, record.value, headers, timestamp)
226         logger.trace(s"Consumed message [$topic:${record.partition}:${record.offset}] $message")
227         message match {
228           case m: Change => toLoad.cache.fireChange(timestamp, m.feature); cache.put(m.feature)
229           case m: Delete => toLoad.cache.fireDelete(timestamp, m.id, cache.query(m.id).orNull); cache.remove(m.id)
230           case _: Clear  => toLoad.cache.fireClear(timestamp); cache.clear()
231           case m => throw new IllegalArgumentException(s"Unknown message: $m")
232         }
233         if (record.offset > 0 && record.offset % 1048576 == 0) { // magic number 2^20
234           logger.info(s"Initial load: consumed [$topic:${record.partition}:${record.offset}]")
235         }
236       }
237     }

239     def start(): Unit = {
240       LoaderStatus.startLoad(this)
241       try {
242         val partitions = consumers.head.partitionsFor(topic).asScala.map(_.partition)
243         logger.info(s"Starting initial load for [$topic] with ${partitions.size} partitions")
244         latch = new CountDownLatch(partitions.size)
245         startConsumers() // kick off the asynchronous consumer threads
246         submission = CachedThreadPool.submit(this)
247       } catch {
248         case NonFatal(e) =>
249           LoaderStatus.completedLoad(this)
250           throw e
251       }
252     }
253 
254     override def run(): Unit = {
255       try {
256         try { latch.await() } finally {
257           // stop the consumer threads, but won't close the consumers due to `closeConsumers`
258           // note: don't call this.close() as it would interrupt this thread
259           super.close()
260         }
261         // set a flag just in case the consumer threads haven't finished spinning down, so that we will
262         // pass any additional messages back to the main loader
263         done = true
264         logger.info(s"Finished initial load, transferring to indexed cache for [$topic]")
265         cache.query(Filter.INCLUDE).foreach(toLoad.cache.put)
266         logger.info(s"Finished transfer for [$topic], starting normal load")
267         toLoad.startConsumers()
268       } finally {
269         LoaderStatus.completedLoad(this)
270       }
271     }
272 
273     override def close(): Unit = {
274       try { super.close() } finally {
275         if (submission != null && !submission.isDone) {
276           submission.cancel(true)
277         }
278       }
279     }

289         extends ConsumerRunnable(id, consumer, handler) {
290 
291       override protected def processPoll(result: ConsumerRecords[Array[Byte], Array[Byte]]): Unit = {
292         try {
293           super.processPoll(result)
294         } finally {
295           result.partitions().asScala.foreach { tp =>
296             val position = consumer.position(tp)
297             // once we've hit the max offset for the partition, remove from the offset map so we don't double count it
298             if (position >= offsets.getOrDefault(tp.partition(), Long.MaxValue)) {
299               offsets.remove(tp.partition)
300               latch.countDown()
301               logger.info(s"Initial load: consumed [$topic:${tp.partition}:${position - 1}]")
302               logger.info(s"Initial load completed for [$topic:${tp.partition}], ${latch.getCount} partitions remaining")
303             }
304           }
305         }
306       }
307     }
