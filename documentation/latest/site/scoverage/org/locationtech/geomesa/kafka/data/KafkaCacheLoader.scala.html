51     private val loading = Collections.newSetFromMap(new ConcurrentHashMap[AnyRef, java.lang.Boolean]())

54     /**
55      * Register a loader starting
56      *
57      * @param loader loader instance
58      * @return
59      */
60     def startLoad(loader: AnyRef): Boolean = synchronized {
61       if (!loading.add(loader)) {
62         logger.warn(s"Called startLoad for a loader that was already registered and has not yet completed: $loader")
63       }
64       firstLoadStartTime.compareAndSet(0L, System.currentTimeMillis())
65     }
66 
67     /**
68      * De-register a loader as having completed
69      *
70      * @param loader loader instance
71      */
72     def completedLoad(loader: AnyRef): Unit = synchronized {
73       if (!loading.remove(loader)) {
74         logger.warn(s"Called completedLoad for a loader that was not registered or already deregistered: $loader")
75       } else if (loading.isEmpty) {
76         logger.info(s"Last active initial load completed.  " +
77           s"Initial loads took ${System.currentTimeMillis() - firstLoadStartTime.get} milliseconds.")
78         firstLoadStartTime.set(0L)
79       }
80     }
81 
82     /**
83      * Indicates if any active loads are ongoing
84      *
85      * @return
86      */
87     def allLoaded(): Boolean = loading.isEmpty

152     override protected def consume(record: ConsumerRecord[Array[Byte], Array[Byte]]): Unit = {
153       val headers = RecordVersions.getHeaders(record)
154       val timestamp = RecordVersions.getTimestamp(record)
155       val message = serializer.deserialize(record.key(), record.value(), headers, timestamp)
156       logger.trace(s"Consumed message [$topic:${record.partition}:${record.offset}] $message")
157       message match {
158         case m: Change =>
159           cache.fireChange(timestamp, m.feature)
160           cache.put(m.feature)
161           updates.increment()
162           dtgMetrics.foreach { case DateMetrics(index, dtg, latency) =>
163             val date = m.feature.getAttribute(index).asInstanceOf[Date]
164             if (date != null) {
165               dtg.updateAndGet(prev => Math.max(prev, date.getTime))
166               latency.record(System.currentTimeMillis() - date.getTime)
167             }
168           }
169 
170         case m: Delete =>
171           cache.fireDelete(timestamp, m.id, cache.query(m.id).orNull)
172           cache.remove(m.id)
173           deletes.increment()
174 
175         case _: Clear =>
176           cache.fireClear(timestamp)
177           cache.clear()
178           clears.increment()
179 
180         case m => throw new IllegalArgumentException(s"Unknown message: $m")
181       }
182     }

184     /**
185      * Handles initial loaded 'from-beginning' without indexing features in the spatial index. Will still
186      * trigger message events.
187      *
188      * @param readBack initial load read back
189      */
190     private class InitialLoader(readBack: scala.concurrent.duration.Duration)
191         extends ThreadedConsumer(consumers, frequency, offsetCommitInterval, false) with Runnable with ConsumerRebalanceListener {

193       import scala.collection.JavaConverters._

195       private val cache = KafkaFeatureCache.nonIndexing(sft, initialLoadConfig)

198       private val updates = Metrics.counter(s"$MetricsPrefix.readback.updates", tags)
199       private val deletes = Metrics.counter(s"$MetricsPrefix.readback.deletes", tags)
200       private val clears = Metrics.counter(s"$MetricsPrefix.readback.clears", tags)
201 
202       // track the offsets that we want to read to
203       private val offsets = new ConcurrentHashMap[Int, Long]()
204       private val assignment = Collections.newSetFromMap(new ConcurrentHashMap[Int, java.lang.Boolean]())
205       @volatile
206       private var done: Boolean = false
207       private var latch: CountDownLatch = _
208       @volatile

213       override def onPartitionsAssigned(topicPartitions: java.util.Collection[TopicPartition]): Unit = {
214         logger.debug(s"Partitions assigned: ${topicPartitions.asScala.mkString(", ")}")
215         topicPartitions.asScala.foreach { tp =>
216           if (assignment.add(tp.partition())) {
217             val consumer = consumers.find(_.assignment().contains(tp)).orNull
218             if (consumer == null) {
219               logger.warn("Partition assigned but no consumer contains the assignment")
220             } else {
221               KafkaConsumerVersions.pause(consumer, tp)
222               try {
223                 logger.debug(s"Checking offsets for [${tp.topic()}:${tp.partition()}]")
224                 // the only reliable way we've found to check max offset is to seek to the end and check the position there
225                 consumer.seekToEnd(Collections.singleton(tp))
226                 val end = consumer.position(tp)
227                 logger.debug(s"Setting max offset to [${tp.topic}:${tp.partition}:${end - 1}]")
228                 offsets.put(tp.partition(), end - 1)
229                 try {
230                   if (!readBack.isFinite) {
231                     KafkaConsumerVersions.seekToBeginning(consumer, tp)
232                   } else {
233                     val offset = Try {
234                       val time = System.currentTimeMillis() - readBack.toMillis
235                       KafkaConsumerVersions.offsetsForTimes(consumer, tp.topic, Seq(tp.partition), time).get(tp.partition)
236                     }
237                     offset match {
238                       case Success(Some(o)) =>
239                         logger.debug(s"Seeking to offset $o for read-back $readBack on [${tp.topic}:${tp.partition}]")
240                         consumer.seek(tp, o)
241 
242                       case Success(None) =>
243                         logger.debug(s"No prior offset found for read-back $readBack on [${tp.topic}:${tp.partition}], " +
244                           "reading from head of queue")
245 
246                       case Failure(e) =>
247                         logger.warn(s"Error finding initial offset: [${tp.topic}:${tp.partition}], seeking to beginning", e)
248                         KafkaConsumerVersions.seekToBeginning(consumer, tp)
249                     }
250                   }
251                 } finally {
252                   checkComplete(consumer, tp)
253                 }
254               } finally {
255                 KafkaConsumerVersions.resume(consumer, tp)
256               }
257             }
258           }
259         }
260       }

269       override protected def consume(record: ConsumerRecord[Array[Byte], Array[Byte]]): Unit = {
270         if (done) { toLoad.consume(record) } else {
271           val headers = RecordVersions.getHeaders(record)
272           val timestamp = RecordVersions.getTimestamp(record)
273           val message = serializer.deserialize(record.key, record.value, headers, timestamp)
274           logger.trace(s"Consumed message [$topic:${record.partition}:${record.offset}] $message")
275           message match {
276             case m: Change =>
277               toLoad.cache.fireChange(timestamp, m.feature)
278               cache.put(m.feature)
279               updates.increment()
280 
281             case m: Delete =>
282               toLoad.cache.fireDelete(timestamp, m.id, cache.query(m.id).orNull)
283               cache.remove(m.id)
284               deletes.increment()
285 
286             case _: Clear =>
287               toLoad.cache.fireClear(timestamp)
288               cache.clear()
289               clears.increment()
290 
291             case m => throw new IllegalArgumentException(s"Unknown message: $m")
292           }
293           if (record.offset > 0 && record.offset % 1048576 == 0) { // magic number 2^20
294             logger.info(s"Initial load: consumed [$topic:${record.partition}:${record.offset}]")
295           }
296         }
297       }

299       def start(): Unit = {
300         LoaderStatus.startLoad(this)
301         try {
302           val partitions = consumers.head.partitionsFor(topic).asScala.map(_.partition)
303           logger.info(s"Starting initial load for [$topic] with ${partitions.size} partitions")
304           latch = new CountDownLatch(partitions.size)
305           startConsumers() // kick off the asynchronous consumer threads
306           submission = CachedThreadPool.submit(this)
307         } catch {
308           case NonFatal(e) =>
309             LoaderStatus.completedLoad(this)
310             throw e
311         }
312       }
313 
314       override def run(): Unit = {
315         try {
316           try { latch.await() } finally {
317             // stop the consumer threads, but won't close the consumers due to `closeConsumers`
318             // note: don't call this.close() as it would interrupt this thread
319             super.close()
320           }
321           // set a flag just in case the consumer threads haven't finished spinning down, so that we will
322           // pass any additional messages back to the main loader
323           done = true
324           logger.info(s"Finished initial load, transferring to indexed cache for [$topic]")
325           cache.query(Filter.INCLUDE).foreach(toLoad.cache.put)
326           logger.info(s"Finished transfer for [$topic], starting normal load")
327           toLoad.startConsumers()
328         } finally {
329           LoaderStatus.completedLoad(this)
330         }
331       }
332 
333       override def close(): Unit = {
334         try { super.close() } finally {
335           if (submission != null && !submission.isDone) {
336             submission.cancel(true)
337           }
338         }
339       }

341       /**
342        * Checks the current position of the consumer, and notifies through the countdown latch when the initial load
343        * is complete
344        *
345        * @param consumer consumer
346        * @param tp topic and partition to check
347        */
348       private def checkComplete(consumer: Consumer[Array[Byte], Array[Byte]], tp: TopicPartition): Unit = {
349         val position = consumer.position(tp)
350         // once we've hit the max offset for the partition, remove from the offset map so we don't double count it
351         if (position > offsets.getOrDefault(tp.partition(), Long.MaxValue)) {
352           offsets.remove(tp.partition)
353           latch.countDown()
354           logger.info(s"Initial load completed for [$topic:${tp.partition}], ${latch.getCount} partitions remaining")
355         }
356       }

358       /**
359        * Consumer runnable that tracks when we have completed the initial load
360        *
361        * @param id id
362        * @param consumer consumer
363        * @param handler error handler
364        */
365       private class InitialLoaderConsumerRunnable(id: String, consumer: Consumer[Array[Byte], Array[Byte]], handler: ConsumerErrorHandler)
366           extends ConsumerRunnable(id, consumer, handler) {
367 
368         override protected def processPoll(result: ConsumerRecords[Array[Byte], Array[Byte]]): Unit = {
369           try { super.processPoll(result) } finally {
370             result.partitions().asScala.foreach(checkComplete(consumer, _))
371           }
372         }
373       }
