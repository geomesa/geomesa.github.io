48     private val loading = Collections.newSetFromMap(new ConcurrentHashMap[AnyRef, java.lang.Boolean]())
49     private val firstLoadStartTime = new AtomicLong(0L)

51     def startLoad(loader: AnyRef): Boolean = synchronized {
52       if (!loading.add(loader)) {
53         logger.warn(s"Called startLoad for a loader that was already registered and has not yet completed: $loader")
54       }
55       firstLoadStartTime.compareAndSet(0L, System.currentTimeMillis())
56     }
57 
58     def completedLoad(loader: AnyRef): Unit = synchronized {
59       if (!loading.remove(loader)) {
60         logger.warn(s"Called completedLoad for a loader that was not registered or already deregistered: $loader")
61       } else if (loading.isEmpty) {
62         logger.info(s"Last active initial load completed.  " +
63           s"Initial loads took ${System.currentTimeMillis() - firstLoadStartTime.get} milliseconds.")
64         firstLoadStartTime.set(0L)
65       }
66     }
67 
68     def allLoaded(): Boolean = loading.isEmpty
69   }

133     override protected def consume(record: ConsumerRecord[Array[Byte], Array[Byte]]): Unit = {
134       val headers = RecordVersions.getHeaders(record)
135       val timestamp = RecordVersions.getTimestamp(record)
136       val message = serializer.deserialize(record.key(), record.value(), headers, timestamp)
137       logger.trace(s"Consumed message [$topic:${record.partition}:${record.offset}] $message")
138       message match {
139         case m: Change =>
140           cache.fireChange(timestamp, m.feature)

143           println("update")
144           dtgMetrics.foreach { case DateMetrics(index, dtg, latency) =>
145             val date = m.feature.getAttribute(index).asInstanceOf[Date]
146             println(s"dtg update: ${date}")

148               dtg.updateAndGet(prev => Math.max(prev, date.getTime))
149               latency.record(System.currentTimeMillis() - date.getTime)

156           deletes.increment()

158         case _: Clear =>
159           cache.fireClear(timestamp)

163         case m => throw new IllegalArgumentException(s"Unknown message: $m")

196       override def onPartitionsAssigned(topicPartitions: java.util.Collection[TopicPartition]): Unit = {
197         logger.debug(s"Partitions assigned: ${topicPartitions.asScala.mkString(", ")}")
198         topicPartitions.asScala.foreach { tp =>
199           if (assignment.add(tp.partition())) {
200             val consumer = consumers.find(_.assignment().contains(tp)).orNull
201             if (consumer == null) {
202               logger.warn("Partition assigned but no consumer contains the assignment")
203             } else {
204               KafkaConsumerVersions.pause(consumer, tp)
205               try {
206                 logger.debug(s"Checking offsets for [${tp.topic()}:${tp.partition()}]")
207                 // the only reliable way we've found to check max offset is to seek to the end and check the position there
208                 consumer.seekToEnd(Collections.singleton(tp))
209                 val end = consumer.position(tp)
210                 logger.debug(s"Setting max offset to [${tp.topic}:${tp.partition}:${end - 1}]")
211                 offsets.put(tp.partition(), end - 1)
212                 if (!readBack.isFinite) {
213                   KafkaConsumerVersions.seekToBeginning(consumer, tp)
214                 } else {
215                   val offset = Try {
216                     val time = System.currentTimeMillis() - readBack.toMillis
217                     KafkaConsumerVersions.offsetsForTimes(consumer, tp.topic, Seq(tp.partition), time).get(tp.partition)
218                   }
219                   offset match {
220                     case Success(Some(o)) =>
221                       logger.debug(s"Seeking to offset $o for read-back $readBack on [${tp.topic}:${tp.partition}]")
222                       consumer.seek(tp, o)
223 
224                     case Success(None) =>
225                       logger.debug(s"No prior offset found for read-back $readBack on [${tp.topic}:${tp.partition}], " +
226                         "reading from head of queue")
227 
228                     case Failure(e) =>
229                       logger.warn(s"Error finding initial offset: [${tp.topic}:${tp.partition}], seeking to beginning", e)
230                       KafkaConsumerVersions.seekToBeginning(consumer, tp)
231                   }
232                 }
233               } finally {
234                 KafkaConsumerVersions.resume(consumer, tp)
235               }
236             }
237           }
238         }
239       }

248       override protected def consume(record: ConsumerRecord[Array[Byte], Array[Byte]]): Unit = {
249         if (done) { toLoad.consume(record) } else {
250           val headers = RecordVersions.getHeaders(record)
251           val timestamp = RecordVersions.getTimestamp(record)

254           message match {
255             case m: Change =>

264               deletes.increment()

267               toLoad.cache.fireClear(timestamp)

271             case m => throw new IllegalArgumentException(s"Unknown message: $m")

273           if (record.offset > 0 && record.offset % 1048576 == 0) { // magic number 2^20
274             logger.info(s"Initial load: consumed [$topic:${record.partition}:${record.offset}]")

279       def start(): Unit = {
280         LoaderStatus.startLoad(this)
281         try {
282           val partitions = consumers.head.partitionsFor(topic).asScala.map(_.partition)
283           logger.info(s"Starting initial load for [$topic] with ${partitions.size} partitions")
284           latch = new CountDownLatch(partitions.size)
285           startConsumers() // kick off the asynchronous consumer threads
286           submission = CachedThreadPool.submit(this)
287         } catch {
288           case NonFatal(e) =>
289             LoaderStatus.completedLoad(this)
290             throw e
291         }
292       }
293 
294       override def run(): Unit = {
295         try {
296           try { latch.await() } finally {
297             // stop the consumer threads, but won't close the consumers due to `closeConsumers`
298             // note: don't call this.close() as it would interrupt this thread
299             super.close()
300           }
301           // set a flag just in case the consumer threads haven't finished spinning down, so that we will
302           // pass any additional messages back to the main loader
303           done = true
304           logger.info(s"Finished initial load, transferring to indexed cache for [$topic]")
305           cache.query(Filter.INCLUDE).foreach(toLoad.cache.put)
306           logger.info(s"Finished transfer for [$topic], starting normal load")
307           toLoad.startConsumers()
308         } finally {
309           LoaderStatus.completedLoad(this)
310         }
311       }
312 
313       override def close(): Unit = {
314         try { super.close() } finally {
315           if (submission != null && !submission.isDone) {
316             submission.cancel(true)
317           }
318         }
319       }

331         override protected def processPoll(result: ConsumerRecords[Array[Byte], Array[Byte]]): Unit = {
332           try {
333             super.processPoll(result)
334           } finally {
335             result.partitions().asScala.foreach { tp =>
336               val position = consumer.position(tp)
337               // once we've hit the max offset for the partition, remove from the offset map so we don't double count it
338               if (position >= offsets.getOrDefault(tp.partition(), Long.MaxValue)) {
339                 offsets.remove(tp.partition)
340                 latch.countDown()
341                 logger.info(s"Initial load: consumed [$topic:${tp.partition}:${position - 1}]")
342                 logger.info(s"Initial load completed for [$topic:${tp.partition}], ${latch.getCount} partitions remaining")
343               }
344             }
345           }
346         }
347       }
