44     private val loading = Collections.newSetFromMap(new ConcurrentHashMap[AnyRef, java.lang.Boolean]())
45     private val firstLoadStartTime = new AtomicLong(0L)
46 
47     def startLoad(loader: AnyRef): Boolean = synchronized {
48       if (!loading.add(loader)) {
49         logger.warn(s"Called startLoad for a loader that was already registered and has not yet completed: $loader")
50       }
51       firstLoadStartTime.compareAndSet(0L, System.currentTimeMillis())
52     }
53 
54     def completedLoad(loader: AnyRef): Unit = synchronized {
55       if (!loading.remove(loader)) {
56         logger.warn(s"Called completedLoad for a loader that was not registered or already deregistered: $loader")
57       } else if (loading.isEmpty) {
58         logger.info(s"Last active initial load completed.  " +
59           s"Initial loads took ${System.currentTimeMillis() - firstLoadStartTime.get} milliseconds.")
60         firstLoadStartTime.set(0L)
61       }
62     }
63 
64     def allLoaded(): Boolean = loading.isEmpty
65   }

72   class KafkaCacheLoaderImpl(
73       sft: SimpleFeatureType,
74       override val cache: KafkaFeatureCache,
75       consumers: Seq[Consumer[Array[Byte], Array[Byte]]],
76       topic: String,
77       frequency: Long,
78       serializer: GeoMessageSerializer,
79       doInitialLoad: Boolean,
80       initialLoadConfig: ExpiryTimeConfig,
81       offsetCommitIntervalMs: Long
82     ) extends ThreadedConsumer(consumers, Duration.ofMillis(frequency), offsetCommitIntervalMs) with KafkaCacheLoader {
83     try { classOf[ConsumerRecord[Any, Any]].getMethod("timestamp") } catch {
84       case _: NoSuchMethodException => logger.warn("This version of Kafka doesn't support timestamps, using system time")
85     }
86 
87     private val initialLoader =
88       if (doInitialLoad) {
89         // for the initial load, don't bother spatially indexing until we have the final state
90         Some(new InitialLoader(sft, consumers, topic, frequency, offsetCommitIntervalMs, serializer, initialLoadConfig, this))
91       } else {
92         None
93       }
94 
95     def start(): Unit = {
96       initialLoader match {
97         case None => startConsumers()
98         case Some(loader) => loader.start()
99       }
100     }
101 
102     override def close(): Unit = {
103       try { super.close() } finally {
104         CloseWithLogging(initialLoader)
105         cache.close()
106       }
107     }
108 
109     override protected [KafkaCacheLoader] def consume(record: ConsumerRecord[Array[Byte], Array[Byte]]): Unit = {
110       val headers = RecordVersions.getHeaders(record)
111       val timestamp = RecordVersions.getTimestamp(record)
112       val message = serializer.deserialize(record.key(), record.value(), headers, timestamp)
113       logger.trace(s"Consumed message [$topic:${record.partition}:${record.offset}] $message")
114       message match {
115         case m: Change => cache.fireChange(timestamp, m.feature); cache.put(m.feature)
116         case m: Delete => cache.fireDelete(timestamp, m.id, cache.query(m.id).orNull); cache.remove(m.id)
117         case _: Clear  => cache.fireClear(timestamp); cache.clear()
118         case m => throw new IllegalArgumentException(s"Unknown message: $m")
119       }
120     }
121   }

123   /**
124     * Handles initial loaded 'from-beginning' without indexing features in the spatial index. Will still
125     * trigger message events.
126     *
127     * @param consumers consumers, won't be closed even on call to 'close()'
128     * @param topic kafka topic
129     * @param frequency polling frequency in milliseconds
130     * @param serializer message serializer
131     * @param toLoad main cache loader, used for callback when bulk loading is done
132     */
133   private class InitialLoader(
134       sft: SimpleFeatureType,
135       consumers: Seq[Consumer[Array[Byte], Array[Byte]]],
136       topic: String,
137       frequency: Long,
138       offsetCommitIntervalMs: Long,
139       serializer: GeoMessageSerializer,
140       ordering: ExpiryTimeConfig,
141       toLoad: KafkaCacheLoaderImpl
142     ) extends ThreadedConsumer(consumers, Duration.ofMillis(frequency), offsetCommitIntervalMs, false) with Runnable {
143 
144     import scala.collection.JavaConverters._
145 
146     private val cache = KafkaFeatureCache.nonIndexing(sft, ordering)
147 
148     // track the offsets that we want to read to
149     private val offsets = new ConcurrentHashMap[Int, Long]()
150     private val done = new AtomicBoolean(false)
151     private var latch: CountDownLatch = _
152     @volatile
153     private var submission: Future[_] = _
154 
155     override protected def createConsumerRunnable(
156         id: String,
157         consumer: Consumer[Array[Byte], Array[Byte]],
158         handler: ConsumerErrorHandler): Runnable = {
159       new InitialLoaderConsumerRunnable(id, consumer, handler)
160     }
161 
162     override protected def consume(record: ConsumerRecord[Array[Byte], Array[Byte]]): Unit = {
163       if (done.get) { toLoad.consume(record) } else {
164         val headers = RecordVersions.getHeaders(record)
165         val timestamp = RecordVersions.getTimestamp(record)
166         val message = serializer.deserialize(record.key, record.value, headers, timestamp)
167         logger.trace(s"Consumed message [$topic:${record.partition}:${record.offset}] $message")
168         message match {
169           case m: Change => toLoad.cache.fireChange(timestamp, m.feature); cache.put(m.feature)
170           case m: Delete => toLoad.cache.fireDelete(timestamp, m.id, cache.query(m.id).orNull); cache.remove(m.id)
171           case _: Clear  => toLoad.cache.fireClear(timestamp); cache.clear()
172           case m => throw new IllegalArgumentException(s"Unknown message: $m")
173         }
174         if (record.offset > 0 && record.offset % 1048576 == 0) { // magic number 2^20
175           logger.info(s"Initial load: consumed [$topic:${record.partition}:${record.offset}]")
176         }
177       }
178     }
179 
180     def start(): Unit = {
181       LoaderStatus.startLoad(this)
182       try {
183         val partitions = consumers.head.partitionsFor(topic).asScala.map(_.partition)
184         val doInitialLoad =
185           try {
186             // note: these methods are not available in kafka 0.9, which will cause it to fall back to normal loading
187             val beginningOffsets = KafkaConsumerVersions.beginningOffsets(consumers.head, topic, partitions.toSeq)
188             val endOffsets = KafkaConsumerVersions.endOffsets(consumers.head, topic, partitions.toSeq)
189             partitions.exists(p => beginningOffsets.getOrElse(p, 0L) < endOffsets.getOrElse(p, 0L))
190           } catch {
191             case e: NoSuchMethodException =>
192               logger.warn(s"Can't support initial bulk loading for current Kafka version: $e")
193               false
194           }
195         if (doInitialLoad) {
196           logger.info(s"Starting initial load for [$topic] with ${partitions.size} partitions")
197           latch = new CountDownLatch(partitions.size)
198           startConsumers() // kick off the asynchronous consumer threads
199           submission = CachedThreadPool.submit(this)
200         } else {
201           // don't bother spinning up the consumer threads if we don't need to actually bulk load anything
202           startNormalLoad()
203           LoaderStatus.completedLoad(this)
204         }
205       } catch {
206         case NonFatal(e) =>
207           LoaderStatus.completedLoad(this)
208           throw e
209       }
210     }
211 
212     override def run(): Unit = {
213       try {
214         try { latch.await() } finally {
215           // stop the consumer threads, but won't close the consumers due to `closeConsumers`
216           // note: don't call this.close() as it would interrupt this thread
217           super.close()
218         }
219         // set a flag just in case the consumer threads haven't finished spinning down, so that we will
220         // pass any additional messages back to the main loader
221         done.set(true)
222         logger.info(s"Finished initial load, transferring to indexed cache for [$topic]")
223         cache.query(Filter.INCLUDE).foreach(toLoad.cache.put)
224         logger.info(s"Finished transfer for [$topic]")
225         startNormalLoad()
226       } finally {
227         LoaderStatus.completedLoad(this)
228       }
229     }
230 
231     override def close(): Unit = {
232       try { super.close() } finally {
233         if (submission != null && !submission.isDone) {
234           submission.cancel(true)
235         }
236       }
237     }
238     // start the normal loading
239     private def startNormalLoad(): Unit = {
240       logger.info(s"Starting normal load for [$topic]")
241       toLoad.startConsumers()
242     }
243 
244     /**
245      * Consumer runnable for the initial load. It's not possible to check assignments until after poll() has been called,
246      * and checking endOffsets is not reliable as a high water mark, due to transaction markers or other issues that
247      * may advance the endOffset past where the consumer can actually reach (unless new messages are written).
248      *
249      * @param id id
250      * @param consumer consumer
251      * @param handler error handler
252      */
253     private class InitialLoaderConsumerRunnable(id: String, consumer: Consumer[Array[Byte], Array[Byte]], handler: ConsumerErrorHandler)
254         extends ConsumerRunnable(id, consumer, handler) {
255 
256       private var checkOffsets = true
257 
258       override protected def processPoll(result: ConsumerRecords[Array[Byte], Array[Byte]]): Unit = {
259         if (checkOffsets) {
260           consumer.assignment().asScala.foreach { tp =>
261             // the only reliable way we've found to check max offset is to seek to the end and check the position there
262             val position = consumer.position(tp)
263             consumer.seekToEnd(Collections.singleton(tp))
264             val end = consumer.position(tp)
265             // return to the original position
266             consumer.seek(tp, position)
267             logger.debug(s"Setting max offset to [${tp.topic}:${tp.partition}:${end - 1}]")
268             offsets.put(tp.partition(), end - 1)
269           }
270           checkOffsets = false
271         }
272         try {
273           super.processPoll(result)
274         } finally {
275           result.partitions().asScala.foreach { tp =>
276             val position = consumer.position(tp)
277             // once we've hit the max offset for the partition, remove from the offset map so we don't double count it
278             if (position >= offsets.getOrDefault(tp.partition(), Long.MaxValue)) {
279               offsets.remove(tp.partition)
280               latch.countDown()
281               logger.info(s"Initial load: consumed [$topic:${tp.partition}:${position - 1}]")
282               logger.info(s"Initial load completed for [$topic:${tp.partition}], ${latch.getCount} partitions remaining")
283             }
284           }
285         }
286       }
287     }
288   }
