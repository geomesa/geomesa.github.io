41   object LoaderStatus {
42 
43     private val loading = Collections.newSetFromMap(new ConcurrentHashMap[AnyRef, java.lang.Boolean]())
44     private val firstLoadStartTime = new AtomicLong(0L)
45 
46     def startLoad(loader: AnyRef): Boolean = synchronized {
47       if (!loading.add(loader)) {
48         logger.warn(s"Called startLoad for a loader that was already registered and has not yet completed: $loader")
49       }
50       firstLoadStartTime.compareAndSet(0L, System.currentTimeMillis())
51     }
52 
53     def completedLoad(loader: AnyRef): Unit = synchronized {
54       if (!loading.remove(loader)) {
55         logger.warn(s"Called completedLoad for a loader that was not registered or already deregistered: $loader")
56       } else if (loading.isEmpty) {
57         logger.info(s"Last active initial load completed.  " +
58           s"Initial loads took ${System.currentTimeMillis() - firstLoadStartTime.get} milliseconds.")
59         firstLoadStartTime.set(0L)
60       }
61     }
62 
63     def allLoaded(): Boolean = loading.isEmpty
64   }

71   class KafkaCacheLoaderImpl(
72       sft: SimpleFeatureType,
73       override val cache: KafkaFeatureCache,
74       consumers: Seq[Consumer[Array[Byte], Array[Byte]]],
75       topic: String,
76       frequency: Long,
77       serializer: GeoMessageSerializer,
78       doInitialLoad: Boolean,

81     ) extends ThreadedConsumer(consumers, Duration.ofMillis(frequency), offsetCommitIntervalMs) with KafkaCacheLoader {
82     try { classOf[ConsumerRecord[Any, Any]].getMethod("timestamp") } catch {
83       case _: NoSuchMethodException => logger.warn("This version of Kafka doesn't support timestamps, using system time")

86     private val initialLoader =
87       if (doInitialLoad) {
88         // for the initial load, don't bother spatially indexing until we have the final state
89         Some(new InitialLoader(sft, consumers, topic, frequency, offsetCommitIntervalMs, serializer, initialLoadConfig, this))

94     def start(): Unit = {

101     override def close(): Unit = {
102       try { super.close() } finally {
103         CloseWithLogging(initialLoader)

108     override protected [KafkaCacheLoader] def consume(record: ConsumerRecord[Array[Byte], Array[Byte]]): Unit = {
109       val headers = RecordVersions.getHeaders(record)
110       val timestamp = RecordVersions.getTimestamp(record)
111       val message = serializer.deserialize(record.key(), record.value(), headers, timestamp)
112       logger.trace(s"Consumed message [$topic:${record.partition}:${record.offset}] $message")
113       message match {
114         case m: Change => cache.fireChange(timestamp, m.feature); cache.put(m.feature)
115         case m: Delete => cache.fireDelete(timestamp, m.id, cache.query(m.id).orNull); cache.remove(m.id)
116         case _: Clear  => cache.fireClear(timestamp); cache.clear()
117         case m => throw new IllegalArgumentException(s"Unknown message: $m")
118       }
119     }
120   }

122   /**
123     * Handles initial loaded 'from-beginning' without indexing features in the spatial index. Will still
124     * trigger message events.
125     *
126     * @param consumers consumers, won't be closed even on call to 'close()'
127     * @param topic kafka topic
128     * @param frequency polling frequency in milliseconds
129     * @param serializer message serializer
130     * @param toLoad main cache loader, used for callback when bulk loading is done
131     */
132   private class InitialLoader(
133       sft: SimpleFeatureType,
134       consumers: Seq[Consumer[Array[Byte], Array[Byte]]],
135       topic: String,
136       frequency: Long,
137       offsetCommitIntervalMs: Long,
138       serializer: GeoMessageSerializer,
139       ordering: ExpiryTimeConfig,
140       toLoad: KafkaCacheLoaderImpl
141     ) extends ThreadedConsumer(consumers, Duration.ofMillis(frequency), offsetCommitIntervalMs, false) with Runnable {
142 
143     import scala.collection.JavaConverters._
144 
145     private val cache = KafkaFeatureCache.nonIndexing(sft, ordering)
146 
147     // track the offsets that we want to read to
148     private val offsets = new ConcurrentHashMap[Int, Long]()
149     private val done = new AtomicBoolean(false)
150     private var latch: CountDownLatch = _
151     @volatile
152     private var submission: Future[_] = _
153 
154     override protected def consume(record: ConsumerRecord[Array[Byte], Array[Byte]]): Unit = {
155       if (done.get) { toLoad.consume(record) } else {
156         val headers = RecordVersions.getHeaders(record)
157         val timestamp = RecordVersions.getTimestamp(record)
158         val message = serializer.deserialize(record.key, record.value, headers, timestamp)
159         logger.trace(s"Consumed message [$topic:${record.partition}:${record.offset}] $message")
160         message match {
161           case m: Change => toLoad.cache.fireChange(timestamp, m.feature); cache.put(m.feature)
162           case m: Delete => toLoad.cache.fireDelete(timestamp, m.id, cache.query(m.id).orNull); cache.remove(m.id)
163           case _: Clear  => toLoad.cache.fireClear(timestamp); cache.clear()
164           case m => throw new IllegalArgumentException(s"Unknown message: $m")
165         }
166         // once we've hit the max offset for the partition, remove from the offset map to indicate we're done
167         val maxOffset = offsets.getOrDefault(record.partition, Long.MaxValue)
168         if (maxOffset <= record.offset) {
169           offsets.remove(record.partition)
170           latch.countDown()
171           logger.info(s"Initial load: consumed [$topic:${record.partition}:${record.offset}] of $maxOffset, " +
172               s"${latch.getCount} partitions remaining")
173         } else if (record.offset > 0 && record.offset % 1048576 == 0) { // magic number 2^20
174           logger.info(s"Initial load: consumed [$topic:${record.partition}:${record.offset}] of $maxOffset")
175         }
176       }
177     }
178 
179     def start(): Unit = {
180       LoaderStatus.startLoad(this)
181       try {
182         val partitions = consumers.head.partitionsFor(topic).asScala.map(_.partition)
183         try {
184           // note: these methods are not available in kafka 0.9, which will cause it to fall back to normal loading
185           val beginningOffsets = KafkaConsumerVersions.beginningOffsets(consumers.head, topic, partitions.toSeq)
186           val endOffsets = KafkaConsumerVersions.endOffsets(consumers.head, topic, partitions.toSeq)
187           partitions.foreach { p =>
188             // end offsets are the *next* offset that will be returned, so subtract one to track the last offset
189             // we will actually consume
190             val endOffset = endOffsets.getOrElse(p, 0L) - 1L
191             // note: not sure if start offsets are also off by one, but at the worst we would skip bulk loading
192             // for the last message per topic
193             val beginningOffset = beginningOffsets.getOrElse(p, 0L)
194             if (beginningOffset < endOffset) {
195               offsets.put(p, endOffset)
196             }
197           }
198         } catch {
199           case e: NoSuchMethodException => logger.warn(s"Can't support initial bulk loading for current Kafka version: $e")
200         }
201         if (offsets.isEmpty) {
202           // don't bother spinning up the consumer threads if we don't need to actually bulk load anything
203           startNormalLoad()
204           LoaderStatus.completedLoad(this)
205         } else {
206           logger.info(s"Starting initial load for [$topic] with ${offsets.size} partitions")
207           latch = new CountDownLatch(offsets.size)
208           startConsumers() // kick off the asynchronous consumer threads
209           submission = CachedThreadPool.submit(this)
210         }
211       } catch {
212         case NonFatal(e) =>
213           LoaderStatus.completedLoad(this)
214           throw e
215       }
216     }
217 
218     override def run(): Unit = {
219       try {
220         try { latch.await() } finally {
221           // stop the consumer threads, but won't close the consumers due to `closeConsumers`
222           // note: don't call this.close() as it would interrupt this thread
223           super.close()
224         }
225         // set a flag just in case the consumer threads haven't finished spinning down, so that we will
226         // pass any additional messages back to the main loader
227         done.set(true)
228         logger.info(s"Finished initial load, transferring to indexed cache for [$topic]")
229         cache.query(Filter.INCLUDE).foreach(toLoad.cache.put)
230         logger.info(s"Finished transfer for [$topic]")
231         startNormalLoad()
232       } finally {
233         LoaderStatus.completedLoad(this)
234       }
235     }
236 
237     override def close(): Unit = {
238       try { super.close() } finally {
239         if (submission != null && !submission.isDone) {
240           submission.cancel(true)
241         }
242       }
243     }
244     // start the normal loading
245     private def startNormalLoad(): Unit = {
246       logger.info(s"Starting normal load for [$topic]")
247       toLoad.startConsumers()
248     }
249   }
