41   object LoaderStatus {
42 
43     private val loading = Collections.newSetFromMap(new ConcurrentHashMap[AnyRef, java.lang.Boolean]())
44     private val firstLoadStartTime = new AtomicLong(0L)
45 
46     def startLoad(loader: AnyRef): Boolean = synchronized {
47       if (!loading.add(loader)) {
48         logger.warn(s"Called startLoad for a loader that was already registered and has not yet completed: $loader")
49       }
50       firstLoadStartTime.compareAndSet(0L, System.currentTimeMillis())
51     }
52 
53     def completedLoad(loader: AnyRef): Unit = synchronized {
54       if (!loading.remove(loader)) {
55         logger.warn(s"Called completedLoad for a loader that was not registered or already deregistered: $loader")
56       } else if (loading.isEmpty) {
57         logger.info(s"Last active initial load completed.  " +
58           s"Initial loads took ${System.currentTimeMillis() - firstLoadStartTime.get} milliseconds.")
59         firstLoadStartTime.set(0L)
60       }
61     }
62 
63     def allLoaded(): Boolean = loading.isEmpty
64   }

71   class KafkaCacheLoaderImpl(
72       sft: SimpleFeatureType,
73       override val cache: KafkaFeatureCache,
74       consumers: Seq[Consumer[Array[Byte], Array[Byte]]],
75       topic: String,
76       frequency: Long,
77       serializer: GeoMessageSerializer,
78       doInitialLoad: Boolean,
79       initialLoadConfig: ExpiryTimeConfig
80     ) extends ThreadedConsumer(consumers, Duration.ofMillis(frequency)) with KafkaCacheLoader {
81     try { classOf[ConsumerRecord[Any, Any]].getMethod("timestamp") } catch {
82       case _: NoSuchMethodException => logger.warn("This version of Kafka doesn't support timestamps, using system time")
83     }
84 
85     private val initialLoader =
86       if (doInitialLoad) {
87         // for the initial load, don't bother spatially indexing until we have the final state
88         Some(new InitialLoader(sft, consumers, topic, frequency, serializer, initialLoadConfig, this))
89       } else {
90         None
91       }
92 
93     def start(): Unit = {
94       initialLoader match {
95         case None => startConsumers()
96         case Some(loader) => loader.start()
97       }
98     }
99 
100     override def close(): Unit = {
101       try { super.close() } finally {
102         CloseWithLogging(initialLoader)
103         cache.close()
104       }
105     }

107     override protected [KafkaCacheLoader] def consume(record: ConsumerRecord[Array[Byte], Array[Byte]]): Unit = {
108       val headers = RecordVersions.getHeaders(record)
109       val timestamp = RecordVersions.getTimestamp(record)
110       val message = serializer.deserialize(record.key(), record.value(), headers, timestamp)
111       logger.trace(s"Consumed message [$topic:${record.partition}:${record.offset}] $message")
112       message match {
113         case m: Change => cache.fireChange(timestamp, m.feature); cache.put(m.feature)
114         case m: Delete => cache.fireDelete(timestamp, m.id, cache.query(m.id).orNull); cache.remove(m.id)
115         case _: Clear  => cache.fireClear(timestamp); cache.clear()
116         case m => throw new IllegalArgumentException(s"Unknown message: $m")
117       }
118     }
119   }

121   /**
122     * Handles initial loaded 'from-beginning' without indexing features in the spatial index. Will still
123     * trigger message events.
124     *
125     * @param consumers consumers, won't be closed even on call to 'close()'
126     * @param topic kafka topic
127     * @param frequency polling frequency in milliseconds
128     * @param serializer message serializer
129     * @param toLoad main cache loader, used for callback when bulk loading is done
130     */
131   private class InitialLoader(
132       sft: SimpleFeatureType,
133       consumers: Seq[Consumer[Array[Byte], Array[Byte]]],
134       topic: String,
135       frequency: Long,
136       serializer: GeoMessageSerializer,
137       ordering: ExpiryTimeConfig,
138       toLoad: KafkaCacheLoaderImpl
139     ) extends ThreadedConsumer(consumers, Duration.ofMillis(frequency), false) with Runnable {
140 
141     import scala.collection.JavaConverters._
142 
143     private val cache = KafkaFeatureCache.nonIndexing(sft, ordering)
144 
145     // track the offsets that we want to read to
146     private val offsets = new ConcurrentHashMap[Int, Long]()
147     private val done = new AtomicBoolean(false)
148     private var latch: CountDownLatch = _
149     @volatile
150     private var submission: Future[_] = _
151 
152     override protected def consume(record: ConsumerRecord[Array[Byte], Array[Byte]]): Unit = {
153       if (done.get) { toLoad.consume(record) } else {
154         val headers = RecordVersions.getHeaders(record)
155         val timestamp = RecordVersions.getTimestamp(record)
156         val message = serializer.deserialize(record.key, record.value, headers, timestamp)
157         logger.trace(s"Consumed message [$topic:${record.partition}:${record.offset}] $message")
158         message match {
159           case m: Change => toLoad.cache.fireChange(timestamp, m.feature); cache.put(m.feature)
160           case m: Delete => toLoad.cache.fireDelete(timestamp, m.id, cache.query(m.id).orNull); cache.remove(m.id)
161           case _: Clear  => toLoad.cache.fireClear(timestamp); cache.clear()
162           case m => throw new IllegalArgumentException(s"Unknown message: $m")
163         }
164         // once we've hit the max offset for the partition, remove from the offset map to indicate we're done
165         val maxOffset = offsets.getOrDefault(record.partition, Long.MaxValue)
166         if (maxOffset <= record.offset) {
167           offsets.remove(record.partition)
168           latch.countDown()
169           logger.info(s"Initial load: consumed [$topic:${record.partition}:${record.offset}] of $maxOffset, " +
170               s"${latch.getCount} partitions remaining")
171         } else if (record.offset > 0 && record.offset % 1048576 == 0) { // magic number 2^20
172           logger.info(s"Initial load: consumed [$topic:${record.partition}:${record.offset}] of $maxOffset")
173         }
174       }
175     }
176 
177     def start(): Unit = {
178       LoaderStatus.startLoad(this)
179       try {
180         val partitions = consumers.head.partitionsFor(topic).asScala.map(_.partition)
181         try {
182           // note: these methods are not available in kafka 0.9, which will cause it to fall back to normal loading
183           val beginningOffsets = KafkaConsumerVersions.beginningOffsets(consumers.head, topic, partitions.toSeq)
184           val endOffsets = KafkaConsumerVersions.endOffsets(consumers.head, topic, partitions.toSeq)
185           partitions.foreach { p =>
186             // end offsets are the *next* offset that will be returned, so subtract one to track the last offset
187             // we will actually consume
188             val endOffset = endOffsets.getOrElse(p, 0L) - 1L
189             // note: not sure if start offsets are also off by one, but at the worst we would skip bulk loading
190             // for the last message per topic
191             val beginningOffset = beginningOffsets.getOrElse(p, 0L)
192             if (beginningOffset < endOffset) {
193               offsets.put(p, endOffset)
194             }
195           }
196         } catch {
197           case e: NoSuchMethodException => logger.warn(s"Can't support initial bulk loading for current Kafka version: $e")
198         }
199         if (offsets.isEmpty) {
200           // don't bother spinning up the consumer threads if we don't need to actually bulk load anything
201           startNormalLoad()
202           LoaderStatus.completedLoad(this)
203         } else {
204           logger.info(s"Starting initial load for [$topic] with ${offsets.size} partitions")
205           latch = new CountDownLatch(offsets.size)
206           startConsumers() // kick off the asynchronous consumer threads
207           submission = CachedThreadPool.submit(this)
208         }
209       } catch {
210         case NonFatal(e) =>
211           LoaderStatus.completedLoad(this)
212           throw e
213       }
214     }
215 
216     override def run(): Unit = {
217       try {
218         try { latch.await() } finally {
219           // stop the consumer threads, but won't close the consumers due to `closeConsumers`
220           // note: don't call this.close() as it would interrupt this thread
221           super.close()
222         }
223         // set a flag just in case the consumer threads haven't finished spinning down, so that we will
224         // pass any additional messages back to the main loader
225         done.set(true)
226         logger.info(s"Finished initial load, transferring to indexed cache for [$topic]")
227         cache.query(Filter.INCLUDE).foreach(toLoad.cache.put)
228         logger.info(s"Finished transfer for [$topic]")
229         startNormalLoad()
230       } finally {
231         LoaderStatus.completedLoad(this)
232       }
233     }
234 
235     override def close(): Unit = {
236       try { super.close() } finally {
237         if (submission != null && !submission.isDone) {
238           submission.cancel(true)
239         }
240       }
241     }
242     // start the normal loading
243     private def startNormalLoad(): Unit = {
244       logger.info(s"Starting normal load for [$topic]")
245       toLoad.startConsumers()
246     }
247   }
