143     override protected def consume(record: ConsumerRecord[Array[Byte], Array[Byte]]): Unit = {
144       val headers = RecordVersions.getHeaders(record)
145       val timestamp = RecordVersions.getTimestamp(record)
146       val message = serializer.deserialize(record.key(), record.value(), headers, timestamp)
147       logger.trace(s"Consumed message [$topic:${record.partition}:${record.offset}] $message")
148       message match {
149         case m: Change =>
150           cache.fireChange(timestamp, m.feature)
151           cache.put(m.feature)
152           updates.increment()
153           latency.foreach(_.apply(m.feature))
154 
155         case m: Delete =>
156           cache.fireDelete(timestamp, m.id, cache.query(m.id).orNull)
157           cache.remove(m.id)
158           deletes.increment()
159 
160         case _: Clear =>
161           cache.fireClear(timestamp)
162           cache.clear()
163           clears.increment()
164 
165         case m => throw new IllegalArgumentException(s"Unknown message: $m")
166       }

169     /**
170      * Handles initial loaded 'from-beginning' without indexing features in the spatial index. Will still
171      * trigger message events.
172      *
173      * Since the entire consumer group is in this process, we shouldn't usually get rebalance events after the initial assignment.
174      * However, it can still occur if the number of partitions is administratively altered. Note that we don't try to
175      * re-read the topic if a partition is revoked and then reassigned.
176      *
177      * @param readBack initial load read back
178      */
179     private class InitialLoader(readBack: scala.concurrent.duration.Duration)
180         extends ThreadedConsumer(consumers, frequency, offsetCommitInterval, false) with Runnable with ConsumerRebalanceListener {
181 
182       import scala.collection.JavaConverters._
183 
184       private val cache = KafkaFeatureCache.nonIndexing(sft, initialLoadConfig)
185       private val toLoad = KafkaCacheLoaderImpl.this
186 
187       private val updates = Metrics.counter(s"$MetricsPrefix.readback", tags.and("op", "update"))
188       private val deletes = Metrics.counter(s"$MetricsPrefix.readback", tags.and("op", "delete"))
189       private val clears = Metrics.counter(s"$MetricsPrefix.readback", tags.and("op", "clear"))
190 
191       // track the offsets that we want to read to
192       private val offsets = new ConcurrentHashMap[Int, java.lang.Long]()
193       private val assignment = Collections.newSetFromMap(new ConcurrentHashMap[Int, java.lang.Boolean]())
194       @volatile
195       private var done: Boolean = false
196       private var latch: CountDownLatch = _
197       @volatile
198       private var submission: Future[_] = _

200       override def onPartitionsAssigned(topicPartitions: java.util.Collection[TopicPartition]): Unit = {
201         logger.debug(s"Partitions assigned: ${topicPartitions.asScala.mkString(", ")}")
202         topicPartitions.asScala.foreach { tp =>
203           if (assignment.add(tp.partition())) {
204             val consumer = consumers.find(_.assignment().contains(tp)).orNull
205             if (consumer == null) {
206               logger.warn("Partition assigned but no consumer contains the assignment")
207             } else {
208               KafkaConsumerVersions.pause(consumer, tp)
209               try {
210                 logger.debug(s"Checking offsets for [${tp.topic()}:${tp.partition()}]")
211                 // the only reliable way we've found to check max offset is to seek to the end and check the position there
212                 consumer.seekToEnd(Collections.singleton(tp))
213                 val end = consumer.position(tp)
214                 logger.debug(s"Setting max offset to [${tp.topic}:${tp.partition}:${end - 1}]")
215                 offsets.put(tp.partition(), end - 1)
216                 try {
217                   if (!readBack.isFinite) {
218                     KafkaConsumerVersions.seekToBeginning(consumer, tp)
219                   } else {
220                     val offset = Try {
221                       val time = System.currentTimeMillis() - readBack.toMillis
222                       KafkaConsumerVersions.offsetsForTimes(consumer, tp.topic, Seq(tp.partition), time).get(tp.partition)
223                     }
224                     offset match {
225                       case Success(Some(o)) =>
226                         logger.debug(s"Seeking to offset $o for read-back $readBack on [${tp.topic}:${tp.partition}]")
227                         consumer.seek(tp, o)
228 
229                       case Success(None) =>
230                         logger.debug(s"No prior offset found for read-back $readBack on [${tp.topic}:${tp.partition}], " +
231                           "reading from head of queue")
232 
233                       case Failure(e) =>
234                         logger.warn(s"Error finding initial offset: [${tp.topic}:${tp.partition}], seeking to beginning", e)
235                         KafkaConsumerVersions.seekToBeginning(consumer, tp)
236                     }
237                   }
238                 } finally {
239                   checkComplete(consumer, tp)
240                 }
241               } finally {
242                 KafkaConsumerVersions.resume(consumer, tp)
243               }
244             }
245           }
246         }
247       }

249       override def onPartitionsRevoked(topicPartitions: java.util.Collection[TopicPartition]): Unit = {
250         logger.debug(s"Partitions revoked: ${topicPartitions.asScala.mkString(", ")}")
251         topicPartitions.asScala.foreach { tp =>
252           if (offsets.remove(tp.partition) != null) {
253             latch.countDown()
254             logger.info(
255               s"Stopping initial load due to revocation of assignment for [$topic:${tp.partition}], " +
256                 s"${latch.getCount} partitions remaining")
257           }
258         }
259       }

268       override protected def consume(record: ConsumerRecord[Array[Byte], Array[Byte]]): Unit = {
269         if (done) { toLoad.consume(record) } else {
270           val headers = RecordVersions.getHeaders(record)
271           val timestamp = RecordVersions.getTimestamp(record)
272           val message = serializer.deserialize(record.key, record.value, headers, timestamp)
273           logger.trace(s"Consumed message [$topic:${record.partition}:${record.offset}] $message")
274           message match {
275             case m: Change =>
276               toLoad.cache.fireChange(timestamp, m.feature)
277               cache.put(m.feature)
278               updates.increment()
279 
280             case m: Delete =>
281               toLoad.cache.fireDelete(timestamp, m.id, cache.query(m.id).orNull)
282               cache.remove(m.id)
283               deletes.increment()
284 
285             case _: Clear =>
286               toLoad.cache.fireClear(timestamp)
287               cache.clear()
288               clears.increment()
289 
290             case m => throw new IllegalArgumentException(s"Unknown message: $m")
291           }
292           if (record.offset > 0 && record.offset % 1048576 == 0) { // magic number 2^20
293             logger.info(s"Initial load: consumed [$topic:${record.partition}:${record.offset}]")
294           }
295         }
296       }

298       def start(): Unit = {
299         LoaderStatus.startLoad(this)
300         try {
301           val partitions = consumers.head.partitionsFor(topic).asScala.map(_.partition)
302           logger.info(s"Starting initial load for [$topic] with ${partitions.size} partitions")
303           latch = new CountDownLatch(partitions.size)
304           startConsumers() // kick off the asynchronous consumer threads
305           submission = CachedThreadPool.submit(this)
306         } catch {
307           case NonFatal(e) =>
308             LoaderStatus.completedLoad(this)
309             throw e
310         }
311       }
312 
313       override def run(): Unit = {
314         try {
315           try { latch.await() } finally {
316             // stop the consumer threads, but won't close the consumers due to `closeConsumers`
317             // note: don't call this.close() as it would interrupt this thread
318             super.close()
319           }
320           // set a flag just in case the consumer threads haven't finished spinning down, so that we will
321           // pass any additional messages back to the main loader
322           done = true
323           logger.info(s"Finished initial load, transferring to indexed cache for [$topic]")
324           cache.query(Filter.INCLUDE).foreach(toLoad.cache.put)
325           logger.info(s"Finished transfer for [$topic], starting normal load")
326           toLoad.startConsumers()
327         } finally {
328           LoaderStatus.completedLoad(this)
329         }
330       }
331 
332       override def close(): Unit = {
333         try { super.close() } finally {
334           if (submission != null && !submission.isDone) {
335             submission.cancel(true)
336           }
337         }
338       }

340       /**
341        * Checks the current position of the consumer, and notifies through the countdown latch when the initial load
342        * is complete
343        *
344        * @param consumer consumer
345        * @param tp topic and partition to check
346        */
347       private def checkComplete(consumer: Consumer[Array[Byte], Array[Byte]], tp: TopicPartition): Unit = {
348         val position = consumer.position(tp)
349         // once we've hit the max offset for the partition, remove from the offset map so we don't double count it
350         if (position > offsets.getOrDefault(tp.partition(), Long.MaxValue) && offsets.remove(tp.partition) != null) {
351           latch.countDown()
352           logger.info(s"Initial load completed for [$topic:${tp.partition}], ${latch.getCount} partitions remaining")
353         }
354       }

356       /**
357        * Consumer runnable that tracks when we have completed the initial load
358        *
359        * @param id id
360        * @param consumer consumer
361        * @param handler error handler
362        */
363       private class InitialLoaderConsumerRunnable(id: String, consumer: Consumer[Array[Byte], Array[Byte]], handler: ConsumerErrorHandler)
364           extends ConsumerRunnable(id, consumer, handler) {
365         override protected def processPoll(result: ConsumerRecords[Array[Byte], Array[Byte]]): Unit = {
366           try { super.processPoll(result) } finally {
367             result.partitions().asScala.foreach(checkComplete(consumer, _))
368           }
369         }
370       }
