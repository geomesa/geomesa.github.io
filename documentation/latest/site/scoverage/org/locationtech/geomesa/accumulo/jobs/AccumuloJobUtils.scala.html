60   /**
61    * Gets a query plan that can be satisfied via AccumuloInputFormat - e.g. only 1 table and configuration.
62    */
63   def getSingleQueryPlan(ds: AccumuloDataStore, query: Query): AccumuloQueryPlan = {
64     // disable join plans
65     JoinIndex.AllowJoinPlans.set(false)
66 
67     try {
68       lazy val fallbackIndex = {
69         val schema = ds.getSchema(query.getTypeName)
70         ds.manager.indices(schema, IndexMode.Read).headOption.getOrElse {
71           throw new IllegalStateException(s"Schema '${schema.getTypeName}' does not have any readable indices")
72         }
73       }
74 
75       val queryPlans = ds.getQueryPlan(query)
76 
77       if (queryPlans.isEmpty) {
78         EmptyPlan(FilterStrategy(fallbackIndex, None, Some(Filter.EXCLUDE), temporal = false, Float.PositiveInfinity))
79       } else if (queryPlans.lengthCompare(1) > 0) {
80         // this query requires multiple scans, which we can't execute from some input formats
81         // instead, fall back to a full table scan
82         logger.warn("Desired query plan requires multiple scans - falling back to full table scan")
83         val qps = ds.getQueryPlan(query, Some(fallbackIndex.identifier))
84         if (qps.lengthCompare(1) > 0 || qps.exists(_.tables.lengthCompare(1) > 0)) {
85           logger.error("The query being executed requires multiple scans, which is not currently " +
86               "supported by GeoMesa. Your result set will be partially incomplete. " +
87               s"Query: ${filterToString(query.getFilter)}")
88         }
89         qps.head
90       } else {
91         val qp = queryPlans.head
92         if (qp.tables.lengthCompare(1) > 0) {
93           logger.error("The query being executed requires multiple scans, which is not currently " +
94               "supported by GeoMesa. Your result set will be partially incomplete. " +
95               s"Query: ${filterToString(query.getFilter)}")
96         }
97         qp
98       }
99     } finally {
100       // make sure we reset the thread locals
101       JoinIndex.AllowJoinPlans.remove()
102     }
103   }
104 
105   /**
106     * Get a sequence of one or more query plans, which is guaranteed not to contain
107     * a JoinPlan (return a fallback in this case). If we get multiple scan plans,
108     * that's groovy.
109     */
110   def getMultipleQueryPlan(ds: AccumuloDataStore, query: Query): Seq[AccumuloQueryPlan] = {
111     // disable join plans
112     JoinIndex.AllowJoinPlans.set(false)
113 
114     try {
115       lazy val fallbackIndex = {
116         val schema = ds.getSchema(query.getTypeName)
117         ds.manager.indices(schema, IndexMode.Read).headOption.getOrElse {
118           throw new IllegalStateException(s"Schema '${schema.getTypeName}' does not have any readable indices")
119         }
120       }
121 
122       val queryPlans = ds.getQueryPlan(query)
123       if (queryPlans.isEmpty) {
124         Seq(EmptyPlan(FilterStrategy(fallbackIndex, None, Some(Filter.EXCLUDE), temporal = false, Float.PositiveInfinity)))
125       } else {
126         queryPlans
127       }
128     } finally {
129       // make sure we reset the thread locals
130       JoinIndex.AllowJoinPlans.remove()
131     }
132   }
