88     val sft = from.ds.getSchema(typeName)

104   lazy private val indexPairs: Seq[(GeoMesaFeatureIndex[_, _], GeoMesaFeatureIndex[_, _])] = {
105     val all = from.ds.manager.indices(sft)
106     val fromIndices = if (indices.isEmpty) { all } else {
107       val builder = Seq.newBuilder[GeoMesaFeatureIndex[_, _]]
108       indices.foreach { ident =>
109         val filtered = all.filter(_.identifier.contains(ident))
110         if (filtered.isEmpty) {
111           throw new IllegalArgumentException(
112             s"Index '$ident' does not exist in the schema. Available indices: ${all.map(_.identifier).mkString(", ")}")
113         }
114         logger.debug(s"Mapped identifier $ident to ${filtered.map(_.identifier).mkString(", ")}")
115         builder ++= filtered
116       }
117       builder.result.distinct
118     }
119     fromIndices.map(from => from -> to.ds.manager.index(sft, from.identifier))
120   }

124   lazy private val fromPartitions: Seq[Option[String]] = {
125     if (sft.isPartitioned) {
126       val builder = ListBuffer.empty[String]
127       lazy val partitioning = TablePartition(from.ds, sft).get
128       lazy val sf = new ScalaSimpleFeature(sft, "")
129       builder ++= partitions.map {
130         case PartitionName(name) => name
131         case PartitionValue(value) =>
132           sf.setAttribute(sft.getDtgIndex.get, value)
133           val partition = partitioning.partition(sf)
134           logger.debug(s"Generated partition $partition from value $value")
135           partition
136       }
137       if (builder.isEmpty) {
138         logger.debug("No partitions specified - loading all partitions from store")
139         builder ++= indexPairs.flatMap(_._1.getPartitions)
140       }
141       builder.result.distinct.sorted.map(Option.apply)
142     } else {
143       if (partitions.nonEmpty) {
144         throw new IllegalArgumentException("partitions are not applicable for a non-partitioned schema")
145       }
146       Seq(None)
147     }

171   def call(resume: Boolean): Set[CopyResult] = {
172     val results = Collections.newSetFromMap(new ConcurrentHashMap[CopyResult, java.lang.Boolean]())
173     CachedThreadPool.executor(options.tableConcurrency) { executor =>
174       fromPartitions.foreach { partition =>
175         indexPairs.foreach { case (fromIndex, toIndex) =>
176           val partitionLogId = s"${partition.fold(s"index")(p => s"partition $p")} ${fromIndex.identifier}"
177           val runnable: Runnable = () => {
178             logger.info(s"Copying $partitionLogId")
179             val result = copy(fromIndex, toIndex, partition, resume, partitionLogId)
180             result.error match {
181               case None =>
182                 logger.info(s"Bulk copy complete for $partitionLogId")
183               case Some(e) =>
184                 logger.error(s"Error copying $partitionLogId: ${e.getMessage}")
185                 logger.debug(s"Error copying $partitionLogId", e)
186             }
187             results.add(result)
188           }
189           executor.submit(runnable)
190         }
191       }
192     }
193     results.asScala.toSet
194   }

228   /**
229    * Copy a single index + partition
230    *
231    * @param fromTable from table
232    * @param toIndex to index
233    * @param partition partition name - must be Some if schema is partitioned
234    * @param resume use any partial results from a previous run, if present
235    * @param partitionLogId identifier for log messages
236    * @param fileResults set to hold files that we've copied successfully
237    * @return result
238    */
239   private def copy(
240       fromTable: String,
241       toIndex: GeoMesaFeatureIndex[_, _],
242       partition: Option[String],
243       resume: Boolean,
244       partitionLogId: String,
245       fileResults: java.util.Set[String]): Unit = {
246 
247     require(sft.isPartitioned == partition.isDefined) // sanity check - this should always be true due to our setup
248 
249     val completeMarker = new Path(exportPath, s"$fromTable.complete")
250     if (exportFs.exists(completeMarker)) {
251       if (resume) {
252         logger.debug("Skipping already completed copy")
253         return
254       } else {
255         exportFs.delete(completeMarker, false)
256       }
257     }
258 
259     val tableExportPath = new Path(exportPath, fromTable)
260     val distcpPath = new Path(tableExportPath, "distcp.txt")
261     val copyToDir = new Path(tableExportPath, "files")
262     val cloneTable = s"${fromTable}_bc_tmp"
263 
264     logger.debug(s"Source table $fromTable (${from.tableOps.tableIdMap().get(fromTable)})")
265     logger.debug(s"Export path $tableExportPath")
266 
267     if (resume && from.tableOps.exists(cloneTable)) {
268       logger.debug(s"Using existing cloned table $cloneTable - ensuring table is offline")
269       from.tableOps.offline(cloneTable, true)
270     } else {
271       // clone the table as we have to take it offline in order to export it
272       // note that cloning is just a metadata op as it shares the underlying data files (until they change)
273       logger.debug(s"Checking for existence and deleting any existing cloned table $cloneTable")
274       from.ds.adapter.deleteTable(cloneTable) // no-op if table doesn't exist
275       logger.debug(s"Cloning $fromTable to $cloneTable")
276       from.tableOps.clone(fromTable, cloneTable, true, Collections.emptyMap(), Collections.emptySet()) // use 2.0 method for compatibility
277       logger.debug(s"Taking $cloneTable offline")
278       from.tableOps.offline(cloneTable, true)
279     }
280 
281     if (resume && exportFs.exists(distcpPath) && exportFs.getFileStatus(distcpPath).getLen > 0) {
282       logger.debug(s"Using existing export results $distcpPath")
283     } else {
284       if (exportFs.exists(tableExportPath)) {
285         logger.debug(s"Deleting existing export directory $tableExportPath")
286         exportFs.delete(tableExportPath, true)
287       }
288       logger.debug(s"Exporting table to $tableExportPath")
289       from.tableOps.exportTable(cloneTable, tableExportPath.toString)
290 
291       if (!exportFs.exists(distcpPath) || exportFs.getFileStatus(distcpPath).getLen == 0) {
292         throw new RuntimeException(s"Could not read table export results at $distcpPath")
293       }
294     }
295 
296     // ensures the destination table exists
297     logger.debug(s"Checking destination for table $fromTable")
298     to.ds.adapter.createTable(toIndex, partition, Seq.empty)
299     val toTable = try { toIndex.getTableName(partition) } catch {
300       case NonFatal(e) => throw new RuntimeException("Could not get destination table", e)
301     }
302     logger.debug(s"Destination table $toTable (${to.tableOps.tableIdMap().get(toTable)})")
303 
304     // create splits, do this separately in case the table already exists
305     val splits = new java.util.TreeSet(from.tableOps.listSplits(cloneTable))
306     val existingSplits = to.tableOps.listSplits(toTable)
307     splits.removeAll(existingSplits)
308     if (!splits.isEmpty) {
309       if (!existingSplits.isEmpty) {
310         logger.warn(s"Detected split mismatch between source ($fromTable) and destination ($toTable) for $partitionLogId")
311       }
312       logger.debug(s"Adding splits to destination table $toTable")
313       to.tableOps.addSplits(toTable, splits)
314     }
315 
316     val copyErrors = Collections.newSetFromMap(new ConcurrentHashMap[Throwable, java.lang.Boolean]())
317     // read the distcp.txt file produced by the table export
318     // consumer: (src, dest) => Unit
319     def distCpConsumer(threads: Int)(consumer: (Path, Path) => Unit): Unit = {
320       logger.debug(s"Reading $distcpPath")
321       WithClose(IOUtils.lineIterator(exportFs.open(distcpPath), StandardCharsets.UTF_8)) { files =>
322         CachedThreadPool.executor(threads) { executor =>
323           files.asScala.foreach { file =>
324             val runnable: Runnable = () => {
325               val path = new Path(file)
326               val copy = new Path(copyToDir, path.getName)
327               try {
328                 if (resume && exportFs.exists(copy) &&
329                   path.getFileSystem(from.conf).getFileStatus(path).getLen == exportFs.getFileStatus(copy).getLen) {
330                   logger.debug(s"Using existing copy of $path at $copy")
331                 } else {
332                   consumer(path, copy)
333                 }
334               } catch {
335                 // catch Throwable so NoClassDefFound still gets logged
336                 case e: Throwable =>
337                   copyErrors.add(e)
338                   logger.error(s"Failed to copy $path to $copy: ${e.getMessage}")
339                   logger.debug(s"Failed to copy $path to $copy", e)
340               }
341             }
342             executor.submit(runnable)
343           }
344         }
345       }
346     }
347 
348     if (options.distCp) {
349       var inputPath = distcpPath
350       val distCpFiles = ArrayBuffer.empty[String]
351       if (resume) {
352         logger.debug(s"Checking copy status of files in $distcpPath")
353         inputPath = new Path(tableExportPath, "distcp-remaining.txt")
354         WithClose(exportFs.create(inputPath, true)) { out =>
355           distCpConsumer(1) { (path, _) =>
356             logger.debug(s"Adding $path to distcp")
357             out.writeUTF(s"$path\n")
358             distCpFiles += path.getName
359           }
360         }
361       } else {
362         logger.debug(s"Checking file list at $distcpPath")
363         distCpConsumer(1) { (path, _) =>
364           distCpFiles += path.getName
365         }
366       }
367       val job = new DistCp(from.conf, DistributedCopyOptions(inputPath, copyToDir)).execute()
368       logger.info(s"Tracking available at ${job.getStatus.getTrackingUrl}")
369       while (!job.isComplete) {
370         Thread.sleep(500)
371       }
372       if (job.isSuccessful) {
373         logger.info(s"Successfully copied data to $copyToDir")
374         fileResults.addAll(distCpFiles.asJava)
375       } else {
376         val msg = s"DistCp job failed with state ${job.getStatus.getState} due to: ${job.getStatus.getFailureInfo}"
377         copyErrors.add(new RuntimeException(msg))
378         logger.error(msg)
379       }
380     } else {
381       distCpConsumer(options.fileConcurrency) { (path, copy) =>
382         logger.debug(s"Copying $path to $copy")
383         val fs = path.getFileSystem(from.conf)
384         if (FileUtil.copy(fs, path, exportFs, copy, false, true, to.conf)) {
385           fileResults.add(path.getName)
386         } else {
387           // consolidate error handling in the catch block
388           throw new IOException(s"Failed to copy $path to $copy, copy returned false")
389         }
390       }
391     }
392     if (!copyErrors.isEmpty) {
393       val e = new RuntimeException("Error copying data files")
394       copyErrors.asScala.foreach(e.addSuppressed)
395       throw e
396     }
397 
398     logger.debug(s"Loading rfiles from $copyToDir to $toTable")
399     val importDir = to.tableOps.importDirectory(copyToDir.toString).to(toTable)
400     try { importDir.ignoreEmptyDir(true) } catch {
401       case _: NoSuchMethodError => // accumulo 2.0, ignore
402     }
403     try { importDir.load() } catch {
404       case e: IllegalArgumentException => logger.trace("Error importing directory:", e) // should mean empty dir
405     }
406 
407     // create marker indicating this copy was successful
408     logger.debug(s"Creating completion marker $completeMarker")
409     exportFs.create(completeMarker).close()
410 
411     // cleanup
412     logger.debug(s"Deleting export path $tableExportPath")
413     exportFs.delete(tableExportPath, true)
414     logger.debug(s"Deleting clone table $cloneTable")
415     from.tableOps.delete(cloneTable)
416   }

419     if (!closed) {
420       closed = true
421       CloseWithLogging(tryFrom.toOption)
422       CloseWithLogging(tryTo.toOption)
423     }
