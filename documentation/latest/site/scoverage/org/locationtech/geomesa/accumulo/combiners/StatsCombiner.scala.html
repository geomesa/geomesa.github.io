25 /**
26  * Combiner for serialized stats. Should be one instance configured per catalog table. Simple feature
27  * types and columns with stats should be set in the configuration.
28  */
29 class StatsCombiner extends Combiner with LazyLogging {
30 
31   import StatsCombiner.{SeparatorOption, SftOption}
32 
33   import scala.collection.JavaConverters._
34 
35   private var serializers: Map[String, StatSerializer] = _
36   private var separator: Char = '~'
37 
38   override def init(source: SortedKeyValueIterator[Key, Value],
39       options: java.util.Map[String, String],
40       env: IteratorEnvironment): Unit = {
41     super.init(source, options, env)
42     serializers = options.asScala.toMap.collect {
43       case (k, v) if k.startsWith(SftOption) =>
44         val typeName = k.substring(SftOption.length)
45         (typeName, StatSerializer(SimpleFeatureTypes.createType(typeName, v)))
46     }
47     separator = Option(options.get(SeparatorOption)).map(_.charAt(0)).getOrElse('~')
48   }
49 
50   override def reduce(key: Key, iter: java.util.Iterator[Value]): Value = {
51     val head = iter.next()
52     if (!iter.hasNext) { head } else {
53       val sftName = try {
54         KeyValueStoreMetadata.decodeRow(key.getRow.getBytes, separator)._1
55       } catch {
56         // back compatible check
57         case NonFatal(_) => StatsCombiner.SingleRowMetadata.getTypeName(key.getRow)
58       }
59       val serializer = serializers(sftName)
60 
61       var stat = deserialize(head, serializer)
62 
63       while (stat == null) {
64         if (iter.hasNext) {
65           stat = deserialize(iter.next, serializer)
66         } else {
67           return head // couldn't parse anything... return head value and let client deal with it
68         }
69       }
70 
71       iter.asScala.foreach { s =>
72         try { stat += serializer.deserialize(s.get) } catch {
73           case NonFatal(e) => logger.error("Error combining stats:", e)
74         }
75       }
76 
77       new Value(serializer.serialize(stat))
78     }
79   }
80 
81   private def deserialize(value: Value, serializer: StatSerializer): Stat = {
82     try { serializer.deserialize(value.get) } catch {
83       case NonFatal(e) => logger.error("Error deserializing stat:", e); null
84     }
85   }
86 }

92   val CombinerName = "stats-combiner"
93 
94   val SftOption = "sft-"
95   val SeparatorOption = "sep"
96 
97   def configure(sft: SimpleFeatureType, connector: AccumuloClient, table: String, separator: String): Unit = {
98     val sftKey = getSftKey(sft)
99     val sftOpt = SimpleFeatureTypes.encodeType(sft)
100 
101     getExisting(connector, table) match {
102       case None => attach(connector, table, options(separator) + (sftKey -> sftOpt))
103       case Some(existing) =>
104         val existingSfts = existing.getOptions.asScala.filter(_._1.startsWith(SftOption))
105         if (!existingSfts.get(sftKey).contains(sftOpt)) {
106           connector.tableOperations().removeIterator(table, CombinerName, java.util.EnumSet.allOf(classOf[IteratorScope]))
107           attach(connector, table, existingSfts.toMap ++ options(separator) + (sftKey -> sftOpt))
108         }
109     }
110   }
111 
112   def remove(sft: SimpleFeatureType, connector: AccumuloClient, table: String, separator: String): Unit = {
113     getExisting(connector, table).foreach { existing =>
114       val sftKey = getSftKey(sft)
115       val existingSfts = existing.getOptions.asScala.filter(_._1.startsWith(SftOption))
116       if (existingSfts.asJava.containsKey(sftKey)) {
117         connector.tableOperations().removeIterator(table, CombinerName, java.util.EnumSet.allOf(classOf[IteratorScope]))
118         if (existingSfts.size > 1) {
119           attach(connector, table, (existingSfts.toMap - sftKey) ++ options(separator))
120         }
121       }
122     }
123   }
124 
125   def list(connector: AccumuloClient, table: String): scala.collection.Map[String, String] = {
126     getExisting(connector, table) match {
127       case None => Map.empty
128       case Some(existing) =>
129         existing.getOptions.asScala.collect {
130           case (k, v) if k.startsWith(SftOption) => (k.substring(SftOption.length), v)
131         }
132     }
133   }
134 
135   private def getExisting(connector: AccumuloClient, table: String): Option[IteratorSetting] = {
136     if (!connector.tableOperations().exists(table)) { None } else {
137       Option(connector.tableOperations().getIteratorSetting(table, CombinerName, IteratorScope.scan))
138     }
139   }
140 
141   private def options(separator: String): Map[String, String] =
142     Map(SeparatorOption -> separator, "all" -> "true")
143 
144   private def getSftKey(sft: SimpleFeatureType): String = s"$SftOption${sft.getTypeName}"
145 
146   private def attach(connector: AccumuloClient, table: String, options: Map[String, String]): Unit = {
147     // priority needs to be less than the versioning iterator at 20
148     // note: we use the old class name to allow full interop between gm versions
149     val is = new IteratorSetting(10, CombinerName, "org.locationtech.geomesa.accumulo.data.stats.StatsCombiner")
150     options.foreach { case (k, v) => is.addOption(k, v) }
151     connector.tableOperations().attachIterator(table, is)
152   }
