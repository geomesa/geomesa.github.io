24 /**
25  * Combiner for serialized stats. Should be one instance configured per catalog table. Simple feature
26  * types and columns with stats should be set in the configuration.
27  */
28 class StatsCombiner extends Combiner with LazyLogging {
29 
30   import StatsCombiner.{SeparatorOption, SftOption}
31 
32   import scala.collection.JavaConverters._
33 
34   private var serializers: Map[String, StatSerializer] = _
35   private var separator: Char = '~'
36 
37   override def init(source: SortedKeyValueIterator[Key, Value],
38       options: java.util.Map[String, String],
39       env: IteratorEnvironment): Unit = {
40     super.init(source, options, env)
41     serializers = options.asScala.toMap.collect {
42       case (k, v) if k.startsWith(SftOption) =>
43         val typeName = k.substring(SftOption.length)
44         (typeName, StatSerializer(SimpleFeatureTypes.createType(typeName, v)))
45     }
46     separator = Option(options.get(SeparatorOption)).map(_.charAt(0)).getOrElse('~')
47   }
48 
49   override def reduce(key: Key, iter: java.util.Iterator[Value]): Value = {
50     val head = iter.next()
51     if (!iter.hasNext) { head } else {
52       val sftName = try {
53         KeyValueStoreMetadata.decodeRow(key.getRow.getBytes, separator)._1
54       } catch {
55         // back compatible check
56         case NonFatal(_) => StatsCombiner.SingleRowMetadata.getTypeName(key.getRow)
57       }
58       val serializer = serializers(sftName)
59 
60       var stat = deserialize(head, serializer)
61 
62       while (stat == null) {
63         if (iter.hasNext) {
64           stat = deserialize(iter.next, serializer)
65         } else {
66           return head // couldn't parse anything... return head value and let client deal with it
67         }
68       }
69 
70       iter.asScala.foreach { s =>
71         try { stat += serializer.deserialize(s.get) } catch {
72           case NonFatal(e) => logger.error("Error combining stats:", e)
73         }
74       }
75 
76       new Value(serializer.serialize(stat))
77     }
78   }
79 
80   private def deserialize(value: Value, serializer: StatSerializer): Stat = {
81     try { serializer.deserialize(value.get) } catch {
82       case NonFatal(e) => logger.error("Error deserializing stat:", e); null
83     }
84   }
85 }

91   val CombinerName = "stats-combiner"
92 
93   val SftOption = "sft-"
94   val SeparatorOption = "sep"
95 
96   def configure(sft: SimpleFeatureType, connector: AccumuloClient, table: String, separator: String): Unit = {
97     val sftKey = getSftKey(sft)
98     val sftOpt = SimpleFeatureTypes.encodeType(sft)
99 
100     getExisting(connector, table) match {
101       case None => attach(connector, table, options(separator) + (sftKey -> sftOpt))
102       case Some(existing) =>
103         val existingSfts = existing.getOptions.asScala.filter(_._1.startsWith(SftOption))
104         if (!existingSfts.get(sftKey).contains(sftOpt)) {
105           connector.tableOperations().removeIterator(table, CombinerName, java.util.EnumSet.allOf(classOf[IteratorScope]))
106           attach(connector, table, existingSfts.toMap ++ options(separator) + (sftKey -> sftOpt))
107         }
108     }
109   }
110 
111   def remove(sft: SimpleFeatureType, connector: AccumuloClient, table: String, separator: String): Unit = {
112     getExisting(connector, table).foreach { existing =>
113       val sftKey = getSftKey(sft)
114       val existingSfts = existing.getOptions.asScala.filter(_._1.startsWith(SftOption))
115       if (existingSfts.asJava.containsKey(sftKey)) {
116         connector.tableOperations().removeIterator(table, CombinerName, java.util.EnumSet.allOf(classOf[IteratorScope]))
117         if (existingSfts.size > 1) {
118           attach(connector, table, (existingSfts.toMap - sftKey) ++ options(separator))
119         }
120       }
121     }
122   }
123 
124   def list(connector: AccumuloClient, table: String): scala.collection.Map[String, String] = {
125     getExisting(connector, table) match {
126       case None => Map.empty
127       case Some(existing) =>
128         existing.getOptions.asScala.collect {
129           case (k, v) if k.startsWith(SftOption) => (k.substring(SftOption.length), v)
130         }
131     }
132   }
133 
134   private def getExisting(connector: AccumuloClient, table: String): Option[IteratorSetting] = {
135     if (!connector.tableOperations().exists(table)) { None } else {
136       Option(connector.tableOperations().getIteratorSetting(table, CombinerName, IteratorScope.scan))
137     }
138   }
139 
140   private def options(separator: String): Map[String, String] =
141     Map(SeparatorOption -> separator, "all" -> "true")
142 
143   private def getSftKey(sft: SimpleFeatureType): String = s"$SftOption${sft.getTypeName}"
144 
145   private def attach(connector: AccumuloClient, table: String, options: Map[String, String]): Unit = {
146     // priority needs to be less than the versioning iterator at 20
147     // note: we use the old class name to allow full interop between gm versions
148     val is = new IteratorSetting(10, CombinerName, "org.locationtech.geomesa.accumulo.data.stats.StatsCombiner")
149     options.foreach { case (k, v) => is.addOption(k, v) }
150     connector.tableOperations().attachIterator(table, is)
151   }
