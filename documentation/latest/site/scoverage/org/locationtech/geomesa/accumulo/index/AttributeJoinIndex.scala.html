39   abstract override def getFilterStrategy(filter: Filter, hints: Hints): Option[FilterStrategy] = {
40     super.getFilterStrategy(filter, hints).flatMap { strategy =>
41       // verify that it's ok to return join plans, and filter them out if not
42       if (!requiresJoin(strategy.secondary, hints.getTransformSchema)) {
43         Some(strategy)
44       } else if (!JoinIndex.AllowJoinPlans.get) {
45         None
46       } else {
47         val primary = strategy.primary.getOrElse(Filter.INCLUDE)
48         val bounds = FilterHelper.extractAttributeBounds(primary, attribute, binding)
49         val joinMultiplier = 10f + (bounds.values.length / 10) // 10 plus 1 per 10 additional ranges being scanned
50         val multiplier = strategy.costMultiplier * joinMultiplier
51         Some(FilterStrategy(strategy.index, strategy.primary, strategy.secondary, strategy.temporal, multiplier, hints))
52       }
53     }

61     * @param transform transform being applied, if any
62     * @return
63     */
64   private def requiresJoin(filter: Option[Filter], transform: Option[SimpleFeatureType]): Boolean =
65     !canUseIndexSchema(filter, transform) && !canUseIndexSchemaPlusKey(filter, transform)
66 
67   /**
68     * Determines if the given filter and transform can operate on index encoded values.
69     */
70   def canUseIndexSchema(filter: Option[Filter], transform: Option[SimpleFeatureType]): Boolean = {
71     // verify that transform *does* exist and only contains fields in the index sft,
72     // and that filter *does not* exist or can be fulfilled by the index sft
73     supportsTransform(transform) && supportsFilter(filter)
74   }
75 
76   /**
77     * Determines if the given filter and transform can operate on index encoded values
78     * in addition to the values actually encoded in the attribute index keys
79     */
80   def canUseIndexSchemaPlusKey(filter: Option[Filter], transform: Option[SimpleFeatureType]): Boolean = {
81     transform.exists { t =>
82       val attributes = t.getAttributeDescriptors.asScala.map(_.getLocalName)
83       attributes.forall(a => a == attribute || indexSft.indexOf(a) != -1) && supportsFilter(filter)
84     }
85   }
86 
87   def supportsTransform(transform: Option[SimpleFeatureType]): Boolean =
88     transform.exists(_.getAttributeDescriptors.asScala.map(_.getLocalName).forall(indexSft.indexOf(_) != -1))
89 
90   def supportsFilter(filter: Option[Filter]): Boolean =
91     filter.forall(FilterHelper.propertyNames(_, sft).forall(indexSft.indexOf(_) != -1))
