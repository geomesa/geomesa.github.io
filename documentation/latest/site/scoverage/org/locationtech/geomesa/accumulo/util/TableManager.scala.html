28 /**
29  * Manages table creation/deletion
30  *
31  * @param client accumulo client - note, this client must be cleaned up externally (as it's usually shared)
32  */
33 class TableManager(client: AccumuloClient) {
34 
35   private val delegate: TableLock = TableSynchronization(TableCreationSync.get) match {
36     case TableSynchronization.ZooKeeper => new TableLockZk(client)
37     case TableSynchronization.Local     => new TableLockLocal(client)
38     case TableSynchronization.None      => new TableLockNone(client)
39   }
40 
41   /**
42    * Create table if it does not exist.
43    *
44    * Note: to avoid having to lock and check namespaces, the namespace for the table must already exist. Generally
45    * all tables for a data store share a namespace, and the namespace is created up front, so it will always exist.
46    *
47    * @param table table name
48    * @param useLogicalTime use logical table time
49    * @return true if table was created, false if table already exists
50    */
51   def ensureTableExists(table: String, useLogicalTime: Boolean = true): Boolean =
52     delegate.ensureTableExists(table, if (useLogicalTime) { TimeType.LOGICAL } else { TimeType.MILLIS })
53 
54   /**
55    * Create namespace if it does not exist
56    *
57    * @param namespace namespace
58    */
59   def ensureNamespaceExists(namespace: String): Unit = delegate.ensureNamespaceExists(namespace)
60 
61   /**
62    * Delete a table
63    *
64    * @param table table to delete
65    */
66   def deleteTable(table: String): Unit = delegate.deleteTable(table)
67 
68   // note: shadows IndexAdapter.renameTable
69   // noinspection ScalaUnusedSymbol
70   def renameTable(from: String, to: String): Unit = delegate.renameTable(from, to)
71 }

130   /**
131    * Table utility class
132    *
133    * @param client accumulo client
134    */
135   private abstract class TableLock(client: AccumuloClient) extends DistributedLocking {
136 
137     private val timeoutMillis = DistributedLockTimeout.toDuration.map(_.toMillis).getOrElse {
138       // note: this property has a valid default value so this exception should never be triggered
139       throw new IllegalArgumentException(s"Couldn't convert '${DistributedLockTimeout.get}' to a duration")
140     }
141 
142     // note: value is not used here, we're treating this as a set
143     private val tableCache: Cache[String, java.lang.Boolean] =
144       Caffeine.newBuilder().expireAfterWrite(TableCacheExpiry.toDuration.get.toMillis, TimeUnit.MILLISECONDS).build()
145 
146     private val nsCache: Cache[String, java.lang.Boolean] =
147       Caffeine.newBuilder().expireAfterWrite(TableCacheExpiry.toDuration.get.toMillis, TimeUnit.MILLISECONDS).build()
148 
149     /**
150      * Create the table if it doesn't exist
151      *
152      * @param table table name
153      * @param timeType table time type
154      * @return true if table was created, false if it already exists
155      */
156     def ensureTableExists(table: String, timeType: TimeType): Boolean = {
157       var created = false
158       var err: Throwable = null
159 
160       // noinspection ScalaUnusedSymbol
161       def load(ignoredKey: String): java.lang.Boolean = {
162         try {
163           withLock(tablePath(table), timeoutMillis, {
164             val tableOps = client.tableOperations()
165             if (!tableOps.exists(table)) {
166               try {
167                 tableOps.create(table, new NewTableConfiguration().setTimeType(timeType))
168                 created = true
169               } catch {
170                 case _: TableExistsException => onTableExists(table)
171               }
172               val start = System.currentTimeMillis()
173               while (!tableOps.exists(table)) {
174                 if (System.currentTimeMillis() - start > timeoutMillis) {
175                   throw new RuntimeException(
176                     s"Tried to create table '$table', but it has not been created after timeout of ${timeoutMillis}ms")
177                 }
178                 Thread.sleep(10)
179               }
180             }
181           })
182           java.lang.Boolean.FALSE
183         } catch {
184           case NonFatal(e) => err = e; throw e
185         }
186       }
187 
188       tableCache.get(table, load)
189       if (err != null) {
190         throw err
191       }
192       created
193     }
194 
195     /**
196      * Creates the namespace if it doesn't exist
197      *
198      * @param namespace namespace (or table name with namespace included)
199      * @return true if namespace was created, false if it already existed
200      */
201     def ensureNamespaceExists(namespace: String): Unit = {
202       val ns = namespace.takeWhile(_ != '.')
203       var err: Throwable = null
204 
205       // noinspection ScalaUnusedSymbol
206       def load(ignoredKey: String): java.lang.Boolean = {
207         try {
208           withLock(nsPath(ns), timeoutMillis, {
209             val nsOps = client.namespaceOperations
210             if (!nsOps.exists(ns)) {
211               try { nsOps.create(ns) } catch {
212                 case _: NamespaceExistsException => onNamespaceExists(ns)
213               }
214               val start = System.currentTimeMillis()
215               while (!nsOps.exists(ns)) {
216                 if (System.currentTimeMillis() - start > timeoutMillis) {
217                   throw new RuntimeException(
218                     s"Tried to create namespace '$ns', but it has not been created after timeout of ${timeoutMillis}ms")
219                 }
220                 Thread.sleep(10)
221               }
222             }
223           })
224           java.lang.Boolean.FALSE
225         } catch {
226           case NonFatal(e) => err = e; throw e
227         }
228       }
229 
230       nsCache.get(ns, load)
231       if (err != null) {
232         throw err
233       }
234     }
235 
236     /**
237      * Rename a table
238      *
239      * @param from current name
240      * @param to new name
241      */
242     def renameTable(from: String, to: String): Unit = {
243       withLock(tablePath(from), timeoutMillis, {
244         val tableOps = client.tableOperations()
245         if (tableOps.exists(from)) {
246           withLock(tablePath(to), timeoutMillis, {
247             tableOps.rename(from, to)
248             tableCache.put(to, java.lang.Boolean.FALSE)
249           })
250         }
251         tableCache.invalidate(from)
252       })
253     }
254 
255     /**
256      * Delete a table
257      *
258      * @param table table to delete
259      */
260     def deleteTable(table: String): Unit = {
261       withLock(tablePath(table), timeoutMillis, {
262         val tableOps = client.tableOperations()
263         if (tableOps.exists(table)) {
264           tableOps.delete(table)
265         }
266         tableCache.invalidate(table)
267       })
268     }
269 
270     // can happen sometimes with multiple threads but usually not a problem
271     protected def onTableExists(table: String): Unit = {}
272     // can happen sometimes with multiple threads but usually not a problem
273     protected def onNamespaceExists(namespace: String): Unit = {}
274 
275     /**
276      * ZK path for acquiring a table lock
277      *
278      * @param table table name
279      * @return
280      */
281     private def tablePath(table: String): String =
282       s"/org.locationtech.geomesa/table-locks/${StringSerialization.alphaNumericSafeString(table)}"
283 
284     /**
285      * ZK path for acquiring a namespace lock
286      *
287      * @param namespace namespace
288      * @return
289      */
290     private def nsPath(namespace: String): String =
291       s"/org.locationtech.geomesa/ns-locks/${StringSerialization.alphaNumericSafeString(namespace)}"
292   }
