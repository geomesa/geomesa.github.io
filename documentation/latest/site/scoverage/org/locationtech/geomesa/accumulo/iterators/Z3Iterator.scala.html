21   /**
22    * Configure the iterator
23    *
24    * @param values index values
25    * @param offset offset for z-value in each row
26    * @param compatibility compatibility mode
27    * @param priority iterator priority
28    * @return
29    */
30   def configure(
31       values: Z3IndexValues,
32       offset: Int,
33       compatibility: Option[FilterCompatibility],
34       priority: Int): IteratorSetting = {
35 
36     val opts = compatibility match {
37       case None =>
38         Z3Filter.serializeToStrings(Z3Filter(values)) + (RowFilterIterator.RowOffsetKey -> offset.toString)
39 
40       case Some(FilterCompatibility.`1.3`) =>
41         val Z3IndexValues(sfc, _, bounds, _, times, _) = values
42         val xyOpts = bounds.map { case (xmin, ymin, xmax, ymax) =>
43           s"${sfc.lon.normalize(xmin)}:${sfc.lat.normalize(ymin)}:" +
44               s"${sfc.lon.normalize(xmax)}:${sfc.lat.normalize(ymax)}"
45         }
46         val tOpts = times.toSeq.sortBy(_._1).map { case (bin, times) =>
47           val time = times.map { case (t1, t2) =>
48             s"${sfc.time.normalize(t1)}:${sfc.time.normalize(t2)}"
49           }
50           s"$bin;${time.mkString(";")}"
51         }
52         Map(
53           "zxy" -> xyOpts.mkString(";"),
54           "zt"  -> tOpts.mkString(","),
55           "zo"  -> offset.toString,
56           "zl"  -> "8"
57         )
58 
59       case Some(c) =>
60         throw new NotImplementedError(s"Unknown compatibility flag: '$c'")
61     }
62 
63     val is = new IteratorSetting(priority, "z3", classOf[Z3Iterator])
64     opts.foreach { case (k, v) => is.addOption(k, v) }
65     is
66   }
