27 /**
28  * Trait for mapping stats to accumulo and back
29  */
30 trait AccumuloEventTransform[T <: AuditedEvent] extends LazyLogging {
31 
32   import AccumuloEventTransform.toRowKey
33 
34   /**
35    * Convert an event to a mutation
36    *
37    * @param event event
38    * @return
39    */
40   def toMutation(event: T): Mutation
41 
42   /**
43    * Convert accumulo scan results into an event
44    *
45    * @param entries scan entries from a single row
46    * @return
47    */
48   def toEvent(entries: Iterable[Entry[Key, Value]]): T
49 
50   /**
51    * Creates an iterator that returns Stats from accumulo scans
52    *
53    * @param scanner accumulo scanner over stored events
54    * @param typeName type name to scan
55    * @param dates dates to scan
56    * @return
57    */
58   def iterator(scanner: Scanner, typeName: String, dates: (ZonedDateTime, ZonedDateTime)): CloseableIterator[T] = {
59     try {
60       scanner.setRange(new Range(toRowKey(typeName, dates._1), true, toRowKey(typeName, dates._2), false))
61       new RowGrouper(scanner).flatMap(_.groupBy(_.getKey.getColumnFamily).values).map(toEvent)
62     } catch {
63       case NonFatal(e) =>
64         CloseQuietly(scanner).foreach(e.addSuppressed)
65         throw e
66     }
67   }
68 }

115   /**
116    * Groups scan entries by row
117    *
118    * @param scanner scanner
119    */
120   private class RowGrouper(scanner: Scanner) extends CloseableIterator[Seq[Entry[Key, Value]]] {
121 
122     private val iter = scanner.iterator()
123     private var nextEntry: Entry[Key, Value] = if (iter.hasNext) { iter.next() } else { null }
124 
125     override def hasNext: Boolean = nextEntry != null
126 
127     override def next(): Seq[Entry[Key, Value]] = {
128       if (nextEntry == null) {
129         return null
130       }
131       val currentRowKey = nextEntry.getKey.getRow.toString
132       val entries = Seq.newBuilder[Entry[Key, Value]]
133       entries += nextEntry
134       var currentEntry: Entry[Key, Value] = null
135       while (iter.hasNext && { currentEntry = iter.next(); currentEntry.getKey.getRow.toString == currentRowKey }) {
136         entries += currentEntry
137         currentEntry = null
138       }
139       nextEntry = currentEntry
140       entries.result()
141     }
142 
143     override def close(): Unit = scanner.close()
144   }
