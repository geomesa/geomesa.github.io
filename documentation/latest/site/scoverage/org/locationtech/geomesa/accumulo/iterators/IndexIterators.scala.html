28   def configure(
29       index: GeoMesaFeatureIndex[_, _],
30       values: Any,
31       priority: Int,
32       compatibility: Option[FilterCompatibility] = None): Option[IteratorSetting] = {
33     // TODO pull this out to be SPI loaded so that new indices can be added seamlessly
34     if (index.name == Z3Index.name) {
35       val offset = index.keySpace.sharding.length + index.keySpace.sharing.length
36       Some(Z3Iterator.configure(values.asInstanceOf[Z3IndexValues], offset, compatibility, priority))
37     } else if (index.name == Z2Index.name) {
38       val offset = index.keySpace.sharding.length + index.keySpace.sharing.length
39       Some(Z2Iterator.configure(values.asInstanceOf[Z2IndexValues], offset, priority))
40     } else if (index.name == S3Index.name) {
41       Some(S3Iterator.configure(values.asInstanceOf[S3IndexValues], index.keySpace.sharding.length, priority))
42     } else if (index.name == S2Index.name) {
43       Some(S2Iterator.configure(values.asInstanceOf[S2IndexValues], index.keySpace.sharding.length, priority))
44     } else {
45       None
46     }
47   }
