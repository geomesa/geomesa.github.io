36   override def rdd(conf: Configuration,
37                    sc: SparkContext,
38                    params: Map[String, String],
39                    query: Query): SpatialRDD = {
40     val paramsAsJava = params.asJava
41     val ds = DataStoreFinder.getDataStore(paramsAsJava).asInstanceOf[AccumuloDataStore]
42 
43     lazy val transform = query.getHints.getTransformSchema
44 
45     def queryPlanToRDD(sft: SimpleFeatureType, qp: AccumuloQueryPlan): RDD[SimpleFeature] = {
46       if (ds == null || sft == null || qp.isInstanceOf[EmptyPlan]) {
47         sc.emptyRDD[SimpleFeature]
48       } else {
49         val config = new JobConf(conf)
50         GeoMesaAccumuloInputFormat.configure(config, paramsAsJava, qp, Some(ds.auths))
51         sc.newAPIHadoopRDD(config, classOf[GeoMesaAccumuloInputFormat], classOf[Text], classOf[SimpleFeature]).map(_._2)
52       }
53     }
54 
55     try {
56       // get the query plan to set up the iterators, ranges, etc
57       // getMultipleQueryPlan will return the fallback if any
58       // element of the plan is a JoinPlan
59       val sft = ds.getSchema(query.getTypeName)
60       val qps = AccumuloJobUtils.getMultipleQueryPlan(ds, query)
61 
62       // can return a union of the RDDs because the query planner *should*
63       // be rewriting ORs to make them logically disjoint
64       // e.g. "A OR B OR C" -> "A OR (B NOT A) OR ((C NOT A) NOT B)"
65       val sfrdd = if (qps.lengthCompare(1) == 0 && qps.head.tables.lengthCompare(1) == 0) {
66         queryPlanToRDD(sft, qps.head) // no union needed for single query plan
67       } else {
68         // flatten and duplicate the query plans so each one only has a single table
69         val expanded = qps.flatMap {
70           case qp: BatchScanPlan => qp.tables.map(t => qp.copy(tables = Seq(t)))
71           case qp: EmptyPlan => Seq(qp)
72           case qp => throw new NotImplementedError(s"Unexpected query plan type: $qp")
73         }
74         sc.union(expanded.map(queryPlanToRDD(sft, _)))
75       }
76       SpatialRDD(sfrdd, transform.getOrElse(sft))
77     } finally {
78       if (ds != null) {
79         ds.dispose()
80       }
81     }
82   }
83 
84   /**
85     * Writes this RDD to a GeoMesa table.
86     * The type must exist in the data store, and all of the features in the RDD must be of this type.
87     *
88     * @param rdd rdd
89     * @param params params
90     * @param typeName type name
91     */
92   def save(rdd: RDD[SimpleFeature], params: Map[String, String], typeName: String): Unit = {
93     WithStore[AccumuloDataStore](params) { ds =>
94       require(ds.getSchema(typeName) != null,
95         "Feature type must exist before calling save. Call createSchema on the DataStore first.")
96     }
97 
98     rdd.foreachPartition { iter =>
99       WithStore[AccumuloDataStore](params) { ds =>
100         WithClose(ds.getFeatureWriterAppend(typeName, Transaction.AUTO_COMMIT)) { writer =>
101           iter.foreach(FeatureUtils.write(writer, _, useProvidedFid = true))
102         }
103       }
104     }
105   }
