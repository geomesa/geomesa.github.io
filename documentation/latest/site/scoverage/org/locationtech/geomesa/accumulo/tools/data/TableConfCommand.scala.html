49 class TableConfUpdateCommand extends AccumuloDataStoreCommand {
50 
51   import TableConfCommand._
52 
53   override val name = "update"
54   override val params = new UpdateParams
55 
56   override def execute(): Unit = {
57     withDataStore { ds =>
58       Command.user.info(s"Reading configuration parameters for index '${params.index}'")
59       val tables = getTableNames(ds, ds.getSchema(params.featureName), params.index)
60       val values = tables.map(getProp(ds, _, params.key)).distinct
61       values.foreach(v => Command.user.info(s"  current value: ${params.key}=$v"))
62 
63       if (values != Seq(params.newValue)) {
64         Command.user.info(s"Updating configuration parameter to '${params.newValue}'...")
65         tables.foreach(setValue(ds, _, params.key, params.newValue))
66         val updated = tables.map(getProp(ds, _, params.key)).distinct
67         updated.foreach(v => Command.user.info(s"  updated value: ${params.key}=$v"))
68       } else {
69         Command.user.info(s"'${params.key}' already set to '${params.newValue}'.")
70       }
71     }
72   }
73 }

77   def getProp(ds: AccumuloDataStore, table: String, key: String): String =
78     getProperties(ds, table).getOrElse((table, key),
79       throw new RuntimeException(s"Parameter '$key' not found in table '$table'"))
80 
81   def setValue(ds: AccumuloDataStore, table: String, key: String, value: String): Unit = {
82     try {
83       ds.connector.tableOperations.setProperty(table, key, value)
84     } catch {
85       case e: Exception => throw new RuntimeException(s"Error updating table property: ${e.getMessage}", e)
86     }
87   }
88 
89   def getProperties(ds: AccumuloDataStore, table: String): Map[(String, String), String] = {
90     try {
91       ds.connector.tableOperations.getProperties(table).asScala.map(e => ((table, e.getKey), e.getValue)).toMap
92     } catch {
93       case e: TableNotFoundException =>
94         throw new RuntimeException(s"Error: table $table does not exist: ${e.getMessage}", e)
95     }
96   }
97 
98   def getTableNames(ds: AccumuloDataStore, sft: SimpleFeatureType, index: String): Seq[String] = {
99     val tables = ds.manager.indices(sft).filter(_.name.equalsIgnoreCase(index)).flatMap(_.getTableNames(None))
100     if (tables.isEmpty) {
101       throw new IllegalArgumentException(s"Index '$index' does not exist for schema '${sft.getTypeName}'. " +
102           s"Available indices: ${ds.manager.indices(sft).map(_.name).distinct.mkString(", ")}")
103     }
104     tables
105   }
