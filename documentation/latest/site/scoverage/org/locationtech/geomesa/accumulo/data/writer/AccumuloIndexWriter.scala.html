17 /**
18   * Accumulo index writer implementation
19   *
20   * @param ds data store
21   * @param indices indices to write to
22   * @param wrapper feature wrapper
23   * @param partition partition to write to (if partitioned schema)
24   */
25 class AccumuloIndexWriter(
26     ds: AccumuloDataStore,
27     indices: Seq[GeoMesaFeatureIndex[_, _]],
28     wrapper: FeatureWrapper[WritableFeature],
29     partition: Option[String]
30   ) extends BaseIndexWriter[WritableFeature](indices, wrapper) {
31 
32   import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
33 
34   private val multiWriter = ds.connector.createMultiTableBatchWriter()
35   private val writers = indices.toArray.map { index =>
36     val table = index.getTableNames(partition) match {
37       case Seq(t) => t // should always be writing to a single table here
38       case tables => throw new IllegalStateException(s"Expected a single table but got: ${tables.mkString(", ")}")
39     }
40     multiWriter.getBatchWriter(table)
41   }
42 
43   private val colFamilyMappings = indices.map(ColumnFamilyMapper.apply).toArray
44   private val timestamps = indices.exists(i => !i.sft.isLogicalTime)
45   private val visCache = new VisibilityCache()
46 
47   override protected def append(feature: WritableFeature, values: Array[RowKeyValue[_]]): Unit = {
48     var i = 0
49     while (i < values.length) {
50       values(i) match {
51         case kv: SingleRowKeyValue[_] =>
52           val mutation = new Mutation(kv.row)
53           kv.values.foreach { v =>
54             mutation.put(colFamilyMappings(i)(v.cf), v.cq, visCache(v.vis), v.value)
55           }
56           writers(i).addMutation(mutation)
57 
58         case mkv: MultiRowKeyValue[_] =>
59           mkv.rows.foreach { row =>
60             val mutation = new Mutation(row)
61             mkv.values.foreach { v =>
62               mutation.put(colFamilyMappings(i)(v.cf), v.cq, visCache(v.vis), v.value)
63             }
64             writers(i).addMutation(mutation)
65           }
66       }
67       i += 1
68     }
69   }
70 
71   override protected def update(
72       feature: WritableFeature,
73       values: Array[RowKeyValue[_]],
74       previous: WritableFeature,
75       previousValues: Array[RowKeyValue[_]]): Unit = {
76     delete(previous, previousValues)
77     if (timestamps) {
78       // for updates, ensure that our timestamps don't clobber each other
79       multiWriter.flush()
80       Thread.sleep(1)
81     }
82     append(feature, values)
83   }
84 
85   override protected def delete(feature: WritableFeature, values: Array[RowKeyValue[_]]): Unit = {
86     var i = 0
87     while (i < values.length) {
88       values(i) match {
89         case kv: SingleRowKeyValue[_] =>
90           val mutation = new Mutation(kv.row)
91           kv.values.foreach { v =>
92             mutation.putDelete(colFamilyMappings(i)(v.cf), v.cq, visCache(v.vis))
93           }
94           writers(i).addMutation(mutation)
95 
96         case kv: MultiRowKeyValue[_] =>
97           kv.rows.foreach { row =>
98             val mutation = new Mutation(row)
99             kv.values.foreach { v =>
100               mutation.putDelete(colFamilyMappings(i)(v.cf), v.cq, visCache(v.vis))
101             }
102             writers(i).addMutation(mutation)
103           }
104       }
105       i += 1
106     }
107   }
108 
109   override def flush(): Unit = multiWriter.flush()
110 
111   override def close(): Unit = multiWriter.close()
112 }
