17 /**
18   * Accumulo index writer implementation
19   *
20   * @param ds data store
21   * @param indices indices to write to
22   * @param wrapper feature wrapper
23   * @param partition partition to write to (if partitioned schema)
24   */
25 class AccumuloIndexWriter(
26     ds: AccumuloDataStore,
27     indices: Seq[GeoMesaFeatureIndex[_, _]],
28     wrapper: FeatureWrapper[WritableFeature],
29     partition: Option[String]
30   ) extends BaseIndexWriter[WritableFeature](indices, wrapper) {
31 
32   import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
33 
34   private val multiWriter = ds.connector.createMultiTableBatchWriter()
35   private val writers = indices.toArray.map { index =>
36     // should always be writing to a single table here
37     multiWriter.getBatchWriter(index.getTableName(partition))
38   }
39 
40   private val colFamilyMappings = indices.map(ColumnFamilyMapper.apply).toArray
41   private val timestamps = indices.exists(i => !i.sft.isLogicalTime)
42   private val visCache = new VisibilityCache()
43 
44   override protected def append(feature: WritableFeature, values: Array[RowKeyValue[_]]): Unit = {
45     var i = 0
46     while (i < values.length) {
47       values(i) match {
48         case kv: SingleRowKeyValue[_] =>
49           val mutation = new Mutation(kv.row)
50           kv.values.foreach { v =>
51             mutation.put(colFamilyMappings(i)(v.cf), v.cq, visCache(v.vis), v.value)
52           }
53           writers(i).addMutation(mutation)
54 
55         case mkv: MultiRowKeyValue[_] =>
56           mkv.rows.foreach { row =>
57             val mutation = new Mutation(row)
58             mkv.values.foreach { v =>
59               mutation.put(colFamilyMappings(i)(v.cf), v.cq, visCache(v.vis), v.value)
60             }
61             writers(i).addMutation(mutation)
62           }
63       }
64       i += 1
65     }
66   }
67 
68   override protected def update(
69       feature: WritableFeature,
70       values: Array[RowKeyValue[_]],
71       previous: WritableFeature,
72       previousValues: Array[RowKeyValue[_]]): Unit = {
73     delete(previous, previousValues)
74     if (timestamps) {
75       // for updates, ensure that our timestamps don't clobber each other
76       multiWriter.flush()
77       Thread.sleep(1)
78     }
79     append(feature, values)
80   }
81 
82   override protected def delete(feature: WritableFeature, values: Array[RowKeyValue[_]]): Unit = {
83     var i = 0
84     while (i < values.length) {
85       values(i) match {
86         case kv: SingleRowKeyValue[_] =>
87           val mutation = new Mutation(kv.row)
88           kv.values.foreach { v =>
89             mutation.putDelete(colFamilyMappings(i)(v.cf), v.cq, visCache(v.vis))
90           }
91           writers(i).addMutation(mutation)
92 
93         case kv: MultiRowKeyValue[_] =>
94           kv.rows.foreach { row =>
95             val mutation = new Mutation(row)
96             kv.values.foreach { v =>
97               mutation.putDelete(colFamilyMappings(i)(v.cf), v.cq, visCache(v.vis))
98             }
99             writers(i).addMutation(mutation)
100           }
101       }
102       i += 1
103     }
104   }
105 
106   override def flush(): Unit = multiWriter.flush()
107 
108   override def close(): Unit = multiWriter.close()
109 }
