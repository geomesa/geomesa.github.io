39   /**
40    * Gets a feature type compatible with the stored index value
41    *
42    * @param sft simple feature type
43    * @return
44    */
45   def getIndexSft(sft: SimpleFeatureType): SimpleFeatureType = {
46     sft match {
47       case immutable: ImmutableSimpleFeatureType =>
48         var indexSft = cache.get(immutable)
49         if (indexSft == null) {
50           indexSft = SimpleFeatureTypes.immutable(buildIndexSft(sft)).asInstanceOf[ImmutableSimpleFeatureType]
51           cache.put(immutable, indexSft)
52         }
53         indexSft
54 
55       case _ => buildIndexSft(sft)
56     }
57   }

59   private def buildIndexSft(sft: SimpleFeatureType): SimpleFeatureType = {
60     val builder = new SimpleFeatureTypeBuilder()
61     builder.setNamespaceURI(null: String)
62     builder.setName(sft.getTypeName + "--index")
63     builder.setAttributes(getIndexValueAttributes(sft).asJava)
64     if (sft.getGeometryDescriptor != null) {
65       builder.setDefaultGeometry(sft.getGeometryDescriptor.getLocalName)
66     }
67     builder.setCRS(sft.getCoordinateReferenceSystem)
68     val indexSft = builder.buildFeatureType()
69     indexSft.getUserData.putAll(sft.getUserData)
70     indexSft
71   }

73   /**
74    * Gets the attributes that are stored in the index value
75    *
76    * @param sft simple feature type
77    * @return
78    */
79   private def getIndexValueAttributes(sft: SimpleFeatureType): Seq[AttributeDescriptor] = {
80     val geom = sft.getGeometryDescriptor
81     val dtg = sft.getDtgField
82     val attributes = scala.collection.mutable.Buffer.empty[AttributeDescriptor]
83     var i = 0
84     while (i < sft.getAttributeCount) {
85       val ad = sft.getDescriptor(i)
86       if (ad == geom || dtg.contains(ad.getLocalName) || ad.isIndexValue()) {
87         attributes.append(ad)
88       }
89       i += 1
90     }
91     attributes.toSeq
92   }

122   }
