88   class AttributeMapper extends Mapper[Text, SimpleFeature, Text, Mutation] {
89 
90     type Context = Mapper[Text, SimpleFeature, Text, Mutation]#Context
91 
92     private var counter: Counter = _
93 
94     // TODO GEOMESA-2545 have a standardized writer that returns mutations instead of using a batch writer
95 
96     private var wrapper: FeatureWrapper[WritableFeature] = _
97     private var converters: Seq[(WriteConverter[_], Int)] = _
98     private var colFamilyMappings: IndexedSeq[Array[Byte] => Array[Byte]] = _
99     private var defaultVis: ColumnVisibility = _
100     private var tables: SimpleFeature => IndexedSeq[Text] = _
101 
102     override protected def setup(context: Context): Unit = {
103       counter = context.getCounter("org.locationtech.geomesa", "attributes-written")
104       WithStore[AccumuloDataStore](GeoMesaConfigurator.getDataStoreOutParams(context.getConfiguration)) { ds =>
105         val sft = SimpleFeatureTypes.mutable(ds.getSchema(getTypeName(context.getConfiguration)))
106         val attributes = getAttributes(context.getConfiguration).toSet
107         val indices = ds.manager.indices(sft, IndexMode.Write).filter { i =>
108           (i.name == AttributeIndex.name || i.name == JoinIndex.name) &&
109               i.attributes.headOption.exists(attributes.contains)
110         }
111         wrapper = WritableFeature.wrapper(sft, ds.adapter.groups)
112         converters = indices.map(_.createConverter()).zipWithIndex
113         colFamilyMappings = indices.map(ColumnFamilyMapper.apply).toIndexedSeq
114         defaultVis = new ColumnVisibility()
115         tables = TablePartition(ds, sft) match {
116           case Some(tp) =>
117             val tables = scala.collection.mutable.Map.empty[String, IndexedSeq[Text]]
118             f => {
119               // create the new partition table if needed, which also writes the metadata for it
120               val partition = tp.partition(f)
121               tables.getOrElseUpdate(partition,
122                 indices.map(i => new Text(i.configureTableName(Some(partition)))).toIndexedSeq)
123             }
124 
125           case None =>
126             val names = indices.map(i => new Text(i.getTableName())).toIndexedSeq
127             _ => names
128         }
129       }
130     }
131 
132     override protected def cleanup(context: Context): Unit = {}
133 
134     override def map(key: Text, value: SimpleFeature, context: Context): Unit = {
135       val writable = wrapper.wrap(value)
136       val out = tables(value)
137       converters.foreach { case (converter, i) =>
138         converter.convert(writable) match {
139           case SingleRowKeyValue(row, _, _, _, _, _, vals) =>
140             val mutation = new Mutation(row)
141             vals.foreach { v =>
142               val vis = if (v.vis.isEmpty) { defaultVis } else { new ColumnVisibility(v.vis) }
143               mutation.put(colFamilyMappings(i)(v.cf), v.cq, vis, v.value)
144             }
145             context.write(out(i), mutation)
146 
147           case MultiRowKeyValue(rows, _, _, _, _, _, vals) =>
148             rows.foreach { row =>
149               val mutation = new Mutation(row)
150               vals.foreach { v =>
151                 val vis = if (v.vis.isEmpty) { defaultVis } else { new ColumnVisibility(v.vis) }
152                 mutation.put(colFamilyMappings(i)(v.cf), v.cq, vis, v.value)
153               }
154               context.write(out(i), mutation)
155             }
156         }
157       }
158       counter.increment(converters.length)
159     }
160   }

175     val attributes = parsedArgs.attributes.asScala
176 
177     val coverage = Option(parsedArgs.coverage).map { c =>
178       try { IndexCoverage.withName(c) } catch {
179         case NonFatal(_) => throw new IllegalArgumentException(s"Invalid coverage value $c")
180       }
181     }.getOrElse(IndexCoverage.FULL)
182 
183     // validation and initialization - ensure the types exist before launching distributed job
184     WithStore[AccumuloDataStore](dsInParams) { ds =>
185       require(ds != null, "The specified input data store could not be created - check your job parameters")

187       require(sft != null, s"The schema '$typeName' does not exist in the input data store")
188 
189       // validate attributes and set the index coverage

192         require(descriptor != null, s"Attribute '$a' does not exist in schema '$typeName'")
193         descriptor.getUserData.put(AttributeOptions.OptIndex, coverage.toString)
194       }
195 
196       // update the schema - this will create the table if it doesn't exist, and add splits
197       ds.updateSchema(sft.getTypeName, sft)
198 
199       // re-load the sft now that we've updated it
200       sft = ds.getSchema(sft.getTypeName)
201 
202       val job = Job.getInstance(conf,
203         s"GeoMesa Attribute Index Job '${sft.getTypeName}' - '${attributes.mkString(", ")}'")
204 
205       AccumuloJobUtils.setLibJars(job.getConfiguration)
206 
207       job.setJarByClass(AttributeIndexJob.getClass)

210       job.setOutputFormatClass(classOf[AccumuloOutputFormat])
211       job.setMapOutputKeyClass(classOf[Text])
212       job.setMapOutputValueClass(classOf[Mutation])
213       job.setNumReduceTasks(0)
214 
215       val plan = AccumuloJobUtils.getSingleQueryPlan(ds, new Query(sft.getTypeName, Filter.INCLUDE))
216       GeoMesaAccumuloInputFormat.configure(job.getConfiguration, dsInParams.asJava, plan)
217       GeoMesaConfigurator.setDataStoreOutParams(job.getConfiguration, dsInParams)
218       AttributeIndexJob.setAttributes(job.getConfiguration, attributes.toSeq)
219       AttributeIndexJob.setTypeName(job.getConfiguration, sft.getTypeName)
220 
221       val props = AccumuloDataStoreFactory.buildAccumuloClientConfig(
222         java.util.Map.of(
223           AccumuloDataStoreParams.InstanceNameParam.key, parsedArgs.inInstanceId,
224           AccumuloDataStoreParams.ZookeepersParam.key, parsedArgs.inZookeepers,
225           AccumuloDataStoreParams.UserParam.key, parsedArgs.inUser,
226           AccumuloDataStoreParams.PasswordParam.key, parsedArgs.inPassword
227         )
228       )
229 
230       AccumuloOutputFormat.configure().clientProperties(props).createTables(true).store(job)
231 
232       val result = job.waitForCompletion(true)
233 
234       if (result) { 0 } else { 1 }
235     }
236   }
