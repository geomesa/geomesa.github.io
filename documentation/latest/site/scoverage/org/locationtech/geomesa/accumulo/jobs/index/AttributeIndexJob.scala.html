85   class AttributeMapper extends Mapper[Text, SimpleFeature, Text, Mutation] {
86 
87     type Context = Mapper[Text, SimpleFeature, Text, Mutation]#Context
88 
89     private var counter: Counter = _
90 
91     // TODO GEOMESA-2545 have a standardized writer that returns mutations instead of using a batch writer
92 
93     private var wrapper: FeatureWrapper[WritableFeature] = _
94     private var converters: Seq[(WriteConverter[_], Int)] = _
95     private var colFamilyMappings: IndexedSeq[Array[Byte] => Array[Byte]] = _
96     private var defaultVis: ColumnVisibility = _
97     private var tables: SimpleFeature => IndexedSeq[Text] = _
98 
99     override protected def setup(context: Context): Unit = {
100       counter = context.getCounter("org.locationtech.geomesa", "attributes-written")
101       WithStore[AccumuloDataStore](GeoMesaConfigurator.getDataStoreOutParams(context.getConfiguration)) { ds =>
102         val sft = SimpleFeatureTypes.mutable(ds.getSchema(getTypeName(context.getConfiguration)))
103         val attributes = getAttributes(context.getConfiguration).toSet
104         val indices = ds.manager.indices(sft, IndexMode.Write).filter { i =>
105           (i.name == AttributeIndex.name || i.name == JoinIndex.name) &&
106               i.attributes.headOption.exists(attributes.contains)
107         }
108         wrapper = WritableFeature.wrapper(sft, ds.adapter.groups)
109         converters = indices.map(_.createConverter()).zipWithIndex
110         colFamilyMappings = indices.map(ColumnFamilyMapper.apply).toIndexedSeq
111         defaultVis = new ColumnVisibility()
112         tables = TablePartition(ds, sft) match {
113           case Some(tp) =>
114             val tables = scala.collection.mutable.Map.empty[String, IndexedSeq[Text]]
115             f => {
116               // create the new partition table if needed, which also writes the metadata for it
117               val partition = tp.partition(f)
118               tables.getOrElseUpdate(partition,
119                 indices.map(i => new Text(i.configureTableName(Some(partition)))).toIndexedSeq)
120             }
121 
122           case None =>
123             val names = indices.map(i => new Text(i.getTableName())).toIndexedSeq
124             _ => names
125         }
126       }
127     }
128 
129     override protected def cleanup(context: Context): Unit = {}
130 
131     override def map(key: Text, value: SimpleFeature, context: Context): Unit = {
132       val writable = wrapper.wrap(value)
133       val out = tables(value)
134       converters.foreach { case (converter, i) =>
135         converter.convert(writable) match {
136           case SingleRowKeyValue(row, _, _, _, _, _, vals) =>
137             val mutation = new Mutation(row)
138             vals.foreach { v =>
139               val vis = if (v.vis.isEmpty) { defaultVis } else { new ColumnVisibility(v.vis) }
140               mutation.put(colFamilyMappings(i)(v.cf), v.cq, vis, v.value)
141             }
142             context.write(out(i), mutation)
143 
144           case MultiRowKeyValue(rows, _, _, _, _, _, vals) =>
145             rows.foreach { row =>
146               val mutation = new Mutation(row)
147               vals.foreach { v =>
148                 val vis = if (v.vis.isEmpty) { defaultVis } else { new ColumnVisibility(v.vis) }
149                 mutation.put(colFamilyMappings(i)(v.cf), v.cq, vis, v.value)
150               }
151               context.write(out(i), mutation)
152             }
153         }
154       }
155       counter.increment(converters.length)
156     }
157   }

175       try { IndexCoverage.withName(c) } catch {
176         case NonFatal(_) => throw new IllegalArgumentException(s"Invalid coverage value $c")
177       }
178     }.getOrElse(IndexCoverage.FULL)

193       // update the schema - this will create the table if it doesn't exist, and add splits
194       ds.updateSchema(sft.getTypeName, sft)

213       GeoMesaAccumuloInputFormat.configure(job.getConfiguration, dsInParams.asJava, plan)
214       GeoMesaConfigurator.setDataStoreOutParams(job.getConfiguration, dsInParams)

216       AttributeIndexJob.setTypeName(job.getConfiguration, sft.getTypeName)
