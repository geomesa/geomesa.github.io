14 object TableUtils {
15 
16   /**
17    * Creates the table if it doesn't exist
18    *
19    * @param connector connector
20    * @param table table name
21    * @param logical use logical time?
22    * @return true if table was created, false if it already existed
23    */
24   def createTableIfNeeded(connector: AccumuloClient, table: String, logical: Boolean = true): Boolean = {
25     val tableOps = connector.tableOperations()
26     if (tableOps.exists(table)) { false } else {
27       val dot = table.indexOf('.')
28       if (dot > 0) {
29         createNamespaceIfNeeded(connector, table.substring(0, dot))
30       }
31       val config = new NewTableConfiguration().setTimeType(if (logical) { TimeType.LOGICAL } else { TimeType.MILLIS })
32       try { tableOps.create(table, config); true } catch {
33         // this can happen with multiple threads but shouldn't cause any issues
34         case _: TableExistsException => false
35       }
36     }
37   }
38 
39   /**
40    * Creates the namespace if it doesn't exist
41    *
42    * @param connector connector
43    * @param namespace namespace
44    * @return true if namespace was created, false if it already existed
45    */
46   def createNamespaceIfNeeded(connector: AccumuloClient, namespace: String): Boolean = {
47     val nsOps = connector.namespaceOperations
48     if (nsOps.exists(namespace)) { false } else {
49       try { nsOps.create(namespace); true } catch {
50         // this can happen with multiple threads but shouldn't cause any issue
51         case _: NamespaceExistsException => false
52       }
53     }
54   }
