33 /**
34  * Audit writer that persists log entries to a table in Accumulo
35  *
36  * @param client accumulo client
37  * @param table table to write to
38  * @param auditProvider audit provider
39  * @param enabled enable table writes (entries will always be written to logs)
40  */
41 class AccumuloAuditWriter(
42     client: AccumuloClient,
43     val table: String,
44     auditProvider: AuditProvider,
45     enabled: Boolean
46   ) extends AuditWriter(StoreType, auditProvider) with Runnable with Closeable with StrictLogging {
47

52   private val queue = new java.util.concurrent.ConcurrentLinkedQueue[(Promise[Unit], QueryEvent, Seq[AccumuloQueryPlan])]
53 
54   private val running = new AtomicBoolean(true)
55   private val writeLock = new ReentrantLock()
56 
57   private val timeout = ShutdownTimeout.toDuration.get.toMillis
58 
59   private val scheduledRun = {
60     val millis = AccumuloAuditWriter.WriteInterval.toDuration.get.toMillis
61     logger.debug(s"Scheduling audit writer for ${millis}ms")
62     AccumuloAuditWriter.executor.scheduleWithFixedDelay(this, millis, millis, TimeUnit.MILLISECONDS)
63   }
64 
65   override def writeQueryEvent(
66       typeName: String,
67       user: String,
68       filter: Filter,
69       hints: Hints,
70       plans: Seq[QueryPlan[_]],
71       startTime: Long,
72       endTime: Long,
73       planTime: Long,
74       scanTime: Long,
75       hits: Long): Future[Unit] = {
76     if (running.get()) {
77       val promise = Promise[Unit]()
78       // note: we populate the metadata in the asynchronous thread
79       val eventNoMeta = QueryEvent(storeType, typeName, user, filter, hints, startTime, endTime, planTime, scanTime, hits)
80       queue.offer((promise, eventNoMeta, plans.asInstanceOf[Seq[AccumuloQueryPlan]])) // unbounded queue so will never fail
81       promise.future
82     } else {
83       Future.failed(new RuntimeException("This AuditWriter instance has been closed"))
84     }
85   }
86 
87   override def run(): Unit = {
88     if (running.get() && writeLock.tryLock()) {
89       try { writeQueuedEvents() } finally {
90         writeLock.unlock()
91       }
92     }
93   }
94 
95   private def writeQueuedEvents(): Unit = {
96     try {
97       var promiseAndEvent = queue.poll()
98       if (promiseAndEvent != null) {
99         val stopTime = System.currentTimeMillis() + timeout
100         if (enabled && writer == null) {
101           new TableManager(client).ensureTableExists(table)
102           val writeConfig = GeoMesaBatchWriterConfig().setMaxMemory(10000L).setMaxWriteThreads(5)
103           writer = client.createBatchWriter(table, writeConfig)
104         }
105         while (promiseAndEvent != null) {
106           val (promise, eventNoMeta, plans) = promiseAndEvent
107           val write =
108             Try {
109               val event = eventNoMeta.copy(metadata = createMetadata(plans))
110               AuditWriter.log(event)
111               if (enabled) {
112                 writer.addMutation(AccumuloQueryEventTransform.toMutation(event))
113               }
114             }
115           promise.complete(write)
116           promiseAndEvent = if (running.get() || System.currentTimeMillis() < stopTime) { queue.poll() } else { null }
117         }
118         if (enabled) {
119           writer.flush()
120         }
121       }
122     } catch {
123       case NonFatal(e) => logger.error("Error writing audit logs:", e)
124     }
125   }
126 
127   private def createMetadata(plans: Seq[AccumuloQueryPlan]): java.util.Map[String, AnyRef] = {
128     // note: use tree maps for consistent ordering in json output
129     val planMeta = plans.map { plan =>
130       val planData = new java.util.TreeMap[String, AnyRef]()
131       planData.put("index", plan.filter.index.identifier)
132       val tables = new java.util.TreeMap[String, String]()
133       planData.put("tables", tables)
134       val locations = new java.util.TreeMap[String, java.util.Set[String]]()
135       planData.put("locations", locations)
136       plan.tables.foreach { table =>
137         tables.put(table, client.tableOperations().tableIdMap().get(table))
138         val locate = client.tableOperations().locate(table, plan.ranges.asJava)
139         locate.groupByTablet().asScala.keys.foreach { tablet =>
140           val location = locate.getTabletLocation(tablet)
141           var tablets = locations.get(location)
142           if (tablets == null) {
143             tablets = new java.util.TreeSet[String]()
144             locations.put(location, tablets)
145           }
146           tablets.add(tablet.toString)
147         }
148       }
149       planData.put("iterators", plan.iterators.map(_.toString).sorted.asJava)
150       plan.join.foreach { case (_, join) => planData.put("join", join.filter.index.identifier) }
151       plan.sort.foreach { sort =>
152         planData.put("sort", sort.map { case (field, descending) => s"$field:${if (descending) { "desc" } else { "asc" }}"}.asJava)
153       }
154       plan.maxFeatures.foreach(m => planData.put("limit", Int.box(m)))
155       planData
156     }
157     java.util.Map.of("queryPlans", planMeta.asJava)
158   }
159 
160   override def close(): Unit = {
161     if (running.compareAndSet(true, false)) {
162       scheduledRun.cancel(false)
163       try {
164         if (writeLock.tryLock()) {
165           // not currently running, so clean up any remaining events
166           writeQueuedEvents()
167         } else {
168           // currently running, wait for run to end
169           writeLock.lock()
170         }
171         if (writer != null) {
172           writer.close()
173         }
174       } finally {
175         writeLock.unlock()
176       }
177     }
178   }
179 }
