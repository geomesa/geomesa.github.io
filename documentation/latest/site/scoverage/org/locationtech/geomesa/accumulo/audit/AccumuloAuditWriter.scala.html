29 /**
30  * Audit writer that persists log entries to a table in Accumulo
31  *
32  * @param client accumulo client
33  * @param table table to write to
34  * @param auditProvider audit provider
35  * @param enabled enable table writes (entries will always be written to logs)
36  */
37 class AccumuloAuditWriter(
38     client: AccumuloClient,
39     val table: String,
40     auditProvider: AuditProvider,
41     enabled: Boolean
42   ) extends AuditLogger(StoreType, auditProvider) with Runnable with Closeable {
43 
44   private var writer: BatchWriter = _
45 
46   private val queue = new java.util.concurrent.ConcurrentLinkedQueue[(Promise[Unit], QueryEvent)]
47 
48   private val running = new AtomicBoolean(enabled)
49   private val writeLock = new ReentrantLock()
50 
51   private val timeout = ShutdownTimeout.toDuration.get.toMillis
52 
53   private val scheduledRun =
54     AccumuloAuditWriter.WriteInterval.toDuration.collect {
55       case d if enabled && d.isFinite =>
56         val millis = d.toMillis
57         logger.debug(s"Scheduling audit writer for ${millis}ms")
58         AccumuloAuditWriter.executor.scheduleWithFixedDelay(this, millis, millis, TimeUnit.MILLISECONDS)
59     }
60 
61   override protected def write(event: AuditedEvent.QueryEvent): Future[Unit] = {
62     val log = super.write(event)
63     if (running.get()) {
64       val promise = Promise[Unit]()
65       queue.offer(promise -> event) // unbounded queue so will never fail
66       promise.future
67     } else {
68       log
69     }
70   }
71 
72   override def run(): Unit = {
73     if (running.get() && writeLock.tryLock()) {
74       try { writeQueuedEvents() } finally {
75         writeLock.unlock()
76       }
77     }
78   }
79 
80   private def writeQueuedEvents(): Unit = {
81     try {
82       var promiseAndEvent = queue.poll()
83       if (promiseAndEvent != null) {
84         val stopTime = System.currentTimeMillis() + timeout
85         if (writer == null) {
86           new TableManager(client).ensureTableExists(table)
87           val writeConfig = GeoMesaBatchWriterConfig().setMaxMemory(10000L).setMaxWriteThreads(5)
88           writer = client.createBatchWriter(table, writeConfig)
89         }
90         while (promiseAndEvent != null) {
91           val (promise, event) = promiseAndEvent
92           promise.complete(Try(writer.addMutation(AccumuloQueryEventTransform.toMutation(event))))
93           promiseAndEvent = if (running.get() || System.currentTimeMillis() < stopTime) { queue.poll() } else { null }
94         }
95         writer.flush()
96       }
97     } catch {
98       case NonFatal(e) => logger.error("Error writing audit logs:", e)
99     }
100   }
101 
102   override def close(): Unit = {
103     if (running.compareAndSet(true, false)) {
104       scheduledRun.foreach(_.cancel(false))
105       try {
106         if (writeLock.tryLock()) {
107           // not currently running, so clean up any remaining events
108           writeQueuedEvents()
109         } else {
110           // currently running, wait for run to end
111           writeLock.lock()
112         }
113         if (writer != null) {
114           writer.close()
115         }
116       } finally {
117         writeLock.unlock()
118       }
119     }
120   }
121 }
