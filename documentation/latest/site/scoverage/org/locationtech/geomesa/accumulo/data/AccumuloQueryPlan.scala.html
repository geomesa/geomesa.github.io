68   // converts a range to a printable string - only includes the row
69   private def rangeToString(r: org.apache.accumulo.core.data.Range): String = {
70     val a = if (r.isStartKeyInclusive) "[" else "("
71     val z = if (r.isEndKeyInclusive) "]" else ")"
72     val start = if (r.isInfiniteStartKey) "-inf" else keyToString(r.getStartKey)
73     val stop = if (r.isInfiniteStopKey) "+inf" else keyToString(r.getEndKey)
74     s"$a$start::$stop$z"
75   }

97       filter: FilterStrategy,
98       tables: Seq[String],
99       ranges: Seq[org.apache.accumulo.core.data.Range],

101       columnFamily: Option[Text],

105       maxFeatures: Option[Int],
106       projection: Option[QueryReferenceSystems],
107       numThreads: Int

110     override def scan(ds: AccumuloDataStore): CloseableIterator[Entry[Key, Value]] = {
111       // convert the relative timeout to an absolute timeout up front
112       val timeout = ds.config.queries.timeout.map(Timeout.apply)
113       // note: calculate authorizations up front so that multi-threading doesn't mess up auth providers
114       scan(ds.connector, ds.auths, ds.config.queries.parallelPartitionScans, timeout)
115     }

117     /**
118      * Scan with pre-computed auths
119      *
120      * @param connector connector
121      * @param auths auths
122      * @param partitionParallelScans parallelize scans across multiple partitions (vs execute them serially)
123      * @param timeout absolute stop time, as sys time
124      * @return
125      */
126     def scan(
127         connector: AccumuloClient,
128         auths: Authorizations,
129         partitionParallelScans: Boolean,
130         timeout: Option[Timeout]): CloseableIterator[Entry[Key, Value]] = {
131       if (partitionParallelScans) {
132         // kick off all the scans at once
133         tables.map(scanner(connector, _, auths, timeout)).foldLeft(CloseableIterator.empty[Entry[Key, Value]])(_ concat _)
134       } else {
135         // kick off the scans sequentially as they finish
136         SelfClosingIterator(tables.iterator).flatMap(scanner(connector, _, auths, timeout))
137       }
138     }
139 
140     /**
141      *
142      * @param connector connector
143      * @param table table
144      * @param auths auths
145      * @param timeout absolute stop time, as sys time
146      * @return
147      */
148     private def scanner(
149         connector: AccumuloClient,
150         table: String,
151         auths: Authorizations,
152         timeout: Option[Timeout]): CloseableIterator[Entry[Key, Value]] = {
153       val scanner = connector.createBatchScanner(table, auths, numThreads)
154       scanner.setRanges(ranges.asJava)
155       configure(scanner)
156       timeout match {
157         case None => new ScanIterator(scanner)
158         case Some(t) => new ManagedScan(new AccumuloScanner(scanner), t, this)
159       }
160     }
161   }

163   // join on multiple tables - requires multiple scans
164   case class JoinPlan(
165       filter: FilterStrategy,
166       tables: Seq[String],
167       ranges: Seq[org.apache.accumulo.core.data.Range],
168       iterators: Seq[IteratorSetting],
169       columnFamily: Option[Text],
170       numThreads: Int,
171       joinFunction: JoinFunction,
172       joinQuery: BatchScanPlan
173     ) extends AccumuloQueryPlan {
174 
175     override val join: Some[(JoinFunction, BatchScanPlan)] = Some((joinFunction, joinQuery))
176     override def resultsToFeatures: ResultsToFeatures[Entry[Key, Value]] = joinQuery.resultsToFeatures
177     override def reducer: Option[FeatureReducer] = joinQuery.reducer
178     override def sort: Option[Seq[(String, Boolean)]] = joinQuery.sort
179     override def maxFeatures: Option[Int] = joinQuery.maxFeatures
180     override def projection: Option[QueryReferenceSystems] = joinQuery.projection
181 
182     override def scan(ds: AccumuloDataStore): CloseableIterator[Entry[Key, Value]] = {
183       // convert the relative timeout to an absolute timeout up front
184       val timeout = ds.config.queries.timeout.map(Timeout.apply)
185       // calculate authorizations up front so that multi-threading doesn't mess up auth providers
186       val auths = ds.auths
187       val joinTables = joinQuery.tables.iterator
188       if (ds.config.queries.parallelPartitionScans) {
189         // kick off all the scans at once
190         tables.map(scanner(ds.connector, _, joinTables.next, auths, partitionParallelScans = true, timeout))
191             .foldLeft(CloseableIterator.empty[Entry[Key, Value]])(_ concat _)
192       } else {
193         // kick off the scans sequentially as they finish
194         SelfClosingIterator(tables.iterator)
195             .flatMap(scanner(ds.connector, _, joinTables.next, auths, partitionParallelScans = false, timeout))
196       }
197     }
198 
199     private def scanner(
200         connector: AccumuloClient,
201         table: String,
202         joinTable: String,
203         auths: Authorizations,
204         partitionParallelScans: Boolean,
205         timeout: Option[Timeout]): CloseableIterator[Entry[Key, Value]] = {
206       val primary = if (ranges.lengthCompare(1) == 0) {
207         val scanner = connector.createScanner(table, auths)
208         scanner.setRange(ranges.head)
209         scanner
210       } else {
211         val scanner = connector.createBatchScanner(table, auths, numThreads)
212         scanner.setRanges(ranges.asJava)
213         scanner
214       }
215       configure(primary)
216       val join = joinQuery.copy(tables = Seq(joinTable))
217       new BatchMultiScanner(connector, primary, join, joinFunction, auths, partitionParallelScans, timeout)
218     }
219   }

224     override def next(): Entry[Key, Value] = iter.next()
225     override def close(): Unit = scanner.close()
226   }
