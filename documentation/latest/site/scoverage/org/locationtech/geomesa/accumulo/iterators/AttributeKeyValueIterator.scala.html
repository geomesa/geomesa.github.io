23 /**
24   * Iterator that operates on kryo encoded attribute index rows. It will modify the value
25   * (kryo encoded simple feature) by setting the attribute from the row key.
26   *
27   * It is expected to run *after* the normal iterator stack - thus the key value received in getTopValue
28   * will be the already transformed feature result, with a null value for the attribute in question. This
29   * iterator will set the attribute value based on the row, then re-serialize the feature
30   */
31 class AttributeKeyValueIterator extends SortedKeyValueIterator[Key, Value] with LazyLogging {
32 
33   import AttributeKeyValueIterator._
34 
35   var source: SortedKeyValueIterator[Key, Value] = _
36   val topValue: Value = new Value
37 
38   var index: AttributeIndex = _
39   var serializer: KryoFeatureSerializer = _
40   var attribute: Int = -1
41 
42   override def init(src: SortedKeyValueIterator[Key, Value],
43                     options: jMap[String, String],
44                     env: IteratorEnvironment): Unit = {
45     this.source = src
46 
47     val spec = options.get(SFT_OPT)
48     attribute = options.get(ATTRIBUTE_OPT).toInt
49     index = IteratorCache.index(IteratorCache.sft(spec), spec, options.get(INDEX_OPT)).asInstanceOf[AttributeIndex]
50     serializer = IteratorCache.serializer(options.get(TRANSFORM_OPT), index.serializedWithId)
51   }
52 
53   override def seek(range: Range, columnFamilies: jCollection[ByteSequence], inclusive: Boolean): Unit =
54     source.seek(range, columnFamilies, inclusive)
55 
56   override def next(): Unit = source.next()
57   override def hasTop: Boolean = source.hasTop
58   override def getTopKey: Key = source.getTopKey
59 
60   override def getTopValue: Value = {
61     val serializedSf = source.getTopValue.get()
62     val row = source.getTopKey.getRow
63     val value = index.keySpace.decodeRowValue(row.getBytes, 0, row.getLength).map { value =>
64       val sf = serializer.deserialize(serializedSf)
65       sf.setAttribute(attribute, value)
66       serializer.serialize(sf)
67     }
68     topValue.set(value.getOrElse(serializedSf))
69     topValue
70   }
71 
72   override def deepCopy(env: IteratorEnvironment): SortedKeyValueIterator[Key, Value] =
73     throw new UnsupportedOperationException()
74 }

76 object AttributeKeyValueIterator {
77 
78   val SFT_OPT       = "sft"
79   val TRANSFORM_OPT = "tsft"
80   val INDEX_OPT     = "index"
81   val ATTRIBUTE_OPT = "attr"
82 
83   val DefaultPriority = 27 // needs to be higher than FilterTransformIterator at 25
84 
85   def configure(index: AttributeIndex,
86                 transform: SimpleFeatureType,
87                 priority: Int = DefaultPriority): IteratorSetting = {
88     val is = new IteratorSetting(priority, "attr-value-iter", classOf[AttributeKeyValueIterator])
89     is.addOption(INDEX_OPT, index.identifier)
90     is.addOption(SFT_OPT, SimpleFeatureTypes.encodeType(index.sft, includeUserData = true))
91     is.addOption(TRANSFORM_OPT, SimpleFeatureTypes.encodeType(transform, includeUserData = true))
92     is.addOption(ATTRIBUTE_OPT, transform.indexOf(index.attributes.head).toString)
93     is
94   }
