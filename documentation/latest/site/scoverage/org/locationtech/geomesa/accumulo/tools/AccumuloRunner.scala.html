56 trait RunnerWithAccumuloEnvironment extends Runner {
57 
58   /**
59     * Loads Accumulo properties from `accumulo-client.properties` or `accumulo.conf`
60     */
61   override def resolveEnvironment(command: Command): Unit = {
62     lazy val config = AccumuloClientConfig.load()
63 
64     Option(command.params).collect { case p: AccumuloConnectionParams => p }.foreach { p =>
65       if (p.user == null) {
66         config.getProperty(ClientProperty.AUTH_PRINCIPAL.getKey) match {
67           case null => throw new ParameterException("Parameter '--user' is required")
68           case u => p.user = u
69         }
70       }
71       if (p.password != null && p.keytab != null) {
72         // error if both password and keytab supplied
73         throw new ParameterException("Cannot specify both password and keytab")
74       } else if (p.password == null && p.keytab == null &&
75           AccumuloClientConfig.PasswordAuthType.equalsIgnoreCase(ClientProperty.AUTH_TYPE.getValue(config))) {
76         // if password not supplied, and not using keytab, prompt for it
77         p.password = Option(config.getProperty(ClientProperty.AUTH_TOKEN.getKey)).getOrElse(Prompt.readPassword())
78       }
79     }
80 
81     Option(command.params).collect { case p: InstanceNameParams => p }.foreach { p =>
82       if (p.instance == null) {
83         Option(config.getProperty(ClientProperty.INSTANCE_NAME.getKey)).foreach(p.instance = _)
84       }
85       if (p.zookeepers == null) {
86         Option(config.getProperty(ClientProperty.INSTANCE_ZOOKEEPERS.getKey)).foreach(p.zookeepers = _)
87         if (p.zkTimeout == null) {
88           Option(config.getProperty(ClientProperty.INSTANCE_ZOOKEEPERS_TIMEOUT.getKey)).foreach(p.zkTimeout = _)
89         }
90       }
91     }
92   }
93 
94   override protected def classpathEnvironments: Seq[String] = Seq("ACCUMULO_HOME", "HADOOP_HOME")
