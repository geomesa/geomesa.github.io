58 trait RunnerWithAccumuloEnvironment extends Runner {
59 
60   /**
61     * Loads Accumulo properties from `accumulo-client.properties` or `accumulo.conf`
62     */
63   override def resolveEnvironment(command: Command): Unit = {
64     lazy val config = AccumuloClientConfig.load()
65 
66     Option(command.params).collect { case p: AccumuloConnectionParams => p }.foreach { p =>
67       if (p.user == null) {
68         config.getProperty(ClientProperty.AUTH_PRINCIPAL.getKey) match {
69           case null => throw new ParameterException("Parameter '--user' is required")
70           case u => p.user = u
71         }
72       }
73       if (p.password != null && p.keytab != null) {
74         // error if both password and keytab supplied
75         throw new ParameterException("Cannot specify both password and keytab")
76       } else if (p.password == null && p.keytab == null &&
77           AccumuloClientConfig.PasswordAuthType.equalsIgnoreCase(ClientProperty.AUTH_TYPE.getValue(config))) {
78         // if password not supplied, and not using keytab, prompt for it
79         p.password = Option(config.getProperty(ClientProperty.AUTH_TOKEN.getKey)).getOrElse(Prompt.readPassword())
80       }
81     }
82 
83     Option(command.params).collect { case p: InstanceNameParams => p }.foreach { p =>
84       if (p.instance == null) {
85         Option(config.getProperty(ClientProperty.INSTANCE_NAME.getKey)).foreach(p.instance = _)
86       }
87       if (p.zookeepers == null) {
88         Option(config.getProperty(ClientProperty.INSTANCE_ZOOKEEPERS.getKey)).foreach(p.zookeepers = _)
89         if (p.zkTimeout == null) {
90           Option(config.getProperty(ClientProperty.INSTANCE_ZOOKEEPERS_TIMEOUT.getKey)).foreach(p.zkTimeout = _)
91         }
92       }
93     }
94   }
95 
96   override protected def classpathEnvironments: Seq[String] = Seq("ACCUMULO_HOME", "HADOOP_HOME")
