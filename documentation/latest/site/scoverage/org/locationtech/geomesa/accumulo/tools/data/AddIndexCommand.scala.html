31 /**
32   *
33   * 1. Add the new index in write-only mode
34   * 2. Pause and indicate that the user should bounce live ingestion to pick up the changes -
35   *    after this it will be writing to both the new and old index
36   * 3. Migrate data through a m/r job, with an optional CQL filter for what gets migrated
37   * 4. Turn old index off, put new index in read/write mode
38   * 5. Pause and indicate that the user should bounce live ingestion again
39   */

43   override val params = new AddIndexParameters
44 
45   override def execute(): Unit = {

48   }

51 object AddIndexCommand {
52 
53   @Parameters(commandDescription = "Add or update indices for an existing GeoMesa feature type")
54   class AddIndexParameters extends AccumuloDataStoreParams with RequiredTypeNameParam with OptionalCqlFilterParam {
55     @Parameter(names = Array("--index"), description = "Name of index(es) to add - comma-separate or use multiple flags", required = true)
56     var indexNames: java.util.List[String] = _
57 
58     @Parameter(names = Array("--no-back-fill"), description = "Do not copy any existing data into the new index")
59     var noBackFill: java.lang.Boolean = _
60   }
61 }

63 class AddIndexCommandExecutor(override val params: AddIndexParameters) extends AccumuloDataStoreCommand {
64 
65   import org.locationtech.geomesa.index.metadata.GeoMesaMetadata.AttributesKey
66   import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
67 
68   override val name = ""
69

72   def addIndex(ds: AccumuloDataStore): Unit  = {
73 
74     import scala.collection.JavaConverters._
75 
76     val sft = Option(ds.getSchema(params.featureName)).map(SimpleFeatureTypes.mutable).orNull
77     require(sft != null, s"Schema '${params.featureName}' does not exist in the specified data store")
78 
79     val indices = {
80       val builder = new SimpleFeatureTypeBuilder()
81       builder.init(sft)
82       val copy = builder.buildFeatureType()
83       copy.getUserData.putAll(sft.getUserData)
84       copy.getUserData.remove(InternalConfigs.IndexVersions)
85       copy.getUserData.put(Configs.EnabledIndices, params.indexNames.asScala.mkString(","))
86       GeoMesaFeatureIndexFactory.indices(copy)
87     }
88 
89     val existing = ds.manager.indices(sft)
90     require(indices.map(GeoMesaFeatureIndex.identifier).forall(i => !existing.map(_.identifier).contains(i)),
91       s"Requested indices already exist: ${existing.map(_.identifier).mkString("[", "][", "]")}")
92 
93     val toDisable = indices.flatMap { i =>
94       existing.find(e => e.name == i.name && e.attributes == i.attributes).map { e =>
95         (i, IndexId(e.name, e.version, e.attributes, e.mode))
96       }
97     }
98 
99     if (toDisable.nonEmpty) {
100       if (!Prompt.confirm("The following index versions will be replaced: " +
101         toDisable.map { case (n, o) => s"[$o] by [${GeoMesaFeatureIndex.identifier(n)}]" }.mkString(", ") +
102         "Continue (y/n)? ")) {
103         return
104       }
105     }
106     if (!Prompt.confirm("If you are ingesting streaming data, you will be required to restart " +
107       "the streaming ingestion when prompted. Continue (y/n)? ")) {
108       return
109     }
110 
111     // write a backup meta-data entry in case the process fails part-way
112     val backupKey = s"$AttributesKey.bak"
113     ds.metadata.insert(sft.getTypeName, backupKey, ds.metadata.readRequired(sft.getTypeName, AttributesKey))
114 
115     val toKeep = sft.getIndices.filter(i => !toDisable.map(_._2).contains(i))
116 
117     if (params.noBackFill != null && params.noBackFill.booleanValue()) {
118       Command.user.info("Adding new indices and disabling old ones")
119       sft.setIndices(indices ++ toKeep)
120       ds.updateSchema(sft.getTypeName, sft)
121     } else {
122       Command.user.info("Adding new indices in write-only mode")
123 
124       // add new index in write-only mode
125       sft.setIndices(indices.map(_.copy(mode = IndexMode.Write)) ++ sft.getIndices)
126       ds.updateSchema(sft.getTypeName, sft)
127 
128       // wait for the user to bounce ingestion
129       Prompt.acknowledge("Indices have been added in write-only mode. To pick up the changes, " +
130         "please bounce any streaming ingestion. Once ingestion has resumed, press 'enter' to continue.")
131 
132       // run migration job
133       Command.user.info("Running index back-fill job")
134 
135       val args = new WriteIndexArgs(Array.empty)
136       args.inZookeepers = params.zookeepers
137       args.inInstanceId = params.instance
138       args.inUser       = params.user
139       args.inPassword   = params.password
140       args.inTableName  = params.catalog
141       args.inFeature    = params.featureName
142       args.inCql        = Option(params.cqlFilter).map(ECQL.toCQL).orNull
143       args.indexNames.addAll(indices.map(GeoMesaFeatureIndex.identifier).asJava)
144 
145       val libjars = Some(AccumuloJobUtils.defaultLibJars, AccumuloJobUtils.defaultSearchPath)
146       val result = try { ToolRunner.run(new WriteIndexJob(libjars), args.unparse()) } catch {
147         case NonFatal(e) => Command.user.error("Error running back-fill job:", e); -1
148       }
149 
150       def setReadWrite(): Unit = {
151         Command.user.info("Setting index to read-write mode and disabling old indices")
152         // set new indices to read-write and turn off disabled indices
153         sft.setIndices(indices.map(_.copy(mode = IndexMode.ReadWrite)) ++ toKeep)
154         Command.user.info(sft.getIndices.toString)
155         ds.updateSchema(sft.getTypeName, sft)
156       }
157 
158       if (result == 0) {
159         setReadWrite()
160       } else {
161         def readResponse(): String = {
162           Prompt.read("Index back-fill job failed. You may:\n" +
163               "  1. Switch the indices to read-write mode without existing data (you may manually back-fill later)\n" +
164               "  2. Roll-back index creation\n" +
165               "Select an option: ")
166         }
167         var response: String = readResponse()
168         while (response != "1" && response != "2") {
169           Command.user.error("Invalid response, please select 1 or 2")
170           response = readResponse()
171         }
172         response match {
173           case "1" => setReadWrite()
174           case "2" =>
175             val bak = ds.metadata.readRequired(sft.getTypeName, backupKey)
176             ds.metadata.insert(sft.getTypeName, AttributesKey, bak)
177         }
178       }
179     }
180 
181     // final bounce
182     Command.user.info("Operation complete. Please bounce any streaming ingestion to pick up the changes.")
183   }
