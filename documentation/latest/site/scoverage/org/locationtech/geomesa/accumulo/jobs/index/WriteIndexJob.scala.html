34 /**
35  * Class to write data to a single index
36  *
37  * Can be used to back-fill data into a new index without re-writing unchanged indices.
38  */
39 object WriteIndexJob {
40 
41   def main(args: Array[String]): Unit = {
42     val result = ToolRunner.run(new WriteIndexJob, args)
43     System.exit(result)
44   }

46   class WriteIndexArgs(args: Array[String]) extends GeoMesaArgs(args)
47       with InputFeatureArgs with InputDataStoreArgs with InputCqlArgs {
48 
49     @Parameter(names = Array("--geomesa.index"), description = "Name of index(es) to add - comma-separate or use multiple flags", required = true)
50     var indexNames: java.util.List[String] = new java.util.ArrayList[String]()
51 
52     override def unparse(): Array[String] = {
53       val names = if (indexNames == null || indexNames.isEmpty) {
54         Array.empty[String]
55       } else {
56         indexNames.asScala.flatMap(n => Seq("--geomesa.index", n)).toArray
57       }
58       Array.concat(super[InputFeatureArgs].unparse(),
59         super[InputDataStoreArgs].unparse(),
60         super[InputCqlArgs].unparse(),
61         names)
62     }
63   }
64 
65   class PassThroughMapper extends Mapper[Text, SimpleFeature, Text, SimpleFeature] {
66 
67     type Context = Mapper[Text, SimpleFeature, Text, SimpleFeature]#Context
68 
69     private val text: Text = new Text
70     private var counter: Counter = _
71 
72     override protected def setup(context: Context): Unit =
73       counter = context.getCounter("org.locationtech.geomesa", "features-written")
74 
75     override protected def cleanup(context: Context): Unit = {}
76 
77     override def map(key: Text, value: SimpleFeature, context: Context) {
78       context.write(text, value)
79       counter.increment(1)
80     }
81   }
82 }

84 class WriteIndexJob(libjars: Option[(Seq[String], Iterator[() => Seq[File]])] = None) extends Tool with LazyLogging {
85 
86   private var conf: Configuration = new Configuration
87 
88   override def run(args: Array[String]): Int = {
89     val parsedArgs = new WriteIndexArgs(args)
90     parsedArgs.parse()
91 
92     val featureIn  = parsedArgs.inFeature
93     val dsInParams = parsedArgs.inDataStore
94     val filter     = Option(parsedArgs.inCql).getOrElse("INCLUDE")
95 
96     // validation and initialization - ensure the types exist before launching distributed job
97     val (sft, indices, plan) = WithStore[AccumuloDataStore](dsInParams) { dsIn =>
98       require(dsIn != null, "The specified input data store could not be created - check your job parameters")
99       val sft = dsIn.getSchema(featureIn)
100       require(sft != null, s"The feature '$featureIn' does not exist in the input data store")
101       val allIndices = dsIn.manager.indices(sft, IndexMode.Write)
102       val indices = parsedArgs.indexNames.asScala.map { name =>
103         allIndices.find(_.identifier == name).orElse(allIndices.find(_.name == name)).getOrElse {
104           throw new IllegalArgumentException(s"Invalid index $name. Valid values are " +
105               allIndices.map(_.identifier).sorted.mkString(", "))
106         }
107       }
108       val plan = AccumuloJobUtils.getSingleQueryPlan(dsIn, new Query(sft.getTypeName, ECQL.toFilter(filter)))
109       (sft, indices.map(_.identifier), plan)
110     }
111 
112     val conf = new Configuration
113     val jobName = s"GeoMesa Index Job [${sft.getTypeName}] ${indices.mkString("[", "][", "]")}"
114     val job = Job.getInstance(conf, jobName)
115 
116     libjars.foreach { case (jars, path) => JobUtils.setLibJars(job.getConfiguration, jars, path) }
117 
118     job.setJarByClass(classOf[WriteIndexJob])
119     job.setMapperClass(classOf[PassThroughMapper])
120     job.setInputFormatClass(classOf[GeoMesaAccumuloInputFormat])
121     job.setOutputFormatClass(classOf[GeoMesaOutputFormat])
122     job.setMapOutputKeyClass(classOf[Text])
123     job.setMapOutputValueClass(classOf[ScalaSimpleFeature])
124     job.setNumReduceTasks(0)
125 
126     GeoMesaAccumuloInputFormat.configure(job.getConfiguration, dsInParams.asJava, plan)
127 
128     // disable writing stats as we're just copying data not creating any
129     val dsOutParams = dsInParams ++ Map(AccumuloDataStoreParams.GenerateStatsParam.getName -> "false")
130     GeoMesaOutputFormat.setOutput(job.getConfiguration, dsOutParams, sft, Some(indices.toSeq))
131 
132     logger.info("Submitting job - please wait...")
133     job.submit()
134     logger.info(s"Tracking available at ${job.getStatus.getTrackingUrl}")
135 
136     val result = job.waitForCompletion(true)
137 
138     if (result) 0 else 1
139   }
140 
141   override def getConf: Configuration = conf
