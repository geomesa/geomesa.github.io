31 /**
32  * Class to copy a schema and all data from one data store to another.
33  *
34  * Can be used to 'update' geomesa data from older versions. It does this by reading data in the old format
35  * and writing it to a new schema which will use the latest format. This way, improvements in serialization,
36  * etc can be leveraged for old data.
37  */
38 object SchemaCopyJob {
39 
40   private val CopySchemaNameKey = "org.locationtech.geomesa.copy.name"
41   private val CopySchemaSpecKey = "org.locationtech.geomesa.copy.spec"
42 
43   def main(args: Array[String]): Unit = {
44     val result = ToolRunner.run(new SchemaCopyJob, args)
45     System.exit(result)
46   }
47 
48   private def setCopySchema(conf: Configuration, sft: SimpleFeatureType): Unit = {
49     conf.set(CopySchemaNameKey, sft.getTypeName)
50     conf.set(CopySchemaSpecKey, SimpleFeatureTypes.encodeType(sft, includeUserData = true))
51   }
52 
53   private def getCopySchema(conf: Configuration): SimpleFeatureType =
54     SimpleFeatureTypes.createType(conf.get(CopySchemaNameKey), conf.get(CopySchemaSpecKey))
55 
56   class SchemaCopyArgs(args: Array[String]) extends GeoMesaArgs(args)
57       with InputFeatureArgs with InputDataStoreArgs with InputCqlArgs
58       with OutputFeatureOptionalArgs with OutputDataStoreArgs {
59 
60     override def unparse(): Array[String] = {

64         super[OutputFeatureOptionalArgs].unparse(),
65         super[OutputDataStoreArgs].unparse()
66       )
67     }
68   }
69 
70   class CopyMapper extends Mapper[Text, SimpleFeature, Text, SimpleFeature] {
71 
72     type Context = Mapper[Text, SimpleFeature, Text, SimpleFeature]#Context
73 
74     private val text: Text = new Text
75     private var counter: Counter = _
76 
77     private var sftOut: SimpleFeatureType = _
78 
79     override protected def setup(context: Context): Unit = {
80       counter = context.getCounter("org.locationtech.geomesa", "features-written")
81       sftOut = getCopySchema(context.getConfiguration)
82     }
83 
84     override protected def cleanup(context: Context): Unit = {}
85 
86     override def map(key: Text, value: SimpleFeature, context: Context) {
87       context.write(text, ScalaSimpleFeature.copy(sftOut, value))
88       counter.increment(1)
89     }
90   }

93 class SchemaCopyJob extends Tool {
94 
95   private var conf: Configuration = new Configuration
96 
97   override def run(args: Array[String]): Int = {
98     val parsedArgs = new SchemaCopyArgs(args)
99     parsedArgs.parse()
100 
101     val featureIn   = parsedArgs.inFeature
102     val featureOut  = Option(parsedArgs.outFeature).getOrElse(featureIn)
103 
104     val dsInParams  = parsedArgs.inDataStore
105     val dsOutParams = parsedArgs.outDataStore
106     val filter      = Option(parsedArgs.inCql).getOrElse("INCLUDE")
107 
108     // validation and initialization - ensure the types exist before launching distributed job
109     val (sftIn, plan) = WithStore[AccumuloDataStore](dsOutParams) { dsIn =>
110       require(dsIn != null, "The specified input data store could not be created - check your job parameters")
111       val sft = dsIn.getSchema(featureIn)
112       require(sft != null, s"The feature '$featureIn' does not exist in the input data store")
113       val plan = AccumuloJobUtils.getSingleQueryPlan(dsIn, new Query(sft.getTypeName, ECQL.toFilter(filter)))
114       (sft, plan)
115     }
116 
117     val sftOut = WithStore[AccumuloDataStore](dsOutParams) { dsOut =>
118       require(dsOut != null, "The specified output data store could not be created - check your job parameters")
119       var sft = dsOut.getSchema(featureOut)
120       if (sft != null) { sft } else {
121         // update the feature name
122         if (featureOut == featureIn) {
123           sft = sftIn
124         } else {
125           sft = SimpleFeatureTypes.createType(featureOut, SimpleFeatureTypes.encodeType(sftIn))
126         }
127         // create the schema in the output datastore
128         dsOut.createSchema(sft)
129         dsOut.getSchema(featureOut)
130       }
131     }
132     require(sftOut != null, "Could not create output type - check your job parameters")
133 
134     val conf = new Configuration()
135     val job = Job.getInstance(conf, s"GeoMesa Schema Copy '${sftIn.getTypeName}' to '${sftOut.getTypeName}'")
136 
137     job.setJarByClass(SchemaCopyJob.getClass)
138     job.setMapperClass(classOf[PassThroughMapper])
139     job.setInputFormatClass(classOf[GeoMesaAccumuloInputFormat])
140     job.setOutputFormatClass(classOf[GeoMesaOutputFormat])
141     job.setMapOutputKeyClass(classOf[Text])
142     job.setMapOutputValueClass(classOf[ScalaSimpleFeature])
143     job.setNumReduceTasks(0)
144 
145     GeoMesaAccumuloInputFormat.configure(job.getConfiguration, dsInParams.asJava, plan)
146     GeoMesaOutputFormat.setOutput(job.getConfiguration, dsOutParams, sftOut)
147     SchemaCopyJob.setCopySchema(job.getConfiguration, sftOut)
148 
149     val result = job.waitForCompletion(true)
150 
151     if (result) 0 else 1
152   }
153 
154   override def getConf: Configuration = conf
155 
156   override def setConf(conf: Configuration): Unit = this.conf = conf
