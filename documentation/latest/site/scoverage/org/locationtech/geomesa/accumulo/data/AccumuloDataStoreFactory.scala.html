137   /**
138    * Build an Accumulo client configuration, based on passed in parameters and any accumulo-client.properties
139    * file on the classpath
140    *
141    * @param params data store params
142    * @return
143    */
144   def buildAccumuloClientConfig(params: java.util.Map[String, _]): Properties = {
145     val config = AccumuloClientConfig.load()
146 
147     def setRequired(param: GeoMesaParam[String], key: ClientProperty): String = {
148       param.lookupOpt(params) match {
149         case Some(v) => config.put(key.getKey, v); v
150         case None =>
151           Option(config.getProperty(key.getKey)).getOrElse {
152             throw new IOException(s"Parameter ${param.key} is required: ${param.description}")
153           }
154       }
155     }
156 
157     def setOptional(param: GeoMesaParam[String], key: ClientProperty): Unit =
158       param.lookupOpt(params).foreach(config.put(key.getKey, _))
159 
160     def getRequired(param: GeoMesaParam[String], key: ClientProperty): String = {
161       param.lookupOpt(params).orElse(Option(config.getProperty(key.getKey))).getOrElse {
162         throw new IOException(s"Parameter ${param.key} is required: ${param.description}")
163       }
164     }
165 
166     setRequired(InstanceNameParam, ClientProperty.INSTANCE_NAME)
167     setRequired(ZookeepersParam, ClientProperty.INSTANCE_ZOOKEEPERS)
168     setOptional(ZookeeperTimeoutParam, ClientProperty.INSTANCE_ZOOKEEPERS_TIMEOUT)
169 
170     val user = setRequired(UserParam, ClientProperty.AUTH_PRINCIPAL)
171 
172     if (PasswordParam.exists(params) && KeytabPathParam.exists(params)) {
173       throw new IllegalArgumentException(
174         s"'${PasswordParam.key}' and '${KeytabPathParam.key}' are mutually exclusive, but are both set")
175     }
176 
177     val authType =
178       if (PasswordParam.exists(params)) {
179         AccumuloClientConfig.PasswordAuthType
180       } else if (KeytabPathParam.exists(params)) {
181         AccumuloClientConfig.KerberosAuthType
182       } else {
183         ClientProperty.AUTH_TYPE.getValue(config).toLowerCase(Locale.US)
184       }
185 
186     val token =
187       if (authType == AccumuloClientConfig.PasswordAuthType) {
188         new PasswordToken(getRequired(PasswordParam, ClientProperty.AUTH_TOKEN).getBytes(StandardCharsets.UTF_8))
189       } else if (authType == AccumuloClientConfig.KerberosAuthType) {
190         val file = new java.io.File(getRequired(KeytabPathParam, ClientProperty.AUTH_TOKEN))
191         // mimic behavior from accumulo 1.9 and earlier:
192         // `public KerberosToken(String principal, File keytab, boolean replaceCurrentUser)`
193         UserGroupInformation.loginUserFromKeytab(user, file.getAbsolutePath)
194         new KerberosToken(user, file)
195       } else {
196          throw new IllegalArgumentException(
197            s"Unsupported auth type: $authType - supported values are " +
198              s"${AccumuloClientConfig.PasswordAuthType}, ${AccumuloClientConfig.KerberosAuthType}")
199       }
200     ClientProperty.setAuthenticationToken(config, token)
201 
202     if (WriteThreadsParam.exists(params)) {
203       config.put(ClientProperty.BATCH_WRITER_THREADS_MAX.getKey, WriteThreadsParam.lookup(params).toString)
204     } else if (ClientProperty.BATCH_WRITER_THREADS_MAX.isEmpty(config)) {
205       BatchWriterProperties.WRITER_THREADS.option.foreach { threads =>
206         config.put(ClientProperty.BATCH_WRITER_THREADS_MAX.getKey, String.valueOf(threads))
207       }
208     }
209     if (ClientProperty.BATCH_WRITER_MEMORY_MAX.isEmpty(config)) {
210       BatchWriterProperties.WRITER_MEMORY_BYTES.toBytes.foreach { memory =>
211         config.put(ClientProperty.BATCH_WRITER_MEMORY_MAX.getKey, String.valueOf(memory))
212       }
213     }
214     if (ClientProperty.BATCH_WRITER_LATENCY_MAX.isEmpty(config)) {
215       BatchWriterProperties.WRITER_LATENCY.toDuration.foreach { duration =>
216         config.put(ClientProperty.BATCH_WRITER_LATENCY_MAX.getKey, s"${duration.toMillis}ms")
217       }
218     }
219     if (ClientProperty.BATCH_WRITER_TIMEOUT_MAX.isEmpty(config)) {
220       BatchWriterProperties.WRITE_TIMEOUT.toDuration.foreach { duration =>
221         config.put(ClientProperty.BATCH_WRITER_TIMEOUT_MAX.getKey, s"${duration.toMillis}ms")
222       }
223     }
224 
225     config
226   }

235   def buildConfig(client: AccumuloClient, params: java.util.Map[String, _]): AccumuloDataStoreConfig = {
236     val catalog = CatalogParam.lookup(params)
237 
238     val authProvider = buildAuthsProvider(client, params)
239     val auditProvider = buildAuditProvider(params)
240     val auditWriter =
241       new AccumuloAuditWriter(client, s"${catalog}_queries", auditProvider, AuditQueriesParam.lookup(params).booleanValue())
242 
243     val queries = AccumuloQueryConfig(
244       threads = QueryThreadsParam.lookup(params),
245       recordThreads = RecordThreadsParam.lookup(params),
246       timeout = QueryTimeoutParam.lookupOpt(params).map(_.toMillis),
247       looseBBox = LooseBBoxParam.lookup(params),
248       parallelPartitionScans = PartitionParallelScansParam.lookup(params)
249     )
250 
251     val remote = RemoteScansEnabled(
252       arrow = RemoteArrowParam.lookup(params),
253       bin = RemoteBinParam.lookup(params),
254       density = RemoteDensityParam.lookup(params),
255       stats = RemoteStatsParam.lookup(params)
256     )
257 
258     AccumuloDataStoreConfig(
259       catalog = catalog,
260       generateStats = GenerateStatsParam.lookup(params),
261       authProvider = authProvider,
262       auditWriter = auditWriter,
263       queries = queries,
264       remote = remote,
265       writeThreads = WriteThreadsParam.lookup(params),
266       namespace = NamespaceParam.lookupOpt(params)
267     )
268   }

278   def buildAuthsProvider(client: AccumuloClient, params: java.util.Map[String, _]): AuthorizationsProvider = {
279     // convert the connector authorizations into a string array - this is the maximum auths this connector can support
280     val securityOps = client.securityOperations
281     val masterAuths = securityOps.getUserAuthorizations(client.whoami).asScala.toSeq.map(b => new String(b))
282 
283     // get the auth params passed in as a comma-delimited string
284     val configuredAuths = AuthsParam.lookupOpt(params).getOrElse("").split(",").filterNot(_.isEmpty).toSeq
285 
286     // verify that the configured auths are valid for the connector we are using (fail-fast)
287     val invalidAuths = configuredAuths.filterNot(masterAuths.contains)
288     if (invalidAuths.nonEmpty) {
289       throw new IllegalArgumentException(s"The authorizations '${invalidAuths.mkString(",")}' " +
290         "are not valid for the Accumulo connection being used")
291     }
292 
293     val forceEmptyAuths = ForceEmptyAuthsParam.lookup(params)
294 
295     // if the caller provided any non-null string for authorizations, use it;
296     // otherwise, grab all authorizations to which the Accumulo user is entitled
297     if (configuredAuths.nonEmpty && forceEmptyAuths) {
298       throw new IllegalArgumentException("Forcing empty auths is checked, but explicit auths are provided")
299     }
300     val auths = if (forceEmptyAuths || configuredAuths.nonEmpty) { configuredAuths } else { masterAuths }
301 
302     AuthUtils.getProvider(params, auths)
303   }
