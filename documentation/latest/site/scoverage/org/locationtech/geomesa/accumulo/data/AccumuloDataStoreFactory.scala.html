124   def buildAccumuloConnector(params: java.util.Map[String, _]): AccumuloClient = {
125     val config = AccumuloClientConfig.load()
126 
127     def setRequired(param: GeoMesaParam[String], key: ClientProperty): Unit = {
128       param.lookupOpt(params) match {
129         case Some(v) => config.put(key.getKey, v)
130         case None =>
131           if (config.get(key.getKey) == null) {
132             throw new IOException(s"Parameter ${param.key} is required: ${param.description}")
133           }
134       }
135     }
136 
137     def setOptional(param: GeoMesaParam[String], key: ClientProperty): Unit =
138       param.lookupOpt(params).foreach(config.put(key.getKey, _))
139 
140     def getRequired(param: GeoMesaParam[String], key: ClientProperty): String = {
141       param.lookupOpt(params).orElse(Option(config.getProperty(key.getKey))).getOrElse {
142         throw new IOException(s"Parameter ${param.key} is required: ${param.description}")
143       }
144     }
145 
146     setRequired(InstanceNameParam, ClientProperty.INSTANCE_NAME)
147     setRequired(ZookeepersParam, ClientProperty.INSTANCE_ZOOKEEPERS)
148     setOptional(ZookeeperTimeoutParam, ClientProperty.INSTANCE_ZOOKEEPERS_TIMEOUT)
149 
150     val user = getRequired(UserParam, ClientProperty.AUTH_PRINCIPAL)
151 
152     if (PasswordParam.exists(params) && KeytabPathParam.exists(params)) {
153       throw new IllegalArgumentException(
154         s"'${PasswordParam.key}' and '${KeytabPathParam.key}' are mutually exclusive, but are both set")
155     }
156 
157     val authType =
158       if (PasswordParam.exists(params)) {
159         AccumuloClientConfig.PasswordAuthType
160       } else if (KeytabPathParam.exists(params)) {
161         AccumuloClientConfig.KerberosAuthType
162       } else {
163         ClientProperty.AUTH_TYPE.getValue(config).toLowerCase(Locale.US)
164       }
165 
166     // build authentication token according to how we are authenticating
167     val auth: AuthenticationToken = if (authType == AccumuloClientConfig.PasswordAuthType) {
168       new PasswordToken(getRequired(PasswordParam, ClientProperty.AUTH_TOKEN).getBytes(StandardCharsets.UTF_8))
169     } else if (authType == AccumuloClientConfig.KerberosAuthType) {
170       val file = new java.io.File(getRequired(KeytabPathParam, ClientProperty.AUTH_TOKEN))
171       // mimic behavior from accumulo 1.9 and earlier:
172       // `public KerberosToken(String principal, File keytab, boolean replaceCurrentUser)`
173       UserGroupInformation.loginUserFromKeytab(user, file.getAbsolutePath)
174       new KerberosToken(user, file)
175     } else {
176       throw new IllegalArgumentException(s"Unsupported auth type: $authType")
177     }
178 
179     if (WriteThreadsParam.exists(params)) {
180       config.put(ClientProperty.BATCH_WRITER_THREADS_MAX.getKey, WriteThreadsParam.lookup(params).toString)
181     } else if (!config.containsKey(ClientProperty.BATCH_WRITER_THREADS_MAX.getKey)) {
182       BatchWriterProperties.WRITER_THREADS.option.foreach { threads =>
183         config.put(ClientProperty.BATCH_WRITER_THREADS_MAX.getKey, String.valueOf(threads))
184       }
185     }
186     if (!config.containsKey(ClientProperty.BATCH_WRITER_MEMORY_MAX.getKey)) {
187       BatchWriterProperties.WRITER_MEMORY_BYTES.toBytes.foreach { memory =>
188         config.put(ClientProperty.BATCH_WRITER_MEMORY_MAX.getKey, String.valueOf(memory))
189       }
190     }
191     if (!config.containsKey(ClientProperty.BATCH_WRITER_LATENCY_MAX.getKey)) {
192       BatchWriterProperties.WRITER_LATENCY.toDuration.foreach { duration =>
193         config.put(ClientProperty.BATCH_WRITER_LATENCY_MAX.getKey, s"${duration.toMillis}ms")
194       }
195     }
196     if (!config.containsKey(ClientProperty.BATCH_WRITER_TIMEOUT_MAX.getKey)) {
197       BatchWriterProperties.WRITE_TIMEOUT.toDuration.foreach { duration =>
198         config.put(ClientProperty.BATCH_WRITER_TIMEOUT_MAX.getKey, s"${duration.toMillis}ms")
199       }
200     }
201 
202     Accumulo.newClient().from(config).as(user, auth).build()
203   }

205   def buildConfig(connector: AccumuloClient, params: java.util.Map[String, _]): AccumuloDataStoreConfig = {
206     val catalog = CatalogParam.lookup(params)
207 
208     val authProvider = buildAuthsProvider(connector, params)
209     val auditProvider = buildAuditProvider(params)
210     val auditQueries = AuditQueriesParam.lookup(params).booleanValue()

212     val auditService = new AccumuloAuditService(connector, authProvider, s"${catalog}_queries", auditQueries)

214     val queries = AccumuloQueryConfig(

216       recordThreads = RecordThreadsParam.lookup(params),
217       timeout = QueryTimeoutParam.lookupOpt(params).map(_.toMillis),

224       bin = RemoteBinParam.lookup(params),
225       density = RemoteDensityParam.lookup(params),
226       stats = RemoteStatsParam.lookup(params)

229     AccumuloDataStoreConfig(
230       catalog = catalog,
231       generateStats = GenerateStatsParam.lookup(params),
232       authProvider = authProvider,
233       audit = Some(auditService, auditProvider, AccumuloAuditService.StoreType),

249   def buildAuthsProvider(connector: AccumuloClient, params: java.util.Map[String, _]): AuthorizationsProvider = {
250     // convert the connector authorizations into a string array - this is the maximum auths this connector can support
251     val securityOps = connector.securityOperations
252     val masterAuths = securityOps.getUserAuthorizations(connector.whoami).asScala.toSeq.map(b => new String(b))
253 
254     // get the auth params passed in as a comma-delimited string
255     val configuredAuths = AuthsParam.lookupOpt(params).getOrElse("").split(",").filterNot(_.isEmpty).toSeq
256 
257     // verify that the configured auths are valid for the connector we are using (fail-fast)
258     val invalidAuths = configuredAuths.filterNot(masterAuths.contains)
259     if (invalidAuths.nonEmpty) {
260       throw new IllegalArgumentException(s"The authorizations '${invalidAuths.mkString(",")}' " +
261         "are not valid for the Accumulo connection being used")
262     }
263 
264     val forceEmptyAuths = ForceEmptyAuthsParam.lookup(params)
265 
266     // if the caller provided any non-null string for authorizations, use it;
267     // otherwise, grab all authorizations to which the Accumulo user is entitled
268     if (configuredAuths.nonEmpty && forceEmptyAuths) {
269       throw new IllegalArgumentException("Forcing empty auths is checked, but explicit auths are provided")
270     }
271     val auths = if (forceEmptyAuths || configuredAuths.nonEmpty) { configuredAuths } else { masterAuths }
272 
273     AuthUtils.getProvider(params, auths)
274   }
