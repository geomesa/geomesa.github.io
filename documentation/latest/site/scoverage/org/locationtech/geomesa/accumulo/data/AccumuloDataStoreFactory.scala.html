129   /**
130    * Build an Accumulo client configuration, based on passed in parameters and any accumulo-client.properties
131    * file on the classpath
132    *
133    * @param params data store params
134    * @return
135    */
136   def buildAccumuloClientConfig(params: java.util.Map[String, _]): Properties = {
137     val config = AccumuloClientConfig.load()
138 
139     def setRequired(param: GeoMesaParam[String], key: ClientProperty): String = {
140       param.lookupOpt(params) match {
141         case Some(v) => config.put(key.getKey, v); v
142         case None =>
143           Option(config.getProperty(key.getKey)).getOrElse {
144             throw new IOException(s"Parameter ${param.key} is required: ${param.description}")
145           }
146       }
147     }
148 
149     def setOptional(param: GeoMesaParam[String], key: ClientProperty): Unit =
150       param.lookupOpt(params).foreach(config.put(key.getKey, _))
151 
152     def getRequired(param: GeoMesaParam[String], key: ClientProperty): String = {
153       param.lookupOpt(params).orElse(Option(config.getProperty(key.getKey))).getOrElse {
154         throw new IOException(s"Parameter ${param.key} is required: ${param.description}")
155       }
156     }
157 
158     setRequired(InstanceNameParam, ClientProperty.INSTANCE_NAME)
159     setRequired(ZookeepersParam, ClientProperty.INSTANCE_ZOOKEEPERS)
160     setOptional(ZookeeperTimeoutParam, ClientProperty.INSTANCE_ZOOKEEPERS_TIMEOUT)
161 
162     val user = setRequired(UserParam, ClientProperty.AUTH_PRINCIPAL)
163 
164     if (PasswordParam.exists(params) && KeytabPathParam.exists(params)) {
165       throw new IllegalArgumentException(
166         s"'${PasswordParam.key}' and '${KeytabPathParam.key}' are mutually exclusive, but are both set")
167     }
168 
169     val authType =
170       if (PasswordParam.exists(params)) {
171         AccumuloClientConfig.PasswordAuthType
172       } else if (KeytabPathParam.exists(params)) {
173         AccumuloClientConfig.KerberosAuthType
174       } else {
175         ClientProperty.AUTH_TYPE.getValue(config).toLowerCase(Locale.US)
176       }
177 
178     val token =
179       if (authType == AccumuloClientConfig.PasswordAuthType) {
180         new PasswordToken(getRequired(PasswordParam, ClientProperty.AUTH_TOKEN).getBytes(StandardCharsets.UTF_8))
181       } else if (authType == AccumuloClientConfig.KerberosAuthType) {
182         val file = new java.io.File(getRequired(KeytabPathParam, ClientProperty.AUTH_TOKEN))
183         // mimic behavior from accumulo 1.9 and earlier:
184         // `public KerberosToken(String principal, File keytab, boolean replaceCurrentUser)`
185         UserGroupInformation.loginUserFromKeytab(user, file.getAbsolutePath)
186         new KerberosToken(user, file)
187       } else {
188          throw new IllegalArgumentException(
189            s"Unsupported auth type: $authType - supported values are " +
190              s"${AccumuloClientConfig.PasswordAuthType}, ${AccumuloClientConfig.KerberosAuthType}")
191       }
192     ClientProperty.setAuthenticationToken(config, token)
193 
194     if (WriteThreadsParam.exists(params)) {
195       config.put(ClientProperty.BATCH_WRITER_THREADS_MAX.getKey, WriteThreadsParam.lookup(params).toString)
196     } else if (ClientProperty.BATCH_WRITER_THREADS_MAX.isEmpty(config)) {
197       BatchWriterProperties.WRITER_THREADS.option.foreach { threads =>
198         config.put(ClientProperty.BATCH_WRITER_THREADS_MAX.getKey, String.valueOf(threads))
199       }
200     }
201     if (ClientProperty.BATCH_WRITER_MEMORY_MAX.isEmpty(config)) {
202       BatchWriterProperties.WRITER_MEMORY_BYTES.toBytes.foreach { memory =>
203         config.put(ClientProperty.BATCH_WRITER_MEMORY_MAX.getKey, String.valueOf(memory))
204       }
205     }
206     if (ClientProperty.BATCH_WRITER_LATENCY_MAX.isEmpty(config)) {
207       BatchWriterProperties.WRITER_LATENCY.toDuration.foreach { duration =>
208         config.put(ClientProperty.BATCH_WRITER_LATENCY_MAX.getKey, s"${duration.toMillis}ms")
209       }
210     }
211     if (ClientProperty.BATCH_WRITER_TIMEOUT_MAX.isEmpty(config)) {
212       BatchWriterProperties.WRITE_TIMEOUT.toDuration.foreach { duration =>
213         config.put(ClientProperty.BATCH_WRITER_TIMEOUT_MAX.getKey, s"${duration.toMillis}ms")
214       }
215     }
216 
217     config
218   }

272   def buildAuthsProvider(client: AccumuloClient, params: java.util.Map[String, _]): AuthorizationsProvider = {
273     // convert the connector authorizations into a string array - this is the maximum auths this connector can support
274     val securityOps = client.securityOperations
275     val masterAuths = securityOps.getUserAuthorizations(client.whoami).asScala.toSeq.map(b => new String(b))
276 
277     // get the auth params passed in as a comma-delimited string
278     val configuredAuths = AuthsParam.lookupOpt(params).getOrElse("").split(",").filterNot(_.isEmpty).toSeq
279 
280     // verify that the configured auths are valid for the connector we are using (fail-fast)
281     val invalidAuths = configuredAuths.filterNot(masterAuths.contains)
282     if (invalidAuths.nonEmpty) {
283       throw new IllegalArgumentException(s"The authorizations '${invalidAuths.mkString(",")}' " +
284         "are not valid for the Accumulo connection being used")
285     }
286 
287     val forceEmptyAuths = ForceEmptyAuthsParam.lookup(params)
288 
289     // if the caller provided any non-null string for authorizations, use it;
290     // otherwise, grab all authorizations to which the Accumulo user is entitled
291     if (configuredAuths.nonEmpty && forceEmptyAuths) {
292       throw new IllegalArgumentException("Forcing empty auths is checked, but explicit auths are provided")
293     }
294     val auths = if (forceEmptyAuths || configuredAuths.nonEmpty) { configuredAuths } else { masterAuths }
295 
296     AuthUtils.getProvider(params, auths)
297   }
