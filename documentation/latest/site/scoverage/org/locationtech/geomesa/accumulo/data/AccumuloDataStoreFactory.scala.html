132   /**
133    * Build an Accumulo client configuration, based on passed in parameters and any accumulo-client.properties
134    * file on the classpath
135    *
136    * @param params data store params
137    * @return
138    */
139   def buildAccumuloClientConfig(params: java.util.Map[String, _]): Properties = {
140     val config = AccumuloClientConfig.load()
141 
142     def setRequired(param: GeoMesaParam[String], key: ClientProperty): String = {
143       param.lookupOpt(params) match {
144         case Some(v) => config.put(key.getKey, v); v
145         case None =>
146           Option(config.getProperty(key.getKey)).getOrElse {
147             throw new IOException(s"Parameter ${param.key} is required: ${param.description}")
148           }
149       }
150     }
151 
152     def setOptional(param: GeoMesaParam[String], key: ClientProperty): Unit =
153       param.lookupOpt(params).foreach(config.put(key.getKey, _))
154 
155     def getRequired(param: GeoMesaParam[String], key: ClientProperty): String = {
156       param.lookupOpt(params).orElse(Option(config.getProperty(key.getKey))).getOrElse {
157         throw new IOException(s"Parameter ${param.key} is required: ${param.description}")
158       }
159     }
160 
161     setRequired(InstanceNameParam, ClientProperty.INSTANCE_NAME)
162     setRequired(ZookeepersParam, ClientProperty.INSTANCE_ZOOKEEPERS)
163     setOptional(ZookeeperTimeoutParam, ClientProperty.INSTANCE_ZOOKEEPERS_TIMEOUT)
164 
165     val user = setRequired(UserParam, ClientProperty.AUTH_PRINCIPAL)
166 
167     if (PasswordParam.exists(params) && KeytabPathParam.exists(params)) {
168       throw new IllegalArgumentException(
169         s"'${PasswordParam.key}' and '${KeytabPathParam.key}' are mutually exclusive, but are both set")
170     }
171 
172     val authType =
173       if (PasswordParam.exists(params)) {
174         AccumuloClientConfig.PasswordAuthType
175       } else if (KeytabPathParam.exists(params)) {
176         AccumuloClientConfig.KerberosAuthType
177       } else {
178         ClientProperty.AUTH_TYPE.getValue(config).toLowerCase(Locale.US)
179       }
180 
181     val token =
182       if (authType == AccumuloClientConfig.PasswordAuthType) {
183         new PasswordToken(getRequired(PasswordParam, ClientProperty.AUTH_TOKEN).getBytes(StandardCharsets.UTF_8))
184       } else if (authType == AccumuloClientConfig.KerberosAuthType) {
185         val file = new java.io.File(getRequired(KeytabPathParam, ClientProperty.AUTH_TOKEN))
186         // mimic behavior from accumulo 1.9 and earlier:
187         // `public KerberosToken(String principal, File keytab, boolean replaceCurrentUser)`
188         UserGroupInformation.loginUserFromKeytab(user, file.getAbsolutePath)
189         new KerberosToken(user, file)
190       } else {
191          throw new IllegalArgumentException(
192            s"Unsupported auth type: $authType - supported values are " +
193              s"${AccumuloClientConfig.PasswordAuthType}, ${AccumuloClientConfig.KerberosAuthType}")
194       }
195     ClientProperty.setAuthenticationToken(config, token)
196 
197     if (WriteThreadsParam.exists(params)) {
198       config.put(ClientProperty.BATCH_WRITER_THREADS_MAX.getKey, WriteThreadsParam.lookup(params).toString)
199     } else if (ClientProperty.BATCH_WRITER_THREADS_MAX.isEmpty(config)) {
200       BatchWriterProperties.WRITER_THREADS.option.foreach { threads =>
201         config.put(ClientProperty.BATCH_WRITER_THREADS_MAX.getKey, String.valueOf(threads))
202       }
203     }
204     if (ClientProperty.BATCH_WRITER_MEMORY_MAX.isEmpty(config)) {
205       BatchWriterProperties.WRITER_MEMORY_BYTES.toBytes.foreach { memory =>
206         config.put(ClientProperty.BATCH_WRITER_MEMORY_MAX.getKey, String.valueOf(memory))
207       }
208     }
209     if (ClientProperty.BATCH_WRITER_LATENCY_MAX.isEmpty(config)) {
210       BatchWriterProperties.WRITER_LATENCY.toDuration.foreach { duration =>
211         config.put(ClientProperty.BATCH_WRITER_LATENCY_MAX.getKey, s"${duration.toMillis}ms")
212       }
213     }
214     if (ClientProperty.BATCH_WRITER_TIMEOUT_MAX.isEmpty(config)) {
215       BatchWriterProperties.WRITE_TIMEOUT.toDuration.foreach { duration =>
216         config.put(ClientProperty.BATCH_WRITER_TIMEOUT_MAX.getKey, s"${duration.toMillis}ms")
217       }
218     }
219 
220     config
221   }

237     val metrics = MetricsRegistryParam.lookupRegistry(params)

252       stats = RemoteStatsParam.lookup(params)

256       catalog = catalog,
257       generateStats = GenerateStatsParam.lookup(params),
258       authProvider = authProvider,
259       auditWriter = auditWriter,

261       queries = queries,

276   def buildAuthsProvider(client: AccumuloClient, params: java.util.Map[String, _]): AuthorizationsProvider = {
277     // convert the connector authorizations into a string array - this is the maximum auths this connector can support
278     val securityOps = client.securityOperations
279     val masterAuths = securityOps.getUserAuthorizations(client.whoami).asScala.toSeq.map(b => new String(b))
280 
281     // get the auth params passed in as a comma-delimited string
282     val configuredAuths = AuthsParam.lookupOpt(params).getOrElse("").split(",").filterNot(_.isEmpty).toSeq
283 
284     // verify that the configured auths are valid for the connector we are using (fail-fast)
285     val invalidAuths = configuredAuths.filterNot(masterAuths.contains)
286     if (invalidAuths.nonEmpty) {
287       throw new IllegalArgumentException(s"The authorizations '${invalidAuths.mkString("', '")}' " +
288         "are not valid for the Accumulo connection being used")
289     }
290 
291     val forceEmptyAuths = ForceEmptyAuthsParam.lookup(params)
292 
293     // if the caller provided any non-null string for authorizations, use it;
294     // otherwise, grab all authorizations to which the Accumulo user is entitled
295     if (configuredAuths.nonEmpty && forceEmptyAuths) {
296       throw new IllegalArgumentException("Forcing empty auths is checked, but explicit auths are provided")
297     }
298     val auths = if (forceEmptyAuths || configuredAuths.nonEmpty) { configuredAuths } else { masterAuths }
299 
300     AuthUtils.getProvider(params, auths)
301   }
