125   def buildAccumuloConnector(params: java.util.Map[String, _]): AccumuloClient = {
126     val config = AccumuloClientConfig.load()
127 
128     def setRequired(param: GeoMesaParam[String], key: ClientProperty): Unit = {
129       param.lookupOpt(params) match {
130         case Some(v) => config.put(key.getKey, v)
131         case None =>
132           if (config.get(key.getKey) == null) {
133             throw new IOException(s"Parameter ${param.key} is required: ${param.description}")
134           }
135       }
136     }
137 
138     def setOptional(param: GeoMesaParam[String], key: ClientProperty): Unit =
139       param.lookupOpt(params).foreach(config.put(key.getKey, _))
140 
141     def getRequired(param: GeoMesaParam[String], key: ClientProperty): String = {
142       param.lookupOpt(params).orElse(Option(config.getProperty(key.getKey))).getOrElse {
143         throw new IOException(s"Parameter ${param.key} is required: ${param.description}")
144       }
145     }
146 
147     setRequired(InstanceNameParam, ClientProperty.INSTANCE_NAME)
148     setRequired(ZookeepersParam, ClientProperty.INSTANCE_ZOOKEEPERS)
149     setOptional(ZookeeperTimeoutParam, ClientProperty.INSTANCE_ZOOKEEPERS_TIMEOUT)
150 
151     val user = getRequired(UserParam, ClientProperty.AUTH_PRINCIPAL)
152 
153     if (PasswordParam.exists(params) && KeytabPathParam.exists(params)) {
154       throw new IllegalArgumentException(
155         s"'${PasswordParam.key}' and '${KeytabPathParam.key}' are mutually exclusive, but are both set")
156     }
157 
158     val authType =
159       if (PasswordParam.exists(params)) {
160         AccumuloClientConfig.PasswordAuthType
161       } else if (KeytabPathParam.exists(params)) {
162         AccumuloClientConfig.KerberosAuthType
163       } else {
164         ClientProperty.AUTH_TYPE.getValue(config).toLowerCase(Locale.US)
165       }
166 
167     // build authentication token according to how we are authenticating
168     val auth: AuthenticationToken = if (authType == AccumuloClientConfig.PasswordAuthType) {
169       new PasswordToken(getRequired(PasswordParam, ClientProperty.AUTH_TOKEN).getBytes(StandardCharsets.UTF_8))
170     } else if (authType == AccumuloClientConfig.KerberosAuthType) {
171       val file = new java.io.File(getRequired(KeytabPathParam, ClientProperty.AUTH_TOKEN))
172       // mimic behavior from accumulo 1.9 and earlier:
173       // `public KerberosToken(String principal, File keytab, boolean replaceCurrentUser)`
174       UserGroupInformation.loginUserFromKeytab(user, file.getAbsolutePath)
175       new KerberosToken(user, file)
176     } else {
177       throw new IllegalArgumentException(s"Unsupported auth type: $authType")
178     }
179 
180     if (WriteThreadsParam.exists(params)) {
181       config.put(ClientProperty.BATCH_WRITER_THREADS_MAX.getKey, WriteThreadsParam.lookup(params).toString)
182     } else if (!config.containsKey(ClientProperty.BATCH_WRITER_THREADS_MAX.getKey)) {
183       BatchWriterProperties.WRITER_THREADS.option.foreach { threads =>
184         config.put(ClientProperty.BATCH_WRITER_THREADS_MAX.getKey, String.valueOf(threads))
185       }
186     }
187     if (!config.containsKey(ClientProperty.BATCH_WRITER_MEMORY_MAX.getKey)) {
188       BatchWriterProperties.WRITER_MEMORY_BYTES.toBytes.foreach { memory =>
189         config.put(ClientProperty.BATCH_WRITER_MEMORY_MAX.getKey, String.valueOf(memory))
190       }
191     }
192     if (!config.containsKey(ClientProperty.BATCH_WRITER_LATENCY_MAX.getKey)) {
193       BatchWriterProperties.WRITER_LATENCY.toDuration.foreach { duration =>
194         config.put(ClientProperty.BATCH_WRITER_LATENCY_MAX.getKey, s"${duration.toMillis}ms")
195       }
196     }
197     if (!config.containsKey(ClientProperty.BATCH_WRITER_TIMEOUT_MAX.getKey)) {
198       BatchWriterProperties.WRITE_TIMEOUT.toDuration.foreach { duration =>
199         config.put(ClientProperty.BATCH_WRITER_TIMEOUT_MAX.getKey, s"${duration.toMillis}ms")
200       }
201     }
202 
203     Accumulo.newClient().from(config).as(user, auth).build()
204   }

249   def buildAuthsProvider(connector: AccumuloClient, params: java.util.Map[String, _]): AuthorizationsProvider = {
250     // convert the connector authorizations into a string array - this is the maximum auths this connector can support
251     val securityOps = connector.securityOperations
252     val masterAuths = securityOps.getUserAuthorizations(connector.whoami).asScala.toSeq.map(b => new String(b))
253 
254     // get the auth params passed in as a comma-delimited string
255     val configuredAuths = AuthsParam.lookupOpt(params).getOrElse("").split(",").filterNot(_.isEmpty).toSeq
256 
257     // verify that the configured auths are valid for the connector we are using (fail-fast)
258     val invalidAuths = configuredAuths.filterNot(masterAuths.contains)
259     if (invalidAuths.nonEmpty) {
260       throw new IllegalArgumentException(s"The authorizations '${invalidAuths.mkString(",")}' " +
261         "are not valid for the Accumulo connection being used")
262     }
263 
264     val forceEmptyAuths = ForceEmptyAuthsParam.lookup(params)
265 
266     // if the caller provided any non-null string for authorizations, use it;
267     // otherwise, grab all authorizations to which the Accumulo user is entitled
268     if (configuredAuths.nonEmpty && forceEmptyAuths) {
269       throw new IllegalArgumentException("Forcing empty auths is checked, but explicit auths are provided")
270     }
271     val auths = if (forceEmptyAuths || configuredAuths.nonEmpty) { configuredAuths } else { masterAuths }
272 
273     AuthUtils.getProvider(params, auths)
274   }
