23   private [audit] val CQ_USER     = new Text("user")
24   private [audit] val CQ_FILTER   = new Text("queryFilter")
25   private [audit] val CQ_HINTS    = new Text("queryHints")
26   private [audit] val CQ_PLANTIME = new Text("timePlanning")
27   private [audit] val CQ_SCANTIME = new Text("timeScanning")
28   private [audit] val CQ_TIME     = new Text("timeTotal")
29   private [audit] val CQ_HITS     = new Text("hits")
30   private [audit] val CQ_DELETED  = new Text("deleted")
31 
32   override def toMutation(event: QueryEvent): Mutation = {
33     val mutation = createMutation(event)
34     val cf = createRandomColumnFamily
35     mutation.put(cf, CQ_USER,     new Value(event.user.getBytes(StandardCharsets.UTF_8)))
36     mutation.put(cf, CQ_FILTER,   new Value(event.filter.getBytes(StandardCharsets.UTF_8)))
37     mutation.put(cf, CQ_HINTS,    new Value(event.hints.getBytes(StandardCharsets.UTF_8)))
38     mutation.put(cf, CQ_PLANTIME, new Value(s"${event.planTime}ms".getBytes(StandardCharsets.UTF_8)))
39     mutation.put(cf, CQ_SCANTIME, new Value(s"${event.scanTime}ms".getBytes(StandardCharsets.UTF_8)))
40     mutation.put(cf, CQ_TIME,     new Value(s"${event.scanTime + event.planTime}ms".getBytes(StandardCharsets.UTF_8)))
41     mutation.put(cf, CQ_HITS,     new Value(event.hits.toString.getBytes(StandardCharsets.UTF_8)))
42     mutation.put(cf, CQ_DELETED,  new Value(event.deleted.toString.getBytes(StandardCharsets.UTF_8)))
43     mutation
44   }
45 
46   override def toEvent(entries: Iterable[Entry[Key, Value]]): QueryEvent = {
47     if (entries.isEmpty) {
48       return null
49     }
50 
51     val (featureName, date) = typeNameAndDate(entries.head.getKey)
52     val values = collection.mutable.Map.empty[Text, Any]
53 
54     entries.foreach { e =>
55       e.getKey.getColumnQualifier match {
56         case CQ_USER     => values.put(CQ_USER, e.getValue.toString)
57         case CQ_FILTER   => values.put(CQ_FILTER, e.getValue.toString)
58         case CQ_HINTS    => values.put(CQ_HINTS, e.getValue.toString)
59         case CQ_PLANTIME => values.put(CQ_PLANTIME, e.getValue.toString.stripSuffix("ms").toLong)
60         case CQ_SCANTIME => values.put(CQ_SCANTIME, e.getValue.toString.stripSuffix("ms").toLong)
61         case CQ_HITS     => values.put(CQ_HITS, e.getValue.toString.toLong)
62         case CQ_DELETED  => values.put(CQ_DELETED, e.getValue.toString.toBoolean)
63         case CQ_TIME     => // time is an aggregate, doesn't need to map back to anything
64         case _           => logger.warn(s"Unmapped entry in query stat: ${e.getKey.getColumnQualifier.toString}")
65       }
66     }
67 
68     val user = values.getOrElse(CQ_USER, "unknown").asInstanceOf[String]
69     val queryHints = values.getOrElse(CQ_HINTS, "").asInstanceOf[String]
70     val queryFilter = values.getOrElse(CQ_FILTER, "").asInstanceOf[String]
71     val planTime = values.getOrElse(CQ_PLANTIME, 0L).asInstanceOf[Long]
72     val scanTime = values.getOrElse(CQ_SCANTIME, 0L).asInstanceOf[Long]
73     val hits = values.getOrElse(CQ_HITS, 0L).asInstanceOf[Long]
74     val deleted = values.getOrElse(CQ_DELETED, false).asInstanceOf[Boolean]
75 
76     QueryEvent(AccumuloAuditService.StoreType, featureName, date, user, queryFilter, queryHints, planTime, scanTime, hits, deleted)
77   }
78 }
