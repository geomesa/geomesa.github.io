22 
23   private val CqUser     = new Text("user")
24   private val CqFilter   = new Text("queryFilter")
25   private val CqHints    = new Text("queryHints")
26   private val CqPlanTime = new Text("timePlanning")
27   private val CqScanTime = new Text("timeScanning")
28   private val CqTime     = new Text("timeTotal")
29   private val CqHits     = new Text("hits")
30 
31   override def toMutation(event: QueryEvent): Mutation = {
32     val mutation = new Mutation(AccumuloEventTransform.toRowKey(event.typeName, event.date))
33     // avoids collisions if timestamp is the same, not 100% foolproof but good enough
34     val cf = AccumuloEventTransform.createRandomColumnFamily
35     mutation.put(cf, CqUser,     new Value(event.user.getBytes(StandardCharsets.UTF_8)))
36     mutation.put(cf, CqFilter,   new Value(event.filter.getBytes(StandardCharsets.UTF_8)))
37     mutation.put(cf, CqHints,    new Value(event.hints.getBytes(StandardCharsets.UTF_8)))
38     mutation.put(cf, CqPlanTime, new Value(s"${event.planTime}ms".getBytes(StandardCharsets.UTF_8)))
39     mutation.put(cf, CqScanTime, new Value(s"${event.scanTime}ms".getBytes(StandardCharsets.UTF_8)))
40     mutation.put(cf, CqTime,     new Value(s"${event.scanTime + event.planTime}ms".getBytes(StandardCharsets.UTF_8)))
41     mutation.put(cf, CqHits,     new Value(event.hits.toString.getBytes(StandardCharsets.UTF_8)))
42     mutation
43   }
44 
45   override def toEvent(entries: Iterable[Entry[Key, Value]]): QueryEvent = {
46     if (entries.isEmpty) {
47       return null
48     }
49 
50     val (featureName, date) = AccumuloEventTransform.typeNameAndDate(entries.head.getKey)
51 
52     var user = "unknown"
53     var queryHints = ""
54     var queryFilter = ""
55     var planTime = 0L
56     var scanTime = 0L
57     var hits = 0L
58 
59     entries.foreach { e =>
60       e.getKey.getColumnQualifier match {
61         case CqUser     => user = e.getValue.toString
62         case CqFilter   => queryFilter = e.getValue.toString
63         case CqHints    => queryHints = e.getValue.toString
64         case CqPlanTime => planTime = e.getValue.toString.stripSuffix("ms").toLong
65         case CqScanTime => scanTime = e.getValue.toString.stripSuffix("ms").toLong
66         case CqHits     => hits = e.getValue.toString.toLong
67         case CqTime     => // time is an aggregate, doesn't need to map back to anything
68         case _          => logger.warn(s"Unmapped entry in query stat: ${e.getKey.getColumnQualifier.toString}")
69       }
70     }
71 
72     QueryEvent(StoreType, featureName, date, user, queryFilter, queryHints, planTime, scanTime, hits)
73   }
74 }
