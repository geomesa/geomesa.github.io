40   override def toMutation(event: QueryEvent): Mutation = {
41     val mutation = new Mutation(AccumuloEventTransform.toRowKey(event.typeName, event.end))
42     // avoids collisions if timestamp is the same, not 100% foolproof but good enough
43     val cf = AccumuloEventTransform.createRandomColumnFamily
44     mutation.put(cf, CqUser,     new Value(event.user.getBytes(StandardCharsets.UTF_8)))
45     mutation.put(cf, CqFilter,   new Value(event.filter.getBytes(StandardCharsets.UTF_8)))
46     mutation.put(cf, CqHints,    new Value(AuditWriter.Gson.toJson(event.hints).getBytes(StandardCharsets.UTF_8)))
47     mutation.put(cf, CqMetadata, new Value(AuditWriter.Gson.toJson(event.metadata).getBytes(StandardCharsets.UTF_8)))
48     mutation.put(cf, CqStart,    new Value(event.start.toString.getBytes(StandardCharsets.UTF_8)))
49     mutation.put(cf, CqEnd,      new Value(event.end.toString.getBytes(StandardCharsets.UTF_8)))
50     mutation.put(cf, CqPlanTime, new Value(s"${event.planTime}ms".getBytes(StandardCharsets.UTF_8)))
51     mutation.put(cf, CqScanTime, new Value(s"${event.scanTime}ms".getBytes(StandardCharsets.UTF_8)))
52     mutation.put(cf, CqTime,     new Value(s"${event.scanTime + event.planTime}ms".getBytes(StandardCharsets.UTF_8)))
53     mutation.put(cf, CqHits,     new Value(event.hits.toString.getBytes(StandardCharsets.UTF_8)))
54     mutation
55   }

57   override def toEvent(entries: Iterable[Entry[Key, Value]]): QueryEvent = {
58     if (entries.isEmpty) {
59       return null
60     }
61 
62     val (featureName, end) = AccumuloEventTransform.typeNameAndDate(entries.head.getKey)
63 
64     var user = "unknown"
65     var hints = Collections.emptyMap[String, String]()
66     var metadata = Collections.emptyMap[String, AnyRef]()
67     var filter = ""
68     var start = -1L
69     var planTime = 0L
70     var scanTime = 0L
71     var hits = 0L
72 
73     entries.foreach { e =>
74       e.getKey.getColumnQualifier match {
75         case CqUser     => user = e.getValue.toString
76         case CqFilter   => filter = e.getValue.toString
77         case CqHints    => hints = parseHints(e.getValue.toString)
78         case CqMetadata => metadata = AuditWriter.Gson.fromJson(e.getValue.toString, classOf[java.util.Map[String, AnyRef]])
79         case CqStart    => start = e.getValue.toString.toLong
80         case CqEnd      => // end is already extracted from the row key
81         case CqPlanTime => planTime = e.getValue.toString.stripSuffix("ms").toLong
82         case CqScanTime => scanTime = e.getValue.toString.stripSuffix("ms").toLong
83         case CqHits     => hits = e.getValue.toString.toLong
84         case CqTime     => // time is an aggregate, doesn't need to map back to anything
85         case _          => logger.warn(s"Unmapped entry in query stat: ${e.getKey.getColumnQualifier.toString}")
86       }
87     }
88 
89     QueryEvent(StoreType, featureName, user, filter, hints, metadata, start, end, planTime, scanTime, hits)
90   }
91 
92   private def parseHints(value: String): java.util.Map[String, String] = {
93     if (value.isEmpty) {
94       Collections.emptyMap()
95     } else if (value.charAt(0) == '{') {
96       AuditWriter.Gson.fromJson(value, classOf[java.util.Map[String, String]])
97     } else {
98       // old format, do our best to parse it out into distinct hints
99       HintRegex.findAllMatchIn(value).map(m => m.group(1) -> m.group(2)).toMap.asJava
100     }
101   }
