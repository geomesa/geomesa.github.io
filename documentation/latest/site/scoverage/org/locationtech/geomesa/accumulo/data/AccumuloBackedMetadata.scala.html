21 class AccumuloBackedMetadata[T](
22     val client: AccumuloClient,
23     val table: String,
24     val serializer: MetadataSerializer[T],
25     consistency: Option[ConsistencyLevel] = None
26   ) extends KeyValueStoreMetadata[T] {
27 
28   import scala.collection.JavaConverters._
29 
30   // note: accumulo client is closed by the owning data store
31 
32   private val config = GeoMesaBatchWriterConfig().setMaxMemory(10000L).setMaxWriteThreads(2)
33 
34   private val empty = new Text()
35 
36   override protected def checkIfTableExists: Boolean = client.tableOperations().exists(table)
37 
38   override protected def createTable(): Unit = new TableManager(client).ensureTableExists(table)
39 
40   override protected def createEmptyBackup(timestamp: String): AccumuloBackedMetadata[T] =
41     new AccumuloBackedMetadata(client, s"${table}_${timestamp}_bak", serializer, consistency)
42 
43   override protected def write(rows: Seq[(Array[Byte], Array[Byte])]): Unit = {
44     WithClose(client.createBatchWriter(table, config)) { writer =>
45       rows.foreach { case (k, v) =>
46         val m = new Mutation(k)
47         m.put(empty, empty, new Value(v))
48         writer.addMutation(m)
49       }
50     }
51   }
52 
53   override protected def delete(rows: Seq[Array[Byte]]): Unit = {
54     val ranges = rows.map(r => Range.exact(new Text(r))).asJava
55     WithClose(client.createBatchDeleter(table, Authorizations.EMPTY, 1, config)) { deleter =>
56       deleter.setRanges(ranges)
57       deleter.delete()
58     }
59   }
60 
61   override protected def scanValue(row: Array[Byte]): Option[Array[Byte]] = {
62     WithClose(client.createScanner(table, Authorizations.EMPTY)) { scanner =>
63       scanner.setRange(Range.exact(new Text(row)))
64       consistency.foreach(scanner.setConsistencyLevel)
65       val iter = scanner.iterator
66       if (iter.hasNext) {
67         Some(iter.next.getValue.get)
68       } else {
69         None
70       }
71     }
72   }
73 
74   override protected def scanRows(prefix: Option[Array[Byte]]): CloseableIterator[(Array[Byte], Array[Byte])] = {
75     // ensure we don't scan any single-row encoded values
76     val range = prefix.map(p => Range.prefix(new Text(p))).getOrElse(new Range("", "~"))
77     val scanner = client.createScanner(table, Authorizations.EMPTY)
78     scanner.setRange(range)
79     consistency.foreach(scanner.setConsistencyLevel)
80     CloseableIterator(scanner.iterator.asScala.map(r => (r.getKey.getRow.copyBytes, r.getValue.get)), scanner.close())
81   }
82 
83   @deprecated("Use `client`")
84   def connector: AccumuloClient = client
85 }
