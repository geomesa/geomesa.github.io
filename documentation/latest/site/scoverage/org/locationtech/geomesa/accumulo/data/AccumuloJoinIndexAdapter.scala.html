48   /**
49    * Create a query plan against a join index - if possible, will use the reduced index-values to scan
50    * the single table, otherwise will require a join against the id index
51    *
52    * @param ds data store
53    * @param index index
54    * @param strategy query strategy
55    * @param tables tables to scan
56    * @param ranges ranges to scan
57    * @param colFamily column family to scan, optional
58    * @param schema simple feature schema being scanned
59    * @param ecql secondary push-down predicates
60    * @param hints query hints
61    * @param numThreads query threads
62    * @return
63    */
64   def createQueryPlan(
65       ds: AccumuloDataStore,
66       index: AttributeJoinIndex,
67       strategy: QueryStrategy,
68       tables: Seq[String],
69       ranges: Seq[org.apache.accumulo.core.data.Range],
70       colFamily: Option[Text],
71       schema: SimpleFeatureType,
72       ecql: Option[Filter],
73       hints: Hints,
74       numThreads: Int): AccumuloQueryPlan = {
75 
76     // TODO seems like this should be using 'schema' here, which may be a reduced version of the indexSft due to col groups
77     val indexSft = index.indexSft
78     val transform = hints.getTransformSchema
79 
80     // for queries that don't require a join, creates a regular batch scan plan
81     def plan(
82         iters: Seq[IteratorSetting],
83         kvsToFeatures: ResultsToFeatures[Entry[Key, Value]],
84         reduce: Option[FeatureReducer]): BatchScanPlan = {
85       // add the attribute-level vis iterator if necessary
86       val iterators = visibilityIter(index) ++ iters
87       val sort = hints.getSortFields
88       val max = hints.getMaxFeatures
89       val project = hints.getProjection
90       BatchScanPlan(ds, strategy, tables, ranges, iterators, colFamily, kvsToFeatures, reduce, sort, max, project, numThreads)
91     }
92 
93     // used when remote processing is disabled
94     def localPlan(overrides: Option[Seq[IteratorSetting]] = None): BatchScanLocalProcessorPlan = {
95       val iters = visibilityIter(index) ++ overrides.getOrElse(FilterTransformIterator.configure(indexSft, index, ecql, hints).toSeq)
96       val processor = LocalProcessor(transform.getOrElse(indexSft), QueryHints.Internal.clearTransforms(hints), None)
97       val projection = hints.getProjection
98       BatchScanLocalProcessorPlan(ds, strategy, tables, ranges, iters, colFamily, processor, projection, numThreads)
99     }
100 
101     val qp = if (hints.isBinQuery) {
102       // check to see if we can execute against the index values
103       if (indexSft.indexOf(hints.getBinTrackIdField) != -1 &&
104           hints.getBinGeomField.forall(indexSft.indexOf(_) != -1) &&
105           hints.getBinLabelField.forall(indexSft.indexOf(_) != -1) &&
106           index.supportsFilter(ecql)) {
107         if (ds.config.remote.bin) {
108           val binIter = BinAggregatingIterator.configure(indexSft, index, ecql, hints)
109           plan(Seq(binIter), new AccumuloBinResultsToFeatures(), None)
110         } else {
111           localPlan()
112         }
113       } else {
114         // have to do a join against the record table
115         createJoinPlan(ds, index, strategy, tables, ranges, colFamily, ecql, hints)
116       }
117     } else if (hints.isArrowQuery) {
118       // check to see if we can execute against the index values
119       if (index.canUseIndexSchema(ecql, transform)) {
120         if (ds.config.remote.arrow) {
121           val (arrowIter, reduce) = ArrowIterator.configure(indexSft, index, ds.stats, strategy.filter.filter, ecql, hints)
122           plan(Seq(arrowIter), new AccumuloArrowResultsToFeatures(), Some(reduce))
123         } else {
124           localPlan()
125         }
126       } else if (index.canUseIndexSchemaPlusKey(ecql, transform)) {
127         val transformSft = transform.getOrElse {
128           throw new IllegalStateException("Must have a transform for attribute key plus value scan")
129         }
130         // first filter and apply the transform
131         val filterTransformIter = FilterTransformIterator.configure(indexSft, index, ecql, hints, 23).get
132         // next add the attribute value from the row key
133         val rowValueIter = AttributeKeyValueIterator.configure(index.asInstanceOf[AttributeIndex], transformSft, 24)
134         if (ds.config.remote.arrow) {
135           // clear the transforms as we've already accounted for them
136           val newHints = new Hints(hints)
137           newHints.clearTransforms()
138           // finally apply the arrow iterator on the resulting features
139           val (arrowIter, reduce) = ArrowIterator.configure(transformSft, index, ds.stats, None, None, newHints)
140           val iters = Seq(filterTransformIter, rowValueIter, arrowIter)
141           plan(iters, new AccumuloArrowResultsToFeatures(), Some(reduce))
142         } else {
143           localPlan(Some(Seq(filterTransformIter, rowValueIter)))
144         }
145       } else {
146         // have to do a join against the record table
147         createJoinPlan(ds, index, strategy, tables, ranges, colFamily, ecql, hints)
148       }
149     } else if (hints.isDensityQuery) {
150       // check to see if we can execute against the index values
151       val weightIsAttribute = hints.getDensityWeight.contains(index.attributes.head)
152       if (index.supportsFilter(ecql) && (weightIsAttribute || hints.getDensityWeight.forall(indexSft.indexOf(_) != -1))) {
153         if (ds.config.remote.density) {
154           val iters = if (weightIsAttribute) {
155             // create a transform sft with the attribute added
156             val transform = {
157               val builder = new SimpleFeatureTypeBuilder()
158               builder.setNamespaceURI(null: String)
159               builder.setName(indexSft.getTypeName + "--attr")
160               builder.setAttributes(indexSft.getAttributeDescriptors)
161               builder.add(index.sft.getDescriptor(index.attributes.head))
162               if (indexSft.getGeometryDescriptor != null) {
163                 builder.setDefaultGeometry(indexSft.getGeometryDescriptor.getLocalName)
164               }
165               builder.setCRS(indexSft.getCoordinateReferenceSystem)
166               val tmp = builder.buildFeatureType()
167               tmp.getUserData.putAll(indexSft.getUserData)
168               tmp
169             }
170             // priority needs to be between vis iter (21) and density iter (25)
171             val keyValueIter = AttributeKeyValueIterator.configure(index.asInstanceOf[AttributeIndex], transform, 23)
172             val densityIter = DensityIterator.configure(transform, index, ecql, hints)
173             Seq(keyValueIter, densityIter)
174           } else {
175             Seq(DensityIterator.configure(indexSft, index, ecql, hints))
176           }
177           plan(iters, new AccumuloDensityResultsToFeatures(), None)
178         } else {
179           localPlan()
180         }
181       } else {
182         // have to do a join against the record table
183         createJoinPlan(ds, index, strategy, tables, ranges, colFamily, ecql, hints)
184       }
185     } else if (hints.isStatsQuery) {
186       // check to see if we can execute against the index values
187       if (Try(Stat(indexSft, hints.getStatsQuery)).isSuccess && index.supportsFilter(ecql)) {
188         if (ds.config.remote.stats) {
189           val statsIter = StatsIterator.configure(indexSft, index, ecql, hints)
190           val reduce = Some(StatsScan.StatsReducer(indexSft, hints))
191           plan(Seq(statsIter), new AccumuloStatsResultsToFeatures(), reduce)
192         } else {
193           localPlan()
194         }
195       } else {
196         // have to do a join against the record table
197         createJoinPlan(ds, index, strategy, tables, ranges, colFamily, ecql, hints)
198       }
199     } else if (index.canUseIndexSchema(ecql, transform)) {
200       // we can use the index value
201       // transform has to be non-empty to get here and can only include items
202       // in the index value (not the index keys aka the attribute indexed)
203       val transformSft = transform.getOrElse {
204         throw new IllegalStateException("Must have a transform for attribute value scan")
205       }
206       val fti = FilterTransformIterator.configure(indexSft, index, ecql, hints.getTransform, hints.getSampling)
207       // need to use transform to convert key/values
208       val toFeatures = AccumuloResultsToFeatures(index, transformSft)
209       plan(fti.toSeq, toFeatures, None)
210     } else if (index.canUseIndexSchemaPlusKey(ecql, transform)) {
211       // we can use the index PLUS the value
212       val transformSft = transform.getOrElse {
213         throw new IllegalStateException("Must have a transform for attribute key plus value scan")
214       }
215       val fti = FilterTransformIterator.configure(indexSft, index, ecql, hints.getTransform, hints.getSampling)
216       val iters = fti.toSeq :+ AttributeKeyValueIterator.configure(index.asInstanceOf[AttributeIndex], transformSft)
217       // need to use transform to convert key/values
218       val toFeatures = AccumuloResultsToFeatures(index, transformSft)
219       plan(iters, toFeatures, None)
220     } else {
221       // have to do a join against the record table
222       createJoinPlan(ds, index, strategy, tables, ranges, colFamily, ecql, hints)
223     }
224 
225     if (ranges.nonEmpty) { qp } else { EmptyPlan(strategy, qp.reducer) }
226   }

228   /**
229     * Gets a query plan comprised of a join against the record table. This is the slowest way to
230     * execute a query, so we avoid it if possible.
231     */
232   private def createJoinPlan(
233       ds: AccumuloDataStore,
234       index: AttributeJoinIndex,
235       strategy: QueryStrategy,
236       tables: Seq[String],
237       ranges: Seq[org.apache.accumulo.core.data.Range],
238       colFamily: Option[Text],
239       ecql: Option[Filter],
240       hints: Hints): AccumuloQueryPlan = {
241     import org.locationtech.geomesa.filter.ff
242     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
243     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
244 
245     // apply any secondary filters or transforms against the record table
246     val recordIndex = ds.manager.indices(index.sft, IndexMode.Read).find(_.name == IdIndex.name).getOrElse {
247       throw new RuntimeException("Id index does not exist for join query: " +
248           ds.manager.indices(index.sft, IndexMode.Read).map(_.identifier).mkString(", "))
249     }
250 
251     // break out the st filter to evaluate against the attribute table
252     val (stFilter, ecqlFilter) = ecql.map { f =>
253       val (geomFilters, otherFilters) = partitionPrimarySpatials(f, index.sft)
254       val (temporalFilters, nonSTFilters) = partitionPrimaryTemporals(otherFilters, index.sft)
255       (andOption(geomFilters ++ temporalFilters), andOption(nonSTFilters))
256     }.getOrElse((None, None))
257 
258     val (recordColFamily, recordSchema) = {
259       val (cf, s) = ds.adapter.groups.group(index.sft, hints.getTransformDefinition, ecqlFilter)
260       (Some(new Text(ColumnFamilyMapper(recordIndex)(cf))), s)
261     }
262 
263     // since each range is a single row, it wouldn't be very efficient to do any aggregating scans
264     // instead, handle them with the local query runner
265     val resultSft = hints.getTransformSchema.getOrElse(index.sft)
266     val recordIterators = {
267       val recordIter = FilterTransformIterator.configure(recordSchema, recordIndex, ecqlFilter, hints).toSeq
268       if (index.sft.getVisibilityLevel != VisibilityLevel.Attribute) { recordIter } else {
269         Seq(KryoVisibilityRowEncoder.configure(recordSchema)) ++ recordIter
270       }
271     }
272     val recordToFeatures = AccumuloResultsToFeatures(recordIndex, resultSft)
273     // transforms are handled by the recordIter
274     val processor = LocalProcessor(resultSft, QueryHints.Internal.clearTransforms(hints), None)
275 
276     val recordTables = recordIndex.getTablesForQuery(strategy.filter.filter)
277     val recordThreads = ds.config.queries.recordThreads
278 
279     // function to join the attribute index scan results to the record table
280     // have to pull the feature id from the row
281     val joinFunction: JoinFunction = {
282       val prefix = index.sft.getTableSharingBytes
283       val idToBytes = GeoMesaFeatureIndex.idToBytes(index.sft)
284       kv => {
285         val row = kv.getKey.getRow
286         new Range(new Text(ByteArrays.concat(prefix, idToBytes(index.getIdFromRow(row.getBytes, 0, row.getLength, null)))))
287       }
288     }
289 
290     val joinQuery =
291       BatchScanPlan(ds, strategy, recordTables, Seq.empty, recordIterators, recordColFamily, recordToFeatures,
292         None, None, None, None, recordThreads)
293 
294     val attributeIters = visibilityIter(index) ++
295         FilterTransformIterator.configure(index.indexSft, index, stFilter, None, hints.getSampling).toSeq
296 
297     JoinPlan(ds, strategy, tables, ranges, attributeIters, colFamily, recordThreads, joinFunction, joinQuery,
298       processor, hints.getProjection)
299   }
