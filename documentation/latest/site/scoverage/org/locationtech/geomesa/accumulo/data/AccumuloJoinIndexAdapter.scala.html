48   /**
49     * Create a query plan against a join index - if possible, will use the reduced index-values to scan
50     * the single table, otherwise will require a join against the id index
51     *
52     * @param filter filter strategy
53     * @param tables tables to scan
54     * @param ranges ranges to scan
55     * @param colFamily column family to scan, optional
56     * @param schema simple feature schema being scanned
57     * @param ecql secondary push-down predicates
58     * @param hints query hints
59     * @param numThreads query threads

77     lazy val max = hints.getMaxFeatures

81     def plan(

83         kvsToFeatures: ResultsToFeatures[Entry[Key, Value]],

92     lazy val resultsToFeatures = AccumuloResultsToFeatures(index, returnSchema)
93     lazy val localReducer = Some(new LocalTransformReducer(returnSchema, None, None, None, hints))

95     val qp = if (hints.isBinQuery) {
96       // check to see if we can execute against the index values
97       if (indexSft.indexOf(hints.getBinTrackIdField) != -1 &&
98           hints.getBinGeomField.forall(indexSft.indexOf(_) != -1) &&
99           hints.getBinLabelField.forall(indexSft.indexOf(_) != -1) &&
100           index.supportsFilter(ecql)) {
101         if (ds.config.remote.bin) {
102           val iter = BinAggregatingIterator.configure(indexSft, index, ecql, hints)
103           val iters = visibilityIter(index) :+ iter
104           plan(iters, new AccumuloBinResultsToFeatures(), None)
105         } else {
106           if (hints.isSkipReduce) {
107             // override the return sft to reflect what we're actually returning,
108             // since the bin sft is only created in the local reduce step
109             hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
110           }
111           plan(fti, resultsToFeatures, localReducer)
112         }
113       } else {
114         // have to do a join against the record table
115         createJoinPlan(ds, index, filter, tables, ranges, colFamily, ecql, hints)
116       }
117     } else if (hints.isArrowQuery) {
118       // check to see if we can execute against the index values
119       if (index.canUseIndexSchema(ecql, transform)) {
120         if (ds.config.remote.bin) {
121           val (iter, reduce) = ArrowIterator.configure(indexSft, index, ds.stats, filter.filter, ecql, hints)
122           val iters = visibilityIter(index) :+ iter
123           plan(iters, new AccumuloArrowResultsToFeatures(), Some(reduce))
124         } else {
125           if (hints.isSkipReduce) {
126             // override the return sft to reflect what we're actually returning,
127             // since the arrow sft is only created in the local reduce step
128             hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
129           }
130           plan(fti, resultsToFeatures, localReducer)
131         }
132       } else if (index.canUseIndexSchemaPlusKey(ecql, transform)) {
133         val transformSft = transform.getOrElse {
134           throw new IllegalStateException("Must have a transform for attribute key plus value scan")

137         val filterTransformIter = FilterTransformIterator.configure(indexSft, index, ecql, hints, 23).get
138         // clear the transforms as we've already accounted for them
139         hints.clearTransforms()

141         val rowValueIter = AttributeKeyValueIterator.configure(index.asInstanceOf[AttributeIndex], transformSft, 24)

146           plan(iters, new AccumuloArrowResultsToFeatures(), Some(reduce))
147         } else {
148           if (hints.isSkipReduce) {
149             // override the return sft to reflect what we're actually returning,

158       }
159     } else if (hints.isDensityQuery) {
160       // check to see if we can execute against the index values
161       val weightIsAttribute = hints.getDensityWeight.contains(index.attributes.head)
162       if (index.supportsFilter(ecql) && (weightIsAttribute || hints.getDensityWeight.forall(indexSft.indexOf(_) != -1))) {

192             // since the density sft is only created in the local reduce step
193             hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)

195           plan(fti, resultsToFeatures, localReducer)
196         }
197       } else {
198         // have to do a join against the record table
199         createJoinPlan(ds, index, filter, tables, ranges, colFamily, ecql, hints)
200       }
201     } else if (hints.isStatsQuery) {
202       // check to see if we can execute against the index values
203       if (Try(Stat(indexSft, hints.getStatsQuery)).isSuccess && index.supportsFilter(ecql)) {
204         if (ds.config.remote.bin) {
205           val iter = StatsIterator.configure(indexSft, index, ecql, hints)
206           val iters = visibilityIter(index) :+ iter
207           val reduce = Some(StatsScan.StatsReducer(indexSft, hints))
208           plan(iters, new AccumuloStatsResultsToFeatures(), reduce)
209         } else {
210           if (hints.isSkipReduce) {
211             // override the return sft to reflect what we're actually returning,
212             // since the stats sft is only created in the local reduce step
213             hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)

216         }
217       } else {
218         // have to do a join against the record table
219         createJoinPlan(ds, index, filter, tables, ranges, colFamily, ecql, hints)
220       }
221     } else if (index.canUseIndexSchema(ecql, transform)) {
222       // we can use the index value
223       // transform has to be non-empty to get here and can only include items
224       // in the index value (not the index keys aka the attribute indexed)
225       val transformSft = transform.getOrElse {
226         throw new IllegalStateException("Must have a transform for attribute value scan")
227       }
228       val iter = FilterTransformIterator.configure(indexSft, index, ecql, hints.getTransform, hints.getSampling)
229       // add the attribute-level vis iterator if necessary
230       val iters = visibilityIter(index) ++ iter.toSeq
231       // need to use transform to convert key/values
232       val toFeatures = AccumuloResultsToFeatures(index, transformSft)
233       plan(iters, toFeatures, None)
234     } else if (index.canUseIndexSchemaPlusKey(ecql, transform)) {
235       // we can use the index PLUS the value
236       val transformSft = transform.getOrElse {
237         throw new IllegalStateException("Must have a transform for attribute key plus value scan")
238       }
239       val iter = FilterTransformIterator.configure(indexSft, index, ecql, hints.getTransform, hints.getSampling)
240       // add the attribute-level vis iterator if necessary
241       val iters =
242         visibilityIter(index) ++ iter.toSeq :+
243             AttributeKeyValueIterator.configure(index.asInstanceOf[AttributeIndex], transformSft)
244       // need to use transform to convert key/values
245       val toFeatures = AccumuloResultsToFeatures(index, transformSft)
246       plan(iters, toFeatures, None)
247     } else {
248       // have to do a join against the record table
249       createJoinPlan(ds, index, filter, tables, ranges, colFamily, ecql, hints)
250     }
251 
252     if (ranges.nonEmpty) { qp } else { EmptyPlan(qp.filter, qp.reducer) }
253   }

255   /**
256     * Gets a query plan comprised of a join against the record table. This is the slowest way to
257     * execute a query, so we avoid it if possible.
258     */
259   private def createJoinPlan(
260       ds: AccumuloDataStore,
261       index: AttributeJoinIndex,
262       filter: FilterStrategy,
263       tables: Seq[String],
264       ranges: Seq[org.apache.accumulo.core.data.Range],
265       colFamily: Option[Text],
266       ecql: Option[Filter],
267       hints: Hints): AccumuloQueryPlan = {
268     import org.locationtech.geomesa.filter.ff
269     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
270     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
271 
272     // apply any secondary filters or transforms against the record table
273     val recordIndex = ds.manager.indices(index.sft, IndexMode.Read).find(_.name == IdIndex.name).getOrElse {
274       throw new RuntimeException("Id index does not exist for join query: " +
275           ds.manager.indices(index.sft, IndexMode.Read).map(_.identifier).mkString(", "))
276     }
277 
278     // break out the st filter to evaluate against the attribute table
279     val (stFilter, ecqlFilter) = ecql.map { f =>
280       val (geomFilters, otherFilters) = partitionPrimarySpatials(f, index.sft)
281       val (temporalFilters, nonSTFilters) = partitionPrimaryTemporals(otherFilters, index.sft)
282       (andOption(geomFilters ++ temporalFilters), andOption(nonSTFilters))
283     }.getOrElse((None, None))
284 
285     val (recordColFamily, recordSchema) = {
286       val (cf, s) = ds.adapter.groups.group(index.sft, hints.getTransformDefinition, ecqlFilter)
287       (Some(new Text(ColumnFamilyMapper(recordIndex)(cf))), s)
288     }
289 
290     // since each range is a single row, it wouldn't be very efficient to do any aggregating scans
291     // instead, handle them with the local query runner
292     val resultSft = hints.getTransformSchema.getOrElse(index.sft)
293     val recordIterators = {
294       val recordIter = FilterTransformIterator.configure(recordSchema, recordIndex, ecqlFilter, hints).toSeq
295       if (index.sft.getVisibilityLevel != VisibilityLevel.Attribute) { recordIter } else {
296         Seq(KryoVisibilityRowEncoder.configure(recordSchema)) ++ recordIter
297       }
298     }
299     val toFeatures = AccumuloResultsToFeatures(recordIndex, resultSft)
300     val reducer = new LocalTransformReducer(resultSft, None, None, None, hints)
301 
302     val recordTables = recordIndex.getTablesForQuery(filter.filter)
303     val recordThreads = ds.config.queries.recordThreads
304 
305     // function to join the attribute index scan results to the record table
306     // have to pull the feature id from the row
307     val joinFunction: JoinFunction = {
308       val prefix = index.sft.getTableSharingBytes
309       val idToBytes = GeoMesaFeatureIndex.idToBytes(index.sft)
310       kv => {
311         val row = kv.getKey.getRow
312         new Range(new Text(ByteArrays.concat(prefix, idToBytes(index.getIdFromRow(row.getBytes, 0, row.getLength, null)))))
313       }
314     }
315 
316     val joinQuery = BatchScanPlan(filter, recordTables, Seq.empty, recordIterators, recordColFamily, toFeatures,
317       Some(reducer), hints.getSortFields, hints.getMaxFeatures, hints.getProjection, recordThreads)
318 
319     val attributeIters = visibilityIter(index) ++
320         FilterTransformIterator.configure(index.indexSft, index, stFilter, None, hints.getSampling).toSeq
321 
322     JoinPlan(filter, tables, ranges, attributeIters, colFamily, recordThreads, joinFunction, joinQuery)
323   }
