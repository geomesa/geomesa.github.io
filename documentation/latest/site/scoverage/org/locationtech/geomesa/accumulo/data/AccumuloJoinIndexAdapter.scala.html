49   /**
50     * Create a query plan against a join index - if possible, will use the reduced index-values to scan
51     * the single table, otherwise will require a join against the id index
52     *
53     * @param filter filter strategy
54     * @param tables tables to scan
55     * @param ranges ranges to scan
56     * @param colFamily column family to scan, optional
57     * @param schema simple feature schema being scanned
58     * @param ecql secondary push-down predicates
59     * @param hints query hints
60     * @param numThreads query threads
61     * @return
62     */
63   def createQueryPlan(
64       ds: AccumuloDataStore,
65       index: AttributeJoinIndex,
66       filter: FilterStrategy,
67       tables: Seq[String],
68       ranges: Seq[org.apache.accumulo.core.data.Range],
69       colFamily: Option[Text],
70       schema: SimpleFeatureType,
71       ecql: Option[Filter],
72       hints: Hints,
73       numThreads: Int): AccumuloQueryPlan = {
74 
75     // TODO seems like this should be using 'schema' here, which may be a reduced version of the indexSft due to col groups
76     val indexSft = index.indexSft
77     lazy val sort = hints.getSortFields
78     lazy val max = hints.getMaxFeatures
79     lazy val project = hints.getProjection
80 
81     // for queries that don't require a join, creates a regular batch scan plan
82     def plan(
83         iters: Seq[IteratorSetting],
84         kvsToFeatures: ResultsToFeatures[Entry[Key, Value]],
85         reduce: Option[FeatureReducer]): BatchScanPlan =
86       BatchScanPlan(filter, tables, ranges, iters, colFamily, kvsToFeatures, reduce, sort, max, project, numThreads)
87 
88     val transform = hints.getTransformSchema
89 
90     // used when remote processing is disabled
91     lazy val returnSchema = hints.getTransformSchema.getOrElse(indexSft)
92     lazy val fti = visibilityIter(index) ++ FilterTransformIterator.configure(indexSft, index, ecql, hints).toSeq
93     lazy val resultsToFeatures = AccumuloResultsToFeatures(index, returnSchema)
94     lazy val localReducer = {
95       val arrowHook = Some(ArrowDictionaryHook(ds.stats, filter.filter))
96       Some(new LocalTransformReducer(returnSchema, None, None, None, hints, arrowHook))
97     }
98 
99     val qp = if (hints.isBinQuery) {
100       // check to see if we can execute against the index values
101       if (indexSft.indexOf(hints.getBinTrackIdField) != -1 &&
102           hints.getBinGeomField.forall(indexSft.indexOf(_) != -1) &&
103           hints.getBinLabelField.forall(indexSft.indexOf(_) != -1) &&
104           index.supportsFilter(ecql)) {
105         if (ds.config.remote.bin) {
106           val iter = BinAggregatingIterator.configure(indexSft, index, ecql, hints)
107           val iters = visibilityIter(index) :+ iter
108           plan(iters, new AccumuloBinResultsToFeatures(), None)
109         } else {
110           if (hints.isSkipReduce) {
111             // override the return sft to reflect what we're actually returning,
112             // since the bin sft is only created in the local reduce step
113             hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
114           }
115           plan(fti, resultsToFeatures, localReducer)
116         }
117       } else {
118         // have to do a join against the record table
119         createJoinPlan(ds, index, filter, tables, ranges, colFamily, ecql, hints)
120       }
121     } else if (hints.isArrowQuery) {
122       // check to see if we can execute against the index values
123       if (index.canUseIndexSchema(ecql, transform)) {
124         if (ds.config.remote.bin) {
125           val (iter, reduce) = ArrowIterator.configure(indexSft, index, ds.stats, filter.filter, ecql, hints)
126           val iters = visibilityIter(index) :+ iter
127           plan(iters, new AccumuloArrowResultsToFeatures(), Some(reduce))
128         } else {
129           if (hints.isSkipReduce) {
130             // override the return sft to reflect what we're actually returning,
131             // since the arrow sft is only created in the local reduce step
132             hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
133           }
134           plan(fti, resultsToFeatures, localReducer)
135         }
136       } else if (index.canUseIndexSchemaPlusKey(ecql, transform)) {
137         val transformSft = transform.getOrElse {
138           throw new IllegalStateException("Must have a transform for attribute key plus value scan")
139         }
140         // first filter and apply the transform
141         val filterTransformIter = FilterTransformIterator.configure(indexSft, index, ecql, hints, 23).get
142         // clear the transforms as we've already accounted for them
143         hints.clearTransforms()
144         // next add the attribute value from the row key
145         val rowValueIter = AttributeKeyValueIterator.configure(index.asInstanceOf[AttributeIndex], transformSft, 24)
146         if (ds.config.remote.bin) {
147           // finally apply the arrow iterator on the resulting features
148           val (iter, reduce) = ArrowIterator.configure(transformSft, index, ds.stats, None, None, hints)
149           val iters = visibilityIter(index) ++ Seq(filterTransformIter, rowValueIter, iter)
150           plan(iters, new AccumuloArrowResultsToFeatures(), Some(reduce))
151         } else {
152           if (hints.isSkipReduce) {
153             // override the return sft to reflect what we're actually returning,
154             // since the arrow sft is only created in the local reduce step
155             hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
156           }
157           plan(fti, resultsToFeatures, localReducer)
158         }
159       } else {
160         // have to do a join against the record table
161         createJoinPlan(ds, index, filter, tables, ranges, colFamily, ecql, hints)
162       }
163     } else if (hints.isDensityQuery) {
164       // check to see if we can execute against the index values
165       val weightIsAttribute = hints.getDensityWeight.contains(index.attributes.head)
166       if (index.supportsFilter(ecql) && (weightIsAttribute || hints.getDensityWeight.forall(indexSft.indexOf(_) != -1))) {
167         if (ds.config.remote.bin) {
168           val visIter = visibilityIter(index)
169           val iters = if (weightIsAttribute) {
170             // create a transform sft with the attribute added
171             val transform = {
172               val builder = new SimpleFeatureTypeBuilder()
173               builder.setNamespaceURI(null: String)
174               builder.setName(indexSft.getTypeName + "--attr")
175               builder.setAttributes(indexSft.getAttributeDescriptors)
176               builder.add(index.sft.getDescriptor(index.attributes.head))
177               if (indexSft.getGeometryDescriptor != null) {
178                 builder.setDefaultGeometry(indexSft.getGeometryDescriptor.getLocalName)
179               }
180               builder.setCRS(indexSft.getCoordinateReferenceSystem)
181               val tmp = builder.buildFeatureType()
182               tmp.getUserData.putAll(indexSft.getUserData)
183               tmp
184             }
185             // priority needs to be between vis iter (21) and density iter (25)
186             val keyValueIter = AttributeKeyValueIterator.configure(index.asInstanceOf[AttributeIndex], transform, 23)
187             val densityIter = DensityIterator.configure(transform, index, ecql, hints)
188             visIter :+ keyValueIter :+ densityIter
189           } else {
190             visIter :+ DensityIterator.configure(indexSft, index, ecql, hints)
191           }
192           plan(iters, new AccumuloDensityResultsToFeatures(), None)
193         } else {
194           if (hints.isSkipReduce) {
195             // override the return sft to reflect what we're actually returning,
196             // since the density sft is only created in the local reduce step
197             hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
198           }
199           plan(fti, resultsToFeatures, localReducer)
200         }
201       } else {
202         // have to do a join against the record table
203         createJoinPlan(ds, index, filter, tables, ranges, colFamily, ecql, hints)
204       }
205     } else if (hints.isStatsQuery) {
206       // check to see if we can execute against the index values
207       if (Try(Stat(indexSft, hints.getStatsQuery)).isSuccess && index.supportsFilter(ecql)) {
208         if (ds.config.remote.bin) {
209           val iter = StatsIterator.configure(indexSft, index, ecql, hints)
210           val iters = visibilityIter(index) :+ iter
211           val reduce = Some(StatsScan.StatsReducer(indexSft, hints))
212           plan(iters, new AccumuloStatsResultsToFeatures(), reduce)
213         } else {
214           if (hints.isSkipReduce) {
215             // override the return sft to reflect what we're actually returning,
216             // since the stats sft is only created in the local reduce step
217             hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
218           }
219           plan(fti, resultsToFeatures, localReducer)
220         }
221       } else {
222         // have to do a join against the record table
223         createJoinPlan(ds, index, filter, tables, ranges, colFamily, ecql, hints)
224       }
225     } else if (index.canUseIndexSchema(ecql, transform)) {
226       // we can use the index value
227       // transform has to be non-empty to get here and can only include items
228       // in the index value (not the index keys aka the attribute indexed)
229       val transformSft = transform.getOrElse {
230         throw new IllegalStateException("Must have a transform for attribute value scan")
231       }
232       val iter = FilterTransformIterator.configure(indexSft, index, ecql, hints.getTransform, hints.getSampling)
233       // add the attribute-level vis iterator if necessary
234       val iters = visibilityIter(index) ++ iter.toSeq
235       // need to use transform to convert key/values
236       val toFeatures = AccumuloResultsToFeatures(index, transformSft)
237       plan(iters, toFeatures, None)
238     } else if (index.canUseIndexSchemaPlusKey(ecql, transform)) {
239       // we can use the index PLUS the value
240       val transformSft = transform.getOrElse {
241         throw new IllegalStateException("Must have a transform for attribute key plus value scan")
242       }
243       val iter = FilterTransformIterator.configure(indexSft, index, ecql, hints.getTransform, hints.getSampling)
244       // add the attribute-level vis iterator if necessary
245       val iters =
246         visibilityIter(index) ++ iter.toSeq :+
247             AttributeKeyValueIterator.configure(index.asInstanceOf[AttributeIndex], transformSft)
248       // need to use transform to convert key/values
249       val toFeatures = AccumuloResultsToFeatures(index, transformSft)
250       plan(iters, toFeatures, None)
251     } else {
252       // have to do a join against the record table
253       createJoinPlan(ds, index, filter, tables, ranges, colFamily, ecql, hints)
254     }
255 
256     if (ranges.nonEmpty) { qp } else { EmptyPlan(qp.filter, qp.reducer) }
257   }

259   /**
260     * Gets a query plan comprised of a join against the record table. This is the slowest way to
261     * execute a query, so we avoid it if possible.
262     */
263   private def createJoinPlan(
264       ds: AccumuloDataStore,
265       index: AttributeJoinIndex,
266       filter: FilterStrategy,
267       tables: Seq[String],
268       ranges: Seq[org.apache.accumulo.core.data.Range],
269       colFamily: Option[Text],
270       ecql: Option[Filter],
271       hints: Hints): AccumuloQueryPlan = {
272     import org.locationtech.geomesa.filter.ff
273     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
274     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
275 
276     // apply any secondary filters or transforms against the record table
277     val recordIndex = ds.manager.indices(index.sft, IndexMode.Read).find(_.name == IdIndex.name).getOrElse {
278       throw new RuntimeException("Id index does not exist for join query: " +
279           ds.manager.indices(index.sft, IndexMode.Read).map(_.identifier).mkString(", "))
280     }
281 
282     // break out the st filter to evaluate against the attribute table
283     val (stFilter, ecqlFilter) = ecql.map { f =>
284       val (geomFilters, otherFilters) = partitionPrimarySpatials(f, index.sft)
285       val (temporalFilters, nonSTFilters) = partitionPrimaryTemporals(otherFilters, index.sft)
286       (andOption(geomFilters ++ temporalFilters), andOption(nonSTFilters))
287     }.getOrElse((None, None))
288 
289     val (recordColFamily, recordSchema) = {
290       val (cf, s) = ds.adapter.groups.group(index.sft, hints.getTransformDefinition, ecqlFilter)
291       (Some(new Text(ColumnFamilyMapper(recordIndex)(cf))), s)
292     }
293 
294     // since each range is a single row, it wouldn't be very efficient to do any aggregating scans
295     // instead, handle them with the local query runner
296     val resultSft = hints.getTransformSchema.getOrElse(index.sft)
297     val recordIterators = {
298       val recordIter = FilterTransformIterator.configure(recordSchema, recordIndex, ecqlFilter, hints).toSeq
299       if (index.sft.getVisibilityLevel != VisibilityLevel.Attribute) { recordIter } else {
300         Seq(KryoVisibilityRowEncoder.configure(recordSchema)) ++ recordIter
301       }
302     }
303     val toFeatures = AccumuloResultsToFeatures(recordIndex, resultSft)
304     val hook = Some(ArrowDictionaryHook(ds.stats, filter.filter))
305     val reducer = new LocalTransformReducer(resultSft, None, None, None, hints, hook)
306 
307     val recordTables = recordIndex.getTablesForQuery(filter.filter)
308     val recordThreads = ds.config.queries.recordThreads
309 
310     // function to join the attribute index scan results to the record table
311     // have to pull the feature id from the row
312     val joinFunction: JoinFunction = {
313       val prefix = index.sft.getTableSharingBytes
314       val idToBytes = GeoMesaFeatureIndex.idToBytes(index.sft)
315       kv => {
316         val row = kv.getKey.getRow
317         new Range(new Text(ByteArrays.concat(prefix, idToBytes(index.getIdFromRow(row.getBytes, 0, row.getLength, null)))))
318       }
319     }
320 
321     val joinQuery = BatchScanPlan(filter, recordTables, Seq.empty, recordIterators, recordColFamily, toFeatures,
322       Some(reducer), hints.getSortFields, hints.getMaxFeatures, hints.getProjection, recordThreads)
323 
324     val attributeIters = visibilityIter(index) ++
325         FilterTransformIterator.configure(index.indexSft, index, stFilter, None, hints.getSampling).toSeq
326 
327     JoinPlan(filter, tables, ranges, attributeIters, colFamily, recordThreads, joinFunction, joinQuery)
328   }
