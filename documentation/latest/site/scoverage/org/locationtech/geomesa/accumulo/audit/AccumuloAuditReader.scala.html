22 /**
23  * An audit reader
24  *
25  * @param client accumulo client - note: assumed to be shared and not cleaned up on closed
26  * @param table table containing audit records
27  * @param authProvider auth provider
28  * @param consistency scan consistency level
29  */
30 class AccumuloAuditReader(
31     client: AccumuloClient,
32     table: String,
33     authProvider: AuthorizationsProvider,
34     consistency: Option[ConsistencyLevel] = None
35   ) extends AuditReader {
36 
37   import scala.collection.JavaConverters._
38 
39   def this(ds: AccumuloDataStore) =
40     this(ds.client, ds.config.auditWriter.table, ds.config.authProvider, ds.config.queries.consistency)
41 
42   @volatile
43   private var tableExists: Boolean = client.tableOperations().exists(table)
44 
45   override def getQueryEvents(typeName: String, dates: (ZonedDateTime, ZonedDateTime)): CloseableIterator[QueryEvent] = {
46     if (!checkTable) { CloseableIterator.empty } else {
47       val scanner = client.createScanner(table, new Authorizations(authProvider.getAuthorizations.asScala.toSeq: _*))
48       consistency.foreach(scanner.setConsistencyLevel)
49       AccumuloQueryEventTransform.iterator(scanner, typeName, dates)
50     }
51   }
52 
53   override def close(): Unit = {}
54 
55   private def checkTable: Boolean = {
56     if (tableExists) {
57       true
58     } else if (client.tableOperations().exists(table)) {
59       tableExists = true
60       true
61     } else {
62       false
63     }
64   }
65 }
