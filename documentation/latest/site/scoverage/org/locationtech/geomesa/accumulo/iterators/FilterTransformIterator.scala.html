30 /**
31  * Iterator that operates on kryo encoded values. It will:
32  *   a) evaluate the feature against an optional filter
33  *   b) apply an optional transform
34  *
35  * Uses lazy evaluation of attributes and binary transforms when possible.
36  */
37 class FilterTransformIterator extends SortedKeyValueIterator[Key, Value] with SamplingIterator with LazyLogging {
38 
39   import org.locationtech.geomesa.index.iterators.AggregatingScan.Configuration._
40 
41   var source: SortedKeyValueIterator[Key, Value] = _
42   val topValue: Value = new Value()
43 
44   var sft: SimpleFeatureType = _
45   var filter: SimpleFeature => Boolean = _
46 
47   var reusableSf: KryoBufferSimpleFeature = _
48   var hasTransform: Boolean = _
49 
50   override def init(src: SortedKeyValueIterator[Key, Value],
51                     options: jMap[String, String],
52                     env: IteratorEnvironment): Unit = {
53     this.source = src
54 
55     val spec = options.get(SftOpt)
56     sft = IteratorCache.sft(spec)
57 
58     val index = {
59       val indexSpec = options.get(IndexSftOpt)
60       if (indexSpec == null) {
61         IteratorCache.index(sft, spec, options.get(IndexOpt))
62       } else {
63         IteratorCache.index(IteratorCache.sft(indexSpec), indexSpec, options.get(IndexOpt))
64       }
65     }
66     // noinspection ScalaDeprecation
67     val kryo = if (index.serializedWithId) { SerializationOptions.none } else { SerializationOptions.withoutId }
68     reusableSf = IteratorCache.serializer(spec, kryo).getReusableFeature
69     reusableSf.setIdParser(index.getIdFromRow(_, _, _, null))
70 
71     val transform = Option(options.get(TransformDefsOpt))
72     val transformSchema = Option(options.get(TransformSchemaOpt))
73     for { t <- transform; ts <- transformSchema } {
74       reusableSf.setTransforms(t, IteratorCache.sft(ts))
75     }
76     hasTransform = transform.isDefined
77 
78     val cql = Option(options.get(CqlOpt)).map(IteratorCache.filter(sft, spec, _))
79     // TODO: can we optimize the configuration of sampling
80     val sampling = sample(options)
81 
82     filter = (cql, sampling) match {
83       case (None, None)       => _ => true
84       case (Some(c), None)    => c.evaluate
85       case (None, Some(s))    => s
86       case (Some(c), Some(s)) => sf => c.evaluate(sf) && s(sf)
87     }
88   }
89 
90   override def seek(range: Range, columnFamilies: jCollection[ByteSequence], inclusive: Boolean): Unit = {
91     source.seek(range, columnFamilies, inclusive)
92     findTop()
93   }
94 
95   override def next(): Unit = {
96     source.next()
97     findTop()
98   }
99 
100   override def hasTop: Boolean = source.hasTop
101   override def getTopKey: Key = source.getTopKey
102   override def getTopValue: Value =
103     if (hasTransform) {
104       topValue.set(reusableSf.transform())
105       topValue
106     } else {
107       source.getTopValue
108     }
109 
110   def findTop(): Unit = {
111     var found = false
112     while (!found && source.hasTop) {
113       reusableSf.setBuffer(source.getTopValue.get())
114       val row = source.getTopKey.getRowData
115       reusableSf.setIdBuffer(row.getBackingArray, row.offset(), row.length())
116 
117       if (filter(reusableSf)) {
118         found = true
119       } else {
120         source.next()
121       }
122     }
123   }
124 
125   override def deepCopy(env: IteratorEnvironment): SortedKeyValueIterator[Key, Value] = {
126     val iter = new FilterTransformIterator
127     iter.source = source.deepCopy(env)
128     iter.sft = sft
129     iter.filter = filter
130     iter.reusableSf = reusableSf.copy()
131     iter.hasTransform = hasTransform
132     iter
133   }
134 }

156   def configure(
157       sft: SimpleFeatureType,
158       index: GeoMesaFeatureIndex[_, _],
159       filter: Option[Filter],
160       transform: Option[(String, SimpleFeatureType)],
161       sampling: Option[(Float, Option[String])],
162       compatibility: Option[FilterCompatibility] = None,
163       priority: Int = DefaultPriority): Option[IteratorSetting] = {
164     if (filter.isEmpty && transform.isEmpty && sampling.isEmpty) { None } else {
165       val is = new IteratorSetting(priority, "filter-transform-iter", classOf[FilterTransformIterator])
166       is.addOption(SftOpt, SimpleFeatureTypes.encodeType(sft, includeUserData = true))
167       if (sft != index.sft) {
168         is.addOption(IndexSftOpt, SimpleFeatureTypes.encodeType(index.sft, includeUserData = true))
169       }
170       is.addOption(IndexOpt, index.identifier)
171       filter.foreach(f => is.addOption(CqlOpt, ECQL.toCQL(f)))
172       transform.foreach { case (tdef, tsft) =>
173         is.addOption(TransformDefsOpt, tdef)
174         is.addOption(TransformSchemaOpt, SimpleFeatureTypes.encodeType(tsft))
175       }
176       sampling.foreach(SamplingIterator.configure(sft, _).foreach { case (k, v) => is.addOption(k, v) })
177 
178       compatibility match {
179         case None =>
180           is.addOption(VersionOpt, GeoMesaProperties.ProjectVersion)
181 
182         case Some(FilterCompatibility.`1.3`) =>
183           is.setIteratorClass("org.locationtech.geomesa.accumulo.iterators.KryoLazyFilterTransformIterator")
184           is.addOption(IndexOpt, s"${index.name}:${index.version}")
185 
186         case Some(c) =>
187           throw new UnsupportedOperationException(s"Unknown compatibility flag: '$c'")
188       }
189       Some(is)
190     }
191   }
