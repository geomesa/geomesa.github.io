29 /**
30  * Iterator that operates on kryo encoded values. It will:
31  *   a) evaluate the feature against an optional filter
32  *   b) apply an optional transform
33  *
34  * Uses lazy evaluation of attributes and binary transforms when possible.
35  */
36 class FilterTransformIterator extends SortedKeyValueIterator[Key, Value] with SamplingIterator with LazyLogging {
37 
38   import org.locationtech.geomesa.index.iterators.AggregatingScan.Configuration._
39 
40   var source: SortedKeyValueIterator[Key, Value] = _
41   val topValue: Value = new Value()
42 
43   var sft: SimpleFeatureType = _
44   var filter: SimpleFeature => Boolean = _
45 
46   var reusableSf: KryoBufferSimpleFeature = _
47   var hasTransform: Boolean = _
48 
49   override def init(src: SortedKeyValueIterator[Key, Value],
50                     options: jMap[String, String],
51                     env: IteratorEnvironment): Unit = {
52     this.source = src
53 
54     val spec = options.get(SftOpt)
55     sft = IteratorCache.sft(spec)
56 
57     val index = {
58       val indexSpec = options.get(IndexSftOpt)
59       if (indexSpec == null) {
60         IteratorCache.index(sft, spec, options.get(IndexOpt))
61       } else {
62         IteratorCache.index(IteratorCache.sft(indexSpec), indexSpec, options.get(IndexOpt))
63       }
64     }
65     reusableSf = IteratorCache.serializer(spec, index.serializedWithId).getReusableFeature
66     reusableSf.setIdParser(index.getIdFromRow(_, _, _, null))
67 
68     val transform = Option(options.get(TransformDefsOpt))
69     val transformSchema = Option(options.get(TransformSchemaOpt))
70     for { t <- transform; ts <- transformSchema } {
71       reusableSf.setTransforms(t, IteratorCache.sft(ts))
72     }
73     hasTransform = transform.isDefined
74 
75     val cql = Option(options.get(CqlOpt)).map(IteratorCache.filter(sft, spec, _))
76     // TODO: can we optimize the configuration of sampling
77     val sampling = sample(options)
78 
79     filter = (cql, sampling) match {
80       case (None, None)       => _ => true
81       case (Some(c), None)    => c.evaluate
82       case (None, Some(s))    => s
83       case (Some(c), Some(s)) => sf => c.evaluate(sf) && s(sf)
84     }
85   }
86 
87   override def seek(range: Range, columnFamilies: jCollection[ByteSequence], inclusive: Boolean): Unit = {
88     source.seek(range, columnFamilies, inclusive)
89     findTop()
90   }
91 
92   override def next(): Unit = {
93     source.next()
94     findTop()
95   }
96 
97   override def hasTop: Boolean = source.hasTop
98   override def getTopKey: Key = source.getTopKey
99   override def getTopValue: Value =
100     if (hasTransform) {
101       topValue.set(reusableSf.transform())
102       topValue
103     } else {
104       source.getTopValue
105     }
106 
107   def findTop(): Unit = {
108     var found = false
109     while (!found && source.hasTop) {
110       reusableSf.setBuffer(source.getTopValue.get())
111       val row = source.getTopKey.getRowData
112       reusableSf.setIdBuffer(row.getBackingArray, row.offset(), row.length())
113 
114       if (filter(reusableSf)) {
115         found = true
116       } else {
117         source.next()
118       }
119     }
120   }
121 
122   override def deepCopy(env: IteratorEnvironment): SortedKeyValueIterator[Key, Value] = {
123     val iter = new FilterTransformIterator
124     iter.source = source.deepCopy(env)
125     iter.sft = sft
126     iter.filter = filter
127     iter.reusableSf = reusableSf.copy()
128     iter.hasTransform = hasTransform
129     iter
130   }
131 }

153   def configure(
154       sft: SimpleFeatureType,
155       index: GeoMesaFeatureIndex[_, _],
156       filter: Option[Filter],
157       transform: Option[(String, SimpleFeatureType)],
158       sampling: Option[(Float, Option[String])],
159       compatibility: Option[FilterCompatibility] = None,
160       priority: Int = DefaultPriority): Option[IteratorSetting] = {
161     if (filter.isEmpty && transform.isEmpty && sampling.isEmpty) { None } else {
162       val is = new IteratorSetting(priority, "filter-transform-iter", classOf[FilterTransformIterator])
163       is.addOption(SftOpt, SimpleFeatureTypes.encodeType(sft, includeUserData = true))
164       if (sft != index.sft) {
165         is.addOption(IndexSftOpt, SimpleFeatureTypes.encodeType(index.sft, includeUserData = true))
166       }
167       is.addOption(IndexOpt, index.identifier)
168       filter.foreach(f => is.addOption(CqlOpt, ECQL.toCQL(f)))
169       transform.foreach { case (tdef, tsft) =>
170         is.addOption(TransformDefsOpt, tdef)
171         is.addOption(TransformSchemaOpt, SimpleFeatureTypes.encodeType(tsft))
172       }
173       sampling.foreach(SamplingIterator.configure(sft, _).foreach { case (k, v) => is.addOption(k, v) })
174 
175       compatibility match {
176         case None =>
177           is.addOption(VersionOpt, GeoMesaProperties.ProjectVersion)
178 
179         case Some(FilterCompatibility.`1.3`) =>
180           is.setIteratorClass("org.locationtech.geomesa.accumulo.iterators.KryoLazyFilterTransformIterator")
181           is.addOption(IndexOpt, s"${index.name}:${index.version}")
182 
183         case Some(c) =>
184           throw new UnsupportedOperationException(s"Unknown compatibility flag: '$c'")
185       }
186       Some(is)
187     }
188   }
