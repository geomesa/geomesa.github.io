38   private val compactionScheduled = new AtomicBoolean(false)
39   private val lastCompaction = new AtomicLong(0L)
40 
41   private val running = new AtomicBoolean(true)
42   private var scheduledCompaction: ScheduledFuture[_] = _

44   private val compactor = new Runnable() {
45     override def run(): Unit = {
46       import org.locationtech.geomesa.accumulo.AccumuloProperties.StatsProperties.STAT_COMPACTION_INTERVAL
47       val compactInterval = STAT_COMPACTION_INTERVAL.toDuration.get.toMillis
48       if (lastCompaction.get < System.currentTimeMillis() - compactInterval &&
49           compactionScheduled.compareAndSet(true, false) ) {
50         compact()
51       }
52       if (running.get) {
53         synchronized{scheduledCompaction = executor.schedule(this, compactInterval, TimeUnit.MILLISECONDS)}
54       }
55     }
56   }
57 
58   compactor.run() // schedule initial compaction

60   override val writer: GeoMesaStatWriter = new AccumuloMetadataStatWriter()
61 
62   override def close(): Unit = {
63     super.close()
64     running.set(false)
65     synchronized{scheduledCompaction.cancel(false)}
66   }

68   /**
69     * Configures the stat combiner to sum stats dynamically.
70     *
71     * Note: should be called with a distributed lock on the stats table
72     *
73     * @param connector accumulo connector
74     * @param sft simple feature type
75     */
76   def configureStatCombiner(connector: AccumuloClient, sft: SimpleFeatureType): Unit = {
77     import MetadataBackedStats._
78 
79     StatsCombiner.configure(sft, connector, metadata.table, metadata.typeNameSeparator.toString)
80 
81     val keys = Seq(CountKey, BoundsKeyPrefix, TopKKeyPrefix, FrequencyKeyPrefix, HistogramKeyPrefix)
82     val splits = new java.util.TreeSet[Text](keys.map(k => new Text(metadata.encodeRow(sft.getTypeName, k))).asJava)
83     connector.tableOperations().addSplits(metadata.table, splits)
84   }

90     *
91     * @param connector accumulo connector
92     * @param sft simple feature type
93     */
94   def removeStatCombiner(connector: AccumuloClient, sft: SimpleFeatureType): Unit =
95     StatsCombiner.remove(sft, connector, metadata.table, metadata.typeNameSeparator.toString)
96 
97   override protected def write(typeName: String, stats: Seq[WritableStat]): Unit = {
98     val (merge, overwrite) = stats.partition(_.merge)
99     metadata.insert(typeName, merge.map(s => s.key -> s.stat).toMap)
100     // invalidate the cache as we would need to reload from accumulo for the combiner to take effect
101     merge.foreach(s => metadata.invalidateCache(typeName, s.key))
102     if (overwrite.nonEmpty) {
103       // due to accumulo issues with combiners, deletes and compactions, we have to:
104       // 1) delete the existing data; 2) compact the table; 3) insert the new value
105       // see: https://issues.apache.org/jira/browse/ACCUMULO-2232
106       metadata.remove(typeName, overwrite.map(_.key))
107       compact()
108       metadata.insert(typeName, overwrite.map(s => s.key -> s.stat).toMap)
109     }
110   }
111 
112   /**
113     * Performs a synchronous compaction of the stats table
114     */
115   private [accumulo] def compact(wait: Boolean = true): Unit = {
116     compactionScheduled.set(false)
117     ds.connector.tableOperations().compact(metadata.table, null, null, true, wait)
118     lastCompaction.set(System.currentTimeMillis())
119   }

121   class AccumuloMetadataStatWriter extends MetadataStatWriter {
122 
123     override def rename(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
124       // the stat combiner should still be configured for the old sft
125       // the call to super() will read the old rows and write new rows, but it doesn't read any
126       // new rows after writing them, so the combiner does not need to be correct yet
127       super.rename(sft, previous)
128       // now remove the old sft and configure the new one
129       removeStatCombiner(ds.connector, previous)
130       configureStatCombiner(ds.connector, sft)
131     }
132 
133     override def updater(sft: SimpleFeatureType): StatUpdater =
134       if (sft.statsEnabled) { new AccumuloStatUpdater(sft) } else { NoopStatUpdater }
135   }
136

138     * Stores stats as metadata entries
139     *
140     * @param sft SimpleFeatureType
141     */
142   class AccumuloStatUpdater(sft: SimpleFeatureType) extends MetadataStatUpdater(sft) {
143     override def close(): Unit = {
144       super.close()
145       // schedule a compaction so our metadata doesn't stack up too much
146       compactionScheduled.set(true)
147     }
148   }
