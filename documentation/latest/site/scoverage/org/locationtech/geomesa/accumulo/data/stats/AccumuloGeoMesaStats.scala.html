39   private val compactionScheduled = new AtomicBoolean(false)
40   private val lastCompaction = new AtomicLong(0L)

42   private val running = new AtomicBoolean(true)
43   private var scheduledCompaction: ScheduledFuture[_] = _

45   private val compactor = new Runnable() {
46     override def run(): Unit = {
47       import org.locationtech.geomesa.accumulo.AccumuloProperties.StatsProperties.STAT_COMPACTION_INTERVAL
48       val compactInterval = STAT_COMPACTION_INTERVAL.toDuration.get.toMillis
49       if (lastCompaction.get < System.currentTimeMillis() - compactInterval &&
50           compactionScheduled.compareAndSet(true, false) ) {
51         compact()
52       }
53       if (running.get) {
54         synchronized{scheduledCompaction = executor.schedule(this, compactInterval, TimeUnit.MILLISECONDS)}
55       }
56     }
57   }

59   compactor.run() // schedule initial compaction

63   override def close(): Unit = {
64     super.close()
65     running.set(false)
66     synchronized{scheduledCompaction.cancel(false)}
67   }

69   /**
70     * Configures the stat combiner to sum stats dynamically.
71     *
72     * Note: should be called with a distributed lock on the stats table
73     *
74     * @param connector accumulo connector
75     * @param sft simple feature type
76     */
77   def configureStatCombiner(connector: AccumuloClient, sft: SimpleFeatureType): Unit = {
78     import MetadataBackedStats._
79 
80     TableUtils.createTableIfNeeded(connector, metadata.table)
81     StatsCombiner.configure(sft, connector, metadata.table, metadata.typeNameSeparator.toString)
82 
83     val keys = Seq(CountKey, BoundsKeyPrefix, TopKKeyPrefix, FrequencyKeyPrefix, HistogramKeyPrefix)
84     val splits = new java.util.TreeSet[Text](keys.map(k => new Text(metadata.encodeRow(sft.getTypeName, k))).asJava)
85     connector.tableOperations().addSplits(metadata.table, splits)
86   }

97     StatsCombiner.remove(sft, connector, metadata.table, metadata.typeNameSeparator.toString)

99   override protected def write(typeName: String, stats: Seq[WritableStat]): Unit = {
100     val (merge, overwrite) = stats.partition(_.merge)
101     metadata.insert(typeName, merge.map(s => s.key -> s.stat).toMap)
102     // invalidate the cache as we would need to reload from accumulo for the combiner to take effect
103     merge.foreach(s => metadata.invalidateCache(typeName, s.key))
104     if (overwrite.nonEmpty) {
105       // due to accumulo issues with combiners, deletes and compactions, we have to:
106       // 1) delete the existing data; 2) compact the table; 3) insert the new value
107       // see: https://issues.apache.org/jira/browse/ACCUMULO-2232
108       metadata.remove(typeName, overwrite.map(_.key))
109       compact()
110       metadata.insert(typeName, overwrite.map(s => s.key -> s.stat).toMap)
111     }
112   }
113 
114   /**
115     * Performs a synchronous compaction of the stats table
116     */
117   private [accumulo] def compact(wait: Boolean = true): Unit = {
118     compactionScheduled.set(false)
119     ds.connector.tableOperations().compact(metadata.table, null, null, true, wait)
120     lastCompaction.set(System.currentTimeMillis())
121   }

123   class AccumuloMetadataStatWriter extends MetadataStatWriter {
124 
125     override def rename(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
126       // the stat combiner should still be configured for the old sft
127       // the call to super() will read the old rows and write new rows, but it doesn't read any
128       // new rows after writing them, so the combiner does not need to be correct yet
129       super.rename(sft, previous)
130       // now remove the old sft and configure the new one
131       removeStatCombiner(ds.connector, previous)
132       configureStatCombiner(ds.connector, sft)
133     }
134 
135     override def updater(sft: SimpleFeatureType): StatUpdater =
136       if (sft.statsEnabled) { new AccumuloStatUpdater(sft) } else { NoopStatUpdater }
137   }
138 
139   /**
140     * Stores stats as metadata entries
141     *
142     * @param sft SimpleFeatureType
143     */
144   class AccumuloStatUpdater(sft: SimpleFeatureType) extends MetadataStatUpdater(sft) {
145     override def close(): Unit = {
146       super.close()
147       // schedule a compaction so our metadata doesn't stack up too much
148       compactionScheduled.set(true)
149     }
150   }
