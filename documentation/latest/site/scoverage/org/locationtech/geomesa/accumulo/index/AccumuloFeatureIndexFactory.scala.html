18 /**
19   * Feature index factory that provides attribute join indices
20   */
21 class AccumuloFeatureIndexFactory extends GeoMesaFeatureIndexFactory {
22 
23   override def indices(sft: SimpleFeatureType, hint: Option[String]): Seq[IndexId] = {
24     def defaults: Seq[IndexId] = JoinIndex.defaults(sft).map(IndexId(JoinIndex.name, JoinIndex.version, _))
25     hint match {
26       case None => defaults
27       case Some(h) =>
28         lazy val Array(name, attributes @ _*) = h.split(":")
29         if (h.equalsIgnoreCase(JoinIndex.name) || h.equalsIgnoreCase(AttributeIndex.name)) {
30           defaults
31         } else if (name.equalsIgnoreCase(JoinIndex.name) && JoinIndex.supports(sft, attributes)) {
32           Seq(IndexId(JoinIndex.name, JoinIndex.version, attributes))
33         } else {
34           Seq.empty
35         }
36     }
37   }
38 
39   override def available(sft: SimpleFeatureType): Seq[(String, Int)] = {
40     if (JoinIndex.defaults(sft).exists(JoinIndex.supports(sft, _))) {
41       Seq((JoinIndex.name, JoinIndex.version))
42     } else {
43       Seq.empty
44     }
45   }
46 
47   override def create[T, U](ds: GeoMesaDataStore[_],
48                             sft: SimpleFeatureType,
49                             index: IndexId): Option[GeoMesaFeatureIndex[T, U]] = {
50     val idx = if (index.name == JoinIndex.name) {
51       val Seq(attribute, secondary @ _*) = index.attributes
52       index.version match {
53         case 8  => Some(new JoinIndex(ds, sft, attribute, secondary, index.mode))
54         case 7  => Some(new JoinIndexV7(ds, sft, attribute, secondary, index.mode))
55         case 6  => Some(new JoinIndexV6(ds, sft, attribute, secondary, index.mode))
56         case 5  => Some(new JoinIndexV5(ds, sft, attribute, secondary, index.mode))
57         case 4  => Some(new JoinIndexV4(ds, sft, attribute, secondary, index.mode))
58         case 3  => Some(new JoinIndexV3(ds, sft, attribute, secondary.headOption, index.mode))
59         case 2  => Some(new JoinIndexV2(ds, sft, attribute, secondary.headOption, index.mode))
60         case _ => None
61       }
62     } else {
63       None
64     }
65     idx.asInstanceOf[Option[GeoMesaFeatureIndex[T, U]]]
66   }
67 }
