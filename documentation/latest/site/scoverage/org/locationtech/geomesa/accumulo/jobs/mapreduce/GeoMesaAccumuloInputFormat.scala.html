48   /**
49     * Gets splits for a job.
50     *
51     * Our delegated AccumuloInputFormat creates a split for each range - because we set a lot of ranges in
52     * geomesa, that creates too many mappers. Instead, we try to group the ranges by tservers. We use the
53     * location assignment of the tablets to tservers to determine the number of splits returned.
54     */
55   override def getSplits(context: JobContext): java.util.List[InputSplit] = {
56     val accumuloSplits = delegate.getSplits(context)
57     // Get the appropriate number of mapper splits using the following priority
58     // 1. Get splits from AccumuloMapperProperties.DESIRED_ABSOLUTE_SPLITS (geomesa.mapreduce.splits.max)
59     // 2. Get splits from #tserver locations * AccumuloMapperProperties.DESIRED_SPLITS_PER_TSERVER (geomesa.mapreduce.splits.tserver.max)
60     // 3. Get splits from AccumuloInputFormat.getSplits(context)
61     def positive(prop: SystemProperty): Option[Int] = {
62       val int = prop.toInt
63       if (int.exists(_ < 1)) {
64         throw new IllegalArgumentException(s"${prop.property} contains an invalid int: ${prop.get}")
65       }
66       int
67     }
68 
69     val grpSplitsMax = positive(AccumuloMapperProperties.DESIRED_ABSOLUTE_SPLITS)
70 
71     lazy val grpSplitsPerTServer = positive(AccumuloMapperProperties.DESIRED_SPLITS_PER_TSERVER).flatMap { perTS =>
72       val numLocations = accumuloSplits.asScala.flatMap(_.getLocations).distinct.length
73       if (numLocations < 1) { None } else { Some(numLocations * perTS) }
74     }
75 
76     grpSplitsMax.orElse(grpSplitsPerTServer) match {
77       case Some(numberOfSplits) =>
78         logger.debug(s"Using desired splits with result of $numberOfSplits splits")
79         val splitSize: Int = math.ceil(accumuloSplits.size().toDouble / numberOfSplits).toInt
80         accumuloSplits.asScala.groupBy(_.getLocations.head).flatMap { case (location, splits) =>
81           splits.grouped(splitSize).map { group =>
82             val split = new GroupedSplit()
83             split.location = location
84             split.splits.append(group.map(f => f.asInstanceOf[RangeInputSplit]).toSeq: _*)
85             split.asInstanceOf[InputSplit]
86           }
87         }.toList.asJava
88 
89       case None =>
90         logger.debug(s"Using default Accumulo Splits with ${accumuloSplits.size} splits")
91         accumuloSplits
92     }
93   }

135   /**
136    * Configure the input format based on a query plan
137    *
138    * @param conf configuration to update
139    * @param params data store parameters
140    * @param plan query plan
141    */
142   def configure(
143       conf: Configuration,
144       params: java.util.Map[String, _],
145       plan: AccumuloQueryPlan,
146       auths: Option[Authorizations]): Unit = {
147     // all accumulo input config methods requires a job
148     // assertion: only the JobConf is updated - to get credentials pass in a JobConf instead of Configuration
149     val job = new Job(conf)
150     job.setInputFormatClass(classOf[GeoMesaAccumuloInputFormat])
151 
152     val props = AccumuloDataStoreFactory.buildAccumuloClientConfig(params)
153     // TODO verify kerberos still works
154     if (ClientProperty.AUTH_TYPE.getValue(props).toLowerCase(Locale.US).contains("kerberos")) {
155       props.put(ClientProperty.SASL_ENABLED.getKey, "true")
156       // // note: for Kerberos, this will create a DelegationToken for us and add it to the Job credentials
157       // AbstractInputFormat.setConnectorInfo(job, user, token)
158     }
159 
160     // use the query plan to set the accumulo input format options
161     require(plan.tables.lengthCompare(1) == 0, s"Can only query from a single table: ${plan.tables.mkString(", ")}")
162 
163     val builder = AccumuloInputFormat.configure().clientProperties(props).table(plan.tables.head).batchScan(true)
164     auths.foreach(builder.auths)
165     if (plan.ranges.nonEmpty) {
166       builder.ranges(plan.ranges.asJava)
167     }
168     plan.columnFamily.foreach { colFamily =>
169       builder.fetchColumns(Collections.singletonList(new Column(colFamily)))
170     }
171     plan.iterators.foreach(builder.addIterator)
172     builder.store(job)
173 
174     // add the configurations back into the original conf
175     conf.addResource(job.getConfiguration)
176 
177     GeoMesaConfigurator.setResultsToFeatures(conf, plan.resultsToFeatures)
178     plan.reducer.foreach(GeoMesaConfigurator.setReducer(conf, _))
179     plan.sort.foreach(GeoMesaConfigurator.setSorting(conf, _))
180     plan.projection.foreach(GeoMesaConfigurator.setProjection(conf, _))
181   }

183   /**
184    * This takes any jars that have been loaded by spark in the context classloader and makes them
185    * available to the general classloader. This is required as not all classes (even spark ones) check
186    * the context classloader.
187    */
188   def ensureSparkClasspath(): Unit = {
189     val sysLoader = ClassLoader.getSystemClassLoader
190     val ccl = Thread.currentThread().getContextClassLoader
191     if (ccl == null || !ccl.getClass.getCanonicalName.startsWith("org.apache.spark.")) {
192       logger.debug("No spark context classloader found")
193     } else if (!ccl.isInstanceOf[URLClassLoader]) {
194       logger.warn(s"Found context classloader, but can't handle type ${ccl.getClass.getCanonicalName}")
195     } else if (!sysLoader.isInstanceOf[URLClassLoader]) {
196       logger.warn(s"Found context classloader, but can't add to type ${sysLoader.getClass.getCanonicalName}")
197     } else {
198       // hack to get around protected visibility of addURL
199       // this might fail if there is a security manager present
200       val addUrl = classOf[URLClassLoader].getDeclaredMethod("addURL", classOf[URL])
201       addUrl.setAccessible(true)
202       val sysUrls = sysLoader.asInstanceOf[URLClassLoader].getURLs.map(_.toString).toSet
203       val (dupeUrls, newUrls) = ccl.asInstanceOf[URLClassLoader].getURLs.filterNot(_.toString.contains("__app__.jar")).partition(url => sysUrls.contains(url.toString))
204       newUrls.foreach(addUrl.invoke(sysLoader, _))
205       logger.debug(s"Loaded ${newUrls.length} urls from context classloader into system classloader " +
206           s"and ignored ${dupeUrls.length} that are already loaded")
207     }
208   }

246   /**
247     * Input split that groups a series of RangeInputSplits. Has to implement Hadoop Writable, thus the vars and
248     * mutable state.
249     */
250   class GroupedSplit extends InputSplit with Writable {
251 
252     // if we're running in spark, we need to load the context classpath before anything else,
253     // otherwise we get classloading and serialization issues
254     if (sys.env.get(GeoMesaAccumuloInputFormat.SYS_PROP_SPARK_LOAD_CP).exists(_.toBoolean)) {
255       GeoMesaAccumuloInputFormat.ensureSparkClasspath()
256     }
257 
258     private [mapreduce] var location: String = _
259     private [mapreduce] val splits: ArrayBuffer[RangeInputSplit] = ArrayBuffer.empty
260 
261     override def getLength: Long = splits.foldLeft(0L)((l: Long, r: RangeInputSplit) => l + r.getLength)
262 
263     override def getLocations: Array[String] = if (location == null) { Array.empty } else { Array(location) }
264 
265     override def write(out: DataOutput): Unit = {
266       out.writeUTF(location)
267       out.writeInt(splits.length)
268       splits.foreach(_.write(out))
269     }
270 
271     override def readFields(in: DataInput): Unit = {
272       location = in.readUTF()
273       splits.clear()
274       var i = 0
275       val size = in.readInt()
276       while (i < size) {
277         val split = new RangeInputSplit()
278         split.readFields(in)
279         splits.append(split)
280         i = i + 1
281       }
282     }
283 
284     override def toString = s"mapreduce.GroupedSplit[$location](${splits.length})"
285   }
