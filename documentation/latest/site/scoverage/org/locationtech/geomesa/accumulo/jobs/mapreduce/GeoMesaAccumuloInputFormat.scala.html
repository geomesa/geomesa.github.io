49   /**
50     * Gets splits for a job.
51     *
52     * Our delegated AccumuloInputFormat creates a split for each range - because we set a lot of ranges in
53     * geomesa, that creates too many mappers. Instead, we try to group the ranges by tservers. We use the
54     * location assignment of the tablets to tservers to determine the number of splits returned.
55     */
56   override def getSplits(context: JobContext): java.util.List[InputSplit] = {
57     val accumuloSplits = delegate.getSplits(context)
58     // Get the appropriate number of mapper splits using the following priority
59     // 1. Get splits from AccumuloMapperProperties.DESIRED_ABSOLUTE_SPLITS (geomesa.mapreduce.splits.max)
60     // 2. Get splits from #tserver locations * AccumuloMapperProperties.DESIRED_SPLITS_PER_TSERVER (geomesa.mapreduce.splits.tserver.max)
61     // 3. Get splits from AccumuloInputFormat.getSplits(context)
62     def positive(prop: SystemProperty): Option[Int] = {
63       val int = prop.toInt
64       if (int.exists(_ < 1)) {
65         throw new IllegalArgumentException(s"${prop.property} contains an invalid int: ${prop.get}")
66       }
67       int
68     }
69 
70     val grpSplitsMax = positive(AccumuloMapperProperties.DESIRED_ABSOLUTE_SPLITS)
71 
72     lazy val grpSplitsPerTServer = positive(AccumuloMapperProperties.DESIRED_SPLITS_PER_TSERVER).flatMap { perTS =>
73       val numLocations = accumuloSplits.asScala.flatMap(_.getLocations).distinct.length
74       if (numLocations < 1) { None } else { Some(numLocations * perTS) }
75     }
76 
77     grpSplitsMax.orElse(grpSplitsPerTServer) match {
78       case Some(numberOfSplits) =>
79         logger.debug(s"Using desired splits with result of $numberOfSplits splits")
80         val splitSize: Int = math.ceil(accumuloSplits.size().toDouble / numberOfSplits).toInt
81         accumuloSplits.asScala.groupBy(_.getLocations.head).flatMap { case (location, splits) =>
82           splits.grouped(splitSize).map { group =>
83             val split = new GroupedSplit()
84             split.location = location
85             split.splits.append(group.map(f => f.asInstanceOf[RangeInputSplit]).toSeq: _*)
86             split.asInstanceOf[InputSplit]
87           }
88         }.toList.asJava
89 
90       case None =>
91         logger.debug(s"Using default Accumulo Splits with ${accumuloSplits.size} splits")
92         accumuloSplits
93     }
94   }

136   /**
137    * Configure the input format based on a query plan
138    *
139    * @param conf configuration to update
140    * @param params data store parameters
141    * @param plan query plan
142    */
143   def configure(
144       conf: Configuration,
145       params: java.util.Map[String, _],
146       plan: AccumuloQueryPlan,
147       auths: Option[Authorizations]): Unit = {
148     // all accumulo input config methods requires a job
149     // assertion: only the JobConf is updated - to get credentials pass in a JobConf instead of Configuration
150     val job = new Job(conf)
151     job.setInputFormatClass(classOf[GeoMesaAccumuloInputFormat])
152 
153     val props = AccumuloDataStoreFactory.buildAccumuloClientConfig(params)
154     // TODO verify kerberos still works
155     if (ClientProperty.AUTH_TYPE.getValue(props).toLowerCase(Locale.US).contains("kerberos")) {
156       props.put(ClientProperty.SASL_ENABLED.getKey, "true")
157       // // note: for Kerberos, this will create a DelegationToken for us and add it to the Job credentials
158       // AbstractInputFormat.setConnectorInfo(job, user, token)
159     }
160 
161     // use the query plan to set the accumulo input format options
162     require(plan.tables.lengthCompare(1) == 0, s"Can only query from a single table: ${plan.tables.mkString(", ")}")
163 
164     val builder = AccumuloInputFormat.configure().clientProperties(props).table(plan.tables.head).batchScan(true)
165     auths.foreach(builder.auths)
166     if (plan.ranges.nonEmpty) {
167       builder.ranges(plan.ranges.asJava)
168     }
169     plan.columnFamily.foreach { colFamily =>
170       builder.fetchColumns(Collections.singletonList(new Column(colFamily)))
171     }
172     plan.iterators.foreach(builder.addIterator)
173     builder.store(job)
174 
175     // add the configurations back into the original conf
176     conf.addResource(job.getConfiguration)
177 
178     GeoMesaConfigurator.setResultsToFeatures(conf, plan.resultsToFeatures)
179     plan.reducer.foreach(GeoMesaConfigurator.setReducer(conf, _))
180     plan.sort.foreach(GeoMesaConfigurator.setSorting(conf, _))
181     plan.projection.foreach(GeoMesaConfigurator.setProjection(conf, _))
182   }

184   /**
185    * This takes any jars that have been loaded by spark in the context classloader and makes them
186    * available to the general classloader. This is required as not all classes (even spark ones) check
187    * the context classloader.
188    */
189   def ensureSparkClasspath(): Unit = {
190     val sysLoader = ClassLoader.getSystemClassLoader
191     val ccl = Thread.currentThread().getContextClassLoader
192     if (ccl == null || !ccl.getClass.getCanonicalName.startsWith("org.apache.spark.")) {
193       logger.debug("No spark context classloader found")
194     } else if (!ccl.isInstanceOf[URLClassLoader]) {
195       logger.warn(s"Found context classloader, but can't handle type ${ccl.getClass.getCanonicalName}")
196     } else if (!sysLoader.isInstanceOf[URLClassLoader]) {
197       logger.warn(s"Found context classloader, but can't add to type ${sysLoader.getClass.getCanonicalName}")
198     } else {
199       // hack to get around protected visibility of addURL
200       // this might fail if there is a security manager present
201       val addUrl = classOf[URLClassLoader].getDeclaredMethod("addURL", classOf[URL])
202       addUrl.setAccessible(true)
203       val sysUrls = sysLoader.asInstanceOf[URLClassLoader].getURLs.map(_.toString).toSet
204       val (dupeUrls, newUrls) = ccl.asInstanceOf[URLClassLoader].getURLs.filterNot(_.toString.contains("__app__.jar")).partition(url => sysUrls.contains(url.toString))
205       newUrls.foreach(addUrl.invoke(sysLoader, _))
206       logger.debug(s"Loaded ${newUrls.length} urls from context classloader into system classloader " +
207           s"and ignored ${dupeUrls.length} that are already loaded")
208     }
209   }

247   /**
248     * Input split that groups a series of RangeInputSplits. Has to implement Hadoop Writable, thus the vars and
249     * mutable state.
250     */
251   class GroupedSplit extends InputSplit with Writable {
252 
253     // if we're running in spark, we need to load the context classpath before anything else,
254     // otherwise we get classloading and serialization issues
255     if (sys.env.get(GeoMesaAccumuloInputFormat.SYS_PROP_SPARK_LOAD_CP).exists(_.toBoolean)) {
256       GeoMesaAccumuloInputFormat.ensureSparkClasspath()
257     }
258 
259     private [mapreduce] var location: String = _
260     private [mapreduce] val splits: ArrayBuffer[RangeInputSplit] = ArrayBuffer.empty
261 
262     override def getLength: Long = splits.foldLeft(0L)((l: Long, r: RangeInputSplit) => l + r.getLength)
263 
264     override def getLocations: Array[String] = if (location == null) { Array.empty } else { Array(location) }
265 
266     override def write(out: DataOutput): Unit = {
267       out.writeUTF(location)
268       out.writeInt(splits.length)
269       splits.foreach(_.write(out))
270     }
271 
272     override def readFields(in: DataInput): Unit = {
273       location = in.readUTF()
274       splits.clear()
275       var i = 0
276       val size = in.readInt()
277       while (i < size) {
278         val split = new RangeInputSplit()
279         split.readFields(in)
280         splits.append(split)
281         i = i + 1
282       }
283     }
284 
285     override def toString = s"mapreduce.GroupedSplit[$location](${splits.length})"
286   }
