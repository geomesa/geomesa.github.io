147     lazy private val exportPath: Path = {

149       // makeQualified is a no-op if the path was already qualified (has a defined scheme and is not a relative path)

153     lazy private val exportFs: FileSystem = exportPath.getFileSystem(to.conf)

155     // sft should be shareable/the same from both datastores
156     lazy private val sft: SimpleFeatureType = from.ds.getSchema(params.featureName)

158     lazy private val partitions: Seq[String] = {
159       val builder = Seq.newBuilder[String]
160       if (params.partitions != null) {
161         builder ++= params.partitions.asScala
162       }
163       if (params.partitionValues != null && !params.partitionValues.isEmpty) {
164         val partitioning = TablePartition(from.ds, sft).get
165         val sf = new ScalaSimpleFeature(sft, "")
166         builder ++= params.partitionValues.asScala.map { value =>
167           sf.setAttribute(sft.getDtgIndex.get, value)
168           val partition = partitioning.partition(sf)
169           logger.debug(s"Generated partition $partition from value $value")
170           partition
171         }
172       }
173       builder.result.distinct.sorted
174     }
175 
176     lazy private val indices = {
177       val all = from.ds.manager.indices(sft)
178       if (params.indices == null || params.indices.isEmpty) { all } else {
179         val builder = Seq.newBuilder[GeoMesaFeatureIndex[_, _]]
180         params.indices.asScala.foreach { ident =>
181           val filtered = all.filter(_.identifier.contains(ident))
182           if (filtered.isEmpty) {
183             throw new ParameterException(
184               s"Index '$ident' does not exist in the schema. Available indices: ${all.map(_.identifier).mkString(", ")}")
185           }
186           logger.debug(s"Mapped identifier $ident to ${filtered.map(_.identifier).mkString(", ")}")
187           builder ++= filtered
188         }
189         builder.result.distinct
190       }
191     }

193     override def run(): Unit = {
194       // validate our params/setup
195       if (exportPath.toUri.getScheme == "file") {
196         throw new RuntimeException("Could not read defaultFS - this may be caused by a missing Hadoop core-site.xml file")
197       }
198       if (sft == null) {
199         throw new ParameterException(s"Schema '${params.featureName}' does not exist in the source store")
200       } else if (!sft.isPartitioned) {
201         throw new ParameterException(s"Schema '${params.featureName}' is not partitioned")
202       } else {
203         val toSft = to.ds.getSchema(params.featureName)
204         if (toSft == null) {
205           throw new ParameterException(s"Schema '${params.featureName}' does not exist in the destination store")
206         } else if (!toSft.isPartitioned) {
207           throw new ParameterException(s"Schema '${params.featureName}' is not partitioned in the destination store")
208         } else if (SimpleFeatureTypes.compare(sft, toSft) != 0) {
209           throw new ParameterException(s"Schema '${params.featureName}' is not the same in the source and destination store")
210         }
211       }
212       if (partitions.isEmpty) {
213         throw new ParameterException("At least one of --partition or --partition-value is required")
214       }
215 
216       // now execute the copy
217       CachedThreadPool.executor(params.tableThreads) { executor =>
218         partitions.foreach { partition =>
219           indices.map { fromIndex =>
220             val toIndex = to.ds.manager.index(sft, fromIndex.identifier)
221             val runnable: Runnable = () => try { copy(fromIndex, toIndex, partition) } catch {
222               // catch Throwable so NoClassDefFound still gets logged
223               case e: Throwable => logger.error(s"Error copying partition $partition ${fromIndex.identifier}", e)
224             }
225             executor.submit(runnable)
226           }
227         }
228       }
229     }

231     private def copy(fromIndex: GeoMesaFeatureIndex[_, _], toIndex: GeoMesaFeatureIndex[_, _], partition: String): Unit = {
232       Command.user.info(s"Copying partition $partition ${fromIndex.identifier}")
233 
234       val fromTable = fromIndex.getTableNames(Some(partition)).headOption.orNull
235       if (fromTable == null) {
236         Command.user.warn(s"Ignoring non-existent partition ${fromIndex.identifier} $partition")
237         return
238       }
239 
240       val tableExportPath = new Path(exportPath, fromTable)
241       logger.debug(s"Source table $fromTable (${from.tableOps.tableIdMap().get(fromTable)})")
242       logger.debug(s"Export path $tableExportPath")
243       if (exportFs.exists(tableExportPath)) {
244         logger.debug(s"Deleting existing export directory $tableExportPath")
245         exportFs.delete(tableExportPath, true)
246       }
247       // clone the table as we have to take it offline in order to export it
248       // note that cloning is just a metadata op as it shares the underlying data files (until they change)
249       val cloneTable = s"${fromTable}_tmp"
250       logger.debug(s"Checking for existence and deleting any existing cloned table $cloneTable")
251       from.ds.adapter.deleteTable(cloneTable) // no-op if table doesn't exist
252       logger.debug(s"Cloning $fromTable to $cloneTable")
253       from.tableOps.clone(fromTable, cloneTable, false, Collections.emptyMap(), Collections.emptySet()) // use 2.0 method for compatibility
254       logger.debug(s"Taking offline $cloneTable")
255       from.tableOps.offline(cloneTable, true)
256       logger.debug(s"Exporting table to $tableExportPath")
257       from.tableOps.exportTable(cloneTable, tableExportPath.toString)
258 
259       val distcpPath = new Path(tableExportPath, "distcp.txt")
260       if (!exportFs.exists(distcpPath)) {
261         throw new RuntimeException(s"Could not read table export results: $distcpPath")
262       }
263 
264       // ensures the destination table exists
265       logger.debug(s"Checking destination table for $fromTable")
266       to.ds.adapter.createTable(toIndex, Some(partition), Seq.empty)
267       val toTable = toIndex.getTableNames(Some(partition)).headOption.getOrElse {
268         throw new RuntimeException(s"Could not get destination table for index ${fromIndex.identifier} and partition $partition")
269       }
270       logger.debug(s"Destination table $toTable (${to.tableOps.tableIdMap().get(toTable)})")
271       // create splits, do this separately in case the table already exists
272       val splits = new java.util.TreeSet(from.tableOps.listSplits(cloneTable))
273       val existingSplits = to.tableOps.listSplits(toTable)
274       splits.removeAll(existingSplits)
275       if (!splits.isEmpty) {
276         if (!existingSplits.isEmpty) {
277           val warning = s"Detected split mismatch between source ($fromTable) and destination ($toTable)"
278           Command.user.warn(warning)
279           logger.warn(warning)
280         }
281         logger.debug(s"Adding splits to destination table $toTable")
282         to.tableOps.addSplits(toTable, splits)
283       }
284 
285       val dirs = Collections.newSetFromMap(new ConcurrentHashMap[String, java.lang.Boolean]())
286       val copyToDir = new Path(tableExportPath, "files")
287       if (params.distCp) {
288         new DistributedCopy(from.conf).copy(distcpPath, copyToDir, TerminalCallback()) match {
289           case JobSuccess(message, counts) =>
290             dirs.add(copyToDir.toString)
291             Command.user.info(message)
292             logger.debug(s"Distributed copy counters: ${counts.mkString("\n  ", "\n  ", "")}")
293 
294           case JobFailure(message) =>
295             Command.user.error(message)
296             logger.error(message)
297         }
298       } else {
299         logger.debug(s"Reading $distcpPath")
300         WithClose(IOUtils.lineIterator(exportFs.open(distcpPath), StandardCharsets.UTF_8)) { files =>
301           CachedThreadPool.executor(params.fileThreads) { executor =>
302             files.asScala.foreach { file =>
303               val runnable: Runnable = () => {
304                 val path = new Path(file)
305                 val copy = new Path(copyToDir, path.getName)
306                 try {
307                   logger.debug(s"Copying $path to $copy")
308                   val fs = path.getFileSystem(from.conf)
309                   if (FileUtil.copy(fs, path, exportFs, copy, false, false, to.conf)) {
310                     dirs.add(copyToDir.toString)
311                   } else {
312                     Command.user.error(s"Failed to copy $path to $copy")
313                     logger.error(s"Failed to copy $path to $copy")
314                   }
315                 } catch {
316                   // catch Throwable so NoClassDefFound still gets logged
317                   case e: Throwable =>
318                     Command.user.error(s"Failed to copy $path to $copy")
319                     logger.error(s"Failed to copy $path to $copy", e)
320                 }
321               }
322               executor.submit(runnable)
323             }
324           }
325         }
326       }
327 
328       dirs.asScala.toSeq.sorted.foreach { dir =>
329         logger.debug(s"Loading rfiles from $dir to $toTable")
330         val importDir = to.tableOps.importDirectory(dir).to(toTable)
331         try { importDir.ignoreEmptyDir(true) } catch {
332           case _: NoSuchMethodError => // accumulo 2.0, ignore
333         }
334         try { importDir.load() } catch {
335           case e: IllegalArgumentException => logger.trace("Error importing directory:", e) // should mean empty dir
336         }
337       }
338 
339       // cleanup
340       logger.debug(s"Deleting clone table $cloneTable")
341       from.tableOps.delete(cloneTable)
342       logger.debug(s"Deleting export path $tableExportPath")
343       exportFs.delete(tableExportPath, true)
344 
345       Command.user.info(s"Bulk copy complete for partition $partition ${fromIndex.identifier}")
346     }

349       CloseWithLogging(tryFrom.toOption)
350       CloseWithLogging(tryTo.toOption)
351     }

394         s"Unable to load datastore using provided values: ${params.map { case (k, v) => s"$k=$v" }.mkString(", ")}")
395     }
