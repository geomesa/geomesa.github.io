22 /**
23  * Copy a partitioned table out of one feature type and into an identical feature type
24  */
25 class AccumuloBulkCopyCommand extends Command with StrictLogging {
26 
27   import scala.collection.JavaConverters._
28 
29   override val name = "bulk-copy"
30   override val params = new AccumuloBulkCopyParams()
31 
32   override def execute(): Unit = {
33     val fromCluster = {
34       val auth = if (params.fromKeytab != null) { ClusterKeytab(params.fromKeytab) } else { ClusterPassword(params.fromPassword) }
35       val configs = if (params.fromConfigs == null) { Seq.empty } else { params.fromConfigs.asScala.toSeq }
36       ClusterConfig(params.fromInstance, params.fromZookeepers, params.fromUser, auth, params.fromCatalog, configs)
37     }
38     val toCluster = {
39       val auth = if (params.toKeytab != null) { ClusterKeytab(params.toKeytab) } else { ClusterPassword(params.toPassword) }
40       val configs = if (params.toConfigs == null) { Seq.empty } else { params.toConfigs.asScala.toSeq }
41       ClusterConfig(params.toInstance, params.toZookeepers, params.toUser, auth, params.toCatalog, configs)
42     }
43     val indices = if (params.indices == null) { Seq.empty } else { params.indices.asScala.toSeq }
44     val partitions: Seq[PartitionId] =
45       Option(params.partitions).fold(Seq.empty[PartitionId])(_.asScala.map(PartitionName.apply).toSeq) ++
46         Option(params.partitionValues).fold(Seq.empty[PartitionId])(_.asScala.map(PartitionValue.apply).toSeq)
47     val opts = CopyOptions(params.tableThreads, params.fileThreads, params.distCp)
48 
49     WithClose(new SchemaCopier(fromCluster, toCluster, params.featureName, params.exportPath, indices, partitions, opts)) { copier =>
50       try {
51         copier.call(params.resume)
52       } catch {
53         case e: IllegalArgumentException => throw new ParameterException(e.getMessage)
54       }
55     }
56 
57   }
58 }
