26 class AccumuloAgeOffCommand extends AccumuloDataStoreCommand {
27 
28   import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
29 
30   override val name = "configure-age-off"
31   override val params = new AccumuloAgeOffParams()
32 
33   override def execute(): Unit = {
34     if (Seq(params.set, params.remove, params.list).count(_ == true) != 1) {
35       throw new ParameterException("Must specify exactly one of 'list', 'set' or 'remove'")
36     } else if (params.set && params.expiry == null) {
37       throw new ParameterException("Must specify 'expiry' when setting age-off")
38     }
39     withDataStore(exec)
40   }
41 
42   private def exec(ds: AccumuloDataStore): Unit = {
43     val sft = ds.getSchema(params.featureName)
44     if (sft == null) {
45       throw new ParameterException(s"SimpleFeatureType '${params.featureName}' does not exist in the data store")
46     }
47     lazy val mutable = SimpleFeatureTypes.mutable(sft)
48     if (params.list) {
49       val tableOps = ds.client.tableOperations()
50       val tables = ds.getAllIndexTableNames(sft.getTypeName).filter(tableOps.exists)
51       val dtgAgeOff = tables.foldLeft[Option[IteratorSetting]](None) { (res, table) =>
52         res.orElse(DtgAgeOffIterator.list(tableOps, table))
53       }
54       val ageOff = tables.foldLeft[Option[IteratorSetting]](None) { (res, table) =>
55         res.orElse(AgeOffIterator.list(tableOps, table))
56       }
57       Command.user.info(s"Attribute age-off: ${dtgAgeOff.getOrElse("None")}")
58       Command.user.info(s"Timestamp age-off: ${ageOff.getOrElse("None")}")
59     } else if (params.set && params.dtgField == null) {
60       if (Prompt.confirm(s"Configuring ingest-time-based age-off for schema '${params.featureName}' " +
61           s"with expiry ${params.expiry}. Continue (y/n)? ")) {
62         mutable.setFeatureExpiration(IngestTimeExpiration(params.expiry))
63         ds.updateSchema(sft.getTypeName, mutable)
64       }
65     } else if (params.set) {
66       if (Prompt.confirm(s"Configuring attribute-based age-off for schema '${params.featureName}' " +
67           s"on field '${params.dtgField}' with expiry ${params.expiry}. Continue (y/n)? ")) {
68         val expiry = FeatureTimeExpiration(params.dtgField, sft.indexOf(params.dtgField), params.expiry)
69         mutable.setFeatureExpiration(expiry)
70         ds.updateSchema(sft.getTypeName, mutable)
71       }
72     } else if (Prompt.confirm(s"Removing age-off for schema '${params.featureName}'. Continue (y/n)? ")) {
73       // clear the iterator configs if expiration wasn't configured in the schema,
74       // as we can't detect that in updateSchema
75       if (mutable.getUserData.remove(Configs.FeatureExpiration) == null) {
76         val tableOps = ds.client.tableOperations()
77         ds.getAllIndexTableNames(sft.getTypeName).filter(tableOps.exists).foreach { table =>
78           AgeOffIterator.clear(tableOps, table)
79           DtgAgeOffIterator.clear(tableOps, table)
80         }
81       } else {
82         ds.updateSchema(sft.getTypeName, mutable)
83       }
84     }
85   }
86 }
