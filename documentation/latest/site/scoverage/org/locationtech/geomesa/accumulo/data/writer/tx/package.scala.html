57   object MutationBuilder {
58 
59     case class AppendBuilder(row: Array[Byte], kvs: Seq[MutationValue]) extends MutationBuilder {
60       override def name: String = "insert"
61       override def apply(): ConditionalMutation = {
62         val mutation = new ConditionalMutation(row)
63         kvs.foreach { kv =>
64           mutation.addCondition(new Condition(kv.cf, kv.cq)) // requires cf+cq to not exist
65           mutation.put(kv.cf, kv.cq, kv.vis, kv.value)
66         }
67         mutation
68       }
69       override def invert(): ConditionalMutation = DeleteBuilder(row, kvs)()
70     }
71 
72     case class DeleteBuilder(row: Array[Byte], kvs: Seq[MutationValue]) extends MutationBuilder {
73       override def name: String = "delete"
74       override def apply(): ConditionalMutation = {
75         val mutation = new ConditionalMutation(row)
76         kvs.foreach { kv =>
77           mutation.addCondition(new Condition(kv.cf, kv.cq).setVisibility(kv.vis).setValue(kv.value))
78           mutation.putDelete(kv.cf, kv.cq, kv.vis)
79         }
80         mutation
81       }
82       override def invert(): ConditionalMutation = AppendBuilder(row, kvs)()
83     }
84 
85     case class UpdateBuilder(row: Array[Byte], kvs: Seq[MutationValue], previous: Seq[MutationValue])
86         extends MutationBuilder {
87       override def name: String = "update"
88       override def apply(): ConditionalMutation = {
89         val mutation = new ConditionalMutation(row)
90         previous.foreach(kv => mutation.addCondition(new Condition(kv.cf, kv.cq).setVisibility(kv.vis).setValue(kv.value)))
91         kvs.foreach(kv => mutation.put(kv.cf, kv.cq, kv.vis, kv.value))
92         mutation
93       }
94       override def invert(): ConditionalMutation = UpdateBuilder(row, previous, kvs)()
95     }
96   }
97 
98   /**
99    * Holder for a mutation's values - i.e. the non-row parts of a key plus the value
100    *
101    * @param cf column family
102    * @param cq column qualifier
103    * @param vis visibility label
104    * @param value value
105    */
106   case class MutationValue(cf: Array[Byte], cq: Array[Byte], vis: ColumnVisibility, value: Array[Byte]) {
107     def equalKey(other: MutationValue): Boolean = {
108       java.util.Arrays.equals(cf, other.cf) &&
109           java.util.Arrays.equals(cq, other.cq) &&
110           java.util.Arrays.equals(vis.getExpression, other.vis.getExpression)
111     }
112     override def toString: String =
113       s"cf: ${new String(cf, StandardCharsets.UTF_8)}, cq: ${new String(cq, StandardCharsets.UTF_8)}, " +
114           s"vis: $vis, value: ${ByteArrays.toHex(value)}"
115   }
