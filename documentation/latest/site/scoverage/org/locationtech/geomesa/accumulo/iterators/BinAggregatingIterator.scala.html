45   /**
46    * Configure based on query hints
47    */
48   def configure(sft: SimpleFeatureType,
49                 index: GeoMesaFeatureIndex[_, _],
50                 filter: Option[Filter],
51                 hints: Hints,
52                 priority: Int = DEFAULT_PRIORITY): IteratorSetting = {
53     val trackId = hints.getBinTrackIdField
54     val geom = hints.getBinGeomField.getOrElse(sft.getGeomField)
55     val dtg = hints.getBinDtgField.orElse(sft.getDtgField)
56     val label = hints.getBinLabelField
57     val batchSize = hints.getBinBatchSize
58     val sort = hints.isBinSorting
59     val sampling = hints.getSampling
60 
61     val is = new IteratorSetting(priority, "bin-iter", classOf[BinAggregatingIterator])
62     BinAggregatingScan.configure(sft, index, filter, trackId, geom, dtg, label, batchSize, sort, sampling).foreach {
63       case (k, v) => is.addOption(k, v)
64     }
65     is
66   }

68   /**
69    * Fallback for when we can't use the aggregating iterator (for example, if the features are avro encoded).
70    * Instead, do bin conversion in client.
71    *
72    * Only encodes one bin (or one bin line) per feature
73    */
74   def nonAggregatedKvsToFeatures(
75       sft: SimpleFeatureType,
76       index: GeoMesaFeatureIndex[_, _],
77       hints: Hints,
78       serializationType: SerializationType): Entry[Key, Value] => SimpleFeature = {
79 
80     // don't use return sft from query hints, as it will be bin_sft
81     val returnSft = hints.getTransformSchema.getOrElse(sft)
82 
83     val trackId = Option(hints.getBinTrackIdField).filter(_ != "id").map(returnSft.indexOf)
84     val geom = hints.getBinGeomField.map(returnSft.indexOf)
85     val dtg = hints.getBinDtgField.map(returnSft.indexOf)
86     val label = hints.getBinLabelField.map(returnSft.indexOf)
87 
88     val encoder = BinaryOutputEncoder(returnSft, EncodingOptions(geom, dtg, trackId, label))
89 
90     // noinspection ScalaDeprecation
91     if (index.serializedWithId) {
92       val deserializer = serializationType match {
93         case SerializationType.KRYO => KryoFeatureSerializer(returnSft)
94         case SerializationType.AVRO => new AvroFeatureSerializer(returnSft)
95       }
96       e: Entry[Key, Value] => {
97         val deserialized = deserializer.deserialize(e.getValue.get())
98         val values = Array[AnyRef](encoder.encode(deserialized), GeometryUtils.zeroPoint)
99         new ScalaSimpleFeature(BinEncodedSft, deserialized.getID, values)
100       }
101     } else {
102       val deserializer = serializationType match {
103         case SerializationType.KRYO => KryoFeatureSerializer(returnSft, SerializationOptions.withoutId)
104         case SerializationType.AVRO => new AvroFeatureSerializer(returnSft, SerializationOptions.withoutId)
105       }
106       e: Entry[Key, Value] => {
107         val deserialized = deserializer.deserialize(e.getValue.get())
108         val row = e.getKey.getRow
109         val id = index.getIdFromRow(row.getBytes, 0, row.getLength, deserialized)
110         deserialized.getIdentifier.asInstanceOf[FeatureIdImpl].setID(id)
111         val values = Array[AnyRef](encoder.encode(deserialized), GeometryUtils.zeroPoint)
112         new ScalaSimpleFeature(BinEncodedSft, deserialized.getID, values)
113       }
114     }
115   }
116

122   }
