33   val DEFAULT_PRIORITY = 25

35   /**
36    * Configure based on query hints
37    */
38   def configure(sft: SimpleFeatureType,
39                 index: GeoMesaFeatureIndex[_, _],
40                 filter: Option[Filter],
41                 hints: Hints,
42                 priority: Int = DEFAULT_PRIORITY): IteratorSetting = {
43     val trackId = hints.getBinTrackIdField
44     val geom = hints.getBinGeomField.getOrElse(sft.getGeomField)
45     val dtg = hints.getBinDtgField.orElse(sft.getDtgField)
46     val label = hints.getBinLabelField
47     val batchSize = hints.getBinBatchSize
48     val sort = hints.isBinSorting
49     val sampling = hints.getSampling
50 
51     val is = new IteratorSetting(priority, "bin-iter", classOf[BinAggregatingIterator])
52     BinAggregatingScan.configure(sft, index, filter, trackId, geom, dtg, label, batchSize, sort, sampling).foreach {
53       case (k, v) => is.addOption(k, v)
54     }
55     is
56   }
57

60     */
61   class AccumuloBinResultsToFeatures extends BinResultsToFeatures[Entry[Key, Value]] {

63   }
