44   /**
45    * Configure based on query hints
46    */
47   def configure(sft: SimpleFeatureType,
48                 index: GeoMesaFeatureIndex[_, _],
49                 filter: Option[Filter],
50                 hints: Hints,
51                 priority: Int = DEFAULT_PRIORITY): IteratorSetting = {
52     val trackId = hints.getBinTrackIdField
53     val geom = hints.getBinGeomField.getOrElse(sft.getGeomField)
54     val dtg = hints.getBinDtgField.orElse(sft.getDtgField)
55     val label = hints.getBinLabelField
56     val batchSize = hints.getBinBatchSize
57     val sort = hints.isBinSorting
58     val sampling = hints.getSampling
59 
60     val is = new IteratorSetting(priority, "bin-iter", classOf[BinAggregatingIterator])
61     BinAggregatingScan.configure(sft, index, filter, trackId, geom, dtg, label, batchSize, sort, sampling).foreach {
62       case (k, v) => is.addOption(k, v)
63     }
64     is
65   }

67   /**
68    * Fallback for when we can't use the aggregating iterator (for example, if the features are avro encoded).
69    * Instead, do bin conversion in client.
70    *
71    * Only encodes one bin (or one bin line) per feature
72    */
73   def nonAggregatedKvsToFeatures(
74       sft: SimpleFeatureType,
75       index: GeoMesaFeatureIndex[_, _],
76       hints: Hints,
77       serializationType: SerializationType): Entry[Key, Value] => SimpleFeature = {
78 
79     // don't use return sft from query hints, as it will be bin_sft
80     val returnSft = hints.getTransformSchema.getOrElse(sft)
81 
82     val trackId = Option(hints.getBinTrackIdField).filter(_ != "id").map(returnSft.indexOf)
83     val geom = hints.getBinGeomField.map(returnSft.indexOf)
84     val dtg = hints.getBinDtgField.map(returnSft.indexOf)
85     val label = hints.getBinLabelField.map(returnSft.indexOf)
86 
87     val encoder = BinaryOutputEncoder(returnSft, EncodingOptions(geom, dtg, trackId, label))
88 
89     // noinspection ScalaDeprecation
90     if (index.serializedWithId) {
91       val deserializer = serializationType match {
92         case SerializationType.KRYO => KryoFeatureSerializer(returnSft)
93         case SerializationType.AVRO => new AvroFeatureSerializer(returnSft)
94       }
95       e: Entry[Key, Value] => {
96         val deserialized = deserializer.deserialize(e.getValue.get())
97         val values = Array[AnyRef](encoder.encode(deserialized), GeometryUtils.zeroPoint)
98         new ScalaSimpleFeature(BinEncodedSft, deserialized.getID, values)
99       }
100     } else {
101       val deserializer = serializationType match {
102         case SerializationType.KRYO => KryoFeatureSerializer(returnSft, SerializationOption.WithoutId)
103         case SerializationType.AVRO => new AvroFeatureSerializer(returnSft, SerializationOption.WithoutId)
104       }
105       e: Entry[Key, Value] => {
106         val deserialized = deserializer.deserialize(e.getValue.get())
107         val row = e.getKey.getRow
108         val id = index.getIdFromRow(row.getBytes, 0, row.getLength, deserialized)
109         deserialized.getIdentifier.asInstanceOf[FeatureIdImpl].setID(id)
110         val values = Array[AnyRef](encoder.encode(deserialized), GeometryUtils.zeroPoint)
111         new ScalaSimpleFeature(BinEncodedSft, deserialized.getID, values)
112       }
113     }
114   }

121   }
