21 /**
22  * Runs a join scan against two tables
23  *
24  * @param in input scan
25  * @param join join scan
26  * @param numThreads threads
27  * @param batchSize batch size
28  */
29 class BatchMultiScanner(
30     in: ScannerBase,
31     join: Seq[Entry[Key, Value]] => CloseableIterator[Entry[Key, Value]],

42   private val executor = Executors.newFixedThreadPool(numThreads)
43 
44   private val inQ  = new LinkedBlockingQueue[Entry[Key, Value]](batchSize)
45   private val outQ = new LinkedBlockingQueue[Entry[Key, Value]](batchSize)
46 
47   private val inDone  = new AtomicBoolean(false)
48   private val outDone = new AtomicBoolean(false)
49 
50   private var prefetch: Entry[Key, Value] = _
51 
52   private def prefetchIfNull(): Unit = {
53     // loop while we might have another and we haven't set prefetch
54     while (prefetch == null && (!outDone.get || outQ.size > 0)) {
55       prefetch = outQ.poll(5, TimeUnit.MILLISECONDS)
56     }
57   }
58 
59   // must attempt a prefetch since we don't know whether or not the outQ
60   // will actually be filled with an item (filters may not match and the
61   // in scanner may never return a range)
62   override def hasNext(): Boolean = {
63     prefetchIfNull()
64     prefetch != null
65   }
66 
67   override def next(): Entry[Key, Value] = {
68     prefetchIfNull()
69     val ret = prefetch
70     prefetch = null
71     ret
72   }

75     override def run(): Unit = {
76       try {

78       } finally {

84   executor.submit(new Runnable {
85     override def run(): Unit = {
86       try {
87         val tasks = ListBuffer.empty[Future[_]]
88         while (!inDone.get || inQ.size() > 0) {
89           val entry = inQ.poll(5, TimeUnit.MILLISECONDS)
90           if (entry != null) {
91             val entries = ListBuffer(entry)
92             inQ.drainTo(entries.asJava)
93             val task = executor.submit(new Runnable {
94               override def run(): Unit = {
95                 val iterator = join(entries.toSeq)
96                 try {
97                   iterator.foreach(outQ.put)
98                 } finally {
99                   iterator.close()
100                 }
101               }
102             })
103             tasks.append(task)
104           }
105         }
106         tasks.foreach(_.get)
107       } catch {
108         case _: InterruptedException =>
109       } finally {
110         executor.shutdown()
111         outDone.set(true)
112       }
113     }
114   })
115 
116   override def close(): Unit = {
117     if (!executor.isShutdown) {
118       executor.shutdownNow()
119     }
120     in.close()
121   }
