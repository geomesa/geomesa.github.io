60   override def createTable(
61       index: GeoMesaFeatureIndex[_, _],
62       partition: Option[String],
63       splits: => Seq[Array[Byte]]): Unit = {
64     val table = index.configureTableName(partition) // writes table name to metadata
65     // create table if it doesn't exist
66     val created = ensureTableExists(table, index.sft.isLogicalTime)
67 
68     def addSplitsAndGroups(): Unit = {
69       // create splits
70       val splitsToAdd = splits.map(new Text(_)).toSet -- tableOps.listSplits(table).asScala.toSet
71       if (splitsToAdd.nonEmpty) {
72         tableOps.addSplits(table, new java.util.TreeSet(splitsToAdd.asJava))
73       }
74 
75       // create locality groups
76       val existingGroups = tableOps.getLocalityGroups(table)
77       val localityGroups = new java.util.HashMap[String, java.util.Set[Text]](existingGroups)
78 
79       def addGroup(cf: Text): Unit = {
80         val key = cf.toString
81         if (localityGroups.containsKey(key)) {
82           val update = new java.util.HashSet[Text](localityGroups.get(key))
83           update.add(cf)
84           localityGroups.put(key, update)
85         } else {
86           localityGroups.put(key, Collections.singleton(cf))
87         }
88       }
89 
90       groups.apply(index.sft).foreach { case (k, _) => addGroup(new Text(k)) }
91 
92       if (localityGroups != existingGroups) {
93         tableOps.setLocalityGroups(table, localityGroups)
94       }
95     }
96 
97     if (created) {
98       addSplitsAndGroups()
99 
100       // enable block cache
101       tableOps.setProperty(table, Property.TABLE_BLOCKCACHE_ENABLED.getKey, "true")
102 
103       if (index.name == IdIndex.name) {
104         // enable the row functor as the feature ID is stored in the Row ID
105         tableOps.setProperty(table, Property.TABLE_BLOOM_KEY_FUNCTOR.getKey, classOf[RowFunctor].getCanonicalName)
106         tableOps.setProperty(table, Property.TABLE_BLOOM_ENABLED.getKey, "true")
107       }
108 
109       if (index.sft.isVisibilityRequired) {
110         VisibilityIterator.set(tableOps, table)
111       }
112     } else if (index.keySpace.sharing.nonEmpty) {
113       // even if the table existed, we still need to check the splits and locality groups if it's shared
114       addSplitsAndGroups()
115     }
116   }

135   override def createQueryPlan(strategy: QueryStrategy): AccumuloQueryPlan = {
136     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
137 
138     val QueryStrategy(filter, byteRanges, _, _, ecql, hints, _) = strategy
139     val index = filter.index
140     // index api defines empty start/end for open-ended range - in accumulo, it's indicated with null
141     // index api defines start row inclusive, end row exclusive
142     val ranges = byteRanges.map {
143       case BoundedByteRange(start, end) =>
144           val startKey = if (start.length == 0) { null } else { new Key(new Text(start)) }
145           val endKey = if (end.length == 0) { null } else { new Key(new Text(end)) }
146           new Range(startKey, true, endKey, false)
147 
148       case SingleRowByteRange(row) =>
149         new Range(new Text(row))
150     }
151     val numThreads = if (index.name == IdIndex.name) { ds.config.queries.recordThreads } else { ds.config.queries.threads }
152     val tables = index.getTablesForQuery(filter.filter)
153     val (colFamily, schema) = {
154       val (cf, s) = groups.group(index.sft, hints.getTransformDefinition, ecql)
155       (Some(new Text(ColumnFamilyMapper(index)(cf))), s)
156     }
157     // used when remote processing is disabled
158     lazy val returnSchema = hints.getTransformSchema.getOrElse(schema)
159     lazy val fti = FilterTransformIterator.configure(schema, index, ecql, hints).toSeq
160     lazy val resultsToFeatures = AccumuloResultsToFeatures(index, returnSchema)
161     lazy val localReducer = Some(new LocalTransformReducer(returnSchema, None, None, None, hints))
162 
163     index match {
164       case i: AttributeJoinIndex =>
165         AccumuloJoinIndexAdapter.createQueryPlan(ds, i, filter, tables, ranges, colFamily, schema, ecql, hints, numThreads)
166 
167       case _ =>
168         val (iter, eToF, reduce) = if (strategy.hints.isBinQuery) {
169           if (ds.config.remote.bin) {
170             val iter = BinAggregatingIterator.configure(schema, index, ecql, hints)
171             (Seq(iter), new AccumuloBinResultsToFeatures(), None)
172           } else {
173             if (hints.isSkipReduce) {
174               // override the return sft to reflect what we're actually returning,
175               // since the bin sft is only created in the local reduce step
176               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
177             }
178             (fti, resultsToFeatures, localReducer)
179           }
180         } else if (strategy.hints.isArrowQuery) {
181           if (ds.config.remote.arrow) {
182             val (iter, reduce) = ArrowIterator.configure(schema, index, ds.stats, filter.filter, ecql, hints)
183             (Seq(iter), new AccumuloArrowResultsToFeatures(), Some(reduce))
184           } else {
185             if (hints.isSkipReduce) {
186               // override the return sft to reflect what we're actually returning,
187               // since the arrow sft is only created in the local reduce step
188               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
189             }
190             (fti, resultsToFeatures, localReducer)
191           }
192         } else if (strategy.hints.isDensityQuery) {
193           if (ds.config.remote.density) {
194             val iter = DensityIterator.configure(schema, index, ecql, hints)
195             (Seq(iter), new AccumuloDensityResultsToFeatures(), None)
196           } else {
197             if (hints.isSkipReduce) {
198               // override the return sft to reflect what we're actually returning,
199               // since the density sft is only created in the local reduce step
200               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
201             }
202             (fti, resultsToFeatures, localReducer)
203           }
204         } else if (strategy.hints.isStatsQuery) {
205           if (ds.config.remote.stats) {
206             val iter = StatsIterator.configure(schema, index, ecql, hints)
207             val reduce = Some(StatsScan.StatsReducer(schema, hints))
208             (Seq(iter), new AccumuloStatsResultsToFeatures(), reduce)
209           } else {
210             if (hints.isSkipReduce) {
211               // override the return sft to reflect what we're actually returning,
212               // since the stats sft is only created in the local reduce step
213               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
214             }
215             (fti, resultsToFeatures, localReducer)
216           }
217         } else {
218           (fti, resultsToFeatures, None)
219         }
220 
221         if (ranges.isEmpty) { EmptyPlan(strategy.filter, reduce) } else {
222           // configure additional iterators based on the index
223           // TODO pull this out to be SPI loaded so that new indices can be added seamlessly
224           val indexIter = if (index.name == Z3Index.name) {
225             strategy.values.toSeq.map { case v: Z3IndexValues =>
226               val offset = index.keySpace.sharding.length + index.keySpace.sharing.length
227               Z3Iterator.configure(v, offset, hints.getFilterCompatibility, ZIterPriority)
228             }
229           } else if (index.name == Z2Index.name) {
230             strategy.values.toSeq.map { case v: Z2IndexValues =>
231               Z2Iterator.configure(v, index.keySpace.sharding.length + index.keySpace.sharing.length, ZIterPriority)
232             }
233           } else if (index.name == S3Index.name) {
234             strategy.values.toSeq.map { case v: S3IndexValues =>
235               S3Iterator.configure(v, index.keySpace.sharding.length, ZIterPriority)
236             }
237           } else if (index.name == S2Index.name) {
238             strategy.values.toSeq.map { case v: S2IndexValues =>
239               S2Iterator.configure(v, index.keySpace.sharding.length, ZIterPriority)
240             }
241           } else {
242             Seq.empty
243           }
244 
245           // add the attribute-level vis iterator if necessary
246           val visIter = index.sft.getVisibilityLevel match {
247             case VisibilityLevel.Attribute => Seq(KryoVisibilityRowEncoder.configure(schema))
248             case _ => Seq.empty
249           }
250 
251           val iters = iter ++ indexIter ++ visIter
252 
253           val sort = hints.getSortFields
254           val max = hints.getMaxFeatures
255           val project = hints.getProjection
256 
257           BatchScanPlan(filter, tables, ranges, iters, colFamily, eToF, reduce, sort, max, project, numThreads)
258         }
259     }
260   }
