60   override def createTable(
61       index: GeoMesaFeatureIndex[_, _],
62       partition: Option[String],
63       splits: => Seq[Array[Byte]]): Unit = {
64     val table = index.configureTableName(partition) // writes table name to metadata
65     // create table if it doesn't exist
66     val created = TableUtils.createTableIfNeeded(ds.connector, table, index.sft.isLogicalTime)
67 
68     def addSplitsAndGroups(): Unit = {
69       // create splits
70       val splitsToAdd = splits.map(new Text(_)).toSet -- tableOps.listSplits(table).asScala.toSet
71       if (splitsToAdd.nonEmpty) {
72         tableOps.addSplits(table, new java.util.TreeSet(splitsToAdd.asJava))
73       }
74 
75       // create locality groups
76       val existingGroups = tableOps.getLocalityGroups(table)
77       val localityGroups = new java.util.HashMap[String, java.util.Set[Text]](existingGroups)
78 
79       def addGroup(cf: Text): Unit = {
80         val key = cf.toString
81         if (localityGroups.containsKey(key)) {
82           val update = new java.util.HashSet[Text](localityGroups.get(key))
83           update.add(cf)
84           localityGroups.put(key, update)
85         } else {
86           localityGroups.put(key, Collections.singleton(cf))
87         }
88       }
89 
90       groups.apply(index.sft).foreach { case (k, _) => addGroup(new Text(k)) }
91 
92       if (localityGroups != existingGroups) {
93         tableOps.setLocalityGroups(table, localityGroups)
94       }
95     }
96 
97     if (created) {
98       addSplitsAndGroups()
99 
100       // enable block cache
101       tableOps.setProperty(table, Property.TABLE_BLOCKCACHE_ENABLED.getKey, "true")
102 
103       if (index.name == IdIndex.name) {
104         // enable the row functor as the feature ID is stored in the Row ID
105         tableOps.setProperty(table, Property.TABLE_BLOOM_KEY_FUNCTOR.getKey, classOf[RowFunctor].getCanonicalName)
106         tableOps.setProperty(table, Property.TABLE_BLOOM_ENABLED.getKey, "true")
107       }
108 
109       if (index.sft.isVisibilityRequired) {
110         VisibilityIterator.set(tableOps, table)
111       }
112     } else if (index.keySpace.sharing.nonEmpty) {
113       // even if the table existed, we still need to check the splits and locality groups if it's shared
114       addSplitsAndGroups()
115     }
116   }

147   override def createQueryPlan(strategy: QueryStrategy): AccumuloQueryPlan = {
148     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
149 
150     val QueryStrategy(filter, byteRanges, _, _, ecql, hints, _) = strategy
151     val index = filter.index
152     // index api defines empty start/end for open-ended range - in accumulo, it's indicated with null
153     // index api defines start row inclusive, end row exclusive
154     val ranges = byteRanges.map {
155       case BoundedByteRange(start, end) =>
156           val startKey = if (start.length == 0) { null } else { new Key(new Text(start)) }
157           val endKey = if (end.length == 0) { null } else { new Key(new Text(end)) }
158           new Range(startKey, true, endKey, false)
159 
160       case SingleRowByteRange(row) =>
161         new Range(new Text(row))
162     }
163     val numThreads = if (index.name == IdIndex.name) { ds.config.queries.recordThreads } else { ds.config.queries.threads }
164     val tables = index.getTablesForQuery(filter.filter)
165     val (colFamily, schema) = {
166       val (cf, s) = groups.group(index.sft, hints.getTransformDefinition, ecql)
167       (Some(new Text(ColumnFamilyMapper(index)(cf))), s)
168     }
169     // used when remote processing is disabled
170     lazy val returnSchema = hints.getTransformSchema.getOrElse(schema)
171     lazy val fti = FilterTransformIterator.configure(schema, index, ecql, hints).toSeq
172     lazy val resultsToFeatures = AccumuloResultsToFeatures(index, returnSchema)
173     lazy val localReducer = {
174       val arrowHook = Some(ArrowDictionaryHook(ds.stats, filter.filter))
175       Some(new LocalTransformReducer(returnSchema, None, None, None, hints, arrowHook))
176     }
177 
178     index match {
179       case i: AttributeJoinIndex =>
180         AccumuloJoinIndexAdapter.createQueryPlan(ds, i, filter, tables, ranges, colFamily, schema, ecql, hints, numThreads)
181 
182       case _ =>
183         val (iter, eToF, reduce) = if (strategy.hints.isBinQuery) {
184           if (ds.config.remote.bin) {
185             val iter = BinAggregatingIterator.configure(schema, index, ecql, hints)
186             (Seq(iter), new AccumuloBinResultsToFeatures(), None)
187           } else {
188             if (hints.isSkipReduce) {
189               // override the return sft to reflect what we're actually returning,
190               // since the bin sft is only created in the local reduce step
191               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
192             }
193             (fti, resultsToFeatures, localReducer)
194           }
195         } else if (strategy.hints.isArrowQuery) {
196           if (ds.config.remote.arrow) {
197             val (iter, reduce) = ArrowIterator.configure(schema, index, ds.stats, filter.filter, ecql, hints)
198             (Seq(iter), new AccumuloArrowResultsToFeatures(), Some(reduce))
199           } else {
200             if (hints.isSkipReduce) {
201               // override the return sft to reflect what we're actually returning,
202               // since the arrow sft is only created in the local reduce step
203               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
204             }
205             (fti, resultsToFeatures, localReducer)
206           }
207         } else if (strategy.hints.isDensityQuery) {
208           if (ds.config.remote.density) {
209             val iter = DensityIterator.configure(schema, index, ecql, hints)
210             (Seq(iter), new AccumuloDensityResultsToFeatures(), None)
211           } else {
212             if (hints.isSkipReduce) {
213               // override the return sft to reflect what we're actually returning,
214               // since the density sft is only created in the local reduce step
215               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
216             }
217             (fti, resultsToFeatures, localReducer)
218           }
219         } else if (strategy.hints.isStatsQuery) {
220           if (ds.config.remote.stats) {
221             val iter = StatsIterator.configure(schema, index, ecql, hints)
222             val reduce = Some(StatsScan.StatsReducer(schema, hints))
223             (Seq(iter), new AccumuloStatsResultsToFeatures(), reduce)
224           } else {
225             if (hints.isSkipReduce) {
226               // override the return sft to reflect what we're actually returning,
227               // since the stats sft is only created in the local reduce step
228               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
229             }
230             (fti, resultsToFeatures, localReducer)
231           }
232         } else {
233           (fti, resultsToFeatures, None)
234         }
235 
236         if (ranges.isEmpty) { EmptyPlan(strategy.filter, reduce) } else {
237           // configure additional iterators based on the index
238           // TODO pull this out to be SPI loaded so that new indices can be added seamlessly
239           val indexIter = if (index.name == Z3Index.name) {
240             strategy.values.toSeq.map { case v: Z3IndexValues =>
241               val offset = index.keySpace.sharding.length + index.keySpace.sharing.length
242               Z3Iterator.configure(v, offset, hints.getFilterCompatibility, ZIterPriority)
243             }
244           } else if (index.name == Z2Index.name) {
245             strategy.values.toSeq.map { case v: Z2IndexValues =>
246               Z2Iterator.configure(v, index.keySpace.sharding.length + index.keySpace.sharing.length, ZIterPriority)
247             }
248           } else if (index.name == S3Index.name) {
249             strategy.values.toSeq.map { case v: S3IndexValues =>
250               S3Iterator.configure(v, index.keySpace.sharding.length, ZIterPriority)
251             }
252           } else if (index.name == S2Index.name) {
253             strategy.values.toSeq.map { case v: S2IndexValues =>
254               S2Iterator.configure(v, index.keySpace.sharding.length, ZIterPriority)
255             }
256           } else {
257             Seq.empty
258           }
259 
260           // add the attribute-level vis iterator if necessary
261           val visIter = index.sft.getVisibilityLevel match {
262             case VisibilityLevel.Attribute => Seq(KryoVisibilityRowEncoder.configure(schema))
263             case _ => Seq.empty
264           }
265 
266           val iters = iter ++ indexIter ++ visIter
267 
268           val sort = hints.getSortFields
269           val max = hints.getMaxFeatures
270           val project = hints.getProjection
271 
272           BatchScanPlan(filter, tables, ranges, iters, colFamily, eToF, reduce, sort, max, project, numThreads)
273         }
274     }
275   }
