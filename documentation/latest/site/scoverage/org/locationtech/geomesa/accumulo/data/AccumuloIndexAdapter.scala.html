62   override def createTable(
63       index: GeoMesaFeatureIndex[_, _],
64       partition: Option[String],
65       splits: => Seq[Array[Byte]]): Unit = {
66     val table = index.configureTableName(partition) // writes table name to metadata
67     // create table if it doesn't exist
68     val created = ensureTableExists(table, index.sft.isLogicalTime)
69 
70     def addSplitsAndGroups(): Unit = {
71       // create splits
72       val splitsToAdd = splits.map(new Text(_)).toSet -- tableOps.listSplits(table).asScala.toSet
73       if (splitsToAdd.nonEmpty) {
74         tableOps.addSplits(table, new java.util.TreeSet(splitsToAdd.asJava))
75       }
76 
77       // create locality groups
78       val existingGroups = tableOps.getLocalityGroups(table)
79       val localityGroups = new java.util.HashMap[String, java.util.Set[Text]](existingGroups)
80 
81       def addGroup(cf: Text): Unit = {
82         val key = cf.toString
83         if (localityGroups.containsKey(key)) {
84           val update = new java.util.HashSet[Text](localityGroups.get(key))
85           update.add(cf)
86           localityGroups.put(key, update)
87         } else {
88           localityGroups.put(key, Collections.singleton(cf))
89         }
90       }
91 
92       groups.apply(index.sft).foreach { case (k, _) => addGroup(new Text(k)) }
93 
94       if (localityGroups != existingGroups) {
95         tableOps.setLocalityGroups(table, localityGroups)
96       }
97     }
98 
99     if (created) {
100       import org.apache.accumulo.core.conf.Property.{TABLE_BLOCKCACHE_ENABLED, TABLE_BLOOM_ENABLED, TABLE_BLOOM_KEY_FUNCTOR}
101 
102       addSplitsAndGroups()
103 
104       // block cache config
105       val enableBlockCache = {
106         val key = if (index.sft.isPartitioned) { InternalConfigs.PartitionTableCache } else { Configs.TableCacheEnabled }
107         val config = index.sft.getUserData.get(key).asInstanceOf[String]
108         if (config == null) { true } else {
109           val enabled = config.split(',').exists { hint =>
110             hint.equalsIgnoreCase(index.name) || hint.equalsIgnoreCase((Seq(index.name) ++ index.attributes).mkString(":")) ||
111               hint.equalsIgnoreCase(index.identifier)
112           }
113           logger.debug(s"Setting ${TABLE_BLOCKCACHE_ENABLED.getKey}=$enabled for index ${index.identifier} based on user config: $config")
114           enabled
115         }
116       }
117       if (enableBlockCache) {
118         tableOps.setProperty(table, TABLE_BLOCKCACHE_ENABLED.getKey, "true")
119       }
120 
121       if (index.name == IdIndex.name) {
122         // enable the row functor as the feature ID is stored in the Row ID
123         tableOps.setProperty(table, TABLE_BLOOM_KEY_FUNCTOR.getKey, classOf[RowFunctor].getCanonicalName)
124         tableOps.setProperty(table, TABLE_BLOOM_ENABLED.getKey, "true")
125       }
126 
127       if (index.sft.isVisibilityRequired) {
128         VisibilityIterator.set(tableOps, table)
129       }
130     } else if (index.keySpace.sharing.nonEmpty) {
131       // even if the table existed, we still need to check the splits and locality groups if it's shared
132       addSplitsAndGroups()
133     }
134   }

153   override def createQueryPlan(strategy: QueryStrategy): AccumuloQueryPlan = {
154     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
155 
156     val QueryStrategy(filter, byteRanges, _, _, ecql, hints, _) = strategy
157     val index = filter.index
158     // index api defines empty start/end for open-ended range - in accumulo, it's indicated with null
159     // index api defines start row inclusive, end row exclusive
160     val ranges = byteRanges.map {
161       case BoundedByteRange(start, end) =>
162           val startKey = if (start.length == 0) { null } else { new Key(new Text(start)) }
163           val endKey = if (end.length == 0) { null } else { new Key(new Text(end)) }
164           new Range(startKey, true, endKey, false)
165 
166       case SingleRowByteRange(row) =>
167         new Range(new Text(row))
168     }
169     val numThreads = if (index.name == IdIndex.name) { ds.config.queries.recordThreads } else { ds.config.queries.threads }
170     val tables = index.getTablesForQuery(filter.filter)
171     val (colFamily, schema) = {
172       val (cf, s) = groups.group(index.sft, hints.getTransformDefinition, ecql)
173       (Some(new Text(ColumnFamilyMapper(index)(cf))), s)
174     }
175     // used when remote processing is disabled
176     lazy val returnSchema = hints.getTransformSchema.getOrElse(schema)
177     lazy val fti = FilterTransformIterator.configure(schema, index, ecql, hints).toSeq
178     lazy val resultsToFeatures = AccumuloResultsToFeatures(index, returnSchema)
179     lazy val localReducer = Some(new LocalTransformReducer(returnSchema, None, None, None, hints))
180 
181     index match {
182       case i: AttributeJoinIndex =>
183         AccumuloJoinIndexAdapter.createQueryPlan(ds, i, filter, tables, ranges, colFamily, schema, ecql, hints, numThreads)
184 
185       case _ =>
186         val (iter, eToF, reduce) = if (strategy.hints.isBinQuery) {
187           if (ds.config.remote.bin) {
188             val iter = BinAggregatingIterator.configure(schema, index, ecql, hints)
189             (Seq(iter), new AccumuloBinResultsToFeatures(), None)
190           } else {
191             if (hints.isSkipReduce) {
192               // override the return sft to reflect what we're actually returning,
193               // since the bin sft is only created in the local reduce step
194               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
195             }
196             (fti, resultsToFeatures, localReducer)
197           }
198         } else if (strategy.hints.isArrowQuery) {
199           if (ds.config.remote.arrow) {
200             val (iter, reduce) = ArrowIterator.configure(schema, index, ds.stats, filter.filter, ecql, hints)
201             (Seq(iter), new AccumuloArrowResultsToFeatures(), Some(reduce))
202           } else {
203             if (hints.isSkipReduce) {
204               // override the return sft to reflect what we're actually returning,
205               // since the arrow sft is only created in the local reduce step
206               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
207             }
208             (fti, resultsToFeatures, localReducer)
209           }
210         } else if (strategy.hints.isDensityQuery) {
211           if (ds.config.remote.density) {
212             val iter = DensityIterator.configure(schema, index, ecql, hints)
213             (Seq(iter), new AccumuloDensityResultsToFeatures(), None)
214           } else {
215             if (hints.isSkipReduce) {
216               // override the return sft to reflect what we're actually returning,
217               // since the density sft is only created in the local reduce step
218               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
219             }
220             (fti, resultsToFeatures, localReducer)
221           }
222         } else if (strategy.hints.isStatsQuery) {
223           if (ds.config.remote.stats) {
224             val iter = StatsIterator.configure(schema, index, ecql, hints)
225             val reduce = Some(StatsScan.StatsReducer(schema, hints))
226             (Seq(iter), new AccumuloStatsResultsToFeatures(), reduce)
227           } else {
228             if (hints.isSkipReduce) {
229               // override the return sft to reflect what we're actually returning,
230               // since the stats sft is only created in the local reduce step
231               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
232             }
233             (fti, resultsToFeatures, localReducer)
234           }
235         } else {
236           (fti, resultsToFeatures, None)
237         }
238 
239         if (ranges.isEmpty) { EmptyPlan(strategy.filter, reduce) } else {
240           // configure additional iterators based on the index
241           // TODO pull this out to be SPI loaded so that new indices can be added seamlessly
242           val indexIter = if (index.name == Z3Index.name) {
243             strategy.values.toSeq.map { case v: Z3IndexValues =>
244               val offset = index.keySpace.sharding.length + index.keySpace.sharing.length
245               Z3Iterator.configure(v, offset, hints.getFilterCompatibility, ZIterPriority)
246             }
247           } else if (index.name == Z2Index.name) {
248             strategy.values.toSeq.map { case v: Z2IndexValues =>
249               Z2Iterator.configure(v, index.keySpace.sharding.length + index.keySpace.sharing.length, ZIterPriority)
250             }
251           } else if (index.name == S3Index.name) {
252             strategy.values.toSeq.map { case v: S3IndexValues =>
253               S3Iterator.configure(v, index.keySpace.sharding.length, ZIterPriority)
254             }
255           } else if (index.name == S2Index.name) {
256             strategy.values.toSeq.map { case v: S2IndexValues =>
257               S2Iterator.configure(v, index.keySpace.sharding.length, ZIterPriority)
258             }
259           } else {
260             Seq.empty
261           }
262 
263           // add the attribute-level vis iterator if necessary
264           val visIter = index.sft.getVisibilityLevel match {
265             case VisibilityLevel.Attribute => Seq(KryoVisibilityRowEncoder.configure(schema))
266             case _ => Seq.empty
267           }
268 
269           val iters = iter ++ indexIter ++ visIter
270 
271           val sort = hints.getSortFields
272           val max = hints.getMaxFeatures
273           val project = hints.getProjection
274 
275           BatchScanPlan(filter, tables, ranges, iters, colFamily, eToF, reduce, sort, max, project, numThreads)
276         }
277     }
278   }

340         applyVisibility(sf, result.getKey)
