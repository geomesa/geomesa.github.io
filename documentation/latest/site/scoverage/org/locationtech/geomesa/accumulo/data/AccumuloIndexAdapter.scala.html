70       index: GeoMesaFeatureIndex[_, _],
71       partition: Option[String],
72       splits: => Seq[Array[Byte]]): Unit = {

77     def addSplitsAndGroups(): Unit = {
78       // create splits
79       val splitsToAdd = splits.map(new Text(_)).toSet -- tableOps.listSplits(table).asScala.toSet
80       if (splitsToAdd.nonEmpty) {
81         tableOps.addSplits(table, new java.util.TreeSet(splitsToAdd.asJava))
82       }
83 
84       // create locality groups
85       val existingGroups = tableOps.getLocalityGroups(table)
86       val localityGroups = new java.util.HashMap[String, java.util.Set[Text]](existingGroups)
87 
88       def addGroup(cf: Text): Unit = {
89         val key = cf.toString
90         if (localityGroups.containsKey(key)) {
91           val update = new java.util.HashSet[Text](localityGroups.get(key))
92           update.add(cf)
93           localityGroups.put(key, update)
94         } else {
95           localityGroups.put(key, Collections.singleton(cf))
96         }
97       }
98 
99       groups.apply(index.sft).foreach { case (k, _) => addGroup(new Text(k)) }
100 
101       if (localityGroups != existingGroups) {
102         tableOps.setLocalityGroups(table, localityGroups)
103       }
104     }
105 
106     if (created) {
107       import org.apache.accumulo.core.conf.Property.{TABLE_BLOCKCACHE_ENABLED, TABLE_BLOOM_ENABLED, TABLE_BLOOM_KEY_FUNCTOR}
108 
109       addSplitsAndGroups()
110 
111       // block cache config
112       val enableBlockCache = {
113         val key = if (index.sft.isPartitioned) { InternalConfigs.PartitionTableCache } else { Configs.TableCacheEnabled }
114         val config = index.sft.getUserData.get(key).asInstanceOf[String]
115         if (config == null) { true } else {
116           val enabled = config.split(',').exists { hint =>
117             hint.equalsIgnoreCase(index.name) || hint.equalsIgnoreCase((Seq(index.name) ++ index.attributes).mkString(":")) ||
118               hint.equalsIgnoreCase(index.identifier)
119           }
120           logger.debug(s"Setting ${TABLE_BLOCKCACHE_ENABLED.getKey}=$enabled for index ${index.identifier} based on user config: $config")

122         }
123       }

125         tableOps.setProperty(table, TABLE_BLOCKCACHE_ENABLED.getKey, "true")
126       }
127 
128       if (index.name == IdIndex.name) {
129         // enable the row functor as the feature ID is stored in the Row ID
130         tableOps.setProperty(table, TABLE_BLOOM_KEY_FUNCTOR.getKey, classOf[RowFunctor].getCanonicalName)
131         tableOps.setProperty(table, TABLE_BLOOM_ENABLED.getKey, "true")
132       }
133 
134       index.sft.getTableProps.foreach { case (k, v) => tableOps.setProperty(table, k, v) }
135 
136       if (index.sft.isVisibilityRequired) {
137         VisibilityIterator.set(tableOps, table)
138       }
139     } else if (index.keySpace.sharing.nonEmpty) {
140       // even if the table existed, we still need to check the splits and locality groups if it's shared
141       addSplitsAndGroups()
142     }
143   }

223           }
224         } else if (hints.isDensityQuery) {

230           }

235             scanPlan(Some(iter), new AccumuloStatsResultsToFeatures(), reduce)
236           } else {
237             semiLocalPlan()
238           }
239         } else {
240           scanPlan(fti, AccumuloResultsToFeatures(index, returnSchema), None)
241         }
242 
243         if (tables.isEmpty || ranges.isEmpty) { EmptyPlan(strategy, plan.reducer) } else { plan }
244     }

261   override def getStrategyCost(strategy: FilterStrategy, explain: Explainer): Option[Long] = {
262     explain.pushLevel(s"Calculating cost for ${strategy.index.identifier}")
263     val start = System.currentTimeMillis()
264     try {
265       val tables = strategy.index.getTablesForQuery(strategy.filter)
266       if (tables.isEmpty) {
267         return Some(0L)
268       }
269       val ranges = strategy.getQueryStrategy(explain).ranges.map(toAccumuloRange).asJava
270       val cost =
271         tables.foldLeft(0d) { case (sum, table) =>
272           val numTablets = tableSizeCache.get(table)
273           val tabletsScanned = tableOps.locate(table, ranges).groupByTablet().size()
274           explain(s"Strategy hits $tabletsScanned/$numTablets tablets for table $table")
275           val cost = 100 * (tabletsScanned.toDouble / numTablets)
276           sum + cost
277         }
278       Some(cost.toLong)
279     } finally {
280       explain(s"Cost calculations took ${System.currentTimeMillis() - start}ms").popLevel()
281     }
282   }

289   /**
290    * Converts a generic index-api range into an Accumulo range
291    *
292    * @param range range
293    * @return
294    */
295   private def toAccumuloRange(range: ByteRange): Range = range match {
296     case BoundedByteRange(lower, upper) =>
297       // index api defines empty start/end for open-ended range - in accumulo, it's indicated with null
298       val start = if (lower.length == 0) { null } else { new Key(new Text(lower)) }
299       val end = if (upper.length == 0) { null } else { new Key(new Text(upper)) }
300       // index api defines start row inclusive, end row exclusive
301       new Range(start, true, end, false)
302 
303     case SingleRowByteRange(row) =>
304       new Range(new Text(row))
305 
306     case _ =>
307       throw new IllegalArgumentException(s"Unexpected range type $range")
308   }
