60   override def createTable(
61       index: GeoMesaFeatureIndex[_, _],
62       partition: Option[String],
63       splits: => Seq[Array[Byte]]): Unit = {
64     val table = index.configureTableName(partition) // writes table name to metadata
65     // create table if it doesn't exist
66     val created = ensureTableExists(table, index.sft.isLogicalTime)
67 
68     def addSplitsAndGroups(): Unit = {
69       // create splits
70       val splitsToAdd = splits.map(new Text(_)).toSet -- tableOps.listSplits(table).asScala.toSet
71       if (splitsToAdd.nonEmpty) {
72         tableOps.addSplits(table, new java.util.TreeSet(splitsToAdd.asJava))
73       }
74 
75       // create locality groups
76       val existingGroups = tableOps.getLocalityGroups(table)
77       val localityGroups = new java.util.HashMap[String, java.util.Set[Text]](existingGroups)
78 
79       def addGroup(cf: Text): Unit = {
80         val key = cf.toString
81         if (localityGroups.containsKey(key)) {
82           val update = new java.util.HashSet[Text](localityGroups.get(key))
83           update.add(cf)
84           localityGroups.put(key, update)
85         } else {
86           localityGroups.put(key, Collections.singleton(cf))
87         }
88       }
89 
90       groups.apply(index.sft).foreach { case (k, _) => addGroup(new Text(k)) }
91 
92       if (localityGroups != existingGroups) {
93         tableOps.setLocalityGroups(table, localityGroups)
94       }
95     }
96 
97     if (created) {
98       addSplitsAndGroups()
99 
100       // enable block cache
101       tableOps.setProperty(table, Property.TABLE_BLOCKCACHE_ENABLED.getKey, "true")
102 
103       if (index.name == IdIndex.name) {
104         // enable the row functor as the feature ID is stored in the Row ID
105         tableOps.setProperty(table, Property.TABLE_BLOOM_KEY_FUNCTOR.getKey, classOf[RowFunctor].getCanonicalName)
106         tableOps.setProperty(table, Property.TABLE_BLOOM_ENABLED.getKey, "true")
107       }
108 
109       if (index.sft.isVisibilityRequired) {
110         VisibilityIterator.set(tableOps, table)
111       }
112     } else if (index.keySpace.sharing.nonEmpty) {
113       // even if the table existed, we still need to check the splits and locality groups if it's shared
114       addSplitsAndGroups()
115     }
116   }

135   override def createQueryPlan(strategy: QueryStrategy): AccumuloQueryPlan = {
136     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
137 
138     val QueryStrategy(filter, byteRanges, _, _, ecql, hints, _) = strategy
139     val index = filter.index
140     // index api defines empty start/end for open-ended range - in accumulo, it's indicated with null
141     // index api defines start row inclusive, end row exclusive
142     val ranges = byteRanges.map {
143       case BoundedByteRange(start, end) =>
144           val startKey = if (start.length == 0) { null } else { new Key(new Text(start)) }
145           val endKey = if (end.length == 0) { null } else { new Key(new Text(end)) }
146           new Range(startKey, true, endKey, false)
147 
148       case SingleRowByteRange(row) =>
149         new Range(new Text(row))
150     }
151     val numThreads = if (index.name == IdIndex.name) { ds.config.queries.recordThreads } else { ds.config.queries.threads }
152     val tables = index.getTablesForQuery(filter.filter)
153     val (colFamily, schema) = {
154       val (cf, s) = groups.group(index.sft, hints.getTransformDefinition, ecql)
155       (Some(new Text(ColumnFamilyMapper(index)(cf))), s)
156     }
157     // used when remote processing is disabled
158     lazy val returnSchema = hints.getTransformSchema.getOrElse(schema)
159     lazy val fti = FilterTransformIterator.configure(schema, index, ecql, hints).toSeq
160     lazy val resultsToFeatures = AccumuloResultsToFeatures(index, returnSchema)
161     lazy val localReducer = {
162       val arrowHook = Some(ArrowDictionaryHook(ds.stats, filter.filter))
163       Some(new LocalTransformReducer(returnSchema, None, None, None, hints, arrowHook))
164     }
165 
166     index match {
167       case i: AttributeJoinIndex =>
168         AccumuloJoinIndexAdapter.createQueryPlan(ds, i, filter, tables, ranges, colFamily, schema, ecql, hints, numThreads)
169 
170       case _ =>
171         val (iter, eToF, reduce) = if (strategy.hints.isBinQuery) {
172           if (ds.config.remote.bin) {
173             val iter = BinAggregatingIterator.configure(schema, index, ecql, hints)
174             (Seq(iter), new AccumuloBinResultsToFeatures(), None)
175           } else {
176             if (hints.isSkipReduce) {
177               // override the return sft to reflect what we're actually returning,
178               // since the bin sft is only created in the local reduce step
179               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
180             }
181             (fti, resultsToFeatures, localReducer)
182           }
183         } else if (strategy.hints.isArrowQuery) {
184           if (ds.config.remote.arrow) {
185             val (iter, reduce) = ArrowIterator.configure(schema, index, ds.stats, filter.filter, ecql, hints)
186             (Seq(iter), new AccumuloArrowResultsToFeatures(), Some(reduce))
187           } else {
188             if (hints.isSkipReduce) {
189               // override the return sft to reflect what we're actually returning,
190               // since the arrow sft is only created in the local reduce step
191               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
192             }
193             (fti, resultsToFeatures, localReducer)
194           }
195         } else if (strategy.hints.isDensityQuery) {
196           if (ds.config.remote.density) {
197             val iter = DensityIterator.configure(schema, index, ecql, hints)
198             (Seq(iter), new AccumuloDensityResultsToFeatures(), None)
199           } else {
200             if (hints.isSkipReduce) {
201               // override the return sft to reflect what we're actually returning,
202               // since the density sft is only created in the local reduce step
203               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
204             }
205             (fti, resultsToFeatures, localReducer)
206           }
207         } else if (strategy.hints.isStatsQuery) {
208           if (ds.config.remote.stats) {
209             val iter = StatsIterator.configure(schema, index, ecql, hints)
210             val reduce = Some(StatsScan.StatsReducer(schema, hints))
211             (Seq(iter), new AccumuloStatsResultsToFeatures(), reduce)
212           } else {
213             if (hints.isSkipReduce) {
214               // override the return sft to reflect what we're actually returning,
215               // since the stats sft is only created in the local reduce step
216               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
217             }
218             (fti, resultsToFeatures, localReducer)
219           }
220         } else {
221           (fti, resultsToFeatures, None)
222         }
223 
224         if (ranges.isEmpty) { EmptyPlan(strategy.filter, reduce) } else {
225           // configure additional iterators based on the index
226           // TODO pull this out to be SPI loaded so that new indices can be added seamlessly
227           val indexIter = if (index.name == Z3Index.name) {
228             strategy.values.toSeq.map { case v: Z3IndexValues =>
229               val offset = index.keySpace.sharding.length + index.keySpace.sharing.length
230               Z3Iterator.configure(v, offset, hints.getFilterCompatibility, ZIterPriority)
231             }
232           } else if (index.name == Z2Index.name) {
233             strategy.values.toSeq.map { case v: Z2IndexValues =>
234               Z2Iterator.configure(v, index.keySpace.sharding.length + index.keySpace.sharing.length, ZIterPriority)
235             }
236           } else if (index.name == S3Index.name) {
237             strategy.values.toSeq.map { case v: S3IndexValues =>
238               S3Iterator.configure(v, index.keySpace.sharding.length, ZIterPriority)
239             }
240           } else if (index.name == S2Index.name) {
241             strategy.values.toSeq.map { case v: S2IndexValues =>
242               S2Iterator.configure(v, index.keySpace.sharding.length, ZIterPriority)
243             }
244           } else {
245             Seq.empty
246           }
247 
248           // add the attribute-level vis iterator if necessary
249           val visIter = index.sft.getVisibilityLevel match {
250             case VisibilityLevel.Attribute => Seq(KryoVisibilityRowEncoder.configure(schema))
251             case _ => Seq.empty
252           }
253 
254           val iters = iter ++ indexIter ++ visIter
255 
256           val sort = hints.getSortFields
257           val max = hints.getMaxFeatures
258           val project = hints.getProjection
259 
260           BatchScanPlan(filter, tables, ranges, iters, colFamily, eToF, reduce, sort, max, project, numThreads)
261         }
262     }
263   }
