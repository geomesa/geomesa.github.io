104   override protected def transitionIndices(sft: SimpleFeatureType): Unit = {
105     // note: versions already correspond to accumulo index versions
106     val dtg = sft.getDtgField.toSeq
107     val geom = Option(sft.getGeomField).toSeq
108     val indices = sft.getIndices.flatMap {
109       case id if id.name == IdIndex.name  => Seq(id) // no update needed
110       case id if id.name == "records"     => Seq(id.copy(name = IdIndex.name))
111       case id if id.name == Z3Index.name  => Seq(id.copy(attributes = geom ++ dtg))
112       case id if id.name == XZ3Index.name => Seq(id.copy(attributes = geom ++ dtg))
113       case id if id.name == Z2Index.name  => Seq(id.copy(attributes = geom))
114       case id if id.name == XZ2Index.name => Seq(id.copy(attributes = geom))
115       case id if id.name == AttributeIndex.name =>
116         lazy val fields = if (id.version < 4) { dtg } else { geom ++ dtg }
117         sft.getAttributeDescriptors.asScala.flatMap { d =>
118           val index = d.getUserData.remove(AttributeOptions.OptIndex).asInstanceOf[String]
119           if (index == null || index.equalsIgnoreCase(IndexCoverage.NONE.toString) || index.equalsIgnoreCase("false")) {
120             Seq.empty
121           } else if (index.equalsIgnoreCase(IndexCoverage.FULL.toString)) {
122             Seq(id.copy(name = AttributeIndex.name, attributes = Seq(d.getLocalName) ++ fields))
123           } else if (index.equalsIgnoreCase(IndexCoverage.JOIN.toString) || java.lang.Boolean.valueOf(index)) {
124             Seq(id.copy(name = JoinIndex.name, attributes = Seq(d.getLocalName) ++ fields))
125           } else {
126             throw new IllegalStateException(s"Expected an index coverage or boolean but got: $index")
127           }
128         }
129     }
130     sft.setIndices(indices)
131   }

155   override protected def preSchemaCreate(sft: SimpleFeatureType): Unit = {
156     import org.locationtech.geomesa.index.conf.SchemaProperties.ValidateDistributedClasspath
157 
158     // call super first so that user data keys are updated
159     super.preSchemaCreate(sft)
160 
161     def getNamespace(prefix: String): String = prefix.indexOf('.') match {
162       case -1 => ""
163       case i  => prefix.substring(0, i)
164     }
165 
166     val prefixes = Seq(config.catalog) ++ sft.getIndices.flatMap(i => sft.getTablePrefix(i.name))
167     prefixes.map(getNamespace).distinct.foreach { namespace =>
168       if (namespace.nonEmpty) {
169         adapter.ensureNamespaceExists(namespace)
170       }
171       // validate that the accumulo runtime is available
172       val canLoad = connector.namespaceOperations().testClassLoad(namespace,
173         classOf[ProjectVersionIterator].getName, classOf[SortedKeyValueIterator[_, _]].getName)
174 
175       if (!canLoad) {
176         val msg = s"Could not load GeoMesa distributed code from the Accumulo classpath"
177         logger.error(s"$msg for catalog ${config.catalog}")
178         if (ValidateDistributedClasspath.toBoolean.contains(true)) {
179           val nsMsg = if (namespace.isEmpty) { "" } else { s" for the namespace '$namespace'" }
180           throw new RuntimeException(s"$msg. You may override this check by setting the system property " +
181             s"'${ValidateDistributedClasspath.property}=false'. Otherwise, please verify that the appropriate " +
182             s"JARs are installed$nsMsg - see https://www.geomesa.org/documentation/stable/user/accumulo/install.html" +
183             "#installing-the-accumulo-distributed-runtime-library")
184         }
185       }
186     }
187 
188     if (sft.getVisibilityLevel == VisibilityLevel.Attribute && sft.getAttributeCount > 255) {
189       throw new IllegalArgumentException("Attribute level visibility only supports up to 255 attributes")
190     }
191 
192     sft.getDtgField.foreach { dtg =>
193       if (sft.getIndices.exists(i => i.name == JoinIndex.name && i.attributes.headOption.contains(dtg))) {
194         if (!GeoMesaSchemaValidator.declared(sft, OverrideDtgJoin)) {
195           throw new IllegalArgumentException("Trying to create a schema with a partial (join) attribute index " +
196               s"on the default date field '$dtg'. This may cause whole-world queries with time bounds to be much " +
197               "slower. If this is intentional, you may override this check by putting Boolean.TRUE into the " +
198               s"SimpleFeatureType user data under the key '$OverrideDtgJoin' before calling createSchema, or by " +
199               s"setting the system property '$OverrideDtgJoin' to 'true'. Otherwise, please either specify a " +
200               "full attribute index or remove it entirely.")
201         }
202       }
203     }
204   }

206   override protected def onSchemaCreated(sft: SimpleFeatureType): Unit = {
207     super.onSchemaCreated(sft)
208     if (sft.statsEnabled) {
209       // configure the stats combining iterator on the table for this sft
210       adapter.ensureTableExists(stats.metadata.table)
211       stats.configureStatCombiner(connector, sft)
212     }
213     sft.getFeatureExpiration.foreach {
214       case IngestTimeExpiration(ttl) =>
215         val tableOps = connector.tableOperations()
216         getAllIndexTableNames(sft.getTypeName).filter(tableOps.exists).foreach { table =>
217           AgeOffIterator.set(tableOps, table, sft, ttl)
218         }
219 
220       case FeatureTimeExpiration(dtg, _, ttl) =>
221         val tableOps = connector.tableOperations()
222         manager.indices(sft).foreach { index =>
223           val indexSft = index match {
224             case joinIndex: AttributeJoinIndex => joinIndex.indexSft
225             case _ => sft
226           }
227           DtgAgeOffIterator.set(tableOps, indexSft, index, ttl, dtg)
228         }
229 
230       case e =>
231         throw new IllegalArgumentException(s"Unexpected feature expiration: $e")
232     }
233   }

235   @throws(classOf[IllegalArgumentException])
236   override protected def preSchemaUpdate(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
237     if (sft.getVisibilityLevel == VisibilityLevel.Attribute && sft.getAttributeCount > 255) {
238       throw new IllegalArgumentException("Attribute level visibility only supports up to 255 attributes")
239     }
240 
241     // check for attributes flagged 'index=join' and convert them to sft-level user data
242     sft.getAttributeDescriptors.asScala.foreach { d =>
243       val index = d.getUserData.get(AttributeOptions.OptIndex).asInstanceOf[String]
244       if (index != null && index.equalsIgnoreCase(IndexCoverage.JOIN.toString)) {
245         d.getUserData.remove(AttributeOptions.OptIndex) // remove it so it's not processed again
246         val fields = Seq(d.getLocalName) ++ Option(sft.getGeomField) ++ sft.getDtgField.filter(_ != d.getLocalName)
247         val attribute = IndexId(JoinIndex.name, JoinIndex.version, fields, IndexMode.ReadWrite)
248         val existing = sft.getIndices.map(GeoMesaFeatureIndex.identifier)
249         if (!existing.contains(GeoMesaFeatureIndex.identifier(attribute))) {
250           sft.setIndices(sft.getIndices :+ attribute)
251         }
252       }
253     }
254 
255     // check any previous age-off - previously age-off wasn't tied to the sft metadata
256     if (!sft.isFeatureExpirationEnabled && !previous.isFeatureExpirationEnabled) {
257       // explicitly set age-off in the feature type if found
258       val tableOps = connector.tableOperations()
259       val tables = getAllIndexTableNames(previous.getTypeName).filter(tableOps.exists)
260       val ageOff = tables.foldLeft[Option[FeatureExpiration]](None) { (res, table) =>
261         res.orElse(AgeOffIterator.expiry(tableOps, table))
262       }
263       val configured = ageOff.orElse {
264         tables.foldLeft[Option[FeatureExpiration]](None) { (res, table) =>
265           res.orElse(DtgAgeOffIterator.expiry(tableOps, previous, table))
266         }
267       }
268       configured.foreach(sft.setFeatureExpiration)
269     }
270 
271     super.preSchemaUpdate(sft, previous)
272   }

274   override protected def onSchemaUpdated(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
275     super.onSchemaUpdated(sft, previous)
276 
277     if (previous.statsEnabled) {
278       stats.removeStatCombiner(connector, previous)
279     }
280     if (sft.statsEnabled) {
281       adapter.ensureTableExists(stats.metadata.table)
282       stats.configureStatCombiner(connector, sft)
283     }
284 
285     val tableOps = connector.tableOperations()
286     val previousTables = getAllIndexTableNames(previous.getTypeName).filter(tableOps.exists)
287     val tables = getAllIndexTableNames(sft.getTypeName).filter(tableOps.exists)
288 
289     if (previous.isVisibilityRequired != sft.isVisibilityRequired) {
290       previousTables.foreach(VisibilityIterator.clear(tableOps, _))
291       if (sft.isVisibilityRequired) {
292         tables.foreach(VisibilityIterator.set(tableOps, _))
293       }
294     }
295 
296     previousTables.foreach { table =>
297       AgeOffIterator.clear(tableOps, table)
298       DtgAgeOffIterator.clear(tableOps, table)
299     }
300 
301     sft.getFeatureExpiration.foreach {
302       case IngestTimeExpiration(ttl) =>
303         tables.foreach(AgeOffIterator.set(tableOps, _, sft, ttl))
304 
305       case FeatureTimeExpiration(dtg, _, ttl) =>
306         manager.indices(sft).foreach { index =>
307           val indexSft = index match {
308             case joinIndex: AttributeJoinIndex => joinIndex.indexSft
309             case _ => sft
310           }
311           DtgAgeOffIterator.set(tableOps, indexSft, index, ttl, dtg)
312         }
313 
314       case e => throw new IllegalArgumentException(s"Unexpected feature expiration: $e")
315     }
316   }

325   // noinspection ScalaDeprecation
326   override def getSchema(typeName: String): SimpleFeatureType = {
327     var sft = super.getSchema(typeName)
328     if (sft == null) {
329       // check for old-style metadata and re-write it if necessary
330       if (oldMetadata.getFeatureTypes.contains(typeName)) {
331         val lock = acquireCatalogLock()
332         try {
333           if (oldMetadata.getFeatureTypes.contains(typeName)) {
334             oldMetadata.migrate(typeName)
335             new SingleRowAccumuloMetadata[Stat](stats.metadata).migrate(typeName)
336           }
337         } finally {
338           lock.close()
339         }
340         sft = super.getSchema(typeName)
341       }
342     }
343     if (sft != null) {
344       // back compatible check for index versions
345       if (sft.getIndices.isEmpty) {
346         // make the sft temporarily mutable so we can update the keys
347         sft = SimpleFeatureTypes.mutable(sft)
348         // back compatible check if user data wasn't encoded with the sft
349         if (!sft.getUserData.containsKey(AccumuloDataStore.DeprecatedSchemaVersionKey)) {
350           metadata.read(typeName, "dtgfield").foreach(sft.setDtgField)
351           sft.getUserData.put(AccumuloDataStore.DeprecatedSchemaVersionKey,
352             metadata.readRequired(typeName, GeoMesaMetadata.VersionKey))
353 
354           // if no data is written, we default to 'false' in order to support old tables
355           if (metadata.read(typeName, "tables.sharing").exists(_.toBoolean)) {
356             sft.getUserData.put(SimpleFeatureTypes.Configs.TableSharing, "true")
357             // use schema id if available or fall back to old type name for backwards compatibility
358             val prefix = metadata.read(typeName, "id").getOrElse(s"${sft.getTypeName}~")
359             sft.getUserData.put(SimpleFeatureTypes.InternalConfigs.TableSharingPrefix, prefix)
360           }
361           // noinspection deprecation
362           SimpleFeatureTypes.Configs.ENABLED_INDEX_OPTS.foreach { i =>
363             metadata.read(typeName, i).foreach(e => sft.getUserData.put(SimpleFeatureTypes.Configs.EnabledIndices, e))
364           }
365         }
366 
367         // set the enabled indices
368         sft.setIndices(AccumuloDataStore.translateSchemaVersion(sft))
369 
370         // store the metadata and reload the sft again to validate indices
371         val encoded = SimpleFeatureTypes.encodeType(sft, includeUserData = true)
372         metadata.insert(typeName, GeoMesaMetadata.AttributesKey, encoded)
373         sft = super.getSchema(typeName)
374       }
375 
376       // back compatibility check for stat configuration
377       if (sft.statsEnabled && metadata.read(typeName, GeoMesaMetadata.StatsGenerationKey).isEmpty) {
378         // configure the stats combining iterator - we only use this key for older data stores
379         val configuredKey = "stats-configured"
380         if (!metadata.read(typeName, configuredKey).contains("true")) {
381           val lock = acquireCatalogLock()
382           try {
383             if (!metadata.read(typeName, configuredKey, cache = false).contains("true")) {
384               adapter.ensureTableExists(stats.metadata.table)
385               stats.configureStatCombiner(connector, sft)
386               metadata.insert(typeName, configuredKey, "true")
387             }
388           } finally {
389             lock.close()
390           }
391         }
392         // kick off asynchronous stats run for the existing data - this will set the stat date
393         // this may get triggered more than once, but should only run one time
394         val statsRunner = new StatsRunner(this)
395         statsRunner.submit(sft)
396         statsRunner.close()
397       }
398     }
399 
400     sft
401   }

406    * Gets a query plan that can be satisfied via AccumuloInputFormat - e.g. only 1 table and configuration.
407    *
408    * @param query query
409    * @return
410    */
411   def getSingleQueryPlan(query: Query): AccumuloQueryPlan = {
412     // disable join plans as those have multiple tables
413     JoinIndex.AllowJoinPlans.set(false)
414 
415     try {
416       lazy val fallbackIndex = {
417         val schema = getSchema(query.getTypeName)
418         manager.indices(schema, IndexMode.Read).headOption.getOrElse {
419           throw new IllegalStateException(s"Schema '${schema.getTypeName}' does not have any readable indices")
420         }
421       }
422 
423       val queryPlans = getQueryPlan(query)
424 
425       if (queryPlans.isEmpty) {
426         EmptyPlan(FilterStrategy(fallbackIndex, None, Some(Filter.EXCLUDE), temporal = false, Float.PositiveInfinity))
427       } else {
428         val qps =
429           if (queryPlans.lengthCompare(1) == 0) { queryPlans } else {
430             // this query requires multiple scans, which we can't execute from some input formats
431             // instead, fall back to a full table scan
432             logger.warn("Desired query plan requires multiple scans - falling back to full table scan")
433             getQueryPlan(query, Some(fallbackIndex.identifier))
434           }
435         if (qps.lengthCompare(1) > 0 || qps.exists(_.tables.lengthCompare(1) > 0)) {
436           logger.error("The query being executed requires multiple scans, which is not currently " +
437             "supported by GeoMesa. Your result set will be partially incomplete. " +
438             s"Query: ${filterToString(query.getFilter)}")
439         }
440         qps.head
441       }
442     } finally {
443       // make sure we reset the thread locals
444       JoinIndex.AllowJoinPlans.remove()
445     }
446   }

463   /**
464     * Converts the old 'index schema' into the appropriate index identifiers
465     *
466     * @param sft simple feature type
467     * @return
468     */
469   private def translateSchemaVersion(sft: SimpleFeatureType): Seq[IndexId] = {
470     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
471     lazy val docs =
472       "https://www.geomesa.org/documentation/stable/user/accumulo/jobs.html#updating-existing-data-to-the-latest-index-format"
473 
474     val geom = Option(sft.getGeomField)
475     val dtg = sft.getDtgField
476 
477     val id = IndexId(IdIndex.name, -1, Seq.empty, IndexMode.ReadWrite)
478     val z3 = for { g <- geom; d <- dtg } yield { IndexId(Z3Index.name, -1, Seq(g, d), IndexMode.ReadWrite) }
479     val xz3 = for { g <- geom; d <- dtg } yield { IndexId(XZ3Index.name, 1, Seq(g, d), IndexMode.ReadWrite) }
480     val z2 = geom.map(g => IndexId(Z2Index.name, -1, Seq(g), IndexMode.ReadWrite))
481     val xz2 = geom.map(g => IndexId(XZ2Index.name, 1, Seq(g), IndexMode.ReadWrite))
482     val attributes = sft.getAttributeDescriptors.asScala.flatMap { d =>
483       val index = d.getUserData.remove(AttributeOptions.OptIndex).asInstanceOf[String]
484       if (index == null || index.equalsIgnoreCase(IndexCoverage.NONE.toString) || index.equalsIgnoreCase("false")) {
485         Seq.empty
486       } else if (index.equalsIgnoreCase(IndexCoverage.FULL.toString)) {
487         Seq(IndexId(AttributeIndex.name, -1, Seq(d.getLocalName) ++ dtg, IndexMode.ReadWrite))
488       } else if (index.equalsIgnoreCase(IndexCoverage.JOIN.toString) || java.lang.Boolean.valueOf(index)) {
489         Seq(IndexId(JoinIndex.name, -1, Seq(d.getLocalName) ++ dtg, IndexMode.ReadWrite))
490       } else {
491         throw new IllegalStateException(s"Expected an index coverage or boolean but got: $index")
492       }
493     }
494 
495     // note: 10 was the last valid value for CURRENT_SCHEMA_VERSION, which is no longer used except
496     // to transition old schemas from the 1.2.5 era
497     val version = {
498       // noinspection ScalaDeprecation
499       val string = sft.getUserData.remove(DeprecatedSchemaVersionKey).asInstanceOf[String]
500       if (string != null) { string.toInt } else { 10 }
501     }
502     val indices: Seq[IndexId] = if (version > 8) {
503       // note: version 9 was never in a release
504       val zs = if (sft.isPoints) { z3.map(_.copy(version = 3)) ++ z2.map(_.copy(version = 2)) } else { xz3 ++ xz2 }
505       zs.toSeq ++ Seq(id.copy(version = 2)) ++ attributes.map(_.copy(version = 3))
506     } else if (version == 8) {
507       z3.map(_.copy(version = 2)).toSeq ++ z2.map(_.copy(version = 1)) ++
508           Seq(id.copy(version = 1)) ++ attributes.map(_.copy(version = 2))
509     } else if (version > 5) {
510       logger.warn("The GeoHash index is no longer supported. Some queries may take longer than normal. To " +
511           s"update your data to a newer format, see $docs")
512       val z = if (version == 7) { z3.map(_.copy(version = 2)) } else { z3.map(_.copy(version = 1)) }
513       z.toSeq ++ Seq(id.copy(version = 1)) ++ attributes.map(_.copy(version = 2))
514     } else {
515       throw new UnsupportedOperationException("This schema format is no longer supported. Please use " +
516           s"GeoMesa 1.2.6 to update you data to a newer format. For more information, see $docs")
517     }
518 
519     // noinspection deprecation
520     SimpleFeatureTypes.Configs.ENABLED_INDEX_OPTS.map(sft.getUserData.get).find(_ != null) match {
521       case None => indices
522       case Some(enabled) =>
523         // check for old index names
524         val e = enabled.toString.toLowerCase(Locale.US).split(",").map(_.trim).filterNot(_.isEmpty).map {
525           case "attr_idx" => AttributeIndex.name
526           case "records" => IdIndex.name
527           case i => i
528         }
