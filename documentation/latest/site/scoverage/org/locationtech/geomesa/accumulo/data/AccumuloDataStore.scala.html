107   override protected def transitionIndices(sft: SimpleFeatureType): Unit = {
108     // note: versions already correspond to accumulo index versions
109     val dtg = sft.getDtgField.toSeq
110     val geom = Option(sft.getGeomField).toSeq
111     val indices = sft.getIndices.flatMap {
112       case id if id.name == IdIndex.name  => Seq(id) // no update needed
113       case id if id.name == "records"     => Seq(id.copy(name = IdIndex.name))
114       case id if id.name == Z3Index.name  => Seq(id.copy(attributes = geom ++ dtg))
115       case id if id.name == XZ3Index.name => Seq(id.copy(attributes = geom ++ dtg))
116       case id if id.name == Z2Index.name  => Seq(id.copy(attributes = geom))
117       case id if id.name == XZ2Index.name => Seq(id.copy(attributes = geom))
118       case id if id.name == AttributeIndex.name =>
119         lazy val fields = if (id.version < 4) { dtg } else { geom ++ dtg }
120         sft.getAttributeDescriptors.asScala.flatMap { d =>
121           val index = d.getUserData.remove(AttributeOptions.OptIndex).asInstanceOf[String]
122           if (index == null || index.equalsIgnoreCase(IndexCoverage.NONE.toString) || index.equalsIgnoreCase("false")) {
123             Seq.empty
124           } else if (index.equalsIgnoreCase(IndexCoverage.FULL.toString)) {
125             Seq(id.copy(name = AttributeIndex.name, attributes = Seq(d.getLocalName) ++ fields))
126           } else if (index.equalsIgnoreCase(IndexCoverage.JOIN.toString) || java.lang.Boolean.valueOf(index)) {
127             Seq(id.copy(name = JoinIndex.name, attributes = Seq(d.getLocalName) ++ fields))
128           } else {
129             throw new IllegalStateException(s"Expected an index coverage or boolean but got: $index")
130           }
131         }
132     }
133     sft.setIndices(indices)
134   }

203   override protected def onSchemaCreated(sft: SimpleFeatureType): Unit = {
204     super.onSchemaCreated(sft)
205     if (sft.statsEnabled) {
206       // configure the stats combining iterator on the table for this sft
207       stats.configureStatCombiner(connector, sft)
208     }
209     sft.getFeatureExpiration.foreach {
210       case IngestTimeExpiration(ttl) =>
211         val tableOps = connector.tableOperations()
212         getAllIndexTableNames(sft.getTypeName).filter(tableOps.exists).foreach { table =>
213           AgeOffIterator.set(tableOps, table, sft, ttl)
214         }
215 
216       case FeatureTimeExpiration(dtg, _, ttl) =>
217         val tableOps = connector.tableOperations()
218         manager.indices(sft).foreach { index =>
219           val indexSft = index match {
220             case joinIndex: AttributeJoinIndex => joinIndex.indexSft
221             case _ => sft
222           }
223           DtgAgeOffIterator.set(tableOps, indexSft, index, ttl, dtg)
224         }
225 
226       case e =>
227         throw new IllegalArgumentException(s"Unexpected feature expiration: $e")
228     }

232   override protected def preSchemaUpdate(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
233     if (sft.getVisibilityLevel == VisibilityLevel.Attribute && sft.getAttributeCount > 255) {
234       throw new IllegalArgumentException("Attribute level visibility only supports up to 255 attributes")
235     }
236 
237     // check for attributes flagged 'index=join' and convert them to sft-level user data
238     sft.getAttributeDescriptors.asScala.foreach { d =>
239       val index = d.getUserData.get(AttributeOptions.OptIndex).asInstanceOf[String]
240       if (index != null && index.equalsIgnoreCase(IndexCoverage.JOIN.toString)) {
241         d.getUserData.remove(AttributeOptions.OptIndex) // remove it so it's not processed again
242         val fields = Seq(d.getLocalName) ++ Option(sft.getGeomField) ++ sft.getDtgField.filter(_ != d.getLocalName)
243         val attribute = IndexId(JoinIndex.name, JoinIndex.version, fields, IndexMode.ReadWrite)
244         val existing = sft.getIndices.map(GeoMesaFeatureIndex.identifier)
245         if (!existing.contains(GeoMesaFeatureIndex.identifier(attribute))) {
246           sft.setIndices(sft.getIndices :+ attribute)
247         }
248       }
249     }
250 
251     // check any previous age-off - previously age-off wasn't tied to the sft metadata
252     if (!sft.isFeatureExpirationEnabled && !previous.isFeatureExpirationEnabled) {
253       // explicitly set age-off in the feature type if found
254       val tableOps = connector.tableOperations()
255       val tables = getAllIndexTableNames(previous.getTypeName).filter(tableOps.exists)
256       val ageOff = tables.foldLeft[Option[FeatureExpiration]](None) { (res, table) =>
257         res.orElse(AgeOffIterator.expiry(tableOps, table))
258       }
259       val configured = ageOff.orElse {
260         tables.foldLeft[Option[FeatureExpiration]](None) { (res, table) =>
261           res.orElse(DtgAgeOffIterator.expiry(tableOps, previous, table))
262         }
263       }
264       configured.foreach(sft.setFeatureExpiration)
265     }
266 
267     super.preSchemaUpdate(sft, previous)
268   }

271     super.onSchemaUpdated(sft, previous)

282     val tables = getAllIndexTableNames(sft.getTypeName).filter(tableOps.exists)
283 
284     if (previous.isVisibilityRequired != sft.isVisibilityRequired) {
285       previousTables.foreach(VisibilityIterator.clear(tableOps, _))
286       if (sft.isVisibilityRequired) {
287         tables.foreach(VisibilityIterator.set(tableOps, _))
288       }
289     }
290 
291     previousTables.foreach { table =>
292       AgeOffIterator.clear(tableOps, table)
293       DtgAgeOffIterator.clear(tableOps, table)
294     }
295 
296     sft.getFeatureExpiration.foreach {
297       case IngestTimeExpiration(ttl) =>
298         tables.foreach(AgeOffIterator.set(tableOps, _, sft, ttl))
299 
300       case FeatureTimeExpiration(dtg, _, ttl) =>
301         manager.indices(sft).foreach { index =>
302           val indexSft = index match {
303             case joinIndex: AttributeJoinIndex => joinIndex.indexSft
304             case _ => sft
305           }
306           DtgAgeOffIterator.set(tableOps, indexSft, index, ttl, dtg)
307         }
308 
309       case e => throw new IllegalArgumentException(s"Unexpected feature expiration: $e")
310     }

321   override def getSchema(typeName: String): SimpleFeatureType = {
322     var sft = super.getSchema(typeName)
323     if (sft == null) {
324       // check for old-style metadata and re-write it if necessary
325       if (oldMetadata.getFeatureTypes.contains(typeName)) {
326         val lock = acquireCatalogLock()
327         try {
328           if (oldMetadata.getFeatureTypes.contains(typeName)) {
329             oldMetadata.migrate(typeName)
330             new SingleRowAccumuloMetadata[Stat](stats.metadata).migrate(typeName)
331           }
332         } finally {
333           lock.release()
334         }
335         sft = super.getSchema(typeName)
336       }
337     }
338     if (sft != null) {
339       // back compatible check for index versions
340       if (sft.getIndices.isEmpty) {
341         // make the sft temporarily mutable so we can update the keys
342         sft = SimpleFeatureTypes.mutable(sft)
343         // back compatible check if user data wasn't encoded with the sft
344         if (!sft.getUserData.containsKey(AccumuloDataStore.DeprecatedSchemaVersionKey)) {
345           metadata.read(typeName, "dtgfield").foreach(sft.setDtgField)
346           sft.getUserData.put(AccumuloDataStore.DeprecatedSchemaVersionKey,
347             metadata.readRequired(typeName, GeoMesaMetadata.VersionKey))
348 
349           // if no data is written, we default to 'false' in order to support old tables
350           if (metadata.read(typeName, "tables.sharing").exists(_.toBoolean)) {
351             sft.getUserData.put(SimpleFeatureTypes.Configs.TableSharing, "true")
352             // use schema id if available or fall back to old type name for backwards compatibility
353             val prefix = metadata.read(typeName, "id").getOrElse(s"${sft.getTypeName}~")
354             sft.getUserData.put(SimpleFeatureTypes.InternalConfigs.TableSharingPrefix, prefix)
355           }
356           // noinspection deprecation
357           SimpleFeatureTypes.Configs.ENABLED_INDEX_OPTS.foreach { i =>
358             metadata.read(typeName, i).foreach(e => sft.getUserData.put(SimpleFeatureTypes.Configs.EnabledIndices, e))
359           }
360         }
361 
362         // set the enabled indices
363         sft.setIndices(AccumuloDataStore.translateSchemaVersion(sft))
364 
365         // store the metadata and reload the sft again to validate indices
366         val encoded = SimpleFeatureTypes.encodeType(sft, includeUserData = true)
367         metadata.insert(typeName, GeoMesaMetadata.AttributesKey, encoded)
368         sft = super.getSchema(typeName)
369       }
370 
371       // back compatibility check for stat configuration
372       if (sft.statsEnabled && metadata.read(typeName, GeoMesaMetadata.StatsGenerationKey).isEmpty) {
373         // configure the stats combining iterator - we only use this key for older data stores
374         val configuredKey = "stats-configured"
375         if (!metadata.read(typeName, configuredKey).contains("true")) {
376           val lock = acquireCatalogLock()
377           try {
378             if (!metadata.read(typeName, configuredKey, cache = false).contains("true")) {
379               stats.configureStatCombiner(connector, sft)
380               metadata.insert(typeName, configuredKey, "true")
381             }
382           } finally {
383             lock.release()
384           }
385         }
386         // kick off asynchronous stats run for the existing data - this will set the stat date
387         // this may get triggered more than once, but should only run one time
388         val statsRunner = new StatsRunner(this)
389         statsRunner.submit(sft)
390         statsRunner.close()
391       }
392     }
393 
394     sft
395   }

412   /**
413     * Converts the old 'index schema' into the appropriate index identifiers
414     *
415     * @param sft simple feature type
416     * @return
417     */
418   private def translateSchemaVersion(sft: SimpleFeatureType): Seq[IndexId] = {
419     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
420     lazy val docs =
421       "http://www.geomesa.org/documentation/user/jobs.html#updating-existing-data-to-the-latest-index-format"
422 
423     val geom = Option(sft.getGeomField)
424     val dtg = sft.getDtgField
425 
426     val id = IndexId(IdIndex.name, -1, Seq.empty, IndexMode.ReadWrite)
427     val z3 = for { g <- geom; d <- dtg } yield { IndexId(Z3Index.name, -1, Seq(g, d), IndexMode.ReadWrite) }
428     val xz3 = for { g <- geom; d <- dtg } yield { IndexId(XZ3Index.name, 1, Seq(g, d), IndexMode.ReadWrite) }
429     val z2 = geom.map(g => IndexId(Z2Index.name, -1, Seq(g), IndexMode.ReadWrite))
430     val xz2 = geom.map(g => IndexId(XZ2Index.name, 1, Seq(g), IndexMode.ReadWrite))
431     val attributes = sft.getAttributeDescriptors.asScala.flatMap { d =>
432       val index = d.getUserData.remove(AttributeOptions.OptIndex).asInstanceOf[String]
433       if (index == null || index.equalsIgnoreCase(IndexCoverage.NONE.toString) || index.equalsIgnoreCase("false")) {
434         Seq.empty
435       } else if (index.equalsIgnoreCase(IndexCoverage.FULL.toString)) {
436         Seq(IndexId(AttributeIndex.name, -1, Seq(d.getLocalName) ++ dtg, IndexMode.ReadWrite))
437       } else if (index.equalsIgnoreCase(IndexCoverage.JOIN.toString) || java.lang.Boolean.valueOf(index)) {
438         Seq(IndexId(JoinIndex.name, -1, Seq(d.getLocalName) ++ dtg, IndexMode.ReadWrite))
439       } else {
440         throw new IllegalStateException(s"Expected an index coverage or boolean but got: $index")
441       }
442     }
443 
444     // note: 10 was the last valid value for CURRENT_SCHEMA_VERSION, which is no longer used except
445     // to transition old schemas from the 1.2.5 era
446     val version = {
447       // noinspection ScalaDeprecation
448       val string = sft.getUserData.remove(DeprecatedSchemaVersionKey).asInstanceOf[String]
449       if (string != null) { string.toInt } else { 10 }
450     }
451     val indices: Seq[IndexId] = if (version > 8) {
452       // note: version 9 was never in a release
453       val zs = if (sft.isPoints) { z3.map(_.copy(version = 3)) ++ z2.map(_.copy(version = 2)) } else { xz3 ++ xz2 }
454       zs.toSeq ++ Seq(id.copy(version = 2)) ++ attributes.map(_.copy(version = 3))
455     } else if (version == 8) {
456       z3.map(_.copy(version = 2)).toSeq ++ z2.map(_.copy(version = 1)) ++
457           Seq(id.copy(version = 1)) ++ attributes.map(_.copy(version = 2))
458     } else if (version > 5) {
459       logger.warn("The GeoHash index is no longer supported. Some queries may take longer than normal. To " +
460           s"update your data to a newer format, see $docs")
461       val z = if (version == 7) { z3.map(_.copy(version = 2)) } else { z3.map(_.copy(version = 1)) }
462       z.toSeq ++ Seq(id.copy(version = 1)) ++ attributes.map(_.copy(version = 2))
463     } else {
464       throw new NotImplementedError("This schema format is no longer supported. Please use " +
465           s"GeoMesa 1.2.6 to update you data to a newer format. For more information, see $docs")
466     }
467 
468     // noinspection deprecation
469     SimpleFeatureTypes.Configs.ENABLED_INDEX_OPTS.map(sft.getUserData.get).find(_ != null) match {
470       case None => indices
471       case Some(enabled) =>
472         // check for old index names
473         val e = enabled.toString.toLowerCase(Locale.US).split(",").map(_.trim).filterNot(_.isEmpty).map {
474           case "attr_idx" => AttributeIndex.name
475           case "records" => IdIndex.name
476           case i => i
477         }
478         indices.filter(i => e.contains(i.name))
479     }
480   }
