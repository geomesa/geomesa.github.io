101   override protected def transitionIndices(sft: SimpleFeatureType): Unit = {
102     // note: versions already correspond to accumulo index versions
103     val dtg = sft.getDtgField.toSeq
104     val geom = Option(sft.getGeomField).toSeq
105     val indices = sft.getIndices.flatMap {
106       case id if id.name == IdIndex.name  => Seq(id) // no update needed
107       case id if id.name == "records"     => Seq(id.copy(name = IdIndex.name))
108       case id if id.name == Z3Index.name  => Seq(id.copy(attributes = geom ++ dtg))
109       case id if id.name == XZ3Index.name => Seq(id.copy(attributes = geom ++ dtg))
110       case id if id.name == Z2Index.name  => Seq(id.copy(attributes = geom))
111       case id if id.name == XZ2Index.name => Seq(id.copy(attributes = geom))
112       case id if id.name == AttributeIndex.name =>
113         lazy val fields = if (id.version < 4) { dtg } else { geom ++ dtg }
114         sft.getAttributeDescriptors.asScala.flatMap { d =>
115           val index = d.getUserData.remove(AttributeOptions.OptIndex).asInstanceOf[String]
116           if (index == null || index.equalsIgnoreCase(IndexCoverage.NONE.toString) || index.equalsIgnoreCase("false")) {
117             Seq.empty
118           } else if (index.equalsIgnoreCase(IndexCoverage.FULL.toString)) {
119             Seq(id.copy(name = AttributeIndex.name, attributes = Seq(d.getLocalName) ++ fields))
120           } else if (index.equalsIgnoreCase(IndexCoverage.JOIN.toString) || java.lang.Boolean.valueOf(index)) {
121             Seq(id.copy(name = JoinIndex.name, attributes = Seq(d.getLocalName) ++ fields))
122           } else {
123             throw new IllegalStateException(s"Expected an index coverage or boolean but got: $index")
124           }
125         }
126     }
127     sft.setIndices(indices)
128   }

203   override protected def onSchemaCreated(sft: SimpleFeatureType): Unit = {
204     super.onSchemaCreated(sft)
205     if (sft.statsEnabled) {
206       // configure the stats combining iterator on the table for this sft
207       adapter.ensureTableExists(stats.metadata.table)
208       stats.configureStatCombiner(connector, sft)
209     }
210     sft.getFeatureExpiration.foreach {
211       case IngestTimeExpiration(ttl) =>
212         val tableOps = connector.tableOperations()
213         getAllIndexTableNames(sft.getTypeName).filter(tableOps.exists).foreach { table =>
214           AgeOffIterator.set(tableOps, table, sft, ttl)
215         }
216 
217       case FeatureTimeExpiration(dtg, _, ttl) =>
218         val tableOps = connector.tableOperations()
219         manager.indices(sft).foreach { index =>
220           val indexSft = index match {
221             case joinIndex: AttributeJoinIndex => joinIndex.indexSft
222             case _ => sft
223           }
224           DtgAgeOffIterator.set(tableOps, indexSft, index, ttl, dtg)
225         }
226 
227       case e =>
228         throw new IllegalArgumentException(s"Unexpected feature expiration: $e")
229     }

233   override protected def preSchemaUpdate(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
234     if (sft.getVisibilityLevel == VisibilityLevel.Attribute && sft.getAttributeCount > 255) {
235       throw new IllegalArgumentException("Attribute level visibility only supports up to 255 attributes")
236     }
237 
238     // check for attributes flagged 'index=join' and convert them to sft-level user data
239     sft.getAttributeDescriptors.asScala.foreach { d =>
240       val index = d.getUserData.get(AttributeOptions.OptIndex).asInstanceOf[String]
241       if (index != null && index.equalsIgnoreCase(IndexCoverage.JOIN.toString)) {
242         d.getUserData.remove(AttributeOptions.OptIndex) // remove it so it's not processed again
243         val fields = Seq(d.getLocalName) ++ Option(sft.getGeomField) ++ sft.getDtgField.filter(_ != d.getLocalName)
244         val attribute = IndexId(JoinIndex.name, JoinIndex.version, fields, IndexMode.ReadWrite)
245         val existing = sft.getIndices.map(GeoMesaFeatureIndex.identifier)
246         if (!existing.contains(GeoMesaFeatureIndex.identifier(attribute))) {
247           sft.setIndices(sft.getIndices :+ attribute)
248         }
249       }
250     }
251 
252     // check any previous age-off - previously age-off wasn't tied to the sft metadata
253     if (!sft.isFeatureExpirationEnabled && !previous.isFeatureExpirationEnabled) {
254       // explicitly set age-off in the feature type if found
255       val tableOps = connector.tableOperations()
256       val tables = getAllIndexTableNames(previous.getTypeName).filter(tableOps.exists)
257       val ageOff = tables.foldLeft[Option[FeatureExpiration]](None) { (res, table) =>
258         res.orElse(AgeOffIterator.expiry(tableOps, table))
259       }
260       val configured = ageOff.orElse {
261         tables.foldLeft[Option[FeatureExpiration]](None) { (res, table) =>
262           res.orElse(DtgAgeOffIterator.expiry(tableOps, previous, table))
263         }
264       }
265       configured.foreach(sft.setFeatureExpiration)
266     }
267 
268     super.preSchemaUpdate(sft, previous)
269   }

272     super.onSchemaUpdated(sft, previous)

274     if (previous.statsEnabled) {
275       stats.removeStatCombiner(connector, previous)

277     if (sft.statsEnabled) {
278       adapter.ensureTableExists(stats.metadata.table)

280     }
281 
282     val tableOps = connector.tableOperations()
283     val previousTables = getAllIndexTableNames(previous.getTypeName).filter(tableOps.exists)
284     val tables = getAllIndexTableNames(sft.getTypeName).filter(tableOps.exists)
285 
286     if (previous.isVisibilityRequired != sft.isVisibilityRequired) {
287       previousTables.foreach(VisibilityIterator.clear(tableOps, _))
288       if (sft.isVisibilityRequired) {
289         tables.foreach(VisibilityIterator.set(tableOps, _))
290       }
291     }
292 
293     previousTables.foreach { table =>
294       AgeOffIterator.clear(tableOps, table)
295       DtgAgeOffIterator.clear(tableOps, table)
296     }
297 
298     sft.getFeatureExpiration.foreach {
299       case IngestTimeExpiration(ttl) =>
300         tables.foreach(AgeOffIterator.set(tableOps, _, sft, ttl))
301 
302       case FeatureTimeExpiration(dtg, _, ttl) =>
303         manager.indices(sft).foreach { index =>
304           val indexSft = index match {
305             case joinIndex: AttributeJoinIndex => joinIndex.indexSft
306             case _ => sft
307           }
308           DtgAgeOffIterator.set(tableOps, indexSft, index, ttl, dtg)
309         }
310 
311       case e => throw new IllegalArgumentException(s"Unexpected feature expiration: $e")
312     }

323   override def getSchema(typeName: String): SimpleFeatureType = {
324     var sft = super.getSchema(typeName)
325     if (sft == null) {
326       // check for old-style metadata and re-write it if necessary
327       if (oldMetadata.getFeatureTypes.contains(typeName)) {
328         val lock = acquireCatalogLock()
329         try {
330           if (oldMetadata.getFeatureTypes.contains(typeName)) {
331             oldMetadata.migrate(typeName)
332             new SingleRowAccumuloMetadata[Stat](stats.metadata).migrate(typeName)
333           }
334         } finally {
335           lock.close()
336         }
337         sft = super.getSchema(typeName)
338       }
339     }
340     if (sft != null) {
341       // back compatible check for index versions
342       if (sft.getIndices.isEmpty) {
343         // make the sft temporarily mutable so we can update the keys
344         sft = SimpleFeatureTypes.mutable(sft)
345         // back compatible check if user data wasn't encoded with the sft
346         if (!sft.getUserData.containsKey(AccumuloDataStore.DeprecatedSchemaVersionKey)) {
347           metadata.read(typeName, "dtgfield").foreach(sft.setDtgField)
348           sft.getUserData.put(AccumuloDataStore.DeprecatedSchemaVersionKey,
349             metadata.readRequired(typeName, GeoMesaMetadata.VersionKey))
350 
351           // if no data is written, we default to 'false' in order to support old tables
352           if (metadata.read(typeName, "tables.sharing").exists(_.toBoolean)) {
353             sft.getUserData.put(SimpleFeatureTypes.Configs.TableSharing, "true")
354             // use schema id if available or fall back to old type name for backwards compatibility
355             val prefix = metadata.read(typeName, "id").getOrElse(s"${sft.getTypeName}~")
356             sft.getUserData.put(SimpleFeatureTypes.InternalConfigs.TableSharingPrefix, prefix)
357           }
358           // noinspection deprecation
359           SimpleFeatureTypes.Configs.ENABLED_INDEX_OPTS.foreach { i =>
360             metadata.read(typeName, i).foreach(e => sft.getUserData.put(SimpleFeatureTypes.Configs.EnabledIndices, e))
361           }
362         }
363 
364         // set the enabled indices
365         sft.setIndices(AccumuloDataStore.translateSchemaVersion(sft))
366 
367         // store the metadata and reload the sft again to validate indices
368         val encoded = SimpleFeatureTypes.encodeType(sft, includeUserData = true)
369         metadata.insert(typeName, GeoMesaMetadata.AttributesKey, encoded)
370         sft = super.getSchema(typeName)
371       }
372 
373       // back compatibility check for stat configuration
374       if (sft.statsEnabled && metadata.read(typeName, GeoMesaMetadata.StatsGenerationKey).isEmpty) {
375         // configure the stats combining iterator - we only use this key for older data stores
376         val configuredKey = "stats-configured"
377         if (!metadata.read(typeName, configuredKey).contains("true")) {
378           val lock = acquireCatalogLock()
379           try {
380             if (!metadata.read(typeName, configuredKey, cache = false).contains("true")) {
381               adapter.ensureTableExists(stats.metadata.table)
382               stats.configureStatCombiner(connector, sft)
383               metadata.insert(typeName, configuredKey, "true")
384             }
385           } finally {
386             lock.close()
387           }
388         }
389         // kick off asynchronous stats run for the existing data - this will set the stat date
390         // this may get triggered more than once, but should only run one time
391         val statsRunner = new StatsRunner(this)
392         statsRunner.submit(sft)
393         statsRunner.close()
394       }
395     }
396 
397     sft
398   }

415   /**
416     * Converts the old 'index schema' into the appropriate index identifiers
417     *
418     * @param sft simple feature type
419     * @return
420     */
421   private def translateSchemaVersion(sft: SimpleFeatureType): Seq[IndexId] = {
422     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
423     lazy val docs =
424       "http://www.geomesa.org/documentation/user/jobs.html#updating-existing-data-to-the-latest-index-format"
425 
426     val geom = Option(sft.getGeomField)
427     val dtg = sft.getDtgField
428 
429     val id = IndexId(IdIndex.name, -1, Seq.empty, IndexMode.ReadWrite)
430     val z3 = for { g <- geom; d <- dtg } yield { IndexId(Z3Index.name, -1, Seq(g, d), IndexMode.ReadWrite) }
431     val xz3 = for { g <- geom; d <- dtg } yield { IndexId(XZ3Index.name, 1, Seq(g, d), IndexMode.ReadWrite) }
432     val z2 = geom.map(g => IndexId(Z2Index.name, -1, Seq(g), IndexMode.ReadWrite))
433     val xz2 = geom.map(g => IndexId(XZ2Index.name, 1, Seq(g), IndexMode.ReadWrite))
434     val attributes = sft.getAttributeDescriptors.asScala.flatMap { d =>
435       val index = d.getUserData.remove(AttributeOptions.OptIndex).asInstanceOf[String]
436       if (index == null || index.equalsIgnoreCase(IndexCoverage.NONE.toString) || index.equalsIgnoreCase("false")) {
437         Seq.empty
438       } else if (index.equalsIgnoreCase(IndexCoverage.FULL.toString)) {
439         Seq(IndexId(AttributeIndex.name, -1, Seq(d.getLocalName) ++ dtg, IndexMode.ReadWrite))
440       } else if (index.equalsIgnoreCase(IndexCoverage.JOIN.toString) || java.lang.Boolean.valueOf(index)) {
441         Seq(IndexId(JoinIndex.name, -1, Seq(d.getLocalName) ++ dtg, IndexMode.ReadWrite))
442       } else {
443         throw new IllegalStateException(s"Expected an index coverage or boolean but got: $index")
444       }
445     }
446 
447     // note: 10 was the last valid value for CURRENT_SCHEMA_VERSION, which is no longer used except
448     // to transition old schemas from the 1.2.5 era
449     val version = {
450       // noinspection ScalaDeprecation
451       val string = sft.getUserData.remove(DeprecatedSchemaVersionKey).asInstanceOf[String]
452       if (string != null) { string.toInt } else { 10 }
453     }
454     val indices: Seq[IndexId] = if (version > 8) {
455       // note: version 9 was never in a release
456       val zs = if (sft.isPoints) { z3.map(_.copy(version = 3)) ++ z2.map(_.copy(version = 2)) } else { xz3 ++ xz2 }
457       zs.toSeq ++ Seq(id.copy(version = 2)) ++ attributes.map(_.copy(version = 3))
458     } else if (version == 8) {
459       z3.map(_.copy(version = 2)).toSeq ++ z2.map(_.copy(version = 1)) ++
460           Seq(id.copy(version = 1)) ++ attributes.map(_.copy(version = 2))
461     } else if (version > 5) {
462       logger.warn("The GeoHash index is no longer supported. Some queries may take longer than normal. To " +
463           s"update your data to a newer format, see $docs")
464       val z = if (version == 7) { z3.map(_.copy(version = 2)) } else { z3.map(_.copy(version = 1)) }
465       z.toSeq ++ Seq(id.copy(version = 1)) ++ attributes.map(_.copy(version = 2))
466     } else {
467       throw new NotImplementedError("This schema format is no longer supported. Please use " +
468           s"GeoMesa 1.2.6 to update you data to a newer format. For more information, see $docs")
469     }
470 
471     // noinspection deprecation
472     SimpleFeatureTypes.Configs.ENABLED_INDEX_OPTS.map(sft.getUserData.get).find(_ != null) match {
473       case None => indices
474       case Some(enabled) =>
475         // check for old index names
476         val e = enabled.toString.toLowerCase(Locale.US).split(",").map(_.trim).filterNot(_.isEmpty).map {
477           case "attr_idx" => AttributeIndex.name
478           case "records" => IdIndex.name
479           case i => i
480         }
481         indices.filter(i => e.contains(i.name))
482     }
483   }
