104   override protected def transitionIndices(sft: SimpleFeatureType): Unit = {
105     // note: versions already correspond to accumulo index versions
106     val dtg = sft.getDtgField.toSeq
107     val geom = Option(sft.getGeomField).toSeq
108     val indices = sft.getIndices.flatMap {
109       case id if id.name == IdIndex.name  => Seq(id) // no update needed
110       case id if id.name == "records"     => Seq(id.copy(name = IdIndex.name))
111       case id if id.name == Z3Index.name  => Seq(id.copy(attributes = geom ++ dtg))
112       case id if id.name == XZ3Index.name => Seq(id.copy(attributes = geom ++ dtg))
113       case id if id.name == Z2Index.name  => Seq(id.copy(attributes = geom))
114       case id if id.name == XZ2Index.name => Seq(id.copy(attributes = geom))
115       case id if id.name == AttributeIndex.name =>
116         lazy val fields = if (id.version < 4) { dtg } else { geom ++ dtg }
117         sft.getAttributeDescriptors.asScala.flatMap { d =>
118           val index = d.getUserData.remove(AttributeOptions.OptIndex).asInstanceOf[String]
119           if (index == null || index.equalsIgnoreCase(IndexCoverage.NONE.toString) || index.equalsIgnoreCase("false")) {
120             Seq.empty
121           } else if (index.equalsIgnoreCase(IndexCoverage.FULL.toString)) {
122             Seq(id.copy(name = AttributeIndex.name, attributes = Seq(d.getLocalName) ++ fields))
123           } else if (index.equalsIgnoreCase(IndexCoverage.JOIN.toString) || java.lang.Boolean.valueOf(index)) {
124             Seq(id.copy(name = JoinIndex.name, attributes = Seq(d.getLocalName) ++ fields))
125           } else {
126             throw new IllegalStateException(s"Expected an index coverage or boolean but got: $index")
127           }
128         }
129     }
130     sft.setIndices(indices)
131   }

153   override protected def preSchemaCreate(sft: SimpleFeatureType): Unit = {
154     import org.locationtech.geomesa.index.conf.SchemaProperties.ValidateDistributedClasspath
155 
156     // call super first so that user data keys are updated
157     super.preSchemaCreate(sft)
158 
159     def getNamespace(prefix: String): String = prefix.indexOf('.') match {
160       case -1 => ""
161       case i  => prefix.substring(0, i)
162     }
163 
164     val prefixes = Seq(config.catalog) ++ sft.getIndices.flatMap(i => sft.getTablePrefix(i.name))
165     prefixes.map(getNamespace).distinct.foreach { namespace =>
166       if (namespace.nonEmpty) {
167         adapter.ensureNamespaceExists(namespace)
168       }
169       // validate that the accumulo runtime is available
170       val canLoad = connector.namespaceOperations().testClassLoad(namespace,
171         classOf[ProjectVersionIterator].getName, classOf[SortedKeyValueIterator[_, _]].getName)
172 
173       if (!canLoad) {
174         val msg = s"Could not load GeoMesa distributed code from the Accumulo classpath"
175         logger.error(s"$msg for catalog ${config.catalog}")
176         if (ValidateDistributedClasspath.toBoolean.contains(true)) {
177           val nsMsg = if (namespace.isEmpty) { "" } else { s" for the namespace '$namespace'" }
178           throw new RuntimeException(s"$msg. You may override this check by setting the system property " +
179             s"'${ValidateDistributedClasspath.property}=false'. Otherwise, please verify that the appropriate " +
180             s"JARs are installed$nsMsg - see https://www.geomesa.org/documentation/stable/user/accumulo/install.html" +
181             "#installing-the-accumulo-distributed-runtime-library")
182         }
183       }
184     }
185 
186     if (sft.getVisibilityLevel == VisibilityLevel.Attribute && sft.getAttributeCount > 255) {
187       throw new IllegalArgumentException("Attribute level visibility only supports up to 255 attributes")
188     }
189 
190     sft.getDtgField.foreach { dtg =>
191       if (sft.getIndices.exists(i => i.name == JoinIndex.name && i.attributes.headOption.contains(dtg))) {
192         if (!GeoMesaSchemaValidator.declared(sft, OverrideDtgJoin)) {
193           throw new IllegalArgumentException("Trying to create a schema with a partial (join) attribute index " +
194               s"on the default date field '$dtg'. This may cause whole-world queries with time bounds to be much " +
195               "slower. If this is intentional, you may override this check by putting Boolean.TRUE into the " +
196               s"SimpleFeatureType user data under the key '$OverrideDtgJoin' before calling createSchema, or by " +
197               s"setting the system property '$OverrideDtgJoin' to 'true'. Otherwise, please either specify a " +
198               "full attribute index or remove it entirely.")
199         }
200       }
201     }
202   }

204   override protected def onSchemaCreated(sft: SimpleFeatureType): Unit = {
205     super.onSchemaCreated(sft)
206     if (sft.statsEnabled) {
207       // configure the stats combining iterator on the table for this sft
208       adapter.ensureTableExists(stats.metadata.table)
209       stats.configureStatCombiner(connector, sft)
210     }
211     sft.getFeatureExpiration.foreach {
212       case IngestTimeExpiration(ttl) =>
213         val tableOps = connector.tableOperations()
214         getAllIndexTableNames(sft.getTypeName).filter(tableOps.exists).foreach { table =>
215           AgeOffIterator.set(tableOps, table, sft, ttl)
216         }
217 
218       case FeatureTimeExpiration(dtg, _, ttl) =>
219         val tableOps = connector.tableOperations()
220         manager.indices(sft).foreach { index =>
221           val indexSft = index match {
222             case joinIndex: AttributeJoinIndex => joinIndex.indexSft
223             case _ => sft
224           }
225           DtgAgeOffIterator.set(tableOps, indexSft, index, ttl, dtg)
226         }
227 
228       case e =>
229         throw new IllegalArgumentException(s"Unexpected feature expiration: $e")
230     }
231   }

233   @throws(classOf[IllegalArgumentException])
234   override protected def preSchemaUpdate(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
235     if (sft.getVisibilityLevel == VisibilityLevel.Attribute && sft.getAttributeCount > 255) {
236       throw new IllegalArgumentException("Attribute level visibility only supports up to 255 attributes")
237     }
238 
239     // check for attributes flagged 'index=join' and convert them to sft-level user data
240     sft.getAttributeDescriptors.asScala.foreach { d =>
241       val index = d.getUserData.get(AttributeOptions.OptIndex).asInstanceOf[String]
242       if (index != null && index.equalsIgnoreCase(IndexCoverage.JOIN.toString)) {
243         d.getUserData.remove(AttributeOptions.OptIndex) // remove it so it's not processed again
244         val fields = Seq(d.getLocalName) ++ Option(sft.getGeomField) ++ sft.getDtgField.filter(_ != d.getLocalName)
245         val attribute = IndexId(JoinIndex.name, JoinIndex.version, fields, IndexMode.ReadWrite)
246         val existing = sft.getIndices.map(GeoMesaFeatureIndex.identifier)
247         if (!existing.contains(GeoMesaFeatureIndex.identifier(attribute))) {
248           sft.setIndices(sft.getIndices :+ attribute)
249         }
250       }
251     }
252 
253     // check any previous age-off - previously age-off wasn't tied to the sft metadata
254     if (!sft.isFeatureExpirationEnabled && !previous.isFeatureExpirationEnabled) {
255       // explicitly set age-off in the feature type if found
256       val tableOps = connector.tableOperations()
257       val tables = getAllIndexTableNames(previous.getTypeName).filter(tableOps.exists)
258       val ageOff = tables.foldLeft[Option[FeatureExpiration]](None) { (res, table) =>
259         res.orElse(AgeOffIterator.expiry(tableOps, table))
260       }
261       val configured = ageOff.orElse {
262         tables.foldLeft[Option[FeatureExpiration]](None) { (res, table) =>
263           res.orElse(DtgAgeOffIterator.expiry(tableOps, previous, table))
264         }
265       }
266       configured.foreach(sft.setFeatureExpiration)
267     }
268 
269     super.preSchemaUpdate(sft, previous)
270   }

272   override protected def onSchemaUpdated(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
273     super.onSchemaUpdated(sft, previous)
274 
275     if (previous.statsEnabled) {
276       stats.removeStatCombiner(connector, previous)
277     }
278     if (sft.statsEnabled) {
279       adapter.ensureTableExists(stats.metadata.table)
280       stats.configureStatCombiner(connector, sft)
281     }
282 
283     val tableOps = connector.tableOperations()
284     val previousTables = getAllIndexTableNames(previous.getTypeName).filter(tableOps.exists)
285     val tables = getAllIndexTableNames(sft.getTypeName).filter(tableOps.exists)
286 
287     if (previous.isVisibilityRequired != sft.isVisibilityRequired) {
288       previousTables.foreach(VisibilityIterator.clear(tableOps, _))
289       if (sft.isVisibilityRequired) {
290         tables.foreach(VisibilityIterator.set(tableOps, _))
291       }
292     }
293 
294     previousTables.foreach { table =>
295       AgeOffIterator.clear(tableOps, table)
296       DtgAgeOffIterator.clear(tableOps, table)
297     }
298 
299     sft.getFeatureExpiration.foreach {
300       case IngestTimeExpiration(ttl) =>
301         tables.foreach(AgeOffIterator.set(tableOps, _, sft, ttl))
302 
303       case FeatureTimeExpiration(dtg, _, ttl) =>
304         manager.indices(sft).foreach { index =>
305           val indexSft = index match {
306             case joinIndex: AttributeJoinIndex => joinIndex.indexSft
307             case _ => sft
308           }
309           DtgAgeOffIterator.set(tableOps, indexSft, index, ttl, dtg)
310         }
311 
312       case e => throw new IllegalArgumentException(s"Unexpected feature expiration: $e")
313     }
314   }

323   // noinspection ScalaDeprecation
324   override def getSchema(typeName: String): SimpleFeatureType = {
325     var sft = super.getSchema(typeName)
326     if (sft == null) {
327       // check for old-style metadata and re-write it if necessary
328       if (oldMetadata.getFeatureTypes.contains(typeName)) {
329         val lock = acquireCatalogLock()
330         try {
331           if (oldMetadata.getFeatureTypes.contains(typeName)) {
332             oldMetadata.migrate(typeName)
333             new SingleRowAccumuloMetadata[Stat](stats.metadata).migrate(typeName)
334           }
335         } finally {
336           lock.close()
337         }
338         sft = super.getSchema(typeName)
339       }
340     }
341     if (sft != null) {
342       // back compatible check for index versions
343       if (sft.getIndices.isEmpty) {
344         // make the sft temporarily mutable so we can update the keys
345         sft = SimpleFeatureTypes.mutable(sft)
346         // back compatible check if user data wasn't encoded with the sft
347         if (!sft.getUserData.containsKey(AccumuloDataStore.DeprecatedSchemaVersionKey)) {
348           metadata.read(typeName, "dtgfield").foreach(sft.setDtgField)
349           sft.getUserData.put(AccumuloDataStore.DeprecatedSchemaVersionKey,
350             metadata.readRequired(typeName, GeoMesaMetadata.VersionKey))
351 
352           // if no data is written, we default to 'false' in order to support old tables
353           if (metadata.read(typeName, "tables.sharing").exists(_.toBoolean)) {
354             sft.getUserData.put(SimpleFeatureTypes.Configs.TableSharing, "true")
355             // use schema id if available or fall back to old type name for backwards compatibility
356             val prefix = metadata.read(typeName, "id").getOrElse(s"${sft.getTypeName}~")
357             sft.getUserData.put(SimpleFeatureTypes.InternalConfigs.TableSharingPrefix, prefix)
358           }
359           // noinspection deprecation
360           SimpleFeatureTypes.Configs.ENABLED_INDEX_OPTS.foreach { i =>
361             metadata.read(typeName, i).foreach(e => sft.getUserData.put(SimpleFeatureTypes.Configs.EnabledIndices, e))
362           }
363         }
364 
365         // set the enabled indices
366         sft.setIndices(AccumuloDataStore.translateSchemaVersion(sft))
367 
368         // store the metadata and reload the sft again to validate indices
369         val encoded = SimpleFeatureTypes.encodeType(sft, includeUserData = true)
370         metadata.insert(typeName, GeoMesaMetadata.AttributesKey, encoded)
371         sft = super.getSchema(typeName)
372       }
373 
374       // back compatibility check for stat configuration
375       if (sft.statsEnabled && metadata.read(typeName, GeoMesaMetadata.StatsGenerationKey).isEmpty) {
376         // configure the stats combining iterator - we only use this key for older data stores
377         val configuredKey = "stats-configured"
378         if (!metadata.read(typeName, configuredKey).contains("true")) {
379           val lock = acquireCatalogLock()
380           try {
381             if (!metadata.read(typeName, configuredKey, cache = false).contains("true")) {
382               adapter.ensureTableExists(stats.metadata.table)
383               stats.configureStatCombiner(connector, sft)
384               metadata.insert(typeName, configuredKey, "true")
385             }
386           } finally {
387             lock.close()
388           }
389         }
390         // kick off asynchronous stats run for the existing data - this will set the stat date
391         // this may get triggered more than once, but should only run one time
392         val statsRunner = new StatsRunner(this)
393         statsRunner.submit(sft)
394         statsRunner.close()
395       }
396     }
397 
398     sft
399   }

403   /**
404    * Gets a query plan that can be satisfied via AccumuloInputFormat - e.g. only 1 table and configuration.
405    *
406    * @param query query
407    * @return
408    */
409   def getSingleQueryPlan(query: Query): AccumuloQueryPlan = {
410     // disable join plans as those have multiple tables
411     JoinIndex.AllowJoinPlans.set(false)
412 
413     try {
414       lazy val fallbackIndex = {
415         val schema = getSchema(query.getTypeName)
416         manager.indices(schema, IndexMode.Read).headOption.getOrElse {
417           throw new IllegalStateException(s"Schema '${schema.getTypeName}' does not have any readable indices")
418         }
419       }
420 
421       val queryPlans = getQueryPlan(query)
422 
423       if (queryPlans.isEmpty) {
424         val filter =
425           FilterStrategy(fallbackIndex, None, Some(Filter.EXCLUDE), temporal = false, Float.PositiveInfinity, query.getHints)
426         EmptyPlan(QueryStrategy(filter, Seq.empty, Seq.empty, Seq.empty, filter.filter, None))
427       } else {
428         val qps =
429           if (queryPlans.lengthCompare(1) == 0) { queryPlans } else {
430             // this query requires multiple scans, which we can't execute from some input formats
431             // instead, fall back to a full table scan
432             logger.warn("Desired query plan requires multiple scans - falling back to full table scan")
433             getQueryPlan(query, Some(fallbackIndex.identifier))
434           }
435         if (qps.lengthCompare(1) > 0 || qps.exists(_.tables.lengthCompare(1) > 0)) {
436           logger.error("The query being executed requires multiple scans, which is not currently " +
437             "supported by GeoMesa. Your result set will be partially incomplete. " +
438             s"Query: ${FilterHelper.toString(query.getFilter)}")
439         }
440         qps.head
441       }
442     } finally {
443       // make sure we reset the thread locals
444       JoinIndex.AllowJoinPlans.remove()
445     }
446   }

463   /**
464     * Converts the old 'index schema' into the appropriate index identifiers
465     *
466     * @param sft simple feature type
467     * @return
468     */
469   private def translateSchemaVersion(sft: SimpleFeatureType): Seq[IndexId] = {
470     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
471     lazy val docs =
472       "https://www.geomesa.org/documentation/stable/user/accumulo/jobs.html#updating-existing-data-to-the-latest-index-format"
473 
474     val geom = Option(sft.getGeomField)
475     val dtg = sft.getDtgField
476 
477     val id = IndexId(IdIndex.name, -1, Seq.empty, IndexMode.ReadWrite)
478     val z3 = for { g <- geom; d <- dtg } yield { IndexId(Z3Index.name, -1, Seq(g, d), IndexMode.ReadWrite) }
479     val xz3 = for { g <- geom; d <- dtg } yield { IndexId(XZ3Index.name, 1, Seq(g, d), IndexMode.ReadWrite) }
480     val z2 = geom.map(g => IndexId(Z2Index.name, -1, Seq(g), IndexMode.ReadWrite))
481     val xz2 = geom.map(g => IndexId(XZ2Index.name, 1, Seq(g), IndexMode.ReadWrite))
482     val attributes = sft.getAttributeDescriptors.asScala.flatMap { d =>
483       val index = d.getUserData.remove(AttributeOptions.OptIndex).asInstanceOf[String]
484       if (index == null || index.equalsIgnoreCase(IndexCoverage.NONE.toString) || index.equalsIgnoreCase("false")) {
485         Seq.empty
486       } else if (index.equalsIgnoreCase(IndexCoverage.FULL.toString)) {
487         Seq(IndexId(AttributeIndex.name, -1, Seq(d.getLocalName) ++ dtg, IndexMode.ReadWrite))
488       } else if (index.equalsIgnoreCase(IndexCoverage.JOIN.toString) || java.lang.Boolean.valueOf(index)) {
489         Seq(IndexId(JoinIndex.name, -1, Seq(d.getLocalName) ++ dtg, IndexMode.ReadWrite))
490       } else {
491         throw new IllegalStateException(s"Expected an index coverage or boolean but got: $index")
492       }
493     }
494 
495     // note: 10 was the last valid value for CURRENT_SCHEMA_VERSION, which is no longer used except
496     // to transition old schemas from the 1.2.5 era
497     val version = {
498       // noinspection ScalaDeprecation
499       val string = sft.getUserData.remove(DeprecatedSchemaVersionKey).asInstanceOf[String]
500       if (string != null) { string.toInt } else { 10 }
501     }
502     val indices: Seq[IndexId] = if (version > 8) {
503       // note: version 9 was never in a release
504       val zs = if (sft.isPoints) { z3.map(_.copy(version = 3)) ++ z2.map(_.copy(version = 2)) } else { xz3 ++ xz2 }
505       zs.toSeq ++ Seq(id.copy(version = 2)) ++ attributes.map(_.copy(version = 3))
506     } else if (version == 8) {
507       z3.map(_.copy(version = 2)).toSeq ++ z2.map(_.copy(version = 1)) ++
508           Seq(id.copy(version = 1)) ++ attributes.map(_.copy(version = 2))
509     } else if (version > 5) {
510       logger.warn("The GeoHash index is no longer supported. Some queries may take longer than normal. To " +
511           s"update your data to a newer format, see $docs")
512       val z = if (version == 7) { z3.map(_.copy(version = 2)) } else { z3.map(_.copy(version = 1)) }
513       z.toSeq ++ Seq(id.copy(version = 1)) ++ attributes.map(_.copy(version = 2))
514     } else {
515       throw new UnsupportedOperationException("This schema format is no longer supported. Please use " +
516           s"GeoMesa 1.2.6 to update you data to a newer format. For more information, see $docs")
517     }
518 
519     // noinspection deprecation
520     SimpleFeatureTypes.Configs.ENABLED_INDEX_OPTS.map(sft.getUserData.get).find(_ != null) match {
521       case None => indices
522       case Some(enabled) =>
523         // check for old index names
524         val e = enabled.toString.toLowerCase(Locale.US).split(",").map(_.trim).filterNot(_.isEmpty).map {
525           case "attr_idx" => AttributeIndex.name
526           case "records" => IdIndex.name
