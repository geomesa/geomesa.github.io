36 class AccumuloQueryAuditCommand extends AccumuloDataStoreCommand {
37 
38   override val name = "query-audit-logs"
39   override val params = new AccumuloQueryAuditParams()
40 
41   override def execute(): Unit = withDataStore(exec)
42 
43   private def exec(ds: AccumuloDataStore): Unit = {
44     if (ds.getSchema(params.featureName) == null) {
45       throw new ParameterException(s"SimpleFeatureType '${params.featureName}' does not exist in the data store")
46     }
47 
48     val cql = Option(params.cqlFilter).filter(_ != Filter.INCLUDE)
49     val dateRanges = {
50       import FilterHelper.ff
51       val startFilter = Option(params.start).map(d => ff.greaterOrEqual(ff.property("end"), ff.literal(d)))
52       val endFilter = Option(params.end).map(d => ff.less(ff.property("end"), ff.literal(d)))
53       val dateFilter = andOption(cql.toSeq ++ startFilter ++ endFilter)
54       val intervals = dateFilter.map { f =>
55         val intervals = FilterHelper.extractIntervals(f, "end")
56         if (intervals.disjoint) {
57           throw new ParameterException(s"Invalid disjoint date filter: ${ECQL.toCQL(f)}")
58         }
59         intervals.values
60       }
61       val bounds = intervals.toSeq.flatten
62       if (bounds.isEmpty) {
63         Seq((Instant.EPOCH.atZone(ZoneOffset.UTC), Instant.now().atZone(ZoneOffset.UTC)))
64       } else {
65         val now = Instant.now().atZone(ZoneOffset.UTC)
66         bounds.map { bound =>
67           val start = bound.lower.value.map(d => if (bound.lower.inclusive) { d } else { d.plusNanos(1000000) })
68           val end = bound.upper.value.map(d => if (bound.upper.exclusive) { d } else { d.plusNanos(1000000) })
69           (start.getOrElse(Instant.EPOCH.atZone(ZoneOffset.UTC)), end.getOrElse(now))
70         }
71       }
72     }
73 
74     WithClose(new AccumuloAuditReader(ds)) { reader=>
75       val writer = if ("json".equalsIgnoreCase(params.outputFormat)) { new JsonWriter() } else { new CsvWriter() }
76       writer.header().foreach(Command.output.info)
77       dateRanges.foreach { case (start, end) =>
78         WithClose(reader.getQueryEvents(params.featureName, (start, end))) { events =>
79           val out = cql match {
80             case None => events
81             case Some(f) => events.filter(e => f.evaluate(AccumuloQueryAuditCommand.toFeature(e)))
82           }
83           out.foreach { event =>
84             Command.output.info(writer.output(event))
85           }
86         }
87       }
88     }
89   }
90 }

98   // note: the cql description needs to be a final string constant to work with annotations, so we extract the spec
99   // from there instead of building it up as we might normally do
100   private val Spec = CqlDescription.substring(CqlDescription.indexOf(": ") + 2)
101   private val Sft = SimpleFeatureTypes.createType("audit-logs", Spec)
102 
103   /**
104    * Convert an audit event to a simple feature
105    *
106    * @param event event
107    * @return
108    */
109   private def toFeature(event: QueryEvent): SimpleFeature = {
110     val hints = AuditWriter.Gson.toJson(event.hints)
111     val metadata = AuditWriter.Gson.toJson(event.metadata)
112     val start = if (event.start == -1) { null } else { new Date(event.start) }
113     val end = new Date(event.end)
114     val planTime = Long.box(event.planTime)
115     val scanTime = Long.box(event.scanTime)
116     val hits = Long.box(event.hits)
117     new ScalaSimpleFeature(Sft, "", Array(event.user, event.filter, hints, metadata, start, end, planTime, scanTime, hits))
118   }

153   private trait OutputWriter {
154     def header(): Option[String]
155     def output(event: QueryEvent): String
156   }
157 
158   private class CsvWriter extends OutputWriter {
159     private val out = new java.lang.StringBuilder()
160     // note: don't need to close the printer, as it just closes the underlying stream (i.e. our StringBuilder)
161     private val printer = new CSVPrinter(out, CSVFormat.DEFAULT)
162 
163     override def header(): Option[String] = Some(Spec.replaceAll(":\\w+", ""))
164     override def output(event: QueryEvent): String = {
165       printer.println() // reset new record
166       out.setLength(0)
167       printer.print(event.user)
168       printer.print(event.filter)
169       printer.print(AuditWriter.Gson.toJson(event.hints))
170       printer.print(AuditWriter.Gson.toJson(event.metadata))
171       printer.print(if (event.start == -1) { null } else { DateParsing.formatMillis(event.start) })
172       printer.print(DateParsing.formatMillis(event.end))
173       printer.print(Long.box(event.planTime))
174       printer.print(Long.box(event.scanTime))
175       printer.print(Long.box(event.hits))
176       // note: we don't want to end the record as that inserts a newline, which our logger output is already doing
177       out.toString
178     }
179   }
180 
181   private class JsonWriter extends OutputWriter {
182     override def header(): Option[String] = None
183     override def output(event: QueryEvent): String = {
184       val model = new java.util.LinkedHashMap[String, AnyRef]()
185       model.put("user", event.user)
186       model.put("filter", event.filter)
187       model.put("hints", event.hints)
188       model.put("metadata", event.metadata)
189       if (event.start != -1) {
190         model.put("start", DateParsing.formatMillis(event.start))
191       }
192       model.put("end", DateParsing.formatMillis(event.end))
193       model.put("planTimeMillis", Long.box(event.planTime))
194       model.put("scanTimeMillis", Long.box(event.scanTime))
195       model.put("hits", Long.box(event.hits))
196       AuditWriter.Gson.toJson(model)
197     }
198   }
