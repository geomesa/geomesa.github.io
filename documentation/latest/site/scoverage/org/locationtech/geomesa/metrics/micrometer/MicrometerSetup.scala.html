30   /**
31    * Add registries to the global registry list, based on the default configuration paths
32    *
33    * @param conf conf
34    */
35   def configure(conf: Config = ConfigFactory.load()): Unit = synchronized {
36     if (conf.hasPath(ConfigPath)) {
37       // noinspection ScalaUnusedSymbol
38       implicit val bindingsReader: ConfigReader[MetricsBindings] = deriveReader[MetricsBindings]
39       implicit val metricsReader: ConfigReader[MetricsConfig] = deriveReader[MetricsConfig]
40       implicit val registryReader: ConfigReader[RegistryConfig] = deriveReader[RegistryConfig]
41       val metricsConfig = ConfigSource.fromConfig(conf.getConfig(ConfigPath)).loadOrThrow[MetricsConfig]
42       metricsConfig.registries.foreach { registryConfig =>
43         val config = ConfigSource.fromConfig(registryConfig).loadOrThrow[RegistryConfig]
44         if (config.enabled) {
45           val configString = registryConfig.root().render(ConfigRenderOptions.concise())
46           if (registries.contains(config.`type`)) {
47             val existing = registries(config.`type`)
48             if (existing != configString) {
49               throw new IllegalArgumentException(
50                 s"Registry type ${config.`type`} already registered with a different configuration:" +
51                   s"\n  existing: $existing\n    update: $configString")
52             }
53           } else {
54             val registry = createRegistry(conf)
55             sys.addShutdownHook(registry.close())
56             Metrics.addRegistry(registry)
57             registries.put(config.`type`, configString)
58           }
59         }
60       }
61 
62       if (metricsConfig.bindings.classloader && bindings.add("classloader")) {
63         new ClassLoaderMetrics().bindTo(Metrics.globalRegistry)
64       }
65       if (metricsConfig.bindings.memory && bindings.add("memory")) {
66         new JvmMemoryMetrics().bindTo(Metrics.globalRegistry)
67       }
68       if (metricsConfig.bindings.gc && bindings.add("gc")) {
69         new JvmGcMetrics().bindTo(Metrics.globalRegistry)
70       }
71       if (metricsConfig.bindings.processor && bindings.add("processor")) {
72         new ProcessorMetrics().bindTo(Metrics.globalRegistry)
73       }
74       if (metricsConfig.bindings.threads && bindings.add("threads")) {
75         new JvmThreadMetrics().bindTo(Metrics.globalRegistry)
76       }
77     }
78   }

80   /**
81    * Create a new registry
82    *
83    * @param conf configuration for the registry
84    * @return
85    */
86   def createRegistry(conf: Config): MeterRegistry = {
87     implicit val reader: ConfigReader[RegistryConfig] = deriveReader[RegistryConfig]
88     val config = ConfigSource.fromConfig(conf).loadOrThrow[RegistryConfig]
89     config.`type`.toLowerCase(Locale.US) match {
90       case "prometheus" => PrometheusFactory(conf)
91       case "cloudwatch" => CloudwatchFactory(conf)
92       case t => throw new IllegalArgumentException(s"No registry type defined for '$t' - valid values are: prometheus, cloudwatch")
93     }
94   }
