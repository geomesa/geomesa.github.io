62 object RegistryFactory {
63 
64   // identifiers for our registry factories
65   // note that we keep these here to avoid a runtime dependency on any unused registries
66   val Cloudwatch = "cloudwatch"

69   /**
70    * Factory for creating registries from config
71    *
72    * @param name Well-known name of this registry
73    */
74   abstract class BaseRegistryFactory(val name: String) extends RegistryFactory with StrictLogging {
75 
76     private val registrations = new ConcurrentHashMap[Config, RegistryHolder]()
77 
78     private val default: Config = ConfigFactory.load(name)
79 
80     /**
81      * Create a registry using the default configuration
82      *
83      * @return
84      */
85     override def apply(): MeterRegistry = apply(default)
86 
87     /**
88      * Create a registry
89      *
90      * @param conf config
91      * @return
92      */
93     override def apply(conf: Config): MeterRegistry = {
94       logger.info(s"Creating $name registry")
95       createRegistry(if (conf.eq(default)) { conf } else { conf.withFallback(default) })
96     }
97 
98     /**
99      * Register with this instance, which will ensure the registry managed by this instance
100      * is added to the global metrics registry, until the returned object is closed
101      */
102     override def register(): Closeable = register(default)
103 
104     /**
105      * Register using the given config
106      *
107      * @param conf registry config
108      * @return
109      */
110     override def register(conf: Config): Closeable =
111       registrations.computeIfAbsent(if (conf.eq(default)) { conf } else { conf.withFallback(default) }, new RegistryHolder(_)).register()
112 
113     /**
114      * Create a new registry
115      *
116      * @param config conf
117      * @return
118      */
119     protected def createRegistry(config: Config): MeterRegistry
120 
121     /**
122      * Holds references to a registry
123      *
124      * @param conf registry config
125      */
126     private class RegistryHolder(conf: Config) {
127 
128       private var registry: MeterRegistry = _
129       private var count = 0
130 
131       /**
132        * Register a consumer of the registry held by this object
133        *
134        * @return a handle to the registration, call `close()` to deregister
135        */
136       def register(): Closeable = synchronized {
137         if (count == 0) {
138           logger.info(s"Creating $name registry")
139           registry = createRegistry(conf)
140           Metrics.addRegistry(registry)
141           Instrumentations.bind(conf)
142         }
143         count += 1
144         new Deregistration()
145       }
146 
147       /**
148        * Deregister the lock, called internally when a registration is closed
149        */
150       private def deregister(): Unit = synchronized {
151         if (count == 1) {
152           Metrics.removeRegistry(registry)
153           Try(registry.close())
154           registry = null
155         }
156         count -= 1
157       }
158 
159       /**
160        * Registration holder that ensures only a single `deregister` is called per `register`
161        */
162       private class Deregistration extends Closeable {
163 
164         private val closed = new AtomicBoolean(false)
165 
166         override def close(): Unit = {
167           // ensure we only deregister once
168           if (closed.compareAndSet(false, true)) {
169             deregister()
170           }
171         }
172       }
173     }
174   }
