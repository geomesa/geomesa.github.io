19 /**
20  * Factory for creating registries from config
21  *
22  * @param name Well-known name of this registry
23  */
24 abstract class RegistryFactory(val name: String) {
25 
26   private val registrations = new ConcurrentHashMap[Config, RegistryHolder]()
27 
28   private val default: Config = ConfigFactory.load(name)
29 
30   /**
31    * Create a registry using the default configuration
32    *
33    * @return
34    */
35   def apply(): MeterRegistry = apply(default)
36 
37   /**
38    * Create a registry
39    *
40    * @param conf config
41    * @return
42    */
43   def apply(conf: Config): MeterRegistry = createRegistry(if (conf.eq(default)) { conf } else { conf.withFallback(default) })
44 
45   /**
46    * Register with this instance, which will ensure the registry managed by this instance
47    * is added to the global metrics registry, until the returned object is closed
48    */
49   def register(): Closeable = register(default)
50 
51   /**
52    * Register using the given config
53    *
54    * @param conf registry config
55    * @return
56    */
57   def register(conf: Config): Closeable =
58     registrations.computeIfAbsent(if (conf.eq(default)) { conf } else { conf.withFallback(default) }, new RegistryHolder(_)).register()
59 
60   /**
61    * Create a new registry
62    *
63    * @param config conf
64    * @return
65    */
66   protected def createRegistry(config: Config): MeterRegistry
67 
68   /**
69    * Holds references to a registry
70    *
71    * @param conf registry config
72    */
73   private class RegistryHolder(conf: Config) {
74 
75     private var registry: MeterRegistry = _
76     private var count = 0
77 
78     /**
79      * Register a consumer of the registry held by this object
80      *
81      * @return a handle to the registration, call `close()` to deregister
82      */
83     def register(): Closeable = synchronized {
84       if (count == 0) {
85         registry = createRegistry(conf)
86         Metrics.addRegistry(registry)
87         Instrumentations.bind(conf)
88       }
89       count += 1
90       new Deregistration()
91     }
92 
93     /**
94      * Deregister the lock, called internally when a registration is closed
95      */
96     private def deregister(): Unit = synchronized {
97       if (count == 1) {
98         Metrics.removeRegistry(registry)
99         Try(registry.close())
100         registry = null
101       }
102       count -= 1
103     }
104 
105     /**
106      * Registration holder that ensures only a single `deregister` is called per `register`
107      */
108     private class Deregistration extends Closeable {
109 
110       private val closed = new AtomicBoolean(false)
111 
112       override def close(): Unit = {
113         // ensure we only deregister once
114         if (closed.compareAndSet(false, true)) {
115           deregister()
116         }
117       }
118     }
119   }
120 }
