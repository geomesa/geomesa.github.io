17 /**
18  * Extension of commons-dbcp data source that works with micrometer/prometheus metrics.
19  * See https://github.com/micrometer-metrics/micrometer/issues/2593 for reference.
20  *
21  * We skip the standard jmx exposure in BasicDataSource, because we only want to register the connection pool,
22  * otherwise gauges don't register correctly due to jmx name overlap.
23  *
24  * BasicDataSource registers jmx mbeans in at least 3 places:
25  *  - itself, using the `jmxName` set by the user
26  *  - its GenericObjectPool, using `config.getJmxNameBase` + `config.getJmxNamePrefix`
27  *  - connections created with its PoolableConnectionFactory, using "jmxName" + ",connectionpool=connections,connection=" + `i`
28  *
29  * Micrometer's CommonsObjectPool2Metrics tries to register anything with the jmx name prefix 'org.apache.commons.pool2:'. It uses
30  * the `name` attribute in the jmx name for a tag, and it tries to read `getFactoryType` from the underlying object as a tag.
31  * It exposes pool-related values from the jmx objects (e.g. NumIdle, etc) as metrics.
32  *
33  * The only thing that makes sense to instrument with those values is the GenericObjectPool. Thus, we skip jmx registration for
34  * this data source and the underlying connections. Indeed, since the metric name is based on the jmx name, and the metric
35  * value is pulled from the object itself, registering the datasource or connections will break the metric since the object
36  * doesn't have the expected pool methods.
37  *
38  * Note: short-lived data sources may cause error logs like `ERROR CommonsObjectPool2Metrics: can not set name tag` due to
39  * already being removed from the mbean server by the time the jmx listener fires.
40  */
41 class MetricsDataSource(jmxNamePrefix: String = "dbcp") extends BasicDataSource {
42 
43   // in order for gauges to work correctly, we need unique metrics tags based on the name
44   // the pool will increment names until it finds one that isn't registered, but if a datasource is closed it will deregister,
45   // causing a potential name conflict
46   // instead we track them here with an incrementing counter
47   val jmxName: String = MetricsDataSource.getUniqueName(jmxNamePrefix)
48 
49   override protected def createObjectPool(
50       factory: PoolableConnectionFactory,
51       poolConfig: GenericObjectPoolConfig[PoolableConnection],
52       abandonedConfig: AbandonedConfig): GenericObjectPool[PoolableConnection] = {
53     poolConfig.setJmxEnabled(true)
54     poolConfig.setJmxNameBase(MetricsDataSource.NameBase)
55     poolConfig.setJmxNamePrefix(jmxName)
56     super.createObjectPool(factory, poolConfig, abandonedConfig)
57   }
58 
59   /**
60    * Register this pool with JMX in order to expose metrics. Not thread safe.
61    */
62   @deprecated("Registration happens automatically")
63   def registerJmx(): Unit = {}
64 }
65 
66 object MetricsDataSource {
67 
68   // required for micrometer metrics:
69   //  - prefix must be `org.apache.commons.pool2`
70   //  - type must be either `GenericObjectPool` or `GenericKeyedObjectPool`
71   private val NameBase =
72     s"org.apache.commons.pool2:type=GenericObjectPool${Constants.JMX_CONNECTION_POOL_BASE_EXT}${Constants.JMX_CONNECTION_POOL_PREFIX},name="
73 
74   private val nameCounters = new ConcurrentHashMap[String, AtomicInteger]()
75 
76   private def getUniqueName(base: String): String = {
77     val i = nameCounters.computeIfAbsent(base, _ => new AtomicInteger(0)).getAndIncrement()
78     if (i == 0) { base } else { s"$base$i" }
79   }
80 }
