20 class CloudWatchReporterFactory extends ReporterFactory {
21 
22   import CloudWatchReporterFactory.{CloudWatchConfig, CloudWatchDefaults, Type}
23 
24   override def apply(
25       conf: Config,
26       registry: MetricRegistry,
27       rates: TimeUnit,
28       durations: TimeUnit): Option[ScheduledReporter] = {
29     if (!conf.hasPath("type") || !conf.getString("type").equalsIgnoreCase(Type)) { None } else {
30       val cloudwatch = ConfigSource.fromConfig(conf.withFallback(CloudWatchDefaults)).loadOrThrow[CloudWatchConfig]
31       val client = AmazonCloudWatchAsyncClientBuilder.defaultClient
32       val reporter =
33         CloudWatchReporter.forRegistry(registry, client, cloudwatch.namespace)
34           .convertRatesTo(rates)
35           .convertDurationsTo(durations)
36 
37       if (cloudwatch.rawCounts) {
38         reporter.withReportRawCountValue()
39       }
40       if (cloudwatch.zeroValues) {
41         reporter.withZeroValuesSubmission()
42       }
43 
44       Some(reporter.build)
45     }
46   }
47 }
