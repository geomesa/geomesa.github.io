30   override protected def createRegistry(conf: Config): MeterRegistry = {
31     logger.info("Creating Prometheus registry")
32     // noinspection ScalaUnusedSymbol
33     implicit val gatewayReader: ConfigReader[PushGatewayConfig] = deriveReader[PushGatewayConfig]
34     implicit val prometheusReader: ConfigReader[PrometheusConfig] = deriveReader[PrometheusConfig]
35     val config = ConfigSource.fromConfig(conf).loadOrThrow[PrometheusConfig]
36     val dependentClose = new AtomicReference[Closeable]()
37     val registry = new PrometheusMeterRegistry(k => config.properties.getOrElse(k, null)) {
38       override def close(): Unit = {
39         val child = dependentClose.get()
40         if (child != null) {
41           try { child.close() } catch {
42             case NonFatal(e) => logger.error("Error on close:", e)
43           }
44         }
45         super.close()
46       }
47     }
48     registry.throwExceptionOnRegistrationFailure()
49     if (config.rename) {
50       registry.config().meterFilter(new PrometheusRenameFilter())
51     }
52     if (config.commonTags.nonEmpty) {
53       val tags = config.commonTags.map { case (k, v) => Tag.of(k, v) }
54       registry.config.commonTags(tags.asJava)
55     }
56     config.pushGateway match {
57       case None =>
58         val server =
59           HTTPServer.builder()
60             .port(config.port)
61             .registry(registry.getPrometheusRegistry)
62             .buildAndStart()
63         dependentClose.set(server)
64 
65       case Some(pg) =>
66         val builder = PushGateway.builder().registry(registry.getPrometheusRegistry).address(pg.host)
67         pg.job.foreach(builder.job)
68         pg.format.foreach(v => builder.format(Format.valueOf(v.toUpperCase(Locale.US))))
69         pg.scheme.foreach(v => builder.scheme(Scheme.fromString(v.toLowerCase(Locale.US))))
70         val pushGateway = builder.build()
71         dependentClose.set(() => pushGateway.pushAdd())
72     }
73     registry
74   }
