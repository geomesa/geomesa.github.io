30   override protected def createRegistry(conf: Config): MeterRegistry = {
31     // noinspection ScalaUnusedSymbol
32     implicit val gatewayReader: ConfigReader[PushGatewayConfig] = deriveReader[PushGatewayConfig]
33     implicit val prometheusReader: ConfigReader[PrometheusConfig] = deriveReader[PrometheusConfig]
34     val config = ConfigSource.fromConfig(conf).loadOrThrow[PrometheusConfig]
35     val dependentClose = new AtomicReference[Closeable]()
36     val registry = new PrometheusMeterRegistry(k => config.properties.getOrElse(k, null)) {
37       override def close(): Unit = {
38         val child = dependentClose.get()
39         if (child != null) {
40           try { child.close() } catch {
41             case NonFatal(e) => logger.error("Error on close:", e)
42           }
43         }
44         super.close()
45       }
46     }
47     registry.throwExceptionOnRegistrationFailure()
48     if (config.rename) {
49       registry.config().meterFilter(new PrometheusRenameFilter())
50     }
51     if (config.commonTags.nonEmpty) {
52       val tags = config.commonTags.map { case (k, v) => Tag.of(k, v) }
53       registry.config.commonTags(tags.asJava)
54     }
55     config.pushGateway match {
56       case None =>
57         val server =
58           HTTPServer.builder()
59             .port(config.port)
60             .registry(registry.getPrometheusRegistry)
61             .buildAndStart()
62         dependentClose.set(server)
63 
64       case Some(pg) =>
65         val builder = PushGateway.builder().registry(registry.getPrometheusRegistry).address(pg.host)
66         pg.job.foreach(builder.job)
67         pg.format.foreach(v => builder.format(Format.valueOf(v.toUpperCase(Locale.US))))
68         pg.scheme.foreach(v => builder.scheme(Scheme.fromString(v.toLowerCase(Locale.US))))
69         val pushGateway = builder.build()
70         dependentClose.set(() => pushGateway.pushAdd())
71     }
72     registry
73   }
