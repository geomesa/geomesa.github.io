33     val config = ConfigSource.fromConfig(conf).loadOrThrow[PrometheusConfig]
34     val dependentClose = new AtomicReference[Closeable]()
35     val registry = new PrometheusMeterRegistry(k => config.properties.getOrElse(k, null)) {
36       override def close(): Unit = {
37         val child = dependentClose.get()
38         if (child != null) {
39           try { child.close() } catch {
40             case NonFatal(e) => logger.error("Error on close:", e)
41           }
42         }
43         super.close()
44       }
45     }
46     registry.throwExceptionOnRegistrationFailure()
47     if (config.rename) {
48       registry.config().meterFilter(new PrometheusRenameFilter())
49     }
50     if (config.commonTags.nonEmpty) {
51       val tags = config.commonTags.map { case (k, v) => Tag.of(k, v) }
52       registry.config.commonTags(tags.asJava)
53     }
54     config.pushGateway match {
55       case None =>
56         val server =
57           HTTPServer.builder()
58             .port(config.port)
59             .registry(registry.getPrometheusRegistry)
60             .buildAndStart()
61         dependentClose.set(server)
62 
63       case Some(pg) =>
64         val builder = PushGateway.builder().registry(registry.getPrometheusRegistry).address(pg.host)
65         pg.job.foreach(builder.job)
66         pg.format.foreach(v => builder.format(Format.valueOf(v.toUpperCase(Locale.US))))
67         pg.scheme.foreach(v => builder.scheme(Scheme.fromString(v.toLowerCase(Locale.US))))
68         val pushGateway = builder.build()
69         dependentClose.set(() => pushGateway.pushAdd())
70     }
71     registry
72   }
