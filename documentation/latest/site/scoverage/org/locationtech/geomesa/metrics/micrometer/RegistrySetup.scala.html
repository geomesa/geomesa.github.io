11 import com.typesafe.config.ConfigFactory

14 import java.util.Locale
15 
16 /**
17  * Class for configuring metrics
18  *
19  * @param name name of registry, and associated resource file
20  */
21 class RegistrySetup(name: String) extends StrictLogging {
22 
23   /**
24    * Configure default metrics. Metrics can be customized with environment variables, or with an application.conf,
25    * see `<name>.conf` for details
26    */
27   def configure(): Unit = {
28     val conf = ConfigFactory.load(name.toLowerCase(Locale.US))
29     if (conf.getBoolean("enabled")) {
30       logger.info(s"Configuring $name metrics")
31       MicrometerSetup.configure(conf.atPath(MicrometerSetup.ConfigPath).withFallback(ConfigFactory.load()))
32     } else {
33       logger.info(s"$name metrics disabled")
34     }
35   }
36 }
