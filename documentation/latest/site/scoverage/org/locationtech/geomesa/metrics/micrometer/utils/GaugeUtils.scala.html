22   /**
23    * Register a map size gauge.
24    *
25    * Adds a unique registration tag to ensure the gauge is captured correctly.
26    *
27    * @param name metric name
28    * @param tags tags
29    * @param map map to measure
30    * @tparam T map type
31    * @return the original map
32    */
33   def mapSizeGauge[T <: java.util.Map[_, _]](name: String, tags: java.lang.Iterable[Tag], map: T): T =
34     Metrics.gaugeMapSize(name, addUidTag(name, tags), map)
35 
36   /**
37    * Register a new time gauge.
38    *
39    * Adds a unique registration tag to ensure the gauge is captured correctly.
40    *
41    * @param name metric name
42    * @param tags tags
43    * @param timeUnit time unit of the gauge
44    * @return
45    */
46   def timeGauge(name: String, tags: java.lang.Iterable[Tag], timeUnit: TimeUnit = TimeUnit.MILLISECONDS): AtomicLong = {
47     val time = new AtomicLong()
48     TimeGauge.builder(name, () => Long.box(time.get), timeUnit).tags(addUidTag(name, tags)).register(Metrics.globalRegistry)
49     time
50   }
51 
52   private def addUidTag(name: String, tags: java.lang.Iterable[Tag]): Tags = {
53     val cacheKey = name + tags.asScala.toList.sorted.mkString(" ", ",", "")
54     val uid = registrations.computeIfAbsent(cacheKey, _ => new AtomicInteger(0)).getAndIncrement().toString
55     Tags.of("uid", uid).and(tags)
56   }
