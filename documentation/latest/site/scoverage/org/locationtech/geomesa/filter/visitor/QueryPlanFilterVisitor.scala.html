55   override def visit(f: Or, data: AnyRef): AnyRef = {
56     val children = new java.util.ArrayList[Filter](f.getChildren.size)
57     var i = 0
58     while (i < f.getChildren.size) {
59       val child = f.getChildren.get(i).accept(this, data).asInstanceOf[Filter]
60       if (child == Filter.INCLUDE) {
61         // INCLUDE OR foo == INCLUDE
62         return Filter.INCLUDE
63       } else if (child != Filter.EXCLUDE) {
64         // EXCLUDE OR foo == foo
65         children.add(child)
66       }
67       i += 1
68     }
69 
70     children.size() match {
71       case 0 => Filter.EXCLUDE
72       case 1 => children.get(0)
73       case _ => getFactory(data).or(children)
74     }
75   }

147   private def handleInArrayEquals(filter: PropertyIsEqualTo, extraData: AnyRef): Option[Filter] = {
148     var inArrayFunction: InArrayFunction = null
149     var hasTrue: Boolean = false
150 
151     def inspectExpression(expression: Expression, extraData: AnyRef): Unit = {
152       expression match {
153         case function: InArrayFunction =>
154           inArrayFunction = function
155         case literal: Literal if java.lang.Boolean.TRUE.equals(literal.getValue) =>
156           hasTrue = true
157         case _ =>
158       }
159     }
160 
161     inspectExpression(filter.getExpression1, extraData)
162     inspectExpression(filter.getExpression2, extraData)
163 
164     if (inArrayFunction != null && hasTrue) {
165       val attribute = inArrayFunction.getParameters.get(0)
166       val array = inArrayFunction.getParameters.get(1)
167       import scala.collection.JavaConverters._
168       // The important part of this optimization is to handle ArrayLists generated by the
169       // GeoServer cross-layer querylayer extension:  E.g.
170       // https://github.com/geoserver/geoserver/blob/2.17.3/src/extension/querylayer/src/main/java/org/geoserver/filter/function/QueryFunction.java#L124-L151
171       val filters = array.evaluate(null).asInstanceOf[util.List[_]].asScala.toSet[Any].map {
172         o => getFactory(extraData).equals(attribute, getFactory(extraData).literal(o)).asInstanceOf[Filter]
173       }.toList
174       val newFilter = getFactory(extraData).or(filters.asJava)
175       Some(newFilter)
176     } else {
177       None
178     }
179   }

374   private def binding(expressions: Seq[Expression]): Class[_] = {
375     val bindings = expressions.flatMap {
376       case _: Literal => Seq.empty // don't consider literals, as we're trying to bind them to the right type
377       case e => Seq(e.accept(typeVisitor, null)).filter(_ != null)
378     }
379     bindings.distinct match {
380       case Seq(b) => b.asInstanceOf[Class[_]]
381       case _ => null // if not exactly one type, we can't bind it
382     }
383   }
384 
385   private def bind(e: Expression, extraData: AnyRef, target: Class[_]): Expression = {
386     if (e.isInstanceOf[Literal]) {
387       val bound = FastConverter.convert(e.evaluate(null), target)
388       if (bound != null) {
389         return getFactory(extraData).literal(bound)
390       }
391     }
392     visit(e, extraData)
393   }
394 
395   private def bindTemporal(e: Expression, extraData: AnyRef): Expression = {
396     if (e.isInstanceOf[Literal]) {
397       val lit = e.evaluate(null)
398       val bound = FastConverter.convertFirst[AnyRef](lit, Iterator(classOf[Period], classOf[Date]))
399       if (bound != null) {
400         return getFactory(extraData).literal(bound)
401       }
402     }
403     visit(e, extraData)
404   }
