51   private def isOperationGeomWholeWorld[Op <: BinarySpatialOperator]
52       (op: Op, order: SpatialOpOrder.SpatialOpOrder = SpatialOpOrder.AnyOrder): Boolean = {
53     val prop = checkOrder(op.getExpression1, op.getExpression2)
54     // validate that property and literal are in the specified order
55     prop.exists { p =>
56       val ordered = order match {
57         case SpatialOpOrder.AnyOrder      => true
58         case SpatialOpOrder.PropertyFirst => !p.flipped
59         case SpatialOpOrder.LiteralFirst  => p.flipped
60       }
61       ordered && Option(FastConverter.evaluate(p.literal, classOf[Geometry])).exists(isWholeWorld)
62     }

103     * Extract geometries from a filter without validating boundaries.

106     * @param attribute attribute to consider
107     * @param intersect intersect AND'd geometries or return them all
108     * @return geometry bounds from spatial filters

110   private def extractUnclippedGeometries(filter: Filter, attribute: String, intersect: Boolean): FilterValues[Geometry] = {
111     filter match {
112       case o: Or  =>
113         val all = o.getChildren.asScala.map(extractUnclippedGeometries(_, attribute, intersect))
114         val join = FilterValues.or[Geometry]((l, r) => l ++ r) _
115         all.reduceLeftOption[FilterValues[Geometry]](join).getOrElse(FilterValues.empty)
116 
117       case a: And =>
118         val all = a.getChildren.asScala.map(extractUnclippedGeometries(_, attribute, intersect)).filter(_.nonEmpty)
119         if (intersect) {
120           val intersect = FilterValues.and[Geometry]((l, r) => Option(l.intersection(r)).filterNot(_.isEmpty)) _
121           all.reduceLeftOption[FilterValues[Geometry]](intersect).getOrElse(FilterValues.empty)
122         } else {
123           FilterValues(all.toSeq.flatMap(_.values))
124         }
125 
126       // Note: although not technically required, all known spatial predicates are also binary spatial operators
127       case f: BinarySpatialOperator if isSpatialFilter(f) =>
128         FilterValues(GeometryProcessing.extract(f, attribute))
129 
130       case _ =>
131         FilterValues.empty
132     }

135   /**
136     * Extracts intervals from a filter. Intervals will be merged where possible - the resulting sequence
137     * is considered to be a union (i.e. OR)
138     *
139     * @param filter filter to evaluate
140     * @param attribute attribute to consider
141     * @param intersect intersect extracted values together, or return them all
142     *                  note if not intersected, 'and/or' distinction will be lost
143     * @return a sequence of intervals, if any. disjoint intervals will result in Seq((null, null))
144     */
145   def extractIntervals(filter: Filter,
146                        attribute: String,
147                        intersect: Boolean = true,
148                        handleExclusiveBounds: Boolean = false): FilterValues[Bounds[ZonedDateTime]] = {
149     extractAttributeBounds(filter, attribute, classOf[Date]).map { bounds =>
150       var lower, upper: Bound[ZonedDateTime] = null
151       // this if check determines if rounding will be used and if we need to account for narrow ranges
152       if (!handleExclusiveBounds || bounds.lower.value.isEmpty || bounds.upper.value.isEmpty ||
153           (bounds.lower.inclusive && bounds.upper.inclusive)) {
154         lower = createDateTime(bounds.lower, roundSecondsUp, handleExclusiveBounds)
155         upper = createDateTime(bounds.upper, roundSecondsDown, handleExclusiveBounds)
156       } else {
157         // check for extremely narrow filters where our rounding makes the result out-of-order
158         // note: both upper and lower are known to be defined based on hitting this else branch
159         val margin = if (bounds.lower.inclusive || bounds.upper.inclusive) { 1000 } else { 2000 }
160         val round = bounds.upper.value.get.getTime - bounds.lower.value.get.getTime > margin
161         lower = createDateTime(bounds.lower, roundSecondsUp, round)
162         upper = createDateTime(bounds.upper, roundSecondsDown, round)
163       }
164       Bounds(lower, upper)
165     }
166   }

168   private def createDateTime(bound: Bound[Date],
169                              round: ZonedDateTime => ZonedDateTime,
170                              roundExclusive: Boolean): Bound[ZonedDateTime] = {
171     if (bound.value.isEmpty) { Bound.unbounded } else {
172       val dt = bound.value.map(d => ZonedDateTime.ofInstant(d.toInstant, ZoneOffset.UTC))
173       if (roundExclusive && !bound.inclusive) {
174         Bound(dt.map(round), inclusive = true)
175       } else {
176         Bound(dt, bound.inclusive)
177       }
178     }
179   }
180 
181   private def roundSecondsUp(dt: ZonedDateTime): ZonedDateTime = dt.plusSeconds(1).withNano(0)
182 
183   private def roundSecondsDown(dt: ZonedDateTime): ZonedDateTime = {
184     val nanos = dt.getNano
185     if (nanos == 0) { dt.minusSeconds(1) } else { dt.withNano(0) }
186   }

188   /**
189     * Extracts bounds from filters that pertain to a given attribute. Bounds will be merged where
190     * possible.
191     *
192     * @param filter filter to evaluate
193     * @param attribute attribute name to consider
194     * @param binding attribute type
195     * @return a sequence of bounds, if any
196     */
197   def extractAttributeBounds[T](filter: Filter, attribute: String, binding: Class[T]): FilterValues[Bounds[T]] = {
198     filter match {
199       case o: Or =>
200         val union = FilterValues.or[Bounds[T]](Bounds.union[T]) _
201         o.getChildren.asScala.map(f =>
202           extractAttributeBounds(f, attribute, binding)
203         ).reduceLeft[FilterValues[Bounds[T]]]((acc, child) => {
204           if (acc.isEmpty || child.isEmpty) {
205             FilterValues.empty
206           } else {
207             union(acc, child)
208           }
209         })
210 
211       case a: And =>
212         val all = a.getChildren.asScala.flatMap { f =>
213           val child = extractAttributeBounds(f, attribute, binding)
214           if (child.isEmpty) { Seq.empty } else { Seq(child) }
215         }
216         val intersection = FilterValues.and[Bounds[T]](Bounds.intersection[T]) _
217         all.reduceLeftOption[FilterValues[Bounds[T]]](intersection).getOrElse(FilterValues.empty)
218 
219       case f: PropertyIsEqualTo =>
220         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
221           case e: PropertyLiteral =>
222             Option(FastConverter.evaluate(e.literal, binding)).map { lit =>
223               val bound = Bound(Some(lit), inclusive = true)
224               FilterValues(Seq(Bounds(bound, bound)))
225             }
226 
227           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = true, binding)
228         }.getOrElse(FilterValues.empty)
229 
230       case f: PropertyIsBetween =>
231         try {
232           val prop = f.getExpression.asInstanceOf[PropertyName].getPropertyName
233           if (prop != attribute) { FilterValues.empty } else {
234             // note that between is inclusive
235             val lower = Bound(Option(FastConverter.evaluate(f.getLowerBoundary, binding)), inclusive = true)
236             val upper = Bound(Option(FastConverter.evaluate(f.getUpperBoundary, binding)), inclusive = true)
237             FilterValues(Seq(Bounds(lower, upper)))
238           }
239         } catch {
240           case e: Exception =>
241             FilterHelperLogger.log.warn(s"Unable to extract bounds from filter '${filterToString(f)}'", e)
242             FilterValues.empty
243         }
244 
245       case f: During if classOf[Date].isAssignableFrom(binding) =>
246         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
247           case e: PropertyLiteral =>
248             Option(FastConverter.evaluate(e.literal, classOf[Period])).map { p =>
249               // note that during is exclusive
250               val lower = Bound(Option(p.getBeginning.getPosition.getDate.asInstanceOf[T]), inclusive = false)
251               val upper = Bound(Option(p.getEnding.getPosition.getDate.asInstanceOf[T]), inclusive = false)
252               FilterValues(Seq(Bounds(lower, upper)))
253             }
254 
255           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = false, binding)
256         }.getOrElse(FilterValues.empty)
257 
258       case f: PropertyIsGreaterThan =>
259         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
260           case e: PropertyLiteral =>
261             Option(FastConverter.evaluate(e.literal, binding)).map { lit =>
262               val bound = Bound(Some(lit), inclusive = false)
263               val (lower, upper) = if (e.flipped) { (Bound.unbounded[T], bound) } else { (bound, Bound.unbounded[T]) }
264               FilterValues(Seq(Bounds(lower, upper)))
265             }
266 
267           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = false, binding)
268         }.getOrElse(FilterValues.empty)
269 
270       case f: PropertyIsGreaterThanOrEqualTo =>
271         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
272           case e: PropertyLiteral =>
273             Option(FastConverter.evaluate(e.literal, binding)).map { lit =>
274               val bound = Bound(Some(lit), inclusive = true)
275               val (lower, upper) = if (e.flipped) { (Bound.unbounded[T], bound) } else { (bound, Bound.unbounded[T]) }
276               FilterValues(Seq(Bounds(lower, upper)))
277             }
278 
279           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = true, binding)
280         }.getOrElse(FilterValues.empty)
281 
282       case f: PropertyIsLessThan =>
283         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
284           case e: PropertyLiteral =>
285             Option(FastConverter.evaluate(e.literal, binding)).map { lit =>
286               val bound = Bound(Some(lit), inclusive = false)
287               val (lower, upper) = if (e.flipped) { (bound, Bound.unbounded[T]) } else { (Bound.unbounded[T], bound) }
288               FilterValues(Seq(Bounds(lower, upper)))
289             }
290 
291           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = false, binding)
292         }.getOrElse(FilterValues.empty)
293 
294       case f: PropertyIsLessThanOrEqualTo =>
295         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
296           case e: PropertyLiteral =>
297             Option(FastConverter.evaluate(e.literal, binding)).map { lit =>
298               val bound = Bound(Some(lit), inclusive = true)
299               val (lower, upper) = if (e.flipped) { (bound, Bound.unbounded[T]) } else { (Bound.unbounded[T], bound) }
300               FilterValues(Seq(Bounds(lower, upper)))
301             }
302 
303           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = true, binding)
304         }.getOrElse(FilterValues.empty)
305 
306       case f: Before =>
307         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
308           case e: PropertyLiteral =>
309             Option(FastConverter.evaluate(e.literal, binding)).map { lit =>
310               // note that before is exclusive
311               val bound = Bound(Some(lit), inclusive = false)
312               val (lower, upper) = if (e.flipped) { (bound, Bound.unbounded[T]) } else { (Bound.unbounded[T], bound) }
313               FilterValues(Seq(Bounds(lower, upper)))
314             }
315 
316           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = false, binding)
317         }.getOrElse(FilterValues.empty)
318 
319       case f: After =>
320         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
321           case e: PropertyLiteral =>
322             Option(FastConverter.evaluate(e.literal, binding)).map { lit =>
323               // note that after is exclusive
324               val bound = Bound(Some(lit), inclusive = false)
325               val (lower, upper) = if (e.flipped) { (Bound.unbounded[T], bound) } else { (bound, Bound.unbounded[T]) }
326               FilterValues(Seq(Bounds(lower, upper)))
327             }
328 
329           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = false, binding)
330         }.getOrElse(FilterValues.empty)
331 
332       case f: PropertyIsLike if binding == classOf[String] =>
333         try {
334           val prop = f.getExpression.asInstanceOf[PropertyName].getPropertyName
335           if (prop != attribute) { FilterValues.empty } else {
336             // find the first wildcard and create a range prefix
337             val literal = f.getLiteral
338             var i = literal.indexWhere(Wildcards.contains)
339             // check for escaped wildcards
340             while (i > 1 && literal.charAt(i - 1) == '\\' && literal.charAt(i - 2) == '\\') {
341               i = literal.indexWhere(Wildcards.contains, i + 1)
342             }
343             if (i == -1) {
344               val literals = if (f.isMatchingCase) { Seq(literal) } else { casePermutations(literal) }
345               val bounds = literals.map { lit =>
346                 val bound = Bound(Some(lit), inclusive = true)
347                 Bounds(bound, bound)
348               }
349               FilterValues(bounds.asInstanceOf[Seq[Bounds[T]]], precise = true)
350             } else {
351               val prefix = literal.substring(0, i)
352               val prefixes = if (f.isMatchingCase) { Seq(prefix) } else { casePermutations(prefix) }
353               val bounds = prefixes.map { p =>
354                 Bounds(Bound(Some(p), inclusive = true), Bound(Some(p + WildcardSuffix), inclusive = true))
355               }
356               // our ranges fully capture the filter if there's a single trailing multi-char wildcard
357               val exact = i == literal.length - 1 && literal.charAt(i) == WildcardMultiChar
358               FilterValues(bounds.asInstanceOf[Seq[Bounds[T]]], precise = exact)
359             }
360           }
361         } catch {
362           case e: Exception =>
363             FilterHelperLogger.log.warn(s"Unable to extract bounds from filter '${filterToString(f)}'", e)
364             FilterValues.empty
365         }
366 
367       case f: Not if f.getFilter.isInstanceOf[PropertyIsNull] =>
368         try {
369           val isNull = f.getFilter.asInstanceOf[PropertyIsNull]
370           val prop = isNull.getExpression.asInstanceOf[PropertyName].getPropertyName
371           if (prop != attribute) { FilterValues.empty } else {
372             FilterValues(Seq(Bounds.everything[T]))
373           }
374         } catch {
375           case e: Exception =>
376             FilterHelperLogger.log.warn(s"Unable to extract bounds from filter '${filterToString(f)}'", e)
377             FilterValues.empty
378         }
379 
380       case f: Not =>
381         // we extract the sub-filter bounds, then invert them
382         val inverted = extractAttributeBounds(f.getFilter, attribute, binding)
383         if (inverted.isEmpty) {
384           inverted
385         } else if (inverted.disjoint) {
386           FilterValues(Seq(Bounds.everything[T])) // equivalent to not null
387         } else if (!inverted.precise) {
388           FilterHelperLogger.log.warn(s"Falling back to full table scan for inverted query: '${filterToString(f)}'")
389           FilterValues(Seq(Bounds.everything[T]), precise = false)
390         } else {
391           // NOT(A OR B) turns into NOT(A) AND NOT(B)
392           val uninverted = inverted.values.map { bounds =>
393             // NOT the single bound
394             val not = bounds.bounds match {
395               case (None, None) => Seq.empty
396               case (Some(lo), None) => Seq(Bounds(Bound.unbounded, Bound(Some(lo), !bounds.lower.inclusive)))
397               case (None, Some(hi)) => Seq(Bounds(Bound(Some(hi), !bounds.upper.inclusive), Bound.unbounded))
398               case (Some(lo), Some(hi)) => Seq(
399                   Bounds(Bound.unbounded, Bound(Some(lo), !bounds.lower.inclusive)),
400                   Bounds(Bound(Some(hi), !bounds.upper.inclusive), Bound.unbounded)
401                 )
402             }
403             FilterValues(not)
404           }
405           // AND together
406           val intersect = FilterValues.and[Bounds[T]](Bounds.intersection[T]) _
407           uninverted.reduceLeft[FilterValues[Bounds[T]]](intersect)
408         }
409 
410       case f: TEquals =>
411         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap { prop =>
412           Option(FastConverter.evaluate(prop.literal, binding)).map { lit =>
413             val bound = Bound(Some(lit), inclusive = true)
414             FilterValues(Seq(Bounds(bound, bound)))
415           }
416         }.getOrElse(FilterValues.empty)
417 
418       case _ => FilterValues.empty
419     }
420   }

429   /**
430    * Calculates all the different case permutations of a string.
431    *
432    * For example, "foo" -> Seq("foo", "Foo", "fOo", "foO", "fOO", "FoO", "FOo", "FOO")
433    *
434    * @param string input string
435    * @return
436    */
437   private def casePermutations(string: String): Seq[String] = {
438     val max = FilterProperties.CaseInsensitiveLimit.toInt.getOrElse {
439       // has a valid default value so should never return a none
440       throw new IllegalStateException(
441         s"Error getting default value for ${FilterProperties.CaseInsensitiveLimit.property}")
442     }
443 
444     val lower = string.toLowerCase(Locale.US)
445     val upper = string.toUpperCase(Locale.US)
446     // account for chars without upper/lower cases, which we don't need to permute
447     val count = (0 until lower.length).count(i => lower(i) != upper(i))
448 
449     if (count > max) {
450       FilterHelperLogger.log.warn(s"Not expanding case-insensitive prefix due to length: $string")
451       Seq.empty
452     } else {
453       // there will be 2^n different permutations, accounting for chars that don't have an upper/lower case
454       val permutations = Array.fill(math.pow(2, count).toInt)(Array(lower: _*))
455       var i = 0 // track the index of the current char
456       var c = 0 // track the index of the bit check, which skips chars that don't have an upper/lower case
457       while (i < string.length) {
458         val upperChar = upper.charAt(i)
459         if (lower.charAt(i) != upperChar) {
460           var j = 0
461           while (j < permutations.length) {
462             // set upper/lower based on the bit
463             if (((j >> c) & 1) != 0) {
464               permutations(j)(i) = upperChar
465             }
466             j += 1
467           }
468           c += 1
469         }
470         i += 1
471       }
472 
473       permutations.map(new String(_))
474     }
475   }

507   /**
508     * Simplifies filters to make them easier to process.
509     *
510     * Current simplifications:
511     *
512     *   1) Extracts out common parts in an OR clause to simplify further processing.
513     *
514     *      Example: OR(AND(1, 2), AND(1, 3), AND(1, 4)) -> AND(1, OR(2, 3, 4))
515     *
516     *   2) N/A - add more simplifications here as needed
517     *
518     * @param filter filter
519     * @return
520     */
521   def simplify(filter: Filter): Filter = {
522     def deduplicateOrs(f: Filter): Filter = f match {
523       case and: And => ff.and(and.getChildren.asScala.map(deduplicateOrs).asJava)
524 
525       case or: Or =>
526         // OR(AND(1,2,3), AND(1,2,4)) -> Seq(Seq(1,2,3), Seq(1,2,4))
527         val decomposed = or.getChildren.asScala.map(decomposeAnd)
528         val clauses = decomposed.head // Seq(1,2,3)
529         val duplicates = clauses.filter(c => decomposed.tail.forall(_.contains(c))) // Seq(1,2)
530         if (duplicates.isEmpty) { or } else {
531           val simplified = decomposed.flatMap(d => andOption(d.filterNot(duplicates.contains)))
532           if (simplified.length < decomposed.length) {
533             // the duplicated filters are an entire clause, so we can ignore the rest of the clauses
534             andFilters(duplicates)
535           } else {
536             andFilters(orOption(simplified.toSeq).toSeq ++ duplicates)
537           }
538         }
539 
540       case _ => f
541     }
542     // TODO GEOMESA-1533 simplify ANDs of ORs for DNF
543     flatten(deduplicateOrs(flatten(filter)))
544   }

562   private [filter] def flattenAnd(filters: Seq[Filter]): ListBuffer[Filter] = {
563     val remaining = ListBuffer.empty[Filter] ++ filters
564     val result = ListBuffer.empty[Filter]
565     while (remaining.nonEmpty) {
