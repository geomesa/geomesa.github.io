187 sealed private abstract class FastComparisonOperator(exp1: Expression, exp2: Expression, private val op: String)
188     extends BinaryComparisonOperator {
189 
190   override def getExpression1: Expression = exp1
191 
192   override def getExpression2: Expression = exp2
193 
194   override def isMatchingCase: Boolean = false
195 
196   override def getMatchAction: MultiValuedFilter.MatchAction = MultiValuedFilter.MatchAction.ANY
197 
198   override def toString: String = s"[ $exp1 $op $exp2 ]"
199 
200   def canEqual(other: Any): Boolean = other.isInstanceOf[FastComparisonOperator]
201 
202   override def equals(other: Any): Boolean = other match {
203     case that: FastComparisonOperator =>
204       (that canEqual this) && exp1 == that.getExpression1 && exp2 == that.getExpression2 && op == that.op
205     case _ => false
206   }
207 
208   override def hashCode(): Int =
