13 /**
14   * Single typed bound. If filter is unbounded on one or both sides, the associated bound will be None.
15   *
16   * For example, bounds for 'foo < 5' would be (None, Some(5))
17   * Special case for 'foo NOT NULL' will have both bounds be None
18   *
19   * @param lower lower bound, if any
20   * @param upper upper bound, if any
21   * @tparam T binding of the attribute type
22   */
23 case class Bounds[T](lower: Bound[T], upper: Bound[T]) {
24 
25   def bounds: (Option[T], Option[T]) = (lower.value, upper.value)
26 
27   /**
28     * Bounded on at least one side
29     *
30     * @return
31     */
32   def isBounded: Boolean = lower.value.nonEmpty || upper.value.nonEmpty
33 
34   /**
35     * Bounded on both sides
36     *
37     * @return
38     */
39   def isBoundedBothSides: Boolean = lower.value.nonEmpty && upper.value.nonEmpty
40 
41   /**
42     * Covers multiple values
43     *
44     * @return
45     */
46   def isRange: Boolean = lower.value.isEmpty || lower.value != upper.value
47 
48   /**
49     * Covers a single value
50     *
51     * @return
52     */
53   def isEquals: Boolean = !isRange
54 
55   /**
56     * Does this bounds cover the other bounds?
57     *
58     * @param other other bounds
59     * @return
60     */
61   def covers(other: Bounds[T]): Boolean = {
62     // if the end is unbounded, it will always cover, so use .forall
63     // if the end is bounded and other.end is unbounded, then it will never cover, so use .exists on the inner check
64     // if both end and other.end are bounded, then compare them
65     upper.value.forall { up =>
66       other.upper.value.exists { oup =>
67         val c = oup.asInstanceOf[Comparable[Any]].compareTo(up)
68         c < 0 || (c == 0 && (upper.inclusive || other.upper.exclusive))
69       }
70     } && lower.value.forall { lo =>
71       other.lower.value.exists { olo =>
72         val c = olo.asInstanceOf[Comparable[Any]].compareTo(lo)
73         c > 0 || (c == 0 && (lower.inclusive || other.lower.exclusive))
74       }
75     }
76   }
77 
78   /**
79     * Does this bounds intersect the other bounds?
80     *
81     * @param other other bounds
82     * @return
83     */
84   def intersects(other: Bounds[T]): Boolean = {
85     // if `largerLowerBound` or `smallerUpperBound` return None, then both this and other are unbounded
86     // on the same end, so they by definition overlap - so use .forall
87     val largerLowerBound = Bounds.largerLowerBound(lower, other.lower)
88     largerLowerBound.value.forall { lo =>
89       val smallerUpperBound = Bounds.smallerUpperBound(upper, other.upper)
90       smallerUpperBound.value.forall { up =>
91         // if largerLowerBound bound is smaller than smallerUpperBound, then they overlap
92         val c = lo.asInstanceOf[Comparable[Any]].compareTo(up)
93         c < 0 || (c == 0 && largerLowerBound.inclusive && smallerUpperBound.inclusive)
94       }
95     }
96   }
97 
98   override def toString: String = {
99     (if (lower.inclusive) { "[" } else { "(" }) + lower.value.getOrElse("-\u221E") + "," +
100       upper.value.getOrElse("+\u221E") + (if (upper.inclusive) { "]" } else { ")" })
101   }
102 }

104 object Bounds {
105 
106   /**
107     * Single bound (lower or upper).
108     *
109     * Bound may be unbounded, in which case value is None. Note by convention unbounded bounds are exclusive
110     *
111     * @param value value of this bound, if bounded
112     * @param inclusive whether the bound is inclusive or exclusive.
113     *                  for example, 'foo < 5' is exclusive, 'foo <= 5' is inclusive
114     */
115   case class Bound[T](value: Option[T], inclusive: Boolean) {
116     def exclusive: Boolean = !inclusive
117   }
118 
119   object Bound {
120     private val unboundedBound = Bound[Any](None, inclusive = false)
121     def unbounded[T]: Bound[T] = unboundedBound.asInstanceOf[Bound[T]]
122   }
123 
124   private val allValues = Bounds(Bound.unbounded, Bound.unbounded)
125 
126   def everything[T]: Bounds[T] = allValues.asInstanceOf[Bounds[T]]
127 
128   /**
129     * Gets the smaller value between two lower bounds, taking into account exclusivity.
130     * If the bounds are equal, the first bound will always be returned
131     *
132     * @param bound1 first bound
133     * @param bound2 second bound
134     * @return smaller bound
135     */
136   def smallerLowerBound[T](bound1: Bound[T], bound2: Bound[T]): Bound[T] = {
137     if (bound1.value.isEmpty) {
138       bound1
139     } else if (bound2.value.isEmpty) {
140       bound2
141     } else {
142       val c = bound1.value.get.asInstanceOf[Comparable[Any]].compareTo(bound2.value.get)
143       if (c < 0 || (c == 0 && (bound1.inclusive || bound2.exclusive))) { bound1 } else { bound2 }
144     }
145   }
146 
147   /**
148     * Gets the larger value between two upper bounds, taking into account exclusivity.
149     * If the bounds are equal, the first bound will always be returned
150     *
151     * @param bound1 first bound
152     * @param bound2 second bound
153     * @return larger bound
154     */
155   def largerUpperBound[T](bound1: Bound[T], bound2: Bound[T]): Bound[T] = {
156     if (bound1.value.isEmpty) {
157       bound1
158     } else if (bound2.value.isEmpty) {

160     } else {
161       val c = bound1.value.get.asInstanceOf[Comparable[Any]].compareTo(bound2.value.get)
162       if (c > 0 || (c == 0 && (bound1.inclusive || bound2.exclusive))) { bound1 } else { bound2 }
163     }
164   }
165 
166   /**
167     * Gets the smaller value between two upper bounds, taking into account exclusivity.
168     * If the bounds are equal, the first bound will always be returned
169     *
170     * @param bound1 first bound
171     * @param bound2 second bound
172     * @return smaller bound
173     */
174   def smallerUpperBound[T](bound1: Bound[T], bound2: Bound[T]): Bound[T] = {
175     if (bound2.value.isEmpty) {
176       bound1
177     } else if (bound1.value.isEmpty) {

179     } else {
180       val c = bound1.value.get.asInstanceOf[Comparable[Any]].compareTo(bound2.value.get)
181       if (c < 0 || (c == 0 && (bound2.inclusive || bound1.exclusive))) { bound1 } else { bound2 }
182     }
183   }
184 
185   /**
186     * Gets the larger value between two upper bounds, taking into account exclusivity.
187     * If the bounds are equal, the first bound will always be returned
188     *
189     * @param bound1 first bound
190     * @param bound2 second bound
191     * @return larger bound
192     */
193   def largerLowerBound[T](bound1: Bound[T], bound2: Bound[T]): Bound[T] = {
194     if (bound2.value.isEmpty) {
195       bound1
196     } else if (bound1.value.isEmpty) {
197       bound2
198     } else {
199       val c = bound1.value.get.asInstanceOf[Comparable[Any]].compareTo(bound2.value.get)
200       if (c > 0 || (c == 0 && (bound2.inclusive || bound1.exclusive))) { bound1 } else { bound2 }
201     }
202   }
203 
204   /**
205     * Takes the intersection of two bounds. If they are disjoint, will return None.
206     *
207     * @param left first bounds
208     * @param right second bounds
209     * @tparam T type parameter
210     * @return intersection
211     */
212   def intersection[T](left: Bounds[T], right: Bounds[T]): Option[Bounds[T]] = {
213     val lower = largerLowerBound(left.lower, right.lower)
214     val upper = smallerUpperBound(right.upper, left.upper)
215     (lower.value, upper.value) match {
216       case (Some(lo), Some(up)) =>
217         lo.asInstanceOf[Comparable[Any]].compareTo(up) match {
218           case i if i > 0 || (i == 0 && (!lower.inclusive || !upper.inclusive)) => None
219           case _ => Some(Bounds(lower, upper))
220         }
221       case _ => Some(Bounds(lower, upper))
222     }
223   }
224 
225   /**
226     * Takes the union of two bound sequences.
227     *
228     * @param left first bounds
229     * @param right second bounds
230     * @tparam T type parameter
231     * @return union
232     */
233   def union[T](left: Seq[Bounds[T]], right: Seq[Bounds[T]]): Seq[Bounds[T]] = {
234 
235     def mergeOverlapping(bound: Bounds[T], rest: Seq[Bounds[T]]): Seq[Bounds[T]] = {
236       rest match {
237         case Seq() => Seq(bound)
238         case head +: tail =>
239           bound.upper match {
240             case Bound(None, _) => Seq(bound)
241             case Bound(Some(x), incX) =>
242               head.lower match {
243                 case Bound(None, _) =>
244                   mergeOverlapping(Bounds(head.lower, largerUpperBound(bound.upper, head.upper)), tail)
245                 case Bound(Some(y), incY) =>
246                   val c = y.asInstanceOf[Comparable[Any]].compareTo(x)
247                   if (c < 0 || (c == 0 && (incX || incY)))
248                     mergeOverlapping(Bounds(bound.lower, largerUpperBound(bound.upper, head.upper)), tail)
249                   else
250                     bound +: mergeOverlapping(head, tail)
251               }
252           }
253       }
254     }
255 
256     val bounds = (left ++ right).sortWith { case (a, b) => smallerLowerBound(a.lower, b.lower) == a.lower }
257     bounds match {
258       case Seq() => bounds
259       case head +: tail => mergeOverlapping(head, tail)
260     }
261   }
