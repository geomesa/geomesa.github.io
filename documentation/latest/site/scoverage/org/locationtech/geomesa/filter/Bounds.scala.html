13 /**
14   * Single typed bound. If filter is unbounded on one or both sides, the associated bound will be None.
15   *
16   * For example, bounds for 'foo < 5' would be (None, Some(5))
17   * Special case for 'foo NOT NULL' will have both bounds be None
18   *
19   * @param lower lower bound, if any
20   * @param upper upper bound, if any
21   * @tparam T binding of the attribute type
22   */
23 case class Bounds[T](lower: Bound[T], upper: Bound[T]) {
24 
25   def bounds: (Option[T], Option[T]) = (lower.value, upper.value)
26 
27   /**
28     * Bounded on at least one side
29     *
30     * @return
31     */
32   def isBounded: Boolean = lower.value.nonEmpty || upper.value.nonEmpty
33 
34   /**
35     * Bounded on both sides
36     *
37     * @return
38     */
39   def isBoundedBothSides: Boolean = lower.value.nonEmpty && upper.value.nonEmpty
40 
41   /**
42     * Covers multiple values
43     *
44     * @return
45     */
46   def isRange: Boolean = lower.value.isEmpty || lower.value != upper.value
47 
48   /**
49     * Covers a single value
50     *
51     * @return
52     */
53   def isEquals: Boolean = !isRange
54 
55   /**
56     * Does this bounds cover the other bounds?
57     *
58     * @param other other bounds
59     * @return
60     */
61   def covers(other: Bounds[T]): Boolean = {
62     // if the end is unbounded, it will always cover, so use .forall
63     // if the end is bounded and other.end is unbounded, then it will never cover, so use .exists on the inner check
64     // if both end and other.end are bounded, then compare them
65     upper.value.forall { up =>
66       other.upper.value.exists { oup =>
67         val c = oup.asInstanceOf[Comparable[Any]].compareTo(up)
68         c < 0 || (c == 0 && (upper.inclusive || other.upper.exclusive))
69       }
70     } && lower.value.forall { lo =>
71       other.lower.value.exists { olo =>
72         val c = olo.asInstanceOf[Comparable[Any]].compareTo(lo)
73         c > 0 || (c == 0 && (lower.inclusive || other.lower.exclusive))
74       }
75     }
76   }
77 
78   /**
79     * Does this bounds intersect the other bounds?
80     *
81     * @param other other bounds
82     * @return
83     */
84   def intersects(other: Bounds[T]): Boolean = {
85     // if `largerLowerBound` or `smallerUpperBound` return None, then both this and other are unbounded
86     // on the same end, so they by definition overlap - so use .forall
87     val largerLowerBound = Bounds.largerLowerBound(lower, other.lower)
88     largerLowerBound.value.forall { lo =>
89       val smallerUpperBound = Bounds.smallerUpperBound(upper, other.upper)
90       smallerUpperBound.value.forall { up =>
91         // if largerLowerBound bound is smaller than smallerUpperBound, then they overlap
92         val c = lo.asInstanceOf[Comparable[Any]].compareTo(up)
93         c < 0 || (c == 0 && largerLowerBound.inclusive && smallerUpperBound.inclusive)
94       }
95     }
96   }
97 
98   override def toString: String = {
99     (if (lower.inclusive) { "[" } else { "(" }) + lower.value.getOrElse("-\u221E") + "," +
100       upper.value.getOrElse("+\u221E") + (if (upper.inclusive) { "]" } else { ")" })
101   }
102 }

104 object Bounds {
105 
106   /**
107     * Single bound (lower or upper).
108     *
109     * Bound may be unbounded, in which case value is None. Note by convention unbounded bounds are exclusive
110     *
111     * @param value value of this bound, if bounded
112     * @param inclusive whether the bound is inclusive or exclusive.
113     *                  for example, 'foo < 5' is exclusive, 'foo <= 5' is inclusive
114     */
115   case class Bound[T](value: Option[T], inclusive: Boolean) {
116     def exclusive: Boolean = !inclusive
117   }
118 
119   object Bound {
120     private val unboundedBound = Bound[Any](None, inclusive = false)
121     def unbounded[T]: Bound[T] = unboundedBound.asInstanceOf[Bound[T]]
122     def inclusive[T](value: T): Bound[T] = Bound(Option(value), inclusive = true)
123   }
124 
125   private val allValues = Bounds(Bound.unbounded, Bound.unbounded)
126 
127   def everything[T]: Bounds[T] = allValues.asInstanceOf[Bounds[T]]
128 
129   /**
130     * Gets the smaller value between two lower bounds, taking into account exclusivity.
131     * If the bounds are equal, the first bound will always be returned
132     *
133     * @param bound1 first bound
134     * @param bound2 second bound
135     * @return smaller bound
136     */
137   def smallerLowerBound[T](bound1: Bound[T], bound2: Bound[T]): Bound[T] = {
138     if (bound1.value.isEmpty) {
139       bound1
140     } else if (bound2.value.isEmpty) {
141       bound2
142     } else {
143       val c = bound1.value.get.asInstanceOf[Comparable[Any]].compareTo(bound2.value.get)
144       if (c < 0 || (c == 0 && (bound1.inclusive || bound2.exclusive))) { bound1 } else { bound2 }
145     }
146   }
147 
148   /**
149     * Gets the larger value between two upper bounds, taking into account exclusivity.
150     * If the bounds are equal, the first bound will always be returned
151     *
152     * @param bound1 first bound

154     * @return larger bound
155     */

167   /**
168     * Gets the smaller value between two upper bounds, taking into account exclusivity.

174     */

176     if (bound2.value.isEmpty) {

178     } else if (bound1.value.isEmpty) {

183     }
184   }

186   /**
187     * Gets the larger value between two upper bounds, taking into account exclusivity.
188     * If the bounds are equal, the first bound will always be returned
189     *
190     * @param bound1 first bound
191     * @param bound2 second bound
192     * @return larger bound
193     */
194   def largerLowerBound[T](bound1: Bound[T], bound2: Bound[T]): Bound[T] = {
195     if (bound2.value.isEmpty) {
196       bound1
197     } else if (bound1.value.isEmpty) {
198       bound2
199     } else {
200       val c = bound1.value.get.asInstanceOf[Comparable[Any]].compareTo(bound2.value.get)
201       if (c > 0 || (c == 0 && (bound2.inclusive || bound1.exclusive))) { bound1 } else { bound2 }
202     }
203   }

205   /**
206     * Takes the intersection of two bounds. If they are disjoint, will return None.
207     *
208     * @param left first bounds
209     * @param right second bounds
210     * @tparam T type parameter
211     * @return intersection
212     */
213   def intersection[T](left: Bounds[T], right: Bounds[T]): Option[Bounds[T]] = {
214     val lower = largerLowerBound(left.lower, right.lower)
215     val upper = smallerUpperBound(right.upper, left.upper)
216     (lower.value, upper.value) match {
217       case (Some(lo), Some(up)) =>
218         lo.asInstanceOf[Comparable[Any]].compareTo(up) match {
219           case i if i > 0 || (i == 0 && (!lower.inclusive || !upper.inclusive)) => None
220           case _ => Some(Bounds(lower, upper))
221         }
222       case _ => Some(Bounds(lower, upper))
223     }
224   }
225 
226   /**
227     * Takes the union of two bound sequences.
228     *
229     * @param left first bounds
230     * @param right second bounds
231     * @tparam T type parameter
232     * @return union
233     */
234   def union[T](left: Seq[Bounds[T]], right: Seq[Bounds[T]]): Seq[Bounds[T]] = {
235 
236     def mergeOverlapping(bound: Bounds[T], rest: Seq[Bounds[T]]): Seq[Bounds[T]] = {
237       rest match {
238         case Seq() => Seq(bound)
239         case head +: tail =>
240           bound.upper match {
241             case Bound(None, _) => Seq(bound)
242             case Bound(Some(x), incX) =>
243               head.lower match {
244                 case Bound(None, _) =>
245                   mergeOverlapping(Bounds(head.lower, largerUpperBound(bound.upper, head.upper)), tail)
246                 case Bound(Some(y), incY) =>
247                   val c = y.asInstanceOf[Comparable[Any]].compareTo(x)
248                   if (c < 0 || (c == 0 && (incX || incY)))
249                     mergeOverlapping(Bounds(bound.lower, largerUpperBound(bound.upper, head.upper)), tail)
250                   else
251                     bound +: mergeOverlapping(head, tail)
252               }
253           }
254       }
255     }
256 
257     val bounds = (left ++ right).sortWith { case (a, b) => smallerLowerBound(a.lower, b.lower) == a.lower }
258     bounds match {
259       case Seq() => bounds
260       case head +: tail => mergeOverlapping(head, tail)
261     }
262   }
