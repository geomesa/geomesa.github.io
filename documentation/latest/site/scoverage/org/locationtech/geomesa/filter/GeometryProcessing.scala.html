41     * Extract geometries from a filter, to use for querying

50 object GeometryProcessing extends GeometryProcessing with LazyLogging {

52   private val SafeGeomString = "gm-safe"
53 
54   private val processor = FilterProperties.GeometryProcessing.get match {
55     case p if p.equalsIgnoreCase("spatial4j")  => Spatial4jStrategy
56     case p if p.equalsIgnoreCase("none") => NoneStrategy
57     case p =>
58       logger.warn(s"Invalid value for '${FilterProperties.GeometryProcessing.property}', using default (spatial4j): $p")
59       Spatial4jStrategy
60   }
61 
62   override def process(op: BinarySpatialOperator, sft: SimpleFeatureType, factory: FilterFactory): Filter =
63     processor.process(op, sft, factory)
64 
65   override def extract(op: BinarySpatialOperator, attribute: String): Seq[Geometry] =
66     processor.extract(op, attribute)
67 
68   /**
69     * Multiplier to convert to meters from a dwithin unit
70     *
71     * @param units units defined in a dwithin filter
72     * @return
73     */
74   def metersMultiplier(units: String): Double = {
75     if (units == null) { 1d } else {
76       units.trim.toLowerCase(Locale.US) match {
77         case "meters"         => 1d
78         case "kilometers"     => 1000d
79         case "feet"           => 0.3048
80         case "statute miles"  => 1609.347
81         case "nautical miles" => 1852d
82         case _                => 1d // not part of ECQL spec...
83       }
84     }
85   }
86 
87   /**
88     * Expand a geometry collection into a seq of geometries
89     *
90     * @param geometry geometry
91     * @return
92     */
93   private def flatten(geometry: Geometry): Seq[Geometry] = geometry match {
94     case g: GeometryCollection => Seq.tabulate(g.getNumGeometries)(g.getGeometryN).flatMap(flatten)
95     case _ => Seq(geometry)
96   }

98   private trait AbstractGeometryProcessing extends GeometryProcessing {
99 
100     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
101 
102     protected def split(geom: Geometry, op: BinarySpatialOperator): Geometry
103 
104     override def process(op: BinarySpatialOperator, sft: SimpleFeatureType, factory: FilterFactory): Filter = {
105       val prop = org.locationtech.geomesa.filter.checkOrderUnsafe(op.getExpression1, op.getExpression2)
106       val geom = FastConverter.evaluate(prop.literal, classOf[Geometry])
107       if (geom.getUserData == SafeGeomString) {
108         op // we've already visited this geom once
109       } else {
110         // check for null or empty attribute and replace with default geometry name
111         val attribute = Option(prop.name).filterNot(_.isEmpty).orElse(Option(sft).map(_.getGeomField)).orNull
112         // trim to world boundaries
113         val trimmedGeom = trimToWorld(geom)
114         if (trimmedGeom.isEmpty) {
115           Filter.EXCLUDE
116         } else {
117           val property = factory.property(attribute)
118           val filters = flatten(split(trimmedGeom, op)).map { geom =>
119             // mark it as being visited
120             geom.setUserData(SafeGeomString) // note: side effect
121             val literal = factory.literal(geom)
122             val (e1, e2) = if (prop.flipped) { (literal, property) } else { (property, literal) }
123             op match {
124               case _: Within     => factory.within(e1, e2)
125               case _: Intersects => factory.intersects(e1, e2)
126               case _: Overlaps   => factory.overlaps(e1, e2)
127               case d: DWithin    => factory.dwithin(e1, e2, d.getDistance, d.getDistanceUnits)
128               // use the direct constructor so that we preserve our geom user data
129               case _: BBOX       => new BBOXImpl(e1, e2)
130               case _: Contains   => factory.contains(e1, e2)
131             }
132           }
133           orFilters(filters)(factory)
134         }
135       }
136     }
137 
138     override def extract(op: BinarySpatialOperator, attribute: String): Seq[Geometry] = {
139       val geometry = for {
140         prop <- checkOrder(op.getExpression1, op.getExpression2)
141         if prop.name == null || prop.name == attribute
142         geom <- Option(FastConverter.evaluate(prop.literal, classOf[Geometry]))
143       } yield {
144         val buffered = op match {
145           case f: DWithin => geom.buffer(distanceDegrees(geom, f.getDistance * metersMultiplier(f.getDistanceUnits))._2)
146           case _          => geom
147         }
148         split(trimToWorld(buffered), op)
149       }
150       geometry.map(flatten).getOrElse(Seq.empty)
151     }
152   }
153 
154   private object Spatial4jStrategy extends AbstractGeometryProcessing {
155     override protected def split(geom: Geometry, op: BinarySpatialOperator): Geometry = {
156       // add waypoints if needed so that IDL is handled correctly
157       val waypoints = if (op.isInstanceOf[BBOX]) { FilterHelper.addWayPointsToBBOX(geom) } else { geom }
158       GeohashUtils.getInternationalDateLineSafeGeometry(waypoints) match {
159         case Success(g) => g
160         case Failure(e) => logger.warn(s"Error splitting geometry on AM for $waypoints", e); waypoints
161       }
162     }
163   }
164 
165   private object NoneStrategy extends AbstractGeometryProcessing {
166     override protected def split(geom: Geometry, op: BinarySpatialOperator): Geometry = geom
167   }
