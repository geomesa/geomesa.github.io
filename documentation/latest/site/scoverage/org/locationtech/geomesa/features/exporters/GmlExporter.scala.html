27 /**
28   * GML exporter implementation.
29   *
30   * The geotools GML export classes only support encoding a feature collection. To support our usage
31   * pattern (start, export n times, end), we create an asynchronous feature collection and do the actual
32   * encoding in a separate thread. The encoder thread will block until there are more features to export,
33   * so that we only get a single feature collection in the xml.
34   *
35   * @param out output stream
36   * @param configuration wfs configuration (gml3 vs gml2)
37   */
38 class GmlExporter private (out: OutputStream, configuration: WFSConfiguration)
39     extends FeatureExporter {
40 
41   private val encoder: Encoder = {
42     val props = configuration.getProperties
43     props.add(org.geotools.gml2.GMLConfiguration.OPTIMIZED_ENCODING)
44     props.add(org.geotools.gml2.GMLConfiguration.NO_FEATURE_BOUNDS)
45     val e = new Encoder(configuration)
46     e.getNamespaces.declarePrefix("geomesa", "http://geomesa.org")
47     e.setEncoding(StandardCharsets.UTF_8)
48     e.setIndenting(true)
49     e
50   }

55   override def start(sft: SimpleFeatureType): Unit = {
56     fc = new AsyncFeatureCollection(sft)
57     val features = if (sft.getName.getNamespaceURI != null) { fc } else {
58       val builder = new SimpleFeatureTypeBuilder()
59       builder.init(sft)
60       builder.setNamespaceURI("http://geomesa.org")

64     collection.getFeature.asInstanceOf[java.util.List[SimpleFeatureCollection]].add(features)
65 
66     def encode(): Unit = encoder.encode(collection, org.geotools.wfs.WFS.FeatureCollection, out)
67 
68     val runnable = new Runnable() {
69       override def run(): Unit = {
70         if (System.getProperty(GmlExporter.TransformerProperty) != null) { encode() } else {
71           // explicitly set the default java transformer, to avoid picking up saxon (which causes errors)
72           // the default class is hard-coded in javax.xml.transform.TransformerFactory.newInstance() ...
73           System.setProperty(GmlExporter.TransformerProperty,
74             "com.sun.org.apache.xalan.internal.xsltc.trax.TransformerFactoryImpl")
75           try { encode() } finally {
76             System.clearProperty(GmlExporter.TransformerProperty)
77           }
78         }
79       }
80     }

82     es.execute(runnable)
83   }
84 
85   override def export(features: Iterator[SimpleFeature]): Option[Long] = {
86     var count = 0L
87     val counting = features.map { f => count += 1; f }
88     while (counting.nonEmpty) {
89       // export in chunks of 100 so that the exporter thread gets notified and doesn't keep blocking
90       fc.addAsync(counting.take(100))
91     }
92     Some(count)
93   }
94 
95   override def close(): Unit = {
96     try {
97       if (fc != null) {
98         fc.endAsync()
99       }
100       es.shutdown()
101       es.awaitTermination(Long.MaxValue, TimeUnit.MILLISECONDS)
102     } finally {
103       out.close()
104     }
105   }

130   /**
131     * Feature collection that lets us add additional features in an asynchronous fashion. The consumer
132     * thread will be blocked on calls to 'hasNext' until the producer thread adds features or indicates
133     * completion
134     *
135     * @param sft simple feature type
136     */
137   private class AsyncFeatureCollection(sft: SimpleFeatureType) extends DataFeatureCollection(null, sft) {
138 
139     private val buffer = new ConcurrentLinkedQueue[SimpleFeature]()
140     private val done = new AtomicBoolean(false)
141     private val lock = new ReentrantLock()
142     private val condition = lock.newCondition()
143 
144     private val iter: SimpleFeatureIterator = new SimpleFeatureIterator() {
145 
146       private var current: SimpleFeature = _
147 
148       override def hasNext: Boolean = {
149         if (current != null) {
150           return true
151         }
152         lock.lock()
153         try {
154           current = buffer.poll()
155           // note: we need to loop here to skip 'spurious wake-ups'
156           while (current == null) {
157             if (done.get) {
158               return false
159             }
160             condition.await()
161             current = buffer.poll()
162           }
163           true
164         } finally {
165           lock.unlock()
166         }
167       }
168 
169       override def next(): SimpleFeature = {
170         // note: we shouldn't need to synchronize this as next/hasNext should be a single caller thread
171         val result = current
172         current = null
173         result
174       }
175 
176       override def close(): Unit = endAsync()
177     }
178 
179     /**
180       * Add features to be returned from this feature collection
181       *
182       * @param features features
183       */
184     def addAsync(features: Iterator[SimpleFeature]): Unit = {
185       lock.lock()
186       try {
187         features.foreach(buffer.add)
188         condition.signal()
189       } finally {
190         lock.unlock()
191       }
192     }
193 
194     /**
195       * Signal that there are no more features that will be added
196       */
197     def endAsync(): Unit = {
198       lock.lock()
199       try {
200         done.set(true)
201         condition.signal()
202       } finally {
203         lock.unlock()
204       }
205     }
206
