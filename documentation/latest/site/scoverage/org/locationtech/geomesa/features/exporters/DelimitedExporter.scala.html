23 class DelimitedExporter(out: OutputStream, format: CSVFormat, withHeader: Boolean, includeIds: Boolean)
24     extends FeatureExporter with LazyLogging {
25 
26   import org.locationtech.geomesa.utils.geotools.GeoToolsDateFormat
27 
28   import scala.collection.JavaConverters._
29 
30   private var printer: CSVPrinter = _
31 
32   override def start(sft: SimpleFeatureType): Unit = {
33     printer = format.print(new OutputStreamWriter(out, StandardCharsets.UTF_8))
34     // write out a header line
35     if (withHeader) {
36       if (includeIds) {
37         printer.print("id")
38       }
39       sft.getAttributeDescriptors.asScala.foreach { descriptor =>
40         printer.print(SimpleFeatureTypes.encodeDescriptor(sft, descriptor))
41       }
42       printer.println()
43       printer.flush()
44     }
45   }
46 
47   override def export(features: Iterator[SimpleFeature]): Option[Long] = {
48     var count = 0L
49     features.foreach { sf =>
50       if (includeIds) {
51         printer.print(sf.getID)
52       }
53       var i = 0
54       while (i < sf.getAttributeCount) {
55         printer.print(stringify(sf.getAttribute(i)))
56         i += 1
57       }
58       printer.println()
59 
60       count += 1
61       if (count % 10000 == 0) {
62         logger.debug(s"wrote $count features")
63       }
64     }
65 
66     printer.flush()
67 
68     logger.debug(s"Exported $count features")
69     Some(count)
70   }
71 
72   override def close(): Unit =  if (printer != null) { printer.close() }
73 
74   private def stringify(obj: Any): String = obj match {
75     case null                   => ""
76     case g: Geometry            => WKTUtils.write(g)
77     case d: Date                => GeoToolsDateFormat.format(Instant.ofEpochMilli(d.getTime).atZone(ZoneOffset.UTC))
78     case l: java.util.List[_]   => l.asScala.map(stringify).mkString(",")
79     case m: java.util.Map[_, _] => m.asScala.map { case (k, v) => s"${stringify(k)}->${stringify(v)}"}.mkString(",")
80     case _                      => obj.toString
81   }
82 }
