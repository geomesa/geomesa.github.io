13 /**
14   * A bit set backed by ints (to reduce memory footprint for small sizes).
15   *
16   * Provides access to the underlying mask, for serialization. Not thread safe.
17   */
18 trait IntBitSet {
19 
20   /**
21     * Checks whether the value is contained in the set or not
22     *
23     * @param value value to check
24     * @return true if contains, false otherwise
25     */
26   def contains(value: Int): Boolean
27 
28   /**
29     * Adds the value to the set, if it is not present. Note that implementations may be constructed with
30     * a fixed capacity - attempting to add a value outside the capacity will result in undefined behavior
31     *
32     * @param value value to add
33     * @return true if value was added, false if value was already present
34     */
35   def add(value: Int): Boolean
36 
37   /**
38     * Removes the value from the set, if it is present
39     *
40     * @param value value to remove
41     * @return true if value was removed, false if value was not present
42     */
43   def remove(value: Int): Boolean
44 
45   /**
46     * Resets all values
47     */
48   def clear(): Unit
49 
50   /**
51     * Serialize to a kryo output stream
52     *
53     * @param output output
54     */
55   def serialize(output: Output): Unit
56 }

58 object IntBitSet {
59 
60   // use `>> 5` instead of `/ 32`
61   private final val Divisor = 5
62 
63   /**
64     * Create a bit set capable of holding numbers up to `length`
65     *
66     * @param length max int to be stored in the bit set
67     * @return
68     */
69   def apply(length: Int): IntBitSet = {
70     size(length) match {
71       case 1 => new BitSet32()
72       case 2 => new BitSet64()
73       case 3 => new BitSet96()
74       case 4 => new BitSet128()
75       case 5 => new BitSet160()
76       case 6 => new BitSet192()
77       case n => new BitSetN(Array.fill(n)(0))
78     }
79   }
80 
81   /**
82     * Deserialize from a kryo input stream
83     *
84     * @param in input
85     * @param length max int that the serialized bit set was capabable of storing
86     * @return
87     */
88   def deserialize(in: Input, length: Int): IntBitSet = {
89     size(length) match {
90       case 1 => new BitSet32(in.readInt())
91       case 2 => new BitSet64(in.readInt(), in.readInt())
92       case 3 => new BitSet96(in.readInt(), in.readInt(), in.readInt())
93       case 4 => new BitSet128(in.readInt(), in.readInt(), in.readInt(), in.readInt())
94       case 5 => new BitSet160(in.readInt(), in.readInt(), in.readInt(), in.readInt(), in.readInt())
95       case 6 => new BitSet192(in.readInt(), in.readInt(), in.readInt(), in.readInt(), in.readInt(), in.readInt())
96       case n => new BitSetN(Array.fill(n)(in.readInt()))
97     }
98   }
99 
100   /**
101     * Gets the number of ints (words) required to hold numbers up to `length`
102     *
103     * @param length max int to be stored in the bit set
104     * @return
105     */
106   def size(length: Int): Int = ((length - 1) >> Divisor) + 1
107 
108   /**
109     * Checks if the mask contains the value
110     *
111     * @param mask mask
112     * @param value value
113     * @return
114     */
115   private def contains(mask: Int, value: Int): Boolean = (mask & (1 << value)) != 0
116 
117   /**
118     * Checks if the mask contains the value, and invokes `update` on the new value if not
119     *
120     * @param mask mask
121     * @param value value
122     * @param update callback for the updated value
123     * @return
124     */
125   private def add(mask: Int, value: Int, update: Int => Unit): Boolean = {
126     val updated = mask | (1 << value)
127     if (updated == mask) { false } else {
128       update(updated)
129       true
130     }
131   }
132 
133   /**
134     * Checks if the mask contains the value, and invokes `update` if it does
135     *
136     * @param mask mask
137     * @param value value
138     * @param update callback for the updated value
139     * @return
140     */
141   private def remove(mask: Int, value: Int, update: Int => Unit): Boolean = {
142     val updated = mask & ~(1 << value)
143     if (updated == mask) { false } else {
144       update(updated)
145       true
146     }
147   }

479   /**
480     * Bit set for tracking any number of values
481     *
482     * @param masks underlying mask values
483     */
484   class BitSetN(masks: Array[Int]) extends IntBitSet {
485 
486     override def contains(value: Int): Boolean = (masks(value >> Divisor) & (1 << value)) != 0
487 
488     override def add(value: Int): Boolean = {
489       val word = value >> Divisor
490       IntBitSet.add(masks(word), value, updated => masks(word) = updated)
491     }
492 
493     override def remove(value: Int): Boolean = {
494       val word = value >> Divisor
495       IntBitSet.remove(masks(word), value, updated => masks(word) = updated)
496     }
497 
498     override def clear(): Unit = {
499       var i = 0
500       while (i < masks.length) {
501         masks(i) = 0
502         i += 1
503       }
504     }
505 
506     override def serialize(output: Output): Unit = masks.foreach(output.writeInt)
507   }
