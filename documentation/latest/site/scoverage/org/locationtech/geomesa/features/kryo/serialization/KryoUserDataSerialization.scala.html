63   private def serialize(
64       out: Output,
65       javaMap: java.util.Map[_ <: AnyRef, _ <: AnyRef],
66       withoutFidHints: Boolean,
67       writeAscii: Boolean): Unit = {
68     import scala.collection.JavaConverters._
69 
70     // write in sorted order to keep consistent output
71     val toWrite = scala.collection.mutable.SortedSet.empty[(AnyRef, AnyRef)]
72 
73     javaMap.asScala.foreach {
74       case (k, v) if k != null && !k.isInstanceOf[Hints.Key] => toWrite += k -> v
75       case (Hints.USE_PROVIDED_FID, _) if withoutFidHints => // no-op
76       case (Hints.PROVIDED_FID, _) if withoutFidHints => // no-op
77       case (k, v) => logger.warn(s"Skipping serialization of entry: $k -> $v")
78     }
79 
80     out.writeInt(toWrite.size) // don't use positive optimized version for back compatibility
81 
82     toWrite.foreach { case (key, value) =>
83       out.writeString(baseClassMappings.getOrElse(key.getClass, key.getClass.getName))
84       write(out, key, writeAscii)
85       if (value == null) {
86         out.writeString(nullMapping)
87       } else {
88         out.writeString(baseClassMappings.getOrElse(value.getClass, value.getClass.getName))
89         write(out, value, writeAscii)
90       }
91     }
92   }

94   def deserialize(in: Input): java.util.Map[AnyRef, AnyRef] = {
95     try {
96       val size = in.readInt()
97       val map = new java.util.HashMap[AnyRef, AnyRef](size)
98       deserializeWithSize(in, map, size)
99       map
100     } catch {
101       case NonFatal(e) =>
102         logger.error("Error reading serialized kryo user data:", e)
103         new java.util.HashMap[AnyRef, AnyRef]()
104     }
105   }

107   def deserialize(in: Input, map: java.util.Map[AnyRef, AnyRef]): Unit = {
108     try {

112         logger.error("Error reading serialized kryo user data:", e)
113         new java.util.HashMap[AnyRef, AnyRef]()

117   private def deserializeWithSize(in: Input, map: java.util.Map[AnyRef, AnyRef], size: Int): Unit = {
118     var i = 0
119     while (i < size) {
120       val keyClass = in.readString()
121       val key = read(in, baseClassLookups.getOrElse(keyClass, Class.forName(keyClass)))
122       val valueClass = in.readString()
123       val value = if (valueClass == nullMapping) { null } else {
124         read(in, baseClassLookups.getOrElse(valueClass, Class.forName(valueClass)))
125       }
126       map.put(key, value)
127       i += 1
128     }
129   }

131   private def write(out: Output, value: AnyRef, writeAscii: Boolean = false): Unit = value match {
132     case v: String                 => if (writeAscii) { out.writeAscii(v) } else { out.writeString(v) }
133     case v: java.lang.Integer      => out.writeInt(v)
134     case v: java.lang.Long         => out.writeLong(v)
135     case v: java.lang.Float        => out.writeFloat(v)
136     case v: java.lang.Double       => out.writeDouble(v)
137     case v: java.lang.Boolean      => out.writeBoolean(v)
138     case v: Date                   => out.writeLong(v.getTime)
139     case v: Array[Byte]            => writeBytes(out, v)
140     case v: Geometry               => KryoGeometrySerialization.serializeWkb(out, v)
141     case v: UUID                   => out.writeLong(v.getMostSignificantBits); out.writeLong(v.getLeastSignificantBits)
142     case v: java.util.List[AnyRef] => writeList(out, v)
143     case _ => throw new IllegalArgumentException(s"Unsupported value: $value (${value.getClass})")
144   }
145 
146   /**
147    * Read a key or value. Strings will be interned, as we expect a lot of duplication in user data,
148    * i.e keys but also visibilities, which is the only user data we generally store
149    *
150    * @param in input
151    * @param clas class of the item to read
152    * @return
153    */
154   private def read(in: Input, clas: Class[_]): AnyRef = clas match {
155     case c if classOf[java.lang.String].isAssignableFrom(c)  => in.readString().intern()
156     case c if classOf[java.lang.Integer].isAssignableFrom(c) => Int.box(in.readInt())
157     case c if classOf[java.lang.Long].isAssignableFrom(c)    => Long.box(in.readLong())
158     case c if classOf[java.lang.Float].isAssignableFrom(c)   => Float.box(in.readFloat())
159     case c if classOf[java.lang.Double].isAssignableFrom(c)  => Double.box(in.readDouble())
160     case c if classOf[java.lang.Boolean].isAssignableFrom(c) => Boolean.box(in.readBoolean())
161     case c if classOf[java.util.Date].isAssignableFrom(c)    => new java.util.Date(in.readLong())
162     case c if classOf[Array[Byte]] == c                      => readBytes(in)
163     case c if classOf[Geometry].isAssignableFrom(c)          => KryoGeometrySerialization.deserializeWkb(in, checkNull = true)
164     case c if classOf[UUID].isAssignableFrom(c)              => new UUID(in.readLong(), in.readLong())
165     case c if classOf[java.util.List[_]].isAssignableFrom(c) => readList(in)
166     case c if classOf[Hints.Key].isAssignableFrom(c)         => HintKeySerialization.idToKey(in.readString())
167     case _ => throw new IllegalArgumentException(s"Unsupported value class: $clas")
168   }
169 
170   private def writeBytes(out: Output, bytes: Array[Byte]): Unit = {
171     out.writeInt(bytes.length)
172     out.writeBytes(bytes)
173   }
174 
175   private def readBytes(in: Input): Array[Byte] = {
176     val bytes = Array.ofDim[Byte](in.readInt)
177     in.readBytes(bytes)
178     bytes
179   }
180 
181   private def writeList(out: Output, list: java.util.List[AnyRef]): Unit = {
182     out.writeInt(list.size)
183     val iterator = list.iterator()
184     while (iterator.hasNext) {
185       val value = iterator.next()
186       if (value == null) {
187         out.writeString(nullMapping)
188       } else {
189         out.writeString(baseClassMappings.getOrElse(value.getClass, value.getClass.getName))
190         write(out, value)
191       }
192     }
193   }
194 
195   private def readList(in: Input): java.util.List[AnyRef] = {
196     val size = in.readInt()
197     val list = new java.util.ArrayList[AnyRef](size)
198     var i = 0
199     while (i < size) {
200       val clas = in.readString()
201       if (clas == nullMapping) { list.add(null) } else {
202         list.add(read(in, baseClassLookups.getOrElse(clas, Class.forName(clas))))
203       }
204       i += 1
205     }
206     list
207   }
