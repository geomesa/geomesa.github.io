36 /**
37  * Kryo input that doesn't mutate the underlying buffer when reading
38  */
39 class NonMutatingInput extends Input {
40 
41   def this(buffer: Array[Byte]) = {
42     this()
43     setBuffer(buffer)
44   }
45 
46   override def readString(): String = {
47     if (position == limit) {
48       require(1)
49     }
50     if ((buffer(position) & 0x80) == 0) {
51       readAsciiString()
52     } else {
53       // null, empty or utf8
54       var charCount = readVarIntLength()
55       if (charCount == 0) {
56         null
57       } else if (charCount == 1) {
58         ""
59       } else {
60         charCount -= 1
61         readUtf8Chars(charCount)
62         new String(chars, 0, charCount)
63       }
64     }
65   }
66 
67   override def readStringBuilder(): java.lang.StringBuilder = {
68     if (position == limit) {
69       require(1)
70     }
71     if ((buffer(position) & 0x80) == 0) {
72       new java.lang.StringBuilder(readAsciiString())
73     } else {
74       // null, empty or utf8
75       var charCount = readVarIntLength()
76       if (charCount == 0) {
77         null
78       } else if (charCount == 1) {
79         new java.lang.StringBuilder(0)
80       } else {
81         charCount -= 1
82         readUtf8Chars(charCount)
83         val builder = new java.lang.StringBuilder(charCount)
84         builder.append(chars, 0, charCount)
85         builder
86       }
87     }
88   }

90   private def readAsciiString(): String = {
91     val chars = this.chars
92     val buffer = this.buffer
93     var charCount = 0
94     var p = position
95     val n = Math.min(chars.length, limit - p)
96     while (charCount < n) {
97       val b = buffer(p)
98       if ((b & 0x80) == 0x80) {
99         position = p + 1
100         chars(charCount) = (b & 0x7F).toChar
101         return new String(chars, 0, charCount + 1)
102       }
103       chars(charCount) = b.toChar
104       charCount += 1
105       p += 1
106     }
107     position = p
108     readAsciiString_slow(charCount)
109   }
110 
111   private def readAsciiString_slow(count: Int): String = {
112     var chars = this.chars
113     val buffer = this.buffer
114     var charCount = count
115     var p = position
116     while (true) {
117       if (position == limit) {
118         require(1)
119       }
120       val b = buffer(p)
121       if (charCount == chars.length) {
122         val newChars = new Array[Char](charCount * 2)
123         System.arraycopy(chars, 0, newChars, 0, charCount)
124         chars = newChars
125         this.chars = chars
126       }
127       if ((b & 0x80) == 0x80) {
128         position = p + 1
129         chars(charCount) = (b & 0x7F).toChar
130         return new String(chars, 0, charCount + 1)
131       }
132       chars(charCount) = b.toChar
133       charCount += 1
134       p += 1
135     }
136     throw new IllegalStateException() // can't ever reach here
137   }

139   private def readUtf8Chars(charCount: Int): Unit = {
140     if (this.chars.length < charCount) {
141       this.chars = Array.ofDim[Char](charCount)
142     }
143     val buffer = this.buffer
144     val chars = this.chars
145     // try to read 7 bit ASCII chars
146     var count = Math.min(require(1), charCount)
147     var charIndex = 0
148     var p = position
149     var b = 0
150     while (charIndex < count) {
151       b = buffer(p)
152       if (b < 0) {
153         count = 0 // break out of the loop
154       } else {
155         chars(charIndex) = b.toChar
156         charIndex += 1
157         p += 1
158       }
159     }
160     position = p
161     // if buffer didn't hold all chars or any were not ASCII, use slow path for remainder.
162     if (charIndex < charCount) {
163       readUtf8Chars_slow(charCount, charIndex)
164     }
165   }
166 
167   private def readUtf8Chars_slow(charCount: Int, index: Int): Unit = {
168     val chars = this.chars
169     val buffer = this.buffer
170     var charIndex = index
171     while (charIndex < charCount) {
172       if (position == limit) {
173         require(1)
174       }
175       val b = buffer(position) & 0xFF
176       position += 1
177       val i = b >> 4
178       if (i >= 0 && i < 8) {
179         chars(charIndex) = b.toChar
180       } else if (i == 12 || i == 13) {
181         if (position == limit) {
182           require(1)
183         }
184         chars(charIndex) = ((b & 0x1F) << 6 | buffer(position) & 0x3F).toChar
185         position += 1
186       } else if (i == 14) {
187         require(2)
188         chars(charIndex) = ((b & 0x0F) << 12 | (buffer(position) & 0x3F) << 6 | buffer(position + 1) & 0x3F).toChar
189         position += 2
190       }
191       charIndex += 1
192     }
193   }
194 
195   private def readVarIntLength(): Int = {
196     if (require(1) < 5) {
197       readVarIntLength_slow()
198     } else {
199       var b = buffer(position)
200       position += 1
201       var result = b & 0x3F // mask first 6 bits
202       if ((b & 0x40) != 0) { // bit 7 means another byte, bit 8 means UTF8
203         b = buffer(position)
204         position += 1
205         result |= (b & 0x7F) << 6
206         if ((b & 0x80) != 0) {
207           b = buffer(position)
208           position += 1
209           result |= (b & 0x7F) << 13
210           if ((b & 0x80) != 0) {
211             b = buffer(position)
212             position += 1
213             result |= (b & 0x7F) << 20
214             if ((b & 0x80) != 0) {
215               b = buffer(position)
216               position += 1
217               result |= (b & 0x7F) << 27
218             }
219           }
220         }
221       }
222       result
223     }
224   }
