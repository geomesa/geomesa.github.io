29   /**
30    * Encoder/decoder for index values. Allows customizable fields to be encoded. Not thread-safe.
31    *
32    * @param sft simple feature type
33    */
34   @deprecated
35   class IndexValueEncoderImpl(sft: SimpleFeatureType)
36     extends org.locationtech.geomesa.features.SimpleFeatureSerializer with LimitedSerialization {
37 
38     import scala.collection.JavaConverters._
39 
40     private val indexSft = getIndexSft(sft)
41     private val encoder = KryoFeatureSerializer(indexSft)
42     private val reusableFeature = new ScalaSimpleFeature(indexSft, "")
43     private val indices = indexSft.getAttributeDescriptors.asScala.map(ad => sft.indexOf(ad.getLocalName)).toArray
44 
45     override val options: Set[SerializationOption] = Set.empty
46 
47     override def serialize(sf: SimpleFeature): Array[Byte] = {
48       reusableFeature.getIdentifier.asInstanceOf[FeatureIdImpl].setID(sf.getID)
49       var i = 0
50       while (i < indices.length) {
51         reusableFeature.setAttribute(i, sf.getAttribute(indices(i)))
52         i += 1
53       }
54       encoder.serialize(reusableFeature)
55     }
56 
57     override def deserialize(value: Array[Byte]): SimpleFeature = encoder.deserialize(value)
58   }
59 }

68   private val cache = new ConcurrentHashMap[ImmutableSimpleFeatureType, ImmutableSimpleFeatureType]()
69 
70   def apply(sft: SimpleFeatureType): org.locationtech.geomesa.features.SimpleFeatureSerializer =
71     new ProjectingKryoFeatureSerializer(sft, getIndexSft(sft), SerializationOption.WithoutId)

73   /**
74    * Gets a feature type compatible with the stored index value
75    *
76    * @param sft simple feature type
77    * @return
78    */
79   def getIndexSft(sft: SimpleFeatureType): SimpleFeatureType = {
80     sft match {
81       case immutable: ImmutableSimpleFeatureType =>
82         var indexSft = cache.get(immutable)
83         if (indexSft == null) {
84           indexSft = SimpleFeatureTypes.immutable(buildIndexSft(sft)).asInstanceOf[ImmutableSimpleFeatureType]
85           cache.put(immutable, indexSft)
86         }
87         indexSft
88 
89       case _ => buildIndexSft(sft)
90     }
91   }
92 
93   private def buildIndexSft(sft: SimpleFeatureType): SimpleFeatureType = {
94     val builder = new SimpleFeatureTypeBuilder()
95     builder.setNamespaceURI(null: String)
96     builder.setName(sft.getTypeName + "--index")
97     builder.setAttributes(getIndexValueAttributes(sft).asJava)
98     if (sft.getGeometryDescriptor != null) {
99       builder.setDefaultGeometry(sft.getGeometryDescriptor.getLocalName)
100     }
101     builder.setCRS(sft.getCoordinateReferenceSystem)
102     val indexSft = builder.buildFeatureType()
103     indexSft.getUserData.putAll(sft.getUserData)
104     indexSft
105   }
106 
107   /**
108    * Gets the attributes that are stored in the index value
109    *
110    * @param sft simple feature type
111    * @return
112    */
113   private def getIndexValueAttributes(sft: SimpleFeatureType): Seq[AttributeDescriptor] = {
114     val geom = sft.getGeometryDescriptor
115     val dtg = sft.getDtgField
116     val attributes = scala.collection.mutable.Buffer.empty[AttributeDescriptor]
117     var i = 0
118     while (i < sft.getAttributeCount) {
119       val ad = sft.getDescriptor(i)
120       if (ad == geom || dtg.contains(ad.getLocalName) || ad.isIndexValue()) {
121         attributes.append(ad)
122       }
123       i += 1
124     }
125     attributes.toSeq
126   }
