17   * Based on the method from geotools WKBWriter. This method is optimized for kryo and simplified from
18   * WKBWriter in the following ways:
19   *
20   * 1. Doesn't save SRID (geomesa didn't use that functionality in WKBWriter)
21   * 2. Doesn't worry about byte order (handled by kryo)  TODO does avro handle byte order?
22   * 3. Doesn't use a precision model
23   */

42   def serializeWkb(out: T, geometry: Geometry): Unit = {
43     if (geometry == null) { out.writeByte(NULL_BYTE) } else {
44       out.writeByte(NOT_NULL_BYTE)
45       geometry match {
46         case g: Point              => writePoint(out, g)
47         case g: LineString         => writeLineString(out, g)
48         case g: Polygon            => writePolygon(out, g)
49         case g: MultiPoint         => writeGeometryCollection(out, WkbSerialization.MultiPoint, g)
50         case g: MultiLineString    => writeGeometryCollection(out, WkbSerialization.MultiLineString, g)
51         case g: MultiPolygon       => writeGeometryCollection(out, WkbSerialization.MultiPolygon, g)
52         case g: GeometryCollection => writeGeometryCollection(out, WkbSerialization.GeometryCollection, g)
53       }
54     }
55   }

57   def deserializeWkb(in: V, checkNull: Boolean = false): Geometry = deserializeWkb(in, checkNull, 0)

59   private def deserializeWkb(in: V, checkNull: Boolean, nesting: Int): Geometry = {
60     if (checkNull && in.readByte() == NULL_BYTE) { null } else {
61       in.readInt(true) match {
62         case Point              => readPoint(in, xySerializer)
63         case LineString         => readLineString(in, xySerializer)
64         case Polygon            => readPolygon(in, xySerializer)
65         case PointXYZ           => readPoint(in, xyzSerializer)
66         case LineStringXYZ      => readLineString(in, xyzSerializer)
67         case PolygonXYZ         => readPolygon(in, xyzSerializer)
68         case PointXYM           => readPoint(in, xymSerializer)
69         case LineStringXYM      => readLineString(in, xymSerializer)
70         case PolygonXYM         => readPolygon(in, xymSerializer)
71         case PointXYZM          => readPoint(in, xyzmSerializer)
72         case LineStringXYZM     => readLineString(in, xyzmSerializer)
73         case PolygonXYZM        => readPolygon(in, xyzmSerializer)
74         case MultiPoint         => factory.createMultiPoint(readGeometryCollection[Point](in, nesting))
75         case MultiLineString    => factory.createMultiLineString(readGeometryCollection[LineString](in, nesting))
76         case MultiPolygon       => factory.createMultiPolygon(readGeometryCollection[Polygon](in, nesting))
77         case GeometryCollection => factory.createGeometryCollection(readGeometryCollection[Geometry](in, nesting))
78         // legacy encodings - dimension serialized as a separate int
79         case 8                  => readLegacyPoint(in)
80         case 9                  => readLegacyLineString(in)
81         case 10                 => readLegacyPolygon(in)
82         case f => throw new IllegalArgumentException(s"Expected geometry type byte, but got $f")
83       }
84     }
85   }

87   private def writePoint(out: T, g: Point): Unit = {
88     val writer = getWriter(g)
89     out.writeInt(writer.flag + Point, optimizePositive = true)
90     writeCoordinateSequence(out, writer, g.getCoordinateSequence, writeLength = false)
91   }

93   private def readPoint(in: V, reader: CoordinateSerializer): Point =
94     factory.createPoint(readCoordinateSequence(in, reader, Some(1)))

96   private def readLegacyPoint(in: V): Point = {
97     in.readInt(optimizePositive = true) match {
98       case 2 => readPoint(in, xySerializer)

100       case i => throw new IllegalArgumentException(s"Expected 2 or 3 dimensions, but got $i")
101     }

123   private def writePolygon(out: T, g: Polygon): Unit = {
124     val writer = getWriter(g)
125     out.writeInt(writer.flag + Polygon, optimizePositive = true)
126     val exterior = g.getExteriorRing.getCoordinateSequence
127     writeCoordinateSequence(out, writer, exterior, writeLength = true)
128     out.writeInt(g.getNumInteriorRing, optimizePositive = true)
129     var i = 0
130     while (i < g.getNumInteriorRing) {
131       writeCoordinateSequence(out, writer, g.getInteriorRingN(i).getCoordinateSequence, writeLength = true)
132       i += 1
133     }
134   }

136   private def readPolygon(in: V, reader: CoordinateSerializer): Polygon = {
137     val exteriorRing = factory.createLinearRing(readCoordinateSequence(in, reader, None))
138     val numInteriorRings = in.readInt(true)
139     if (numInteriorRings == 0) {
140       factory.createPolygon(exteriorRing)
141     } else if (numInteriorRings > maxLength) {
142       throw new IllegalArgumentException(
143         s"Attempting to deserialize a polygon of size $numInteriorRings " +
144             s"with '${GeometryLengthThreshold.property}' = $maxLength'")
145     } else {
146       val interiorRings = Array.ofDim[LinearRing](numInteriorRings)
147       var i = 0
148       while (i < numInteriorRings) {
149         interiorRings(i) = factory.createLinearRing(readCoordinateSequence(in, reader, None))
150         i += 1
151       }
152       factory.createPolygon(exteriorRing, interiorRings)
153     }
154   }

156   private def readLegacyPolygon(in: V): Polygon = {
157     def readLinearRing(): LinearRing = {
158       val length = in.readInt(optimizePositive = true)
159       in.readInt(optimizePositive = true) match {
160         case 2 => factory.createLinearRing(readCoordinateSequence(in, xySerializer, Some(length)))
161         case 3 => factory.createLinearRing(readCoordinateSequence(in, xyzSerializer, Some(length)))
162         case i => throw new IllegalArgumentException(s"Expected 2 or 3 dimensions, but got $i")
163       }
164     }
165     val exteriorRing = readLinearRing()
166     val numInteriorRings = in.readInt(true)
167     if (numInteriorRings == 0) {
168       factory.createPolygon(exteriorRing)
169     } else if (numInteriorRings > maxLength) {
170       throw new IllegalArgumentException(
171         s"Attempting to deserialize a polygon of size $numInteriorRings " +
172             s"with '${GeometryLengthThreshold.property}' = $maxLength'")
173     } else {
174       val interiorRings = Array.ofDim[LinearRing](numInteriorRings)
175       var i = 0
176       while (i < numInteriorRings) {
177         interiorRings(i) = readLinearRing()
178         i += 1
179       }
180       factory.createPolygon(exteriorRing, interiorRings)
181     }
182   }

184   private def writeGeometryCollection(out: T, typ: Int, g: GeometryCollection): Unit = {
185     out.writeInt(typ, optimizePositive = true)
186     out.writeInt(g.getNumGeometries, optimizePositive = true)
187     var i = 0
188     while (i < g.getNumGeometries) {
189       serializeWkb(out, g.getGeometryN(i))
190       i += 1
191     }
192   }

194   private def readGeometryCollection[U <: Geometry: ClassTag](in: V, nesting: Int): Array[U] = {
195     if (nesting > maxNesting) {
196       throw new IllegalArgumentException(s"Detected recursive deserialization loop of $maxNesting")
197     }
198     val numGeoms = in.readInt(true)
199     if (numGeoms > maxLength) {
200       throw new IllegalArgumentException(
201         s"Attempting to deserialize a geometry of size $numGeoms " +
202             s"with '${GeometryLengthThreshold.property}' = $maxLength'")
203     }
204     val geoms = Array.ofDim[U](numGeoms)
205     var i = 0
206     while (i < numGeoms) {
207       geoms.update(i, deserializeWkb(in, checkNull = true, nesting + 1).asInstanceOf[U])
208       i += 1
209     }
210     geoms
211   }

213   private def writeCoordinateSequence(
214       out: T,
215       writer: CoordinateSerializer,
216       coords: CoordinateSequence,
217       writeLength: Boolean): Unit = {
218     if (writeLength) {
219       out.writeInt(coords.size(), optimizePositive = true)
220     }
221     var i = 0
222     while (i < coords.size()) {
223       writer.write(out, coords.getCoordinate(i))
224       i += 1
225     }
226   }
227 
228   private def readCoordinateSequence(in: V, reader: CoordinateSerializer, length: Option[Int]): CoordinateSequence = {
229     val numCoords = length.getOrElse(in.readInt(true))
230     if (numCoords > maxLength) {
231       throw new IllegalArgumentException(
232         s"Attempting to deserialize a geometry of size $numCoords " +
233             s"with '${GeometryLengthThreshold.property}' = $maxLength'")
234     }
235     val coords = Array.ofDim[Coordinate](numCoords)
236     var i = 0
237     while (i < numCoords) {
238       coords(i) = reader.read(in)
239       i += 1
240     }
241     csFactory.create(coords)
242   }

244   private def getWriter(geometry: Geometry): CoordinateSerializer = {
245     // don't trust coord.getDimensions - it always returns 3 in jts
246     // instead, check for NaN for the z dimension
247     // note that we only check the first coordinate - if a geometry is written with different
248     // dimensions in each coordinate, some information may be lost
249     val coord = geometry.getCoordinate
250     if (coord == null) { xySerializer } else {
251       // check for dimensions - use NaN != NaN to verify coordinate dimensions
252       val hasZ = !java.lang.Double.isNaN(coord.getZ)
253       val hasM = !java.lang.Double.isNaN(coord.getM)
254       if (hasZ) {
255         if (hasM) {
256           xyzmSerializer
257         } else {
258           xyzSerializer
259         }
260       } else if (hasM) {
261         xymSerializer
262       } else {
263         xySerializer
264       }
265     }
266   }
