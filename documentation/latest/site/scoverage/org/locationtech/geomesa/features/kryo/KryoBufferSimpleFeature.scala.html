331 
332     private var metadata: Metadata = _
333     private var transformer: Transformer = _
334 
335     override def reset(): Unit = metadata = Metadata(input) // reads count, size, nulls, etc
336 
337     override def setTransforms(schema: SimpleFeatureType, transforms: Array[Transform]): Unit = {

383     // if we are just returning a subset of attributes, we can copy the bytes directly
384     private class BinaryTransformer(indices: Array[Int]) extends Transformer {
385 
386       private val positionsAndLengths = Array.ofDim[(Int, Int)](indices.length)
387 
388       override def transform(original: SimpleFeature): Array[Byte] = {
389         // +1 for version, +2 for count, +1 for size
390         var length = (metadata.size * (indices.length + 1)) + (IntBitSet.size(indices.length) * 4) + 4
391         val id = if (serializer.withoutId) { null } else {
392           val pos = metadata.setIdPosition() + metadata.offset
393           val id = input.readString()
394           length += input.position() - pos
395           id
396         }
397 
398         val nulls = IntBitSet(indices.length)
399 
400         // track the write position for copying attributes in the transformed array
401         var resultCursor = length
402 
403         var i = 0
404         while (i < indices.length) {
405           val index = indices(i) // the index of the attribute in the original feature
406           if (index >= metadata.count || metadata.nulls.contains(index)) { nulls.add(i) } else {
407             // read the offset and the subsequent offset to get the length
408             val pos = metadata.setPosition(index)
409             val len = metadata.setPosition(index + 1) - pos
410             length += len
411             positionsAndLengths(i) = (metadata.offset + pos, len)
412           }
413           i += 1
414         }
415 
416         // note: the input buffer is the raw buffer. we need to ensure that we use the
417         // offset into the raw buffer rather than the raw buffer directly
418         val buf = input.getBuffer
419 
420         val result = Array.ofDim[Byte](length)
421         val output = new Output(result, length)
422         output.writeByte(KryoFeatureSerializer.Version3)
423         output.writeShort(indices.length) // track the number of attributes
424         output.write(metadata.size)
425         i = 0
426         while (i < positionsAndLengths.length) {
427           // note: offset is always 4 here since we're using the full result array
428           if (metadata.size == 2) {
429             output.writeShort(resultCursor - 4)
430           } else {
431             output.writeInt(resultCursor - 4)
432           }
433           if (!nulls.contains(i)) {
434             val (pos, len) = positionsAndLengths(i)
435             System.arraycopy(buf, pos, result, resultCursor, len)
436             resultCursor += len
437           }
438           i += 1
439         }
440         // user data offset
441         if (metadata.size == 2) {
442           output.writeShort(resultCursor)
443         } else {
444           output.writeInt(resultCursor)
445         }
446 
447         // TODO user data?
448 
449         // write out nulls
450         nulls.serialize(output)
451 
452         // we're already positioned to write out the feature id
453         if (id != null) {
454           output.writeString(id)
455         }
456 
457         result
458       }
459     }

463     * Serialiation version 2 delegate
464     *
465     * @param serializer serializer
466     * @param input input
467     */
468   private class KryoBufferV2(serializer: KryoFeatureDeserialization, input: Input) extends KryoBufferDelegate {
469 
470     private val offsets = Array.ofDim[Int](serializer.out.getAttributeCount)
471     private var offset: Int = -1
472     private var startOfOffsets: Int = -1
473     private var missingAttributes: Boolean = false
474     private var userDataOffset: Int = -1
475 
476     private var reserializeTransform: Transformer = _
477     private var binaryTransform: Transformer = _
478 
479     override def reset(): Unit = {
480       // reset our offsets
481       offset = input.position() - 1 // we've already read the version byte
482       startOfOffsets = offset + input.readInt()
483       input.setPosition(startOfOffsets) // set to offsets start
484       var i = 0
485       while (i < offsets.length && input.position < input.limit) {
486         offsets(i) = offset + input.readInt(true)
487         i += 1
488       }
489       if (i < offsets.length) {
490         // attributes have been added to the sft since this feature was serialized
491         missingAttributes = true
492         while ({{ offsets(i) = -1; i += 1 }; i < offsets.length })()
493       } else {
494         missingAttributes = false
495       }
496       userDataOffset = input.position()
497     }
498 
499     override def setTransforms(schema: SimpleFeatureType, transforms: Array[Transform]): Unit = {
500       val indices = transforms.map {
501         case t: PropertyTransform => t.i
502         case t: RenameTransform => t.i
503         case _ => -1
504       }
505 
506       // transforms by evaluating the transform expressions and then serializing the resulting feature
507       // we use this for transform expressions and for data that was written using an old schema
508       reserializeTransform = new ReserializeTransformer(schema, transforms, serializer.options)
509       // if we are just returning a subset of attributes, we can copy the bytes directly
510       // and avoid creating new objects, reserializing, etc
511       binaryTransform =
512           if (indices.contains(-1)) { reserializeTransform } else { new BinaryTransformerV2(input, indices, offsets) }
513     }
514 
515     override def id(): String = {
516       input.setPosition(5)
517       input.readString()
518     }
519 
520     override def getAttribute(index: Int): AnyRef = {
521       val offset = offsets(index)
522       if (offset == -1) { null } else {
523         input.setPosition(offset)
524         serializer.readersV2(index)(input)
525       }
526     }
527 
528     override def getUserData: java.util.Map[AnyRef, AnyRef] = {
529       input.setPosition(userDataOffset)
530       KryoUserDataSerialization.deserialize(input)
531     }
532 
533     override def getDateAsLong(index: Int): Long = {
534       val offset = offsets(index)
535       if (offset == -1) { 0L } else {
536         input.setPosition(offset)
537         KryoFeatureDeserializationV2.LongReader.apply(input)
538       }
539     }
540 
541     override def getInput(index: Int): Option[Input] = {
542       val offset = offsets(index)
543       if (offset == -1) { None } else {
544         input.setPosition(offset)
545         Some(input)
546       }
547     }
548 
549     override def transform(original: SimpleFeature): Array[Byte] = {
550       // if attributes have been added to the sft, we have to reserialize to get the null serialized values
551       val transformer = if (missingAttributes) { reserializeTransform } else { binaryTransform }
552       transformer.transform(original)
553     }

555     /**
556       * Serialization version 2 binary transformer. Copies serialized attribute bytes directly without
557       * deserializing them
558       *
559       * @param input input
560       * @param indices transform indices
561       * @param offsets attribute offsets
562       */
563     private class BinaryTransformerV2(input: Input, indices: Array[Int], offsets: Array[Int]) extends Transformer {
564 
565       private val mutableOffsetsAndLength = Array.ofDim[(Int,Int)](indices.length)
566 
567       override def transform(original: SimpleFeature): Array[Byte] = {
568         // NOTE: the input buffer is the raw buffer. we need to ensure that we use the
569         // offset into the raw buffer rather than the raw buffer directly
570         val buf = input.getBuffer
571         var length = offsets(0) - offset // space for version, offset block and ID
572         var idx = 0
573         while (idx < mutableOffsetsAndLength.length) {
574           val i = indices(idx)
575           val el = (if (i < offsets.length - 1) { offsets(i + 1) } else { startOfOffsets }) - offsets(i)
576           length += el
577           mutableOffsetsAndLength(idx) = (offsets(i), el)
578           idx += 1
579         }
580 
581         val dst = Array.ofDim[Byte](length)
582         // copy the version, offset block and id
583         var dstPos = offsets(0) - offset
584         System.arraycopy(buf, offset, dst, 0, dstPos)
585         mutableOffsetsAndLength.foreach { case (o, l) =>
586           System.arraycopy(buf, o, dst, dstPos, l)
587           dstPos += l
588         }
589         // note that the offset block is incorrect - we couldn't use this in another lazy feature
590         // but the normal serializer doesn't care
591         dst
592       }
593     }
594   }

618       }

620     }
