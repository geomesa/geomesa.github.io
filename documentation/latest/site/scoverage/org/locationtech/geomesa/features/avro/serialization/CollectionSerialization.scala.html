22   /**
23    * Encodes a list of primitives or Dates into a byte buffer. The list items must be all of the same
24    * class.
25    *
26    * @param list list
27    * @return
28    */
29   def encodeList(list: java.util.List[_], binding: String): ByteBuffer = {
30     val size = Option(list).map(_.size)
31     size match {
32       case Some(s) if s == 0 => encodeEmptyCollection
33       case Some(s)           => encodeNonEmptyList(list, s, binding)
34       case None              => encodeNullCollection
35     }
36   }

38   /**
39    * Decodes a byte buffer created with @see encodeList back into a list
40    *
41    * @param bb buffer
42    * @return
43    */
44   def decodeList(bb: ByteBuffer): java.util.List[AnyRef] = {
45     val size = bb.getInt
46     if (size < 0) {
47       null
48     } else if (size == 0) {
49       java.util.Collections.emptyList()
50     } else {
51       val list = new java.util.ArrayList[Object](size)
52       val label = getString(bb)
53       val readMethod = getReadMethod(label, bb)
54       var i = 0
55       while (i < size) {
56         list.add(readMethod())
57         i += 1
58       }
59       list
60     }
61   }

63   /**
64    * Encodes a map of primitives or Dates into a byte buffer. The map keys must be all of the same
65    * class, and the map values must all be of the same class.
66    *
67    * @param map map
68    * @return
69    */
70   def encodeMap(map: java.util.Map[_, _], keyBinding: String, valueBinding: String): ByteBuffer = {

72     size match {
73       case Some(s) if s == 0 => encodeEmptyCollection
74       case Some(s)           => encodeNonEmptyMap(map, s, keyBinding, valueBinding)
75       case None              => encodeNullCollection
76     }
77   }

79   /**
80    * Decodes a byte buffer created with @see encodeMap back into a map.
81    *
82    * @param bb buffer
83    * @return
84    */
85   def decodeMap(bb: ByteBuffer): java.util.Map[AnyRef, AnyRef] = {
86     val size = bb.getInt
87     if (size < 0) {
88       null
89     } else if (size == 0) {
90       java.util.Collections.emptyMap()
91     } else {
92       val map = new java.util.HashMap[Object, Object](size)
93       val keyType = getString(bb)
94       val valueType = getString(bb)
95       val keyReadMethod = getReadMethod(keyType, bb)
96       val valueReadMethod = getReadMethod(valueType, bb)
97       var i = 0
98       while (i < size) {
99         val key = keyReadMethod()
100         val value = valueReadMethod()
101         map.put(key, value)
102         i += 1
103       }
104       map
105     }
106   }

114   /**
115    * Encodes a list that has entries.
116    *
117    * @param list list
118    * @param size size of list
119    * @return
120    */
121   private def encodeNonEmptyList(list: java.util.List[_], size: Int, label: String): ByteBuffer = {
122     // get the appropriate write method for the list type
123     val (bytesPerItem, putMethod): (Int, (ByteBuffer, Any) => Unit) = getWriteMethod(label)
124     // calculate the total size needed to encode the list
125     val totalBytes = getTotalBytes(bytesPerItem, size, list.iterator().asScala, label)
126 
127     val labelBytes = label.getBytes(StandardCharsets.UTF_8)
128     // 4 bytes for list size + 4 bytes for label bytes size + label bytes + item bytes
129     val bb = ByteBuffer.allocate(4 + 4 + labelBytes.size + totalBytes)
130     // first put the size of the list
131     bb.putInt(size)
132     // put the type of the list
133     putString(bb, label)
134     // put each item
135     list.asScala.foreach(v => putMethod(bb, v))
136     // flip (reset) the buffer so that it's ready for reading
137     bb.flip
138     bb
139   }

141   /**
142    * Encodes a map that has entries.
143    *
144    * @param map map
145    * @param size size of map
146    * @return
147    */
148   private def encodeNonEmptyMap(
149       map: java.util.Map[_, _],
150       size: Int,
151       keyLabel: String,
152       valueLabel: String): ByteBuffer = {
153     // get the appropriate write methods and approximate sizes for keys and values
154     val (bytesPerKeyItem, keyPutMethod)     = getWriteMethod(keyLabel)
155     val (bytesPerValueItem, valuePutMethod) = getWriteMethod(valueLabel)
156 
157     // get the exact size in bytes for keys and values
158     val totalKeyBytes   = getTotalBytes(bytesPerKeyItem, size, map.asScala.keysIterator, keyLabel)
159     val totalValueBytes = getTotalBytes(bytesPerValueItem, size, map.asScala.valuesIterator, valueLabel)
160 
161     val keyLabelBytes = keyLabel.getBytes(StandardCharsets.UTF_8)
162     val valueLabelBytes = valueLabel.getBytes(StandardCharsets.UTF_8)
163     // 4 bytes for map size + 8 bytes for label bytes size + label bytes + key bytes + value bytes
164     val totalBytes = 4 + 8 + keyLabelBytes.size + valueLabelBytes.size + totalKeyBytes + totalValueBytes
165     val bb = ByteBuffer.allocate(totalBytes)
166     // first put the size of the map
167     bb.putInt(size)
168     // put the types of the keys and values
169     putString(bb, keyLabel)
170     putString(bb, valueLabel)
171     // put each key value pair
172     map.asScala.foreach { case (k, v) =>
173       keyPutMethod(bb, k)
174       valuePutMethod(bb, v)
175     }
176     // flip (reset) the buffer so that it's ready for reading
177     bb.flip
178     bb
179   }

181   /**
182    * Gets the appropriate byte buffer method for the given object type.
183    *
184    * @param label class type
185    * @return size per item (if known, otherwise -1) + read method
186    */
187   private def getWriteMethod(label: String): (Int, (ByteBuffer, Any) => Unit) =
188     label.toLowerCase(Locale.US) match {
189       case "string"  => (-1, (bb, v) => putString(bb, v.asInstanceOf[String]))
190       case "int" |
191            "integer" => (4, (bb, v) => bb.putInt(v.asInstanceOf[Int]))
192       case "double"  => (8, (bb, v) => bb.putDouble(v.asInstanceOf[Double]))
193       case "long"    => (8, (bb, v) => bb.putLong(v.asInstanceOf[Long]))
194       case "float"   => (4, (bb, v) => bb.putFloat(v.asInstanceOf[Float]))
195       case "date"    => (8, (bb, v) => bb.putLong(v.asInstanceOf[Date].getTime))
196       case "boolean" => (1, (bb, v) => if (v.asInstanceOf[Boolean]) bb.put(1.toByte) else bb.put(0.toByte))
197       case "uuid"    => (16, (bb, v) => putUUID(bb, v.asInstanceOf[UUID]))
198       case "byte[]"  => (-1, (bb, v) => putBytes(bb, v.asInstanceOf[Array[Byte]]))
199       case _         =>
200         val msg = s"Invalid collection type: '$label'. Only primitives and Dates are supported."
201         throw new IllegalArgumentException(msg)
202     }

204   /**
205    * Gets the appropriate byte buffer method for the given object type.
206    *
207    * @param label class type
208    * @param bb buffer
209    * @return
210    */
211   private def getReadMethod(label: String, bb: ByteBuffer): () => Object =
212     label.toLowerCase(Locale.US) match {
213       case "string"  => () => getString(bb)
214       case "int" |
215            "integer" => () => bb.getInt.asInstanceOf[Object]
216       case "double"  => () => bb.getDouble.asInstanceOf[Object]
217       case "long"    => () => bb.getLong.asInstanceOf[Object]
218       case "float"   => () => bb.getFloat.asInstanceOf[Object]
219       case "boolean" => () => java.lang.Boolean.valueOf(bb.get > 0)
220       case "date"    => () => new Date(bb.getLong())
221       case "uuid"    => () => getUUID(bb)
222       case "byte[]"  => () => getBytes(bb)
223       case _         =>
224         val msg = s"Invalid collection type: '$label'. Only primitives and Dates are supported."
225         throw new IllegalArgumentException(msg)
226     }

228   /**
229    * Gets the total bytes needed to encode the given values. For most types, the size is fixed, but
230    * Strings and bytes are encoded with a dynamic length.
231    *
232    * @param bytesPerItem bytes per item
233    * @param size number of items
234    * @param values values
235    * @return
236    */
237   private def getTotalBytes(bytesPerItem: Int, size: Int, values: Iterator[_], label: String): Int =
238     if (bytesPerItem == -1) {
239       // bytes are variable, we need to calculate them based on content
240       // this only happens with strings
241       // add 4 to each to use for length encoding
242       label.toLowerCase match {
243         case "string" => values.map(_.asInstanceOf[String].getBytes(StandardCharsets.UTF_8).length + 4).sum
244         case "byte[]" => values.map(_.asInstanceOf[Array[Byte]].length + 4).sum
245         case _ => throw new IllegalArgumentException("invalid type")
246       }
247     } else {
248       bytesPerItem * size
249     }

251   /**
252    * Reads a string from a byte buffer that has been written using @see putString.
253    *
254    * @param bb buffer
255    * @return
256    */
257   private def getString(bb: ByteBuffer): String = {
258     val size = bb.getInt
259     val buf = new Array[Byte](size)
260     bb.get(buf)
261     new String(buf, StandardCharsets.UTF_8)
262   }

289     val sz = bb.getInt

295   private def putUUID(bb: ByteBuffer, uuid: UUID): ByteBuffer =
296     bb.putLong(uuid.getMostSignificantBits).putLong(uuid.getLeastSignificantBits)
