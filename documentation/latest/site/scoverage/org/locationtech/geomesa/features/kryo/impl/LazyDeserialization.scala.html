247   override def deserialize(id: String, bytes: Array[Byte], offset: Int, length: Int): SimpleFeature = {
248     bytes(offset) match {
249       case KryoFeatureSerializer.Version3 => readFeatureV3(id, bytes, offset, length)
250       case KryoFeatureSerializer.Version2 => readFeatureV2(id, bytes, offset, length)
251       case b => throw new IllegalArgumentException(s"Can't process features serialized with version: $b")
252     }
253   }

262   private def readFeatureV3(id: String, bytes: Array[Byte], offset: Int, length: Int): SimpleFeature = {
263     // skip the version byte, which we've already read
264     val input = new NonMutatingInput()
265     input.setBuffer(bytes, offset + 1, length - 1)
266     val metadata = Metadata(input) // read count, size, nulls, etc
267 
268     // we should now be positioned to read the feature id
269     val finalId = if (withoutId) { id } else { input.readString() }
270 
271     val remaining = input.limit - metadata.offset
272 
273     var reader: LazyAttributeReader = null
274     var userData: LazyUserDataReader = null
275 
276     if (metadata.size == 2) {
277       reader = new LazyShortReaderV3(readers, metadata.nulls, metadata.count, bytes, metadata.offset, remaining)
278       userData = if (withoutUserData) { WithoutUserDataReader } else {
279         new LazyShortUserDataReaderV3(metadata.count, bytes, metadata.offset, remaining)
280       }
281     } else {
282       reader = new LazyIntReaderV3(readers, metadata.nulls, metadata.count, bytes, metadata.offset, remaining)
283       userData = if (withoutUserData) { WithoutUserDataReader } else {
284         new LazyIntUserDataReaderV3(metadata.count, bytes, metadata.offset, remaining)
285       }
286     }
287 
288     createFeature(finalId, reader, userData)
289   }
290 
291   private def readFeatureV2(id: String, bytes: Array[Byte], offset: Int, length: Int): SimpleFeature = {
292     val input = new NonMutatingInput()
293     input.setBuffer(bytes, offset + 1, length - 1) // skip the version byte
294     // read the start of the offsets, then the feature id
295     val offsets = Array.ofDim[Int](readersV2.length)
296     val offsetStarts = offset + input.readInt()
297     val finalId = if (withoutId) { id } else { input.readString() }
298     // now read our offsets
299     input.setPosition(offsetStarts) // set to offsets start
300     var i = 0
301     while (i < offsets.length && input.position < input.limit) {
302       offsets(i) = offset + input.readInt(true)
303       i += 1
304     }
305     if (i < offsets.length) {
306       // attributes have been added to the sft since this feature was serialized
307       while ({{ offsets(i) = -1; i += 1 }; i < offsets.length })()
308     }
309     val userDataOffset = input.position()
310 
311     val reader = new LazyReaderV2(readersV2, offsets, bytes, length)
312     val userData = if (withoutUserData) { WithoutUserDataReader } else {
313       new LazyUserDataReaderV2(bytes, userDataOffset, length)
314     }
315 
316     createFeature(finalId, reader, userData)
317   }
