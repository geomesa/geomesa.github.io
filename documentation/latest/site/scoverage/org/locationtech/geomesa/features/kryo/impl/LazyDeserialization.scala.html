249   override def deserialize(id: String, bytes: Array[Byte], offset: Int, length: Int): SimpleFeature = {
250     bytes(offset) match {
251       case KryoFeatureSerializer.Version3 => readFeatureV3(id, bytes, offset, length)
252       case KryoFeatureSerializer.Version2 => readFeatureV2(id, bytes, offset, length)
253       case b => throw new IllegalArgumentException(s"Can't process features serialized with version: $b")
254     }
255   }

264   private def readFeatureV3(id: String, bytes: Array[Byte], offset: Int, length: Int): SimpleFeature = {
265     // skip the version byte, which we've already read
266     val input = new NonMutatingInput()
267     input.setBuffer(bytes, offset + 1, length - 1)
268     val metadata = Metadata(input) // read count, size, nulls, etc
269 
270     // we should now be positioned to read the feature id
271     val finalId = if (withoutId) { id } else { input.readString() }
272 
273     val remaining = input.limit - metadata.offset
274 
275     var reader: LazyAttributeReader = null
276     var userData: LazyUserDataReader = null
277 
278     if (metadata.size == 2) {
279       reader = new LazyShortReaderV3(readers, metadata.nulls, metadata.count, bytes, metadata.offset, remaining)
280       userData = if (withoutUserData) { WithoutUserDataReader } else {
281         new LazyShortUserDataReaderV3(metadata.count, bytes, metadata.offset, remaining)
282       }
283     } else {
284       reader = new LazyIntReaderV3(readers, metadata.nulls, metadata.count, bytes, metadata.offset, remaining)
285       userData = if (withoutUserData) { WithoutUserDataReader } else {
286         new LazyIntUserDataReaderV3(metadata.count, bytes, metadata.offset, remaining)
287       }
288     }
289 
290     createFeature(finalId, reader, userData)
291   }
292 
293   private def readFeatureV2(id: String, bytes: Array[Byte], offset: Int, length: Int): SimpleFeature = {
294     val input = new NonMutatingInput()
295     input.setBuffer(bytes, offset + 1, length - 1) // skip the version byte
296     // read the start of the offsets, then the feature id
297     val offsets = Array.ofDim[Int](readersV2.length)
298     val offsetStarts = offset + input.readInt()
299     val finalId = if (withoutId) { id } else { input.readString() }
300     // now read our offsets
301     input.setPosition(offsetStarts) // set to offsets start
302     var i = 0
303     while (i < offsets.length && input.position < input.limit) {
304       offsets(i) = offset + input.readInt(true)
305       i += 1
306     }
307     if (i < offsets.length) {
308       // attributes have been added to the sft since this feature was serialized
309       while ({{ offsets(i) = -1; i += 1 }; i < offsets.length })()
310     }
311     val userDataOffset = input.position()
312 
313     val reader = new LazyReaderV2(readersV2, offsets, bytes, length)
314     val userData = if (withoutUserData) { WithoutUserDataReader } else {
315       new LazyUserDataReaderV2(bytes, userDataOffset, length)
316     }
317 
318     createFeature(finalId, reader, userData)
319   }
