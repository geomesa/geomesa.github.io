22   /**
23    * Create a filter
24    *
25    * @param in input
26    * @param root root node of the document being evaluated (for absolute path references)
27    * @param op the filter op
28    * @return
29    */
30   def apply(in: Input, root: ValuePointer, op: PathFilter.FilterOp): KryoJsonPathFilter =
31     new FilterBuilder(in, root)(op)
32 
33   /**
34    * An expression in a filter - either a path, or a literal
35    */
36   private type Expression = ValuePointer => Any

38   /**
39    * Builder
40    *
41    * @param in input
42    * @param root root document path (for absolute path expressions)
43    */
44   private class FilterBuilder(in: Input, root: ValuePointer) {
45 
46     def apply(op: PathFilter.FilterOp): KryoJsonPathFilter = op match {
47       case EqualsOp(left, right)              => BinaryFilter.Equals(expression(left), expression(right))
48       case NotEqualsOp(left, right)           => BinaryFilter.NotEquals(expression(left), expression(right))
49       case LessThanOp(left, right)            => ComparisonFilter.LessThan(expression(left), expression(right))
50       case LessThanOrEqualsOp(left, right)    => ComparisonFilter.LessThanOrEquals(expression(left), expression(right))
51       case GreaterThanOp(left, right)         => ComparisonFilter.GreaterThan(expression(left), expression(right))
52       case GreaterThanOrEqualsOp(left, right) => ComparisonFilter.GreaterThanOrEquals(expression(left), expression(right))
53       case RegexOp(path, regex)               => RegexFilter(expression(path), regex)
54       case InOp(left, right)                  => ScalarArrayFilter.In(expression(left), expression(right))
55       case NotInOp(left, right)               => ScalarArrayFilter.NotIn(expression(left), expression(right))
56       case IsSubsetOp(left, right)            => ArrayFilter.Subset(expression(left), expression(right))
57       case AnyOfOp(left, right)               => ArrayFilter.AnyOf(expression(left), expression(right))
58       case NoneOfOp(left, right)              => ArrayFilter.NoneOf(expression(left), expression(right))
59       case EqualSizeOp(left, right)           => SizeFilter(expression(left), expression(right))
60       case IsEmptyOp(value, empty)            => IsEmptyFilter(expression(value), expression(empty))
61       case ExistsOp(value)                    => ExistsFilter(expression(value))
62       case AndFilterOp(filters)               => AndFilter(filters.map(this.apply))
63       case OrFilterOp(filters)                => OrFilter(filters.map(this.apply))
64       case NotFilterOp(filter)                => NotFilter(this.apply(filter))
65     }
66 
67     private def expression(exp: FilterExpression): Expression = exp match {
68       case p: PathExpression =>
69         if (p.path.elements.isEmpty) {
70           Constant(null)
71         } else if (p.absolute) {
72           Constant(new KryoJsonPath(in, root).deserialize(p.path))
73         } else {
74           Path(in, p.path)
75         }
76       case lit: LiteralExpression =>
77         Constant(lit.value)
78     }
79   }

100   /**
101    * Filter op between two arbitrary expressions
102    *
103    * @param left left expression
104    * @param right right expression
105    * @param op comparison op
106    */
107   private case class BinaryFilter(left: Expression, right: Expression, op: BinaryFilter.BinaryPredicate)
108       extends KryoJsonPathFilter {
109     override def apply(pointer: ValuePointer): Boolean = op(left(pointer), right(pointer))
110   }
111 
112   private object BinaryFilter {
113     trait BinaryPredicate extends ((Any, Any) => Boolean) {
114       // constructor
115       def apply(left: Expression, right: Expression): BinaryFilter = BinaryFilter(left, right, this)
116     }
117 
118     object Equals extends BinaryPredicate {
119       override def apply(left: Any, right: Any): Boolean = left == right
120     }
121     object NotEquals extends BinaryPredicate {
122       override def apply(left: Any, right: Any): Boolean = left != right
123     }
124   }
125 
126   /**
127    * Filter op between two 'comparable' expressions - expressions must both evaluate to either strings or numbers
128    *
129    * @param left left expression
130    * @param right right expression
131    * @param op comparison op
132    */
133   private case class ComparisonFilter(left: Expression, right: Expression, op: ComparisonFilter.ComparisonPredicate)
134     extends KryoJsonPathFilter {
135     override def apply(pointer: ValuePointer): Boolean = {
136       left(pointer) match {
137         case leftValue: Number =>
138           right(pointer) match {
139             case rightValue: Number => op(leftValue.doubleValue(), rightValue.doubleValue())
140             case _ => false
141           }
142         case leftValue: String =>
143           right(pointer) match {
144             case rightValue: String => op(leftValue, rightValue)
145             case _ => false
146           }
147         case _ => false
148       }
149     }
150   }
151 
152   private object ComparisonFilter {
153 
154     trait ComparisonPredicate {
155       // constructor
156       def apply(left: Expression, right: Expression): ComparisonFilter = ComparisonFilter(left, right, this)
157       def apply(left: Double, right: Double): Boolean
158       def apply(left: String, right: String): Boolean
159     }
160 
161     object LessThan extends ComparisonPredicate {
162       override def apply(left: Double, right: Double): Boolean = left < right
163       override def apply(left: String, right: String): Boolean = left < right
164     }
165     object LessThanOrEquals extends ComparisonPredicate {
166       override def apply(left: Double, right: Double): Boolean = left <= right
167       override def apply(left: String, right: String): Boolean = left <= right
168     }
169     object GreaterThan extends ComparisonPredicate {
170       override def apply(left: Double, right: Double): Boolean = left > right
171       override def apply(left: String, right: String): Boolean = left > right
172     }
173     object GreaterThanOrEquals extends ComparisonPredicate {
174       override def apply(left: Double, right: Double): Boolean = left >= right
175       override def apply(left: String, right: String): Boolean = left >= right
176     }
177   }
178 
179   /**
180    * Regex filter op
181    *
182    * `=~` - left matches regular expression `[?(@.name =~ /foo.*?/i)]`
183    *
184    * @param left path expression, must evaluate to a string
185    * @param right regex to match
186    */
187   private case class RegexFilter(left: Expression, right: Pattern) extends KryoJsonPathFilter {
188     override def apply(pointer: ValuePointer): Boolean = {
189       left(pointer) match {
190         case s: String => right.matcher(s).matches()
191         case _ => false
192       }
193     }
194   }

196   private trait ArrayOps {
197     import scala.collection.JavaConverters._
198 
199     def toSeq(expression: Any): Option[Seq[Any]] = expression match {
200       case s: Seq[_] => Some(s)
201       case s: java.util.List[_] => Some(s.asScala.toSeq)
202       case _ => None
203     }
204 
205     def toString(expression: Any): Option[Seq[Any]] = expression match {
206       case s: String => Some(s)
207       case _ => None
208     }
209   }

211   /**
212    * Filter op between a scalar (number, string) and an array
213    *
214    * @param left scalar expression
215    * @param right array expression
216    * @param op comparison op
217    */
218   private case class ScalarArrayFilter(left: Expression, right: Expression, op: ScalarArrayFilter.ScalarArrayPredicate)
219     extends KryoJsonPathFilter with ArrayOps {
220     override def apply(pointer: ValuePointer): Boolean = toSeq(right(pointer)).exists(op(left(pointer), _))
221   }
222 
223   private object ScalarArrayFilter {
224 
225     trait ScalarArrayPredicate extends ((Any, Seq[Any]) => Boolean) {
226       // constructor
227       def apply(left: Expression, right: Expression): ScalarArrayFilter = ScalarArrayFilter(left, right, this)
228     }
229 
230     // in - left exists in right [?(@.size in ['S', 'M'])]
231     object In extends ScalarArrayPredicate {
232       override def apply(left: Any, right: Seq[Any]): Boolean = right.contains(left)
233     }
234 
235     // nin - left does not exists in right
236     object NotIn extends ScalarArrayPredicate {
237       override def apply(left: Any, right: Seq[Any]): Boolean = !right.contains(left)
238     }
239   }

241   /**
242    * Filter op between two arrays
243    *
244    * @param left left array
245    * @param right right array
246    * @param op comparison op
247    */
248   private case class ArrayFilter(left: Expression, right: Expression, op: ArrayFilter.ArrayPredicate)
249       extends KryoJsonPathFilter with ArrayOps {
250     override def apply(pointer: ValuePointer): Boolean = {
251       val seqs = for { s1 <- toSeq(left(pointer)); s2 <- toSeq(right(pointer)) } yield { op(s1, s2) }
252       seqs.getOrElse(false)
253     }
254   }
255 
256   private object ArrayFilter {
257 
258     trait ArrayPredicate extends ((Seq[Any], Seq[Any]) => Boolean) {
259       // constructor
260       def apply(left: Expression, right: Expression): ArrayFilter = ArrayFilter(left, right, this)
261     }
262 
263     // subsetof - left is a subset of right [?(@.sizes subsetof ['S', 'M', 'L'])]
264     object Subset extends ArrayPredicate {
265       override def apply(left: Seq[Any], right: Seq[Any]): Boolean = left.forall(right.contains)
266     }
267 
268     // anyof - left has an intersection with right [?(@.sizes anyof ['M', 'L'])]
269     object AnyOf extends ArrayPredicate {
270       override def apply(left: Seq[Any], right: Seq[Any]): Boolean = left.exists(right.contains)
271     }
272 
273     //noneof - left has no intersection with right [?(@.sizes noneof ['M', 'L'])]
274     object NoneOf extends ArrayPredicate {
275       override def apply(left: Seq[Any], right: Seq[Any]): Boolean = !left.exists(right.contains)
276     }
277   }

279   /**
280    * Size filter op
281    *
282    * `size` - size of left (array or string) should match right
283    *
284    * @param left left expression, array or string
285    * @param right right expression, array or string
286    */
287   private case class SizeFilter(left: Expression, right: Expression) extends KryoJsonPathFilter with ArrayOps {
288     override def apply(pointer: ValuePointer): Boolean = {
289       val leftValue = left(pointer)
290       val rightValue = right(pointer)
291       val arraySize = for { s1 <- toSeq(leftValue); s2 <- toSeq(rightValue) } yield { s1.size == s2.size }
292       lazy val stringSize = for { s1 <- toString(leftValue); s2 <- toString(rightValue) } yield { s1.size == s2.size }
293       arraySize.orElse(stringSize).getOrElse(false)
294     }
295   }
296 
297   /**
298    * Empty check filter op
299    *
300    * `empty` - left (array or string) should be empty
301    *
302    * @param value array or string
303    * @param empty empty or not empty
304    */
305   private case class IsEmptyFilter(value: Expression, empty: Expression) extends KryoJsonPathFilter {
306     override def apply(pointer: ValuePointer): Boolean = {
307       empty(pointer) match {
308         case b: Boolean =>
309           value(pointer) match {
310             case s: Seq[_] => s.isEmpty == b
311             case s: java.util.List[_] => s.isEmpty == b
312             case s: String => s.isEmpty == b
313             case null => b
314             case _ => false
315           }
316         case _ => false
317       }
318     }
319   }
320 
321   /**
322    * Filter op for whether expression exists (i.e. is not null)
323    *
324    * @param value any expression
325    */
326   private case class ExistsFilter(value: Expression) extends KryoJsonPathFilter {
327     override def apply(pointer: ValuePointer): Boolean = value(pointer) != null
328   }
329 
330   private case class AndFilter(filters: Seq[KryoJsonPathFilter]) extends KryoJsonPathFilter {
331     override def apply(pointer: ValuePointer): Boolean = filters.forall(_.apply(pointer))
332   }
333   private case class OrFilter(filters: Seq[KryoJsonPathFilter]) extends KryoJsonPathFilter {
334     override def apply(pointer: ValuePointer): Boolean = filters.exists(_.apply(pointer))
335   }
336   private case class NotFilter(filter: KryoJsonPathFilter) extends KryoJsonPathFilter {
337     override def apply(pointer: ValuePointer): Boolean = !filter.apply(pointer)
338   }
