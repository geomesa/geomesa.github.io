46   override def serialize(sf: SimpleFeature, out: OutputStream): Unit = {
47     // note: don't write directly to the output stream, we need to be able to reposition in the output buffer,
48     // and if we use the output stream it might be flushed and reset during our write
49     val output = KryoFeatureSerialization.getOutput(null)
50     writeFeature(sf, output)
51     out.write(output.getBuffer, 0, output.position())
52   }
53 
54   private def writeFeature(sf: SimpleFeature, output: Output): Unit = {
55     output.writeByte(KryoFeatureSerializer.Version3)
56     val offset = Metadata.write(output, count, 2)
57     if (withId) {
58       output.writeString(sf.getID) // TODO optimize for uuids?
59     }
60     // write attributes and keep track of offset into the byte array
61     val offsets = Array.ofDim[Int](count + 1)
62     val nulls = IntBitSet(count)
63     var i = 0
64     while (i < count) {
65       offsets(i) = output.position() - offset
66       val attribute = sf.getAttribute(i)
67       if (attribute == null) {
68         nulls.add(i)
69       } else {
70         writers(i).apply(output, attribute)
71       }
72       i += 1
73     }
74     offsets(i) = output.position() - offset // user data position
75     if (withUserData) {
76       KryoUserDataSerialization.serialize(output, sf.getUserData, withoutFidHints)
77     }
78     val end = output.position()
79     if (end - offset > KryoFeatureSerialization.MaxUnsignedShort) {
80       // we need to shift the bytes rightwards to add space for writing ints instead of shorts for the offsets
81       val shift = 2 * (count + 1)
82       if (output.getBuffer.length < end + shift) {
83         val expanded = Array.ofDim[Byte](end + shift)
84         System.arraycopy(output.getBuffer, 0, expanded, 0, offset)
85         System.arraycopy(output.getBuffer, offset, expanded, offset + shift, end - offset)
86         output.setBuffer(expanded)
87       } else {
88         val buffer = output.getBuffer
89         // end is the position of the next byte to write, so we want to copy from the previous byte
90         var i = end - 1
91         while (i > offset) {
92           buffer(i + shift) = buffer(i)
93           i -= 1
94         }
95       }
96       // go back and write the offsets and nulls
97       output.setPosition(offset - 1)
98       output.write(4) // 4 bytes per offset
99       offsets.foreach(o => output.writeInt(o + shift))
100       nulls.serialize(output)
101       // reset the position back to the end of the buffer so the bytes aren't lost
102       output.setPosition(end + shift)
103     } else {
104       // go back and write the offsets and nulls
105       output.setPosition(offset)
106       offsets.foreach(output.writeShort)
107       nulls.serialize(output)
108       // reset the position back to the end of the buffer so the bytes aren't lost
109       output.setPosition(end)
110     }
111   }
112 }

150   private [geomesa] def writer(bindings: Seq[ObjectType], ad: AttributeDescriptor): KryoAttributeWriter = {
151     bindings.head match {
152       case ObjectType.STRING   => if (bindings.last == ObjectType.JSON) { KryoJsonWriter } else { KryoStringWriter }
153       case ObjectType.INT      => KryoIntWriter
154       case ObjectType.LONG     => KryoLongWriter
155       case ObjectType.FLOAT    => KryoFloatWriter
156       case ObjectType.DOUBLE   => KryoDoubleWriter
157       case ObjectType.DATE     => KryoDateWriter
158       case ObjectType.GEOMETRY =>
159         ad.getPrecision match {
160           case GeometryPrecision.FullPrecision => KryoGeometryWkbWriter
161           case precision: GeometryPrecision.TwkbPrecision => KryoGeometryTwkbWriter(precision)
162         }
163       case ObjectType.UUID     => KryoUuidWriter
164       case ObjectType.BYTES    => KryoBytesWriter
165       case ObjectType.BOOLEAN  => KryoBooleanWriter
166       case ObjectType.LIST     => KryoListWriter(writer(bindings.drop(1), ad))
167       case ObjectType.MAP      => KryoMapWriter(writer(bindings.slice(1, 2), ad), writer(bindings.drop(2), ad))
168 
169       case b => throw new UnsupportedOperationException(s"Unexpected attribute type binding: $b")
170     }
171   }

236   case class KryoListWriter(elements: KryoAttributeWriter) extends KryoAttributeWriter {
237     override def apply(output: Output, value: AnyRef): Unit = {
238       val list = value.asInstanceOf[java.util.List[AnyRef]]
239       val pos = output.position()
240       var size = list.size()
241       output.writeInt(size, true)
242       val start = output.position()
243       val iter = list.iterator()
244       while (iter.hasNext) {
245         val next = iter.next()
246         if (next == null) {
247           size -= 1
248         } else {
249           elements(output, next)
250         }
251       }
252       if (size != list.size()) {
253         logger.warn(s"Dropping ${list.size() - size} null elements from serialized list")
254         resize(output, size, pos, start)
255       }
256     }
257   }
258 
259   case class KryoMapWriter(keys: KryoAttributeWriter, values: KryoAttributeWriter) extends KryoAttributeWriter {
260     override def apply(output: Output, value: AnyRef): Unit = {
261       val map = value.asInstanceOf[java.util.Map[AnyRef, AnyRef]]
262       val pos = output.position()
263       var size = map.size()
264       output.writeInt(size, true)
265       val start = output.position()
266       val iter = map.entrySet.iterator()
267       while (iter.hasNext) {
268         val entry = iter.next()
269         if (entry.getKey == null || entry.getValue == null) {
270           size -= 1
271         } else {
272           keys(output, entry.getKey)
273           values(output, entry.getValue)
274         }
275       }
276       if (size != map.size()) {
277         logger.warn(s"Dropping ${map.size() - size} entries with null keys or values from serialized map")
278         resize(output, size, pos, start)
279       }
280     }
281   }
282 
283   private def resize(output: Output, size: Int, position: Int, start: Int): Unit = {
284     val end = output.position()
285     output.setPosition(position)
286     output.writeInt(size, true)
287     val shift = start - output.position()
288     if (shift != 0) {
289       // the var-int encoding resulting in fewer bytes, we have to shift the result left
290       val buf = output.getBuffer
291       var i = output.position()
292       while (i + shift < end) {
293         buf(i) = buf(i + shift)
294         i -= 1
295       }
296       output.setPosition(end - shift)
297     }
298   }
