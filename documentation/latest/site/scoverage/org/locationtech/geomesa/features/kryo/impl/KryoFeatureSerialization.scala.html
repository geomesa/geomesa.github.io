47   override def serialize(sf: SimpleFeature, out: OutputStream): Unit = {
48     // note: don't write directly to the output stream, we need to be able to reposition in the output buffer,
49     // and if we use the output stream it might be flushed and reset during our write
50     val output = KryoFeatureSerialization.getOutput(null)
51     writeFeature(sf, output)
52     out.write(output.getBuffer, 0, output.position())
53   }
54 
55   private def writeFeature(sf: SimpleFeature, output: Output): Unit = {
56     output.writeByte(KryoFeatureSerializer.Version3)
57     val offset = Metadata.write(output, count, 2)
58     if (withId) {
59       output.writeString(sf.getID) // TODO optimize for uuids?
60     }
61     // write attributes and keep track of offset into the byte array
62     val offsets = Array.ofDim[Int](count + 1)
63     val nulls = IntBitSet(count)
64     var i = 0
65     while (i < count) {
66       offsets(i) = output.position() - offset
67       val attribute = sf.getAttribute(i)
68       if (attribute == null) {
69         nulls.add(i)
70       } else {
71         writers(i).apply(output, attribute)
72       }
73       i += 1
74     }
75     offsets(i) = output.position() - offset // user data position
76     if (withUserData) {
77       KryoUserDataSerialization.serialize(output, sf.getUserData, withoutFidHints)
78     }
79     val end = output.position()
80     if (end - offset > KryoFeatureSerialization.MaxUnsignedShort) {
81       // we need to shift the bytes rightwards to add space for writing ints instead of shorts for the offsets
82       val shift = 2 * (count + 1)
83       if (output.getBuffer.length < end + shift) {
84         val expanded = Array.ofDim[Byte](end + shift)
85         System.arraycopy(output.getBuffer, 0, expanded, 0, offset)
86         System.arraycopy(output.getBuffer, offset, expanded, offset + shift, end - offset)
87         output.setBuffer(expanded)
88       } else {
89         val buffer = output.getBuffer
90         // end is the position of the next byte to write, so we want to copy from the previous byte
91         var i = end - 1
92         while (i > offset) {
93           buffer(i + shift) = buffer(i)
94           i -= 1
95         }
96       }
97       // go back and write the offsets and nulls
98       output.setPosition(offset - 1)
99       output.write(4) // 4 bytes per offset
100       offsets.foreach(o => output.writeInt(o + shift))
101       nulls.serialize(output)
102       // reset the position back to the end of the buffer so the bytes aren't lost
103       output.setPosition(end + shift)
104     } else {
105       // go back and write the offsets and nulls
106       output.setPosition(offset)
107       offsets.foreach(output.writeShort)
108       nulls.serialize(output)
109       // reset the position back to the end of the buffer so the bytes aren't lost
110       output.setPosition(end)
111     }
112   }
113 }

148   private [geomesa] def writer(descriptor: AttributeDescriptor): KryoAttributeWriter =
149     writer(ObjectType.selectType(descriptor), descriptor)

151   private [geomesa] def writer(bindings: Seq[ObjectType], ad: AttributeDescriptor): KryoAttributeWriter = {
152     bindings.head match {
153       case ObjectType.STRING   => if (bindings.last == ObjectType.JSON) { KryoJsonWriter } else { KryoStringWriter }
154       case ObjectType.INT      => KryoIntWriter
155       case ObjectType.LONG     => KryoLongWriter
156       case ObjectType.FLOAT    => KryoFloatWriter
157       case ObjectType.DOUBLE   => KryoDoubleWriter
158       case ObjectType.DATE     => KryoDateWriter
159       case ObjectType.GEOMETRY =>
160         ad.getPrecision match {
161           case GeometryPrecision.FullPrecision => KryoGeometryWkbWriter
162           case precision: GeometryPrecision.TwkbPrecision => KryoGeometryTwkbWriter(precision)
163         }
164       case ObjectType.UUID     => KryoUuidWriter
165       case ObjectType.BYTES    => KryoBytesWriter
166       case ObjectType.BOOLEAN  => KryoBooleanWriter
167       case ObjectType.LIST     => KryoListWriter(writer(bindings.drop(1), ad))
168       case ObjectType.MAP      => KryoMapWriter(writer(bindings.slice(1, 2), ad), writer(bindings.drop(2), ad))
169 
170       case b => throw new UnsupportedOperationException(s"Unexpected attribute type binding: $b")
171     }
172   }

237   case class KryoListWriter(elements: KryoAttributeWriter) extends KryoAttributeWriter {
238     override def apply(output: Output, value: AnyRef): Unit = {
239       val list = value.asInstanceOf[java.util.List[AnyRef]]
240       val pos = output.position()
241       var size = list.size()
242       output.writeInt(size, true)
243       val start = output.position()
244       val iter = list.iterator()
245       while (iter.hasNext) {
246         val next = iter.next()
247         if (next == null) {
248           size -= 1
249         } else {
250           elements(output, next)
251         }
252       }
253       if (size != list.size()) {
254         logger.warn(s"Dropping ${list.size() - size} null elements from serialized list")
255         resize(output, size, pos, start)
256       }
257     }
258   }
259 
260   case class KryoMapWriter(keys: KryoAttributeWriter, values: KryoAttributeWriter) extends KryoAttributeWriter {
261     override def apply(output: Output, value: AnyRef): Unit = {
262       val map = value.asInstanceOf[java.util.Map[AnyRef, AnyRef]]
263       val pos = output.position()
264       var size = map.size()
265       output.writeInt(size, true)
266       val start = output.position()
267       val iter = map.entrySet.iterator()
268       while (iter.hasNext) {
269         val entry = iter.next()
270         if (entry.getKey == null || entry.getValue == null) {
271           size -= 1
272         } else {
273           keys(output, entry.getKey)
274           values(output, entry.getValue)
275         }
276       }
277       if (size != map.size()) {
278         logger.warn(s"Dropping ${map.size() - size} entries with null keys or values from serialized map")
279         resize(output, size, pos, start)
280       }
281     }
282   }
283 
284   private def resize(output: Output, size: Int, position: Int, start: Int): Unit = {
285     val end = output.position()
286     output.setPosition(position)
287     output.writeInt(size, true)
288     val shift = start - output.position()
289     if (shift != 0) {
290       // the var-int encoding resulting in fewer bytes, we have to shift the result left
291       val buf = output.getBuffer
292       var i = output.position()
293       while (i + shift < end) {
294         buf(i) = buf(i + shift)
295         i -= 1
296       }
297       output.setPosition(end - shift)
298     }
299   }
