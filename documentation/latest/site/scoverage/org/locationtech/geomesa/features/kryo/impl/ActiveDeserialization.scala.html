74   private def readFeature(id: String, input: Input): SimpleFeature = {
75     input.readByte() match {
76       case KryoFeatureSerializer.Version3 => readFeatureV3(id, input)
77       case KryoFeatureSerializer.Version2 => readFeatureV2(id, input)
78       case b => throw new IllegalArgumentException(s"Can't process features serialized with version: $b")
79     }
80   }
81 
82   private def readFeatureV3(id: String, input: Input): SimpleFeature = {
83     val metadata = Metadata(input) // read count, size, nulls, etc
84 
85     // we should now be positioned to read the feature id
86     val finalId = if (withoutId) { id } else { input.readString() }
87 
88     // read our attributes
89     val attributes = Array.ofDim[AnyRef](out.getAttributeCount) // note: may not match the serialized count
90     var i = 0
91     while (i < metadata.count) {
92       if (!metadata.nulls.contains(i)) {
93         attributes(i) = readers(i).apply(input)
94       }
95       i += 1
96     }
97 
98     val userData = if (withoutUserData) { new java.util.HashMap[AnyRef, AnyRef](1) } else {
99       KryoUserDataSerialization.deserialize(input)
100     }
101 
102     createFeature(finalId, attributes, userData)
103   }
104 
105   private def readFeatureV2(id: String, input: Input): SimpleFeature = {
106     val offset = input.position() - 1 // we've already read our version byte
107     // read the start of the offsets - we'll stop reading when we hit this
108     val limit = offset + input.readInt()
109     val finalId = if (withoutId) { id } else { input.readString() }
110     val attributes = Array.ofDim[AnyRef](readersV2.length)
111     var i = 0
112     while (i < readersV2.length && input.position < limit) {
113       attributes(i) = readersV2(i)(input)
114       i += 1
115     }
116 
117     val userData = if (withoutUserData) { new java.util.HashMap[AnyRef, AnyRef] } else {
118       // skip offset data
119       try {
120         i = 0
121         while (i < readersV2.length) {
122           input.readInt(true)
123           i += 1
124         }
125         KryoUserDataSerialization.deserialize(input)
126       } catch {
127         case NonFatal(e) =>
128           logger.error("Error reading serialized kryo user data:", e)
129           new java.util.HashMap[AnyRef, AnyRef]()
130       }
131     }
132 
133     createFeature(finalId, attributes, userData)
134   }
