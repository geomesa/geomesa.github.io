17 /**
18  * Kryo serialization implementation for simple features. This class shouldn't be used directly -
19  * see @KryoFeatureSerializer
20  *
21  * @param sft the type of simple feature
22  * @param opts the encoding options (optional)
23  */
24 class SimpleFeatureSerializer(sft: SimpleFeatureType, opts: Set[SerializationOption] = Set.empty)
25     extends Serializer[SimpleFeature] {
26 
27   private val serializer = KryoFeatureSerializer(sft, opts)
28 
29   override def write(kryo: Kryo, output: Output, sf: SimpleFeature): Unit = {
30     val bytes = serializer.serialize(sf)
31     output.writeInt(bytes.length, true)
32     output.write(bytes)
33   }
34 
35   override def read(kryo: Kryo, input: Input, typ: Class[SimpleFeature]): SimpleFeature = {
36     val bytes = Array.ofDim[Byte](input.readInt(true))
37     input.read(bytes)
38     serializer.deserialize(bytes)
39   }
40 }
