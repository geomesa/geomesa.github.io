21   def serialize(out: Encoder, map: java.util.Map[_ <: AnyRef, _ <: AnyRef]): Unit = {
22     def write(value: Any): Option[() => Unit] = {
23       value match {
24         case null                 => Some(() => { out.writeIndex(0); out.writeNull() })
25         case v: String            => Some(() => { out.writeIndex(1); out.writeString(v) })
26         case v: java.lang.Integer => Some(() => { out.writeIndex(2); out.writeInt(v) })
27         case v: java.lang.Long    => Some(() => { out.writeIndex(3); out.writeLong(v) })
28         case v: java.lang.Float   => Some(() => { out.writeIndex(4); out.writeFloat(v) })
29         case v: java.lang.Double  => Some(() => { out.writeIndex(5); out.writeDouble(v) })
30         case v: java.lang.Boolean => Some(() => { out.writeIndex(6); out.writeBoolean(v) })
31         case v: Array[Byte]       => Some(() => { out.writeIndex(7); out.writeBytes(v) })
32         case v: Hints.Key if v == Hints.USE_PROVIDED_FID => logger.debug("Dropping USE_PROVIDED_FID hint"); None
33         case _ =>
34           throw new IllegalArgumentException(s"Serialization not implemented for '$value' of type ${value.getClass}")
35       }
36     }
37 
38     val writes = map.asScala.flatMap { case (key, value) =>
39       for (k <- write(key); v <- write(value)) yield { () => { out.startItem(); k(); v() }}
40     }
41 
42     out.writeArrayStart()
43     out.setItemCount(writes.size)
44     writes.foreach(_.apply())
45     out.writeArrayEnd()
46   }

58   private def deserializeWithSize(in: Decoder, size: Int, map: java.util.Map[AnyRef, AnyRef]): Unit = {
59     var bb: ByteBuffer = null
60 
61     def read(): AnyRef = {
62       in.readIndex() match {
63         case 0 => in.readNull(); null
64         case 1 => in.readString()
65         case 2 => Int.box(in.readInt())
66         case 3 => Long.box(in.readLong())
67         case 4 => Float.box(in.readFloat())
68         case 5 => Double.box(in.readDouble())
69         case 6 => Boolean.box(in.readBoolean())
70         case 7 => bb = in.readBytes(bb); val array = Array.ofDim[Byte](bb.remaining()); bb.get(array); array
71         case i => throw new IllegalArgumentException(s"Unexpected union type: $i")
72       }
73     }
74 
75     var remaining = size
76     while (remaining > 0) {
77       val key = read()
78       val value = read()
79       map.put(key, value)
80       remaining -= 1
81       if (remaining == 0) {
82         remaining = in.arrayNext().toInt
83       }
84     }
85   }
