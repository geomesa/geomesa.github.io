22 class LeafletMapExporter(out: OutputStream) extends FeatureExporter with LazyLogging {
23 
24   private val coordMap = scala.collection.mutable.Map.empty[Coordinate, Int].withDefaultValue(0)
25 
26   private var writer: OutputStreamWriter = _
27   private var jsonWriter: JsonWriter = _
28   private var jsonSerializer: GeoJsonSerializer = _
29 
30   private var featureInfo = ""
31 
32   override def start(sft: SimpleFeatureType): Unit = {
33     featureInfo = LeafletMapExporter.getFeatureInfo(sft)
34     writer = new OutputStreamWriter(out, StandardCharsets.UTF_8)
35     jsonWriter = GeoJsonSerializer.writer(writer)
36     writer.write(LeafletMapExporter.IndexHead)
37     writer.write("var points = ")
38     writer.flush()
39     jsonSerializer = new GeoJsonSerializer(sft)
40     jsonSerializer.startFeatureCollection(jsonWriter)
41   }
42 
43   override def export(features: Iterator[SimpleFeature]): Option[Long] = {
44     var count = 0L
45     while (features.hasNext) {
46       val feature = features.next
47       jsonSerializer.write(jsonWriter, feature)
48       val geom = feature.getDefaultGeometry.asInstanceOf[Geometry]
49       if (geom != null) {
50         geom.getCoordinates.foreach(c => coordMap(c) += 1)
51       }
52       count += 1L
53     }
54     jsonWriter.flush()
55     Some(count)
56   }
57 
58   override def close(): Unit  = {
59     // finish writing GeoJson
60     if (writer != null) {
61       try {
62         jsonSerializer.endFeatureCollection(jsonWriter)
63         jsonWriter.flush()
64 
65         writer.write(";\n\n")
66         writer.write(featureInfo)
67         // Write Heatmap Data
68         writer.write("var heat = L.heatLayer([\n")
69         if (coordMap.isEmpty) {
70           logger.warn("No features were exported - the map will not render correctly")
71         } else {
72           val max = coordMap.maxBy(_._2)._2
73           val iter = coordMap.iterator
74           iter.take(1).foreach { case (coord, weight) =>
75             writer.write(s"        [${coord.y}, ${coord.x}, ${weight / max}]")
76           }
77           iter.foreach { case (coord, weight) =>
78             writer.write(s",\n        [${coord.y}, ${coord.x}, ${weight / max}]")
79           }
80         }
81         writer.write("\n    ], { radius: 25 });\n\n")
82         writer.write(LeafletMapExporter.IndexTail)
83       } finally {
84         jsonWriter.close() // also closes writer and output stream
85       }
86     }
87   }
88 }
