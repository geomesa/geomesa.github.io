38   /**
39     * Deserialize the result of a json-path.
40     *
41     * If the path selects leaf elements, they will be returned as primitives. If the path
42     * selects objects, they will be returned as strings of json. If more than one item is
43     * selected, they will be returned in a java List. If nothing is selected, it will return null.
44     *
45     * @param path json path to evaluate
46     * @return result of the path, if any
47     */
48   def deserialize(path: JsonPath): Any = {
49     require(path.nonEmpty, "Path must not be empty")
50 
51     if (root == null || root.typ == NullByte) {
52       return null
53     }
54 
55     val matches = path.elements.foldLeft(Seq(root)) {
56       case (nodes, _) if nodes.isEmpty       => nodes
57       case (nodes, PathAttribute(name, _))   => matchPathAttribute(nodes, Some(name))
58       case (nodes, PathAttributeWildCard)    => matchPathAttribute(nodes, None)
59       case (nodes, PathIndices(indices))     => matchPathIndex(nodes, indices)
60       case (nodes, PathIndexRange(from, to)) => matchPathIndexRange(nodes, from, to)
61       case (nodes, PathIndexWildCard)        => matchAllPathIndices(nodes)
62       case (nodes, PathDeepScan)             => matchDeep(nodes)
63       case (nodes, PathFilter(op))           => matchFilter(nodes, op)
64     }
65 
66     if (matches.isEmpty) {
67       // special handling to return 0 for length functions without matches
68       path.function match {
69         case Some(PathFunction.LengthFunction) => 0
70         case _ => null
71       }
72     } else {
73       val values = matches.map(m => KryoJsonPath.readPathValue(in, m.typ, m.pos))
74       val result = if (values.lengthCompare(1) == 0) { values.head } else { values.asJava }
75       path.function match {
76         case None => result
77         case Some(f) => f.apply(result)
78       }
79     }
80   }

84   /**
85     * Given a set of input nodes defined by their types and offsets, return a list of output nodes which
86     * match the given attribute name
87     *
88     * @param nodes input node types and offsets
89     * @param name attribute name to match - if None will match all names
90     * @return matching node types and offsets
91     */
92   private def matchPathAttribute(nodes: Seq[ValuePointer], name: Option[String]): Seq[ValuePointer] = {
93     val docs = nodes.collect { case ValuePointer(t, p) if t == DocByte || t == ArrayByte => p }
94     matchObjectPath(docs, name.map(n => () => n == in.readName()))
95   }

97   /**
98     * Given a set of input nodes defined by their types and offsets, return a list of output nodes which
99     * match the given index
100     *
101     * @param nodes input node types and offsets
102     * @param indices array indices to match
103     * @return matching node types and offsets
104     */
105   private def matchPathIndex(nodes: Seq[ValuePointer], indices: Seq[Int]): Seq[ValuePointer] = {
106     if (indices.forall(_ >= 0)) {
107       matchPathIndices(nodes, Some(() => indices.contains(in.readName().toInt)))
108     } else {
109       // have to do post-filtering to read backwards from end
110       val all = matchPathIndices(nodes, None)
111       val res = Seq.newBuilder[ValuePointer]
112       indices.foreach { i =>
113         res += (if (i < 0) { all(all.length + i) } else { all(i) })
114       }
115       res.result()
116     }
117   }
118 
119   /**
120    * Given a set of input nodes defined by their types and offsets, return a list of output nodes which
121    * match the given index range
122    *
123    * @param nodes input node types and offsets
124    * @param from from index, inclusive, or None to start from the first element
125    * @param to to index, exclusive, or None to end with the last element
126    * @return matching node types and offsets
127    */
128   private def matchPathIndexRange(nodes: Seq[ValuePointer], from: Option[Int], to: Option[Int]): Seq[ValuePointer] = {
129     if (from.forall(_ >= 0) && to.forall(_ >= 0)) {
130       matchPathIndices(nodes, Some(() => { val i = in.readName().toInt; from.forall(_ <= i) && to.forall(_ > i) }))
131     } else {
132       // have to do post-filtering to read backwards from end
133       val all = matchPathIndices(nodes, None)
134       def offset(bound: Int): Int = if (bound >= 0) { bound } else { all.length + bound }
135       all.slice(from.map(offset).getOrElse(0), to.map(offset).getOrElse(all.length))
136     }
137   }

142    * @param nodes input node types and offsets
143    * @return matching node types and offsets

145   private def matchAllPathIndices(nodes: Seq[ValuePointer]): Seq[ValuePointer] =

148   private def matchPathIndices(nodes: Seq[ValuePointer], predicate: Option[() => Boolean]): Seq[ValuePointer] =
149     matchObjectPath(nodes.collect { case ValuePointer(ArrayByte, p) => p }, predicate)

151   /**
152     * Given a set of input nodes defined by their offsets, return a list of matching nodes
153     * which match the predicate. Input nodes must be either arrays or objects
154     *
155     * Note: predicate must consume the 'name' from the input stream
156     *
157     * @param positions input node offsets - must be arrays or objects
158     * @param nameConsumingPredicate optional predicate to match - must consume the object name from the input stream
159     * @return matching node types and offsets
160     */
161   private def matchObjectPath(positions: Seq[Int], nameConsumingPredicate: Option[() => Boolean]): Seq[ValuePointer] = {
162     val predicate: () => Boolean = nameConsumingPredicate.getOrElse(() => { in.skipName(); true })
163     val result = Seq.newBuilder[ValuePointer]
164     positions.foreach { position =>
165       in.setPosition(position)
166       val end = position + in.readInt() - 1 // last byte is the terminal byte
167       while (in.position() < end) {
168         val switch = in.readByte()
169         if (predicate()) {
170           result += ValuePointer(switch, in.position())
171         }
172         switch match {
173           case StringByte   => skipString(in)
174           case DocByte      => skipDocument(in)
175           case ArrayByte    => skipDocument(in) // arrays are stored as docs
176           case DoubleByte   => in.skip(8)
177           case IntByte      => in.skip(4)
178           case LongByte     => in.skip(8)
179           case NullByte     => // no-op
180           case BooleanByte  => skipBoolean(in)
181         }
182       }
183     }
184     result.result()
185   }

187   /**
188    * Given a set of input nodes defined by their types and offsets, returns all nodes contained by the inputs.
189    *
190    * Note: this is implemented as a depth-first search to match jayway json-path behavior
191    *
192    * @param nodes input node types and offsets
193    * @return matching node types and offsets
194    */
195   private def matchDeep(nodes: Seq[ValuePointer]): Seq[ValuePointer] = {
196     nodes.flatMap { node =>
197       node.typ match {
198         case DocByte | ArrayByte => Seq(node) ++ matchDeep(matchObjectPath(Seq(node.pos), None))
199         case _ => Seq(node) // primitives
200       }
201     }
202   }
203 
204   /**
205    * Given a set of input nodes defined by their types and offsets, return all nodes which match the given filter
206    *
207    * @param nodes input node types and offsets
208    * @param op filter op
209    * @return matching node types and offsets
210    */
211   private def matchFilter(nodes: Seq[ValuePointer], op: PathFilter.FilterOp): Seq[ValuePointer] = {
212     val predicate = KryoJsonPathFilter(in, root, op)
213     nodes.flatMap {
214       case ValuePointer(ArrayByte, pos) => matchObjectPath(Seq(pos), None).filter(predicate.apply)
215       case p if predicate(p) => Seq(p)
216       case _ => Nil
217     }
218   }

235   /**
236    * Gets the root element that the input is currently pointing to
237    *
238    * @param in input
239    * @return
240    */
241   private def root(in: Input): ValuePointer = {
242     in.readByte match {
243       case BooleanTrue  => ValuePointer(DocByte, in.position())
244       case BooleanFalse => ValuePointer(NullByte, -1)
245       case NonDoc =>
246         val switch = in.readByte()
247         in.skipName()
248         ValuePointer(switch, in.position())
249     }
250   }

252   /**
253    * Read the value for a given node path - does not return the wrapped JElement
254    *
255    * @param in input
256    * @param typed type of value being read
257    * @param position position of the value being read in the input
258    * @return value
259    */
260   private def readPathValue(in: Input, typed: Byte, position: Int): Any = {
261     import org.json4s.native.JsonMethods._
262 
263     in.setPosition(position)
264     typed match {
265       case StringByte   => readString(in)
266       case DocByte      => compact(render(readDocument(in)))
267       case ArrayByte    => unwrapArray(readArray(in))
268       case DoubleByte   => in.readDouble()
269       case IntByte      => in.readInt()
270       case LongByte     => in.readLong()
271       case NullByte     => null
272       case BooleanByte  => readBoolean(in)
273     }
274   }
275 
276   /**
277    * Unwrap a JArray into the primitive types contained within. Compared to JArray.values, this
278    * serializes documents into strings
279    *
280    * @param array array to unwrap
281    * @return
282    */
283   private def unwrapArray(array: JArray): java.util.List[Any] = {
284     array.arr.map {
285       case JString(s)              => s
286       case j: JObject              => compact(render(j))
287       case j: JArray               => unwrapArray(j)
288       case JDouble(d)              => d
289       case JInt(i) if i.isValidInt => i.intValue // note: this check needs to be a separate line to avoid auto-casting to long
290       case JInt(i)                 => i.longValue
291       case JLong(i)                => i
292       case JNull                   => null
293       case JBool(b)                => b
294     }.asJava
295   }
