39   /**
40     * Deserialize the result of a json-path.
41     *
42     * If the path selects leaf elements, they will be returned as primitives. If the path
43     * selects objects, they will be returned as strings of json. If more than one item is
44     * selected, they will be returned in a java List. If nothing is selected, it will return null.
45     *
46     * @param path json path to evaluate
47     * @return result of the path, if any
48     */
49   def deserialize(path: JsonPath): Any = {
50     require(path.nonEmpty, "Path must not be empty")
51 
52     if (root == null || root.typ == NullByte) {
53       return null
54     }
55 
56     val matches = path.elements.foldLeft(Seq(root)) {
57       case (nodes, _) if nodes.isEmpty       => nodes
58       case (nodes, PathAttribute(name, _))   => matchPathAttribute(nodes, Some(name))
59       case (nodes, PathAttributeWildCard)    => matchPathAttribute(nodes, None)
60       case (nodes, PathIndices(indices))     => matchPathIndex(nodes, indices)
61       case (nodes, PathIndexRange(from, to)) => matchPathIndexRange(nodes, from, to)
62       case (nodes, PathIndexWildCard)        => matchAllPathIndices(nodes)
63       case (nodes, PathDeepScan)             => matchDeep(nodes)
64       case (nodes, PathFilter(op))           => matchFilter(nodes, op)
65     }
66 
67     if (matches.isEmpty) {
68       // special handling to return 0 for length functions without matches
69       path.function match {
70         case Some(PathFunction.LengthFunction) => 0
71         case _ => null
72       }
73     } else {
74       val values = matches.map(m => KryoJsonPath.readPathValue(in, m.typ, m.pos))
75       val result = if (values.lengthCompare(1) == 0) { values.head } else { values.asJava }
76       path.function match {
77         case None => result
78         case Some(f) => f.apply(result)
79       }
80     }
81   }

85   /**
86     * Given a set of input nodes defined by their types and offsets, return a list of output nodes which
87     * match the given attribute name
88     *
89     * @param nodes input node types and offsets
90     * @param name attribute name to match - if None will match all names
91     * @return matching node types and offsets
92     */
93   private def matchPathAttribute(nodes: Seq[ValuePointer], name: Option[String]): Seq[ValuePointer] = {
94     val docs = nodes.collect { case ValuePointer(t, p) if t == DocByte || t == ArrayByte => p }
95     matchObjectPath(docs, name.map(n => () => n == in.readName()))
96   }

98   /**
99     * Given a set of input nodes defined by their types and offsets, return a list of output nodes which
100     * match the given index
101     *
102     * @param nodes input node types and offsets
103     * @param indices array indices to match
104     * @return matching node types and offsets
105     */
106   private def matchPathIndex(nodes: Seq[ValuePointer], indices: Seq[Int]): Seq[ValuePointer] = {
107     if (indices.forall(_ >= 0)) {
108       matchPathIndices(nodes, Some(() => indices.contains(in.readName().toInt)))
109     } else {
110       // have to do post-filtering to read backwards from end
111       val all = matchPathIndices(nodes, None)
112       val res = Seq.newBuilder[ValuePointer]
113       indices.foreach { i =>
114         res += (if (i < 0) { all(all.length + i) } else { all(i) })
115       }
116       res.result()
117     }
118   }

120   /**
121    * Given a set of input nodes defined by their types and offsets, return a list of output nodes which
122    * match the given index range
123    *
124    * @param nodes input node types and offsets
125    * @param from from index, inclusive, or None to start from the first element
126    * @param to to index, exclusive, or None to end with the last element
127    * @return matching node types and offsets
128    */
129   private def matchPathIndexRange(nodes: Seq[ValuePointer], from: Option[Int], to: Option[Int]): Seq[ValuePointer] = {
130     if (from.forall(_ >= 0) && to.forall(_ >= 0)) {
131       matchPathIndices(nodes, Some(() => { val i = in.readName().toInt; from.forall(_ <= i) && to.forall(_ > i) }))
132     } else {
133       // have to do post-filtering to read backwards from end
134       val all = matchPathIndices(nodes, None)
135       def offset(bound: Int): Int = if (bound >= 0) { bound } else { all.length + bound }
136       all.slice(from.map(offset).getOrElse(0), to.map(offset).getOrElse(all.length))
137     }
138   }

149   private def matchPathIndices(nodes: Seq[ValuePointer], predicate: Option[() => Boolean]): Seq[ValuePointer] =
150     matchObjectPath(nodes.collect { case ValuePointer(ArrayByte, p) => p }, predicate)

152   /**
153     * Given a set of input nodes defined by their offsets, return a list of matching nodes
154     * which match the predicate. Input nodes must be either arrays or objects
155     *
156     * Note: predicate must consume the 'name' from the input stream
157     *
158     * @param positions input node offsets - must be arrays or objects
159     * @param nameConsumingPredicate optional predicate to match - must consume the object name from the input stream
160     * @return matching node types and offsets
161     */
162   private def matchObjectPath(positions: Seq[Int], nameConsumingPredicate: Option[() => Boolean]): Seq[ValuePointer] = {
163     val predicate: () => Boolean = nameConsumingPredicate.getOrElse(() => { in.skipName(); true })
164     val result = Seq.newBuilder[ValuePointer]
165     positions.foreach { position =>
166       in.setPosition(position)
167       val end = position + in.readInt() - 1 // last byte is the terminal byte
168       while (in.position() < end) {
169         val switch = in.readByte()
170         if (predicate()) {
171           result += ValuePointer(switch, in.position())
172         }
173         switch match {
174           case StringByte   => skipString(in)
175           case DocByte      => skipDocument(in)
176           case ArrayByte    => skipDocument(in) // arrays are stored as docs
177           case DoubleByte   => in.skip(8)
178           case IntByte      => in.skip(4)
179           case LongByte     => in.skip(8)
180           case NullByte     => // no-op
181           case BooleanByte  => skipBoolean(in)
182         }
183       }
184     }
185     result.result()
186   }

188   /**
189    * Given a set of input nodes defined by their types and offsets, returns all nodes contained by the inputs.
190    *
191    * Note: this is implemented as a depth-first search to match jayway json-path behavior
192    *
193    * @param nodes input node types and offsets
194    * @return matching node types and offsets
195    */
196   private def matchDeep(nodes: Seq[ValuePointer]): Seq[ValuePointer] = {
197     nodes.flatMap { node =>
198       node.typ match {
199         case DocByte | ArrayByte => Seq(node) ++ matchDeep(matchObjectPath(Seq(node.pos), None))
200         case _ => Seq(node) // primitives
201       }
202     }
203   }
204 
205   /**
206    * Given a set of input nodes defined by their types and offsets, return all nodes which match the given filter
207    *
208    * @param nodes input node types and offsets
209    * @param op filter op
210    * @return matching node types and offsets
211    */
212   private def matchFilter(nodes: Seq[ValuePointer], op: PathFilter.FilterOp): Seq[ValuePointer] = {
213     val predicate = KryoJsonPathFilter(in, root, op)
214     nodes.flatMap {
215       case ValuePointer(ArrayByte, pos) => matchObjectPath(Seq(pos), None).filter(predicate.apply)
216       case p if predicate(p) => Seq(p)
217       case _ => Nil
218     }
219   }

234   /**
235    * Gets the root element that the input is currently pointing to
236    *
237    * @param in input
238    * @return
239    */
240   private def root(in: Input): ValuePointer = {
241     in.readByte match {
242       case BooleanTrue  => ValuePointer(DocByte, in.position())
243       case BooleanFalse => ValuePointer(NullByte, -1)
244       case NonDoc =>
245         val switch = in.readByte()
246         in.skipName()
247         ValuePointer(switch, in.position())
248     }
249   }

251   /**
252    * Read the value for a given node path - does not return the wrapped JElement
253    *
254    * @param in input
255    * @param typed type of value being read
256    * @param position position of the value being read in the input
257    * @return value
258    */
259   private def readPathValue(in: Input, typed: Byte, position: Int): Any = {
260     in.setPosition(position)
261     typed match {
262       case StringByte   => readString(in)
263       case DocByte      => mapper.writeValueAsString(readDocument(in))
264       case ArrayByte    => unwrapArray(readArray(in))
265       case DoubleByte   => in.readDouble()
266       case IntByte      => in.readInt()
267       case LongByte     => in.readLong()
268       case NullByte     => null
269       case BooleanByte  => readBoolean(in)
270     }
271   }

273   /**
274    * Unwrap a JArray into the primitive types contained within. Compared to JArray.values, this
275    * serializes documents into strings
276    *
277    * @param array array to unwrap
278    * @return
279    */
280   private def unwrapArray(array: ArrayNode): java.util.List[Any] = {
281     val list = new java.util.ArrayList[Any](array.size())
282     var i = 0
283     while (i < array.size()) {
284       val value = array.get(i)
285       val unwrapped =
286         if (value.isTextual || value.isBinary) {
287           value.asText()
288         } else if (value.isObject) {
289           mapper.writeValueAsString(value)
290         } else if (value.isArray) {
291           unwrapArray(value.asInstanceOf[ArrayNode])
292         } else if (value.isDouble || value.isFloat) {
293           value.asDouble()
294         } else if (value.isInt || value.isShort) {
295           value.intValue()
296         } else if (value.isLong) {
297           value.longValue()
298         } else if (value.isNull) {
299           null
300         } else if (value.isBoolean) {
301           value.booleanValue()
302         } else if (value.isBigDecimal) {
303           value.asDouble()
304         } else if (value.isBigInteger) {
305           if (value.canConvertToInt) {
306             value.intValue()
307           } else if (value.canConvertToLong) {
308             value.longValue()
309           } else {
310             logger.warn(s"Skipping int value that does not fit in a long: $value")
311             null
312           }
313         } else {
314           logger.warn(s"Unhandled JsonNode: $value")
315           null
316         }
317       if (unwrapped != null) {
318         list.add(unwrapped)
319       }
320       i += 1
321     }
322     Collections.unmodifiableList(list)
323   }
