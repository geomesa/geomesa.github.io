23 /**
24  * Class that writes out Avro SimpleFeature Data Files which carry the SimpleFeatureType Schema
25  * along with them.
26  *
27  * @param os output stream.
28  * @param sft simple feature type being written
29  * @param compression compression level, from -1 to 9. @see java.util.zip.Deflater
30  */
31 class AvroDataFileWriter(
32     os: OutputStream,
33     sft: SimpleFeatureType,
34     compression: Int = Deflater.DEFAULT_COMPRESSION,
35     opts: Set[SerializationOption] = Set.empty
36   ) extends Closeable with Flushable {
37 
38   // constructors for java interop
39   def this(os: OutputStream, sft: SimpleFeatureType, compression: Int) = this(os, sft, compression, Set.empty)
40 
41   private val writer = new SimpleFeatureDatumWriter(sft, SerializationOption.WithUserData ++ opts)
42   private val dfw    = new DataFileWriter[SimpleFeature](writer)
43 
44   if (compression != Deflater.NO_COMPRESSION) {
45     dfw.setCodec(CodecFactory.deflateCodec(compression))
46   }
47   AvroDataFile.setMetaData(dfw, sft)
48   dfw.create(writer.getSchema, os)
49 
50   def append(fc: SimpleFeatureCollection): Unit = CloseableIterator(fc.features()).foreach(dfw.append)
51 
52   def append(sf: SimpleFeature): Unit = dfw.append(sf)
53 
54   override def close(): Unit = if (dfw != null) { dfw.close() }
55 
56   override def flush(): Unit = if (dfw != null) { dfw.flush() }
57 }
