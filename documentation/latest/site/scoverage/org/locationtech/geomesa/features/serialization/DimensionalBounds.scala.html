14 /**
15   * Extracts the bounds from a geometry. The generic geometry envelope only deals with X and Y, this
16   * also supports Z and M
17   *
18   * Operations expect a non-empty geometry
19   *
20   * @tparam T geometry type
21   */
22 trait DimensionalBounds[T <: Geometry] {
23 
24   /**
25     * Get bounds for the x dimension
26     *
27     * @param geometry geometry, not null and not empty
28     * @return (min, max)
29     */
30   def x(geometry: T): (Double, Double) = (geometry.getEnvelopeInternal.getMinX, geometry.getEnvelopeInternal.getMaxX)
31 
32   /**
33     * Get bounds for the y dimension
34     *
35     * @param geometry geometry, not null and not empty
36     * @return (min, max)
37     */
38   def y(geometry: T): (Double, Double) = (geometry.getEnvelopeInternal.getMinY, geometry.getEnvelopeInternal.getMaxY)
39 
40   /**
41     * Get bounds for the z dimension
42     *
43     * @param geometry geometry, not null and not empty
44     * @return (min, max)
45     */
46   def z(geometry: T): (Double, Double) = bounds(geometry, ZAccessor)
47 
48   /**
49     * Get bounds for the m dimension
50     *
51     * @param geometry geometry, not null and not empty
52     * @return (min, max)
53     */
54   def m(geometry: T): (Double, Double) = bounds(geometry, MAccessor)
55 
56   private [serialization] def bounds(geometry: T, accessor: CoordAccessor): (Double, Double)
57 }

61   implicit object LineStringBounds extends DimensionalBounds[LineString] {
62     override private [serialization] def bounds(
63         geometry: LineString,
64         accessor: CoordAccessor): (Double, Double) = {
65       var min = accessor(geometry.getCoordinateN(0))
66       var max = min
67       var i = 1
68       while (i < geometry.getNumPoints) {
69         val next = accessor(geometry.getCoordinateN(i))
70         if (next < min) {
71           min = next
72         } else if (next > max) {
73           max = next
74         }
75         i += 1
76       }
77       (min, max)
78     }
79   }
80
