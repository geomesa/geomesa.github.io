27   private lazy val sort = hints.getArrowSort
28   private lazy val encoding = SimpleFeatureEncoding.min(hints.isArrowIncludeFid, hints.isArrowProxyFid)
29   private lazy val ipc = hints.getArrowFormatVersion.getOrElse(FormatVersion.ArrowFormatVersion.get)
30   private lazy val batchSize = hints.getArrowBatchSize.getOrElse(ArrowProperties.BatchSize.get.toInt)
31   private lazy val dictionaryFields = hints.getArrowDictionaryFields
32 
33   private var delegate: FeatureExporter = _
34 
35   override def start(sft: SimpleFeatureType): Unit = {
36     delegate = if (sft == org.locationtech.geomesa.arrow.ArrowEncodedSft) {
37       new EncodedDelegate(out)
38     } else if (dictionaryFields.isEmpty) {
39       // note: features should be sorted already, even if arrow encoding wasn't performed
40       new BatchDelegate(out, encoding, FormatVersion.options(ipc), sort, batchSize, Map.empty)
41     } else {
42       if (sort.isDefined) {
43         throw new NotImplementedError("Sorting and calculating dictionaries at the same time is not supported")
44       }
45       new DictionaryDelegate(out, dictionaryFields, encoding, FormatVersion.options(ipc), batchSize)
46     }
47     delegate.start(sft)
48   }
49 
50   override def export(features: Iterator[SimpleFeature]): Option[Long] = delegate.export(features)
51 
52   override def close(): Unit = {
53     CloseWithLogging(Option(delegate))
54     out.close()
55   }

60   private class EncodedDelegate(os: OutputStream) extends FeatureExporter {
61     override def start(sft: SimpleFeatureType): Unit = {}
62     override def export(features: Iterator[SimpleFeature]): Option[Long] = {
63       // just copy bytes directly out
64       features.foreach(f => os.write(f.getAttribute(0).asInstanceOf[Array[Byte]]))
65       None // we don't know the actual count

67     override def close(): Unit = {}

70   private class DictionaryDelegate(
71       os: OutputStream,
72       dictionaryFields: Seq[String],
73       encoding: SimpleFeatureEncoding,
74       ipcOpts: IpcOption,
75       batchSize: Int
76     ) extends FeatureExporter {
77 
78     private var writer: DictionaryBuildingWriter = _
79     private var count = 0L
80 
81     override def start(sft: SimpleFeatureType): Unit = {
82       writer = new DictionaryBuildingWriter(sft, dictionaryFields, encoding, ipcOpts)
83     }
84 
85     override def export(features: Iterator[SimpleFeature]): Option[Long] = {
86       val start = count
87       features.foreach { f =>
88         writer.add(f)
89         count += 1
90         if (count % batchSize == 0) {
91           writer.encode(os)
92           writer.clear()
93         }
94       }
95       Some(count - start)
96     }
97 
98     override def close(): Unit = {
99       if (writer != null) {
100         if (count % batchSize != 0) {
101           writer.encode(os)
102           writer.clear()
103         }
104         CloseWithLogging(writer)
105       }
106     }
107   }
108 
109   private class BatchDelegate(
110       os: OutputStream,
111       encoding: SimpleFeatureEncoding,
112       ipcOpts: IpcOption,
113       sort: Option[(String, Boolean)],
114       batchSize: Int,
115       dictionaries: Map[String, ArrowDictionary]
116     ) extends FeatureExporter {
117 
118     private var writer: SimpleFeatureArrowFileWriter = _
119     private var count = 0L
120 
121     override def start(sft: SimpleFeatureType): Unit = {
122       writer = SimpleFeatureArrowFileWriter(os, sft, dictionaries, encoding, ipcOpts, sort)
123     }
124 
125     override def export(features: Iterator[SimpleFeature]): Option[Long] = {
126       val start = count
127       features.foreach { f =>
128         writer.add(f)
129         count += 1
130         if (count % batchSize == 0) {
131           writer.flush()
132         }
133       }
134       Some(count - start)
135     }
136 
137     override def close(): Unit = {
138       if (writer != null) {
139         CloseWithLogging(writer)
140       }
141     }
142   }
