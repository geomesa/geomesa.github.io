23 class ArrowExporter(out: OutputStream, hints: Hints) extends FeatureExporter {
24 
25   import org.locationtech.geomesa.index.conf.QueryHints.RichHints
26 
27   private lazy val sort = hints.getArrowSort
28   private lazy val encoding = SimpleFeatureEncoding.min(hints.isArrowIncludeFid, hints.isArrowProxyFid, hints.isFlipAxisOrder)
29   private lazy val ipc = hints.getArrowFormatVersion.getOrElse(FormatVersion.ArrowFormatVersion.get)
30   private lazy val batchSize = hints.getArrowBatchSize.getOrElse(ArrowProperties.BatchSize.get.toInt)
31   private lazy val dictionaryFields = hints.getArrowDictionaryFields
32   private lazy val flattenFields = hints.isArrowFlatten
33 
34   private var delegate: FeatureExporter = _
35 
36   override def start(sft: SimpleFeatureType): Unit = {
37     delegate = if (sft == org.locationtech.geomesa.arrow.ArrowEncodedSft) {
38       new EncodedDelegate(out)
39     } else if (dictionaryFields.isEmpty) {
40       // note: features should be sorted already, even if arrow encoding wasn't performed
41       new BatchDelegate(out, encoding, FormatVersion.options(ipc), sort, batchSize, Map.empty, flattenFields)
42     } else {
43       if (sort.isDefined) {
44         throw new NotImplementedError("Sorting and calculating dictionaries at the same time is not supported")
45       }
46       new DictionaryDelegate(out, dictionaryFields, encoding, FormatVersion.options(ipc), batchSize, flattenFields)
47     }
48     delegate.start(sft)
49   }
50 
51   override def export(features: Iterator[SimpleFeature]): Option[Long] = delegate.export(features)
52 
53   override def close(): Unit = {
54     CloseWithLogging(Option(delegate))
55     out.close()
56   }
57 }

61   private class EncodedDelegate(os: OutputStream) extends FeatureExporter {
62     override def start(sft: SimpleFeatureType): Unit = {}
63     override def export(features: Iterator[SimpleFeature]): Option[Long] = {
64       // just copy bytes directly out
65       features.foreach(f => os.write(f.getAttribute(0).asInstanceOf[Array[Byte]]))
66       None // we don't know the actual count
67     }
68     override def close(): Unit = {}
69   }
70 
71   private class DictionaryDelegate(
72       os: OutputStream,
73       dictionaryFields: Seq[String],
74       encoding: SimpleFeatureEncoding,
75       ipcOpts: IpcOption,
76       batchSize: Int,
77       flattenStruct: Boolean = false
78     ) extends FeatureExporter {
79 
80     private var writer: DictionaryBuildingWriter = _
81     private var count = 0L
82 
83     override def start(sft: SimpleFeatureType): Unit = {
84       writer = new DictionaryBuildingWriter(sft, dictionaryFields, encoding, ipcOpts, flattenStruct = flattenStruct)
85     }
86 
87     override def export(features: Iterator[SimpleFeature]): Option[Long] = {
88       val start = count
89       features.foreach { f =>
90         writer.add(f)
91         count += 1
92         if (count % batchSize == 0) {
93           writer.encode(os)
94           writer.clear()
95         }
96       }
97       Some(count - start)
98     }
99 
100     override def close(): Unit = {
101       if (writer != null) {
102         if (count % batchSize != 0) {
103           writer.encode(os)
104           writer.clear()
105         }
106         CloseWithLogging(writer)
107       }
108     }
109   }
110 
111   private class BatchDelegate(
112       os: OutputStream,
113       encoding: SimpleFeatureEncoding,
114       ipcOpts: IpcOption,
115       sort: Option[(String, Boolean)],
116       batchSize: Int,
117       dictionaries: Map[String, ArrowDictionary],
118       flattenStruct: Boolean = false
119     ) extends FeatureExporter {
120 
121     private var writer: SimpleFeatureArrowFileWriter = _
122     private var count = 0L
123 
124     override def start(sft: SimpleFeatureType): Unit = {
125       writer = SimpleFeatureArrowFileWriter(os, sft, dictionaries, encoding, ipcOpts, sort, flattenStruct = flattenStruct)
126     }
127 
128     override def export(features: Iterator[SimpleFeature]): Option[Long] = {
129       val start = count
130       features.foreach { f =>
131         writer.add(f)
132         count += 1
133         if (count % batchSize == 0) {
134           writer.flush()
135         }
136       }
137       Some(count - start)
138     }
139 
140     override def close(): Unit = {
141       if (writer != null) {
142         CloseWithLogging(writer)
143       }
144     }
145   }
