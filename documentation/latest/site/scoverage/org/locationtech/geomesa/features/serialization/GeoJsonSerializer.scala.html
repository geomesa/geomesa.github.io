21 /**
22  * Serializer to write out geojson from simple features. There are two valid usage patterns.
23  *
24  * To encode a feature collection with an array of features:
25  *
26  * &lt;ol&gt;
27  *   &lt;li&gt;`startFeatureCollection`&lt;/li&gt;
28  *   &lt;li&gt;`write` (0-n)&lt;/li&gt;
29  *   &lt;li&gt;`endFeatureCollection`&lt;/li&gt;
30  * &lt;/ol&gt;
31  *
32  * To encode individual features:
33  *
34  * &lt;ol&gt;
35  *   &lt;li&gt;`write` (0-n)&lt;/li&gt;
36  * &lt;/ol&gt;
37  *
38  * Note that encoding individual features is not valid json, so generally you would
39  * want to split up the output after each feature.
40  *
41  * @param sft simple feature type
42  */
43 class GeoJsonSerializer(sft: SimpleFeatureType) extends LazyLogging {
44 
45   import GeoJsonSerializer._
46 
47   private val defaultGeomIndex = sft.indexOf(sft.getGeometryDescriptor.getLocalName)
48 
49   private val geometryWriter = new GeometryWriter("geometry", defaultGeomIndex)
50 
51   private val attributeWriters = Seq.tabulate(sft.getAttributeCount - 1) { i =>
52     GeoJsonSerializer.createAttributeWriter(sft, if (i < defaultGeomIndex) { i } else { i + 1 })
53   }
54 
55   /**
56    * Start a feature collection type
57    *
58    * @param writer json writer
59    */
60   def startFeatureCollection(writer: JsonWriter): Unit =
61     writer.beginObject().name("type").value("FeatureCollection").name("features").beginArray()
62 
63   /**
64    * Write a single feature
65    *
66    * @param writer json writer
67    * @param feature feature
68    */
69   def write(writer: JsonWriter, feature: SimpleFeature): Unit = {
70     writer.beginObject()
71     writer.name("type").value("Feature")
72     writer.name("id").value(feature.getID)
73     geometryWriter.apply(feature, writer)
74     writer.name("properties").beginObject()
75     attributeWriters.foreach(_.apply(feature, writer))
76     writer.endObject()
77     writer.endObject()
78   }
79 
80   /**
81    * End a previously started feature collection
82    *
83    * @param writer json writer
84    */
85   def endFeatureCollection(writer: JsonWriter): Unit = writer.endArray().endObject()
86 }

99   private val JsonObjectPattern = Pattern.compile("^\\s*[{\\[]")

101   /**
102    * Create a JsonWriter from a standard writer.
103    *
104    * Note that closing the JsonWriter will close the wrapped writer. Technically the JsonWriter does not need
105    * to be closed if you close the wrapped writer directly, although this is mainly a gson implementation
106    * detail and may change in the future.
107    *
108    * @param wrapped writer
109    * @return
110    */
111   def writer(wrapped: Writer): JsonWriter = {
112     val writer = new JsonWriter(wrapped)
113     writer.setSerializeNulls(true)
114     writer.setLenient(true) // allow multiple top-level objects
115     writer
116   }

118   private def createAttributeWriter(sft: SimpleFeatureType, i: Int): JsonAttributeWriter = {
119     val descriptor = sft.getDescriptor(i)
120     val name = descriptor.getLocalName
121     val bindings = ObjectType.selectType(descriptor)
122 
123     bindings.head match {
124       case ObjectType.STRING if bindings.last == ObjectType.JSON => new JsonStringWriter(name, i)
125       case ObjectType.STRING   => new StringWriter(name, i)
126       case ObjectType.INT      => new NumberWriter(name, i)
127       case ObjectType.LONG     => new NumberWriter(name, i)
128       case ObjectType.FLOAT    => new NumberWriter(name, i)
129       case ObjectType.DOUBLE   => new NumberWriter(name, i)
130       case ObjectType.DATE     => new DateWriter(name, i)
131       case ObjectType.GEOMETRY => new GeometryWriter(name, i)
132       case ObjectType.UUID     => new UuidWriter(name, i)
133       case ObjectType.BYTES    => new BytesWriter(name, i)
134       case ObjectType.BOOLEAN  => new BooleanWriter(name, i)
135       case ObjectType.LIST     => new ListWriter(name, i, bindings(1))
136       case ObjectType.MAP      => new MapWriter(name, i, bindings(1), bindings(2))
137 
138       case _ =>
139         logger.warn(s"Dropping unsupported attribute '$name:${descriptor.getType.getBinding.getName}'")
140         NoopWriter
141     }
142   }
143 
144   sealed private trait JsonAttributeWriter {
145     def apply(feature: SimpleFeature, writer: JsonWriter): Unit
146   }
147 
148   private object NoopWriter extends JsonAttributeWriter {
149     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit = {}
150   }
151 
152   private class StringWriter(name: String, i: Int) extends JsonAttributeWriter {
153     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit =
154       writer.name(name).value(feature.getAttribute(i).asInstanceOf[String])
155   }
156 
157   private class JsonStringWriter(name: String, i: Int) extends JsonAttributeWriter {
158     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit = {
159       writer.name(name)
160       feature.getAttribute(i).asInstanceOf[String] match {
161         case null => writer.nullValue()
162         // note: this check isn't exhaustive, and may still produce invalid json if the value is badly formatted
163         case obj if JsonObjectPattern.matcher(obj).find() => writer.jsonValue(obj.trim)
164         // note: we don't support primitive types, but this is here to prevent bad input from producing invalid json
165         case primitive => writer.value(primitive)
166       }
167     }
168   }
169 
170   private class NumberWriter(name: String, i: Int) extends JsonAttributeWriter {
171     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit =
172       writer.name(name).value(feature.getAttribute(i).asInstanceOf[Number])
173   }
174 
175   private class BooleanWriter(name: String, i: Int) extends JsonAttributeWriter {
176     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit =
177       writer.name(name).value(feature.getAttribute(i).asInstanceOf[java.lang.Boolean])
178   }
179 
180   private class DateWriter(name: String, i: Int) extends JsonAttributeWriter {
181     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit = {
182       writer.name(name)
183       val date = feature.getAttribute(i).asInstanceOf[Date]
184       if (date == null) { writer.nullValue() } else {
185         writer.value(DateParsing.formatDate(date))
186       }
187     }
188   }
189 
190   private class GeometryWriter(name: String, i: Int) extends JsonAttributeWriter {
191     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit = {
192       writer.name(name)
193       val geom = feature.getAttribute(i).asInstanceOf[Geometry]
194       if (geom == null) { writer.nullValue() } else {
195         writer.jsonValue(GeometryWriter.write(geom))
196       }
197     }
198   }
199 
200   private class UuidWriter(name: String, i: Int) extends JsonAttributeWriter {
201     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit = {
202       writer.name(name)
203       val uuid = feature.getAttribute(i).asInstanceOf[UUID]
204       if (uuid == null) { writer.nullValue() } else {
205         writer.value(uuid.toString)
206       }
207     }
208   }
209 
210   private class BytesWriter(name: String, i: Int) extends JsonAttributeWriter {
211     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit = {
212       writer.name(name)
213       val bytes = feature.getAttribute(i).asInstanceOf[Array[Byte]]
214       if (bytes == null) { writer.nullValue() } else {
215         writer.value(Base64.getEncoder.encodeToString(bytes))
216       }
217     }
218   }
219 
220   private class ListWriter(name: String, i: Int, binding: ObjectType) extends JsonAttributeWriter {
221 
222     import scala.collection.JavaConverters._
223 
224     private val elementWriter: (JsonWriter, Any) => Unit = subWriter(binding)
225 
226     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit = {
227       writer.name(name)
228       val list = feature.getAttribute(i).asInstanceOf[java.util.List[_]]
229       if (list == null) {
230         writer.nullValue()
231       } else {
232         writer.beginArray()
233         list.asScala.foreach(elementWriter.apply(writer, _))
234         writer.endArray()
235       }
236     }
237   }
238 
239   private class MapWriter(name: String, i: Int, keyBinding: ObjectType, valueBinding: ObjectType)
240       extends JsonAttributeWriter {
241 
242     import scala.collection.JavaConverters._
243 
244     private val valueWriter: (JsonWriter, Any) => Unit = subWriter(valueBinding)
245 
246     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit = {
247       writer.name(name)
248       val map = feature.getAttribute(i).asInstanceOf[java.util.Map[_, _]]
249       if (map == null) {
250         writer.nullValue()
251       } else {
252         writer.beginObject()
253         map.asScala.foreach { case (k, v) =>
254           writer.name(k.toString)
255           valueWriter(writer, v)
256         }
257         writer.endObject()
258       }
259     }
260   }
261 
262   private def subWriter(binding: ObjectType): (JsonWriter, Any) => Unit = binding match {
263     case ObjectType.INT      => (writer, elem) => writer.value(elem.asInstanceOf[Number])
264     case ObjectType.LONG     => (writer, elem) => writer.value(elem.asInstanceOf[Number])
265     case ObjectType.FLOAT    => (writer, elem) => writer.value(elem.asInstanceOf[Number])
266     case ObjectType.DOUBLE   => (writer, elem) => writer.value(elem.asInstanceOf[Number])
267     case ObjectType.DATE     => (writer, elem) => writer.value(DateParsing.formatDate(elem.asInstanceOf[Date]))
268     case ObjectType.GEOMETRY => (writer, elem) => writer.jsonValue(GeometryWriter.write(elem.asInstanceOf[Geometry]))
269     case ObjectType.BYTES    => (writer, elem) => writer.value(Base64.getEncoder.encodeToString(elem.asInstanceOf[Array[Byte]]))
270     case ObjectType.BOOLEAN  => (writer, elem) => writer.value(elem.asInstanceOf[java.lang.Boolean])
271     case _                   => (writer, elem) => writer.value(elem.toString)
272   }
