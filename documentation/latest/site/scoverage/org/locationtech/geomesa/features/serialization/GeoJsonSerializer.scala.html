22 /**
23  * Serializer to write out geojson from simple features. There are two valid usage patterns.
24  *
25  * To encode a feature collection with an array of features:
26  *
27  * &lt;ol&gt;
28  *   &lt;li&gt;`startFeatureCollection`&lt;/li&gt;
29  *   &lt;li&gt;`write` (0-n)&lt;/li&gt;
30  *   &lt;li&gt;`endFeatureCollection`&lt;/li&gt;
31  * &lt;/ol&gt;
32  *
33  * To encode individual features:
34  *
35  * &lt;ol&gt;
36  *   &lt;li&gt;`write` (0-n)&lt;/li&gt;
37  * &lt;/ol&gt;
38  *
39  * Note that encoding individual features is not valid json, so generally you would
40  * want to split up the output after each feature.
41  *
42  * @param sft simple feature type
43  */
44 class GeoJsonSerializer(sft: SimpleFeatureType) extends LazyLogging {
45 
46   import GeoJsonSerializer._
47 
48   private val defaultGeomIndex = sft.indexOf(sft.getGeometryDescriptor.getLocalName)
49 
50   private val geometryWriter = new GeometryWriter("geometry", defaultGeomIndex)
51 
52   private val attributeWriters = Seq.tabulate(sft.getAttributeCount - 1) { i =>
53     GeoJsonSerializer.createAttributeWriter(sft, if (i < defaultGeomIndex) { i } else { i + 1 })
54   }
55 
56   /**
57    * Start a feature collection type
58    *
59    * @param writer json writer
60    */
61   def startFeatureCollection(writer: JsonWriter): Unit =
62     writer.beginObject().name("type").value("FeatureCollection").name("features").beginArray()
63 
64   /**
65    * Write a single feature
66    *
67    * @param writer json writer
68    * @param feature feature
69    */
70   def write(writer: JsonWriter, feature: SimpleFeature): Unit = {
71     writer.beginObject()
72     writer.name("type").value("Feature")
73     writer.name("id").value(feature.getID)
74     geometryWriter.apply(feature, writer)
75     writer.name("properties").beginObject()
76     attributeWriters.foreach(_.apply(feature, writer))
77     writer.endObject()
78     writer.endObject()
79   }
80 
81   /**
82    * End a previously started feature collection
83    *
84    * @param writer json writer
85    */
86   def endFeatureCollection(writer: JsonWriter): Unit = writer.endArray().endObject()
87 }

100   private val JsonObjectPattern = Pattern.compile("^\\s*[{\\[]")

102   /**
103    * Create a JsonWriter from a standard writer.
104    *
105    * Note that closing the JsonWriter will close the wrapped writer. Technically the JsonWriter does not need
106    * to be closed if you close the wrapped writer directly, although this is mainly a gson implementation
107    * detail and may change in the future.
108    *
109    * @param wrapped writer
110    * @return
111    */
112   def writer(wrapped: Writer): JsonWriter = {
113     val writer = new JsonWriter(wrapped)
114     writer.setSerializeNulls(true)
115     writer.setLenient(true) // allow multiple top-level objects
116     writer
117   }

119   private def createAttributeWriter(sft: SimpleFeatureType, i: Int): JsonAttributeWriter = {
120     val descriptor = sft.getDescriptor(i)
121     val name = descriptor.getLocalName
122     val bindings = ObjectType.selectType(descriptor)
123 
124     bindings.head match {
125       case ObjectType.STRING if bindings.last == ObjectType.JSON => new JsonStringWriter(name, i)
126       case ObjectType.STRING   => new StringWriter(name, i)
127       case ObjectType.INT      => new NumberWriter(name, i)
128       case ObjectType.LONG     => new NumberWriter(name, i)
129       case ObjectType.FLOAT    => new NumberWriter(name, i)
130       case ObjectType.DOUBLE   => new NumberWriter(name, i)
131       case ObjectType.DATE     => new DateWriter(name, i)
132       case ObjectType.GEOMETRY => new GeometryWriter(name, i)
133       case ObjectType.UUID     => new UuidWriter(name, i)
134       case ObjectType.BYTES    => new BytesWriter(name, i)
135       case ObjectType.BOOLEAN  => new BooleanWriter(name, i)
136       case ObjectType.LIST     => new ListWriter(name, i, bindings(1))
137       case ObjectType.MAP      => new MapWriter(name, i, bindings(1), bindings(2))
138 
139       case _ =>
140         logger.warn(s"Dropping unsupported attribute '${SimpleFeatureTypes.encodeDescriptor(sft, descriptor)}'")
141         NoopWriter
142     }
143   }
144 
145   sealed private trait JsonAttributeWriter {
146     def apply(feature: SimpleFeature, writer: JsonWriter): Unit
147   }
148 
149   private object NoopWriter extends JsonAttributeWriter {
150     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit = {}
151   }
152 
153   private class StringWriter(name: String, i: Int) extends JsonAttributeWriter {
154     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit =
155       writer.name(name).value(feature.getAttribute(i).asInstanceOf[String])
156   }
157 
158   private class JsonStringWriter(name: String, i: Int) extends JsonAttributeWriter {
159     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit = {
160       writer.name(name)
161       feature.getAttribute(i).asInstanceOf[String] match {
162         case null => writer.nullValue()
163         // note: this check isn't exhaustive, and may still produce invalid json if the value is badly formatted
164         case obj if JsonObjectPattern.matcher(obj).find() => writer.jsonValue(obj.trim)
165         // note: we don't support primitive types, but this is here to prevent bad input from producing invalid json
166         case primitive => writer.value(primitive)
167       }
168     }
169   }
170 
171   private class NumberWriter(name: String, i: Int) extends JsonAttributeWriter {
172     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit =
173       writer.name(name).value(feature.getAttribute(i).asInstanceOf[Number])
174   }
175 
176   private class BooleanWriter(name: String, i: Int) extends JsonAttributeWriter {
177     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit =
178       writer.name(name).value(feature.getAttribute(i).asInstanceOf[java.lang.Boolean])
179   }
180 
181   private class DateWriter(name: String, i: Int) extends JsonAttributeWriter {
182     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit = {
183       writer.name(name)
184       val date = feature.getAttribute(i).asInstanceOf[Date]
185       if (date == null) { writer.nullValue() } else {
186         writer.value(DateParsing.formatDate(date))
187       }
188     }
189   }
190 
191   private class GeometryWriter(name: String, i: Int) extends JsonAttributeWriter {
192     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit = {
193       writer.name(name)
194       val geom = feature.getAttribute(i).asInstanceOf[Geometry]
195       if (geom == null) { writer.nullValue() } else {
196         writer.jsonValue(GeometryWriter.write(geom))
197       }
198     }
199   }

201   private class UuidWriter(name: String, i: Int) extends JsonAttributeWriter {
202     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit = {
203       writer.name(name)
204       val uuid = feature.getAttribute(i).asInstanceOf[UUID]
205       if (uuid == null) { writer.nullValue() } else {
206         writer.value(uuid.toString)
207       }
208     }
209   }

211   private class BytesWriter(name: String, i: Int) extends JsonAttributeWriter {
212     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit = {
213       writer.name(name)
214       val bytes = feature.getAttribute(i).asInstanceOf[Array[Byte]]
215       if (bytes == null) { writer.nullValue() } else {
216         writer.value(Base64.getEncoder.encodeToString(bytes))
217       }
218     }
219   }

221   private class ListWriter(name: String, i: Int, binding: ObjectType) extends JsonAttributeWriter {
222 
223     import scala.collection.JavaConverters._
224 
225     private val elementWriter: (JsonWriter, Any) => Unit = subWriter(binding)
226 
227     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit = {
228       writer.name(name)
229       val list = feature.getAttribute(i).asInstanceOf[java.util.List[_]]
230       if (list == null) {
231         writer.nullValue()
232       } else {
233         writer.beginArray()
234         list.asScala.foreach(elementWriter.apply(writer, _))
235         writer.endArray()
236       }
237     }
238   }
239 
240   private class MapWriter(name: String, i: Int, keyBinding: ObjectType, valueBinding: ObjectType)
241       extends JsonAttributeWriter {
242 
243     import scala.collection.JavaConverters._
244 
245     private val valueWriter: (JsonWriter, Any) => Unit = subWriter(valueBinding)
246 
247     override def apply(feature: SimpleFeature, writer: JsonWriter): Unit = {
248       writer.name(name)
249       val map = feature.getAttribute(i).asInstanceOf[java.util.Map[_, _]]
250       if (map == null) {
251         writer.nullValue()
252       } else {
253         writer.beginObject()
254         map.asScala.foreach { case (k, v) =>
255           writer.name(k.toString)
256           valueWriter(writer, v)
257         }
258         writer.endObject()
259       }
260     }
261   }
262 
263   private def subWriter(binding: ObjectType): (JsonWriter, Any) => Unit = binding match {
264     case ObjectType.INT      => (writer, elem) => writer.value(elem.asInstanceOf[Number])
265     case ObjectType.LONG     => (writer, elem) => writer.value(elem.asInstanceOf[Number])
266     case ObjectType.FLOAT    => (writer, elem) => writer.value(elem.asInstanceOf[Number])
267     case ObjectType.DOUBLE   => (writer, elem) => writer.value(elem.asInstanceOf[Number])
268     case ObjectType.DATE     => (writer, elem) => writer.value(DateParsing.formatDate(elem.asInstanceOf[Date]))
269     case ObjectType.GEOMETRY => (writer, elem) => writer.jsonValue(GeometryWriter.write(elem.asInstanceOf[Geometry]))
270     case ObjectType.BYTES    => (writer, elem) => writer.value(Base64.getEncoder.encodeToString(elem.asInstanceOf[Array[Byte]]))
271     case ObjectType.BOOLEAN  => (writer, elem) => writer.value(elem.asInstanceOf[java.lang.Boolean])
272     case _                   => (writer, elem) => writer.value(elem.toString)
273   }
