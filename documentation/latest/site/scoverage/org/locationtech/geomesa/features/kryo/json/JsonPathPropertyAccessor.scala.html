28 /**
29   * Access values from a json-type string field. Syntax must start with '$.'. The first part of the path
30   * selects the simple feature attribute, and the rest of the path selects within the json contained in
31   * that attribute.
32   *
33   * Note: this class is optimized for `KryoBufferSimpleFeature`s. It will work on standard simple features,
34   * but will incur a serialization cost.
35   */

42   override def canHandle(obj: AnyRef, xpath: String, target: Class[_]): Boolean = {
43     val path = try { JsonPathPropertyAccessor.paths.get(xpath) } catch { case NonFatal(e) => e.printStackTrace; JsonPath(Seq.empty, None) }
44 
45     if (path.isEmpty) { false } else {
46       path.head match {
47         case PathAttribute(name: String, _) =>
48           val descriptor = obj match {
49             case s: SimpleFeature => s.getFeatureType.getDescriptor(name)
50             case s: SimpleFeatureType => s.getDescriptor(name)
51             case _ => null
52           }
53           descriptor != null && descriptor.getType.getBinding == classOf[String]
54 
55         case PathAttributeWildCard | PathDeepScan =>
56           val sft = obj match {
57             case s: SimpleFeature => s.getFeatureType
58             case s: SimpleFeatureType => s
59             case _ => null
60           }
61           sft != null && sft.getAttributeDescriptors.asScala.exists(_.getType.getBinding == classOf[String])
62 
63         case _ => false
64       }
65     }
66   }

68   override def get[T](obj: AnyRef, xpath: String, target: Class[T]): T = {
69 
70     val path = JsonPathPropertyAccessor.paths.get(xpath)
71 
72     val result = obj match {
73       case s: KryoBufferSimpleFeature =>
74         val i = attribute(s.getFeatureType, path.head)
75         if (s.getFeatureType.getDescriptor(i).isJson()) {
76           s.getInput(i).map(KryoJsonSerialization.deserialize(_, path.tail)).orNull
77         } else {
78           JsonPathPropertyAccessor.evaluateJsonPath(s.getAttribute(i).asInstanceOf[String], path.tail)
79         }
80 
81       case s: SimpleFeature =>
82         JsonPathPropertyAccessor.evaluateJsonPath(s.getAttribute(attribute(s.getFeatureType, path.head)).asInstanceOf[String], path.tail)
83 
84       case s: SimpleFeatureType =>
85         // remove the json flag, so that transform serializations don't try to serialize
86         // json path results that aren't valid json objects or arrays
87         val descriptor = s.getDescriptor(attribute(s, path.head))
88         val builder = new AttributeTypeBuilder()
89         builder.init(descriptor)
90         val result = builder.buildDescriptor(descriptor.getLocalName)
91         result.getUserData.remove(SimpleFeatureTypes.AttributeOptions.OptJson)
92         result
93     }
94 
95     if (target == null) {
96       result.asInstanceOf[T]
97     } else {
98       FastConverter.convert(result, target)
99     }
100   }
101 
102   override def set[T](obj: Any, xpath: String, value: T, target: Class[T]): Unit = throw new UnsupportedOperationException()

104   private def attribute(sft: SimpleFeatureType, head: PathElement): Int = {
105     head match {
106       case PathAttribute(name: String, _) =>
107         sft.indexOf(name)
108 
109       case _ =>
110         // we know it will be a wildcard due to canHandle
111         // prioritize fields marked json over generic strings
112         // note: will only match first json attribute if more than 1
113         val i = sft.getAttributeDescriptors.asScala.indexWhere(_.isJson())
114         if (i != -1) { i } else {
115           sft.getAttributeDescriptors.asScala.indexWhere(_.getType.getBinding == classOf[String])
116         }
117     }
118   }

154   /**
155    * Evaluate a json path using the json-path lib. This method is a fallback for our custom kryo-deserialization,
156    * and requires parsing the input into the json-path AST. The result has been modified to be consistent
157    * with our custom serialization (for better interop with geotools), i.e. json objects are serialized to json
158    * strings.
159    *
160    * @param json json doc
161    * @param path json path
162    * @return
163    */
164   private[json] def evaluateJsonPath(json: String, path: JsonPath): AnyRef = {
165     val parsed = com.jayway.jsonpath.JsonPath.using(pathConfig).parse(json)
166     val list = parsed.read[java.util.List[AnyRef]](JsonPathParser.print(path))
167     if (list == null || list.isEmpty) {
168       // special handling to return 0 for length functions without matches
169       path.function match {
170         case Some(PathFunction.LengthFunction) => Int.box(0)
171         case _ => null
172       }
173     } else {
174       val transformed = list.asScala.map {
175         case o: java.util.Map[String, AnyRef] => JSONObject.toJSONString(o)
176         case a: java.util.List[AnyRef] => unwrapArray(a)
177         case p => p
178       }
179       if (transformed.lengthCompare(1) == 0) { transformed.head } else { transformed.asJava }
180     }
181   }
182 
183   private def unwrapArray(array: java.util.List[AnyRef]): java.util.List[AnyRef] = {
184     array.asScala.map {
185       case o: java.util.Map[String, AnyRef] => JSONObject.toJSONString(o)
186       case a: java.util.List[AnyRef] => unwrapArray(a)
187       case p => p
188     }.asJava
189   }
