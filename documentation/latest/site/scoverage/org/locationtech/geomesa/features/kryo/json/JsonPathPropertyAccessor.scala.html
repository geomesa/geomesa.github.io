26 /**
27   * Access values from a json-type string field. Syntax must start with '$.'. The first part of the path
28   * selects the simple feature attribute, and the rest of the path selects within the json contained in
29   * that attribute.
30   *
31   * Note: this class is optimized for `KryoBufferSimpleFeature`s. It will work on standard simple features,
32   * but will incur a serialization cost.
33   */
34 trait JsonPathPropertyAccessor extends PropertyAccessor {
35 
36   import org.locationtech.geomesa.utils.geotools.RichAttributeDescriptors.RichAttributeDescriptor
37 
38   import scala.collection.JavaConverters._
39 
40   override def canHandle(obj: AnyRef, xpath: String, target: Class[_]): Boolean = {
41     val path = try { pathFor(xpath) } catch { case NonFatal(_) => Seq.empty }
42 
43     if (path.isEmpty) { false } else {
44       path.head match {
45         case PathAttribute(name: String, _) =>
46           val descriptor = obj match {
47             case s: SimpleFeature => s.getFeatureType.getDescriptor(name)
48             case s: SimpleFeatureType => s.getDescriptor(name)
49             case _ => null
50           }
51           descriptor != null && descriptor.getType.getBinding == classOf[String]
52 
53         case PathAttributeWildCard | PathDeepScan =>
54           val sft = obj match {
55             case s: SimpleFeature => s.getFeatureType
56             case s: SimpleFeatureType => s
57             case _ => null
58           }
59           sft != null && sft.getAttributeDescriptors.asScala.exists(_.getType.getBinding == classOf[String])
60 
61         case _ => false
62       }
63     }
64   }
65 
66   override def get[T](obj: AnyRef, xpath: String, target: Class[T]): T = {
67 
68     val path = pathFor(xpath)
69 
70     val result = obj match {
71       case s: KryoBufferSimpleFeature =>
72         val i = attribute(s.getFeatureType, path.head)
73         if (s.getFeatureType.getDescriptor(i).isJson()) {
74           s.getInput(i).map(KryoJsonSerialization.deserialize(_, path.tail)).orNull
75         } else {
76           parse(s.getAttribute(i).asInstanceOf[String], path.tail)
77         }
78 
79       case s: SimpleFeature =>
80         parse(s.getAttribute(attribute(s.getFeatureType, path.head)).asInstanceOf[String], path.tail)
81 
82       case s: SimpleFeatureType =>
83         // remove the json flag, so that transform serializations don't try to serialize
84         // json path results that aren't valid json objects or arrays
85         val descriptor = s.getDescriptor(attribute(s, path.head))
86         val builder = new AttributeTypeBuilder()
87         builder.init(descriptor)
88         val result = builder.buildDescriptor(descriptor.getLocalName)
89         result.getUserData.remove(SimpleFeatureTypes.AttributeOptions.OptJson)
90         result
91     }
92 
93     if (target == null) {
94       result.asInstanceOf[T]
95     } else {
96       FastConverter.convert(result, target)
97     }
98   }
99 
100   override def set[T](obj: Any, xpath: String, value: T, target: Class[T]): Unit = throw new NotImplementedError()
101 
102   private def attribute(sft: SimpleFeatureType, head: PathElement): Int = {
103     head match {
104       case PathAttribute(name: String, _) =>
105         sft.indexOf(name)
106 
107       case _ =>
108         // we know it will be a wildcard due to canHandle
109         // prioritize fields marked json over generic strings
110         // note: will only match first json attribute if more than 1
111         val i = sft.getAttributeDescriptors.asScala.indexWhere(_.isJson())
112         if (i != -1) { i } else {
113           sft.getAttributeDescriptors.asScala.indexWhere(_.getType.getBinding == classOf[String])
114         }
115     }
116   }
117 
118   private def parse(json: String, path: Seq[PathElement]): AnyRef = {
119     val list = JsonPath.using(pathConfig).parse(json).read[java.util.List[AnyRef]](JsonPathParser.print(path))
120     if (list == null || list.isEmpty) { null } else if (list.size == 1) { list.get(0) } else { list }
121   }
122 }

151     * @param path path to parse

160       val parsed = JsonPathParser.parse(path, report = false)
161       paths.put(path, new SoftReference(parsed))
162       parsed
163     }
