29 /**
30   * Access values from a json-type string field. Syntax must start with '$.'. The first part of the path
31   * selects the simple feature attribute, and the rest of the path selects within the json contained in
32   * that attribute.
33   *
34   * Note: this class is optimized for `KryoBufferSimpleFeature`s. It will work on standard simple features,
35   * but will incur a serialization cost.
36   */
37 trait JsonPathPropertyAccessor extends PropertyAccessor with LazyLogging {
38 
39   import org.locationtech.geomesa.utils.geotools.RichAttributeDescriptors.RichAttributeDescriptor
40 
41   import scala.collection.JavaConverters._
42 
43   override def canHandle(obj: AnyRef, xpath: String, target: Class[_]): Boolean = {
44     val path =
45       try {
46         JsonPathPropertyAccessor.paths.get(xpath)
47       } catch {
48         case NonFatal(e) =>
49           logger.warn(s"Error parsing path: $xpath", e)
50           JsonPath(Seq.empty, None)
51       }
52 
53     if (path.isEmpty) { false } else {
54       path.head match {
55         case PathAttribute(name: String, _) =>
56           val descriptor = obj match {
57             case s: SimpleFeature => s.getFeatureType.getDescriptor(name)
58             case s: SimpleFeatureType => s.getDescriptor(name)
59             case _ => null
60           }
61           descriptor != null && descriptor.getType.getBinding == classOf[String]
62 
63         case PathAttributeWildCard | PathDeepScan =>
64           val sft = obj match {
65             case s: SimpleFeature => s.getFeatureType
66             case s: SimpleFeatureType => s
67             case _ => null
68           }
69           sft != null && sft.getAttributeDescriptors.asScala.exists(_.getType.getBinding == classOf[String])
70 
71         case _ => false
72       }
73     }
74   }
75 
76   override def get[T](obj: AnyRef, xpath: String, target: Class[T]): T = {
77 
78     val path = JsonPathPropertyAccessor.paths.get(xpath)
79 
80     val result = obj match {
81       case s: KryoBufferSimpleFeature =>
82         val i = attribute(s.getFeatureType, path.head)
83         if (s.getFeatureType.getDescriptor(i).isJson()) {
84           s.getInput(i).map(KryoJsonSerialization.deserialize(_, path.tail)).orNull
85         } else {
86           JsonPathPropertyAccessor.evaluateJsonPath(s.getAttribute(i).asInstanceOf[String], path.tail)
87         }
88 
89       case s: SimpleFeature =>
90         JsonPathPropertyAccessor.evaluateJsonPath(s.getAttribute(attribute(s.getFeatureType, path.head)).asInstanceOf[String], path.tail)
91 
92       case s: SimpleFeatureType =>
93         // remove the json flag, so that transform serializations don't try to serialize
94         // json path results that aren't valid json objects or arrays
95         val descriptor = s.getDescriptor(attribute(s, path.head))
96         val builder = new AttributeTypeBuilder()
97         builder.init(descriptor)
98         val result = builder.buildDescriptor(descriptor.getLocalName)
99         result.getUserData.remove(SimpleFeatureTypes.AttributeOptions.OptJson)
100         result
101     }
102 
103     if (target == null) {
104       result.asInstanceOf[T]
105     } else {
106       FastConverter.convert(result, target)
107     }
108   }
109 
110   override def set[T](obj: Any, xpath: String, value: T, target: Class[T]): Unit = throw new UnsupportedOperationException()
111 
112   private def attribute(sft: SimpleFeatureType, head: PathElement): Int = {
113     head match {
114       case PathAttribute(name: String, _) =>
115         sft.indexOf(name)
116 
117       case _ =>
118         // we know it will be a wildcard due to canHandle
119         // prioritize fields marked json over generic strings
120         // note: will only match first json attribute if more than 1
121         val i = sft.getAttributeDescriptors.asScala.indexWhere(_.isJson())
122         if (i != -1) { i } else {
123           sft.getAttributeDescriptors.asScala.indexWhere(_.getType.getBinding == classOf[String])
124         }
125     }
126   }
127 }

162   /**
163    * Evaluate a json path using the json-path lib. This method is a fallback for our custom kryo-deserialization,
164    * and requires parsing the input into the json-path AST. The result has been modified to be consistent
165    * with our custom serialization (for better interop with geotools), i.e. json objects are serialized to json
166    * strings.
167    *
168    * @param json json doc
169    * @param path json path
170    * @return
171    */
172   private[json] def evaluateJsonPath(json: String, path: JsonPath): AnyRef = {
173     val parsed = com.jayway.jsonpath.JsonPath.using(pathConfig).parse(json)
174     val list = parsed.read[java.util.List[AnyRef]](JsonPathParser.print(path))
175     if (list == null || list.isEmpty) {
176       // special handling to return 0 for length functions without matches
177       path.function match {
178         case Some(PathFunction.LengthFunction) => Int.box(0)
179         case _ => null
180       }
181     } else {
182       val transformed = list.asScala.map {
183         case o: java.util.Map[String, AnyRef] => JSONObject.toJSONString(o)
184         case a: java.util.List[AnyRef] => unwrapArray(a)
185         case p => p
186       }
187       if (transformed.lengthCompare(1) == 0) { transformed.head } else { transformed.asJava }
188     }
189   }
190 
191   private def unwrapArray(array: java.util.List[AnyRef]): java.util.List[AnyRef] = {
192     array.asScala.map {
193       case o: java.util.Map[String, AnyRef] => JSONObject.toJSONString(o)
194       case a: java.util.List[AnyRef] => unwrapArray(a)
195       case p => p
196     }.asJava
197   }
