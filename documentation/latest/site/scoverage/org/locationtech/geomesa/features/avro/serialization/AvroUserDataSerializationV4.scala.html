26   @deprecated("does not match declared schema")
27   private[serialization] def serialize(out: Encoder, map: java.util.Map[_ <: AnyRef, _ <: AnyRef]): Unit = {
28     // may not be able to write all entries - must pre-filter to know correct count
29     val filtered = map.asScala.filter { case (key, value) =>
30       if (canSerialize(key)) {
31         true
32       } else {
33         logger.warn(s"Can't serialize Map entry ($key,$value) - it will be skipped.")
34         false
35       }
36     }
37 
38     out.writeArrayStart()
39     out.setItemCount(filtered.size)
40 
41     filtered.foreach { case (key, value) =>
42       out.startItem()
43       if (key == null) {
44         out.writeString(NullMarkerString)
45       } else {
46         out.writeString(key.getClass.getName)
47         write(out, key)
48       }
49       if (value == null) {
50         out.writeString(NullMarkerString)
51       } else {
52         out.writeString(value.getClass.getName)
53         write(out, value)
54       }
55     }
56 
57     out.writeArrayEnd()
58   }

71   private def deserializeWithSize(in: Decoder, size: Int, map: java.util.Map[AnyRef, AnyRef]): Unit = {
72     var remaining = size
73     while (remaining > 0) {
74       val keyClass = in.readString()
75       val key =
76         if (keyClass == NullMarkerString) { null }
77         else if (keyClass == "org.geotools.factory.Hints$Key") { HintKeySerialization.idToKey(in.readString()) }
78         else { read(in, Class.forName(keyClass)) }
79       val valueClass = in.readString()
80       val value = if (valueClass == NullMarkerString) { null } else { read(in, Class.forName(valueClass))}
81       map.put(key, value)
82       remaining -= 1
83       if (remaining == 0) {
84         remaining = in.arrayNext().toInt
85       }
86     }
87   }

89   private def write(out: Encoder, value: AnyRef): Unit = value match {
90     case v: String                 => out.writeString(v)
91     case v: java.lang.Integer      => out.writeInt(v)
92     case v: java.lang.Long         => out.writeLong(v)
93     case v: java.lang.Float        => out.writeFloat(v)
94     case v: java.lang.Double       => out.writeDouble(v)
95     case v: java.lang.Boolean      => out.writeBoolean(v)
96     case v: Date                   => out.writeLong(v.getTime)
97     case v: Array[Byte]            => out.writeBytes(v)
98     case v: Geometry               => out.writeBytes(WKBUtils.write(v))
99     case v: UUID                   => out.writeLong(v.getMostSignificantBits); out.writeLong(v.getLeastSignificantBits)
100     case v: java.util.List[AnyRef] => writeList(out, v)
101     case _ => throw new IllegalArgumentException(s"Unsupported value: $value (${value.getClass})")
102   }

104   /**
105    * Read a key or value. Strings will be interned, as we expect a lot of duplication in user data,
106    * i.e keys but also visibilities, which is the only user data we generally store
107    *
108    * @param in input
109    * @param clas class of the item to read
110    * @return
111    */
112   private def read(in: Decoder, clas: Class[_]): AnyRef = clas match {
113     case c if classOf[java.lang.String].isAssignableFrom(c)  => in.readString().intern()
114     case c if classOf[java.lang.Integer].isAssignableFrom(c) => Int.box(in.readInt())
115     case c if classOf[java.lang.Long].isAssignableFrom(c)    => Long.box(in.readLong())
116     case c if classOf[java.lang.Float].isAssignableFrom(c)   => Float.box(in.readFloat())
117     case c if classOf[java.lang.Double].isAssignableFrom(c)  => Double.box(in.readDouble())
118     case c if classOf[java.lang.Boolean].isAssignableFrom(c) => Boolean.box(in.readBoolean())
119     case c if classOf[java.util.Date].isAssignableFrom(c)    => new java.util.Date(in.readLong())
120     case c if classOf[Array[Byte]] == c                      => readBytes(in)
121     case c if classOf[Geometry].isAssignableFrom(c)          => WKBUtils.read(readBytes(in))
122     case c if classOf[UUID].isAssignableFrom(c)              => new UUID(in.readLong(), in.readLong())
123     case c if classOf[java.util.List[_]].isAssignableFrom(c) => readList(in)
124     case c if classOf[Hints.Key].isAssignableFrom(c)         => HintKeySerialization.idToKey(in.readString())
125     case _ => throw new IllegalArgumentException(s"Unsupported value class: $clas")
126   }

128   private def readBytes(in: Decoder): Array[Byte] = {
129     val buffer = in.readBytes(null)
130     val bytes = Array.ofDim[Byte](buffer.remaining())
131     buffer.get(bytes)
132     bytes
133   }

135   private def writeList(out: Encoder, list: java.util.List[AnyRef]): Unit = {
136     out.writeArrayStart()

150   private def readList(in: Decoder): java.util.List[AnyRef] = {
151     val size = in.readArrayStart().toInt
152     val list = new java.util.ArrayList[AnyRef](size)
153     var remaining = size
154     while (remaining > 0) {
155       val clas = in.readString()
156       val value = if (clas == NullMarkerString) { null } else { read(in, Class.forName(clas)) }
157       list.add(value)
158       remaining -= 1
159       if (remaining == 0) {
160         remaining = in.arrayNext().toInt
161       }
162     }
163     list
164   }
