34   /**
35     * Serialize a geometry
36     *
37     * For explanation of precisions, see `org.locationtech.geomesa.utils.geometry.GeometryPrecision`
38     *
39     * @param out output
40     * @param geometry geometry
41     * @param precision precision for encoding x, y, z, m
42     */
43   def serialize(out: T, geometry: Geometry, precision: TwkbPrecision = TwkbPrecision()): Unit = {
44     if (geometry == null) {
45       out.writeByte(ZeroByte)
46     } else {
47       // choose our state to correspond with the dimensions in the geometry
48       implicit val state: DeltaState = {
49         // note that we only check the first coordinate - if a geometry is written with different
50         // dimensions in each coordinate, some information may be lost
51         val coord = geometry.getCoordinate
52         if (coord == null) {
53           new XYState(precision.xy)
54         } else {
55           // check for dimensions - use NaN != NaN to verify presence of coordinates
56           val hasZ = !java.lang.Double.isNaN(coord.getZ)
57           val hasM = !java.lang.Double.isNaN(coord.getM)
58           if (hasZ) {
59             if (hasM) {
60               new XYZMState(precision.xy, precision.z, precision.m)
61             } else {
62               new XYZState(precision.xy, precision.z)
63             }
64           } else if (hasM) {
65             new XYMState(precision.xy, precision.m)
66           } else {
67             new XYState(precision.xy)
68           }
69         }
70       }
71 
72       geometry match {
73         case g: Point =>
74           if (g.isEmpty) {
75             state.writeMetadata(out, TwkbPoint, empty = true, bbox = false)
76           } else {
77             state.writeMetadata(out, TwkbPoint, empty = false, bbox = false)
78             state.writeCoordinate(out, g.getCoordinate)
79           }
80 
81         case g: LineString =>
82           if (g.isEmpty) {
83             state.writeMetadata(out, TwkbLineString, empty = true, bbox = false)
84           } else {
85             state.writeMetadata(out, TwkbLineString, empty = false, bbox = true)
86             state.writeBoundingBox(out, g)
87           }
88           writeLineString(out, g)
89 
90         case g: Polygon =>
91           if (g.isEmpty) {
92             state.writeMetadata(out, TwkbPolygon, empty = true, bbox = false)
93           } else {
94             state.writeMetadata(out, TwkbPolygon, empty = false, bbox = true)
95             state.writeBoundingBox(out, g)
96           }
97           writePolygon(out, g)
98 
99         case g: MultiPoint =>
100           if (g.isEmpty) {
101             state.writeMetadata(out, TwkbMultiPoint, empty = true, bbox = false)
102           } else {
103             state.writeMetadata(out, TwkbMultiPoint, empty = false, bbox = true)
104             state.writeBoundingBox(out, g)
105           }
106           writeMultiPoint(out, g)
107 
108         case g: MultiLineString =>
109           if (g.isEmpty) {
110             state.writeMetadata(out, TwkbMultiLineString, empty = true, bbox = false)
111           } else {
112             state.writeMetadata(out, TwkbMultiLineString, empty = false, bbox = true)
113             state.writeBoundingBox(out, g)
114           }
115           writeMultiLineString(out, g)
116 
117         case g: MultiPolygon =>
118           if (g.isEmpty) {
119             state.writeMetadata(out, TwkbMultiPolygon, empty = true, bbox = false)
120           } else {
121             state.writeMetadata(out, TwkbMultiPolygon, empty = false, bbox = true)
122             state.writeBoundingBox(out, g)
123           }
124           writeMultiPolygon(out, g)
125 
126         case g: GeometryCollection =>
127           if (g.isEmpty) {
128             state.writeMetadata(out, TwkbCollection, empty = true, bbox = false)
129           } else {
130             state.writeMetadata(out, TwkbCollection, empty = false, bbox = true)
131             state.writeBoundingBox(out, g)
132           }
133           writeCollection(out, g)
134       }
135     }
136   }

150   private def deserialize(in: V, nesting: Int): Geometry = {
151     val precisionAndType = in.readByte()
152     if (precisionAndType == ZeroByte) {
153       null
154     } else if (precisionAndType == NOT_NULL_BYTE) {
155       // TODO this overlaps with twkb point type with precision 0
156       deserializeWkb(in)
157     } else {
158       // first byte contains the geometry type in the first 4 bits and the x-y precision in the second 4 bits
159       val geomType = (precisionAndType & 0x0F).toByte
160       val precision = VarIntEncoding.zigzagDecode((precisionAndType & 0xF0) >>> 4)
161 
162       // second byte contains flags for optional elements
163       val flags = in.readByte()
164       val hasBoundingBox = (flags & BoundingBoxFlag) != 0
165       val hasExtendedDims = (flags & ExtendedDimsFlag) != 0
166       val isEmpty = (flags & EmptyFlag) != 0
167 
168       // extended dims indicates the presence of z and/or m
169       // we create our state tracker based on the dimensions that are present
170       implicit val state: DeltaState = if (hasExtendedDims) {
171         // z and m precisions are indicated in the next byte, where (from right to left):
172         //   bit 0 indicates presence of z dimension
173         //   bit 1 indicates presence of m dimension
174         //   bits 2-5 indicate z precision
175         //   bits 6-8 indicate m precision
176         val extendedDims = in.readByte()
177         if ((extendedDims & 0x01) != 0) { // indicates z dimension
178           if ((extendedDims & 0x02) != 0) { // indicates m dimension
179             new XYZMState(precision, (extendedDims & 0x1C) >> 2, (extendedDims & 0xE0) >>> 5)
180           } else {
181             new XYZState(precision, (extendedDims & 0x1C) >> 2)
182           }
183         } else if ((extendedDims & 0x02) != 0) {  // indicates m dimension
184           new XYMState(precision, (extendedDims & 0xE0) >>> 5)
185         } else {
186           // not sure why anyone would indicate extended dims but set them all false...
187           new XYState(precision)
188         }
189       } else {
190         new XYState(precision)
191       }
192 
193       // size is the length of the remainder of the geometry, after the size attribute
194       // we don't currently use size - parsing will fail if size is actually present
195 
196       // val hasSize = (flags & FlagBytes.SizeFlag) != 0
197       // if (hasSize) {
198       //   val size = readUnsignedVarInt(in)
199       // }
200 
201       // bounding box is not currently used, but we write it in anticipation of future filter optimizations
202       if (hasBoundingBox) {
203         state.skipBoundingBox(in)
204       }
205 
206       // children geometries can be written with an id list
207       // we don't currently use ids - parsing will fail if ids are actually present
208       // val hasIds = (flags & FlagBytes.IdsFlag) != 0
209 
210       geomType match {
211         case TwkbPoint => factory.createPoint(if (isEmpty) { null } else { csFactory.create(readPointArray(in, 1)) })
212         case TwkbLineString      => readLineString(in)
213         case TwkbPolygon         => readPolygon(in)
214         case TwkbMultiPoint      => readMultiPoint(in)
215         case TwkbMultiLineString => readMultiLineString(in)
216         case TwkbMultiPolygon    => readMultiPolygon(in)
217         case TwkbCollection      => readCollection(in, nesting)
218         case _ => throw new IllegalArgumentException(s"Invalid TWKB geometry type $geomType")
219       }
220     }
221   }

229   private def writePolygon(out: T, g: Polygon)(implicit state: DeltaState): Unit = {
230     if (g.isEmpty) {
231       writeUnsignedVarInt(out, 0)
232     } else {
233       val numRings = g.getNumInteriorRing
234       writeUnsignedVarInt(out, numRings + 1) // include exterior ring in count
235       // note: don't write final point for each ring, as they should duplicate the first point
236       var ring = g.getExteriorRing.getCoordinateSequence
237       writePointArray(out, ring, ring.size() - 1)
238       var j = 0
239       while (j < numRings) {
240         ring = g.getInteriorRingN(j).getCoordinateSequence
241         writePointArray(out, ring, ring.size() - 1)
242         j += 1
243       }
244     }
245   }

247   private def readPolygon(in: V)(implicit state: DeltaState): Polygon = {
248     val numRings = readUnsignedVarInt(in)
249     if (numRings == 0) {
250       factory.createPolygon(null, null)
251     } else if (numRings > maxLength) {
252       throw new IllegalArgumentException(
253         s"Attempting to deserialize a polygon of size $numRings " +
254             s"with '${GeometryLengthThreshold.property}' = $maxLength'")
255     } else {
256       val exteriorRing = readLinearRing(in, readUnsignedVarInt(in))
257       val interiorRings = Array.ofDim[LinearRing](numRings - 1)
258       var i = 1
259       while (i < numRings) {
260         interiorRings(i - 1) = readLinearRing(in, readUnsignedVarInt(in))
261         i += 1
262       }
263       factory.createPolygon(exteriorRing, interiorRings)
264     }
265   }

267   private def writeMultiPoint(out: T, g: MultiPoint)(implicit state: DeltaState): Unit = {
268     val length = g.getNumPoints
269     writeUnsignedVarInt(out, length)
270     var i = 0
271     while (i < length) {
272       state.writeCoordinate(out, g.getGeometryN(i).asInstanceOf[Point].getCoordinate)
273       i += 1
274     }

277   private def readMultiPoint(in: V)(implicit state: DeltaState): MultiPoint = {
278     val numPoints = readUnsignedVarInt(in)
279     if (numPoints == 0) { factory.createMultiPoint(null: CoordinateSequence) } else {
280       // note: id list would go here, with one ID per point
281       factory.createMultiPoint(readPointArray(in, numPoints).map(factory.createPoint))
282     }

285   private def writeMultiLineString(out: T, g: MultiLineString)(implicit state: DeltaState): Unit = {
286     val length = g.getNumGeometries
287     writeUnsignedVarInt(out, length)
288     var i = 0

290       val line = g.getGeometryN(i).asInstanceOf[LineString].getCoordinateSequence
291       writePointArray(out, line, line.size())
292       i += 1
293     }

356   private def readCollection(in: V, nesting: Int): GeometryCollection = {
357     if (nesting > maxNesting) {
358       throw new IllegalArgumentException(s"Detected recursive deserialization loop of $maxNesting")
359     }
360     val numGeoms = readUnsignedVarInt(in)
361     if (numGeoms == 0) {
362       factory.createGeometryCollection(null)
363     } else if (numGeoms > maxLength) {
364       throw new IllegalArgumentException(
365         s"Attempting to deserialize a geometry collection of size $numGeoms " +
366             s"with '${GeometryLengthThreshold.property}' = $maxLength'")
367     } else {
368       // note: id list would go here, with one ID per sub geometry
369       val geoms = Array.ofDim[Geometry](numGeoms)
370       var i = 0
371       while (i < numGeoms) {
372         geoms(i) = deserialize(in, nesting + 1)
373         i += 1
374       }
375       factory.createGeometryCollection(geoms)
376     }
377   }

379   private def writePointArray(out: T, coords: CoordinateSequence, length: Int)(implicit state: DeltaState): Unit = {
380     writeUnsignedVarInt(out, length)
381     var i = 0
382     while (i < length) {
383       state.writeCoordinate(out, coords.getCoordinate(i))
384       i += 1
385     }
386   }

388   private def readPointArray(in: V, length: Int)(implicit state: DeltaState): Array[Coordinate] = {
389     if (length > maxLength) {
390       throw new IllegalArgumentException(
391         s"Attempting to deserialize a point array of size $length with " +
392             s"'${GeometryLengthThreshold.property}' = $maxLength'")
393     }
394     val result = Array.ofDim[Coordinate](length)
395     var i = 0
396     while (i < length) {
397       result(i) = state.readCoordinate(in)
398       i += 1
399     }
400     result
401   }

403   private def readLinearRing(in: V, length: Int)(implicit state: DeltaState): LinearRing = {
404     if (length == 0) {
405       factory.createLinearRing(null: CoordinateSequence)
406     } else if (length > maxLength) {
407       throw new IllegalArgumentException(
408         s"Attempting to deserialize a linear ring of size $length " +
409             s"with '${GeometryLengthThreshold.property}' = $maxLength'")
410     } else {
411       val result = Array.ofDim[Coordinate](length + 1)
412       var i = 0
413       while (i < length) {
414         result(i) = state.readCoordinate(in)
415         i += 1
416       }
417       // linear rings should not store the final, duplicate point, but still need it for the geometry
418       result(length) = result(0)
419       factory.createLinearRing(csFactory.create(result))
420     }
421   }

423   /**
424     * TWKB only reads and writes the delta from one coordinate to the next, which generally saves space
425     * over absolute values. This trait tracks the values used for delta calculations over a single
426     * read or write operation
427     */
428   private sealed trait DeltaState {
429 
430     protected def readCoordinateState(in: V): Unit
431     protected def createCoordinate(): Coordinate
432 
433     /**
434       * Write metadata, which includes the geometry and precision byte, the flag byte, and optionally
435       * an extended precision byte
436       *
437       * @param out output
438       * @param geometryType geometry type
439       * @param empty indicate that the geometry is empty
440       * @param bbox indicate that a bbox will be written
441       */
442     def writeMetadata(out: T, geometryType: Byte, empty: Boolean, bbox: Boolean): Unit
443 
444     /**
445       * Writes out a bounding box. Each dimension stores a min value and a delta to the max value
446       *
447       * @param out output
448       * @param geometry geometry
449       * @param bounds bounds operation
450       * @tparam G geometry type
451       */
452     def writeBoundingBox[G <: Geometry](out: T, geometry: G)(implicit bounds: DimensionalBounds[G]): Unit
453 
454     /**
455       * Skips over a bounding box. We don't currently use the bounding box when reading
456       *
457       * @param in in
458       */
459     def skipBoundingBox(in: V): Unit
460 
461     /**
462       * Write a coordinate
463       *
464       * @param out output
465       * @param coordinate coordinate
466       */
467     def writeCoordinate(out: T, coordinate: Coordinate): Unit
468 
469     /**
470       * Read a coordinate
471       *
472       * @param in input
473       * @return
474       */
475     def readCoordinate(in: V): Coordinate = {
476       readCoordinateState(in)
477       createCoordinate()
478     }
479 
480     /**
481       * Reset the state back to its original state, suitable for re-use
482       */
483     def reset(): Unit
484   }

486   private class XYState(precision: Int) extends DeltaState {
487 
488     protected val p: Double = math.pow(10, precision)
489     protected var x: Int = 0
490     protected var y: Int = 0
491 
492     protected val boundingBoxFlag: Byte = BoundingBoxFlag
493     protected val emptyFlag: Byte = EmptyFlag
494     protected val dimensionsFlag: Byte = ZeroByte
495 
496     override protected def readCoordinateState(in: V): Unit = {
497       x = x + readVarInt(in)
498       y = y + readVarInt(in)
499     }
500 
501     override protected def createCoordinate(): Coordinate = new CoordinateXY(x / p, y / p)
502 
503     override def writeMetadata(out: T, geometryType: Byte, empty: Boolean, bbox: Boolean): Unit = {
504       // write the geometry type and the main precision
505       out.writeByte(((VarIntEncoding.zigzagEncode(precision) << 4) | geometryType).toByte)
506       // write the flag byte
507       out.writeByte(if (bbox) { boundingBoxFlag } else if (empty) { emptyFlag } else { dimensionsFlag })
508     }
509 
510     override def writeCoordinate(out: T, coordinate: Coordinate): Unit = {
511       val cx = math.round(coordinate.x * p).toInt
512       val cy = math.round(coordinate.y * p).toInt
513       writeVarInt(out, cx - x)
514       writeVarInt(out, cy - y)
515       x = cx
516       y = cy
517     }
518 
519     override def writeBoundingBox[G <: Geometry](out: T, geometry: G)(implicit bounds: DimensionalBounds[G]): Unit = {
520       val (minX, maxX) = bounds.x(geometry)
521       val intX = math.round(minX * p).toInt
522       writeVarInt(out, intX)
523       writeVarInt(out, math.round(maxX * p).toInt - intX)
524       val (minY, maxY) = bounds.y(geometry)
525       val intY = math.round(minY * p).toInt
526       writeVarInt(out, intY)
527       writeVarInt(out, math.round(maxY * p).toInt - intY)
528     }
529 
530     override def skipBoundingBox(in: V): Unit = {
531       skipVarInt(in)
532       skipVarInt(in)
533       skipVarInt(in)
534       skipVarInt(in)
535     }
536 
537     override def reset(): Unit = {
538       x = 0
539       y = 0
540     }
541   }

544 
545     protected def extendedDims: Byte

548     override protected val emptyFlag: Byte = (ExtendedDimsFlag | EmptyFlag).toByte
549     override protected val dimensionsFlag: Byte = ExtendedDimsFlag

558   private class XYZState(precision: Int, zPrecision: Int) extends ExtendedState(precision) {
559 
560     protected val pz: Double = math.pow(10, zPrecision)
561     protected var z: Int = 0
562 
563     // sets bits for z dim, and its precisions
564     override protected val extendedDims: Byte = (0x01 | ((zPrecision & 0x03) << 2)).toByte
565 
566     override protected def readCoordinateState(in: V): Unit = {
567       super.readCoordinateState(in)
568       z = z + readVarInt(in)
569     }
570 
571     override protected def createCoordinate(): Coordinate = new Coordinate(x / p, y / p, z / pz)
572 
573     override def writeBoundingBox[G <: Geometry](out: T, geometry: G)(implicit bounds: DimensionalBounds[G]): Unit = {
574       super.writeBoundingBox(out, geometry)
575       val (minZ, maxZ) = bounds.z(geometry)
576       val intZ = math.round(minZ * pz).toInt
577       writeVarInt(out, intZ)
578       writeVarInt(out, math.round(maxZ * pz).toInt - intZ)
579     }
580 
581     override def writeCoordinate(out: T, coordinate: Coordinate): Unit = {
582       super.writeCoordinate(out, coordinate)
583       val cz = math.round(coordinate.getZ * pz).toInt
584       writeVarInt(out, cz - z)
585       z = cz
586     }
587 
588     override def skipBoundingBox(in: V): Unit = {
589       super.skipBoundingBox(in)
590       skipVarInt(in)
591       skipVarInt(in)
592     }
593 
594     override def reset(): Unit = {
595       super.reset()
596       z = 0
597     }
598   }

642   private class XYZMState(precision: Int, zPrecision: Int, mPrecision: Int) extends XYZState(precision, zPrecision) {
643 
644     protected val pm: Double = math.pow(10, mPrecision)
645     protected var m: Int = 0
646
