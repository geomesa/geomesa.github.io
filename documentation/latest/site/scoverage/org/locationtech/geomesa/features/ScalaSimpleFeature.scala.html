88   /**
89     * Copy the feature with a new feature type. Attributes will be copied by name, and converted
90     * as necessary. As compared to `copy`, the new feature type does not have to have a compatible schema.
91     *
92     * If the feature already has the desired feature type, it will be returned as-is and not copied.
93     *
94     * This is a shallow copy, in that the attributes and user data values will be shared between the two features
95     *
96     * @param sft new feature type
97     * @param in feature to copy
98     * @return
99     */
100   def retype(sft: SimpleFeatureType, in: SimpleFeature): SimpleFeature = {
101     if (sft == in.getFeatureType) { in } else {
102       val out = new ScalaSimpleFeature(sft, in.getID)
103       var i = 0
104       sft.getAttributeDescriptors.asScala.foreach { d =>
105         out.setAttribute(i, in.getAttribute(d.getLocalName))
106         i += 1
107       }
108       out.getUserData.putAll(in.getUserData)
109       out
110     }
111   }

131   def create(sft: SimpleFeatureType, id: String, values: Any*): ScalaSimpleFeature = {
132     val sf = new ScalaSimpleFeature(sft, id)
133     var i = 0
134     while (i < values.length) {

136       i += 1
137     }

277   trait LazySimpleFeature extends AbstractSimpleFeature {
278 
279     // we synchronize on the low-level words, in order to minimize contention
280     protected val bits: WordBitSet = WordBitSet(getFeatureType.getAttributeCount + 1)
281     protected var attributes: Array[AnyRef] = _
282     protected var userData: java.util.Map[AnyRef, AnyRef] = _

285     protected var reader: LazyAttributeReader = _
286     protected var userDataReader: LazyUserDataReader = _

288     override def getAttribute(index: Int): AnyRef = {
289       if (reader != null) {
290         val word = bits.word(index)
291         word.synchronized {
292           if (word.add(index)) {
293             bits.synchronized {
294               if (attributes == null) {

296               }
297             }

299             if (count.decrementAndGet() == 0) {
300               // once all attributes have been read, dereference any backing resources so that they can be gc'd
301               reader = null

303             }
304           }
305         }

308     }
309 
310     override def getUserData: java.util.Map[AnyRef, AnyRef] = {
311       if (userDataReader != null) {

313           if (userData == null) {
314             userData = userDataReader.read()

317         }

320     }
