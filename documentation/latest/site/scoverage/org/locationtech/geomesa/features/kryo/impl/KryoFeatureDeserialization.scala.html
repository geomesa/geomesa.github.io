47 object KryoFeatureDeserialization extends LazyLogging {
48 
49   private val inputBytes   = new SoftThreadLocal[Input]()
50   private val inputStreams = new SoftThreadLocal[Input]()
51   private val readers      = new ThreadLocalCache[String, Array[KryoAttributeReader]](SerializerCacheExpiry)
52 
53   def getInput(bytes: Array[Byte], offset: Int, count: Int): Input = {
54     val in = inputBytes.getOrElseUpdate(new NonMutatingInput())
55     in.setBuffer(bytes, offset, count)
56     in
57   }
58 
59   def getInput(stream: InputStream): Input = {
60     val in = inputStreams.getOrElseUpdate(new NonMutatingInput(Array.ofDim(1024)))
61     in.setInputStream(stream)
62     in
63   }
64 
65   def getReaders(key: String, sft: SimpleFeatureType): Array[KryoAttributeReader] = {
66     readers.getOrElseUpdate(key, sft.getAttributeDescriptors.toArray.map {
67       case ad: AttributeDescriptor => reader(ObjectType.selectType(ad))
68     })
69   }
70 
71   private [geomesa] def reader(bindings: Seq[ObjectType]): KryoAttributeReader = {
72     bindings.head match {
73       case ObjectType.STRING   => if (bindings.last == ObjectType.JSON) { KryoJsonReader } else { KryoStringReader }
74       case ObjectType.INT      => KryoIntReader
75       case ObjectType.LONG     => KryoLongReader
76       case ObjectType.FLOAT    => KryoFloatReader
77       case ObjectType.DOUBLE   => KryoDoubleReader
78       case ObjectType.DATE     => KryoDateReader
79       case ObjectType.GEOMETRY => KryoGeometryReader
80       case ObjectType.BYTES    => KryoBytesReader
81       case ObjectType.UUID     => KryoUuidReader
82       case ObjectType.BOOLEAN  => KryoBooleanReader
83       case ObjectType.LIST     => KryoListReader(reader(bindings.drop(1)))
84       case ObjectType.MAP      => KryoMapReader(reader(bindings.slice(1, 2)), reader(bindings.drop(2)))
85 
86       case b => throw new UnsupportedOperationException(s"Unexpected attribute type binding: $b")
87     }
88   }

160   case class KryoListReader(elements: KryoAttributeReader) extends KryoAttributeReader {
161     override def apply(input: Input): java.util.List[AnyRef] = try {
162       val size = input.readInt(true)
163       val list = new java.util.ArrayList[AnyRef](size)
164       var index = 0
165       while (index < size) {
166         list.add(elements.apply(input))
167         index += 1
168       }
169       list
170     } catch {
171       case NonFatal(e) => logger.error("Error reading serialized kryo bytes:", e); null
172     }
173   }
174 
175   case class KryoMapReader(keys: KryoAttributeReader, values: KryoAttributeReader) extends KryoAttributeReader {
176     override def apply(input: Input): java.util.Map[AnyRef, AnyRef] = try {
177       val size = input.readInt(true)
178       val map = new java.util.HashMap[AnyRef, AnyRef](size)
179       var index = 0
180       while (index < size) {
181         map.put(keys.apply(input), values.apply(input))
182         index += 1
183       }
184       map
185     } catch {
186       case NonFatal(e) => logger.error("Error reading serialized kryo bytes:", e); null
187     }
188   }
