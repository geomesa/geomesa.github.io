17 /**
18  * Config used for serializing simple features
19  *
20  * @param version serialization version, see org.locationtech.geomesa.features.avro.SerializationVersions
21  * @param schema the avro schema
22  * @param fid the feature id field, if present
23  * @param fields fields for each attribute in the feature
24  * @param userData the user data field, if present
25  */
26 case class AvroSerialization(
27     version: Int,
28     schema: Schema,
29     fid: Option[AvroField[String]],
30     fields: Seq[AvroField[AnyRef]],
31     userData: Option[AvroField[java.util.Map[AnyRef, AnyRef]]]
32   )
33 
34 object AvroSerialization {
35 
36   import org.locationtech.geomesa.features.avro.serialization.AvroField.{FidField, UserDataField}
37   import org.locationtech.geomesa.utils.geotools.RichAttributeDescriptors.RichAttributeDescriptor
38 
39   import scala.collection.JavaConverters._
40 
41   // array of name encoders, indexed by the encoder version (minus 1)
42   private val nameEncoders = Array.tabulate(SerializationVersions.MaxVersion)(i => new FieldNameEncoder(i + 1))
43 
44   /**
45    * Create a serialization config
46    *
47    * @param sft simple feature type
48    * @param opts options
49    * @return
50    */
51   def apply(sft: SimpleFeatureType, opts: Set[SerializationOption]): AvroSerialization = {
52     val version: Int =
53       if (opts.useNativeCollections && sft.getAttributeDescriptors.asScala.exists(d => d.isList || d.isMap)) {
54         SerializationVersions.NativeCollectionVersion
55       } else {
56         SerializationVersions.DefaultVersion
57       }
58 
59     val fields = sft.getAttributeDescriptors.asScala.map(AvroField.apply(_).withVersion(version)).toSeq
60 
61     val fid = if (opts.withoutId) { None } else { Some(FidField.withVersion(version)) }
62     val userData = if (opts.withUserData) { Some(UserDataField.withVersion(version)) } else { None }
63 
64     val schema = AvroSerialization.schema(sft, version, fields, fid.isDefined, userData.isDefined)
65 
66     AvroSerialization(version, schema, fid, fields, userData)
67   }
68 
69   /**
70    * Calculate the schema corresponding to a given feature type
71    *
72    * @param sft feature type
73    * @param version serialization version
74    * @param fields field definitions
75    * @param includeFid include feature id
76    * @param includeUserData include user data
77    * @return
78    */
79   def schema(
80       sft: SimpleFeatureType,
81       version: Int,
82       fields: Seq[AvroField[AnyRef]],
83       includeFid: Boolean,
84       includeUserData: Boolean): Schema = {
85 
86     import AvroField.{FidField, UserDataField, VersionField}
87 
88     require(version > 0 && version <= SerializationVersions.MaxVersion,
89       s"Unknown version $version - valid versions are 1 to ${SerializationVersions.MaxVersion}")
90     require(fields.length == sft.getAttributeCount, "Mismatch between fields and feature type")
91 
92     val nameEncoder = nameEncoders(version - 1)
93     val builder =
94       SchemaBuilder.record(nameEncoder.encode(sft.getTypeName))
95           .namespace(Option(sft.getName.getNamespaceURI).getOrElse(AvroNamespace))
96           .fields
97 
98     builder.name(VersionField.name).`type`(VersionField.schema).withDefault(version)
99     if (includeFid) {
100       builder.name(FidField.name).`type`(FidField.schema).noDefault
101     }
102 
103     val types = fields.map(_.schema).iterator
104     sft.getAttributeDescriptors.asScala.foreach { d =>
105       builder.name(nameEncoder.encode(d.getLocalName)).`type`(types.next).noDefault()
106     }
107 
108     if (includeUserData) {
109       builder.name(UserDataField.name).`type`(UserDataField.schema).noDefault()
110     }
111 
112     builder.endRecord()
113   }
114 
115   /**
116    * For a given schema, checks if collections are encoded natively or as binary
117    *
118    * @param schema schema
119    * @return
120    */
121   def usesNativeCollections(schema: Schema): Boolean = {
122     Option(schema.getField(VersionField.name))
123         .exists(_.defaultVal() == SerializationVersions.NativeCollectionVersion)
124   }
125 }
