255   /**
256     * Plans the query
257     *
258     * @param filter filter strategy
259     * @param explain explainer
260     * @return
261     */
262   def getQueryStrategy(filter: FilterStrategy, explain: Explainer = ExplainNull): QueryStrategy = {
263     val sharing = keySpace.sharing
264     val indexValues = filter.primary.map(keySpace.getIndexValues(_, explain))
265 
266     val useFullFilter = keySpace.useFullFilter(indexValues, Some(ds.config), filter.hints)
267     val ecql = if (useFullFilter) { filter.filter } else { filter.secondary }
268 
269     indexValues match {
270       case None =>
271         if (filter.filter.exists(_.isInstanceOf[ExcludeFilter])) {
272           QueryStrategy(filter, Seq.empty, Seq.empty, Seq.empty, ecql, indexValues)
273         } else {
274           val byteRanges = Seq(BoundedByteRange(sharing, ByteArrays.rowFollowingPrefix(sharing)))
275           QueryStrategy(filter, byteRanges, FullTableKeyRange, Seq.empty, ecql, indexValues)
276         }
277 
278       case Some(values) =>
279         val keyRanges = keySpace.getRanges(values).toSeq
280         val bytes = keySpace.getRangeBytes(keyRanges.iterator, tieredKeySpace.isDefined).toSeq
281         val tier = tieredKeySpace.orNull.asInstanceOf[IndexKeySpace[Any, Any]]
282         val secondary = filter.secondary.orNull
283         lazy val tiers = {
284           val multiplier = math.max(1, bytes.count(_.isInstanceOf[SingleRowByteRange]))
285           tier.getRangeBytes(tier.getRanges(tier.getIndexValues(secondary, explain), multiplier)).toSeq
286         }
287 
288         if (tier == null) {
289           QueryStrategy(filter, bytes, keyRanges, Seq.empty, ecql, indexValues)
290         } else if (secondary == null || tiers.exists(_.isInstanceOf[UnboundedByteRange])) {
291           val byteRanges = keySpace.getRangeBytes(keyRanges.iterator, tier = true).map {
292             case BoundedByteRange(lo, hi)      => BoundedByteRange(lo, ByteArrays.concat(hi, ByteRange.UnboundedUpperRange))
293             case SingleRowByteRange(row)       => BoundedByteRange(row, ByteArrays.concat(row, ByteRange.UnboundedUpperRange))
294             case UpperBoundedByteRange(lo, hi) => BoundedByteRange(lo, ByteArrays.concat(hi, ByteRange.UnboundedUpperRange))
295             case LowerBoundedByteRange(lo, hi) => BoundedByteRange(lo, hi)
296             case UnboundedByteRange(lo, hi)    => BoundedByteRange(lo, hi)
297             case r => throw new IllegalArgumentException(s"Unexpected range type $r")
298           }.toSeq
299           QueryStrategy(filter, byteRanges, keyRanges, Seq.empty, ecql, indexValues)
300         } else if (tiers.isEmpty) {
301           // disjoint secondary filter
302           QueryStrategy(filter, Seq.empty, Seq.empty, Seq.empty, ecql, indexValues)
303         } else {
304           lazy val minTier = ByteRange.min(tiers)
305           lazy val maxTier = ByteRange.max(tiers)
306 
307           val byteRanges = bytes.flatMap {
308             case SingleRowByteRange(row) =>
309               // single row - we can use all the tiered ranges appended to the end
310               tiers.map {
311                 case BoundedByteRange(lo, hi) => BoundedByteRange(ByteArrays.concat(row, lo), ByteArrays.concat(row, hi))
312                 case SingleRowByteRange(trow) => SingleRowByteRange(ByteArrays.concat(row, trow))
313               }
314 
315             case BoundedByteRange(lo, hi) =>
316               // bounded ranges - we can use the min/max tier on the endpoints
317               Iterator.single(BoundedByteRange(ByteArrays.concat(lo, minTier), ByteArrays.concat(hi, maxTier)))
318 
319             case LowerBoundedByteRange(lo, hi) =>
320               // we can't use the upper tier
321               Iterator.single(BoundedByteRange(ByteArrays.concat(lo, minTier), hi))
322 
323             case UpperBoundedByteRange(lo, hi) =>
324               // we can't use the lower tier
325               Iterator.single(BoundedByteRange(lo, ByteArrays.concat(hi, maxTier)))
326 
327             case UnboundedByteRange(lo, hi) =>
328               // we can't use either side of the tiers
329               Iterator.single(BoundedByteRange(lo, hi))
330 
331             case r =>
332               throw new IllegalArgumentException(s"Unexpected range type $r")
333           }
334 
335           QueryStrategy(filter, byteRanges, keyRanges, tiers, ecql, indexValues)
336         }
337     }
338   }

340   /**
341    * Check that full table scans are allowed
342    *
343    * @param strategy strategy
344    * @param blockByDefault default behavior in case no system props are configured
345    * @return
346    */
347   private[index] def checkQueryBlock(strategy: QueryStrategy, blockByDefault: Boolean): Option[IllegalArgumentException] = {
348     if (!isFullTableScan(strategy)) {
349       return None
350     }
351     lazy val filterString = strategy.filter.filter.fold("INCLUDE")(FilterHelper.toString)
352     val block = QueryProperties.blockFullTableScansForFeatureType(sft.getTypeName)
353     if (block.getOrElse(blockByDefault)) {
354       Some(new IllegalArgumentException(s"Full-table scans are disabled. Query being stopped for ${sft.getTypeName}: $filterString"))
355     } else {
356       lazy val warning = s"Running full table scan for schema '${sft.getTypeName}' with filter: $filterString"
357       if (block.isDefined) {
358         logger.info(warning) // turn down log level if full table scans are explicitly allowed
359       } else {
360         logger.warn(warning)
361       }
362       None
363     }
364   }

377   /**
378     * Creates a valid, unique string for the underlying table
379     *
380     * @param partition partition
381     * @param limit limit on the length of a table name in the underlying database
382     * @return
383     */
384   private def generateTableName(partition: Option[String] = None, limit: Option[Int] = None): String = {
385     import StringSerialization.alphaNumericSafeString
386     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
387 
388     val namespace = sft.getTablePrefix(name).getOrElse(ds.config.catalog)
389     val prefix = (namespace +: Seq(sft.getTypeName, name).map(alphaNumericSafeString)).mkString("_")
390     val suffix = s"v$version${partition.map(p => s"_$p").getOrElse("")}"
391 
392     def build(attrs: Seq[String]): String = (prefix +: attrs :+ suffix).mkString("_")
393 
394     val full = build(attributes.map(alphaNumericSafeString))
395 
396     limit match {
397       case Some(lim) if full.lengthCompare(lim) > 0 =>
398         // try using the attribute numbers instead
399         val nums = build(attributes.map(a => s"${sft.indexOf(a)}"))
400         if (nums.lengthCompare(lim) <= 0) { nums } else {
401           logger.warn(s"Table name length exceeds configured limit ($lim), falling back to UUID: $full")
402           IndexAdapter.truncateTableName(full, lim)
403         }
404 
405       case _ => full
406     }
407   }

503     /**
504       * Return the id from the given row
505       *
506       * @param row row value as bytes
507       * @param offset start of the row in the byte array
508       * @param length length of the row in the byte array
509       * @param feature a simple feature used to cache the feature id, optional (may be null)

511       */

515       * Return the start index of the id in the row
516       *
517       * @param row row value as bytes
518       * @param offset start of the row in the byte array
519       * @param length length of the row in the byte array
520       * @return the start of the id in the row, relative to the row offset
521       */

525   object IdFromRow {
526 
527     def apply(
528         sft: SimpleFeatureType,
529         keySpace: IndexKeySpace[_, _],
530         tieredKeySpace: Option[IndexKeySpace[_, _]]): IdFromRow = {
531       val tieredLength = tieredKeySpace.map(_.indexKeyByteLength).getOrElse(Right(0))
532       (keySpace.indexKeyByteLength, tieredLength) match {
533         case (Right(i), Right(j)) =>
534           new FixedIdFromRow(idFromBytes(sft), i + j)
535 
536         case (Right(i), Left(j)) =>
537           val dynamic: (Array[Byte], Int, Int) => Int = (row, offset, length) => i + j(row, offset + i, length - i)
538           new DynamicIdFromRow(idFromBytes(sft), dynamic)
539 
540         case (Left(i), Right(j)) =>
541           val dynamic: (Array[Byte], Int, Int) => Int = (row, offset, length) => i(row, offset, length - j) + j
542           new DynamicIdFromRow(idFromBytes(sft), dynamic)
543 
544         case (Left(i), Left(j))   =>
545           val dynamic: (Array[Byte], Int, Int) => Int =
546             (row, offset, length) => {
547               val first = i(row, offset, length)
548               j(row, offset + first, length - first)
549             }
550           new DynamicIdFromRow(idFromBytes(sft), dynamic)
551       }
552     }
553 
554     /**
555       * Id is always at a fixed offset into the row
556       *
557       * @param idFromBytes deserialization for the feature id
558       * @param prefix the length of the fixed prefix preceding the feature id bytes
559       */
560     final class FixedIdFromRow(idFromBytes: (Array[Byte], Int, Int, SimpleFeature) => String, prefix: Int)
561         extends IdFromRow {
562 
563       override def apply(row: Array[Byte], offset: Int, length: Int, feature: SimpleFeature): String =
564         idFromBytes(row, offset + prefix, length - prefix, feature)
565 
566       override def start(row: Array[Byte], offset: Int, length: Int): Int = prefix
567     }
568 
569     /**
570       * Id is dynamically located in the row
571       *
572       * @param idFromBytes deserialization of the feature id
573       * @param prefix a function to find the length of the prefix preceding the feature id bytes
574       */
575     final class DynamicIdFromRow(
576         idFromBytes: (Array[Byte], Int, Int, SimpleFeature) => String,
577         prefix: (Array[Byte], Int, Int) => Int
578       ) extends IdFromRow {
579 
580       override def apply(row: Array[Byte], offset: Int, length: Int, feature: SimpleFeature): String = {
581         val prefix = this.prefix(row, offset, length)
582         idFromBytes(row, offset + prefix, length - prefix, feature)
583       }
584 
585       override def start(row: Array[Byte], offset: Int, length: Int): Int = prefix(row, offset, length)
586     }
