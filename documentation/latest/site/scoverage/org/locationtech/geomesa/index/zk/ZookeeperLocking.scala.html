20 trait ZookeeperLocking extends DistributedLocking {
21 
22   protected def zookeepers: String
23 
24   /**
25     * Gets and acquires a distributed lock based on the key.
26     * Make sure that you 'release' the lock in a finally block.
27     *
28     * @param key key to lock on - equivalent to a path in zookeeper
29     * @return the lock
30     */
31   override protected def acquireDistributedLock(key: String): Closeable = {
32     val (client, lock) = distributedLock(key)
33     try {
34       lock.acquire()
35       ZookeeperLocking.releasable(lock, client)
36     } catch {
37       case e: Exception => CloseQuietly(client).foreach(e.addSuppressed); throw e
38     }
39   }
40 
41   /**
42     * Gets and acquires a distributed lock based on the key.
43     * Make sure that you 'release' the lock in a finally block.
44     *
45     * @param key key to lock on - equivalent to a path in zookeeper
46     * @param timeOut how long to wait to acquire the lock, in millis
47     * @return the lock, if obtained
48     */
49   override protected def acquireDistributedLock(key: String, timeOut: Long): Option[Closeable] = {
50     val (client, lock) = distributedLock(key)
51     try {
52       if (lock.acquire(timeOut, TimeUnit.MILLISECONDS)) {
53         Some(ZookeeperLocking.releasable(lock, client))
54       } else {
55         None
56       }
57     } catch {
58       case e: Exception => CloseQuietly(client).foreach(e.addSuppressed); throw e
59     }
60   }
61 
62   private def distributedLock(key: String): (CuratorFramework, InterProcessSemaphoreMutex) = {
63     val lockPath = if (key.startsWith("/")) key else s"/$key"
64     val client =
65       CuratorFrameworkFactory.builder()
66         .connectString(zookeepers)
67         .retryPolicy(new ExponentialBackoffRetry(1000, 3))
68         .dontUseContainerParents()
69         .build()
70     client.start()
71     val lock = new InterProcessSemaphoreMutex(client, lockPath)
72     (client, lock)
73   }
74 }
75 
76 object ZookeeperLocking {
77 
78   // delegate lock that will close the curator client upon release
79   def releasable(lock: InterProcessSemaphoreMutex, client: CuratorFramework): Closeable =
80     () => try { lock.release() } finally { client.close() }
81 }
