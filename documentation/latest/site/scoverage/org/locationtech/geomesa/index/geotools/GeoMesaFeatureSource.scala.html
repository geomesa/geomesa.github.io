39   /**
40     * The default behavior for getCount is to use estimated statistics if available, or -1 to indicate
41     * that the operation would be expensive (@see org.geotools.api.data.FeatureSource#getCount(org.geotools.api.data.Query)).
42     *
43     * Since users may want <b>exact</b> counts, there are two ways to force exact counts:
44     *   1. use the system property "geomesa.force.count"
45     *   2. use the EXACT_COUNT query hint
46     *
47     * @param query query
48     * @return
49     */
50   override def getCount(query: Query): Int = {
51     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
52     import org.locationtech.geomesa.index.conf.QueryProperties.QueryExactCount
53 
54     // configure the query hints
55     val hints = QueryRunner.configureQuery(sft, query).getHints
56     val useExactCount = hints.isExactCount.getOrElse(QueryExactCount.get.toBoolean)
57 
58     val count = if (useExactCount && hints.getMaxFeatures.exists(_ < QueryExactCountMaxFeatures.get.toInt)) {
59       SelfClosingIterator(getFeatures(query)).size
60     } else {
61       val statsCount = ds.stats.getCount(getSchema, query.getFilter, useExactCount, hints).getOrElse(-1L)
62       hints.getMaxFeatures match {
63         case None => statsCount
64         case Some(m) => if (statsCount < 0) { m } else { math.min(statsCount, m) }
65       }
66     }
67 
68     if (count > Int.MaxValue) {
69       logger.warn(s"Truncating count $count to Int.MaxValue (${Int.MaxValue})")
70       Int.MaxValue
71     } else {
72       count.toInt
73     }
74   }
