14 /**
15   * If the stats parser receives a string with multiple stats, a SeqStat will be used.
16   *
17   * @param sft simple feature type
18   * @param stats a Sequence of individual Stat objects
19   */
20 class SeqStat(val sft: SimpleFeatureType, val stats: Seq[Stat]) extends Stat {
21 
22   override type S = SeqStat
23 
24   override def observe(sf: SimpleFeature): Unit = stats.foreach(_.observe(sf))
25 
26   override def unobserve(sf: SimpleFeature): Unit = stats.foreach(_.unobserve(sf))
27 
28   override def +(other: SeqStat): SeqStat = {
29     val builder = Seq.newBuilder[Stat]
30     builder.sizeHint(stats.length)
31     val iter = other.stats.iterator
32     stats.foreach { stat =>
33       if (iter.hasNext) {
34         builder += (stat + iter.next())
35       }
36     }
37     new SeqStat(sft, builder.result())
38   }
39 
40 
41   override def +=(other: SeqStat): Unit = {
42     val iter = other.stats.iterator
43     stats.foreach { stat =>
44       if (iter.hasNext) {
45         stat += iter.next()
46       }
47     }
48   }
49 
50   override def toJsonObject: Seq[Any] = stats.map(_.toJsonObject)
51 
52   override def isEmpty: Boolean = stats.forall(_.isEmpty)
53 
54   override def clear(): Unit = stats.foreach(_.clear())
55 
56   override def isEquivalent(other: Stat): Boolean = other match {
57     case that: SeqStat =>
58       stats.length == that.stats.length && {
59         val iter = that.stats.iterator
60         stats.forall(stat => iter.hasNext && stat.isEquivalent(iter.next))
61       }
62 
63     case _ => false
64   }
65 }
