92   /**
93     * Encodes a sparse matrix into a byte array
94     */
95   def encodeResult(result: RenderingGrid): Array[Byte] = {
96     val output = KryoFeatureSerialization.getOutput(null)
97     result.iterator.toList.groupBy(_._1._1).foreach { case (row, cols) =>
98       output.writeInt(row, true)
99       output.writeInt(cols.size, true)
100       cols.foreach { case (xy, weight) =>
101         output.writeInt(xy._2, true)
102         output.writeDouble(weight)
103       }
104     }
105     output.toBytes
106   }

115   /**
116     * Decodes a result feature into an iterator of (x, y, weight)
117     */
118   def decodeResult(gridSnap: GridSnap)(sf: SimpleFeature): Iterator[(Double, Double, Double)] = {
119     val result = sf.getUserData.get(DensityValueKey).asInstanceOf[Array[Byte]]
120     val input = KryoFeatureDeserialization.getInput(result, 0, result.length)
121     new Iterator[(Double, Double, Double)]() {
122       private var x = 0.0
123       private var colCount = 0
124       override def hasNext: Boolean = input.position < input.limit
125       override def next(): (Double, Double, Double) = {
126         if (colCount == 0) {
127           x = gridSnap.x(input.readInt(true))
128           colCount = input.readInt(true)
129         }
130         val y = gridSnap.y(input.readInt(true))
131         val weight = input.readDouble()
132         colCount -= 1
133         (x, y, weight)
134       }
135     }
136   }

151   /**
152    * Gets a renderer for the associated geometry binding
153    *
154    * @param sft simple feature type
155    * @param geom geometry field to render
156    * @param weight field to use for weighting features
157    * @return
158    */
159   def getRenderer(sft: SimpleFeatureType, geom: String, weight: Option[String]): GeometryRenderer = {
160     // function to get the weight from the feature - defaults to 1.0 unless an attribute/exp is specified
161     val weigher = weight match {
162       case None => EqualWeight
163       case Some(w) =>
164         val i = sft.indexOf(w)
165         if (i == -1) {
166           new WeightByExpression(ECQL.toExpression(w))
167         } else if (classOf[Number].isAssignableFrom(sft.getDescriptor(i).getType.getBinding)) {
168           new WeightByNumber(i)
169         } else {
170           new WeightByNonNumber(i)
171         }
172     }
173 
174     val i = sft.indexOf(geom)
175     sft.getDescriptor(i).getType.getBinding match {
176       case b if b == classOf[Point]           => new PointRenderer(i, weigher)
177       case b if b == classOf[MultiPoint]      => new MultiPointRenderer(i, weigher)
178       case b if b == classOf[LineString]      => new LineStringRenderer(i, weigher)
179       case b if b == classOf[MultiLineString] => new MultiLineStringRenderer(i, weigher)
180       case b if b == classOf[Polygon]         => new PolygonRenderer(i, weigher)
181       case b if b == classOf[MultiPolygon]    => new MultiPolygonRenderer(i, weigher)
182       case _                                  => new MultiRenderer(i, weigher)
183     }
184   }

243   private sealed trait Weigher {
244     def weight(sf: SimpleFeature): Double
245   }
246 
247   private case object EqualWeight extends Weigher {
248     override def weight(sf: SimpleFeature): Double = 1d
249   }
250 
251   /**
252     * Gets the weight for a feature from a numeric attribute
253     */
254   private class WeightByNumber(i: Int) extends Weigher {
255     override def weight(sf: SimpleFeature): Double = {
256       val d = sf.getAttribute(i).asInstanceOf[Number]
257       if (d == null) { 0.0 } else { d.doubleValue }
258     }
259   }
260 
261   /**
262     * Tries to convert a non-double attribute into a double
263     */
264   private class WeightByNonNumber(i: Int) extends Weigher {
265     override def weight(sf: SimpleFeature): Double = {
266       val d = sf.getAttribute(i)
267       if (d == null) { 0.0 } else {
268         val converted = FastConverter.convert(d, classOf[java.lang.Double])
269         if (converted == null) { 1.0 } else { converted.doubleValue() }
270       }
271     }
272   }
273 
274   /**
275     * Evaluates an arbitrary expression against the simple feature to return a weight
276     */
277   private class WeightByExpression(e: Expression) extends Weigher {
278     override def weight(sf: SimpleFeature): Double = {
279       val d = e.evaluate(sf, classOf[java.lang.Double])
280       if (d == null) { 0.0 } else { d }
281     }
282   }
