141   /**
142     * Reduces computed simple features which contain stat information into one on the client
143     *
144     * @param sft sft used for the stat query
145     * @param query stat query
146     * @param encode encode results or return as json
147     */
148   class StatsReducer(
149       private var sft: SimpleFeatureType,
150       private var query: String,
151       private var encode: Boolean
152     ) extends FeatureReducer {
153 
154     def this() = this(null, null, false) // no-arg constructor required for serialization
155 
156     override def init(state: Map[String, String]): Unit = {
157       sft = SimpleFeatureTypes.createType(state("sft"), state("spec"))
158       query = state("q")
159       encode = state("e").toBoolean
160     }
161 
162     override def state: Map[String, String] = Map(
163       "sft"  -> sft.getTypeName,
164       "spec" -> SimpleFeatureTypes.encodeType(sft, includeUserData = true),
165       "q"    -> query,
166       "e"    -> encode.toString
167     )
168 
169     override def apply(features: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = {
170       try {
171         // if no results, create empty stat based on the original input so that we always return something
172         val sum = if (features.isEmpty) { Stat(sft, query) } else {
173           val decode = decodeStat(sft)
174           features.map(f => decode(f.getAttribute(0).asInstanceOf[String])).reduceLeft(reducer)
175         }
176         val result = if (encode) { encodeStat(sft)(sum) } else { sum.toJson }
177         CloseableIterator.single(new ScalaSimpleFeature(StatsSft, "stat", Array(result, GeometryUtils.zeroPoint)))
178       } finally {
179         CloseWithLogging(features)
180       }
181     }
182 
183     private def reducer(sum: Stat, next: Stat): Stat = { sum += next; sum }
184 
185     def canEqual(other: Any): Boolean = other.isInstanceOf[StatsReducer]
186 
187     override def equals(other: Any): Boolean = other match {
188       case that: StatsReducer if that.canEqual(this) => sft == that.sft && query == that.query && encode == that.encode
189       case _ => false
190     }
191 
192     override def hashCode(): Int = {
193       val state = Seq(sft, query, encode)
194       state.map(Objects.hashCode).foldLeft(0)((a, b) => 31 * a + b)
195     }
196   }
