74     val serializer = StatSerializer(sft)

142   /**
143     * Reduces computed simple features which contain stat information into one on the client
144     *
145     * @param sft sft used for the stat query
146     * @param query stat query
147     * @param encode encode results or return as json
148     */
149   class StatsReducer(
150       private var sft: SimpleFeatureType,
151       private var query: String,
152       private var encode: Boolean
153     ) extends FeatureReducer {
154 
155     def this() = this(null, null, false) // no-arg constructor required for serialization
156 
157     override def init(state: Map[String, String]): Unit = {
158       sft = SimpleFeatureTypes.createType(state("sft"), state("spec"))
159       query = state("q")
160       encode = state("e").toBoolean
161     }
162 
163     override def state: Map[String, String] = Map(
164       "sft"  -> sft.getTypeName,
165       "spec" -> SimpleFeatureTypes.encodeType(sft, includeUserData = true),
166       "q"    -> query,
167       "e"    -> encode.toString
168     )
169 
170     override def apply(features: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = {
171       try {
172         // if no results, create empty stat based on the original input so that we always return something
173         val sum = if (features.isEmpty) { Stat(sft, query) } else {
174           val decode = decodeStat(sft)
175           features.map(f => decode(f.getAttribute(0).asInstanceOf[String])).reduceLeft(reducer)
176         }
177         val result = if (encode) { encodeStat(sft)(sum) } else { sum.toJson }
178         CloseableIterator.single(new ScalaSimpleFeature(StatsSft, "stat", Array(result, GeometryUtils.zeroPoint)))
179       } finally {
180         CloseWithLogging(features)
181       }
182     }
183 
184     private def reducer(sum: Stat, next: Stat): Stat = { sum += next; sum }
185 
186     def canEqual(other: Any): Boolean = other.isInstanceOf[StatsReducer]
187 
188     override def equals(other: Any): Boolean = other match {
189       case that: StatsReducer if that.canEqual(this) => sft == that.sft && query == that.query && encode == that.encode
190       case _ => false
191     }
192 
193     override def hashCode(): Int = {
194       val state = Seq(sft, query, encode)
195       state.map(Objects.hashCode).foldLeft(0)((a, b) => 31 * a + b)
196     }
197   }
