44 object AttributeIndexV3 {
45 
46   private val MinDateTime = ZonedDateTime.of(0, 1, 1, 0, 0, 0, 0, ZoneOffset.UTC)
47   private val MaxDateTime = ZonedDateTime.of(9999, 12, 31, 23, 59, 59, 999000000, ZoneOffset.UTC)
48 
49   // note: add 1 to exclude null values
50   private val MinLowerBound = Long.MinValue + 1
51   private val MaxUpperBound = Long.MaxValue
52 
53   private val Unbounded =
54     UnboundedByteRange(
55       ByteArrays.toOrderedBytes(MinLowerBound),
56       ByteArrays.rowFollowingPrefix(ByteArrays.toOrderedBytes(MaxUpperBound)))
57 
58   class LegacyDateIndexKeySpace(val sft: SimpleFeatureType, dtgField: String)
59       extends IndexKeySpace[FilterValues[Bounds[ZonedDateTime]], Long] {
60 
61     require(classOf[Date].isAssignableFrom(sft.getDescriptor(dtgField).getType.getBinding),
62       s"Expected field $dtgField to have a date binding, but instead it has: " +
63           sft.getDescriptor(dtgField).getType.getBinding.getSimpleName)
64 
65     private val dtgIndex = sft.indexOf(dtgField)
66 
67     override val attributes: Seq[String] = Seq(dtgField)
68 
69     override val sharing: Array[Byte] = Array.empty
70 
71     override val sharding: ShardStrategy = NoShardStrategy
72 
73     override val indexKeyByteLength: Right[(Array[Byte], Int, Int) => Int, Int] = Right(12)
74 
75     override def toIndexKey(writable: WritableFeature,
76                             tier: Array[Byte],
77                             id: Array[Byte],
78                             lenient: Boolean): RowKeyValue[Long] = {
79       // note: only used as a tiered keyspace so tier and id will be empty
80       val dtg = writable.getAttribute[Date](dtgIndex)
81       val time = if (dtg == null) { 0L } else { dtg.getTime }
82       SingleRowKeyValue(timeToBytes(time), sharing, sharing, time, tier, sharing, Seq.empty)
83     }
84 
85     override def getIndexValues(filter: Filter, explain: Explainer): FilterValues[Bounds[ZonedDateTime]] =
86       FilterHelper.extractIntervals(filter, dtgField)
87 
88     override def getRanges(values: FilterValues[Bounds[ZonedDateTime]], multiplier: Int): Iterator[ScanRange[Long]] = {
89       if (values.isEmpty) {
90         Iterator.single(UnboundedRange(-1))
91       } else if (values.disjoint) {
92         Iterator.empty
93       } else {
94         values.values.iterator.map { bounds =>
95           val lower = bounds.lower.value.getOrElse(MinDateTime).toInstant.toEpochMilli
96           val upper = bounds.upper.value.getOrElse(MaxDateTime).toInstant.toEpochMilli
97           BoundedRange(lower, upper)
98         }
99       }
100     }
101 
102     override def getRangeBytes(ranges: Iterator[ScanRange[Long]], tier: Boolean): Iterator[ByteRange] = {
103       ranges.map {
104         case BoundedRange(lo, hi) => BoundedByteRange(timeToBytes(lo), roundUpTime(timeToBytes(hi)))
105         case UnboundedRange(_)    => Unbounded
106         case r => throw new IllegalArgumentException(s"Unexpected range type $r")
107       }
108     }
109 
110     override def useFullFilter(values: Option[FilterValues[Bounds[ZonedDateTime]]],
111                                config: Option[GeoMesaDataStoreConfig],
112                                hints: Hints): Boolean = false // TODO for functions?
113 
114     // store the first 12 hex chars of the time - that is roughly down to the minute interval
115     private def timeToBytes(t: Long): Array[Byte] =
116       AttributeIndexKey.typeEncode(t).substring(0, 12).getBytes(StandardCharsets.UTF_8)
117 
118     // rounds up the time to ensure our range covers all possible times given our time resolution
119     private def roundUpTime(time: Array[Byte]): Array[Byte] = {
120       // find the last byte in the array that is not 0xff
121       var changeIndex: Int = time.length - 1
122       while (changeIndex > -1 && time(changeIndex) == 0xff.toByte) { changeIndex -= 1 }
123 
124       if (changeIndex < 0) {
125         // the array is all 1s - it's already at time max given our resolution
126         time
127       } else {
128         // increment the selected byte
129         time.updated(changeIndex, (time(changeIndex) + 1).asInstanceOf[Byte])
130       }
131     }
132   }
