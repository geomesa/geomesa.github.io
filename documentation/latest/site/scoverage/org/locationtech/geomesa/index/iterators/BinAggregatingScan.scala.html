31   override protected def createResult(
32       sft: SimpleFeatureType,
33       transform: Option[SimpleFeatureType],
34       batchSize: Int,
35       options: Map[String, String]): ResultCallback = {
36     val geom = options.get(GeomOpt).map(_.toInt).filter(_ != -1)
37     val dtg = options.get(DateOpt).map(_.toInt).filter(_ != -1)
38     val track = options.get(TrackOpt).map(_.toInt).filter(_ != -1)
39     val label = options.get(LabelOpt).map(_.toInt).filter(_ != -1)
40 
41     val encoder = BinaryOutputEncoder(sft, EncodingOptions(geom, dtg, track, label))
42 
43     val binSize = if (label.isEmpty) { 16 } else { 24 }
44     val sort = options(SortOpt).toBoolean
45 
46     val buffer = ByteBuffer.wrap(Array.ofDim(batchSize * binSize)).order(ByteOrder.LITTLE_ENDIAN)
47     val overflow = ByteBuffer.wrap(Array.ofDim(binSize * 16)).order(ByteOrder.LITTLE_ENDIAN)
48 
49     new ResultCallback(buffer, overflow, encoder, binSize, sort)
50   }
51 
52   override protected def defaultBatchSize: Int =
53     throw new IllegalArgumentException("Batch scan is specified per scan")

121   class ResultCallback(
122       buffer: ByteBuffer,
123       private var overflow: ByteBuffer,
124       encoder: BinaryOutputEncoder,
125       binSize: Int,
126       sort: Boolean
127     ) extends AggregatingScan.Result with BinaryOutputCallback {
128 
129     override def apply(trackId: Int, lat: Float, lon: Float, dtg: Long): Unit =
130       put(ensureCapacity(16), trackId, lat, lon, dtg)
131 
132     override def apply(trackId: Int, lat: Float, lon: Float, dtg: Long, label: Long): Unit =
133       put(ensureCapacity(24), trackId, lat, lon, dtg, label)
134 
135     override def init(): Unit = {}
136 
137     override def aggregate(sf: SimpleFeature): Int = {
138       val pos = buffer.position() + overflow.position()
139       encoder.encode(sf, this)
140       (buffer.position() + overflow.position() - pos) / binSize
141     }
142 
143     override def encode(): Array[Byte] = {
144       val bytes = try {
145         if (overflow.position() > 0) {
146           // overflow bytes - copy the two buffers into one
147           val copy = Array.ofDim[Byte](buffer.position() + overflow.position())
148           System.arraycopy(buffer.array, 0, copy, 0, buffer.position())
149           System.arraycopy(overflow.array, 0, copy, buffer.position(), overflow.position())
150           copy
151         } else if (buffer.position() == buffer.limit()) {
152           // use the existing buffer if possible
153           buffer.array
154         } else {
155           // if not, we have to copy it - values do not allow you to specify a valid range
156           val copy = Array.ofDim[Byte](buffer.position())
157           System.arraycopy(buffer.array, 0, copy, 0, buffer.position())
158           copy
159         }
160       } finally {
161         buffer.clear()
162         overflow.clear()
163       }
164       if (sort) {
165         BinSorter.quickSort(bytes, 0, bytes.length - binSize, binSize)
166       }
167       bytes
168     }
169 
170     override def cleanup(): Unit = {}
171 
172     private def ensureCapacity(size: Int): ByteBuffer = {
173       if (buffer.position() < buffer.limit() - size) {
174         buffer
175       } else if (overflow.position() < overflow.limit() - size) {
176         overflow
177       } else {
178         val expanded = Array.ofDim[Byte](overflow.limit() * 2)
179         System.arraycopy(overflow.array, 0, expanded, 0, overflow.limit())
180         val order = overflow.order()
181         val position = overflow.position()
182         overflow = ByteBuffer.wrap(expanded).order(order).position(position).asInstanceOf[ByteBuffer]
183         overflow
184       }
185     }
186   }
