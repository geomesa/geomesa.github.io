49     private val rangePrefixes = {
50       if (sharing.isEmpty && sharding.length == 0) {
51         Seq.empty
52       } else if (sharing.isEmpty) {
53         sharding.shards
54       } else if (sharding.length == 0) {
55         Seq(sharing)
56       } else {
57         sharding.shards.map(ByteArrays.concat(sharing, _))
58       }
59     }
60 
61     override val indexKeyByteLength: Right[(Array[Byte], Int, Int) => Int, Int] =
62       Right(10 + sharing.length + sharding.length)
63 
64     override def toIndexKey(writable: WritableFeature,
65                             tier: Array[Byte],
66                             id: Array[Byte],
67                             lenient: Boolean): RowKeyValue[Z3IndexKey] = {
68       val geom = writable.getAttribute[Geometry](geomIndex)
69       if (geom == null) {
70         throw new IllegalArgumentException(s"Null geometry in feature ${writable.feature.getID}")
71       }
72       val envelope = geom.getEnvelopeInternal
73       val dtg = writable.getAttribute[Date](dtgIndex)
74       val time = if (dtg == null) { 0L } else { dtg.getTime }
75       val BinnedTime(b, t) = timeToIndex(time)
76       val xz = try {
77         sfc.index(envelope.getMinX, envelope.getMinY, t, envelope.getMaxX, envelope.getMaxY, t, lenient)
78       } catch {
79         case NonFatal(e) => throw new IllegalArgumentException(s"Invalid xz value from geometry/time: $geom,$dtg", e)
80       }
81       val shard = sharding(writable)
82 
83       // create the byte array - allocate a single array up front to contain everything
84       // ignore tier, not used here
85       val bytes = Array.ofDim[Byte](sharing.length + shard.length + 10 + id.length)
86 
87       var i = 0
88       if (!sharing.isEmpty) {
89         bytes(0) = sharing.head // sharing is only a single byte
90         i += 1
91       }
92       if (!shard.isEmpty) {
93         bytes(i) = shard.head // shard is only a single byte
94         i += 1
95       }
96 
97       ByteArrays.writeShort(b, bytes, i)
98       ByteArrays.writeLong(xz, bytes, i + 2)
99       System.arraycopy(id, 0, bytes, i + 10, id.length)
100 
101       SingleRowKeyValue(bytes, sharing, shard, Z3IndexKey(b, xz), tier, id, writable.values)
102     }
103 
104     override def getRangeBytes(ranges: Iterator[ScanRange[Z3IndexKey]], tier: Boolean): Iterator[ByteRange] = {
105       if (rangePrefixes.isEmpty) {
106         ranges.map {
107           case BoundedRange(lo, hi) =>
108             BoundedByteRange(ByteArrays.toBytes(lo.bin, lo.z), ByteArrays.toBytesFollowingPrefix(hi.bin, hi.z))
109 
110           case r =>
111             throw new IllegalArgumentException(s"Unexpected range type $r")
112         }
113       } else {
114         ranges.flatMap {
115           case BoundedRange(lo, hi) =>
116             val lower = ByteArrays.toBytes(lo.bin, lo.z)
117             val upper = ByteArrays.toBytesFollowingPrefix(hi.bin, hi.z)
118             rangePrefixes.map(p => BoundedByteRange(ByteArrays.concat(p, lower), ByteArrays.concat(p, upper)))
119 
120           case r =>
121             throw new IllegalArgumentException(s"Unexpected range type $r")
122         }
123       }
124     }
125   }
