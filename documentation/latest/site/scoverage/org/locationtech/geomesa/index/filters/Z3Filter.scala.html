26   override def inBounds(buf: Array[Byte], offset: Int): Boolean = {
27     val keyZ = ByteArrays.readLong(buf, offset + 2) // account for epoch - first 2 bytes
28     pointInBounds(keyZ) && timeInBounds(ByteArrays.readShort(buf, offset), keyZ)
29   }

31   private def pointInBounds(z: Long): Boolean = {
32     val x = Z3(z).d0
33     val y = Z3(z).d1
34     var i = 0
35     while (i < xy.length) {
36       val xyi = xy(i)
37       if (x >= xyi(0) && x <= xyi(2) && y >= xyi(1) && y <= xyi(3)) {
38         return true
39       }
40       i += 1
41     }
42     false
43   }
44 
45   private def timeInBounds(epoch: Short, z: Long): Boolean = {
46     // we know we're only going to scan appropriate epochs, so leave out whole epochs
47     if (epoch > maxEpoch || epoch < minEpoch) { true } else {
48       val tEpoch = t(epoch - minEpoch)
49       if (tEpoch == null) { true } else {
50         val time = Z3(z).d2
51         var i = 0
52         while (i < tEpoch.length) {
53           val ti = tEpoch(i)
54           if (time >= ti(0) && time <= ti(1)) {
55             return true
56           }
57           i += 1
58         }
59         false
60       }
61     }
62   }

78   def apply(values: Z3IndexValues): Z3Filter = {
79     val Z3IndexValues(sfc, _, spatialBounds, _, temporalBounds, _) = values
80 
81     val xy: Array[Array[Int]] = spatialBounds.map { case (xmin, ymin, xmax, ymax) =>
82       Array(sfc.lon.normalize(xmin), sfc.lat.normalize(ymin), sfc.lon.normalize(xmax), sfc.lat.normalize(ymax))
83     }.toArray
84 
85     // we know we're only going to scan appropriate periods, so leave out whole ones
86     val wholePeriod = Seq((sfc.time.min.toLong, sfc.time.max.toLong))
87     var minEpoch: Short = Short.MaxValue
88     var maxEpoch: Short = Short.MinValue
89     val epochsAndTimes = temporalBounds.toSeq.filter(_._2 != wholePeriod).sortBy(_._1).map { case (epoch, times) =>
90       // set min/max epochs - note: side effect in map
91       if (epoch < minEpoch) {
92         minEpoch = epoch
93       }
94       if (epoch > maxEpoch) {
95         maxEpoch = epoch
96       }
97       (epoch, times.map { case (t1, t2) => Array(sfc.time.normalize(t1), sfc.time.normalize(t2)) }.toArray)
98     }
99 
100     val t: Array[Array[Array[Int]]] =
101       if (minEpoch == Short.MaxValue && maxEpoch == Short.MinValue) {
102         Array.empty
103       } else {
104         Array.ofDim(maxEpoch - minEpoch + 1)
105       }
106 
107     epochsAndTimes.foreach { case (w, times) => t(w - minEpoch) = times }
108 
109     new Z3Filter(xy, t, minEpoch, maxEpoch)
110   }

112   override def serializeToBytes(filter: Z3Filter): Array[Byte] = {
113     // 4 bytes for length plus 16 bytes for each xy val (4 ints)
114     val xyLength = 4 + filter.xy.length * 16
115     // 4 bytes for length, then per-epoch 4 bytes for length plus 8 bytes for each t val (2 ints)
116     val tLength = 4 + filter.t.map(bounds => if (bounds == null) { 4 } else { 4 + bounds.length * 8 } ).sum
117     // additional 4 bytes for min/max epoch
118     val buffer = ByteBuffer.allocate(xyLength + tLength + 4)
119 
120     buffer.putInt(filter.xy.length)
121     filter.xy.foreach(bounds => bounds.foreach(buffer.putInt))
122 
123     buffer.putInt(filter.t.length)
124     filter.t.foreach { bounds =>
125       if (bounds == null) {
126         buffer.putInt(-1)
127       } else {
128         buffer.putInt(bounds.length)
129         bounds.foreach(inner => inner.foreach(buffer.putInt))
130       }
131     }
132 
133     buffer.putShort(filter.minEpoch)
134     buffer.putShort(filter.maxEpoch)
135 
136     buffer.array()
137   }
