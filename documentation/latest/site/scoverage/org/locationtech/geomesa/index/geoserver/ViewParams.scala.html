36   /**
37     * Serialize hints into a string
38     *
39     * @param hints hints
40     * @return
41     */
42   def serialize(hints: Hints): String = {
43     val map = AllHintsMap.flatMap { case (name, hint) =>
44       Option(hints.get(hint)).flatMap {
45         case v: String             => Some(name -> v)
46         case v: java.lang.Boolean  => Some(name -> v.toString)
47         case v: java.lang.Integer  => Some(name -> v.toString)
48         case v: java.lang.Float    => Some(name -> v.toString)
49         case v: ReferencedEnvelope => Some(name -> toString(v))
50         case v: CostEvaluation     => Some(name -> v.toString)
51         case v: SimpleFeatureType  => Some(name -> SimpleFeatureTypes.serialize(v))
52         case _ => logger.warn(s"Unhandled hint type for '$name'"); None
53       }
54     }
55     StringSerialization.encodeMap(map)
56   }

96   def getReadableHints(hints: Hints): java.util.Map[String, String] = {
97     val readable = new java.util.TreeMap[String, String]() // use tree map for consistent ordering by key
98     hints.asScala.map { case (k: Hints.Key, v) =>
99       val key = hintToString(k)
100       val value = v match {
101         case null => "null"
102         case sft: SimpleFeatureType => SimpleFeatureTypes.encodeType(sft)
103         case s => s.toString
104       }
105       readable.put(key, value)
106     }
107     readable
108   }

112   private def setHints(hints: Hints, params: Map[String, String], lookup: Map[String, Hints.Key]): Unit = {
113     params.foreach { case (original, value) =>
114       val key = if (original == "STRATEGY") { "QUERY_INDEX" } else { original }
115       lookup.get(key) match {
116         case None => logger.debug(s"Ignoring view param $key=$value")
117         case Some(hint) =>
118           try {
119             hint.getValueClass match {
120               case c if c == classOf[String]             => setHint(hints, key, hint, value)
121               case c if c == classOf[java.lang.Boolean]  => toBoolean(key, value).foreach(setHint(hints, key, hint, _))
122               case c if c == classOf[java.lang.Integer]  => toInt(key, value).foreach(setHint(hints, key, hint, _))
123               case c if c == classOf[java.lang.Float]    => toFloat(key, value).foreach(setHint(hints, key, hint, _))
124               case c if c == classOf[ReferencedEnvelope] => toEnvelope(key, value).foreach(setHint(hints, key, hint, _))
125               case c if c == classOf[CostEvaluation]     => toCost(value).foreach(setHint(hints, key, hint, _))
126               case c if c == classOf[SimpleFeatureType]  => toFeatureType(key, value).foreach(setHint(hints, key, hint, _))
127               case _ => logger.warn(s"Unhandled hint type for '$key'")
128             }
129           } catch {
130             case NonFatal(e) => logger.warn(s"Error invoking query hint for $key=$value", e)
131           }
132       }
133     }
134   }

136   private def setHint(hints: Hints, name: String, hint: Hints.Key, value: Any): Unit = {
137     val old = hints.get(hint)
138     if (old == null) {
139       logger.debug(s"Using query hint from geoserver view params: $name=$value")
140       hints.put(hint, value)
141     } else if (old != value) {
142       logger.warn("Ignoring query hint from geoserver in favor of hint directly set in query. " +
143           s"Using $name=$old and disregarding $value")
144     }
145   }

147   private def toCost(name: String): Option[CostEvaluation] = {
148     val check = name.toLowerCase(Locale.US)
149     val value = CostEvaluation.values.find(_.toString.toLowerCase(Locale.US) == check)

151       logger.error(s"Ignoring invalid cost type from view params: $name. Valid values " +

160   private def toEnvelope(name: String, geom: String): Option[ReferencedEnvelope] = {
161     import org.locationtech.geomesa.utils.geotools.CRS_EPSG_4326
162 
163     def fromBbox: Try[ReferencedEnvelope] = Try {
164       // note: underlines are internal capture groups enclosed in the values we're already pulling out
165       val envelope(xmin, _, ymin, _, xmax, _, ymax, _) = geom
166       new ReferencedEnvelope(xmin.toDouble, xmax.toDouble, ymin.toDouble, ymax.toDouble, CRS_EPSG_4326)
167     }
168 
169     def fromWkt: Try[ReferencedEnvelope] =
170       Try(new ReferencedEnvelope(WKTUtils.read(geom).getEnvelopeInternal, CRS_EPSG_4326))
171 
172     val value = fromBbox.orElse(fromWkt).toOption
173     if (value.isEmpty) {
174       logger.error(s"Ignoring invalid envelope from view params: $name=$geom. Envelope should be " +
175           "WKT or in the form [-180.0,-90.0,180.0,90.0]")
176     }
177     value
178   }

204   private def buildHintsMap(obj: Object): Map[String, Hints.Key] = {
205     val methods = obj.getClass.getDeclaredMethods.filter { m =>
206       m.getParameterCount == 0 && classOf[Hints.Key].isAssignableFrom(m.getReturnType)
207     }
208     val map = Map.newBuilder[String, Hints.Key]
209     map.sizeHint(methods.length)
210     // note: keys in the view params map are always uppercase
211     methods.foreach(m => map += m.getName.toUpperCase(Locale.US) -> m.invoke(obj).asInstanceOf[Hints.Key])
212     map.result()
213   }
