36   /**
37     * Serialize hints into a string
38     *
39     * @param hints hints
40     * @return
41     */
42   def serialize(hints: Hints): String = {
43     val map = AllHintsMap.flatMap { case (name, hint) =>
44       Option(hints.get(hint)).flatMap {
45         case v: String             => Some(name -> v)
46         case v: java.lang.Boolean  => Some(name -> v.toString)
47         case v: java.lang.Integer  => Some(name -> v.toString)
48         case v: java.lang.Float    => Some(name -> v.toString)
49         case v: ReferencedEnvelope => Some(name -> toString(v))
50         case v: CostEvaluation     => Some(name -> v.toString)
51         case v: SimpleFeatureType  => Some(name -> SimpleFeatureTypes.serialize(v))
52         case _ => logger.warn(s"Unhandled hint type for '$name'"); None
53       }
54     }
55     StringSerialization.encodeMap(map)
56   }

90   /**
91    * Maps query hints to readable strings
92    *
93    * @param hints hints
94    * @return
95    */
96   def getReadableHints(hints: Hints): String = {
97     val readable = Seq.newBuilder[String]
98     readable.sizeHint(hints.size())
99     hints.asScala.foreach { case (k: Hints.Key, v) =>
100       val key = hintToString(k)
101       val value = v match {
102         case null => "null"
103         case sft: SimpleFeatureType => SimpleFeatureTypes.encodeType(sft)
104         case s => s.toString
105       }
106       readable += s"$key=$value"
107     }
108     readable.result.sorted.mkString(", ")
109   }

113   private def setHints(hints: Hints, params: Map[String, String], lookup: Map[String, Hints.Key]): Unit = {
114     params.foreach { case (original, value) =>
115       val key = if (original == "STRATEGY") { "QUERY_INDEX" } else { original }
116       lookup.get(key) match {
117         case None => logger.debug(s"Ignoring view param $key=$value")
118         case Some(hint) =>
119           try {
120             hint.getValueClass match {
121               case c if c == classOf[String]             => setHint(hints, key, hint, value)
122               case c if c == classOf[java.lang.Boolean]  => toBoolean(key, value).foreach(setHint(hints, key, hint, _))
123               case c if c == classOf[java.lang.Integer]  => toInt(key, value).foreach(setHint(hints, key, hint, _))
124               case c if c == classOf[java.lang.Float]    => toFloat(key, value).foreach(setHint(hints, key, hint, _))
125               case c if c == classOf[ReferencedEnvelope] => toEnvelope(key, value).foreach(setHint(hints, key, hint, _))
126               case c if c == classOf[CostEvaluation]     => toCost(value).foreach(setHint(hints, key, hint, _))
127               case c if c == classOf[SimpleFeatureType]  => toFeatureType(key, value).foreach(setHint(hints, key, hint, _))
128               case _ => logger.warn(s"Unhandled hint type for '$key'")
129             }
130           } catch {
131             case NonFatal(e) => logger.warn(s"Error invoking query hint for $key=$value", e)
132           }
133       }
134     }
135   }

137   private def setHint(hints: Hints, name: String, hint: Hints.Key, value: Any): Unit = {
138     val old = hints.get(hint)
139     if (old == null) {
140       logger.debug(s"Using query hint from geoserver view params: $name=$value")
141       hints.put(hint, value)
142     } else if (old != value) {
143       logger.warn("Ignoring query hint from geoserver in favor of hint directly set in query. " +
144           s"Using $name=$old and disregarding $value")
145     }
146   }

148   private def toCost(name: String): Option[CostEvaluation] = {
149     val check = name.toLowerCase(Locale.US)

152       logger.error(s"Ignoring invalid cost type from view params: $name. Valid values " +
153           s"are ${CostEvaluation.values.mkString(", ")}")

161   private def toEnvelope(name: String, geom: String): Option[ReferencedEnvelope] = {
162     import org.locationtech.geomesa.utils.geotools.CRS_EPSG_4326
163 
164     def fromBbox: Try[ReferencedEnvelope] = Try {
165       // note: underlines are internal capture groups enclosed in the values we're already pulling out
166       val envelope(xmin, _, ymin, _, xmax, _, ymax, _) = geom
167       new ReferencedEnvelope(xmin.toDouble, xmax.toDouble, ymin.toDouble, ymax.toDouble, CRS_EPSG_4326)
168     }
169 
170     def fromWkt: Try[ReferencedEnvelope] =
171       Try(new ReferencedEnvelope(WKTUtils.read(geom).getEnvelopeInternal, CRS_EPSG_4326))
172 
173     val value = fromBbox.orElse(fromWkt).toOption
174     if (value.isEmpty) {
175       logger.error(s"Ignoring invalid envelope from view params: $name=$geom. Envelope should be " +
176           "WKT or in the form [-180.0,-90.0,180.0,90.0]")
177     }
178     value
179   }

205   private def buildHintsMap(obj: Object): Map[String, Hints.Key] = {
206     val methods = obj.getClass.getDeclaredMethods.filter { m =>
207       m.getParameterCount == 0 && classOf[Hints.Key].isAssignableFrom(m.getReturnType)
208     }
209     val map = Map.newBuilder[String, Hints.Key]
210     map.sizeHint(methods.length)
211     // note: keys in the view params map are always uppercase
212     methods.foreach(m => map += m.getName.toUpperCase(Locale.US) -> m.invoke(obj).asInstanceOf[Hints.Key])
213     map.result()
214   }
