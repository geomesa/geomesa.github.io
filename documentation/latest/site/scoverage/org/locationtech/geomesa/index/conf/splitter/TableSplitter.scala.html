21   def getSplits(sft: SimpleFeatureType, index: String, partition: Option[String] = None): Array[Array[Byte]] = {
22     partition match {
23       case None =>
24         val splitter = create(sft.getUserData.get(TableSplitterClass).asInstanceOf[String])
25         splitter.getSplits(sft, index, sft.getUserData.get(TableSplitterOpts).asInstanceOf[String])
26 
27       case Some(p) =>
28         val splitter = create(sft.getUserData.get(PartitionSplitterClass).asInstanceOf[String])
29         splitter.getSplits(sft, index, p, sft.getUserData.get(PartitionSplitterOpts).asInstanceOf[String])
30     }
31   }
32 
33   private def create(clas: String): TableSplitter =
34     if (clas == null) { DefaultSplitter.Instance } else { Class.forName(clas).newInstance().asInstanceOf[TableSplitter] }
