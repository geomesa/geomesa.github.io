16 class GroupBy[T](val sft: SimpleFeatureType, val property: String, val stat: String)(implicit val ct: ClassTag[T])
17     extends Stat {
18 
19   override type S = GroupBy[T]
20 
21   private val i = sft.indexOf(property)
22   private [stats] val groups = scala.collection.mutable.Map.empty[T, Stat]
23 
24   def size: Int = groups.size
25   def get(key: T): Option[Stat] = groups.get(key)
26   def getOrElse[U >: Stat](key: T, default: => U): U = groups.getOrElse(key, default)
27   def iterator: Iterator[(T, Stat)] = groups.iterator
28 
29   private def buildNewStat: Stat = StatParser.parse(sft, stat)
30 
31   /**
32     * Compute statistics based upon the given simple feature.
33     * This method will be called for every SimpleFeature a query returns.
34     *
35     * @param sf feature to evaluate
36     */
37   override def observe(sf: SimpleFeature): Unit = {
38     val key = sf.getAttribute(i).asInstanceOf[T]
39     if (key != null) {
40       groups.getOrElseUpdate(key, buildNewStat).observe(sf)
41     }
42   }
43 
44   /**
45     * Tries to remove the given simple feature from the compiled statistics.
46     * Note: may not be possible to un-observe a feature, in which case this method will
47     * have no effect.
48     *
49     * @param sf feature to un-evaluate
50     */
51   override def unobserve(sf: SimpleFeature): Unit = {
52     val key = sf.getAttribute(i).asInstanceOf[T]
53     if (key != null) {
54       groups.get(key).foreach(groupedStat => groupedStat.unobserve(sf))
55     }
56   }
57 
58   /**
59     * Add another stat to this stat. Avoids allocating another object.
60     *
61     * @param other the other stat to add
62     */
63   override def +=(other: GroupBy[T]): Unit = {
64     other.groups.foreach { case (key, s) =>
65       groups.getOrElseUpdate(key, buildNewStat) += s
66     }
67   }
68 
69   /**
70     * Combine two stats into a new stat
71     *
72     * @param other the other stat to add
73     */
74   override def +(other: GroupBy[T]): GroupBy[T] = {
75     val sum = new GroupBy[T](sft, property, stat)
76     sum += this
77     sum += other
78     sum
79   }
80 
81   override def toJsonObject: Map[T, Any] = {
82     val builder = collection.immutable.ListMap.newBuilder[T, Any]
83     val keyClass = groups.headOption.map(_._1.getClass).getOrElse(ct.runtimeClass)
84     val ordered: Iterable[(T, Stat)] = if (classOf[Comparable[_]].isAssignableFrom(keyClass)) {
85       val ordering = new Ordering[T] {
86         override def compare(x: T, y: T): Int = x.asInstanceOf[Comparable[Any]].compareTo(y)
87       }
88       groups.toSeq.sortBy(_._1)(ordering)
89     } else {
90       groups
91     }
92     ordered.foreach { case (k, v) => builder += k -> v.toJsonObject }
93     builder.result
94   }
95 
96   /**
97     * Necessary method used by the StatIterator. Indicates if the stat has any values or not
98     *
99     * @return true if stat contains values
100     */
101   override def isEmpty: Boolean = groups.values.forall(_.isEmpty)
102 
103   /**
104     * Compares the two stats for equivalence. We don't use standard 'equals' as it gets messy with
105     * mutable state and hash codes
106     *
107     * @param other other stat to compare
108     * @return true if equals
109     */
110   override def isEquivalent(other: Stat): Boolean = {
111     other match {
112       case other: GroupBy[T] =>
113         groups.keys == other.groups.keys &&
114           groups.forall { case (key, s) => other.groups(key).isEquivalent(s) }
115       case _ => false
116     }
117   }
118 
119   /**
120     * Clears the stat to its original state when first initialized.
121     * Necessary method used by the StatIterator.
122     */
123   override def clear(): Unit = groups.clear()
124 }
