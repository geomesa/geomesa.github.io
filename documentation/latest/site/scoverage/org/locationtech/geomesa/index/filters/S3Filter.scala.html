29   /**
30     * Determine if the point is inside the rectangle
31     * @param buf row
32     * @param offset offset
33     * @return
34     */
35   override def inBounds(buf: Array[Byte], offset: Int): Boolean = {
36     // account for epoch - first 2 bytes
37     val keyS = ByteArrays.readLong(buf, offset + 2)
38     val timeOffset = ByteArrays.readInt(buf, offset + 10)
39     pointInBounds(keyS) && timeInBounds(ByteArrays.readShort(buf, offset), timeOffset)
40   }
41 
42   private def pointInBounds(s3Values: Long): Boolean = {
43     val s2CellId = new S2CellId(s3Values)
44     val x = s2CellId.toLatLng.lngDegrees()
45     val y = s2CellId.toLatLng.latDegrees()
46     var i = 0
47     while (i < xy.length) {
48       val xyi = xy(i)
49       if (x >= xyi(0) && x <= xyi(2) && y >= xyi(1) && y <= xyi(3)) {
50         return true
51       }
52       i += 1
53     }
54     false
55   }
56 
57   private def timeInBounds(epoch: Short, offset: Int): Boolean = {
58     // we know we're only going to scan appropriate epochs, so leave out whole epochs
59     if (epoch > maxEpoch || epoch < minEpoch) { false } else {
60       val tEpoch = t(epoch - minEpoch)
61       if (tEpoch == null) { true } else {
62         val time = offset
63         var i = 0
64         while (i < tEpoch.length) {
65           val ti = tEpoch(i)
66           if (time >= ti(0) && time <= ti(1)) {
67             return true
68           }
69           i += 1
70         }
71         false
72       }
73     }
74   }

89   def apply(values: S3IndexValues): S3Filter = {
90     val S3IndexValues(_, maxTime, _, spatialBounds, _, temporalBounds, _) = values
91 
92     val xy: Array[Array[Double]] = spatialBounds.map { case (xmin, ymin, xmax, ymax) =>
93       Array(xmin, ymin, xmax, ymax)
94     }.toArray
95 
96     // we know we're only going to scan appropriate periods, so leave out whole ones
97     val wholePeriod = Seq((0L, maxTime))
98     var minEpoch: Short = Short.MaxValue
99     var maxEpoch: Short = Short.MinValue
100     val epochsAndTimes = temporalBounds.toSeq.filter(_._2 != wholePeriod).sortBy(_._1).map { case (epoch, times) =>
101       // set min/max epochs - note: side effect in map
102       if (epoch < minEpoch) {
103         minEpoch = epoch
104       }
105       if (epoch > maxEpoch) {
106         maxEpoch = epoch
107       }
108       (epoch, times.map { case (t1, t2) => Array(t1, t2)}.toArray)
109     }
110 
111     val t: Array[Array[Array[Int]]] =
112       if (minEpoch == Short.MaxValue && maxEpoch == Short.MinValue) {
113         Array.empty
114       } else {
115         Array.ofDim(maxEpoch - minEpoch + 1)
116       }
117 
118     epochsAndTimes.foreach { case (w, times) => t(w - minEpoch) = times }
119 
120     new S3Filter(xy, t, minEpoch, maxEpoch)
121   }

123   override def serializeToBytes(filter: S3Filter): Array[Byte] = {
124     // 4 bytes for length plus 32 bytes for each xy val (4 doubles)
125     val xyLength = 4 + filter.xy.length * 32
126     // 4 bytes for length, then per-epoch 4 bytes for length plus 8 bytes for each t val (2 ints)
127     val tLength = 4 + filter.t.map(bounds => if (bounds == null) { 4 } else { 4 + bounds.length * 8 } ).sum
128     // additional 4 bytes for min/max epoch
129     val buffer = ByteBuffer.allocate(xyLength + tLength + 4)
130 
131     buffer.putInt(filter.xy.length)
132     filter.xy.foreach(bounds => bounds.foreach(buffer.putDouble))
133     buffer.putInt(filter.t.length)
134     filter.t.foreach { bounds =>
135       if (bounds == null) {
136         buffer.putInt(-1)
137       } else {
138         buffer.putInt(bounds.length)
139         bounds.foreach(inner => inner.foreach(buffer.putInt))
140       }
141     }
142 
143     buffer.putShort(filter.minEpoch)
144     buffer.putShort(filter.maxEpoch)
145 
146     buffer.array()
147   }
