23 /**
24  * Feature visitor for calculating min/max values
25  *
26  * @param features features
27  * @param attribute attribute to calculate
28  * @param cached use cached stats, if available, vs calculating on the data
29  */
30 class MinMaxVisitor(features: SimpleFeatureCollection, attribute: String, cached: Boolean)
31   extends GeoMesaProcessVisitor with LazyLogging {
32 
33   private lazy val stat: Stat = Stat(features.getSchema, Stat.MinMax(attribute))
34 
35   private var resultCalc: FeatureResult = _
36 
37   // non-optimized visit
38   override def visit(feature: Feature): Unit = stat.observe(feature.asInstanceOf[SimpleFeature])
39 
40   override def getResult: FeatureResult = {
41     if (resultCalc != null) {
42       resultCalc
43     } else {
44       createResult(stat.toJson)
45     }
46   }
47 
48   override def execute(source: SimpleFeatureSource, query: Query): Unit = {
49     logger.debug(s"Running Geomesa min/max process on source type ${source.getClass.getName}")
50 
51     source.getDataStore match {
52       case ds: HasGeoMesaStats =>
53         resultCalc = ds.stats.getMinMax[Any](source.getSchema, attribute, query.getFilter, !cached) match {
54           case None     => createResult("{}")
55           case Some(mm) => createResult(mm.toJson)
56         }
57 
58       case ds =>
59         logger.warn(s"Running unoptimized min/max query on ${ds.getClass.getName}")
60         CloseableIterator(features.features).foreach(visit)
61     }
62   }
63 
64   private def createResult(stat: String): FeatureResult = {
65     val sf = new ScalaSimpleFeature(StatsScan.StatsSft, "", Array(stat, GeometryUtils.zeroPoint))
66     FeatureResult(new ListFeatureCollection(StatsScan.StatsSft, sf))
67   }
68 }
