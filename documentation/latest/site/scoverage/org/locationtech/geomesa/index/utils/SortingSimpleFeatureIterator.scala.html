29 /**
30   * In memory sorting of simple features, with optional spill to disk
31   *
32   * @param features unsorted feature iterator
33   * @param sortBy attributes to sort by, in the form: (name, reverse).
34   *               for sort by feature id (e.g. natural sort), use an empty string for name
35   */
36 class SortingSimpleFeatureIterator(features: CloseableIterator[SimpleFeature], sortBy: Seq[(String, Boolean)])
37     extends CloseableIterator[SimpleFeature] {
38 
39   import SortingSimpleFeatureIterator.{sortInMemory, sortWithSpillover}
40 
41   private val closed = new AtomicBoolean(false)
42 
43   private lazy val sorted: CloseableIterator[SimpleFeature] = {
44     if (closed.get || !features.hasNext) { features } else {
45       val head = features.next()
46       if (closed.get || !features.hasNext) { CloseableIterator.single(head, features.close()) } else {
47         val ordering = SimpleFeatureOrdering(head.getFeatureType, sortBy)
48         QueryProperties.SortMemoryThreshold.toBytes match {
49           case None => sortInMemory(head, features, ordering, closed)
50           case Some(threshold) => sortWithSpillover(head, features, ordering, closed, threshold)
51         }
52       }
53     }
54   }
55 
56   override def hasNext: Boolean = sorted.hasNext
57 
58   override def next(): SimpleFeature = sorted.next()
59 
60   override def close(): Unit = {
61     if (closed.compareAndSet(false, true)) {
62       sorted.close() // also closes original features iterator
63     }
64   }
65 }

71   /**
72    * Sorts the iterator in memory
73    *
74    * @param head head of iterator
75    * @param tail tail of iterator
76    * @param ordering ordering
77    * @param closed checked to determine if we should stop sorting and return early
78    * @return
79    */
80   private def sortInMemory(
81       head: SimpleFeature,
82       tail: CloseableIterator[SimpleFeature],
83       ordering: Ordering[SimpleFeature],
84       closed: AtomicBoolean): CloseableIterator[SimpleFeature] = {
85     // use ArrayList for sort-in-place of the underlying array
86     val list = new java.util.ArrayList[SimpleFeature](100)
87     list.add(head)
88 
89     while (tail.hasNext && !closed.get) {
90       list.add(tail.next())
91     }
92 
93     if (closed.get) {
94       // don't bother sorting, just return an empty iterator
95       CloseableIterator(Iterator.empty, tail.close())
96     } else {
97       list.sort(ordering)
98       CloseableIterator(list.iterator.asScala, tail.close())
99     }
100   }
101 
102   /**
103    * Sorts the iterator in memory, but spills over to disk if the size in memory exceeds the specified threshold
104    *
105    * @param head head of iterator
106    * @param tail tail of iterator
107    * @param ordering ordering
108    * @param closed checked to determine if we should stop sorting and return early
109    * @param threshold max memory to use before spilling to disk
110    * @return
111    */
112   private def sortWithSpillover(
113       head: SimpleFeature,
114       tail: CloseableIterator[SimpleFeature],
115       ordering: Ordering[SimpleFeature],
116       closed: AtomicBoolean,
117       threshold: Long): CloseableIterator[SimpleFeature] = {
118 
119     // grouping ID for tmp files, doesn't have to be unique
120     lazy val id = {
121       val name = head.getFeatureType.getTypeName.replaceAll("[^A-Za-z0-9_-]", "").take(20)
122       f"$name-${Random.nextInt(10000)}%04d"
123     }
124 
125     val sizable: FeatureIsSizable[SimpleFeature] = head match {
126       case _: SimpleFeature with Sizable =>
127         SizableFeatureIsSizable.asInstanceOf[FeatureIsSizable[SimpleFeature]]
128       case _ =>
129         // currently, all of our uses of this class pass in Sizable features... this means we missed one
130         logger.warn(
131           s"Feature class '${head.getClass.getName}' doesn't implement Sizable - " +
132               "using estimated size for memory threshold calculations")
133         UnSizableFeatureIsSizable
134     }
135 
136     val files = ArrayBuffer.empty[File] // tmp files we create when we exceed in-memory threshold
137     lazy val serializer = KryoFeatureSerializer(head.getFeatureType, SerializationOptions.withUserData)
138     // use ArrayList for sort-in-place of the underlying array
139     val list = new java.util.ArrayList[SimpleFeature](100)
140     list.add(head)
141     var size = sizable.sizeOf(head)
142 
143     while (tail.hasNext && !closed.get) {
144       if (size >= threshold) {
145         // write out the sorted list to disk
146         list.sort(ordering)
147         val file = Files.createTempFile(s"gm-sort-$id-", ".kryo").toFile
148         files += file
149         logger.trace(s"Created temp sort file '${file.getAbsolutePath}'")
150         WithClose(new FileOutputStream(file)) { os =>
151           var i = 0
152           while (i < list.size()) {
153             val bytes = serializer.serialize(list.get(i))
154             os.write(ByteArrays.toBytes(bytes.length))
155             os.write(bytes)
156             i += 1
157           }
158         }
159         list.clear()
160         size = 0
161       }
162       val next = tail.next()
163       list.add(next)
164       size += sizable.sizeOf(next)
165     }
166 
167     if (closed.get) {
168       files.foreach { file =>
169         try {
170           if (!file.delete()) {
171             logger.warn(s"Unable to delete tmp file '${file.getAbsolutePath}''")
172             file.deleteOnExit()
173           }
174         } catch {
175           case NonFatal(e) => logger.warn(s"Unable to delete tmp file '${file.getAbsolutePath}''", e)
176         }
177       }
178       // don't bother sorting, just return an empty iterator
179       CloseableIterator(Iterator.empty, tail.close())
180     } else {
181       if (!list.isEmpty) {
182         list.sort(ordering)
183       }
184       if (files.isEmpty) {
185         CloseableIterator(list.iterator.asScala, tail.close())
186       } else {
187         new MergeSortingIterator(files.toIndexedSeq, serializer, list.iterator.asScala, tail, ordering)
188       }
189     }
190   }
191 
192   /**
193    * Does a merge sort on a group of locally sorted files and a left-over in-memory iterator
194    *
195    * @param files files to merge
196    * @param serializer serializer
197    * @param mem left-over in-memory features, already sorted
198    * @param closeable link to the original (now-empty) feature iterator, for cleaning up on close
199    * @param ordering feature ordering
200    */
201   private class MergeSortingIterator(
202       files: IndexedSeq[File],
203       serializer: SimpleFeatureSerializer,
204       mem: Iterator[SimpleFeature],
205       closeable: CloseableIterator[_],
206       ordering: Ordering[SimpleFeature]
207     ) extends CloseableIterator[SimpleFeature] {
208 
209     // list of feature iterators to merge
210     private val merging = IndexedSeq(CloseableIterator(mem)) ++ files.map(new FileIterator(_, serializer))
211 
212     // priority queue containing the next feature from each file
213     private val heads = {
214       val o = Ordering.by[(SimpleFeature, Int), SimpleFeature](_._1)(ordering)
215       val q = new java.util.PriorityQueue[(SimpleFeature, Int)](files.size + 1, o)
216       var i = 0
217       while (i < merging.length) {
218         val m = merging(i)
219         if (m.hasNext) {
220           q.add(m.next -> i)
221         }
222         i += 1
223       }
224       q
225     }
226 
227     override def hasNext: Boolean = !heads.isEmpty
228 
229     override def next(): SimpleFeature = {
230       val (f, i) = heads.poll()
231       val m = merging(i)
232       if (m.hasNext) {
233         heads.add(m.next -> i)
234       }
235       f
236     }
237 
238     override def close(): Unit = CloseQuietly.raise(merging :+ closeable)
239   }

241   /**
242    * Reads a tmp file of serialized simple features
243    *
244    * @param file file
245    * @param serializer serializer
246    */
247   private class FileIterator(file: File, serializer: SimpleFeatureSerializer)
248       extends CloseableIterator[SimpleFeature] with LazyLogging {
249 
250     private val is = new FileInputStream(file)
251     private var buf = Array.ofDim[Byte](4)
252 
253     override def hasNext: Boolean = is.available() > 0
254 
255     override def next(): SimpleFeature = {
256       is.read(buf, 0, 4)
257       val size = ByteArrays.readInt(buf)
258       if (buf.length < size) {
259         buf = Array.ofDim((size * 1.2).toInt)
260       }
261       is.read(buf, 0, size)
262       serializer.deserialize(buf, 0, size)
263     }
264 
265     override def close(): Unit = {
266       try { is.close() } finally {
267         if (!file.delete()) {
268           logger.warn(s"Unable to delete tmp file '${file.getAbsolutePath}''")
269           file.deleteOnExit()
270         }
271       }
272     }
273   }
