30 /**
31  * Visitor that tracks unique attribute values and counts
32  *
33  * @param features features to evaluate
34  * @param attributeDescriptor attribute to evaluate
35  * @param filter optional filter to apply to features before evaluating
36  * @param histogram return counts or not
37  */
38 class AttributeValuesVisitor(
39     val features: SimpleFeatureCollection,
40     val attributeDescriptor: AttributeDescriptor,
41     val filter: Option[Filter],
42     histogram: Boolean
43   ) extends GeoMesaProcessVisitor with LazyLogging {
44 
45   import org.locationtech.geomesa.utils.geotools.RichAttributeDescriptors.RichAttributeDescriptor
46 
47   import scala.collection.JavaConverters._
48 
49   private val attribute    = attributeDescriptor.getLocalName
50   private val uniqueValues = mutable.Map.empty[AnyRef, Long].withDefaultValue(0)
51 
52   private var attributeIdx: Int = -1
53 
54   // normally handled in our query planner, but we are going to use the filter directly here
55   private lazy val manualFilter = filter.map(FastFilterFactory.optimize(features.getSchema, _))
56 
57   private def getAttribute[T](f: SimpleFeature): T = {
58     if (attributeIdx == -1) {
59       attributeIdx = f.getType.indexOf(attribute)
60     }
61     f.getAttribute(attributeIdx).asInstanceOf[T]
62   }
63 
64   private def addSingularValue(f: SimpleFeature): Unit = {
65     val value = getAttribute[AnyRef](f)
66     if (value != null) {
67       uniqueValues(value) += 1
68     }
69   }
70 
71   private def addMultiValue(f: SimpleFeature): Unit = {
72     val values = getAttribute[java.util.Collection[AnyRef]](f)
73     if (values != null) {
74       values.asScala.foreach(uniqueValues(_) += 1)
75     }
76   }
77 
78   private val addValue: SimpleFeature => Unit =
79     if (attributeDescriptor.isList) { addMultiValue } else { addSingularValue }
80 
81   // non-optimized visit
82   override def visit(feature: Feature): Unit = {
83     val f = feature.asInstanceOf[SimpleFeature]
84     if (manualFilter.forall(_.evaluate(f))) {
85       addValue(f)
86     }
87   }
88 
89   override def getResult: AttributeResult = new AttributeResult(uniqueValues.toMap)
90 
91   override def execute(source: SimpleFeatureSource, query: Query): Unit = {
92 
93     import org.locationtech.geomesa.filter.mergeFilters
94 
95     logger.debug(s"Running Geomesa histogram process on source type ${source.getClass.getName}")
96 
97     // combine filters from this process and any input collection
98     filter.foreach(f => query.setFilter(mergeFilters(query.getFilter, f)))
99 
100     val sft = source.getSchema
101 
102     val enumerated = if (attributeDescriptor.isMultiValued) {
103       // stats don't support list types
104       uniqueV5(source, query)
105     } else {
106       // TODO if !histogram, we could write a new unique skipping iterator
107       query.getHints.put(QueryHints.STATS_STRING, Stat.Enumeration(attribute))
108       query.getHints.put(QueryHints.ENCODE_STATS, java.lang.Boolean.TRUE)
109 
110       // execute the query
111       val reader = source.getFeatures(query).features()
112 
113       val enumeration = try {
114         // stats should always return exactly one result, even if there are no features in the table
115         val encoded = reader.next.getAttribute(0).asInstanceOf[String]
116         StatsScan.decodeStat(sft)(encoded).asInstanceOf[EnumerationStat[AnyRef]]
117       } finally {
118         reader.close()
119       }
120 
121       enumeration.frequencies
122     }
123 
124     uniqueValues.clear()
125     enumerated.foreach { case (k, v) => uniqueValues.put(k, v) }
126   }
127 
128   private def uniqueV5(source: SimpleFeatureSource, query: Query): Iterable[(AnyRef, Long)] = {
129     import org.locationtech.geomesa.filter.ff
130 
131     // only return the attribute we are interested in to reduce bandwidth
132     query.setPropertyNames(Seq(attribute).asJava)
133 
134     // if there is no filter, try to force an attribute scan - should be the fastest query
135     if (query.getFilter == Filter.INCLUDE && AttributeIndex.indexed(features.getSchema, attribute)) {
136       // a filter that is equivalent to Filter.INCLUDE, but against the attribute index
137       query.setFilter(ff.greaterOrEqual(ff.property(attribute), ff.literal("")))
138     }
139 
140     // execute the query
141     CloseableIterator(source.getFeatures(query).features()).foreach(addValue)
142     uniqueValues.toMap
143   }
144 }

146 object AttributeValuesVisitor {
147 
148   import scala.collection.JavaConverters._
149 
150   /**
151    * Result class to hold the attribute histogram
152    *
153    * @param attributes result
154    */
155   class AttributeResult(val attributes: Map[AnyRef, Long]) extends AbstractCalcResult {
156 
157     override def getValue: java.util.Map[AnyRef, Long] = attributes.asJava
158 
159     override def isCompatible(targetResults: CalcResult): Boolean =
160       targetResults.isInstanceOf[AttributeResult] || targetResults == CalcResult.NULL_RESULT
161 
162     override def merge(resultsToAdd: CalcResult): CalcResult = {
163       if (!isCompatible(resultsToAdd)) {
164         throw new IllegalArgumentException("Parameter is not a compatible type")
165       } else if (resultsToAdd == CalcResult.NULL_RESULT) {
166         this
167       } else if (resultsToAdd.isInstanceOf[AttributeResult]) {
168         val toAdd = resultsToAdd.getValue.asInstanceOf[Map[AnyRef, Long]]
169         // note ++ on maps will get all keys with second maps values if exists, if not first map values
170         val merged = attributes ++ toAdd.map {
171           case (attr, count) => attr -> (count + attributes.getOrElse(attr, 0L))
172         }
173         new AttributeResult(merged)
174       } else {
175         throw new IllegalArgumentException(
176           "The CalcResults claim to be compatible, but the appropriate merge method has not been implemented.")
177       }
178     }
179   }
