101   /**
102     * Checks a table name for a max limit. If the table name exceeds the limit, then it will be
103     * truncated and a UUID appended. Note that if the limit is less than 34 (one char prefix,
104     * an underscore separator, and 32 chars for a UUID), this method will throw an exception
105     *
106     * @param name desired name
107     * @param limit database limit on the length of a table name
108     * @return
109     */
110   @throws[IllegalArgumentException]("Limit does not fit a UUID (34 chars)")
111   def truncateTableName(name: String, limit: Int): String = {
112     val offset = limit - 33
113     if (offset <= 0) {
114       throw new IllegalArgumentException(s"Limit is too small to fit a UUID, must be at least 34 chars: $limit")
115     }
116     s"${name.substring(0, offset)}_${UUID.randomUUID().toString.replaceAllLiterally("-", "")}"
117   }

146   /**
147    * Writes features to a particular back-end data store implementation
148    *
149    * @param indices indices being written to
150    * @param wrapper creates writable feature
151    */
152   abstract class BaseIndexWriter[T <: WritableFeature](
153       indices: Seq[GeoMesaFeatureIndex[_, _]],
154       wrapper: FeatureWrapper[T]
155     ) extends IndexWriter {
156 
157     private val converters = indices.map(_.createConverter()).toArray
158     private val values = Array.ofDim[RowKeyValue[_]](indices.length)
159     private val previousValues = Array.ofDim[RowKeyValue[_]](indices.length)
160 
161     override def append(feature: SimpleFeature): Unit = {
162       val writable = wrapper.wrap(feature)
163 
164       try {
165         var i = 0
166         // calculate all the mutations up front to ensure that there aren't any validation errors
167         while (i < converters.length) {
168           values(i) = converters(i).convert(writable)
169           i +=1
170         }
171       } catch {
172         case NonFatal(e) => throw new IllegalArgumentException("Error creating keys for insert:", e)
173       }
174 
175       append(writable, values)
176     }
177 
178     override def update(updated: SimpleFeature, previous: SimpleFeature): Unit = {
179       val writable = wrapper.wrap(updated)
180       val removable = wrapper.wrap(previous)
181 
182       try {
183         var i = 0
184         // calculate all the mutations up front to ensure that there aren't any validation errors
185         while (i < converters.length) {
186           values(i) = converters(i).convert(writable)
187           previousValues(i) = converters(i).convert(removable)
188           i +=1
189         }
190       } catch {
191         case NonFatal(e) => throw new IllegalArgumentException("Error creating keys for insert:", e)
192       }
193 
194       update(writable, values, removable, previousValues)
195     }
196 
197     override def delete(feature: SimpleFeature): Unit = {
198       val writable = wrapper.wrap(feature, delete = true)
199 
200       try {
201         var i = 0
202         // we assume that all converters will pass as this feature was already written once
203         while (i < converters.length) {
204           values(i) = converters(i).convert(writable, lenient = true)
205           i += 1
206         }
207       } catch {
208         case NonFatal(e) => throw new IllegalArgumentException("Error creating keys for delete:", e)
209       }
210 
211       delete(writable, values)
212     }
213 
214     /**
215       * Write values derived from the feature
216       *
217       * @param feature feature being written
218       * @param values derived values, one per index
219       */
220     protected def append(feature: T, values: Array[RowKeyValue[_]]): Unit
221 
222     /**
223      * Write values derived from the feature
224      *
225      * @param feature feature being written
226      * @param values derived values, one per index
227      * @param previous the previous feature being updated/replaced
228      * @param previousValues derived values for the previous feature
229      */
230     protected def update(feature: T, values: Array[RowKeyValue[_]], previous: T, previousValues: Array[RowKeyValue[_]]): Unit
231 
232     /**
233       * Delete values derived from the feature
234       *
235       * @param feature feature being deleted
236       * @param values derived values, one per index
237       */
238     protected def delete(feature: T, values: Array[RowKeyValue[_]]): Unit
239   }
