90   /**
91     * Checks a table name for a max limit. If the table name exceeds the limit, then it will be
92     * truncated and a UUID appended. Note that if the limit is less than 34 (one char prefix,
93     * an underscore separator, and 32 chars for a UUID), this method will throw an exception
94     *
95     * @param name desired name
96     * @param limit database limit on the length of a table name
97     * @return
98     */
99   @throws[IllegalArgumentException]("Limit does not fit a UUID (34 chars)")
100   def truncateTableName(name: String, limit: Int): String = {
101     val offset = limit - 33
102     if (offset <= 0) {
103       throw new IllegalArgumentException(s"Limit is too small to fit a UUID, must be at least 34 chars: $limit")
104     }
105     s"${name.substring(0, offset)}_${UUID.randomUUID().toString.replaceAllLiterally("-", "")}"
106   }

129       *

135   /**
136    * Writes features to a particular back-end data store implementation
137    *
138    * @param indices indices being written to
139    * @param wrapper creates writable feature
140    */
141   abstract class BaseIndexWriter[T <: WritableFeature](
142       indices: Seq[GeoMesaFeatureIndex[_, _]],
143       wrapper: FeatureWrapper[T]
144     ) extends IndexWriter {
145 
146     private val converters = indices.map(_.createConverter()).toArray
147     private val values = Array.ofDim[RowKeyValue[_]](indices.length)
148     private val previousValues = Array.ofDim[RowKeyValue[_]](indices.length)
149 
150     override def append(feature: SimpleFeature): Unit = {
151       val writable = wrapper.wrap(feature)
152 
153       try {
154         var i = 0
155         // calculate all the mutations up front to ensure that there aren't any validation errors
156         while (i < converters.length) {
157           values(i) = converters(i).convert(writable)
158           i +=1
159         }
160       } catch {
161         case NonFatal(e) => throw new IllegalArgumentException("Error creating keys for insert:", e)
162       }
163 
164       append(writable, values)
165     }
166 
167     override def update(updated: SimpleFeature, previous: SimpleFeature): Unit = {
168       val writable = wrapper.wrap(updated)
169       val removable = wrapper.wrap(previous)
170 
171       try {
172         var i = 0
173         // calculate all the mutations up front to ensure that there aren't any validation errors
174         while (i < converters.length) {
175           values(i) = converters(i).convert(writable)
176           previousValues(i) = converters(i).convert(removable)
177           i +=1
178         }
179       } catch {
180         case NonFatal(e) => throw new IllegalArgumentException("Error creating keys for insert:", e)
181       }
182 
183       update(writable, values, removable, previousValues)
184     }
185 
186     override def delete(feature: SimpleFeature): Unit = {
187       val writable = wrapper.wrap(feature, delete = true)
188 
189       try {
190         var i = 0
191         // we assume that all converters will pass as this feature was already written once
192         while (i < converters.length) {
193           values(i) = converters(i).convert(writable, lenient = true)
194           i += 1
195         }
196       } catch {
197         case NonFatal(e) => throw new IllegalArgumentException("Error creating keys for delete:", e)
198       }
199 
200       delete(writable, values)
201     }
202 
203     /**
204       * Write values derived from the feature
205       *
206       * @param feature feature being written
207       * @param values derived values, one per index
208       */
209     protected def append(feature: T, values: Array[RowKeyValue[_]]): Unit
210 
211     /**
212      * Write values derived from the feature
213      *
214      * @param feature feature being written
215      * @param values derived values, one per index
216      * @param previous the previous feature being updated/replaced
217      * @param previousValues derived values for the previous feature
218      */
219     protected def update(feature: T, values: Array[RowKeyValue[_]], previous: T, previousValues: Array[RowKeyValue[_]]): Unit
220 
221     /**
222       * Delete values derived from the feature
223       *
224       * @param feature feature being deleted
225       * @param values derived values, one per index
226       */
227     protected def delete(feature: T, values: Array[RowKeyValue[_]]): Unit
228   }
