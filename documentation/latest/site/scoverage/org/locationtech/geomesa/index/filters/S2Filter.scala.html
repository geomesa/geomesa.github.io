18 class S2Filter(val xy: Array[Array[Double]]) extends RowFilter {
19   override def inBounds(buf: Array[Byte], offset: Int): Boolean = {
20     val latLon = new S2CellId(ByteArrays.readLong(buf, offset)).toLatLng
21     val x = latLon.lngDegrees()
22     val y = latLon.latDegrees()
23     var i = 0
24     while (i < xy.length) {
25       val xyi = xy(i)
26       if (x >= xyi(0) && x <= xyi(2) && y >= xyi(1) && y <= xyi(3)) {
27         return true
28       }
29       i += 1
30     }
31     false
32   }
33 }
