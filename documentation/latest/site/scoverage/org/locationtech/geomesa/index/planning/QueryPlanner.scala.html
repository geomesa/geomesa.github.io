62     val plans = getQueryPlans(sft, query, original.getFilter, None, explain)

66   private def run(query: Query, plans: Seq[QueryPlan[DS]])(): CloseableIterator[SimpleFeature] = {
67     var iterator = SelfClosingIterator(plans.iterator).flatMap(p => p.scan(ds).map(p.resultsToFeatures.apply))
68 
69     if (!query.getHints.isSkipReduce) {
70       plans.headOption.flatMap(_.reducer).foreach { reducer =>
71         require(plans.tail.forall(_.reducer.contains(reducer)), "Reduce must be the same in all query plans")
72         iterator = reducer.apply(iterator)
73       }
74     }
75 
76     plans.headOption.flatMap(_.sort).foreach { sort =>
77       require(plans.tail.forall(_.sort.contains(sort)), "Sort must be the same in all query plans")
78       iterator = new SortingSimpleFeatureIterator(iterator, sort)
79     }
80 
81     plans.headOption.flatMap(_.maxFeatures).foreach { maxFeatures =>
82       require(plans.tail.forall(_.maxFeatures.contains(maxFeatures)),
83         "Max features must be the same in all query plans")
84       if (query.getHints.getReturnSft == BinaryOutputEncoder.BinEncodedSft) {
85         // bin queries pack multiple records into each feature
86         // to count the records, we have to count the total bytes coming back, instead of the number of features
87         val label = query.getHints.getBinLabelField.isDefined
88         iterator = new BinaryOutputEncoder.FeatureLimitingIterator(iterator, maxFeatures, label)
89       } else {
90         iterator = iterator.take(maxFeatures)
91       }
92     }
93 
94     plans.headOption.flatMap(_.projection).foreach { projection =>
95       require(plans.tail.forall(_.projection.contains(projection)), "Projection must be the same in all query plans")
96       val project = Reprojection(query.getHints.getReturnSft, projection)
97       iterator = iterator.map(project.apply)
98     }
99 
100     // wrap in an exceptional iterator to prevent geoserver from suppressing any exceptions
101     ExceptionalIterator(iterator)
102   }

104   /**
105     * Set up the query plans and strategies used to execute them
106     *
107     * @param sft simple feature type
108     * @param query query to plan - must have been run through `configureQuery` to set expected hints, etc
109     * @param requested override index to use for executing the query
110     * @param output planning explanation output
111     * @return
112     */
113   private def getQueryPlans(
114       sft: SimpleFeatureType,
115       query: Query,
116       original: Filter,
117       requested: Option[String],
118       output: Explainer): Seq[QueryPlan[DS]] = {
119     import org.locationtech.geomesa.filter.filterToString
120 
121     profile(time => output(s"Query planning took ${time}ms")) {
122       val hints = query.getHints
123 
124       output.pushLevel(s"Planning '${query.getTypeName}' ${filterToString(query.getFilter)}")
125       output(s"Original filter: ${filterToString(original)}")
126       output(s"Hints: bin[${hints.isBinQuery}] arrow[${hints.isArrowQuery}] density[${hints.isDensityQuery}] " +
127           s"stats[${hints.isStatsQuery}] " +
128           s"sampling[${hints.getSampling.map { case (s, f) => s"$s${f.map(":" + _).getOrElse("")}"}.getOrElse("none")}]")
129       output(s"Sort: ${query.getHints.getSortFields.map(QueryHints.sortReadableString).getOrElse("none")}")
130       output(s"Transforms: ${query.getHints.getTransformDefinition.map(t => if (t.isEmpty) { "empty" } else { t }).getOrElse("none")}")
131       output(s"Max features: ${query.getHints.getMaxFeatures.getOrElse("none")}")
132       hints.getFilterCompatibility.foreach(c => output(s"Filter compatibility: $c"))
133 
134       output.pushLevel("Strategy selection:")
135       val requestedIndex = requested.orElse(hints.getRequestedIndex)
136       val transform = query.getHints.getTransformSchema
137       val evaluation = query.getHints.getCostEvaluation
138       val strategies =
139         StrategyDecider.getFilterPlan(ds, sft, query.getFilter, transform, evaluation, requestedIndex, output)
140       output.popLevel()
141 
142       var strategyCount = 1
143       strategies.map { strategy =>
144         def complete(plan: QueryPlan[DS], time: Long): Unit = {
145           plan.explain(output)
146           output(s"Plan creation took ${time}ms").popLevel()
147         }
148 
149         output.pushLevel(s"Strategy $strategyCount of ${strategies.length}: ${strategy.index}")
150         strategyCount += 1
151         output(s"Strategy filter: $strategy")
152         profile(complete _) {
153           val qs = strategy.getQueryStrategy(hints, output)
154           // query guard hook
155           interceptors(sft).foreach(_.guard(qs).foreach(e => throw e))
156           ds.adapter.createQueryPlan(qs)
157         }
158       }
159     }
160   }

196   /**
197    * Extract and parse transforms from the query
198    *
199    * @param sft simple feature type
200    * @param query query
201    * @return
202    */
203   def extractQueryTransforms(
204       sft: SimpleFeatureType,
205       query: Query): Option[(SimpleFeatureType, Seq[Transform], String)] = {
206     // even if a transform is not specified, some queries only use a subset of attributes
207     // specify them here so that it's easier to pick the best column group later
208     def fromQueryType: Option[Seq[String]] = {
209       val hints = query.getHints
210       if (hints.isBinQuery) {
211         Some(BinAggregatingScan.propertyNames(hints, sft))
212       } else if (hints.isDensityQuery) {
213         Some(DensityScan.propertyNames(hints, sft))
214       } else if (hints.isStatsQuery) {
215         Some(StatParser.propertyNames(sft, hints.getStatsQuery))
216       } else {
217         None
218       }
219     }
220 
221     // since we do sorting on the client, just add any sort-by attributes to the transform
222     // TODO GEOMESA-2655 we should sort and then transform back to the requested props, but it's complicated...
223     def withSort(props: Array[String]): Seq[String] = {
224       val names = props.map(p => if (p.contains('=')) { p.substring(0, p.indexOf('=')) } else { p })
225       props ++ Option(query.getSortBy).toSeq.flatMap { sort =>
226         sort.flatMap(s => Option(s.getPropertyName).flatMap(p => Option(p.getPropertyName).filterNot(names.contains)))
227       }
228     }
229 
230     // ignore transforms that don't actually do anything
231     def noop(props: Seq[String]): Boolean = props == sft.getAttributeDescriptors.asScala.map(_.getLocalName)
232 
233     Option(query.getPropertyNames).map(withSort).filterNot(noop).orElse(fromQueryType).map { props =>
234       val transforms = Transforms(sft, props)
235       val schema = Transforms.schema(sft, transforms)
236       val definition = props.mkString(Transform.DefinitionDelimiter)
237       (schema, transforms, definition)
238     }
239   }

241   /**
242     * Sets query hints for sorting
243     *
244     * @param sft sft
245     * @param query query
246     */
247   def setQuerySort(sft: SimpleFeatureType, query: Query): Unit = {
248     val sortBy = query.getSortBy
249     if (sortBy != null && sortBy.nonEmpty) {
250       val hints = query.getHints
251       if (hints.isArrowQuery) {
252         if (sortBy.lengthCompare(1) > 0) {
253           throw new IllegalArgumentException("Arrow queries only support sort by a single field: " +
254               sortBy.mkString(", "))
255         } else if (sortBy.head.getPropertyName == null) {
256           throw new IllegalArgumentException("Arrow queries only support sort by properties: " +
257               sortBy.mkString(", "))
258         }
259         val field = sortBy.head.getPropertyName.getPropertyName
260         val reverse = sortBy.head.getSortOrder == SortOrder.DESCENDING
261         hints.getArrowSort.foreach { case (f, r) =>
262           if (f != field || r != reverse) {
263             throw new IllegalArgumentException(s"Query sort does not match Arrow hints sort: " +
264                 s"${sortBy.mkString(", ")} != $f:${if (r) "DESC" else "ASC"}")
265           }
266         }
267         hints.put(QueryHints.ARROW_SORT_FIELD, field)
268         hints.put(QueryHints.ARROW_SORT_REVERSE, reverse)
269       } else if (hints.isBinQuery) {
270         val dtg = hints.getBinDtgField.orElse(sft.getDtgField).orNull
271         if (dtg == null ||
272             sortBy.map(s => Option(s.getPropertyName).map(_.getPropertyName).orNull).toSeq != Seq(dtg)) {
273           throw new IllegalArgumentException("BIN queries only support sort by a date-type field: " +
274               sortBy.mkString(", "))
275         }
276         if (sortBy.head.getSortOrder == SortOrder.DESCENDING) {
277           throw new IllegalArgumentException("BIN queries only support sort in ASCENDING order: " +
278               sortBy.mkString(", "))
279         }
280         if (hints.get(QueryHints.BIN_SORT) != null && !hints.isBinSorting) {
281           throw new IllegalArgumentException("Query sort order contradicts BIN sorting hint: " +
282               sortBy.mkString(", "))
283         }
284         hints.put(QueryHints.BIN_SORT, java.lang.Boolean.TRUE)
285       } else {
286         hints.put(QueryHints.Internal.SORT_FIELDS, QueryHints.Internal.toSortHint(sortBy))
287       }
288     }
289   }
