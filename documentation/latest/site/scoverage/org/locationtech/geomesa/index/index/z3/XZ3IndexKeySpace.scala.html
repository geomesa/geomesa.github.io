60   override def toIndexKey(writable: WritableFeature,
61                           tier: Array[Byte],
62                           id: Array[Byte],
63                           lenient: Boolean): RowKeyValue[Z3IndexKey] = {
64     val geom = writable.getAttribute[Geometry](geomIndex)
65     if (geom == null) {
66       throw new IllegalArgumentException(s"Null geometry in feature ${writable.feature.getID}")
67     }
68     val envelope = geom.getEnvelopeInternal
69     // TODO support date intervals (remember to remove disjoint data check in getRanges)
70     val dtg = writable.getAttribute[Date](dtgIndex)
71     val time = if (dtg == null) { 0L } else { dtg.getTime }
72     val BinnedTime(b, t) = timeToIndex(time)
73     val xz = try {
74       sfc.index(envelope.getMinX, envelope.getMinY, t, envelope.getMaxX, envelope.getMaxY, t, lenient)
75     } catch {
76       case NonFatal(e) => throw new IllegalArgumentException(s"Invalid xz value from geometry/time: $geom,$dtg", e)
77     }
78     val shard = sharding(writable)
79 
80     // create the byte array - allocate a single array up front to contain everything
81     // ignore tier, not used here
82     val bytes = Array.ofDim[Byte](shard.length + 10 + id.length)
83 
84     if (shard.isEmpty) {
85       ByteArrays.writeShort(b, bytes, 0)
86       ByteArrays.writeLong(xz, bytes, 2)
87       System.arraycopy(id, 0, bytes, 10, id.length)
88     } else {
89       bytes(0) = shard.head // shard is only a single byte
90       ByteArrays.writeShort(b, bytes, 1)
91       ByteArrays.writeLong(xz, bytes, 3)
92       System.arraycopy(id, 0, bytes, 11, id.length)
93     }
94 
95     SingleRowKeyValue(bytes, sharing, shard, Z3IndexKey(b, xz), tier, id, writable.values)
96   }

98   override def getIndexValues(filter: Filter, explain: Explainer): XZ3IndexValues = {
99     import org.locationtech.geomesa.filter.FilterHelper._
100 
101     // standardize the two key query arguments:  polygon and date-range
102 
103     val geometries: FilterValues[Geometry] = {
104       val extracted = extractGeometries(filter, geomField, isPoints)
105       if (extracted.nonEmpty) { extracted } else { FilterValues(Seq(WholeWorldPolygon)) }
106     }
107 
108     // since we don't apply a temporal filter, we pass handleExclusiveBounds to
109     // make sure we exclude the non-inclusive endpoints of a during filter.
110     // note that this isn't completely accurate, as we only index down to the second
111     val intervals = extractIntervals(filter, dtgField, handleExclusiveBounds = true)
112 
113     explain(s"Geometries: $geometries")
114     explain(s"Intervals: $intervals")
115 
116     // disjoint geometries are ok since they could still intersect a polygon
117     if (intervals.disjoint) {
118       explain("Disjoint dates extracted, short-circuiting to empty query")
119       return XZ3IndexValues(sfc, geometries, Seq.empty, intervals, Map.empty, Seq.empty)
120     }
121 
122     // compute our ranges based on the coarse bounds for our query
123     val xy: Seq[(Double, Double, Double, Double)] = {
124       val multiplier = QueryProperties.PolygonDecompMultiplier.toInt.get
125       val bits = QueryProperties.PolygonDecompBits.toInt.get
126       geometries.values.flatMap(GeometryUtils.bounds(_, multiplier, bits))
127     }
128 
129     // calculate map of weeks to time intervals in that week
130     val timesByBin = scala.collection.mutable.Map.empty[Short, (Double, Double)]
131     val unboundedBins = Seq.newBuilder[(Short, Short)]
132 
133     def updateTime(bin: Short, lt: Double, ut: Double): Unit = {
134       val times = timesByBin.get(bin) match {
135         case None => (lt, ut)
136         case Some((min, max)) => (math.min(min, lt), math.max(max, ut))
137       }
138       timesByBin(bin) = times
139     }
140 
141     // note: intervals shouldn't have any overlaps
142     intervals.foreach { interval =>
143       val (lower, upper) = boundsToDates(interval.bounds)
144       val BinnedTime(lb, lt) = dateToIndex(lower)
145       val BinnedTime(ub, ut) = dateToIndex(upper)
146 
147       if (interval.isBoundedBothSides) {
148         if (lb == ub) {
149           updateTime(lb, lt, ut)
150         } else {
151           updateTime(lb, lt, sfc.zBounds._2)
152           updateTime(ub, sfc.zBounds._1, ut)
153           Range.inclusive(lb + 1, ub - 1).foreach(b => timesByBin(b.toShort) = sfc.zBounds)
154         }
155       } else if (interval.lower.value.isDefined) {
156         updateTime(lb, lt, sfc.zBounds._2)
157         unboundedBins += (((lb + 1).toShort, Short.MaxValue))
158       } else if (interval.upper.value.isDefined) {
159         updateTime(ub, sfc.zBounds._1, ut)
160         unboundedBins += ((0, (ub - 1).toShort))
161       }
162     }
163 
164     // make our underlying index values available to other classes in the pipeline for processing
165     XZ3IndexValues(sfc, geometries, xy, intervals, timesByBin.toMap, unboundedBins.result())
166   }

168   override def getRanges(values: XZ3IndexValues, multiplier: Int): Iterator[ScanRange[Z3IndexKey]] = {
169     val XZ3IndexValues(sfc, geoms, xy, intervals, timesByBin, unboundedBins) = values
170 
171     if (geoms.disjoint || intervals.disjoint) {
172       Iterator.empty
173     } else if (timesByBin.isEmpty && unboundedBins.isEmpty) {
174       Iterator.single(UnboundedRange(null))
175     } else {
176       // note: `target` will always be Some, as ScanRangesTarget has a default value
177       val target = QueryProperties.ScanRangesTarget.option.map { t =>
178         math.max(1, if (timesByBin.isEmpty) { t.toInt } else { t.toInt / timesByBin.size } / multiplier)
179       }
180 
181       def toZRanges(t: (Double, Double)): Seq[IndexRange] =
182         sfc.ranges(xy.map { case (xmin, ymin, xmax, ymax) => (xmin, ymin, t._1, xmax, ymax, t._2) }, target)
183 
184       lazy val wholePeriodRanges = toZRanges(sfc.zBounds)
185 
186       val bounded = timesByBin.iterator.flatMap { case (bin, times) =>
187         val zs = if (times.eq(sfc.zBounds)) { wholePeriodRanges } else { toZRanges(times) }
188         zs.map(r => BoundedRange(Z3IndexKey(bin, r.lower), Z3IndexKey(bin, r.upper)))
189       }
190 
191       val unbounded = unboundedBins.iterator.map {
192         case (lower, Short.MaxValue) => LowerBoundedRange(Z3IndexKey(lower, 0L))
193         case (0, upper)              => UpperBoundedRange(Z3IndexKey(upper, Long.MaxValue))
194         case (lower, upper) =>
195           logger.error(s"Unexpected unbounded bin endpoints: $lower:$upper")
196           UnboundedRange(Z3IndexKey(0, 0L))
197       }
198 
199       bounded ++ unbounded
200     }
201   }

203   override def getRangeBytes(ranges: Iterator[ScanRange[Z3IndexKey]], tier: Boolean): Iterator[ByteRange] = {
204     if (sharding.length == 0) {
205       ranges.map {
206         case BoundedRange(lo, hi) =>
207           BoundedByteRange(ByteArrays.toBytes(lo.bin, lo.z), ByteArrays.toBytesFollowingPrefix(hi.bin, hi.z))
208 
209         case LowerBoundedRange(lo) =>
210           BoundedByteRange(ByteArrays.toBytes(lo.bin, lo.z), ByteRange.UnboundedUpperRange)
211 
212         case UpperBoundedRange(hi) =>
213           BoundedByteRange(ByteRange.UnboundedLowerRange, ByteArrays.toBytesFollowingPrefix(hi.bin, hi.z))
214 
215         case UnboundedRange(_) =>
216           BoundedByteRange(ByteRange.UnboundedLowerRange, ByteRange.UnboundedUpperRange)
217 
218         case r =>
219           throw new IllegalArgumentException(s"Unexpected range type $r")
220       }
221     } else {
222       ranges.flatMap {
223         case BoundedRange(lo, hi) =>
224           val lower = ByteArrays.toBytes(lo.bin, lo.z)
225           val upper = ByteArrays.toBytesFollowingPrefix(hi.bin, hi.z)
226           sharding.shards.map(p => BoundedByteRange(ByteArrays.concat(p, lower), ByteArrays.concat(p, upper)))
227 
228         case LowerBoundedRange(lo) =>
229           val lower = ByteArrays.toBytes(lo.bin, lo.z)
230           val upper = ByteRange.UnboundedUpperRange
231           sharding.shards.map(p => BoundedByteRange(ByteArrays.concat(p, lower), ByteArrays.concat(p, upper)))
232 
233         case UpperBoundedRange(hi) =>
234           val lower = ByteRange.UnboundedLowerRange
235           val upper = ByteArrays.toBytesFollowingPrefix(hi.bin, hi.z)
236           sharding.shards.map(p => BoundedByteRange(ByteArrays.concat(p, lower), ByteArrays.concat(p, upper)))
237 
238         case UnboundedRange(_) =>
239           Seq(BoundedByteRange(ByteRange.UnboundedLowerRange, ByteRange.UnboundedUpperRange))
240 
241         case r =>
242           throw new IllegalArgumentException(s"Unexpected range type $r")
243       }
244     }
