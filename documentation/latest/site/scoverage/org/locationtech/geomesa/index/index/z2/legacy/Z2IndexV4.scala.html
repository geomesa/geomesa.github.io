38 object Z2IndexV4 {
39 
40   class Z2IndexKeySpaceV4(sft: SimpleFeatureType,
41                           override val sharing: Array[Byte],
42                           sharding: ShardStrategy,
43                           geomField: String) extends Z2IndexKeySpace(sft, sharding, geomField) {
44 
45     private val rangePrefixes = {
46       if (sharding.length == 0 && sharing.isEmpty) {
47         Seq.empty
48       } else if (sharing.isEmpty) {
49         sharding.shards
50       } else if (sharding.length == 0) {
51         Seq(sharing)
52       } else {
53         sharding.shards.map(ByteArrays.concat(sharing, _))
54       }
55     }
56 
57     override val indexKeyByteLength: Right[(Array[Byte], Int, Int) => Int, Int] =
58       Right(8 + sharing.length + sharding.length)
59 
60     override def toIndexKey(writable: WritableFeature,
61                             tier: Array[Byte],
62                             id: Array[Byte],
63                             lenient: Boolean): RowKeyValue[Long] = {
64       val geom = writable.getAttribute[Point](geomIndex)
65       if (geom == null) {
66         throw new IllegalArgumentException(s"Null geometry in feature ${writable.feature.getID}")
67       }
68       val z = try { sfc.index(geom.getX, geom.getY, lenient) } catch {
69         case NonFatal(e) => throw new IllegalArgumentException(s"Invalid z value from geometry: $geom", e)
70       }
71       val shard = sharding(writable)
72 
73       // create the byte array - allocate a single array up front to contain everything
74       // ignore tier, not used here
75       val bytes = Array.ofDim[Byte](sharing.length + shard.length + 8 + id.length)
76       var i = 0
77       if (!sharing.isEmpty) {
78         bytes(0) = sharing.head // sharing is only a single byte
79         i += 1
80       }
81       if (!shard.isEmpty) {
82         bytes(i) = shard.head // shard is only a single byte
83         i += 1
84       }
85       ByteArrays.writeLong(z, bytes, i)
86       System.arraycopy(id, 0, bytes, i + 8, id.length)
87 
88       SingleRowKeyValue(bytes, sharing, shard, z, tier, id, writable.values)
89     }
90 
91     override def getRangeBytes(ranges: Iterator[ScanRange[Long]], tier: Boolean): Iterator[ByteRange] = {
92       if (rangePrefixes.isEmpty) {
93         ranges.map {
94           case BoundedRange(lo, hi) => BoundedByteRange(ByteArrays.toBytes(lo), ByteArrays.toBytesFollowingPrefix(hi))
95           case r => throw new IllegalArgumentException(s"Unexpected range type $r")
96         }
97       } else {
98         ranges.flatMap {
99           case BoundedRange(lo, hi) =>
100             val lower = ByteArrays.toBytes(lo)
101             val upper = ByteArrays.toBytesFollowingPrefix(hi)
102             rangePrefixes.map(p => BoundedByteRange(ByteArrays.concat(p, lower), ByteArrays.concat(p, upper)))
103 
104           case r => throw new IllegalArgumentException(s"Unexpected range type $r")
105         }
106       }
107     }
108   }
