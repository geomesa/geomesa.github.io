28 /**
29   * Runnable stats implementation, doesn't persist stat values. DataStore needs to be able to handle stat
30   * query hints
31   *
32   * @param ds datastore
33   */
34 class RunnableStats(ds: DataStore) extends GeoMesaStats with LazyLogging {
35 
36   override val writer: GeoMesaStatWriter = NoopStatWriter
37 
38   override def getCount(
39       sft: SimpleFeatureType,
40       filter: Filter,
41       exact: Boolean,
42       queryHints: Hints): Option[Long] = {
43     if (exact) { query[CountStat](sft, filter, Stat.Count(), queryHints).map(_.count) } else { None }
44   }
45 
46   override def getMinMax[T](
47       sft: SimpleFeatureType,
48       attribute: String,
49       filter: Filter,
50       exact: Boolean): Option[MinMax[T]] = {
51     if (exact) { query[MinMax[T]](sft, filter, Stat.MinMax(attribute)) } else { None }
52   }
53 
54   override def getEnumeration[T](
55       sft: SimpleFeatureType,
56       attribute: String,
57       filter: Filter,
58       exact: Boolean): Option[EnumerationStat[T]] = {
59     if (exact) { query[EnumerationStat[T]](sft, filter, Stat.Enumeration(attribute)) } else { None }
60   }
61 
62   override def getFrequency[T](
63       sft: SimpleFeatureType,
64       attribute: String,
65       precision: Int,
66       filter: Filter,
67       exact: Boolean): Option[Frequency[T]] = {
68     if (exact) { query[Frequency[T]](sft, filter, Stat.Frequency(attribute, precision)) } else { None }
69   }
70 
71   override def getTopK[T](
72       sft: SimpleFeatureType,
73       attribute: String,
74       filter: Filter,
75       exact: Boolean): Option[TopK[T]] = {
76     if (exact) { query[TopK[T]](sft, filter, Stat.TopK(attribute)) } else { None }
77   }
78 
79   override def getHistogram[T](
80       sft: SimpleFeatureType,
81       attribute: String,
82       bins: Int,
83       min: T,
84       max: T,
85       filter: Filter,
86       exact: Boolean): Option[Histogram[T]] = {
87     if (exact) {
88       this.query[Histogram[T]](sft, filter, Stat.Histogram(attribute, bins, min, max)(ClassTag(min.getClass)))
89     } else {
90       None
91     }
92   }
93 
94   override def getZ3Histogram(
95       sft: SimpleFeatureType,
96       geom: String,
97       dtg: String,
98       period: TimePeriod,
99       bins: Int,
100       filter: Filter,
101       exact: Boolean): Option[Z3Histogram] = {
102     if (exact) { query(sft, filter, Stat.Z3Histogram(geom, dtg, period, bins)) } else { None }
103   }
104 
105   override def getStat[T <: Stat](
106       sft: SimpleFeatureType,
107       query: String,
108       filter: Filter,
109       exact: Boolean): Option[T] = {
110     if (exact) { this.query(sft, filter, query) } else { None }
111   }
112 
113   override def close(): Unit = {}
114 
115   /**
116     * Run an exact query against the data store
117     *
118     * @param sft simple feature type
119     * @param filter filter
120     * @param query stat string
121     * @tparam T stat type
122     * @return
123     */
124   protected def query[T <: Stat](sft: SimpleFeatureType, filter: Filter, query: String, queryHints: Hints = new Hints()): Option[T] = {
125     val q = new Query(sft.getTypeName, filter)
126     q.setHints(queryHints)
127     q.getHints.put(QueryHints.STATS_STRING, query)
128     q.getHints.put(QueryHints.ENCODE_STATS, java.lang.Boolean.TRUE)
129 
130     try {
131       WithClose(ds.getFeatureReader(q, Transaction.AUTO_COMMIT)) { reader =>
132         // stats should always return exactly one result, even if there are no features in the table
133         val result = if (reader.hasNext) { reader.next.getAttribute(0).asInstanceOf[String] } else {
134           throw new IllegalStateException("Stats scan didn't return any rows")
135         }
136         Some(StatsScan.decodeStat(sft)(result).asInstanceOf[T])
137       }
138     } catch {
139       case NonFatal(e) =>
140         logger.error(s"Error running stats query with stats '$query' and filter '${filterToString(filter)}'", e)
141         None
142     }
143   }
144 }

148   /**
149     * Allows for running of stats against data stores that don't support stat queries
150     *
151     * @param ds datastore
152     */
153   class UnoptimizedRunnableStats(ds: DataStore) extends RunnableStats(ds) {
154     override protected def query[T <: Stat](sft: SimpleFeatureType, filter: Filter, query: String, queryHints: Hints): Option[T] = {
155       try {
156         val q = new Query(sft.getTypeName, filter, StatParser.propertyNames(sft, query): _*)
157         q.setHints(queryHints)
158         WithClose(ds.getFeatureReader(q, Transaction.AUTO_COMMIT)) { reader =>
159           val stat = Stat(reader.getFeatureType, query).asInstanceOf[T]
160           CloseableIterator(reader).foreach(stat.observe)
161           Some(stat)
162         }
163       } catch {
164         case NonFatal(e) =>
165           logger.error(s"Error running stats query with stats '$query' and filter '${filterToString(filter)}'", e)
166           None
167       }
168     }
169   }
