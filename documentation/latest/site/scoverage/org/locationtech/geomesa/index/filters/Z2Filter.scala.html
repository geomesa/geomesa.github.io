22   override def inBounds(buf: Array[Byte], offset: Int): Boolean = {
23     val z = ByteArrays.readLong(buf, offset)
24     val x = Z2(z).d0
25     val y = Z2(z).d1
26     var i = 0
27     while (i < xy.length) {
28       val xyi = xy(i)
29       if (x >= xyi(0) && x <= xyi(2) && y >= xyi(1) && y <= xyi(3)) {
30         return true
31       }
32       i += 1
33     }
34     false
35   }
