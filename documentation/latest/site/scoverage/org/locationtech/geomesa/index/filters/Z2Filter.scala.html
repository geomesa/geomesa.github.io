22   override def inBounds(buf: Array[Byte], offset: Int): Boolean = {
23     val z = ByteArrays.readLong(buf, offset)

25     val y = Z2(z).d1

27     while (i < xy.length) {
28       val xyi = xy(i)

30         return true

35   }
