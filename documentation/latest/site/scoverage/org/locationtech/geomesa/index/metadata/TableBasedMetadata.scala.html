27 /**
28   * Metadata persisted in a database table. The underlying table will be lazily created when required.
29   * Metadata values are cached with a configurable timeout to save repeated database reads.
30   *
31   * @tparam T type param
32   */
33 trait TableBasedMetadata[T] extends GeoMesaMetadata[T] with LazyLogging {
34 
35   import scala.collection.JavaConverters._
36 
37   /**
38     * Serializer
39     *
40     * @return
41     */
42   def serializer: MetadataSerializer[T]
43 
44   /**
45     * Checks if the underlying table exists
46     *
47     * @return
48     */
49   protected def checkIfTableExists: Boolean
50 
51   /**
52     * Creates the underlying table. Must be idempotent.
53     */
54   protected def createTable(): Unit
55 
56   /**
57     * Create an instance to use for backup
58     *
59     * @param timestamp formatted timestamp for the current time
60     * @return
61     */
62   protected def createEmptyBackup(timestamp: String): TableBasedMetadata[T]
63 
64   /**
65     * Writes key/value pairs
66     *
67     * @param rows keys/values
68     */
69   protected def write(typeName: String, rows: Seq[(String, Array[Byte])]): Unit

71   /**
72     * Deletes multiple rows
73     *
74     * @param typeName simple feature type name
75     * @param keys keys
76     */
77   protected def delete(typeName: String, keys: Seq[String]): Unit

79   /**
80     * Reads a value from the underlying table

88   /**
89     * Reads row keys from the underlying table

93     * @return matching tuples of (key, value)
94     */
95   protected def scanValues(typeName: String, prefix: String = ""): CloseableIterator[(String, Array[Byte])]
96 
97   /**
98     * Reads all row keys from the underlying table
99     *
100     * @return matching tuples of (typeName, key)
101     */
102   protected def scanKeys(): CloseableIterator[(String, String)]

104   private val expiry = TableBasedMetadata.Expiry.toDuration.get.toMillis
105 
106   private val tableExists = new TableExists().init()
107 
108   // cache for our metadata - invalidate every 10 minutes so we keep things current
109   private val metaDataCache: LoadingCache[(String, String), Option[T]] =
110     Caffeine.newBuilder().expireAfterWrite(expiry, TimeUnit.MILLISECONDS).build(
111       new CacheLoader[(String, String), Option[T]] {
112         override def load(typeNameAndKey: (String, String)): Option[T] = {
113           if (!tableExists.get) { None } else {
114             val (typeName, key) = typeNameAndKey
115             scanValue(typeName, key).map(serializer.deserialize(typeName, _))
116           }
117         }
118       }
119     )
120 
121   // keep a separate cache for scan queries vs point lookups, so that the point lookups don't cache
122   // partial values for a scan result
123   private val metaDataScanCache: LoadingCache[(String, String), Seq[(String, T)]] =
124     Caffeine.newBuilder().expireAfterWrite(expiry, TimeUnit.MILLISECONDS).build(
125       new CacheLoader[(String, String), Seq[(String, T)]] {
126         override def load(typeNameAndPrefix: (String, String)): Seq[(String, T)] = {
127           if (!tableExists.get) { Seq.empty } else {
128             val (typeName, prefix) = typeNameAndPrefix
129             WithClose(scanValues(typeName, prefix)) { iter =>
130               val builder = Seq.newBuilder[(String, T)]
131               iter.foreach { case (k, v) => builder += k -> serializer.deserialize(typeName, v) }
132               builder.result()
133             }
134           }
135         }
136       }
137     )
138 
139   private lazy val formatter =
140     new DateTimeFormatterBuilder()
141       .parseCaseInsensitive()
142       .appendValue(ChronoField.YEAR, 4)
143       .appendValue(ChronoField.MONTH_OF_YEAR, 2)
144       .appendValue(ChronoField.DAY_OF_MONTH, 2)
145       .appendLiteral('T')
146       .appendValue(ChronoField.HOUR_OF_DAY, 2)
147       .appendValue(ChronoField.MINUTE_OF_HOUR, 2)
148       .appendValue(ChronoField.SECOND_OF_MINUTE, 2)
149       .toFormatter(Locale.US)
150       .withZone(ZoneOffset.UTC)

152   override def getFeatureTypes: Array[String] = {
153     if (!tableExists.get) { Array.empty } else {
154       WithClose(scanKeys()) { keys =>
155         keys.collect { case (typeName, key) if key == GeoMesaMetadata.AttributesKey => typeName }.toArray
156       }
157     }
158   }
159 
160   override def read(typeName: String, key: String, cache: Boolean): Option[T] = {
161     if (!cache) {
162       metaDataCache.invalidate((typeName, key))
163     }
164     metaDataCache.get((typeName, key))
165   }
166 
167   override def scan(typeName: String, prefix: String, cache: Boolean): Seq[(String, T)] = {
168     if (!cache) {
169       metaDataScanCache.invalidate((typeName, prefix))
170     }
171     metaDataScanCache.get((typeName, prefix))
172   }
173 
174   override def insert(typeName: String, key: String, value: T): Unit = insert(typeName, Map(key -> value))
175 
176   override def insert(typeName: String, kvPairs: Map[String, T]): Unit = {
177     ensureTableExists()
178     val strings = kvPairs.map { case (k, v) =>
179       metaDataCache.put((typeName, k), Option(v)) // note: side effect in map
180       (k, serializer.serialize(typeName, v))
181     }
182     write(typeName, strings.toSeq)
183     invalidate(metaDataScanCache, typeName)
184   }
185 
186   override def invalidateCache(typeName: String, key: String): Unit = {
187     metaDataCache.invalidate((typeName, key))
188     invalidate(metaDataScanCache, typeName)
189   }
190 
191   override def remove(typeName: String, key: String): Unit = remove(typeName, Seq(key))
192 
193   override def remove(typeName: String, keys: Seq[String]): Unit = {
194     if (tableExists.get()) {
195       delete(typeName, keys)
196       // also remove from the cache
197       keys.foreach(k => metaDataCache.invalidate((typeName, k)))
198       invalidate(metaDataScanCache, typeName)
199     } else {
200       logger.debug(s"Trying to delete '$typeName: ${keys.mkString(", ")}' but table does not exist")
201     }
202   }
203 
204   override def delete(typeName: String): Unit = {
205     if (tableExists.get()) {
206       WithClose(scanValues(typeName)) { rows =>
207         if (rows.nonEmpty) {
208           delete(typeName, rows.map(_._1).toSeq)
209         }
210       }
211     } else {
212       logger.debug(s"Trying to delete type '$typeName' but table does not exist")
213     }
214     Seq(metaDataCache, metaDataScanCache).foreach(invalidate(_, typeName))
215   }
216 
217   override def backup(typeName: String): Unit = {
218     if (tableExists.get()) {
219       WithClose(scanValues(typeName)) { rows =>
220         if (rows.nonEmpty) {
221           WithClose(createEmptyBackup(DateParsing.formatInstant(Instant.now, formatter))) { metadata =>
222             metadata.ensureTableExists()
223             metadata.write(typeName, rows.toSeq)
224           }
225         }
226       }
227     } else {
228       logger.debug(s"Trying to back up type '$typeName' but table does not exist")
229     }
230   }
231 
232   /**
233    * Checks that the table is already created, and creates it if not
234    */
235   def ensureTableExists(): Unit = tableExists.createIfNeeded()
236 
237   override def resetCache(): Unit = {
238     tableExists.run()
239     metaDataCache.invalidateAll()
240     metaDataScanCache.invalidateAll()
241   }
242 
243   override def close(): Unit = tableExists.close()
244 
245   /**
246     * Invalidate all keys for the given feature type
247     *
248     * @param cache cache to invalidate
249     * @param typeName feature type name
250     */
251   private def invalidate(cache: Cache[(String, String), _], typeName: String): Unit = {
252     cache.asMap.asScala.keys.foreach { k =>
253       if (k._1 == typeName) {
254         cache.invalidate(k)
255       }
256     }
257   }
258 
259   /**
260    * Tracks existence of the metadata table
261    */
262   private class TableExists extends Runnable with Closeable {
263 
264     @volatile
265     private var exists = false
266     private var es: ScheduledExecutorService = _
267     private var future: ScheduledFuture[_] = _
268 
269     /**
270      * Initialization
271      *
272      * @return
273      */
274     def init(): TableExists = { run(); this }
275 
276     /**
277      * Does the table exist
278      *
279      * @return
280      */
281     def get(): Boolean = exists
282 
283     /**
284      * Create the table if it doesn't exist.
285      *
286      * Note that we don't synchronize this method, but creating the table should be idempotent
287      */
288     def createIfNeeded(): Unit = {
289       if (!exists) {
290         createTable()
291         run()
292       }
293     }
294 
295     /**
296      * Check for table existence
297      */
298     override def run(): Unit = {
299       exists = checkIfTableExists
300       if (exists) {
301         cancel()
302       } else {
303         schedule()
304       }
305     }
306 
307     override def close(): Unit = cancel()
308 
309     private def schedule(): Unit = synchronized {
310       if (es == null) {
311         es = ExitingExecutor(new ScheduledThreadPoolExecutor(1))
312       }
313       if (future == null) {
314         future = es.scheduleAtFixedRate(this, expiry, expiry, TimeUnit.MILLISECONDS)
315       }
316     }
317 
318     /**
319      * Cancel any scheduled tasks
320      */
321     private def cancel(): Unit = synchronized {
322       if (future != null) {
323         Try(future.cancel(true))
324         future = null
325       }
326       if (es != null) {
327         CloseWithLogging(es)
328         es = null
329       }
330     }
331   }
332 }
