24 /**
25  * Visitor for calculating stats
26  *
27  * @param features features
28  * @param statString stats to calculate
29  * @param encode encode the result as binary, vs return as json
30  * @param properties transform properties, or null for no transform
31  */
32 class StatsVisitor(features: SimpleFeatureCollection, statString: String, encode: Boolean, properties: Array[String])
33   extends GeoMesaProcessVisitor with LazyLogging {
34 
35   private val origSft = features.getSchema
36 
37   private lazy val transformDefinitions = Transforms(origSft, properties)
38   private lazy val transformSft = Transforms.schema(origSft, transformDefinitions)
39   private lazy val transformSf = TransformSimpleFeature(transformSft, transformDefinitions)
40 
41   private lazy val statSft = if (properties == null) { origSft } else { transformSft }
42 
43   private lazy val stat: Stat = Stat(statSft, statString)
44 
45   private var resultCalc: FeatureResult = _
46 
47   // non-optimized visit
48   override def visit(feature: Feature): Unit = {
49     val sf = feature.asInstanceOf[SimpleFeature]
50     if (properties != null) {
51       // There are transforms!
52       transformSf.setFeature(sf)
53       stat.observe(transformSf)
54     } else {
55       stat.observe(sf)
56     }
57   }
58 
59   override def getResult: FeatureResult = {
60     if (resultCalc != null) {
61       resultCalc
62     } else {
63       val stats = if (encode) {
64         StatsScan.encodeStat(statSft)(stat)
65       } else {
66         stat.toJson
67       }
68 
69       val sf = new ScalaSimpleFeature(StatsScan.StatsSft, "", Array(stats, GeometryUtils.zeroPoint))
70       val manualVisitResults = new ListFeatureCollection(StatsScan.StatsSft)
71       manualVisitResults.add(sf)
72       FeatureResult(manualVisitResults)
73     }
74   }
75 
76   override def execute(source: SimpleFeatureSource, query: Query): Unit = {
77     logger.debug(s"Running Geomesa stats iterator process on source type ${source.getClass.getName}")
78 
79     if (properties != null) {
80       if (query.getProperties != Query.ALL_PROPERTIES) {
81         logger.warn(s"Overriding inner query's properties (${query.getProperties}) " +
82           s"with properties/transforms ${properties.mkString(",")}.")
83       }
84       query.setPropertyNames(properties: _*)
85     }
86     query.getHints.put(QueryHints.STATS_STRING, statString)
87     query.getHints.put(QueryHints.ENCODE_STATS, Boolean.box(encode))
88     resultCalc = FeatureResult(source.getFeatures(query))
89   }
90 }
