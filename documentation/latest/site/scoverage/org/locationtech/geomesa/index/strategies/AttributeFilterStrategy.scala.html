23 trait AttributeFilterStrategy[T, U] extends GeoMesaFeatureIndex[T, U] {
24 
25   import AttributeFilterStrategy.isTemporal
26   import org.locationtech.geomesa.utils.geotools.RichAttributeDescriptors.RichAttributeDescriptor
27 
28   private val Seq(attribute, tiered @ _*) = attributes
29   private val descriptor = sft.getDescriptor(attribute)
30   private val isList = descriptor.isList
31   private val binding = if (isList) { descriptor.getListType() } else { descriptor.getType.getBinding }
32 
33   override def getFilterStrategy(filter: Filter, hints: Hints): Option[FilterStrategy] = {
34 
35     val (primary, secondary) =
36       FilterExtractingVisitor(filter, attribute, sft, AttributeFilterStrategy.attributeCheck(sft))
37 
38     primary.flatMap { extracted =>
39       val bounds = FilterHelper.extractAttributeBounds(extracted, attribute, binding)
40       if (bounds.isEmpty) { None } else {
41         val isEquals = bounds.precise && bounds.forall(_.isEquals)
42         lazy val secondaryFilterAttributes = tiered.union(secondary.toSeq.flatMap(FilterHelper.propertyNames))
43         val temporal = isTemporal(sft, attribute) || (isEquals && secondaryFilterAttributes.exists(isTemporal(sft, _)))
44         val basePriority =
45           if (isEquals || (bounds.disjoint && !isList)) {
46             1f // TODO account for secondary index
47           } else if (!bounds.forall(_.isBounded)) {
48             1000f // not null
49           } else {
50             2.5f // range query
51           }
52         // prioritize attributes based on cardinality hint
53         val priority = descriptor.getCardinality() match {
54           case Cardinality.HIGH    => basePriority / 10f
55           case Cardinality.UNKNOWN => basePriority
56           case Cardinality.LOW     => basePriority * 10f
57         }
58         Some(FilterStrategy(this, primary, secondary, temporal, priority, hints))
59       }
60     }
61   }
62 }

64 object AttributeFilterStrategy {
65 
66   /**
67     * Checks for attribute filters that we can satisfy using the attribute index strategy
68     *
69     * @param filter filter to evaluate
70     * @return true if we can process it as an attribute query
71     */
72   def attributeCheck(sft: SimpleFeatureType)(filter: Filter): Boolean = {
73     filter match {
74       case _: And | _: Or => true // note: implies further processing of children
75       case _: PropertyIsEqualTo => true
76       case _: PropertyIsBetween => true
77       case _: PropertyIsGreaterThan | _: PropertyIsLessThan => true
78       case _: PropertyIsGreaterThanOrEqualTo | _: PropertyIsLessThanOrEqualTo => true
79       case _: During |  _: Before | _: After | _: TEquals => true
80       case _: PropertyIsNull => true // we need this to be able to handle 'not null'
81       case f: PropertyIsLike => isStringProperty(sft, f.getExpression) && likeEligible(f)
82       case f: Not => f.getFilter.isInstanceOf[PropertyIsNull]
83       case _ => false
84     }
85   }
86 
87   def isStringProperty(sft: SimpleFeatureType, e: Expression): Boolean = e match {
88     case p: PropertyName => Option(sft.getDescriptor(p.getPropertyName)).exists(_.getType.getBinding == classOf[String])
89     case _ => false
90   }
91 
92   private def isTemporal(sft: SimpleFeatureType, attribute: String): Boolean =
93     classOf[Date].isAssignableFrom(sft.getDescriptor(attribute).getType.getBinding)
