22 trait AttributeFilterStrategy[T, U] extends GeoMesaFeatureIndex[T, U] {
23 
24   import AttributeFilterStrategy.isTemporal
25   import org.locationtech.geomesa.utils.geotools.RichAttributeDescriptors.RichAttributeDescriptor
26 
27   private val Seq(attribute, tiered @ _*) = attributes
28   private val descriptor = sft.getDescriptor(attribute)
29   private val isList = descriptor.isList
30   private val binding = if (isList) { descriptor.getListType() } else { descriptor.getType.getBinding }
31 
32   override def getFilterStrategy(filter: Filter, transform: Option[SimpleFeatureType]): Option[FilterStrategy] = {
33 
34     val (primary, secondary) =
35       FilterExtractingVisitor(filter, attribute, sft, AttributeFilterStrategy.attributeCheck(sft))
36 
37     primary.flatMap { extracted =>
38       val bounds = FilterHelper.extractAttributeBounds(extracted, attribute, binding)
39       if (bounds.isEmpty) { None } else {
40         val isEquals = bounds.precise && bounds.forall(_.isEquals)
41         lazy val secondaryFilterAttributes = tiered.union(secondary.toSeq.flatMap(FilterHelper.propertyNames))
42         val temporal = isTemporal(sft, attribute) || (isEquals && secondaryFilterAttributes.exists(isTemporal(sft, _)))
43         val basePriority =
44           if (isEquals || (bounds.disjoint && !isList)) {
45             1f // TODO account for secondary index
46           } else if (!bounds.forall(_.isBounded)) {
47             1000f // not null
48           } else {
49             2.5f // range query
50           }
51         // prioritize attributes based on cardinality hint
52         val priority = descriptor.getCardinality() match {
53           case Cardinality.HIGH    => basePriority / 10f
54           case Cardinality.UNKNOWN => basePriority
55           case Cardinality.LOW     => basePriority * 10f
56         }
57         Some(FilterStrategy(this, primary, secondary, temporal, priority))
58       }
59     }
60   }
61 }

63 object AttributeFilterStrategy {
64 
65   /**
66     * Checks for attribute filters that we can satisfy using the attribute index strategy
67     *
68     * @param filter filter to evaluate
69     * @return true if we can process it as an attribute query
70     */
71   def attributeCheck(sft: SimpleFeatureType)(filter: Filter): Boolean = {
72     filter match {
73       case _: And | _: Or => true // note: implies further processing of children
74       case _: PropertyIsEqualTo => true
75       case _: PropertyIsBetween => true
76       case _: PropertyIsGreaterThan | _: PropertyIsLessThan => true
77       case _: PropertyIsGreaterThanOrEqualTo | _: PropertyIsLessThanOrEqualTo => true
78       case _: During |  _: Before | _: After | _: TEquals => true
79       case _: PropertyIsNull => true // we need this to be able to handle 'not null'
80       case f: PropertyIsLike => isStringProperty(sft, f.getExpression) && likeEligible(f)
81       case f: Not => f.getFilter.isInstanceOf[PropertyIsNull]
82       case _ => false
83     }
84   }
85 
86   def isStringProperty(sft: SimpleFeatureType, e: Expression): Boolean = e match {
87     case p: PropertyName => Option(sft.getDescriptor(p.getPropertyName)).exists(_.getType.getBinding == classOf[String])
88     case _ => false
89   }
90 
91   private def isTemporal(sft: SimpleFeatureType, attribute: String): Boolean =
92     classOf[Date].isAssignableFrom(sft.getDescriptor(attribute).getType.getBinding)
