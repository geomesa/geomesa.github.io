97   override def toIndexKey(
98       writable: WritableFeature,
99       tier: Array[Byte],
100       id: Array[Byte],
101       lenient: Boolean): RowKeyValue[S3IndexKey] = {
102     val geom = writable.getAttribute[Point](geomIndex)
103     if (geom == null) {
104       throw new IllegalArgumentException(s"Null geometry in feature ${writable.feature.getID}")
105     }
106     val dtg = writable.getAttribute[Date](dtgIndex)
107     val time = if (dtg == null) { 0 } else { dtg.getTime }
108     val BinnedTime(b, t) = timeToIndex(time)
109     val s = try { sfc.index(geom.getX, geom.getY, lenient) } catch {
110       case NonFatal(e) => throw new IllegalArgumentException(s"Invalid s value from geometry/time: $geom,$dtg", e)
111     }
112     val shard = sharding(writable)
113 
114     // create the byte array - allocate a single array up front to contain everything
115     // ignore tier, not used here
116     val bytes = Array.ofDim[Byte](shard.length + 14 + id.length)
117 
118     if (shard.isEmpty) {
119       ByteArrays.writeShort(b, bytes)
120       ByteArrays.writeLong(s, bytes, 2)
121       ByteArrays.writeInt(t.toInt, bytes, 10)
122       System.arraycopy(id, 0, bytes, 14, id.length)
123     } else {
124       bytes(0) = shard.head // shard is only a single byte
125       ByteArrays.writeShort(b, bytes, 1)
126       ByteArrays.writeLong(s, bytes, 3)
127       ByteArrays.writeInt(t.toInt, bytes, 11)
128       System.arraycopy(id, 0, bytes, 15, id.length)
129     }
130 
131     SingleRowKeyValue(bytes, sharing, shard, S3IndexKey(b, s, t.toInt), tier, id, writable.values)
132   }

134   /**
135     * Extracts values out of the filter used for range and push-down predicate creation
136     *
137     * @param filter  query filter
138     * @param explain explainer
139     * @return
140     */
141   override def getIndexValues(filter: Filter, explain: Explainer): S3IndexValues = {
142 
143     import org.locationtech.geomesa.filter.FilterHelper._
144 
145     // standardize the two key query arguments:  polygon and date-range
146 
147     val geometries: FilterValues[Geometry] = {
148       val extracted = extractGeometries(filter, geomField) // intersect since we have points
149       if (extracted.nonEmpty) { extracted } else { FilterValues(Seq(WholeWorldPolygon)) }
150     }
151 
152     // since we don't apply a temporal filter, we pass handleExclusiveBounds to
153     // make sure we exclude the non-inclusive endpoints of a during filter.
154     // note that this isn't completely accurate, as we only index down to the second
155     val intervals = extractIntervals(filter, dtgField, handleExclusiveBounds = true)
156 
157     explain(s"Geometries: $geometries")
158     explain(s"Intervals: $intervals")
159 
160     if (geometries.disjoint || intervals.disjoint) {
161       explain("Disjoint geometries or dates extracted, short-circuiting to empty query")
162       return S3IndexValues(sfc, maxTime, geometries, Seq.empty, intervals, Map.empty, Seq.empty)
163     }
164 
165     // compute our ranges based on the coarse bounds for our query
166     val xy: Seq[(Double, Double, Double, Double)] = {
167       val multiplier = QueryProperties.PolygonDecompMultiplier.toInt.get
168       val bits = QueryProperties.PolygonDecompBits.toInt.get
169       geometries.values.flatMap(GeometryUtils.bounds(_, multiplier, bits))
170     }
171 
172     // calculate map of weeks to time intervals in that week
173     val timesByBin = scala.collection.mutable.Map.empty[Short, Seq[(Int, Int)]].withDefaultValue(Seq.empty)
174     val unboundedBins = Seq.newBuilder[(Short, Short)]
175 
176     // note: intervals shouldn't have any overlaps
177     intervals.foreach { interval =>
178       val (lower, upper) = boundsToDates(interval.bounds)
179       val BinnedTime(lb, lt) = dateToIndex(lower)
180       val BinnedTime(ub, ut) = dateToIndex(upper)
181 
182       if (interval.isBoundedBothSides) {
183         if (lb == ub) {
184           timesByBin(lb) ++= Seq((lt.toInt, ut.toInt))
185         } else {
186           timesByBin(lb) ++= Seq((lt.toInt, maxTime))
187           timesByBin(ub) ++= Seq((0, ut.toInt))
188           Range.inclusive(lb + 1, ub - 1).foreach(b => timesByBin(b.toShort) = Seq((0, maxTime)))
189         }
190       } else if (interval.lower.value.isDefined) {
191         timesByBin(lb) ++= Seq((lt.toInt, maxTime))
192         unboundedBins += (((lb + 1).toShort, Short.MaxValue))
193       } else if (interval.upper.value.isDefined) {
194         timesByBin(ub) ++= Seq((0, ut.toInt))
195         unboundedBins += ((0, (ub - 1).toShort))
196       }
197     }
198 
199     S3IndexValues(sfc, maxTime, geometries, xy, intervals, timesByBin.toMap, unboundedBins.result())
200   }

202   /**
203     * Creates ranges over the index keys
204     *
205     * @param values     index values @see getIndexValues
206     * @param multiplier hint for how many times the ranges will be multiplied. can be used to
207     *                   inform the number of ranges generated
208     * @return
209     */
210   override def getRanges(values: S3IndexValues, multiplier: Int): Iterator[ScanRange[S3IndexKey]] = {
211     val S3IndexValues(s3, _, geoms, xy, intervals, timesByBin, unboundedBins) = values
212 
213     if (geoms.disjoint || intervals.disjoint) {
214       Iterator.empty
215     } else if (timesByBin.isEmpty && unboundedBins.isEmpty) {
216       Iterator.single(UnboundedRange(null))
217     } else {
218       // note: `target` will always be Some, as ScanRangesTarget has a default value
219       val target = QueryProperties.ScanRangesTarget.option.map { t =>
220         math.max(1, if (timesByBin.isEmpty) { t.toInt } else { t.toInt / timesByBin.size } / multiplier)
221       }
222 
223       val s2CellId = s3.ranges(xy, -1, target)
224 
225       val bounded = timesByBin.iterator.flatMap { case (bin, times) =>
226         times.flatMap { time =>
227           s2CellId.map(s => BoundedRange(S3IndexKey(bin, s.lower, time._1), S3IndexKey(bin, s.upper, time._2)))
228         }
229       }
230 
231       val unbounded = unboundedBins.iterator.map {
232         case (0, Short.MaxValue)     => UnboundedRange(S3IndexKey(0, 0L, 0))
233         case (lower, Short.MaxValue) => LowerBoundedRange(S3IndexKey(lower, 0L, 0))
234         case (0, upper)              => UpperBoundedRange(S3IndexKey(upper, 0L, Int.MaxValue))
235         case (lower, upper) =>
236           logger.error(s"Unexpected unbounded bin endpoints: $lower:$upper")
237           UnboundedRange(S3IndexKey(0, 0L, 0))
238       }
239 
240       bounded ++ unbounded
241     }
242   }

244   /**
245     * Creates bytes from ranges
246     *
247     * @param ranges typed scan ranges. @see `getRanges`
248     * @param tier   will the ranges have tiered ranges appended, or not
249     * @return
250     */
251   override def getRangeBytes(ranges: Iterator[ScanRange[S3IndexKey]], tier: Boolean): Iterator[ByteRange] = {
252     if (sharding.length == 0) {
253       ranges.map {
254         case BoundedRange(lo, hi) =>
255           val lower = ByteArrays.toBytes(lo.bin, lo.s, lo.offset)
256           val upper = ByteArrays.toBytesFollowingPrefix(hi.bin, hi.s, hi.offset)
257           BoundedByteRange(lower, upper)
258 
259         case LowerBoundedRange(lo) =>
260           BoundedByteRange(ByteArrays.toBytes(lo.bin, lo.s, lo.offset), ByteRange.UnboundedUpperRange)
261 
262         case UpperBoundedRange(hi) =>
263           BoundedByteRange(ByteRange.UnboundedLowerRange, ByteArrays.toBytesFollowingPrefix(hi.bin, hi.s, hi.offset))
264 
265         case UnboundedRange(_) =>
266           BoundedByteRange(ByteRange.UnboundedLowerRange, ByteRange.UnboundedUpperRange)
267 
268         case r =>
269           throw new IllegalArgumentException(s"Unexpected range type $r")
270       }
271     } else {
272       ranges.flatMap {
273         case BoundedRange(lo, hi) =>
274           val lower = ByteArrays.toBytes(lo.bin, lo.s, lo.offset)
275           val upper = ByteArrays.toBytesFollowingPrefix(hi.bin, hi.s, hi.offset)
276           sharding.shards.map(p => BoundedByteRange(ByteArrays.concat(p, lower), ByteArrays.concat(p, upper)))
277 
278         case LowerBoundedRange(lo) =>
279           val lower = ByteArrays.toBytes(lo.bin, lo.s, lo.offset)
280           val upper = ByteRange.UnboundedUpperRange
281           sharding.shards.map(p => BoundedByteRange(ByteArrays.concat(p, lower), ByteArrays.concat(p, upper)))
282 
283         case UpperBoundedRange(hi) =>
284           val lower = ByteRange.UnboundedLowerRange
285           val upper = ByteArrays.toBytesFollowingPrefix(hi.bin, hi.s, hi.offset)
286           sharding.shards.map(p => BoundedByteRange(ByteArrays.concat(p, lower), ByteArrays.concat(p, upper)))
287 
288         case UnboundedRange(_) =>
289           Seq(BoundedByteRange(ByteRange.UnboundedLowerRange, ByteRange.UnboundedUpperRange))
290 
291         case r =>
292           throw new IllegalArgumentException(s"Unexpected range type $r")
293       }
294     }
