49   override val tieredKeySpace: Option[IndexKeySpace[_, _]] = {
50     if (secondaries.isEmpty) { None } else {
51       val opt = AttributeIndex.tiers.collectFirst {
52         case ks if ks.supports(sft, secondaries) => ks.apply(sft, secondaries, tier = true)
53       }
54       if (opt.isEmpty) {
55         throw new IllegalArgumentException(s"No key space matched tiering for attributes: ${secondaries.mkString(", ")}")
56       }
57       opt
58     }
59   }

76   override def supports(sft: SimpleFeatureType, attributes: Seq[String]): Boolean = {
77     if (attributes.isEmpty) { false } else {
78       val primary = attributes.take(1)
79       val secondary = attributes.tail
80       AttributeIndexKeySpace.supports(sft, primary) && (secondary.isEmpty || tiers.exists(_.supports(sft, secondary)))
81     }
82   }
83 
84   override def defaults(sft: SimpleFeatureType): Seq[Seq[String]] = {
85     sft.getAttributeDescriptors.asScala.flatMap { d =>
86       val index = d.getUserData.get(AttributeOptions.OptIndex).asInstanceOf[String]
87       if (index != null &&
88           (index.equalsIgnoreCase(IndexCoverage.FULL.toString) || java.lang.Boolean.valueOf(index)) &&
89           AttributeIndexKey.encodable(d)) {
90         Seq(Seq(d.getLocalName) ++ Option(sft.getGeomField) ++ sft.getDtgField.filter(_ != d.getLocalName))
91       } else {
92         Seq.empty
93       }
94     }.toSeq
95   }

97   /**
98     * Checks if the given field is attribute indexed

100     * @param sft simple feature type
101     * @param attribute attribute to check
102     * @return
103     */
104   def indexed(sft: SimpleFeatureType, attribute: String): Boolean = {
105     sft.getIndices.exists { i =>
106       (i.name == name || i.name == JoinIndexName) && i.attributes.headOption.contains(attribute)
107     }
108   }
