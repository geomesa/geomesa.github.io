17 /**
18   * Selects routes based on the attributes present in the query filter
19   */
20 class RouteSelectorByAttribute extends RouteSelector {
21 
22   import RouteSelectorByAttribute.RouteAttributes
23 
24   import scala.collection.JavaConverters._
25 
26   private var mappings: Seq[(Set[String], DataStore)] = Seq.empty
27   private var id: Option[DataStore] = None
28   private var include: Option[DataStore] = None
29 
30   override def init(stores: Seq[(DataStore, java.util.Map[String, _ <: AnyRef])]): Unit = {
31     val builder = Seq.newBuilder[(Set[String], DataStore)]
32 
33     stores.foreach { case (ds, config) =>
34       Option(config.get(RouteAttributes)).foreach { attributes =>
35         try {
36           attributes.asInstanceOf[java.util.List[AnyRef]].asScala.foreach {
37             case s: String if s.equalsIgnoreCase("id") => setId(ds)
38             case s: String => builder += Set(s) -> ds
39             case list: java.util.List[String] if list.isEmpty => setInclude(ds)
40             case list: java.util.List[String] if list.size() == 1 && list.get(0).equalsIgnoreCase("id") => setId(ds)
41             case list: java.util.List[String] => builder += list.asScala.toSet -> ds
42             case i => throw new IllegalArgumentException(s"Expected List[String] bug got: $i")
43           }
44         } catch {
45           case NonFatal(e) => throw new IllegalArgumentException(s"Invalid store routing: $attributes", e)
46         }
47       }
48     }
49     mappings = builder.result
50 
51     val check = scala.collection.mutable.Set.empty[Set[String]]
52     mappings.foreach { case (m, _) =>
53       if (!check.add(m)) {
54         throw new IllegalArgumentException("Invalid store routing: " +
55             s"route [${m.mkString(", ")}] is defined more than once")
56       }
57     }
58   }
59 
60   override def route(sft: SimpleFeatureType, query: Query): Option[DataStore] = {
61     def attributes: Option[DataStore] = {
62       val names = FilterHelper.propertyNames(query.getFilter, sft).toSet
63       if (names.isEmpty) { None } else {
64         mappings.collectFirst { case (routes, ds) if routes.forall(names.contains) => ds }
65       }
66     }
67 
68     if (FilterHelper.hasIdFilter(query.getFilter)) {
69       id.orElse(attributes.orElse(include))
70     } else {
71       attributes.orElse(include)
72     }
73   }
74 
75   private def setId(ds: DataStore): Unit = {
76     if (id.isDefined) {
77       throw new IllegalArgumentException("Invalid store routing: 'id' route is defined more than once")
78     }
79     id = Some(ds)
80   }
81 
82   private def setInclude(ds: DataStore): Unit = {
83     if (include.isDefined) {
84       throw new IllegalArgumentException("Invalid store routing: 'include' route is defined more than once")
85     }
86     include = Some(ds)
87   }
88 }
