63   override def toIndexKey(writable: WritableFeature,
64                           tier: Array[Byte],
65                           id: Array[Byte],
66                           lenient: Boolean): RowKeyValue[Z3IndexKey] = {
67     val geom = writable.getAttribute[Point](geomIndex)
68     if (geom == null) {
69       throw new IllegalArgumentException(s"Null geometry in feature ${writable.feature.getID}")
70     }
71     val dtg = writable.getAttribute[Date](dtgIndex)
72     val time = if (dtg == null) { 0 } else { dtg.getTime }
73     val BinnedTime(b, t) = timeToIndex(time)
74     val z = try { sfc.index(geom.getX, geom.getY, t, lenient) } catch {
75       case NonFatal(e) => throw new IllegalArgumentException(s"Invalid z value from geometry/time: $geom,$dtg", e)
76     }
77     val shard = sharding(writable)
78 
79     // create the byte array - allocate a single array up front to contain everything
80     // ignore tier, not used here
81     val bytes = Array.ofDim[Byte](shard.length + 10 + id.length)
82 
83     if (shard.isEmpty) {
84       ByteArrays.writeShort(b, bytes, 0)
85       ByteArrays.writeLong(z, bytes, 2)
86       System.arraycopy(id, 0, bytes, 10, id.length)
87     } else {
88       bytes(0) = shard.head // shard is only a single byte
89       ByteArrays.writeShort(b, bytes, 1)
90       ByteArrays.writeLong(z, bytes, 3)
91       System.arraycopy(id, 0, bytes, 11, id.length)
92     }
93 
94     SingleRowKeyValue(bytes, sharing, shard, Z3IndexKey(b, z), tier, id, writable.values)
95   }

97   override def getIndexValues(filter: Filter, explain: Explainer): Z3IndexValues = {
98 
99     import org.locationtech.geomesa.filter.FilterHelper._
100 
101     // standardize the two key query arguments:  polygon and date-range
102 
103     val geometries: FilterValues[Geometry] = {
104       val extracted = extractGeometries(filter, geomField, intersect = true) // intersect since we have points
105       if (extracted.nonEmpty) { extracted } else { FilterValues(Seq(WholeWorldPolygon)) }
106     }
107 
108     // since we don't apply a temporal filter, we pass handleExclusiveBounds to
109     // make sure we exclude the non-inclusive endpoints of a during filter.
110     // note that this isn't completely accurate, as we only index down to the second
111     val intervals = extractIntervals(filter, dtgField, handleExclusiveBounds = true)
112 
113     explain(s"Geometries: $geometries")
114     explain(s"Intervals: $intervals")
115 
116     if (geometries.disjoint || intervals.disjoint) {
117       explain("Disjoint geometries or dates extracted, short-circuiting to empty query")
118       return Z3IndexValues(sfc, geometries, Seq.empty, intervals, Map.empty, Seq.empty)
119     }
120 
121     // compute our ranges based on the coarse bounds for our query
122     val xy: Seq[(Double, Double, Double, Double)] = {
123       val multiplier = QueryProperties.PolygonDecompMultiplier.toInt.get
124       val bits = QueryProperties.PolygonDecompBits.toInt.get
125       geometries.values.flatMap(GeometryUtils.bounds(_, multiplier, bits))
126     }
127 
128     val minTime = sfc.time.min.toLong
129     val maxTime = sfc.time.max.toLong
130 
131     // calculate map of weeks to time intervals in that week
132     val timesByBin = scala.collection.mutable.Map.empty[Short, Seq[(Long, Long)]].withDefaultValue(Seq.empty)
133     val unboundedBins = Seq.newBuilder[(Short, Short)]
134 
135     // note: intervals shouldn't have any overlaps
136     intervals.foreach { interval =>
137       val (lower, upper) = boundsToDates(interval.bounds)
138       val BinnedTime(lb, lt) = dateToIndex(lower)
139       val BinnedTime(ub, ut) = dateToIndex(upper)
140 
141       if (interval.isBoundedBothSides) {
142         if (lb == ub) {
143           timesByBin(lb) ++= Seq((lt, ut))
144         } else {
145           timesByBin(lb) ++= Seq((lt, maxTime))
146           timesByBin(ub) ++= Seq((minTime, ut))
147           Range.inclusive(lb + 1, ub - 1).foreach(b => timesByBin(b.toShort) = sfc.wholePeriod)
148         }
149       } else if (interval.lower.value.isDefined) {
150         timesByBin(lb) ++= Seq((lt, maxTime))
151         unboundedBins += (((lb + 1).toShort, Short.MaxValue))
152       } else if (interval.upper.value.isDefined) {
153         timesByBin(ub) ++= Seq((minTime, ut))
154         unboundedBins += ((0, (ub - 1).toShort))
155       }
156     }
157 
158     Z3IndexValues(sfc, geometries, xy, intervals, timesByBin.toMap, unboundedBins.result())
159   }

161   override def getRanges(values: Z3IndexValues, multiplier: Int): Iterator[ScanRange[Z3IndexKey]] = {
162     val Z3IndexValues(z3, geoms, xy, intervals, timesByBin, unboundedBins) = values
163 
164     if (geoms.disjoint || intervals.disjoint) {
165       Iterator.empty
166     } else if (timesByBin.isEmpty && unboundedBins.isEmpty) {
167       Iterator.single(UnboundedRange(null))
168     } else {
169       // note: `target` will always be Some, as ScanRangesTarget has a default value
170       val target = QueryProperties.ScanRangesTarget.option.map { t =>
171         math.max(1, if (timesByBin.isEmpty) { t.toInt } else { t.toInt / timesByBin.size } / multiplier)
172       }
173 
174       def toZRanges(t: Seq[(Long, Long)]): Seq[IndexRange] = z3.ranges(xy, t, 64, target)
175 
176       lazy val wholePeriodRanges = toZRanges(z3.wholePeriod)
177 
178       val bounded = timesByBin.iterator.flatMap { case (bin, times) =>
179         val zs = if (times.eq(z3.wholePeriod)) { wholePeriodRanges } else { toZRanges(times) }
180         zs.map(range => BoundedRange(Z3IndexKey(bin, range.lower), Z3IndexKey(bin, range.upper)))
181       }
182 
183       val unbounded = unboundedBins.iterator.map {
184         case (0, Short.MaxValue)     => UnboundedRange(Z3IndexKey(0, 0L))
185         case (lower, Short.MaxValue) => LowerBoundedRange(Z3IndexKey(lower, 0L))
186         case (0, upper)              => UpperBoundedRange(Z3IndexKey(upper, Long.MaxValue))
187         case (lower, upper) =>
188           logger.error(s"Unexpected unbounded bin endpoints: $lower:$upper")
189           UnboundedRange(Z3IndexKey(0, 0L))
190       }
191 
192       bounded ++ unbounded
193     }
194   }

196   override def getRangeBytes(ranges: Iterator[ScanRange[Z3IndexKey]], tier: Boolean): Iterator[ByteRange] = {
197     if (sharding.length == 0) {
198       ranges.map {
199         case BoundedRange(lo, hi) =>
200           BoundedByteRange(ByteArrays.toBytes(lo.bin, lo.z), ByteArrays.toBytesFollowingPrefix(hi.bin, hi.z))
201 
202         case LowerBoundedRange(lo) =>
203           BoundedByteRange(ByteArrays.toBytes(lo.bin, lo.z), ByteRange.UnboundedUpperRange)
204 
205         case UpperBoundedRange(hi) =>
206           BoundedByteRange(ByteRange.UnboundedLowerRange, ByteArrays.toBytesFollowingPrefix(hi.bin, hi.z))
207 
208         case UnboundedRange(_) =>
209           BoundedByteRange(ByteRange.UnboundedLowerRange, ByteRange.UnboundedUpperRange)
210 
211         case r =>
212           throw new IllegalArgumentException(s"Unexpected range type $r")
213       }
214     } else {
215       ranges.flatMap {
216         case BoundedRange(lo, hi) =>
217           val lower = ByteArrays.toBytes(lo.bin, lo.z)
218           val upper = ByteArrays.toBytesFollowingPrefix(hi.bin, hi.z)
219           sharding.shards.map(p => BoundedByteRange(ByteArrays.concat(p, lower), ByteArrays.concat(p, upper)))
220 
221         case LowerBoundedRange(lo) =>
222           val lower = ByteArrays.toBytes(lo.bin, lo.z)
223           val upper = ByteRange.UnboundedUpperRange
224           sharding.shards.map(p => BoundedByteRange(ByteArrays.concat(p, lower), ByteArrays.concat(p, upper)))
225 
226         case UpperBoundedRange(hi) =>
227           val lower = ByteRange.UnboundedLowerRange
228           val upper = ByteArrays.toBytesFollowingPrefix(hi.bin, hi.z)
229           sharding.shards.map(p => BoundedByteRange(ByteArrays.concat(p, lower), ByteArrays.concat(p, upper)))
230 
231         case UnboundedRange(_) =>
232           Seq(BoundedByteRange(ByteRange.UnboundedLowerRange, ByteRange.UnboundedUpperRange))
233 
234         case r =>
235           throw new IllegalArgumentException(s"Unexpected range type $r")
236       }
237     }

240   override def useFullFilter(values: Option[Z3IndexValues],
241                              config: Option[GeoMesaDataStoreConfig],
242                              hints: Hints): Boolean = {
243     // if the user has requested strict bounding boxes, we apply the full filter
244     !Option(hints.get(LOOSE_BBOX)).map(Boolean.unbox).getOrElse(config.forall(_.queries.looseBBox)) ||
245       // if we have unbounded dates, we need to pass them through as we don't have z-values for all periods
246       values.exists(_.temporalUnbounded.nonEmpty) ||
247       // non-rectangular geometries don't map to ranges exactly
248       // if the spatial predicate is rectangular (e.g. a bbox), the index is fine enough that we
249       // don't need to apply the filter on top of it. this may cause some minor errors at extremely
250       // fine resolutions, but the performance is worth it
251       values.exists(_.geometries.values.exists(g => !GeometryUtils.isRectangular(g))) ||
252       // for attribute-level vis, we need to re-evaluate the filter to account for visibility of the row key
253       (sft.getVisibilityLevel == VisibilityLevel.Attribute && values.exists(_.geometries.nonEmpty))
254   }
