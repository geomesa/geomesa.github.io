25 /**
26   * Routed querying against multiple data stores
27   *
28   * @param stores delegate stores
29   * @param router query router
30   * @param namespace schema namespace
31   */
32 class RoutedDataStoreView(val stores: Seq[DataStore], val router: RouteSelector, namespace: Option[String])
33     extends MergedDataStoreSchemas(stores, namespace) with HasGeoMesaStats {
34 
35   require(stores.nonEmpty, "No delegate stores configured")
36 
37   override val stats: GeoMesaStats = new RoutedStats(stores, router)
38 
39   override def getFeatureSource(name: Name): SimpleFeatureSource = getFeatureSource(name.getLocalPart)
40 
41   override def getFeatureSource(typeName: String): SimpleFeatureSource =
42     new RoutedFeatureSourceView(this, getSchema(typeName))
43 
44   override def getFeatureReader(
45       query: Query,
46       transaction: Transaction): FeatureReader[SimpleFeatureType, SimpleFeature] = {
47     val sft = getSchema(query.getTypeName)
48     router.route(sft, query) match {
49       case None => new EmptySimpleFeatureReader(sft)
50       case Some(store) => store.getFeatureReader(query, transaction)
51     }
52   }
53 }

59     private val delegates = stores.toStream.map {

122         filter: Filter,
123         exact: Boolean): Option[T] = route(sft, filter, _.getStat(sft, query, filter, exact))
124 
125     override def close(): Unit = delegates.foreach { case (_, stat) => CloseWithLogging(stat) }
126 
127     private def route[T](sft: SimpleFeatureType, filter: Filter, fn: GeoMesaStats => Option[T]): Option[T] = {
128       val ordered: Stream[GeoMesaStats] = router.route(sft, new Query(sft.getTypeName, filter)) match {
129         case None => delegates.map(_._2)
130         case Some(ds) =>
131           val (head, tail) = delegates.partition(_._1.eq(ds))
132           (head ++ tail).map(_._2)
133       }
134       ordered.flatMap(fn.apply).headOption
135     }
136   }
