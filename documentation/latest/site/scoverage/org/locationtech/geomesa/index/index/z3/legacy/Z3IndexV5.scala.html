54     private val rangePrefixes = {
55       if (sharding.length == 0 && sharing.isEmpty) {
56         Seq.empty
57       } else if (sharing.isEmpty) {
58         sharding.shards
59       } else if (sharding.length == 0) {
60         Seq(sharing)
61       } else {
62         sharding.shards.map(ByteArrays.concat(sharing, _))
63       }
64     }
65 
66     override val indexKeyByteLength: Right[(Array[Byte], Int, Int) => Int, Int] =
67       Right(10 + sharing.length + sharding.length)
68 
69     override def toIndexKey(writable: WritableFeature,
70                             tier: Array[Byte],
71                             id: Array[Byte],
72                             lenient: Boolean): RowKeyValue[Z3IndexKey] = {
73       val geom = writable.getAttribute[Point](geomIndex)
74       if (geom == null) {
75         throw new IllegalArgumentException(s"Null geometry in feature ${writable.feature.getID}")
76       }
77       val dtg = writable.getAttribute[Date](dtgIndex)
78       val time = if (dtg == null) { 0 } else { dtg.getTime }
79       val BinnedTime(b, t) = timeToIndex(time)
80       val z = try { sfc.index(geom.getX, geom.getY, t, lenient) } catch {
81         case NonFatal(e) => throw new IllegalArgumentException(s"Invalid z value from geometry/time: $geom,$dtg", e)
82       }
83       val shard = sharding(writable)
84 
85       // create the byte array - allocate a single array up front to contain everything
86       // ignore tier, not used here
87       val bytes = Array.ofDim[Byte](sharing.length + shard.length + 10 + id.length)
88 
89       var i = 0
90       if (!sharing.isEmpty) {
91         bytes(0) = sharing.head // sharing is only a single byte
92         i += 1
93       }
94       if (!shard.isEmpty) {
95         bytes(i) = shard.head // shard is only a single byte
96         i += 1
97       }
98 
99       ByteArrays.writeShort(b, bytes, i)
100       ByteArrays.writeLong(z, bytes, i + 2)
101       System.arraycopy(id, 0, bytes, i + 10, id.length)
102 
103       SingleRowKeyValue(bytes, sharing, shard, Z3IndexKey(b, z), tier, id, writable.values)
104     }
105 
106     override def getRangeBytes(ranges: Iterator[ScanRange[Z3IndexKey]], tier: Boolean): Iterator[ByteRange] = {
107       if (rangePrefixes.isEmpty) {
108         ranges.map {
109           case BoundedRange(lo, hi) =>
110             BoundedByteRange(ByteArrays.toBytes(lo.bin, lo.z), ByteArrays.toBytesFollowingPrefix(hi.bin, hi.z))
111 
112           case LowerBoundedRange(lo) =>
113             BoundedByteRange(ByteArrays.toBytes(lo.bin, lo.z), ByteRange.UnboundedUpperRange)
114 
115           case UpperBoundedRange(hi) =>
116             BoundedByteRange(ByteRange.UnboundedLowerRange, ByteArrays.toBytesFollowingPrefix(hi.bin, hi.z))
117 
118           case UnboundedRange(_) =>
119             BoundedByteRange(ByteRange.UnboundedLowerRange, ByteRange.UnboundedUpperRange)
120 
121           case r =>
122             throw new IllegalArgumentException(s"Unexpected range type $r")
123         }
124       } else {
125         ranges.flatMap {
126           case BoundedRange(lo, hi) =>
127             val lower = ByteArrays.toBytes(lo.bin, lo.z)
128             val upper = ByteArrays.toBytesFollowingPrefix(hi.bin, hi.z)
129             rangePrefixes.map(p => BoundedByteRange(ByteArrays.concat(p, lower), ByteArrays.concat(p, upper)))
130 
131           case LowerBoundedRange(lo) =>
132             val lower = ByteArrays.toBytes(lo.bin, lo.z)
133             val upper = ByteRange.UnboundedUpperRange
134             rangePrefixes.map(p => BoundedByteRange(ByteArrays.concat(p, lower), ByteArrays.concat(p, upper)))
135 
136           case UpperBoundedRange(hi) =>
137             val lower = ByteRange.UnboundedLowerRange
138             val upper = ByteArrays.toBytesFollowingPrefix(hi.bin, hi.z)
139             rangePrefixes.map(p => BoundedByteRange(ByteArrays.concat(p, lower), ByteArrays.concat(p, upper)))
140 
141           case UnboundedRange(_) =>
142             Seq(BoundedByteRange(ByteRange.UnboundedLowerRange, ByteRange.UnboundedUpperRange))
143 
144           case r =>
145             throw new IllegalArgumentException(s"Unexpected range type $r")
146         }
147       }
148     }
149   }
