15   /**
16    * Returns a sampling function that will indicate if a feature will be kept or discarded.
17    *
18    * Since the current implementation rounds heavily, large percentages will not actually sample out any
19    * features, so will return None
20    *
21    * @param percent percent of features to keep, between [0, 1]
22    * @param field field to use for threading of sampling
23    * @return sampling function
24    */
25   def sample(percent: Float, field: Option[Int]): Option[SimpleFeature => Boolean] = {
26     if (!(percent > 0 && percent < 1f)) {
27       throw new IllegalArgumentException(s"Sampling must be a percentage between (0, 1): $percent")
28     }
29     val nth = math.round(1 / percent)
30     if (nth <= 1) { None } else {
31       Some(sample(nth, field))
32     }
33   }
34 
35   /**
36    * Returns a sampling function that will indicate if a feature should be kept or discarded
37    *
38    * @param nth will keep every nth feature
39    * @param field field to use for threading of samples
40    * @return sampling function
41    */
42   def sample(nth: Int, field: Option[Int]): SimpleFeature => Boolean = {
43     field match {
44       case None =>
45         var i = 1
46         _ => if (i == 1) { i += 1; true } else if (i < nth) { i += 1; false } else { i = 1; false }
47       case Some(f) =>
48         val i = scala.collection.mutable.HashMap.empty[String, Int].withDefaultValue(1)
49         sf => {
50           val value = sf.getAttribute(f)
51           val key = if (value == null) "" else value.toString
52           val count = i(key)
53           if (count < nth) { i(key) = count + 1; count == 1 } else { i(key) = 1; false }
54         }
55     }
56   }
