38   override def createNewDataStore(params: java.util.Map[String, _]): DataStore = {
39     val config = ConfigFactory.parseString(ConfigParam.lookup(params)).resolve()
40     val configs = if (config.hasPath("stores")) { config.getConfigList("stores").asScala } else { Seq.empty }
41     if (configs.isEmpty) {
42       throw new IllegalArgumentException("No 'stores' element defined in configuration")
43     }
44 
45     val namespace = NamespaceParam.lookupOpt(params)
46     val nsConfig = namespace.map(ConfigValueFactory.fromAnyRef)
47 
48     val stores = Seq.newBuilder[(DataStore, java.util.Map[String, _ <: AnyRef])]
49     stores.sizeHint(configs.length)
50 
51     try {
52       configs.foreach { config =>
53         lazy val error = new IllegalArgumentException(s"Could not load store using configuration:\n" +
54             config.root().render(ConfigRenderOptions.concise().setFormatted(true)))
55         // inject the namespace into the underlying stores
56         val storeParams = nsConfig.map(config.withValue(NamespaceParam.key, _)).getOrElse(config).root().unwrapped()
57         Try(DataStoreFinder.getDataStore(storeParams)) match {
58           case Success(null)  => throw error
59           case Success(store) => stores += store -> storeParams
60           case Failure(e)     => throw error.initCause(e)
61         }
62       }
63     } catch {
64       case NonFatal(e) => stores.result.foreach(_._1.dispose()); throw e
65     }
66 
67     val storesAndConfigs = stores.result
68 
69     try {
70       val router = {
71         val impl = RouterParam.lookup(params)
72         ServiceLoader.load[RouteSelector]().find(_.getClass.getName == impl).getOrElse {
73           throw new IllegalArgumentException(s"Could not load route selector '$impl'. Available implementations: " +
74               ServiceLoader.load[RouteSelector]().map(_.getClass.getName).mkString(", "))
75         }
76       }
77 
78       router.init(storesAndConfigs)
79 
80       new RoutedDataStoreView(storesAndConfigs.map(_._1), router, namespace)
81     } catch {
82       case NonFatal(e) => storesAndConfigs.foreach(_._1.dispose()); throw e
83     }
84   }
