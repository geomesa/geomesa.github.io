117   /**
118     * Validates the schema and writes metadata to catalog.If the schema already exists,
119     * continue without error.
120     *
121     * This method uses distributed locking to ensure a schema is only created once.
122     *
123     * @see org.geotools.data.DataAccess#createSchema(org.geotools.api.feature.type.FeatureType)
124     * @param schema type to create
125     */
126   override def createSchema(schema: SimpleFeatureType): Unit = {
127     if (getSchema(schema.getTypeName) == null) {
128       val lock = acquireCatalogLock()
129       try {
130         // check a second time now that we have the lock
131         if (getSchema(schema.getTypeName) == null) {
132           // ensure that we have a mutable type so we can set user data
133           val sft = SimpleFeatureTypes.mutable(schema)
134           // inspect and update the simple feature type for various components
135           // do this before anything else so that any modifications will be in place
136           GeoMesaSchemaValidator.validate(sft)
137 
138           // set the enabled indices
139           preSchemaCreate(sft)
140 
141           try {
142             // write out the metadata to the catalog table
143             // compute the metadata values - IMPORTANT: encode type has to be called after all user data is set
144             val metadataMap = Map(
145               AttributesKey      -> SimpleFeatureTypes.encodeType(sft, includeUserData = true),
146               StatsGenerationKey -> GeoToolsDateFormat.format(Instant.now().atOffset(ZoneOffset.UTC))
147             )
148             metadata.insert(sft.getTypeName, metadataMap)
149 
150             // reload the sft so that we have any default metadata,
151             // then copy over any additional keys that were in the original sft.
152             // avoid calling getSchema directly, as that may trigger a remote version
153             // check for indices that haven't been created yet
154             val attributes = metadata.readRequired(sft.getTypeName, AttributesKey)
155             val reloadedSft = SimpleFeatureTypes.createType(sft.getTypeName, attributes)
156             val userData = new java.util.HashMap[AnyRef, AnyRef]()
157             userData.putAll(reloadedSft.getUserData)
158             reloadedSft.getUserData.putAll(sft.getUserData)
159             reloadedSft.getUserData.putAll(userData)
160 
161             // create the tables
162             onSchemaCreated(reloadedSft)
163           } catch {
164             case NonFatal(e) =>
165               // If there was an error creating a schema, clean up.
166               try {
167                 metadata.delete(sft.getTypeName)
168               } catch {
169                 case NonFatal(e2) => e.addSuppressed(e2)
170               }
171               throw e
172           }
173         }
174       } finally {
175         lock.close()
176       }
177     }
178   }

217     * Allows the following modifications to the schema:
218     *   renaming the feature type
219     *   renaming attributes
220     *   appending new attributes
221     *   enabling/disabling indices through RichSimpleFeatureType.setIndexVersion
222     *   modifying keywords through user-data
223     *
224     * Other modifications are not supported.
225     *
226     * @see org.geotools.data.DataAccess#updateSchema(org.geotools.api.feature.type.Name, org.geotools.api.feature.type.FeatureType)
227     * @param typeName simple feature type name
228     * @param schema new simple feature type
229     */
230   override def updateSchema(typeName: Name, schema: SimpleFeatureType): Unit = {
231     // validate that the type name has not changed, or that the new name is not already in use
232     if (typeName.getLocalPart != schema.getTypeName && getTypeNames.contains(schema.getTypeName)) {
233       throw new IllegalArgumentException(
234         s"Updated type name already exists: '$typeName' changed to '${schema.getTypeName}'")
235     }
236 
237     val lock = acquireCatalogLock()
238     try {
239       // get previous schema and user data
240       val previousSft = getSchema(typeName)
241 
242       if (previousSft == null) {
243         throw new IllegalArgumentException(s"Schema '$typeName' does not exist")
244       }
245 
246       GeoMesaSchemaValidator.validate(schema)
247 
248       validateSchemaUpdate(previousSft, schema).left.foreach(e => throw e)
249 
250       val sft = SimpleFeatureTypes.mutable(schema)
251 
252       // validation and normalization of the schema
253       preSchemaUpdate(sft, previousSft)
254 
255       // if all is well, update the metadata - first back it up
256       if (FastConverter.convertOrElse[java.lang.Boolean](sft.getUserData.get(Configs.UpdateBackupMetadata), true)) {
257         metadata.backup(typeName.getLocalPart)
258       }
259 
260       // rename the old rows if the type name has changed
261       if (typeName.getLocalPart != schema.getTypeName) {
262         metadata.scan(typeName.getLocalPart, "", cache = false).foreach { case (k, v) =>
263           metadata.insert(sft.getTypeName, k, v)
264           metadata.remove(typeName.getLocalPart, k)
265         }
266       }
267       // now insert the new spec string
268       metadata.insert(sft.getTypeName, AttributesKey, SimpleFeatureTypes.encodeType(sft, includeUserData = true))
269 
270       onSchemaUpdated(sft, previousSft)
271     } finally {
272       lock.close()
273     }
274   }

414   /**
415    * Validate a call to updateSchema, throwing errors on failed validation
416    *
417    * @param existing existing schema
418    * @param schema updated sft
419    * @return validation result
420    */
421   protected def validateSchemaUpdate(
422       existing: SimpleFeatureType,
423       schema: SimpleFeatureType): Either[UnsupportedOperationException, TypeComparison.Compatible] = {
424     // validate that default geometry and date have not changed (rename is ok)
425     if (schema.getGeomIndex != existing.getGeomIndex) {
426       return Left(new UnsupportedOperationException("Changing the default geometry attribute is not supported"))
427     }
428     if (schema.getDtgIndex != existing.getDtgIndex) {
429       return Left(new UnsupportedOperationException("Changing the default date attribute is not supported"))
430     }
431 
432     // check that unmodifiable user data has not changed
433     val userDataChanges = MetadataBackedDataStore.UnmodifiableUserDataKeys.flatMap { key =>
434       if (schema.userData[Any](key) == existing.userData[Any](key)) { None } else {
435         Some(s"'$key'")
436       }
437     }
438     if (userDataChanges.nonEmpty) {
439       val msg = s"${if (userDataChanges.size == 1) { "" } else { "s" }} ${userDataChanges.mkString(", ")}"
440       return Left(new UnsupportedOperationException(s"Updating user data key$msg is not supported"))
441     }
442 
443     SimpleFeatureTypeComparator.compare(existing, schema) match {
444       case TypeComparison.AttributeRemoved =>
445         Left(new UnsupportedOperationException("Removing attributes from the schema is not supported"))
446 
447       case TypeComparison.AttributeTypeChanged(changes) =>
448         val msg = changes.map { case (name, (from, to)) => s"$name from ${from.getName} to ${to.getName}" }
449         Left(new UnsupportedOperationException(s"Incompatible schema column type changes: ${msg.mkString(", ")}"))
450 
451       case c: TypeComparison.Compatible =>
452         // check for reserved words - only check for new/renamed attributes
453         val reserved = schema.getAttributeDescriptors.asScala.map(_.getLocalName).exists { name =>
454           existing.getDescriptor(name) == null && FeatureUtils.ReservedWords.contains(name.toUpperCase(Locale.US))
455         }
456         if (reserved) {
457           try { ReservedWordCheck.validateAttributeNames(schema) } catch {
458             case NonFatal(e) =>
459               return Left(new UnsupportedOperationException(e.getMessage))
460           }
461         }
462         Right(c)
463     }
464   }

466   /**
467    * Acquires a distributed lock for all data stores sharing this catalog table.
468    * Make sure that you 'release' the lock in a finally block.
469    */
470   protected[geomesa] def acquireCatalogLock(): Closeable = {
471     import org.locationtech.geomesa.index.DistributedLockTimeout
472     val dsTypeName = getClass.getSimpleName.replaceAll("[^A-Za-z]", "")
473     val path = s"/org.locationtech.geomesa/ds/$dsTypeName/${config.catalog}"
474     val timeout = DistributedLockTimeout.toDuration.getOrElse {
475       // note: should always be a valid fallback value so this exception should never be triggered
476       throw new IllegalArgumentException(s"Couldn't convert '${DistributedLockTimeout.get}' to a duration")
477     }
478     acquireDistributedLock(path, timeout.toMillis).getOrElse {
479       throw new RuntimeException(s"Could not acquire distributed lock at '$path' within $timeout")
480     }
481   }
