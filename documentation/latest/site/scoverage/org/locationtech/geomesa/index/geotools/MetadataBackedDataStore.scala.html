117   /**
118     * Validates the schema and writes metadata to catalog.If the schema already exists,
119     * continue without error.
120     *
121     * This method uses distributed locking to ensure a schema is only created once.
122     *
123     * @see org.geotools.data.DataAccess#createSchema(org.geotools.api.feature.type.FeatureType)
124     * @param schema type to create
125     */
126   override def createSchema(schema: SimpleFeatureType): Unit = {
127     if (getSchema(schema.getTypeName) == null) {
128       val lock = acquireCatalogLock()
129       try {
130         // check a second time now that we have the lock
131         if (getSchema(schema.getTypeName) == null) {
132           // ensure that we have a mutable type so we can set user data
133           val sft = SimpleFeatureTypes.mutable(schema)
134           // inspect and update the simple feature type for various components
135           // do this before anything else so that any modifications will be in place
136           GeoMesaSchemaValidator.validate(sft)
137 
138           // set the enabled indices
139           preSchemaCreate(sft)
140 
141           // compute the metadata values - IMPORTANT: encode type has to be called after all user data is set
142           val encodedAttributes = SimpleFeatureTypes.encodeType(sft, includeUserData = true)
143           // validate we can read out the encoded sft - invalid user data keys can break the parsing
144           try { SimpleFeatureTypes.createType("", encodedAttributes) } catch {
145             case NonFatal(e) => throw new IllegalArgumentException("Invalid schema:", e)
146           }
147           val metadataMap = Map(
148             AttributesKey      -> encodedAttributes,
149             StatsGenerationKey -> GeoToolsDateFormat.format(Instant.now().atOffset(ZoneOffset.UTC))
150           )
151           try {
152             // write out the metadata to the catalog table
153             metadata.insert(sft.getTypeName, metadataMap)
154 
155             // reload the sft so that we have any default metadata,
156             // then copy over any additional keys that were in the original sft.
157             // avoid calling getSchema directly, as that may trigger a remote version
158             // check for indices that haven't been created yet
159             val attributes = metadata.readRequired(sft.getTypeName, AttributesKey)
160             val reloadedSft = SimpleFeatureTypes.createType(sft.getTypeName, attributes)
161             val userData = new java.util.HashMap[AnyRef, AnyRef]()
162             userData.putAll(reloadedSft.getUserData)
163             reloadedSft.getUserData.putAll(sft.getUserData)
164             reloadedSft.getUserData.putAll(userData)
165 
166             // create the tables
167             onSchemaCreated(reloadedSft)
168           } catch {
169             case NonFatal(e) =>
170               // If there was an error creating a schema, clean up.
171               try {
172                 metadata.delete(sft.getTypeName)
173               } catch {
174                 case NonFatal(e2) => e.addSuppressed(e2)
175               }
176               throw e
177           }
178         }
179       } finally {
180         lock.close()
181       }
182     }
183   }

221   /**
222     * Allows the following modifications to the schema:
223     *   renaming the feature type
224     *   renaming attributes
225     *   appending new attributes
226     *   enabling/disabling indices through RichSimpleFeatureType.setIndexVersion
227     *   modifying keywords through user-data
228     *
229     * Other modifications are not supported.
230     *
231     * @see org.geotools.data.DataAccess#updateSchema(org.geotools.api.feature.type.Name, org.geotools.api.feature.type.FeatureType)
232     * @param typeName simple feature type name
233     * @param schema new simple feature type
234     */
235   override def updateSchema(typeName: Name, schema: SimpleFeatureType): Unit = {
236     // validate that the type name has not changed, or that the new name is not already in use
237     if (typeName.getLocalPart != schema.getTypeName && getTypeNames.contains(schema.getTypeName)) {
238       throw new IllegalArgumentException(
239         s"Updated type name already exists: '$typeName' changed to '${schema.getTypeName}'")
240     }
241 
242     val lock = acquireCatalogLock()
243     try {
244       // get previous schema and user data
245       val previousSft = getSchema(typeName)
246 
247       if (previousSft == null) {
248         throw new IllegalArgumentException(s"Schema '$typeName' does not exist")
249       }
250 
251       GeoMesaSchemaValidator.validate(schema)
252 
253       validateSchemaUpdate(previousSft, schema).left.foreach(e => throw e)
254 
255       val sft = SimpleFeatureTypes.mutable(schema)
256 
257       // validation and normalization of the schema
258       preSchemaUpdate(sft, previousSft)
259 
260       val encodedAttributes = SimpleFeatureTypes.encodeType(sft, includeUserData = true)
261       // validate we can read out the encoded sft - invalid user data keys can break the parsing
262       try { SimpleFeatureTypes.createType("", encodedAttributes) } catch {
263         case NonFatal(e) => throw new IllegalArgumentException("Invalid schema:", e)
264       }
265 
266       // if all is well, update the metadata - first back it up
267       if (FastConverter.convertOrElse[java.lang.Boolean](sft.getUserData.get(Configs.UpdateBackupMetadata), true)) {
268         metadata.backup(typeName.getLocalPart)
269       }
270 
271       // rename the old rows if the type name has changed
272       if (typeName.getLocalPart != schema.getTypeName) {
273         metadata.scan(typeName.getLocalPart, "", cache = false).foreach { case (k, v) =>
274           metadata.insert(sft.getTypeName, k, v)
275           metadata.remove(typeName.getLocalPart, k)
276         }
277       }
278       // now insert the new spec string
279       metadata.insert(sft.getTypeName, AttributesKey, encodedAttributes)
280 
281       onSchemaUpdated(sft, previousSft)
282     } finally {
283       lock.close()
284     }
285   }

425   /**
426    * Validate a call to updateSchema, throwing errors on failed validation
427    *
428    * @param existing existing schema
429    * @param schema updated sft
430    * @return validation result
431    */
432   protected def validateSchemaUpdate(
433       existing: SimpleFeatureType,
434       schema: SimpleFeatureType): Either[UnsupportedOperationException, TypeComparison.Compatible] = {
435     // validate that default geometry and date have not changed (rename is ok)
436     if (schema.getGeomIndex != existing.getGeomIndex) {
437       return Left(new UnsupportedOperationException("Changing the default geometry attribute is not supported"))
438     }
439     if (schema.getDtgIndex != existing.getDtgIndex) {
440       return Left(new UnsupportedOperationException("Changing the default date attribute is not supported"))
441     }
442 
443     // check that unmodifiable user data has not changed
444     val userDataChanges = MetadataBackedDataStore.UnmodifiableUserDataKeys.flatMap { key =>
445       if (schema.userData[Any](key) == existing.userData[Any](key)) { None } else {
446         Some(s"'$key'")
447       }
448     }
449     if (userDataChanges.nonEmpty) {
450       val msg = s"${if (userDataChanges.size == 1) { "" } else { "s" }} ${userDataChanges.mkString(", ")}"
451       return Left(new UnsupportedOperationException(s"Updating user data key$msg is not supported"))
452     }
453 
454     SimpleFeatureTypeComparator.compare(existing, schema) match {
455       case TypeComparison.AttributeRemoved =>
456         Left(new UnsupportedOperationException("Removing attributes from the schema is not supported"))
457 
458       case TypeComparison.AttributeTypeChanged(changes) =>
459         val msg = changes.map { case (name, (from, to)) => s"$name from ${from.getName} to ${to.getName}" }
460         Left(new UnsupportedOperationException(s"Incompatible schema column type changes: ${msg.mkString(", ")}"))
461 
462       case c: TypeComparison.Compatible =>
463         // check for reserved words - only check for new/renamed attributes
464         val reserved = schema.getAttributeDescriptors.asScala.map(_.getLocalName).exists { name =>
465           existing.getDescriptor(name) == null && FeatureUtils.ReservedWords.contains(name.toUpperCase(Locale.US))
466         }
467         if (reserved) {
468           try { ReservedWordCheck.validateAttributeNames(schema) } catch {
469             case NonFatal(e) =>
470               return Left(new UnsupportedOperationException(e.getMessage))
471           }
472         }
473         Right(c)
474     }
475   }

479    * Make sure that you 'release' the lock in a finally block.

481   protected[geomesa] def acquireCatalogLock(): Closeable = {
482     import org.locationtech.geomesa.index.DistributedLockTimeout
483     val dsTypeName = getClass.getSimpleName.replaceAll("[^A-Za-z]", "")
484     val path = s"/org.locationtech.geomesa/ds/$dsTypeName/${config.catalog}"
485     val timeout = DistributedLockTimeout.toDuration.getOrElse {
486       // note: should always be a valid fallback value so this exception should never be triggered
487       throw new IllegalArgumentException(s"Couldn't convert '${DistributedLockTimeout.get}' to a duration")
488     }
489     acquireDistributedLock(path, timeout.toMillis).getOrElse {
490       throw new RuntimeException(s"Could not acquire distributed lock at '$path' within $timeout")
