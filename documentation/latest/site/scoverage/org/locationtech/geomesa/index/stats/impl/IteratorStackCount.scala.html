14 /**
15   * The IteratorStackCount keeps track of the number of times Accumulo sets up an iterator stack
16   * as a result of a query.
17   *
18   * @param sft simple feature type
19   */
20 class IteratorStackCount private [stats] (val sft: SimpleFeatureType) extends Stat {
21 
22   private [stats] var counter: Long = 1
23 
24   private var empty = false
25 
26   override type S = IteratorStackCount
27 
28   def count: Long = counter
29 
30   override def observe(sf: SimpleFeature): Unit = {}
31 
32   override def unobserve(sf: SimpleFeature): Unit = {}
33 
34   override def +(other: IteratorStackCount): IteratorStackCount = {
35     val plus = new IteratorStackCount(sft)
36     plus.counter += this.counter
37     plus.counter += other.counter
38     plus
39   }
40 
41   override def +=(other: IteratorStackCount): Unit = counter += other.counter
42 
43   override def toJsonObject: Map[String, Long] = Map("count" -> counter)
44 
45   // this logic is tied to the scan lifecycle in accumulo/hbase, and ensures that a single
46   // result is returned for each scan that is created, regardless of if there is data or not
47   override def isEmpty: Boolean = if (empty) { true } else { empty = true; false }
48 
49   override def clear(): Unit = counter = 1L
50 
51   override def isEquivalent(other: Stat): Boolean = other match {
52     case that: IteratorStackCount => counter == that.counter
53     case _ => false
54   }
55 }
