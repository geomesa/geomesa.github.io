22 /**
23   * Attribute index with secondary z-curve indexing. Z-indexing is based on the sft and will be
24   * one of Z3, XZ3, Z2, XZ2. Shards come after the attribute number, instead of before it.
25   */
26 class AttributeIndexV6 protected (ds: GeoMesaDataStore[_],
27                                   sft: SimpleFeatureType,
28                                   version: Int,
29                                   attribute: String,
30                                   secondaries: Seq[String],
31                                   mode: IndexMode)
32     extends AttributeIndexV7(ds, sft, version, attribute, secondaries, mode)
33         with LegacyTableNaming[AttributeIndexValues[Any], AttributeIndexKey] {
34 
35   def this(ds: GeoMesaDataStore[_], sft: SimpleFeatureType, attribute: String, secondaries: Seq[String], mode: IndexMode) =
36     this(ds, sft, 6, attribute, secondaries, mode)
37 
38   override val keySpace: AttributeIndexKeySpace = {
39     val sharding = AttributeShardStrategy(sft)
40     if (sharding.shards.nonEmpty) {
41       // if sharding, we need to swap the shard bytes with the idx bytes
42       new AttributeIndexKeySpaceV6(sft, sft.getTableSharingBytes, sharding, attribute)
43     } else {
44       // otherwise we can skip the swap and use the parent class
45       new AttributeIndexKeySpaceV7(sft, sft.getTableSharingBytes, sharding, attribute)
46     }
47   }
48 
49   override protected val fallbackTableNameKey: String = "tables.idx.attr.name"
50 }
51 
52 object AttributeIndexV6 {
53 
54   /**
55     * Map from (sharing, shard, idx0, idx1) to (sharing, idx0, idx1, shard)
56     *
57     * Rows in the attribute table have the following layout:
58     *
59     * - 1 byte identifying the sft (OPTIONAL - only if table is shared)
60     * - 2 bytes storing the index of the attribute in the sft
61     * - 1 byte shard (OPTIONAL)
62     * - n bytes storing the lexicoded attribute value
63     * - NULLBYTE as a separator
64     * - n bytes storing the secondary z-index of the feature - identified by getSecondaryIndexKeyLength
65     * - n bytes storing the feature ID
66     *
67     */
68   class AttributeIndexKeySpaceV6(sft: SimpleFeatureType,
69                                  sharing: Array[Byte],
70                                  sharding: ShardStrategy,
71                                  attributeField: String)
72       extends AttributeIndexKeySpaceV7(sft, sharing, sharding, attributeField) {
73 
74     private val offset = sharing.length
75 
76     override def toIndexKey(writable: WritableFeature,
77                             tier: Array[Byte],
78                             id: Array[Byte],
79                             lenient: Boolean): RowKeyValue[AttributeIndexKey] = {
80       val key = super.toIndexKey(writable, tier, id, lenient)
81       key match {
82         case kv: SingleRowKeyValue[AttributeIndexKey] => swap(kv.row)
83         case kv: MultiRowKeyValue[AttributeIndexKey] => kv.rows.foreach(swap)
84       }
85       key
86     }
87 
88     override def getRangeBytes(ranges: Iterator[ScanRange[AttributeIndexKey]], tier: Boolean): Iterator[ByteRange] = {
89       super.getRangeBytes(ranges, tier).map {
90         case BoundedByteRange(lower, upper)      => BoundedByteRange(swap(lower), swap(upper))
91         case SingleRowByteRange(row)             => SingleRowByteRange(swap(row))
92         case LowerBoundedByteRange(lower, upper) => LowerBoundedByteRange(swap(lower), swap(upper))
93         case UpperBoundedByteRange(lower, upper) => UpperBoundedByteRange(swap(lower), swap(upper))
94         case UnboundedByteRange(lower, upper)    => UnboundedByteRange(swap(lower), swap(upper))
95         case r => throw new NotImplementedError(s"Unexpected byte range: $r")
96       }
97     }
98 
99     private def swap(bytes: Array[Byte]): Array[Byte] = {
100       if (bytes.length > 0) {
101         val shard = bytes(offset)
102         bytes(offset) = bytes(offset + 1)
103         bytes(offset + 1) = bytes(offset + 2)
104         bytes(offset + 2) = shard
105       }
106       bytes
107     }
108   }
