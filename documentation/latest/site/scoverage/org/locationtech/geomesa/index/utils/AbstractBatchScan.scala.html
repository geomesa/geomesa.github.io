17 /**
18   * Provides parallelism for scanning multiple ranges at a given time, for systems that don't
19   * natively support that.
20   *
21   * Subclasses should generally only expose the `CloseableIterator` interface, and make sure to
22   * invoke `start()` before returning to the caller.
23   *
24   * @param ranges ranges to scan
25   * @param threads number of client threads to use for scanning
26   * @param buffer max size of the buffer for storing results before they are read by the caller
27   * @param sentinel singleton sentinel value used to indicate the completion of scanning threads
28   * @tparam T range type
29   * @tparam R scan result type
30   */
31 abstract class AbstractBatchScan[T, R <: AnyRef](ranges: Seq[T], threads: Int, buffer: Int, sentinel: R)
32     extends CloseableIterator[R] {
33 
34   import scala.collection.JavaConverters._
35 
36   require(threads > 0, "Thread count must be greater than 0")
37 
38   private val inQueue = new ConcurrentLinkedQueue(ranges.asJava)
39   private val outQueue = new LinkedBlockingQueue[R](buffer)
40 
41   private val pool = new CachedThreadPool(threads)
42   private val latch = new CountDownLatch(threads)
43   private val terminator = new Terminator()
44 
45   private var retrieved: R = _
46   private var error: Throwable = _
47 
48   @volatile
49   protected var closed: Boolean = false
50 
51   /**
52     * Scan a single range
53     *
54     * @param range range to scan
55     */
56   protected def scan(range: T): CloseableIterator[R]
57 
58   /**
59     * Start the threaded scans executing
60     */
61   protected def start(): CloseableIterator[R] = {
62     var i = 0
63     while (i < threads) {
64       pool.submit(new SingleThreadScan())
65       i += 1
66     }
67     pool.submit(terminator)
68     pool.shutdown()
69     this
70   }
71 
72   override def hasNext: Boolean = {
73     if (retrieved != null) {
74       true
75     } else {
76       retrieved = outQueue.take
77       if (!retrieved.eq(sentinel)) {
78         true
79       } else {
80         outQueue.put(sentinel) // re-queue in case hasNext is called again
81         retrieved = null.asInstanceOf[R]
82         this.synchronized {
83           if (error != null) {
84             throw error
85           }
86         }
87         false
88       }
89     }
90   }
91 
92   override def next(): R = {
93     val n = retrieved
94     retrieved = null.asInstanceOf[R]
95     n
96   }
97 
98   override def close(): Unit = {
99     closed = true
100     inQueue.clear()
101     terminator.terminate(true)
102   }

104   /**
105     * Exposed for testing only
106     *
107     * @param timeout timeout to wait
108     * @return true if all threads have terminated, otherwise false
109     */
110   private [utils] def waitForDone(timeout: Long): Boolean = {
111     val start = System.currentTimeMillis()
112     while (true) {
113       if (pool.isTerminated) {
114         return true
115       } else if (System.currentTimeMillis() - start > timeout) {

122   }

144   /**
145     * Pulls ranges off the queue and executes them
146     */
147   private class SingleThreadScan extends Runnable {
148     override def run(): Unit = {
149       try {
150         var range = inQueue.poll()
151         while (range != null) {
152           val result = scan(range)
153           try {
154             while (result.hasNext) {
155               val r = result.next
156               while (!outQueue.offer(r, 100, TimeUnit.MILLISECONDS)) {
157                 if (closed) {
158                   return
159                 }
160               }
161             }
162           } finally {
163             result.close()
164           }
165           range = inQueue.poll()
166         }
167       } catch {
168         case NonFatal(e) =>
169           AbstractBatchScan.this.synchronized {
170             if (error == null) { error = e } else { error.addSuppressed(e) }
171           }
172           close()
173       } finally {
174         latch.countDown()
175       }
176     }
177   }
178 
179   /**
180     * Injects the terminal value into the output buffer, once all the scans are complete
181     */
182   private class Terminator extends Runnable {
183 
184     @volatile
185     private var done = false
186 
187     override def run(): Unit = try { latch.await() } finally { terminate(false) }
188 
189     final def terminate(drop: Boolean): Unit = {
190       // it's possible that the queue is full, in which case we can't immediately
191       // add the sentinel to the queue to indicate to the client that scans are done
192       // if the scan has been closed, then the client is done
193       // reading and we don't mind dropping some results
194       // otherwise we wait and give the client a chance to empty the queue
195       if (!done && (drop || closed || !outQueue.offer(sentinel, 1000, TimeUnit.MILLISECONDS))) {
196         // terminate with drops
197         while (!outQueue.offer(sentinel)) { outQueue.poll() }
198       }
199       done = true
200     }
201   }
