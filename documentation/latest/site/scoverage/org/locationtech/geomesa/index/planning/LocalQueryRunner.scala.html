132   /**
133    * Local processor, for datastores that don't support remote filtering/transforms. Handles filtering, transforms,
134    * sorting, max features, and analytic processing (may be partially encapsulated in the reducer).
135    *
136    * Note: will not perform any auth filtering if the authProvider is not defined
137    *
138    * @param sft feature type
139    * @param hints query hints
140    * @param authProvider auth provider, if auth filtering is desired
141    */
142   case class LocalProcessor(sft: SimpleFeatureType, hints: Hints, authProvider: Option[AuthorizationsProvider])
143       extends LocalScanProcessor {
144 
145     private val filterFunction: Option[SimpleFeature => Boolean] = {
146       val visible = authProvider.map(VisibilityUtils.visible)
147       val sampler = hints.getSampling.flatMap { case (percent, field) =>
148         FeatureSampler.sample(percent, field.map(sft.indexOf).filter(_ != -1))
149       }
150       (visible, sampler) match {
151         case (None, None) => None
152         case (Some(v), None) => Some(f => v(f))
153         case (None, Some(s)) => Some(f => s(f))
154         case (Some(v), Some(s)) => Some(f => v(f) && s(f)) // note: make sure sampling is checked after filtering so it's correct
155       }
156     }
157 
158     private val sort = hints.getSortFields.orElse {
159       // pull out distributed sort hints into regular sort hints that we process here
160       if (hints.isArrowQuery) {
161         hints.getArrowSort.map { case (field, reverse) => Seq(field -> reverse) }
162       } else if (hints.isBinQuery && hints.isBinSorting) {
163         hints.getBinDtgField.orElse(sft.getDtgField).map(dtg => Seq(dtg -> false))
164       } else {
165         None
166       }
167     }
168 
169     private val processor = {
170       val inputSchema = hints.getTransformSchema.getOrElse(sft)
171       if (hints.isArrowQuery) {
172         Some(new ArrowProcessor(inputSchema, hints))
173       } else if (hints.isBinQuery) {
174         Some(new BinProcessor(inputSchema, hints))
175       } else if (hints.isDensityQuery) {
176         Some(new DensityProcessor(inputSchema, hints))
177       } else if (hints.isStatsQuery) {
178         Some(new StatsProcessor(inputSchema, hints))
179       } else {
180         None
181       }
182     }
183 
184     override val reducer: Option[FeatureReducer] = processor.flatMap(_.reducer)
185 
186     override def apply(features: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = {
187       val filtered = filterFunction.fold(features)(f => features.filter(f.apply))
188       val transformed = hints.getTransform.fold(filtered) {  case (tdefs, tsft) =>
189         // note: we do the transform here and not in the query plan, as the processor step expects already transformed features
190         val transform = TransformSimpleFeature(sft, tsft, tdefs)
191         filtered.map(f => ScalaSimpleFeature.copy(transform.setFeature(f)))
192       }
193       // handle sorting + max features here before any encoding complicates things
194       val sorted = sort.fold(transformed)(s => new SortingSimpleFeatureIterator(transformed, s))
195       val limited = hints.getMaxFeatures.fold(sorted)(m => sorted.take(m))
196       processor.fold(limited)(_.apply(limited))
197     }
198 
199     def explain(explainer: Explainer): Unit = {
200       explainer(s"Transform: ${hints.getTransform.fold("none")(t => s"${t._1} ${SimpleFeatureTypes.encodeType(t._2)}")}")
201       explainer(s"Sort: ${hints.getSortFields.fold("none")(_.mkString(", "))}")
202       explainer(s"Max Features: ${hints.getMaxFeatures.getOrElse("none")}")
203     }
204   }

207    * Processor for local arrow transforms

209    * @param sft feature type

211    * @param encoding arrow encoding options
212    * @param ipcOpts arrow ipc format options

215   class ArrowProcessor(
216       sft: SimpleFeatureType,
217       dictionaryFields: Seq[String],
218       encoding: SimpleFeatureEncoding,
219       batchSize: Int,
220       ipcOpts: IpcOption,
221       sort: Option[(String, Boolean)],
222       processDeltas: Boolean,
223     ) extends LocalScanProcessor {
224 
225     def this(sft: SimpleFeatureType, hints: Hints) =
226       this(sft, hints.getArrowDictionaryFields, ArrowScan.getEncoding(hints), ArrowScan.getBatchSize(hints),
227         ArrowScan.getIpcOpts(hints), hints.getSortFields.map(_.head).orElse(hints.getArrowSort), hints.isArrowProcessDeltas)
228 
229     override val reducer: Option[FeatureReducer] =
230       Some(new DeltaReducer(sft, dictionaryFields, encoding, ipcOpts, batchSize, sort, sorted = true, processDeltas))
231 
232     override def apply(features: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = {
233       val writer = new DeltaWriter(sft, dictionaryFields, encoding, ipcOpts, None, batchSize)
234       val array = Array.ofDim[SimpleFeature](batchSize)
235 
236       val sf = ArrowScan.resultFeature()
237 
238       val arrows = features.grouped(batchSize).map { group =>
239         var index = 0
240         group.foreach { sf =>
241           array(index) = sf
242           index += 1
243         }
244         sf.setAttribute(0, writer.encode(array, index))
245         sf
246       }
247       CloseableIterator(arrows, CloseWithLogging(Seq(writer, features)))
248     }
249   }

252    * Processor for local BIN transforms
253    *
254    * @param sft simple feature type being queried
255    * @param trackId bin track id
256    * @param geom bin geom field
257    * @param dtg bin dtg field
258    * @param label bin label field

260   class BinProcessor(
261       sft: SimpleFeatureType,
262       trackId: Option[Int],
263       geom: Option[Int],
264       dtg: Option[Int],
265       label: Option[Int],
266     ) extends LocalScanProcessor {
267 
268     def this(sft: SimpleFeatureType, hints: Hints) =
269       this(sft, Option(hints.getBinTrackIdField).filter(_ != "id").map(sft.indexOf), hints.getBinGeomField.map(sft.indexOf),
270         hints.getBinDtgField.map(sft.indexOf), hints.getBinLabelField.map(sft.indexOf))
271 
272     override def reducer: Option[FeatureReducer] = None
273 
274     override def apply(features: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = {
275       val encoder = BinaryOutputEncoder(sft, EncodingOptions(geom, dtg, trackId, label))
276       val os = new ByteArrayOutputStream(1024)
277       val callback = new ByteStreamCallback(os)
278 
279       val bins = features.grouped(64).map { group =>
280         os.reset()
281         group.foreach(encoder.encode(_, callback))
282         new ScalaSimpleFeature(BinaryOutputEncoder.BinEncodedSft, "", Array(os.toByteArray, GeometryUtils.zeroPoint))
283       }
284       CloseableIterator(bins, features.close())
285     }
286   }
