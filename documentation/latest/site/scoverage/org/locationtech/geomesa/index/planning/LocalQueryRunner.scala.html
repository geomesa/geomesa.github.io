71   private def run(sft: SimpleFeatureType, query: Query, explain: Explainer)(): CloseableIterator[SimpleFeature] = {
72     explain.pushLevel(s"$name query: '${sft.getTypeName}' ${org.locationtech.geomesa.filter.filterToString(query.getFilter)}")
73     explain(s"bin[${query.getHints.isBinQuery}] arrow[${query.getHints.isArrowQuery}] " +
74         s"density[${query.getHints.isDensityQuery}] stats[${query.getHints.isStatsQuery}] " +
75         s"sampling[${query.getHints.getSampling.map { case (s, f) => s"$s${f.map(":" + _).getOrElse("")}"}.getOrElse("none")}]")
76     explain(s"Transforms: ${query.getHints.getTransformDefinition.getOrElse("None")}")
77     explain(s"Sort: ${query.getHints.getSortFields.map(QueryHints.sortReadableString).getOrElse("none")}")
78     explain.popLevel()
79 
80     val filter = Option(query.getFilter).filter(_ != Filter.INCLUDE)
81     val visible = LocalQueryRunner.visible(authProvider)
82     val iter = features(sft, filter).filter(visible.apply)
83 
84     val hook = Some(ArrowDictionaryHook(stats, filter))
85     var result = transform(sft, iter, query.getHints.getTransform, query.getHints, hook)
86 
87     query.getHints.getMaxFeatures.foreach { maxFeatures =>
88       if (query.getHints.getReturnSft == BinaryOutputEncoder.BinEncodedSft) {
89         // bin queries pack multiple records into each feature
90         // to count the records, we have to count the total bytes coming back, instead of the number of features
91         val label = query.getHints.getBinLabelField.isDefined
92         result = new BinaryOutputEncoder.FeatureLimitingIterator(result, maxFeatures, label)
93       } else {
94         result = result.take(maxFeatures)
95       }
96     }
97 
98     query.getHints.getProjection.foreach { crs =>
99       val r = Reprojection(query.getHints.getReturnSft, crs)
100       result = result.map(r.apply)
101     }
102 
103     result
104   }

117     }
118   }

150       private var sft: SimpleFeatureType,
151       private var filter: Option[Filter],
152       private var visibility: Option[SimpleFeature => Boolean],
153       private var transform: Option[(String, SimpleFeatureType)],

160     override def init(state: Map[String, String]): Unit = {
161       sft = SimpleFeatureTypes.createType(state("name"), state("spec"))
162       filter = state.get("filt").filterNot(_.isEmpty).map(ECQL.toFilter)
163       hints = ViewParams.deserialize(state("hint"))
164       transform = for {
165         tdef <- state.get("tdef").filterNot(_.isEmpty)
166         tnam <- state.get("tnam").filterNot(_.isEmpty)
167         spec <- state.get("tsft").filterNot(_.isEmpty)
168       } yield {
169         (tdef, SimpleFeatureTypes.createType(tnam, spec))
170       }
171     }

173     override def state: Map[String, String] = {
174       if (visibility.isDefined) {
175         throw new NotImplementedError("Visibility filtering is not serializable")
176       }
177       Map(
178         "name" -> sft.getTypeName,
179         "spec" -> SimpleFeatureTypes.encodeType(sft, includeUserData = true),
180         "filt" -> filter.map(ECQL.toCQL).getOrElse(""),
181         "tdef" -> transform.map(_._1).getOrElse(""),
182         "tnam" -> transform.map(_._2.getTypeName).getOrElse(""),
183         "tsft" -> transform.map(t => SimpleFeatureTypes.encodeType(t._2, includeUserData = true)).getOrElse(""),
184         "hint" -> ViewParams.serialize(hints)
185       )
186     }

188     override def apply(features: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = {
189       val filtered = (filter, visibility) match {
190         case (None, None)       => features
191         case (Some(f), None)    => features.filter(f.evaluate)
192         case (None, Some(v))    => features.filter(v.apply)
193         case (Some(f), Some(v)) => features.filter(feature => v(feature) && f.evaluate(feature))
194       }
195       LocalQueryRunner.transform(sft, filtered, transform, hints, arrow)
196     }
197   }

199   /**
200     * Transform plain features into the appropriate return type, based on the hints
201     *
202     * @param sft simple feature type being queried
203     * @param features plain, untransformed features matching the simple feature type
204     * @param hints query hints
205     * @param arrow stats hook and cql filter - used for dictionary building in certain arrow queries
206     * @return
207     */
208   def transform(
209       sft: SimpleFeatureType,
210       features: CloseableIterator[SimpleFeature],
211       transform: Option[(String, SimpleFeatureType)],
212       hints: Hints,
213       arrow: Option[ArrowDictionaryHook] = None): CloseableIterator[SimpleFeature] = {
214     val sampled = hints.getSampling match {
215       case None => features
216       case Some((percent, field)) => sample(sft, percent, field)(features)
217     }
218     if (hints.isBinQuery) {
219       val trackId = Option(hints.getBinTrackIdField).filter(_ != "id").map(sft.indexOf)
220       val geom = hints.getBinGeomField.map(sft.indexOf)
221       val dtg = hints.getBinDtgField.map(sft.indexOf)
222       val sorting = hints.isBinSorting
223       binTransform(sampled, sft, trackId, geom, dtg, hints.getBinLabelField.map(sft.indexOf), sorting)
224     } else if (hints.isArrowQuery) {
225       arrowTransform(sampled, sft, transform, hints, arrow)
226     } else if (hints.isDensityQuery) {
227       val Some(envelope) = hints.getDensityEnvelope
228       val Some((width, height)) = hints.getDensityBounds
229       val geom = DensityScan.getDensityGeometry(sft, hints)
230       densityTransform(sampled, sft, geom, envelope, width, height, hints.getDensityWeight)
231     } else if (hints.isStatsQuery) {
232       statsTransform(sampled, sft, transform, hints.getStatsQuery, hints.isStatsEncode || hints.isSkipReduce)
233     } else {
234       transform match {
235         case None => noTransform(sampled, hints.getSortFields)
236         case Some((defs, tsft)) => projectionTransform(sampled, sft, tsft, defs, hints.getSortFields)
237       }
238     }
239   }

241   private def binTransform(
242       features: CloseableIterator[SimpleFeature],
243       sft: SimpleFeatureType,
244       trackId: Option[Int],
245       geom: Option[Int],
246       dtg: Option[Int],
247       label: Option[Int],
248       sorting: Boolean): CloseableIterator[SimpleFeature] = {
249     val encoder = BinaryOutputEncoder(sft, EncodingOptions(geom, dtg, trackId, label))
250     val sorted = if (!sorting) { features } else {
251       val i = dtg.orElse(sft.getDtgIndex).getOrElse(throw new IllegalArgumentException("Can't sort BIN features by date"))
252       new SortingSimpleFeatureIterator(features, Seq(sft.getDescriptor(i).getLocalName -> false))
253     }
254 
255     val os = new ByteArrayOutputStream(1024)
256     val callback = new ByteStreamCallback(os)
257 
258     new CloseableIterator[SimpleFeature] {
259       override def hasNext: Boolean = sorted.hasNext
260       override def next(): SimpleFeature = {
261         os.reset()
262         sorted.take(64).foreach(encoder.encode(_, callback))
263         new ScalaSimpleFeature(BinaryOutputEncoder.BinEncodedSft, "", Array(os.toByteArray, GeometryUtils.zeroPoint))
264       }
265       override def close(): Unit = sorted.close()
266     }
267   }

269   private def arrowTransform(
270       original: CloseableIterator[SimpleFeature],
271       sft: SimpleFeatureType,
272       transform: Option[(String, SimpleFeatureType)],
273       hints: Hints,
274       hook: Option[ArrowDictionaryHook]): CloseableIterator[SimpleFeature] = {
275 
276     val sort = hints.getArrowSort.map(Seq.fill(1)(_))
277     val batchSize = ArrowScan.getBatchSize(hints)
278     val encoding = SimpleFeatureEncoding.min(hints.isArrowIncludeFid, hints.isArrowProxyFid)
279     val ipcOpts = FormatVersion.options(hints.getArrowFormatVersion.getOrElse(FormatVersion.ArrowFormatVersion.get))
280 
281     val (features, arrowSft) = transform match {
282       case None => (noTransform(original, sort), sft)
283       case Some((definitions, tsft)) => (projectionTransform(original, sft, tsft, definitions, sort), tsft)
284     }
285 
286     lazy val ArrowDictionaryHook(stats, filter) = hook.getOrElse {
287       throw new IllegalStateException("Arrow query called without required hooks for dictionary lookups")
288     }
289 
290     val dictionaryFields = hints.getArrowDictionaryFields
291     val writer = new DeltaWriter(arrowSft, dictionaryFields, encoding, ipcOpts, None, batchSize)
292     val array = Array.ofDim[SimpleFeature](batchSize)
293 
294     val sf = ArrowScan.resultFeature()
295 
296     val arrows = new CloseableIterator[SimpleFeature] {
297       override def hasNext: Boolean = features.hasNext
298       override def next(): SimpleFeature = {
299         var index = 0
300         while (index < batchSize && features.hasNext) {
301           array(index) = features.next
302           index += 1
303         }
304         sf.setAttribute(0, writer.encode(array, index))
305         sf
306       }
307       override def close(): Unit = CloseWithLogging(Seq(features, writer))
308     }
309     if (hints.isSkipReduce) { arrows } else {
310       val process = hints.isArrowProcessDeltas
311       new ArrowScan.DeltaReducer(arrowSft, dictionaryFields, encoding, ipcOpts, batchSize, sort.map(_.head), sorted = true, process)(arrows)
312     }
313   }

316       features: CloseableIterator[SimpleFeature],

333   private def statsTransform(features: CloseableIterator[SimpleFeature],
334                              sft: SimpleFeatureType,
335                              transform: Option[(String, SimpleFeatureType)],
336                              query: String,
337                              encode: Boolean): CloseableIterator[SimpleFeature] = {
338     val (statSft, toObserve) = transform match {
339       case None                => (sft, features)
340       case Some((tdefs, tsft)) => (tsft, projectionTransform(features, sft, tsft, tdefs, None))
341     }
342     val stat = Stat(statSft, query)
343     try { toObserve.foreach(stat.observe) } finally { toObserve.close() }
344     val encoded = if (encode) { StatsScan.encodeStat(statSft)(stat) } else { stat.toJson }
345     CloseableIterator(Iterator(new ScalaSimpleFeature(StatsScan.StatsSft, "stat", Array(encoded, GeometryUtils.zeroPoint))))
346   }

348   private def projectionTransform(
349       features: CloseableIterator[SimpleFeature],
350       sft: SimpleFeatureType,
351       transform: SimpleFeatureType,
352       definitions: String,
353       sort: Option[Seq[(String, Boolean)]]): CloseableIterator[SimpleFeature] = {
354     val transformSf = TransformSimpleFeature(sft, transform, definitions)
355 
356     def setValues(from: SimpleFeature, to: ScalaSimpleFeature): ScalaSimpleFeature = {
357       transformSf.setFeature(from)
358       var i = 0
359       while (i < transform.getAttributeCount) {
360         to.setAttributeNoConvert(i, transformSf.getAttribute(i))
361         i += 1
362       }
363       to.setId(from.getID)
364       to
365     }
366 
367     val result = features.map(setValues(_, new ScalaSimpleFeature(transform, "")))
368 
369     sort match {
370       case None    => result
371       case Some(s) => new SortingSimpleFeatureIterator(result, s)
372     }
373   }

384   /**
385     * Sample the features by selecting a subset
386     *
387     * @param sft simple feature type
388     * @param percent percent of features to keep
389     * @param by field to group by for sampling
390     * @param features features to sample
391     * @return
392     */
393   private def sample(sft: SimpleFeatureType, percent: Float, by: Option[String])
394                     (features: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = {
395     if (!(percent > 0 && percent < 1f)) {
396       throw new IllegalArgumentException(s"Sampling must be a percentage between (0, 1): $percent")
397     }
398     val nth = (1 / percent.toFloat).toInt
399     val field = by.map { name =>
400       val i = sft.indexOf(name)
401       if (i == -1) {
402         throw new IllegalArgumentException(s"Invalid sampling field '$name' for schema " +
403             s"${sft.getTypeName} ${SimpleFeatureTypes.encodeType(sft)}")
404       }
405       i
406     }
407 
408     if (nth <= 1) { features } else {
409       val sample = FeatureSampler.sample(nth, field)
410       features.filter(sample.apply)
411     }
412   }
