39   override def fromAttributeFlag(sft: SimpleFeatureType, descriptor: AttributeDescriptor, flag: String): Seq[IndexId] = {
40     if (java.lang.Boolean.valueOf(flag) || flag.equalsIgnoreCase(IndexCoverage.FULL.toString)) {
41       val indices = Seq(Z3Index, XZ3Index, Z2Index, XZ2Index, AttributeIndex).flatMap(_.indexFor(sft, descriptor))
42       if (indices.isEmpty) {
43         throw new IllegalArgumentException(
44           s"Attribute '${descriptor.getLocalName}' is configured for indexing but it is not a supported type: " +
45             descriptor.getType.getBinding.getName)
46       }
47       indices
48     } else {
49       super.fromAttributeFlag(sft, descriptor, flag)
50     }
51   }

105 /**
106  * Base feature index factory trait
107  */
108 trait DefaultFeatureIndexFactory extends GeoMesaFeatureIndexFactory with LazyLogging {
109 
110   override def all(): Seq[ConfiguredIndex]
111 
112   override def fromFeatureFlag(sft: SimpleFeatureType, flag: String): Seq[IndexId] = {
113     // check for name:attr[:attr] and name:version[:attr]
114     val Array(name, secondary@_*) = flag.split(":")
115     all().find(_.name.equalsIgnoreCase(name)).toSeq.flatMap { i =>
116       lazy val version = Try(secondary.head.toInt).toOption
117       if (secondary.isEmpty) {
118         i.defaultIndicesFor(sft)
119       } else if (i.supports(sft, secondary)) {
120         Some(IndexId(i.name, i.version, secondary))
121       } else if (version.isDefined && i.supports(sft, secondary.tail)) {
122         Some(IndexId(i.name, version.get, secondary.tail))
123       } else {
124         None
125       }
126     }
127   }
128 
129   override def fromAttributeFlag(sft: SimpleFeatureType, descriptor: AttributeDescriptor, flag: String): Seq[IndexId] = {
130     if (flag.equalsIgnoreCase("false") || flag.equalsIgnoreCase("none")) {
131       Seq(IndexId(EmptyIndex.name, EmptyIndex.version, Seq(descriptor.getLocalName)))
132     } else {
133       // check for name:attr[:attr] and name:version[:attr]
134       val Array(name, secondary@_*) = flag.split(":")
135       all().find(_.name.equalsIgnoreCase(name)).toSeq.flatMap { i =>
136         lazy val version = Try(secondary.head.toInt).toOption
137         if (secondary.isEmpty) {
138           i.indexFor(sft, descriptor)
139         } else if (i.supports(sft, Seq(descriptor.getLocalName) ++ secondary)) {
140           Some(IndexId(i.name, i.version, Seq(descriptor.getLocalName) ++ secondary))
141         } else if (version.isDefined && i.supports(sft, Seq(descriptor.getLocalName) ++ secondary.tail)) {
142           Some(IndexId(i.name, version.get, Seq(descriptor.getLocalName) ++ secondary.tail))
143         } else {
144           None
145         }
146       }
147     }
148   }
