28 /**
29   * Feature index factory providing the default indices shipped with geomesa. Note: this class is not SPI loaded

31 object DefaultFeatureIndexFactory extends GeoMesaFeatureIndexFactory {

33   private val available = Seq(Z3Index, XZ3Index, Z2Index, XZ2Index, S3Index, S2Index, IdIndex, AttributeIndex)
34 
35   override def indices(sft: SimpleFeatureType, hint: Option[String]): Seq[IndexId] = {
36     hint match {
37       case None => available.flatMap(i => i.defaults(sft).map(IndexId(i.name, i.version, _)))
38       case Some(h) => fromId(h).orElse(fromName(sft, h)).orElse(fromNameAndAttributes(sft, h)).getOrElse(Seq.empty)
39     }
40   }
41 
42   override def available(sft: SimpleFeatureType): Seq[(String, Int)] =
43     available.collect { case i if i.defaults(sft).exists(i.supports(sft, _)) => (i.name, i.version) }
44

98   private def fromId(hint: String): Option[Seq[IndexId]] = Try(Seq(IndexId.id(hint))).toOption
99 
100   private def fromName(sft: SimpleFeatureType, hint: String): Option[Seq[IndexId]] = {
101     available.find(i => hint.equalsIgnoreCase(i.name)).flatMap { i =>
102       val defaults = i.defaults(sft).collect {
103         case attributes if i.supports(sft, attributes) => IndexId(i.name, i.version, attributes)
104       }
105       if (defaults.isEmpty) { None } else { Some(defaults) }
106     }
107   }
108 
109   private def fromNameAndAttributes(sft: SimpleFeatureType, hint: String): Option[Seq[IndexId]] = {
110     val Array(name, attributes @ _*) = hint.split(":")
111     available.find(i => name.equalsIgnoreCase(i.name)).flatMap { i =>
112       if (i.supports(sft, attributes)) { Some(Seq(IndexId(i.name, i.version, attributes))) } else { None }
113     }
114   }
