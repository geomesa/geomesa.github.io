22 /**
23   * The MinMax stat merely returns the min/max of an attribute's values.
24   * Works with dates, integers, longs, doubles, and floats.
25   *
26   * @param sft simple feature type
27   * @param property property name for the attribute being min/maxed
28   * @tparam T the type of the attribute the stat is targeting (needs to be comparable)
29  */
30 class MinMax[T] private [stats] (val sft: SimpleFeatureType,
31                                  val property: String,
32                                  private [stats] var minValue: T,
33                                  private [stats] var maxValue: T,
34                                  private [stats] val hpp: HyperLogLog)
35                                 (implicit val defaults: MinMax.MinMaxDefaults[T])
36     extends Stat with LazyLogging with Serializable {
37 
38   // use a secondary constructor instead of companion apply to allow mixin types (i.e. ImmutableStat)
39   def this(sft: SimpleFeatureType, attribute: String)(implicit defaults: MinMax.MinMaxDefaults[T]) =
40     this(sft, attribute, defaults.max, defaults.min, HyperLogLog(CardinalityBits))(defaults)
41 
42   override type S = MinMax[T]
43 
44   private val i = sft.indexOf(property)
45 
46   def min: T = if (isEmpty) { maxValue } else { minValue }
47   def max: T = if (isEmpty) { minValue } else { maxValue }
48   def bounds: (T, T) = (min, max)
49   def cardinality: Long = hpp.cardinality
50   def tuple: (T, T, Long) = (min, max, cardinality)
51 
52   override def observe(sf: SimpleFeature): Unit = {
53     val value = sf.getAttribute(i).asInstanceOf[T]
54     if (value != null) {
55       try {
56         minValue = defaults.min(value, minValue)
57         maxValue = defaults.max(value, maxValue)
58         hpp.offer(value)
59       } catch {
60         case e: Exception => logger.warn(s"Error observing value '$value': ${e.toString}")
61       }
62     }
63   }
64 
65   // note: can't unobserve min/max without storing a lot more data
66   override def unobserve(sf: SimpleFeature): Unit = {}
67 
68   override def +(other: MinMax[T]): MinMax[T] = {
69     if (other.isEmpty) {
70       new MinMax[T](sft, property, minValue, maxValue, hpp.merge())
71     } else if (this.isEmpty) {
72       new MinMax[T](sft, property, other.minValue, other.maxValue, other.hpp.merge())
73     } else {
74       val plus = new MinMax[T](sft, property, minValue, maxValue, hpp.merge())
75       plus += other
76       plus
77     }
78   }
79 
80   override def +=(other: MinMax[T]): Unit = {
81     if (other.isEmpty) {
82       // no-op
83     } else if (isEmpty) {
84       minValue = other.minValue
85       maxValue = other.maxValue
86       hpp += other.hpp
87     } else {
88       minValue = defaults.min(minValue, other.minValue)
89       maxValue = defaults.max(maxValue, other.maxValue)
90       hpp += other.hpp
91     }
92   }
93 
94   override def toJsonObject: Any =
95     if (isEmpty) {
96       ListMap("min" -> null, "max" -> null, "cardinality" -> 0)
97     } else {
98       ListMap("min" -> minValue, "max" -> maxValue, "cardinality" -> cardinality)
99     }
100 
101   override def isEmpty: Boolean = minValue == defaults.max
102 
103   override def clear(): Unit = {
104     minValue = defaults.max
105     maxValue = defaults.min
106     java.util.Arrays.fill(hpp.registerSet.rawBits, 0)
107   }
108 
109   override def isEquivalent(other: Stat): Boolean = other match {
110     case that: MinMax[T] =>
111       property == that.property && minValue == that.minValue &&
112           maxValue == that.maxValue && cardinality == that.cardinality
113     case _ => false
114   }
115 }

121   trait MinMaxDefaults[T] {
122     def min: T
123     def max: T
124     def min(left: T, right: T): T
125     def max(left: T, right: T): T
126   }

128   object MinMaxDefaults {
129     def apply[T](binding: Class[_]): MinMaxDefaults[T] = {
130       if (binding == classOf[String]) {
131         MinMaxString.asInstanceOf[MinMaxDefaults[T]]
132       } else if (binding == classOf[Integer]) {
133         MinMaxInt.asInstanceOf[MinMaxDefaults[T]]
134       } else if (binding == classOf[java.lang.Long]) {
135         MinMaxLong.asInstanceOf[MinMaxDefaults[T]]
136       } else if (binding == classOf[java.lang.Float]) {
137         MinMaxFloat.asInstanceOf[MinMaxDefaults[T]]
138       } else if (binding == classOf[java.lang.Double]) {
139         MinMaxDouble.asInstanceOf[MinMaxDefaults[T]]
140       } else if (classOf[Date].isAssignableFrom(binding)) {
141         MinMaxDate.asInstanceOf[MinMaxDefaults[T]]
142       } else if (classOf[Geometry].isAssignableFrom(binding)) {
143         MinMaxGeometry.asInstanceOf[MinMaxDefaults[T]]
144       } else {
145         throw new IllegalArgumentException(s"No implicit default available for type: $binding")
146       }
147     }
148   }
149 
150   abstract class ComparableMinMax[T <: Comparable[T]] extends MinMaxDefaults[T] with Serializable {
151     override def min(left: T, right: T): T = if (left.compareTo(right) > 0) right else left
152     override def max(left: T, right: T): T = if (left.compareTo(right) < 0) right else left
153   }
154

185   /**
186     * Geometry min/max tracks the bounding box of each geometry, not the geometries themselves.

188   implicit object MinMaxGeometry extends MinMaxDefaults[Geometry] with Serializable {
189

193     override val max: Geometry = gf.createPoint(new Coordinate(180.0, 90.0))

195     override def min(left: Geometry, right: Geometry): Geometry = {
196       val (lx, ly) = { val e = left.getEnvelopeInternal; (e.getMinX, e.getMinY) }
197       val (rx, ry) = { val e = right.getEnvelopeInternal; (e.getMinX, e.getMinY) }
198 
199       val x = math.min(lx, rx)
200       val y = math.min(ly, ry)
201 
202       if (x == lx && y == ly) {
203         left
204       } else if (x == rx && y == ry) {
205         right
206       } else {
207         gf.createPoint(new Coordinate(x, y))
208       }
209     }
210

212       val (lx, ly) = { val e = left.getEnvelopeInternal; (e.getMaxX, e.getMaxY) }
213       val (rx, ry) = { val e = right.getEnvelopeInternal; (e.getMaxX, e.getMaxY) }
214 
215       val x = math.max(lx, rx)
216       val y = math.max(ly, ry)
217

219         left
220       } else if (x == rx && y == ry) {
221         right

224       }
225     }
