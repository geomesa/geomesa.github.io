22 /**
23  * Writes an audited event
24  *
25  * @param storeType store type
26  * @param auditProvider audit provider
27  */
28 abstract class AuditWriter(val storeType: String, val auditProvider: AuditProvider) extends Closeable {
29 
30   /**
31    * Writes a query event asynchronously
32    *
33    * @param typeName type name
34    * @param user user making the query
35    * @param filter query filter
36    * @param hints query hints
37    * @param plans query plans being executed
38    * @param startTime time query started, in millis since the epoch
39    * @param endTime time query completed, in millis since the epoch
40    * @param planTime time spent planning the query execution, in millis
41    * @param scanTime time spent executing the query, in millis
42    * @param hits number of results
43    * @return
44    */
45   def writeQueryEvent(
46       typeName: String,
47       user: String,
48       filter: Filter,
49       hints: Hints,
50       plans: Seq[QueryPlan],
51       startTime: Long,
52       endTime: Long,
53       planTime: Long,
54       scanTime: Long,
55       hits: Long): Future[Unit]
56 
57   override def close(): Unit = {}
58 }

60 object AuditWriter {

62   val Gson: Gson = new GsonBuilder().disableHtmlEscaping().serializeNulls().create()

66   /**
67    * Log a query event
