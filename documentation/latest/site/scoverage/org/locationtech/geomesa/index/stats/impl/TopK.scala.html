18 /**
19   * TopK stat
20   *
21   * @param sft simple feature type
22   * @param property property name of the attribute to track
23   * @param summary stream summary object
24   * @tparam T attribute type binding
25   */
26 class TopK[T] private [stats] (
27     val sft: SimpleFeatureType,
28     val property: String,
29     private [stats] val summary: StreamSummary[T]
30   ) extends Stat with LazyLogging {
31 
32   import TopK.StreamCapacity
33 
34   def this(sft: SimpleFeatureType, property: String) = this(sft, property, StreamSummary[T](TopK.StreamCapacity))
35 
36   override type S = TopK[T]
37 
38   private val i = sft.indexOf(property)
39 
40   def topK(k: Int): Iterator[(T, Long)] = summary.topK(k)
41   def size: Int = summary.size
42 
43   override def observe(sf: SimpleFeature): Unit = {
44     val value = sf.getAttribute(i).asInstanceOf[T]
45     if (value != null) {
46       summary.offer(value)
47     }
48   }
49 
50   override def unobserve(sf: SimpleFeature): Unit = {
51     val value = sf.getAttribute(i).asInstanceOf[T]
52     if (value != null) {
53       summary.offer(value, -1)
54     }
55   }
56 
57   override def +(other: TopK[T]): TopK[T] = {
58     val merged = new TopK[T](sft, property)
59     merged += this
60     merged += other
61     merged
62   }
63 
64   override def +=(other: TopK[T]): Unit =
65     other.summary.topK(StreamCapacity).foreach { case (item, count) => summary.offer(item, count) }
66 
67   override def clear(): Unit = summary.clear()
68 
69   override def isEmpty: Boolean = summary.size == 0
70 
71   override def toJsonObject: Any = {
72     val maps = summary.topK(10).zipWithIndex.map { case ((item, count), rank) =>
73       (rank, ListMap( "value" -> item, "count" -> count))
74     }
75     ListMap(maps.toSeq:_*)
76   }
77 
78   override def isEquivalent(other: Stat): Boolean = other match {
79     case s: TopK[T] if summary.size == s.summary.size =>
80       s.summary.topK(summary.size).sameElements(summary.topK(summary.size))
81     case _ => false
82   }
83 }
