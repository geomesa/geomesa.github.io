80     override def getCount(sft: SimpleFeatureType, filter: Filter, exact: Boolean, queryHints: Hints): Option[Long] = {
81       // note: unlike most methods in this class, this will return if any of the merged stores provide a response
82       def getSingle(statAndFilter: (GeoMesaStats, Option[Filter])): Option[Long] =
83         statAndFilter._1.getCount(sft, mergeFilter(sft, filter, statAndFilter._2), exact, queryHints)

169     private def merge[T <: Stat](query: (GeoMesaStats, Option[Filter]) => Option[T]): Option[T] = {
170       if (parallel) {
171         val results = new CopyOnWriteArrayList[Option[T]]()
172         stats.toList.map { case (s, f) => CachedThreadPool.submit(() => results.add(query(s, f))) }.foreach(_.get)
173         results.asScala.reduceLeft((res, next) => for { r <- res; n <- next } yield { (r + n).asInstanceOf[T] })
174       } else {
175         // lazily evaluate each stat as we only return Some if all the child stores do
176         val head = query(stats.head._1, stats.head._2)
177         stats.tail.foldLeft(head) { case (result, (stat, filter)) =>
178           for { r <- result; n <- query(stat, filter) } yield { (r + n).asInstanceOf[T] }
179         }
180       }
181     }
182   }
