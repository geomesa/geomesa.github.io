17 /**
18   * An enumeration is merely a HashMap mapping values to number of occurrences
19   *
20   * @param sft simple feature type
21   * @param property property name the enumeration is being made for
22   * @tparam T some type T (which is restricted by the stat parser upstream of Histogram instantiation)
23   */
24 class EnumerationStat[T](val sft: SimpleFeatureType, val property: String)(implicit val ct: ClassTag[T]) extends Stat {
25 
26   override type S = EnumerationStat[T]
27 
28   private val i = sft.indexOf(property)
29   private [stats] val enumeration = scala.collection.mutable.HashMap.empty[T, Long].withDefaultValue(0)
30 
31   def size: Int = enumeration.size
32   def values: Iterable[T] = enumeration.keys
33   def frequency(value: T): Long = enumeration(value)
34   def frequencies: Iterable[(T, Long)] = enumeration
35 
36   override def observe(sf: SimpleFeature): Unit = {
37     val value = sf.getAttribute(i).asInstanceOf[T]
38     if (value != null) {
39       enumeration(value) += 1
40     }
41   }
42 
43   override def unobserve(sf: SimpleFeature): Unit = {
44     val value = sf.getAttribute(i).asInstanceOf[T]
45     if (value != null) {
46       val current = enumeration(value)
47       if (current == 1) {
48         enumeration.remove(value)
49       } else {
50         enumeration(value) = current - 1
51       }
52     }
53   }
54 
55   override def +(other: EnumerationStat[T]): EnumerationStat[T] = {
56     val plus = new EnumerationStat[T](sft, property)
57     plus += this
58     plus += other
59     plus
60   }
61 
62   override def +=(other: EnumerationStat[T]): Unit =
63     other.enumeration.foreach { case (key, count) => enumeration(key) += count }
64 
65   override def toJsonObject: Map[T, Long] =
66     if (enumeration.isEmpty) { Map.empty } else { ListMap(enumeration.toSeq.sortBy(_.toString):_*) }
67 
68   override def isEmpty: Boolean = enumeration.isEmpty
69 
70   override def clear(): Unit = enumeration.clear()
71 
72   override def isEquivalent(other: Stat): Boolean = other match {
73     case that: EnumerationStat[_] => property == that.property && enumeration == that.enumeration
74     case _ => false
75   }
76 }
