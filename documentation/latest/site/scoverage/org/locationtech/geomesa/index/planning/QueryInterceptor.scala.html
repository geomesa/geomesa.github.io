24 /**
25   * Provides a hook to modify a query before executing it
26   */
27 trait QueryInterceptor extends Closeable {
28 
29   /**
30     * Called exactly once after the interceptor is instantiated
31     *
32     * @param ds data store
33     * @param sft simple feature type
34     */
35   def init(ds: DataStore, sft: SimpleFeatureType): Unit
36 
37   /**
38     * Modifies the query in place
39     *
40     * @param query query
41     */
42   def rewrite(query: Query): Unit
43 
44   /**
45    * Hook to allow interception of a query after extracting the query values
46    *
47    * @param strategy query strategy
48    * @return an exception if the query should be stopped
49    */
50   def guard(strategy: QueryStrategy): Option[IllegalArgumentException] = None
51 }
52 
53 object QueryInterceptor extends LazyLogging {
54 
55   /**
56     * Manages query interceptors
57     */
58   trait QueryInterceptorFactory extends Closeable {
59     def apply(sft: SimpleFeatureType): Seq[QueryInterceptor]
60   }
61 
62   object QueryInterceptorFactory {
63 
64     private val Empty: QueryInterceptorFactory = new QueryInterceptorFactory {
65       override def apply(sft: SimpleFeatureType): Seq[QueryInterceptor] = Seq.empty
66       override def close(): Unit = {}
67     }
68 
69     def apply(ds: DataStore): QueryInterceptorFactory = new QueryInterceptorFactoryImpl(ds)
70 
71     def empty(): QueryInterceptorFactory = Empty
72 
73     /**
74       * Manages query interceptor lifecycles for a given data store
75       *
76       * @param ds data store
77       */
78     private class QueryInterceptorFactoryImpl(ds: DataStore) extends QueryInterceptorFactory {
79 
80       import scala.collection.JavaConverters._
81 
82       private val expiry = TableBasedMetadata.Expiry.toDuration.get.toMillis
83 
84       private val loader = new CacheLoader[String, Seq[QueryInterceptor]]() {
85         override def load(key: String): Seq[QueryInterceptor] = QueryInterceptorFactoryImpl.this.load(key)
86         override def reload(key: String, oldValue: Seq[QueryInterceptor]): Seq[QueryInterceptor] = {
87           // only recreate the interceptors if they have changed
88           Option(ds.getSchema(key)).flatMap(s => Option(s.getUserData.get(Configs.QueryInterceptors))) match {
89             case None if oldValue.isEmpty => oldValue
90             case Some(classes) if classes == oldValue.map(_.getClass.getName).mkString(",") => oldValue
91             case _ => CloseWithLogging(oldValue); QueryInterceptorFactoryImpl.this.load(key)
92           }
93         }
94       }
95 
96       private val cache: LoadingCache[String, Seq[QueryInterceptor]] =
97         Caffeine.newBuilder().refreshAfterWrite(expiry, TimeUnit.MILLISECONDS).build(loader)
98 
99       override def apply(sft: SimpleFeatureType): Seq[QueryInterceptor] = cache.get(sft.getTypeName)
100 
101       override def close(): Unit = {
102         cache.asMap.asScala.foreach { case (_, interceptors) => CloseWithLogging(interceptors) }
103         cache.invalidateAll()
104       }
105 
106       private def load(typeName: String): Seq[QueryInterceptor] = {
107         val sft = ds.getSchema(typeName)
108         if (sft == null) { Seq.empty } else {
109           val classes = sft.getUserData.get(Configs.QueryInterceptors).asInstanceOf[String]
110           if (classes == null || classes.isEmpty) { Seq.empty } else {
111             classes.split(",").toSeq.flatMap { c =>
112               var interceptor: QueryInterceptor = null
113               try {
114                 interceptor = Class.forName(c).getDeclaredConstructor().newInstance().asInstanceOf[QueryInterceptor]
115                 interceptor.init(ds, sft)
116                 Seq(interceptor)
117               } catch {
118                 case NonFatal(e) =>
119                   logger.error(s"Error creating query interceptor '$c':", e)
120                   if (interceptor != null) {
121                     CloseWithLogging(interceptor)
122                   }
123                   Seq.empty
124               }
125             }
126           }
127         }
128       }
129     }
130   }
