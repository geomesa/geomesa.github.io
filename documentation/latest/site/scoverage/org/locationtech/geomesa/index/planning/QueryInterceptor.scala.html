25 /**
26   * Provides a hook to modify a query before executing it
27   */
28 trait QueryInterceptor extends Closeable {
29 
30   /**
31     * Called exactly once after the interceptor is instantiated
32     *
33     * @param ds data store
34     * @param sft simple feature type
35     */
36   def init(ds: DataStore, sft: SimpleFeatureType): Unit
37 
38   /**
39     * Modifies the query in place
40     *
41     * @param query query
42     */
43   def rewrite(query: Query): Unit
44 
45   /**
46    * Hook to allow interception of a query after extracting the query values
47    *
48    * @param strategy query strategy
49    * @return an exception if the query should be stopped
50    */
51   def guard(strategy: QueryStrategy): Option[IllegalArgumentException] = None
52 }
53 
54 object QueryInterceptor extends LazyLogging {
55 
56   /**
57     * Manages query interceptors
58     */
59   trait QueryInterceptorFactory extends Closeable {
60 
61     def apply(sft: SimpleFeatureType): Seq[QueryInterceptor]
62 
63     /**
64      * Runs any configured query guards, throwing any exceptions raised
65      *
66      * @param strategy query strategy
67      */
68     @throws[IllegalArgumentException]
69     def runGuards(strategy: QueryStrategy): Unit =
70       apply(strategy.index.sft).foreach(_.guard(strategy).foreach(e => throw e))
71   }
72 
73   object QueryInterceptorFactory {
74 
75     private val Empty: QueryInterceptorFactory = new QueryInterceptorFactory {
76       override def apply(sft: SimpleFeatureType): Seq[QueryInterceptor] = Seq.empty
77       override def close(): Unit = {}
78     }
79 
80     def apply(ds: DataStore): QueryInterceptorFactory = new QueryInterceptorFactoryImpl(ds)
81 
82     def empty(): QueryInterceptorFactory = Empty
83 
84     /**
85       * Manages query interceptor lifecycles for a given data store
86       *
87       * @param ds data store
88       */
89     private class QueryInterceptorFactoryImpl(ds: DataStore) extends QueryInterceptorFactory {
90 
91       import scala.collection.JavaConverters._
92 
93       private val expiry = TableBasedMetadata.Expiry.toDuration.get.toMillis
94 
95       private val cache =
96         Caffeine.newBuilder().refreshAfterWrite(expiry, TimeUnit.MILLISECONDS).build[String, Seq[QueryInterceptor]](
97           new CacheLoader[String, Seq[QueryInterceptor]]() {
98             override def load(typeName: String): Seq[QueryInterceptor] = reload(typeName, Seq.empty)
99             override def reload(typeName: String, oldValue: Seq[QueryInterceptor]): Seq[QueryInterceptor] = {
100               // only recreate the interceptors if they have changed
101               val sft = ds.getSchema(typeName)
102               if (sft == null) {
103                 CloseWithLogging(oldValue)
104                 return Seq.empty
105               }
106               val classes = getInterceptorClasses(sft)
107               if (classes == oldValue.map(_.getClass.getName)) {
108                 oldValue
109               } else {
110                 CloseWithLogging(oldValue)
111                 classes.flatMap(createInterceptor(ds, sft, _))
112               }
113             }
114           }
115         )
116 
117       override def apply(sft: SimpleFeatureType): Seq[QueryInterceptor] = cache.get(sft.getTypeName)
118 
119       override def close(): Unit = {
120         cache.asMap.asScala.foreach { case (_, interceptors) => CloseWithLogging(interceptors) }
121         cache.invalidateAll()
122       }
123     }
124   }
125 
126   private def getInterceptorClasses(sft: SimpleFeatureType) : Seq[String] = {
127     val configured = sft.getQueryInterceptors
128     // add default full table scan check, if the more restrictive FullTableScanQueryGuard doesn't make it redundant
129     if (Seq(classOf[FullTableScanQueryGuard], classOf[DefaultQueryGuard]).map(_.getName).exists(configured.contains)) {
130       configured
131     } else {
132       configured :+ classOf[DefaultQueryGuard].getName
133     }
134   }
135 
136   private def createInterceptor(ds: DataStore, sft: SimpleFeatureType, className: String): Option[QueryInterceptor] = {
137     var interceptor: QueryInterceptor = null
138     try {
139       interceptor = Class.forName(className).getDeclaredConstructor().newInstance().asInstanceOf[QueryInterceptor]
140       interceptor.init(ds, sft)
141       Some(interceptor)
142     } catch {
143       case NonFatal(e) =>
144         logger.error(s"Error creating query interceptor '$className':", e)
145         if (interceptor != null) {
146           CloseWithLogging(interceptor)
147         }
148         None
149     }
150   }
