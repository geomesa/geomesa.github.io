25 /**
26   * Provides a hook to modify a query before executing it
27   */
28 trait QueryInterceptor extends Closeable {
29 
30   /**
31     * Called exactly once after the interceptor is instantiated
32     *
33     * @param ds data store
34     * @param sft simple feature type
35     */
36   def init(ds: DataStore, sft: SimpleFeatureType): Unit
37 
38   /**
39     * Modifies the query in place
40     *
41     * @param query query
42     */
43   def rewrite(query: Query): Unit
44 
45   /**
46    * Hook to allow interception of a query after extracting the query values
47    *
48    * @param strategy query strategy
49    * @return an exception if the query should be stopped
50    */
51   def guard(strategy: QueryStrategy): Option[IllegalArgumentException] = None
52 }
53 
54 object QueryInterceptor extends LazyLogging {
55 
56   /**
57     * Manages query interceptors
58     */
59   trait QueryInterceptorFactory extends Closeable {
60     def apply(sft: SimpleFeatureType): Seq[QueryInterceptor]
61   }
62 
63   object QueryInterceptorFactory {
64 
65     private val Empty: QueryInterceptorFactory = new QueryInterceptorFactory {
66       override def apply(sft: SimpleFeatureType): Seq[QueryInterceptor] = Seq.empty
67       override def close(): Unit = {}
68     }
69 
70     def apply(ds: DataStore): QueryInterceptorFactory = new QueryInterceptorFactoryImpl(ds)
71 
72     def empty(): QueryInterceptorFactory = Empty
73 
74     /**
75       * Manages query interceptor lifecycles for a given data store
76       *
77       * @param ds data store
78       */
79     private class QueryInterceptorFactoryImpl(ds: DataStore) extends QueryInterceptorFactory {
80 
81       import scala.collection.JavaConverters._
82 
83       private val expiry = TableBasedMetadata.Expiry.toDuration.get.toMillis
84 
85       private val cache =
86         Caffeine.newBuilder().refreshAfterWrite(expiry, TimeUnit.MILLISECONDS).build[String, Seq[QueryInterceptor]](
87           new CacheLoader[String, Seq[QueryInterceptor]]() {
88             override def load(typeName: String): Seq[QueryInterceptor] = reload(typeName, Seq.empty)
89             override def reload(typeName: String, oldValue: Seq[QueryInterceptor]): Seq[QueryInterceptor] = {
90               // only recreate the interceptors if they have changed
91               val sft = ds.getSchema(typeName)
92               if (sft == null) {
93                 CloseWithLogging(oldValue)
94                 return Seq.empty
95               }
96               val classes = getInterceptorClasses(sft)
97               if (classes == oldValue.map(_.getClass.getName)) {
98                 oldValue
99               } else {
100                 CloseWithLogging(oldValue)
101                 classes.flatMap(createInterceptor(ds, sft, _))
102               }
103             }
104           }
105         )
106 
107       override def apply(sft: SimpleFeatureType): Seq[QueryInterceptor] = cache.get(sft.getTypeName)
108 
109       override def close(): Unit = {
110         cache.asMap.asScala.foreach { case (_, interceptors) => CloseWithLogging(interceptors) }
111         cache.invalidateAll()
112       }
113     }
114   }
115 
116   private def getInterceptorClasses(sft: SimpleFeatureType) : Seq[String] = {
117     val configured = sft.getQueryInterceptors
118     // add default full table scan check, if the more restrictive FullTableScanQueryGuard doesn't make it redundant
119     if (Seq(classOf[FullTableScanQueryGuard], classOf[DefaultQueryGuard]).map(_.getName).exists(configured.contains)) {
120       configured
121     } else {
122       configured :+ classOf[DefaultQueryGuard].getName
123     }
124   }
125 
126   private def createInterceptor(ds: DataStore, sft: SimpleFeatureType, className: String): Option[QueryInterceptor] = {
127     var interceptor: QueryInterceptor = null
128     try {
129       interceptor = Class.forName(className).getDeclaredConstructor().newInstance().asInstanceOf[QueryInterceptor]
130       interceptor.init(ds, sft)
131       Some(interceptor)
132     } catch {
133       case NonFatal(e) =>
134         logger.error(s"Error creating query interceptor '$className':", e)
135         if (interceptor != null) {
136           CloseWithLogging(interceptor)
137         }
138         None
139     }
140   }
141 }
