25 /**
26   * Encodes dates as lexicoded longs. Null values are encoded as Long.MinValue. This class is only used for
27   * secondary tiering, so it never has sharing or shards
28   */
29 class DateIndexKeySpace(val sft: SimpleFeatureType, dtgField: String)
30     extends IndexKeySpace[FilterValues[Bounds[ZonedDateTime]], Long] {
31 
32   import DateIndexKeySpace.{Empty, MaxUpperBound, MinLowerBound}
33 
34   require(classOf[Date].isAssignableFrom(sft.getDescriptor(dtgField).getType.getBinding),
35     s"Expected field $dtgField to have a date binding, but instead it has: " +
36         sft.getDescriptor(dtgField).getType.getBinding.getName)
37 
38   private val dtgIndex = sft.indexOf(dtgField)
39 
40   override val attributes: Seq[String] = Seq(dtgField)
41 
42   override val sharing: Array[Byte] = Empty
43 
44   override val sharding: ShardStrategy = NoShardStrategy
45 
46   override val indexKeyByteLength: Right[(Array[Byte], Int, Int) => Int, Int] = Right(8)
47 
48   override def toIndexKey(writable: WritableFeature,
49                           tier: Array[Byte],
50                           id: Array[Byte],
51                           lenient: Boolean): RowKeyValue[Long] = {
52     // note: only used as a tiered keyspace so tier and id will be empty
53     val dtg = writable.getAttribute[Date](dtgIndex)
54     val time = if (dtg == null) { Long.MinValue } else { dtg.getTime }
55     SingleRowKeyValue(ByteArrays.toOrderedBytes(time), Empty, Empty, time, tier, Empty, Seq.empty)
56   }
57 
58   override def getIndexValues(filter: Filter, explain: Explainer): FilterValues[Bounds[ZonedDateTime]] =
59     FilterHelper.extractIntervals(filter, dtgField)
60 
61   override def getRanges(values: FilterValues[Bounds[ZonedDateTime]], multiplier: Int): Iterator[ScanRange[Long]] = {
62     if (values.isEmpty) {
63       Iterator.single(UnboundedRange(-1))
64     } else if (values.disjoint) {
65       Iterator.empty
66     } else {
67       values.values.iterator.map { bounds =>
68         if (bounds.isEquals) {
69           SingleRowRange(bounds.lower.value.get.toInstant.toEpochMilli)
70         } else {
71           val lower = bounds.lower.value.map { v =>
72             val millis = v.toInstant.toEpochMilli
73             if (bounds.lower.inclusive) { millis } else { millis + 1L }
74           }
75           val upper = bounds.upper.value.map { v =>
76             val millis = v.toInstant.toEpochMilli
77             if (bounds.upper.inclusive) { millis + 1L } else { millis }
78           }
79           BoundedRange(lower.getOrElse(MinLowerBound), upper.getOrElse(MaxUpperBound))
80         }
81       }
82     }
83   }
84 
85   override def getRangeBytes(ranges: Iterator[ScanRange[Long]], tier: Boolean): Iterator[ByteRange] = {
86     ranges.map {
87       case BoundedRange(lo, hi) =>
88         BoundedByteRange(ByteArrays.toOrderedBytes(lo), ByteArrays.toOrderedBytes(hi))
89 
90       case SingleRowRange(row) =>
91         val ordered = ByteArrays.toOrderedBytes(row)
92         if (tier) {
93           SingleRowByteRange(ordered)
94         } else {
95           BoundedByteRange(ordered, ByteArrays.rowFollowingPrefix(ordered))
96         }
97 
98       case UnboundedRange(_) =>
99         val max = ByteArrays.rowFollowingPrefix(ByteArrays.toOrderedBytes(MaxUpperBound))
100         UnboundedByteRange(ByteArrays.toOrderedBytes(MinLowerBound), max)
101 
102       case r => throw new IllegalArgumentException(s"Unexpected range type $r")
103     }
104   }
105 
106   override def useFullFilter(
107       values: Option[FilterValues[Bounds[ZonedDateTime]]],
108       config: Option[GeoMesaDataStoreConfig],
109       hints: Hints): Boolean = false
110 }

125     new DateIndexKeySpace(sft, attributes.head)
