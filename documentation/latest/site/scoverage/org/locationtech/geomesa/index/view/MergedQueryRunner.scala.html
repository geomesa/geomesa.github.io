35 /**
36  * Query runner for merging results from multiple stores
37  *
38  * @param ds merged data store
39  * @param stores delegate stores
40  * @param deduplicate deduplicate the results between stores
41  * @param parallel run scans in parallel (vs sequentially)
42  */
43 class MergedQueryRunner(
44     ds: HasGeoMesaStats,
45     stores: Seq[(Queryable, Option[Filter])],
46     deduplicate: Boolean,
47     parallel: Boolean
48   ) extends QueryRunner with LazyLogging {
49 
50   import org.locationtech.geomesa.index.conf.QueryHints.RichHints
51 
52   // query interceptors are handled by the individual data stores
53   override protected val interceptors: QueryInterceptorFactory = QueryInterceptorFactory.empty()
54 
55   override protected def tags(typeName: String): Tags = TagUtils.typeNameTag(typeName).and("store", "merged", "catalog", "")
56 
57   override protected def getQueryPlans(sft: SimpleFeatureType, query: Query, explain: Explainer): Seq[QueryPlan] = {
58     // TODO deduplicate arrow, bin, density queries...
59     val hints = query.getHints
60     val toFeatures = new IdentityResultsToFeatures(hints.getReturnSft)
61     val plan = if (hints.isArrowQuery) {
62       val scan = () => scanner(sft, query, org.locationtech.geomesa.arrow.ArrowEncodedSft, new ArrowProcessor(_, hints))
63       val reducer = new DeltaReducer(hints.getTransformSchema.getOrElse(sft), hints, sorted = false)
64       MergedQueryPlan(scan, toFeatures, Some(reducer), hints.getMaxFeatures)
65     } else if (hints.isBinQuery) {
66       if (query.getSortBy != null && !query.getSortBy.isEmpty) {
67         logger.warn("Ignoring sort for BIN query")
68       }
69       val scan = () => scanner(sft, query, BinaryOutputEncoder.BinEncodedSft, new BinProcessor(_, hints))
70       MergedQueryPlan(scan, toFeatures, None, hints.getMaxFeatures)
71     } else if (hints.isDensityQuery) {
72       val scan = () => scanner(sft, query, DensityScan.DensitySft, new DensityProcessor(_, hints))
73       MergedQueryPlan(scan, toFeatures, None, hints.getMaxFeatures)
74     } else if (hints.isStatsQuery) {
75       val scan = () => scanner(sft, query, StatsScan.StatsSft, new StatsProcessor(_, hints))
76       val reducer = StatsScan.StatsReducer(sft, hints)
77       MergedQueryPlan(scan, toFeatures, Some(reducer), hints.getMaxFeatures)
78     } else {
79       // we assume the delegate stores can handle normal transforms/etc appropriately
80       val scanner = () => {
81         val readers = getReaders(sft, query).map(CloseableIterator(_))
82         val deduped = if (deduplicate) {
83           // we re-use the feature id cache across readers
84           val cache = scala.collection.mutable.HashSet.empty[String]
85           readers.map(new DeduplicatingSimpleFeatureIterator(_, cache))
86         } else {
87           readers
88         }
89         val sorted = hints.getSortFields match {
90           case None => doParallelScan(deduped)
91           // the delegate stores should sort their results, so we can sort merge them
92           case Some(sort) => new SortedMergeIterator(deduped)(SimpleFeatureOrdering(hints.getReturnSft, sort))
93         }
94         sorted
95       }
96       MergedQueryPlan(scanner, toFeatures, None, hints.getMaxFeatures)
97     }
98     Seq(plan)
99   }
100 
101   // query each delegate store
102   private def getReaders(sft: SimpleFeatureType, query: Query): Seq[FeatureReader[SimpleFeatureType, SimpleFeature]] = {
103     stores.map { case (store, filter) =>
104       val copy = new Query(query) // make sure to coy the hints so they aren't shared
105       // suppress the reduce step for gm stores so that we can do the merge here
106       copy.getHints.put(QueryHints.Internal.SKIP_REDUCE, java.lang.Boolean.TRUE)
107       store.getFeatureReader(mergeFilter(sft, copy, filter), Transaction.AUTO_COMMIT)
108     }
109   }
110 
111   private def scanner(
112       sft: SimpleFeatureType,
113       query: Query,
114       encodedType: SimpleFeatureType,
115       fallbackProcessor: SimpleFeatureType => LocalScanProcessor): CloseableIterator[SimpleFeature] = {
116     val readers = getReaders(sft, query).map { reader =>
117       val schema = reader.getFeatureType
118       if (schema == encodedType) {
119         // processing has been handled by the store already
120         CloseableIterator(reader)
121       } else {
122         // the store just returned normal features, do the processing here
123         val copy = SimpleFeatureTypes.immutable(schema, sft.getUserData) // copy default dtg, etc if necessary
124         fallbackProcessor(copy).apply(CloseableIterator(reader))
125       }
126     }
127     doParallelScan(readers)
128   }
129 
130   private def doParallelScan(readers: Seq[CloseableIterator[SimpleFeature]]): CloseableIterator[SimpleFeature] = {
131     if (parallel) {
132       // not truly parallel but should kick them all off up front
133       readers.foreach(_.hasNext)
134     }
135     CloseableIterator(readers.iterator).flatMap(i => i)
136   }
137 }

151       scanner: () => CloseableIterator[SimpleFeature],
152       resultsToFeatures: ResultsToFeatures[SimpleFeature],
153       reducer: Option[FeatureReducer],
154       maxFeatures: Option[Int],
155     ) extends QueryPlan {
156     override type Results = SimpleFeature
157     override def sort: Option[Seq[(String, Boolean)]] = None
158     override def projection: Option[QueryReferenceSystems] = None
159     override def scan(): CloseableIterator[SimpleFeature] = scanner()
160     override def explain(explainer: Explainer): Unit = {
161       explainer.pushLevel("MergedQueryPlan:")
162       explainer(s"Reduce: ${reducer.getOrElse("none")}")
163       explainer.popLevel()
164     }
165   }
