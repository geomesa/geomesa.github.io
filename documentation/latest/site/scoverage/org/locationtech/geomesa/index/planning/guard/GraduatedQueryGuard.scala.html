30   private var disabled: Boolean = false

32   override def init(ds: DataStore, sft: SimpleFeatureType): Unit = {
33     disabled = GraduatedQueryGuard.disabled(sft.getTypeName)
34     if (disabled) {
35       logger.info(s"This guard is disabled for schema '${sft.getTypeName}' via system property")
36     }
37     // let any errors bubble up and disable this guard
38     guardLimits = buildLimits(ConfigFactory.load().getConfigList(s"$ConfigPath.${sft.getTypeName}"), sft)
39     // this should be ensured during the loading of limits, but just double check so we can use `.last` safely below
40     require(guardLimits.nonEmpty)
41   }
42 
43   override def rewrite(query: Query): Unit = {
44     if (!disabled) {
45       val bounds = query.getFilter.accept(ExtractBoundsFilterVisitor.BOUNDS_VISITOR, null).asInstanceOf[Envelope]
46       val spatialExtent = (bounds.getMaxX - bounds.getMinX) * (bounds.getMaxY - bounds.getMinY)
47       val limit = guardLimits.find(_.sizeLimit >= spatialExtent).getOrElse{
48         logger.warn(s"Invalid extents/limits: ${query.getFilter.toString} / ${guardLimits.mkString(", ")}")
49         guardLimits.last
50       }
51       limit.percentageLimit.map{ percentage =>
52         query.getHints.put(QueryHints.SAMPLING, percentage.toFloat)
53         limit.sampleAttribute.map(attr => query.getHints.put(QueryHints.SAMPLE_BY, attr))
54       }
55     }
56   }
57 
58   override def guard(strategy: QueryStrategy): Option[IllegalArgumentException] = {
59     val msg = if (disabled || !strategy.index.isInstanceOf[SpatioTemporalIndex[_, _]]) { None } else {
60       val values = strategy.values.collect {
61         case v: SpatialIndexValues with TemporalIndexValues => (v.spatialBounds, v.intervals)
62       }
63       values match {
64         case None =>
65           Some(s"Query does not have a temporal filter. Maximum allowed filter duration for " +
66               s"whole world queries is ${guardLimits.last.durationLimit}")
67 
68         case Some((s, i)) =>
69           val spatialExtent = s.map { case (lx, ly, ux, uy) => (ux - lx) * (uy - ly) }.sum
70           val limit = guardLimits.find(_.sizeLimit >= spatialExtent).getOrElse {
71             // should always be a valid limit due to our checks building the limit...
72             logger.warn(s"Invalid extents/limits: ${s.mkString(", ")} / ${guardLimits.mkString(", ")}")
73             guardLimits.last
74           }
75           limit.durationLimit.flatMap { l =>
76             if (!validate(i, l)) Some(s"Query exceeds maximum allowed filter duration of ${limit.durationLimit} " +
77               s"at ${limit.sizeLimit} degrees") else None
78           }
79       }
80     }
81     msg.map(m => new IllegalArgumentException(s"$m: ${filterString(strategy)}"))
82   }
83 
84   override def close(): Unit = {}
85 }

91   /**
92    *
93    * @param sizeLimit in square degrees
94    * @param durationLimit Maximum duration for a query at or below the spatial size
95    * @param samplePercent Percentage of total records to return
96    * @param sampleAttribute Attribute name to use a threading key for subsampling
97    */
98   class SizeAndLimits(val sizeLimit: Int, val durationLimit: Option[Duration],
99                       samplePercent: Option[Double], val sampleAttribute: Option[String]) {
100     require(samplePercent match {
101       case Some(p) => 0 < p && p <= 1
102       case None => true
103     }, "Graduated query guard percentages must be in range (0,1]")
104 
105     val percentageLimit: Option[Double] = samplePercent
106 
107     override def toString: String =
108       s"SizeAndLimits(sizeLimit=$sizeLimit, durationLimit=$durationLimit, " +
109           s"percentageLimit=$percentageLimit, sampleAttribute=$sampleAttribute)"
110   }
111 
112   def disabled(typeName: String): Boolean =
113     SystemProperty(s"geomesa.guard.graduated.$typeName.disable").toBoolean.contains(true)
114 
115   /**
116    * This function checks conditions on the limits.
117    * 1. Sizes must not be repeated.
118    * 2. Smaller sizes must allow for longer durations.
119    * 3. If a percentage is included it must decrease with larger sizes.
120    * @param limits Sequence of limits
121    * @return Sorted list of limits or throws an IllegalArgumentException
122    */
123   private def evaluateLimits(limits: Seq[SizeAndLimits], sft: SimpleFeatureType): Seq[SizeAndLimits] = {
124     val candidate = limits.sortBy(_.sizeLimit)
125     if (candidate.size > 1) {
126       // Once a duration or percentage is specified, all following configurations must specify it.
127       var hasDuration   = false
128       var hasPercentage = false
129       candidate.sliding(2).foreach { case Seq(first, second) =>
130         if (first.sizeLimit == second.sizeLimit) {
131           throw new IllegalArgumentException(s"Graduated query guard configuration " +
132             s"has repeated size: ${first.sizeLimit}")
133         }
134 
135         if (first.durationLimit.isDefined || hasDuration) {
136           hasDuration = true
137           if (second.durationLimit.isEmpty) {
138             throw new IllegalArgumentException(s"Graduated query guard configuration " +
139               s"has missing duration in size = ${second.sizeLimit}")
140           }
141           if (first.durationLimit.get.compareTo(second.durationLimit.get) <= 0) {
142             throw new IllegalArgumentException(s"Graduated query guard configuration " +
143               s"has durations out of order: ${first.durationLimit.get} is less than ${second.durationLimit.get}")
144           }
145         }
146 
147         if (first.percentageLimit.isDefined || hasPercentage) {
148           hasPercentage = true
149           if (second.percentageLimit.isEmpty) {
150             throw new IllegalArgumentException(s"Graduated query guard configuration " +
151               s"has missing percentage in size = ${second.sizeLimit}")
152           }
153           if (first.percentageLimit.get < second.percentageLimit.get) {
154             throw new IllegalArgumentException(s"Graduated query guard configuration " +
155               s"has percentages out of order or missing: " +
156               s"${first.percentageLimit.get} is less than ${second.durationLimit.get}")
157           }
158         }
159 
160         Seq(first.sampleAttribute, second.sampleAttribute).foreach{
161           case Some(attr) if sft.indexOf(attr) == -1 =>
162             throw new IllegalArgumentException(s"Graduated query guard configuration " +
163               s"has invalid attribute name for filter. ${first.sampleAttribute.get} not in sft ${sft.getTypeName}")
164           case _ =>
165         }
166       }
167     } else if (candidate.isEmpty) {
168       throw new IllegalArgumentException(s"Graduated query guard configuration is empty.")
169     }
170     if (candidate.last.sizeLimit != Int.MaxValue) {
171       throw new IllegalArgumentException(s"Graduated query guard configuration must include " +
172         "unbounded restriction.")
173     }
174     candidate
175   }
176 
177   def buildLimits(guardConfig: java.util.List[_ <: Config], sft: SimpleFeatureType): Seq[SizeAndLimits] = {
178     import scala.collection.JavaConverters._
179     val confs = guardConfig.asScala.map { limitsConfig =>
180       val size: Int = if (limitsConfig.hasPath("size")) {
181         limitsConfig.getInt("size")
182       } else {
183         Int.MaxValue
184       }
185 
186       val duration: Option[Duration] = if (limitsConfig.hasPath("duration")) {
187         Some(Duration(limitsConfig.getString("duration")))
188       } else None
189 
190       val percentage: Option[Double] = if (limitsConfig.hasPath("sampling-percentage")) {
191         Some(limitsConfig.getDouble("sampling-percentage"))
192       } else None
193 
194       val percentageAttr: Option[String] = if (limitsConfig.hasPath("sampling-attribute")) {
195         Some(limitsConfig.getString("sampling-attribute"))
196       } else None
197 
198       new SizeAndLimits(size, duration, percentage, percentageAttr)
199     }
200     evaluateLimits(confs.toSeq, sft)
201   }
