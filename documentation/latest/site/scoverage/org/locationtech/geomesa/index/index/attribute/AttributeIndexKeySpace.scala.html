150   override def getRanges(values: AttributeIndexValues[Any],
151                          multiplier: Int): Iterator[ScanRange[AttributeIndexKey]] = {
152 
153     import AttributeIndexKey.encodeForQuery
154     import org.locationtech.geomesa.filter.WildcardSuffix
155 
156     if (values.values.isEmpty) {
157       // we have an attribute, but weren't able to extract any bounds... scan all values
158       Iterator.single(UnboundedRange(AttributeIndexKey(fieldIndexShort, null, inclusive = false)))
159     } else if (values.values.disjoint && !isList) {
160       Iterator.empty
161     } else {
162       values.values.values.iterator.flatMap { bounds =>
163         bounds.bounds match {
164           case (None, None) => // not null
165             Iterator.single(UnboundedRange(AttributeIndexKey(fieldIndexShort, null, inclusive = false)))
166 
167           case (Some(lower), None) =>
168             val start = AttributeIndexKey(fieldIndexShort, encodeForQuery(lower, binding), bounds.lower.inclusive)
169             Iterator.single(LowerBoundedRange(start))
170 
171           case (None, Some(upper)) =>
172             val end = AttributeIndexKey(fieldIndexShort, encodeForQuery(upper, binding), bounds.upper.inclusive)
173             Iterator.single(UpperBoundedRange(end))
174 
175           case (Some(lower), Some(upper)) =>
176             if (lower == upper) {
177               val row = AttributeIndexKey(fieldIndexShort, encodeForQuery(lower, binding), inclusive = true)
178               Iterator.single(SingleRowRange(row))
179             } else if (lower + WildcardSuffix == upper) {
180               val row = AttributeIndexKey(fieldIndexShort, encodeForQuery(lower, binding), inclusive = true)
181               Iterator.single(PrefixRange(row))
182             } else {
183               val start = AttributeIndexKey(fieldIndexShort, encodeForQuery(lower, binding), bounds.lower.inclusive)
184               val end = AttributeIndexKey(fieldIndexShort, encodeForQuery(upper, binding), bounds.upper.inclusive)
185               Iterator.single(BoundedRange(start, end))
186             }
187         }
188       }
189     }

224   protected def getTieredRangeBytes(ranges: Iterator[ScanRange[AttributeIndexKey]],
225                                     prefixes: Seq[Array[Byte]]): Iterator[ByteRange] = {
226     import org.locationtech.geomesa.utils.index.ByteArrays.concat
227 
228     val bytes = ranges.map {
229       case SingleRowRange(row) =>
230         SingleRowByteRange(lower(row))
231 
232       case BoundedRange(lo, hi) =>
233         tieredUpper(hi) match {
234           case None     => LowerBoundedByteRange(lower(lo), upper(hi))
235           case Some(up) => BoundedByteRange(lower(lo), up)
236         }
237 
238       case PrefixRange(prefix) =>
239         UnboundedByteRange(lower(prefix, prefix = true), upper(prefix, prefix = true))
240 
241       case LowerBoundedRange(lo) =>
242         LowerBoundedByteRange(lower(lo), upper(AttributeIndexKey(lo.i, null)))
243 
244       case UpperBoundedRange(hi) =>
245         tieredUpper(hi) match {
246           case None     => UnboundedByteRange(lower(AttributeIndexKey(hi.i, null)), upper(hi))
247           case Some(up) => UpperBoundedByteRange(lower(AttributeIndexKey(hi.i, null)), up)
248         }
249 
250       case UnboundedRange(empty) =>
251         UnboundedByteRange(lower(empty), upper(empty))
252     }
253 
254     if (prefixes.isEmpty) { bytes } else {
255       bytes.flatMap {
256         case SingleRowByteRange(row)       => prefixes.map(p => SingleRowByteRange(concat(p, row)))
257         case BoundedByteRange(lo, hi)      => prefixes.map(p => BoundedByteRange(concat(p, lo), concat(p, hi)))
258         case LowerBoundedByteRange(lo, hi) => prefixes.map(p => LowerBoundedByteRange(concat(p, lo), concat(p, hi)))
259         case UpperBoundedByteRange(lo, hi) => prefixes.map(p => UpperBoundedByteRange(concat(p, lo), concat(p, hi)))
260         case UnboundedByteRange(lo, hi)    => prefixes.map(p => UnboundedByteRange(concat(p, lo), concat(p, hi)))
261       }
262     }
263   }

265   protected def getStandardRangeBytes(ranges: Iterator[ScanRange[AttributeIndexKey]],
266                                       prefixes: Seq[Array[Byte]]): Iterator[ByteRange] = {
267 
268     import org.locationtech.geomesa.utils.index.ByteArrays.concat
269 
270     val bytes = ranges.map {
271       case SingleRowRange(row)   => BoundedByteRange(lower(row), upper(row))
272       case BoundedRange(lo, hi)  => BoundedByteRange(lower(lo), upper(hi))
273       case PrefixRange(prefix)   => BoundedByteRange(lower(prefix, prefix = true), upper(prefix, prefix = true))
274       case LowerBoundedRange(lo) => BoundedByteRange(lower(lo), upper(AttributeIndexKey(lo.i, null)))
275       case UpperBoundedRange(hi) => BoundedByteRange(lower(AttributeIndexKey(hi.i, null)), upper(hi))
276       case UnboundedRange(empty) => BoundedByteRange(lower(empty), upper(empty))
277       case r => throw new IllegalArgumentException(s"Unexpected range type $r")
278     }
279 
280     if (prefixes.isEmpty) { bytes } else {
281       bytes.flatMap {
282         case BoundedByteRange(lo, hi) => prefixes.map(p => BoundedByteRange(concat(p, lo), concat(p, hi)))
283       }
284     }

287   /**
288     * Gets a lower range bound for an attribute value. The bound can be used with additional tiering or not
289     *
290     * @param key attribute value
291     * @param prefix if this is a prefix scan or not
292     * @return
293     */
294   protected def lower(key: AttributeIndexKey, prefix: Boolean = false): Array[Byte] = {
295     if (key.value == null || key.value.isEmpty) {
296       Array.empty
297     } else if (prefix) {
298       // note: inclusive doesn't make sense for prefix ranges
299       key.value.getBytes(StandardCharsets.UTF_8)
300     } else if (key.inclusive) {
301       ByteArrays.concat(key.value.getBytes(StandardCharsets.UTF_8), ZeroByteArray)
302     } else {
303       ByteArrays.concat(key.value.getBytes(StandardCharsets.UTF_8), OneByteArray)
304     }
305   }
306 
307   /**
308     * Gets an upper range bound for an attribute value. The bound is only suitable when there is no additional tiering
309     *
310     * @param key attribute value
311     * @param prefix if this is a prefix scan or not
312     * @return
313     */
314   protected def upper(key: AttributeIndexKey, prefix: Boolean = false): Array[Byte] = {
315     if (key.value == null || key.value.isEmpty) {
316       ByteRange.UnboundedUpperRange
317     } else if (prefix) {
318       // get the row following the prefix, then get the next row
319       // note: inclusiveness doesn't really make sense for prefix ranges
320       ByteArrays.rowFollowingPrefix(key.value.getBytes(StandardCharsets.UTF_8))
321     } else if (key.inclusive) {
322       // row following prefix, after the delimiter
323       ByteArrays.concat(key.value.getBytes(StandardCharsets.UTF_8), OneByteArray)
324     } else {
325       // exclude the row
326       ByteArrays.concat(key.value.getBytes(StandardCharsets.UTF_8), ZeroByteArray)
327     }
328   }

330   /**
331     * Gets an upper bound for a range that will be tiered. A bound will only be returned if it
332     * supports additional tiering.
333     *
334     * @param key attribute value
335     * @return
336     */
337   protected def tieredUpper(key: AttributeIndexKey): Option[Array[Byte]] = {
338     // note: we can't tier exclusive end points, as we can't calculate previous rows
339     if (key.value == null || !key.inclusive) { None } else {
340       // match the final row, and count on remaining range to exclude the rest
341       Some(ByteArrays.concat(key.value.getBytes(StandardCharsets.UTF_8), ZeroByteArray))
342     }
343   }
