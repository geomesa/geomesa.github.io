150                          multiplier: Int): Iterator[ScanRange[AttributeIndexKey]] = {

153     import org.locationtech.geomesa.filter.WildcardSuffix

155     if (values.values.isEmpty) {

157       Iterator.single(UnboundedRange(AttributeIndexKey(fieldIndexShort, null, inclusive = false)))
158     } else if (values.values.disjoint && !isList) {
159       Iterator.empty

162         bounds.bounds match {
163           case (None, None) => // not null

174           case (Some(lower), Some(upper)) =>

221   protected def getTieredRangeBytes(ranges: Iterator[ScanRange[AttributeIndexKey]],
222                                     prefixes: Seq[Array[Byte]]): Iterator[ByteRange] = {
223     import org.locationtech.geomesa.utils.index.ByteArrays.concat
224 
225     val bytes = ranges.map {
226       case SingleRowRange(row) =>
227         SingleRowByteRange(lower(row))
228 
229       case BoundedRange(lo, hi) =>
230         tieredUpper(hi) match {
231           case None     => LowerBoundedByteRange(lower(lo), upper(hi))
232           case Some(up) => BoundedByteRange(lower(lo), up)
233         }
234 
235       case PrefixRange(prefix) =>
236         UnboundedByteRange(lower(prefix, prefix = true), upper(prefix, prefix = true))
237 
238       case LowerBoundedRange(lo) =>
239         LowerBoundedByteRange(lower(lo), upper(AttributeIndexKey(lo.i, null)))
240 
241       case UpperBoundedRange(hi) =>
242         tieredUpper(hi) match {
243           case None     => UnboundedByteRange(lower(AttributeIndexKey(hi.i, null)), upper(hi))
244           case Some(up) => UpperBoundedByteRange(lower(AttributeIndexKey(hi.i, null)), up)
245         }
246 
247       case UnboundedRange(empty) =>
248         UnboundedByteRange(lower(empty), upper(empty))
249     }
250 
251     if (prefixes.isEmpty) { bytes } else {
252       bytes.flatMap {
253         case SingleRowByteRange(row)       => prefixes.map(p => SingleRowByteRange(concat(p, row)))
254         case BoundedByteRange(lo, hi)      => prefixes.map(p => BoundedByteRange(concat(p, lo), concat(p, hi)))
255         case LowerBoundedByteRange(lo, hi) => prefixes.map(p => LowerBoundedByteRange(concat(p, lo), concat(p, hi)))
256         case UpperBoundedByteRange(lo, hi) => prefixes.map(p => UpperBoundedByteRange(concat(p, lo), concat(p, hi)))
257         case UnboundedByteRange(lo, hi)    => prefixes.map(p => UnboundedByteRange(concat(p, lo), concat(p, hi)))
258       }
259     }
260   }

263                                       prefixes: Seq[Array[Byte]]): Iterator[ByteRange] = {
264 
265     import org.locationtech.geomesa.utils.index.ByteArrays.concat

267     val bytes = ranges.map {
268       case SingleRowRange(row)   => BoundedByteRange(lower(row), upper(row))
269       case BoundedRange(lo, hi)  => BoundedByteRange(lower(lo), upper(hi))
270       case PrefixRange(prefix)   => BoundedByteRange(lower(prefix, prefix = true), upper(prefix, prefix = true))
271       case LowerBoundedRange(lo) => BoundedByteRange(lower(lo), upper(AttributeIndexKey(lo.i, null)))

274       case r => throw new IllegalArgumentException(s"Unexpected range type $r")
275     }
276 
277     if (prefixes.isEmpty) { bytes } else {
278       bytes.flatMap {
279         case BoundedByteRange(lo, hi) => prefixes.map(p => BoundedByteRange(concat(p, lo), concat(p, hi)))
280       }
281     }

284   /**
285     * Gets a lower range bound for an attribute value. The bound can be used with additional tiering or not
286     *
287     * @param key attribute value
288     * @param prefix if this is a prefix scan or not
289     * @return
290     */
291   protected def lower(key: AttributeIndexKey, prefix: Boolean = false): Array[Byte] = {
292     if (key.value == null || key.value.isEmpty) {
293       Array.empty
294     } else if (prefix) {
295       // note: inclusive doesn't make sense for prefix ranges
296       key.value.getBytes(StandardCharsets.UTF_8)
297     } else if (key.inclusive) {
298       ByteArrays.concat(key.value.getBytes(StandardCharsets.UTF_8), ZeroByteArray)
299     } else {
300       ByteArrays.concat(key.value.getBytes(StandardCharsets.UTF_8), OneByteArray)
301     }
302   }
303 
304   /**
305     * Gets an upper range bound for an attribute value. The bound is only suitable when there is no additional tiering
306     *
307     * @param key attribute value
308     * @param prefix if this is a prefix scan or not
309     * @return
310     */
311   protected def upper(key: AttributeIndexKey, prefix: Boolean = false): Array[Byte] = {
312     if (key.value == null || key.value.isEmpty) {
313       ByteRange.UnboundedUpperRange
314     } else if (prefix) {
315       // get the row following the prefix, then get the next row
316       // note: inclusiveness doesn't really make sense for prefix ranges
317       ByteArrays.rowFollowingPrefix(key.value.getBytes(StandardCharsets.UTF_8))
318     } else if (key.inclusive) {
319       // row following prefix, after the delimiter
320       ByteArrays.concat(key.value.getBytes(StandardCharsets.UTF_8), OneByteArray)
321     } else {
322       // exclude the row
323       ByteArrays.concat(key.value.getBytes(StandardCharsets.UTF_8), ZeroByteArray)
324     }
325   }

327   /**
328     * Gets an upper bound for a range that will be tiered. A bound will only be returned if it
329     * supports additional tiering.
330     *
331     * @param key attribute value
332     * @return
333     */
334   protected def tieredUpper(key: AttributeIndexKey): Option[Array[Byte]] = {
335     // note: we can't tier exclusive end points, as we can't calculate previous rows
336     if (key.value == null || !key.inclusive) { None } else {
337       // match the final row, and count on remaining range to exclude the rest
338       Some(ByteArrays.concat(key.value.getBytes(StandardCharsets.UTF_8), ZeroByteArray))
339     }
340   }
