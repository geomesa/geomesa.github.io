39   private val exceptions: ArrayBuffer[Throwable] = ArrayBuffer.empty[Throwable]

41   protected val statUpdater: StatUpdater = ds.stats.writer.updater(sft)
42 
43   override def getFeatureType: SimpleFeatureType = sft
44 
45   protected def getWriter(feature: SimpleFeature): IndexWriter
46 
47   protected def updateFeature(update: FastSettableFeature, previous: SimpleFeature): Unit = {
48     // see if there's a suggested ID to use for this feature, else create one based on the feature
49     val writable = GeoMesaFeatureWriter.featureWithFid(update)
50     try {
51       val writer = getWriter(writable)
52       val remover = getWriter(previous)
53       if (writer.eq(remover)) {
54         // `update` will calculate all mutations up front in case the feature is not valid, so we don't write partial entries
55         writer.update(writable, previous)
56       } else {
57         remover.delete(previous)
58         writer.append(writable)
59       }
60     } catch {
61       case NonFatal(e) => throwWriteErrors(e, writable)
62     }
63     statUpdater.add(writable)
64   }
65 
66   protected def appendFeature(feature: FastSettableFeature): Unit = {
67     // see if there's a suggested ID to use for this feature, else create one based on the feature
68     val writable = GeoMesaFeatureWriter.featureWithFid(feature)
69     // `append` will calculate all mutations up front in case the feature is not valid, so we don't write partial entries
70     try { getWriter(writable).append(writable) } catch {
71       case NonFatal(e) => throwWriteErrors(e, writable)
72     }
73     statUpdater.add(writable)
74   }
75 
76   protected def removeFeature(feature: SimpleFeature): Unit = {
77     // the feature has come directly from our reader, so it should be valid and already have a FID
78     getWriter(feature).delete(feature)
79     statUpdater.remove(feature)
80   }
81 
82   protected def suppressException(e: Throwable): Unit = exceptions += e
83 
84   protected def propagateExceptions(): Unit = {
85     if (exceptions.nonEmpty) {
86       val all = new RuntimeException(s"Error writing features:")
87       exceptions.foreach(all.addSuppressed)
88       exceptions.clear()
89       throw all
90     }
91   }
92 
93   // returns a temporary id - we will replace it just before write
94   protected def nextFeatureId: String = GeoMesaFeatureWriter.tempFeatureIds.getAndIncrement().toString

96   @throws[Exception]
97   private def throwWriteErrors(e: Throwable, feature: SimpleFeature): Unit = {
98     val msg = s"Error indexing feature '${feature.getID}:${DataUtilities.encodeFeature(feature, false)}'"
99     e match {
100       case _: WriteException => throw e
101       case _: IllegalArgumentException => throw new IllegalArgumentException(msg, e)
102       case _ => throw new RuntimeException(msg, e)
103     }
104   }

159   /**
160    * Sets the feature ID on the feature. If the user has requested a specific ID, that will be used,
161    * otherwise one will be generated. If possible, the original feature will be modified and returned.
162    */
163   def featureWithFid(feature: SimpleFeature): SimpleFeature = {
164     val provided = feature.getUserData.get(Hints.PROVIDED_FID)
165     if (provided != null) {
166       withFid(feature, provided.toString)
167     } else if (feature.getUserData.get(Hints.USE_PROVIDED_FID) != java.lang.Boolean.TRUE) {
168       withFid(feature, idGenerator.createId(feature.getFeatureType, feature))
169     } else {
170       feature
171     }
172   }

174   /**
175    * Sets the feature ID on the feature. If the user has requested a specific ID, that will be used,
176    * otherwise one will be generated. The original feature will be modified and returned.
177    *
178    * @param feature feature
179    * @return
180    */
181   def featureWithFid(feature: FastSettableFeature): FastSettableFeature = {
182     val provided = feature.getUserData.get(Hints.PROVIDED_FID)
183     if (provided != null) {
184       feature.setId(provided.toString)
185     } else if (feature.getUserData.get(Hints.USE_PROVIDED_FID) != java.lang.Boolean.TRUE) {
186       feature.setId(idGenerator.createId(feature.getFeatureType, feature))
187     }
188     feature
189   }

191   private def withFid(feature: SimpleFeature, fid: String): SimpleFeature = {
192     feature match {
193       case f: ScalaSimpleFeature => f.setId(fid); f
194       case _ =>
195         feature.getIdentifier match {
196           case f: FeatureIdImpl => f.setID(fid); feature
197           case f =>
198             logger.warn(s"Unknown FeatureID implementation found, rebuilding feature: $f '${f.getClass.getName}'")
199             val copy = ScalaSimpleFeature.copy(feature)
200             copy.setId(fid)
201             copy
202         }
203     }
204   }

243   /**
244     * Support for writing to partitioned tables
245     *
246     */
247   private abstract class PartitionFeatureWriter[DS <: GeoMesaDataStore[DS]](
248       val ds: DS,
249       val sft: SimpleFeatureType,
250       val indices: Seq[GeoMesaFeatureIndex[_, _]],
251       val atomic: Boolean
252     ) extends GeoMesaFeatureWriter[DS] {
253 
254     import scala.collection.JavaConverters._
255 
256     private val partition = TablePartition(ds, sft).getOrElse {
257       throw new IllegalStateException("Creating a partitioned writer for a non-partitioned schema")
258     }
259 
260     private val cache = new java.util.HashMap[String, IndexWriter]()
261     private val view = cache.asScala
262 
263     override protected def getWriter(feature: SimpleFeature): IndexWriter = {
264       val p = partition.partition(feature)
265       var writer = cache.get(p)
266       if (writer == null) {
267         // reconfigure the partition each time - this should be idempotent, and block
268         // until it is fully created (which may happen in some other thread)
269         def createOne(index: GeoMesaFeatureIndex[_, _]): Unit =
270           ds.adapter.createTable(index, Some(p), index.getSplits(Some(p)))
271         indices.toList.map(i => CachedThreadPool.submit(() => createOne(i))).foreach(_.get)
272         writer = ds.adapter.createWriter(sft, indices, Some(p), atomic)
273         cache.put(p, writer)
274       }
275       writer
276     }
277 
278     override def flush(): Unit = {
279       view.foreach { case (_, writer) => FlushQuietly(writer).foreach(suppressException) }
280       FlushQuietly(statUpdater).foreach(suppressException)
281       propagateExceptions()
282     }
283 
284     override def close(): Unit = {
285       view.foreach { case (_, writer) => CloseQuietly(writer).foreach(suppressException) }
286       CloseQuietly(statUpdater).foreach(suppressException)
287       propagateExceptions()
288     }
289   }

294   private trait GeoMesaAppendFeatureWriter[DS <: GeoMesaDataStore[DS]] extends GeoMesaFeatureWriter[DS] {
295 
296     private var currentFeature: ScalaSimpleFeature = _
297 
298     override def hasNext: Boolean = false // per geotools spec, always return false

306       if (currentFeature == null) {

317   /**
318     * Modifies or deletes existing features. Per the data store api, does not allow appending new features.
319     */
320   private trait GeoMesaModifyFeatureWriter[DS <: GeoMesaDataStore[DS]] extends GeoMesaFeatureWriter[DS] {
321 
322     import org.locationtech.geomesa.security.SecureSimpleFeature
323 
324     def filter: Filter
325 
326     private val reader = ds.getFeatureReader(new Query(sft.getTypeName, filter), Transaction.AUTO_COMMIT)
327 
328     // feature returned from reader
329     private var original: SimpleFeature = _
330 
331     // feature that caller will modify
332     private var live: FastSettableFeature = _
333 
334     override def hasNext: Boolean = reader.hasNext
335 
336     override def next(): FastSettableFeature = {
337       original = reader.next()
338       // set the use provided FID hint - allows user to update fid if desired,
339       // but if not we'll use the existing one
340       original.getUserData.put(Hints.USE_PROVIDED_FID, java.lang.Boolean.TRUE)
341       live = ScalaSimpleFeature.copy(sft, original) // this copies user data as well
342       live
343     }
344 
345     override def write(): Unit = {
346       if (original == null) {
347         throw new IllegalStateException("next() must be called before write()")
348       }
349       // update the feature id based on hints before we compare for changes
350       live = GeoMesaFeatureWriter.featureWithFid(live)
351       // only write if feature has actually changed...
352       // comparison of feature ID and attributes - doesn't consider concrete class used
353       if (!ScalaSimpleFeature.equalIdAndAttributes(live, original) || live.visibility != original.visibility) {
354         updateFeature(live, original)
355       }
356       original = null
357       live = null
358     }
359 
360     override def remove(): Unit = {
361       if (original == null) {
362         throw new IllegalStateException("next() must be called before remove()")
363       }
364       removeFeature(original)
365       original = null
366       live = null
367     }
368 
369     abstract override def close(): Unit = {
370       CloseQuietly(reader).foreach(suppressException)
371       super.close() // closes writer
372     }
