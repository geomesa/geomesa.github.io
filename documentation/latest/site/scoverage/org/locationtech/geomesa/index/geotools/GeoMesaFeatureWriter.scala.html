80   protected def updateFeature(update: FastSettableFeature, previous: SimpleFeature): Unit = {
81     val start = System.nanoTime()
82     try {
83       // see if there's a suggested ID to use for this feature, else create one based on the feature
84       val writable = GeoMesaFeatureWriter.featureWithFid(update)
85       try {
86         val writer = getWriter(writable)
87         val remover = getWriter(previous)
88         if (writer.eq(remover)) {
89           // `update` will calculate all mutations up front in case the feature is not valid, so we don't write partial entries
90           writer.update(writable, previous)
91         } else {
92           remover.delete(previous)
93           writer.append(writable)
94         }
95       } catch {
96         case NonFatal(e) => throwWriteErrors(e, writable)
97       }
98       statUpdater.add(writable)
99     } finally {
100       updateTimer.record(System.nanoTime() - start, TimeUnit.NANOSECONDS)
101     }
102   }

104   protected def appendFeature(feature: FastSettableFeature): Unit = {
105     val start = System.nanoTime()
106     try {
107       // see if there's a suggested ID to use for this feature, else create one based on the feature
108       val writable = GeoMesaFeatureWriter.featureWithFid(feature)
109       try {
110         // `append` will calculate all mutations up front in case the feature is not valid, so we don't write partial entries
111         getWriter(writable).append(writable)
112       } catch {
113         case NonFatal(e) => throwWriteErrors(e, writable)
114       }
115       statUpdater.add(writable)
116     } finally {
117       appendTimer.record(System.nanoTime() - start, TimeUnit.NANOSECONDS)
118     }
119   }
120 
121   protected def removeFeature(feature: SimpleFeature): Unit = {
122     val start = System.nanoTime()
123     try {
124       // the feature has come directly from our reader, so it should be valid and already have a FID
125       getWriter(feature).delete(feature)
126       statUpdater.remove(feature)
127     } finally {
128       deleteTimer.record(System.nanoTime() - start, TimeUnit.NANOSECONDS)
129     }
130   }

205    * Sets the feature ID on the feature. If the user has requested a specific ID, that will be used,
206    * otherwise one will be generated. If possible, the original feature will be modified and returned.

208   def featureWithFid(feature: SimpleFeature): SimpleFeature = {
209     val provided = feature.getUserData.get(Hints.PROVIDED_FID)
210     if (provided != null) {
211       withFid(feature, provided.toString)
212     } else if (feature.getUserData.get(Hints.USE_PROVIDED_FID) != java.lang.Boolean.TRUE) {
213       withFid(feature, idGenerator.createId(feature.getFeatureType, feature))
214     } else {
215       feature
216     }
217   }

236   private def withFid(feature: SimpleFeature, fid: String): SimpleFeature = {
237     feature match {
238       case f: ScalaSimpleFeature => f.setId(fid); f
239       case _ =>
240         feature.getIdentifier match {
241           case f: FeatureIdImpl => f.setID(fid); feature
242           case f =>
243             logger.warn(s"Unknown FeatureID implementation found, rebuilding feature: $f '${f.getClass.getName}'")
244             val copy = ScalaSimpleFeature.copy(feature)
245             copy.setId(fid)
246             copy
247         }
248     }
249   }

295   /**
296     * Support for writing to partitioned tables
297     *
298     */
299   private abstract class PartitionFeatureWriter[DS <: GeoMesaDataStore[DS]](
300       val ds: DS,
301       val sft: SimpleFeatureType,
302       val indices: Seq[GeoMesaFeatureIndex[_, _]],
303       val atomic: Boolean
304     ) extends GeoMesaFeatureWriter[DS] {
305 
306     import scala.collection.JavaConverters._
307 
308     private val partition = TablePartition(ds, sft).getOrElse {
309       throw new IllegalStateException("Creating a partitioned writer for a non-partitioned schema")
310     }
311 
312     private val cache = new java.util.HashMap[String, IndexWriter]()
313     private val view = cache.asScala
314 
315     override protected def getWriter(feature: SimpleFeature): IndexWriter = {
316       val p = partition.partition(feature)
317       var writer = cache.get(p)
318       if (writer == null) {
319         // reconfigure the partition each time - this should be idempotent, and block
320         // until it is fully created (which may happen in some other thread)
321         def createOne(index: GeoMesaFeatureIndex[_, _]): Unit =
322           ds.adapter.createTable(index, Some(p), index.getSplits(Some(p)))
323         indices.toList.map(i => CachedThreadPool.submit(() => createOne(i))).foreach(_.get)
324         writer = ds.adapter.createWriter(sft, indices, Some(p), atomic)
325         cache.put(p, writer)
326       }
327       writer
328     }
329 
330     override protected def flushables(): Seq[Flushable] = view.values.toSeq :+ statUpdater
331     override protected def closeables(): Seq[Closeable] = view.values.toSeq :+ statUpdater
332   }

334   /**
335     * Appends new features - can't modify or delete existing features
336     */
337   private trait GeoMesaAppendFeatureWriter[DS <: GeoMesaDataStore[DS]] extends GeoMesaFeatureWriter[DS] {
338 
339     private var currentFeature: ScalaSimpleFeature = _
340 
341     override def hasNext: Boolean = false // per geotools spec, always return false
342 
343     override def next(): FastSettableFeature = {
344       currentFeature = new ScalaSimpleFeature(sft, nextFeatureId)
345       currentFeature
346     }
347 
348     override def write(): Unit = {
349       if (currentFeature == null) {
350         throw new IllegalStateException("next() must be called before write()")
351       }

354     }

360   /**
361     * Modifies or deletes existing features. Per the data store api, does not allow appending new features.
362     */
363   private trait GeoMesaModifyFeatureWriter[DS <: GeoMesaDataStore[DS]] extends GeoMesaFeatureWriter[DS] {
364 
365     import org.locationtech.geomesa.security.SecureSimpleFeature
366 
367     def filter: Filter
368 
369     private val reader = ds.getFeatureReader(new Query(sft.getTypeName, filter), Transaction.AUTO_COMMIT)
370 
371     // feature returned from reader
372     private var original: SimpleFeature = _
373 
374     // feature that caller will modify
375     private var live: FastSettableFeature = _
376 
377     override def hasNext: Boolean = reader.hasNext
378 
379     override def next(): FastSettableFeature = {
380       original = reader.next()
381       // set the use provided FID hint - allows user to update fid if desired,
382       // but if not we'll use the existing one
383       original.getUserData.put(Hints.USE_PROVIDED_FID, java.lang.Boolean.TRUE)
384       live = ScalaSimpleFeature.copy(sft, original) // this copies user data as well
385       live
386     }
387 
388     override def write(): Unit = {
389       if (original == null) {
390         throw new IllegalStateException("next() must be called before write()")
391       }
392       // update the feature id based on hints before we compare for changes
393       live = GeoMesaFeatureWriter.featureWithFid(live)
394       // only write if feature has actually changed...
395       // comparison of feature ID and attributes - doesn't consider concrete class used
396       if (!ScalaSimpleFeature.equalIdAndAttributes(live, original) || live.visibility != original.visibility) {
397         updateFeature(live, original)
398       }
399       original = null
400       live = null
401     }
402 
403     override def remove(): Unit = {
404       if (original == null) {
405         throw new IllegalStateException("next() must be called before remove()")
406       }
407       removeFeature(original)
408       original = null
409       live = null
410     }
411 
412     abstract override protected def closeables(): Seq[Closeable] = super.closeables() :+ reader
413   }
