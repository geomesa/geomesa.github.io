81   protected def updateFeature(update: FastSettableFeature, previous: SimpleFeature): Unit = {
82     val start = System.nanoTime()
83     try {
84       // see if there's a suggested ID to use for this feature, else create one based on the feature
85       val writable = GeoMesaFeatureWriter.featureWithFid(update)
86       try {
87         val writer = getWriter(writable)
88         val remover = getWriter(previous)
89         if (writer.eq(remover)) {
90           // `update` will calculate all mutations up front in case the feature is not valid, so we don't write partial entries
91           writer.update(writable, previous)
92         } else {
93           remover.delete(previous)
94           writer.append(writable)
95         }
96       } catch {
97         case NonFatal(e) => throwWriteErrors(e, writable)
98       }
99       statUpdater.add(writable)
100     } finally {
101       updateTimer.record(System.nanoTime() - start, TimeUnit.NANOSECONDS)
102     }
103   }

105   protected def appendFeature(feature: FastSettableFeature): Unit = {
106     val start = System.nanoTime()
107     try {
108       // see if there's a suggested ID to use for this feature, else create one based on the feature
109       val writable = GeoMesaFeatureWriter.featureWithFid(feature)
110       try {
111         // `append` will calculate all mutations up front in case the feature is not valid, so we don't write partial entries
112         getWriter(writable).append(writable)
113       } catch {
114         case NonFatal(e) => throwWriteErrors(e, writable)
115       }
116       statUpdater.add(writable)
117     } finally {
118       appendTimer.record(System.nanoTime() - start, TimeUnit.NANOSECONDS)
119     }
120   }
121 
122   protected def removeFeature(feature: SimpleFeature): Unit = {
123     val start = System.nanoTime()
124     try {
125       // the feature has come directly from our reader, so it should be valid and already have a FID
126       getWriter(feature).delete(feature)
127       statUpdater.remove(feature)
128     } finally {
129       deleteTimer.record(System.nanoTime() - start, TimeUnit.NANOSECONDS)
130     }
131   }

171    *
172    * @param ds datastore
173    * @param sft simple feature type
174    * @param indices indices to write
175    * @param filter filter for selecting features for updating writes, or None for appending writes
176    * @param atomic enforce atomic writes

180   def apply[DS <: GeoMesaDataStore[DS]](
181       ds: DS,
182       sft: SimpleFeatureType,
183       indices: Seq[GeoMesaFeatureIndex[_, _]],
184       filter: Option[Filter],
185       atomic: Boolean): GeoMesaFeatureWriter[DS] = {
186     if (TablePartition.partitioned(sft)) {
187       filter match {
188         case None => new PartitionFeatureWriter(ds, sft, indices, atomic) with GeoMesaAppendFeatureWriter[DS]
189         case Some(f) =>
190           new PartitionFeatureWriter(ds, sft, indices, atomic) with GeoMesaModifyFeatureWriter[DS] {
191             override def filter: Filter = f
192           }
193       }
194     } else {
195       filter match {
196         case None => new TableFeatureWriter(ds, sft, indices, atomic) with GeoMesaAppendFeatureWriter[DS]
197         case Some(f) =>
198           new TableFeatureWriter(ds, sft, indices, atomic) with GeoMesaModifyFeatureWriter[DS] {
199             override def filter: Filter = f

201       }

205   /**
206    * Sets the feature ID on the feature. If the user has requested a specific ID, that will be used,
207    * otherwise one will be generated. If possible, the original feature will be modified and returned.
208    */
209   def featureWithFid(feature: SimpleFeature): SimpleFeature = {
210     val provided = feature.getUserData.get(Hints.PROVIDED_FID)
211     if (provided != null) {
212       withFid(feature, provided.toString)
213     } else if (feature.getUserData.get(Hints.USE_PROVIDED_FID) != java.lang.Boolean.TRUE) {
214       withFid(feature, idGenerator.createId(feature.getFeatureType, feature))
215     } else {
216       feature
217     }
218   }

237   private def withFid(feature: SimpleFeature, fid: String): SimpleFeature = {
238     feature match {
239       case f: ScalaSimpleFeature => f.setId(fid); f
240       case _ =>
241         feature.getIdentifier match {
242           case f: FeatureIdImpl => f.setID(fid); feature
243           case f =>
244             logger.warn(s"Unknown FeatureID implementation found, rebuilding feature: $f '${f.getClass.getName}'")
245             val copy = ScalaSimpleFeature.copy(feature)
246             copy.setId(fid)
247             copy
248         }
249     }

296   /**
297     * Support for writing to partitioned tables
298     *
299     */
300   private abstract class PartitionFeatureWriter[DS <: GeoMesaDataStore[DS]](
301       val ds: DS,
302       val sft: SimpleFeatureType,
303       val indices: Seq[GeoMesaFeatureIndex[_, _]],
304       val atomic: Boolean
305     ) extends GeoMesaFeatureWriter[DS] {
306 
307     import scala.collection.JavaConverters._
308 
309     private val partition = TablePartition(ds, sft).getOrElse {
310       throw new IllegalStateException("Creating a partitioned writer for a non-partitioned schema")
311     }
312 
313     private val cache = new java.util.HashMap[String, IndexWriter]()
314     private val view = cache.asScala
315 
316     override protected def getWriter(feature: SimpleFeature): IndexWriter = {
317       val p = partition.partition(feature)
318       var writer = cache.get(p)
319       if (writer == null) {
320         // reconfigure the partition each time - this should be idempotent, and block
321         // until it is fully created (which may happen in some other thread)
322         def createOne(index: GeoMesaFeatureIndex[_, _]): Unit =
323           ds.adapter.createTable(index, Some(p), index.getSplits(Some(p)))
324         indices.toList.map(i => CachedThreadPool.submit(() => createOne(i))).foreach(_.get)
325         writer = ds.adapter.createWriter(sft, indices, Some(p), atomic)
326         cache.put(p, writer)
327       }
328       writer
329     }
330 
331     override protected def flushables(): Seq[Flushable] = view.values.toSeq :+ statUpdater
332     override protected def closeables(): Seq[Closeable] = view.values.toSeq :+ statUpdater
333   }

336     * Appends new features - can't modify or delete existing features

338   private trait GeoMesaAppendFeatureWriter[DS <: GeoMesaDataStore[DS]] extends GeoMesaFeatureWriter[DS] {
339 
340     private var currentFeature: ScalaSimpleFeature = _
341 
342     override def hasNext: Boolean = false // per geotools spec, always return false
343 
344     override def next(): FastSettableFeature = {
345       currentFeature = new ScalaSimpleFeature(sft, nextFeatureId)
346       currentFeature
347     }
348 
349     override def write(): Unit = {
350       if (currentFeature == null) {
351         throw new IllegalStateException("next() must be called before write()")
352       }
353       appendFeature(currentFeature)
354       currentFeature = null
355     }

357     override def remove(): Unit =
358       throw new UnsupportedOperationException("Use getFeatureWriter instead of getFeatureWriterAppend")
359   }

361   /**
362     * Modifies or deletes existing features. Per the data store api, does not allow appending new features.
363     */
364   private trait GeoMesaModifyFeatureWriter[DS <: GeoMesaDataStore[DS]] extends GeoMesaFeatureWriter[DS] {
365 
366     def filter: Filter
367 
368     private val reader = ds.getFeatureReader(new Query(sft.getTypeName, filter), Transaction.AUTO_COMMIT)
369 
370     // feature returned from reader
371     private var original: SimpleFeature = _
372 
373     // feature that caller will modify
374     private var live: FastSettableFeature = _
375 
376     override def hasNext: Boolean = reader.hasNext
377 
378     override def next(): FastSettableFeature = {
379       original = reader.next()
380       // set the use provided FID hint - allows user to update fid if desired,
381       // but if not we'll use the existing one
382       original.getUserData.put(Hints.USE_PROVIDED_FID, java.lang.Boolean.TRUE)
383       live = ScalaSimpleFeature.copy(sft, original) // this copies user data as well
384       live
385     }
386 
387     override def write(): Unit = {
388       if (original == null) {
389         throw new IllegalStateException("next() must be called before write()")
390       }
391       // update the feature id based on hints before we compare for changes
392       live = GeoMesaFeatureWriter.featureWithFid(live)
393       // only write if feature has actually changed...
394       // comparison of feature ID and attributes - doesn't consider concrete class used
395       if (!ScalaSimpleFeature.equalIdAndAttributes(live, original) ||
396             SecurityUtils.getVisibility(live) != SecurityUtils.getVisibility(original)) {
397         updateFeature(live, original)
398       }
399       original = null
400       live = null
401     }
402 
403     override def remove(): Unit = {
404       if (original == null) {
405         throw new IllegalStateException("next() must be called before remove()")
406       }
407       removeFeature(original)
408       original = null
409       live = null
410     }
411 
412     abstract override protected def closeables(): Seq[Closeable] = super.closeables() :+ reader
413   }
