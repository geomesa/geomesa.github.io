41   protected val statUpdater: StatUpdater = ds.stats.writer.updater(sft)

43   override def getFeatureType: SimpleFeatureType = sft

47   protected def updateFeature(update: SimpleFeature, previous: SimpleFeature): Unit = {
48     // see if there's a suggested ID to use for this feature, else create one based on the feature
49     val writable = GeoMesaFeatureWriter.featureWithFid(update)
50     try {
51       val writer = getWriter(writable)
52       val remover = getWriter(previous)
53       if (writer.eq(remover)) {
54         // `update` will calculate all mutations up front in case the feature is not valid, so we don't write partial entries
55         writer.update(writable, previous)
56       } else {
57         remover.delete(previous)
58         writer.append(writable)
59       }
60     } catch {
61       case NonFatal(e) => throwWriteErrors(e, writable)
62     }
63     statUpdater.add(writable)
64   }
65 
66   protected def appendFeature(feature: SimpleFeature): Unit = {
67     // see if there's a suggested ID to use for this feature, else create one based on the feature
68     val writable = GeoMesaFeatureWriter.featureWithFid(feature)
69     // `append` will calculate all mutations up front in case the feature is not valid, so we don't write partial entries
70     try { getWriter(writable).append(writable) } catch {
71       case NonFatal(e) => throwWriteErrors(e, writable)
72     }
73     statUpdater.add(writable)
74   }
75 
76   protected def removeFeature(feature: SimpleFeature): Unit = {
77     // the feature has come directly from our reader, so it should be valid and already have a FID
78     getWriter(feature).delete(feature)
79     statUpdater.remove(feature)
80   }
81 
82   protected def suppressException(e: Throwable): Unit = exceptions += e
83 
84   protected def propagateExceptions(): Unit = {
85     if (exceptions.nonEmpty) {
86       val all = new RuntimeException(s"Error writing features:")
87       exceptions.foreach(all.addSuppressed)
88       exceptions.clear()
89       throw all
90     }
91   }

101       case _: IllegalArgumentException => throw new IllegalArgumentException(msg, e)
102       case _ => throw new RuntimeException(msg, e)
103     }
104   }

124    * Create a feature writer
125    *
126    * @param ds datastore
127    * @param sft simple feature type
128    * @param indices indices to write
129    * @param filter filter for selecting features for updating writes, or None for appending writes
130    * @param atomic enforce atomic writes

134   def apply[DS <: GeoMesaDataStore[DS]](
135       ds: DS,
136       sft: SimpleFeatureType,
137       indices: Seq[GeoMesaFeatureIndex[_, _]],
138       filter: Option[Filter],
139       atomic: Boolean): GeoMesaFeatureWriter[DS] = {
140     if (TablePartition.partitioned(sft)) {
141       filter match {
142         case None => new PartitionFeatureWriter(ds, sft, indices, atomic) with GeoMesaAppendFeatureWriter[DS]
143         case Some(f) =>
144           new PartitionFeatureWriter(ds, sft, indices, atomic) with GeoMesaModifyFeatureWriter[DS] {
145             override def filter: Filter = f
146           }
147       }
148     } else {
149       filter match {
150         case None => new TableFeatureWriter(ds, sft, indices, atomic) with GeoMesaAppendFeatureWriter[DS]
151         case Some(f) =>
152           new TableFeatureWriter(ds, sft, indices, atomic) with GeoMesaModifyFeatureWriter[DS] {
153             override def filter: Filter = f
154           }
155       }
156     }
157   }

159   /**
160    * Sets the feature ID on the feature. If the user has requested a specific ID, that will be used,
161    * otherwise one will be generated. If possible, the original feature will be modified and returned.
162    */
163   def featureWithFid(feature: SimpleFeature): SimpleFeature = {
164     if (feature.getUserData.containsKey(Hints.PROVIDED_FID)) {
165       withFid(feature, feature.getUserData.get(Hints.PROVIDED_FID).toString)
166     } else if (feature.getUserData.containsKey(Hints.USE_PROVIDED_FID) &&
167         feature.getUserData.get(Hints.USE_PROVIDED_FID).asInstanceOf[Boolean]) {
168       feature
169     } else {
170       withFid(feature, idGenerator.createId(feature.getFeatureType, feature))
171     }
172   }

184   private def withFid(feature: SimpleFeature, fid: String): SimpleFeature = {
185     feature match {
186       case f: ScalaSimpleFeature => f.setId(fid); f
187       case _ =>
188         feature.getIdentifier match {
189           case f: FeatureIdImpl => f.setID(fid); feature
190           case f =>
191             logger.warn(s"Unknown FeatureID implementation found, rebuilding feature: $f '${f.getClass.getName}'")
192             val copy = ScalaSimpleFeature.copy(feature)
193             copy.setId(fid)
194             copy
195         }
196     }

227   /**
228     * Support for writing to partitioned tables
229     *
230     */
231   private abstract class PartitionFeatureWriter[DS <: GeoMesaDataStore[DS]](
232       val ds: DS,
233       val sft: SimpleFeatureType,
234       val indices: Seq[GeoMesaFeatureIndex[_, _]],
235       val atomic: Boolean
236     ) extends GeoMesaFeatureWriter[DS] {
237 
238     import scala.collection.JavaConverters._
239 
240     private val partition = TablePartition(ds, sft).getOrElse {
241       throw new IllegalStateException("Creating a partitioned writer for a non-partitioned schema")
242     }
243 
244     private val cache = new java.util.HashMap[String, IndexWriter]()
245     private val view = cache.asScala
246 
247     override protected def getWriter(feature: SimpleFeature): IndexWriter = {
248       val p = partition.partition(feature)
249       var writer = cache.get(p)
250       if (writer == null) {
251         // reconfigure the partition each time - this should be idempotent, and block
252         // until it is fully created (which may happen in some other thread)
253         def createOne(index: GeoMesaFeatureIndex[_, _]): Unit =
254           ds.adapter.createTable(index, Some(p), index.getSplits(Some(p)))
255         indices.toList.map(i => CachedThreadPool.submit(() => createOne(i))).foreach(_.get)
256         writer = ds.adapter.createWriter(sft, indices, Some(p), atomic)
257         cache.put(p, writer)
258       }
259       writer
260     }
261 
262     override def flush(): Unit = {
263       view.foreach { case (_, writer) => FlushQuietly(writer).foreach(suppressException) }
264       FlushQuietly(statUpdater).foreach(suppressException)
265       propagateExceptions()
266     }
267 
268     override def close(): Unit = {
269       view.foreach { case (_, writer) => CloseQuietly(writer).foreach(suppressException) }
270       CloseQuietly(statUpdater).foreach(suppressException)
271       propagateExceptions()
272     }
273   }

301   /**
302     * Modifies or deletes existing features. Per the data store api, does not allow appending new features.
303     */
304   private trait GeoMesaModifyFeatureWriter[DS <: GeoMesaDataStore[DS]] extends GeoMesaFeatureWriter[DS] {
305 
306     import org.locationtech.geomesa.security.SecureSimpleFeature
307 
308     def filter: Filter
309 
310     private val reader = ds.getFeatureReader(new Query(sft.getTypeName, filter), Transaction.AUTO_COMMIT)
311 
312     // feature returned from reader
313     private var original: SimpleFeature = _
314 
315     // feature that caller will modify
316     private var live: SimpleFeature = _
317 
318     override def hasNext: Boolean = reader.hasNext
319 
320     override def next: SimpleFeature = {
321       original = reader.next()
322       // set the use provided FID hint - allows user to update fid if desired,
323       // but if not we'll use the existing one
324       original.getUserData.put(Hints.USE_PROVIDED_FID, java.lang.Boolean.TRUE)
325       live = ScalaSimpleFeature.copy(sft, original) // this copies user data as well
326       live
327     }
328 
329     override def write(): Unit = {
330       if (original == null) {
331         throw new IllegalStateException("next() must be called before write()")
332       }
333       // update the feature id based on hints before we compare for changes
334       live = GeoMesaFeatureWriter.featureWithFid(live)
335       // only write if feature has actually changed...
336       // comparison of feature ID and attributes - doesn't consider concrete class used
337       if (!ScalaSimpleFeature.equalIdAndAttributes(live, original) || live.visibility != original.visibility) {
338         updateFeature(live, original)
339       }
340       original = null
341       live = null
342     }
343 
344     override def remove(): Unit = {
345       if (original == null) {
346         throw new IllegalStateException("next() must be called before remove()")
347       }
348       removeFeature(original)
349       original = null
350       live = null
351     }
352 
353     abstract override def close(): Unit = {
354       CloseQuietly(reader).foreach(suppressException)
355       super.close() // closes writer
356     }
