23   override def init(options: Map[String, String]): Unit = {
24     import AgeOffFilter.Configuration.ExpiryOpt

26   }

28   override def accept(row: Array[Byte],
29                       rowOffset: Int,
30                       rowLength: Int,
31                       value: Array[Byte],

33                       valueLength: Int,
34                       timestamp: Long): Boolean = timestamp > expiry
35 }
