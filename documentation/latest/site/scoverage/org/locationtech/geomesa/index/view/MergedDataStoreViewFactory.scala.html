41   override def createNewDataStore(params: java.util.Map[String, _]): DataStore = {
42     val configs: Seq[Config] = {
43       val explicit = Option(ConfigParam.lookup(params)).map(c => ConfigFactory.parseString(c).resolve())
44       val loaded = ConfigLoaderParam.flatMap(_.lookupOpt(params)).flatMap { name =>
45         ServiceLoader.load[MergedViewConfigLoader]().find(_.getClass.getName == name).map(_.load())
46       }
47       Seq(explicit, loaded).flatten.flatMap { config =>
48         if (config.hasPath("stores")) { config.getConfigList("stores").asScala } else { Seq.empty }
49       }
50     }
51 
52     if (configs.isEmpty) {
53       throw new IllegalArgumentException("No 'stores' element defined in configuration")
54     }
55 
56     val namespace = NamespaceParam.lookupOpt(params)
57     val nsConfig = namespace.map(ConfigValueFactory.fromAnyRef)
58 
59     val stores = Seq.newBuilder[(DataStore, Option[Filter])]
60     stores.sizeHint(configs.length)
61 
62     try {
63       configs.foreach { config =>
64         lazy val error =
65           new IllegalArgumentException(s"Could not load store using configuration:\n" +
66               config.root().render(ConfigRenderOptions.concise().setFormatted(true)))
67         // inject the namespace into the underlying stores
68         val paramConf = nsConfig.map(config.withValue(NamespaceParam.key, _)).getOrElse(config)
69         val storeParams = new java.util.HashMap[String, AnyRef](paramConf.root().unwrapped())
70         // allow for unquoted keys
71         storeParams.putAll(paramConf.toStringMap().asJava)
72         val filter = try { StoreFilterParam.lookupOpt(storeParams).map(ECQL.toFilter) } catch {
73           case NonFatal(e) =>
74             throw new IllegalArgumentException(s"Invalid store filter '${storeParams.get(StoreFilterParam.key)}'", e)
75         }
76         Try(DataStoreFinder.getDataStore(storeParams)) match {
77           case Success(null)  => throw error
78           case Success(store) => stores += store -> filter
79           case Failure(e)     => throw error.initCause(e)
80         }
81       }
82     } catch {
83       case NonFatal(e) => stores.result.foreach(_._1.dispose()); throw e
84     }
85 
86     val deduplicate = DeduplicateParam.lookup(params).booleanValue()
87     val parallel = ParallelScanParam.lookup(params).booleanValue()
88 
89     new MergedDataStoreView(stores.result, deduplicate, parallel, namespace)
90   }
