20 /**
21  * Visitor for sampling a percent of features
22  *
23  * @param features features to sample
24  * @param percent percent of features to retain
25  * @param threading threading key
26  */
27 class SamplingVisitor(features: SimpleFeatureCollection, percent: Float, threading: Option[String])
28   extends GeoMesaProcessVisitor with LazyLogging {
29 
30   private val manualVisitResults = new ListFeatureCollection(features.getSchema)
31 
32   private var resultCalc = FeatureResult(manualVisitResults)
33 
34   private val nth = (1f / percent).toInt
35   private val thread = threading.map(features.getSchema.indexOf).filter(_ != -1)
36 
37   private val sampling = FeatureSampler.sample(nth, thread)
38 
39   // non-optimized visit
40   override def visit(feature: Feature): Unit = {
41     val sf = feature.asInstanceOf[SimpleFeature]
42     if (sampling(sf)) {
43       manualVisitResults.add(sf)
44     }
45   }
46 
47   override def getResult: FeatureResult = resultCalc
48 
49   override def execute(source: SimpleFeatureSource, query: Query): Unit = {
50     logger.debug(s"Running Geomesa sampling process on source type ${source.getClass.getName}")
51     query.getHints.put(QueryHints.SAMPLING, percent)
52     threading.foreach(query.getHints.put(QueryHints.SAMPLE_BY, _))
53     resultCalc = FeatureResult(source.getFeatures(query))
54   }
55 }
