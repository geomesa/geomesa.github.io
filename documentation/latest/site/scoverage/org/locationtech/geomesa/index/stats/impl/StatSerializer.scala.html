30 /**
31   * Serialize and deserialize stats
32   */
33 trait StatSerializer {
34   def serialize(stat: Stat): Array[Byte]
35   def deserialize(bytes: Array[Byte], immutable: Boolean = false): Stat =
36     deserialize(bytes, 0, bytes.length, immutable)
37   def deserialize(bytes: Array[Byte], offset: Int, length: Int, immutable: Boolean): Stat
38 }
39 
40 object StatSerializer {
41 
42   private val serializers = scala.collection.mutable.Map.empty[String, StatSerializer]
43 
44   def apply(sft: SimpleFeatureType): StatSerializer = serializers.synchronized {
45     serializers.getOrElseUpdate(CacheKeyGenerator.cacheKey(sft), new KryoStatSerializer(sft))
46   }

48   /**
49     * Kryo implementation of stat serializer. Thread-safe.
50     *
51     * @param sft simple feature type
52     */
53   class KryoStatSerializer(sft: SimpleFeatureType) extends StatSerializer {
54 
55     override def serialize(stat: Stat): Array[Byte] = {
56       val output = KryoStatSerializer.outputs.getOrElseUpdate(new Output(1024, -1))
57       output.setOutputStream(null) // resets the buffer
58       KryoStatSerializer.write(output, sft, stat)
59       output.toBytes
60     }
61 
62     override def deserialize(bytes: Array[Byte], offset: Int, length: Int, immutable: Boolean): Stat = {
63       val input = KryoStatSerializer.inputs.getOrElseUpdate(new NonMutatingInput())
64       input.setBuffer(bytes, offset, length)
65       KryoStatSerializer.read(input, sft, immutable)
66     }
67   }

112     private def write(output: Output, sft: SimpleFeatureType, stat: Stat): Unit = {
113       stat match {
114         case s: CountStat           => output.writeByte(CountByte);           writeCount(output, s)
115         case s: MinMax[_]           => output.writeByte(MinMaxByte);          writeMinMax(output, sft, s)
116         case s: EnumerationStat[_]  => output.writeByte(EnumerationByte);     writeEnumeration(output, sft, s)
117         case s: TopK[_]             => output.writeByte(TopKByte);            writeTopK(output, sft, s)
118         case s: Histogram[_]        => output.writeByte(HistogramByte);       writeHistogram(output, sft, s)
119         case s: Frequency[_]        => output.writeByte(FrequencyByte);       writeFrequency(output, sft, s)
120         case s: Z3Histogram         => output.writeByte(Z3HistogramByte);     writeZ3Histogram(output, sft, s)
121         case s: Z3Frequency         => output.writeByte(Z3FrequencyByte);     writeZ3Frequency(output, sft, s)
122         case s: IteratorStackCount  => output.writeByte(IteratorStackByte);   writeIteratorStackCount(output, s)
123         case s: SeqStat             => output.writeByte(SeqStatByte);         writeSeqStat(output, sft, s)
124         case s: DescriptiveStats    => output.writeByte(DescriptiveStatByte); writeDescriptiveStats(output, sft, s)
125         case s: GroupBy[_]          => output.writeByte(GroupByByte);         writeGroupBy(output, sft, s)
126         case _ => throw new UnsupportedOperationException(s"Unhandled stat $stat")
127       }
128     }

135         case TopKByte              => readTopK(input, sft, immutable, 3)

139         case Z3FrequencyByte       => readZ3Frequency(input, sft, immutable, 3)
140         case IteratorStackByte     => readIteratorStackCount(input, sft, immutable)

173     private def readGroupBy(input: Input, sft: SimpleFeatureType, immutable: Boolean, version: Int): GroupBy[_] = {
174       val attribute = version match {
175         case 2 => input.readString
176         case 1 => sft.getDescriptor(input.readInt(true)).getLocalName
177         case _ => throw new IllegalArgumentException(s"Invalid group by serialization version: $version")
178       }
179       val exampleStat = input.readString()
180       val keyLength   = input.readInt(true)
181 
182       val classTag = ClassTag[Any](sft.getDescriptor(attribute).getType.getBinding)
183       val stat = if (immutable) {
184         new GroupBy(sft, attribute, exampleStat)(classTag) with ImmutableStat
185       } else {
186         new GroupBy(sft, attribute, exampleStat)(classTag)
187       }
188 
189       val keyReader = reader(input, sft.getDescriptor(attribute).getType.getBinding)
190 
191       var i = 0
192       while (i < keyLength) {
193         val key = keyReader.apply()
194         val groupedStat = read(input, sft, immutable)
195         stat.groups.put(key, groupedStat)
196         i += 1
197       }
198       stat
199     }

290     private def readMinMax(input: Input, sft: SimpleFeatureType, immutable: Boolean, version: Int): MinMax[_] = {
291       val attribute = version match {
292         case 3     => input.readString()
293         case 1 | 2 => sft.getDescriptor(input.readInt(true)).getLocalName
294         case _ => throw new IllegalArgumentException(s"Invalid min/max serialization version: $version")
295       }
296       val hpp = if (version > 1) {
297         val log2m = input.readInt(true)
298         val size = input.readInt(true)
299         val bytes = Array.fill(size)(input.readInt)
300         HyperLogLog(log2m, bytes)
301       } else {
302         val hppBytes = Array.ofDim[Byte](input.readInt(true))
303         input.read(hppBytes)
304         val clearspring = com.clearspring.analytics.stream.cardinality.HyperLogLog.Builder.build(hppBytes)
305         // use reflection to access private variables
306         def getField[T](name: String): T = {
307           val field = clearspring.getClass.getDeclaredField(name)
308           field.setAccessible(true)
309           field.get(clearspring).asInstanceOf[T]
310         }
311         val log2m = getField[Int]("log2m")
312         val registerSet = getField[RegisterSet]("registerSet").bits
313         HyperLogLog(log2m, registerSet)
314       }
315 
316       val binding = sft.getDescriptor(attribute).getType.getBinding
317       val read = reader(input, binding)
318       val min = read()
319       val max = read()
320 
321       val defaults = MinMaxDefaults[Any](binding)
322 
323       if (immutable) {
324         new MinMax[Any](sft, attribute, min, max, hpp)(defaults) with ImmutableStat
325       } else {
326         new MinMax[Any](sft, attribute, min, max, hpp)(defaults)
327       }
328     }

510     private def readFrequency(input: Input, sft: SimpleFeatureType, immutable: Boolean, version: Int): Frequency[_] = {
511       def name(i: Int): String = sft.getDescriptor(i).getLocalName
512       val (attribute, dtg) = version match {
513         case 3     => (input.readString(), Option(input.readString()))
514         case 1 | 2 => (name(input.readInt(true)), Some(input.readInt(true)).filter(_ != -1).map(name))
515         case _ => throw new IllegalArgumentException(s"Invalid frequency serialization version: $version")
516       }
517       val period = if (version > 1) { TimePeriod.withName(input.readString()) } else { TimePeriod.Week }
518       val precision = input.readInt(true)
519       val eps = input.readDouble()
520       val confidence = input.readDouble()
521 
522       val binding = sft.getDescriptor(attribute).getType.getBinding
523       val stat = if (immutable) {
524         new Frequency[Any](sft, attribute, dtg, period, precision, eps, confidence)(ClassTag[Any](binding)) with ImmutableStat
525       } else {
526         new Frequency[Any](sft, attribute, dtg, period, precision, eps, confidence)(ClassTag[Any](binding))
527       }
528 
529       val sketchCount = input.readInt(true)
530       var c = 0
531       while (c < sketchCount) {
532         val week = input.readShort
533         val sketch = stat.newSketch
534         stat.sketchMap.put(week, sketch)
535         var i = 0
536         while (i < sketch.table.length) {
537           readCountArray(input, sketch.table(i))
538           i += 1
539         }
540         sketch._size = input.readLong(true)
541         c += 1
542       }
543 
544       stat
545     }
