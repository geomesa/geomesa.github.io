26 /**
27  * Singleton for registering and managing running queries.
28  */
29 object ThreadManagement {
30 
31   private val logger = Logger(LoggerFactory.getLogger(ThreadManagement.getClass.getName.replace("$", "")))
32 
33   private val executor = {
34     val ex = new ScheduledThreadPoolExecutor(2)
35     ex.setRemoveOnCancelPolicy(true)
36     ExitingExecutor(ex, force = true)
37   }
38 
39   /**
40    * Class for scans that are managed, i.e. tracked and terminated if they exceed a specified timeout
41    *
42    * @param underlying low-level scan to be stopped
43    * @param timeout scan timeout
44    * @param typeName type name, used for log messages
45    * @param filter query filter, used for log messages
46    * @tparam T type
47    */
48   class ManagedScan[T](underlying: LowLevelScanner[T], timeout: Timeout, typeName: String, filter: Option[Filter])
49       extends CloseableIterator[T] {
50 
51     def this(underlying: LowLevelScanner[T], timeout: Timeout, plan: QueryStrategyPlan) =
52       this(underlying, timeout, plan.filter.index.sft.getTypeName, plan.filter.filter)
53 
54     // we can use a volatile var since we only update the value with a single thread
55     @volatile
56     private var terminated = timeout.absolute <= System.currentTimeMillis()
57 
58     private val iter = ExceptionalIterator(if (terminated) { Iterator.empty } else { underlying.iterator })
59     private val cancel = if (terminated) { None } else {
60       val timeout = math.max(1, this.timeout.absolute - System.currentTimeMillis())
61       Some(executor.schedule(new QueryKiller(this), timeout, TimeUnit.MILLISECONDS))
62     }
63 
64     // note: check iter.hasNext first so we get updated terminated flag
65     override def hasNext: Boolean = iter.hasNext || terminated
66 
67     override def next(): T = {
68       if (terminated) {
69         val e = new RuntimeException(s"Scan terminated due to timeout of ${timeout.relative}ms")
70         iter.suppressed.foreach(e.addSuppressed)
71         throw e
72       } else {
73         iter.next()
74       }
75     }
76 
77     /**
78      * Forcibly terminate the scan
79      */
80     def terminate(): Unit = {
81       terminated = true
82       try {
83         logger.warn(
84           s"Stopping scan on schema '$typeName' with filter '${filter.fold("INCLUDE")(FilterHelper.toString)}' " +
85               s"based on timeout of ${timeout.relative}ms")
86         underlying.close()
87       } catch {
88         case NonFatal(e) => logger.warn("Error cancelling scan:", e)
89       }
90     }
91 
92     /**
93      * Was the scan terminated due to timeout
94      *
95      * @return
96      */
97     def isTerminated: Boolean = terminated
98 
99     override def close(): Unit = {
100       cancel.foreach(_.cancel(false))
101       // if terminated, we've already closed the iterator
102       if (!terminated) {
103         underlying.close()
104       }
105     }
106   }
107 
108   /**
109    * Low level scanner that can be closed to terminate a scan
110    *
111    * @tparam T type
112    */
113   trait LowLevelScanner[T] extends Closeable {
114     def iterator: Iterator[T]
115   }
116 
117   /**
118    * Timeout holder
119    *
120    * @param relative relative timeout, in millis
121    * @param absolute absolute timeout, in system millis since epoch
122    */
123   case class Timeout(relative: Long, absolute: Long)
124 
125   object Timeout {
126     def apply(relative: Long): Timeout = Timeout(relative, System.currentTimeMillis() + relative)
127     def apply(relative: Duration): Timeout = apply(relative.toMillis)
128     def apply(relative: String): Timeout = apply(Duration(relative))
129   }
130 
131   /**
132    * Runnable to handle terminating a scan
133    *
134    * @param scan scan to terminate
135    */
136   private class QueryKiller(val scan: ManagedScan[_]) extends Runnable {
137     override def run(): Unit = scan.terminate()
138   }
