23 /**
24   * The histogram's state is stored in an indexed array, where the index is the bin number
25   * and the values are the counts.
26   *
27   * Tracks geometry and date attributes as a single value.
28   *
29   * @param sft simple feature type
30   * @param geom geometry attribute in the sft
31   * @param dtg date attribute in the sft
32   * @param period time period to use for z index
33   * @param length number of bins the histogram has, per period
34  */
35 class Z3Histogram(
36     val sft: SimpleFeatureType,
37     val geom: String,
38     val dtg: String,
39     val period: TimePeriod,
40     val length: Int
41   ) extends Stat with LazyLogging {
42 
43   import Z3Histogram._
44 
45   override type S = Z3Histogram
46 
47   private val g = sft.indexOf(geom)
48   private val d = sft.indexOf(dtg)
49 
50   private val sfc = Z3SFC(period)
51   private val timeToBin = BinnedTime.timeToBinnedTime(period)
52   private val binToDate = BinnedTime.binnedTimeToDate(period)
53   private val minZ = sfc.index(minGeom.getX, minGeom.getY, sfc.time.min.toLong)
54   private val maxZ = sfc.index(maxGeom.getX, maxGeom.getY, sfc.time.max.toLong)
55 
56   private lazy val jsonFormat = period match {
57     case TimePeriod.Day   => s"$period-%05d"
58     case TimePeriod.Week  => s"$period-%04d"
59     case TimePeriod.Month => s"$period-%03d"
60     case TimePeriod.Year  => s"$period-%02d"
61   }
62 
63   private [stats] val binMap = scala.collection.mutable.Map.empty[Short, BinnedArray[java.lang.Long]]
64   private [stats] def newBins = new BinnedArray(new LongBinning(length, (minZ, maxZ)))
65 
66   def timeBins: Seq[Short] = binMap.keys.toSeq.sorted
67   def count(timeBin: Short, i: Int): Long = binMap.get(timeBin).map(_.counts(i)).getOrElse(0L)
68 
69   def directIndex(timeBin: Short, value: Long): Int = binMap.get(timeBin).map(_.indexOf(value)).getOrElse(-1)
70 
71   def indexOf(value: (Geometry, Date)): (Short, Int) = {
72     val (timeBin, z) = toKey(value._1, value._2, lenient = false)
73     (timeBin, directIndex(timeBin, z))
74   }
75 
76   def medianValue(timeBin: Short, i: Int): (Geometry, Date) = fromKey(timeBin, binMap(timeBin).medianValue(i))
77 
78   private def toKey(geom: Geometry, dtg: Date, lenient: Boolean): (Short, Long) = {
79     import org.locationtech.geomesa.utils.geotools.Conversions.RichGeometry
80     val BinnedTime(bin, offset) = timeToBin(dtg.getTime)
81     val centroid = geom.safeCentroid()
82     val z = sfc.index(centroid.getX, centroid.getY, offset, lenient)
83     (bin, z)
84   }
85 
86   private def fromKey(timeBin: Short, z: Long): (Geometry, Date) = {
87     val (x, y, t) = sfc.invert(z)
88     val dtg = Date.from(binToDate(BinnedTime(timeBin, t)).toInstant)
89     val geom = Z3Histogram.gf.createPoint(new Coordinate(x, y))
90     (geom, dtg)
91   }
92 
93   /**
94     * Split the stat into a separate stat per time bin of z data. Allows for separate handling of the reduced
95     * data set.
96     *
97     * @return
98     */
99   def splitByTime: Seq[(Short, Z3Histogram)] = {
100     binMap.toSeq.map { case (w, bins) =>
101       val hist = new Z3Histogram(sft, geom, dtg, period, length)
102       hist.binMap.put(w, bins)
103       (w, hist)
104     }
105   }
106 
107   override def observe(sf: SimpleFeature): Unit = {
108     val geom = sf.getAttribute(g).asInstanceOf[Geometry]
109     val dtg  = sf.getAttribute(d).asInstanceOf[Date]
110     if (geom != null && dtg != null) {
111       try {
112         val (timeBin, z3) = toKey(geom, dtg, lenient = false)
113         binMap.getOrElseUpdate(timeBin, newBins).add(z3, 1L)
114       } catch {
115         case e: Exception => logger.warn(s"Error observing geom '$geom' and date '$dtg': ${e.toString}")
116       }
117     }
118   }
119 
120   override def unobserve(sf: SimpleFeature): Unit = {
121     val geom = sf.getAttribute(g).asInstanceOf[Geometry]
122     val dtg  = sf.getAttribute(d).asInstanceOf[Date]
123     if (geom != null && dtg != null) {
124       try {
125         val (timeBin, z3) = toKey(geom, dtg, lenient = true)
126         binMap.get(timeBin).foreach(_.add(z3, -1L))
127       } catch {
128         case e: Exception => logger.warn(s"Error un-observing geom '$geom' and date '$dtg': ${e.toString}")
129       }
130     }
131   }
132 
133   /**
134     * Creates a new histogram by combining another histogram with this one
135     */
136   override def +(other: Z3Histogram): Z3Histogram = {
137     val plus = new Z3Histogram(sft, geom, dtg, period, length)
138     plus += this
139     plus += other
140     plus
141   }
142 
143   /**
144     * Copies another histogram into this one
145     */
146   override def +=(other: Z3Histogram): Unit = {
147     if (length != other.length) {
148       throw new UnsupportedOperationException("Can only add z3 histograms with the same length")
149     }
150     other.binMap.foreach { case (w, bins) =>
151       binMap.get(w) match {
152         case None => binMap.put(w, bins) // note: sharing a reference now
153         case Some(b) =>
154           var i = 0
155           while (i < b.length) {
156             b.counts(i) += bins.counts(i)
157             i += 1
158           }
159       }
160     }
161   }
162 
163   override def toJsonObject: Any =
164     binMap.toSeq.sortBy(_._1)
165       .map { case (p, bins) => (String.format(jsonFormat, Short.box(p)), bins) }
166       .map { case (label, bins) => Map(label-> Map("bins" -> bins.counts)) }
167 
168   override def isEmpty: Boolean = binMap.values.forall(_.counts.forall(_ == 0))
169 
170   override def clear(): Unit = binMap.values.foreach(_.clear())
171 
172   override def isEquivalent(other: Stat): Boolean = other match {
173     case that: Z3Histogram =>
174       g == that.g && d == that.d && period == that.period &&
175           length == that.length && binMap.keySet == that.binMap.keySet &&
176           binMap.forall { case (w, bins) => java.util.Arrays.equals(bins.counts, that.binMap(w).counts) }
177     case _ => false
178   }
179 }
