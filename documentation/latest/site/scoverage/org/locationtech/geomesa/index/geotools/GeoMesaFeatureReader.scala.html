25 class GeoMesaFeatureReader(result: QueryResult) {
26 
27   def schema: SimpleFeatureType = result.schema
28   def hints: Hints = result.hints
29 
30   def reader(): SimpleFeatureReader = new ResultReader()
31 
32   private class ResultReader extends SimpleFeatureReader {
33 
34     private val iter = result.iterator()
35 
36     override def getFeatureType: SimpleFeatureType = result.schema
37     override def hasNext: Boolean = iter.hasNext
38     override def next(): SimpleFeature = iter.next()
39     override def close(): Unit = iter.close()
40   }
41 }

43 object GeoMesaFeatureReader extends MethodProfiling {
44 
45   def apply(
46       sft: SimpleFeatureType,
47       query: Query,
48       qp: QueryRunner,
49       audit: Option[AuditWriter]): GeoMesaFeatureReader = {
50     audit match {
51       case None => new GeoMesaFeatureReader(qp.runQuery(sft, query))
52       case Some(a) =>
53         val timings = new TimingsImpl()
54         val result = profile(time => timings.occurrence("planning", time))(qp.runQuery(sft, query))
55         new GeoMesaFeatureReaderWithAudit(result, timings, a, sft.getTypeName, query.getFilter)
56     }
57   }

63      * widely as we want to ensure that the sft has been loaded from our catalog
64      *
65      * @param sft simple feature type
66      * @param transaction transaction
67      * @param query query
68      * @return
69      */
70     private[geomesa] def getFeatureReader(
71         sft: SimpleFeatureType,

74   }

76   class GeoMesaFeatureReaderWithAudit(
77       result: QueryResult,
78       timings: Timings,
79       auditWriter: AuditWriter,
80       typeName: String,
81       filter: Filter
82     ) extends GeoMesaFeatureReader(result) {
83 
84     override def reader(): SimpleFeatureReader = new ResultReaderWithAudit()
85 
86     private class ResultReaderWithAudit extends SimpleFeatureReader with MethodProfiling {
87 
88       private val start = System.currentTimeMillis()
89       private val user = auditWriter.auditProvider.getCurrentUserId
90       private val closed = new AtomicBoolean(false)
91       private val count = new AtomicLong(0L)
92       private val iter = profile(time => timings.occurrence("planning", time)) {
93         val base = result.iterator()
94         // note: has to be done after running the query so hints are set
95         if (result.schema == BinaryOutputEncoder.BinEncodedSft) {
96           // bin queries pack multiple records into each feature
97           // to count the records, we have to count the total bytes coming back, instead of the number of features
98           val bytesPerHit = if (result.hints.getBinLabelField.isDefined) { 24 } else { 16 }
99           base.map { sf => count.addAndGet(sf.getAttribute(0).asInstanceOf[Array[Byte]].length / bytesPerHit); sf }
100         } else {
101           base.map { sf => count.incrementAndGet(); sf }
102         }
103       }
104 
105       override def getFeatureType: SimpleFeatureType = result.schema
106 
107       override def hasNext: Boolean = profile(time => timings.occurrence("hasNext", time))(iter.hasNext)
108       override def next(): SimpleFeature = profile(time => timings.occurrence("next", time))(iter.next())
109 
110       override def close(): Unit = {
111         if (closed.compareAndSet(false, true)) {
112           try { iter.close() } finally {
113             val plans = result match {
114               case r: QueryPlanResult[_] => r.plans
115               case _ => Seq.empty
116             }
117             // note: implementations should be asynchronous
118             auditWriter.writeQueryEvent(typeName, user, filter, hints, plans, start, System.currentTimeMillis(),
119               timings.time("planning"), timings.time("next") + timings.time("hasNext"), count.get)
120           }
121         }
122       }
123     }
124   }
125 }
