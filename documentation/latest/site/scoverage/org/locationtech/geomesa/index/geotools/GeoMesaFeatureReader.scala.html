40 object GeoMesaFeatureReader extends MethodProfiling {

42   def apply(
43       sft: SimpleFeatureType,
44       query: Query,

46       audit: Option[AuditWriter]): GeoMesaFeatureReader = {

70   private class GeoMesaFeatureReaderWithAudit(
71       result: QueryResult,
72       auditWriter: AuditWriter,
73       typeName: String,
74       filter: Filter,
75     ) extends GeoMesaFeatureReader(result) {
76 
77     override def reader(): SimpleFeatureReader = new ResultReaderWithAudit()
78 
79     private class ResultReaderWithAudit extends SimpleFeatureReader with MethodProfiling {
80 
81       private val start = System.currentTimeMillis()
82       private val user = auditWriter.auditProvider.getCurrentUserId
83       private val closed = new AtomicBoolean(false)
84       private val count = new AtomicLong(0L)
85       private val runtimeNanos = new AtomicLong(0L)
86       private val iter = result.iterator(runtimeNanos.set, count.set)
87 
88       override def getFeatureType: SimpleFeatureType = result.schema
89       override def hasNext: Boolean = iter.hasNext
90       override def next(): SimpleFeature = iter.next()
91 
92       override def close(): Unit = {
93         if (closed.compareAndSet(false, true)) {
94           try { iter.close() } finally {
95             // note: implementations should be asynchronous
96             auditWriter.writeQueryEvent(typeName, user, filter, hints, result.plans, start, System.currentTimeMillis(),
97               result.planTimeNanos / 1000000L, runtimeNanos.get() / 1000000L, count.get)
98           }
99         }
100       }
101     }
102   }
