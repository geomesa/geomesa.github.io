24 class GeoMesaFeatureReader(result: QueryResult) {
25 
26   def schema: SimpleFeatureType = result.schema
27   def hints: Hints = result.hints
28 
29   def reader(): SimpleFeatureReader = new ResultReader()
30 
31   private class ResultReader extends SimpleFeatureReader {
32 
33     private val iter = result.iterator()
34 
35     override def getFeatureType: SimpleFeatureType = result.schema
36     override def hasNext: Boolean = iter.hasNext
37     override def next(): SimpleFeature = iter.next()
38     override def close(): Unit = iter.close()
39   }
40 }

42 object GeoMesaFeatureReader extends MethodProfiling {
43 
44   def apply(
45       sft: SimpleFeatureType,
46       query: Query,
47       qp: QueryRunner,
48       audit: Option[AuditWriter]): GeoMesaFeatureReader = {
49     audit match {
50       case None => new GeoMesaFeatureReader(qp.runQuery(sft, query))
51       case Some(a) =>
52         val timings = new TimingsImpl()
53         val result = profile(time => timings.occurrence("planning", time))(qp.runQuery(sft, query))
54         new GeoMesaFeatureReaderWithAudit(result, timings, a, sft.getTypeName, query.getFilter)
55     }
56   }

74 
75   class GeoMesaFeatureReaderWithAudit(
76       result: QueryResult,
77       timings: Timings,
78       auditWriter: AuditWriter,
79       typeName: String,
80       filter: Filter
81     ) extends GeoMesaFeatureReader(result) {
82 
83     override def reader(): SimpleFeatureReader = new ResultReaderWithAudit()
84 
85     private class ResultReaderWithAudit extends SimpleFeatureReader with MethodProfiling {
86 
87       private val closed = new AtomicBoolean(false)
88       private val count = new AtomicLong(0L)
89       private val iter = profile(time => timings.occurrence("planning", time)) {
90         val base = result.iterator()
91         // note: has to be done after running the query so hints are set
92         if (result.schema == BinaryOutputEncoder.BinEncodedSft) {
93           // bin queries pack multiple records into each feature
94           // to count the records, we have to count the total bytes coming back, instead of the number of features
95           val bytesPerHit = if (result.hints.getBinLabelField.isDefined) { 24 } else { 16 }
96           base.map { sf => count.addAndGet(sf.getAttribute(0).asInstanceOf[Array[Byte]].length / bytesPerHit); sf }
97         } else {
98           base.map { sf => count.incrementAndGet(); sf }
99         }
100       }
101 
102       override def getFeatureType: SimpleFeatureType = result.schema
103 
104       override def hasNext: Boolean = profile(time => timings.occurrence("hasNext", time))(iter.hasNext)
105       override def next(): SimpleFeature = profile(time => timings.occurrence("next", time))(iter.next())
106 
107       override def close(): Unit = {
108         if (closed.compareAndSet(false, true)) {
109           try { iter.close() } finally {
110             // note: implementations should be asynchronous
111             auditWriter.writeQueryEvent(typeName, filter, hints, timings.time("planning"),
112               timings.time("next") + timings.time("hasNext"), count.get)
113           }
114         }
115       }
116     }
117   }
118 }
