140     * Returns batches of [threading key][dictionary deltas][record batch]. Will sort each batch,

149   class DeltaAggregate(
150       sft: SimpleFeatureType,
151       dictionaryFields: Seq[String],
152       encoding: SimpleFeatureEncoding,
153       ipcOpts: IpcOption,
154       sort: Option[(String, Boolean)],
155       batchSize: Int
156     ) extends ArrowAggregate {
157 
158     private val features = Array.ofDim[SimpleFeature](batchSize)
159 
160     private var writer: DeltaWriter = _
161 
162     private var index = 0
163 
164     override def init(): Unit = {
165       writer = new DeltaWriter(sft, dictionaryFields, encoding, ipcOpts, sort, batchSize)
166     }
167 
168     override def aggregate(sf: SimpleFeature): Int = {
169       // we have to copy since the feature might be re-used
170       // TODO we could probably optimize this...
171       features(index) = ScalaSimpleFeature.copy(sf)
172       index += 1
173       1
174     }
175 
176     override def encode(): Array[Byte] = try { writer.encode(features, index) } finally { index = 0 }
177 
178     override def cleanup(): Unit = {
179       CloseWithLogging(writer)
180       writer = null
181     }
182   }

184   /**
185     * Reduce function for delta batches.
186     *
187     * First feature contains metadata for arrow file and dictionary batch, subsequent features
188     * contain record batches, final feature contains EOF indicator
189     *
190     * @param sft simple feature type
191     * @param dictionaryFields dictionaries
192     * @param encoding encoding
193     * @param batchSize batch size
194     * @param sort sort

197   class DeltaReducer(
198       private var sft: SimpleFeatureType,
199       private var dictionaryFields: Seq[String],
200       private var encoding: SimpleFeatureEncoding,
201       private var ipcOpts: IpcOption,
202       private var batchSize: Int,
203       private var sort: Option[(String, Boolean)],
204       private var sorted: Boolean,
205       private var process: Boolean
206     ) extends FeatureReducer {
207 
208     def this() = this(null, null, null, null, -1, null, false, true) // no-arg constructor required for serialization
209 
210     override def init(state: Map[String, String]): Unit = {
211       sft = ReducerConfig.sft(state)
212       dictionaryFields = StringSerialization.decodeSeq(state(ReducerConfig.DictionariesKey))
213       encoding = ReducerConfig.encoding(state)
214       ipcOpts = ReducerConfig.ipcOption(state)
215       batchSize = ReducerConfig.batch(state)
216       sort = ReducerConfig.sort(state)
217       sorted = ReducerConfig.sorted(state)
218       process = ReducerConfig.process(state)
219     }
220 
221     override def state: Map[String, String] = Map(
222       ReducerConfig.sftName(sft),
223       ReducerConfig.sftSpec(sft),
224       ReducerConfig.DictionariesKey -> StringSerialization.encodeSeq(dictionaryFields),
225       ReducerConfig.encoding(encoding),
226       ReducerConfig.ipcOption(ipcOpts),
227       ReducerConfig.batch(batchSize),
228       ReducerConfig.sort(sort),
229       ReducerConfig.sorted(sorted),
230       ReducerConfig.process(process)
231     )
232 
233     override def apply(features: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = {
234       val files = features.map(_.getAttribute(0).asInstanceOf[Array[Byte]])
235       val result = DeltaWriter.reduce(sft, dictionaryFields, encoding, ipcOpts, sort, sorted, batchSize, process, files)
236       val sf = resultFeature()
237       result.map { bytes => sf.setAttribute(0, bytes); sf }
238     }
239 
240     def canEqual(other: Any): Boolean = other.isInstanceOf[DeltaReducer]
241 
242     override def equals(other: Any): Boolean = other match {
243       case that: DeltaReducer if that.canEqual(this) =>
244         sft == that.sft && dictionaryFields == that.dictionaryFields && encoding == that.encoding &&
245             batchSize == that.batchSize && sort == that.sort && sorted == that.sorted && that.process == process
246       case _ => false
247     }
248 
249     override def hashCode(): Int = {
250       val state = Seq(sft, dictionaryFields, encoding, batchSize, sort, process)
251       state.map(Objects.hashCode).foldLeft(0)((a, b) => 31 * a + b)
252     }
253   }

303     def process(options: Map[String, String]): Boolean = options.get(ProcessKey).forall(_.toBoolean)
304   }
