71    * Default indices for a feature type, based on user data hints and the schema attributes
72    *
73    * @param sft simple feature type
74    * @return
75    */
76   @deprecated("Replaced with fromIndexFlag and fromAttributeFlag")
77   def indices(sft: SimpleFeatureType, hint: Option[String]): Seq[IndexId] = {
78     hint match {
79       case Some(h) => fromFeatureFlag(sft, h)
80       case None =>
81         sft.getAttributeDescriptors.asScala.toSeq.flatMap { d =>
82           val indices = d.getIndexFlags.flatMap(fromAttributeFlag(sft, d, _))
83           if (indices.isEmpty && sft.getGeomField == d.getLocalName) {
84             // add in default indices on geom
85             all().collect { case i: ConfiguredIndex => i.indexFor(sft, d) }.flatten
86           } else {
87             indices.filterNot(_.name == EmptyIndex.name)
88           }
89         }
90     }
91   }

113   private val factories = ServiceLoader.load[GeoMesaFeatureIndexFactory]() :+ DefaultFeatureIndexFactory

115   /**
116     * Indices for a feature type, based on user data hints and the schema attributes
117     *
118     * @param sft simple feature type
119     * @return
120     */
121   def indices(sft: SimpleFeatureType): Seq[IndexId] = {
122     val enabled = fromIndexFlags(sft)
123     if (enabled.nonEmpty) {
124       enabled
125     } else {
126       // add in the ID index since there's no attribute descriptor for it
127       fromAttributeFlags(sft) ++ IdIndex.defaultIndicesFor(sft)
128     }
129   }
130 
131   /**
132    * Configures indices based on the EnabledIndices feature-level user data
133    *
134    * @param sft simple feature type
135    * @return
136    */
137   private def fromIndexFlags(sft: SimpleFeatureType): Seq[IndexId] = {
138     Option(sft.getUserData.get(Configs.EnabledIndices)).toSeq.flatMap(_.asInstanceOf[String].split(",").map(_.trim)).flatMap { flag =>
139       val indices = factories.flatMap(_.fromFeatureFlag(sft, flag))
140       if (indices.isEmpty) {
141         throw new IllegalArgumentException(
142           s"Invalid index flag '$flag' does not exist or does not support the schema: ${SimpleFeatureTypes.encodeType(sft)}")
143       }
144       indices
145     }
146   }
147 
148   /**
149    * Configures indices based on attribute-level 'index' user data
150    *
151    * @param sft simple feature type
152    * @return
153    */
154   private def fromAttributeFlags(sft: SimpleFeatureType): Seq[IndexId] = {
155     sft.getAttributeDescriptors.asScala.toSeq.flatMap { d =>
156       val indices = d.getIndexFlags.flatMap { flag =>
157         val indices = factories.flatMap(_.fromAttributeFlag(sft, d, flag))
158         if (indices.isEmpty) {
159           throw new IllegalArgumentException(
160             s"Invalid index flag '$flag' on attribute ${d.getLocalName} does not exist or does not support the type: " +
161               d.getType.getBinding.getSimpleName)
162         }
163         indices
164       }
165       if (indices.isEmpty && sft.getGeomField == d.getLocalName) {
166         // add in default indices on geom
167         Seq(Z3Index, XZ3Index, Z2Index, XZ2Index).flatMap(i => i.indexFor(sft, d))
168       } else {
169         indices.filterNot(_.name == EmptyIndex.name)
170       }
171     }
172   }
173 
174   /**
175     * Gets the names of available indices, based on the schema attributes. Can be used with 'geomesa.indices.enabled'
176     * to specify default indices of the given type
177     *
178     * @param sft simple feature type
179     * @return
180     */
181   @deprecated("Replaced with all()")
182   def available(sft: SimpleFeatureType): Seq[(String, Int)] = factories.flatMap(_.available(sft)).distinct
183 
184   /**
185    * Gets all available indices
186    *
187    * @return
188    */
189   def all(): Seq[NamedIndex] = factories.flatMap(_.all())
190 
191   /**
192     * Create an index instance
193     *
194     * @param ds data store
195     * @param sft simple feature type
196     * @param indices index identifiers
197     * @return
198     */
199   def create(ds: GeoMesaDataStore[_], sft: SimpleFeatureType, indices: Seq[IndexId]): Seq[GeoMesaFeatureIndex[_, _]] = {
200     indices.map { index =>
201       factories.flatMap(_.create(ds, sft, index).toSeq) match {
202         case Seq(i) => i
203         case Nil    => throw new IllegalArgumentException(s"No index found using identifier '$index'")
204         case _      => throw new IllegalArgumentException(s"Multiple indices found using identifier '$index'")
205       }
206     }
207   }
