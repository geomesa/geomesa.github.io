41   override def init(options: Map[String, String]): Unit = {
42     val spec = options(SftOpt)
43     val sft = IteratorCache.sft(spec)
44     // note: index won't have a hook to the data store, so some operations aren't available
45     val index = options.get(IndexSftOpt) match {
46       case None => IteratorCache.index(sft, spec, options(IndexOpt))
47       case Some(ispec) => IteratorCache.index(IteratorCache.sft(ispec), ispec, options(IndexOpt))
48     }
49 
50     // noinspection ScalaDeprecation
51     val kryo = if (index.serializedWithId) { SerializationOptions.none } else { SerializationOptions.withoutId }
52     reusableSf = IteratorCache.serializer(spec, kryo).getReusableFeature
53     reusableSf.setIdParser(index.getIdFromRow(_, _, _, null))
54     aggregateSf = reusableSf
55 
56     val transformSft = options.get(TransformDefsOpt).map { t =>
57       val ts = options.getOrElse(TransformSchemaOpt,
58         throw new IllegalArgumentException("Defined a transform but no transform schema"))
59       val transformSft = IteratorCache.sft(ts)
60       val reusableTransformSf = TransformSimpleFeature(IteratorCache.sft(spec), transformSft, t)
61       reusableTransformSf.setFeature(reusableSf)
62       aggregateSf = reusableTransformSf // note: side-effect in map
63       transformSft
64     }
65     val sampling = sample(options)
66     val cql = options.get(CqlOpt).map(IteratorCache.filter(sft, spec, _))
67     validate = (cql, sampling) match {
68       case (None, None)             => ValidateAll
69       case (Some(filt), None)       => new CqlValidator(filt)
70       case (None, Some(samp))       => new SampleValidator(samp)
71       case (Some(filt), Some(samp)) => new CqlSampleValidator(filt,  samp)
72     }
73     batchSize = options.get(BatchSizeOpt).map(_.toInt).getOrElse(defaultBatchSize)
74     result = createResult(sft, transformSft, batchSize, options)
75   }

77   /**
78    * Aggregates a batch of data. May not exhaust the underlying data
79    *
80    * @param callback callback to provide for results
81    * @return callback
82    */
83   def aggregate[A <: AggregateCallback](callback: A): A = {
84     // noinspection LanguageFeature
85     result.init()
86 
87     val status = new AggregateStatus(callback)
88 
89     var rowValue = try { if (hasNextData) { nextData() } else { null } } catch {
90       case NonFatal(e) => logger.error("Error in underlying scan while aggregating value:", e); null
91     }
92 
93     while (rowValue != null) {
94       try {
95         reusableSf.setIdBuffer(rowValue.row, rowValue.rowOffset, rowValue.rowLength)
96         reusableSf.setBuffer(rowValue.value, rowValue.valueOffset, rowValue.valueLength)
97         if (validate(reusableSf)) {
98           // write the record to our aggregated results
99           status.aggregated(result.aggregate(aggregateSf))
100         } else {
101           status.skipped()
102         }
103       } catch {
104         case NonFatal(e) =>
105           logger.error(s"Error aggregating value for ${debugSf()}:", e)
106           status.skipped()
107       }
108 
109       rowValue = try { if (status.continue() && hasNextData) { nextData() } else { null } } catch {
110         case NonFatal(e) => logger.error("Error in underlying scan while aggregating value:", e); null
111       }
112     }
113 
114     status.done()
115 
116     callback
117   }

144   /**
145    * Class for tracking status of current aggregation
146    *
147    * @param callback results callback
148    */
149   private class AggregateStatus(callback: AggregateCallback) {
150 
151     private var count = 0
152     private var skip = 0
153 
154     def aggregated(count: Int): Unit = this.count += count
155 
156     def skipped(): Unit = skip += 1
157 
158     // noinspection LanguageFeature
159     def continue(): Boolean = {
160       if (count >= batchSize) {
161         callback.batch(bytes())
162       } else if (skip >= batchSize) {
163         skip = 0
164         callback.partial(bytes())
165       } else {
166         true
167       }
168     }
169 
170     def done(): Unit = {
171       if (count > 0) {
172         callback.batch(bytes())
173       }
174       result.cleanup()
175     }
176 
177     // noinspection LanguageFeature
178     private def bytes(): Array[Byte] = {
179       count = 0
180       skip = 0
181       result.encode()
182     }
183   }
