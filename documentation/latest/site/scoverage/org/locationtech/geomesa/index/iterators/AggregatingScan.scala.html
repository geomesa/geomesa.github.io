40   override def init(options: Map[String, String]): Unit = {
41     val spec = options(SftOpt)
42     val sft = IteratorCache.sft(spec)
43     // note: index won't have a hook to the data store, so some operations aren't available
44     val index = options.get(IndexSftOpt) match {
45       case None => IteratorCache.index(sft, spec, options(IndexOpt))
46       case Some(ispec) => IteratorCache.index(IteratorCache.sft(ispec), ispec, options(IndexOpt))
47     }
48 
49     reusableSf = IteratorCache.serializer(spec, index.serializedWithId).getReusableFeature
50     reusableSf.setIdParser(index.getIdFromRow(_, _, _, null))
51     aggregateSf = reusableSf
52 
53     val transformSft = options.get(TransformDefsOpt).map { t =>
54       val ts = options.getOrElse(TransformSchemaOpt,
55         throw new IllegalArgumentException("Defined a transform but no transform schema"))
56       val transformSft = IteratorCache.sft(ts)
57       val reusableTransformSf = TransformSimpleFeature(IteratorCache.sft(spec), transformSft, t)
58       reusableTransformSf.setFeature(reusableSf)
59       aggregateSf = reusableTransformSf // note: side-effect in map
60       transformSft
61     }
62     val sampling = sample(options)
63     val cql = options.get(CqlOpt).map(IteratorCache.filter(sft, spec, _))
64     validate = (cql, sampling) match {
65       case (None, None)             => ValidateAll
66       case (Some(filt), None)       => new CqlValidator(filt)
67       case (None, Some(samp))       => new SampleValidator(samp)
68       case (Some(filt), Some(samp)) => new CqlSampleValidator(filt,  samp)
69     }
70     batchSize = options.get(BatchSizeOpt).map(_.toInt).getOrElse(defaultBatchSize)
71     result = createResult(sft, transformSft, batchSize, options)
72   }

74   /**
75    * Aggregates a batch of data. May not exhaust the underlying data
76    *
77    * @param callback callback to provide for results
78    * @return callback
79    */
80   def aggregate[A <: AggregateCallback](callback: A): A = {
81     // noinspection LanguageFeature
82     result.init()
83 
84     val status = new AggregateStatus(callback)
85 
86     var rowValue = try { if (hasNextData) { nextData() } else { null } } catch {
87       case NonFatal(e) => logger.error("Error in underlying scan while aggregating value:", e); null
88     }
89 
90     while (rowValue != null) {
91       try {
92         reusableSf.setIdBuffer(rowValue.row, rowValue.rowOffset, rowValue.rowLength)
93         reusableSf.setBuffer(rowValue.value, rowValue.valueOffset, rowValue.valueLength)
94         if (validate(reusableSf)) {
95           // write the record to our aggregated results
96           status.aggregated(result.aggregate(aggregateSf))
97         } else {
98           status.skipped()
99         }
100       } catch {
101         case NonFatal(e) =>
102           logger.error(s"Error aggregating value for ${debugSf()}:", e)
103           status.skipped()
104       }
105 
106       rowValue = try { if (status.continue() && hasNextData) { nextData() } else { null } } catch {
107         case NonFatal(e) => logger.error("Error in underlying scan while aggregating value:", e); null
108       }
109     }
110 
111     status.done()
112 
113     callback
114   }

141   /**
142    * Class for tracking status of current aggregation
143    *
144    * @param callback results callback
145    */
146   private class AggregateStatus(callback: AggregateCallback) {
147 
148     private var count = 0
149     private var skip = 0
150 
151     def aggregated(count: Int): Unit = this.count += count
152 
153     def skipped(): Unit = skip += 1
154 
155     // noinspection LanguageFeature
156     def continue(): Boolean = {
157       if (count >= batchSize) {
158         callback.batch(bytes())
159       } else if (skip >= batchSize) {
160         skip = 0
161         callback.partial(bytes())
162       } else {
163         true
164       }
165     }
166 
167     def done(): Unit = {
168       if (count > 0) {
169         callback.batch(bytes())
170       }
171       result.cleanup()
172     }
173 
174     // noinspection LanguageFeature
175     private def bytes(): Array[Byte] = {
176       count = 0
177       skip = 0
178       result.encode()
179     }
180   }
