38     * @param explain explain output

41   def runQuery(

53   /**
54     * Configure the query - set hints, transforms, etc.
55     *
56     * @param original query to configure
57     * @param sft simple feature type associated with the query
58     */
59   protected[geomesa] def configureQuery(sft: SimpleFeatureType, original: Query): Query = {
60     import org.locationtech.geomesa.filter.{andFilters, ff, orFilters}
61     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
62     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
63 
64     val query = new Query(original)
65 
66     // query rewriting
67     interceptors(sft).foreach { interceptor =>
68       interceptor.rewrite(query)
69       QueryRunner.logger.trace(s"Query rewritten by $interceptor to: $query")
70     }
71 
72     // handle any params passed in through geoserver
73     ViewParams.setHints(query)
74 
75     // set transformations in the query
76     QueryPlanner.setQueryTransforms(sft, query)
77     // set return SFT in the query
78     query.getHints.put(QueryHints.Internal.RETURN_SFT, getReturnSft(sft, query.getHints))
79 
80     // set sorting in the query
81     QueryPlanner.setQuerySort(sft, query)
82     QueryPlanner.setProjection(sft, query)
83     QueryPlanner.setMaxFeatures(query)
84 
85     // add the bbox from the density query to the filter, if there is no more restrictive filter
86     query.getHints.getDensityEnvelope.foreach { env =>
87       val geom = query.getHints.getDensityGeometry.getOrElse(sft.getGeomField)
88       val geoms = FilterHelper.extractGeometries(query.getFilter, geom)
89       if (geoms.isEmpty || geoms.exists(g => !env.contains(g.getEnvelopeInternal))) {
90         val split = GeometryUtils.splitBoundingBox(env.asInstanceOf[ReferencedEnvelope])
91         val bbox = orFilters(split.map(ff.bbox(ff.property(geom), _)))
92         if (query.getFilter == Filter.INCLUDE) {
93           query.setFilter(bbox)
94         } else {
95           query.setFilter(andFilters(Seq(query.getFilter, bbox)))
96         }
97       }
98     }
99 
100     if (query.getFilter != null && query.getFilter != Filter.INCLUDE) {
101       // bind the literal values to the appropriate type, so that it isn't done every time the filter is evaluated
102       // update the filter to remove namespaces, handle null property names, and tweak topological filters
103       // replace filters with 'fast' implementations where possible
104       query.setFilter(FastFilterFactory.optimize(sft, query.getFilter))
105     }
106 
107     query
108   }
109 
110   /**
111     * Provides the simple feature type that will be returned from a query. Extension point for subclasses
112     *
113     * @param sft simple feature type
114     * @param hints query hints, which have been configured using @see configureQuery
115     * @return simple feature that will be returned from a query using the hints
116     */
117   protected[geomesa] def getReturnSft(sft: SimpleFeatureType, hints: Hints): SimpleFeatureType = {
118     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
119     if (hints.isBinQuery) {
120       BinaryOutputEncoder.BinEncodedSft
121     } else if (hints.isArrowQuery) {
122       org.locationtech.geomesa.arrow.ArrowEncodedSft
123     } else if (hints.isDensityQuery) {
124       DensityScan.DensitySft
125     } else if (hints.isStatsQuery) {
126       StatsScan.StatsSft
127     } else {
128       hints.getTransformSchema.getOrElse(sft)
129     }
130   }
131 }
