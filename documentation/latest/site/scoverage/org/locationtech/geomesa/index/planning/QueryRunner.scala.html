44       explain: Explainer = new ExplainLogging): QueryResult

53   /**
54     * Configure the query - set hints, transforms, etc.
55     *
56     * @param original query to configure
57     * @param sft simple feature type associated with the query
58     */
59   protected[geomesa] def configureQuery(sft: SimpleFeatureType, original: Query): Query = {
60     import org.locationtech.geomesa.filter.{andFilters, ff, orFilters}
61     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
62     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
63 
64     val query = new Query(original)
65 
66     // query rewriting
67     interceptors(sft).foreach { interceptor =>
68       interceptor.rewrite(query)
69       QueryRunner.logger.trace(s"Query rewritten by $interceptor to: $query")
70     }
71 
72     // set query hints - we need this in certain situations where we don't have access to the query directly
73     QueryPlanner.threadedHints.get.foreach { hints =>
74       hints.foreach { case (k, v) => query.getHints.put(k, v) }
75       // clear any configured hints so we don't process them again
76       QueryPlanner.threadedHints.clear()
77     }
78 
79     // handle any params passed in through geoserver
80     ViewParams.setHints(query)
81 
82     // set transformations in the query
83     QueryPlanner.setQueryTransforms(sft, query)
84     // set return SFT in the query
85     query.getHints.put(QueryHints.Internal.RETURN_SFT, getReturnSft(sft, query.getHints))
86 
87     // set sorting in the query
88     QueryPlanner.setQuerySort(sft, query)
89     QueryPlanner.setProjection(sft, query)
90     QueryPlanner.setMaxFeatures(query)
91 
92     // add the bbox from the density query to the filter, if there is no more restrictive filter
93     query.getHints.getDensityEnvelope.foreach { env =>
94       val geom = query.getHints.getDensityGeometry.getOrElse(sft.getGeomField)
95       val geoms = FilterHelper.extractGeometries(query.getFilter, geom)
96       if (geoms.isEmpty || geoms.exists(g => !env.contains(g.getEnvelopeInternal))) {
97         val split = GeometryUtils.splitBoundingBox(env.asInstanceOf[ReferencedEnvelope])
98         val bbox = orFilters(split.map(ff.bbox(ff.property(geom), _)))
99         if (query.getFilter == Filter.INCLUDE) {
100           query.setFilter(bbox)
101         } else {
102           query.setFilter(andFilters(Seq(query.getFilter, bbox)))
103         }
104       }
105     }
106 
107     if (query.getFilter != null && query.getFilter != Filter.INCLUDE) {
108       // bind the literal values to the appropriate type, so that it isn't done every time the filter is evaluated
109       // update the filter to remove namespaces, handle null property names, and tweak topological filters
110       // replace filters with 'fast' implementations where possible
111       query.setFilter(FastFilterFactory.optimize(sft, query.getFilter))
112     }
113 
114     query
115   }
116 
117   /**
118     * Provides the simple feature type that will be returned from a query. Extension point for subclasses
119     *
120     * @param sft simple feature type
121     * @param hints query hints, which have been configured using @see configureQuery
122     * @return simple feature that will be returned from a query using the hints
123     */
124   protected[geomesa] def getReturnSft(sft: SimpleFeatureType, hints: Hints): SimpleFeatureType = {
125     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
126     if (hints.isBinQuery) {
127       BinaryOutputEncoder.BinEncodedSft
128     } else if (hints.isArrowQuery) {
129       org.locationtech.geomesa.arrow.ArrowEncodedSft
130     } else if (hints.isDensityQuery) {
131       DensityScan.DensitySft
132     } else if (hints.isStatsQuery) {
133       StatsScan.StatsSft
134     } else {
135       hints.getTransformSchema.getOrElse(sft)
136     }
137   }
138 }
