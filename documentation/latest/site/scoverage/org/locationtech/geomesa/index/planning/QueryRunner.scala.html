45   /**
46     * Execute a query
47     *
48     * @param sft simple feature type
49     * @param original query to run
50     * @param explain explain output
51     * @return
52     */
53   def query(sft: SimpleFeatureType, original: Query, explain: Explainer = new ExplainLogging): QueryResult = {
54     val start = System.nanoTime()
55     val query = configureQuery(sft, original)
56     val hints = query.getHints
57 
58     explain.pushLevel(s"Planning '${query.getTypeName}' ${FilterHelper.toString(query.getFilter)}")
59     explain(s"Original filter: ${FilterHelper.toString(original.getFilter)}")
60     explain(s"Hints: bin[${hints.isBinQuery}] arrow[${hints.isArrowQuery}] density[${hints.isDensityQuery}] " +
61       s"stats[${hints.isStatsQuery}] " +
62       s"sampling[${hints.getSampling.map { case (s, f) => s"$s${f.map(":" + _).getOrElse("")}"}.getOrElse("none")}]")
63     explain(s"Sort: ${hints.getSortFields.map(QueryHints.sortReadableString).getOrElse("none")}")
64     explain(s"Transforms: ${hints.getTransformDefinition.map(t => if (t.isEmpty) { "empty" } else { t }).getOrElse("none")}")
65     explain(s"Max features: ${hints.getMaxFeatures.getOrElse("none")}")
66     hints.getFilterCompatibility.foreach(c => explain(s"Filter compatibility: $c"))
67 
68     val plans = getQueryPlans(sft, query, explain)
69 
70     // ensure we're not making incorrect assumptions by only looking at the head plan
71     sanityCheck(plans, _.sort, "Sort")
72     sanityCheck(plans, _.maxFeatures, "Max features")
73     sanityCheck(plans, _.projection, "Projection")
74     sanityCheck(plans, _.reducer, "Reduce")
75 
76     // processing steps we apply to the scan results
77     val steps = plans.headOption.toSeq.flatMap { plan =>
78       val sort: Option[QueryStep] = plan.sort.map(s => new SortingSimpleFeatureIterator(_, s))
79       val limit: Option[QueryStep] = plan.maxFeatures.map { max =>
80         if (hints.getReturnSft == org.locationtech.geomesa.arrow.ArrowEncodedSft) {
81           // TODO handle arrow queries
82           _.take(max)
83         } else if (hints.getReturnSft == BinaryOutputEncoder.BinEncodedSft) {
84           // bin queries pack multiple records into each feature
85           // to count the records, we have to count the total bytes coming back, instead of the number of features
86           new BinaryOutputEncoder.FeatureLimitingIterator(_, max, hints.getBinLabelField.isDefined)
87         } else {
88           _.take(max)
89         }
90       }
91       val reproject: Option[QueryStep] = plan.projection.map(Reprojection(hints.getReturnSft, _)).map(r => _.map(r.apply))
92       val reduce: Option[QueryStep] = plan.reducer.filterNot(_ => hints.isSkipReduce).map(r => r.apply)
93 
94       sort ++ limit ++ reproject ++ reduce
95     }
96 
97     val timer: FinalQueryStep = new TimedIterator(_, Timers.scanning(query.getTypeName).record(_, TimeUnit.NANOSECONDS))
98 
99     val planCreateTime = System.nanoTime() - start
100     Timers.planning(query.getTypeName).record(planCreateTime, TimeUnit.NANOSECONDS)
101     explain(s"Query planning took ${planCreateTime / 1000000L}ms")
102 
103     new QueryResult(hints.getReturnSft, hints, plans, planCreateTime, steps, timer)
104   }

136     * Configure the query - set hints, transforms, etc.
137     *
138     * @param original query to configure
139     * @param sft simple feature type associated with the query
140     */
141   protected[geomesa] def configureQuery(sft: SimpleFeatureType, original: Query): Query = {
142     import org.locationtech.geomesa.filter.{andFilters, ff, orFilters}
143     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
144 
145     val query = new Query(original)
146 
147     // query rewriting
148     interceptors(sft).foreach { interceptor =>
149       interceptor.rewrite(query)
150       QueryRunner.logger.trace(s"Query rewritten by $interceptor to: $query")
151     }
152 
153     // handle any params passed in through geoserver
154     ViewParams.setHints(query)
155 
156     // handle transforms and store them in the query hints
157     extractQueryTransforms(sft, query).foreach { case (schema, _, transforms) =>
158       query.getHints.put(QueryHints.Internal.TRANSFORMS, transforms)
159       query.getHints.put(QueryHints.Internal.TRANSFORM_SCHEMA, schema)
160     }
161 
162     // set the return schema
163     query.getHints.put(QueryHints.Internal.RETURN_SFT,
164       if (query.getHints.isBinQuery) {
165         BinaryOutputEncoder.BinEncodedSft
166       } else if (query.getHints.isArrowQuery) {
167         org.locationtech.geomesa.arrow.ArrowEncodedSft
168       } else if (query.getHints.isDensityQuery) {
169         DensityScan.DensitySft
170       } else if (query.getHints.isStatsQuery) {
171         StatsScan.StatsSft
172       } else {
173         query.getHints.getTransformSchema.getOrElse(sft)
174       }
175     )
176 
177     QueryRunner.setQuerySort(sft, query)
178 
179     // set any reprojection into the hints
180     QueryReferenceSystems(query).foreach { crs =>
181       query.getHints.put(QueryHints.Internal.REPROJECTION, QueryHints.Internal.toProjectionHint(crs))
182     }
183 
184     // set max features into the hints
185     if (!query.isMaxFeaturesUnlimited) {
186       query.getHints.put(QueryHints.Internal.MAX_FEATURES, Int.box(query.getMaxFeatures))
187     }
188 
189     // add the bbox from the density query to the filter, if there is no more restrictive filter
190     query.getHints.getDensityEnvelope.foreach { env =>
191       val geom = query.getHints.getDensityGeometry.getOrElse(sft.getGeomField)
192       val geoms = FilterHelper.extractGeometries(query.getFilter, geom)
193       if (geoms.isEmpty || geoms.exists(g => !env.contains(g.getEnvelopeInternal))) {
194         val split = GeometryUtils.splitBoundingBox(env.asInstanceOf[ReferencedEnvelope])
195         val bbox = orFilters(split.map(ff.bbox(ff.property(geom), _)))
196         if (query.getFilter == Filter.INCLUDE) {
197           query.setFilter(bbox)
198         } else {
199           query.setFilter(andFilters(Seq(query.getFilter, bbox)))
200         }
201       }
202     }
203 
204     // optimize the filter
205     if (query.getFilter != null && query.getFilter != Filter.INCLUDE) {
206       // bind the literal values to the appropriate type, so that it isn't done every time the filter is evaluated
207       // update the filter to remove namespaces, handle null property names, and tweak topological filters
208       // replace filters with 'fast' implementations where possible
209       query.setFilter(FastFilterFactory.optimize(sft, query.getFilter))
210     }
211 
212     query
213   }

278   /**
279    * Extract and parse transforms from the query
280    *
281    * @param sft simple feature type
282    * @param query query
283    * @return
284    */
285   private def extractQueryTransforms(sft: SimpleFeatureType, query: Query): Option[(SimpleFeatureType, Seq[Transform], String)] = {
286     // even if a transform is not specified, some queries only use a subset of attributes
287     // specify them here so that it's easier to pick the best column group later
288     def fromQueryType: Option[Seq[String]] = {
289       val hints = query.getHints
290       if (hints.isBinQuery) {
291         Some(BinAggregatingScan.propertyNames(hints, sft))
292       } else if (hints.isDensityQuery) {
293         Some(DensityScan.propertyNames(hints, sft))
294       } else if (hints.isStatsQuery) {
295         Some(StatParser.propertyNames(sft, hints.getStatsQuery))
296       } else {
297         None
298       }
299     }
300 
301     // since we do sorting on the client, just add any sort-by attributes to the transform
302     // TODO GEOMESA-2655 we should sort and then transform back to the requested props, but it's complicated...
303     def withSort(props: Array[String]): Seq[String] = {
304       val names = props.map(p => if (p.contains('=')) { p.substring(0, p.indexOf('=')) } else { p })
305       props ++ Option(query.getSortBy).toSeq.flatMap { sort =>
306         sort.flatMap(s => Option(s.getPropertyName).flatMap(p => Option(p.getPropertyName).filterNot(names.contains)))
307       }
308     }
309 
310     // ignore transforms that don't actually do anything
311     def noop(props: Seq[String]): Boolean = props == sft.getAttributeDescriptors.asScala.map(_.getLocalName)
312 
313     Option(query.getPropertyNames).map(withSort).filterNot(noop).orElse(fromQueryType).map { props =>
314       val transforms = Transforms(sft, props)
315       val schema = Transforms.schema(sft, transforms)
316       val definition = props.mkString(Transform.DefinitionDelimiter)
317       (schema, transforms, definition)
318     }
319   }

321   /**
322    * Sets query hints for sorting
323    *
324    * @param sft sft
325    * @param query query
326    */
327   private def setQuerySort(sft: SimpleFeatureType, query: Query): Unit = {
328     val sortBy = query.getSortBy
329     if (sortBy != null && sortBy.nonEmpty) {
330       val hints = query.getHints
331       if (hints.isArrowQuery) {
332         if (sortBy.lengthCompare(1) > 0) {
333           throw new IllegalArgumentException("Arrow queries only support sort by a single field: " +
334             sortBy.mkString(", "))
335         } else if (sortBy.head.getPropertyName == null) {
336           throw new IllegalArgumentException("Arrow queries only support sort by properties: " +
337             sortBy.mkString(", "))
338         }
339         val field = sortBy.head.getPropertyName.getPropertyName
340         val reverse = sortBy.head.getSortOrder == SortOrder.DESCENDING
341         hints.getArrowSort.foreach { case (f, r) =>
342           if (f != field || r != reverse) {
343             throw new IllegalArgumentException(s"Query sort does not match Arrow hints sort: " +
344               s"${sortBy.mkString(", ")} != $f:${if (r) "DESC" else "ASC"}")
345           }
346         }
347         hints.put(QueryHints.ARROW_SORT_FIELD, field)
348         hints.put(QueryHints.ARROW_SORT_REVERSE, reverse)
349       } else if (hints.isBinQuery) {
350         val dtg = hints.getBinDtgField.orElse(sft.getDtgField).orNull
351         if (dtg == null ||
352           sortBy.map(s => Option(s.getPropertyName).map(_.getPropertyName).orNull).toSeq != Seq(dtg)) {
353           throw new IllegalArgumentException("BIN queries only support sort by a date-type field: " +
354             sortBy.mkString(", "))
355         }
356         if (sortBy.head.getSortOrder == SortOrder.DESCENDING) {
357           throw new IllegalArgumentException("BIN queries only support sort in ASCENDING order: " +
358             sortBy.mkString(", "))
359         }
360         if (hints.get(QueryHints.BIN_SORT) != null && !hints.isBinSorting) {
361           throw new IllegalArgumentException("Query sort order contradicts BIN sorting hint: " +
362             sortBy.mkString(", "))
363         }
364         hints.put(QueryHints.BIN_SORT, java.lang.Boolean.TRUE)
365       } else {
366         hints.put(QueryHints.Internal.SORT_FIELDS, QueryHints.Internal.toSortHint(sortBy))
367       }
368     }
369   }

371   /**
372    * Holds all the necessary pieces to actually run a query, but doesn't run it yet
373    *
374    * @param schema return schema of the iterators returned by this class
375    * @param hints query hints
376    * @param plans query plans for executing the query
377    * @param planTimeNanos time spent planning the query
378    * @param querySteps processing steps to apply on top of the query plans
379    * @param timer final step to apply, creating a timed iterator for metrics tracking
380    */
381   class QueryResult(
382       val schema: SimpleFeatureType,
383       val hints: Hints,
384       val plans: Seq[QueryPlan],
385       val planTimeNanos: Long,
386       querySteps: Seq[QueryStep],
387       timer: FinalQueryStep) {
388 
389     /**
390      * Executes the query
391      *
392      * @return
393      */
394     def iterator(): CloseableIterator[SimpleFeature] = ExceptionalIterator(runQuery())
395 
396     /**
397      * Executes the query
398      *
399      * @param timerCallback callback for time spent running the query, will be invoked when the iterator is closed with the time in nanoseconds
400      * @param counterCallback callback for the number of results returned by the query, will be invoked when the iterator is closed
401      * @return
402      */
403     def iterator(timerCallback: Long => Unit, counterCallback: Long => Unit): CloseableIterator[SimpleFeature] = {
404       val counter: SimpleFeature => Int =
405         if (schema == org.locationtech.geomesa.arrow.ArrowEncodedSft) {
406           // TODO handle arrow queries
407           StandardCounter
408         } else if (schema == BinaryOutputEncoder.BinEncodedSft) {
409           BinAggregatingScan.BinCounter(hints)
410         } else {
411           StandardCounter
412         }
413       ExceptionalIterator(new CountingIterator(runQuery().addCallback(timerCallback), counter, counterCallback))
414     }
415 
416     private def runQuery(): TimedIterator[SimpleFeature] = {
417       val iter = CloseableIterator(plans.iterator).flatMap(p => p.scan().map(p.resultsToFeatures.apply))
418       timer(querySteps.foldLeft(iter) { case (i, step) => step(i) })
419     }
420 
421     override def toString: String =
422       s"QueryResult(schema=${SimpleFeatureTypes.encodeType(schema)},hints=${gson.toJson(ViewParams.getReadableHints(hints))})"
423   }
424 
425   private object StandardCounter extends (SimpleFeature => Int) {
426     override def apply(f: SimpleFeature): Int = 1
427   }
