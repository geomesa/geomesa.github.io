33   /**
34     * Execute a query
35     *
36     * @param sft simple feature type
37     * @param query query to run
38     * @param explain explain output
39     * @return
40     */
41   def runQuery(
42       sft: SimpleFeatureType,
43       query: Query,
44       explain: Explainer = new ExplainLogging): QueryResult
45 
46   /**
47     * Hook for query interceptors
48     *
49     * @return
50     */
51   protected def interceptors: QueryInterceptorFactory
52 
53   /**
54     * Configure the query - set hints, transforms, etc.
55     *
56     * @param original query to configure
57     * @param sft simple feature type associated with the query
58     */
59   protected[geomesa] def configureQuery(sft: SimpleFeatureType, original: Query): Query = {
60     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
61     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
62 
63     val query = new Query(original)
64 
65     // query rewriting
66     interceptors(sft).foreach { interceptor =>
67       interceptor.rewrite(query)
68       QueryRunner.logger.trace(s"Query rewritten by $interceptor to: $query")
69     }
70 
71     // set query hints - we need this in certain situations where we don't have access to the query directly
72     QueryPlanner.threadedHints.get.foreach { hints =>
73       hints.foreach { case (k, v) => query.getHints.put(k, v) }
74       // clear any configured hints so we don't process them again
75       QueryPlanner.threadedHints.clear()
76     }
77 
78     // handle any params passed in through geoserver
79     ViewParams.setHints(query)
80 
81     // set transformations in the query
82     QueryPlanner.setQueryTransforms(sft, query)
83     // set return SFT in the query
84     query.getHints.put(QueryHints.Internal.RETURN_SFT, getReturnSft(sft, query.getHints))
85 
86     // set sorting in the query
87     QueryPlanner.setQuerySort(sft, query)
88     QueryPlanner.setProjection(sft, query)
89     QueryPlanner.setMaxFeatures(query)
90 
91     // add the bbox from the density query to the filter, if there is no more restrictive filter
92     query.getHints.getDensityEnvelope.foreach { env =>
93       val geom = query.getHints.getDensityGeometry.getOrElse(sft.getGeomField)
94       val geoms = FilterHelper.extractGeometries(query.getFilter, geom)
95       if (geoms.isEmpty || geoms.exists(g => !env.contains(g.getEnvelopeInternal))) {
96         val split = GeometryUtils.splitBoundingBox(env.asInstanceOf[ReferencedEnvelope])
97         val bbox = orFilters(split.map(ff.bbox(ff.property(geom), _)))
98         if (query.getFilter == Filter.INCLUDE) {
99           query.setFilter(bbox)
100         } else {
101           query.setFilter(andFilters(Seq(query.getFilter, bbox)))
102         }
103       }
104     }
105 
106     if (query.getFilter != null && query.getFilter != Filter.INCLUDE) {
107       // bind the literal values to the appropriate type, so that it isn't done every time the filter is evaluated
108       // update the filter to remove namespaces, handle null property names, and tweak topological filters
109       // replace filters with 'fast' implementations where possible
110       query.setFilter(FastFilterFactory.optimize(sft, query.getFilter))
111     }
112 
113     query
114   }
115 
116   /**
117     * Provides the simple feature type that will be returned from a query. Extension point for subclasses
118     *
119     * @param sft simple feature type
120     * @param hints query hints, which have been configured using @see configureQuery
121     * @return simple feature that will be returned from a query using the hints
122     */
123   protected [geomesa] def getReturnSft(sft: SimpleFeatureType, hints: Hints): SimpleFeatureType = {
124     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
125     if (hints.isBinQuery) {
126       BinaryOutputEncoder.BinEncodedSft
127     } else if (hints.isArrowQuery) {
128       org.locationtech.geomesa.arrow.ArrowEncodedSft
129     } else if (hints.isDensityQuery) {
130       DensityScan.DensitySft
131     } else if (hints.isStatsQuery) {
132       StatsScan.StatsSft
133     } else {
134       hints.getTransformSchema.getOrElse(sft)
135     }
136   }
137 }
