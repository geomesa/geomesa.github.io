40   override def getSplits(sft: SimpleFeatureType,
41                          index: String,
42                          partition: String,
43                          options: String): Array[Array[Byte]] = {
44     val splits = Try(IndexId.id(index)).toOption.flatMap { id =>
45       val opts = Option(options).map(KVPairParser.parse).getOrElse(Map.empty)
46       id.name match {
47         case IdIndex.name                 => Some(idBytes(opts))
48         case Z3Index.name | XZ3Index.name => Some(z3Bytes(sft, Option(partition), opts))
49         case Z2Index.name | XZ2Index.name => Some(z2Bytes(opts))
50         case AttributeIndex.name          => Some(attributeBytes(sft, id.attributes.head, opts))
51         case AttributeIndex.JoinIndexName => Some(attributeBytes(sft, id.attributes.head, opts))
52         case _ => None
53       }
54     }
55     splits.getOrElse { logger.warn(s"Unhandled index type $index"); Array(Array.empty[Byte]) }
56   }

92     /**
93       * Creates splits suitable for an attribute index. Each indexed attribute can be split individually.
94       *
95       *  Options are:
96       *
97       * * 'attr.&lt;attribute&gt;.pattern' - pattern used for a given attribute
98       *
99       * Additional patterns can be specified with 'attr.&lt;attribute&gt;.pattern2', etc
100       *
101       * @param name attribute name
102       * @param binding attribute type
103       * @param options user data containing split configuration
104       * @return
105       */
106     def attributeSplits(name: String, binding: Class[_], options: Map[String, String]): Seq[String] = {
107       val patterns = DefaultSplitter.patternPairs(s"${AttributeIndex.name}.$name.", "pattern", "date-range", options)
108       val ranges = patterns.map { case (pattern, datePattern) =>
109         (SplitPatternParser.parse(pattern), datePattern.map(SplitPatternParser.parse))
110       }
111       val getSplits: SplitPattern => Seq[String] =
112         if (classOf[Number].isAssignableFrom(binding)) {
113           pattern =>
114             try { numberPatternSplits(pattern, binding) } catch {
115               case e: NumberFormatException =>
116                 throw new IllegalArgumentException(s"Trying to create splits for attribute '$name' " +
117                     s"of type ${binding.getName}, but splits could not be parsed as a number: " +
118                     patterns.mkString(" "), e)
119             }
120         } else {
121           pattern => pattern.range
122         }
123       ranges.flatMap { case (primary, secondary) =>
124         val splits = getSplits(primary)
125         secondary match {
126           case None => splits
127           case Some(s) => for { a <- splits; b <- s.range } yield { a + ZeroByteString + b }
128         }
129       }
130     }

132     /**
133       * Creates splits suitable for a z3 index. Generally, a split will be created per time interval (e.g. week).
134       * Further splits can be created by specifying the number of bits that will be used for splits, per week.
135       * The total number of splits is: (number of time intervals) * 2^^(number of bits)
136       *
137       *  Options are:
138       *
139       * * 'z3.min' - min date for data
140       * * 'z3.max' - max date for data (must also include min date) - will default to current date if not specified
141       * * 'z3.bits' - number of bits used to create splits. e.g. 2 bits will create 4 splits.
142       *
143       * @param interval interval of the z3 index
144       * @param options user data containing split configuration
145       * @return
146       */
147     def z3Splits(interval: TimePeriod, options: Map[String, String]): Seq[(Short, Option[Long])] = {
148 
149       def date(key: String): Option[Date] = {
150         options.get(key).map { d =>
151           val converted = FastConverter.convert(d, classOf[Date])
152           if (converted == null) {
153             throw new IllegalArgumentException(s"Could not convert date '$d' for splits")
154           }
155           converted
156         }
157       }
158 
159       date(Z3MinDateOption).map(_.getTime) match {
160         case None => Seq.empty
161         case Some(min) =>
162           val max = date(Z3MaxDateOption).map(_.getTime).getOrElse(System.currentTimeMillis())
163 
164           val toBin = BinnedTime.timeToBinnedTime(interval)
165           val minBin = toBin(min).bin
166           val maxBin = toBin(max).bin
167           val times = Seq.range(minBin, maxBin + 1).map(_.toShort)
168 
169           val zs = z3BitSplits(options)
170           if (zs.isEmpty) {
171             times.map(t => (t, None))
172           } else {
173             times.flatMap(t => zs.map(z => (t, Some(z))))
174           }
175       }
176     }

247   private def patterns(base: String, options: Map[String, String]): Seq[String] = {
248     val keys = Iterator.single(base) ++ Iterator.range(2, Int.MaxValue).map(i => s"$base$i")

252   private def patternPairs(
253       prefix: String,
254       primary: String,
255       secondary: String,
256       options: Map[String, String]): Seq[(String, Option[String])] = {
257     val firstKey = s"$prefix$primary"
258     val secondKey = s"$prefix$secondary"
259     val transforms: Iterator[String => String] =
260       Iterator.single[String => String](b => b) ++ Iterator.range(2, Int.MaxValue).map(i => b => s"$b$i")
261     val patterns = transforms.map { transform =>
262       options.get(transform(firstKey)) match {
263         case None => null
264         case Some(pattern) => (pattern, options.get(transform(secondKey)))
265       }
266     }
267     patterns.takeWhile(_ != null).toSeq
268   }

279   /**
280     * Create splits based on individual bits. Will create 2^^(bits) splits.
281     *
282     * @param key key for number of bits to consider
283     * @param options config options
284     * @param maskedBits number of unused bits at the start of the first byte
285     * @return
286     */
287   private def bitSplits(key: String, options: Map[String, String], maskedBits: Int): Seq[Long] = {
288     options.get(key).map(_.toInt) match {
289       case None => Seq.empty
290       case Some(bits) =>
291         require(bits > 0 && bits < 64, "Bit split must be between 1 and 63")
292 
293         // recursive function to create all bit permutations
294         def add(result: Seq[Seq[Int]], bits: Seq[Int], remaining: Int): Seq[Seq[Int]] = {
295           if (remaining == 0) { Seq(bits) } else {
296             val zero = add(result, bits :+ 0, remaining - 1)
297             val one = add(result, bits :+ 1, remaining - 1)
298             result ++ zero ++ one
299           }
300         }
301         // note: first bit(s) in z value are not used, and are always 0
302         def toLong(bits: Seq[Int]): Long = {
303           val binaryString = (Seq.fill(maskedBits)(0) ++ bits).padTo(64, 0).mkString("")
304           java.lang.Long.parseLong(binaryString, 2)
305         }
306 
307         add(Seq.empty, Seq.empty, bits).map(toLong)
308     }
