43                          index: String,
44                          partition: String,
45                          options: String): Array[Array[Byte]] = {
46     val splits = Try(IndexId.id(index)).toOption.flatMap { id =>
47       val opts = Option(options).map(KVPairParser.parse).getOrElse(Map.empty)
48       id.name match {

50         case Z3Index.name | XZ3Index.name        => Some(z3Bytes(sft, Option(partition), opts))
51         case Z2Index.name | XZ2Index.name        => Some(z2Bytes(opts))
52         case AttributeIndex.name | JoinIndexName => Some(attributeBytes(sft, id.attributes.head, Option(partition), opts))
53         case _ => None
54       }
55     }
56     splits.getOrElse { logger.warn(s"Unhandled index type $index"); Array(Array.empty[Byte]) }
57   }

93     /**
94       * Creates splits suitable for an attribute index. Each indexed attribute can be split individually.
95       *
96       *  Options are:
97       *
98       * * 'attr.&lt;attribute&gt;.pattern' - pattern used for a given attribute
99       *
100       * Additional patterns can be specified with 'attr.&lt;attribute&gt;.pattern2', etc
101       *
102       * @param name attribute name
103       * @param binding attribute type
104       * @param options user data containing split configuration
105       * @return
106       */
107     def attributeSplits(name: String, binding: Class[_], options: Map[String, String]): Seq[String] = {
108       val patterns = DefaultSplitter.patternPairs(s"${AttributeIndex.name}.$name.", "pattern", "date-range", options)
109       val ranges = patterns.map { case (pattern, datePattern) =>
110         (SplitPatternParser.parse(pattern), datePattern.map(SplitPatternParser.parse))
111       }
112       val getSplits: SplitPattern => Seq[String] =
113         if (classOf[Number].isAssignableFrom(binding)) {
114           pattern =>
115             try { numberPatternSplits(pattern, binding) } catch {
116               case e: NumberFormatException =>
117                 throw new IllegalArgumentException(s"Trying to create splits for attribute '$name' " +
118                     s"of type ${binding.getName}, but splits could not be parsed as a number: " +
119                     patterns.mkString(" "), e)
120             }
121         } else {
122           pattern => pattern.range
123         }
124       ranges.flatMap { case (primary, secondary) =>
125         val splits = getSplits(primary)
126         secondary match {
127           case None => splits
128           case Some(s) => for { a <- splits; b <- s.range } yield { a + ZeroByteString + b }
129         }
130       }
131     }

133     /**
134       * Creates splits suitable for a z3 index. Generally, a split will be created per time interval (e.g. week).
135       * Further splits can be created by specifying the number of bits that will be used for splits, per week.
136       * The total number of splits is: (number of time intervals) * 2^^(number of bits)
137       *
138       *  Options are:
139       *
140       * * 'z3.min' - min date for data
141       * * 'z3.max' - max date for data (must also include min date) - will default to current date if not specified
142       * * 'z3.bits' - number of bits used to create splits. e.g. 2 bits will create 4 splits.
143       *
144       * @param interval interval of the z3 index
145       * @param options user data containing split configuration
146       * @return
147       */
148     def z3Splits(interval: TimePeriod, options: Map[String, String]): Seq[(Short, Option[Long])] = {
149 
150       def date(key: String): Option[Date] = {
151         options.get(key).map { d =>
152           val converted = FastConverter.convert(d, classOf[Date])
153           if (converted == null) {
154             throw new IllegalArgumentException(s"Could not convert date '$d' for splits")
155           }
156           converted
157         }
158       }
159 
160       date(Z3MinDateOption).map(_.getTime) match {
161         case None => Seq.empty
162         case Some(min) =>
163           val max = date(Z3MaxDateOption).map(_.getTime).getOrElse(System.currentTimeMillis())
164 
165           val toBin = BinnedTime.timeToBinnedTime(interval)
166           val minBin = toBin(min).bin
167           val maxBin = toBin(max).bin
168           val times = Seq.range(minBin, maxBin + 1).map(_.toShort)
169 
170           val zs = z3BitSplits(options)
171           if (zs.isEmpty) {
172             times.map(t => (t, None))
173           } else {
174             times.flatMap(t => zs.map(z => (t, Some(z))))
175           }
176       }
177     }

287   /**
288     * Create splits based on individual bits. Will create 2^^(bits) splits.
289     *
290     * @param key key for number of bits to consider
291     * @param options config options
292     * @param maskedBits number of unused bits at the start of the first byte
293     * @return
294     */
295   private def bitSplits(key: String, options: Map[String, String], maskedBits: Int): Seq[Long] = {
296     options.get(key).map(_.toInt) match {
297       case None => Seq.empty
298       case Some(bits) =>
299         require(bits > 0 && bits < 64, "Bit split must be between 1 and 63")
300 
301         // recursive function to create all bit permutations
302         def add(result: Seq[Seq[Int]], bits: Seq[Int], remaining: Int): Seq[Seq[Int]] = {
303           if (remaining == 0) { Seq(bits) } else {
304             val zero = add(result, bits :+ 0, remaining - 1)
305             val one = add(result, bits :+ 1, remaining - 1)
306             result ++ zero ++ one
307           }
308         }
309         // note: first bit(s) in z value are not used, and are always 0
310         def toLong(bits: Seq[Int]): Long = {
311           val binaryString = (Seq.fill(maskedBits)(0) ++ bits).padTo(64, 0).mkString("")
312           java.lang.Long.parseLong(binaryString, 2)
313         }
314 
315         add(Seq.empty, Seq.empty, bits).map(toLong)
316     }
317   }
