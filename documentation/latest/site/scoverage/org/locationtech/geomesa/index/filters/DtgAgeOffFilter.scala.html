25 /**
26   * Age-off a feature based on an attribute time
27   */
28 trait DtgAgeOffFilter extends AgeOffFilter with LazyLogging {
29 
30   protected var sft: SimpleFeatureType = _
31   protected var index: GeoMesaFeatureIndex[_, _] = _
32 
33   protected var reusableSf: KryoBufferSimpleFeature = _
34   protected var dtgIndex: Int = -1
35 
36   override def init(options: Map[String, String]): Unit = {
37     import DtgAgeOffFilter.Configuration.{DtgOpt, IndexOpt, IndexSftOpt, SftOpt}
38 
39     super.init(options)
40 
41     val spec = options(SftOpt)
42     sft = IteratorCache.sft(spec)
43     index = options.get(IndexSftOpt) match {
44       case None => IteratorCache.index(sft, spec, options(IndexOpt))
45       case Some(ispec) => IteratorCache.index(IteratorCache.sft(ispec), ispec, options(IndexOpt))
46     }
47 
48     reusableSf = IteratorCache.serializer(spec, index.serializedWithId).getReusableFeature
49     dtgIndex = options(DtgOpt).toInt // note: keep this last, for back-compatibility with DtgAgeOffIterator
50   }
51 
52   override def accept(row: Array[Byte],
53                       rowOffset: Int,
54                       rowLength: Int,
55                       value: Array[Byte],
56                       valueOffset: Int,
57                       valueLength: Int,
58                       timestamp: Long): Boolean = {
59     try {
60       reusableSf.setBuffer(value, valueOffset, valueLength)
61       reusableSf.getDateAsLong(dtgIndex) > expiry
62     } catch {
63       case NonFatal(e) =>
64         logger.error(s"Error checking age-off for " +
65             Option(value).getOrElse(Array.empty).mkString("[", ",", s"], offset: $valueOffset, length: $valueLength"))
66         false
67     }
68   }
69 }

77     val IndexSftOpt = "index-sft"

79   }

81   def configure(sft: SimpleFeatureType,
82                 index: GeoMesaFeatureIndex[_, _],
83                 expiry: scala.concurrent.duration.Duration,
84                 dtgField: Option[String]): Map[String, String] = {
85     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
86 
87     require(!sft.isTableSharing || SystemProperty("geomesa.age-off.override").option.exists(_.toBoolean),
88       "DtgAgeOff filter should only be applied to features that don't use table sharing. You may override this check" +
89           "by setting the system property 'geomesa.age-off.override=true', however please note that age-off" +
90           "will affect all shared feature types in the same catalog, and may not work correctly with multiple feature" +
91           "types")
92 
93     val dtgIndex = dtgField match {
94       case None =>
95         sft.getDtgIndex.getOrElse {
96           throw new IllegalArgumentException("Simple feature type does not have a valid date field")
97         }
98       case Some(dtg) =>
99         val i = sft.indexOf(dtg)
100         if (i == -1 || !classOf[Date].isAssignableFrom(sft.getDescriptor(i).getType.getBinding)) {
101           throw new IllegalArgumentException(s"Simple feature type does not have a valid date field '$dtg'")
102         }
103         i
104     }
105 
106     val indexSftOpt = Some(index.sft).collect {
107       case s if s != sft => SimpleFeatureTypes.encodeType(s, includeUserData = true)
108     }
109 
110     AgeOffFilter.configure(sft, expiry) ++ Map (
111       Configuration.SftOpt   -> SimpleFeatureTypes.encodeType(sft),
112       Configuration.IndexOpt -> index.identifier,
113       Configuration.DtgOpt   -> dtgIndex.toString
114     ) ++ optionalMap(Configuration.IndexSftOpt -> indexSftOpt)
115   }
