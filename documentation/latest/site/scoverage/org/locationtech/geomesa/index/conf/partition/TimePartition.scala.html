25 /**
26   * Partition by time period. Partitions consist of 0-padded 5-digit strings
27   *
28   * @param metadata used for registering outside tables for query
29   * @param typeName simple feature type name
30   * @param dtg date attribute
31   * @param dtgIndex date attribute index
32   * @param period time period
33   */
34 class TimePartition(metadata: GeoMesaMetadata[String], typeName: String, dtg: String, dtgIndex: Int, period: TimePeriod)
35     extends TablePartition {
36 
37   import TimePartition.PartitionKeyPrefix
38 
39   private val toBin = BinnedTime.dateToBin(period)
40   private val toDate = BinnedTime.binnedTimeToDate(period)
41 
42   private val cache = new CustomPartitionCache(metadata, typeName)
43 
44   /**
45     * Register a new partition
46     *
47     * @param partition partition name
48     * @param start start time for the partition
49     * @param end end time for the partition
50     */
51   def register(partition: String, start: ZonedDateTime, end: ZonedDateTime): Unit = {
52     val dates = s"${DateParsing.format(start)}/${DateParsing.format(end)}"
53     metadata.insert(typeName, s"$PartitionKeyPrefix$partition", dates)
54     cache.invalidate()
55   }
56 
57   override def partition(feature: SimpleFeature): String = {
58     val date = feature.getAttribute(dtgIndex).asInstanceOf[Date]
59     f"${toBin(ZonedDateTime.ofInstant(date.toInstant, ZoneOffset.UTC))}%05d" // a short should fit into 5 digits
60   }
61 
62   override def partitions(filter: Filter): Option[Seq[String]] = {
63     val intervals = FilterHelper.extractIntervals(filter, dtg)
64     if (intervals.disjoint) {
65       Some(Seq.empty)
66     } else if (intervals.isEmpty || !intervals.forall(_.isBoundedBothSides)) {
67       None
68     } else {
69       val bins = intervals.values.flatMap(i => Range.inclusive(toBin(i.lower.value.get), toBin(i.upper.value.get)))
70       val names = bins.distinct.map(b => f"$b%05d") ++ cache.customPartitions().collect {
71         case (p, start, end) if overlaps(intervals, start, end) => p
72       }
73       Some(names)
74     }
75   }
76 
77   override def recover(partition: String): AnyRef =
78     new Date(toDate(BinnedTime(partition.toShort, 0L)).toInstant.toEpochMilli)
79 
80   private def overlaps(intervals: FilterValues[Bounds[ZonedDateTime]], start: ZonedDateTime, end: ZonedDateTime): Boolean =
81     intervals.exists(i => i.lower.value.forall(_.isBefore(end)) && i.upper.value.forall(_.isAfter(start)))
82 }

111   /**
112     * Caches parsed partition bounds
113     *
114     * @param metadata metadata for reading bounds
115     * @param typeName simple feature type name
116     */
117   private class CustomPartitionCache(metadata: GeoMesaMetadata[String], typeName: String) {
118 
119     private val expiry = TableBasedMetadata.Expiry.toDuration.get.toMillis
120 
121     private var partitions: Seq[(String, ZonedDateTime, ZonedDateTime)] = _
122     private var reload = 0L
123 
124     def customPartitions(): Seq[(String, ZonedDateTime, ZonedDateTime)] = {
125       if (System.currentTimeMillis() < reload) { partitions } else {
126         reload = System.currentTimeMillis() + expiry
127         partitions = metadata.scan(typeName, PartitionKeyPrefix).map { case (k, v) =>
128           val Array(start, end) = v.split("/").map(DateParsing.parse(_))
129           (k.substring(PartitionKeyPrefix.length), start, end)
130         }
131         partitions
132       }
133     }
134 
135     def invalidate(): Unit = reload = 0L
136   }
