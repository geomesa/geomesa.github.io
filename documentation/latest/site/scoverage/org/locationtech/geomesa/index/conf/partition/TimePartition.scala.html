26 /**
27   * Partition by time period. Partitions consist of 0-padded 5-digit strings
28   *
29   * @param metadata used for registering outside tables for query
30   * @param typeName simple feature type name
31   * @param dtg date attribute
32   * @param dtgIndex date attribute index
33   * @param period time period
34   */
35 class TimePartition(metadata: GeoMesaMetadata[String], typeName: String, dtg: String, dtgIndex: Int, period: TimePeriod)
36     extends TablePartition {
37 
38   import TimePartition.PartitionKeyPrefix
39 
40   private val toBin = BinnedTime.dateToBin(period)
41   private val toDate = BinnedTime.binnedTimeToDate(period)
42 
43   private val cache = new CustomPartitionCache(metadata, typeName)
44 
45   /**
46     * Register a new partition
47     *
48     * @param partition partition name
49     * @param start start time for the partition
50     * @param end end time for the partition
51     */
52   def register(partition: String, start: ZonedDateTime, end: ZonedDateTime): Unit = {
53     val dates = s"${DateParsing.format(start)}/${DateParsing.format(end)}"
54     metadata.insert(typeName, s"$PartitionKeyPrefix$partition", dates)
55     cache.invalidate()
56   }
57 
58   override def partition(feature: SimpleFeature): String = {
59     val date = feature.getAttribute(dtgIndex).asInstanceOf[Date]
60     f"${toBin(ZonedDateTime.ofInstant(toInstant(date), ZoneOffset.UTC))}%05d" // a short should fit into 5 digits
61   }
62 
63   override def partitions(filter: Filter): Option[Seq[String]] = {
64     val intervals = FilterHelper.extractIntervals(filter, dtg)
65     if (intervals.disjoint) {
66       Some(Seq.empty)
67     } else if (intervals.isEmpty || !intervals.forall(_.isBoundedBothSides)) {
68       None
69     } else {
70       val bins = intervals.values.flatMap(i => Range.inclusive(toBin(i.lower.value.get), toBin(i.upper.value.get)))
71       val names = bins.distinct.map(b => f"$b%05d") ++ cache.customPartitions().collect {
72         case (p, start, end) if overlaps(intervals, start, end) => p
73       }
74       Some(names)
75     }
76   }
77 
78   override def recover(partition: String): AnyRef =
79     new Date(toDate(BinnedTime(partition.toShort, 0L)).toInstant.toEpochMilli)
80 
81   private def overlaps(intervals: FilterValues[Bounds[ZonedDateTime]], start: ZonedDateTime, end: ZonedDateTime): Boolean =
82     intervals.exists(i => i.lower.value.forall(_.isBefore(end)) && i.upper.value.forall(_.isAfter(start)))
83 }

110   }

112   /**
113     * Caches parsed partition bounds
114     *
115     * @param metadata metadata for reading bounds
116     * @param typeName simple feature type name
117     */
118   private class CustomPartitionCache(metadata: GeoMesaMetadata[String], typeName: String) {
119 
120     private val expiry = TableBasedMetadata.Expiry.toDuration.get.toMillis
121 
122     private var partitions: Seq[(String, ZonedDateTime, ZonedDateTime)] = _
123     private var reload = 0L
124 
125     def customPartitions(): Seq[(String, ZonedDateTime, ZonedDateTime)] = {
126       if (System.currentTimeMillis() < reload) { partitions } else {
127         reload = System.currentTimeMillis() + expiry
128         partitions = metadata.scan(typeName, PartitionKeyPrefix).map { case (k, v) =>
129           val Array(start, end) = v.split("/").map(DateParsing.parse(_))
130           (k.substring(PartitionKeyPrefix.length), start, end)
131         }
132         partitions
133       }
134     }
135 
136     def invalidate(): Unit = reload = 0L
137   }
