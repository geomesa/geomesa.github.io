29 class Z2IndexKeySpace(val sft: SimpleFeatureType, val sharding: ShardStrategy, geomField: String)
30     extends IndexKeySpace[Z2IndexValues, Long] {
31 
32   import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
33 
34   require(classOf[Point].isAssignableFrom(sft.getDescriptor(geomField).getType.getBinding),
35     s"Expected field $geomField to have a point binding, but instead it has: " +
36         sft.getDescriptor(geomField).getType.getBinding.getSimpleName)

38   protected val sfc: Z2SFC = Z2SFC
39 
40   protected val geomIndex: Int = sft.indexOf(geomField)
41 
42   override val attributes: Seq[String] = Seq(geomField)
43 
44   override val indexKeyByteLength: Right[(Array[Byte], Int, Int) => Int, Int] = Right(8 + sharding.length)
45 
46   override val sharing: Array[Byte] = Array.empty
47 
48   override def toIndexKey(writable: WritableFeature,
49                           tier: Array[Byte],
50                           id: Array[Byte],
51                           lenient: Boolean): RowKeyValue[Long] = {
52     val geom = writable.getAttribute[Point](geomIndex)
53     if (geom == null) {
54       throw new IllegalArgumentException(s"Null geometry in feature ${writable.feature.getID}")
55     }
56     val z = try { sfc.index(geom.getX, geom.getY, lenient) } catch {
57       case NonFatal(e) => throw new IllegalArgumentException(s"Invalid z value from geometry: $geom", e)
58     }
59     val shard = sharding(writable)
60 
61     // create the byte array - allocate a single array up front to contain everything
62     // ignore tier, not used here
63     val bytes = Array.ofDim[Byte](shard.length + 8 + id.length)
64 
65     if (shard.isEmpty) {
66       ByteArrays.writeLong(z, bytes, 0)
67       System.arraycopy(id, 0, bytes, 8, id.length)
68     } else {
69       bytes(0) = shard.head // shard is only a single byte
70       ByteArrays.writeLong(z, bytes, 1)
71       System.arraycopy(id, 0, bytes, 9, id.length)
72     }
73 
74     SingleRowKeyValue(bytes, sharing, shard, z, tier, id, writable.values)
75   }
76 
77   override def getIndexValues(filter: Filter, explain: Explainer): Z2IndexValues = {
78 
79     val geometries: FilterValues[Geometry] = {
80       val extracted = FilterHelper.extractGeometries(filter, geomField, intersect = true) // intersect since we have points
81       if (extracted.nonEmpty) { extracted } else { FilterValues(Seq(WholeWorldPolygon)) }
82     }
83 
84     explain(s"Geometries: $geometries")
85 
86     if (geometries.disjoint) {
87       explain("Non-intersecting geometries extracted, short-circuiting to empty query")
88       return Z2IndexValues(sfc, geometries, Seq.empty)
89     }
90 
91     // compute our ranges based on the coarse bounds for our query
92     val xy: Seq[(Double, Double, Double, Double)] = {
93       val multiplier = QueryProperties.PolygonDecompMultiplier.toInt.get
94       val bits = QueryProperties.PolygonDecompBits.toInt.get
95       geometries.values.flatMap(GeometryUtils.bounds(_, multiplier, bits))
96     }
97 
98     Z2IndexValues(sfc, geometries, xy)
99   }
100 
101   override def getRanges(values: Z2IndexValues, multiplier: Int): Iterator[ScanRange[Long]] = {
102     val Z2IndexValues(_, geoms, xy) = values
103     if (geoms.disjoint) {
104       Iterator.empty
105     } else {
106       // note: `target` will always be Some, as ScanRangesTarget has a default value
107       val target = QueryProperties.ScanRangesTarget.option.map(t => math.max(1, t.toInt / multiplier))
108       sfc.ranges(xy, 64, target).iterator.map(r => BoundedRange(r.lower, r.upper))
109     }
110   }
111 
112   override def getRangeBytes(ranges: Iterator[ScanRange[Long]], tier: Boolean): Iterator[ByteRange] = {
113     if (sharding.length == 0) {
114       ranges.map {
115         case BoundedRange(lo, hi) => BoundedByteRange(ByteArrays.toBytes(lo), ByteArrays.toBytesFollowingPrefix(hi))
116         case r => throw new IllegalArgumentException(s"Unexpected range type $r")
117       }
118     } else {
119       ranges.flatMap {
120         case BoundedRange(lo, hi) =>
121           val lower = ByteArrays.toBytes(lo)
122           val upper = ByteArrays.toBytesFollowingPrefix(hi)
123           sharding.shards.map(p => BoundedByteRange(ByteArrays.concat(p, lower), ByteArrays.concat(p, upper)))
124         case r => throw new IllegalArgumentException(s"Unexpected range type $r")
125       }
126     }
127   }
128 
129   override def useFullFilter(values: Option[Z2IndexValues],
130                              config: Option[GeoMesaDataStoreConfig],
131                              hints: Hints): Boolean = {
132     // if the user has requested strict bounding boxes, we apply the full filter
133     !Option(hints.get(LOOSE_BBOX)).map(Boolean.unbox).getOrElse(config.forall(_.queries.looseBBox)) ||
134       // if the spatial predicate is rectangular (e.g. a bbox), the index is fine enough that we
135       // don't need to apply the filter on top of it. this may cause some minor errors at extremely
136       // fine resolutions, but the performance is worth it
137       // if we have a complicated geometry predicate, we need to pass it through to be evaluated
138       values.exists(_.geometries.values.exists(g => !GeometryUtils.isRectangular(g))) ||
139       // for attribute-level vis, we need to re-evaluate the filter to account for visibility of the row key
140       (sft.getVisibilityLevel == VisibilityLevel.Attribute && values.exists(_.geometries.nonEmpty))
141   }
142 }

153   }
