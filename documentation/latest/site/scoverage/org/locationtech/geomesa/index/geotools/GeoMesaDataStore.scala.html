119   @throws(classOf[IllegalArgumentException])
120   override protected def preSchemaCreate(sft: SimpleFeatureType): Unit = {
121     // check for old user data keys and re-map them
122     InternalConfigs.DeprecatedConfigMappings.foreach { case (from, to) =>
123       val v = sft.getUserData.remove(from)
124       if (v != null) {
125         sft.getUserData.put(to, v)
126       }
127     }
128 
129     // validate column groups
130     adapter.groups.validate(sft)
131 
132     // disable table sharing, no longer supported
133     // noinspection ScalaDeprecation
134     if (sft.isTableSharing) {
135       logger.warn("Table sharing is no longer supported - disabling table sharing")
136       sft.getUserData.remove(Configs.TableSharing)
137       sft.getUserData.remove(InternalConfigs.TableSharingPrefix)
138     }
139 
140     // configure the indices to use
141     if (sft.getIndices.isEmpty) {
142       val indices = GeoMesaFeatureIndexFactory.indices(sft)
143       if (indices.isEmpty) {
144         throw new IllegalArgumentException("There are no available indices that support the schema " +
145             SimpleFeatureTypes.encodeType(sft))
146       }
147       sft.setIndices(indices)
148     }
149 
150     // try to create the indices up front to ensure they are valid for the sft
151     GeoMesaFeatureIndexFactory.create(this, sft, sft.getIndices)
152 
153     // remove the enabled indices after configuration so we don't persist them
154     sft.getUserData.remove(SimpleFeatureTypes.Configs.EnabledIndices)
155     // remove any 'index' flags in the attribute metadata, they have already been captured in the indices above
156     sft.getAttributeDescriptors.asScala.foreach(_.getUserData.remove(AttributeOptions.OptIndex))
157 
158     // for partitioned schemas, persist the table partitioning keys
159     if (sft.isPartitioned) {
160       InternalConfigs.PartitionConfigMappings.foreach { case (from, to) =>
161         Option(sft.getUserData.get(from)).foreach(sft.getUserData.put(to, _))
162       }
163       InternalConfigs.PartitionConfigPrefixMappings.foreach { case (from, to) =>
164         sft.getUserData.asScala.toMap.collect {
165           case (k: String, v) if k.startsWith(from) => sft.getUserData.put(to + k.substring(from.length), v)
166         }
167       }
168     }
169 
170     // set stats enabled based on the data store config if not explicitly set
171     if (!sft.getUserData.containsKey(SimpleFeatureTypes.Configs.StatsEnabled)) {
172       sft.setStatsEnabled(config.generateStats)
173     }
174 
175     sft.getFeatureExpiration // validate any configured age-off
176   }

179   override protected def preSchemaUpdate(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
180     updateSchemaUserData(sft, previous)

182     try { GeoMesaFeatureIndexFactory.create(this, sft, sft.getIndices) } catch {

200   // create the new index tables (if not using partitioned tables)
201   override protected def onSchemaUpdated(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
202     val partitioned = TablePartition.partitioned(sft)
203 
204     // check for column renaming
205     val colMap = getColumnMap(previous, sft).map { case (k, v) => (v, k) }
206 
207     // rewrite the table name keys for the renamed cols
208     if (colMap.nonEmpty) {
209       sft.getIndices.foreach { i =>
210         if (i.attributes.exists(colMap.contains)) {
211           val prev = i.copy(attributes = i.attributes.map(a => colMap.getOrElse(a, a)))
212           val old = manager.index(previous, GeoMesaFeatureIndex.identifier(prev))
213           val index = GeoMesaFeatureIndexFactory.create(this, sft, Seq(i)).headOption.getOrElse {
214             throw new IllegalArgumentException(
215               s"Error configuring new feature index: ${GeoMesaFeatureIndex.identifier(i)}")
216           }
217           val partitions = if (!partitioned) { Seq(None) } else {
218             // have to use the old table name key but the new sft name for looking up the partitions
219             val tableNameKey = old.tableNameKey(Some(""))
220             val offset = tableNameKey.length
221             metadata.scan(sft.getTypeName, tableNameKey).map { case (k, _) => Some(k.substring(offset)) }
222           }
223           partitions.foreach { p =>
224             metadata.read(sft.getTypeName, old.tableNameKey(p)).foreach { v =>
225               metadata.insert(sft.getTypeName, index.tableNameKey(p), v)
226             }
227           }
228         }
229       }
230     }
231 
232     // configure any new indices
233     if (partitioned) {
234       logger.debug("Delaying creation of partitioned indices")
235     } else {
236       logger.debug(s"Ensuring indices ${manager.indices(sft).map(_.identifier).mkString(", ")}")
237       manager.indices(sft).toList.map(i => CachedThreadPool.submit(() => adapter.createTable(i, None, i.getSplits(None)))).foreach(_.get)
238     }
239 
240     // update stats
241     if (previous.statsEnabled) {
242       if (!sft.statsEnabled) {
243         stats.writer.clear(previous)
244       } else if (sft.getTypeName != previous.getTypeName || colMap.nonEmpty) {
245         stats.writer.rename(sft, previous)
246       }
247     }
248 
249     // rename tables to match the new sft name
250     if (sft.getTypeName != previous.getTypeName || sft.getIndices != previous.getIndices) {
251       if (FastConverter.convertOrElse[java.lang.Boolean](sft.getUserData.get(Configs.UpdateRenameTables), false)) {
252         manager.indices(sft).foreach { index =>
253           val partitions = if (partitioned) { index.getPartitions.map(Option.apply) } else { Seq(None) }
254           partitions.foreach { partition =>
255             val key = index.tableNameKey(partition)
256             metadata.read(sft.getTypeName, key).foreach { table =>
257               metadata.remove(sft.getTypeName, key)
258               val renamed = index.configureTableName(partition, adapter.tableNameLimit)
259               if (renamed != table) {
260                 logger.debug(s"Renaming table from '$table' to '$renamed'")
261                 adapter.renameTable(table, renamed)
262               }
263             }
264           }
265         }
266       }
267     }
268   }

270   // delete the index tables
271   override protected def onSchemaDeleted(sft: SimpleFeatureType): Unit = {
272     def deleteFull(index: GeoMesaFeatureIndex[_, _]): Unit =
273       adapter.deleteTables(index.deleteTableNames(None))
274     def deleteShared(index: GeoMesaFeatureIndex[_, _]): Unit = {
275       if (index.keySpace.sharing.isEmpty) { deleteFull(index) } else {
276         adapter.clearTables(index.deleteTableNames(None), Some(index.keySpace.sharing))
277       }
278     }
279 
280     val indices = manager.indices(sft).toList
281     // noinspection ScalaDeprecation
282     if (sft.isTableSharing && getTypeNames.exists(t => t != sft.getTypeName && getSchema(t).isTableSharing)) {
283       indices.map(i => CachedThreadPool.submit(() => deleteShared(i))).foreach(_.get)
284     } else {
285       indices.map(i => CachedThreadPool.submit(() => deleteFull(i))).foreach(_.get)
286     }
287     if (sft.statsEnabled) {
288       stats.writer.clear(sft)
289     }
290   }

294   /**
295    * @see org.geotools.api.data.DataStore#getSchema(java.lang.String)
296    * @param typeName feature type name
297    * @return feature type, or null if it does not exist
298    */
299   override def getSchema(typeName: String): SimpleFeatureType = {
300     var sft = super.getSchema(typeName)
301     if (sft != null) {
302       // ensure index metadata is correct
303       if (sft.getIndices.exists(i => i.attributes.isEmpty && i.name != IdIndex.name)) {
304         sft = SimpleFeatureTypes.mutable(sft)
305         // migrate index metadata to standardized versions and attributes
306         transitionIndices(sft)
307         sft = SimpleFeatureTypes.immutable(sft)
308         // validate indices
309         try { manager.indices(sft) } catch {
310           case NonFatal(e) =>
311             throw new IllegalStateException(s"The schema ${sft.getTypeName} was written with a older " +
312                 "version of GeoMesa that is no longer supported. You may continue to use an older client, or " +
313                 s"manually edit the metadata for '${InternalConfigs.IndexVersions}' to exclude the invalid indices.", e)
314         }
315       } else {
316         // validate indices
317         try { manager.indices(sft) } catch {
318           case NonFatal(e) =>
319             val versions = sft.getIndices.map(i => s"${i.name}:${i.version}").mkString(",")
320             val available = GeoMesaFeatureIndexFactory.available(sft).map(i => s"${i._1}:${i._2}").mkString(",")
321             logger.error(s"Trying to access schema ${sft.getTypeName} with invalid index versions '$versions' - " +
322                 s"available indices are '$available'", e)
323             throw new IllegalStateException(s"The schema ${sft.getTypeName} was written with a newer " +
324                 "version of GeoMesa than this client can handle. Please ensure that you are using the " +
325                 "same GeoMesa jar versions across your entire workflow. For more information, see " +
326                 "https://www.geomesa.org/documentation/stable/user/upgrade.html")
327         }
328       }
329 
330       // check for sft-level stats flag and set it if not present
331       if (!sft.getUserData.containsKey(SimpleFeatureTypes.Configs.StatsEnabled)) {
332         val extra = Collections.singletonMap(SimpleFeatureTypes.Configs.StatsEnabled, config.generateStats.toString)
333         sft = SimpleFeatureTypes.immutable(sft, extra)
334       }
335 
336       // get the remote version if it's available, but don't wait for it
337       GeoMesaDataStore.versions.get(VersionKey(this)).getNow(Right(None)) match {
338         case Left(e) => throw e
339         case Right(version) =>
340           version.foreach { v =>
341             val userData = Collections.singletonMap[AnyRef, AnyRef](InternalConfigs.RemoteVersion, v.toString)
342             sft = SimpleFeatureTypes.immutable(sft, userData)
343           }
344       }
345     }
346     sft
347   }

461    * Checks a simple feature type against an existing schema
462    *
463    * @param typeName type name
464    * @param sft udpated simple feature type
465    * @return
466    */
467   def checkSchemaCompatibility(typeName: String, sft: SimpleFeatureType): SchemaCompatibility = {
468     val previous = getSchema(typeName)
469     if (previous == null) {
470       new SchemaCompatibility.DoesNotExist(this, sft)
471     } else {
472       validateSchemaUpdate(previous, sft) match {
473         case Right(TypeComparison.Compatible(extension, renamed, _)) =>
474           val copy = SimpleFeatureTypes.copy(sft)
475           updateSchemaUserData(copy, previous)
476           if (!extension && !renamed && copy.getUserData == previous.getUserData) {
477             SchemaCompatibility.Unchanged
478           } else {
479             new SchemaCompatibility.Compatible(this, typeName, sft)
480           }
481 
482         case Left(e) =>
483           SchemaCompatibility.Incompatible(e)
484       }
485     }
486   }

490   /**
491    * Updates the user data for a schema update prior to persistence. Handles converting existing
492    * user data based on new/updated attributes, copying existing user data over, etc. Feature
493    * type will be updated in place
494    *
495    * @param sft schema update
496    * @param previous existing schema
497    * @return
498    */
499   private def updateSchemaUserData(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
500 
501     // check for column renaming
502     val colMap = getColumnMap(previous, sft)
503 
504     def remapCol(name: String): String = colMap.getOrElse(name, name)
505 
506     // check for attributes flagged 'index' and convert them to sft-level user data
507     def indexed(d: AttributeDescriptor): Boolean = {
508       d.getUserData.get(AttributeOptions.OptIndex) match {
509         case i: String if Seq("true", "full").exists(_.equalsIgnoreCase(i)) => true
510         case i if i == null || Seq("false", "none").exists(_.equalsIgnoreCase(i.toString)) => false
511         case i => throw new IllegalArgumentException(s"Configured index coverage '$i' is not valid: expected 'true'")
512       }
513     }
514     sft.getAttributeDescriptors.asScala.foreach { d =>
515       if (indexed(d)) {
516         val existing = {
517           val explicit = sft.getIndices
518           if (explicit.nonEmpty) { explicit } else { previous.getIndices }
519         }
520         if (!existing.exists(e => e.name == AttributeIndex.name && remapCol(e.attributes.head) == d.getLocalName)) {
521           val fields = Seq(d.getLocalName) ++ Option(sft.getGeomField) ++ sft.getDtgField
522           val id = IndexId(AttributeIndex.name, AttributeIndex.version, fields, IndexMode.ReadWrite)
523           sft.setIndices(existing :+ id)
524         }
525       }
526     }
527 
528     // check for new indices and 'enabled indices' changes
529     val indices = {
530       val enabled = if (!sft.getUserData.containsKey(Configs.EnabledIndices)) { Seq.empty } else {
531         GeoMesaFeatureIndexFactory.indices(sft)
532       }
533       val remapped = (previous.getIndices ++ sft.getIndices).map(i => i.copy(attributes = i.attributes.map(remapCol)))
534       (remapped ++ enabled).foldLeft(Seq.empty[IndexId]) { (sum, next) =>
535         // note: ignore index version
536         if (sum.exists(i => i.name == next.name && i.attributes == next.attributes)) { sum } else { sum :+ next }
537       }
538     }
539     if (indices != previous.getIndices) {
540       sft.setIndices(indices)
541     }
542 
543     // preserve any existing user data but overwrite any keys we redefine
544     val userData = new java.util.HashMap[AnyRef, AnyRef](previous.getUserData)
545     userData.putAll(sft.getUserData)
546     sft.getUserData.putAll(userData)
547     // remove enabled indices as we don't need to persist it
548     sft.getUserData.remove(Configs.EnabledIndices)
549     // remove any null/empty keys as a way to delete existing user data
550     sft.getUserData.asScala.collect { case (k, null | "") => k }.foreach(sft.getUserData.remove)
551     // remove any 'index' flags in the attribute user data - we need them above for checking enabled indices
552     sft.getAttributeDescriptors.asScala.foreach(_.getUserData.remove(AttributeOptions.OptIndex))
553   }

555   /**
556    * Gets a map of column renames, for use during schema updates
557    *
558    * @param previous previous feature type
559    * @param sft updated feature type
560    * @return map of old name -> new name
561    */
562   private def getColumnMap(previous: SimpleFeatureType, sft: SimpleFeatureType): Map[String, String] = {
563     previous.getAttributeDescriptors.asScala.zipWithIndex.toMap.flatMap { case (prev, i) =>
564       val cur = sft.getDescriptor(i)
565       if (prev.getLocalName != cur.getLocalName) {
566         Map(prev.getLocalName -> cur.getLocalName)
567       } else {
568         Map.empty[String, String]
569       }
570     }
571   }
