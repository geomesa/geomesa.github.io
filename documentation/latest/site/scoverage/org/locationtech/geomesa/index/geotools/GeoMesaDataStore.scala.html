118   @throws(classOf[IllegalArgumentException])
119   override protected def preSchemaCreate(sft: SimpleFeatureType): Unit = {
120     // check for old user data keys and re-map them
121     InternalConfigs.DeprecatedConfigMappings.foreach { case (from, to) =>
122       val v = sft.getUserData.remove(from)
123       if (v != null) {
124         sft.getUserData.put(to, v)
125       }
126     }
127 
128     // validate column groups
129     adapter.groups.validate(sft)
130 
131     // disable table sharing, no longer supported
132     // noinspection ScalaDeprecation
133     if (sft.isTableSharing) {
134       logger.warn("Table sharing is no longer supported - disabling table sharing")
135       sft.getUserData.remove(Configs.TableSharing)
136       sft.getUserData.remove(InternalConfigs.TableSharingPrefix)
137     }
138 
139     // configure the indices to use
140     if (sft.getIndices.isEmpty) {
141       val indices = GeoMesaFeatureIndexFactory.indices(sft)
142       if (indices.isEmpty) {
143         throw new IllegalArgumentException("There are no available indices that support the schema " +
144             SimpleFeatureTypes.encodeType(sft))
145       }
146       sft.setIndices(indices)
147     }
148 
149     // try to create the indices up front to ensure they are valid for the sft
150     GeoMesaFeatureIndexFactory.create(this, sft, sft.getIndices)
151 
152     // remove the enabled indices after configuration so we don't persist them
153     sft.getUserData.remove(SimpleFeatureTypes.Configs.EnabledIndices)
154     // remove any 'index' flags in the attribute metadata, they have already been captured in the indices above
155     sft.getAttributeDescriptors.asScala.foreach(_.getUserData.remove(AttributeOptions.OptIndex))
156 
157     // for partitioned schemas, persist the table partitioning keys
158     if (sft.isPartitioned) {
159       InternalConfigs.PartitionConfigMappings.foreach { case (from, to) =>
160         Option(sft.getUserData.get(from)).foreach(sft.getUserData.put(to, _))
161       }
162       InternalConfigs.PartitionConfigPrefixMappings.foreach { case (from, to) =>
163         sft.getUserData.asScala.toMap.collect {
164           case (k: String, v) if k.startsWith(from) => sft.getUserData.put(to + k.substring(from.length), v)
165         }
166       }
167     }
168 
169     // set stats enabled based on the data store config if not explicitly set
170     if (!sft.getUserData.containsKey(SimpleFeatureTypes.Configs.StatsEnabled)) {
171       sft.setStatsEnabled(config.generateStats)
172     }
173 
174     sft.getFeatureExpiration // validate any configured age-off
175   }

199   // create the new index tables (if not using partitioned tables)
200   override protected def onSchemaUpdated(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
201     val partitioned = TablePartition.partitioned(sft)
202 
203     // check for column renaming
204     val colMap = getColumnMap(previous, sft).map { case (k, v) => (v, k) }
205 
206     // rewrite the table name keys for the renamed cols
207     if (colMap.nonEmpty) {
208       sft.getIndices.foreach { i =>
209         if (i.attributes.exists(colMap.contains)) {
210           val prev = i.copy(attributes = i.attributes.map(a => colMap.getOrElse(a, a)))
211           val old = manager.index(previous, GeoMesaFeatureIndex.identifier(prev))
212           val index = GeoMesaFeatureIndexFactory.create(this, sft, Seq(i)).headOption.getOrElse {
213             throw new IllegalArgumentException(
214               s"Error configuring new feature index: ${GeoMesaFeatureIndex.identifier(i)}")
215           }
216           val partitions = if (!partitioned) { Seq(None) } else {
217             // have to use the old table name key but the new sft name for looking up the partitions
218             val tableNameKey = old.tableNameKey(Some(""))
219             val offset = tableNameKey.length
220             metadata.scan(sft.getTypeName, tableNameKey).map { case (k, _) => Some(k.substring(offset)) }
221           }
222           partitions.foreach { p =>
223             metadata.read(sft.getTypeName, old.tableNameKey(p)).foreach { v =>
224               metadata.insert(sft.getTypeName, index.tableNameKey(p), v)
225             }
226           }
227         }
228       }
229     }
230 
231     // configure any new indices
232     if (partitioned) {
233       logger.debug("Delaying creation of partitioned indices")
234     } else {
235       logger.debug(s"Ensuring indices ${manager.indices(sft).map(_.identifier).mkString(", ")}")
236       manager.indices(sft).toList.map(i => CachedThreadPool.submit(() => adapter.createTable(i, None, i.getSplits(None)))).foreach(_.get)
237     }
238 
239     // update stats
240     if (previous.statsEnabled) {
241       if (!sft.statsEnabled) {
242         stats.writer.clear(previous)
243       } else if (sft.getTypeName != previous.getTypeName || colMap.nonEmpty) {
244         stats.writer.rename(sft, previous)
245       }
246     }
247 
248     // rename tables to match the new sft name
249     if (sft.getTypeName != previous.getTypeName || sft.getIndices != previous.getIndices) {
250       if (FastConverter.convertOrElse[java.lang.Boolean](sft.getUserData.get(Configs.UpdateRenameTables), false)) {
251         manager.indices(sft).foreach { index =>
252           val partitions = if (partitioned) { index.getPartitions.map(Option.apply) } else { Seq(None) }
253           partitions.foreach { partition =>
254             val key = index.tableNameKey(partition)
255             metadata.read(sft.getTypeName, key).foreach { table =>
256               metadata.remove(sft.getTypeName, key)
257               val renamed = index.configureTableName(partition, adapter.tableNameLimit)
258               if (renamed != table) {
259                 logger.debug(s"Renaming table from '$table' to '$renamed'")
260                 adapter.renameTable(table, renamed)
261               }
262             }
263           }
264         }
265       }
266     }
267   }

269   // delete the index tables
270   override protected def onSchemaDeleted(sft: SimpleFeatureType): Unit = {
271     def deleteFull(index: GeoMesaFeatureIndex[_, _]): Unit =
272       adapter.deleteTables(index.deleteTableNames(None))
273     def deleteShared(index: GeoMesaFeatureIndex[_, _]): Unit = {
274       if (index.keySpace.sharing.isEmpty) { deleteFull(index) } else {
275         adapter.clearTables(index.deleteTableNames(None), Some(index.keySpace.sharing))
276       }
277     }
278 
279     val indices = manager.indices(sft).toList
280     // noinspection ScalaDeprecation
281     if (sft.isTableSharing && getTypeNames.exists(t => t != sft.getTypeName && getSchema(t).isTableSharing)) {
282       indices.map(i => CachedThreadPool.submit(() => deleteShared(i))).foreach(_.get)
283     } else {
284       indices.map(i => CachedThreadPool.submit(() => deleteFull(i))).foreach(_.get)
285     }
286     if (sft.statsEnabled) {
287       stats.writer.clear(sft)
288     }
289   }

293   /**
294    * @see org.geotools.api.data.DataStore#getSchema(java.lang.String)
295    * @param typeName feature type name
296    * @return feature type, or null if it does not exist
297    */
298   override def getSchema(typeName: String): SimpleFeatureType = {
299     var sft = super.getSchema(typeName)
300     if (sft != null) {
301       // ensure index metadata is correct
302       if (sft.getIndices.exists(i => i.attributes.isEmpty && i.name != IdIndex.name)) {
303         sft = SimpleFeatureTypes.mutable(sft)
304         // migrate index metadata to standardized versions and attributes
305         transitionIndices(sft)
306         sft = SimpleFeatureTypes.immutable(sft)
307         // validate indices
308         try { manager.indices(sft) } catch {
309           case NonFatal(e) =>
310             throw new IllegalStateException(s"The schema ${sft.getTypeName} was written with a older " +
311                 "version of GeoMesa that is no longer supported. You may continue to use an older client, or " +
312                 s"manually edit the metadata for '${InternalConfigs.IndexVersions}' to exclude the invalid indices.", e)
313         }
314       } else {
315         // validate indices
316         try { manager.indices(sft) } catch {
317           case NonFatal(e) =>
318             val versions = sft.getIndices.map(i => s"${i.name}:${i.version}").mkString(",")
319             val available = GeoMesaFeatureIndexFactory.available(sft).map(i => s"${i._1}:${i._2}").mkString(",")
320             logger.error(s"Trying to access schema ${sft.getTypeName} with invalid index versions '$versions' - " +
321                 s"available indices are '$available'", e)
322             throw new IllegalStateException(s"The schema ${sft.getTypeName} was written with a newer " +
323                 "version of GeoMesa than this client can handle. Please ensure that you are using the " +
324                 "same GeoMesa jar versions across your entire workflow. For more information, see " +
325                 "https://www.geomesa.org/documentation/stable/user/upgrade.html")
326         }
327       }
328 
329       // check for sft-level stats flag and set it if not present
330       if (!sft.getUserData.containsKey(SimpleFeatureTypes.Configs.StatsEnabled)) {
331         val extra = Collections.singletonMap(SimpleFeatureTypes.Configs.StatsEnabled, config.generateStats.toString)
332         sft = SimpleFeatureTypes.immutable(sft, extra)
333       }
334 
335       // get the remote version if it's available, but don't wait for it
336       GeoMesaDataStore.versions.get(VersionKey(this)).getNow(Right(None)) match {
337         case Left(e) => throw e
338         case Right(version) =>
339           version.foreach { v =>
340             val userData = Collections.singletonMap[AnyRef, AnyRef](InternalConfigs.RemoteVersion, v.toString)
341             sft = SimpleFeatureTypes.immutable(sft, userData)
342           }
343       }
344     }
345     sft
346   }

464    * @param typeName type name
465    * @param sft udpated simple feature type
466    * @return
467    */
468   def checkSchemaCompatibility(typeName: String, sft: SimpleFeatureType): SchemaCompatibility = {
469     val previous = getSchema(typeName)
470     if (previous == null) {
471       new SchemaCompatibility.DoesNotExist(this, sft)
472     } else {
473       validateSchemaUpdate(previous, sft) match {
474         case Right(TypeComparison.Compatible(extension, renamed, _)) =>
475           val copy = SimpleFeatureTypes.copy(sft)
476           updateSchemaUserData(copy, previous)
477           if (!extension && !renamed && copy.getUserData == previous.getUserData) {
478             SchemaCompatibility.Unchanged
479           } else {
480             new SchemaCompatibility.Compatible(this, typeName, sft)
481           }
482 
483         case Left(e) =>
484           SchemaCompatibility.Incompatible(e)
485       }
486     }
487   }

491   /**
492    * Updates the user data for a schema update prior to persistence. Handles converting existing
493    * user data based on new/updated attributes, copying existing user data over, etc. Feature
494    * type will be updated in place
495    *
496    * @param sft schema update
497    * @param previous existing schema
498    * @return
499    */
500   private def updateSchemaUserData(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
501 
502     // check for column renaming
503     val colMap = getColumnMap(previous, sft)
504 
505     def remapCol(name: String): String = colMap.getOrElse(name, name)
506 
507     // check for attributes flagged 'index' and convert them to sft-level user data
508     def indexed(d: AttributeDescriptor): Boolean = {
509       d.getUserData.get(AttributeOptions.OptIndex) match {
510         case i: String if Seq("true", "full").exists(_.equalsIgnoreCase(i)) => true
511         case i if i == null || Seq("false", "none").exists(_.equalsIgnoreCase(i.toString)) => false
512         case i => throw new IllegalArgumentException(s"Configured index coverage '$i' is not valid: expected 'true'")
513       }
514     }
515     sft.getAttributeDescriptors.asScala.foreach { d =>
516       if (indexed(d)) {
517         val existing = {
518           val explicit = sft.getIndices
519           if (explicit.nonEmpty) { explicit } else { previous.getIndices }
520         }
521         if (!existing.exists(e => e.name == AttributeIndex.name && remapCol(e.attributes.head) == d.getLocalName)) {
522           val fields = Seq(d.getLocalName) ++ Option(sft.getGeomField) ++ sft.getDtgField
523           val id = IndexId(AttributeIndex.name, AttributeIndex.version, fields, IndexMode.ReadWrite)
524           sft.setIndices(existing :+ id)
525         }
526       }
527     }
528 
529     // check for new indices and 'enabled indices' changes
530     val indices = {
531       val enabled = if (!sft.getUserData.containsKey(Configs.EnabledIndices)) { Seq.empty } else {
532         GeoMesaFeatureIndexFactory.indices(sft)
533       }
534       val remapped = (previous.getIndices ++ sft.getIndices).map(i => i.copy(attributes = i.attributes.map(remapCol)))
535       (remapped ++ enabled).foldLeft(Seq.empty[IndexId]) { (sum, next) =>
536         // note: ignore index version
537         if (sum.exists(i => i.name == next.name && i.attributes == next.attributes)) { sum } else { sum :+ next }
538       }
539     }
540     if (indices != previous.getIndices) {
541       sft.setIndices(indices)
542     }
543 
544     // preserve any existing user data but overwrite any keys we redefine
545     val userData = new java.util.HashMap[AnyRef, AnyRef](previous.getUserData)
546     userData.putAll(sft.getUserData)
547     sft.getUserData.putAll(userData)
548     // remove enabled indices as we don't need to persist it
549     sft.getUserData.remove(Configs.EnabledIndices)
550     // remove any null/empty keys as a way to delete existing user data
551     sft.getUserData.asScala.collect { case (k, null | "") => k }.foreach(sft.getUserData.remove)
552     // remove any 'index' flags in the attribute user data - we need them above for checking enabled indices
553     sft.getAttributeDescriptors.asScala.foreach(_.getUserData.remove(AttributeOptions.OptIndex))
554   }

556   /**
557    * Gets a map of column renames, for use during schema updates
558    *
559    * @param previous previous feature type
560    * @param sft updated feature type
561    * @return map of old name -> new name
562    */
563   private def getColumnMap(previous: SimpleFeatureType, sft: SimpleFeatureType): Map[String, String] = {
564     previous.getAttributeDescriptors.asScala.zipWithIndex.toMap.flatMap { case (prev, i) =>
565       val cur = sft.getDescriptor(i)
566       if (prev.getLocalName != cur.getLocalName) {
567         Map(prev.getLocalName -> cur.getLocalName)
568       } else {
569         Map.empty[String, String]
570       }
571     }
572   }
