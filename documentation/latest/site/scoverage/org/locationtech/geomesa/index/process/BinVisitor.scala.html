24 /**
25  * Binary format visitor
26  *
27  * @param sft simple feature type
28  * @param options encoding options
29  */
30 class BinVisitor(sft: SimpleFeatureType, options: EncodingOptions) extends GeoMesaProcessVisitor with LazyLogging {
31 
32   import scala.collection.JavaConverters._
33 
34   // for collecting results manually
35   private val manualResults = scala.collection.mutable.Queue.empty[Array[Byte]]
36   private val manualConversion = BinaryOutputEncoder(sft, options)
37 
38   private var result: java.util.Iterator[Array[Byte]] = _
39 
40   override def getResult: BinResult = {
41     if (result != null) {
42       BinResult(result)
43     } else {
44       BinResult(manualResults.iterator.asJava)
45     }
46   }
47 
48   // manually called for non-accumulo feature collections
49   override def visit(feature: Feature): Unit =
50     manualResults += manualConversion.encode(feature.asInstanceOf[SimpleFeature])
51 
52   override def execute(source: SimpleFeatureSource, query: Query): Unit = {
53     logger.debug(s"Visiting source type: ${source.getClass.getName}")
54 
55     query.getHints.put(QueryHints.BIN_TRACK, options.trackIdField.map(sft.getDescriptor(_).getLocalName).getOrElse("id"))
56     options.geomField.foreach(i => query.getHints.put(QueryHints.BIN_GEOM, sft.getDescriptor(i).getLocalName))
57     options.dtgField.foreach(i => query.getHints.put(QueryHints.BIN_DTG, sft.getDescriptor(i).getLocalName))
58     options.labelField.foreach(i => query.getHints.put(QueryHints.BIN_LABEL, sft.getDescriptor(i).getLocalName))
59 
60     val features =
61       CloseableIterator(source.getFeatures(query).features()).map(_.getAttribute(BIN_ATTRIBUTE_INDEX).asInstanceOf[Array[Byte]])
62     // try to ensure the iterator is closed, by making it Closeable (only discoverable via type matching), and by making it
63     // auto-closing when fully read
64     // TODO investigate options for returning Closeable things from a VectorProcess
65     result = new java.util.Iterator[Array[Byte]] with Closeable {
66       override def hasNext: Boolean = {
67         val more = features.hasNext
68         if (!more) {
69           close()
70         }
71         more
72       }
73       override def next(): Array[Byte] = features.next()
74       override def close(): Unit = features.close()
75     }
76   }
77 }

81     override def getValue: java.util.Iterator[Array[Byte]] = results
82   }
