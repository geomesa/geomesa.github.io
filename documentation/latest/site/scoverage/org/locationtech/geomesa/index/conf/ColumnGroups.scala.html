36   /**
37     * Gets the column groups for a simple feature type. The default group will contain all columns
38     *
39     * @param sft simple feature type
40     * @return
41     */
42   def apply(sft: SimpleFeatureType): Seq[(Array[Byte], SimpleFeatureType)] = {
43     val key = CacheKeyGenerator.cacheKey(sft)
44     var groups = ColumnGroups.cache.getIfPresent(key)
45     if (groups == null) {
46       if (sft.getVisibilityLevel == VisibilityLevel.Attribute) {
47         groups = IndexedSeq((ColumnGroups.Attributes, sft))
48       } else {
49         val map = scala.collection.mutable.Map.empty[String, SimpleFeatureTypeBuilder]
50 
51         sft.getAttributeDescriptors.asScala.foreach { descriptor =>
52           descriptor.getColumnGroups().foreach { group =>
53             map.getOrElseUpdate(group, new SimpleFeatureTypeBuilder()).add(descriptor)
54           }
55         }
56 
57         val sfts = map.map { case (group, builder) =>
58           builder.setName(sft.getTypeName)
59           val subset = SimpleFeatureTypes.immutable(builder.buildFeatureType(), sft.getUserData)
60           (group.getBytes(StandardCharsets.UTF_8), subset)
61         } + (ColumnGroups.Default -> sft)
62 
63         // return the smallest groups first, for consistency tiebreaker is string comparison of group
64         groups = sfts.toIndexedSeq.sortBy { case (group, subset) =>
65           (subset.getAttributeCount, new String(group, StandardCharsets.UTF_8))
66         }
67       }
68 
69       ColumnGroups.cache.put(key, groups)
70     }
71     groups
72   }

74   /**
75     * Get serializers for each column group
76     *
77     * @param sft simple feature type
78     * @return
79     */
80   def serializers(sft: SimpleFeatureType): Seq[(Array[Byte], SimpleFeatureSerializer)] = {
81     apply(sft).map { case (colFamily, subset) =>
82       if (colFamily.eq(ColumnGroups.Default) || colFamily.eq(ColumnGroups.Attributes)) {
83         (colFamily, KryoFeatureSerializer(subset, SerializationOptions.withoutId))
84       } else {
85         (colFamily, new ProjectingKryoFeatureSerializer(sft, subset, SerializationOptions.withoutId))
86       }
87     }
88   }

90   /**
91     * Find a column group that supports the given transform and filter
92     *
93     * @param sft simple feature type
94     * @param transform transform definitions
95     * @param ecql filter, if any
96     * @return
97     */
98   def group(sft: SimpleFeatureType, transform: Option[String], ecql: Option[Filter]): (Array[Byte], SimpleFeatureType) = {
99     val groups = apply(sft)
100     transform.map(Transforms(sft, _)) match {
101       case None => groups.last
102       case Some(definitions) =>
103         val iter = groups.iterator
104         var group = iter.next
105         // last group has all the columns, so just return the last one if nothing else matches
106         while (iter.hasNext && !supports(group._2, definitions, ecql)) {
107           group = iter.next
108         }
109         group
110     }
111   }

113   /**
114     * Validate that the column groups do not overlap with reserved column groups
115     *
116     * @param sft simple feature type
117     */
118   def validate(sft: SimpleFeatureType): Unit = {
119     val groups = sft.getAttributeDescriptors.asScala.flatMap(_.getColumnGroups()).distinct
120     groups.foreach { group =>
121       if (group == ColumnGroups.DefaultString || group == ColumnGroups.AttributesString) {
122         throw new IllegalArgumentException(s"Column group '$group' is reserved for internal use - " +
123             "please choose another name")
124       }
125     }
126     if (sft.getVisibilityLevel == VisibilityLevel.Attribute && groups.nonEmpty) {
127       throw new IllegalArgumentException("Column groups are not supported when using attribute-level visibility")
128     }
129   }

131   /**
132    * Does the simple feature type contain the fields required to evaluate the transform and filter
133    *
134    * @param sft simple feature type
135    * @param transforms transform definitions
136    * @param filter filter
137    * @return
138    */
139   private def supports(sft: SimpleFeatureType, transforms: Seq[Transform], filter: Option[Filter]): Boolean = {
140     filter.forall(FilterHelper.propertyNames(_, sft).forall(sft.indexOf(_) != -1)) &&
141         transforms.flatMap(_.properties).forall(sft.indexOf(_) != -1)
142   }
