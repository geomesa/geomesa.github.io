22 /**
23   * Estimates frequency counts at scale. Tracks geometry and date attributes as a single value.
24   *
25   * @param sft simple feature type
26   * @param geom geometry attribute in the sft
27   * @param dtg date attribute in the sft
28   * @param period time period to use for z index
29   * @param precision number of bits of z-index that will be used
30   * @param eps (epsilon) with probability at least @see confidence, estimates will be within eps * N
31   * @param confidence percent - with probability at least confidence, estimates will be within @see eps * N
32   */
33 class Z3Frequency(
34     val sft: SimpleFeatureType,
35     val geom: String,
36     val dtg: String,
37     val period: TimePeriod,
38     val precision: Int,
39     val eps: Double = 0.005,
40     val confidence: Double = 0.95
41   ) extends Stat with LazyLogging {
42 
43   override type S = Z3Frequency
44 
45   private val g = sft.indexOf(geom)
46   private val d = sft.indexOf(dtg)
47 
48   private val mask = Frequency.getMask(precision)
49   private val sfc = Z3SFC(period)
50   private val timeToBin = BinnedTime.timeToBinnedTime(period)
51 
52   private [stats] val sketches = scala.collection.mutable.Map.empty[Short, CountMinSketch]
53   private [stats] def newSketch: CountMinSketch = CountMinSketch(eps, confidence, Frequency.Seed)
54 
55   private def toKey(geom: Geometry, dtg: Date): (Short, Long) = {
56     import org.locationtech.geomesa.utils.geotools.Conversions.RichGeometry
57     val BinnedTime(b, o) = timeToBin(dtg.getTime)
58     val centroid = geom.safeCentroid()
59     val z = sfc.index(centroid.getX, centroid.getY, o) & mask
60     (b, z)
61   }
62 
63   /**
64     * Gets the count for the given values
65     *
66     * @param geom geometry
67     * @param dtg date
68     * @return count of the values
69     */
70   def count(geom: Geometry, dtg: Date): Long = {
71     val (bin, z3) = toKey(geom, dtg)
72     countDirect(bin, z3)
73   }
74 
75   /**
76     * Gets the count for a time bin and z3. Useful if the values are known ahead of time.
77     *
78     * @param bin period since the epoch
79     * @param z3 z value
80     * @return count of the values
81     */
82   def countDirect(bin: Short, z3: Long): Long = sketches.get(bin).map(_.estimateCount(z3)).getOrElse(0L)
83 
84   /**
85     * Number of observations in the frequency map
86     *
87     * @return number of observations
88     */
89   def size: Long = sketches.values.map(_.size).sum
90 
91   /**
92     * Split the stat into a separate stat per time bin of z data. Allows for separate handling of the reduced
93     * data set.
94     *
95     * @return
96     */
97   def splitByTime: Seq[(Short, Z3Frequency)] = {
98     sketches.toSeq.map { case (w, sketch) =>
99       val freq = new Z3Frequency(sft, geom, dtg, period, precision, eps, confidence)
100       freq.sketches.put(w, sketch)
101       (w, freq)
102     }
103   }
104 
105   override def observe(sf: SimpleFeature): Unit = {
106     val geom = sf.getAttribute(g).asInstanceOf[Geometry]
107     val dtg  = sf.getAttribute(d).asInstanceOf[Date]
108     if (geom != null && dtg != null) {
109       try {
110         val (bin, z3) = toKey(geom, dtg)
111         sketches.getOrElseUpdate(bin, newSketch).add(z3, 1L)
112       } catch {
113         case e: Exception => logger.warn(s"Error observing geom '$geom' and date '$dtg': ${e.toString}")
114       }
115     }
116   }
117 
118   // no-op
119   override def unobserve(sf: SimpleFeature): Unit = {}
120 
121   override def +(other: Z3Frequency): Z3Frequency = {
122     val plus = new Z3Frequency(sft, geom, dtg, period, precision, eps, confidence)
123     plus += this
124     plus += other
125     plus
126   }
127 
128   override def +=(other: Z3Frequency): Unit = {
129     other.sketches.filter(_._2.size > 0).foreach { case (w, sketch) =>
130       sketches.getOrElseUpdate(w, newSketch) += sketch
131     }
132   }
133 
134   override def clear(): Unit = sketches.values.foreach(_.clear())
135 
136   override def isEmpty: Boolean = sketches.values.forall(_.size == 0)
137 
138   override def toJsonObject: Any = {
139     val (e, c) = sketches.values.headOption.map(s => (s.eps, s.confidence)).getOrElse((0d, 0d))
140     ListMap("eps" -> e, "confidence" -> c, "size" -> size)
141   }
142 
143   override def isEquivalent(other: Stat): Boolean = {
144     other match {
145       case s: Z3Frequency =>
146         geom == s.geom && dtg == s.dtg && period == s.period && precision == s.precision && {
147           val nonEmpty = sketches.filter(_._2.size > 0)
148           val sNonEmpty = s.sketches.filter(_._2.size > 0)
149           nonEmpty.keys == sNonEmpty.keys && nonEmpty.keys.forall { k =>
150             sketches(k).isEquivalent(s.sketches(k))
151           }
152         }
153       case _ => false
154     }
155   }
