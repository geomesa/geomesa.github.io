26 /**
27   * Plan for querying a GeoMesaDataStore
28   *
29   * @tparam DS type of this data store
30   */
31 trait QueryPlan[DS <: GeoMesaDataStore[DS]] {
32 
33   /**
34     * Type of raw results returned from the underlying database
35     */
36   type Results
37 
38   /**
39     * Reference back to the strategy
40     *
41     * @return
42     */
43   def filter: FilterStrategy
44 
45   /**
46     * Runs the query plan against the underlying database
47     *
48     * @param ds data store - provides connection object and metadata
49     * @return
50     */
51   def scan(ds: DS): CloseableIterator[Results]
52 
53   /**
54     * Transform results coming back from a raw scan into features
55     *
56     * @return
57     */
58   def resultsToFeatures: ResultsToFeatures[Results]
59 
60   /**
61     * Optional reduce step for simple features coming back
62     *
63     * @return
64     */
65   def reducer: Option[FeatureReducer]
66 
67   /**
68     * Sort fields and reverse order flags (true flips order to descending, false keeps order as ascending)
69     *
70     * @return
71     */
72   def sort: Option[Seq[(String, Boolean)]]
73 
74   /**
75     * Max features to return from the scan
76     *
77     * @return
78     */
79   def maxFeatures: Option[Int]
80 
81   /**
82     * Geometry projection
83     *
84     * @return
85     */
86   def projection: Option[QueryReferenceSystems]
87 
88   /**
89     * Explains details on how this query plan will be executed
90     *
91     * @param explainer explainer to use for explanation
92     * @param prefix prefix for explanation lines, used for nesting explanations
93     */
94   def explain(explainer: Explainer, prefix: String = ""): Unit
95 }

251   /**
252     * Abstract base class for converting the results from a normal feature index
253     *
254     * @param index index
255     * @param sft simple feature type returned from the scan
256     * @tparam T result type
257     */
258   abstract class IndexResultsToFeatures[T](
259       protected var index: GeoMesaFeatureIndex[_, _],
260       protected var sft: SimpleFeatureType
261     ) extends ResultsToFeatures[T] {
262 
263     protected var serializer: KryoFeatureSerializer = if (index == null) { null } else { createSerializer }
264 
265     override def init(state: Map[String, String]): Unit = {
266       val spec = state("spec")
267       sft = SimpleFeatureTypes.createType(state("name"), spec)
268       index = state.get("isft") match {
269         case None => IteratorCache.index(sft, spec, state("idx"))
270         case Some(isft) => IteratorCache.index(IteratorCache.sft(isft), isft, state("idx"))
271       }
272       serializer = createSerializer
273     }
274 
275     override def state: Map[String, String] = {
276       val base = Map(
277         "name" -> sft.getTypeName,
278         "spec" -> SimpleFeatureTypes.encodeType(sft, includeUserData = true),
279         "idx"  -> index.identifier
280       )
281       if (index.sft == sft) { base } else {
282         base.updated("isft", SimpleFeatureTypes.encodeType(index.sft, includeUserData = true))
283       }
284     }
285 
286     override def schema: SimpleFeatureType = sft
287 
288     protected def createSerializer: KryoFeatureSerializer = {
289       val hasAttributeLevelVis = sft.getVisibilityLevel == VisibilityLevel.Attribute
290       val builder = KryoFeatureSerializer.builder(sft)
291       // note: attribute level vis are serialized into the user data since they can't be held by a single key
292       val toBuild = (index.serializedWithId, hasAttributeLevelVis) match {
293         case (true, true)   => builder.withId.withUserData
294         case (true, false)  => builder.withId.withoutUserData
295         case (false, true)  => builder.withoutId.withUserData
296         case (false, false) => builder.withoutId.withoutUserData
297       }
298       toBuild.build()
299     }
300 
301     def canEqual(other: Any): Boolean = other.isInstanceOf[IndexResultsToFeatures[T]]
302 
303     override def equals(other: Any): Boolean = other match {
304       case that: IndexResultsToFeatures[T] if that.canEqual(this) =>
305         sft == that.sft && {
306           if (index == null) { that.index == null } else if (that.index == null) { false } else {
307             index.identifier == that.index.identifier && index.sft == that.index.sft
308           }
309         }
310 
311       case _ => false
312     }
313 
314     override def hashCode(): Int = {
315       val state = Seq(index, sft)
316       state.map(Objects.hashCode).foldLeft(0)((a, b) => 31 * a + b)
317     }
318   }
