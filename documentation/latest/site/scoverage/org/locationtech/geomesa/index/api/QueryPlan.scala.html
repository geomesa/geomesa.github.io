26 /**
27   * Plan for querying a GeoMesaDataStore
28   *
29   * @tparam DS type of this data store
30   */
31 trait QueryPlan[DS <: GeoMesaDataStore[DS]] {
32 
33   /**
34     * Type of raw results returned from the underlying database
35     */
36   type Results
37 
38   /**
39     * Reference back to the query strategy
40     *
41     * @return
42     */
43   def strategy: QueryStrategy
44 
45   /**
46    * Reference back to the filter strategy
47    *
48    * @return
49    */
50   def filter: FilterStrategy = strategy.filter
51 
52   /**
53     * Runs the query plan against the underlying database
54     *
55     * @param ds data store - provides connection object and metadata
56     * @return
57     */
58   def scan(ds: DS): CloseableIterator[Results]
59 
60   /**
61     * Transform results coming back from a raw scan into features
62     *
63     * @return
64     */
65   def resultsToFeatures: ResultsToFeatures[Results]
66 
67   /**
68     * Optional reduce step for simple features coming back
69     *
70     * @return
71     */
72   def reducer: Option[FeatureReducer]
73 
74   /**
75     * Sort fields and reverse order flags (true flips order to descending, false keeps order as ascending)
76     *
77     * @return
78     */
79   def sort: Option[Seq[(String, Boolean)]]
80 
81   /**
82     * Max features to return from the scan
83     *
84     * @return
85     */
86   def maxFeatures: Option[Int]
87 
88   /**
89     * Geometry projection
90     *
91     * @return
92     */
93   def projection: Option[QueryReferenceSystems]
94 
95   /**
96     * Explains details on how this query plan will be executed
97     *
98     * @param explainer explainer to use for explanation
99     * @param prefix prefix for explanation lines, used for nesting explanations
100     */
101   def explain(explainer: Explainer, prefix: String = ""): Unit
102 }

127       *
128       * @param result result
129       * @return
130       */

132   }

258   /**
259     * Abstract base class for converting the results from a normal feature index
260     *
261     * @param index index
262     * @param sft simple feature type returned from the scan
263     * @tparam T result type
264     */
265   abstract class IndexResultsToFeatures[T](
266       protected var index: GeoMesaFeatureIndex[_, _],
267       protected var sft: SimpleFeatureType
268     ) extends ResultsToFeatures[T] {
269 
270     protected var serializer: KryoFeatureSerializer = if (index == null) { null } else { createSerializer }
271 
272     override def init(state: Map[String, String]): Unit = {
273       val spec = state("spec")
274       sft = SimpleFeatureTypes.createType(state("name"), spec)
275       index = state.get("isft") match {
276         case None => IteratorCache.index(sft, spec, state("idx"))
277         case Some(isft) => IteratorCache.index(IteratorCache.sft(isft), isft, state("idx"))
278       }
279       serializer = createSerializer
280     }
281 
282     override def state: Map[String, String] = {
283       val base = Map(
284         "name" -> sft.getTypeName,
285         "spec" -> SimpleFeatureTypes.encodeType(sft, includeUserData = true),
286         "idx"  -> index.identifier
287       )
288       if (index.sft == sft) { base } else {
289         base.updated("isft", SimpleFeatureTypes.encodeType(index.sft, includeUserData = true))
290       }
291     }
292 
293     override def schema: SimpleFeatureType = sft
294 
295     protected def createSerializer: KryoFeatureSerializer = {
296       val hasAttributeLevelVis = sft.getVisibilityLevel == VisibilityLevel.Attribute
297       val builder = KryoFeatureSerializer.builder(sft)
298       // note: attribute level vis are serialized into the user data since they can't be held by a single key
299       val toBuild = (index.serializedWithId, hasAttributeLevelVis) match {
300         case (true, true)   => builder.withId.withUserData
301         case (true, false)  => builder.withId.withoutUserData
302         case (false, true)  => builder.withoutId.withUserData
303         case (false, false) => builder.withoutId.withoutUserData
304       }
305       toBuild.build()
306     }
307 
308     def canEqual(other: Any): Boolean = other.isInstanceOf[IndexResultsToFeatures[T]]
309 
310     override def equals(other: Any): Boolean = other match {
311       case that: IndexResultsToFeatures[T] if that.canEqual(this) =>
312         sft == that.sft && {
313           if (index == null) { that.index == null } else if (that.index == null) { false } else {
314             index.identifier == that.index.identifier && index.sft == that.index.sft
315           }
316         }
317 
318       case _ => false
319     }
320 
321     override def hashCode(): Int = {
322       val state = Seq(index, sft)
323       state.map(Objects.hashCode).foldLeft(0)((a, b) => 31 * a + b)
324     }
325   }
