84     def toSortHint(sortBy: Array[SortBy]): String = {
85       val strings = sortBy.flatMap { sb =>
86         val prop = sb.getPropertyName
87         Seq(if (prop == null) { "" } else { prop.getPropertyName }, s"${sb.getSortOrder == SortOrder.DESCENDING}")
88       }
89       StringSerialization.encodeSeq(strings)
90     }
