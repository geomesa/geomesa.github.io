81     def toSortHint(sortBy: Array[SortBy]): String = {
82       val strings = sortBy.flatMap { sb =>
83         val prop = sb.getPropertyName
84         Seq(if (prop == null) { "" } else { prop.getPropertyName }, s"${sb.getSortOrder == SortOrder.DESCENDING}")
85       }
86       StringSerialization.encodeSeq(strings)
87     }
88 
89     def fromSortHint(hint: String): Seq[(String, Boolean)] =
90       StringSerialization.decodeSeq(hint).grouped(2).toSeq.map { case Seq(f, r) => (f, r.toBoolean) }

92     def toProjectionHint(crs: QueryReferenceSystems): String =

95     def fromProjectionHint(hint: String): QueryReferenceSystems = {
