83     def toSortHint(sortBy: Array[SortBy]): String = {
84       val strings = sortBy.flatMap { sb =>
85         val prop = sb.getPropertyName
86         Seq(if (prop == null) { "" } else { prop.getPropertyName }, s"${sb.getSortOrder == SortOrder.DESCENDING}")
87       }
88       StringSerialization.encodeSeq(strings)
89     }

91     def fromSortHint(hint: String): Seq[(String, Boolean)] =
