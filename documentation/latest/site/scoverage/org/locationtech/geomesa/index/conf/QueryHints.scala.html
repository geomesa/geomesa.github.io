85     def toSortHint(sortBy: Array[SortBy]): String = {
86       val strings = sortBy.flatMap { sb =>
87         val prop = sb.getPropertyName
88         Seq(if (prop == null) { "" } else { prop.getPropertyName }, s"${sb.getSortOrder == SortOrder.DESCENDING}")
89       }
90       StringSerialization.encodeSeq(strings)
91     }
