31   override def addFeatures(collection: FeatureCollection[SimpleFeatureType, SimpleFeature]): java.util.List[FeatureId] = {
32     if (collection.isEmpty) {
33       return Collections.emptyList[FeatureId]()
34     }
35 
36     val fids = new java.util.ArrayList[FeatureId](collection.size())
37     val errors = ArrayBuffer.empty[Throwable]
38 
39     WithClose(collection.features, writer(None)) { case (features, writer) =>
40       while (features.hasNext) {
41         try { fids.add(FeatureUtils.write(writer, features.next()).getIdentifier) } catch {
42           // validation errors in indexing will throw an IllegalArgumentException
43           // make the caller handle other errors, which are likely related to the underlying database,
44           // as we wouldn't know which features were actually written or not due to write buffering
45           case e: IllegalArgumentException => errors.append(e)
46         }
47       }
48     }
49 
50     if (errors.isEmpty) { fids } else {
51       val e = new IllegalArgumentException("Some features were not written:")
52       // suppressed exceptions should contain feature ids and attributes
53       errors.foreach(e.addSuppressed)
54       throw e
55     }
56   }
57 
58   override def setFeatures(reader: FeatureReader[SimpleFeatureType, SimpleFeature]): Unit = {
59     removeFeatures(Filter.INCLUDE)
60     try {
61       WithClose(writer(None)) { writer =>
62         while (reader.hasNext) {
63           FeatureUtils.write(writer, reader.next())
64         }
65       }
66     } finally {
67       reader.close()
68     }
69   }

80   override def modifyFeatures(attributes: Array[Name], values: Array[AnyRef], filter: Filter): Unit = {
81     val sft = getSchema
82 
83     require(filter != null, "Filter must not be null")
84     attributes.foreach(a => require(sft.getDescriptor(a) != null, s"$a is not an attribute of ${sft.getName}"))
85     require(attributes.length == values.length, "Modified names and values don't match")
86 
87     WithClose(writer(Some(filter))) { writer =>
88       while (writer.hasNext) {
89         val sf = writer.next()
90         var i = 0
91         while (i < attributes.length) {
92           try {
93             sf.setAttribute(attributes(i), values(i))
94           } catch {
95             case e: Exception =>
96               throw new DataSourceException("Error updating feature " +
97                   s"'${sf.getID}' with ${attributes(i)}=${values(i)}", e)
98           }
99           i += 1
100         }
101         writer.write()
102       }
103     }
104   }

106   override def removeFeatures(filter: Filter): Unit = {
107     ds match {
108       case gm: GeoMesaDataStore[_] if filter == Filter.INCLUDE =>
109         val indices = gm.manager.indices(sft).toList
110         if (TablePartition.partitioned(sft)) {
111           def deleteOne(index: GeoMesaFeatureIndex[_, _]): Unit =
112             gm.adapter.deleteTables(index.deleteTableNames(None))
113           indices.map(i => CachedThreadPool.submit(() => deleteOne(i))).foreach(_.get)
114         } else {
115           def deleteOne(index: GeoMesaFeatureIndex[_, _]): Unit = {
116             val prefix = Some(index.keySpace.sharing).filterNot(_.isEmpty)
117             gm.adapter.clearTables(index.getTableNames(), prefix)
118           }
119           indices.map(i => CachedThreadPool.submit(() => deleteOne(i))).foreach(_.get)
120         }
121         gm.stats.writer.clear(sft)
122 
123       case _ =>
124         WithClose(writer(Some(filter))) { writer =>
125           while (writer.hasNext) {
126             writer.next()
127             writer.remove()
128           }
129         }
130     }
131   }
132

146   private def writer(filter: Option[Filter]): FeatureWriter[SimpleFeatureType, SimpleFeature] = {
147     ds match {
148       case gm: MetadataBackedDataStore => gm.getFeatureWriter(sft, transaction, filter)
149       case _ if filter.isEmpty => ds.getFeatureWriterAppend(sft.getTypeName, transaction)
150       case _ => ds.getFeatureWriter(sft.getTypeName, filter.get, transaction)
151     }
152   }
