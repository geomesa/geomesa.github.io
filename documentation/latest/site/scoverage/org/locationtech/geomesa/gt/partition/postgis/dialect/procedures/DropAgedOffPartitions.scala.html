15 /**
16  * Drops any partitions older than the configured maximum
17  */
18 object DropAgedOffPartitions extends SqlProcedure {
19 
20   override def name(info: TypeInfo): FunctionName = FunctionName(s"${info.typeName}_age_off")
21 
22   override protected def createStatements(info: TypeInfo): Seq[String] = Seq(proc(info))
23 
24   private def proc(info: TypeInfo): String = {
25     s"""CREATE OR REPLACE PROCEDURE ${info.schema.quoted}.${name(info).quoted}(cur_time timestamp without time zone) LANGUAGE plpgsql AS
26        |  $$BODY$$
27        |    DECLARE
28        |      num_partitions int;                          -- number of partitions to keep
29        |      partition_size int;                          -- size of the partition, in hours
30        |      main_cutoff timestamp without time zone;     -- max age of the records for main tables
31        |      partition_start timestamp without time zone; -- start bounds for the partition we're writing
32        |      partition_name regclass;                     -- partition table name
33        |    BEGIN
34        |      SELECT value::int FROM ${info.schema.quoted}.${UserDataTable.Name.quoted}
35        |        WHERE key = ${literal(SftUserData.MaxPartitions.key)}
36        |          AND type_name = ${literal(info.typeName)}
37        |        INTO num_partitions;
38        |      IF FOUND THEN
39        |        SELECT COALESCE(
40        |          (SELECT value::int FROM ${info.schema.quoted}.${UserDataTable.Name.quoted}
41        |            WHERE type_name = ${literal(info.typeName)} AND key = ${literal(SftUserData.IntervalHours.key)}),
42        |          ${SftUserData.IntervalHours.default})
43        |          INTO partition_size;
44        |        main_cutoff := ${info.schema.quoted}.truncate_to_partition(cur_time, partition_size) - make_interval(hours => partition_size);
45        |        -- remove any partitions that have aged out
46        |        partition_start := main_cutoff - (make_interval(hours => partition_size) * num_partitions);
47        |        FOR partition_name IN
48        |          SELECT relid
49        |            FROM pg_partition_tree(${info.tables.mainPartitions.name.asRegclass})
50        |            WHERE parentrelid IS NOT NULL
51        |            AND (SELECT relname FROM pg_class WHERE oid = relid) <= ${literal(info.tables.mainPartitions.name.raw + "_")} || to_char(partition_start, 'YYYY_MM_DD_HH24')
52        |          UNION ALL
53        |            SELECT relid
54        |              FROM pg_partition_tree(${info.tables.spillPartitions.name.asRegclass})
55        |              WHERE parentrelid IS NOT NULL
56        |              AND (SELECT relname FROM pg_class WHERE oid = relid) <= ${literal(info.tables.spillPartitions.name.raw + "_")} || to_char(partition_start, 'YYYY_MM_DD_HH24')
57        |        LOOP
58        |          RAISE NOTICE 'Aging off partition %', partition_name;
59        |          IF EXISTS(SELECT FROM pg_class WHERE oid = partition_name) THEN
60        |            -- cast to text will handle quoting the table name appropriately
61        |            EXECUTE 'DROP TABLE IF EXISTS ${info.schema.quoted}.' || partition_name::text;
62        |            RAISE NOTICE 'A partition has been deleted %', partition_name;
63        |          END IF;
64        |        END LOOP;
65        |      END IF;
66        |    END;
67        |  $$BODY$$;
68        |""".stripMargin
69   }
70 }
