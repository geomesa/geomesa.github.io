12 /**
13  * Write ahead log, partitioned using inheritance to avoid any restraints on data overlap between partitions.
14  * All writes get directed to the main partition, which is identified with the suffix `_writes`. Every 10 minutes,
15  * the current writes partition is renamed based on a sequence number, and a new writes partition is created
16  */

43          |    ' ON ${info.schema.quoted}.' || quote_ident(partition) || ' (${info.cols.dtg.quoted})$tableTs';
44          |${info.cols.geoms.map { col =>
