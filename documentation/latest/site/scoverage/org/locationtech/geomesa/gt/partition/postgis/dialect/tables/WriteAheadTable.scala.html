12 /**
13  * Write ahead log, partitioned using inheritance to avoid any restraints on data overlap between partitions.
14  * All writes get directed to the main partition, which is identified with the suffix `_writes`. Every 10 minutes,
15  * the current writes partition is renamed based on a sequence number, and a new writes partition is created
16  */
17 object WriteAheadTable extends SqlStatements {

19   override protected def createStatements(info: TypeInfo): Seq[String] = {
20     val table = info.tables.writeAhead
21     val (tableTs, indexTs) = table.tablespace match {
22       case None => ("", "")
23       case Some(ts) => (s" TABLESPACE ${ts.quoted}", s" USING INDEX TABLESPACE ${ts.quoted}")
24     }
25     val logging = if (info.tables.writeAhead.logged) { "" } else { "UNLOGGED" }

29     val block =

32          |  seq_val smallint;
33          |  partition text;
34          |BEGIN
35          |  SELECT value from ${info.schema.quoted}.${SequenceTable.Name.quoted}
36          |    WHERE type_name = ${literal(info.typeName)} INTO seq_val;
37          |  partition := ${literal(table.name.raw + "_")} || lpad(seq_val::text, 3, '0');
38          |
39          |  EXECUTE 'CREATE $logging TABLE IF NOT EXISTS ${info.schema.quoted}.' || quote_ident(partition) || '(' ||
40          |    'CONSTRAINT ' || quote_ident(partition || '_pkey') ||
41          |    ' PRIMARY KEY (fid, ${info.cols.dtg.quoted})$indexTs ' ||
42          |    ') INHERITS (${table.name.qualified})${table.storage.opts}$tableTs';
43          |  EXECUTE 'CREATE INDEX IF NOT EXISTS ' || quote_ident(partition || '_' || ${info.cols.dtg.asLiteral}) ||
44          |    ' ON ${info.schema.quoted}.' || quote_ident(partition) || ' (${info.cols.dtg.quoted})$tableTs';
45          |${info.cols.geoms.map { col =>
46       s"""  EXECUTE 'CREATE INDEX IF NOT EXISTS ' || quote_ident(partition || '_spatial_' || ${col.asLiteral}) ||
47          |    ' ON ${info.schema.quoted}.' || quote_ident(partition) || ' USING gist(${col.quoted})$tableTs';""".stripMargin}.mkString("\n")}
48          |${info.cols.indexed.map { col =>
49       s"""  EXECUTE 'CREATE INDEX IF NOT EXISTS ' || quote_ident(partition || '_' || ${col.asLiteral}) ||
50          |    ' ON ${info.schema.quoted}.' || quote_ident(partition) || '(${col.quoted})$tableTs';""".stripMargin}.mkString("\n")}
51          |END $$$$;""".stripMargin
52 
53     move ++ Seq(block)
54   }

56   override protected def dropStatements(info: TypeInfo): Seq[String] = {

60     )
61   }
