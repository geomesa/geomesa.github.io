33   override def createDataSource(params: java.util.Map[String, _]): BasicDataSource = {
34     val source = super.createDataSource(params)

36       Seq(IdleInTransactionTimeout)

47   override protected def createDataStoreInternal(store: JDBCDataStore, baseParams: java.util.Map[String, _]): JDBCDataStore = {
48     val params = new java.util.HashMap[String, Any](baseParams)
49     // default to using prepared statements, if not specified
50     if (!params.containsKey(PreparedStatements.key)) {
51       params.put(PreparedStatements.key, java.lang.Boolean.TRUE)
52     }
53     // set default schema, if not specified - postgis store doesn't actually use its own default
54     if (!params.containsKey(PostgisNGDataStoreFactory.SCHEMA.key)) {
55       // need to set it in the store, as the key has already been processed
56       store.setDatabaseSchema("public")
57       // also set in the params for consistency, although it's not used anywhere
58       params.put(PostgisNGDataStoreFactory.SCHEMA.key, "public")
59     }
60     val ds = super.createDataStoreInternal(store, params)
61     val dialect = new PartitionedPostgisDialect(ds)
62 
63     ds.getSQLDialect match {
64       case d: PostGISDialect =>
65         dialect.setEncodeBBOXFilterAsEnvelope(d.isEncodeBBOXFilterAsEnvelope)
66         dialect.setEstimatedExtentsEnabled(d.isEstimatedExtentsEnabled)
67         dialect.setFunctionEncodingEnabled(d.isFunctionEncodingEnabled)
68         dialect.setLooseBBOXEnabled(d.isLooseBBOXEnabled)
69         dialect.setSimplifyEnabled(d.isSimplifyEnabled)
70         ds.setSQLDialect(dialect)
71 
72       case d: PostGISPSDialect =>
73         dialect.setEncodeBBOXFilterAsEnvelope(d.isEncodeBBOXFilterAsEnvelope)
74         dialect.setFunctionEncodingEnabled(d.isFunctionEncodingEnabled)
75         dialect.setLooseBBOXEnabled(d.isLooseBBOXEnabled)
76 
77         // these configs aren't exposed through the PS dialect so re-calculate them from the params
78         val est = PostgisNGDataStoreFactory.ESTIMATED_EXTENTS.lookUp(params)
79         dialect.setEstimatedExtentsEnabled(est == null || est == java.lang.Boolean.TRUE)
80         val simplify = PostgisNGDataStoreFactory.SIMPLIFY.lookUp(params)
81         dialect.setSimplifyEnabled(simplify == null || simplify == java.lang.Boolean.TRUE)
82 
83         ds.setSQLDialect(new PartitionedPostgisPsDialect(ds, dialect))
84 
85       case d => throw new IllegalArgumentException(s"Expected PostGISDialect but got: ${d.getClass.getName}")
86     }
87 
88     ds
89   }
90
