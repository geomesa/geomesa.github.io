27   override protected def setupParameters(parameters: java.util.Map[String, AnyRef]): Unit = {

33   override def createDataSource(params: java.util.Map[String, _]): BasicDataSource = {
34     val source = super.createDataSource(params)
35     val options =
36       Seq(IdleInTransactionTimeout)
37           .flatMap(p => p.opt(params).map(t => s"-c ${p.key}=${t.millis}"))
38 
39     logger.debug(s"Connection options: ${options.mkString(" ")}")
40 
41     if (options.nonEmpty) {
42       source.addConnectionProperty("options", options.mkString(" "))
43     }
44     source
45   }

47   override protected def createDataStoreInternal(store: JDBCDataStore, baseParams: java.util.Map[String, _]): JDBCDataStore = {
48     val params = new java.util.HashMap[String, Any](baseParams)
49     // default to using prepared statements, if not specified
50     if (!params.containsKey(PreparedStatements.key)) {
51       params.put(PreparedStatements.key, java.lang.Boolean.TRUE)
52     }
53     // set default schema, if not specified - postgis store doesn't actually use its own default
54     if (!params.containsKey(PostgisNGDataStoreFactory.SCHEMA.key)) {
55       // need to set it in the store, as the key has already been processed
56       store.setDatabaseSchema("public")
57       // also set in the params for consistency, although it's not used anywhere
58       params.put(PostgisNGDataStoreFactory.SCHEMA.key, "public")
59     }
60     val roles = Option(ReadAccessRoles.lookUp(params).asInstanceOf[String]).toSeq.flatMap(_.split(",")).map(r => RoleName(r.trim))
61 
62     val ds = super.createDataStoreInternal(store, params)
63     val dialect = new PartitionedPostgisDialect(ds, roles)
64 
65     ds.getSQLDialect match {
66       case d: PostGISDialect =>
67         dialect.setEncodeBBOXFilterAsEnvelope(d.isEncodeBBOXFilterAsEnvelope)
68         dialect.setEstimatedExtentsEnabled(d.isEstimatedExtentsEnabled)
69         dialect.setFunctionEncodingEnabled(d.isFunctionEncodingEnabled)
70         dialect.setLooseBBOXEnabled(d.isLooseBBOXEnabled)
71         dialect.setSimplifyEnabled(d.isSimplifyEnabled)
72         ds.setSQLDialect(dialect)
73 
74       case d: PostGISPSDialect =>
75         dialect.setEncodeBBOXFilterAsEnvelope(d.isEncodeBBOXFilterAsEnvelope)
76         dialect.setFunctionEncodingEnabled(d.isFunctionEncodingEnabled)
77         dialect.setLooseBBOXEnabled(d.isLooseBBOXEnabled)
78 
79         // these configs aren't exposed through the PS dialect so re-calculate them from the params
80         val est = PostgisNGDataStoreFactory.ESTIMATED_EXTENTS.lookUp(params)
81         dialect.setEstimatedExtentsEnabled(est == null || est == java.lang.Boolean.TRUE)
82         val simplify = PostgisNGDataStoreFactory.SIMPLIFY.lookUp(params)
83         dialect.setSimplifyEnabled(simplify == null || simplify == java.lang.Boolean.TRUE)
84 
85         ds.setSQLDialect(new PartitionedPostgisPsDialect(ds, dialect))
86 
87       case d => throw new IllegalArgumentException(s"Expected PostGISDialect but got: ${d.getClass.getName}")
88     }
89 
90     ds
91   }
92

96     new PostGISDialect(dataStore)
