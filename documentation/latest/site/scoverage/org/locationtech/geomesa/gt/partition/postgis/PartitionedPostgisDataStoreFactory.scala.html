51   override protected def createDataSource(params: java.util.Map[String, _], dialect: SQLDialect): DataSource = {
52     val dataSource = super.createDataSource(params, dialect)
53 
54     val metrics = MetricsRegistryParam.lookupRegistry(params).flatMap { registry =>
55       var metrics: MetricsDataSource = null
56       try {
57         // try to re-create the datasource as a commons-dbcp2 instance so we can instrument it
58         val basic = dataSource match {
59           case ds: AbstractManageableDataSource => ds.unwrap(classOf[BasicDataSource])
60           case ds: BasicDataSource => ds
61           case ds => throw new UnsupportedOperationException(s"Unexpected data source class: ${ds.getClass.getName} $ds")
62         }
63         metrics = new MetricsDataSource("postgis")
64 
65         // these are the options that are configured in our parent factories
66         metrics.setDriverClassName(basic.getDriverClassName)
67         metrics.setUrl(basic.getUrl)
68         metrics.setUsername(basic.getUsername)
69         metrics.setPassword(basic.getPassword)
70         metrics.setMaxWait(Duration.ofMillis(basic.getMaxWait))
71         metrics.setMinIdle(basic.getMinIdle)
72         metrics.setMaxTotal(basic.getMaxActive)
73         metrics.setTestOnBorrow(basic.getTestOnBorrow)
74         metrics.setValidationQuery(basic.getValidationQuery)
75         metrics.setTestWhileIdle(basic.getTestWhileIdle)
76         metrics.setDurationBetweenEvictionRuns(Duration.ofMillis(basic.getTimeBetweenEvictionRunsMillis))
77         metrics.setMinEvictableIdle(Duration.ofMillis(basic.getMinEvictableIdleTimeMillis))
78         metrics.setNumTestsPerEvictionRun(basic.getNumTestsPerEvictionRun)
79         metrics.setAccessToUnderlyingConnectionAllowed(basic.isAccessToUnderlyingConnectionAllowed)
80         metrics.setMaxOpenPreparedStatements(basic.getMaxOpenPreparedStatements)
81         metrics.setPoolPreparedStatements(basic.isPoolPreparedStatements)
82 
83         // re-add connection properties, as we can't access those from the BasicDataSource
84         createConnectionOptions(params).foreach { case (k, v) => metrics.addConnectionProperty(k, v) }
85 
86         val registration = registry.register()
87 
88         tryBindPostgresMetrics(metrics, params)
89 
90         // we attach the registry to the ManagedDataSource b/c that's the only hook we have into datastore.dispose()
91         Some(new Dbcp2ManagedDataSource(metrics, registration))
92       } catch {
93         case NonFatal(e) =>
94           logger.warn("Unable to instrument data source, metrics will not be available:", e)
95           CloseWithLogging(Option(metrics))
96           None
97       }
98     }
99 
100     metrics.getOrElse(dataSource)
101   }

103   private def tryBindPostgresMetrics(dataSource: MetricsDataSource, params: java.util.Map[String, _]): Unit = {
104     try {
105       val conf =
106         MetricsRegistryConfigParam.lookupOpt(params)
107           .getOrElse(ConfigFactory.empty)
108           .withFallback(ConfigFactory.load("postgres-metrics"))
109           .getConfig("postgres")
110       if (conf.getBoolean("enabled")) {
111         val tags = Tags.of(conf.getConfig("tags").root().unwrapped().asScala.map { case (k, v) => Tag.of(k, v.toString) }.asJava)
112         val db =
113           Option(DATABASE.lookUp(params).asInstanceOf[String])
114             .orElse(Option(USER.lookUp(params).asInstanceOf[String]))
115             .getOrElse(throw new IllegalArgumentException(s"${DATABASE.key} was not found in the parameter map"))
116         if (PartitionedPostgisDataStoreFactory.bindings.add(db -> tags)) {
117           logger.info("Adding postgres metrics")
118           new PostgreSQLDatabaseMetrics(dataSource, db, tags).bindTo(Metrics.globalRegistry)
119         }
120       }
121     } catch {
122       case NonFatal(e) => logger.warn("Error binding postgres metrics:", e)
123     }
124   }

132   override protected def createDataStoreInternal(store: JDBCDataStore, baseParams: java.util.Map[String, _]): JDBCDataStore = {
133     val params = new java.util.HashMap[String, Any](baseParams)
134     // default to using prepared statements, if not specified
135     if (!params.containsKey(PreparedStatements.key)) {
136       params.put(PreparedStatements.key, java.lang.Boolean.TRUE)
137     }
138     // set default schema, if not specified - postgis store doesn't actually use its own default
139     if (!params.containsKey(PostgisNGDataStoreFactory.SCHEMA.key)) {
140       // need to set it in the store, as the key has already been processed
141       store.setDatabaseSchema("public")
142       // also set in the params for consistency, although it's not used anywhere
143       params.put(PostgisNGDataStoreFactory.SCHEMA.key, "public")
144     }
145     val roles = Option(ReadAccessRoles.lookUp(params).asInstanceOf[String]).toSeq.flatMap(_.split(",")).map(r => RoleName(r.trim))
146 
147     val ds = super.createDataStoreInternal(store, params)
148     val dialect = new PartitionedPostgisDialect(ds, roles)
149 
150     ds.getSQLDialect match {
151       case d: PostGISDialect =>
152         dialect.setEncodeBBOXFilterAsEnvelope(d.isEncodeBBOXFilterAsEnvelope)
153         dialect.setEstimatedExtentsEnabled(d.isEstimatedExtentsEnabled)
154         dialect.setFunctionEncodingEnabled(d.isFunctionEncodingEnabled)
155         dialect.setLooseBBOXEnabled(d.isLooseBBOXEnabled)
156         dialect.setSimplifyEnabled(d.isSimplifyEnabled)
157         ds.setSQLDialect(dialect)
158 
159       case d: PostGISPSDialect =>
160         dialect.setEncodeBBOXFilterAsEnvelope(d.isEncodeBBOXFilterAsEnvelope)
161         dialect.setFunctionEncodingEnabled(d.isFunctionEncodingEnabled)
162         dialect.setLooseBBOXEnabled(d.isLooseBBOXEnabled)
163 
164         // these configs aren't exposed through the PS dialect so re-calculate them from the params
165         val est = PostgisNGDataStoreFactory.ESTIMATED_EXTENTS.lookUp(params)
166         dialect.setEstimatedExtentsEnabled(est == null || est == java.lang.Boolean.TRUE)
167         val simplify = PostgisNGDataStoreFactory.SIMPLIFY.lookUp(params)
168         dialect.setSimplifyEnabled(simplify == null || simplify == java.lang.Boolean.TRUE)
169 
170         ds.setSQLDialect(new PartitionedPostgisPsDialect(ds, dialect))
171 
172       case d => throw new IllegalArgumentException(s"Expected PostGISDialect but got: ${d.getClass.getName}")
173     }
174 
175     ds
176   }

183   private def createConnectionOptions(params: java.util.Map[String, _]): Map[String, String] = {
184     val options =
185       Seq(IdleInTransactionTimeout)

188     logger.debug(s"Connection options: ${options.mkString(" ")}")

190     if (options.isEmpty) {
191       Map.empty
192     } else {
193       Map("options" -> options.mkString(" "))
194     }
195   }

203   /**
204    * Managed data source for apache dbcp2. Managed data sources expose a `close` method on data sources.
205    *
206    * @param wrapped wrapped data source
207    * @param registry metrics registry
208    */
209   class Dbcp2ManagedDataSource(wrapped: org.apache.commons.dbcp2.BasicDataSource, registry: Closeable)
210       extends AbstractManageableDataSource(wrapped) {
211 
212     override def close(): Unit = {
213       CloseWithLogging(registry)
214       wrapped.close()
215     }
216 
217     override def isWrapperFor(c: Class[_]): Boolean = classOf[DataSource].isAssignableFrom(c)
218 
219     override def unwrap[T](c: Class[T]): T = {
220       if (isWrapperFor(c)) {
221         wrapped.asInstanceOf[T]
222       } else {
223         null.asInstanceOf[T]
224       }
225     }
