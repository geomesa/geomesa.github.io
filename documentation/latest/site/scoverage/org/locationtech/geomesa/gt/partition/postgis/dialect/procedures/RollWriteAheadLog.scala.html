14 /**
15  * Moves the current write ahead table to a sequential name, and creates a new write ahead table
16  * to accept writes going forward
17  */
18 object RollWriteAheadLog extends SqlProcedure with CronSchedule {
19 
20   override def name(info: TypeInfo): FunctionName = FunctionName(s"${info.typeName}_roll_wa")
21 
22   override def jobName(info: TypeInfo): SqlLiteral = SqlLiteral(s"${info.typeName}-roll-wa")
23 
24   override protected def schedule(info: TypeInfo): SqlLiteral = SqlLiteral("9,19,29,39,49,59 * * * *")
25 
26   override protected def invocation(info: TypeInfo): SqlLiteral = SqlLiteral(s"CALL ${name(info).quoted}()")
27 
28   override protected def createStatements(info: TypeInfo): Seq[String] =
29     Seq(proc(info)) ++ super.createStatements(info)
30 
31   private def proc(info: TypeInfo): String = {
32     val table = info.tables.writeAhead
33     val partitionsTable = s"${info.schema.quoted}.${PartitionTablespacesTable.Name.quoted}"
34 
35     s"""CREATE OR REPLACE PROCEDURE ${name(info).quoted}() LANGUAGE plpgsql AS
36        |  $$BODY$$
37        |    DECLARE
38        |      seq_val smallint;
39        |      cur_partition text;        -- rename of the _writes table
40        |      next_partition text;       -- next partition that will be created from the _writes table
41        |      partition_tablespace text; -- table space command
42        |      index_space text;          -- index table space command
43        |      pexists boolean;           -- table exists check
44        |    BEGIN
45        |
46        |      SELECT value FROM ${info.schema.quoted}.${SequenceTable.Name.quoted}
47        |        WHERE type_name = ${literal(info.typeName)} INTO seq_val;
48        |      cur_partition := ${literal(table.name.raw + "_")} || lpad(seq_val::text, 3, '0');
49        |
50        |      -- get the required locks up front to avoid deadlocks with inserts
51        |      LOCK TABLE ONLY ${table.name.qualified} IN SHARE UPDATE EXCLUSIVE MODE;
52        |
53        |      -- don't re-create the table if there hasn't been any data inserted
54        |      EXECUTE 'SELECT EXISTS(SELECT 1 FROM ${info.schema.quoted}.' || quote_ident(cur_partition) || ')'
55        |          INTO pexists;
56        |      IF pexists THEN
57        |
58        |        IF seq_val < 999 THEN
59        |          seq_val := seq_val + 1;
60        |        ELSE
61        |          seq_val := 0;
62        |        END IF;
63        |
64        |        UPDATE ${info.schema.quoted}.${SequenceTable.Name.quoted} SET value = seq_val
65        |          WHERE type_name = ${literal(info.typeName)};
66        |
67        |        -- format the table name to be 3 digits, with leading zeros as needed
68        |        next_partition := ${literal(table.name.raw + "_")} || lpad(seq_val::text, 3, '0');
69        |        SELECT table_space INTO partition_tablespace FROM $partitionsTable
70        |          WHERE type_name = ${literal(info.typeName)} AND table_type = ${WriteAheadTableSuffix.quoted};
71        |        IF partition_tablespace IS NULL THEN
72        |          partition_tablespace := '';
73        |          index_space := '';
74        |        ELSE
75        |          partition_tablespace := ' TABLESPACE ' || quote_ident(partition_tablespace);
76        |          index_space := ' USING INDEX' || partition_tablespace;
77        |        END IF;
78        |
79        |        -- requires SHARE UPDATE EXCLUSIVE
80        |        EXECUTE 'CREATE ${info.walLogSQL} TABLE IF NOT EXISTS ${info.schema.quoted}.' || quote_ident(next_partition) || '(' ||
81        |          'CONSTRAINT ' || quote_ident(next_partition || '_pkey') ||
82        |          ' PRIMARY KEY (fid, ${info.cols.dtg.quoted})' || index_space || ')' ||
83        |          ' INHERITS (${table.name.qualified})${table.storage.opts}' || partition_tablespace;
84        |        EXECUTE 'CREATE INDEX IF NOT EXISTS ' || quote_ident(next_partition || '_' || ${info.cols.dtg.asLiteral}) ||
85        |          ' ON ${info.schema.quoted}.' || quote_ident(next_partition) || ' (${info.cols.dtg.quoted})' ||
86        |          partition_tablespace;
87        |${info.cols.geoms.map { col =>
88     s"""        EXECUTE 'CREATE INDEX IF NOT EXISTS ' || quote_ident(next_partition || '_spatial_' || ${col.asLiteral}) ||
89        |          ' ON ${info.schema.quoted}.' || quote_ident(next_partition) || ' USING gist(${col.quoted})' ||
90        |          partition_tablespace;""".stripMargin}.mkString("\n")}
91        |${info.cols.indexed.map { col =>
92     s"""        EXECUTE 'CREATE INDEX IF NOT EXISTS ' || quote_ident(next_partition || '_' || ${col.asLiteral}) ||
93        |          ' ON ${info.schema.quoted}.' || quote_ident(next_partition) || '(${col.quoted})' ||
94        |          partition_tablespace;""".stripMargin}.mkString("\n")}
95        |
96        |        COMMIT; -- releases our locks
97        |
98        |        EXECUTE 'ANALYZE ' || quote_ident(cur_partition);
99        |
100        |      END IF;
101        |    END;
102        |  $$BODY$$;""".stripMargin
103   }
104 }
