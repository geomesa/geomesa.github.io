15 /**
16  * Moves the current write ahead table to a sequential name, and creates a new write ahead table
17  * to accept writes going forward
18  */
19 object RollWriteAheadLog extends SqlProcedure with CronSchedule {
20 
21   override def name(info: TypeInfo): FunctionName = FunctionName(s"${info.typeName}_roll_wa")
22 
23   override def jobName(info: TypeInfo): SqlLiteral = SqlLiteral(s"${info.typeName}-roll-wa")
24 
25   override protected def schedule(info: TypeInfo): SqlLiteral = SqlLiteral("9,19,29,39,49,59 * * * *")
26 
27   override protected def invocation(info: TypeInfo): SqlLiteral = SqlLiteral(s"CALL ${name(info).quoted}()")
28 
29   override protected def createStatements(info: TypeInfo): Seq[String] =
30     Seq(proc(info)) ++ super.createStatements(info)
31 
32   private def proc(info: TypeInfo): String = {
33     s"""CREATE OR REPLACE PROCEDURE ${info.schema.quoted}.${name(info).quoted}() LANGUAGE plpgsql AS
34        |  $$BODY$$
35        |    DECLARE
36        |      seq_val smallint;
37        |      cur_partition text;        -- rename of the _writes table
38        |      next_partition text;       -- next partition that will be created from the _writes table
39        |      partition_tablespace text; -- table space command
40        |      index_space text;          -- index table space command
41        |      table_wa_logging text;     -- wa log options
42        |      pexists boolean;           -- table exists check
43        |    BEGIN
44        |
45        |      SELECT value FROM ${info.schema.quoted}.${SequenceTable.Name.quoted}
46        |        WHERE type_name = ${literal(info.typeName)} INTO seq_val;
47        |      cur_partition := ${literal(info.tables.writeAhead.name.raw + "_")} || lpad(seq_val::text, 3, '0');
48        |
49        |      -- get the required locks up front to avoid deadlocks with inserts
50        |      LOCK TABLE ONLY ${info.tables.writeAhead.name.qualified} IN SHARE UPDATE EXCLUSIVE MODE;
51        |
52        |      -- don't re-create the table if there hasn't been any data inserted
53        |      EXECUTE 'SELECT EXISTS(SELECT 1 FROM ${info.schema.quoted}.' || quote_ident(cur_partition) || ')'
54        |          INTO pexists;
55        |      IF pexists THEN
56        |
57        |        IF seq_val < 999 THEN
58        |          seq_val := seq_val + 1;
59        |        ELSE
60        |          seq_val := 0;
61        |        END IF;
62        |
63        |        UPDATE ${info.schema.quoted}.${SequenceTable.Name.quoted} SET value = seq_val
64        |          WHERE type_name = ${literal(info.typeName)};
65        |
66        |        -- format the table name to be 3 digits, with leading zeros as needed
67        |        next_partition := ${literal(info.tables.writeAhead.name.raw + "_")} || lpad(seq_val::text, 3, '0');
68        |        SELECT value FROM ${info.schema.quoted}.${UserDataTable.Name.quoted}
69        |          WHERE type_name = ${literal(info.typeName)} AND key = ${literal(SftUserData.WriteAheadTableSpace.key)}
70        |          INTO partition_tablespace;
71        |        IF partition_tablespace IS NULL THEN
72        |          partition_tablespace := '';
73        |          index_space := '';
74        |        ELSE
75        |          partition_tablespace := ' TABLESPACE ' || quote_ident(partition_tablespace);
76        |          index_space := ' USING INDEX' || partition_tablespace;
77        |        END IF;
78        |
79        |        SELECT value FROM ${info.schema.quoted}.${UserDataTable.Name.quoted}
80        |          WHERE type_name = ${literal(info.typeName)} AND key = ${literal(SftUserData.WalLogEnabled.key)}
81        |          INTO table_wa_logging;
82        |        IF table_wa_logging IS NULL OR NOT table_wa_logging::boolean THEN
83        |          table_wa_logging := '';
84        |        ELSE
85        |          table_wa_logging := 'UNLOGGED ';
86        |        END IF;
87        |
88        |        -- requires SHARE UPDATE EXCLUSIVE
89        |        EXECUTE 'CREATE ' || table_wa_logging || 'TABLE IF NOT EXISTS ${info.schema.quoted}.' || quote_ident(next_partition) || '(' ||
90        |          'CONSTRAINT ' || quote_ident(next_partition || '_pkey') ||
91        |          ' PRIMARY KEY (fid, ${info.cols.dtg.quoted})' || index_space || ')' ||
92        |          ' INHERITS (${info.tables.writeAhead.name.qualified})${info.tables.writeAhead.storage.opts}' || partition_tablespace;
93        |        EXECUTE 'CREATE INDEX IF NOT EXISTS ' || quote_ident(next_partition || '_' || ${info.cols.dtg.asLiteral}) ||
94        |          ' ON ${info.schema.quoted}.' || quote_ident(next_partition) || ' (${info.cols.dtg.quoted})' ||
95        |          partition_tablespace;
96        |${info.cols.geoms.map { col =>
97     s"""        EXECUTE 'CREATE INDEX IF NOT EXISTS ' || quote_ident(next_partition || '_spatial_' || ${col.asLiteral}) ||
98        |          ' ON ${info.schema.quoted}.' || quote_ident(next_partition) || ' USING gist(${col.quoted})' ||
99        |          partition_tablespace;""".stripMargin}.mkString("\n")}
100        |${info.cols.indexed.map { col =>
101     s"""        EXECUTE 'CREATE INDEX IF NOT EXISTS ' || quote_ident(next_partition || '_' || ${col.asLiteral}) ||
102        |          ' ON ${info.schema.quoted}.' || quote_ident(next_partition) || '(${col.quoted})' ||
103        |          partition_tablespace;""".stripMargin}.mkString("\n")}
104        |
105        |        COMMIT; -- releases our locks
106        |
107        |        EXECUTE 'ANALYZE ' || quote_ident(cur_partition);
108        |
109        |      END IF;
110        |    END;
111        |  $$BODY$$;""".stripMargin
112   }
