14 /**
15  * Partitions the write ahead table into the recent and/or main partition tables
16  */
17 object PartitionWriteAheadLog extends SqlProcedure {
18 
19   override def name(info: TypeInfo): FunctionName = FunctionName(s"${info.typeName}_partition_wa")
20 
21   override protected def createStatements(info: TypeInfo): Seq[String] = Seq(proc(info))
22 
23   private def proc(info: TypeInfo): String = {
24     val hours = info.partitions.hoursPerPartition
25     val writeAhead = info.tables.writeAhead
26     val writeAheadPartitions = info.tables.writeAheadPartitions
27     val mainPartitions = info.tables.mainPartitions
28     val spillPartitions = info.tables.spillPartitions
29     val tablespaceTable = s"${info.schema.quoted}.${PartitionTablespacesTable.Name.quoted}"
30     val dtgCol = info.cols.dtg.quoted
31     val geomCol = info.cols.geom.quoted
32 
33     s"""CREATE OR REPLACE PROCEDURE ${name(info).quoted}(cur_time timestamp without time zone) LANGUAGE plpgsql AS
34        |  $$BODY$$
35        |    DECLARE
36        |      seq_val smallint;
37        |      write_partition text;                        -- current partition receiving writes
38        |      min_dtg timestamp without time zone;         -- min date in our partitioned tables
39        |      main_cutoff timestamp without time zone;     -- max age of the records for main tables
40        |      write_ahead record;
41        |      partition_start timestamp without time zone; -- start bounds for the partition we're writing
42        |      partition_end timestamp without time zone;   -- end bounds for the partition we're writing
43        |      partition_name text;                         -- partition table name
44        |      partition_parent text;                       -- partition parent table name
45        |      partition_tablespace text;                   -- partition tablespace
46        |      index_tablespace text;                       -- index tablespace
47        |      pexists boolean;                             -- table exists check
48        |      unsorted_count bigint;
49        |    BEGIN
50        |      -- constants
51        |      main_cutoff := truncate_to_partition(cur_time, $hours) - INTERVAL '$hours HOURS';
52        |
53        |      SELECT value from ${info.schema.quoted}.${SequenceTable.Name.quoted}
54        |        WHERE type_name = ${literal(info.typeName)} INTO seq_val;
55        |      write_partition := ${literal(writeAhead.name.raw + "_")} || lpad(seq_val::text, 3, '0');
56        |
57        |      -- check for write ahead partitions and move the data into the time partitioned tables
58        |      FOR write_ahead IN
59        |        SELECT pg_class.relname AS name
60        |          FROM pg_catalog.pg_inherits
61        |          INNER JOIN pg_catalog.pg_class ON (pg_inherits.inhrelid = pg_class.oid)
62        |          INNER JOIN pg_catalog.pg_namespace ON (pg_class.relnamespace = pg_namespace.oid)
63        |          WHERE inhparent = ${writeAhead.name.asRegclass}
64        |          AND relname != write_partition
65        |          ORDER BY name
66        |      LOOP
67        |
68        |        RAISE INFO '% Checking write ahead table %', timeofday()::timestamp, write_ahead.name;
69        |        -- get a lock on the table - this mode won't prevent reads but will prevent writes
70        |        -- (there shouldn't be any writes though) and will synchronize this method
71        |        LOCK TABLE ONLY ${mainPartitions.name.qualified} IN SHARE UPDATE EXCLUSIVE MODE;
72        |        LOCK TABLE ONLY ${spillPartitions.name.qualified} IN SHARE UPDATE EXCLUSIVE MODE;
73        |        EXECUTE 'LOCK TABLE ${info.schema.quoted}.' || quote_ident(write_ahead.name) ||
74        |          ' IN SHARE UPDATE EXCLUSIVE MODE';
75        |        RAISE INFO '% Locked write ahead table % for migration', timeofday()::timestamp, write_ahead.name;
76        |
77        |        -- wait until the table doesn't contain any recent records
78        |        EXECUTE 'SELECT EXISTS(SELECT 1 FROM ${info.schema.quoted}.' || quote_ident(write_ahead.name) ||
79        |          ' WHERE $dtgCol >= ' || quote_literal(truncate_to_ten_minutes(cur_time)) || ')'
80        |          INTO pexists;
81        |
82        |        IF pexists THEN
83        |          -- should only happen if data is inserted with timestamps from the future
84        |          RAISE NOTICE '% Skipping write ahead table % due to min date', timeofday()::timestamp, write_ahead.name;
85        |        ELSE
86        |          partition_end := '-infinity'::timestamp without time zone;
87        |          LOOP
88        |            -- find the range of dates in the write ahead partition
89        |            EXECUTE 'SELECT min($dtgCol) FROM ${info.schema.quoted}.' || quote_ident(write_ahead.name) ||
90        |              ' WHERE $dtgCol >= ' || quote_literal(partition_end) INTO min_dtg;
91        |            EXIT WHEN min_dtg IS NULL;
92        |
93        |            -- calculate the partition bounds for the min date
94        |            IF min_dtg < main_cutoff THEN
95        |              partition_start := truncate_to_partition(min_dtg, $hours);
96        |              partition_end := partition_start + INTERVAL '$hours HOURS';
97        |              partition_parent := ${mainPartitions.name.asLiteral};
98        |              partition_name := partition_parent || '_' || to_char(partition_start, 'YYYY_MM_DD_HH24');
99        |              SELECT table_space INTO partition_tablespace FROM $tablespaceTable
100        |                WHERE type_name = ${literal(info.typeName)} AND table_type = ${PartitionedTableSuffix.quoted};
101        |              IF partition_tablespace IS NULL THEN
102        |                index_tablespace := '';
103        |                partition_tablespace := '${mainPartitions.storage.opts}';
104        |              ELSE
105        |                index_tablespace := ' USING INDEX TABLESPACE '|| quote_ident(partition_tablespace);
106        |                partition_tablespace := '${mainPartitions.storage.opts} TABLESPACE ' ||
107        |                  quote_ident(partition_tablespace);
108        |              END IF;
109        |
110        |              -- check for existing partition and switch to spill table instead
111        |              SELECT EXISTS(SELECT FROM pg_tables WHERE schemaname = ${info.schema.asLiteral} AND tablename = partition_name)
112        |                INTO pexists;
113        |              IF pexists THEN
114        |                partition_parent := ${spillPartitions.name.asLiteral};
115        |                partition_name := partition_parent || '_' || to_char(partition_start, 'YYYY_MM_DD_HH24');
116        |              END IF;
117        |            ELSE
118        |              partition_start := truncate_to_ten_minutes(min_dtg);
119        |              partition_end := partition_start + INTERVAL '10 MINUTES';
120        |              partition_parent := ${writeAheadPartitions.name.asLiteral};
121        |              partition_name := partition_parent || '_' || to_char(partition_start, 'YYYY_MM_DD_HH24_MI');
122        |              SELECT table_space INTO partition_tablespace FROM $tablespaceTable
123        |                WHERE type_name = ${literal(info.typeName)} AND table_type = ${PartitionedWriteAheadTableSuffix.quoted};
124        |              IF partition_tablespace IS NULL THEN
125        |                index_tablespace := '';
126        |                partition_tablespace := '${writeAheadPartitions.storage.opts}';
127        |              ELSE
128        |                index_tablespace := ' USING INDEX TABLESPACE '|| quote_ident(partition_tablespace);
129        |                partition_tablespace := '${writeAheadPartitions.storage.opts} TABLESPACE ' ||
130        |                  quote_ident(partition_tablespace);
131        |              END IF;
132        |            END IF;
133        |
134        |            RAISE INFO '% Writing to partition %', timeofday()::timestamp, partition_name;
135        |
136        |            SELECT EXISTS(SELECT FROM pg_tables WHERE schemaname = ${info.schema.asLiteral} AND tablename = partition_name)
137        |              INTO pexists;
138        |
139        |            -- create the partition table if it doesn't exist
140        |            -- note: normally the partition will not exist unless time-latent data was inserted
141        |            -- we create it unattached so that it doesn't lock the _recent table on insert
142        |            -- then we attach it after inserting the rows
143        |            -- since this is all within a transaction it should all happen "at once"
144        |            -- see https://www.postgresql.org/docs/13/ddl-partitioning.html#DDL-PARTITIONING-DECLARATIVE-MAINTENANCE
145        |
146        |            IF pexists THEN
147        |              -- copy rows from write ahead table to partition table
148        |              RAISE INFO '% Copying rows to partition %', timeofday()::timestamp, partition_name;
149        |              EXECUTE 'INSERT INTO ${info.schema.quoted}.' || quote_ident(partition_name) ||
150        |                ' SELECT * FROM ' || quote_ident(write_ahead.name) ||
151        |                '   WHERE $dtgCol >= ' || quote_literal(partition_start) ||
152        |                '     AND $dtgCol < ' || quote_literal(partition_end) ||
153        |                '   ORDER BY _st_sortablehash($geomCol)' ||
154        |                '   ON CONFLICT DO NOTHING';
155        |              GET DIAGNOSTICS unsorted_count := ROW_COUNT;
156        |            ELSE
157        |              RAISE INFO '% Creating partition with insert % (unattached)', timeofday()::timestamp, partition_name;
158        |              -- upper bounds are exclusive
159        |              -- use "create table as" (vs create then insert) for performance benefits related to WAL skipping
160        |              EXECUTE 'CREATE ${info.walLogSQL} TABLE ${info.schema.quoted}.' || quote_ident(partition_name) ||
161        |                partition_tablespace || ' AS SELECT * FROM ' || quote_ident(write_ahead.name) ||
162        |                '   WHERE $dtgCol >= ' || quote_literal(partition_start) ||
163        |                '     AND $dtgCol < ' || quote_literal(partition_end) ||
164        |                '   ORDER BY _st_sortablehash($geomCol)';
165        |              GET DIAGNOSTICS unsorted_count := ROW_COUNT;
166        |              EXECUTE 'ALTER TABLE ${info.schema.quoted}.' || quote_ident(partition_name) ||
167        |                ' ADD CONSTRAINT ' || quote_ident(partition_name || '_pkey') ||
168        |                ' PRIMARY KEY (fid, ${info.cols.dtg.quoted})' || index_tablespace;
169        |              -- creating a constraint allows it to be attached to the parent without any additional checks
170        |              EXECUTE 'ALTER TABLE  ${info.schema.quoted}.' || quote_ident(partition_name) ||
171        |                ' ADD CONSTRAINT ' || quote_ident(partition_name || '_constraint') ||
172        |                ' CHECK ( $dtgCol >= ' || quote_literal(partition_start) ||
173        |                ' AND $dtgCol < ' || quote_literal(partition_end) || ' )';
174        |            END IF;
175        |            RAISE INFO '% Done writing % rows to partition %', timeofday()::timestamp, unsorted_count, partition_name;
176        |
177        |            IF partition_parent = ${spillPartitions.name.asLiteral} THEN
178        |              -- store record of unsorted row counts which could negatively impact BRIN index scans
179        |              INSERT INTO ${info.tables.sortQueue.name.qualified}(partition_name, unsorted_count, enqueued)
180        |                VALUES (partition_name, unsorted_count, now());
181        |              RAISE NOTICE 'Inserting % rows into spill partition %, queries may be impacted',
182        |                unsorted_count, partition_name;
183        |            END IF;
184        |
185        |            -- attach the partition table to the parent
186        |            IF NOT pexists THEN
187        |              RAISE INFO '% Attaching partition % to parent', timeofday()::timestamp, partition_name;
188        |              EXECUTE 'ALTER TABLE ${info.schema.quoted}.' || quote_ident(partition_parent) ||
189        |                ' ATTACH PARTITION ${info.schema.quoted}.' || quote_ident(partition_name) ||
190        |                ' FOR VALUES FROM (' || quote_literal(partition_start) ||
191        |                ') TO (' || quote_literal(partition_end) || ' );';
192        |              -- once the table is attached we can drop the redundant constraint
193        |              -- however, this requires ACCESS EXCLUSIVE - since constraints are only checked on inserts
194        |              -- or updates, and partition tables are 'immutable' (only written to once), it shouldn't
195        |              -- affect anything to leave it. note that for 'spill' tables, there may be some redundant checks
196        |              -- EXECUTE 'ALTER TABLE ${info.schema.quoted}.' || quote_ident(partition_name) ||
197        |              --  ' DROP CONSTRAINT ' || quote_ident(partition_name || '_constraint');
198        |              RAISE NOTICE 'A partition has been created %', partition_name;
199        |            END IF;
200        |
201        |            -- mark the partition to be analyzed in a separate thread
202        |            INSERT INTO ${info.tables.analyzeQueue.name.qualified}(partition_name, enqueued)
203        |              VALUES (partition_name, now());
204        |          END LOOP;
205        |
206        |          RAISE INFO '% Dropping write ahead table %', timeofday()::timestamp, write_ahead.name;
207        |          -- requires ACCESS EXCLUSIVE lock
208        |          EXECUTE 'DROP TABLE ' || quote_ident(write_ahead.name);
209        |
210        |        END IF;
211        |
212        |        COMMIT; -- releases the lock
213        |      END LOOP;
214        |    END;
215        |  $$BODY$$;
216        |""".stripMargin
217   }
