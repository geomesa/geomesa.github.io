26        |  $$BODY$$
27        |    DECLARE
28        |      seq_val smallint;
29        |      write_partition text;                        -- current partition receiving writes
30        |      min_dtg timestamp without time zone;         -- min date in our partitioned tables
31        |      main_cutoff timestamp without time zone;     -- max age of the records for main tables
32        |      write_ahead record;
33        |      partition_size int;                          -- size of the partition, in hours
34        |      partition_start timestamp without time zone; -- start bounds for the partition we're writing
35        |      partition_end timestamp without time zone;   -- end bounds for the partition we're writing
36        |      partition_name text;                         -- partition table name
37        |      partition_parent text;                       -- partition parent table name
38        |      partition_tablespace text;                   -- partition tablespace
39        |      index_tablespace text;                       -- index tablespace
40        |      table_wa_logging text;                       -- wa log options
41        |      pexists boolean;                             -- table exists check
42        |      unsorted_count bigint;
43        |    BEGIN
44        |      -- constants
45        |      SELECT COALESCE(
46        |        (SELECT value::int FROM ${info.schema.quoted}.${UserDataTable.Name.quoted}
47        |          WHERE type_name = ${literal(info.typeName)} AND key = ${literal(SftUserData.IntervalHours.key)}),
48        |        ${SftUserData.IntervalHours.default})
49        |        INTO partition_size;
50        |      main_cutoff := ${info.schema.quoted}.truncate_to_partition(cur_time, partition_size) - make_interval(hours => partition_size);
51        |
52        |      SELECT value from ${info.schema.quoted}.${SequenceTable.Name.quoted}
53        |        WHERE type_name = ${literal(info.typeName)} INTO seq_val;
54        |      write_partition := ${literal(info.tables.writeAhead.name.raw + "_")} || lpad(seq_val::text, 3, '0');
55        |
56        |      -- check for write ahead partitions and move the data into the time partitioned tables
57        |      FOR write_ahead IN
58        |        SELECT pg_class.relname AS name
59        |          FROM pg_catalog.pg_inherits
60        |          INNER JOIN pg_catalog.pg_class ON (pg_inherits.inhrelid = pg_class.oid)
61        |          INNER JOIN pg_catalog.pg_namespace ON (pg_class.relnamespace = pg_namespace.oid)
62        |          WHERE inhparent = ${info.tables.writeAhead.name.asRegclass}
63        |          AND relname != write_partition
64        |          ORDER BY name
65        |      LOOP
66        |
67        |        RAISE INFO '% Checking write ahead table %', timeofday()::timestamp, write_ahead.name;
68        |        -- get a lock on the table - this mode won't prevent reads but will prevent writes
69        |        -- (there shouldn't be any writes though) and will synchronize this method
70        |        LOCK TABLE ONLY ${info.tables.mainPartitions.name.qualified} IN SHARE UPDATE EXCLUSIVE MODE;
71        |        LOCK TABLE ONLY ${info.tables.spillPartitions.name.qualified} IN SHARE UPDATE EXCLUSIVE MODE;
72        |        EXECUTE 'LOCK TABLE ${info.schema.quoted}.' || quote_ident(write_ahead.name) ||
73        |          ' IN SHARE UPDATE EXCLUSIVE MODE';
74        |        RAISE INFO '% Locked write ahead table % for migration', timeofday()::timestamp, write_ahead.name;
75        |
76        |        -- wait until the table doesn't contain any recent records
77        |        EXECUTE 'SELECT EXISTS(SELECT 1 FROM ${info.schema.quoted}.' || quote_ident(write_ahead.name) ||
78        |          ' WHERE ${info.cols.dtg.quoted} >= ' || quote_literal(${info.schema.quoted}.truncate_to_ten_minutes(cur_time)) || ')'
79        |          INTO pexists;
80        |
81        |        IF pexists THEN
82        |          -- should only happen if data is inserted with timestamps from the future
83        |          RAISE NOTICE '% Skipping write ahead table % due to min date', timeofday()::timestamp, write_ahead.name;
84        |        ELSE
85        |          partition_end := '-infinity'::timestamp without time zone;
86        |          LOOP
87        |            -- find the range of dates in the write ahead partition
88        |            EXECUTE 'SELECT min(${info.cols.dtg.quoted}) FROM ${info.schema.quoted}.' || quote_ident(write_ahead.name) ||
89        |              ' WHERE ${info.cols.dtg.quoted} >= ' || quote_literal(partition_end) INTO min_dtg;
90        |            EXIT WHEN min_dtg IS NULL;
91        |
92        |            -- calculate the partition bounds for the min date
93        |            IF min_dtg < main_cutoff THEN
94        |              partition_start := ${info.schema.quoted}.truncate_to_partition(min_dtg, partition_size);
95        |              partition_end := partition_start + make_interval(hours => partition_size);
96        |              partition_parent := ${info.tables.mainPartitions.name.asLiteral};
97        |              partition_name := partition_parent || '_' || to_char(partition_start, 'YYYY_MM_DD_HH24');
98        |              SELECT value FROM ${info.schema.quoted}.${UserDataTable.Name.quoted}
99        |                WHERE type_name = ${literal(info.typeName)} AND key = ${literal(SftUserData.MainTableSpace.key)}
100        |                INTO partition_tablespace;
101        |              IF partition_tablespace IS NULL THEN
102        |                index_tablespace := '';
103        |                partition_tablespace := '${info.tables.mainPartitions.storage.opts}';
104        |              ELSE
105        |                index_tablespace := ' USING INDEX TABLESPACE '|| quote_ident(partition_tablespace);
106        |                partition_tablespace := '${info.tables.mainPartitions.storage.opts} TABLESPACE ' ||
107        |                  quote_ident(partition_tablespace);
108        |              END IF;
109        |
110        |              -- check for existing partition and switch to spill table instead
111        |              SELECT EXISTS(SELECT FROM pg_tables WHERE schemaname = ${info.schema.asLiteral} AND tablename = partition_name)
112        |                INTO pexists;
113        |              IF pexists THEN
114        |                partition_parent := ${info.tables.spillPartitions.name.asLiteral};
115        |                partition_name := partition_parent || '_' || to_char(partition_start, 'YYYY_MM_DD_HH24');
116        |              END IF;
117        |            ELSE
118        |              partition_start := ${info.schema.quoted}.truncate_to_ten_minutes(min_dtg);
119        |              partition_end := partition_start + INTERVAL '10 MINUTES';
120        |              partition_parent := ${info.tables.writeAheadPartitions.name.asLiteral};
121        |              partition_name := partition_parent || '_' || to_char(partition_start, 'YYYY_MM_DD_HH24_MI');
122        |              SELECT value FROM ${info.schema.quoted}.${UserDataTable.Name.quoted}
123        |                WHERE type_name = ${literal(info.typeName)} AND key = ${literal(SftUserData.WriteAheadPartitionsTableSpace.key)}
124        |                INTO partition_tablespace;
125        |              IF partition_tablespace IS NULL THEN
126        |                index_tablespace := '';
127        |                partition_tablespace := '${info.tables.writeAheadPartitions.storage.opts}';
128        |              ELSE
129        |                index_tablespace := ' USING INDEX TABLESPACE '|| quote_ident(partition_tablespace);
130        |                partition_tablespace := '${info.tables.writeAheadPartitions.storage.opts} TABLESPACE ' ||
131        |                  quote_ident(partition_tablespace);
132        |              END IF;
133        |            END IF;
134        |
135        |            RAISE INFO '% Writing to partition %', timeofday()::timestamp, partition_name;
136        |
137        |            SELECT EXISTS(SELECT FROM pg_tables WHERE schemaname = ${info.schema.asLiteral} AND tablename = partition_name)
138        |              INTO pexists;
139        |
140        |            -- create the partition table if it doesn't exist
141        |            -- note: normally the partition will not exist unless time-latent data was inserted
142        |            -- we create it unattached so that it doesn't lock the _recent table on insert
143        |            -- then we attach it after inserting the rows
144        |            -- since this is all within a transaction it should all happen "at once"
145        |            -- see https://www.postgresql.org/docs/13/ddl-partitioning.html#DDL-PARTITIONING-DECLARATIVE-MAINTENANCE
146        |
147        |            IF pexists THEN
148        |              -- copy rows from write ahead table to partition table
149        |              RAISE INFO '% Copying rows to partition %', timeofday()::timestamp, partition_name;
150        |              EXECUTE 'INSERT INTO ${info.schema.quoted}.' || quote_ident(partition_name) ||
151        |                ' SELECT * FROM ' || quote_ident(write_ahead.name) ||
152        |                '   WHERE ${info.cols.dtg.quoted} >= ' || quote_literal(partition_start) ||
153        |                '     AND ${info.cols.dtg.quoted} < ' || quote_literal(partition_end) ||
154        |                '   ORDER BY _st_sortablehash(${info.cols.geom.quoted})' ||
155        |                '   ON CONFLICT DO NOTHING';
156        |              GET DIAGNOSTICS unsorted_count := ROW_COUNT;
157        |            ELSE
158        |              RAISE INFO '% Creating partition with insert % (unattached)', timeofday()::timestamp, partition_name;
159        |
160        |              SELECT value FROM ${info.schema.quoted}.${UserDataTable.Name.quoted}
161        |                WHERE type_name = ${literal(info.typeName)} AND key = ${literal(SftUserData.WalLogEnabled.key)}
162        |                INTO table_wa_logging;
163        |              IF table_wa_logging IS NULL OR NOT table_wa_logging::boolean THEN
164        |                table_wa_logging := '';
165        |              ELSE
166        |                table_wa_logging := 'UNLOGGED ';
167        |              END IF;
168        |
169        |              -- upper bounds are exclusive
170        |              -- use "create table as" (vs create then insert) for performance benefits related to WAL skipping
171        |              EXECUTE 'CREATE ' || table_wa_logging || 'TABLE ${info.schema.quoted}.' || quote_ident(partition_name) ||
172        |                partition_tablespace || ' AS SELECT * FROM ' || quote_ident(write_ahead.name) ||
173        |                '   WHERE ${info.cols.dtg.quoted} >= ' || quote_literal(partition_start) ||
174        |                '     AND ${info.cols.dtg.quoted} < ' || quote_literal(partition_end) ||
175        |                '   ORDER BY _st_sortablehash(${info.cols.geom.quoted})';
176        |              GET DIAGNOSTICS unsorted_count := ROW_COUNT;
177        |              EXECUTE 'ALTER TABLE ${info.schema.quoted}.' || quote_ident(partition_name) ||
178        |                ' ADD CONSTRAINT ' || quote_ident(partition_name || '_pkey') ||
179        |                ' PRIMARY KEY (fid, ${info.cols.dtg.quoted})' || index_tablespace;
180        |              -- creating a constraint allows it to be attached to the parent without any additional checks
181        |              EXECUTE 'ALTER TABLE  ${info.schema.quoted}.' || quote_ident(partition_name) ||
182        |                ' ADD CONSTRAINT ' || quote_ident(partition_name || '_constraint') ||
183        |                ' CHECK ( ${info.cols.dtg.quoted} >= ' || quote_literal(partition_start) ||
184        |                ' AND ${info.cols.dtg.quoted} < ' || quote_literal(partition_end) || ' )';
185        |            END IF;
186        |            RAISE INFO '% Done writing % rows to partition %', timeofday()::timestamp, unsorted_count, partition_name;
187        |
188        |            IF partition_parent = ${info.tables.spillPartitions.name.asLiteral} THEN
189        |              -- store record of unsorted row counts which could negatively impact BRIN index scans
190        |              INSERT INTO ${info.tables.sortQueue.name.qualified}(partition_name, unsorted_count, enqueued)
191        |                VALUES (partition_name, unsorted_count, now());
192        |              RAISE NOTICE 'Inserting % rows into spill partition %, queries may be impacted',
193        |                unsorted_count, partition_name;
194        |            END IF;
195        |
196        |            -- attach the partition table to the parent
197        |            IF NOT pexists THEN
198        |              RAISE INFO '% Attaching partition % to parent', timeofday()::timestamp, partition_name;
199        |              EXECUTE 'ALTER TABLE ${info.schema.quoted}.' || quote_ident(partition_parent) ||
200        |                ' ATTACH PARTITION ${info.schema.quoted}.' || quote_ident(partition_name) ||
201        |                ' FOR VALUES FROM (' || quote_literal(partition_start) ||
202        |                ') TO (' || quote_literal(partition_end) || ' );';
203        |              -- once the table is attached we can drop the redundant constraint
204        |              -- however, this requires ACCESS EXCLUSIVE - since constraints are only checked on inserts
205        |              -- or updates, and partition tables are 'immutable' (only written to once), it shouldn't
206        |              -- affect anything to leave it. note that for 'spill' tables, there may be some redundant checks
207        |              -- EXECUTE 'ALTER TABLE ${info.schema.quoted}.' || quote_ident(partition_name) ||
208        |              --  ' DROP CONSTRAINT ' || quote_ident(partition_name || '_constraint');
209        |              RAISE NOTICE 'A partition has been created %', partition_name;
210        |            END IF;
211        |
212        |            -- mark the partition to be analyzed in a separate thread
213        |            INSERT INTO ${info.tables.analyzeQueue.name.qualified}(partition_name, enqueued)
214        |              VALUES (partition_name, now());
215        |          END LOOP;
216        |
217        |          RAISE INFO '% Dropping write ahead table %', timeofday()::timestamp, write_ahead.name;
218        |          -- requires ACCESS EXCLUSIVE lock
219        |          EXECUTE 'DROP TABLE ' || quote_ident(write_ahead.name);
220        |
221        |        END IF;
222        |
223        |        COMMIT; -- releases the lock
224        |      END LOOP;
225        |    END;
226        |  $$BODY$$;
227        |""".stripMargin
228   }
