15 /**
16  * Merge recent write-ahead partitions and move them into the main partition table
17  */
18 object MergeWriteAheadPartitions extends SqlProcedure {
19 
20   override def name(info: TypeInfo): FunctionName = FunctionName(s"${info.typeName}_merge_wa_partitions")
21 
22   override protected def createStatements(info: TypeInfo): Seq[String] = Seq(proc(info))

24   private def proc(info: TypeInfo): String = {
25     s"""CREATE OR REPLACE PROCEDURE ${name(info).quoted}(cur_time timestamp without time zone) LANGUAGE plpgsql AS
26        |  $$BODY$$
27        |    DECLARE
28        |      partition_size int;                          -- size of the partition, in hours
29        |      min_dtg timestamp without time zone;         -- min date in our partitioned tables
30        |      main_cutoff timestamp without time zone;     -- max age of the records for main tables
31        |      partition_start timestamp without time zone; -- start bounds for the partition we're writing
32        |      partition_end timestamp without time zone;   -- end bounds for the partition we're writing
33        |      partition_name text;                         -- partition table name
34        |      partition_parent text;                       -- partition table to attach to
35        |      partition_tablespace text;                   -- partition tablespace
36        |      index_tablespace text;                       -- index tablespace
37        |      table_wa_logging text;                       -- wa log options
38        |      write_ahead_partitions text[];               -- names of the partitions we're migrating
39        |      write_ahead_partition text;                  -- name of current partition
40        |      pexists boolean;                             -- table exists check
41        |      unsorted_count bigint;
42        |    BEGIN
43        |      -- constants
44        |      SELECT COALESCE(
45        |          (SELECT value::int FROM ${info.schema.quoted}.${UserDataTable.Name.quoted}
46        |            WHERE type_name = ${literal(info.typeName)} AND key = ${literal(SftUserData.IntervalHours.key)}),
47        |          ${SftUserData.IntervalHours.default}
48        |        ) INTO partition_size;
49        |      main_cutoff := truncate_to_partition(cur_time, partition_size) - make_interval(hours => partition_size);
50        |
51        |      -- move data from the write ahead partitions to the main partitions
52        |      LOOP
53        |        -- find the range of dates in the write ahead partition tables
54        |        SELECT min(${info.cols.dtg.quoted}) INTO min_dtg FROM ${info.tables.writeAheadPartitions.name.qualified}
55        |          WHERE ${info.cols.dtg.quoted} < main_cutoff;
56        |        EXIT WHEN min_dtg IS NULL;
57        |
58        |        partition_start := truncate_to_partition(min_dtg, partition_size);
59        |        partition_end := partition_start + make_interval(hours => partition_size);
60        |        partition_parent := ${info.tables.mainPartitions.name.asLiteral};
61        |        partition_name := partition_parent || '_' || to_char(partition_start, 'YYYY_MM_DD_HH24');
62        |
63        |        SELECT EXISTS(SELECT FROM pg_tables WHERE schemaname = ${info.schema.asLiteral} AND tablename = partition_name)
64        |          INTO pexists;
65        |
66        |        -- if the partition already exists, write to the spill partition instead to avoid messing up the BRIN index
67        |        IF pexists THEN
68        |          partition_parent := ${info.tables.spillPartitions.name.asLiteral};
69        |          partition_name := partition_parent || '_' || to_char(partition_start, 'YYYY_MM_DD_HH24');
70        |          SELECT EXISTS(SELECT FROM pg_tables WHERE schemaname = ${info.schema.asLiteral} AND tablename = partition_name)
71        |            INTO pexists;
72        |        END IF;
73        |
74        |        -- find the write ahead partitions we're copying from
75        |        -- order the results to ensure we get locks in a consistent order to avoid deadlocks
76        |        write_ahead_partitions := Array(
77        |          SELECT '${info.schema.quoted}.' || quote_ident(pg_class.relname)
78        |            FROM pg_catalog.pg_inherits
79        |            INNER JOIN pg_catalog.pg_class ON (pg_inherits.inhrelid = pg_class.oid)
80        |            INNER JOIN pg_catalog.pg_namespace ON (pg_class.relnamespace = pg_namespace.oid)
81        |            WHERE inhparent = ${info.tables.writeAheadPartitions.name.asRegclass}
82        |              AND pg_class.relname >= ${info.tables.writeAheadPartitions.name.asLiteral} || '_' || to_char(partition_start, 'YYYY_MM_DD_HH24_MI')
83        |              AND pg_class.relname < ${info.tables.writeAheadPartitions.name.asLiteral} || '_' || to_char(partition_end, 'YYYY_MM_DD_HH24_MI')
84        |            ORDER BY 1
85        |          );
86        |
87        |        -- get a lock on the tables - this mode won't prevent reads but will prevent writes
88        |        -- (there shouldn't be any writes though) and will synchronize this method
89        |        -- TODO we really just need to sync this method for safety in manual invocations
90        |        -- FOREACH write_ahead_partition IN ARRAY write_ahead_partitions LOOP
91        |        --   EXECUTE 'LOCK TABLE ' || write_ahead_partition || ' IN SHARE ROW EXCLUSIVE MODE';
92        |        --   RAISE INFO '% Locked write ahead partition % for migration', timeofday()::timestamp, write_ahead_partition;
93        |        -- END LOOP;
94        |
95        |        -- create a view from the tables so that we can sort the result by an expression (geohash)
96        |        EXECUTE 'CREATE TEMP VIEW ' || quote_ident(partition_name || '_tmp_migrate') ||
97        |          ' AS SELECT * FROM ' || array_to_string(write_ahead_partitions, ' UNION ALL SELECT * FROM ');
98        |
99        |        -- copy rows from write ahead partitions to main partition table
100        |        IF pexists THEN
101        |          RAISE INFO '% Copying rows to partition %', timeofday()::timestamp, partition_name;
102        |          EXECUTE 'INSERT INTO ${info.schema.quoted}.' || quote_ident(partition_name) ||
103        |            ' SELECT * FROM ' || quote_ident(partition_name || '_tmp_migrate') ||
104        |            '   ORDER BY _st_sortablehash(${info.cols.geom.quoted})' ||
105        |            '   ON CONFLICT DO NOTHING';
106        |          GET DIAGNOSTICS unsorted_count := ROW_COUNT;
107        |        ELSE
108        |          RAISE INFO '% Creating partition with insert % (unattached)', timeofday()::timestamp, partition_name;
109        |
110        |          SELECT value FROM ${info.schema.quoted}.${UserDataTable.Name.quoted}
111        |            WHERE type_name = ${literal(info.typeName)} AND key = ${literal(SftUserData.MainTableSpace.key)}
112        |            INTO partition_tablespace;
113        |          IF partition_tablespace IS NULL THEN
114        |            index_tablespace := '';
115        |            partition_tablespace := '';
116        |          ELSE
117        |            index_tablespace := ' USING INDEX TABLESPACE '|| quote_ident(partition_tablespace);
118        |            partition_tablespace := ' TABLESPACE ' || quote_ident(partition_tablespace);
119        |          END IF;
120        |
121        |          SELECT COALESCE(
122        |            (SELECT value FROM ${info.schema.quoted}.${UserDataTable.Name.quoted}
123        |              WHERE type_name = ${literal(info.typeName)} AND key = ${literal(SftUserData.WalLogEnabled.key)}),
124        |              '${SftUserData.WalLogEnabled.default}'
125        |            ) INTO table_wa_logging;
126        |          IF table_wa_logging IS NULL OR NOT table_wa_logging::boolean THEN
127        |            table_wa_logging := '';
128        |          ELSE
129        |            table_wa_logging := 'UNLOGGED ';
130        |          END IF;
131        |
132        |          -- upper bounds are exclusive
133        |          -- this won't have any indices until we attach it to the parent partition table
134        |          -- use "create table as" (vs create then insert) for performance benefits related to WAL skipping
135        |          -- we need a "select distinct" to avoid primary key conflicts - this should be fairly cheap since
136        |          --   we're already sorting and there should be few or no conflicts
137        |          -- create the partition table with a 'create as' for improved performance
138        |          EXECUTE 'CREATE ' || table_wa_logging || 'TABLE ${info.schema.quoted}.' || quote_ident(partition_name) ||
139        |            partition_tablespace || ' AS SELECT DISTINCT ON' ||
140        |            ' (_st_sortablehash(${info.cols.geom.quoted}), fid, ${info.cols.dtg.quoted}) * FROM ' ||
141        |            quote_ident(partition_name || '_tmp_migrate') || ' ORDER BY _st_sortablehash(${info.cols.geom.quoted})';
142        |          GET DIAGNOSTICS unsorted_count := ROW_COUNT;
143        |          EXECUTE 'ALTER TABLE ${info.schema.quoted}.' || quote_ident(partition_name) ||
144        |            ' ADD CONSTRAINT ' || quote_ident(partition_name || '_pkey') ||
145        |            ' PRIMARY KEY (fid, ${info.cols.dtg.quoted})' || index_tablespace;
146        |          -- creating a constraint allows it to be attached to the parent without any additional checks
147        |          EXECUTE 'ALTER TABLE  ${info.schema.quoted}.' || quote_ident(partition_name) ||
148        |            ' ADD CONSTRAINT ' || quote_ident(partition_name || '_constraint') ||
149        |            ' CHECK ( ${info.cols.dtg.quoted} >= ' || quote_literal(partition_start) ||
150        |            ' AND ${info.cols.dtg.quoted} < ' || quote_literal(partition_end) || ' )';
151        |        END IF;
152        |        RAISE INFO '% Done writing % rows to partition %', timeofday()::timestamp, unsorted_count, partition_name;
153        |
154        |        IF partition_parent = ${info.tables.spillPartitions.name.asLiteral} THEN
155        |          -- store record of unsorted row counts which could negatively impact BRIN index scans
156        |          INSERT INTO ${info.tables.sortQueue.name.qualified}(partition_name, unsorted_count, enqueued)
157        |            VALUES (partition_name, unsorted_count, now());
158        |          RAISE NOTICE 'Inserting % rows into spill partition %, queries may be impacted',
159        |                unsorted_count, partition_name;
160        |        END IF;
161        |
162        |        IF NOT pexists THEN
163        |          EXECUTE 'ALTER TABLE ${info.schema.quoted}.' || quote_ident(partition_parent) ||
164        |            ' ATTACH PARTITION ${info.schema.quoted}.' || quote_ident(partition_name) ||
165        |            ' FOR VALUES FROM (' || quote_literal(partition_start) ||
166        |            ') TO (' || quote_literal(partition_end) || ' );';
167        |          -- now that we've attached the table we can drop the redundant constraint
168        |          -- however, this requires ACCESS EXCLUSIVE - since constraints are only checked on inserts
169        |          -- or updates, and partition tables are 'immutable' (only written to once), it shouldn't
170        |          -- affect anything to leave it. note that for 'spill' tables, there may be some redundant checks
171        |          -- EXECUTE 'ALTER TABLE ${info.schema.quoted}.' || quote_ident(partition_name) ||
172        |          --  ' DROP CONSTRAINT ' || quote_ident(partition_name || '_constraint');
173        |          RAISE NOTICE 'A partition has been created %', partition_name;
174        |        END IF;
175        |
176        |        -- drop the tables that we've copied out
177        |        EXECUTE 'DROP VIEW ' || quote_ident(partition_name || '_tmp_migrate');
178        |        -- TODO this requires ACCESS EXCLUSIVE
179        |        FOREACH write_ahead_partition IN ARRAY write_ahead_partitions LOOP
180        |          EXECUTE 'DROP TABLE ' || write_ahead_partition;
181        |          RAISE NOTICE 'A partition has been deleted %', write_ahead_partition;
182        |        END LOOP;
183        |
184        |        -- mark the partition to be analyzed in a separate thread
185        |        INSERT INTO ${info.tables.analyzeQueue.name.qualified}(partition_name, enqueued)
186        |          VALUES (partition_name, now());
187        |
188        |        -- commit after each move, also releases the table locks
189        |        COMMIT;
190        |
191        |      END LOOP;
192        |    END;
193        |  $$BODY$$;
194        |""".stripMargin
195   }
