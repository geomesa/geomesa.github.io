15  * Merge recent write-ahead partitions and move them into the main partition table

23   private def proc(info: TypeInfo): String = {
24     val hours = info.partitions.hoursPerPartition
25     val writeAheadPartitions = info.tables.writeAheadPartitions
26     val mainPartitions = info.tables.mainPartitions
27     val spillPartitions = info.tables.spillPartitions
28     val tablespacesTable = s"${info.schema.quoted}.${PartitionTablespacesTable.Name.quoted}"
29     val dtgCol = info.cols.dtg.quoted
30     val geomCol = info.cols.geom.quoted

32     s"""CREATE OR REPLACE PROCEDURE ${name(info).quoted}(cur_time timestamp without time zone) LANGUAGE plpgsql AS
33        |  $$BODY$$
34        |    DECLARE
35        |      min_dtg timestamp without time zone;         -- min date in our partitioned tables
36        |      main_cutoff timestamp without time zone;     -- max age of the records for main tables
37        |      partition_start timestamp without time zone; -- start bounds for the partition we're writing
38        |      partition_end timestamp without time zone;   -- end bounds for the partition we're writing
39        |      partition_name text;                         -- partition table name
40        |      partition_parent text;                       -- partition table to attach to
41        |      partition_tablespace text;                   -- partition tablespace
42        |      index_tablespace text;                       -- index tablespace
43        |      write_ahead_partitions text[];               -- names of the partitions we're migrating
44        |      write_ahead_partition text;                  -- name of current partition
45        |      pexists boolean;                             -- table exists check
46        |      unsorted_count bigint;
47        |    BEGIN
48        |      -- constants
49        |      main_cutoff := truncate_to_partition(cur_time, $hours) - INTERVAL '$hours HOURS';
50        |
51        |      -- move data from the write ahead partitions to the main partitions
52        |      LOOP
53        |        -- find the range of dates in the write ahead partition tables
54        |        SELECT min($dtgCol) INTO min_dtg FROM ${writeAheadPartitions.name.qualified}
55        |          WHERE $dtgCol < main_cutoff;
56        |        EXIT WHEN min_dtg IS NULL;
57        |
58        |        partition_start := truncate_to_partition(min_dtg, $hours);
59        |        partition_end := partition_start + INTERVAL '$hours HOURS';
60        |        partition_parent := ${mainPartitions.name.asLiteral};
61        |        partition_name := partition_parent || '_' || to_char(partition_start, 'YYYY_MM_DD_HH24');
62        |
63        |        SELECT EXISTS(SELECT FROM pg_tables WHERE schemaname = ${info.schema.asLiteral} AND tablename = partition_name)
64        |          INTO pexists;
65        |
66        |        -- if the partition already exists, write to the spill partition instead to avoid messing up the BRIN index
67        |        IF pexists THEN
68        |          partition_parent := ${spillPartitions.name.asLiteral};
69        |          partition_name := partition_parent || '_' || to_char(partition_start, 'YYYY_MM_DD_HH24');
70        |          SELECT EXISTS(SELECT FROM pg_tables WHERE schemaname = ${info.schema.asLiteral} AND tablename = partition_name)
71        |            INTO pexists;
72        |        END IF;
73        |
74        |        -- find the write ahead partitions we're copying from
75        |        -- order the results to ensure we get locks in a consistent order to avoid deadlocks
76        |        write_ahead_partitions := Array(
77        |          SELECT '${info.schema.quoted}.' || quote_ident(pg_class.relname)
78        |            FROM pg_catalog.pg_inherits
79        |            INNER JOIN pg_catalog.pg_class ON (pg_inherits.inhrelid = pg_class.oid)
80        |            INNER JOIN pg_catalog.pg_namespace ON (pg_class.relnamespace = pg_namespace.oid)
81        |            WHERE inhparent = ${writeAheadPartitions.name.asRegclass}
82        |              AND pg_class.relname >= ${writeAheadPartitions.name.asLiteral} || '_' || to_char(partition_start, 'YYYY_MM_DD_HH24_MI')
83        |              AND pg_class.relname < ${writeAheadPartitions.name.asLiteral} || '_' || to_char(partition_end, 'YYYY_MM_DD_HH24_MI')
84        |            ORDER BY 1
85        |          );
86        |
87        |        -- get a lock on the tables - this mode won't prevent reads but will prevent writes
88        |        -- (there shouldn't be any writes though) and will synchronize this method
89        |        -- TODO we really just need to sync this method for safety in manual invocations
90        |        -- FOREACH write_ahead_partition IN ARRAY write_ahead_partitions LOOP
91        |        --   EXECUTE 'LOCK TABLE ' || write_ahead_partition || ' IN SHARE ROW EXCLUSIVE MODE';
92        |        --   RAISE INFO '% Locked write ahead partition % for migration', timeofday()::timestamp, write_ahead_partition;
93        |        -- END LOOP;
94        |
95        |        -- create a view from the tables so that we can sort the result by an expression (geohash)
96        |        EXECUTE 'CREATE TEMP VIEW ' || quote_ident(partition_name || '_tmp_migrate') ||
97        |          ' AS SELECT * FROM ' || array_to_string(write_ahead_partitions, ' UNION ALL SELECT * FROM ');
98        |
99        |        -- copy rows from write ahead partitions to main partition table
100        |        IF pexists THEN
101        |          RAISE INFO '% Copying rows to partition %', timeofday()::timestamp, partition_name;
102        |          EXECUTE 'INSERT INTO ${info.schema.quoted}.' || quote_ident(partition_name) ||
103        |            ' SELECT * FROM ' || quote_ident(partition_name || '_tmp_migrate') ||
104        |            '   ORDER BY _st_sortablehash($geomCol)' ||
105        |            '   ON CONFLICT DO NOTHING';
106        |          GET DIAGNOSTICS unsorted_count := ROW_COUNT;
107        |        ELSE
108        |          RAISE INFO '% Creating partition with insert % (unattached)', timeofday()::timestamp, partition_name;
109        |          -- create the partition table with a 'create as' for improved performance
110        |          SELECT table_space INTO partition_tablespace FROM $tablespacesTable
111        |            WHERE type_name = ${literal(info.typeName)} AND table_type = ${PartitionedTableSuffix.quoted};
112        |          IF partition_tablespace IS NULL THEN
113        |            index_tablespace := '';
114        |            partition_tablespace := '';
115        |          ELSE
116        |            index_tablespace := ' USING INDEX TABLESPACE '|| quote_ident(partition_tablespace);
117        |            partition_tablespace := ' TABLESPACE ' || quote_ident(partition_tablespace);
118        |          END IF;
119        |          -- upper bounds are exclusive
120        |          -- this won't have any indices until we attach it to the parent partition table
121        |          -- use "create table as" (vs create then insert) for performance benefits related to WAL skipping
122        |          -- we need a "select distinct" to avoid primary key conflicts - this should be fairly cheap since
123        |          --   we're already sorting and there should be few or no conflicts
124        |          EXECUTE 'CREATE ${info.walLogSQL} TABLE ${info.schema.quoted}.' || quote_ident(partition_name) ||
125        |            partition_tablespace || ' AS SELECT DISTINCT ON' ||
126        |            ' (_st_sortablehash($geomCol), fid, ${info.cols.dtg.quoted}) * FROM ' ||
127        |            quote_ident(partition_name || '_tmp_migrate') || ' ORDER BY _st_sortablehash($geomCol)';
128        |          GET DIAGNOSTICS unsorted_count := ROW_COUNT;
129        |          EXECUTE 'ALTER TABLE ${info.schema.quoted}.' || quote_ident(partition_name) ||
130        |            ' ADD CONSTRAINT ' || quote_ident(partition_name || '_pkey') ||
131        |            ' PRIMARY KEY (fid, ${info.cols.dtg.quoted})' || index_tablespace;
132        |          -- creating a constraint allows it to be attached to the parent without any additional checks
133        |          EXECUTE 'ALTER TABLE  ${info.schema.quoted}.' || quote_ident(partition_name) ||
134        |            ' ADD CONSTRAINT ' || quote_ident(partition_name || '_constraint') ||
135        |            ' CHECK ( $dtgCol >= ' || quote_literal(partition_start) ||
136        |            ' AND $dtgCol < ' || quote_literal(partition_end) || ' )';
137        |        END IF;
138        |        RAISE INFO '% Done writing % rows to partition %', timeofday()::timestamp, unsorted_count, partition_name;
139        |
140        |        IF partition_parent = ${spillPartitions.name.asLiteral} THEN
141        |          -- store record of unsorted row counts which could negatively impact BRIN index scans
142        |          INSERT INTO ${info.tables.sortQueue.name.qualified}(partition_name, unsorted_count, enqueued)
143        |            VALUES (partition_name, unsorted_count, now());
144        |          RAISE NOTICE 'Inserting % rows into spill partition %, queries may be impacted',
145        |                unsorted_count, partition_name;
146        |        END IF;
147        |
148        |        IF NOT pexists THEN
149        |          EXECUTE 'ALTER TABLE ${info.schema.quoted}.' || quote_ident(partition_parent) ||
150        |            ' ATTACH PARTITION ${info.schema.quoted}.' || quote_ident(partition_name) ||
151        |            ' FOR VALUES FROM (' || quote_literal(partition_start) ||
152        |            ') TO (' || quote_literal(partition_end) || ' );';
153        |          -- now that we've attached the table we can drop the redundant constraint
154        |          -- however, this requires ACCESS EXCLUSIVE - since constraints are only checked on inserts
155        |          -- or updates, and partition tables are 'immutable' (only written to once), it shouldn't
156        |          -- affect anything to leave it. note that for 'spill' tables, there may be some redundant checks
157        |          -- EXECUTE 'ALTER TABLE ${info.schema.quoted}.' || quote_ident(partition_name) ||
158        |          --  ' DROP CONSTRAINT ' || quote_ident(partition_name || '_constraint');
159        |          RAISE NOTICE 'A partition has been created %', partition_name;
160        |        END IF;
161        |
162        |        -- drop the tables that we've copied out
163        |        EXECUTE 'DROP VIEW ' || quote_ident(partition_name || '_tmp_migrate');
164        |        -- TODO this requires ACCESS EXCLUSIVE
165        |        FOREACH write_ahead_partition IN ARRAY write_ahead_partitions LOOP
166        |          EXECUTE 'DROP TABLE ' || write_ahead_partition;
167        |          RAISE NOTICE 'A partition has been deleted %', write_ahead_partition;
168        |        END LOOP;
169        |
170        |        -- mark the partition to be analyzed in a separate thread
171        |        INSERT INTO ${info.tables.analyzeQueue.name.qualified}(partition_name, enqueued)
172        |          VALUES (partition_name, now());
173        |
174        |        -- commit after each move, also releases the table locks
175        |        COMMIT;
176        |
177        |      END LOOP;
178        |    END;
179        |  $$BODY$$;
180        |""".stripMargin
181   }
