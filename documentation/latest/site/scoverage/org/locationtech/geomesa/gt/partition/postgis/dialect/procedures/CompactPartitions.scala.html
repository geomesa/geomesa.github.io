14 /**
15  * Re-sorts a partition table. Can be used for maintenance if time-latent data comes in and degrades performance
16  * of the BRIN index
17  */
18 object CompactPartitions extends SqlProcedure {
19 
20   override def name(info: TypeInfo): FunctionName = FunctionName(s"${info.typeName}_compact_partitions")
21 
22   override protected def createStatements(info: TypeInfo): Seq[String] = Seq(proc(info))
23 
24   private def proc(info: TypeInfo): String = {
25     val dtgCol = info.cols.dtg.quoted
26     val geomCol = info.cols.geom.quoted
27     val hours = info.partitions.hoursPerPartition
28     val partitionsTable = s"${info.schema.quoted}.${PartitionTablespacesTable.Name.quoted}"
29     s"""CREATE OR REPLACE PROCEDURE ${name(info).quoted}(for_date timestamp without time zone) LANGUAGE plpgsql AS
30        |  $$BODY$$
31        |    DECLARE
32        |      partition_name text;
33        |      spill_partition text;
34        |      min_dtg timestamp without time zone;         -- min date in our partitioned tables
35        |      partition_start timestamp without time zone; -- start bounds for the partition we're writing
36        |      partition_end timestamp without time zone;   -- end bounds for the partition we're writing
37        |      partition_tablespace text;
38        |      index_tablespace text;                       -- index tablespace
39        |      unsorted_count bigint;
40        |      pexists boolean;                             -- table exists check
41        |    BEGIN
42        |      IF for_date IS NULL THEN
43        |        RAISE EXCEPTION 'date is null' USING HINT = 'Please use a valid date';
44        |      END IF;
45        |
46        |      partition_name :=  ${info.tables.mainPartitions.name.asLiteral} || '_' ||
47        |        to_char(truncate_to_partition(for_date, $hours), 'YYYY_MM_DD_HH24');
48        |
49        |      SELECT table_space INTO partition_tablespace FROM $partitionsTable
50        |        WHERE type_name = ${literal(info.typeName)} AND table_type = ${PartitionedTableSuffix.quoted};
51        |      IF partition_tablespace IS NULL THEN
52        |        index_tablespace := '';
53        |        partition_tablespace := '';
54        |      ELSE
55        |        index_tablespace := ' USING INDEX TABLESPACE '|| quote_ident(partition_tablespace);
56        |        partition_tablespace := ' TABLESPACE ' || quote_ident(partition_tablespace);
57        |      END IF;
58        |
59        |      RAISE INFO '% Compacting partition table %', timeofday()::timestamp, partition_name;
60        |      LOCK TABLE ONLY ${info.tables.mainPartitions.name.qualified} IN SHARE UPDATE EXCLUSIVE MODE;
61        |      -- lock the child table to prevent any inserts that would be lost
62        |      EXECUTE 'LOCK TABLE ${info.schema.quoted}.' || quote_ident(partition_name) ||
63        |        ' IN SHARE ROW EXCLUSIVE MODE';
64        |      EXECUTE 'SELECT $dtgCol FROM ${info.schema.quoted}.' || quote_ident(partition_name) ||
65        |        ' LIMIT 1' INTO min_dtg;
66        |      partition_start := truncate_to_partition(min_dtg, $hours);
67        |      partition_end := partition_start + INTERVAL '$hours HOURS';
68        |      spill_partition := ${info.tables.spillPartitions.name.asLiteral} || '_' || to_char(partition_start, 'YYYY_MM_DD_HH24');
69        |
70        |      SELECT EXISTS(SELECT FROM pg_tables WHERE schemaname = ${info.schema.asLiteral} AND tablename = spill_partition)
71        |        INTO pexists;
72        |
73        |      -- use "create table as" (vs create then insert) for performance benefits related to WAL skipping
74        |      IF pexists THEN
75        |        -- lock the child table to prevent any inserts that would be lost
76        |        EXECUTE 'LOCK TABLE ${info.schema.quoted}.' || quote_ident(spill_partition) ||
77        |          ' IN SHARE ROW EXCLUSIVE MODE';
78        |        EXECUTE 'CREATE TABLE ${info.schema.quoted}.' || quote_ident(partition_name || '_tmp_sort') ||
79        |          partition_tablespace || ' AS SELECT * FROM' ||
80        |          ' (SELECT * FROM ${info.schema.quoted}.' || quote_ident(partition_name) ||
81        |          ' UNION ALL SELECT * FROM ${info.schema.quoted}.' || quote_ident(spill_partition) ||
82        |          ') results' ||
83        |          ' ORDER BY _st_sortablehash($geomCol)';
84        |        GET DIAGNOSTICS unsorted_count := ROW_COUNT;
85        |      ELSE
86        |        EXECUTE 'CREATE TABLE ${info.schema.quoted}.' || quote_ident(partition_name || '_tmp_sort') ||
87        |          partition_tablespace || ' AS SELECT * FROM ' || quote_ident(partition_name) ||
88        |          ' ORDER BY _st_sortablehash($geomCol)';
89        |        GET DIAGNOSTICS unsorted_count := ROW_COUNT;
90        |      END IF;
91        |
92        |      EXECUTE 'ALTER TABLE ${info.schema.quoted}.' || quote_ident(partition_name || '_tmp_sort') ||
93        |        ' ADD CONSTRAINT ' || quote_ident(partition_name || '_pkey_tmp_sort') ||
94        |        ' PRIMARY KEY (fid, $dtgCol)' || index_tablespace;
95        |      -- creating a constraint allows it to be attached to the parent without any additional checks
96        |      EXECUTE 'ALTER TABLE  ${info.schema.quoted}.' || quote_ident(partition_name || '_tmp_sort') ||
97        |        ' ADD CONSTRAINT ' || quote_ident(partition_name || '_constraint_tmp_sort') ||
98        |        ' CHECK ( $dtgCol >= ' || quote_literal(partition_start) ||
99        |        ' AND $dtgCol < ' || quote_literal(partition_end) || ' )';
100        |
101        |      -- create indices before attaching to minimize time to attach, copied from PartitionTables code
102        |      EXECUTE 'CREATE INDEX IF NOT EXISTS ' || quote_ident(partition_name || '_${info.cols.geom.raw}_tmp_sort') ||
103        |        ' ON ${info.schema.quoted}.' || quote_ident(partition_name || '_tmp_sort') ||
104        |        ' USING BRIN($geomCol)' || partition_tablespace;
105        |      EXECUTE 'CREATE INDEX IF NOT EXISTS ' || quote_ident(partition_name || '_${info.cols.dtg.raw}_tmp_sort') ||
106        |        ' ON ${info.schema.quoted}.' || quote_ident(partition_name || '_tmp_sort') ||
107        |        ' ($dtgCol)' || partition_tablespace;
108        |${info.cols.indexed.map { col =>
109     s"""      EXECUTE 'CREATE INDEX IF NOT EXISTS ' || quote_ident(partition_name || '_${col.raw}_tmp_sort') ||
110        |        ' ON ${info.schema.quoted}.' || quote_ident(partition_name || '_tmp_sort') ||
111        |        ' (${col.quoted})' || partition_tablespace;""".stripMargin}.mkString("\n")}
112        |
113        |      RAISE INFO '% Dropping old partition table (queries will be blocked) %', timeofday()::timestamp, partition_name;
114        |      EXECUTE 'DROP TABLE IF EXISTS ${info.schema.quoted}.' || quote_ident(partition_name);
115        |      IF pexists THEN
116        |        EXECUTE 'DROP TABLE IF EXISTS ${info.schema.quoted}.' || quote_ident(spill_partition);
117        |      END IF;
118        |
119        |      RAISE INFO '% Renaming newly sorted partition table %', timeofday()::timestamp, partition_name;
120        |      EXECUTE 'ALTER TABLE ${info.schema.quoted}.' || quote_ident(partition_name || '_tmp_sort') ||
121        |        ' RENAME TO ' || quote_ident(partition_name);
122        |      EXECUTE 'ALTER TABLE ${info.schema.quoted}.' || quote_ident(partition_name) ||
123        |        ' RENAME CONSTRAINT ' || quote_ident(partition_name || '_pkey_tmp_sort') ||
124        |        ' TO ' || quote_ident(partition_name || '_pkey');
125        |      EXECUTE 'ALTER INDEX ' || quote_ident(partition_name || '_${info.cols.geom.raw}_tmp_sort') ||
126        |        ' RENAME TO ' || quote_ident(partition_name || '_${info.cols.geom.raw}');
127        |      EXECUTE 'ALTER INDEX ' || quote_ident(partition_name || '_${info.cols.dtg.raw}_tmp_sort') ||
128        |        ' RENAME TO ' || quote_ident(partition_name || '_${info.cols.dtg.raw}');
129        |${info.cols.indexed.map { col =>
130     s"""      EXECUTE 'ALTER INDEX ' || quote_ident(partition_name || '_${col.raw}_tmp_sort') ||
131        |        ' RENAME TO ' || quote_ident(partition_name || '_${col.raw}');""".stripMargin}.mkString("\n")}
132        |
133        |      RAISE INFO '% Attaching newly sorted partition table %', timeofday()::timestamp, partition_name;
134        |      EXECUTE 'ALTER TABLE ${info.tables.mainPartitions.name.qualified}' ||
135        |        ' ATTACH PARTITION ${info.schema.quoted}.' || quote_ident(partition_name) ||
136        |        ' FOR VALUES FROM (' || quote_literal(partition_start) || ') TO (' ||
137        |        quote_literal(partition_end) || ' );';
138        |      RAISE INFO '% Done compacting partition table %', timeofday()::timestamp, partition_name;
139        |
140        |      -- now that we've attached the table we can drop the redundant constraint
141        |      RAISE INFO '% Dropping constraint for partition table %', timeofday()::timestamp, partition_name;
142        |      EXECUTE 'ALTER TABLE ${info.schema.quoted}.' || quote_ident(partition_name) ||
143        |        ' DROP CONSTRAINT ' || quote_ident(partition_name || '_constraint_tmp_sort');
144        |
145        |      -- mark the partition to be analyzed in a separate thread
146        |      INSERT INTO ${info.tables.analyzeQueue.name.qualified}(partition_name, enqueued)
147        |        VALUES (partition_name, now());
148        |
149        |      -- commit to release our locks
150        |      COMMIT;
151        |      RAISE INFO '% Done compacting % rows in partition %', timeofday()::timestamp, unsorted_count, partition_name;
152        |
153        |    END;
154        |  $$BODY$$;
155        |""".stripMargin
156   }
