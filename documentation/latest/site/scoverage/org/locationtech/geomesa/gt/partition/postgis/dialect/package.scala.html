91 
92     private val statement = cx.createStatement()
93     private var rollback = false
94 
95     /**
96      * Execute a sql command
97      *
98      * @param sql sql string
99      */
100     def execute(sql: String): Unit = {
101       try {
102         logger.debug(sql)
103         statement.execute(sql)
104       } catch {
105         case NonFatal(e) => rollback = true; throw e
106       }
107     }
108 
109     /**
110      * Execute an update statements
111      *
112      * @param sql sql string, with arguments denoted by '?'
113      * @param args arguments to the update statement, must match the argument placeholders in the sql string
114      */
115     def executeUpdate(sql: String, args: Seq[Any]): Unit = {
116       val ps: PreparedStatement = cx.prepareStatement(sql)
117       try {
118         var i = 1
119         args.foreach { arg =>
120           ps.setObject(i, arg)
121           i += 1
122         }
123         logger.debug(s"$sql, ${args.mkString(", ")}")
124         ps.executeUpdate()
125       } catch {
126         case NonFatal(e) => rollback = true; throw e
127       } finally {
128         ps.close()
129       }
130     }
131 
132     override def close(): Unit = {
133       if (!cx.getAutoCommit) {
134         if (rollback) {
135           Try(cx.rollback())
136         } else {
137           cx.commit()
138         }
139       }
140       statement.close()
141     }

322     def apply(original: SimpleFeatureType): Columns = {

324       TemporalIndexCheck.validateDtgField(sft)
325       val dtg = sft.getDtgField.map(ColumnName.apply).getOrElse {

334       //    Seq(dtg) ++ sft.getAttributeDescriptors.asScala.collect { case d if d.isIndexValue() => ColumnName(d) }

339             // accept any index flags, so config can be consistent across data stores
340             !i.isBlank && !"false".equalsIgnoreCase(i) && !"none".equalsIgnoreCase(i)

583            |BEGIN
584            |  IF NOT EXISTS (SELECT FROM pg_trigger WHERE tgname = ${tgName.asLiteral} AND tgrelid = ${tgTable.asRegclass}) THEN
585            |    CREATE TRIGGER ${tgName.quoted}
586            |      $action ON ${tgTable.qualified}
587            |      FOR EACH ROW EXECUTE PROCEDURE ${info.schema.quoted}.${name(info).quoted}();
588            |  END IF;
589            |END$$$$;""".stripMargin
