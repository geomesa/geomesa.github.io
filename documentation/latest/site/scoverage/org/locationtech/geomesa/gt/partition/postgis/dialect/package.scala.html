87    * Note: does not do any validation, sql must be sanitized by the caller.

91   class ExecutionContext(val cx: Connection) extends Closeable with StrictLogging {
92 
93     private val statement = cx.createStatement()
94     private var rollback = false
95 
96     /**
97      * Execute a sql command
98      *
99      * @param sql sql string
100      */
101     def execute(sql: String): Unit = {
102       try {
103         logger.debug(sql)
104         statement.execute(sql)
105       } catch {
106         case NonFatal(e) => rollback = true; throw e
107       }
108     }
109 
110     /**
111      * Execute an update statements
112      *
113      * @param sql sql string, with arguments denoted by '?'
114      * @param args arguments to the update statement, must match the argument placeholders in the sql string
115      */
116     def executeUpdate(sql: String, args: Seq[Any]): Unit = {
117       val ps: PreparedStatement = cx.prepareStatement(sql)
118       try {
119         var i = 1
120         args.foreach { arg =>
121           ps.setObject(i, arg)
122           i += 1
123         }
124         logger.debug(s"$sql, ${args.mkString(", ")}")
125         ps.executeUpdate()
126       } catch {
127         case NonFatal(e) => rollback = true; throw e
128       } finally {
129         ps.close()
130       }
131     }
132 
133     override def close(): Unit = {
134       if (!cx.getAutoCommit) {
135         if (rollback) {
136           Try(cx.rollback())
137         } else {
138           cx.commit()
139         }
140       }
141       statement.close()
142     }
