123     // Throw an error if the sft name is longer than 31 characters

177   override def postCreateFeatureType(
178       sft: SimpleFeatureType,
179       metadata: DatabaseMetaData,
180       schemaName: String,
181       cx: Connection): Unit = {
182 
183     import PartitionedPostgisDialect.Config._
184 
185     // normally views get set to read-only, override that here since we use triggers to delegate writes
186     sft.getUserData.remove(JDBCDataStore.JDBC_READ_ONLY)
187 
188     // populate user data
189     val userDataSql = s"select key, value from ${escape(schemaName)}.${UserDataTable.Name.quoted} where type_name = ?"
190     WithClose(cx.prepareStatement(userDataSql)) { statement =>
191       statement.setString(1, sft.getTypeName)
192       WithClose(statement.executeQuery()) { rs =>
193         while (rs.next()) {
194           sft.getUserData.put(rs.getString(1), rs.getString(2))
195         }
196       }
197     }
198 
199     // populate tablespaces
200     val tablespaceSql =
201       s"select table_space, table_type from " +
202           s"${escape(schemaName)}.${PartitionTablespacesTable.Name.quoted} where type_name = ?"
203     WithClose(cx.prepareStatement(tablespaceSql)) { statement =>
204       statement.setString(1, sft.getTypeName)
205       WithClose(statement.executeQuery()) { rs =>
206         while (rs.next()) {
207           val ts = rs.getString(1)
208           if (ts != null && ts.nonEmpty) {
209             rs.getString(2) match {
210               case WriteAheadTableSuffix.raw => sft.getUserData.put(WriteAheadTableSpace, ts)
211               case PartitionedWriteAheadTableSuffix.raw => sft.getUserData.put(WriteAheadPartitionsTableSpace, ts)
212               case PartitionedTableSuffix.raw => sft.getUserData.put(MainTableSpace, ts)
213               case s => logger.warn(s"Ignoring unexpected tablespace table: $s")
214             }
215           }
216         }
217       }
218     }

222     // due to the JDBCDataStore hard-coding "DROP TABLE" we have to redirect it away from the main view,

272   override def encodePostColumnCreateTable(att: AttributeDescriptor, sql: StringBuffer): Unit = {
273     import PartitionedPostgisDialect.Config.GeometryAttributeConversions
274     att match {
275       case gd: GeometryDescriptor =>
276         val nullable = gd.getMinOccurs <= 0 || gd.isNillable
277         val i = sql.lastIndexOf("geometry")
278         // expect `geometry NOT NULL` or `geometry` depending on nullable flag
279         if (i == -1 || (nullable && i != sql.length() - 8) || (!nullable && i != sql.length() - 17)) {
280           logger.warn(s"Found geometry-type attribute but no geometry column binding: $sql")
281         } else {
282           val srid = gd.getSrid.getOrElse(-1)
283           val geomType = PartitionedPostgisDialect.GeometryMappings.getOrElse(gd.getType.getBinding, "GEOMETRY")
284           val geomTypeWithDims = gd.getCoordinateDimensions match {
285             case None | Some(2) => geomType
286             case Some(3) => s"${geomType}Z"
287             case Some(4) => s"${geomType}ZM"
288             case Some(d) =>
289               throw new IllegalArgumentException(
290                 s"PostGIS only supports geometries with 2, 3 and 4 dimensions, but found: $d")
291           }
292           sql.insert(i + 8, s" ($geomTypeWithDims, $srid)")
293         }
294 
295       case _ if att.isJson() =>
296         // replace 'VARCHAR' with jsonb
297         val i = sql.lastIndexOf(" VARCHAR")
298         if (i == sql.length() - 8) {
299           sql.replace(i + 1, i + 8, "JSONB")
300         } else {
301           logger.warn(s"Found JSON-type attribute but no CHARACTER VARYING column binding: $sql")
302         }
303 
304       case _ if att.isList =>
305         // go back and encode the array type in the CQL create statement
306         val i = sql.lastIndexOf(" ARRAY")
307         if (i == sql.length() - 6) {
308           sql.insert(i, " " + getListTypeMapping(att.getListType()))
309         } else {
310           logger.warn(s"Found list-type attribute but no ARRAY column binding: $sql")
311         }
312 
313       case _ => // no-op
314     }
315   }

326   /**
327    * Gets the array type for a list/array column
328    *
329    * @param binding list-type binding
330    * @return
331    */
332   private def getListTypeMapping(binding: Class[_]): String = {
333     val mappings = new java.util.HashMap[String, Class[_]]()
334     registerSqlTypeNameToClassMappings(mappings)
335     var mapping: String = null
336     var partial: String = null
337     val iter = mappings.asScala.iterator
338     while (iter.hasNext && mapping == null) {
339       val (name, clas) = iter.next
340       if (clas == binding) {
341         mapping = name
342       } else if (partial == null && clas.isAssignableFrom(binding)) {
343         partial = name
344       }
345     }
346     if (mapping != null) {
347       mapping
348     } else if (partial != null) {
349       partial
350     } else {
351       logger.warn(s"Could not find list-type column for type ${binding.getName}")
352       "text"
353     }
354   }
