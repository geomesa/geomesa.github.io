123   override def postCreateTable(schemaName: String, sft: SimpleFeatureType, cx: Connection): Unit = {
124     // Throw an error if the sft name is longer than 31 characters
125     if (sft.getTypeName.length() > 31) {
126       val errorMsg = "Can't create schema: type name exceeds max supported length of 31 characters"
127       throw new IllegalArgumentException(errorMsg)
128     }
129 
130     // note: we skip the call to `super`, which creates a spatial index (that we don't want), and which
131     // alters the geometry column types (which we handle in the create statement)
132     val info = TypeInfo(schemaName, sft)
133     implicit val ex: ExecutionContext = new ExecutionContext(cx)
134     try {
135       PartitionedPostgisDialect.Commands.foreach(_.create(info))
136     } finally {
137       ex.close()
138     }

178   private def querySftAttributesWithIndices(sft: SimpleFeatureType, cx: Connection): List[String] = {
179     val attributesWithIndicesSql =
180       s"""
181          |select distinct(att.attname) as indexed_attribute_name
182          |from pg_class obj
183          |join pg_index idx on idx.indrelid = obj.oid
184          |join pg_attribute att on att.attrelid = obj.oid and att.attnum = any(idx.indkey)
185          |join pg_views v on v.viewname = ?
186          |where obj.relname = concat(?, ${PartitionedTableSuffix.quoted})
187          |order by att.attname
188          |""".stripMargin
189     val sftTypeName = sft.getTypeName
190     Try(
191       WithClose(cx.prepareStatement(attributesWithIndicesSql)) { statement =>
192         statement.setString(1, sftTypeName)
193         statement.setString(2, sftTypeName)
194         WithClose(statement.executeQuery()) { rs =>
195           Iterator.continually(rs.next(), rs)
196             .takeWhile(_._1)
197             .map(_._2.getString(1))
198             .toList
199         }
200       }
201     ).fold(
202       exception => {
203         logger.error(s"${getClass.getSimpleName}: Error loading attributes with indices for SFT: $sftTypeName due to: ${exception.getMessage}", exception)
204         List.empty
205       },
206       (indexedAttributeNames: List[String]) => indexedAttributeNames
207     )
208   }

210   override def postCreateFeatureType(
211       sft: SimpleFeatureType,
212       metadata: DatabaseMetaData,
213       schemaName: String,
214       cx: Connection): Unit = {
215 
216     import PartitionedPostgisDialect.Config._
217 
218     // normally views get set to read-only, override that here since we use triggers to delegate writes
219     sft.getUserData.remove(JDBCDataStore.JDBC_READ_ONLY)
220 
221     // populate user data
222     val userDataSql = s"select key, value from ${escape(schemaName)}.${UserDataTable.Name.quoted} where type_name = ?"
223     WithClose(cx.prepareStatement(userDataSql)) { statement =>
224       statement.setString(1, sft.getTypeName)
225       WithClose(statement.executeQuery()) { rs =>
226         while (rs.next()) {
227           sft.getUserData.put(rs.getString(1), rs.getString(2))
228         }
229       }
230     }
231 
232     // populate tablespaces
233     val tablespaceSql =
234       s"select table_space, table_type from " +
235           s"${escape(schemaName)}.${PartitionTablespacesTable.Name.quoted} where type_name = ?"
236     WithClose(cx.prepareStatement(tablespaceSql)) { statement =>
237       statement.setString(1, sft.getTypeName)
238       WithClose(statement.executeQuery()) { rs =>
239         while (rs.next()) {
240           val ts = rs.getString(1)
241           if (ts != null && ts.nonEmpty) {
242             rs.getString(2) match {
243               case WriteAheadTableSuffix.raw => sft.getUserData.put(WriteAheadTableSpace, ts)
244               case PartitionedWriteAheadTableSuffix.raw => sft.getUserData.put(WriteAheadPartitionsTableSpace, ts)
245               case PartitionedTableSuffix.raw => sft.getUserData.put(MainTableSpace, ts)
246               case s => logger.warn(s"Ignoring unexpected tablespace table: $s")
247             }
248           }
249         }
250       }
251     }
252 
253     // populate indices data
254     querySftAttributesWithIndices(sft, cx).foreach { idxAttribute =>
255       for {
256         attDesc <- Option(sft.getDescriptor(idxAttribute))
257         userData <- Option(attDesc.getUserData)
258       } userData.put(AttributeOptions.OptIndex, "true")
259     }
260   }

262   override def preDropTable(schemaName: String, sft: SimpleFeatureType, cx: Connection): Unit = {
263     // due to the JDBCDataStore hard-coding "DROP TABLE" we have to redirect it away from the main view,
264     // and we can't drop the write ahead table so that it has something to drop
265     dropping.set(true)
266     val info = TypeInfo(schemaName, sft)
267 
268     implicit val ex: ExecutionContext = new ExecutionContext(cx)
269     try {
270       PartitionedPostgisDialect.Commands.reverse.filter(_ != WriteAheadTable).foreach(_.drop(info))
271     } finally {
272       ex.close()
273     }

313   override def encodePostColumnCreateTable(att: AttributeDescriptor, sql: StringBuffer): Unit = {
314     import PartitionedPostgisDialect.Config.GeometryAttributeConversions
315     att match {
316       case gd: GeometryDescriptor =>
317         val nullable = gd.getMinOccurs <= 0 || gd.isNillable
318         val i = sql.lastIndexOf("geometry")
319         // expect `geometry NOT NULL` or `geometry` depending on nullable flag
320         if (i == -1 || (nullable && i != sql.length() - 8) || (!nullable && i != sql.length() - 17)) {
321           logger.warn(s"Found geometry-type attribute but no geometry column binding: $sql")
322         } else {
323           val srid = gd.getSrid.getOrElse(-1)
324           val geomType = PartitionedPostgisDialect.GeometryMappings.getOrElse(gd.getType.getBinding, "GEOMETRY")
325           val geomTypeWithDims = gd.getCoordinateDimensions match {
326             case None | Some(2) => geomType
327             case Some(3) => s"${geomType}Z"
328             case Some(4) => s"${geomType}ZM"
329             case Some(d) =>
330               throw new IllegalArgumentException(
331                 s"PostGIS only supports geometries with 2, 3 and 4 dimensions, but found: $d")
332           }
333           sql.insert(i + 8, s" ($geomTypeWithDims, $srid)")
334         }
335 
336       case _ if att.isJson() =>
337         // replace 'VARCHAR' with jsonb
338         val i = sql.lastIndexOf(" VARCHAR")
339         if (i == sql.length() - 8) {
340           sql.replace(i + 1, i + 8, "JSONB")
341         } else {
342           logger.warn(s"Found JSON-type attribute but no CHARACTER VARYING column binding: $sql")
343         }
344 
345       case _ if att.isList =>
346         // go back and encode the array type in the CQL create statement
347         val i = sql.lastIndexOf(" ARRAY")
348         if (i == sql.length() - 6) {
349           sql.insert(i, " " + getListTypeMapping(att.getListType()))
350         } else {
351           logger.warn(s"Found list-type attribute but no ARRAY column binding: $sql")
352         }
353 
354       case _ => // no-op
355     }
356   }

367   /**
368    * Gets the array type for a list/array column
369    *
370    * @param binding list-type binding
371    * @return
372    */
373   private def getListTypeMapping(binding: Class[_]): String = {
374     val mappings = new java.util.HashMap[String, Class[_]]()
375     registerSqlTypeNameToClassMappings(mappings)
376     var mapping: String = null
377     var partial: String = null
378     val iter = mappings.asScala.iterator
379     while (iter.hasNext && mapping == null) {
380       val (name, clas) = iter.next
381       if (clas == binding) {
382         mapping = name
383       } else if (partial == null && clas.isAssignableFrom(binding)) {
384         partial = name
385       }
386     }
387     if (mapping != null) {
388       mapping
389     } else if (partial != null) {
390       partial
391     } else {
392       logger.warn(s"Could not find list-type column for type ${binding.getName}")
393       "text"
394     }
395   }
