133   override def postCreateTable(schemaName: String, sft: SimpleFeatureType, cx: Connection): Unit = {
134     // Throw an error if the sft name is longer than 31 characters
135     if (sft.getTypeName.length() > 31) {
136       val errorMsg = "Can't create schema: type name exceeds max supported length of 31 characters"
137       throw new IllegalArgumentException(errorMsg)
138     }
139 
140     // note: we skip the call to `super`, which creates a spatial index (that we don't want), and which
141     // alters the geometry column types (which we handle in the create statement)
142     val info = TypeInfo(schemaName, sft)
143     implicit val ex: ExecutionContext = new ExecutionContext(cx)
144     try {
145       PartitionedPostgisDialect.Commands.foreach(_.create(info))
146       if (grants.nonEmpty) {
147         val roles = grants.map(_.quoted).mkString(", ")
148         val tables =
149           Seq(
150             info.tables.view.name,
151             info.tables.writeAhead.name,
152             info.tables.writeAheadPartitions.name,
153             info.tables.mainPartitions.name,
154             info.tables.spillPartitions.name,
155             TableIdentifier(schemaName, PrimaryKeyTable.Name.raw),
156             TableIdentifier(schemaName, UserDataTable.Name.raw)
157           )
158         tables.foreach { table =>
159           ex.execute(s"GRANT SELECT ON ${table.qualified} TO $roles;")
160         }
161       }
162     } finally {
163       ex.close()
164     }
165   }

225   override def preDropTable(schemaName: String, sft: SimpleFeatureType, cx: Connection): Unit = {
226     // due to the JDBCDataStore hard-coding "DROP TABLE" we have to redirect it away from the main view,
227     // and we can't drop the write ahead table so that it has something to drop
228     dropping.set(true)
229     val info = TypeInfo(schemaName, sft)
230 
231     implicit val ex: ExecutionContext = new ExecutionContext(cx)
232     try {
233       PartitionedPostgisDialect.Commands.reverse.filter(_ != WriteAheadTable).foreach(_.drop(info))
234     } finally {
235       ex.close()
236     }
237   }

275   override def encodePostColumnCreateTable(att: AttributeDescriptor, sql: StringBuffer): Unit = {
276     import PartitionedPostgisDialect.GeometryAttributeConversions
277     att match {
278       case gd: GeometryDescriptor =>
279         val nullable = gd.getMinOccurs <= 0 || gd.isNillable
280         val i = sql.lastIndexOf("geometry")
281         // expect `geometry NOT NULL` or `geometry` depending on nullable flag
282         if (i == -1 || (nullable && i != sql.length() - 8) || (!nullable && i != sql.length() - 17)) {
283           logger.warn(s"Found geometry-type attribute but no geometry column binding: $sql")
284         } else {
285           val srid = gd.getSrid.getOrElse(-1)
286           val geomType = PartitionedPostgisDialect.GeometryMappings.getOrElse(gd.getType.getBinding, "GEOMETRY")
287           val geomTypeWithDims = gd.getCoordinateDimensions match {
288             case None | Some(2) => geomType
289             case Some(3) => s"${geomType}Z"
290             case Some(4) => s"${geomType}ZM"
291             case Some(d) =>
292               throw new IllegalArgumentException(
293                 s"PostGIS only supports geometries with 2, 3 and 4 dimensions, but found: $d")
294           }
295           sql.insert(i + 8, s" ($geomTypeWithDims, $srid)")
296         }
297 
298       case _ if att.isJson() =>
299         // replace 'VARCHAR' with jsonb
300         val i = sql.lastIndexOf(" VARCHAR")
301         if (i == sql.length() - 8) {
302           sql.replace(i + 1, i + 8, "JSONB")
303         } else {
304           logger.warn(s"Found JSON-type attribute but no CHARACTER VARYING column binding: $sql")
305         }
306 
307       case _ if att.isList =>
308         // go back and encode the array type in the CQL create statement
309         val i = sql.lastIndexOf(" ARRAY")
310         if (i == sql.length() - 6) {
311           sql.insert(i, " " + getListTypeMapping(att.getListType()))
312         } else {
313           logger.warn(s"Found list-type attribute but no ARRAY column binding: $sql")
314         }
315 
316       case _ => // no-op
317     }
318   }

329   /**
330    * Gets the array type for a list/array column
331    *
332    * @param binding list-type binding
333    * @return
334    */
335   private def getListTypeMapping(binding: Class[_]): String = {
336     val mappings = new java.util.HashMap[String, Class[_]]()
337     registerSqlTypeNameToClassMappings(mappings)
338     var mapping: String = null
339     var partial: String = null
340     val iter = mappings.asScala.iterator
341     while (iter.hasNext && mapping == null) {
342       val (name, clas) = iter.next
343       if (clas == binding) {
344         mapping = name
345       } else if (partial == null && clas.isAssignableFrom(binding)) {
346         partial = name
347       }
348     }
349     if (mapping != null) {
350       mapping
351     } else if (partial != null) {
352       partial
353     } else {
354       logger.warn(s"Could not find list-type column for type ${binding.getName}")
355       "text"
356     }
357   }

363    * @param typeName feature type name

366   private def getIndexedColumns(cx: Connection, typeName: String): Seq[String] = {
367     val attributesWithIndicesSql =
368       s"""select distinct(att.attname) as indexed_attribute_name
369          |from pg_class obj
370          |join pg_index idx on idx.indrelid = obj.oid
371          |join pg_attribute att on att.attrelid = obj.oid and att.attnum = any(idx.indkey)
372          |join pg_views v on v.viewname = ?
373          |where obj.relname = concat(?, ${PartitionedTableSuffix.quoted})
374          |order by att.attname;""".stripMargin
375     try {
376       WithClose(cx.prepareStatement(attributesWithIndicesSql)) { statement =>
377         statement.setString(1, typeName)
378         statement.setString(2, typeName)
379         WithClose(statement.executeQuery()) { rs =>
380           Iterator.continually(rs).takeWhile(_.next()).map(_.getString(1)).toList
381         }
382       }
383     } catch {
384       case NonFatal(e) =>
385         logger.warn(s"Error loading attributes with indices for schema $typeName:", e)
386         Seq.empty
387     }
388   }
