126   override def postCreateTable(schemaName: String, sft: SimpleFeatureType, cx: Connection): Unit = {
127     // Throw an error if the sft name is longer than 31 characters
128     if (sft.getTypeName.length() > 31) {
129       val errorMsg = "Can't create schema: type name exceeds max supported length of 31 characters"
130       throw new IllegalArgumentException(errorMsg)
131     }
132 
133     // note: we skip the call to `super`, which creates a spatial index (that we don't want), and which
134     // alters the geometry column types (which we handle in the create statement)
135     val info = TypeInfo(schemaName, sft)
136     implicit val ex: ExecutionContext = new ExecutionContext(cx)
137     try {
138       PartitionedPostgisDialect.Commands.foreach(_.create(info))
139     } finally {
140       ex.close()

202   override def preDropTable(schemaName: String, sft: SimpleFeatureType, cx: Connection): Unit = {
203     // due to the JDBCDataStore hard-coding "DROP TABLE" we have to redirect it away from the main view,
204     // and we can't drop the write ahead table so that it has something to drop

209     try {
210       PartitionedPostgisDialect.Commands.reverse.filter(_ != WriteAheadTable).foreach(_.drop(info))
211     } finally {
212       ex.close()
213     }

252   override def encodePostColumnCreateTable(att: AttributeDescriptor, sql: StringBuffer): Unit = {
253     import PartitionedPostgisDialect.GeometryAttributeConversions
254     att match {
255       case gd: GeometryDescriptor =>
256         val nullable = gd.getMinOccurs <= 0 || gd.isNillable
257         val i = sql.lastIndexOf("geometry")
258         // expect `geometry NOT NULL` or `geometry` depending on nullable flag
259         if (i == -1 || (nullable && i != sql.length() - 8) || (!nullable && i != sql.length() - 17)) {
260           logger.warn(s"Found geometry-type attribute but no geometry column binding: $sql")
261         } else {
262           val srid = gd.getSrid.getOrElse(-1)
263           val geomType = PartitionedPostgisDialect.GeometryMappings.getOrElse(gd.getType.getBinding, "GEOMETRY")
264           val geomTypeWithDims = gd.getCoordinateDimensions match {
265             case None | Some(2) => geomType
266             case Some(3) => s"${geomType}Z"
267             case Some(4) => s"${geomType}ZM"
268             case Some(d) =>
269               throw new IllegalArgumentException(
270                 s"PostGIS only supports geometries with 2, 3 and 4 dimensions, but found: $d")
271           }
272           sql.insert(i + 8, s" ($geomTypeWithDims, $srid)")
273         }
274 
275       case _ if att.isJson() =>
276         // replace 'VARCHAR' with jsonb
277         val i = sql.lastIndexOf(" VARCHAR")
278         if (i == sql.length() - 8) {
279           sql.replace(i + 1, i + 8, "JSONB")
280         } else {
281           logger.warn(s"Found JSON-type attribute but no CHARACTER VARYING column binding: $sql")
282         }
283 
284       case _ if att.isList =>
285         // go back and encode the array type in the CQL create statement
286         val i = sql.lastIndexOf(" ARRAY")
287         if (i == sql.length() - 6) {
288           sql.insert(i, " " + getListTypeMapping(att.getListType()))
289         } else {
290           logger.warn(s"Found list-type attribute but no ARRAY column binding: $sql")
291         }
292 
293       case _ => // no-op
294     }
295   }

306   /**
307    * Gets the array type for a list/array column
308    *
309    * @param binding list-type binding
310    * @return
311    */
312   private def getListTypeMapping(binding: Class[_]): String = {
313     val mappings = new java.util.HashMap[String, Class[_]]()
314     registerSqlTypeNameToClassMappings(mappings)
315     var mapping: String = null
316     var partial: String = null
317     val iter = mappings.asScala.iterator
318     while (iter.hasNext && mapping == null) {
319       val (name, clas) = iter.next
320       if (clas == binding) {
321         mapping = name
322       } else if (partial == null && clas.isAssignableFrom(binding)) {
323         partial = name
324       }
325     }
326     if (mapping != null) {
327       mapping
328     } else if (partial != null) {
329       partial
330     } else {
331       logger.warn(s"Could not find list-type column for type ${binding.getName}")
332       "text"
333     }
334   }

338    *
339    * @param cx connection
340    * @param typeName feature type name
341    * @return
342    */
343   private def getIndexedColumns(cx: Connection, typeName: String): Seq[String] = {
344     val attributesWithIndicesSql =
345       s"""select distinct(att.attname) as indexed_attribute_name
346          |from pg_class obj
347          |join pg_index idx on idx.indrelid = obj.oid
348          |join pg_attribute att on att.attrelid = obj.oid and att.attnum = any(idx.indkey)
349          |join pg_views v on v.viewname = ?
350          |where obj.relname = concat(?, ${PartitionedTableSuffix.quoted})
351          |order by att.attname;""".stripMargin
352     try {
353       WithClose(cx.prepareStatement(attributesWithIndicesSql)) { statement =>
354         statement.setString(1, typeName)
355         statement.setString(2, typeName)
356         WithClose(statement.executeQuery()) { rs =>
357           Iterator.continually(rs).takeWhile(_.next()).map(_.getString(1)).toList
358         }
359       }
360     } catch {
361       case NonFatal(e) =>
362         logger.warn(s"Error loading attributes with indices for schema $typeName:", e)
363         Seq.empty
364     }
365   }
