132   override def postCreateTable(schemaName: String, sft: SimpleFeatureType, cx: Connection): Unit = {
133     // Throw an error if the sft name is longer than 31 characters
134     if (sft.getTypeName.length() > 31) {
135       val errorMsg = "Can't create schema: type name exceeds max supported length of 31 characters"
136       throw new IllegalArgumentException(errorMsg)
137     }
138 
139     // note: we skip the call to `super`, which creates a spatial index (that we don't want), and which
140     // alters the geometry column types (which we handle in the create statement)
141     val info = TypeInfo(schemaName, sft)
142     implicit val ex: ExecutionContext = new ExecutionContext(cx)
143     try {
144       PartitionedPostgisDialect.Commands.foreach(_.create(info))
145       if (grants.nonEmpty) {
146         val roles = grants.map(_.quoted).mkString(", ")
147         val tables =
148           Seq(
149             info.tables.view.name,
150             info.tables.writeAhead.name,
151             info.tables.writeAheadPartitions.name,
152             info.tables.mainPartitions.name,
153             info.tables.spillPartitions.name,
154             TableIdentifier(schemaName, PrimaryKeyTable.Name.raw),
155             TableIdentifier(schemaName, UserDataTable.Name.raw)
156           )
157         tables.foreach { table =>
158           ex.execute(s"GRANT SELECT ON ${table.qualified} TO $roles;")
159         }
160       }
161     } finally {
162       ex.close()
163     }
164   }

237   override def preDropTable(schemaName: String, sft: SimpleFeatureType, cx: Connection): Unit = {
238     // due to the JDBCDataStore hard-coding "DROP TABLE" we have to redirect it away from the main view,
239     // and we can't drop the write ahead table so that it has something to drop
240     dropping.set(true)
241     val info = TypeInfo(schemaName, sft)
242 
243     implicit val ex: ExecutionContext = new ExecutionContext(cx)
244     try {
245       PartitionedPostgisDialect.Commands.reverse.filter(_ != WriteAheadTable).foreach(_.drop(info))
246       PartitionTablespacesTable.drop(info)
247     } finally {
248       ex.close()
249     }
250   }

288   override def encodePostColumnCreateTable(att: AttributeDescriptor, sql: StringBuffer): Unit = {
289     import PartitionedPostgisDialect.GeometryAttributeConversions
290     att match {
291       case gd: GeometryDescriptor =>
292         val nullable = gd.getMinOccurs <= 0 || gd.isNillable
293         val i = sql.lastIndexOf("geometry")
294         // expect `geometry NOT NULL` or `geometry` depending on nullable flag
295         if (i == -1 || (nullable && i != sql.length() - 8) || (!nullable && i != sql.length() - 17)) {
296           logger.warn(s"Found geometry-type attribute but no geometry column binding: $sql")
297         } else {
298           val srid = gd.getSrid.getOrElse(-1)
299           val geomType = PartitionedPostgisDialect.GeometryMappings.getOrElse(gd.getType.getBinding, "GEOMETRY")
300           val geomTypeWithDims = gd.getCoordinateDimensions match {
301             case None | Some(2) => geomType
302             case Some(3) => s"${geomType}Z"
303             case Some(4) => s"${geomType}ZM"
304             case Some(d) =>
305               throw new IllegalArgumentException(
306                 s"PostGIS only supports geometries with 2, 3 and 4 dimensions, but found: $d")
307           }
308           sql.insert(i + 8, s" ($geomTypeWithDims, $srid)")
309         }
310 
311       case _ if att.isJson() =>
312         // replace 'VARCHAR' with jsonb
313         val i = sql.lastIndexOf(" VARCHAR")
314         if (i == sql.length() - 8) {
315           sql.replace(i + 1, i + 8, "JSONB")
316         } else {
317           logger.warn(s"Found JSON-type attribute but no CHARACTER VARYING column binding: $sql")
318         }
319 
320       case _ if att.isList =>
321         // go back and encode the array type in the CQL create statement
322         val i = sql.lastIndexOf(" ARRAY")
323         if (i == sql.length() - 6) {
324           sql.insert(i, " " + getListTypeMapping(att.getListType()))
325         } else {
326           logger.warn(s"Found list-type attribute but no ARRAY column binding: $sql")
327         }
328 
329       case _ => // no-op
330     }
331   }

342   /**
343    * Gets the array type for a list/array column
344    *
345    * @param binding list-type binding
346    * @return
347    */
348   private def getListTypeMapping(binding: Class[_]): String = {
349     val mappings = new java.util.HashMap[String, Class[_]]()
350     registerSqlTypeNameToClassMappings(mappings)
351     var mapping: String = null
352     var partial: String = null
353     val iter = mappings.asScala.iterator
354     while (iter.hasNext && mapping == null) {
355       val (name, clas) = iter.next
356       if (clas == binding) {
357         mapping = name
358       } else if (partial == null && clas.isAssignableFrom(binding)) {
359         partial = name
360       }
361     }
362     if (mapping != null) {
363       mapping
364     } else if (partial != null) {
365       partial
366     } else {
367       logger.warn(s"Could not find list-type column for type ${binding.getName}")
368       "text"
369     }
370   }

482   def getIndexedColumns(cx: Connection, typeName: String): Try[List[String]] = {
483     val attributesWithIndicesSql =
484       s"""select distinct(att.attname) as indexed_attribute_name
485          |from pg_class obj
486          |join pg_index idx on idx.indrelid = obj.oid
487          |join pg_attribute att on att.attrelid = obj.oid and att.attnum = any(idx.indkey)
488          |where obj.relname = concat(?, ${PartitionedTableSuffix.quoted})
489          |order by att.attname;""".stripMargin
490     Try {
491       WithClose(cx.prepareStatement(attributesWithIndicesSql)) { statement =>
492         statement.setString(1, typeName)
493         WithClose(statement.executeQuery()) { rs =>
494           Iterator.continually(rs).takeWhile(_.next()).map(_.getString(1)).toList
495         }
496       }
