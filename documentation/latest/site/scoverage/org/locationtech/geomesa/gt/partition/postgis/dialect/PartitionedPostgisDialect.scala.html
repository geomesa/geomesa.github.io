133   override def postCreateTable(schemaName: String, sft: SimpleFeatureType, cx: Connection): Unit = {
134     // Throw an error if the sft name is longer than 31 characters
135     if (sft.getTypeName.length() > 31) {
136       val errorMsg = "Can't create schema: type name exceeds max supported length of 31 characters"
137       throw new IllegalArgumentException(errorMsg)
138     }
139 
140     // note: we skip the call to `super`, which creates a spatial index (that we don't want), and which
141     // alters the geometry column types (which we handle in the create statement)
142     val info = TypeInfo(schemaName, sft)
143     implicit val ex: ExecutionContext = new ExecutionContext(cx)
144     try {
145       PartitionedPostgisDialect.Commands.foreach(_.create(info))
146       if (grants.nonEmpty) {
147         val roles = grants.map(_.quoted).mkString(", ")
148         val tables =
149           Seq(
150             info.tables.view.name,
151             info.tables.writeAhead.name,
152             info.tables.writeAheadPartitions.name,
153             info.tables.mainPartitions.name,
154             info.tables.spillPartitions.name,
155             TableIdentifier(schemaName, PrimaryKeyTable.Name.raw),
156             TableIdentifier(schemaName, UserDataTable.Name.raw)
157           )
158         tables.foreach { table =>
159           ex.execute(s"GRANT SELECT ON ${table.qualified} TO $roles;")
160         }
161       }
162     } finally {
163       ex.close()
164     }
165   }

235   override def preDropTable(schemaName: String, sft: SimpleFeatureType, cx: Connection): Unit = {
236     // due to the JDBCDataStore hard-coding "DROP TABLE" we have to redirect it away from the main view,
237     // and we can't drop the write ahead table so that it has something to drop
238     dropping.set(true)
239     val info = TypeInfo(schemaName, sft)
240 
241     implicit val ex: ExecutionContext = new ExecutionContext(cx)
242     try {
243       PartitionedPostgisDialect.Commands.reverse.filter(_ != WriteAheadTable).foreach(_.drop(info))
244     } finally {
245       ex.close()
246     }
247   }

285   override def encodePostColumnCreateTable(att: AttributeDescriptor, sql: StringBuffer): Unit = {
286     import PartitionedPostgisDialect.GeometryAttributeConversions
287     att match {
288       case gd: GeometryDescriptor =>
289         val nullable = gd.getMinOccurs <= 0 || gd.isNillable
290         val i = sql.lastIndexOf("geometry")
291         // expect `geometry NOT NULL` or `geometry` depending on nullable flag
292         if (i == -1 || (nullable && i != sql.length() - 8) || (!nullable && i != sql.length() - 17)) {
293           logger.warn(s"Found geometry-type attribute but no geometry column binding: $sql")
294         } else {
295           val srid = gd.getSrid.getOrElse(-1)
296           val geomType = PartitionedPostgisDialect.GeometryMappings.getOrElse(gd.getType.getBinding, "GEOMETRY")
297           val geomTypeWithDims = gd.getCoordinateDimensions match {
298             case None | Some(2) => geomType
299             case Some(3) => s"${geomType}Z"
300             case Some(4) => s"${geomType}ZM"
301             case Some(d) =>
302               throw new IllegalArgumentException(
303                 s"PostGIS only supports geometries with 2, 3 and 4 dimensions, but found: $d")
304           }
305           sql.insert(i + 8, s" ($geomTypeWithDims, $srid)")
306         }
307 
308       case _ if att.isJson() =>
309         // replace 'VARCHAR' with jsonb
310         val i = sql.lastIndexOf(" VARCHAR")
311         if (i == sql.length() - 8) {
312           sql.replace(i + 1, i + 8, "JSONB")
313         } else {
314           logger.warn(s"Found JSON-type attribute but no CHARACTER VARYING column binding: $sql")
315         }
316 
317       case _ if att.isList =>
318         // go back and encode the array type in the CQL create statement
319         val i = sql.lastIndexOf(" ARRAY")
320         if (i == sql.length() - 6) {
321           sql.insert(i, " " + getListTypeMapping(att.getListType()))
322         } else {
323           logger.warn(s"Found list-type attribute but no ARRAY column binding: $sql")
324         }
325 
326       case _ => // no-op
327     }
328   }

339   /**
340    * Gets the array type for a list/array column
341    *
342    * @param binding list-type binding
343    * @return
344    */
345   private def getListTypeMapping(binding: Class[_]): String = {
346     val mappings = new java.util.HashMap[String, Class[_]]()
347     registerSqlTypeNameToClassMappings(mappings)
348     var mapping: String = null
349     var partial: String = null
350     val iter = mappings.asScala.iterator
351     while (iter.hasNext && mapping == null) {
352       val (name, clas) = iter.next
353       if (clas == binding) {
354         mapping = name
355       } else if (partial == null && clas.isAssignableFrom(binding)) {
356         partial = name
357       }
358     }
359     if (mapping != null) {
360       mapping
361     } else if (partial != null) {
362       partial
363     } else {
364       logger.warn(s"Could not find list-type column for type ${binding.getName}")
365       "text"
366     }
367   }

480   def getIndexedColumns(cx: Connection, typeName: String): Try[List[String]] = {
481     val attributesWithIndicesSql =
482       s"""select distinct(att.attname) as indexed_attribute_name
483          |from pg_class obj
484          |join pg_index idx on idx.indrelid = obj.oid
485          |join pg_attribute att on att.attrelid = obj.oid and att.attnum = any(idx.indkey)
486          |where obj.relname = concat(?, ${PartitionedTableSuffix.quoted})
487          |order by att.attname;""".stripMargin
488     Try {
489       WithClose(cx.prepareStatement(attributesWithIndicesSql)) { statement =>
490         statement.setString(1, typeName)
491         WithClose(statement.executeQuery()) { rs =>
492           Iterator.continually(rs).takeWhile(_.next()).map(_.getString(1)).toList
493         }
494       }
