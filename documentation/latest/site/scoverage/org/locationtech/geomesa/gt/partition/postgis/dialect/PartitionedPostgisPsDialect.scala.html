29   // cache for tracking json-type columns
30   private val jsonColumns: LoadingCache[PreparedStatementKey, java.lang.Boolean] =
31     Caffeine.newBuilder()
32         .expireAfterAccess(1, TimeUnit.MINUTES)
33         .build(new CacheLoader[PreparedStatementKey, java.lang.Boolean]() {
34           override def load(key: PreparedStatementKey): java.lang.Boolean = {
35             key.ps.getParameterMetaData.getParameterTypeName(key.column).toLowerCase(Locale.US) match {
36               case "jsonb" | "json" => true
37               case _ => false
38             }
39           }
40         })
41 
42   override def setValue(value: Any, binding: Class[_], ps: PreparedStatement, column: Int, cx: Connection): Unit = {
43     // json columns are string type in geotools, but we have to use setObject or else we get a binding error
44     if (binding == classOf[String] && jsonColumns.get(new PreparedStatementKey(ps, column))) {
45       ps.setObject(column, value, Types.OTHER)
46     } else if (binding == classOf[java.util.List[_]]) {
47       // handle bug in jdbc store not calling setArrayValue in update statements
48       value match {
49         case null =>
50           ps.setNull(column, Types.ARRAY)
51 
52         case list: java.util.Collection[_] =>
53           if (list.isEmpty) {
54             ps.setNull(column, Types.ARRAY)
55           } else {
56             setArray(list.toArray(), ps, column, cx)
57           }
58 
59         case array: Array[_] =>
60           if (array.isEmpty) {
61             ps.setNull(column, Types.ARRAY)
62           } else {
63             setArray(array, ps, column, cx)
64           }
65 
66         case _ =>
67           // this will almost certainly fail...
68           super.setValue(value, binding, ps, column, cx)
69       }
70     } else {
71       super.setValue(value, binding, ps, column, cx)
72     }
73   }
74 
75   // based on setArrayValue, but we don't have the attribute descriptor to use
76   private def setArray(array: Array[_], ps: PreparedStatement, column: Int, cx: Connection): Unit = {
77     val componentType = array(0).getClass
78     val sqlType = dataStore.getSqlTypeNameToClassMappings.asScala.collectFirst { case (k, v) if v == componentType => k }
79     val componentTypeName = sqlType.getOrElse {
80       throw new java.sql.SQLException(s"Failed to find a SQL type for $componentType")
81     }
82     ps.setArray(column, super.convertToArray(array, componentTypeName, componentType, cx))
83   }
84 
85   override protected def convertToArray(
86       value: Any, componentTypeName: String, componentType: Class[_], connection: Connection): java.sql.Array = {
87     val array = value match {
88       case list: java.util.List[_] => list.toArray()
89       case _ => value
90     }
91     super.convertToArray(array, componentTypeName, componentType, connection)
92   }
