39   override def setValue(
40       value: AnyRef,
41       binding: Class[_],
42       att: AttributeDescriptor,
43       ps: PreparedStatement,
44       column: Int,
45       cx: Connection): Unit = {
46     // json columns are string type in geotools, but we have to use setObject or else we get a binding error
47     if (binding == classOf[String] && att != null && att.isJson) {
48       ps.setObject(column, value, Types.OTHER)
49     } else if (binding.isArray || binding == classOf[java.util.List[_]]) {
50       // handle bug in jdbc store not calling setArrayValue in update statements
51       value match {
52         case null =>
53           ps.setNull(column, Types.ARRAY)
54 
55         case list: java.util.Collection[_] =>
56           if (list.isEmpty) {
57             ps.setNull(column, Types.ARRAY)
58           } else {
59             setArrayValue(list.toArray(), att, ps, column, cx)
60           }
61 
62         case array: Array[_] =>
63           if (array.isEmpty) {
64             ps.setNull(column, Types.ARRAY)
65           } else {
66             setArrayValue(array, att, ps, column, cx)
67           }
68 
69         case singleton =>
70           setArrayValue(Array(singleton), att, ps, column, cx)
71       }
72     } else {
73       super.setValue(value, binding, att, ps, column, cx)
74     }
75   }

117     override def setFeatureType(featureType: SimpleFeatureType): Unit = {
118       // convert List-type attributes to Array-types so that prepared statement bindings work correctly
119       if (featureType.getAttributeDescriptors.asScala.exists(_.getType.getBinding == classOf[java.util.List[_]])) {
120         val builder = new SimpleFeatureTypeBuilder() {
121           override def init(`type`: SimpleFeatureType): Unit = {
122             super.init(`type`)
123             attributes().clear()
124           }
125         }
126         builder.init(featureType)
127         featureType.getAttributeDescriptors.asScala.foreach { descriptor =>
128           val ab = new AttributeTypeBuilder(builder.getFeatureTypeFactory)
129           ab.init(descriptor)
130           if (descriptor.getType.getBinding == classOf[java.util.List[_]]) {
131             ab.setBinding(java.lang.reflect.Array.newInstance(Option(descriptor.getListType()).getOrElse(classOf[String]), 0).getClass)
132           }
133           builder.add(ab.buildDescriptor(descriptor.getLocalName))
134         }
135         this.featureType = builder.buildFeatureType()
136         this.featureType.getUserData.putAll(featureType.getUserData)
137       } else {
138         this.featureType = featureType
139       }
140     }
141 
142     // note: this would be a cleaner solution, but it doesn't get invoked due to explicit calls to
143     // super.getExpressionType in PostgisPSFilterToSql :/
144     override def getExpressionType(expression: Expression): Class[_] = {
145       val result = Option(expression).collect { case p: PropertyName => p }.flatMap { p =>
146         Option(p.evaluate(featureType).asInstanceOf[AttributeDescriptor]).map { descriptor =>
147           val binding = descriptor.getType.getBinding
148           if (binding == classOf[java.util.List[_]]) {
149             val listType = descriptor.getListType()
150             if (listType == null) {
151               classOf[Array[String]]
152             } else {
153               java.lang.reflect.Array.newInstance(listType, 0).getClass
154             }
155           } else {
156             binding
157           }
158         }
159       }
160 
161       result.getOrElse(super.getExpressionType(expression))
162     }
163 
164     override def visit(filter: Or, extraData: AnyRef): AnyRef = {
165       // for array-types, skip the super-class implementation, which merges ORs into INs, as it breaks array OR queries
166       // for other types, keep the super handling as INs may be more efficient that ORs
167       if (FilterHelper.propertyNames(filter).flatMap(name => Option(featureType.getDescriptor(name))).exists(_.getType.getBinding.isArray)) {
168         visit(filter.asInstanceOf[BinaryLogicOperator], "OR")
169       } else {
170         super.visit(filter, extraData)
171       }
172     }
