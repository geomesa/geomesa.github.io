40   override def setValue(
41       value: AnyRef,
42       binding: Class[_],
43       att: AttributeDescriptor,
44       ps: PreparedStatement,
45       column: Int,
46       cx: Connection): Unit = {
47     // json columns are string type in geotools, but we have to use setObject or else we get a binding error
48     if (binding == classOf[String] && att != null && att.isJson) {
49       ps.setObject(column, value, Types.OTHER)
50     } else if (binding.isArray || binding == classOf[java.util.List[_]]) {
51       // handle bug in jdbc store not calling setArrayValue in update statements
52       value match {
53         case null =>
54           ps.setNull(column, Types.ARRAY)
55 
56         case list: java.util.Collection[_] =>
57           if (list.isEmpty) {
58             ps.setNull(column, Types.ARRAY)
59           } else {
60             setArrayValue(list.toArray(), att, ps, column, cx)
61           }
62 
63         case array: Array[_] =>
64           if (array.isEmpty) {
65             ps.setNull(column, Types.ARRAY)
66           } else {
67             setArrayValue(array, att, ps, column, cx)
68           }
69 
70         case singleton =>
71           setArrayValue(Array(singleton), att, ps, column, cx)
72       }
73     } else {
74       super.setValue(value, binding, att, ps, column, cx)
75     }
76   }

120     override def setFeatureType(featureType: SimpleFeatureType): Unit = {
121       // convert List-type attributes to Array-types so that prepared statement bindings work correctly
122       if (featureType.getAttributeDescriptors.asScala.exists(_.getType.getBinding == classOf[java.util.List[_]])) {
123         val builder = new SimpleFeatureTypeBuilder() {
124           override def init(`type`: SimpleFeatureType): Unit = {
125             super.init(`type`)
126             attributes().clear()
127           }
128         }
129         builder.init(featureType)
130         featureType.getAttributeDescriptors.asScala.foreach { descriptor =>
131           val ab = new AttributeTypeBuilder(builder.getFeatureTypeFactory)
132           ab.init(descriptor)
133           if (descriptor.getType.getBinding == classOf[java.util.List[_]]) {
134             ab.setBinding(java.lang.reflect.Array.newInstance(Option(descriptor.getListType()).getOrElse(classOf[String]), 0).getClass)
135           }
136           builder.add(ab.buildDescriptor(descriptor.getLocalName))
137         }
138         this.featureType = builder.buildFeatureType()
139         this.featureType.getUserData.putAll(featureType.getUserData)
140       } else {
141         this.featureType = featureType
142       }
143     }
144 
145     // note: this would be a cleaner solution, but it doesn't get invoked due to explicit calls to
146     // super.getExpressionType in PostgisPSFilterToSql :/
147     override def getExpressionType(expression: Expression): Class[_] = {
148       val result = Option(expression).collect { case p: PropertyName => p }.flatMap { p =>
149         Option(p.evaluate(featureType).asInstanceOf[AttributeDescriptor]).map { descriptor =>
150           val binding = descriptor.getType.getBinding
151           if (binding == classOf[java.util.List[_]]) {
152             val listType = descriptor.getListType()
153             if (listType == null) {
154               classOf[Array[String]]
155             } else {
156               java.lang.reflect.Array.newInstance(listType, 0).getClass
157             }
158           } else {
159             binding
160           }
161         }
162       }
163 
164       result.getOrElse(super.getExpressionType(expression))
165     }
166 
167     override def visit(filter: Or, extraData: AnyRef): AnyRef = {
168       // for array-types, skip the super-class implementation, which merges ORs into INs, as it breaks array OR queries
169       // for other types, keep the super handling as INs may be more efficient that ORs
170       if (FilterHelper.propertyNames(filter).flatMap(name => Option(featureType.getDescriptor(name))).exists(_.getType.getBinding.isArray)) {
171         visit(filter.asInstanceOf[BinaryLogicOperator], "OR")
172       } else {
173         super.visit(filter, extraData)
174       }
175     }
