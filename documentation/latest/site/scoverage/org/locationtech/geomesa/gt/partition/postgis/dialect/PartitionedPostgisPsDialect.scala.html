64   override def setValue(
65       value: AnyRef,
66       binding: Class[_],
67       att: AttributeDescriptor,
68       ps: PreparedStatement,
69       column: Int,
70       cx: Connection): Unit = {
71     // json columns are string type in geotools, but we have to use setObject or else we get a binding error
72     if (binding == classOf[String] && jsonColumns.get(new PreparedStatementKey(ps, column))) {
73       ps.setObject(column, value, Types.OTHER)
74     } else if (binding.isArray || binding == classOf[java.util.List[_]]) {
75       // handle bug in jdbc store not calling setArrayValue in update statements
76       value match {
77         case null =>
78           ps.setNull(column, Types.ARRAY)
79 
80         case list: java.util.Collection[_] =>
81           if (list.isEmpty) {
82             ps.setNull(column, Types.ARRAY)
83           } else {
84             setArray(list.toArray(), ps, column, cx)
85           }
86 
87         case array: Array[_] =>
88           if (array.isEmpty) {
89             ps.setNull(column, Types.ARRAY)
90           } else {
91             setArray(array, ps, column, cx)
92           }
93 
94         case singleton =>
95           setArray(Array(singleton), ps, column, cx)
96       }
97     } else {
98       super.setValue(value, binding, att, ps, column, cx)
99     }
100   }

186     override def setFeatureType(featureType: SimpleFeatureType): Unit = {
187       // convert List-type attributes to Array-types so that prepared statement bindings work correctly
188       if (featureType.getAttributeDescriptors.asScala.exists(_.getType.getBinding == classOf[java.util.List[_]])) {
189         val builder = new SimpleFeatureTypeBuilder() {
190           override def init(`type`: SimpleFeatureType): Unit = {
191             super.init(`type`)
192             attributes().clear()
193           }
194         }
195         builder.init(featureType)
196         featureType.getAttributeDescriptors.asScala.foreach { descriptor =>
197           val ab = new AttributeTypeBuilder(builder.getFeatureTypeFactory)
198           ab.init(descriptor)
199           if (descriptor.getType.getBinding == classOf[java.util.List[_]]) {
200             ab.setBinding(java.lang.reflect.Array.newInstance(Option(descriptor.getListType()).getOrElse(classOf[String]), 0).getClass)
201           }
202           builder.add(ab.buildDescriptor(descriptor.getLocalName))
203         }
204         this.featureType = builder.buildFeatureType()
205         this.featureType.getUserData.putAll(featureType.getUserData)
206       } else {
207         this.featureType = featureType
208       }
209     }

211     // note: this would be a cleaner solution, but it doesn't get invoked due to explicit calls to
212     // super.getExpressionType in PostgisPSFilterToSql :/
213     override def getExpressionType(expression: Expression): Class[_] = {
214       val result = Option(expression).collect { case p: PropertyName => p }.flatMap { p =>
215         Option(p.evaluate(featureType).asInstanceOf[AttributeDescriptor]).map { descriptor =>
216           val binding = descriptor.getType.getBinding
217           if (binding == classOf[java.util.List[_]]) {
218             val listType = descriptor.getListType()
219             if (listType == null) {
220               classOf[Array[String]]
221             } else {
222               java.lang.reflect.Array.newInstance(listType, 0).getClass
223             }
224           } else {
225             binding
226           }
227         }
228       }
229 
230       result.getOrElse(super.getExpressionType(expression))
231     }
