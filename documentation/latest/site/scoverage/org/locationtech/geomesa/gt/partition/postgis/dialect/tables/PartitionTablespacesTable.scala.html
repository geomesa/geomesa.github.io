44   /**
45    * Read partition table spaces for the specified type
46    *
47    * @param cx connection
48    * @param metadata metadata
49    * @param schema database schema name
50    * @param typeName feature type name
51    * @return
52    */
53   def read(cx: Connection, metadata: DatabaseMetaData, schema: String, typeName: String): Map[String, String] = {
54     val map = Map.newBuilder[String, String]
55     if (exists(schema, metadata)) {
56       val tablespaceSql =
57         s"select table_space, table_type from " +
58           s"${escape(schema)}.${PartitionTablespacesTable.Name.quoted} where type_name = ?"
59       WithClose(cx.prepareStatement(tablespaceSql)) { statement =>
60         statement.setString(1, typeName)
61         WithClose(statement.executeQuery()) { rs =>
62           while (rs.next()) {
63             val ts = rs.getString(1)
64             if (ts != null && ts.nonEmpty) {
65               rs.getString(2) match {
66                 case WriteAheadTableSuffix.raw => map += SftUserData.WriteAheadTableSpace.key -> ts
67                 case PartitionedWriteAheadTableSuffix.raw => map += SftUserData.WriteAheadPartitionsTableSpace.key -> ts
68                 case PartitionedTableSuffix.raw => map += SftUserData.MainTableSpace.key -> ts
69                 case s => logger.warn(s"Ignoring unexpected tablespace table: $s")
70               }
71             }
72           }
73         }
74       }
75     }
76     map.result()
77   }
78 
79   private def exists(info: TypeInfo)(implicit ex: ExecutionContext): Boolean =
80     exists(info.schema.raw, ex.cx.getMetaData)
81 
82   private def exists(schema: String, metadata: DatabaseMetaData): Boolean =
83     WithClose(metadata.getTables(null, schema, Name.raw, null))(_.next())
