42   /**
43    * Read partition table spaces for the specified type
44    *
45    * @param cx connection
46    * @param metadata metadata
47    * @param schema database schema name
48    * @param typeName feature type name
49    * @return
50    */
51   def read(cx: Connection, metadata: DatabaseMetaData, schema: String, typeName: String): Map[String, String] = {
52     val map = Map.newBuilder[String, String]
53     if (exists(schema, metadata)) {
54       val tablespaceSql =
55         s"select table_space, table_type from " +
56           s"${escape(schema)}.${PartitionTablespacesTable.Name.quoted} where type_name = ?"
57       WithClose(cx.prepareStatement(tablespaceSql)) { statement =>
58         statement.setString(1, typeName)
59         WithClose(statement.executeQuery()) { rs =>
60           while (rs.next()) {
61             val ts = rs.getString(1)
62             if (ts != null && ts.nonEmpty) {
63               rs.getString(2) match {
64                 case WriteAheadTableSuffix.raw => map += SftUserData.WriteAheadTableSpace.key -> ts
65                 case PartitionedWriteAheadTableSuffix.raw => map += SftUserData.WriteAheadPartitionsTableSpace.key -> ts
66                 case PartitionedTableSuffix.raw => map += SftUserData.MainTableSpace.key -> ts
67                 case s => logger.warn(s"Ignoring unexpected tablespace table: $s")
68               }
69             }
70           }
71         }
72       }
73     }
74     map.result()
75   }
76 
77   private def exists(info: TypeInfo)(implicit ex: ExecutionContext): Boolean =
78     exists(info.schema.raw, ex.cx.getMetaData)
79 
80   private def exists(schema: String, metadata: DatabaseMetaData): Boolean =
81     WithClose(metadata.getTables(null, schema, Name.raw, null))(_.next())
