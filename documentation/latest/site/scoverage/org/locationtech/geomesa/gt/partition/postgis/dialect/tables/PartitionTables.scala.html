30   private def statements(info: TypeInfo, table: TableConfig, indexType: String): Seq[String] = {
31     // note: don't include storage opts since these are parent partition tables
32     val (tableTs, indexTs) = table.tablespace match {
33       case None => ("", "")
34       case Some(ts) => (s" TABLESPACE ${ts.quoted}", s" USING INDEX TABLESPACE ${ts.quoted}")
35     }
36 
37     val logging = if (table.logged) { "" } else { "UNLOGGED" }
38     val create =
39       s"""CREATE $logging TABLE IF NOT EXISTS ${table.name.qualified} (
40          |  LIKE ${info.tables.writeAhead.name.qualified} INCLUDING DEFAULTS INCLUDING CONSTRAINTS,
41          |  CONSTRAINT ${escape(table.name.raw, "pkey")} PRIMARY KEY (fid, ${info.cols.dtg.quoted})$indexTs
42          |) PARTITION BY RANGE(${info.cols.dtg.quoted})$tableTs;""".stripMargin
43     val pagesPerRange =
44       if (indexType == "brin") { s" with (pages_per_range = ${info.partitions.pagesPerRange})" } else { "" }
45     // note: brin doesn't support 'include' cols
46     val geomIndex =
47       s"""CREATE INDEX IF NOT EXISTS ${escape(table.name.raw, info.cols.geom.raw)}
48          |  ON ${table.name.qualified}
49          |  USING $indexType(${info.cols.geom.quoted})$pagesPerRange$tableTs;""".stripMargin
50     val dtgIndex =
51       s"""CREATE INDEX IF NOT EXISTS ${escape(table.name.raw, info.cols.dtg.raw)}
52          |  ON ${table.name.qualified} (${info.cols.dtg.quoted})$tableTs;""".stripMargin
53     val indices = info.cols.indexed.map { col =>
54       s"""CREATE INDEX IF NOT EXISTS ${escape(table.name.raw, col.raw)}
55          |  ON ${table.name.qualified} (${col.quoted})$tableTs;""".stripMargin
56     }
57     Seq(create, geomIndex, dtgIndex) ++ indices
58   }
59 
60   override protected def dropStatements(info: TypeInfo): Seq[String] =
61     tablesAndTypes(info).map { case (table, _) => s"DROP TABLE IF EXISTS ${table.name.qualified};" }
