28   private def statements(info: TypeInfo, table: TableConfig, indexType: String): Seq[String] = {
29     // note: don't include storage opts since these are parent partition tables
30     val (tableTs, indexTs) = table.tablespace match {
31       case None => ("", "")
32       case Some(ts) => (s" TABLESPACE ${ts.quoted}", s" USING INDEX TABLESPACE ${ts.quoted}")
33     }
34     val create =
35       s"""CREATE ${info.walLogSQL} TABLE IF NOT EXISTS ${table.name.qualified} (
36          |  LIKE ${info.tables.writeAhead.name.qualified} INCLUDING DEFAULTS INCLUDING CONSTRAINTS,
37          |  CONSTRAINT ${escape(table.name.raw, "pkey")} PRIMARY KEY (fid, ${info.cols.dtg.quoted})$indexTs
38          |) PARTITION BY RANGE(${info.cols.dtg.quoted})$tableTs;""".stripMargin
39     val pagesPerRange =
40       if (indexType == "brin") { s" with (pages_per_range = ${info.partitions.pagesPerRange})" } else { "" }
41     // note: brin doesn't support 'include' cols
42     val geomIndex =
43       s"""CREATE INDEX IF NOT EXISTS ${escape(table.name.raw, info.cols.geom.raw)}
44          |  ON ${table.name.qualified}
45          |  USING $indexType(${info.cols.geom.quoted})$pagesPerRange$tableTs;""".stripMargin
46     val dtgIndex =
47       s"""CREATE INDEX IF NOT EXISTS ${escape(table.name.raw, info.cols.dtg.raw)}
48          |  ON ${table.name.qualified} (${info.cols.dtg.quoted})$tableTs;""".stripMargin
49     val indices = info.cols.indexed.map { col =>
50       s"""CREATE INDEX IF NOT EXISTS ${escape(table.name.raw, col.raw)}
51          |  ON ${table.name.qualified} (${col.quoted})$tableTs;""".stripMargin
52     }
53     Seq(create, geomIndex, dtgIndex) ++ indices
54   }
55 
56   override protected def dropStatements(info: TypeInfo): Seq[String] =
57     tablesAndTypes(info).map { case (table, _) => s"DROP TABLE IF EXISTS ${table.name.qualified};" }
