27 /**
28   * Persists expired entries to the data store
29   *   1. checks for expired entries
30   *   2. gets zk lock
31   *   3. gets offsets from zk
32   *   4. writes expired entries to data store
33   *   5. updates offsets in zk
34   *   6. releases zk lock
35   *
36   * @param ds data store to write to
37   * @param sft simple feature type
38   * @param offsetManager offset manager
39   * @param cache shared state
40   * @param topic kafka topic
41   * @param ageOffMillis age off for expiring features
42   * @param clock clock used for checking expiration
43   */
44 class DataStorePersistence(ds: DataStore,
45                            sft: SimpleFeatureType,
46                            offsetManager: OffsetManager,
47                            cache: ExpiringFeatureCache,
48                            topic: String,
49                            ageOffMillis: Long,
50                            persistExpired: Boolean)
51                           (implicit clock: Clock = Clock.systemUTC())
52     extends Runnable with Closeable with MethodProfiling with LazyLogging {
53 
54   private val frequency = SystemProperty("geomesa.lambda.persist.interval").toDuration.map(_.toMillis).getOrElse(60000L)
55   private val lockTimeout = SystemProperty("geomesa.lambda.persist.lock.timeout").toDuration.map(_.toMillis).getOrElse(1000L)
56 
57   private val executor = Executors.newSingleThreadScheduledExecutor()
58   private val schedule = executor.scheduleWithFixedDelay(this, frequency, frequency, TimeUnit.MILLISECONDS)
59 
60   override def run(): Unit = {
61     val expired = cache.expired(clock.millis() - ageOffMillis)
62     logger.trace(s"Found partition(s) with expired entries in [$topic]: " +
63         (if (expired.isEmpty) { "none" } else { expired.mkString(",") }))
64     // lock per-partition to allow for multiple write threads
65     // randomly access the partitions to avoid contention if multiple data stores are all on the same schedule
66     Random.shuffle(expired).foreach { partition =>
67       // if we don't get the lock just try again next run
68       logger.trace(s"Acquiring lock for [$topic:$partition]")
69       offsetManager.acquireLock(topic, partition, lockTimeout) match {
70         case None => logger.trace(s"Could not acquire lock for [$topic:$partition] within ${lockTimeout}ms")
71         case Some(lock) =>
72           try {
73             logger.trace(s"Acquired lock for [$topic:$partition]")
74             persist(partition, clock.millis() - ageOffMillis)
75           } finally {
76             lock.close()
77             logger.trace(s"Released lock for [$topic:$partition]")
78           }
79       }
80     }
81   }
82 
83   private def persist(partition: Int, expiry: Long): Unit = {
84     import org.locationtech.geomesa.filter.ff
85 
86     val ExpiredFeatures(nextOffset, expired) = cache.expired(partition, expiry)
87 
88     logger.trace(s"Found ${expired.size} expired entries for [$topic:$partition]:\n\t" +
89         expired.map(e => s"offset ${e.offset}: ${e.feature}").mkString("\n\t"))
90 
91     val lastOffset = offsetManager.getOffset(topic, partition)
92     logger.trace(s"Last persisted offsets for [$topic:$partition]: $lastOffset")
93 
94     if (expired.nonEmpty) {
95       if (!persistExpired) {
96         logger.trace(s"Persist disabled for $topic")
97       } else {
98         // check that features haven't been persisted yet
99         val toPersist = scala.collection.mutable.Map.empty[String, OffsetFeature]
100         expired.foreach { e =>
101           if (e.offset > lastOffset) {
102             toPersist += e.feature.getID -> e
103           }
104         }
105 
106         logger.trace(
107           s"Offsets to persist for [$topic:$partition]: ${toPersist.values.map(_.offset).toSeq.sorted.mkString(",")}")
108 
109         def complete(modified: Long, time: Long): Unit =
110           logger.debug(s"Wrote $modified updated feature(s) to persistent storage in ${time}ms")
111 
112         profile(complete _) {
113           // do an update query first
114           val filter = ff.id(toPersist.keys.map(ff.featureId).toSeq: _*)
115           WithClose(ds.getFeatureWriter(sft.getTypeName, filter, Transaction.AUTO_COMMIT)) { writer =>
116             var count = 0L
117             while (writer.hasNext) {
118               val next = writer.next()
119               toPersist.get(next.getID).foreach { p =>
120                 logger.trace(s"Persistent store modify [$topic:$partition:${p.offset}] ${p.feature}")
121                 FeatureUtils.copyToFeature(next, p.feature, useProvidedFid = true)
122                 try { writer.write() } catch {
123                   case NonFatal(e) => logger.error(s"Error persisting feature: ${p.feature}", e)
124                 }
125                 toPersist.remove(p.feature.getID)
126               }
127               count += 1
128             }
129             count
130           }
131         }
132 
133         // if any weren't updates, add them as inserts
134         if (toPersist.nonEmpty) {
135           def complete(appended: Long, time: Long): Unit =
136             logger.debug(s"Wrote $appended new feature(s) to persistent storage in ${time}ms")
137 
138           profile(complete _) {
139             WithClose(ds.getFeatureWriterAppend(sft.getTypeName, Transaction.AUTO_COMMIT)) { writer =>
140               var count = 0L
141               toPersist.values.foreach { p =>
142                 logger.trace(s"Persistent store append [$topic:$partition:${p.offset}] ${p.feature}")
143                 try { FeatureUtils.write(writer, p.feature, useProvidedFid = true) } catch {
144                   case NonFatal(e) => logger.error(s"Error persisting feature: ${p.feature}", e)
145                 }
146                 count += 1
147               }
148               count
149             }
150           }
151         }
152       }
153     }
154 
155     if (nextOffset > lastOffset) {
156       logger.trace(s"Committing offset [$topic:$partition:$nextOffset]")
157       offsetManager.setOffset(topic, partition, nextOffset)
158     }
159   }
160 
161   override def close(): Unit = {
162     schedule.cancel(true)
163     executor.shutdownNow()
164     executor.awaitTermination(1, TimeUnit.SECONDS)
165   }
