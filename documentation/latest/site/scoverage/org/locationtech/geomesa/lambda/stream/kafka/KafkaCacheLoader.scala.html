19 /**
20   * Consumes from kafka and populates the local cache
21   *   1. reads offsets stored in zk on startup
22   *   2. scheduled repeating - reads features from kafka, adds to in-memory cache
23   *   3. listens for offsets change in zk, removes expired features from in-memory cache
24   *
25   * @param consumers consumers
26   * @param topic kafka topic
27   * @param frequency kafka poll delay, in milliseconds
28   * @param serializer feature serializer
29   * @param cache shared state
30   */
31 class KafkaCacheLoader(
32     consumers: Seq[Consumer[Array[Byte], Array[Byte]]],
33     topic: String,
34     frequency: Long,
35     serializer: KryoFeatureSerializer,
36     cache: WritableFeatureCache,
37     offsetCommitIntervalMs: Long) extends ThreadedConsumer(consumers, Duration.ofMillis(frequency), offsetCommitIntervalMs) {
38 
39   startConsumers()
40 
41   override protected def consume(record: ConsumerRecord[Array[Byte], Array[Byte]]): Unit = {
42     val (time, action) = KafkaStore.deserializeKey(record.key)
43     val feature = serializer.deserialize(record.value)
44     action match {
45       case MessageTypes.Write  => cache.add(feature, record.partition, record.offset, time)
46       case MessageTypes.Delete => cache.delete(feature, record.partition, record.offset, time)
47       case _ => logger.error(s"Unhandled message type: $action")
48     }
49   }
50 }
