70   override def createSchema(sft: SimpleFeatureType): Unit = {
71     persistence.createSchema(sft)
72     // TODO for some reason lambda qs consumers don't rebalance when the topic is created after the consumers...
73     // transients.get(sft.getTypeName).createSchema()
74     val topic = KafkaStore.topic(config.zkNamespace, sft)
75     val props = new Properties()
76     config.producerConfig.foreach { case (k, v) => props.put(k, v) }
77 
78     WithClose(AdminClient.create(props)) { admin =>
79       if (admin.listTopics().names().get.contains(topic)) {
80         logger.warn(s"Topic [$topic] already exists - it may contain stale data")
81       } else {
82         val replication = SystemProperty("geomesa.kafka.replication").option.map(_.toInt).getOrElse(1)
83         val newTopic = new NewTopic(topic, config.partitions, replication.toShort)
84         admin.createTopics(Collections.singletonList(newTopic)).all().get
85       }
86     }
87   }

91   override def getSchema(typeName: String): SimpleFeatureType = {

100   override def updateSchema(typeName: Name, featureType: SimpleFeatureType): Unit =

103   override def updateSchema(typeName: String, featureType: SimpleFeatureType): Unit = {
104     val transient = transients.get(typeName)
105     if (typeName != featureType.getTypeName) {
106       // ensure that we've loaded the entire kafka topic
107       logger.debug("Update schema: entering quiet period")
108       Thread.sleep(SystemProperty("geomesa.lambda.update.quiet.period", "10 seconds").toDuration.get.toMillis)
109       WithClose(transient.read().iterator()) { toPersist =>
110         if (toPersist.nonEmpty) {
111           logger.debug("Update schema: persisting transient features")
112           WithClose(persistence.getFeatureWriter(typeName, Transaction.AUTO_COMMIT)) { writer =>
113             toPersist.foreach(FeatureUtils.write(writer, _, useProvidedFid = true))
114           }
115         }
116       }
117     }
118     CloseWithLogging(transient)
119     transients.invalidate(typeName)
120     persistence.updateSchema(typeName, featureType)
121   }

149   override def getFeatureWriterAppend(typeName: String, transaction: Transaction): SimpleFeatureWriter = {
150     val transient = transients.get(typeName)
151     if (transient.sft.isVisibilityRequired) {
152       new AppendLambdaFeatureWriter(transient) with RequiredVisibilityWriter
153     } else {
154       new AppendLambdaFeatureWriter(transient)
155     }

161   override def getFeatureWriter(typeName: String,
162                                 filter: Filter,
163                                 transaction: Transaction): SimpleFeatureWriter= {
164     val query = new Query(typeName, filter)
165     val features = SelfClosingIterator(getFeatureReader(query, transaction))
166     val transient = transients.get(typeName)
167     if (transient.sft.isVisibilityRequired) {
168       new ModifyLambdaFeatureWriter(transient, features) with RequiredVisibilityWriter
169     } else {
170       new ModifyLambdaFeatureWriter(transient, features)
171     }
172   }

174   override def dispose(): Unit = {
175     CloseWithLogging(transients.asMap.asScala.values)
176     transients.invalidateAll()
177     CloseWithLogging(config.offsetManager)
178     persistence.dispose()
179   }
