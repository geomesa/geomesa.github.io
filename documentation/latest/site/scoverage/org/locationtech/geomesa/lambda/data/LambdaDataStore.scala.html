70   override def createSchema(sft: SimpleFeatureType): Unit = {
71     val topic = LambdaDataStore.topic(sft, config.zkNamespace)
72     if (topic.contains("/")) {
73       // note: kafka doesn't allow slashes in topic names
74       throw new IllegalArgumentException(s"Topic cannot contain '/': $topic")
75     }
76     persistence.createSchema(sft)
77     // TODO for some reason lambda qs consumers don't rebalance when the topic is created after the consumers...
78     // transients.get(sft.getTypeName).createSchema()
79     val props = new Properties()
80     config.producerConfig.foreach { case (k, v) => props.put(k, v) }
81     WithClose(AdminClient.create(props)) { admin =>
82       if (admin.listTopics().names().get.contains(topic)) {
83         logger.warn(s"Topic [$topic] already exists - it may contain stale data")
84       } else {
85         val replication = SystemProperty("geomesa.kafka.replication").option.map(_.toInt).getOrElse(1)
86         val newTopic = new NewTopic(topic, config.partitions, replication.toShort)
87         admin.createTopics(Collections.singletonList(newTopic)).all().get
88       }
89     }
90   }

106   override def updateSchema(typeName: String, featureType: SimpleFeatureType): Unit = {
107     val transient = transients.get(typeName)
108     if (typeName != featureType.getTypeName) {
109       // ensure that we've loaded the entire kafka topic
110       logger.debug("Update schema: entering quiet period")
111       Thread.sleep(SystemProperty("geomesa.lambda.update.quiet.period", "10 seconds").toDuration.get.toMillis)
112       WithClose(transient.read().iterator()) { toPersist =>
113         if (toPersist.nonEmpty) {
114           logger.debug("Update schema: persisting transient features")
115           WithClose(persistence.getFeatureWriter(typeName, Transaction.AUTO_COMMIT)) { writer =>
116             val helper = FeatureWriterHelper(writer, useProvidedFids = true)
117             toPersist.foreach(helper.write)
118           }
119         }
120       }
121     }
122     CloseWithLogging(transient)
123     transients.invalidate(typeName)
124     persistence.updateSchema(typeName, featureType)
125   }

199   /**
200    * Gets the kafka topic configured in the sft, or a default topic if nothing is configured.
201    *
202    * @param sft simple feature type
203    * @param namespace namespace to use for default topic
204    * @return

206   def topic(sft: SimpleFeatureType, namespace: String): String = {
207     sft.getUserData.get(TopicKey) match {
208       case topic: String => topic
209       case _ => s"${namespace}_${sft.getTypeName}".replaceAll("[^a-zA-Z0-9_\\-]", "_")
210     }
211   }
