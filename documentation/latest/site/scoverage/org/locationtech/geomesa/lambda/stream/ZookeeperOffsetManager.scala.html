26 class ZookeeperOffsetManager(zookeepers: String, namespace: String = "geomesa") extends OffsetManager {
27 
28   import ZookeeperOffsetManager.offsetsPath

30   private val client = CuratorHelper.client(zookeepers).namespace(namespace).build()

33   private val listeners = scala.collection.mutable.Map.empty[String, CuratorOffsetListener]

35   override def getOffset(topic: String, partition: Int): Long = {
36     val path = ZookeeperOffsetManager.offsetsPath(topic, partition)
37     if (client.checkExists().forPath(path) == null) { -1L } else {
38       ZookeeperOffsetManager.deserializeOffsets(client.getData.forPath(path))
39     }
40   }
41 
42   override def setOffset(topic: String, partition: Int, offset: Long): Unit = {
43     val path = ZookeeperOffsetManager.offsetsPath(topic, partition)
44     if (client.checkExists().forPath(path) == null) {
45       client.create().creatingParentsIfNeeded().forPath(path)
46     }
47     client.setData().forPath(path, ZookeeperOffsetManager.serializeOffset(offset))
48   }
49 
50   override def deleteOffsets(topic: String): Unit = {
51     val path = ZookeeperOffsetManager.offsetsPath(topic)
52     if (client.checkExists().forPath(path) != null) {
53       client.delete().deletingChildrenIfNeeded().forPath(path)
54     }
55   }
56 
57   override def addOffsetListener(topic: String, listener: OffsetListener): Unit = synchronized {
58     listeners.getOrElseUpdate(topic, new CuratorOffsetListener(client, offsetsPath(topic))).addListener(listener)
59   }
60 
61   override def removeOffsetListener(topic: String, listener: OffsetListener): Unit = synchronized {
62     listeners.get(topic).foreach(_.removeListener(listener))
63   }
64 
65   override def close(): Unit = synchronized {
66     CloseWithLogging(listeners.values)
67     listeners.clear()
68     CloseWithLogging(client)
69   }
70 
71   override protected def acquireDistributedLock(path: String): Releasable =
72     acquireLock(path, lock => { lock.acquire(); true })
73 
74   override protected def acquireDistributedLock(path: String, timeOut: Long): Option[Releasable] =
75     Option(acquireLock(path, lock => lock.acquire(timeOut, TimeUnit.MILLISECONDS)))

77   private def acquireLock(path: String, acquire: InterProcessSemaphoreMutex => Boolean): Releasable = {
78     val lock = new InterProcessSemaphoreMutex(client, s"/$path/locks")
79     if (acquire(lock)) {
80       new Releasable { override def release(): Unit = lock.release() }
81     } else {
82       null
83     }
84   }

94   private def partitionFromPath(path: String): Int = path.substring(path.lastIndexOf("/") + 1).toInt

96   private class CuratorOffsetListener(client: CuratorFramework, path: String)
97       extends PathChildrenCacheListener with Closeable with LazyLogging {
98 
99     private val executor = Executors.newCachedThreadPool()
100 
101     private val listeners = scala.collection.mutable.Set.empty[OffsetListener]
102 
103     private var cache: PathChildrenCache = _
104 
105     def addListener(listener: OffsetListener): Unit = {
106       closeCache()
107       listeners += listener
108       cache = new PathChildrenCache(client, path, true) {
109         // override ensure path to avoid using containers, which aren't supported in zk 3.4
110         override protected def ensurePath(): Unit = {
111           if (client.checkExists().forPath(path) == null) {
112             client.create().creatingParentsIfNeeded().forPath(path)
113           }
114         }
115       }
116       cache.getListenable.addListener(this)
117       cache.start()
118     }
119 
120     def removeListener(listener: OffsetListener): Unit = {
121       listeners -= listener
122       if (listeners.isEmpty) {
123         closeCache()
124       }
125     }
126 
127     override def childEvent(client: CuratorFramework, event: PathChildrenCacheEvent): Unit = {
128       import PathChildrenCacheEvent.Type.{CHILD_ADDED, CHILD_UPDATED}
129       try {
130         val eventPath = Option(event.getData).map(_.getPath).getOrElse("")
131         if ((event.getType == CHILD_ADDED || event.getType == CHILD_UPDATED) && eventPath.startsWith(path)) {
132           logger.trace(s"ZK event triggered for: $eventPath")
133           val partition = partitionFromPath(eventPath)
134           val offset = ZookeeperOffsetManager.deserializeOffsets(event.getData.getData)
135           listeners.foreach { listener =>
136             executor.execute(new Runnable {
137               override def run(): Unit = {
138                 try { listener.offsetChanged(partition, offset) } catch {
139                   case NonFatal(e) => logger.warn("Error calling offset listener", e)
140                 }
141               }
142             })
143           }
144         }
145       } catch {
146         case NonFatal(e) => logger.warn("Error handling ZK event", e)
147       }
148     }
149 
150     override def close(): Unit = {
151       executor.shutdown()
152       closeCache()
153     }
154 
155     private def closeCache(): Unit = synchronized {
156       if (cache != null) {
157         cache.getListenable.removeListener(this)
158         CloseWithLogging(cache)
159         cache = null
160       }
161     }
162   }
