27 class ZookeeperOffsetManager(zookeepers: String, namespace: String = "geomesa") extends OffsetManager {
28 
29   import ZookeeperOffsetManager.offsetsPath
30 
31   private val client = CuratorHelper.client(zookeepers).namespace(namespace).build()
32   client.start()
33 
34   private val listeners = scala.collection.mutable.Map.empty[String, CuratorOffsetListener]
35 
36   override def getOffset(topic: String, partition: Int): Long = {
37     val path = ZookeeperOffsetManager.offsetsPath(topic, partition)
38     if (client.checkExists().forPath(path) == null) { -1L } else {
39       ZookeeperOffsetManager.deserializeOffsets(client.getData.forPath(path))
40     }
41   }
42 
43   override def setOffset(topic: String, partition: Int, offset: Long): Unit = {
44     val path = ZookeeperOffsetManager.offsetsPath(topic, partition)
45     if (client.checkExists().forPath(path) == null) {
46       client.create().creatingParentsIfNeeded().forPath(path)
47     }
48     client.setData().forPath(path, ZookeeperOffsetManager.serializeOffset(offset))
49   }
50 
51   override def deleteOffsets(topic: String): Unit = {
52     val path = ZookeeperOffsetManager.offsetsPath(topic)
53     if (client.checkExists().forPath(path) != null) {
54       client.delete().deletingChildrenIfNeeded().forPath(path)
55     }
56   }
57 
58   override def addOffsetListener(topic: String, listener: OffsetListener): Unit = synchronized {
59     listeners.getOrElseUpdate(topic, new CuratorOffsetListener(client, offsetsPath(topic))).addListener(listener)
60   }
61 
62   override def removeOffsetListener(topic: String, listener: OffsetListener): Unit = synchronized {
63     listeners.get(topic).foreach(_.removeListener(listener))
64   }
65 
66   override def close(): Unit = synchronized {
67     CloseWithLogging(listeners.values)
68     listeners.clear()
69     CloseWithLogging(client)
70   }
71 
72   override protected def acquireDistributedLock(path: String): Closeable =
73     acquireLock(path, lock => { lock.acquire(); true })
74 
75   override protected def acquireDistributedLock(path: String, timeOut: Long): Option[Closeable] =
76     Option(acquireLock(path, lock => lock.acquire(timeOut, TimeUnit.MILLISECONDS)))
77 
78   private def acquireLock(path: String, acquire: InterProcessSemaphoreMutex => Boolean): Closeable = {
79     val lock = new InterProcessSemaphoreMutex(client, s"/$path/locks")
80     if (acquire(lock)) {
81       () => lock.release()
82     } else {
83       null
84     }
85   }

88 object ZookeeperOffsetManager {
89 
90   def serializeOffset(offset: Long): Array[Byte] = offset.toString.getBytes(StandardCharsets.UTF_8)
91   def deserializeOffsets(bytes: Array[Byte]): Long = new String(bytes, StandardCharsets.UTF_8).toLong
92 
93   private def offsetsPath(topic: String): String = s"/$topic/offsets"
94   private def offsetsPath(topic: String, partition: Int): String = s"${offsetsPath(topic)}/$partition"
95   private def partitionFromPath(path: String): Int = path.substring(path.lastIndexOf("/") + 1).toInt
96 
97   private class CuratorOffsetListener(client: CuratorFramework, path: String)
98       extends PathChildrenCacheListener with Closeable with LazyLogging {
99 
100     private val executor = Executors.newCachedThreadPool()
101 
102     private val listeners = scala.collection.mutable.Set.empty[OffsetListener]
103 
104     private var cache: PathChildrenCache = _
105 
106     def addListener(listener: OffsetListener): Unit = {
107       closeCache()
108       listeners += listener
109       cache = new PathChildrenCache(client, path, true) {
110         // override ensure path to avoid using containers, which aren't supported in zk 3.4
111         override protected def ensurePath(): Unit = {
112           if (client.checkExists().forPath(path) == null) {
113             client.create().creatingParentsIfNeeded().forPath(path)
114           }
115         }
116       }
117       cache.getListenable.addListener(this)
118       cache.start()
119     }
120 
121     def removeListener(listener: OffsetListener): Unit = {
122       listeners -= listener
123       if (listeners.isEmpty) {
124         closeCache()
125       }
126     }
127 
128     override def childEvent(client: CuratorFramework, event: PathChildrenCacheEvent): Unit = {
129       import PathChildrenCacheEvent.Type.{CHILD_ADDED, CHILD_UPDATED}
130       try {
131         val eventPath = Option(event.getData).map(_.getPath).getOrElse("")
132         if ((event.getType == CHILD_ADDED || event.getType == CHILD_UPDATED) && eventPath.startsWith(path)) {
133           logger.trace(s"ZK event triggered for: $eventPath")
134           val partition = partitionFromPath(eventPath)
135           val data = event.getData.getData
136           Try(ZookeeperOffsetManager.deserializeOffsets(data)) match {
137             case Success(offset) =>
138               listeners.foreach { listener =>
139                 executor.execute(() => {
140                   try { listener.offsetChanged(partition, offset) } catch {
141                     case NonFatal(e) => logger.warn("Error calling offset listener", e)
142                   }
143                 })
144               }
145 
146             case Failure(e) =>
147               // for some reason we get IP addresses sometimes... seems to be something curator is doing.
148               // we generally get the correct data immediately afterwards, so it appears to be harmless
149               logger.warn(
150                 s"Error deserializing offset data " +
151                   s"'${Try(new String(data, StandardCharsets.UTF_8)).getOrElse(ByteArrays.toHex(data))}': ${e.getClass.getName}")
152           }
153         }
154       } catch {
155         case NonFatal(e) => logger.warn("Error handling ZK event", e)
156       }
157     }
158 
159     override def close(): Unit = {
160       executor.shutdown()
161       closeCache()
162     }
163 
164     private def closeCache(): Unit = synchronized {
165       if (cache != null) {
166         cache.getListenable.removeListener(this)
167         CloseWithLogging(cache)
168         cache = null
169       }
170     }
171   }
