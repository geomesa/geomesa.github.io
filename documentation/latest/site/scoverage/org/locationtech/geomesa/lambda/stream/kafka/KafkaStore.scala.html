85   override def createSchema(): Unit = {
86     val props = new Properties()
87     config.producerConfig.foreach { case (k, v) => props.put(k, v) }
88 
89     WithClose(AdminClient.create(props)) { admin =>
90       if (admin.listTopics().names().get.contains(topic)) {

100   override def removeSchema(): Unit = {

102     val props = new Properties()
103     config.producerConfig.foreach { case (k, v) => props.put(k, v) }
104 
105     WithClose(AdminClient.create(props)) { admin =>
106       if (admin.listTopics().names().get.contains(topic)) {
107         admin.deleteTopics(Collections.singletonList(topic)).all().get

114   override def read(
115       filter: Option[Filter] = None,
116       transforms: Option[Array[String]] = None,
117       hints: Option[Hints] = None,
118       explain: Explainer = new ExplainLogging): QueryResult = {
119     val query = new Query()
120     filter.foreach(query.setFilter)
121     transforms.foreach(query.setPropertyNames(_: _*))
122     hints.foreach(query.setHints)
123     queryRunner.runQuery(sft, query, explain)
124   }

126   override def write(original: SimpleFeature): Unit = {
127     val feature = GeoMesaFeatureWriter.featureWithFid(original)
128     val key = KafkaStore.serializeKey(clock.millis(), MessageTypes.Write)
129     producer.send(new ProducerRecord(topic, key, serializer.serialize(feature)))
130     logger.trace(s"Wrote feature to [$topic]: $feature")
131   }
132 
133   override def delete(original: SimpleFeature): Unit = {
134     import org.locationtech.geomesa.filter.ff
135     // send a message to delete from all transient stores
136     val feature = GeoMesaFeatureWriter.featureWithFid(original)
137     val key = KafkaStore.serializeKey(clock.millis(), MessageTypes.Delete)
138     producer.send(new ProducerRecord(topic, key, serializer.serialize(feature)))
139     // also delete from persistent store
140     if (config.persist) {
141       val filter = ff.id(ff.featureId(feature.getID))
142       WithClose(ds.getFeatureWriter(sft.getTypeName, filter, Transaction.AUTO_COMMIT)) { writer =>
143         while (writer.hasNext) {
144           writer.next()
145           writer.remove()
146         }
147       }
148     }
149   }
150 
151   override def persist(): Unit = persistence match {
152     case Some(p) => p.run()
153     case None => throw new IllegalStateException("Persistence disabled for this store")
154   }
155 
156   override def flush(): Unit = producer.flush()
157 
158   override def close(): Unit = {
159     CloseWithLogging(loader)
160     CloseWithLogging(interceptors)
161     CloseWithLogging(persistence)
162     offsetManager.removeOffsetListener(topic, cache)
163   }

211   private [kafka] def consumers(connect: Map[String, String],
212                                 topic: String,
213                                 manager: OffsetManager,
214                                 parallelism: Int,
215                                 callback: (Int, Long) => Unit): Seq[Consumer[Array[Byte], Array[Byte]]] = {
216     require(parallelism > 0, "Parallelism must be greater than 0")
217 
218     val group = UUID.randomUUID().toString
219 
220     Seq.fill(parallelism) {
221       val consumer = KafkaStore.consumer(connect, group)
222       val listener = new OffsetRebalanceListener(consumer, manager, callback)
223       KafkaConsumerVersions.subscribe(consumer, topic, listener)
224       consumer
225     }
226   }

228   private [kafka] def serializeKey(time: Long, action: Byte): Array[Byte] = {
229     val result = Array.ofDim[Byte](9)
230 
231     result(0) = ((time >> 56) & 0xff).asInstanceOf[Byte]
232     result(1) = ((time >> 48) & 0xff).asInstanceOf[Byte]
233     result(2) = ((time >> 40) & 0xff).asInstanceOf[Byte]
234     result(3) = ((time >> 32) & 0xff).asInstanceOf[Byte]
235     result(4) = ((time >> 24) & 0xff).asInstanceOf[Byte]
236     result(5) = ((time >> 16) & 0xff).asInstanceOf[Byte]
237     result(6) = ((time >> 8)  & 0xff).asInstanceOf[Byte]
238     result(7) = (time & 0xff        ).asInstanceOf[Byte]
239     result(8) = action
240 
241     result
242   }

244   private [kafka] def deserializeKey(key: Array[Byte]): (Long, Byte) = (ByteArrays.readLong(key), key(8))

246   private [kafka] class OffsetRebalanceListener(consumer: Consumer[Array[Byte], Array[Byte]],
247                                                 manager: OffsetManager,
248                                                 callback: (Int, Long) => Unit)
249       extends ConsumerRebalanceListener with LazyLogging {
250 
251     override def onPartitionsRevoked(topicPartitions: java.util.Collection[TopicPartition]): Unit = {}
252 
253     override def onPartitionsAssigned(topicPartitions: java.util.Collection[TopicPartition]): Unit = {
254       import scala.collection.JavaConverters._
255 
256       // ensure we have queues for each partition
257       // read our last committed offsets and seek to them
258       topicPartitions.asScala.foreach { tp =>
259 
260         // seek to earliest existing offset and return the offset
261         def seekToBeginning(): Long = {
262           KafkaConsumerVersions.seekToBeginning(consumer, tp)
263           consumer.position(tp) - 1
264         }
265 
266         val lastRead = manager.getOffset(tp.topic(), tp.partition())
267 
268         KafkaConsumerVersions.pause(consumer, tp)
269 
270         val offset = if (lastRead < 0) { seekToBeginning() } else {
271           try { consumer.seek(tp, lastRead + 1); lastRead } catch {
272             case NonFatal(e) =>
273               logger.warn(s"Error seeking to initial offset: [${tp.topic}:${tp.partition}:$lastRead]" +
274                   s", seeking to beginning: $e")
275               seekToBeginning()
276           }
277         }
278         callback.apply(tp.partition, offset)
279 
280         KafkaConsumerVersions.resume(consumer, tp)
281       }
282     }
283   }

288   class FeatureIdPartitioner extends Partitioner {
289 
290     private var serializer: KryoFeatureSerializer = _
291 
292     private val features = new ThreadLocal[KryoBufferSimpleFeature]() {
293       override def initialValue(): KryoBufferSimpleFeature = serializer.getReusableFeature
294     }
295 
296     override def partition(
297         topic: String,
298         key: scala.Any,
299         keyBytes: Array[Byte],
300         value: scala.Any,
301         valueBytes: Array[Byte],
302         cluster: Cluster): Int = {
303       val numPartitions = cluster.partitionsForTopic(topic).size
304       if (numPartitions < 2) { 0 } else {
305         val feature = features.get
306         feature.setBuffer(valueBytes)
307         Math.abs(MurmurHash3.stringHash(feature.getID)) % numPartitions
308       }
309     }
310 
311     override def configure(configs: java.util.Map[String, _]): Unit = {
312       val spec = configs.get(SimpleFeatureSpecConfig) match {
313         case s: String => s
314         case s => throw new IllegalStateException(s"Invalid spec config for $SimpleFeatureSpecConfig: $s")
315       }
316       val options = SerializationOptions.builder.immutable.`lazy`.build
317       serializer = KryoFeatureSerializer(SimpleFeatureTypes.createType("", spec), options)
318     }
