83   override def createSchema(): Unit = {
84     val props = new Properties()

188   private [kafka] def consumers(connect: Map[String, String],
189                                 topic: String,
190                                 manager: OffsetManager,
191                                 parallelism: Int,
192                                 callback: (Int, Long) => Unit): Seq[Consumer[Array[Byte], Array[Byte]]] = {
193     require(parallelism > 0, "Parallelism must be greater than 0")
194 
195     val group = UUID.randomUUID().toString
196 
197     Seq.fill(parallelism) {
198       val consumer = KafkaStore.consumer(connect, group)
199       val listener = new OffsetRebalanceListener(consumer, manager, callback)
200       KafkaConsumerVersions.subscribe(consumer, topic, listener)
201       consumer
202     }
203   }

205   private [kafka] def serializeKey(time: Long, action: Byte): Array[Byte] = {
206     val result = Array.ofDim[Byte](9)
207 
208     result(0) = ((time >> 56) & 0xff).asInstanceOf[Byte]
209     result(1) = ((time >> 48) & 0xff).asInstanceOf[Byte]
210     result(2) = ((time >> 40) & 0xff).asInstanceOf[Byte]
211     result(3) = ((time >> 32) & 0xff).asInstanceOf[Byte]
212     result(4) = ((time >> 24) & 0xff).asInstanceOf[Byte]
213     result(5) = ((time >> 16) & 0xff).asInstanceOf[Byte]
214     result(6) = ((time >> 8)  & 0xff).asInstanceOf[Byte]
215     result(7) = (time & 0xff        ).asInstanceOf[Byte]
216     result(8) = action
217 
218     result
219   }

223   private [kafka] class OffsetRebalanceListener(consumer: Consumer[Array[Byte], Array[Byte]],
224                                                 manager: OffsetManager,
225                                                 callback: (Int, Long) => Unit)
226       extends ConsumerRebalanceListener with LazyLogging {
227 
228     override def onPartitionsRevoked(topicPartitions: java.util.Collection[TopicPartition]): Unit = {}
229 
230     override def onPartitionsAssigned(topicPartitions: java.util.Collection[TopicPartition]): Unit = {
231       import scala.collection.JavaConverters._
232 
233       // ensure we have queues for each partition
234       // read our last committed offsets and seek to them
235       topicPartitions.asScala.foreach { tp =>
236 
237         // seek to earliest existing offset and return the offset
238         def seekToBeginning(): Long = {
239           KafkaConsumerVersions.seekToBeginning(consumer, tp)
240           consumer.position(tp) - 1
241         }
242 
243         val lastRead = manager.getOffset(tp.topic(), tp.partition())
244 
245         KafkaConsumerVersions.pause(consumer, tp)
246 
247         val offset = if (lastRead < 0) { seekToBeginning() } else {
248           try { consumer.seek(tp, lastRead + 1); lastRead } catch {
249             case NonFatal(e) =>
250               logger.warn(s"Error seeking to initial offset: [${tp.topic}:${tp.partition}:$lastRead]" +
251                   s", seeking to beginning: $e")
252               seekToBeginning()
253           }
254         }
255         callback.apply(tp.partition, offset)
256 
257         KafkaConsumerVersions.resume(consumer, tp)
258       }
259     }
260   }

267     private var serializer: KryoFeatureSerializer = _

269     private val features = new ThreadLocal[KryoBufferSimpleFeature]() {

273     override def partition(
274         topic: String,
275         key: scala.Any,
276         keyBytes: Array[Byte],
277         value: scala.Any,
278         valueBytes: Array[Byte],
279         cluster: Cluster): Int = {
280       val numPartitions = cluster.partitionsForTopic(topic).size
281       if (numPartitions < 2) { 0 } else {
282         val feature = features.get
283         feature.setBuffer(valueBytes)
284         Math.abs(MurmurHash3.stringHash(feature.getID)) % numPartitions
285       }
286     }

292       }
293       val options = SerializationOption.builder.immutable.`lazy`.build()
294       serializer = KryoFeatureSerializer(SimpleFeatureTypes.createType("", spec), options)
295     }
