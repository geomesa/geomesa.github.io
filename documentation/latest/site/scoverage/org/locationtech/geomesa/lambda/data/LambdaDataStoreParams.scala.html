114   def parse(params: java.util.Map[String, _], namespace: String): LambdaConfig = {
115     val brokers = BrokersParam.lookup(params)
116     val expiry = if (!PersistParam.lookup(params).booleanValue) { None } else {
117       ExpiryParam.lookupOpt(params).collect { case d: FiniteDuration => d}
118     }
119 
120     val partitions = PartitionsParam.lookup(params).intValue
121     val consumers = ConsumersParam.lookup(params).intValue
122     val batchSize = BatchSizeParam.lookupOpt(params).map(_.intValue())
123 
124     val bootstrap = Map("bootstrap.servers" -> brokers)
125     val consumerConfig = ConsumerOptsParam.lookupOpt(params).map(_.asScala.toMap).getOrElse(Map.empty) ++ bootstrap
126     val producerConfig = ProducerOptsParam.lookupOpt(params).map(_.asScala.toMap).getOrElse(Map.empty) ++ bootstrap
127 
128     val zk = ZookeepersParam.lookup(params)
129     val zkNamespace = s"gm_lambda_$namespace"
130 
131     val offsetCommitIntervalMs : Long = ConsumerOffsetCommitIntervalMs.lookupOpt(params).map(_.toLong).getOrElse(10000)
132 
133     LambdaConfig(zk, zkNamespace, producerConfig, consumerConfig, partitions, consumers, expiry, batchSize, offsetCommitIntervalMs)
134   }
