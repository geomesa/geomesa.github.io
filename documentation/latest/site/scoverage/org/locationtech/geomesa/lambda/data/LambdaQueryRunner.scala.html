35   override def runQuery(sft: SimpleFeatureType, original: Query, explain: Explainer): QueryResult = {
36     // configure the query so we get viewparams and threaded hints
37     val query = configureQuery(sft, original)
38     val result = super.runQuery(sft, query, explain)
39     if (query.getHints.isLambdaQueryPersistent && query.getHints.isLambdaQueryTransient) {
40       result
41     } else if (query.getHints.isLambdaQueryPersistent) {
42       result.copy(iterator = () => SelfClosingIterator(persistence.getFeatureReader(query, Transaction.AUTO_COMMIT)))
43     } else {
44       def run(): CloseableIterator[SimpleFeature] = {
45         // ensure that we still audit the query
46         val audit = Option(persistence).collect { case ds: GeoMesaDataStore[_] => ds.config.audit }.flatten
47         audit.foreach { case (writer, provider, typ) =>
48           val stat = QueryEvent(
49             s"$typ-lambda",
50             sft.getTypeName,
51             System.currentTimeMillis(),
52             provider.getCurrentUserId,
53             filterToString(query.getFilter),
54             ViewParams.getReadableHints(query),
55             0,
56             0,
57             0
58           )
59           writer.writeEvent(stat) // note: implementations should be asynchronous
60         }
61         transients.get(sft.getTypeName)
62             .read(Option(query.getFilter), Option(query.getPropertyNames), Option(query.getHints), explain)
63             .iterator()
64       }
65       result.copy(iterator = run)
66     }
67   }
68 }
