103   override def offsetChanged(partition: Int, offset: Long): Unit = {
104     if (offsets.length <= partition) {
105       logger.debug(s"Offsets changed for [$topic:$partition]: (unassigned) -> $offset")
106       ensurePartition(partition, offset)

111     var last = offsets(partition).get
112     logger.debug(s"Offsets changed for [$topic:$partition]: $last -> $offset")
113     while (last < offset && !offsets(partition).compareAndSet(last, offset)) {
114       last = offsets(partition).get

163   private class Persistence(expiryMillis: Long, batchSize: Int) extends Closeable {
164 
165     private val frequency = KafkaFeatureCache.PersistInterval.toMillis.get
166     private val executor = ExitingExecutor(new ScheduledThreadPoolExecutor(1))
167     private val schedule = executor.scheduleWithFixedDelay(() => persist(), frequency, frequency, TimeUnit.MILLISECONDS)

169     /**
170      * Persist any expired features
171      *
172      * @return list of last persisted offsets per partition
173      */
174     def persist(): Seq[PartitionOffset] = {
175       logger.debug(s"Running persistence for [$topic]")
176       // lock per-partition to allow for multiple write threads
177       // randomly access the partitions to avoid contention if multiple data stores are all on the same schedule
178       Random.shuffle(offsets.indices.toList).flatMap { partition =>
179         // if we don't get the lock just try again next run
180         logger.trace(s"Acquiring lock for [$topic:$partition]")
181         offsetManager.acquireLock(topic, partition, lockTimeout) match {
182           case None =>
183             logger.trace(s"Could not acquire lock for [$topic:$partition] within ${lockTimeout}ms")
184             None
185           case Some(lock) =>
186             try {
187               logger.trace(s"Acquired lock for [$topic:$partition]")
188               persist(partition).map(PartitionOffset(partition, _))
189             } finally {
190               lock.close()
191               logger.trace(s"Released lock for [$topic:$partition]")
192             }
193         }
194       }
195     }

215     /**
216      * Persist any expired features that haven't yet been persisted
217      *
218      * @param partition partition to persist
219      * @param lastPersistedOffset result from previous recursive run, if any
220      * @return offset of latest persisted feature
221      */
222     @tailrec
223     private def persist(partition: Int, lastPersistedOffset: Option[Long] = None): Option[Long] = {
224       val expiry = clock.millis() - expiryMillis
225 
226       val lastOffset = offsetManager.getOffset(topic, partition)
227       logger.debug(s"Last persisted offsets for [$topic:$partition]: $lastOffset")
228 
229       var nextOffset = -1L
230       // note: copy to a new collection so that any updates don't affect our persistence here
231       val expired = {
232         val builder = Map.newBuilder[String, FeatureReference]
233         features.asScala.foreach { case (id, f) =>
234           // note: offset > lastOffset ensures we don't operate on features that have already been processed
235           if (f.partition == partition && f.created <= expiry && f.offset > lastOffset) {
236             nextOffset = math.max(nextOffset, f.offset)
237             builder += id -> f
238           }
239         }
240         builder.result()
241       }
242       logger.debug(s"Found ${expired.size} expired entries for [$topic:$partition]")
243 
244       if (expired.isEmpty) {
245         lastPersistedOffset
246       } else {
247         logger.trace(expired.values.map(e => s"offset ${e.offset}: ${e.feature}").mkString("Expired entries:\n\t", "\n\t", ""))
248         val batch =
249           if (expired.size <= batchSize) {
250             expired
251           } else {
252             logger.trace(s"Skipping persistence for ${expired.size - batchSize} features based on batch size of $batchSize")
253             // sort by offset since we track persistence based on offset
254             val sorted = expired.toList.sortBy(_._2.offset).take(batchSize)
255             nextOffset = sorted.last._2.offset
256             sorted.toMap
257           }
258         persist(partition, batch)
259 
260         logger.debug(s"Committing offset [$topic:$partition:$nextOffset]")
261         // this will trigger our listener and cause the feature to be removed from the cache
262         offsetManager.setOffset(topic, partition, nextOffset)
263         if (batch.size < expired.size) {
264           // we skipped some expired features, so run again immediately
265           persist(partition, Some(nextOffset))
266         } else {
267           Some(nextOffset)
268         }
269       }
270     }

272     /**
273      * Persist expired features
274      *
275      * @param partition partition
276      * @param batch expired features
277      */
278     private def persist(partition: Int, batch: Map[String, FeatureReference]): Unit = {
279       // do an update query first
280       val remaining = persistUpdates(partition, batch)
281       // if any weren't updates, add them as inserts
282       if (remaining.nonEmpty) {
283         persistAppends(partition, remaining)
284       }
285     }

287     /**
288      * Attempt to persist expired features through modifying writes
289      *
290      * @param partition partition being persisted
291      * @param batch expired feature
292      * @return any features that were not persisted
293      */
294     private def persistUpdates(partition: Int, batch: Map[String, FeatureReference]): Iterable[FeatureReference] = {
295       logger.debug(s"Attempting persistent updates on ${batch.size} features")
296       val persistedKeys = scala.collection.mutable.Set.empty[String]
297       profile((c: Int, time: Long) => logger.debug(s"Wrote $c updated feature(s) to persistent storage in ${time}ms")) {
298         val filter = ff.id(batch.keys.map(ff.featureId).toSeq: _*)
299         WithClose(ds.getFeatureWriter(sft.getTypeName, filter, Transaction.AUTO_COMMIT)) { writer =>
300           var count = 0
301           while (writer.hasNext) {
302             val next = writer.next()
303             batch.get(next.getID).foreach { p =>
304               logger.trace(s"Persistent store modify [$topic:$partition:${p.offset}] ${p.feature}")
305               persistedKeys += next.getID
306               FeatureUtils.copyToFeature(next, p.feature, useProvidedFid = true)
307               try {
308                 writer.write()
309                 count += 1
310               } catch {
311                 case NonFatal(e) => logger.error(s"Error persisting feature: ${p.feature}", e)
312               }
313             }
314           }
315           count
316         }
317       }
318       batch.collect { case (k, v) if !persistedKeys.contains(k) => v }
319     }

321     /**
322      * Attempt to persist expired features through appending writes
323      *
324      * @param partition partition being persisted
325      * @param batch expired feature
326      */
327     private def persistAppends(partition: Int, batch: Iterable[FeatureReference]): Unit = {
328       logger.debug(s"Attempting persistent appends on ${batch.size} features")
329       profile((c: Int, time: Long) => logger.debug(s"Wrote $c new feature(s) to persistent storage in ${time}ms")) {
330         WithClose(ds.getFeatureWriterAppend(sft.getTypeName, Transaction.AUTO_COMMIT)) { writer =>
331           var count = 0
332           val helper = FeatureWriterHelper(writer, useProvidedFids = true)
333           batch.foreach { p =>
334             logger.trace(s"Persistent store append [$topic:$partition:${p.offset}] ${p.feature}")
335             try {
336               helper.write(p.feature)
337               count += 1
338             } catch {
339               case NonFatal(e) => logger.error(s"Error persisting feature: ${p.feature}", e)
340             }
341           }
342           count
343         }
344       }
345     }
