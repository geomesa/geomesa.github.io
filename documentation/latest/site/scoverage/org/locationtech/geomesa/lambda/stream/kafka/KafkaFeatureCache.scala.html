105   override def offsetChanged(partition: Int, offset: Long): Unit = {
106     if (offsets.length <= partition) {
107       logger.debug(s"Offsets changed for [$topic:$partition]: (unassigned) -> $offset")
108       ensurePartition(partition, offset)
109       return

112     // update the valid offset
113     var last = offsets(partition).get
114     logger.debug(s"Offsets changed for [$topic:$partition]: $last -> $offset")
115     while (last < offset && !offsets(partition).compareAndSet(last, offset)) {
116       last = offsets(partition).get
117     }

162   /**
163    * Wrapper for managing scheduled persistence runs
164    */
165   private class Persistence(expiryMillis: Long) extends Closeable {
166 
167     private val frequency = KafkaFeatureCache.PersistInterval.toMillis.get
168     private val executor = ExitingExecutor(new ScheduledThreadPoolExecutor(1))
169     private val schedule = executor.scheduleWithFixedDelay(() => persist(), frequency, frequency, TimeUnit.MILLISECONDS)

171     /**
172      * Persist any expired features
173      *
174      * @return list of last persisted offsets per partition
175      */
176     def persist(): Seq[PartitionOffset] = {
177       logger.debug(s"Running persistence for [$topic]")
178       // lock per-partition to allow for multiple write threads
179       // randomly access the partitions to avoid contention if multiple data stores are all on the same schedule
180       Random.shuffle(Range(0, offsets.length).toList).flatMap { partition =>
181         // if we don't get the lock just try again next run
182         logger.trace(s"Acquiring lock for [$topic:$partition]")
183         offsetManager.acquireLock(topic, partition, lockTimeout) match {
184           case None =>
185             logger.trace(s"Could not acquire lock for [$topic:$partition] within ${lockTimeout}ms")
186             None
187           case Some(lock) =>
188             try {
189               logger.trace(s"Acquired lock for [$topic:$partition]")
190               persist(partition).map(PartitionOffset(partition, _))
191             } finally {
192               lock.close()
193               logger.trace(s"Released lock for [$topic:$partition]")
194             }
195         }
196       }
197     }

217     /**
218      * Persist any expired features that haven't yet been persisted
219      *
220      * @param partition partition to persist
221      * @param lastPersistedOffset result from previous recursive run, if any
222      * @return offset of latest persisted feature
223      */
224     @tailrec
225     private def persist(partition: Int, lastPersistedOffset: Option[Long] = None): Option[Long] = {
226       val expiry = clock.millis() - expiryMillis
227 
228       val lastOffset = offsetManager.getOffset(topic, partition)
229       logger.debug(s"Last persisted offsets for [$topic:$partition]: $lastOffset")
230 
231       var nextOffset = -1L
232       // note: copy to a new collection so that any updates don't affect our persistence here
233       val expired = {
234         val builder = Map.newBuilder[String, FeatureReference]
235         features.asScala.foreach { case (id, f) =>
236           // note: offset > lastOffset ensures we don't operate on features that have already been processed
237           if (f.partition == partition && f.created <= expiry && f.offset > lastOffset) {
238             nextOffset = math.max(nextOffset, f.offset)
239             builder += id -> f
240           }
241         }
242         builder.result()
243       }
244       logger.debug(s"Found ${expired.size} expired entries for [$topic:$partition]")
245 
246       if (expired.isEmpty) {
247         lastPersistedOffset
248       } else {
249         logger.trace(expired.values.map(e => s"offset ${e.offset}: ${e.feature}").mkString("Expired entries:\n\t", "\n\t", ""))
250         val batch = batchSize match {
251           case Some(max) if expired.size > max =>
252             logger.trace(s"Skipping persistence for ${expired.size - max} features based on batch size of $max")
253             // sort by offset since we track persistence based on offset
254             val sorted = expired.toList.sortBy(_._2.offset).take(max)
255             nextOffset = sorted.last._2.offset
256             sorted.toMap
257           case _ =>
258             expired
259         }
260         persist(partition, batch)
261 
262         logger.debug(s"Committing offset [$topic:$partition:$nextOffset]")
263         // this will trigger our listener and cause the feature to be removed from the cache
264         offsetManager.setOffset(topic, partition, nextOffset)
265         if (batch.size < expired.size) {
266           // we skipped some expired features, so run again immediately
267           persist(partition, Some(nextOffset))
268         } else {
269           Some(nextOffset)
270         }
271       }
272     }

274     /**
275      * Persist expired features
276      *
277      * @param partition partition
278      * @param batch expired features
279      */
280     private def persist(partition: Int, batch: Map[String, FeatureReference]): Unit = {
281       // do an update query first
282       val remaining = persistUpdates(partition, batch)
283       // if any weren't updates, add them as inserts
284       if (remaining.nonEmpty) {
285         persistAppends(partition, remaining)
286       }
287     }

289     /**
290      * Attempt to persist expired features through modifying writes
291      *
292      * @param partition partition being persisted
293      * @param batch expired feature
294      * @return any features that were not persisted
295      */
296     private def persistUpdates(partition: Int, batch: Map[String, FeatureReference]): Iterable[FeatureReference] = {
297       logger.debug(s"Attempting persistent updates on ${batch.size} features")
298       val persistedKeys = scala.collection.mutable.Set.empty[String]
299       profile((c: Int, time: Long) => logger.debug(s"Wrote $c updated feature(s) to persistent storage in ${time}ms")) {
300         val filter = ff.id(batch.keys.map(ff.featureId).toSeq: _*)
301         WithClose(ds.getFeatureWriter(sft.getTypeName, filter, Transaction.AUTO_COMMIT)) { writer =>
302           var count = 0
303           while (writer.hasNext) {
304             val next = writer.next()
305             batch.get(next.getID).foreach { p =>
306               logger.trace(s"Persistent store modify [$topic:$partition:${p.offset}] ${p.feature}")
307               persistedKeys += next.getID
308               FeatureUtils.copyToFeature(next, p.feature, useProvidedFid = true)
309               try {
310                 writer.write()
311                 count += 1
312               } catch {
313                 case NonFatal(e) => logger.error(s"Error persisting feature: ${p.feature}", e)
314               }
315             }
316           }
317           count
318         }
319       }
320       batch.collect { case (k, v) if !persistedKeys.contains(k) => v }
321     }
322 
323     /**
324      * Attempt to persist expired features through appending writes
325      *
326      * @param partition partition being persisted
327      * @param batch expired feature
328      */
329     private def persistAppends(partition: Int, batch: Iterable[FeatureReference]): Unit = {
330       logger.debug(s"Attempting persistent appends on ${batch.size} features")
331       profile((c: Int, time: Long) => logger.debug(s"Wrote $c new feature(s) to persistent storage in ${time}ms")) {
332         WithClose(ds.getFeatureWriterAppend(sft.getTypeName, Transaction.AUTO_COMMIT)) { writer =>
333           var count = 0
334           batch.foreach { p =>
335             logger.trace(s"Persistent store append [$topic:$partition:${p.offset}] ${p.feature}")
336             try {
337               FeatureUtils.write(writer, p.feature, useProvidedFid = true)
338               count += 1
339             } catch {
340               case NonFatal(e) => logger.error(s"Error persisting feature: ${p.feature}", e)
341             }
342           }
343           count
344         }
345       }
346     }
