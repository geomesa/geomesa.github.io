55   override def add(feature: SimpleFeature, partition: Int, offset: Long, created: Long): Unit = {
56     if (offsets(partition).get < offset) {
57       logger.trace(s"Adding [$partition:$offset] $feature created at " +
58           s"${ZonedDateTime.ofInstant(Instant.ofEpochMilli(created), ZoneOffset.UTC)}")
59       val id = feature.getID
60       val ref = new FeatureReference(feature, id, partition, offset, created)
61       features.put(id, ref)
62       val (lock, queue) = queues(partition)
63       lock.lock()
64       // make the feature null so that we don't keep it around in memory when we don't need to
65       try { queue.addLast(new FeatureReference(null, id, partition, offset, created)) } finally {
66         lock.unlock()
67       }
68     } else {
69       logger.trace(s"Ignoring [$partition:$offset] $feature created at " +
70           s"${ZonedDateTime.ofInstant(Instant.ofEpochMilli(created), ZoneOffset.UTC)}")
71     }
72   }

74   override def delete(feature: SimpleFeature, partition: Int, offset: Long, created: Long): Unit = {
75     logger.trace(s"Deleting [$partition:$offset] $feature created at " +
76         s"${ZonedDateTime.ofInstant(Instant.ofEpochMilli(created), ZoneOffset.UTC)}")

80   override def expired(expiry: Long): Seq[Int] = {
81     val result = ArrayBuffer.empty[Int]
82     var i = 0
83     while (i < this.queues.length) {
84       val (lock, queue) = queues(i)
85       lock.lock()
86       val peek = try { queue.peek } finally { lock.unlock() }
87       peek match {
88         case null => // no-op
89         case ref => if (expiry > ref.created) { result += i }
90       }
91       i += 1
92     }
93     result.toSeq
94   }
95 
96   override def expired(partition: Int, expiry: Long): ExpiredFeatures = {
97     val expired = ArrayBuffer.empty[FeatureReference]
98     val (lock, queue) = this.queues(partition)
99 
100     var loop = true
101     while (loop) {
102       lock.lock()
103       val poll = queue.poll()
104       if (poll == null) {
105         lock.unlock()
106         loop = false
107       } else if (poll.created > expiry) {
108         // note: add back to the queue before unlocking
109         try { queue.addFirst(poll) } finally {
110           lock.unlock()
111         }
112         loop = false
113       } else {
114         lock.unlock()
115         expired += poll
116       }
117     }
118 
119     logger.debug(
120       s"Checking [$topic:$partition] for expired entries: found ${expired.size} expired and ${queue.size} remaining")
121 
122     val maxExpiredOffset = if (expired.isEmpty) { -1L } else { expired(expired.length - 1).offset }
123 
124     // only remove from feature cache (and persist) if there haven't been additional updates
125     val latest = expired.flatMap(ref => remove(ref))
126 
127     ExpiredFeatures(maxExpiredOffset, latest.toSeq)
128   }
129 
130   override def offsetChanged(partition: Int, offset: Long): Unit = {
131     logger.debug(s"Offsets changed for [$topic:$partition]: -> $offset")
132 
133     if (queues.length <= partition) {
134       ensurePartition(partition, offset)
135       return
136     }
137 
138     val (lock, queue) = queues(partition)
139 
140     // remove the expired features from the cache
141     val (featureSize, queueSize, start) = if (!debug) { (0, 0, 0L) } else {
142       (features.size, queue.size, System.currentTimeMillis())
143     }
144 
145     var loop = true
146     while (loop) {
147       lock.lock()
148       val poll = queue.poll()
149       if (poll == null) {
150         lock.unlock()
151         loop = false
152       } else if (poll.offset > offset) {
153         // note: add back to the queue before unlocking
154         try { queue.addFirst(poll) } finally {
155           lock.unlock()
156         }
157         loop = false
158       } else {
159         lock.unlock()
160         // only remove from feature cache if there haven't been additional updates
161         remove(poll)
162       }
163     }
164 
165     // update the valid offset
166     var last = offsets(partition).get
167     while(last < offset && !offsets(partition).compareAndSet(last, offset)) {
168       last = offsets(partition).get
169     }
170 
171     logger.debug(s"Size of cached state for [$topic:$partition]: features (total): " +
172         s"${diff(featureSize, features.size)}, offsets: ${diff(queueSize, queue.size)} in " +
173         s"${System.currentTimeMillis() - start}ms")
174   }
175 
176   private def ensurePartition(partition: Int, offset: Long): Unit = synchronized {
177     while (queues.length <= partition) {
178       queues += new ReentrantLock -> new java.util.ArrayDeque[FeatureReference]
179       offsets += new AtomicLong(-1L)
180     }
181     offsets(partition).set(offset)
182   }
183 
184   /**
185    * Conditionally removes the simple feature from the feature cache if it is the latest version. This
186    * should only be called with items pulled of the expiry queue
187    *
188    * @param feature feature reference
189    * @return
190    */
191   private def remove(feature: FeatureReference): Option[FeatureReference] = {
192     // since the features are added to the queue after they are added to the cache,
193     // this must be either the feature corresponding to the reference, or a later feature
194     val current = features.get(feature.id)
195     // the remove is based on the offset and partition, so if it is successful, we know that we
196     // got the right feature above
197     if (features.remove(feature.id, feature)) { Option(current) } else { None }
198   }
199 
200   // debug message
201   private def diff(original: Int, updated: Int): String = f"$updated%d (${updated - original}%+d)"
