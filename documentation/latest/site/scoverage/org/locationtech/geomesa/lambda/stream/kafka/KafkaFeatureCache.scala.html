106   override def offsetChanged(partition: Int, offset: Long): Unit = {
107     if (offsets.length <= partition) {
108       logger.debug(s"Offsets changed for [$topic:$partition]: (unassigned) -> $offset")
109       ensurePartition(partition, offset)
110       return
111     }

113     // update the valid offset
114     var last = offsets(partition).get
115     logger.debug(s"Offsets changed for [$topic:$partition]: $last -> $offset")
116     while (last < offset && !offsets(partition).compareAndSet(last, offset)) {
117       last = offsets(partition).get
118     }

164    * Wrapper for managing scheduled persistence runs
165    */
166   private class Persistence(expiryMillis: Long) extends Closeable {
167 
168     private val frequency = KafkaFeatureCache.PersistInterval.toMillis.get
169     private val executor = ExitingExecutor(new ScheduledThreadPoolExecutor(1))
170     private val schedule = executor.scheduleWithFixedDelay(() => persist(), frequency, frequency, TimeUnit.MILLISECONDS)

172     /**
173      * Persist any expired features
174      *
175      * @return list of last persisted offsets per partition
176      */
177     def persist(): Seq[PartitionOffset] = {
178       logger.debug(s"Running persistence for [$topic]")
179       // lock per-partition to allow for multiple write threads
180       // randomly access the partitions to avoid contention if multiple data stores are all on the same schedule
181       Random.shuffle(Range(0, offsets.length).toList).flatMap { partition =>
182         // if we don't get the lock just try again next run
183         logger.trace(s"Acquiring lock for [$topic:$partition]")
184         offsetManager.acquireLock(topic, partition, lockTimeout) match {
185           case None =>
186             logger.trace(s"Could not acquire lock for [$topic:$partition] within ${lockTimeout}ms")
187             None
188           case Some(lock) =>
189             try {
190               logger.trace(s"Acquired lock for [$topic:$partition]")
191               persist(partition).map(PartitionOffset(partition, _))
192             } finally {
193               lock.close()
194               logger.trace(s"Released lock for [$topic:$partition]")
195             }
196         }
197       }
198     }

218     /**
219      * Persist any expired features that haven't yet been persisted
220      *
221      * @param partition partition to persist
222      * @param lastPersistedOffset result from previous recursive run, if any
223      * @return offset of latest persisted feature
224      */
225     @tailrec
226     private def persist(partition: Int, lastPersistedOffset: Option[Long] = None): Option[Long] = {
227       val expiry = clock.millis() - expiryMillis
228 
229       val lastOffset = offsetManager.getOffset(topic, partition)
230       logger.debug(s"Last persisted offsets for [$topic:$partition]: $lastOffset")
231 
232       var nextOffset = -1L
233       // note: copy to a new collection so that any updates don't affect our persistence here
234       val expired = {
235         val builder = Map.newBuilder[String, FeatureReference]
236         features.asScala.foreach { case (id, f) =>
237           // note: offset > lastOffset ensures we don't operate on features that have already been processed
238           if (f.partition == partition && f.created <= expiry && f.offset > lastOffset) {
239             nextOffset = math.max(nextOffset, f.offset)
240             builder += id -> f
241           }
242         }
243         builder.result()
244       }
245       logger.debug(s"Found ${expired.size} expired entries for [$topic:$partition]")
246 
247       if (expired.isEmpty) {
248         lastPersistedOffset
249       } else {
250         logger.trace(expired.values.map(e => s"offset ${e.offset}: ${e.feature}").mkString("Expired entries:\n\t", "\n\t", ""))
251         val batch = batchSize match {
252           case Some(max) if expired.size > max =>
253             logger.trace(s"Skipping persistence for ${expired.size - max} features based on batch size of $max")
254             // sort by offset since we track persistence based on offset
255             val sorted = expired.toList.sortBy(_._2.offset).take(max)
256             nextOffset = sorted.last._2.offset
257             sorted.toMap
258           case _ =>
259             expired
260         }
261         persist(partition, batch)
262 
263         logger.debug(s"Committing offset [$topic:$partition:$nextOffset]")
264         // this will trigger our listener and cause the feature to be removed from the cache
265         offsetManager.setOffset(topic, partition, nextOffset)
266         if (batch.size < expired.size) {
267           // we skipped some expired features, so run again immediately
268           persist(partition, Some(nextOffset))
269         } else {
270           Some(nextOffset)
271         }
272       }
273     }

275     /**
276      * Persist expired features
277      *
278      * @param partition partition
279      * @param batch expired features
280      */
281     private def persist(partition: Int, batch: Map[String, FeatureReference]): Unit = {
282       // do an update query first
283       val remaining = persistUpdates(partition, batch)
284       // if any weren't updates, add them as inserts
285       if (remaining.nonEmpty) {
286         persistAppends(partition, remaining)
287       }
288     }

290     /**
291      * Attempt to persist expired features through modifying writes
292      *
293      * @param partition partition being persisted
294      * @param batch expired feature
295      * @return any features that were not persisted
296      */
297     private def persistUpdates(partition: Int, batch: Map[String, FeatureReference]): Iterable[FeatureReference] = {
298       logger.debug(s"Attempting persistent updates on ${batch.size} features")
299       val persistedKeys = scala.collection.mutable.Set.empty[String]
300       profile((c: Int, time: Long) => logger.debug(s"Wrote $c updated feature(s) to persistent storage in ${time}ms")) {
301         val filter = ff.id(batch.keys.map(ff.featureId).toSeq: _*)
302         WithClose(ds.getFeatureWriter(sft.getTypeName, filter, Transaction.AUTO_COMMIT)) { writer =>
303           var count = 0
304           while (writer.hasNext) {
305             val next = writer.next()
306             batch.get(next.getID).foreach { p =>
307               logger.trace(s"Persistent store modify [$topic:$partition:${p.offset}] ${p.feature}")
308               persistedKeys += next.getID
309               FeatureUtils.copyToFeature(next, p.feature, useProvidedFid = true)
310               try {
311                 writer.write()
312                 count += 1
313               } catch {
314                 case NonFatal(e) => logger.error(s"Error persisting feature: ${p.feature}", e)
315               }
316             }
317           }
318           count
319         }
320       }
321       batch.collect { case (k, v) if !persistedKeys.contains(k) => v }
322     }
323 
324     /**
325      * Attempt to persist expired features through appending writes
326      *
327      * @param partition partition being persisted
328      * @param batch expired feature
329      */
330     private def persistAppends(partition: Int, batch: Iterable[FeatureReference]): Unit = {
331       logger.debug(s"Attempting persistent appends on ${batch.size} features")
332       profile((c: Int, time: Long) => logger.debug(s"Wrote $c new feature(s) to persistent storage in ${time}ms")) {
333         WithClose(ds.getFeatureWriterAppend(sft.getTypeName, Transaction.AUTO_COMMIT)) { writer =>
334           var count = 0
335           val helper = FeatureWriterHelper(writer, useProvidedFids = true)
336           batch.foreach { p =>
337             logger.trace(s"Persistent store append [$topic:$partition:${p.offset}] ${p.feature}")
338             try {
339               helper.write(p.feature)
340               count += 1
341             } catch {
342               case NonFatal(e) => logger.error(s"Error persisting feature: ${p.feature}", e)
343             }
344           }
345           count
346         }
347       }
348     }
