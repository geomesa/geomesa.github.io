15 /**
16  * Class for gathering timing metrics on an iterator
17  *
18  * @param delegate iterator to time
19  * @param callback timer callback, will be passed nanoseconds spent running the iterator
20  * @tparam T type binding of iterator
21  */
22 class TimedIterator[T](delegate: CloseableIterator[T], callback: Long => Unit) extends CloseableIterator[T] {
23 
24   private var total = 0L
25 
26   private val callbacks = ArrayBuffer(callback)
27 
28   /**
29    * Add another callback
30    *
31    * @param callback callback to be executed when iterator is closed, will be passed nanoseconds spent running the iterator
32    * @return this iterator, for method chaining
33    */
34   def addCallback(callback: Long => Unit): TimedIterator[T] = {
35     callbacks += callback
36     this
37   }
38 
39   override def hasNext: Boolean = {
40     val start = System.nanoTime()
41     try { delegate.hasNext } finally {
42       total += (System.nanoTime() - start)
43     }
44   }
45 
46   override def next(): T = {
47     val start = System.nanoTime()
48     try { delegate.next() } finally {
49       total += (System.nanoTime() - start)
50     }
51   }
52 
53   override def close(): Unit = {
54     val start = System.nanoTime()
55     try { delegate.close() } finally {
56       total += (System.nanoTime() - start)
57       callbacks.foreach(_.apply(total))
58     }
59   }
60 }
61 
62 object TimedIterator {
63   def empty[T](): TimedIterator[T] = new TimedIterator(CloseableIterator.empty, _ => ())
