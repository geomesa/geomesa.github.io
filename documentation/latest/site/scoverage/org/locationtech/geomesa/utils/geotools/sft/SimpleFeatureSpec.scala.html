72       * @return a spec map

128   def attribute(sft: SimpleFeatureType, descriptor: AttributeDescriptor): AttributeSpec = {
129     import org.locationtech.geomesa.utils.geotools.RichAttributeDescriptors.RichAttributeDescriptor
130 
131     import scala.collection.JavaConverters._
132 
133     val name = descriptor.getLocalName
134     val binding = descriptor.getType.getBinding
135     val options = descriptor.getUserData.asScala.map { case (k, v) => k.toString -> v.toString }.toMap
136 
137     if (simpleTypeMap.contains(binding.getSimpleName)) {
138       SimpleAttributeSpec(name, binding, options)
139     } else if (geometryTypeMap.contains(binding.getSimpleName)) {
140       val opts = if (sft != null && sft.getGeometryDescriptor == descriptor) { options + (OptDefault -> "true") } else { options }
141       GeomAttributeSpec(name, binding.asInstanceOf[Class[_ <: Geometry]], opts)
142     } else if (classOf[java.util.List[_]].isAssignableFrom(binding) || binding.isArray) {
143       val itemClass = Option(descriptor.getListType()).getOrElse(classOf[String])
144       ListAttributeSpec(name, itemClass, options)
145     } else if (classOf[java.util.Map[_, _]].isAssignableFrom(binding)) {
146       val (keyBinding, valueBinding) = descriptor.getMapTypes()
147       val keyClass = Option(keyBinding).getOrElse(classOf[String])
148       val valueClass = Option(valueBinding).getOrElse(classOf[String])
149       MapAttributeSpec(name, keyClass, valueClass, options)
150     } else {
151       throw new IllegalArgumentException(s"Unknown type binding $binding")
152     }
153   }

155   /**
156    * Encode an attribute option for a spec string
157    *
158    * @param key key for the option
159    * @param value value for the option
160    * @return
161    */
162   def encodeAttributeOption(key: String, value: String): String = {
163     if (simpleOptionPattern.matcher(value).matches()) {
164       s":$key=$value"
165     } else {
166       s":$key='${StringEscapeUtils.escapeJava(value)}'"
167     }
168   }

178   case class GeomAttributeSpec(name: String, clazz: Class[_ <: Geometry], options: Map[String, String])
179       extends AttributeSpec {
180 
181     val default: Boolean = options.get(OptDefault).exists(_.toBoolean)
182 
183     override def toSpec: String = if (default) { s"*${super.toSpec}" } else { super.toSpec }
184 
185     override def builderHook(builder: AttributeTypeBuilder): Unit = {
186       require(!options.get(OptSrid).exists(_.toInt != 4326),
187         s"Invalid SRID '${options(OptSrid)}'. Only 4326 is supported.")
188       builder.crs(org.locationtech.geomesa.utils.geotools.CRS_EPSG_4326)
189     }
190 
191     // default geoms are indicated by the *
192     // we don't allow attribute indexing for geometries
193     override protected def specOptions: Map[String, String] = options - OptDefault - OptIndex
194     override protected def configOptions: Map[String, String] = options - OptIndex
195     override protected def descriptorOptions: Map[String, String] = options - OptIndex
196   }
