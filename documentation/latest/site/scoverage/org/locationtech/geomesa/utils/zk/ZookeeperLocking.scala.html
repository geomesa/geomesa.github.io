18 trait ZookeeperLocking extends DistributedLocking {
19 
20   protected def zookeepers: String
21 
22   /**
23     * Gets and acquires a distributed lock based on the key.
24     * Make sure that you 'release' the lock in a finally block.
25     *
26     * @param key key to lock on - equivalent to a path in zookeeper
27     * @return the lock
28     */
29   override protected def acquireDistributedLock(key: String): Releasable = {
30     val (client, lock) = distributedLock(key)
31     try {
32       lock.acquire()
33       ZookeeperLocking.releasable(lock, client)
34     } catch {
35       case e: Exception => CloseQuietly(client).foreach(e.addSuppressed); throw e
36     }
37   }
38 
39   /**
40     * Gets and acquires a distributed lock based on the key.
41     * Make sure that you 'release' the lock in a finally block.
42     *
43     * @param key key to lock on - equivalent to a path in zookeeper
44     * @param timeOut how long to wait to acquire the lock, in millis
45     * @return the lock, if obtained
46     */
47   override protected def acquireDistributedLock(key: String, timeOut: Long): Option[Releasable] = {
48     val (client, lock) = distributedLock(key)
49     try {
50       if (lock.acquire(timeOut, TimeUnit.MILLISECONDS)) {
51         Some(ZookeeperLocking.releasable(lock, client))
52       } else {
53         None
54       }
55     } catch {
56       case e: Exception => CloseQuietly(client).foreach(e.addSuppressed); throw e
57     }
58   }
59 
60   private def distributedLock(key: String): (CuratorFramework, InterProcessSemaphoreMutex) = {
61     val lockPath = if (key.startsWith("/")) key else s"/$key"
62     val client = CuratorHelper.client(zookeepers).build()
63     client.start()
64     val lock = new InterProcessSemaphoreMutex(client, lockPath)
65     (client, lock)
66   }
67 }
68 
69 object ZookeeperLocking {
70 
71   // delegate lock that will close the curator client upon release
72   def releasable(lock: InterProcessSemaphoreMutex, client: CuratorFramework): Releasable =
73     new Releasable { override def release(): Unit = try { lock.release() } finally { client.close() } }
74 }
