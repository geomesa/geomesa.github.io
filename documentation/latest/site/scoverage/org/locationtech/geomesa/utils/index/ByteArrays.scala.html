427   /**
428     * Returns a row that sorts just after all rows beginning with a prefix. Copied from Accumulo Range
429     *
430     * @param prefix to follow
431     * @return prefix that immediately follows the given prefix when sorted, or an empty array if no prefix can follow
432     *         (i.e., the string is all 0xff bytes)
433     */
434   def rowFollowingPrefix(prefix: Array[Byte]): Array[Byte] = {
435     // find the last byte in the array that is not 0xff
436     var changeIndex = prefix.length - 1
437     while (changeIndex >= 0 && prefix(changeIndex) == MaxByte) {
438       changeIndex -= 1
439     }
440     if (changeIndex < 0) { Array.empty } else {
441       // copy prefix bytes into new array
442       val following = Array.ofDim[Byte](changeIndex + 1)
443       System.arraycopy(prefix, 0, following, 0, changeIndex + 1)
444       // increment the selected byte
445       following(changeIndex) = (following(changeIndex) + 1).toByte
446       following
447     }
448   }
