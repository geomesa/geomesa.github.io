21 /**
22   * Builder class for creating simple feature types
23   *
24   * Adding an attribute returns an AttributeBuilder class with additional options for that attribute.
25   * There is an implicit conversion back to SchemaBuilder to allow for operation chaining.
26   *
27   * Example usage:
28   *
29   *   SchemaBuilder.builder().addString("foo").withIndex().addInt("bar").build("baz")
30   *
31   */
32 class SchemaBuilder extends AbstractSchemaBuilder[AttributeBuilder, UserDataBuilder] {
33   override protected def createAttributeBuilder(spec: StringBuilder): AttributeBuilder =
34     new AttributeBuilder(this, spec)
35   override protected def createUserDataBuilder(userData: StringBuilder): UserDataBuilder =
36     new UserDataBuilder(this, userData)
37 }

330     private def classy[T: ClassTag]: Class[_] = {
331       classTag[T].runtimeClass match {
332         case java.lang.Byte.TYPE      => classOf[java.lang.Byte]
333         case java.lang.Short.TYPE     => classOf[java.lang.Short]
334         case java.lang.Character.TYPE => classOf[java.lang.Character]
335         case java.lang.Integer.TYPE   => classOf[java.lang.Integer]
336         case java.lang.Long.TYPE      => classOf[java.lang.Long]
337         case java.lang.Float.TYPE     => classOf[java.lang.Float]
338         case java.lang.Double.TYPE    => classOf[java.lang.Double]
339         case java.lang.Boolean.TYPE   => classOf[java.lang.Boolean]
340         case java.lang.Void.TYPE      => classOf[java.lang.Void]
341         case c                        => c
342       }
343     }

349   /**
350     * Builder class for configuring per-attribute options
351     */
352   class AbstractAttributeBuilder[A <: AbstractAttributeBuilder[A]](parent: AbstractSchemaBuilder[_, _],
353                                                                    specification: StringBuilder) {
354 
355     this: A =>
356 
357     import SimpleFeatureTypes.AttributeOptions
358 
359     /**
360       * Add an index on the current attribute, to facilitate  querying on that attribute
361       *
362       * @return attribute builder for chaining calls
363       */
364     def withIndex(): A = withOption(AttributeOptions.OptIndex, "true")
365 
366     /**
367       * Add an index on the current attribute, to facilitate querying on that attribute
368       *
369       * @param cardinality a cardinality hint for the attribute - will be considered when picking an index
370       *                    during query planning
371       * @return attribute builder for chaining calls
372       */
373     def withIndex(cardinality: Cardinality): A =
374       withOptions(AttributeOptions.OptIndex -> "true", AttributeOptions.OptCardinality -> cardinality.toString)
375 
376     /**
377       * Specify column groups for a particular attribute, to speed up querying for subsets of attributes
378       *
379       * @param groups column groups - preferably short strings (one character is best), case sensitive
380       * @return
381       */
382     def withColumnGroups(groups: String*): A = withOptions(AttributeOptions.OptColumnGroups -> groups.mkString(","))
383 
384     /**
385       * Add any attribute-level option
386       *
387       * @param key option key
388       * @param value option value
389       * @return attribute builder for chaining calls
390       */
391     def withOption(key: String, value: String): A = {
392       specification.append(SimpleFeatureSpec.encodeAttributeOption(key, value)); this
393     }
394 
395     /**
396       * Add multiple attribute-level options at once
397       *
398       * @param options key/value attribute-level options
399       * @return attribute builder for chaining calls
400       */
401     def withOptions(options: (String, String)*): A = {
402       options.foreach { case (k, v) => withOption(k, v) }
403       this
404     }
405 
406     /**
407       * End the current attribute and return to the schema builder. Useful for chaining calls. Note that
408       * there is an implicit conversion back to SchemaBuilder, so this method does not normally need to be invoked
409       *
410       * @return schema builder for chaining calls
411       */
412     def end[B <: AbstractSchemaBuilder[A, _ <: AbstractUserDataBuilder[_]]](): B = parent.asInstanceOf[B]
413   }
