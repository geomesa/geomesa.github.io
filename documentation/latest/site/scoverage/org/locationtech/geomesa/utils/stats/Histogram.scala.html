22 /**
23   * The histogram's state is stored in an indexed array, where the index is the bin number
24   * and the values are the counts.
25   *
26   * e.g. a range of 0 to 3 with 3 bins will result in these bins: [0, 1), [1, 2), [2, 3) and the
27   * array will contain three entries.
28   *
29   * @param sft simple feature type
30   * @param property property name for the attribute the histogram is being made for
31   * @param initialBins number of bins the histogram has
32   * @param initialEndpoints lower/upper end of histogram
33   * @tparam T a comparable type which must have a StatHelperFunctions type class
34   */
35 class Histogram[T](
36     val sft: SimpleFeatureType,
37     val property: String,
38     initialBins: Int,
39     initialEndpoints: (T, T)
40   )(
41     implicit val defaults: MinMax.MinMaxDefaults[T],
42     val ct: ClassTag[T]
43   ) extends Stat with LazyLogging {
44 
45   override type S = Histogram[T]
46 
47   private val i = sft.indexOf(property)
48   private [stats] var bins: BinnedArray[T] = BinnedArray[T](initialBins, initialEndpoints)
49 
50   def length: Int = bins.length
51   def directIndex(value: Long): Int = bins.directIndex(value)
52   def indexOf(value: T): Int = bins.indexOf(value)
53   def count(i: Int): Long = bins.counts(i)
54   def min: T = bins.bounds._1
55   def max: T = bins.bounds._2
56   def bounds: (T, T) = bins.bounds
57   def bounds(i: Int): (T, T) = bins.bounds(i)
58   def medianValue(i: Int): T = bins.medianValue(i)

60   /**
61     * Copies another histogram into this one. In comparison to +=, this method will preserve the
62     * current bounds and length.
63     *
64     * @param other other histogram
65     */
66   def addCountsFrom(other: Histogram[T]): Unit = {
67     if (length == other.length && bounds == other.bounds) {
68       this += other
69     } else {
70       Histogram.copyInto(bins, other.bins)
71     }
72   }

74   override def observe(sf: SimpleFeature): Unit = {
75     val value = sf.getAttribute(i)
76     if (value != null) {
77       try {
78         val i = bins.indexOf(value.asInstanceOf[T])
79         if (i == -1) {
80           bins = Histogram.expandBins(value.asInstanceOf[T], bins)
81           bins.add(value.asInstanceOf[T])
82         } else {
83           bins.counts(i) += 1
84         }
85       } catch {
86         case e: Exception => logger.warn(s"Error observing value '$value': ${e.toString}")
87       }
88     }
89   }

91   override def unobserve(sf: SimpleFeature): Unit = {
92     val value = sf.getAttribute(i)
93     if (value != null) {
94       try {
95         val i = bins.indexOf(value.asInstanceOf[T])
96         if (i == -1) {
97           bins = Histogram.expandBins(value.asInstanceOf[T], bins)
98           bins.add(value.asInstanceOf[T], -1)
99         } else {
100           bins.counts(i) -= 1
101         }
102       } catch {
103         case e: Exception => logger.warn(s"Error un-observing value '$value': ${e.toString}")
104       }
105     }
106   }

108   /**
109     * Creates a new histogram by combining another histogram with this one.
110     * Bounds and length will both be the greater from each histogram.
111     */
112   override def +(other: Histogram[T]): Histogram[T] = {
113     val plus = new Histogram(sft, property, length, bounds)
114     plus += this
115     plus += other
116     plus

119   /**
120     * Copies another histogram into this one.
121     * Current bounds and length will be expanded if necessary.
122     */
123   override def +=(other: Histogram[T]): Unit = {
124     if (length == other.length && bounds == other.bounds) {
125       // hists match - we can just copy counts in directly
126       var i = 0
127       while (i < bins.length) {
128         bins.counts(i) += other.bins.counts(i)
129         i += 1
130       }
131     } else if (other.isEmpty) {
132       // no-op
133     } else if (isEmpty) {
134       // copy the data from the other histogram
135       bins = BinnedArray(other.length, other.bounds)
136       var i = 0
137       while (i < bins.length) {
138         bins.counts(i) = other.bins.counts(i)
139         i += 1
140       }
141     } else {
142       // figure out the new bounds and size
143       val newEndpoints = Histogram.checkEndpoints(bins, other.bins)
144       val newLength = math.max(length, other.length)
145       if (newEndpoints != bounds || newLength != length) {
146         // if the other hist was not 'contained' in this one, we have to re-create the bins
147         val newBins = BinnedArray(newLength, newEndpoints)
148         Histogram.copyInto(newBins, bins)
149         bins = newBins
150       }
151       // now copy over the other bins
152       Histogram.copyInto(bins, other.bins)
153     }
154   }

182   /**
183     * Takes a single value and creates a range from it. Since the range histogram lower bound has to be
184     * strictly less than the upper bound, this lets us use a single value to create a valid histogram.
185     *
186     * @param value value to buffer
187     * @tparam T type of the value
188     * @return valid bounds for a histogram
189     */
190   def buffer[T](value: T): (T, T) = {
191     import org.locationtech.geomesa.utils.stats.BinnedArray.StringBinning.{Base36Highest, Base36Lowest}
192     val buf = value match {
193       case v: Int    => (v - 100, v + 100)
194       case v: Long   => (v - 100, v + 100)
195       case v: Float  => (v - 100, v + 100)
196       case v: Double => (v - 100, v + 100)
197       case v: String => (s"$v$Base36Lowest", s"$v$Base36Highest")
198       case v: Date   => (new Date(v.getTime - 60000), new Date(v.getTime + 60000))
199       case v: Geometry =>
200         import org.locationtech.geomesa.utils.geotools.Conversions.RichGeometry
201         val env = v.safeCentroid().buffer(10.0).getEnvelopeInternal
202         val min = GeometryUtils.geoFactory.createPoint(new Coordinate(env.getMinX, env.getMinY))
203         val max = GeometryUtils.geoFactory.createPoint(new Coordinate(env.getMaxX, env.getMaxY))
204         (min, max)
205     }
206     buf.asInstanceOf[(T, T)]
207   }
208 
209   /**
210    * Checks if 2 values map to the same bin
211    *
212    * @param o1 object 1
213    * @param o2 object 2
214    * @param bins number of binds
215    * @tparam T type binding
216    * @return
217    */
218   def equivalent[T: ClassTag](o1: T, o2: T, bins: Int): Boolean =
219      Try(Binning(bins, (o1, o2))).isFailure // this will fail if the two values map to the same bin
220 
221   /**
222     * Creates a new binned array that encompasses the new value.
223     *
224     * Assumes that the value is not already within the bounds for the existing binned array.
225     */
226   def expandBins[T](value: T, old: BinnedArray[T])(implicit defaults: MinMax.MinMaxDefaults[T],ct: ClassTag[T]): BinnedArray[T] = {
227     val min = defaults.min(value, old.bounds._1)
228     val max = defaults.max(value, old.bounds._2)
229     val bins = BinnedArray[T](old.length, (min, max))
230     copyInto(bins, old)
231     bins
232   }
233 
234   /**
235     * Gets new endpoints that encompass both arrays. If either array has empty values to start or end, the
236     * bounds will be trimmed down to be the start/end of non-empty values.
237     */
238   def checkEndpoints[T](left: BinnedArray[T], right: BinnedArray[T])(implicit defaults: MinMax.MinMaxDefaults[T]): (T, T) = {
239     val (lMin, lMax) = getActualBounds(left)
240     val (rMin, rMax) = getActualBounds(right)
241     (defaults.min(lMin, rMin), defaults.max(lMax, rMax))
242   }
243 
244   /**
245     * Gets the bounds of the array that actually contain values.
246     */
247   private def getActualBounds[T](bins: BinnedArray[T]): (T, T) = {
248     val minIndex = bins.counts.indexWhere(_ != 0)
249     val maxIndex = bins.counts.length - bins.counts.reverse.indexWhere(_ != 0) - 1
250     val min = if (minIndex <= 0) bins.bounds._1 else bins.bounds(minIndex)._1
251     val max = if (maxIndex >= bins.counts.length -1) bins.bounds._2 else bins.bounds(maxIndex)._2
252     (min, max)
253   }
254 
255   /**
256     * Copies data from one binned array into the other. Arrays are assumed to have different
257     * sizes and/or endpoints. If arrays have the same characteristics, this method is
258     * needlessly expensive/complicated/inexact.
259     */
260   def copyInto[T](to: BinnedArray[T], from: BinnedArray[T]): Unit = {
261     def toIndex(value: T): Int = {
262       val i = to.indexOf(value)
263       if (i != -1) i else if (to.isBelow(value)) 0 else to.length - 1
264     }
265 
266     var i = 0
267     while (i < from.length) {
268       val count = from.counts(i)
269       if (count > 0) {
270         val (min, max) = from.bounds(i)
271         val lo = toIndex(min)
272         val hi = toIndex(max)
273         if (lo == hi) {
274           to.counts(lo) += count
275         } else {
276           val size = hi - lo + 1
277           require(size > 0,
278             s"Error calculating bounds for ${min.getClass.getSimpleName} from ${from.bounds} to ${to.bounds}")
279           val avgCount = count / size
280           val remainingCount = count % size
281           val mid = lo + (size / 2)
282           var j = lo
283           while (j <= hi) {
284             to.counts(j) += avgCount
285             j += 1
286           }
287           to.counts(mid) += remainingCount
288         }
289       }
290       i += 1
291     }
292   }
