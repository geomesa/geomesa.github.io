11 import java.util.concurrent.atomic.AtomicInteger
12 import javax.annotation.concurrent.ThreadSafe

14 /**
15   * Creates unique file names generated from a base name, by appending a sequence number
16   * before the file extension.
17   *
18   * For example, given the file name 'foo.txt', will return 'foo_000.txt', 'foo_001.txt', etc.
19   * If the iterator exceeds the specified number of digits, it will start to append additional
20   * digits to ensure uniqueness, e.g. 'foo_999.txt', 'foo_1000.txt', 'foo_1001.txt', etc.
21   *
22   * @param path file name path
23   * @param digits number of digits used to format the sequence number
24   */
25 @ThreadSafe
26 class IncrementingFileName(path: String, digits: Int = 3) extends Iterator[String] {
27 
28   private val i = new AtomicInteger(0)
29   private val format = s"_%0${digits}d"
30   private val (prefix, suffix) = PathUtils.getBaseNameAndExtension(path)
31 
32   override def hasNext: Boolean = true
33 
34   override def next(): String = s"$prefix${format.format(i.getAndIncrement())}$suffix"
35 }
