36   override def interpretPath(path: String): Seq[FileHandle] = {
37     val firstWildcard = path.indexOf('*')
38     if (firstWildcard == -1) {
39       val file = new File(path)
40       if (file.isDirectory) {
41         logger.warn(s"Input file is a directory: ${file.getAbsolutePath}")
42         Seq.empty
43       } else {
44         Seq(createHandle(file))
45       }
46     } else {
47       // find the base directory to search from based on the non-wildcard prefix
48       val lastSep = path.length - 1 - path.reverse.indexOf('/', path.length - firstWildcard - 1)
49       val (basepath, glob) = if (lastSep == path.length) {
50         (new File(".").toPath, path)
51       } else {
52         (new File(path.substring(0, lastSep)).toPath, path.substring(lastSep + 1))
53       }
54       if (glob.indexOf('/') == -1 && !glob.contains("**")) {
55         // we can just look in the current directory
56         WithClose(Files.newDirectoryStream(basepath, glob)) { stream =>
57           stream.asScala.toList.flatMap{ p =>
58             val file = p.toFile
59             if (file.isDirectory) { Nil } else { List(createHandle(file)) }
60           }
61         }
62       } else {
63         // we have to walk the file tree
64         val matcher = FileSystems.getDefault.getPathMatcher("glob:" + glob)
65         val result = ArrayBuffer.empty[FileHandle]
66         val visitor = new SimpleFileVisitor[Path] {
67           override def visitFile(file: Path, attributes: BasicFileAttributes): FileVisitResult = {
68             if (matcher.matches(file) && attributes.isRegularFile && !attributes.isDirectory) {
69               result += createHandle(file.toFile)
70             }
71             FileVisitResult.CONTINUE
72           }
73         }
74         Files.walkFileTree(basepath, visitor)
75         result.toSeq
76       }
77     }
78   }

82   private def createHandle(file: File): FileHandle = {
83     PathUtils.getUncompressedExtension(file.getName).toLowerCase(Locale.US) match {
84       case TAR       => new LocalTarHandle(file)
85       case ZIP | JAR => new LocalZipHandle(file)
86       case _         => new LocalFileHandle(file)
87     }
88   }

95   class LocalFileHandle(file: File) extends FileHandle {
96 
97     override def path: String = file.getAbsolutePath
98 
99     override def exists: Boolean = file.exists()
100 
101     override def length: Long = file.length()
102 
103     override def open: CloseableIterator[(Option[String], InputStream)] = {
104       val is = PathUtils.handleCompression(new FileInputStream(file), file.getName)
105       CloseableIterator.single(None -> is, is.close())
106     }
107 
108     override def write(mode: CreateMode): OutputStream = {
109       mode.validate()
110       if (file.exists()) {
111         if (mode.append) {
112           new FileOutputStream(file, true)
113         } else if (mode.overwrite) {
114           new FileOutputStream(file, false)
115         } else {
116           // due to validation, must be mode.create
117           throw new FileAlreadyExistsException(s"File already exists for mode 'create': $path")
118         }
119       } else if (!mode.create) {
120         throw new FileNotFoundException(s"File does not exist: $path")
121       } else {
122         val parent = file.getParentFile
123         if (parent != null && !parent.exists() && !parent.mkdirs()) {
124           throw new IOException(s"Parent file does not exist and could not be created: $path")
125         }
126         new FileOutputStream(file)
127       }
128     }
129 
130     override def delete(recursive: Boolean): Unit = {
131       if (recursive) {
132         PathUtils.deleteRecursively(file.toPath)
133       } else if (!file.delete()) {
134         throw new IOException(s"Could not delete file: $path")
135       }
136     }
137   }

178   // A class that caches any bytes read from the stdin input stream
179   private class CachingStdInHandle(in: InputStream) extends StdInHandle(in) {
180     private val is = new CopyingInputStream(in, 1024)
181     private var cachedBytes: Array[Byte] = Array.empty
182 
183     override def length: Long = super.length + cachedBytes.length
184 
185     override def open: CloseableIterator[(Option[String], InputStream)] =
186       CloseableIterator.single(None -> new CopyOnClose())
187 
188     private class CopyOnClose
189         extends SequenceInputStream(new ByteArrayInputStream(cachedBytes), CloseShieldInputStream.wrap(is)) {
190       override def close(): Unit = {
191         try { super.close() } finally {
192           if (is.copied > 0) {
193             cachedBytes ++= is.replay(is.copied)
194           }
195         }
196       }
197     }
198   }
