33   /**
34     * Encode a sequence of string values to a single string
35     *
36     * @param values values
37     * @return
38     */
39   def encodeSeq(values: Seq[String]): String = {
40     if (values.isEmpty) { "" } else {
41       val sb = new java.lang.StringBuilder()
42       val printer = new CSVPrinter(sb, CSVFormat.DEFAULT)
43       values.foreach(printer.print)
44       sb.toString
45     }
46   }

50     *
51     * @param values encoded string
52     * @return
53     */
54   def decodeSeq(values: String): Seq[String] = {
55     if (values.isEmpty) { Seq.empty } else {
56       WithClose(CSVParser.parse(values, CSVFormat.DEFAULT))(_.iterator.next.iterator.asScala.toList)
57     }
58   }

78   /**
79     * Encode a map of sequences as a string
80     *
81     * @param map map of keys to sequences of values
82     * @return
83     */
84   def encodeSeqMap(map: Map[String, Seq[AnyRef]]): String = {
85     val sb = new java.lang.StringBuilder
86     val printer = new CSVPrinter(sb, CSVFormat.DEFAULT)
87     map.foreach { case (k, v) =>
88       val strings = v.headOption match {
89         case Some(_: Date) => v.map(d => ZonedDateTime.ofInstant(toInstant(d.asInstanceOf[Date]), ZoneOffset.UTC).format(dateFormat))
90         case _ => v
91       }
92       printer.print(k)
93       strings.foreach(printer.print)
94       printer.println()
95     }
96     sb.toString
97   }

111   /**
112     * Decode a map of sequences from a string encoded by @see encodeSeqMap
113     *
114     * @param encoded encoded map
115     * @return decoded map
116     */
117   def decodeSeqMap(encoded: String, bindings: Map[String, Class[_]]): Map[String, Array[AnyRef]] = {
118     // encoded as CSV, first element of each row is key, rest is value
119     WithClose(CSVParser.parse(encoded, CSVFormat.DEFAULT)) { parser =>
120       parser.iterator.asScala.map { record =>
121         val iter = record.iterator.asScala
122         val key = iter.next
123         val values = bindings.get(key) match {
124           case Some(c) if c == classOf[String]              => iter.toArray[AnyRef]
125           case Some(c) if c == classOf[Integer]             => iter.map(Integer.valueOf).toArray[AnyRef]
126           case Some(c) if c == classOf[java.lang.Long]      => iter.map(java.lang.Long.valueOf).toArray[AnyRef]
127           case Some(c) if c == classOf[java.lang.Float]     => iter.map(java.lang.Float.valueOf).toArray[AnyRef]
128           case Some(c) if c == classOf[java.lang.Double]    => iter.map(java.lang.Double.valueOf).toArray[AnyRef]
129           case Some(c) if classOf[Date].isAssignableFrom(c) => iter.map(v => Date.from(ZonedDateTime.parse(v, dateFormat).toInstant)).toArray[AnyRef]
130           case Some(c) if c == classOf[java.lang.Boolean]   => iter.map(java.lang.Boolean.valueOf).toArray[AnyRef]
131           case c => logger.warn(s"No conversion defined for encoded attribute '$key' of type ${c.orNull}"); iter.toArray[AnyRef]
132         }
133         key -> values
134       }.toMap
135     }
136   }

138   /**
139     * Encode non-alphanumeric characters in a string with
140     * underscore plus hex digits representing the bytes. Note
141     * that multibyte characters will be represented with multiple
142     * underscores and bytes...e.g. _8a_2f_3b
143     */
144   def alphaNumericSafeString(input: String): String = {
145     if (AlphaNumericPattern.matcher(input).matches()) { input } else {
146       val sb = new StringBuilder
147       input.foreach { c =>
148         if (AlphaNumeric.contains(c)) { sb.append(c) } else {
149           val hex = Hex.encodeHex(c.toString.getBytes(StandardCharsets.UTF_8))
150           val encoded = hex.grouped(2).map(arr => "_" + arr(0) + arr(1)).mkString.toLowerCase(Locale.US)
151           sb.append(encoded)
152         }
153       }
154       sb.toString()
155     }
156   }
157 
158   def decodeAlphaNumericSafeString(input: String): String = {
159     if (AlphaNumericPattern.matcher(input).matches()) { input } else {
160       val sb = new StringBuilder
161       var i = 0
162       while (i < input.length) {
163         val c = input.charAt(i)
164         if (c != '_') { sb.append(c) } else {
165           i += 2
166           sb.append(new String(Hex.decodeHex(Array(input.charAt(i - 1), input.charAt(i))), StandardCharsets.UTF_8))
167         }
168         i += 1
169       }
170       sb.toString()
171     }
172   }
