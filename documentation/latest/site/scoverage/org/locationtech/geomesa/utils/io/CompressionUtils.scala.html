17 /**
18   * Common trait for Apache compression utilities. Each compression type (gzip, xz, bzip) has the same
19   * static method signatures, but they do not implement a common interface. This trait unifies them, so
20   * that code can just deal with a CompressionUtils instance and not have to explicitly deal with each
21   * compression type
22   */
23 trait CompressionUtils {
24   def isCompressedFilename(filename: String): Boolean
25   def getUncompressedFilename(filename: String): String
26   def getCompressedFilename(filename: String): String
27   def compress(is: InputStream): InputStream
28 }
29 
30 object CompressionUtils {
31 
32   val Utils: Seq[CompressionUtils] = Seq(GzipUtils, XZUtils, BZip2Utils)
33 
34   /**
35     * Gets the uncompressed file name. E.g. 'foo.tgz -> foo.tar', 'foo.tar.gz -> foo.tar', etc
36     *
37     * @param filename file name
38     * @return
39     */
40   def getUncompressedFilename(filename: String): String = {
41     val uncompressed = Utils.collectFirst {
42       case u if u.isCompressedFilename(filename) => u.getUncompressedFilename(filename)
43     }
44     uncompressed.getOrElse(filename)
45   }
46 
47   /**
48     * Does the filename indicate compression? E.g. 'foo.tgz', 'foo.tar.gz', etc
49     *
50     * @param filename file name
51     * @return
52     */
53   def isCompressedFilename(filename: String): Boolean = Utils.exists(_.isCompressedFilename(filename))
