37 /**
38  * Class to hold timing results
39  */
40 class Timing extends Serializable {
41 
42   private var total = 0L
43   private var count = 0L
44 
45   /**
46    * Updates this instance with a new timing
47    *
48    * @param time time in millis
49    * @return
50    */
51   def occurrence(time: Long): Unit = {
52     total += time
53     count += 1
54   }
55 
56   /**
57    * Gets the total time
58    *
59    * @return
60    */
61   def time: Long = total
62 
63   /**
64    * Gets the number of event occurrences
65    *
66    * @return
67    */
68   def occurrences: Long = count
69 
70   /**
71    * Computes the average for this instance
72    *
73    * @return
74    */
75   def average(): Double = total / count.toDouble
76 }

119 /**
120  * Class to hold timing results. Not thread-safe.
121  */
122 class TimingsImpl extends Timings {
123 
124   private val map = scala.collection.mutable.Map.empty[String, Timing]
125 
126   override def occurrence(identifier: String, time: Long): Unit =
127     map.getOrElseUpdate(identifier, new Timing).occurrence(time)
128 
129   override def time(identifier: String): Long = map.getOrElseUpdate(identifier, new Timing).time
130 
131   override def occurrences(identifier: String): Long = map.getOrElseUpdate(identifier, new Timing).occurrences
132 
133   override def averageOccurrences(): String = if (map.isEmpty) {
134     "No occurrences"
135   } else {
136     val entries = map.toList.sortBy(_._1)
137     val total = entries.map(_._2.occurrences).sum
138     val percentOccurrences = entries.map { case (id, timing) =>
139       s"$id: ${(timing.occurrences * 100 / total.toDouble).formatted("%.1f%%")}"
140     }
141     percentOccurrences.mkString(s"Total occurrences: $total. Percent of occurrences - ", ", ", "")
142   }
143 
144   override def averageTimes(): String = if (map.isEmpty) {
145     "No occurrences"
146   } else {
147     val entries = map.toList.sortBy(_._1)
148     val total = entries.map(_._2.time).sum
149     val percentTimes = entries.map { case (id, timing) =>
150       s"$id: ${(timing.time * 100 / total.toDouble).formatted("%.1f%%")}" +
151           s" ${timing.occurrences} times at ${timing.average().formatted("%.4f")} ms avg"
152     }
153     percentTimes.mkString(s"Total time: $total ms. Percent of time - ", ", ", "")
154   }
155 }

202  * @param moduloToLog how many events to skip between logging
203  */
204 class AutoLoggingTimings(moduloToLog: Int = 1000) extends ThreadSafeTimingsImpl with LazyLogging {
205 
206   val count = new AtomicLong()
207 
208   override def occurrence(identifier: String, time: Long): Unit = {
209     super.occurrence(identifier, time)
210     if (count.incrementAndGet() % moduloToLog == 0) {
211       logger.debug(averageTimes())
212     }
213   }
