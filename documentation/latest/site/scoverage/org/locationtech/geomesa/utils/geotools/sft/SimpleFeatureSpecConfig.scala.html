35   /**
36     * Parse a SimpleFeatureType spec from a typesafe Config
37     *
38     * @param conf config
39     * @param path instead of parsing the root config, parse the nested config at this path
40     * @return
41     */
42   def parse(conf: Config, path: Option[String]): (Option[String], SimpleFeatureSpec) = {
43 
44     val toParse = path.filter(conf.hasPath) match {
45       case Some(p) => conf.withFallback(conf.getConfig(p))
46       case None    => conf
47     }
48     parse(toParse)
49   }

51   /**
52     * Convert a simple feature type to a typesafe config
53     *
54     * @param sft simple feature type
55     * @param includeUserData include user data
56     * @param includePrefix include the geomesa.sfts.XXX prefix
57     * @return
58     */
59   def toConfig(sft: SimpleFeatureType, includeUserData: Boolean, includePrefix: Boolean): Config = {
60     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
61 
62     // Update "default" options (dtg and geom)
63     val defaults = sft.getDtgField.toSeq ++ Option(sft.getGeomField)
64     val attributes = sft.getAttributeDescriptors.asScala.map { ad =>
65       val config = SimpleFeatureSpec.attribute(sft, ad).toConfigMap
66       if (defaults.contains(ad.getLocalName)) {
67         config.updated("default", "true").asJava
68       } else {
69         config.asJava
70       }
71     }
72 
73     val base = ConfigFactory.empty()
74       .withValue(TypeNamePath, ConfigValueFactory.fromAnyRef(sft.getTypeName))
75       .withValue(AttributesPath, ConfigValueFactory.fromIterable(attributes.asJava))
76 
77     val updated = if (includeUserData) {
78       val prefixes = sft.getUserDataPrefixes
79       // special handling for keywords delimiter
80       val keywords = Map(Keywords -> sft.getKeywords.asJava).filterNot(_._2.isEmpty)
81       val toConvert = keywords ++ sft.getUserData.asScala.collect {
82         case (k, v) if v != null && prefixes.exists(k.toString.startsWith) && k != Keywords => (k.toString, v)
83       }
84       val userData = ConfigValueFactory.fromMap(toConvert.asJava)
85       base.withValue(UserDataPath, userData)
86     } else {
87       base
88     }
89 
90     if (includePrefix) {
91       updated.atPath(s"${ConfigSftParsing.path}.${sft.getTypeName}")
92     } else {
93       updated
94     }
95   }
96 
97   /**
98     * Convert a simple feature type to a typesafe config and renders it as a string
99     *
100     * @param sft simple feature type
101     * @param includeUserData include user data
102     * @param concise concise or verbose string
103     * @return
104     */
105   def toConfigString(sft: SimpleFeatureType,
106                      includeUserData: Boolean,
107                      concise: Boolean,
108                      includePrefix: Boolean,
109                      json: Boolean): String = {
110     val opts = if (concise) {
111       ConfigRenderOptions.concise.setJson(json)
112     } else {
113       ConfigRenderOptions.defaults().setFormatted(true).setComments(false).setOriginComments(false).setJson(json)
114     }
115     toConfig(sft, includeUserData, includePrefix).root().render(opts)
116   }

118   private def parse(conf: Config): (Option[String], SimpleFeatureSpec) = {
119     import org.locationtech.geomesa.utils.conf.ConfConversions.RichConfig
120 
121     val name = if (conf.hasPath(TypeNamePath)) { Option(conf.getString(TypeNamePath)) } else { None }
122     val attributes = {
123       val fields = if (conf.hasPath("fields")) { conf.getConfigList("fields") } else { conf.getConfigList(AttributesPath) }
124       fields.asScala.map(buildField).toSeq
125     }
126     val opts = {
127       val userDataConfig = if (conf.hasPath(UserDataPath)) { conf.getConfig(UserDataPath) } else { ConfigFactory.empty }
128       val base = userDataConfig.toStringMap()
129       if (!base.contains(Keywords)) { base } else {
130         // special case to handle keywords
131         base ++ userDataConfig.withOnlyPath(Keywords).toStringMap(KeywordsDelimiter)
132       }
133     }
134 
135     (name, SimpleFeatureSpec(attributes, opts))
136   }
137 
138   private def buildField(conf: Config): AttributeSpec = {
139     import org.locationtech.geomesa.utils.conf.ConfConversions.RichConfig
140 
141     val attribute = SimpleFeatureSpecParser.parseAttribute(s"${conf.getString(NamePath)}:${conf.getString(TypePath)}")
142     val options = conf.toStringMap().filterNot { case (k, _) => NonOptions.contains(k) }
143 
144     attribute match {
145       case s: SimpleAttributeSpec => s.copy(options = options)
146       case s: GeomAttributeSpec   => s.copy(options = options)
147       case s: ListAttributeSpec   => s.copy(options = options)
148       case s: MapAttributeSpec    => s.copy(options = options)
149     }
150   }
