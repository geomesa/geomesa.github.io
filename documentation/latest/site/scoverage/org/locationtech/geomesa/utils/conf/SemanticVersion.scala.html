14 case class SemanticVersion(major: Int,
15                            minor: Int,
16                            patch: Int,
17                            preRelease: Option[String] = None,
18                            build: Option[String] = None) extends Ordered[SemanticVersion] {
19 
20   import SemanticVersion.comparePreReleases
21 
22   // see https://semver.org/#spec-item-11
23   override def compare(that: SemanticVersion): Int = {
24     if (this.major > that.major) { 1 } else if (this.major < that.major) { -1 } else
25     if (this.minor > that.minor) { 1 } else if (this.minor < that.minor) { -1 } else
26     if (this.patch > that.patch) { 1 } else if (this.patch < that.patch) { -1 } else
27     { comparePreReleases(this.preRelease, that.preRelease) }
28     // note: 'build' is not used for comparison
29   }
30 
31   override def toString: String =
32     s"$major.$minor.$patch${preRelease.map(v => s"-$v").getOrElse("")}${build.map(v => s"+$v").getOrElse("")}"
33 }

35 object SemanticVersion {
36 
37   private val Regex = """(\d+)\.(\d+)\.(\d+)(-([0-9A-Za-z-.]+))?(\+([0-9A-Za-z-.]+))?""".r
38 
39   private val LenientRegex = """(\d+)\.?""".r
40 
41   def apply(version: String): SemanticVersion = {
42     val Regex(major, minor, patch, _, preRelease, _, build) = version
43     SemanticVersion(major.toInt, minor.toInt, patch.toInt, Option(preRelease), Option(build))
44   }
45 
46   /**
47     * Can be used to parse '.' delimited numbers that don't exactly match SemVer, e.g. 1.3.5.1-SNAPSHOT.
48     * The first 3 digit groups found will be used for major, minor, and patch version, respectively.
49     * Everything else is ignored.
50     */
51   def apply(version: String, lenient: Boolean): SemanticVersion = {
52     try { apply(version) } catch {
53       case NonFatal(e) if lenient =>
54         val versions = LenientRegex.findAllMatchIn(version).flatMap(m => Try(m.group(1).toInt).toOption.iterator)
55         if (versions.hasNext) {
56           val Seq(major, minor, patch) = (versions ++ Iterator(0, 0)).take(3).toSeq
57           SemanticVersion(major, minor, patch)
58         } else {
59           throw e
60         }
61     }
62   }
63 
64   /**
65     * Comparison based only on the major version number
66     */
67   object MajorOrdering extends Ordering[SemanticVersion] {
68     override def compare(x: SemanticVersion, y: SemanticVersion): Int = Integer.compare(x.major, y.major)
69   }
70 
71   /**
72     * Comparison based only on the major and minor version numbers
73     */
74   object MinorOrdering extends Ordering[SemanticVersion] {
75     override def compare(x: SemanticVersion, y: SemanticVersion): Int = {
76       if (x.major > y.major) { 1 } else if (x.major < y.major) { -1 } else
77       if (x.minor > y.minor) { 1 } else if (x.minor < y.minor) { -1 } else
78       { 0 }
79     }
80   }
81 
82   // see https://semver.org/#spec-item-11
83   private def comparePreReleases(first: Option[String], second: Option[String]): Int = {
84     if (first.isEmpty && second.isEmpty) { 0 }
85     else if (first.isEmpty && second.isDefined) { 1 }
86     else if (first.isDefined && second.isEmpty) { -1 }
87     else {
88       val firstParsed = first.get.split('.').map(parsePreRelease)
89       val secondParsed = second.get.split('.').map(parsePreRelease)
90       // compare left to right until a difference is found
91       val compare = firstParsed.zip(secondParsed).map { case (firstInt, secondInt) =>
92         firstInt match {
93           case Left(f) =>
94             secondInt match {
95               case Left(s)  => f.compareTo(s)
96               case Right(_) => 1 // numeric values always have lower precedence than non-numeric ones
97             }
98           case Right(f) =>
99             secondInt match {
100               case Left(_)  => -1 // numeric values always have lower precedence than non-numeric ones
101               case Right(s) => if (f > s) { 1 } else if (f < s) { -1 } else { 0 }
102             }
103         }
104       }
105       // if no difference, the longer version is higher precedence
106       compare.find(_ != 0).getOrElse(firstParsed.lengthCompare(secondParsed.length))
107     }
108   }
109 
110   // pre-release values can either be numbers or non-numbers, and are treated differently during compare
111   private def parsePreRelease(version: String): Either[String, Int] =
112     try { Right(version.toInt) } catch { case NonFatal(_) => Left(version) }
113 }
