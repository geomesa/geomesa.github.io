24   * Replacement for geotools `Converters`, which caches the converters for each pair of (from -> to) instead of
25   * re-creating them each time.

33   private val cache: LoadingCache[(Class[_], Class[_]), Array[Converter]] =

53   /**
54     * Convert the value into the given type
55     *
56     * @param value value to convert
57     * @param binding type to convert to
58     * @tparam T type binding
59     * @return converted value, or null if it could not be converted
60     */
61   def convert[T](value: Any, binding: Class[T]): T = {
62     if (value == null || binding.isAssignableFrom(value.getClass)) {
63       return value.asInstanceOf[T]
64     }
65 
66     val converters = cache.get((value.getClass, binding))
67 
68     val result = tryConvert(value, binding, converters)
69     if (result == null) {
70       val msg =
71         s"Could not convert '$value' (of type ${value.getClass.getName}) to ${binding.getName} " +
72           s"using ${converters.map(_.getClass.getName).mkString(", ")}"
73       logger.warn(msg)
74       logger.debug(msg, new RuntimeException())
75     }
76     result
77   }
78 
79   /**
80    * Convert the value into one of the given type
81    *
82    * @param value value to convert
83    * @param bindings type to convert to, in order of preference
84    * @tparam T type binding
85    * @return converted value, or null if it could not be converted
86    */
87   def convertFirst[T](value: Any, bindings: Iterator[Class[_ <: T]]): T = {
88     if (value == null) {
89       return null.asInstanceOf[T]
90     }
91 
92     val clas = value.getClass
93     val errors = ArrayBuffer.empty[(Class[_], Array[Converter])]
94 
95     while (bindings.hasNext) {
96       val binding = bindings.next
97       if (binding.isAssignableFrom(clas)) {
98         return value.asInstanceOf[T]
99       }
100 
101       val converters = cache.get((clas, binding))
102 
103       val result = tryConvert(value, binding, converters)
104       if (result != null) {
105         return result
106       } else {
107         errors += binding -> converters
108       }
109     }
110 
111     val msg =
112       s"Could not convert '$value' (of type ${value.getClass.getName}) to any of:" +
113         errors.map { case (b, c) => s"${b.getClass.getName} using ${c.map(_.getClass.getName).mkString(", ")}"}.mkString("\n  ", "\n  ", "")
114 
115     logger.warn(msg)
116     logger.debug(msg, new RuntimeException())
117 
118     null.asInstanceOf[T]
119   }

121   /**
122     * Convert the value into the given type, returning the default if it could not be converted or is null
123     *
124     * @param value value to convert
125     * @param default value to return if convert results in null
126     * @param ct class tag
127     * @tparam T type to convert to
128     * @return
129     */
130   def convertOrElse[T <: AnyRef](value: Any, default: => T)(implicit ct: ClassTag[T]): T = {
131     val attempt = convert(value, ct.runtimeClass.asInstanceOf[Class[T]])
132     if (attempt == null) { default } else { attempt }
133   }

135   /**
136     * Evaluate and convert an expression
137     *
138     * @param expression expression to evaluate
139     * @param binding type to convert to
140     * @tparam T type binding
141     * @return converted value, or null if it could not be converted
142     */
143   def evaluate[T](expression: Expression, binding: Class[T]): T = convert(expression.evaluate(null), binding)
144 
145   /**
146    * Try to convert the value
147    *
148    * @param value value
149    * @param binding expected return type
150    * @param converters converters to use
151    * @tparam T expected return type
152    * @return the converted value as type T, or null if could not convert
153    */
154   private def tryConvert[T](value: Any, binding: Class[T], converters: Array[Converter]): T = {
155     var i = 0
156     while (i < converters.length) {
157       try {
158         val result = converters(i).convert(value, binding)
159         if (result != null) {
160           return result
161         }
162       } catch {
163         case NonFatal(e) =>
164           logger.trace(s"Error converting $value (of type ${value.getClass.getName}) " +
165             s"to ${binding.getName} using converter ${converters(i).getClass.getName}:", e)
166       }
167       i += 1
168     }
169     null.asInstanceOf[T]
170   }
171 
172   private object ToStringConverter extends Converter {
173     override def convert[T](source: Any, target: Class[T]): T = source.toString.asInstanceOf[T]
174   }
