23 /**
24   * Replacement for geotools `Converters`, which caches the converters for each pair of (from -> to) instead of
25   * re-creating them each time.
26   */

36   private val cache: LoadingCache[(Class[_], Class[_]), Array[Converter]] =
37     Caffeine.newBuilder().expireAfterAccess(ConverterCacheExpiry.toDuration.get.toMillis, TimeUnit.MILLISECONDS).build(
38       new CacheLoader[(Class[_], Class[_]), Array[Converter]]() {
39         override def load(key: (Class[_], Class[_])): Array[Converter] = {
40           val (from, to) = key
41           val converters = factories.flatMap(factory => Option(factory.createConverter(from, to, null)))
42           logger.debug(
43             s"Found ${converters.length} converters for ${from.getName}->${to.getName}: " +
44               s"${converters.map(_.getClass.getName).mkString(", ")}")
45           if (to == classOf[String]) {
46             converters :+ ToStringConverter // add toString as a final fallback
47           } else {
48             converters
49           }
50         }
51       }
52     )

54   /**
55     * Convert the value into the given type
56     *
57     * @param value value to convert
58     * @param binding type to convert to
59     * @tparam T type binding
60     * @return converted value, or null if it could not be converted
61     */
62   def convert[T](value: Any, binding: Class[T]): T = {
63     if (value == null || binding.isAssignableFrom(value.getClass)) {
64       return value.asInstanceOf[T]
65     }
66 
67     val converters = cache.get((value.getClass, binding))
68 
69     val result = tryConvert(value, binding, converters)
70     if (result == null) {
71       val msg =
72         s"Could not convert '$value' (of type ${value.getClass.getName}) to ${binding.getName} " +
73           s"using ${converters.map(_.getClass.getName).mkString(", ")}"
74       logger.warn(msg)
75       logger.debug(msg, new RuntimeException())
76     }
77     result
78   }
79 
80   /**
81    * Convert the value into one of the given type
82    *
83    * @param value value to convert
84    * @param bindings type to convert to, in order of preference
85    * @tparam T type binding
86    * @return converted value, or null if it could not be converted
87    */
88   def convertFirst[T](value: Any, bindings: Iterator[Class[_ <: T]]): T = {
89     if (value == null) {
90       return null.asInstanceOf[T]
91     }
92 
93     val clas = value.getClass
94     val errors = ArrayBuffer.empty[(Class[_], Array[Converter])]
95 
96     while (bindings.hasNext) {
97       val binding = bindings.next
98       if (binding.isAssignableFrom(clas)) {
99         return value.asInstanceOf[T]
100       }
101 
102       val converters = cache.get((clas, binding))
103 
104       val result = tryConvert(value, binding, converters)
105       if (result != null) {
106         return result
107       } else {
108         errors += binding -> converters
109       }
110     }
111 
112     val msg =
113       s"Could not convert '$value' (of type ${value.getClass.getName}) to any of:" +
114         errors.map { case (b, c) => s"${b.getClass.getName} using ${c.map(_.getClass.getName).mkString(", ")}"}.mkString("\n  ", "\n  ", "")
115 
116     logger.warn(msg)
117     logger.debug(msg, new RuntimeException())
118 
119     null.asInstanceOf[T]
120   }

122   /**
123     * Convert the value into the given type, returning the default if it could not be converted or is null
124     *
125     * @param value value to convert
126     * @param default value to return if convert results in null
127     * @param ct class tag
128     * @tparam T type to convert to
129     * @return
130     */
131   def convertOrElse[T <: AnyRef](value: Any, default: => T)(implicit ct: ClassTag[T]): T = {
132     val attempt = convert(value, ct.runtimeClass.asInstanceOf[Class[T]])
133     if (attempt == null) { default } else { attempt }
134   }

136   /**
137     * Evaluate and convert an expression
138     *
139     * @param expression expression to evaluate
140     * @param binding type to convert to
141     * @tparam T type binding
142     * @return converted value, or null if it could not be converted
143     */
144   def evaluate[T](expression: Expression, binding: Class[T]): T = convert(expression.evaluate(null), binding)
145 
146   /**
147    * Try to convert the value
148    *
149    * @param value value
150    * @param binding expected return type
151    * @param converters converters to use
152    * @tparam T expected return type
153    * @return the converted value as type T, or null if could not convert
154    */
155   private def tryConvert[T](value: Any, binding: Class[T], converters: Array[Converter]): T = {
156     var i = 0
157     while (i < converters.length) {
158       try {
159         val result = converters(i).convert(value, binding)
160         if (result != null) {
161           return result
162         }
163       } catch {
164         case NonFatal(e) =>
165           logger.trace(s"Error converting $value (of type ${value.getClass.getName}) " +
166             s"to ${binding.getName} using converter ${converters(i).getClass.getName}:", e)
167       }
168       i += 1
169     }
170     null.asInstanceOf[T]
171   }
172 
173   private object ToStringConverter extends Converter {
174     override def convert[T](source: Any, target: Class[T]): T = source.toString.asInstanceOf[T]
175   }
