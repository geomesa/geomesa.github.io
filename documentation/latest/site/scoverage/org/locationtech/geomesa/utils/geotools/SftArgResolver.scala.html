22 /**
23  * Resolves SimpleFeatureType specification from a variety of arguments
24  * including sft strings (e.g. name:String,age:Integer,*geom:Point)
25  * and typesafe config.
26  */
27 object SftArgResolver extends ArgResolver[SimpleFeatureType, SftArgs] with LazyLogging {
28 
29   import org.locationtech.geomesa.utils.conf.ArgResolver.ArgTypes._
30 
31   private val confStrings = Seq("geomesa{", "geomesa {", "geomesa.sfts")
32   private val specStrReg = """^[a-zA-Z0-9]+[:][String|Integer|Double|Point|Date|Map|List].*""" // e.g. "foo:String..."
33   private val fileNameReg = """([^.]*)\.([^.]*)""" // e.g. "foo.bar"
34   private val specStrRegError = """^[a-zA-Z0-9]+[:][a-zA-Z0-9]+.*""" // e.g. "foo:Sbartring..."
35 
36   override def argType(args: SftArgs): ArgTypes = {
37     // guess the type we are trying to parse, to determine which error we show for failures
38     // order is important here
39     if (confStrings.exists(args.spec.contains)) {
40       CONFSTR
41     } else if (args.spec.matches(specStrReg)) {
42       SPECSTR
43     } else if (args.spec.matches(fileNameReg) || args.spec.contains("/")) {
44       PATH
45     } else if (args.spec.matches(specStrRegError)) {
46       SPECSTR
47     } else {
48       NAME
49     }
50   }
51 
52   override val parseMethodList: Seq[SftArgs => ResEither] = Seq[SftArgs => ResEither](
53     getLoadedSft,
54     parseSpecString,
55     parseConfStr,
56     parseSpecStringFile,
57     parseConfFile
58   )
59 
60   // gets an sft from simple feature type providers on the classpath
61   private [SftArgResolver] def getLoadedSft(args: SftArgs): ResEither = {
62     SimpleFeatureTypeLoader.sfts.find(_.getTypeName == args.spec).map { sft =>
63       if (args.featureName == null || args.featureName == sft.getTypeName) { sft } else {
64         SimpleFeatureTypes.renameSft(sft, args.featureName)
65       }
66     } match {
67       case Some(sft) => Right(sft)
68       case None =>
69         val error =
70           new RuntimeException(s"Unable to find SFT with name: ${args.spec}\n  " +
71               s"Available types: ${SimpleFeatureTypeLoader.sfts.map(_.getTypeName).mkString(", ")}")
72         Left((s"Unable to get loaded SFT using ${args.spec}", error, NAME))
73     }
74   }
75 
76   // gets an sft based on a spec string
77   private [SftArgResolver] def parseSpecString(args: SftArgs): ResEither = {
78     try {
79       val name = Option(args.featureName).getOrElse {
80         throw new IllegalArgumentException("Feature name was not provided")
81       }
82       Right(SimpleFeatureTypes.createType(name, args.spec))
83     } catch {
84       case NonFatal(e) => Left((s"Unable to parse sft spec from: ${args.spec}", e, SPECSTR))
85     }
86   }
87 
88   // gets an sft based on a spec string
89   private [SftArgResolver] def parseSpecStringFile(args: SftArgs): ResEither = {
90     try {
91       val name = Option(args.featureName).getOrElse {
92         throw new IllegalArgumentException("Feature name was not provided")
93       }
94       val file = Option(args.spec).getOrElse(throw new IllegalArgumentException("No input file specified"))
95       val spec = FileUtils.readFileToString(new File(file), StandardCharsets.UTF_8)
96       Right(SimpleFeatureTypes.createType(name, spec))
97     } catch {
98       case NonFatal(e) => Left((s"Unable to parse sft spec from file: ${args.spec}", e, PATH))
99     }
100   }
101 
102   private [SftArgResolver] def parseConf(input: Reader, name: String): Either[Throwable, SimpleFeatureType] = {
103     try {
104       val sfts = ConfigSftParsing.parseConf(ConfigFactory.parseReader(input, parseOpts).resolve())
105       if (sfts.isEmpty) {
106         throw new RuntimeException("No feature types parsed from config string")
107       }
108       if (name == null) {
109         if (sfts.lengthCompare(1) > 0) {
110           logger.warn(s"Found more than one SFT conf in input arg")
111         }
112         Right(sfts.head)
113       } else {
114         sfts.find(_.getTypeName == name) match {
115           case Some(sft) => Right(sft)
116           case None =>
117             if (sfts.lengthCompare(1) > 0) {
118               logger.warn("Found more than one SFT conf in input arg")
119             }
120             Right(SimpleFeatureTypes.renameSft(sfts.head, name))
121         }
122       }
123     } catch {
124       case NonFatal(e) => Left(e)
125     }
126   }
127 
128   // gets an sft based on a spec conf string
129   private [SftArgResolver] def parseConfStr(args: SftArgs): ResEither =
130     parseConf(new StringReader(args.spec), args.featureName).left.map { e =>
131       (s"Unable to parse sft spec from configuration: ${args.spec}", e, CONFSTR)
132     }
133 
134   // parse spec conf file
135   private [SftArgResolver] def parseConfFile(args: SftArgs): ResEither = {
136     try {
137       val handle = PathUtils.interpretPath(args.spec).headOption.getOrElse {
138         throw new RuntimeException(s"Could not read file at ${args.spec}")
139       }
140       WithClose(handle.open) { streams =>
141         if (streams.hasNext) {
142           val reader = new InputStreamReader(streams.next._2, StandardCharsets.UTF_8)
143           parseConf(reader, args.featureName).left.map { e =>
144             (s"Unable to parse sft spec from file: ${args.spec}", e, PATH)
145           }
146         } else {
147           throw new RuntimeException(s"Could not read file at ${args.spec}")
148         }
149       }
150     } catch {
151       case NonFatal(e) => Left((s"Unable to load file: ${args.spec}", e, PATH))
152     }
153   }
154 }
