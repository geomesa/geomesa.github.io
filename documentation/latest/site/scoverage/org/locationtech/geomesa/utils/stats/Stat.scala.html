156     def serialize(double: jDouble, t: Type, jsc: JsonSerializationContext): JsonElement = double match {

313   /**
314     * Combines a sequence of stats. This will not modify any of the inputs.
315     *
316     * @param stats stats to combine
317     * @return
318     */
319   def combine[T <: Stat](stats: Seq[T]): Option[T] = {
320     if (stats.length < 2) {
321       stats.headOption
322     } else {
323       // create a new stat so that we don't modify the existing ones
324       val summed = stats.head + stats.tail.head
325       stats.drop(2).foreach(summed += _)
326       Some(summed.asInstanceOf[T])
327     }
328   }

330   // note: adds quotes around the string
331   private def safeString(s: String): String = s""""${StringEscapeUtils.escapeJava(s)}""""

333   /**
334     * Converts a value to a string
335     *
336     * @param clas class of the input value
337     * @param json will the result be used in json? will quote appropriately if so
338     * @tparam T type of input
339     * @return
340     */
341   def stringifier[T](clas: Class[T], json: Boolean = false): Any => String = {
342     val toString: Any => String = if (classOf[Geometry].isAssignableFrom(clas)) {
343       v => WKTUtils.write(v.asInstanceOf[Geometry])
344     } else if (classOf[Date].isAssignableFrom(clas)) {
345       v => GeoToolsDateFormat.format(v.asInstanceOf[Date].toInstant)
346     } else {
347       v => v.toString
348     }
349 
350     // add quotes to json strings if needed
351     if (json && (!classOf[Number].isAssignableFrom(clas) && clas != classOf[java.lang.Boolean])) {
352       v => if (v == null) { "null" } else { s""""${toString(v)}"""" }
353     } else {
354       v => if (v == null) { "null" } else { toString(v) }
355     }
356   }

358   /**
359     * Converts a string back to a value
360     *
361     * @param clas class of the stringified value
362     * @tparam T type of the value class
363     * @return
364     */
365   def destringifier[T](clas: Class[T]): String => T = {
366     if (clas == classOf[String]) {
367       s => if (s == "null") { null.asInstanceOf[T] } else { s.asInstanceOf[T] }
368     } else if (clas == classOf[Integer]) {
369       s => if (s == "null") { null.asInstanceOf[T] } else { s.toInt.asInstanceOf[T] }
370     } else if (clas == classOf[jLong]) {
371       s => if (s == "null") { null.asInstanceOf[T] } else { s.toLong.asInstanceOf[T] }
372     } else if (clas == classOf[jFloat]) {
373       s => if (s == "null") { null.asInstanceOf[T] } else { s.toFloat.asInstanceOf[T] }
374     } else if (clas == classOf[jDouble]) {
375       s => if (s == "null") { null.asInstanceOf[T] } else { s.toDouble.asInstanceOf[T] }
376     } else if (clas == classOf[java.lang.Boolean]) {
377       s => if (s == "null") { null.asInstanceOf[T] } else { s.toBoolean.asInstanceOf[T] }
378     } else if (classOf[Geometry].isAssignableFrom(clas)) {
379       s => if (s == "null") { null.asInstanceOf[T] } else { WKTUtils.read(s).asInstanceOf[T] }
380     } else if (classOf[Date].isAssignableFrom(clas)) {
381       s => if (s == "null") { null.asInstanceOf[T] } else {
382         Date.from(LocalDateTime.parse(s, GeoToolsDateFormat).toInstant(ZoneOffset.UTC)).asInstanceOf[T]
383       }
384     } else {
385       throw new RuntimeException(s"Unexpected class binding for stat attribute: $clas")
386     }
387   }
