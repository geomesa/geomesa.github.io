15 /**
16   * Proxies an input stream, copying any bytes read. The number of copied bytes can be checked with `copied`,
17   * discarded with `drop`, and returned with `replay` (which will also drop the bytes).
18   *
19   * Note: this class is not thread-safe
20   *
21   * @param wrapped input stream to proxy
22   * @param initialBuffer initial size of the array allocated for copying bytes on read. The array will grow as needed
23   */
24 class CopyingInputStream(wrapped: InputStream, initialBuffer: Int = 16) extends InputStream {
25 
26   // circular buffer holding our copied bytes
27   private val copy = new CircularByteQueue(initialBuffer)
28   private var skipped = Array.empty[Byte]
29 
30   /**
31     * The number of bytes currently available for replaying from the underlying stream
32     *
33     * @return
34     */
35   def copied: Int = copy.size
36 
37   /**
38     * Discard bytes that have been copied from the underlying stream
39     *
40     * @param count number of bytes to discard
41     */
42   def drop(count: Int): Unit = copy.drop(count)
43 
44   /**
45     * Return bytes copied from the underlying stream, discarding them afterwards. If more bytes are requested
46     * than have been copied, only copied bytes will be returned
47     *
48     * @param count number of bytes to return
49     * @return
50     */
51   def replay(count: Int): Array[Byte] = copy.dequeue(count)
52 
53   override def read(): Int = {
54     val c = wrapped.read()
55     if (c != -1) {
56       copy.enqueue(c.toByte)
57     }
58     c
59   }
60 
61   override def read(b: Array[Byte]): Int = {
62     val count = wrapped.read(b)
63     if (count != -1) {
64       copy.enqueue(b, 0, count)
65     }
66     count
67   }
68 
69   override def read(b: Array[Byte], off: Int, len: Int): Int = {
70     val count = wrapped.read(b, off, len)
71     if (count != -1) {
72       copy.enqueue(b, off, count)
73     }
74     count
75   }
76 
77   override def skip(n: Long): Long = {
78     if (n > Int.MaxValue) {
79       var remaining = n
80       var count = 0
81       var total = 0L
82       while (remaining > 0) {
83         val toRead = math.min(remaining, Int.MaxValue.toLong).toInt
84         remaining -= toRead
85         if (toRead > skipped.length) {
86           skipped = Array.ofDim(toRead)
87         }
88         count = wrapped.read(skipped, 0, toRead)
89         if (count != -1) {
90           copy.enqueue(skipped, 0, count)
91           total += count.toLong
92         } else {
93           remaining = 0
94         }
95       }
96       // release the byte buffer, as it is now Int.MaxValue length
97       skipped = Array.empty
98       total
99     } else {
100       if (n > skipped.length) {
101         skipped = Array.ofDim(n.toInt)
102       }
103       val count = wrapped.read(skipped, 0, n.toInt)
104       if (count != -1) {
105         copy.enqueue(skipped, 0, count)
106       }
107       count.toLong
108     }
109   }
110 
111   override def readNBytes(len: Int): Array[Byte] = {
112     val buf = super.readNBytes(len)
113     if (buf.nonEmpty) {
114       copy.enqueue(buf, 0, buf.length)
115     }
116     buf
117   }
118 
119   override def readNBytes(b: Array[Byte], off: Int, len: Int): Int = {
120     val count = super.readNBytes(b, off, len)
121     if (count > 0) {
122       copy.enqueue(b, off, len)
123     }
124     count
125   }
126 
127   override def readAllBytes(): Array[Byte] = {
128     val buf = super.readAllBytes()
129     if (buf.nonEmpty) {
130       copy.enqueue(buf, 0, buf.length)
131     }
132     buf
133   }
134 
135   override def available(): Int = wrapped.available()
136 
137   // note: mark/reset not supported, defer to default implementation that throws IllegalArgumentExceptions
138 
139   override def close(): Unit = wrapped.close()
140 }
