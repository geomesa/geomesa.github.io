69     if (ct.runtimeClass == classOf[Duration]) {
70       v => GeoMesaParam.parseDuration(v.asInstanceOf[String]).asInstanceOf[T]
71     } else if (ct.runtimeClass == classOf[FiniteDuration]) {

83     if (classOf[Duration].isAssignableFrom(ct.runtimeClass)) {

106   /**
107     * Returns the typed value from the map. Priority is:
108     *
109     *   1. primary key from the map
110     *   2. deprecated keys from the map
111     *   3. system property, if defined
112     *   4. default value, if defined
113     *   5. null
114     *
115     * Required parameters must be contained in the map, they will not fall back to
116     * default values or system properties
117     *
118     * @param params parameter map
119     * @return
120     */
121   def lookup(params: java.util.Map[String, _]): T = {
122     val value = if (params.containsKey(key)) {
123       lookUp(params)
124     } else if (deprecated.exists(params.containsKey)) {
125       val oldKey = deprecated.find(params.containsKey).get
126       logger.warn(s"Parameter '$oldKey' is deprecated, please use '$key' instead")
127       if (deprecatedKeys.contains(oldKey)) {
128         lookUp(Collections.singletonMap(key, params.get(oldKey)))
129       } else {
130         Option(deprecatedParams.dropWhile(_.key != oldKey).head.lookup(params, required)).map(fromTypedValue).orNull
131       }
132     } else if (required) {
133       throw new IOException(s"Parameter $key is required: $description")
134     } else {
135       systemProperty.flatMap(_.option) match {
136         case Some(v) => fromTypedValue(v)
137         case None    => null
138       }
139     }
140     if (value == null) { default } else {
141       val typed = try { toTypedValue(value) } catch {
142         case NonFatal(e) => throw new IOException(s"Invalid property for parameter '$key': $value", e)
143       }
144       if (enumerations.nonEmpty && !enumerations.contains(typed)) {
145         throw new IOException(s"Invalid property for parameter '$key': $value\n" +
146             s"  Accepted values are: ${enumerations.mkString(", ")}")
147       }
148       typed
149     }
150   }

153     * Lookup for optional parameters. If a default or system property is defined, will be returned
154     *
155     * @param params parameter map
156     * @return
157     */
158   def lookupOpt(params: java.util.Map[String, _]): Option[T] = Option(lookup(params))

160   // handle FQDN for interface classes, instead of the `new Class(String)` constructor that geotools tries
161   override def parse(text: String): AnyRef = {
162     try {
163       super.parse(text)
164     } catch {
165       case NonFatal(e) =>
166         try {
167           val modifiers = ct.runtimeClass.getModifiers
168           if (Modifier.isAbstract(modifiers) || Modifier.isInterface(modifiers)) {
169             return Class.forName(text).getConstructor().newInstance().asInstanceOf[AnyRef]
170           }
171         } catch {
172           case NonFatal(suppressed) => e.addSuppressed(suppressed)
173         }
174         throw e
175     }
176   }

282   def metadata(
283       password: Boolean,
284       largeText: Boolean,
285       extension: String,
286       enumerations: Seq[AnyRef],
287       supportsNiFiExpressions: Boolean,
288       deprecated: Boolean = false): java.util.Map[String, AnyRef] = {
289     var opt: Option[java.util.Map[String, AnyRef]] = None
290     lazy val map = {
291       val m = new java.util.HashMap[String, AnyRef]
292       opt = Some(m) // note: side effect
293       m
294     }
295     if (password) {
296       map.put(Parameter.IS_PASSWORD, java.lang.Boolean.TRUE)
297     }
298     if (largeText) {
299       map.put(Parameter.IS_LARGE_TEXT, java.lang.Boolean.TRUE)
300     }
301     if (extension != null) {
302       map.put(Parameter.EXT, extension)
303     }
304     if (enumerations.nonEmpty) {
305       // convert to a mutable java list, as geoserver tries to sort it in place
306       val enums = new java.util.ArrayList[AnyRef](enumerations.length)
307       enumerations.foreach(enums.add)
308       map.put(Parameter.OPTIONS, enums)
309     }
310     if (supportsNiFiExpressions) {
311       map.put(SupportsNiFiExpressions, java.lang.Boolean.TRUE)
312     }
313     if (deprecated) {
314       map.put(Parameter.DEPRECATED, java.lang.Boolean.TRUE)
315     }
316     opt.map(Collections.unmodifiableMap[String,AnyRef]).orNull
317   }
