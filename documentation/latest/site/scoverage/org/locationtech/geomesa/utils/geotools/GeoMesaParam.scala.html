97   /**
98     * Returns the typed value from the map. Priority is:
99     *
100     *   1. primary key from the map
101     *   2. deprecated keys from the map
102     *   3. system property, if defined
103     *   4. default value, if defined
104     *   5. null
105     *
106     * Required parameters must be contained in the map, they will not fall back to
107     * default values or system properties
108     *
109     * @param params parameter map
110     * @return
111     */
112   def lookup(params: java.util.Map[String, _]): T = {
113     val value = if (params.containsKey(key)) {
114       lookUp(params)
115     } else if (deprecated.exists(params.containsKey)) {
116       val oldKey = deprecated.find(params.containsKey).get
117       deprecationWarning(oldKey)
118       if (deprecatedKeys.contains(oldKey)) {
119         lookUp(Collections.singletonMap(key, params.get(oldKey)))
120       } else {
121         Option(deprecatedParams.dropWhile(_.key != oldKey).head.lookup(params, required)).map(fromTypedValue).orNull
122       }
123     } else if (required) {
124       throw new IOException(s"Parameter $key is required: $description")
125     } else {
126       systemProperty.flatMap(_.option) match {
127         case Some(v) => fromTypedValue(v)
128         case None    => null
129       }
130     }
131     if (value == null) { default } else {
132       val typed = try { toTypedValue(value) } catch {
133         case NonFatal(e) => throw new IOException(s"Invalid property for parameter '$key': $value", e)
134       }
135       if (enumerations.nonEmpty && !enumerations.contains(typed)) {
136         throw new IOException(s"Invalid property for parameter '$key': $value\n" +
137             s"  Accepted values are: ${enumerations.mkString(", ")}")
138       }
139       typed
140     }
141   }

263   def metadata(
264       password: Boolean,
265       largeText: Boolean,
266       extension: String,
267       enumerations: Seq[AnyRef],
268       supportsNiFiExpressions: Boolean,
269       deprecated: Boolean = false): java.util.Map[String, AnyRef] = {
270     var opt: Option[java.util.Map[String, AnyRef]] = None
271     lazy val map = {
272       val m = new java.util.HashMap[String, AnyRef]
273       opt = Some(m) // note: side effect
274       m
275     }
276     if (password) {
277       map.put(Parameter.IS_PASSWORD, java.lang.Boolean.TRUE)
278     }
279     if (largeText) {
280       map.put(Parameter.IS_LARGE_TEXT, java.lang.Boolean.TRUE)
281     }
282     if (extension != null) {
283       map.put(Parameter.EXT, extension)
284     }
285     if (enumerations.nonEmpty) {
286       // convert to a mutable java list, as geoserver tries to sort it in place
287       val enums = new java.util.ArrayList[AnyRef](enumerations.length)
288       enumerations.foreach(enums.add)
289       map.put(Parameter.OPTIONS, enums)
290     }
291     if (supportsNiFiExpressions) {
292       map.put(SupportsNiFiExpressions, java.lang.Boolean.TRUE)
293     }
294     if (deprecated) {
295       map.put(Parameter.DEPRECATED, java.lang.Boolean.TRUE)
296     }
297     opt.map(Collections.unmodifiableMap[String,AnyRef]).orNull
298   }
