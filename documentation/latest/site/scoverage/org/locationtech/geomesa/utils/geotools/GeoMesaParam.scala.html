66     if (ct.runtimeClass == classOf[Duration]) {

96   def exists(params: java.util.Map[String, _]): Boolean =

99   /**
100     * Returns the typed value from the map. Priority is:
101     *
102     *   1. primary key from the map
103     *   2. deprecated keys from the map
104     *   3. system property, if defined
105     *   4. default value, if defined
106     *   5. null
107     *
108     * Required parameters must be contained in the map, they will not fall back to
109     * default values or system properties
110     *
111     * @param params parameter map
112     * @return
113     */
114   def lookup(params: java.util.Map[String, _]): T = {
115     val value = if (params.containsKey(key)) {
116       lookUp(params)
117     } else if (deprecated.exists(params.containsKey)) {
118       val oldKey = deprecated.find(params.containsKey).get
119       deprecationWarning(oldKey)
120       if (deprecatedKeys.contains(oldKey)) {
121         lookUp(Collections.singletonMap(key, params.get(oldKey)))
122       } else {
123         Option(deprecatedParams.dropWhile(_.key != oldKey).head.lookup(params, required)).map(fromTypedValue).orNull
124       }
125     } else if (required) {
126       throw new IOException(s"Parameter $key is required: $description")
127     } else {
128       systemProperty.flatMap(_.option) match {
129         case Some(v) => fromTypedValue(v)
130         case None    => null
131       }
132     }
133     if (value == null) { default } else {
134       val typed = try { toTypedValue(value) } catch {
135         case NonFatal(e) => throw new IOException(s"Invalid property for parameter '$key': $value", e)
136       }
137       if (enumerations.nonEmpty && !enumerations.contains(typed)) {
138         throw new IOException(s"Invalid property for parameter '$key': $value\n" +
139             s"  Accepted values are: ${enumerations.mkString(", ")}")
140       }
141       typed
142     }
143   }

265   def metadata(
266       password: Boolean,
267       largeText: Boolean,
268       extension: String,
269       enumerations: Seq[AnyRef],
270       supportsNiFiExpressions: Boolean,
271       deprecated: Boolean = false): java.util.Map[String, AnyRef] = {
272     var opt: Option[java.util.Map[String, AnyRef]] = None
273     lazy val map = {
274       val m = new java.util.HashMap[String, AnyRef]
275       opt = Some(m) // note: side effect
276       m
277     }
278     if (password) {
279       map.put(Parameter.IS_PASSWORD, java.lang.Boolean.TRUE)
280     }
281     if (largeText) {
282       map.put(Parameter.IS_LARGE_TEXT, java.lang.Boolean.TRUE)
283     }
284     if (extension != null) {
285       map.put(Parameter.EXT, extension)
286     }
287     if (enumerations.nonEmpty) {
288       // convert to a mutable java list, as geoserver tries to sort it in place
289       val enums = new java.util.ArrayList[AnyRef](enumerations.length)
290       enumerations.foreach(enums.add)
291       map.put(Parameter.OPTIONS, enums)
292     }
293     if (supportsNiFiExpressions) {
294       map.put(SupportsNiFiExpressions, java.lang.Boolean.TRUE)
295     }
296     if (deprecated) {
297       map.put(Parameter.DEPRECATED, java.lang.Boolean.TRUE)
298     }
299     opt.map(Collections.unmodifiableMap[String,AnyRef]).orNull
300   }
