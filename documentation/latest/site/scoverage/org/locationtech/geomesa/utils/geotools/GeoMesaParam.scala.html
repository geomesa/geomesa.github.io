67   private val toTypedValue: AnyRef => T = {
68     if (ct.runtimeClass == classOf[Duration]) {
69       v => GeoMesaParam.parseDuration(v.asInstanceOf[String]).asInstanceOf[T]
70     } else if (ct.runtimeClass == classOf[FiniteDuration]) {

72     } else if (ct.runtimeClass == classOf[Properties]) {

74     } else if (ct.runtimeClass == classOf[Config]) {
75       v => ConfigFactory.parseString(v.asInstanceOf[String]).asInstanceOf[T]

81   private val fromTypedValue: T => AnyRef = {
82     if (classOf[Duration].isAssignableFrom(ct.runtimeClass)) {
83       v => GeoMesaParam.printDuration(v.asInstanceOf[Duration])
84     } else if (ct.runtimeClass == classOf[Properties]) {

86     } else if (ct.runtimeClass == classOf[Config]) {
87       v => v.asInstanceOf[Config].root().render(ConfigRenderOptions.concise())
88     } else {
89       v => v

96   /**
97     * Checks that the parameter is contained in the map, but does not do type conversion
98     *
99     * @param params parameter map
100     * @return
101     */
102   def exists(params: java.util.Map[String, _]): Boolean =
103     params.get(key) != null || deprecated.exists(params.get(_) != null)

105   /**
106     * Returns the typed value from the map. Priority is:
107     *
108     *   1. primary key from the map
109     *   2. deprecated keys from the map
110     *   3. system property, if defined
111     *   4. default value, if defined
112     *   5. null
113     *
114     * Required parameters must be contained in the map, they will not fall back to
115     * default values or system properties
116     *
117     * @param params parameter map
118     * @return
119     */
120   def lookup(params: java.util.Map[String, _]): T = {
121     val value = if (params.containsKey(key)) {
122       lookUp(params)
123     } else if (deprecated.exists(params.containsKey)) {
124       val oldKey = deprecated.find(params.containsKey).get
125       deprecationWarning(oldKey)
126       if (deprecatedKeys.contains(oldKey)) {
127         lookUp(Collections.singletonMap(key, params.get(oldKey)))
128       } else {
129         Option(deprecatedParams.dropWhile(_.key != oldKey).head.lookup(params, required)).map(fromTypedValue).orNull
130       }
131     } else if (required) {
132       throw new IOException(s"Parameter $key is required: $description")
133     } else {
134       systemProperty.flatMap(_.option) match {
135         case Some(v) => fromTypedValue(v)
136         case None    => null
137       }
138     }
139     if (value == null) { default } else {
140       val typed = try { toTypedValue(value) } catch {
141         case NonFatal(e) => throw new IOException(s"Invalid property for parameter '$key': $value", e)
142       }
143       if (enumerations.nonEmpty && !enumerations.contains(typed)) {
144         throw new IOException(s"Invalid property for parameter '$key': $value\n" +
145             s"  Accepted values are: ${enumerations.mkString(", ")}")
146       }
147       typed
148     }
149   }

271   def metadata(
272       password: Boolean,
273       largeText: Boolean,
274       extension: String,
275       enumerations: Seq[AnyRef],
276       supportsNiFiExpressions: Boolean,
277       deprecated: Boolean = false): java.util.Map[String, AnyRef] = {
278     var opt: Option[java.util.Map[String, AnyRef]] = None
279     lazy val map = {
280       val m = new java.util.HashMap[String, AnyRef]
281       opt = Some(m) // note: side effect
282       m
283     }
284     if (password) {
285       map.put(Parameter.IS_PASSWORD, java.lang.Boolean.TRUE)
286     }
287     if (largeText) {
288       map.put(Parameter.IS_LARGE_TEXT, java.lang.Boolean.TRUE)
289     }
290     if (extension != null) {
291       map.put(Parameter.EXT, extension)
292     }
293     if (enumerations.nonEmpty) {
294       // convert to a mutable java list, as geoserver tries to sort it in place
295       val enums = new java.util.ArrayList[AnyRef](enumerations.length)
296       enumerations.foreach(enums.add)
297       map.put(Parameter.OPTIONS, enums)
298     }
299     if (supportsNiFiExpressions) {
300       map.put(SupportsNiFiExpressions, java.lang.Boolean.TRUE)
301     }
302     if (deprecated) {
303       map.put(Parameter.DEPRECATED, java.lang.Boolean.TRUE)
304     }
305     opt.map(Collections.unmodifiableMap[String,AnyRef]).orNull
306   }
