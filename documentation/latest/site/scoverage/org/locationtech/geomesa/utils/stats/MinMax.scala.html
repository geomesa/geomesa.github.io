21 /**
22   * The MinMax stat merely returns the min/max of an attribute's values.
23   * Works with dates, integers, longs, doubles, and floats.
24   *
25   * @param sft simple feature type
26   * @param property property name for the attribute being min/maxed
27   * @tparam T the type of the attribute the stat is targeting (needs to be comparable)
28  */
29 class MinMax[T] private [stats] (val sft: SimpleFeatureType,
30                                  val property: String,
31                                  private [stats] var minValue: T,
32                                  private [stats] var maxValue: T,
33                                  private [stats] val hpp: HyperLogLog)
34                                 (implicit val defaults: MinMax.MinMaxDefaults[T])
35     extends Stat with LazyLogging with Serializable {
36 
37   // use a secondary constructor instead of companion apply to allow mixin types (i.e. ImmutableStat)
38   def this(sft: SimpleFeatureType, attribute: String)(implicit defaults: MinMax.MinMaxDefaults[T]) =
39     this(sft, attribute, defaults.max, defaults.min, HyperLogLog(CardinalityBits))(defaults)
40 
41   override type S = MinMax[T]
42 
43   private val i = sft.indexOf(property)
44 
45   def min: T = if (isEmpty) { maxValue } else { minValue }
46   def max: T = if (isEmpty) { minValue } else { maxValue }
47   def bounds: (T, T) = (min, max)
48   def cardinality: Long = hpp.cardinality()
49   def tuple: (T, T, Long) = (min, max, cardinality)
50 
51   override def observe(sf: SimpleFeature): Unit = {
52     val value = sf.getAttribute(i).asInstanceOf[T]
53     if (value != null) {
54       try {
55         minValue = defaults.min(value, minValue)
56         maxValue = defaults.max(value, maxValue)
57         hpp.offer(value)
58       } catch {
59         case e: Exception => logger.warn(s"Error observing value '$value': ${e.toString}")
60       }
61     }
62   }
63 
64   // note: can't unobserve min/max without storing a lot more data
65   override def unobserve(sf: SimpleFeature): Unit = {}
66 
67   override def +(other: MinMax[T]): MinMax[T] = {
68     if (other.isEmpty) {
69       new MinMax[T](sft, property, minValue, maxValue, hpp.merge())
70     } else if (this.isEmpty) {
71       new MinMax[T](sft, property, other.minValue, other.maxValue, other.hpp.merge())
72     } else {
73       val plus = new MinMax[T](sft, property, minValue, maxValue, hpp.merge())
74       plus += other
75       plus
76     }
77   }
78 
79   override def +=(other: MinMax[T]): Unit = {
80     if (other.isEmpty) {
81       // no-op
82     } else if (isEmpty) {
83       minValue = other.minValue
84       maxValue = other.maxValue
85       hpp += other.hpp
86     } else {
87       minValue = defaults.min(minValue, other.minValue)
88       maxValue = defaults.max(maxValue, other.maxValue)
89       hpp += other.hpp
90     }
91   }
92 
93   override def toJsonObject: Any =
94     if (isEmpty) {
95       ListMap("min" -> null, "max" -> null, "cardinality" -> 0)
96     } else {
97       ListMap("min" -> minValue, "max" -> maxValue, "cardinality" -> cardinality)
98     }
99 
100   override def isEmpty: Boolean = minValue == defaults.max
101 
102   override def clear(): Unit = {
103     minValue = defaults.max
104     maxValue = defaults.min
105     java.util.Arrays.fill(hpp.registerSet.rawBits, 0)
106   }
107 
108   override def isEquivalent(other: Stat): Boolean = other match {
109     case that: MinMax[T] =>
110       property == that.property && minValue == that.minValue &&
111           maxValue == that.maxValue && cardinality == that.cardinality
112     case _ => false
113   }
114 }

116 object MinMax {

118   val CardinalityBits: Int = 10

120   trait MinMaxDefaults[T] {
121     def min: T
122     def max: T
123     def min(left: T, right: T): T
124     def max(left: T, right: T): T
125   }
126 
127   object MinMaxDefaults {
128     def apply[T](binding: Class[_]): MinMaxDefaults[T] = {
129       if (binding == classOf[String]) {
130         MinMaxString.asInstanceOf[MinMaxDefaults[T]]
131       } else if (binding == classOf[Integer]) {
132         MinMaxInt.asInstanceOf[MinMaxDefaults[T]]
133       } else if (binding == classOf[java.lang.Long]) {
134         MinMaxLong.asInstanceOf[MinMaxDefaults[T]]
135       } else if (binding == classOf[java.lang.Float]) {
136         MinMaxFloat.asInstanceOf[MinMaxDefaults[T]]
137       } else if (binding == classOf[java.lang.Double]) {
138         MinMaxDouble.asInstanceOf[MinMaxDefaults[T]]
139       } else if (classOf[Date].isAssignableFrom(binding)) {
140         MinMaxDate.asInstanceOf[MinMaxDefaults[T]]
141       } else if (classOf[Geometry].isAssignableFrom(binding)) {
142         MinMaxGeometry.asInstanceOf[MinMaxDefaults[T]]
143       } else {
144         throw new IllegalArgumentException(s"No implicit default available for type: $binding")
145       }
146     }
147   }
148 
149   abstract class ComparableMinMax[T <: Comparable[T]] extends MinMaxDefaults[T] with Serializable {
150     override def min(left: T, right: T): T = if (left.compareTo(right) > 0) right else left
151     override def max(left: T, right: T): T = if (left.compareTo(right) < 0) right else left
152   }
153 
154   implicit object MinMaxString extends ComparableMinMax[String] with Serializable {
155     override val min: String = ""
156     override val max: String = "\uFFFF\uFFFF\uFFFF"
157   }

194     override def min(left: Geometry, right: Geometry): Geometry = {
195       val (lx, ly) = { val e = left.getEnvelopeInternal; (e.getMinX, e.getMinY) }
196       val (rx, ry) = { val e = right.getEnvelopeInternal; (e.getMinX, e.getMinY) }
197 
198       val x = math.min(lx, rx)
199       val y = math.min(ly, ry)
200 
201       if (x == lx && y == ly) {
202         left
203       } else if (x == rx && y == ry) {
204         right
205       } else {
206         gf.createPoint(new Coordinate(x, y))
207       }
208     }

223       }
224     }
