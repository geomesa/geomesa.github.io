13 /**
14   * A bit set backed by ints (to reduce memory footprint for small sizes).
15   *
16   * Provides access to the underlying words, for low-level synchronization. Not thread safe.
17   */
18 trait WordBitSet {
19 
20   /**
21     * Gets the word for the given value. The word can be synchronized on for any operations on the given value.
22     *
23     * @param value value
24     * @return word
25     */
26   def word(value: Int): Word
27 }
28 
29 object WordBitSet {
30 
31   // use `>> 5` instead of `/ 32`
32   private final val Divisor = 5
33 
34   /**
35     * Create a bit set capable of holding numbers up to `length`
36     *
37     * @param length max int to be stored in the bit set
38     * @return
39     */
40   def apply(length: Int): WordBitSet = {
41     size(length) match {
42       case 1 => new WordBitSet32()
43       case 2 => new WordBitSet64()
44       case 3 => new WordBitSet96()
45       case 4 => new WordBitSet128()
46       case 5 => new WordBitSet160()
47       case 6 => new WordBitSet192()
48       case n => new WordBitSetN(n)
49     }
50   }
51 
52   /**
53    * Gets the number of ints (words) required to hold numbers up to `length`
54    *
55    * @param length max int to be stored in the bit set
56    * @return
57    */
58   def size(length: Int): Int = ((length - 1) >> Divisor) + 1
59 
60   /**
61     * A single word in the bit set
62     *
63     * @param mask the mask for this word
64     */
65   class Word(private var mask: Int = 0) {
66 
67     /**
68       * Checks whether the value is contained in the set or not
69       *
70       * @param value value to check
71       * @return true if contains, false otherwise
72       */
73     def contains(value: Int): Boolean = (mask & (1 << value)) != 0
74 
75     /**
76       * Adds the value to the set, if it is not present. Note that implementations may be constructed with
77       * a fixed capacity - attempting to add a value outside the capacity will result in undefined behavior
78       *
79       * @param value value to add
80       * @return true if value was added, false if value was already present
81       */
82     def add(value: Int): Boolean = {
83       val updated = mask | (1 << value)
84       if (updated == mask) { false } else {
85         mask = updated
86         true
87       }
88     }
89 
90     /**
91       * Removes the value from the set, if it is present
92       *
93       * @param value value to remove
94       * @return true if value was removed, false if value was not present
95       */
96     def remove(value: Int): Boolean = {
97       val updated = mask & ~(1 << value)
98       if (updated == mask) { false } else {
99         mask = updated
100         true
101       }
102     }
103   }
104 
105   /**
106     * Bit set for tracking values < 32
107     */
108   class WordBitSet32 extends WordBitSet {
109 
110     private val word0 = new Word()
111 
112     override def word(value: Int): Word =  {
113       value >> Divisor match {
114         case 0 => word0
115         case _ => throw new ArrayIndexOutOfBoundsException(value)
116       }
117     }
118   }
