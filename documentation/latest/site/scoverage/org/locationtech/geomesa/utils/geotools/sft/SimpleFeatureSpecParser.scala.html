37   private def parse[T](spec: String, rule: Rule1[T], report: Boolean): T = {
38     if (spec == null) {
39       throw new IllegalArgumentException("Invalid spec: null")
40     }
41     val runner = if (report) { ReportingParseRunner(rule) } else { BasicParseRunner(rule) }
42     val parsing = runner.run(spec.stripMarginAndWhitespace())
43     parsing.result.getOrElse(throw new ParsingException(constructErrorMessage(parsing, report, spec)))
44   }
45 
46   private def constructErrorMessage[T](result: ParsingResult[T], report: Boolean, spec: String): String = {
47     lazy val fallback = s"Invalid spec: ${ErrorUtils.printParseErrors(result)}\n$spec"
48     if (!report) { fallback } else {
49       result.parseErrors.collectFirst { case e: InvalidInputError =>
50         import scala.collection.JavaConverters._
51         // add a caret pointing to the invalid location
52         val specAndIndex = s"$spec\n${Seq.fill(math.max(0, e.getStartIndex - 1))(" ").mkString}^"
53         // determine what paths the parser partially matched
54         val matchers = e.getFailedMatchers.asScala.map(getFailedMatcher).distinct
55         if (matchers.isEmpty) {
56           s"Invalid spec at index ${e.getStartIndex}\n$specAndIndex"
57         } else {
58           val expected = if (matchers.lengthCompare(1) > 0) { s"one of: ${matchers.mkString(", ")}" } else { matchers.head }
59           s"Invalid spec at index ${e.getStartIndex} - expected $expected\n$specAndIndex"
60         }
61       }.getOrElse(fallback)
62     }
63   }

67     * we get to a meaningful parser, as defined in ReportableParserMessages.

70     * @param path failed matcher path

76       case Some(m) => m
77       case None    => getFailedMatcher(path.parent)

97 private class SimpleFeatureSpecParser extends BasicParser {
98 
99   // Valid specs can have attributes that look like the following:
100   // "id:Integer:opt1=v1:opt2=v2,*geom:Geometry:srid=4326,ct:List[String],mt:Map[String,Double]"
101 
102   import SimpleFeatureTypes.AttributeOptions.OptDefault
103   import org.parboiled.scala._

105   // full simple feature spec
106   def spec: Rule1[SimpleFeatureSpec] = rule("Specification") {
107     (whitespace ~ zeroOrMore(attribute, whitespace ~ "," ~ whitespace) ~ whitespace ~ sftOptions ~ whitespace) ~ EOI ~~> {
108       (attributes, sftOpts) => SimpleFeatureSpec(attributes, sftOpts)
109     }
110   }
111 
112   // any attribute
113   def attribute: Rule1[AttributeSpec] = rule("Attribute") {
114     simpleAttribute | geometryAttribute | listAttribute | mapAttribute
115   }
116 
117   // builds a SimpleAttributeSpec for primitive types
118   private def simpleAttribute: Rule1[AttributeSpec] = rule("SimpleAttribute") {
119     (name ~ whitespace ~ simpleType ~ whitespace ~ attributeOptions) ~~> SimpleAttributeSpec
120   }
121 
122   // builds a GeometrySpec
123   private def geometryAttribute: Rule1[AttributeSpec] = rule("GeometryAttribute") {
124     (default ~ name ~ whitespace ~ geometryType ~ whitespace ~ attributeOptions) ~~> {
125       (default, name, geom, opts) => {
126         GeomAttributeSpec(name, geom, opts + (OptDefault -> default.getOrElse(false).toString))
127       }
128     }
129   }
130 
131   // builds a ListAttributeSpec for complex types
132   private def listAttribute: Rule1[AttributeSpec] = rule("ListAttribute") {
133     (name ~ whitespace ~ listType ~ whitespace ~ attributeOptions) ~~> ListAttributeSpec
134   }
135 
136   // builds a MapAttributeSpec for complex types
137   private def mapAttribute: Rule1[AttributeSpec] = rule("MapAttribute") {
138     (name ~ whitespace ~ mapType ~ whitespace ~ attributeOptions) ~~> {
139       (name, types, opts) => {
140         MapAttributeSpec(name, types._1, types._2, opts)
141       }
142     }
143   }
144 
145   // an attribute name
146   private def name: Rule1[String] = rule("AttributeName") { oneOrMore(noneOf("*:, ")) ~> { s => s } }
147 
148   private def default: Rule1[Option[Boolean]] = rule("DefaultAttribute") { optional("*" ~ push(true)) }
149 
150   private def simpleType: Rule1[Class[_]] = rule("SimpleTypeBinding") { ":" ~ whitespace ~ simpleBinding }
151 
152   // matches any of the primitive types defined in simpleTypeMap
153   // order matters so that Integer is matched before Int
154   private def simpleBinding: Rule1[Class[_]] = rule("SimpleBinding") {
155     simpleTypeMap.keys.toList.sorted.reverse.map(str).reduce(_ | _) ~> simpleTypeMap.apply
156   }
157 
158   // matches any of the geometry types defined in geometryTypeMap
159   private def geometryType: Rule1[Class[_ <: Geometry]] = rule("GeometryTypeBinding") {
160     ":" ~ whitespace ~ geometryTypeMap.keys.toList.sorted.reverse.map(str).reduce(_ | _) ~> geometryTypeMap.apply
161   }
162 
163   // matches "List[Foo]" or "List" (which defaults to [String])
164   private def listType: Rule1[Class[_]] = rule("ListTypeBinding") {
165     ":" ~ whitespace ~ listTypeMap.keys.map(str).reduce(_ | _) ~ whitespace ~
166         optional("[" ~ whitespace ~ simpleBinding ~ whitespace ~ "]") ~~> {
167       inner => inner.getOrElse(classOf[String])
168     }
169   }
170 
171   // matches "Map[Foo,Bar]" or Map (which defaults to [String,String])
172   private def mapType: Rule1[(Class[_], Class[_])] = rule("MapTypeBinding") {
173     ":" ~ whitespace ~ mapTypeMap.keys.map(str).reduce(_ | _) ~ whitespace ~
174         optional("[" ~ whitespace ~ simpleBinding ~ whitespace ~ "," ~ whitespace ~ simpleBinding ~ whitespace ~ "]") ~~> {
175       types => types.getOrElse(classOf[String], classOf[String])
176     }
177   }
178 
179   // options appended to a specific attribute
180   private def attributeOptions: Rule1[Map[String, String]] = rule("AttributeOptions") {
181     optional(":" ~ oneOrMore(attributeOption, whitespace ~ ":" ~ whitespace) ~~> { _.toMap }) ~~> { _.getOrElse(Map.empty) }
182   }
183 
184   // single attribute option
185   private def attributeOption: Rule2[String, String] = rule("AttributeOption") {
186     oneOrMore(char | anyOf(".-")) ~> { s => s } ~ whitespace ~ "=" ~ whitespace ~ (quotedString | singleQuotedString | unquotedString)
187   }
188 
189   // options for the simple feature type
190   private def sftOptions: Rule1[Map[String, String]] = rule("FeatureTypeOptions") {
191     optional(";" ~ whitespace ~ zeroOrMore(sftOption, whitespace ~ "," ~ whitespace)) ~~>  { o => o.map(_.toMap).getOrElse(Map.empty) }
192   }
193 
194   // single sft option
195   private def sftOption: Rule1[(String, String)] = rule("FeatureTypeOption") {
196     (oneOrMore(char | anyOf(".-")) ~> { s => s } ~ whitespace ~ "=" ~ whitespace ~ sftOptionValue) ~~> { (k, v) => (k, v) }
197   }
198 
199   // value for an sft option
200   private def sftOptionValue: Rule1[String] = rule("FeatureTypeOptionValue") {
201     singleQuotedString | quotedString | oneOrMore(noneOf("'\",")) ~> { s => s }
202   }
