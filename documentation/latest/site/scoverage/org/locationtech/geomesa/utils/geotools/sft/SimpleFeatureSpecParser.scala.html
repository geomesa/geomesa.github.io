37   private def parse[T](spec: String, rule: Rule1[T], report: Boolean): T = {
38     if (spec == null) {
39       throw new IllegalArgumentException("Invalid spec string: null")
40     }
41     val runner = if (report) { ReportingParseRunner(rule) } else { BasicParseRunner(rule) }
42     val parsing = runner.run(spec.stripMarginAndWhitespace())
43     parsing.result.getOrElse(throw new ParsingException(constructErrorMessage(parsing, report)))
44   }
45 
46   private def constructErrorMessage[T](result: ParsingResult[T], report: Boolean): String = {
47     lazy val fallback = s"Invalid spec string: ${ErrorUtils.printParseErrors(result)}"
48     if (!report) { fallback } else {
49       result.parseErrors.collectFirst { case e: InvalidInputError =>
50         import scala.collection.JavaConverters._
51         // determine what paths the parser partially matched
52         val matchers = e.getFailedMatchers.asScala.map(getFailedMatcher).distinct
53         if (matchers.isEmpty) {
54           s"Invalid spec string at index ${e.getStartIndex}."
55         } else {
56           val expected = if (matchers.lengthCompare(1) > 0) { s"one of: ${matchers.mkString(", ")}" } else { matchers.head }
57           s"Invalid spec string at index ${e.getStartIndex}. Expected $expected."
58         }
59       }.getOrElse(fallback)
60     }
61   }

63   /**
64     * Start at the failure point and go up the parse tree until
65     * we get to a meaningful parser, as defined in ReportableParserMessages.
66     * We should always match at least the top-level parser (spec or attribute)
67     *
68     * @param path failed matcher path
69     * @return message indicating the last-matched matcher
70     */
71   @tailrec
72   private def getFailedMatcher(path: MatcherPath): String = {
73     ReportableParserMessages.get(path.element.matcher.getLabel) match {
74       case Some(m) => m
75       case None    => getFailedMatcher(path.parent)
76     }
77   }

95 private class SimpleFeatureSpecParser extends BasicParser {
96 
97   // Valid specs can have attributes that look like the following:
98   // "id:Integer:opt1=v1:opt2=v2,*geom:Geometry:srid=4326,ct:List[String],mt:Map[String,Double]"
99 
100   import SimpleFeatureTypes.AttributeOptions.OptDefault
101   import org.parboiled.scala._
102 
103   // full simple feature spec
104   def spec: Rule1[SimpleFeatureSpec] = rule("Specification") {
105     (whitespace ~ zeroOrMore(attribute, whitespace ~ "," ~ whitespace) ~ whitespace ~ sftOptions ~ whitespace) ~ EOI ~~> {
106       (attributes, sftOpts) => SimpleFeatureSpec(attributes, sftOpts)
107     }
108   }
109 
110   // any attribute
111   def attribute: Rule1[AttributeSpec] = rule("Attribute") {
112     simpleAttribute | geometryAttribute | listAttribute | mapAttribute
113   }
114 
115   // builds a SimpleAttributeSpec for primitive types
116   private def simpleAttribute: Rule1[AttributeSpec] = rule("SimpleAttribute") {
117     (name ~ whitespace ~ simpleType ~ whitespace ~ attributeOptions) ~~> SimpleAttributeSpec
118   }
119 
120   // builds a GeometrySpec
121   private def geometryAttribute: Rule1[AttributeSpec] = rule("GeometryAttribute") {
122     (default ~ name ~ whitespace ~ geometryType ~ whitespace ~ attributeOptions) ~~> {
123       (default, name, geom, opts) => {
124         GeomAttributeSpec(name, geom, opts + (OptDefault -> default.getOrElse(false).toString))
125       }
126     }
127   }
128 
129   // builds a ListAttributeSpec for complex types
130   private def listAttribute: Rule1[AttributeSpec] = rule("ListAttribute") {
131     (name ~ whitespace ~ listType ~ whitespace ~ attributeOptions) ~~> ListAttributeSpec
132   }
133 
134   // builds a MapAttributeSpec for complex types
135   private def mapAttribute: Rule1[AttributeSpec] = rule("MapAttribute") {
136     (name ~ whitespace ~ mapType ~ whitespace ~ attributeOptions) ~~> {
137       (name, types, opts) => {
138         MapAttributeSpec(name, types._1, types._2, opts)
139       }
140     }
141   }
142 
143   // an attribute name
144   private def name: Rule1[String] = rule("AttributeName") { oneOrMore(noneOf("*:, ")) ~> { s => s } }
145 
146   private def default: Rule1[Option[Boolean]] = rule("DefaultAttribute") { optional("*" ~ push(true)) }
147 
148   private def simpleType: Rule1[Class[_]] = rule("SimpleTypeBinding") { ":" ~ whitespace ~ simpleBinding }
149 
150   // matches any of the primitive types defined in simpleTypeMap
151   // order matters so that Integer is matched before Int
152   private def simpleBinding: Rule1[Class[_]] = rule("SimpleBinding") {
153     simpleTypeMap.keys.toList.sorted.reverse.map(str).reduce(_ | _) ~> simpleTypeMap.apply
154   }
155 
156   // matches any of the geometry types defined in geometryTypeMap
157   private def geometryType: Rule1[Class[_ <: Geometry]] = rule("GeometryTypeBinding") {
158     ":" ~ whitespace ~ geometryTypeMap.keys.toList.sorted.reverse.map(str).reduce(_ | _) ~> geometryTypeMap.apply
159   }
160 
161   // matches "List[Foo]" or "List" (which defaults to [String])
162   private def listType: Rule1[Class[_]] = rule("ListTypeBinding") {
163     ":" ~ whitespace ~ listTypeMap.keys.map(str).reduce(_ | _) ~ whitespace ~
164         optional("[" ~ whitespace ~ simpleBinding ~ whitespace ~ "]") ~~> {
165       inner => inner.getOrElse(classOf[String])
166     }
167   }
168 
169   // matches "Map[Foo,Bar]" or Map (which defaults to [String,String])
170   private def mapType: Rule1[(Class[_], Class[_])] = rule("MapTypeBinding") {
171     ":" ~ whitespace ~ mapTypeMap.keys.map(str).reduce(_ | _) ~ whitespace ~
172         optional("[" ~ whitespace ~ simpleBinding ~ whitespace ~ "," ~ whitespace ~ simpleBinding ~ whitespace ~ "]") ~~> {
173       types => types.getOrElse(classOf[String], classOf[String])
174     }
175   }
176 
177   // options appended to a specific attribute
178   private def attributeOptions: Rule1[Map[String, String]] = rule("AttributeOptions") {
179     optional(":" ~ oneOrMore(attributeOption, whitespace ~ ":" ~ whitespace) ~~> { _.toMap }) ~~> { _.getOrElse(Map.empty) }
180   }
181 
182   // single attribute option
183   private def attributeOption: Rule2[String, String] = rule("AttributeOption") {
184     oneOrMore(char | anyOf(".-")) ~> { s => s } ~ whitespace ~ "=" ~ whitespace ~ (quotedString | singleQuotedString | unquotedString)
185   }
186 
187   // options for the simple feature type
188   private def sftOptions: Rule1[Map[String, String]] = rule("FeatureTypeOptions") {
189     optional(";" ~ whitespace ~ zeroOrMore(sftOption, whitespace ~ "," ~ whitespace)) ~~>  { o => o.map(_.toMap).getOrElse(Map.empty) }
190   }
191 
192   // single sft option
193   private def sftOption: Rule1[(String, String)] = rule("FeatureTypeOption") {
194     (oneOrMore(char | anyOf(".-")) ~> { s => s } ~ whitespace ~ "=" ~ whitespace ~ sftOptionValue) ~~> { (k, v) => (k, v) }
195   }
196 
197   // value for an sft option
198   private def sftOptionValue: Rule1[String] = rule("FeatureTypeOptionValue") {
199     singleQuotedString | quotedString | oneOrMore(noneOf("'\",")) ~> { s => s }
200   }
