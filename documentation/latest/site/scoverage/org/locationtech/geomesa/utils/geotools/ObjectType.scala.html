39   /**
40    * Turns a SimpleFeatureType attribute class binding into an enumeration.
41    *
42    * The first element in the result will be the primary binding. For geometries, lists and maps,
43    * the result will also contain secondary types.
44    *
45    * Lists will contain the type of the list elements.
46    * Maps will contain the type of the map keys, then the type of the map values.
47    * Geometries will contain the specific geometry type.
48    *
49    * Note: geometries will always return GEOMETRY as the primary type to allow for generic matching.
50    *
51    * @param clazz class, must be valid for a SimpleFeatureType attribute
52    * @param metadata attribute metadata (user data)
53    * @return binding
54    */
55   def selectType(clazz: Class[_], metadata: jMap[_, _] = jCollections.emptyMap()): Seq[ObjectType] = {
56     clazz match {
57       case c if classOf[java.lang.String].isAssignableFrom(c) =>
58         if (metadata.get(OptJson) == "true") { Seq(STRING, JSON) } else { Seq(STRING) }
59       case c if classOf[java.lang.Integer].isAssignableFrom(c) => Seq(INT)
60       case c if classOf[java.lang.Long].isAssignableFrom(c) => Seq(LONG)
61       case c if classOf[java.lang.Float].isAssignableFrom(c) => Seq(FLOAT)
62       case c if classOf[java.lang.Double].isAssignableFrom(c) => Seq(DOUBLE)
63       case c if classOf[java.lang.Boolean].isAssignableFrom(c) => Seq(BOOLEAN)
64       case c if classOf[java.util.Date].isAssignableFrom(c) => Seq(DATE)
65       case c if classOf[UUID].isAssignableFrom(c) => Seq(UUID)
66       case c if classOf[Geometry].isAssignableFrom(c) => geometryType(c.asInstanceOf[Class[_ <: Geometry]])
67       case c if classOf[Array[Byte]].isAssignableFrom(c) => Seq(BYTES)
68       case c if classOf[jList[_]].isAssignableFrom(c) => listType(metadata)
69       case c if classOf[jMap[_, _]].isAssignableFrom(c) => mapType(metadata)
70 
71       case _ => throw new IllegalArgumentException(s"Class $clazz can't be serialized")
72     }
73   }
74 
75   private def geometryType(clazz: Class[_ <: Geometry]): Seq[ObjectType] = {
76     val subtype = clazz match {
77       case c if c == classOf[Point]              => POINT
78       case c if c == classOf[LineString]         => LINESTRING
79       case c if c == classOf[Polygon]            => POLYGON
80       case c if c == classOf[MultiLineString]    => MULTILINESTRING
81       case c if c == classOf[MultiPolygon]       => MULTIPOLYGON
82       case c if c == classOf[MultiPoint]         => MULTIPOINT
83       case c if c == classOf[GeometryCollection] => GEOMETRY_COLLECTION
84       case _                                     => GEOMETRY
85     }
86     Seq(GEOMETRY, subtype)
87   }
88 
89   private def listType(metadata: jMap[_, _]): Seq[ObjectType] = {
90     val clazz = Class.forName(metadata.get(UserDataListType).asInstanceOf[String])
91     selectType(clazz) match {
92       case Seq(binding) => Seq(LIST, binding)
93       case _ => throw new IllegalArgumentException(s"Can't serialize list sub-type of ${clazz.getName}")
94     }
95   }
96 
97   private def mapType(metadata: jMap[_, _]): Seq[ObjectType] = {
98     val keyClass   = Class.forName(metadata.get(UserDataMapKeyType).asInstanceOf[String])
99     val keyType = selectType(keyClass) match {
100       case Seq(binding) => binding
101       case _ => throw new IllegalArgumentException(s"Can't serialize map key type of ${keyClass.getName}")
102     }
103     val valueClass = Class.forName(metadata.get(UserDataMapValueType).asInstanceOf[String])
104     val valueType = selectType(valueClass) match {
105       case Seq(binding) => binding
106       case _ => throw new IllegalArgumentException(s"Can't serialize map value type of ${valueClass.getName}")
107     }
108     Seq(MAP, keyType, valueType)
109   }
