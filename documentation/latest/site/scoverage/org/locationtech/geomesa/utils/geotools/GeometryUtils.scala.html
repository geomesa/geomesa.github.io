88   /**
89     * Calculate distance from a point, in degrees.
90     *
91     * Note: normally east/west would be the largest in degrees, but sometimes it can be smaller
92     * due to variances in the ellipsoid
93     *
94     * @param point point to buffer
95     * @param meters distance to buffer, in meters
96     * @param calc calculator, for re-use
97     * @return (east/west, north/south) distance in degrees
98     */
99   def directionalDegrees(point: Point, meters: Double, calc: GeodeticCalculator): (Double, Double) = {
100     calc.setStartingGeographicPoint(point.getX, point.getY)
101     def degrees(azimuth: Double): Double = {
102       calc.setDirection(azimuth, meters)
103       val dest = calc.getDestinationGeographicPoint
104       point.distance(geoFactory.createPoint(new Coordinate(dest.getX, dest.getY)))
105     }
106     // take min distance to account for crossing the AM/poles
107     (math.min(degrees(90), degrees(-90)), math.min(degrees(0), degrees(180)))
108   }

110   /** Adds way points to Seq[Coordinates] so that they remain valid with Spatial4j, useful for BBOX */
111   def addWayPoints(coords: Seq[Coordinate]): List[Coordinate] = {
112     unfoldRight(coords) {
113       case Seq() => None
114       case Seq(pt) => Some((pt, Seq()))
115       case Seq(first, second, rest @ _*) => second.x - first.x match {
116         case dx if dx > 120 =>

142     * Returns the rough bounds of a geometry, decomposing the geometry to provide better accuracy
143     *
144     * @param geometry geometry
145     * @param maxBounds maximum number of bounds that will be returned
146     * @param maxBitResolution maximum bit resolution to decompose to
147     *                         must be between 1-63, inclusive
148     * @return seq of (xmin, ymin, xmax, ymax)
149     */
150   def bounds(geometry: Geometry, maxBounds: Int, maxBitResolution: Int): Seq[(Double, Double, Double, Double)] = {
151     if (maxBounds < 2 || GeometryUtils.isRectangular(geometry)) {
152       return Seq(bounds(geometry))
153     }

156       // use `maxBitResolution | 1` to ensure oddness, which is required by GeohashUtils
157       val resolution = ResolutionRange(0, maxBitResolution | 1, 5)
158       GeohashUtils.decomposeGeometry(geometry, maxBounds, resolution)
159     } catch {
160       case NonFatal(e) =>
161         logger.error("Error decomposing geometry, falling back to envelope bounds:", e)
162         Seq(bounds(geometry))
163     }
164   }

179   /**
180     * Checks that a polygon is rectangular and has no interior holes
181     *
182     * @param p polygon
183     * @return
184     */
185   def isRectangular(p: Polygon): Boolean = {
186     if (p.isEmpty) {
187       true
188     } else if (p.getNumInteriorRing != 0) {
189       // checks that there are no interior holes
190       false
191     } else {
192       val env = p.getEnvelopeInternal
193       val (xmin, ymin, xmax, ymax) = (env.getMinX, env.getMinY, env.getMaxX, env.getMaxY)
194 
195       // checks that all points are on the exterior envelope of the polygon
196       def cutout(c: Coordinate): Boolean = c.x != xmin && c.x != xmax && c.y != ymin && c.y != ymax
197 
198       val coords = p.getCoordinates // note: getCoordinates constructs an array so just call once
199       var i = 1
200       while (i < coords.length) {
201         val c = coords(i)
202         // checks that there aren't any cutouts or angled lines
203         if (cutout(c) || (c.x != coords(i - 1).x && c.y != coords(i - 1).y)) {
204           return false
205         }
206         i += 1
207       }
208       // check final coord cutout
209       !cutout(coords(0))
210     }
211   }

223   /**
224     * Calculate the latitude at which the segment intercepts the IDL.
225     * This function assumes that the provided points do actually cross the IDL.
226     * @param point1 The first point in the segment
227     * @param point2 The second point in the segment
228     * @return a double representing the intercept latitude
229     */
230   def calcIDLIntercept(point1: Coordinate, point2: Coordinate): Double = {
231     if (point1.x < 0) {
232       calcCrossLat(point1, new Coordinate(point2.x - 360, point2.y), -180)
233     } else {
234       calcCrossLat(point1, new Coordinate(point2.x + 360, point2.y), 180)
235     }

238   /**
239     * Calculate the latitude at which a segment intercepts a given latitude.
240     * @param point1 The first point in the segment
241     * @param point2 The second point in the segment
242     * @param crossLon The longitude of intercept
243     * @return a double representing the intercept latitude
244     */
245   def calcCrossLat(point1: Coordinate, point2: Coordinate, crossLon: Double): Double = {
246     val slope = (point1.y - point2.y) / (point1.x - point2.x)
247     val intercept = point1.y - (slope * point1.x)
248     (slope * crossLon) + intercept
249   }

251   /**
252     * Split a bounding box envelope, which may extend outside [-180,180], into one or more envelopes
253     * that are contained within [-180,180]
254     *
255     * @param envelope envelope of a bounding box
256     * @return
257     */
258   def splitBoundingBox(envelope: ReferencedEnvelope): Seq[ReferencedEnvelope] = {
259     try {
260       val crs = envelope.getCoordinateReferenceSystem
261 
262       // if the bbox is completely outside world bounds, translate to bring it back in
263       val translated = if (envelope.getMinX >= 180d) {
264         val multiplier = math.ceil((envelope.getMinX - 180d) / 360d)
265         val left = envelope.getMinX - 360d * multiplier
266         val right = envelope.getMaxX - 360d * multiplier
267         new ReferencedEnvelope(left, right, envelope.getMinY, envelope.getMaxY, crs)
268       } else if (envelope.getMaxX <= -180d) {
269         val multiplier = math.ceil((envelope.getMaxX + 180d) / -360d)
270         val left = envelope.getMinX + 360d * multiplier
271         val right = envelope.getMaxX + 360d * multiplier
272         new ReferencedEnvelope(left, right, envelope.getMinY, envelope.getMaxY, crs)
273       } else {
274         envelope
275       }
276 
277       // if the bbox extends past world bounds, split and wrap it
278       if (translated.getMinX < -180d && translated.getMaxX < 180d) {
279         val trimmed = new ReferencedEnvelope(-180d , translated.getMaxX, envelope.getMinY, envelope.getMaxY, crs)
280         val wrapped = new ReferencedEnvelope(translated.getMinX + 360d , 180d, envelope.getMinY, envelope.getMaxY, crs)
281         Seq(trimmed, wrapped)
282       } else if (translated.getMaxX > 180d && translated.getMinX > -180d) {
283         val trimmed = new ReferencedEnvelope(translated.getMinX, 180d, envelope.getMinY, envelope.getMaxY, crs)
284         val wrapped = new ReferencedEnvelope(-180d, translated.getMaxX - 360d, envelope.getMinY, envelope.getMaxY, crs)
285         Seq(trimmed, wrapped)
286       } else {
287         Seq(translated)
288       }
289     } catch {
290       case NonFatal(e) => logger.warn(s"Error splitting bounding box envelope '$envelope':", e); Seq(envelope)
291     }
292   }
