88   /**
89     * Calculate distance from a point, in degrees.
90     *
91     * Note: normally east/west would be the largest in degrees, but sometimes it can be smaller
92     * due to variances in the ellipsoid
93     *
94     * @param point point to buffer
95     * @param meters distance to buffer, in meters
96     * @param calc calculator, for re-use
97     * @return (east/west, north/south) distance in degrees
98     */
99   def directionalDegrees(point: Point, meters: Double, calc: GeodeticCalculator): (Double, Double) = {
100     calc.setStartingGeographicPoint(point.getX, point.getY)
101     def degrees(azimuth: Double): Double = {
102       calc.setDirection(azimuth, meters)
103       val dest = calc.getDestinationGeographicPoint
104       point.distance(geoFactory.createPoint(new Coordinate(dest.getX, dest.getY)))
105     }
106     // take min distance to account for crossing the AM/poles
107     (math.min(degrees(90), degrees(-90)), math.min(degrees(0), degrees(180)))
108   }

112     unfoldRight(coords) {

141   /**
142     * Returns the rough bounds of a geometry, decomposing the geometry to provide better accuracy
143     *
144     * @param geometry geometry
145     * @param maxBounds maximum number of bounds that will be returned
146     * @param maxBitResolution maximum bit resolution to decompose to
147     *                         must be between 1-63, inclusive
148     * @return seq of (xmin, ymin, xmax, ymax)
149     */
150   def bounds(geometry: Geometry, maxBounds: Int, maxBitResolution: Int): Seq[(Double, Double, Double, Double)] = {
151     if (maxBounds < 2 || GeometryUtils.isRectangular(geometry)) {
152       return Seq(bounds(geometry))
153     }
154 
155     try {
156       // use `maxBitResolution | 1` to ensure oddness, which is required by GeohashUtils
157       val resolution = ResolutionRange(0, maxBitResolution | 1, 5)
158       val geohashes = GeohashUtils.decomposeGeometry(geometry, maxBounds, resolution, relaxFit = true)
159       geohashes.map(gh => (gh.bbox.ll.getX, gh.bbox.ll.getY, gh.bbox.ur.getX, gh.bbox.ur.getY))
160     } catch {
161       case NonFatal(e) =>
162         logger.error("Error decomposing geometry, falling back to envelope bounds:", e)
163         Seq(bounds(geometry))
164     }
165   }

180   /**
181     * Checks that a polygon is rectangular and has no interior holes
182     *
183     * @param p polygon
184     * @return
185     */
186   def isRectangular(p: Polygon): Boolean = {
187     if (p.isEmpty) {
188       true
189     } else if (p.getNumInteriorRing != 0) {
190       // checks that there are no interior holes
191       false
192     } else {
193       val env = p.getEnvelopeInternal
194       val (xmin, ymin, xmax, ymax) = (env.getMinX, env.getMinY, env.getMaxX, env.getMaxY)
195 
196       // checks that all points are on the exterior envelope of the polygon
197       def cutout(c: Coordinate): Boolean = c.x != xmin && c.x != xmax && c.y != ymin && c.y != ymax
198 
199       val coords = p.getCoordinates // note: getCoordinates constructs an array so just call once
200       var i = 1
201       while (i < coords.length) {
202         val c = coords(i)
203         // checks that there aren't any cutouts or angled lines
204         if (cutout(c) || (c.x != coords(i - 1).x && c.y != coords(i - 1).y)) {
205           return false
206         }
207         i += 1
208       }
209       // check final coord cutout
210       !cutout(coords(0))
211     }
212   }

226     * This function assumes that the provided points do actually cross the IDL.

239   /**
240     * Calculate the latitude at which a segment intercepts a given latitude.
241     * @param point1 The first point in the segment
242     * @param point2 The second point in the segment
243     * @param crossLon The longitude of intercept
244     * @return a double representing the intercept latitude
245     */
246   def calcCrossLat(point1: Coordinate, point2: Coordinate, crossLon: Double): Double = {
247     val slope = (point1.y - point2.y) / (point1.x - point2.x)
248     val intercept = point1.y - (slope * point1.x)
249     (slope * crossLon) + intercept
250   }

252   /**
253     * Split a bounding box envelope, which may extend outside [-180,180], into one or more envelopes
254     * that are contained within [-180,180]
255     *
256     * @param envelope envelope of a bounding box
257     * @return
258     */
259   def splitBoundingBox(envelope: ReferencedEnvelope): Seq[ReferencedEnvelope] = {
260     try {
261       val crs = envelope.getCoordinateReferenceSystem
262 
263       // if the bbox is completely outside world bounds, translate to bring it back in
264       val translated = if (envelope.getMinX >= 180d) {
265         val multiplier = math.ceil((envelope.getMinX - 180d) / 360d)
266         val left = envelope.getMinX - 360d * multiplier
267         val right = envelope.getMaxX - 360d * multiplier
268         new ReferencedEnvelope(left, right, envelope.getMinY, envelope.getMaxY, crs)
269       } else if (envelope.getMaxX <= -180d) {
270         val multiplier = math.ceil((envelope.getMaxX + 180d) / -360d)
271         val left = envelope.getMinX + 360d * multiplier
272         val right = envelope.getMaxX + 360d * multiplier
273         new ReferencedEnvelope(left, right, envelope.getMinY, envelope.getMaxY, crs)
274       } else {
275         envelope
276       }
277 
278       // if the bbox extends past world bounds, split and wrap it
279       if (translated.getMinX < -180d && translated.getMaxX < 180d) {
280         val trimmed = new ReferencedEnvelope(-180d , translated.getMaxX, envelope.getMinY, envelope.getMaxY, crs)
281         val wrapped = new ReferencedEnvelope(translated.getMinX + 360d , 180d, envelope.getMinY, envelope.getMaxY, crs)
282         Seq(trimmed, wrapped)
283       } else if (translated.getMaxX > 180d && translated.getMinX > -180d) {
284         val trimmed = new ReferencedEnvelope(translated.getMinX, 180d, envelope.getMinY, envelope.getMaxY, crs)
285         val wrapped = new ReferencedEnvelope(-180d, translated.getMaxX - 360d, envelope.getMinY, envelope.getMaxY, crs)
286         Seq(trimmed, wrapped)
287       } else {
288         Seq(translated)
289       }
290     } catch {
291       case NonFatal(e) => logger.warn(s"Error splitting bounding box envelope '$envelope':", e); Seq(envelope)
292     }
293   }
