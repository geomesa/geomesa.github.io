342   /**
343     * Creates an immutable copy of the simple feature type
344     *
345     * Note that immutable types will never be `equals` to a mutable type, due to specific class checking
346     * in the geotools implementation
347     *
348     * Note that some parts of the feature type may still be mutable - in particular AttributeType,
349     * GeometryType and SuperType are not used by geomesa so we don't bother with them. In addition,
350     * user data keys and values may be mutable objects, so while the user data map will not change,
351     * the values inside may
352     *
353     * @param sft simple feature type
354     * @param extraData additional user data to add to the simple feature type
355     * @return immutable copy of the simple feature type
356     */
357   def immutable(
358       sft: SimpleFeatureType,
359       extraData: java.util.Map[_ <: AnyRef, _ <: AnyRef] = null): SimpleFeatureType = {
360     sft match {
361       case immutable: ImmutableSimpleFeatureType if extraData == null || extraData.isEmpty => immutable
362       case _ =>
363         val schema = new java.util.ArrayList[AttributeDescriptor](sft.getAttributeCount)
364         var geom: GeometryDescriptor = null
365         var i = 0
366         while (i < sft.getAttributeCount) {
367           val descriptor = sft.getDescriptor(i)
368           val im = immutable(descriptor)
369           if (descriptor == sft.getGeometryDescriptor) {
370             geom = im.asInstanceOf[GeometryDescriptor]
371           }
372           schema.add(im)
373           i += 1
374         }
375         val userData = Option(extraData).filterNot(_.isEmpty).map { data =>
376           val map = new java.util.HashMap[AnyRef, AnyRef](data.size() + sft.getUserData.size)
377           map.putAll(sft.getUserData)
378           map.putAll(data)
379           map
380         }
381 
382         new ImmutableSimpleFeatureType(sft.getName, schema, geom, sft.isAbstract, sft.getRestrictions, sft.getSuper,
383           sft.getDescription, userData.getOrElse(sft.getUserData))
384     }
385   }

464   /**
465     * Compare two feature types. This method compares the the schemas of each feature type, i.e. the
466     * names and type bindings of the attributes. The result will be:
467     *
468     * <ul>
469     *   <li>a positive int if type A is a sub type or reorder of type B</li>
470     *   <li>zero if type A has the same schema as type B</li>
471     *   <li>a negative int if none of the above</li>
472     * </ul>
473     *
474     * Note that this shouldn't be used for sorting.
475     *
476     * Note: GeoTools DataUtilities has a similar method that does not work correctly
477     *
478     * @param a schema A
479     * @param b schema B
480     * @return comparison
481     */
482   def compare(a: SimpleFeatureType, b: SimpleFeatureType): Int = {
483     if (a.getAttributeCount > b.getAttributeCount) {
484       return -1 // not equals, subtype or reorder
485     }
486 
487     var exact = a.getAttributeCount == b.getAttributeCount
488 
489     var i = 0
490     while (i < a.getAttributeCount) {
491       val ad = a.getDescriptor(i)
492       val bd = b.getDescriptor(i)
493       if (ad.getLocalName == bd.getLocalName) {
494         if (ad.getType.getBinding != bd.getType.getBinding) {
495           if (bd.getType.getBinding.isAssignableFrom(ad.getType.getBinding)) {
496             exact = false // we've seen a subtyping
497           } else {
498             return -1 // not a subtype/reorder
499           }
500         }
501       } else {
502         val reorder = b.getDescriptor(ad.getLocalName)
503         if (reorder == null || !reorder.getType.getBinding.isAssignableFrom(ad.getType.getBinding)) {
504           return -1 // not a subtype/reorder
505         } else {
506           exact = false // we've seen a reordering
507         }
508       }
509       i += 1
510     }
511 
512     if (exact) { 0 } else {
513       1  // we haven't hit an absolute inequality, but we've found a reordering or subtype
514     }
515   }

517   /**
518    * Compares that index tables are the same format (key and value) between the two feature types
519    *
520    * @param a first type
521    * @param b second type
522    * @return 0 if indices are the same, non-zero otherwise
523    */
524   def compareIndexConfigs(a: SimpleFeatureType, b: SimpleFeatureType): Int = {
525     def compareAttributes(i: Int): Boolean = {
526       val (ad, bd) = (a.getDescriptor(i), b.getDescriptor(i))
527       ad.getType.getBinding == bd.getType.getBinding && AttributeChecks.forall(c => c(ad) == c(bd))
528     }
529     if (a.getAttributeCount == b.getAttributeCount &&
530         IndexChecks.forall(c => c(a) == c(b)) &&
531         Range(0, a.getAttributeCount).forall(compareAttributes)) {
532       0
533     } else {
534       1
535     }
536   }

538   private def createFeatureType(namespace: String,
539                                 name: String,
540                                 spec: SimpleFeatureSpec,
541                                 factory: Option[FeatureTypeFactory] = None): SimpleFeatureType = {
542     import AttributeOptions.OptDefault
543 
544     val defaultGeom = {
545       val geomAttributes = spec.attributes.collect { case g: GeomAttributeSpec => g }
546       geomAttributes.find(_.options.get(OptDefault).exists(_.toBoolean))
547           .orElse(geomAttributes.headOption)
548           .map(_.name)
549     }
550     val defaultDate = if (spec.options.get(IndexIgnoreDtg).exists(toBoolean)) { None } else {
551       val dateAttributes = spec.attributes.filter(a => classOf[Date].isAssignableFrom(a.clazz))
552       spec.options.get(DefaultDtgField).flatMap(dtg => dateAttributes.find(_.name == dtg))
553           .orElse(dateAttributes.find(_.options.get(OptDefault).exists(_.toBoolean)))
554           .orElse(dateAttributes.headOption)
555           .map(_.name)
556     }
557 
558     val b = factory.map(new SimpleFeatureTypeBuilder(_)).getOrElse(new SimpleFeatureTypeBuilder())
559     b.setNamespaceURI(namespace)
560     b.setName(name)
561     b.addAll(spec.attributes.map(_.toDescriptor).asJava)
562     defaultGeom.foreach(b.setDefaultGeometry)
563 
564     val sft = b.buildFeatureType()
565     sft.getUserData.putAll(spec.options.asJava)
566     defaultDate.foreach(sft.setDtgField)
567     sft
568   }
