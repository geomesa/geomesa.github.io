341   /**
342     * Creates an immutable copy of the simple feature type
343     *
344     * Note that immutable types will never be `equals` to a mutable type, due to specific class checking
345     * in the geotools implementation
346     *
347     * Note that some parts of the feature type may still be mutable - in particular AttributeType,
348     * GeometryType and SuperType are not used by geomesa so we don't bother with them. In addition,
349     * user data keys and values may be mutable objects, so while the user data map will not change,
350     * the values inside may
351     *
352     * @param sft simple feature type
353     * @param extraData additional user data to add to the simple feature type
354     * @return immutable copy of the simple feature type
355     */
356   def immutable(
357       sft: SimpleFeatureType,
358       extraData: java.util.Map[_ <: AnyRef, _ <: AnyRef] = null): SimpleFeatureType = {
359     sft match {
360       case immutable: ImmutableSimpleFeatureType if extraData == null || extraData.isEmpty => immutable
361       case _ =>
362         val schema = new java.util.ArrayList[AttributeDescriptor](sft.getAttributeCount)
363         var geom: GeometryDescriptor = null
364         var i = 0
365         while (i < sft.getAttributeCount) {
366           val descriptor = sft.getDescriptor(i)
367           val im = immutable(descriptor)
368           if (descriptor == sft.getGeometryDescriptor) {
369             geom = im.asInstanceOf[GeometryDescriptor]
370           }
371           schema.add(im)
372           i += 1
373         }
374         val userData = Option(extraData).filterNot(_.isEmpty).map { data =>
375           val map = new java.util.HashMap[AnyRef, AnyRef](data.size() + sft.getUserData.size)
376           map.putAll(sft.getUserData)
377           map.putAll(data)
378           map
379         }
380 
381         new ImmutableSimpleFeatureType(sft.getName, schema, geom, sft.isAbstract, sft.getRestrictions, sft.getSuper,
382           sft.getDescription, userData.getOrElse(sft.getUserData))
383     }
384   }

463   /**
464     * Compare two feature types. This method compares the the schemas of each feature type, i.e. the
465     * names and type bindings of the attributes. The result will be:
466     *
467     * <ul>
468     *   <li>a positive int if type A is a sub type or reorder of type B</li>
469     *   <li>zero if type A has the same schema as type B</li>
470     *   <li>a negative int if none of the above</li>
471     * </ul>
472     *
473     * Note that this shouldn't be used for sorting.
474     *
475     * Note: GeoTools DataUtilities has a similar method that does not work correctly
476     *
477     * @param a schema A
478     * @param b schema B
479     * @return comparison
480     */
481   def compare(a: SimpleFeatureType, b: SimpleFeatureType): Int = {
482     if (a.getAttributeCount > b.getAttributeCount) {
483       return -1 // not equals, subtype or reorder
484     }
485 
486     var exact = a.getAttributeCount == b.getAttributeCount
487 
488     var i = 0
489     while (i < a.getAttributeCount) {
490       val ad = a.getDescriptor(i)
491       val bd = b.getDescriptor(i)
492       if (ad.getLocalName == bd.getLocalName) {
493         if (ad.getType.getBinding != bd.getType.getBinding) {
494           if (bd.getType.getBinding.isAssignableFrom(ad.getType.getBinding)) {
495             exact = false // we've seen a subtyping
496           } else {
497             return -1 // not a subtype/reorder
498           }
499         }
500       } else {
501         val reorder = b.getDescriptor(ad.getLocalName)
502         if (reorder == null || !reorder.getType.getBinding.isAssignableFrom(ad.getType.getBinding)) {
503           return -1 // not a subtype/reorder
504         } else {
505           exact = false // we've seen a reordering
506         }
507       }
508       i += 1
509     }
510 
511     if (exact) { 0 } else {
512       1  // we haven't hit an absolute inequality, but we've found a reordering or subtype
513     }
514   }

516   /**
517    * Compares that index tables are the same format (key and value) between the two feature types
518    *
519    * @param a first type
520    * @param b second type
521    * @return 0 if indices are the same, non-zero otherwise

523   def compareIndexConfigs(a: SimpleFeatureType, b: SimpleFeatureType): Int = {
524     def compareAttributes(i: Int): Boolean = {
525       val (ad, bd) = (a.getDescriptor(i), b.getDescriptor(i))
526       ad.getType.getBinding == bd.getType.getBinding && AttributeChecks.forall(c => c(ad) == c(bd))
527     }
528     if (a.getAttributeCount == b.getAttributeCount &&
529         IndexChecks.forall(c => c(a) == c(b)) &&
530         Range(0, a.getAttributeCount).forall(compareAttributes)) {
531       0
532     } else {
533       1
534     }
535   }

537   private def createFeatureType(namespace: String,
538                                 name: String,
539                                 spec: SimpleFeatureSpec,
540                                 factory: Option[FeatureTypeFactory] = None): SimpleFeatureType = {
541     import AttributeOptions.OptDefault
542 
543     val defaultGeom = {
544       val geomAttributes = spec.attributes.collect { case g: GeomAttributeSpec => g }
545       geomAttributes.find(_.options.get(OptDefault).exists(_.toBoolean))
546           .orElse(geomAttributes.headOption)
547           .map(_.name)
548     }
549     val defaultDate = if (spec.options.get(IndexIgnoreDtg).exists(toBoolean)) { None } else {
550       val dateAttributes = spec.attributes.filter(a => classOf[Date].isAssignableFrom(a.clazz))
551       spec.options.get(DefaultDtgField).flatMap(dtg => dateAttributes.find(_.name == dtg))
552           .orElse(dateAttributes.find(_.options.get(OptDefault).exists(_.toBoolean)))
553           .orElse(dateAttributes.headOption)
554           .map(_.name)
555     }
556 
557     val b = factory.map(new SimpleFeatureTypeBuilder(_)).getOrElse(new SimpleFeatureTypeBuilder())
558     b.setNamespaceURI(namespace)
559     b.setName(name)
560     b.addAll(spec.attributes.map(_.toDescriptor).asJava)
561     defaultGeom.foreach(b.setDefaultGeometry)
562 
563     val sft = b.buildFeatureType()
564     sft.getUserData.putAll(spec.options.asJava)
565     defaultDate.foreach(sft.setDtgField)
566     sft
567   }
