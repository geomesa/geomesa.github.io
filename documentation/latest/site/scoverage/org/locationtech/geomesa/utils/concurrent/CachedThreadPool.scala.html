18 /**
19  * Executor service that will grow up to the number of threads specified.
20  *
21  * Assertion: core java executors offer 3 features that we want, but you can only use two of them at once:
22  *   1. Add threads as needed but re-use idle threads if available
23  *   2. Limit the total number of threads available
24  *   3. Accept and queue tasks once the thread limit is reached
25  *
26  * This executor allows us to do all three. It is backed by an unlimited, cached thread pool that will handle
27  * re-use and expiration of threads, while it tracks the number of threads used. It is fairly lightweight, and
28  * can be instantiated for a given task and then shutdown after use.
29  *
30  * @param maxThreads max threads to use at once
31  */
32 class CachedThreadPool(maxThreads: Int) extends AbstractExecutorService with LazyLogging {
33 
34   @volatile
35   private var available = maxThreads
36   private val queue = new java.util.LinkedList[TrackableFutureTask[_]]()
37   private val tasks = new java.util.HashSet[TrackableFutureTask[_]]()
38   private val stopped = new AtomicBoolean(false)
39   private val lock = new ReentrantLock()
40   private val done = lock.newCondition()
41 
42   override def shutdown(): Unit = stopped.set(true)
43 
44   override def shutdownNow(): java.util.List[Runnable] = {
45     stopped.set(true)
46     lock.lock()
47     try {
48       val waiting = new java.util.ArrayList[Runnable](queue)
49       queue.clear()
50       // copy the running tasks to prevent concurrent modification errors in the synchronous cancel call
51       val running = new java.util.ArrayList[Future[_]](tasks).iterator()
52       while (running.hasNext) {
53         running.next.cancel(true)
54       }
55       waiting
56     } finally {
57       lock.unlock()
58     }
59   }
60 
61   override def isShutdown: Boolean = stopped.get
62 
63   override def isTerminated: Boolean = stopped.get && available == maxThreads // should be safe to read a volatile primitive
64 
65   override def awaitTermination(timeout: Long, unit: TimeUnit): Boolean = {
66     lock.lock()
67     try {
68       if (isTerminated) { true } else {
69         done.await(timeout, unit)
70         isTerminated
71       }
72     } finally {
73       lock.unlock()
74     }
75   }
76 
77   override def execute(command: Runnable): Unit = {
78     if (stopped.get) {
79       throw new RejectedExecutionException("Trying to execute a task but executor service is shut down")
80     }
81     val task = command match {
82       case t: TrackableFutureTask[_] => t
83       case c => newTaskFor(c, null)
84     }
85     runOrQueueTask(task)
86   }
87 
88   private def runOrQueueTask(task: TrackableFutureTask[_]): Unit = {
89     lock.lock()
90     try {
91       if (available > 0) {
92         // note that we could fairly easily create a global thread limit by backing this with a different pool
93         try { CachedThreadPool.pool.execute(task) } catch {
94           case e: RejectedExecutionException =>
95             // we still need to execute the task to fulfill the executor API
96             logger.warn(
97               "CachedThreadPool rejected queued task (likely due to shutdown)," +
98                 s"creating new single thread executor: $task: $e")
99             val pool = Executors.newSingleThreadExecutor()
100             pool.execute(task)
101             pool.shutdown()
102         }
103         available -= 1
104         tasks.add(task)
105       } else {
106         queue.offer(task) // unbounded queue so should always succeed
107       }
108     } finally {
109       lock.unlock()
110     }
111   }
112 
113   // noinspection SameParameterValue
114   override protected def newTaskFor[T](runnable: Runnable, value: T): TrackableFutureTask[T] =
115     new TrackableFutureTask[T](runnable, value)
116 
117   class TrackableFutureTask[T](runnable: Runnable, result: T) extends FutureTask[T](runnable, result) {
118     override def done(): Unit = {
119       lock.lock()
120       try {
121         available += 1
122         tasks.remove(this)
123         val next = queue.poll()
124         if (next != null) {
125           runOrQueueTask(next) // note: this may briefly use more than maxThreads as this thread finishes up
126         } else if (isTerminated) {
127           CachedThreadPool.this.done.signalAll()
128         }
129       } finally {
130         lock.unlock()
131       }
132     }
133 
134     override def toString: String = s"TrackableFutureTask[$runnable]"
135   }
136 }

138 object CachedThreadPool {
139 
140   // unlimited size but re-uses cached threads
141   private val pool = ExitingExecutor(Executors.newCachedThreadPool().asInstanceOf[ThreadPoolExecutor])
142 
143   /**
144    * Execute a single command in a potentially cached thread
145    *
146    * @param command command
147    */
148   def execute(command: Runnable): Unit = pool.execute(command)
149 
150   /**
151    * Submit a single command to run in a potentially cached thread
152    *
153    * @param command command
154    * @return
155    */
156   def submit(command: Runnable): Future[_] = pool.submit(command)
157 
158   /**
159    * Run commands in a executor with a fixed level of concurrency, potentially re-using threads. Will block
160    * until any submitted tasks are complete.
161    *
162    * @param threads number of concurrent threads to use
163    * @param func function
164    */
165   def executor(threads: Int)(func: ExecutorService => Unit): Unit = {
166     val executor = new CachedThreadPool(threads)
167     val shutdown = sys.addShutdownHook(executor.shutdownNow())
168     try { func(executor) } finally {
169       executor.shutdown()
170     }
171     while (!executor.isTerminated) {
172       executor.awaitTermination(1, TimeUnit.MINUTES)
173     }
174     Try(shutdown.remove())
175   }
