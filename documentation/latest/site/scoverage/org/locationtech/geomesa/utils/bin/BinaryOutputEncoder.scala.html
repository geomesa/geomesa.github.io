39   def encode(f: Iterator[SimpleFeature], os: OutputStream, sort: Boolean): Long = {
40     if (sort) {
41       val byteStream = new ByteArrayOutputStream
42       val callback = new ByteStreamCallback(byteStream)
43       f.foreach(toValues(_, callback))
44       val count = callback.result
45       val bytes = byteStream.toByteArray
46       val size = (bytes.length / count).toInt
47       bytes.grouped(size).toSeq.sorted(BinaryOutputEncoder.DateOrdering).foreach(os.write)
48       count
49     } else {
50       val callback = new ByteStreamCallback(os)
51       f.foreach(toValues(_, callback))
52       callback.result
53     }
54   }

109   /**
110     * BIN queries pack multiple records into each feature. To count the records, we have to count
111     * the total bytes coming back, instead of the number of features
112     *
113     * @param iter aggregated bin iter
114     * @param maxFeatures max features
115     * @param hasLabel bin results have labels (extended format) or not
116     */
117   class FeatureLimitingIterator(iter: CloseableIterator[SimpleFeature], maxFeatures: Int, hasLabel: Boolean)
118       extends CloseableIterator[SimpleFeature] {
119 
120     private val bytesPerHit = if (hasLabel) { 24 } else { 16 }
121     private var seen = 0L
122 
123     override def hasNext: Boolean = seen < maxFeatures && iter.hasNext
124 
125     override def next(): SimpleFeature = {
126       if (hasNext) {
127         val sf = iter.next()
128         val bytes = sf.getAttribute(0).asInstanceOf[Array[Byte]]
129         val count = bytes.length / bytesPerHit
130         seen += count
131         if (seen > maxFeatures) {
132           // remove the extra aggregated features so that we hit our exact feature limit
133           val trimmed = Array.ofDim[Byte]((count - (seen - maxFeatures).toInt) * bytesPerHit)
134           System.arraycopy(bytes, 0, trimmed, 0, trimmed.length)
135           sf.setAttribute(0, trimmed)
136         }
137         sf
138       } else {
139         Iterator.empty.next()
140       }
141     }
142 
143     override def close(): Unit = iter.close()
144   }

161       var sum = 0L

181     val lon = buf.getFloat
182     if (encoded.length > 16) {
183       val label = buf.getLong
184       callback(trackId, lat, lon, time, label)
185     } else {
186       callback(trackId, lat, lon, time)
187     }

202     * Creates the function to map a simple feature to a bin-encoded buffer

204     * @param sft simple feature type
205     * @param options encoding options
206     * @return

208   private def toValues(sft: SimpleFeatureType, options: EncodingOptions): ToValues = {
209 
210     val geomIndex = options.geomField.getOrElse(sft.getGeomIndex)
211     if (geomIndex == -1) {
212       throw new IllegalArgumentException(s"Invalid geometry field requested for feature type ${sft.getTypeName}")
213     }
214     val dtgIndex = options.dtgField.orElse(sft.getDtgIndex).getOrElse(-1)
215     if (dtgIndex == -1) {
216       throw new RuntimeException(s"Invalid date field requested for feature type ${sft.getTypeName}")
217     }
218     val isSingleDate = classOf[Date].isAssignableFrom(sft.getDescriptor(dtgIndex).getType.getBinding)
219     val axisOrder = options.axisOrder.getOrElse(AxisOrder.LonLat)
220 
221     val (isPoint, isLineString) = {
222       val binding = sft.getDescriptor(geomIndex).getType.getBinding
223       (binding == classOf[Point], binding == classOf[LineString])
224     }
225 
226     // noinspection ExistsEquals
227     if (options.trackIdField.exists(_ == -1)) {
228       throw new IllegalArgumentException(s"Invalid track field requested for feature type ${sft.getTypeName}")
229     } else if (options.labelField.exists(_ == -1)) {
230       throw new IllegalArgumentException(s"Invalid label field requested for feature type ${sft.getTypeName}")
231     } else if (!isSingleDate) {
232       if (isLineString) {
233         val dtgField = sft.getDescriptor(dtgIndex).getLocalName
234         val sftAttributes = SimpleFeatureSpecParser.parse(SimpleFeatureTypes.encodeType(sft)).attributes
235         sftAttributes.find(_.name == dtgField).foreach { spec =>
236           if (!spec.isInstanceOf[ListAttributeSpec] ||
237                 !classOf[Date].isAssignableFrom(spec.asInstanceOf[ListAttributeSpec].subClass)) {
238             throw new RuntimeException(s"Invalid date field requested for feature type ${sft.getTypeName}")
239           }
240         }
241       } else {
242         throw new RuntimeException(s"Invalid date field requested for feature type ${sft.getTypeName}")
243       }
244     }
245 
246     // gets the track id from a feature
247     val getTrackId: (SimpleFeature) => Int = options.trackIdField match {
248       case None => (f) => f.getID.hashCode
249       case Some(trackId) => convertToTrack(_, trackId)
250     }
251 
252     // gets the label from a feature
253     val getLabelOption: Option[(SimpleFeature) => Long] = options.labelField.map { labelIndex =>
254       convertToLabel(_, labelIndex)
255     }
256 
257     if (isLineString) {
258       // for linestrings, we return each point - use an array so we get constant-time lookup
259       // depending on srs requested and wfs versions, axis order can be flipped
260       val getLineLatLon: (SimpleFeature) => Array[(Float, Float)] = axisOrder match {
261         case LatLon => lineToXY(_, geomIndex)
262         case LonLat => lineToYX(_, geomIndex)
263       }
264 
265       if (isSingleDate) {
266         getLabelOption match {
267           case None => new ToValuesLines(getTrackId, getLineLatLon, dtgIndex)
268           case Some(getLabel) => new ToValuesLinesLabels(getTrackId, getLineLatLon, getLabel, dtgIndex)
269         }
270       } else {
271         // for line strings, we need an array of dates corresponding to the points in the line
272         val getLineDtg: (SimpleFeature) => Array[Long] = dateArray(_, dtgIndex)
273         getLabelOption match {
274           case None => new ToValuesLinesDates(getTrackId, getLineLatLon, getLineDtg)
275           case Some(getLabel) => new ToValuesLinesDatesLabels(getTrackId, getLineLatLon, getLineDtg, getLabel)
276         }
277       }
278     } else {
279       // get lat/lon as floats
280       // depending on srs requested and wfs versions, axis order can be flipped
281       val getLatLon: (SimpleFeature) => (Float, Float) = (isPoint, axisOrder) match {
282         case (true,  LatLon) => pointToXY(_, geomIndex)
283         case (true,  LonLat) => pointToYX(_, geomIndex)
284         case (false, LatLon) => geomToXY(_, geomIndex)
285         case (false, LonLat) => geomToYX(_, geomIndex)
286       }
287 
288       getLabelOption match {
289         case None => new ToValuesPoints(getTrackId, getLatLon, dtgIndex)
290         case Some(getLabel) => new ToValuesPointsLabels(getTrackId, getLatLon, getLabel, dtgIndex)
291       }
292     }
293   }
