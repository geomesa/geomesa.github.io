18 /**
19   * Creates an index suitable for geometries with extents. The index is broken up into tiers, where each
20   * geometry is assigned a tier based on its envelope size. When querying, each tier must be evaluated, so
21   * if possible tiers should be matched closely to the envelopes of the entries.
22   *
23   * Values are indexed by the centroid of their envelope. When querying, the bounding box is expanded based
24   * on the max envelope size of the tier, to ensure that all potential results are found. Thus, the false
25   * positive rate tends to go up with larger tiers, and post-filtering is recommended.
26   *
27   * @param sizes (x, y) max envelope size for each tier
28   * @param xBucketMultiplier multiplier for number of x buckets to create per tier
29   * @param yBucketMultiplier multiplier for number of y buckets to create per tier
30   * @param extents total area being indexed
31   * @tparam T item type
32   */
33 class SizeSeparatedBucketIndex[T](sizes: Seq[(Double, Double)] = SizeSeparatedBucketIndex.DefaultTiers,
34                                   xBucketMultiplier: Double = 1,
35                                   yBucketMultiplier: Double = 1,
36                                   extents: Envelope = new Envelope(-180.0, 180.0, -90.0, 90.0))
37     extends SpatialIndex[T] with StrictLogging {
38 
39   // TODO https://geomesa.atlassian.net/browse/GEOMESA-2323 better anti-meridian handling
40 
41   require(sizes.nonEmpty, "No valid tier sizes specified")
42   require(sizes.lengthCompare(1) == 0 ||
43       sizes.sliding(2).forall { case Seq((x1, y1), (x2, y2)) => x1 <= x2 && y1 <= y2 },
44     "Tiers must be ordered by increasing size")
45 
46   // note: for point ops, we always use the first (smallest) tier
47 
48   private val tiers = sizes.map { case (width, height) =>
49     val xSize = math.ceil(extents.getWidth * xBucketMultiplier / width).toInt
50     val ySize = math.ceil(extents.getHeight * yBucketMultiplier / height).toInt
51     // create the buckets up front to avoid having to synchronize the whole array
52     // we use a ConcurrentHashMap, which gives us iterators that aren't affected by modifications to the backing map
53     val buckets = Array.fill(xSize, ySize)(new ConcurrentHashMap[String, T]())
54     logger.debug(s"Creating tier for size ($width $height) with buckets [${xSize}x$ySize]")
55     new Tier(width, height, buckets, new GridSnap(extents, xSize, ySize))
56   }
57

59   override def insert(geom: Geometry, key: String, value: T): Unit = {
60     val envelope = geom.getEnvelopeInternal
61     val tier = selectTier(envelope)
62     // volatile reads should be cheaper than writes, so only update the variable if necessary
63     if (tier.empty) {
64       tier.empty = false
65     }
66     tier.bucket(envelope).put(key, value)
67   }
68 
69   override def remove(geom: Geometry, key: String): T = {
70     val envelope = geom.getEnvelopeInternal
71     selectTier(envelope).bucket(envelope).remove(key)
72   }
73 
74   override def get(geom: Geometry, key: String): T = {
75     val envelope = geom.getEnvelopeInternal
76     selectTier(envelope).bucket(envelope).get(key)
77   }
78 
79   override def query(xmin: Double, ymin: Double, xmax: Double, ymax: Double): Iterator[T] =
80     tiers.iterator.flatMap(_.iterator(xmin, ymin, xmax, ymax))
81 
82   override def query(): Iterator[T] = query(extents.getMinX, extents.getMinY, extents.getMaxX, extents.getMaxY)
83 
84   override def size(): Int = {
85     var size = 0
86     tiers.foreach(tier => size += tier.size())
87     size
88   }
89 
90   override def clear(): Unit = tiers.foreach(_.clear())
91 
92   private def selectTier(envelope: Envelope): Tier = {
93     val width = envelope.getWidth
94     val height = envelope.getHeight
95     tiers.find(t => t.maxSizeX >= width && t.maxSizeY >= height).getOrElse {
96       throw new IllegalArgumentException(s"Envelope $envelope exceeds the max tier size ${sizes.last}")
97     }
98   }
99 
100   private class Tier(val maxSizeX: Double,
101                      val maxSizeY: Double,
102                      buckets: Array[Array[ConcurrentHashMap[String, T]]],
103                      gridSnap: GridSnap) {
104 
105     // we can safely use volatile instead of synchronized here, as this is a primitive boolean whose
106     // state doesn't depend on its own value
107     @volatile
108     var empty: Boolean = true
109 
110     private val maxX = buckets.length - 1
111     private val maxY = buckets(0).length - 1
112 
113     def bucket(x: Double, y: Double): ConcurrentHashMap[String, T] = buckets(snapX(x))(snapY(y))
114 
115     // the bucket is selected based on the envelope centroid
116     def bucket(envelope: Envelope): ConcurrentHashMap[String, T] =
117       buckets(snapX((envelope.getMinX + envelope.getMaxX) / 2.0))(snapY((envelope.getMinY + envelope.getMaxY) / 2.0))
118 
119     def iterator(xmin: Double, ymin: Double, xmax: Double, ymax: Double): Iterator[T] =
120       if (empty) { Iterator.empty } else { new TierIterator(xmin, ymin, xmax, ymax) }
121 
122     def size(): Int = {
123       if (empty) { 0 } else {
124         var size = 0
125         var i = 0
126         while (i <= maxX) {
127           var j = 0
128           while (j <= maxY) {
129             size += buckets(i)(j).size()
130             j += 1
131           }
132           i += 1
133         }
134         size
135       }
136     }
137 
138     def clear(): Unit = {
139       var i = 0
140       while (i <= maxX) {
141         var j = 0
142         while (j <= maxY) {
143           buckets(i)(j).clear()
144           j += 1
145         }
146         i += 1
147       }
148     }
149 
150     private def snapX(x: Double): Int = {
151       val i = gridSnap.i(x)
152       if (i != -1) { i } else if (x < extents.getMinX) { 0 } else { maxX }
153     }
154 
155     private def snapY(y: Double): Int = {
156       val j = gridSnap.j(y)
157       if (j != -1) { j } else if (y < extents.getMinY) { 0 } else { maxY }
158     }
159 
160     /**
161       * Iterator over a range of buckets
162       */
163     class TierIterator (xmin: Double, ymin: Double, xmax: Double, ymax: Double) extends Iterator[T] {
164 
165       private val maxi = snapX(xmax + maxSizeX)
166       private val minj = snapY(ymin - maxSizeY)
167       private val maxj = snapY(ymax + maxSizeY)
168 
169       private var i = snapX(xmin - maxSizeX)
170       private var j = minj
171       private var iter = buckets(i)(j).values.iterator() // note: `.values` is a cached view
172 
173       @tailrec
174       override final def hasNext: Boolean = iter.hasNext || {
175         if (i == maxi && j == maxj) { false } else {
176           if (j < maxj) {
177             j += 1
178           } else {
179             j = minj
180             i += 1
181           }
182           iter = buckets(i)(j).values.iterator() // note: `.values` is a cached view
183           hasNext
184         }
185       }
186 
187       override def next(): T = iter.next()
188     }
189   }
190 }
