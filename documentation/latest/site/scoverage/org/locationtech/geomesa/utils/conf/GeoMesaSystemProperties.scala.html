19   case class SystemProperty(property: String, default: String = null) {
20 
21     val threadLocalValue = new InheritableThreadLocal[String]()
22 
23     def set(value: String): Unit = System.setProperty(property, value)
24 
25     def clear(): Unit = System.clearProperty(property)
26 
27     def get: String = ConfigLoader.Config.get(property) match {
28       case Some((value, true))  => value // final value - can't be overridden
29       case Some((value, false)) => fromSysProps.getOrElse(value)
30       case None => fromSysProps.getOrElse(default)
31     }
32 
33     def option: Option[String] = Option(get)
34 
35     def toDuration: Option[Duration] = option.flatMap { value =>
36       Try(DurationParsing.caseInsensitive(value)) match {
37         case Success(v) if v.isFinite && v > Duration.Zero => Some(v)
38         case _ =>
39           logger.warn(s"Invalid duration for property $property: $value")
40           Option(default).map(Duration.apply)
41       }
42     }
43 
44     def toMillis: Option[Long] = toDuration.map(_.toMillis)
45 
46     def toUnboundedDuration: Option[Duration] = option.flatMap { value =>
47       Try(DurationParsing.caseInsensitive(value)) match {
48         case Success(v) => Some(v)
49         case _ =>
50           logger.warn(s"Invalid duration for property $property: $value")
51           Option(default).map(Duration.apply)
52       }
53     }
54 
55     def toBytes: Option[Long] = option.flatMap { value =>
56       Suffixes.Memory.bytes(value) match {
57         case Success(b) => Some(b)
58         case Failure(e) =>
59           logger.warn(s"Invalid bytes for property $property=$value: ${e.toString}")
60           Option(default).flatMap(Suffixes.Memory.bytes(_).toOption)
61       }
62     }
63 
64     def toInt: Option[Int] = option.flatMap { value =>
65       Try(value.toInt) match {

100     def toBoolean: Option[Boolean] = option.flatMap { value =>
101       Try(value.toBoolean) match {
102         case Success(v) => Some(v)
103         case Failure(_) =>
104           logger.warn(s"Invalid Boolean for property $property: $value")
105           Option(default).map(java.lang.Boolean.parseBoolean)
106       }
107     }
108 
109     private def fromSysProps: Option[String] =
110       Option(threadLocalValue.get).orElse(sys.props.get(property).filterNot(_.isEmpty))
111   }
