47   override def getHandle(path: String): FileHandle = {
48     val p = new Path(path)
49     val fs = getSharedFs(p)
50     PathUtils.getUncompressedExtension(p.getName).toLowerCase(Locale.US) match {
51       case TAR       => new HadoopTarHandle(fs, p)
52       case ZIP | JAR => new HadoopZipHandle(fs, p)
53       case _         => new HadoopFileHandle(fs, p)
54     }
55   }

57   // based on logic from hadoop FileInputFormat
58   override def interpretPath(path: String): Seq[FileHandle] = {
59     val p = new Path(path)
60     val fs = getSharedFs(p)
61     val files = fs.globStatus(p, HiddenFileFilter)
62 
63     if (files == null) {
64       throw new IllegalArgumentException(s"Input path does not exist: $path")
65     } else if (files.isEmpty) {
66       throw new IllegalArgumentException(s"Input path does not match any files: $path")
67     }
68 
69     val remaining = scala.collection.mutable.Queue(files: _*)
70     val result = ListBuffer.empty[FileHandle]
71 
72     while (remaining.nonEmpty) {
73       val file = remaining.dequeue()
74       if (file.isDirectory) {
75         if (recursive) {
76           val children = fs.listLocatedStatus(file.getPath)
77           val iter = new Iterator[LocatedFileStatus] {
78             override def hasNext: Boolean = children.hasNext
79             override def next(): LocatedFileStatus = children.next
80           }
81           remaining ++= iter.filter(f => HiddenFileFilter.accept(f.getPath))
82         }
83       } else {
84         PathUtils.getUncompressedExtension(file.getPath.getName).toLowerCase(Locale.US) match {
85           case TAR       => result += new HadoopTarHandle(fs, file.getPath)
86           case ZIP | JAR => result += new HadoopZipHandle(fs, file.getPath)
87           case _         => result += new HadoopFileHandle(fs, file.getPath)
88         }
89       }
90     }
91 
92     result.result
93   }

101   private def getSharedFs(path: Path): FileSystem = {
102     // standardize the uri so we get the same cached instance
103     val uri = path.toUri
104     val root = if (uri.getScheme == null || uri.getAuthority == null) { uri } else { uri.resolve("/") }
105     FileSystem.get(root, conf)
106   }

138   class HadoopFileHandle(fs: FileSystem, file: Path) extends FileHandle {
139 
140     override def path: String = file.toString
141 
142     override def exists: Boolean = fs.exists(file)
143 
144     override def length: Long = if (exists) { fs.getFileStatus(file).getLen } else { 0L }

146     override def open: CloseableIterator[(Option[String], InputStream)] = {
147       val is = PathUtils.handleCompression(fs.open(file), file.getName)
148       CloseableIterator.single(None -> is, is.close())
149     }

151     override def write(mode: CreateMode): OutputStream = {
152       mode.validate()
153       if (mode.append) {
154         fs.append(file)
155       } else {
156         fs.create(file, mode.overwrite) // TODO do we need to hsync/hflush?
157       }
158     }
159 
160     override def delete(recursive: Boolean): Unit = {
161       if (!fs.delete(file, recursive)) {
162         throw new IOException(s"Could not delete file: $path")
163       }
164     }

167   class HadoopZipHandle(fs: FileSystem, file: Path) extends HadoopFileHandle(fs, file) {
168     override def open: CloseableIterator[(Option[String], InputStream)] = {
169       // we have to read the bytes into memory to get random access reads
170       val bytes = WithClose(PathUtils.handleCompression(fs.open(file), file.getName)) { is =>
171         IOUtils.toByteArray(is)
172       }
173       new ZipFileIterator(new ZipFile(new SeekableInMemoryByteChannel(bytes)), file.toString)
174     }
175 
176     override def write(mode: CreateMode): OutputStream =
177       factory.createArchiveOutputStream(ArchiveStreamFactory.ZIP, super.write(mode))
178   }
179 
180   class HadoopTarHandle(fs: FileSystem, file: Path) extends HadoopFileHandle(fs, file) {
181     override def open: CloseableIterator[(Option[String], InputStream)] = {
182       val uncompressed = PathUtils.handleCompression(fs.open(file), file.getName)
183       val archive: ArchiveInputStream[_ <: ArchiveEntry] =
184         factory.createArchiveInputStream(ArchiveStreamFactory.TAR, uncompressed)
185       new ArchiveFileIterator(archive, file.toString)
186     }
187 
188     override def write(mode: CreateMode): OutputStream =
189       factory.createArchiveOutputStream(ArchiveStreamFactory.TAR, super.write(mode))
190   }
