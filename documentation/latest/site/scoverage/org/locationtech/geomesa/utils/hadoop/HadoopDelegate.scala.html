48   override def getHandle(path: String): FileHandle = {
49     val p = new Path(path)
50     val fc = FileContext.getFileContext(p.toUri, conf)
51     PathUtils.getUncompressedExtension(p.getName).toLowerCase(Locale.US) match {
52       case TAR       => new HadoopTarHandle(fc, p)
53       case ZIP | JAR => new HadoopZipHandle(fc, p)
54       case _         => new HadoopFileHandle(fc, p)
55     }
56   }

58   // based on logic from hadoop FileInputFormat
59   override def interpretPath(path: String): Seq[FileHandle] = {
60     val p = new Path(path)
61     val fc = FileContext.getFileContext(p.toUri, conf)
62     val files = fc.util.globStatus(p, HiddenFileFilter)
63 
64     if (files == null) {
65       throw new IllegalArgumentException(s"Input path does not exist: $path")
66     } else if (files.isEmpty) {
67       throw new IllegalArgumentException(s"Input path does not match any files: $path")
68     }
69 
70     val remaining = scala.collection.mutable.Queue(files: _*)
71     val result = ListBuffer.empty[FileHandle]
72 
73     while (remaining.nonEmpty) {
74       val file = remaining.dequeue()
75       if (file.isDirectory) {
76         if (recursive) {
77           val children = fc.listLocatedStatus(file.getPath)
78           val iter = new Iterator[LocatedFileStatus] {
79             override def hasNext: Boolean = children.hasNext
80             override def next(): LocatedFileStatus = children.next
81           }
82           remaining ++= iter.filter(f => HiddenFileFilter.accept(f.getPath))
83         }
84       } else {
85         PathUtils.getUncompressedExtension(file.getPath.getName).toLowerCase(Locale.US) match {
86           case TAR       => result += new HadoopTarHandle(fc, file.getPath)
87           case ZIP | JAR => result += new HadoopZipHandle(fc, file.getPath)
88           case _         => result += new HadoopFileHandle(fc, file.getPath)
89         }
90       }
91     }
92 
93     result.result
94   }

132   class HadoopFileHandle(fc: FileContext, file: Path) extends FileHandle {
133 
134     override def path: String = file.toString
135 
136     override def exists: Boolean = fc.util.exists(file)
137 
138     override def length: Long = if (exists) { fc.getFileStatus(file).getLen } else { 0L }
139 
140     override def open: CloseableIterator[(Option[String], InputStream)] = {
141       val is = PathUtils.handleCompression(fc.open(file), file.getName)
142       CloseableIterator.single(None -> is, is.close())
143     }
144 
145     override def write(mode: CreateMode, createParents: Boolean): OutputStream = {
146       mode.validate()
147       val flags = java.util.EnumSet.noneOf(classOf[CreateFlag])
148       if (mode.append) {
149         flags.add(CreateFlag.APPEND)
150       } else if (mode.overwrite) {
151         flags.add(CreateFlag.OVERWRITE)
152       }
153       if (mode.create) {
154         flags.add(CreateFlag.CREATE)
155       }
156       val ops = if (createParents) { CreateOpts.createParent() } else { CreateOpts.donotCreateParent() }
157       fc.create(file, flags, ops) // TODO do we need to hsync/hflush?
158     }
159 
160     override def delete(recursive: Boolean): Unit = {
161       if (!fc.delete(file, recursive)) {
162         throw new IOException(s"Could not delete file: $path")
163       }
164     }
165   }

167   class HadoopZipHandle(fc: FileContext, file: Path) extends HadoopFileHandle(fc, file) {
168     override def open: CloseableIterator[(Option[String], InputStream)] = {
169       // we have to read the bytes into memory to get random access reads
170       val bytes = WithClose(PathUtils.handleCompression(fc.open(file), file.getName)) { is =>
171         IOUtils.toByteArray(is)
172       }
173       new ZipFileIterator(new ZipFile(new SeekableInMemoryByteChannel(bytes)), file.toString)
174     }
175 
176     override def write(mode: CreateMode, createParents: Boolean): OutputStream =
177       factory.createArchiveOutputStream(ArchiveStreamFactory.ZIP, super.write(mode, createParents))
178   }
179 
180   class HadoopTarHandle(fc: FileContext, file: Path) extends HadoopFileHandle(fc, file) {
181     override def open: CloseableIterator[(Option[String], InputStream)] = {
182       val uncompressed = PathUtils.handleCompression(fc.open(file), file.getName)
183       val archive: ArchiveInputStream[_ <: ArchiveEntry] =
184         factory.createArchiveInputStream(ArchiveStreamFactory.TAR, uncompressed)
185       new ArchiveFileIterator(archive, file.toString)
186     }
187 
188     override def write(mode: CreateMode, createParents: Boolean): OutputStream =
189       factory.createArchiveOutputStream(ArchiveStreamFactory.TAR, super.write(mode, createParents))
190   }
