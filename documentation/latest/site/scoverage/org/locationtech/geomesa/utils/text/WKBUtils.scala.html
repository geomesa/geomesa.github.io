18   private[this] val readerPool = new ThreadLocal[WKBReader]{

21

23     override def initialValue = new WKBWriter(2)
24   }
25 
26   private[this] val writer3dPool = new ThreadLocal[WKBWriter]{
27     override def initialValue = new WKBWriter(3)
28   }
29 
30   def read(s: String): Geometry = read(s.getBytes)
31   def read(b: Array[Byte]): Geometry = readerPool.get.read(b)
32 
33   def write(g: Geometry): Array[Byte] = {
34     val writer = if (is2d(g)) { writer2dPool } else { writer3dPool }
35     writer.get.write(g)
36   }
37 
38   private def is2d(geometry: Geometry): Boolean = {
39     // don't trust coord.getDimensions - it always returns 3 in jts
40     // instead, check for NaN for the z dimension
41     // note that we only check the first coordinate - if a geometry is written with different
42     // dimensions in each coordinate, some information may be lost
43     if (geometry == null) { true } else {
44       val coord = geometry.getCoordinate
45       // check for dimensions
46       // TODO WKBWriter still only supports z or m but not both
47       coord == null || (java.lang.Double.isNaN(coord.getZ) && java.lang.Double.isNaN(coord.getM))
48     }
49   }
50 }
