45   /**
46    * Create validators for the given feature type
47    *
48    * @param sft simple feature type
49    * @param names validator names and options
50    * @param tags tags used for metrics
51    * @return
52    */
53   def apply(sft: SimpleFeatureType, names: Seq[String], tags: Tags): SimpleFeatureValidator = {
54     val validators = names.filterNot(_.equalsIgnoreCase(NoneValidatorFactory.Name)).map { full =>
55       val i = full.indexOf('(')
56       val (name, options) = if (i == -1) { (full, None) } else {
57         require(full.last == ')', s"Invalid option parentheses: $full")
58         (full.substring(0, i), Some(full.substring(i + 1, full.length - 1)))
59       }
60       val factory = factories.find(_.name.equalsIgnoreCase(name)).getOrElse {
61         throw new IllegalArgumentException(s"No factory found for name '$name'. " +
62           s"Available factories: ${factories.map(_.name).mkString(", ")}")
63       }
64       factory.apply(sft, options, tags)
65     }
66 
67     if (validators.lengthCompare(2) < 0) {
68       validators.headOption.getOrElse(NoValidator)
69     } else {
70       new CompositeValidator(validators)
71     }
72   }

121   /**
122     * Evaluates multiple validators
123     *
124     * @param validators validators
125     */
126   class CompositeValidator(validators: Seq[SimpleFeatureValidator]) extends SimpleFeatureValidator {
127     override def validate(sf: SimpleFeature): String = {
128       var error: String = null
129       validators.foreach { validator =>
130         val e = validator.validate(sf)
131         if (e != null) {
132           error = if (error == null) { e } else { s"$error, $e" }
133         }
134       }
135       error
136     }
137     override def close(): Unit = CloseWithLogging(validators)
138   }
