18 /**
19   * Provides namespaced access to reporting metrics
20   *
21   * @param registry metric registry
22   * @param prefix namespace prefix
23   * @param typeName simple feature type name being processed
24   * @param reporters metric reporters
25   */
26 @deprecated("Use micrometer global registry")
27 class ConverterMetrics(
28     registry: MetricRegistry,
29     prefix: Option[String],
30     typeName: String,
31     reporters: Seq[ScheduledReporter]
32   ) extends GeoMesaMetrics(registry, s"${prefix.map(_ + ".").getOrElse("")}geomesa.convert", reporters) {
33 
34   /**
35     * Creates a prefixed counter
36     *
37     * @param id short identifier for the metric being counted
38     * @return
39     */
40   def counter(id: String): Counter = super.counter(typeName, id)
41 
42   /**
43     * Gets an updatable gauge
44     *
45     * @param id short identifier for hte metric being gauged
46     * @tparam T gauge type
47     * @return
48     */
49   def gauge[T](id: String): ConverterMetrics.SimpleGauge[T] =
50     super.gauge(typeName, id, new ConverterMetrics.SimpleGauge()).asInstanceOf[ConverterMetrics.SimpleGauge[T]]
51 
52   /**
53     * Creates a prefixed histogram
54     *
55     * @param id short identifier for the metric being histogram-ed
56     * @return
57     */
58   def histogram(id: String): Histogram = super.histogram(typeName, id)
59 
60   /**
61     * Creates a prefixed meter
62     *
63     * @param id short identifier for the metric being metered
64     * @return
65     */
66   def meter(id: String): Meter = super.meter(typeName, id)
67 
68   /**
69     * Creates a prefixed timer
70     *
71     * @param id short identifier for the metric being timed
72     * @return
73     */
74   def timer(id: String): Timer = super.timer(typeName, id)
75 
76   /**
77     * Register a metric
78     *
79     * @param id short identifier for the metric
80     * @param metric metric to register
81     * @tparam T metric type
82     * @return
83     */
84   def register[T <: Metric](id: String, metric: T): T = super.register(typeName, id, metric)
85 
86   override def close(): Unit = {
87     // execute a final report before closing, for situations where the converter runs too quickly to report anything
88     try { reporters.foreach(_.report()) } finally {
89       super.close()
90     }
91   }
92 }

94 object ConverterMetrics {
95 
96   @deprecated("replaced with micrometer/MetricsNamePrefix")
97   val MetricsPrefix: SystemProperty = SystemProperty("geomesa.convert.validators.prefix")
98 
99   val MetricsNamePrefix: SystemProperty = SystemProperty("geomesa.convert.metrics.prefix", "geomesa.convert")
100 
101   /**
102    * Gets a standard name for a converter-based metric, i.e. prefixing it with `geomesa.convert`
103    *
104    * @param name short name
105    * @return
106    */
107   def name(name: String): String = {
108     val prefix = MetricsNamePrefix.get
109     if (prefix.isEmpty) { name } else { s"$prefix.$name" }
110   }
111 
112   /**
113    * Gets the converter name as a tag
114    *
115    * @param name converter name
116    * @return
117    */
118   def converterNameTag(name: String): Tags = Tags.of("converter.name", name)
119 
120   /**
121     * Creates an empty registry with no namespace or reporters
122     *
123     * @return
124     */
125   @deprecated("Use micrometer global registry")
126   def empty: ConverterMetrics = new ConverterMetrics(new MetricRegistry(), None, "", Seq.empty)
127 
128   /**
129     * Create a registry for the provided feature type
130     *
131     * @param sft simple feature type
132     * @param reporters configs for metric reporters
133     * @return
134     */
135   @deprecated("Use micrometer global registry")
136   def apply(sft: SimpleFeatureType, reporters: Seq[Config]): ConverterMetrics = {
137     val registry = new MetricRegistry()
138     val reps = reporters.map(org.locationtech.geomesa.metrics.core.ReporterFactory.apply(_, registry)).toList
139     new ConverterMetrics(registry, MetricsPrefix.option, sft.getTypeName, reps)
140   }
141 
142   /**
143     * Simple gauge that can be updated
144     *
145     * @tparam T value
146     */
147   @deprecated("Use micrometer global registry")
148   class SimpleGauge[T] extends Gauge[T] {
149 
150     @volatile
151     private var value: T = _
152 
153     override def getValue: T = value
154 
155     def set(value: T): Unit = this.value = value
156   }
