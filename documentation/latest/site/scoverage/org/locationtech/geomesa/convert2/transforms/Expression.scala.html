50     *
51     * @param expressions expressions
52     * @return
53     */
54   def flatten(expressions: Seq[Expression]): Seq[Expression] = {
55     val toCheck = scala.collection.mutable.Queue(expressions: _*)
56     val result = scala.collection.mutable.Set.empty[Expression]
57     while (toCheck.nonEmpty) {
58       val next = toCheck.dequeue()
59       if (result.add(next)) {
60         toCheck ++= next.children()
61       }
62     }
63     result.toSeq
64   }

100     override def apply(args: Array[_ <: AnyRef]): Integer = {
101       e.apply(args) match {
102         case n: Integer          => n
103         case n: java.lang.Number => n.intValue()

199   case class FieldLookup(n: String, accessor: FieldAccessor = NullFieldAccessor) extends Expression {
200     override def apply(args: Array[_ <: AnyRef]): AnyRef = accessor.apply()
201     override def withContext(ec: EvaluationContext): Expression = FieldLookup(n, ec.accessor(n))
202     override def dependencies(stack: Set[Field], fieldMap: Map[String, Field]): Set[Field] = {
203       fieldMap.get(n) match {
204         case None => Set.empty
205         case Some(field) =>
206           if (stack.contains(field)) {
207             throw new IllegalArgumentException(s"Cyclical dependency detected in field $field")
208           } else {
209             field.transforms.toSeq.flatMap(_.dependencies(stack + field, fieldMap)).toSet + field
210           }
211       }
212     }
213     override def toString: String = s"$$$n"
214   }

224   case class FunctionExpression(f: TransformerFunction, arguments: Array[Expression]) extends Expression {
225 
226     @volatile private var contextDependent: Int = -1
227 
228     private def this(f: TransformerFunction, arguments: Array[Expression], contextDependent: Int) = {
229       this(f, arguments)
230       this.contextDependent = contextDependent
231     }
232 
233     override def apply(args: Array[_ <: AnyRef]): AnyRef = f.apply(arguments.map(_.apply(args)))
234 
235     override def withContext(ec: EvaluationContext): Expression = {
236       // this code is thread-safe, in that it will ensure correctness, but does not guarantee
237       // that the dependency check is only performed once
238       if (contextDependent == 0) { this } else {
239         lazy val fwc = f.withContext(ec)
240         lazy val awc = arguments.map(_.withContext(ec))
241         if (contextDependent == 1) {
242           new FunctionExpression(fwc, awc, 1)
243         } else {
244           if (!fwc.eq(f)) {
245             contextDependent = 1
246           } else {
247             var i = 0
248             while (i < arguments.length) {
249               if (!awc(i).eq(arguments(i))) {
250                 contextDependent = 1
251                 i = Int.MaxValue
252               } else {
253                 i += 1
254               }
255             }
256             if (i == arguments.length) {
257               contextDependent = 0
258             }
259           }
260           if (contextDependent == 0) { this } else { new FunctionExpression(fwc, awc, 1) }
261         }
262       }
263     }
264 
265     override def dependencies(stack: Set[Field], fieldMap: Map[String, Field]): Set[Field] =
266       arguments.flatMap(_.dependencies(stack, fieldMap)).toSet
267     override def children(): Seq[Expression] = arguments
268     override def toString: String = s"${f.names.head}${arguments.mkString("(", ",", ")")}"
269   }

271   case class TryExpression(toTry: Expression, fallback: Expression) extends Expression {
272 
273     @volatile private var contextDependent: Int = -1
274 
275     private def this(toTry: Expression, fallback: Expression, contextDependent: Int) = {
276       this(toTry, fallback)
277       this.contextDependent = contextDependent
278     }
279 
280     override def apply(args: Array[_ <: AnyRef]): AnyRef = Try(toTry.apply(args)).getOrElse(fallback.apply(args))
281 
282     override def withContext(ec: EvaluationContext): Expression = {
283       // this code is thread-safe, in that it will ensure correctness, but does not guarantee
284       // that the dependency check is only performed once
285       if (contextDependent == 0) { this } else {
286         lazy val twc = toTry.withContext(ec)
287         lazy val fwc = fallback.withContext(ec)
288         if (contextDependent == 1) {
289           new TryExpression(twc, fwc, 1)
290         } else {
291           contextDependent = if (twc.eq(toTry) && fwc.eq(fallback)) { 0 } else { 1 }
292           if (contextDependent == 0) { this } else { new TryExpression(twc, fwc, 1) }
293         }
294       }
295     }
296 
297     override def dependencies(stack: Set[Field], fieldMap: Map[String, Field]): Set[Field] =
298       toTry.dependencies(stack, fieldMap) ++ fallback.dependencies(stack, fieldMap)
299     override def children(): Seq[Expression] = Seq(toTry, fallback)
300     override def toString: String = s"try($toTry,$fallback)"
301   }
