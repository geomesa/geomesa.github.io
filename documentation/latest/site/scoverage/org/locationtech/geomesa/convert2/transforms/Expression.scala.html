62   def flatten(expressions: Seq[Expression]): Seq[Expression] = {
63     val toCheck = scala.collection.mutable.Queue(expressions: _*)
64     val result = scala.collection.mutable.Set.empty[Expression]
65     while (toCheck.nonEmpty) {
66       val next = toCheck.dequeue()
67       if (result.add(next)) {
68         toCheck ++= next.children()
69       }
70     }
71     result.toSeq
72   }

254       // this code is thread-safe, in that it will ensure correctness, but does not guarantee
255       // that the dependency check is only performed once
256       if (contextDependent == 0) { this } else {
257         lazy val fwc = f.withContext(ec)
258         lazy val awc = arguments.map(_.withContext(ec))
259         if (contextDependent == 1) {
260           new FunctionExpression(fwc, awc, 1)
261         } else {
262           if (!fwc.eq(f)) {

264           } else {
265             var i = 0

267               if (!awc(i).eq(arguments(i))) {
268                 contextDependent = 1

270               } else {
271                 i += 1
272               }
273             }
274             if (i == arguments.length) {

276             }
277           }
