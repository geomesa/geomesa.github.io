29   private val listParserFn = TransformerFunction.pure("parseList") { args =>
30     args(1) match {
31       case s: String if s.nonEmpty =>
32         val clazz = determineClazz(args(0).asInstanceOf[String])
33         val delim = if (args.length >= 3) { args(2).asInstanceOf[String] } else { defaultListDelim }
34         s.split(delim).map(i => convert(i.trim(), clazz)).toList.asJava

44   private val mapParserFn = TransformerFunction.pure("parseMap") { args =>
45     args(1) match {
46       case s: String if s.nonEmpty =>
47         val types = args(0).asInstanceOf[String].split("->").map(_.trim)
48         val keyClazz = determineClazz(types(0))
49         val valueClazz = determineClazz(types(1))
50         val kvDelim = if (args.length >= 3) { args(2).asInstanceOf[String] } else { defaultKVDelim }
51         val pairDelim = if (args.length >= 4) { args(3).asInstanceOf[String] } else { defaultListDelim }
52         val pairs = s.split(pairDelim).map { pair =>
53           pair.split(kvDelim) match {
54             case Array(key, value) => (convert(key.trim(), keyClazz), convert(value.trim(), valueClazz))
55             case _ => throw new IllegalArgumentException(s"Unexpected key/value pair: $pair")
56           }
57         }
58         pairs.toMap.asJava
59 
60       case "" => Collections.emptyMap()
61 
62       case null => null
63 
64       case s => throw new IllegalArgumentException(s"Expected a String but got $s:${s.getClass.getName}")
65     }

95     protected def convert(value: Any, clazz: Class[_]): Any = {
96       val converted = FastConverter.convert(value, clazz)
97       if (converted == null) {
98         throw new IllegalArgumentException(s"Could not convert value '$value' to type ${clazz.getName})")
99       }
100       converted
101     }
102 
103     protected def determineClazz(s: String): Class[_] = s.toLowerCase match {
104       case "string" | "str"   => classOf[String]
105       case "int" | "integer"  => classOf[java.lang.Integer]
106       case "long"             => classOf[java.lang.Long]
107       case "double"           => classOf[java.lang.Double]
108       case "float"            => classOf[java.lang.Float]
109       case "bool" | "boolean" => classOf[java.lang.Boolean]
110       case "bytes"            => classOf[Array[Byte]]
111       case "uuid"             => classOf[UUID]
112       case "date"             => classOf[java.util.Date]
113     }
114   }
