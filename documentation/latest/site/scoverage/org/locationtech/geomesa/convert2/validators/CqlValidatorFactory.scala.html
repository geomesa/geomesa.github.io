18 /**
19   * Validator for arbitrary CQL filters

21 class CqlValidatorFactory extends SimpleFeatureValidatorFactory {
22 
23   override def name: String = CqlValidatorFactory.Name
24 
25   override def apply(sft: SimpleFeatureType, metrics: ConverterMetrics, config: Option[String]): SimpleFeatureValidator =
26     apply(sft, config, Tags.empty())
27 
28   override def apply(sft: SimpleFeatureType, config: Option[String], tags: Tags): SimpleFeatureValidator = {
29     val ecql = config.getOrElse(throw new IllegalArgumentException("No filter specified for CQL Validator"))
30     val filter = FastFilterFactory.toFilter(sft, ecql)
31     new CqlValidator(filter, ecql, tags)
32   }
33 }

35 object CqlValidatorFactory {
36 
37   val Name = "cql"
38 
39   class CqlValidator(filter: Filter, ecql: String, tags: Tags) extends SimpleFeatureValidator {
40 
41     private val attributes = FilterHelper.propertyNames(filter).sorted.mkString(",")
42 
43     private val success = successCounter("cql", attributes, tags)
44     private val failure = failureCounter("cql", attributes, "filtered", tags)
45     private val error = s"failed filter: $ecql"
46 
47     override def validate(sf: SimpleFeature): String = {
48       if (filter.evaluate(sf)) {
49         success.increment()
50         null
51       } else {
52         failure.increment()
53         error
54       }
55     }
56 
57     override def close(): Unit = {}
58   }
