18 /**
19   * Validator for arbitrary CQL filters
20   */
21 class CqlValidatorFactory extends SimpleFeatureValidatorFactory {
22 
23   override def name: String = CqlValidatorFactory.Name
24 
25   override def apply(sft: SimpleFeatureType, config: Option[String], tags: Tags): SimpleFeatureValidator = {
26     val ecql = config.getOrElse(throw new IllegalArgumentException("No filter specified for CQL Validator"))
27     val filter = FastFilterFactory.toFilter(sft, ecql)
28     new CqlValidator(filter, ecql, tags)
29   }
30 }

32 object CqlValidatorFactory {
33 
34   val Name = "cql"

36   class CqlValidator(filter: Filter, ecql: String, tags: Tags) extends SimpleFeatureValidator {
37 
38     private val attributes = FilterHelper.propertyNames(filter).sorted.mkString(",")
39 
40     private val success = successCounter("cql", attributes, tags)
41     private val failure = failureCounter("cql", attributes, "filtered", tags)
42     private val error = s"failed filter: $ecql"
43 
44     override def validate(sf: SimpleFeature): String = {
45       if (filter.evaluate(sf)) {
46         success.increment()
47         null
48       } else {
49         failure.increment()
50         error
51       }
52     }
53 
54     override def close(): Unit = {}
55   }
