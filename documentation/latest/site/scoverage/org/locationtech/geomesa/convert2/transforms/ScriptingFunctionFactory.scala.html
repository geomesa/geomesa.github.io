24 /**
25   * Provides TransformerFunctions that execute javax.scripts compatible functions defined
26   * on the classpath or in external directories.  Scripting languages are determined by
27   * the extension of the file.  For instance, 'name.js' will be interpreted as a javascript file.
28   */
29 class ScriptingFunctionFactory extends TransformerFunctionFactory with LazyLogging {
30 
31   import ScriptingFunctionFactory._
32 
33   override lazy val functions: Seq[TransformerFunction] = {
34     val curClassLoader = Thread.currentThread().getContextClassLoader
35     val scriptURIs = loadScriptsFromEnvironment ++ loadScripts(curClassLoader)
36 
37     logger.debug(s"Script URIs found: ${scriptURIs.map(_.toString).mkString(",")}")
38     val mgr = new ScriptEngineManager()
39     scriptURIs
40       .map { f =>
41         // Always use the scheme specific part for URIs
42         val ext = FilenameUtils.getExtension(f.getSchemeSpecificPart)
43         (f, ext)
44       }.groupBy { case (_, ext) => ext }
45       .flatMap { case (ext, files) =>
46         val engine = Option(mgr.getEngineByExtension(ext))
47         engine.map { e => evaluateScriptsForEngine(curClassLoader, files, e, ext) }.getOrElse(Seq())
48       }.toList
49   }
50 
51   private def evaluateScriptsForEngine(loader: ClassLoader, files: Seq[(URI, String)], e: ScriptEngine, ext: String) = {
52     files.foreach { case (f, _) => evalScriptFile(loader, e, f) }
53     e.getBindings(ScriptContext.ENGINE_SCOPE).asScala.map { case (k, v) =>
54       new ScriptTransformerFn(ext, k, e.asInstanceOf[Invocable])
55     }
56   }
57 
58   def evalScriptFile(loader: ClassLoader, e: ScriptEngine, f: URI): AnyRef =
59     e.eval(IOUtils.toString(f, "UTF-8"))
60 }

69   /**
70     * <p>Load scripts from the environment using the property "geomesa.convert.scripts.path"
71     * Entries are colon (:) separated. Entries can be files or directories. Directories will
72     * be recursively searched for script files. The extension of script files defines what
73     * kind of script they are (e.g. js = javascript)</p>
74     */
75   def loadScriptsFromEnvironment: Seq[URI] = {
76     ConvertScriptsPath.option.toSeq.flatMap(_.split(":")).flatMap { path =>
77       val f = new File(path)
78       if (!f.exists() || !f.canRead) {
79         Seq.empty
80       } else if (f.isDirectory) {
81         if (!f.canExecute) { Seq.empty } else {
82           FileUtils.listFiles(f, TrueFileFilter.TRUE, TrueFileFilter.TRUE).asScala.map(_.toURI)
83         }
84       } else {
85         Seq(f.toURI)
86       }
87     }
88   }
89 
90   /**
91     * Load scripts from the resource geomesa-convert-scripts from a classloader. To use this
92     * create a folder in the jar named "geomesa-convert-scripts" and place script files
93     * within that directory.
94     *
95     * @param loader
96     * @return
97     */
98   def loadScripts(loader: ClassLoader): Seq[URI] = {
99     Option(loader.getResources(ConvertScriptsClassPath + "/").asScala).toSeq.flatten.flatMap { url =>
100       val uri = url.toURI
101       uri.getScheme match {
102         case "jar" =>
103           val fs = FileSystems.newFileSystem(uri, Collections.emptyMap[String, AnyRef](), loader)
104           val p = fs.getPath(ConvertScriptsClassPath + "/")
105           val uris = java.nio.file.Files.walk(p).iterator().asScala.toSeq
106           val files = uris.filterNot(java.nio.file.Files.isDirectory(_)).map(_.toUri)
107           logger.debug(s"Loaded scripts ${files.mkString(",")} from jar ${uri.toString}")
108           files
109         case "file" =>
110           IOUtils.readLines(url.openStream(), "UTF-8")
111             .asScala.map { s => loader.getResource(s"$ConvertScriptsClassPath/$s").toURI }
112       }
113     }
114   }
