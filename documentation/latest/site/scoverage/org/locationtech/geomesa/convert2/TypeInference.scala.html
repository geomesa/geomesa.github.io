43   /**
44    * Infer types
45    *
46    * @param pathsAndValues paths (xpath, jsonpath, etc) to values
47    * @param sft simple feature type, or name for inferred type
48    * @param namer naming for inferred attributes
49    * @param failureRate allowed failure rate for conversion to a given type
50    * @return
51    */
52   def infer(
53       pathsAndValues: Seq[PathWithValues],
54       sft: Either[String, SimpleFeatureType],
55       namer: String => String = new Namer(),
56       failureRate: Float = 0.1f): InferredTypesWithPaths = {
57     val inferred = pathsAndValues.map { case PathWithValues(path, values) =>
58       val rawTypes = values.map(InferredType.infer)
59       // merge the types for this column from each row
60       val typed = merge(rawTypes.toSeq, failureRate)
61       val baseName = if (path.nonEmpty) { path } else { String.valueOf(typed.typed).toLowerCase(Locale.US) }
62       TypeWithPath(path, typed.copy(name = namer(baseName)))
63     }
64     val geom = deriveGeometry(inferred.map(_.inferredType), namer).map(TypeWithPath("", _))
65 
66     val typesWithPaths = inferred ++ geom
67 
68     sft match {
69       case Left(name) =>
70         val schema = TypeInference.schema(name, typesWithPaths.map(_.inferredType))
71         InferredTypesWithPaths(schema, typesWithPaths)
72 
73       case Right(sft) =>
74         // validate the existing schema
75         AbstractConverterFactory.validateInferredType(sft, typesWithPaths.map(_.inferredType.typed))
76         val renamed = typesWithPaths.zip(sft.getAttributeDescriptors.asScala.map(_.getLocalName)).map {
77           case (inferred, name) => inferred.copy(inferredType = inferred.inferredType.copy(name = name))
78         }
79         InferredTypesWithPaths(sft, renamed)
80     }
81   }

83   /**
84     * Try to derive a geometry field, if one does not already exist
85     *
86     * @param types known types
87     * @return
88     */
89   def deriveGeometry(types: Seq[InferredType], namer: String => String): Option[InferredType] = {
90     // if there is no geometry field, see if we can derive one
91     if (types.map(_.typed).exists(geometries.contains)) { None } else {
92       val nums = types.filter(t => t.typed == ObjectType.DOUBLE || t.typed == ObjectType.FLOAT)
93       if (nums.lengthCompare(2) < 0) { None } else {
94         def findBestMatch(names: Seq[String]): Option[InferredType] = {
95           // determine priority order
96           val potentials = nums.flatMap { t =>
97             val name = t.name.toLowerCase(Locale.US)
98             val i = names.indexOf(name)
99             val j = names.indexWhere(n => name.endsWith(s"_$n"))
100             if (i != -1) {
101               Some(i -> t)
102             } else if (j != -1) {
103               Some(j + names.length -> t) // add names.length to make exact matches sort first
104             } else {
105               None
106             }
107           }
108           potentials.sortBy(_._1).collectFirst { case (_, t) => t }
109         }
110         for {
111           lat <- findBestMatch(latitudeNames)
112           lon <- findBestMatch(longitudeNames)
113         } yield {
114           InferredType(namer("geom"), POINT, DerivedTransform("point", lon.name, lat.name))
115         }
116       }
117     }
118   }

141   /**
142     * Merge the values from a given column into a single type
143     *
144     * @param types types extracted from the column
145     * @param failureRate max percentage of values that can fail a conversion to a particular type,
146     *                    and still consider the column to be that type
147     * @return
148     */
149   private def merge(types: Seq[InferredType], failureRate: Float): InferredType = {
150 
151     val typeCounts = scala.collection.mutable.Map.empty[InferredType, Int].withDefaultValue(0)
152     types.foreach(t => typeCounts(t) += 1)
153 
154     if (typeCounts.size == 1) {
155       // if only one type, we're good
156       return types.head
157     }
158 
159     var (mostFrequentType, mostFrequentTypeCount) = typeCounts.maxBy(_._2)
160 
161     if (1f - mostFrequentTypeCount.toFloat / types.length < failureRate) {
162       // if the most frequent type passes the error threshold, use it
163       return mostFrequentType
164     }
165 
166     // try to combine types up into more general ones
167     val mergedTypeCounts = typeCounts.foldLeft(Map.empty[InferredType, Int]) { case (counts, (typ, count)) =>
168       val merged = scala.collection.mutable.Map.empty[InferredType, (InferredType, Int)] // from -> to
169       counts.foreach { case (mergedType, mergedCount) =>
170         merge(typ, mergedType).foreach(t => merged += mergedType -> (t, mergedCount + count))
171       }
172       if (merged.isEmpty) {
173         counts + (typ -> count)
174       } else {
175         (counts -- merged.keys) ++ merged.values
176       }
177     }
178 
179     mergedTypeCounts.maxBy(_._2) match {
180       case (t, c) => mostFrequentType = t; mostFrequentTypeCount = c
181     }
182 
183     if (1f - mostFrequentTypeCount.toFloat / types.length < failureRate) {
184       // if the most frequent merged type passes the error threshold, use it
185       return mostFrequentType
186     }
187 
188     // if nothing else, fall back to string
189     InferredType("", STRING, CastToString)
190   }

192   /**
193     * Merge two types into a single super-type, if one exists
194     *
195     * for example:
196     *
197     *   merge(float, double) == double
198     *   merge(point, linestring) == geometry
199     *
200     * @param left first type
201     * @param right second type
202     * @return
203     */
204   private def merge(left: InferredType, right: InferredType): Option[InferredType] = {
205     // string + foo => string, null + foo => foo
206     if (left == right || left.typed == STRING || right.typed == null) {
207       Some(left)
208     } else if (left.typed == null || right.typed == STRING) {
209       Some(right)
210     } else if (geometries.contains(left.typed) && geometries.contains(right.typed)) {
211       Some(InferredType("", GEOMETRY, IdentityTransform))
212     } else {
213       for {
214         (leftPriority, leftTransform) <- mergeUpNumbers.get(left.typed)
215         (rightPriority, rightTransform) <- mergeUpNumbers.get(right.typed)
216       } yield {
217         if (leftPriority <= rightPriority) {
218           left.copy(transform = leftTransform)
219         } else {
220           right.copy(transform = rightTransform)
221         }
222       }
223     }
224   }

258   class Namer(existing: Seq[String] = Seq.empty) extends (String => String) {
259 
260     // track the names we use for each column to ensure no duplicates
261     private val uniqueNames = scala.collection.mutable.HashSet[String](existing: _*)
262 
263     /**
264      * Get a valid attribute name based on the element name
265      *
266      * @param key json key
267      * @return
268      */
269     override def apply(key: String): String = {
270       val base = key.replaceAll("[^A-Za-z0-9]+", "_").replaceAll("^_|_$", "")
271       var candidate = base
272       var i = 0
273       while (FeatureUtils.ReservedWords.contains(candidate.toUpperCase(Locale.US)) || !uniqueNames.add(candidate)) {
274         candidate = s"${base}_$i"
275         i += 1
276       }
277       candidate
278     }
279   }

358     private val dateParsers = TransformerFunction.functions.values.collect { case f: StandardDateParser => f }.toArray
359 
360     /**
361       * Infer a type from a value. Returned type will have an empty name
362       *
363       * @param value value
364       * @return
365       */
366     def infer(value: Any): InferredType = {
367       value match {
368         case null | ""                => InferredType("", null, IdentityTransform)
369         case _: Int | _: Integer      => InferredType("", INT, IdentityTransform)
370         case _: Long | _: jLong       => InferredType("", LONG, IdentityTransform)
371         case _: Float | _: jFloat     => InferredType("", FLOAT, IdentityTransform)
372         case _: Double | _: jDouble   => InferredType("", DOUBLE, IdentityTransform)
373         case _: Boolean | _: jBoolean => InferredType("", BOOLEAN, IdentityTransform)
374         case _: Date                  => InferredType("", DATE, IdentityTransform)
375         case _: Array[Byte]           => InferredType("", BYTES, IdentityTransform)
376         case _: java.util.UUID        => InferredType("", UUID, IdentityTransform)
377         case _: Point                 => InferredType("", POINT, IdentityTransform)
378         case _: LineString            => InferredType("", LINESTRING, IdentityTransform)
379         case _: Polygon               => InferredType("", POLYGON, IdentityTransform)
380         case _: MultiPoint            => InferredType("", MULTIPOINT, IdentityTransform)
381         case _: MultiLineString       => InferredType("", MULTILINESTRING, IdentityTransform)
382         case _: MultiPolygon          => InferredType("", MULTIPOLYGON, IdentityTransform)
383         case _: GeometryCollection    => InferredType("", GEOMETRY_COLLECTION, IdentityTransform)
384         case _: java.util.List[_]     => InferredType("", LIST, IdentityTransform)
385 
386         case s: String =>
387           val trimmed = s.trim
388           tryNumberParsing(trimmed)
389               .orElse(tryDateParsing(trimmed))
390               .orElse(tryGeometryParsing(trimmed))
391               .orElse(tryBooleanParsing(trimmed))
392               .orElse(tryUuidParsing(trimmed))
393               .getOrElse(InferredType("", STRING, IdentityTransform))
394 
395         case _ => InferredType("", STRING, CastToString)
396       }
397     }
398 
399     private def tryNumberParsing(s: String): Option[InferredType] = {
400       Try(BigDecimal(s)).toOption.collect {
401         case n if s.indexOf('.') == -1 && n.isValidInt  => InferredType("", INT, CastToInt)
402         case n if s.indexOf('.') == -1 && n.isValidLong => InferredType("", LONG, CastToLong)
403         // don't consider floats - even valid floats cause rounding changes when using as doubles in geometries
404         case n if n.isDecimalDouble                     => InferredType("", DOUBLE, CastToDouble)
405       }
406     }
407 
408     private def tryDateParsing(s: String): Option[InferredType] = {
409       var i = 0
410       while (i < dateParsers.length) {
411         val p = dateParsers(i)
412         if (Try(DateParsing.parseDate(s, p.format)).isSuccess) {
413           return Some(InferredType("", DATE, FunctionTransform(s"${p.names.head}(", ")")))
414         }
415         i += 1
416       }
417       None
418     }
419 
420     private def tryGeometryParsing(s: String): Option[InferredType] = {
421       Try(WKTUtils.read(s)).toOption.map {
422         case _: Point              => InferredType("", POINT, FunctionTransform("point(", ")"))
423         case _: LineString         => InferredType("", LINESTRING, FunctionTransform("linestring(", ")"))
424         case _: Polygon            => InferredType("", POLYGON, FunctionTransform("polygon(", ")"))
425         case _: MultiPoint         => InferredType("", MULTIPOINT, FunctionTransform("multipoint(", ")"))
426         case _: MultiLineString    => InferredType("", MULTILINESTRING, FunctionTransform("multilinestring(", ")"))
427         case _: MultiPolygon       => InferredType("", MULTIPOLYGON, FunctionTransform("multipolygon(", ")"))
428         case _: GeometryCollection => InferredType("", GEOMETRY_COLLECTION, FunctionTransform("geometrycollection(", ")"))
429         case _                     => InferredType("", GEOMETRY, FunctionTransform("geometry(", ")"))
430       }
431     }
432 
433     private def tryBooleanParsing(s: String): Option[InferredType] =
434       Try(s.toBoolean).toOption.map(_ => InferredType("", BOOLEAN, CastToBoolean))
435 
436     private def tryUuidParsing(s: String): Option[InferredType] =
437       Try(java.util.UUID.fromString(s)).toOption.map(_ => InferredType("", UUID, IdentityTransform))
438   }
