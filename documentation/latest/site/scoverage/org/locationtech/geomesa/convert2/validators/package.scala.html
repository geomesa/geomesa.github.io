58   /**
59    * Validates an attribute is not null
60    *
61    * @param attribute attribute to validate
62    * @param name attribute name
63    * @param error error message
64    * @param tags metric tags
65    */
66   class NullValidator(name: String, attribute: Attribute, error: String, tags: Tags) extends SimpleFeatureValidator {
67 
68     private val success = successCounter(name, attribute.name, tags)
69     private val failure = failureCounter(name, attribute.name, "null", tags)
70 
71     override def validate(sf: SimpleFeature): String = {
72       if (sf.getAttribute(attribute.i) != null) {
73         success.increment()
74         null
75       } else {
76         failure.increment()

78       }
79     }

81     override def close(): Unit = {}
