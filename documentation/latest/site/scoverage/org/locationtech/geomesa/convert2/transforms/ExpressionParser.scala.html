25   @throws(classOf[ParsingException])
26   def parse(expression: String, report: Boolean = true): Expression = {
27     logger.trace(s"Parsing expression: $expression")
28     if (expression == null) {
29       throw new IllegalArgumentException("Invalid expression: null")
30     }
31     val runner = if (report) { ReportingParseRunner(Parser.expression) } else { BasicParseRunner(Parser.expression) }
32     val parsing = runner.run(expression)
33     parsing.result.getOrElse {
34       throw new ParsingException(s"Invalid expression: ${ErrorUtils.printParseErrors(parsing)}")
35     }
36   }
37 }
