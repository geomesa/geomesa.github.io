23 /**
24  * Converts input streams into simple features. SimpleFeatureConverters should be thread-safe. However,
25  * a given EvaluationContext should only be used in a single thread at once.
26  */
27 trait SimpleFeatureConverter extends Closeable with LazyLogging {
28 
29   import scala.collection.JavaConverters._
30 
31   /**
32     * Result feature type
33     */
34   def targetSft: SimpleFeatureType
35 
36   /**
37    * Process an input stream into simple features.
38    *
39    * This method should be thread-safe, as long as different evaluation contexts are used for each request.
40    *
41    * @param is input
42    * @param ec evaluation context
43    * @return
44    */
45   def process(is: InputStream, ec: EvaluationContext = createEvaluationContext()): CloseableIterator[SimpleFeature]
46 
47   /**
48    * Create a context used for local state while processing. A context object is not thread-safe, and should
49    * only be used in one thread at a time.
50    *
51    * @param globalParams global key-values to make accessible through the evaluation context
52    * @return
53    */
54   def createEvaluationContext(globalParams: Map[String, Any] = Map.empty): EvaluationContext
55 
56   /**
57    * Java API for `createEvaluationContext`
58    *
59    * @param globalParams global key-values to make accessible through the evaluation context
60    * @return
61    */
62   final def createEvaluationContext(globalParams: java.util.Map[String, Any]): EvaluationContext =
63     createEvaluationContext(globalParams.asScala.toMap)
64 }
65 
66 object SimpleFeatureConverter extends StrictLogging {
67 
68   val factories: List[SimpleFeatureConverterFactory] = ServiceLoader.load[SimpleFeatureConverterFactory]()
69 
70   logger.debug(s"Found ${factories.size} factories: ${factories.map(_.getClass.getName).mkString(", ")}")
71 
72   /**
73     * Create a converter
74     *
75     * @param sft simple feature type
76     * @param config converter configuration
77     * @return
78     */
79   def apply(sft: SimpleFeatureType, config: Config): SimpleFeatureConverter = {
80     factories.toStream.flatMap(_.apply(sft, config)).headOption.getOrElse {
81       throw new IllegalArgumentException(s"Cannot find converter factory for ${sft.getTypeName}")
82     }
83   }
84 
85   /**
86     * Create a converter by name
87     *
88     * @param typeName simple feature type name
89     * @param converterName converter name
90     * @return
91     */
92   def apply(typeName: String, converterName: String): SimpleFeatureConverter = {
93     val sft = SimpleFeatureTypeLoader.sftForName(typeName).getOrElse {
94       throw new IllegalArgumentException(s"Unable to load SimpleFeatureType for typeName '$typeName'")
95     }
96     apply(sft, converterName)
97   }
98 
99   /**
100     * Create a converter by name
101     *
102     * @param sft simple feature type
103     * @param converterName converter name
104     * @return
105     */
106   def apply(sft: SimpleFeatureType, converterName: String): SimpleFeatureConverter = {
107     val converter = ConverterConfigLoader.configForName(converterName).getOrElse {
108       throw new IllegalArgumentException(s"Unable to load converter config for converter named '$converterName'")
109     }
110     apply(sft, converter)
111   }
112 
113   /**
114     * Infer a converter based on a data sample
115     *
116     * @param is input stream to convert
117     * @param sft simple feature type, if known
118     * @return
119     */
120   @deprecated("replaced with `infer(() => InputStream, Option[SimpleFeatureType], Map[String, Any])`")
121   def infer(
122       is: () => InputStream,
123       sft: Option[SimpleFeatureType],
124       path: Option[String] = None): Option[(SimpleFeatureType, Config)] = {
125     val hints = path match {
126       case None    => Map.empty[String, AnyRef]
127       case Some(p) => Map(EvaluationContext.InputFilePathKey -> p)
128     }
129     infer(is, sft, hints).toOption
130   }
131 
132   /**
133    * Infer a converter based on a data sample
134    *
135    * @param is input stream to convert
136    * @param sft simple feature type, if known
137    * @param hints implementation specific hints
138    * @return
139    */
140   def infer(
141       is: () => InputStream,
142       sft: Option[SimpleFeatureType],
143       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = infer(is, sft, hints, None)
144 
145   /**
146    * Infer a converter based on a data sample
147    *
148    * @param is input stream to convert
149    * @param sft simple feature type, if known
150    * @param hints implementation specific hints
151    * @param priority priority order for trying factories
152    * @return
153    */
154   def infer(
155       is: () => InputStream,
156       sft: Option[SimpleFeatureType],
157       hints: Map[String, AnyRef],
158       priority: Option[Ordering[SimpleFeatureConverterFactory]]): Try[(SimpleFeatureType, Config)] = {
159     if (factories.isEmpty) {
160       Failure(new RuntimeException("There are no converters available on the classpath"))
161     } else {
162       val sorted = priority.fold(factories)(factories.sorted(_))
163       val attempts = sorted.iterator.map { factory =>
164         WithClose(is()) { is =>
165           val res = try { factory.infer(is, sft, hints) } catch {
166             // in case a particular converter's dependencies aren't on the classpath
167             case e: NoClassDefFoundError => Failure(e)
168           }
169           if (res.isSuccess) { res } else {
170             val msg = s"${factory.getClass.getSimpleName}: could not infer a converter:"
171             Failure(new RuntimeException(msg, res.failed.get))
172           }
173         }
174       }
175       multiTry(attempts, new RuntimeException("Unable to infer a converter"))
176     }
177   }
178 }
