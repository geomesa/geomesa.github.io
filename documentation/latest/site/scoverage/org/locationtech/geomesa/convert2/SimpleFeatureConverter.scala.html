24 /**
25  * Converts input streams into simple features. SimpleFeatureConverters should be thread-safe. However,
26  * a given EvaluationContext should only be used in a single thread at once.
27  */
28 trait SimpleFeatureConverter extends Closeable with LazyLogging {
29 
30   import scala.collection.JavaConverters._
31 
32   /**
33     * Result feature type
34     */
35   def targetSft: SimpleFeatureType
36 
37   /**
38    * Process an input stream into simple features.
39    *
40    * This method should be thread-safe, as long as different evaluation contexts are used for each request.
41    *
42    * @param is input
43    * @param ec evaluation context
44    * @return
45    */
46   def process(is: InputStream, ec: EvaluationContext = createEvaluationContext()): CloseableIterator[SimpleFeature]
47 
48   /**
49    * Create a context used for local state while processing. A context object is not thread-safe, and should
50    * only be used in one thread at a time.
51    *
52    * @param globalParams global key-values to make accessible through the evaluation context
53    * @return
54    */
55   def createEvaluationContext(globalParams: Map[String, Any] = Map.empty): EvaluationContext
56 
57   /**
58    * Create a context used for local state while processing. A context object is not thread-safe, and should
59    * only be used in one thread at a time.
60    *
61    * @param globalParams global key-values to make accessible through the evaluation context
62    * @param success counter for tracking successful conversions
63    * @param failure counter for tracking failed conversions
64    * @return
65    */
66   def createEvaluationContext(globalParams: Map[String, Any], success: Counter, failure: Counter): EvaluationContext
67 
68   /**
69    * Java API for `createEvaluationContext`
70    *
71    * @param globalParams global key-values to make accessible through the evaluation context
72    * @return
73    */
74   final def createEvaluationContext(globalParams: java.util.Map[String, Any]): EvaluationContext =
75     createEvaluationContext(globalParams.asScala.toMap)
76 }

78 object SimpleFeatureConverter extends StrictLogging {
79 
80   val factories: List[SimpleFeatureConverterFactory] = ServiceLoader.load[SimpleFeatureConverterFactory]()
81 
82   logger.debug(s"Found ${factories.size} factories: ${factories.map(_.getClass.getName).mkString(", ")}")
83 
84   /**
85     * Create a converter
86     *
87     * @param sft simple feature type
88     * @param config converter configuration
89     * @return
90     */
91   def apply(sft: SimpleFeatureType, config: Config): SimpleFeatureConverter = {
92     factories.toStream.flatMap(_.apply(sft, config)).headOption.getOrElse {
93       throw new IllegalArgumentException(s"Cannot find converter factory for ${sft.getTypeName}")
94     }
95   }
96 
97   /**
98     * Create a converter by name
99     *
100     * @param typeName simple feature type name
101     * @param converterName converter name
102     * @return
103     */
104   def apply(typeName: String, converterName: String): SimpleFeatureConverter = {
105     val sft = SimpleFeatureTypeLoader.sftForName(typeName).getOrElse {
106       throw new IllegalArgumentException(s"Unable to load SimpleFeatureType for typeName '$typeName'")
107     }
108     apply(sft, converterName)
109   }
110 
111   /**
112     * Create a converter by name
113     *
114     * @param sft simple feature type
115     * @param converterName converter name
116     * @return
117     */
118   def apply(sft: SimpleFeatureType, converterName: String): SimpleFeatureConverter = {
119     val converter = ConverterConfigLoader.configForName(converterName).getOrElse {
120       throw new IllegalArgumentException(s"Unable to load converter config for converter named '$converterName'")
121     }
122     apply(sft, converter)
123   }
124 
125   /**
126     * Infer a converter based on a data sample
127     *
128     * @param is input stream to convert
129     * @param sft simple feature type, if known
130     * @return
131     */
132   @deprecated("replaced with `infer(() => InputStream, Option[SimpleFeatureType], Map[String, Any])`")
133   def infer(
134       is: () => InputStream,
135       sft: Option[SimpleFeatureType],
136       path: Option[String] = None): Option[(SimpleFeatureType, Config)] = {
137     val hints = path match {
138       case None    => Map.empty[String, AnyRef]
139       case Some(p) => Map(EvaluationContext.InputFilePathKey -> p)
140     }
141     infer(is, sft, hints).toOption
142   }
143 
144   /**
145    * Infer a converter based on a data sample
146    *
147    * @param is input stream to convert
148    * @param sft simple feature type, if known
149    * @param hints implementation specific hints
150    * @return
151    */
152   def infer(
153       is: () => InputStream,
154       sft: Option[SimpleFeatureType],
155       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = infer(is, sft, hints, None)
156 
157   /**
158    * Infer a converter based on a data sample
159    *
160    * @param is input stream to convert
161    * @param sft simple feature type, if known
162    * @param hints implementation specific hints
163    * @param priority priority order for trying factories
164    * @return
165    */
166   def infer(
167       is: () => InputStream,
168       sft: Option[SimpleFeatureType],
169       hints: Map[String, AnyRef],
170       priority: Option[Ordering[SimpleFeatureConverterFactory]]): Try[(SimpleFeatureType, Config)] = {
171     if (factories.isEmpty) {
172       Failure(new RuntimeException("There are no converters available on the classpath"))
173     } else {
174       val sorted = priority.fold(factories)(factories.sorted(_))
175       val attempts = sorted.iterator.map { factory =>
176         WithClose(is()) { is =>
177           val res = try { factory.infer(is, sft, hints) } catch {
178             // in case a particular converter's dependencies aren't on the classpath
179             case e: NoClassDefFoundError => Failure(e)
180           }
181           if (res.isSuccess) { res } else {
182             val msg = s"${factory.getClass.getSimpleName}: could not infer a converter:"
183             Failure(new RuntimeException(msg, res.failed.get))
184           }
185         }
186       }
187       multiTry(attempts, new RuntimeException("Unable to infer a converter"))
188     }
189   }
