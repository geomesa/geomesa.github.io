24 /**
25  * Converts input streams into simple features. SimpleFeatureConverters should be thread-safe. However,
26  * a given EvaluationContext should only be used in a single thread at once.
27  */
28 trait SimpleFeatureConverter extends Closeable with LazyLogging {
29 
30   import scala.collection.JavaConverters._
31 
32   /**
33     * Result feature type
34     */
35   def targetSft: SimpleFeatureType
36 
37   /**
38    * Process an input stream into simple features.
39    *
40    * This method should be thread-safe, as long as different evaluation contexts are used for each request.
41    *
42    * @param is input
43    * @param ec evaluation context
44    * @return
45    */
46   def process(is: InputStream, ec: EvaluationContext = createEvaluationContext()): CloseableIterator[SimpleFeature]
47 
48   /**
49    * Create a context used for local state while processing. A context object is not thread-safe, and should
50    * only be used in one thread at a time.
51    *
52    * @param globalParams global key-values to make accessible through the evaluation context
53    * @return
54    */
55   def createEvaluationContext(globalParams: Map[String, Any] = Map.empty): EvaluationContext
56 
57   /**
58    * Create a context used for local state while processing. A context object is not thread-safe, and should
59    * only be used in one thread at a time.
60    *
61    * @param globalParams global key-values to make accessible through the evaluation context
62    * @param success counter for tracking successful conversions
63    * @param failure counter for tracking failed conversions
64    * @return
65    */
66   @deprecated("Custom counters are no longer supported, use a ContextListener instead")
67   def createEvaluationContext(globalParams: Map[String, Any], success: Counter, failure: Counter): EvaluationContext
68 
69   /**
70    * Java API for `createEvaluationContext`
71    *
72    * @param globalParams global key-values to make accessible through the evaluation context
73    * @return
74    */
75   final def createEvaluationContext(globalParams: java.util.Map[String, Any]): EvaluationContext =
76     createEvaluationContext(globalParams.asScala.toMap)
77 }

79 object SimpleFeatureConverter extends StrictLogging {
80 
81   val factories: List[SimpleFeatureConverterFactory] = ServiceLoader.load[SimpleFeatureConverterFactory]()
82 
83   logger.debug(s"Found ${factories.size} factories: ${factories.map(_.getClass.getName).mkString(", ")}")
84 
85   /**
86     * Create a converter
87     *
88     * @param sft simple feature type
89     * @param config converter configuration
90     * @return
91     */
92   def apply(sft: SimpleFeatureType, config: Config): SimpleFeatureConverter = {
93     factories.toStream.flatMap(_.apply(sft, config)).headOption.getOrElse {
94       throw new IllegalArgumentException(s"Cannot find converter factory for ${sft.getTypeName}")
95     }
96   }
97 
98   /**
99     * Create a converter by name
100     *
101     * @param typeName simple feature type name
102     * @param converterName converter name
103     * @return
104     */
105   def apply(typeName: String, converterName: String): SimpleFeatureConverter = {
106     val sft = SimpleFeatureTypeLoader.sftForName(typeName).getOrElse {
107       throw new IllegalArgumentException(s"Unable to load SimpleFeatureType for typeName '$typeName'")
108     }
109     apply(sft, converterName)
110   }
111 
112   /**
113     * Create a converter by name
114     *
115     * @param sft simple feature type
116     * @param converterName converter name
117     * @return
118     */
119   def apply(sft: SimpleFeatureType, converterName: String): SimpleFeatureConverter = {
120     val converter = ConverterConfigLoader.configForName(converterName).getOrElse {
121       throw new IllegalArgumentException(s"Unable to load converter config for converter named '$converterName'")
122     }
123     apply(sft, converter)
124   }
125 
126   /**
127     * Infer a converter based on a data sample
128     *
129     * @param is input stream to convert
130     * @param sft simple feature type, if known
131     * @return
132     */
133   @deprecated("replaced with `infer(() => InputStream, Option[SimpleFeatureType], Map[String, Any])`")
134   def infer(
135       is: () => InputStream,
136       sft: Option[SimpleFeatureType],
137       path: Option[String] = None): Option[(SimpleFeatureType, Config)] = {
138     val hints = path match {
139       case None    => Map.empty[String, AnyRef]
140       case Some(p) => Map(EvaluationContext.InputFilePathKey -> p)
141     }
142     infer(is, sft, hints).toOption
143   }
144 
145   /**
146    * Infer a converter based on a data sample
147    *
148    * @param is input stream to convert
149    * @param sft simple feature type, if known
150    * @param hints implementation specific hints
151    * @return
152    */
153   def infer(
154       is: () => InputStream,
155       sft: Option[SimpleFeatureType],
156       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = infer(is, sft, hints, None)
157 
158   /**
159    * Infer a converter based on a data sample
160    *
161    * @param is input stream to convert
162    * @param sft simple feature type, if known
163    * @param hints implementation specific hints
164    * @param priority priority order for trying factories
165    * @return
166    */
167   def infer(
168       is: () => InputStream,
169       sft: Option[SimpleFeatureType],
170       hints: Map[String, AnyRef],
171       priority: Option[Ordering[SimpleFeatureConverterFactory]]): Try[(SimpleFeatureType, Config)] = {
172     if (factories.isEmpty) {
173       Failure(new RuntimeException("There are no converters available on the classpath"))
174     } else {
175       val sorted = priority.fold(factories)(factories.sorted(_))
176       val attempts = sorted.iterator.map { factory =>
177         WithClose(is()) { is =>
178           val res = try { factory.infer(is, sft, hints) } catch {
179             // in case a particular converter's dependencies aren't on the classpath
180             case e: NoClassDefFoundError => Failure(e)
181           }
182           if (res.isSuccess) { res } else {
183             val msg = s"${factory.getClass.getSimpleName}: could not infer a converter:"
184             Failure(new RuntimeException(msg, res.failed.get))
185           }
186         }
187       }
188       multiTry(attempts, new RuntimeException("Unable to infer a converter"))
189     }
190   }
191 }
