33   override def createEvaluationContext(globalParams: Map[String, Any]): EvaluationContext =
34     createEvaluationContext(delegates.map(_._2.createEvaluationContext(globalParams)), Stats(tags))

36   private def createEvaluationContext(contexts: Seq[EvaluationContext], theseStats: Stats): EvaluationContext = {
37     val stats = new Stats() {
38       override def success(i: Int): Long = theseStats.success(i) + contexts.map(_.stats.success(0)).sum
39       override def failure(i: Int): Long = theseStats.failure(i) + contexts.map(_.stats.failure(0)).sum
40     }
41     CompositeEvaluationContext(contexts, stats)
42   }

44   override def process(is: InputStream, ec: EvaluationContext): CloseableIterator[SimpleFeature] = {
45     val contexts = ec match {
46       case cec: CompositeEvaluationContext => cec.contexts.iterator
47       case _ => Iterator.continually(ec)
48     }
49     val predicates = delegates.map { case (predicate, converter) =>
50       if (!contexts.hasNext) {
51         throw new IllegalArgumentException(s"Invalid evaluation context doesn't match this converter: $ec")
52       }
53       val context = contexts.next()
54       PredicateContext(predicate.withContext(context), converter, context)
55     }
56 
57     val toEval = Array.ofDim[AnyRef](1)
58 
59     val lines = IOUtils.lineIterator(is, StandardCharsets.UTF_8)
60 
61     new CloseableIterator[SimpleFeature] {
62 
63       private var delegate: CloseableIterator[SimpleFeature] = CloseableIterator.empty
64 
65       @tailrec
66       override def hasNext: Boolean = delegate.hasNext || {
67         if (!lines.hasNext) {
68           false
69         } else {
70           toEval(0) = lines.next()
71           ec.line += 1
72           delegate.close()
73           delegate = null
74           lazy val in = new ByteArrayInputStream(toEval(0).asInstanceOf[String].getBytes(StandardCharsets.UTF_8))
75           predicates.foreach { p =>
76             if (delegate == null && Try(p.predicate(toEval)).getOrElse(false)) {
77               delegate = p.converter.process(in, p.context)
78             } else {
79               p.context.line += 1
80             }
81           }
82           if (delegate == null) {
83             ec.stats.failure()
84             delegate = CloseableIterator.empty
85           }
86           hasNext
87         }
88       }
89 
90       override def next(): SimpleFeature = if (hasNext) { delegate.next } else { Iterator.empty.next }
91 
92       override def close(): Unit = {
93         CloseWithLogging(delegate)
94         is.close()
95       }
96     }
97   }
98
