40   override def apply(sft: SimpleFeatureType, conf: Config): Option[SimpleFeatureConverter] = {
41     if (!conf.hasPath("type") || !conf.getString("type").equalsIgnoreCase(TypeToProcess)) { None } else {
42       val (config, fields, opts) = try {
43         val c = ConfigSource.fromConfig(AbstractConverterFactory.standardDefaults(conf, logger))
44         val config = c.loadOrThrow[FeatureToFeatureConfig]
45         val fields = c.loadOrThrow[Seq[BasicField]]
46         val opts = c.loadOrThrow[BasicOptions]
47         (config, fields, opts)
48       } catch {
49         case NonFatal(e) => throw new IllegalArgumentException(s"Invalid configuration: ${e.getMessage}")
50       }
51 
52       val inputSft = SimpleFeatureTypeLoader.sftForName(config.inputSft).getOrElse {
53         throw new IllegalArgumentException(s"Could not load input sft ${config.inputSft}")
54       }
55       // FID is put in as the last attribute, we copy it over here
56       // TODO: does this have any implications for global params in the evaluation context?
57       val id = if (config.idField.isDefined) { config } else {
58         config.copy(idField = Some(Expression.Column(inputSft.getAttributeCount)))
59       }
60 
61       // add transform expressions to look up the attribute
62       val visitor = new SftLookupVisitor(inputSft)
63       val columns = fields.map { field =>
64         val transforms = field.transforms match {
65           case Some(transforms) => Some(transforms.accept(visitor))
66           case None =>
67             val i = inputSft.indexOf(field.name)
68             if (i == -1) { None } else { Some(Expression.Column(i)) }
69         }
70         field.copy(transforms = transforms)
71       }
72 
73       // any matching fields that aren't explicitly defined will be copied over by default
74       val defaults = {
75         val names = fields.map(_.name).toSet
76         sft.getAttributeDescriptors.asScala.map(_.getLocalName).filterNot(names.contains).map { name =>
77           val i = inputSft.indexOf(name)
78           if (i == -1) {
79             BasicField(name, Some(Expression.LiteralNull))
80           } else {
81             BasicField(name, Some(Expression.Column(i)))
82           }
83         }
84       }
85 
86       Some(new FeatureToFeatureConverter(sft, id, columns ++ defaults, opts))
87     }
88   }
89 }

97   case class FeatureToFeatureConfig(
98       `type`: String,
99       converterName: Option[String],

103       userData: Map[String, Expression]

106   object FeatureToFeatureConfigConvert extends ConverterConfigConvert[FeatureToFeatureConfig] with StrictLogging {
107 
108     override protected def decodeConfig(
109         cur: ConfigObjectCursor,
110         `type`: String,
111         idField: Option[Expression],
112         caches: Map[String, Config],
113         userData: Map[String, Expression]): Either[ConfigReaderFailures, FeatureToFeatureConfig] = {
114       for {
115         converterName <- converterName(cur).right
116         sftName       <- cur.atKey(InputSftPath).right.flatMap(_.asString).right
117       } yield {
118         FeatureToFeatureConfig(`type`, converterName, sftName, idField, caches, userData)
119       }
120     }
121 
122     override protected def encodeConfig(config: FeatureToFeatureConfig, base: java.util.Map[String, AnyRef]): Unit =
123       base.put(InputSftPath, config.inputSft)
124   }
125 
126   /**
127    * Replaces field name lookups with column lookups referencing the input sft
128    *
129    * @param sft input sft
130    */
131   private class SftLookupVisitor(sft: SimpleFeatureType) extends ExpressionTreeVisitor {
132     override def visit(e: FieldLookup): Expression = {
133       val i = sft.indexOf(e.n)
134       if (i == -1) { e } else { Expression.Column(i) }
135     }
136   }
