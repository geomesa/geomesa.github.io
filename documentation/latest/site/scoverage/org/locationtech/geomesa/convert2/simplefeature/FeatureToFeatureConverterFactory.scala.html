40   override def apply(sft: SimpleFeatureType, conf: Config): Option[SimpleFeatureConverter] = {
41     if (!conf.hasPath("type") || !conf.getString("type").equalsIgnoreCase(TypeToProcess)) { None } else {
42       val (config, fields, opts) = try {
43         val c = ConfigSource.fromConfig(AbstractConverterFactory.standardDefaults(conf, logger))
44         val config = c.loadOrThrow[FeatureToFeatureConfig]
45         val fields = c.loadOrThrow[Seq[BasicField]]
46         val opts = c.loadOrThrow[BasicOptions]
47         (config, fields, opts)
48       } catch {
49         case NonFatal(e) => throw new IllegalArgumentException(s"Invalid configuration: ${e.getMessage}")
50       }
51 
52       val inputSft = SimpleFeatureTypeLoader.sftForName(config.inputSft).getOrElse {
53         throw new IllegalArgumentException(s"Could not load input sft ${config.inputSft}")
54       }
55       // FID is put in as the last attribute, we copy it over here
56       // TODO: does this have any implications for global params in the evaluation context?
57       val id = if (config.idField.isDefined) { config } else {
58         config.copy(idField = Some(Expression.Column(inputSft.getAttributeCount)))
59       }
60 
61       // add transform expressions to look up the attribute
62       val columns = fields.map { field =>
63         field.transforms match {
64           case None => field.copy(transforms = Some(Expression.Column(inputSft.indexOf(field.name))))
65           case Some(Expression.FieldLookup(n, _)) => field.copy(transforms = Some(Expression.Column(inputSft.indexOf(n))))
66           case _ => field
67         }
68       }
69 
70       // any matching fields that aren't explicitly defined will be copied over by default
71       val defaults = {
72         val names = fields.map(_.name).toSet
73         sft.getAttributeDescriptors.asScala.map(_.getLocalName).filterNot(names.contains).map { name =>
74           val i = inputSft.indexOf(name)
75           if (i == -1) {
76             BasicField(name, Some(Expression.LiteralNull))
77           } else {
78             BasicField(name, Some(Expression.Column(i)))
79           }
80         }
81       }
82 
83       Some(new FeatureToFeatureConverter(sft, id, columns ++ defaults, opts))
84     }
85   }

89       sft: Option[SimpleFeatureType],
90       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = Failure(new NotImplementedError())

109     override protected def decodeConfig(
110         cur: ConfigObjectCursor,
111         `type`: String,
112         idField: Option[Expression],
113         caches: Map[String, Config],
114         userData: Map[String, Expression]): Either[ConfigReaderFailures, FeatureToFeatureConfig] = {
115       for { sftName <- cur.atKey(InputSftPath).right.flatMap(_.asString).right } yield {

118     }

121       base.put(InputSftPath, config.inputSft)
