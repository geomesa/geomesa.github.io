33 /**
34   * Abstract converter factory implementation. Subclasses need to implement `typeToProcess` and make available
35   * pureconfig readers for the converter configuration
36  *
37  *  The converter to use is identified by the 'type' field in the config, e.g. 'xml' or 'json'
38   */
39 abstract class AbstractConverterFactory[S <: AbstractConverter[_, C, F, O] : ClassTag, C <: ConverterConfig : ClassTag, F <: Field : ClassTag, O <: ConverterOptions : ClassTag](
40     val typeToProcess: String,
41     protected val configConvert: ConverterConfigConvert[C],
42     protected val fieldConvert: FieldConvert[F],
43     protected val optsConvert: ConverterOptionsConvert[O]
44   ) extends SimpleFeatureConverterFactory {
45 
46   private def loadConfig(config: ConfigObjectSource): C = config.loadOrThrow[C](classTag[C], configConvert)
47   private def loadFields(config: ConfigObjectSource): Seq[F] = config.loadOrThrow[Seq[F]](classTag[Seq[F]], fieldConvert)
48   private def loadOptions(config: ConfigObjectSource): O = config.loadOrThrow[O](classTag[O], optsConvert)
49 
50   override def apply(sft: SimpleFeatureType, conf: Config): Option[SimpleFeatureConverter] = {
51     if (!conf.hasPath("type") || !conf.getString("type").equalsIgnoreCase(typeToProcess)) { None } else {
52       val (config, fields, opts) = try {
53         val c = ConfigSource.fromConfig(withDefaults(conf))
54         (loadConfig(c), loadFields(c), loadOptions(c))
55       } catch {
56         case NonFatal(e) => throw new IllegalArgumentException(s"Invalid configuration: ${e.getMessage}")
57       }
58       val args = Array(classOf[SimpleFeatureType], implicitly[ClassTag[C]].runtimeClass,
59         classOf[Seq[F]], implicitly[ClassTag[O]].runtimeClass)
60       val constructor = implicitly[ClassTag[S]].runtimeClass.getConstructor(args: _*)
61       try {
62         Some(constructor.newInstance(sft, config, fields, opts).asInstanceOf[SimpleFeatureConverter])
63       } catch {
64         case e: InvocationTargetException => throw e.getCause
65       }
66     }
67   }

118     if (conf.hasPath("options.validation-mode")) {
119       logger.warn(s"Using deprecated option 'validation-mode'. Prefer 'error-mode'")
120       updates.append(c => c.withValue("options.error-mode", conf.getValue("options.validation-mode")))
121     }
122     if (conf.hasPath("options.validating")) {
123       logger.warn(s"Using deprecated validation key 'validating'")
124       val validators = if (conf.getBoolean("options.validating")) {
125         ConfigValueFactory.fromIterable(Seq(HasGeoValidatorFactory.Name, HasDtgValidatorFactory.Name).asJava)
126       } else {
127         ConfigValueFactory.fromIterable(Collections.emptyList())
128       }
129       updates.append(c => c.withValue("options.validators", validators))
130     }
131 
132     updates.foldLeft(conf)((c, mod) => mod.apply(c)).withFallback(ConfigFactory.load("base-converter-defaults"))
133   }

189     * Pureconfig convert that parses out basic config. Subclasses must implement `decodeConfig` and `encodeConfig`
190     * to read/write any custom config
191     *
192     * @tparam C config class
193     */
194   abstract class ConverterConfigConvert[C <: ConverterConfig]
195       extends ConfigConvert[C] with ExpressionConvert with ConfigMapConvert {
196 
197     protected def decodeConfig(
198         cur: ConfigObjectCursor,
199         `type`: String,
200         idField: Option[Expression],
201         caches: Map[String, Config],
202         userData: Map[String, Expression]): Either[ConfigReaderFailures, C]
203 
204     protected def encodeConfig(config: C, base: java.util.Map[String, AnyRef]): Unit
205 
206     override def from(cur: ConfigCursor): Either[ConfigReaderFailures, C] = {
207       for {
208         obj      <- cur.asObjectCursor.right
209         typ      <- obj.atKey("type").right.flatMap(_.asString).right
210         idField  <- idFieldFrom(obj.atKeyOrUndefined("id-field")).right
211         userData <- userDataFrom(obj.atKeyOrUndefined("user-data")).right
212         caches   <- configMapFrom(obj.atKeyOrUndefined("caches")).right
213         config   <- decodeConfig(obj, typ, idField, caches, userData).right
214       } yield {
215         config
216       }
217     }
218 
219     override def to(obj: C): ConfigObject = ConfigValueFactory.fromMap(configTo(obj))
220 
221     protected def converterName(cur: ConfigObjectCursor): Either[ConfigReaderFailures, Option[String]] =
222       optionalStringFrom(cur.atKeyOrUndefined(ConverterConfigLoader.ConverterNameKey))
223 
224     private def configTo(config: C): java.util.Map[String, AnyRef] = {
225       val map = new java.util.HashMap[String, AnyRef]
226       map.put("type", config.`type`)
227       config.idField.foreach(f => map.put("id-field", f.toString))
228       if (config.userData.nonEmpty) {
229         map.put("user-data", config.userData.map { case (k, v) => (k, v.toString.asInstanceOf[AnyRef]) }.asJava)
230       }
231       if (config.caches.nonEmpty) {
232         map.put("caches", config.caches.map { case (k, v) => (k, v.root().unwrapped()) })
233       }
234       encodeConfig(config, map)
235       map
236     }
237 
238     private def optionalStringFrom(cur: ConfigCursor): Either[ConfigReaderFailures, Option[String]] =
239       if (cur.isUndefined) { Right(None) } else { cur.asString.right.map(s => Option(s)) }
240 
241     private def idFieldFrom(cur: ConfigCursor): Either[ConfigReaderFailures, Option[Expression]] = {
242       if (cur.isUndefined) { Right(None) } else {
243         for { expr <- exprFrom(cur).right } yield { Some(expr) }
244       }
245     }

247     private def userDataFrom(cur: ConfigCursor): Either[ConfigReaderFailures, Map[String, Expression]] = {
248       import org.locationtech.geomesa.utils.conf.ConfConversions.RichConfig
249 
250       if (cur.isUndefined) { Right(Map.empty) } else {
251         def merge(cur: ConfigObjectCursor): Either[ConfigReaderFailures, Map[String, Expression]] = {
252           val map = cur.valueOpt.get.toConfig.toStringMap() // handles quoting keys
253           map.foldLeft[Either[ConfigReaderFailures, Map[String, Expression]]](Right(Map.empty)) {
254             case (map, (k, v)) =>
255               // convert back to a cursor for parsing the expression
256               val path = cur.pathElems ++ ConfigUtil.splitPath(k).asScala
257               val valueCursor = ConfigCursor(ConfigValueFactory.fromAnyRef(v), path)
258               for { m <- map.right; d <- exprFrom(valueCursor).right } yield { m + (k -> d) }
259           }
260         }
261         for { obj <- cur.asObjectCursor.right; data <- merge(obj).right } yield { data }
262       }
263     }

268     * to read/write any custom field values

272   abstract class FieldConvert[F <: Field] extends ConfigConvert[Seq[F]] with ExpressionConvert {
273 
274     protected def decodeField(cur: ConfigObjectCursor,
275                               name: String,
276                               transform: Option[Expression]): Either[ConfigReaderFailures, F]
277 
278     protected def encodeField(field: F, base: java.util.Map[String, AnyRef]): Unit
279 
280     override def from(cur: ConfigCursor): Either[ConfigReaderFailures, Seq[F]] = {
281       for {
282         obj    <- cur.asObjectCursor.right
283         fields <- obj.atKey("fields").right.flatMap(_.asListCursor).right.flatMap(fieldsFrom).right
284       } yield {
285         fields
286       }
287     }
288 
289     override def to(obj: Seq[F]): ConfigObject = {
290       val map = new java.util.HashMap[String, AnyRef]
291       map.put("fields", obj.map(fieldTo).asJava)
292       ConfigValueFactory.fromMap(map)
293     }
294 
295     private def fieldsFrom(cur: ConfigListCursor): Either[ConfigReaderFailures, Seq[F]] = {
296       cur.list.foldLeft[Either[ConfigReaderFailures, Seq[F]]](Right(Seq.empty)) {
297         (list, field) => for { li <- list.right; f <- fieldFrom(field).right } yield { li :+ f }
298       }
299     }
300 
301     private def fieldFrom(cur: ConfigCursor): Either[ConfigReaderFailures, F] = {
302       def transformFrom(cur: ConfigCursor): Either[ConfigReaderFailures, Option[Expression]] = {
303         if (cur.isUndefined) { Right(None) } else {
304           exprFrom(cur).right.map(t => Some(t))
305         }
306       }
307 
308       for {
309         obj       <- cur.asObjectCursor.right
310         name      <- obj.atKey("name").right.flatMap(_.asString).right
311         transform <- transformFrom(obj.atKeyOrUndefined("transform")).right
312         field     <- decodeField(obj, name, transform).right
313       } yield {
314         field
315       }
316     }
317 
318     private def fieldTo(field: F): java.util.Map[String, AnyRef] = {
319       val map = new java.util.HashMap[String, AnyRef]
320       map.put("name", field.name)
321       field.transforms.foreach(t => map.put("transform", t.toString))
322       encodeField(field, map)
323       map
324     }
325   }

328     * Pureconfig convert that parses out basic options. Subclasses must implement `decodeOptions` and `encodeOptions`
329     * to read/write any custom converter options
330     *
331     * @tparam O options class
332     */
333   abstract class ConverterOptionsConvert[O <: ConverterOptions] extends ConfigConvert[O] with ConfigMapConvert {
334 
335     protected def decodeOptions(
336         cur: ConfigObjectCursor,
337         validators: Seq[String],
338         parseMode: ParseMode,
339         errorMode: ErrorMode,
340         encoding: Charset): Either[ConfigReaderFailures, O]
341 
342     protected def encodeOptions(options: O, base: java.util.Map[String, AnyRef]): Unit
343 
344     override def from(cur: ConfigCursor): Either[ConfigReaderFailures, O] = {
345       for {
346         obj     <- cur.asObjectCursor.right
347         options <- obj.atKey("options").right.flatMap(_.asObjectCursor).right.flatMap(optionsFrom).right
348       } yield {
349         options
350       }
351     }
352 
353     override def to(obj: O): ConfigObject = {
354       val map = new java.util.HashMap[String, AnyRef]
355       map.put("options", optionsTo(obj))
356       ConfigValueFactory.fromMap(map)
357     }
358 
359     private def optionsFrom(cur: ConfigObjectCursor): Either[ConfigReaderFailures, O] = {
360 
361       def mergeValidators(cur: ConfigListCursor): Either[ConfigReaderFailures, Seq[String]] = {
362         cur.list.foldLeft[Either[ConfigReaderFailures, Seq[String]]](Right(Seq.empty)) {
363           case (seq, v) => for { s <- seq.right; string <- v.asString.right } yield { s :+ string }
364         }
365       }
366 
367       def parse[T](key: String, values: Iterable[T], fallback: Map[String, T] = Map.empty[String, T]): Either[ConfigReaderFailures, T] = {
368         cur.atKey(key).right.flatMap { value =>
369           value.asString.right.flatMap { string =>
370             val fb = fallback.collectFirst { case (k, v) if k.equalsIgnoreCase(string) => v }
371             values.find(_.toString.equalsIgnoreCase(string)).orElse(fb) match {
372               case Some(v) => Right(v)
373               case None => value.failed(CannotConvert(value.valueOpt.map(_.toString).orNull, values.head.getClass.getSimpleName, s"Must be one of: ${values.mkString(", ")}"))
374             }
375           }
376         }
377       }
378 
379       if (cur.atKey("verbose").isRight) {
380         logger.warn("'verbose' option is deprecated - please use logging levels instead")
381       }
382       if (cur.atKey("reporters").isRight) {
383         logger.warn(
384           "'reporters' option is no longer supported, please see https://www.geomesa.org/documentation/stable/user/appendix/metrics.html")
385       }
386 
387       for {
388         validators <- cur.atKey("validators").right.flatMap(_.asListCursor).right.flatMap(mergeValidators).right
389         parseMode  <- parse("parse-mode", ParseMode.values).right
390         errorMode  <- parse("error-mode", ErrorMode.values, Map("skip-bad-records" -> ErrorMode.LogErrors)).right
391         encoding   <- cur.atKey("encoding").right.flatMap(_.asString).right.map(Charset.forName).right
392         options    <- decodeOptions(cur, validators, parseMode, errorMode, encoding).right
393       } yield {
394         options
395       }
396     }
397 
398     private def optionsTo(options: O): java.util.Map[String, AnyRef] = {
399       val map = new java.util.HashMap[String, AnyRef]
400       map.put("parse-mode", options.parseMode.toString)
401       map.put("error-mode", options.errorMode.toString)
402       map.put("encoding", options.encoding.name)
403       map.put("validators", options.validators.asJava)
404       encodeOptions(options, map)
405       map
406     }
407   }
