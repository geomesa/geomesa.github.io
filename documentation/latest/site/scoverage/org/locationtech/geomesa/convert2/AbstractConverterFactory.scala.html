32 /**
33   * Abstract converter factory implementation. Subclasses need to implement `typeToProcess` and make available
34   * pureconfig readers for the converter configuration
35  *
36  *  The converter to use is identified by the 'type' field in the config, e.g. 'xml' or 'json'
37   */
38 abstract class AbstractConverterFactory[S <: AbstractConverter[_, C, F, O] : ClassTag, C <: ConverterConfig : ClassTag, F <: Field : ClassTag, O <: ConverterOptions : ClassTag](
39     val typeToProcess: String,
40     protected val configConvert: ConverterConfigConvert[C],
41     protected val fieldConvert: FieldConvert[F],
42     protected val optsConvert: ConverterOptionsConvert[O]
43   ) extends SimpleFeatureConverterFactory {
44 
45   private def loadConfig(config: ConfigObjectSource): C = config.loadOrThrow[C](classTag[C], configConvert)
46   private def loadFields(config: ConfigObjectSource): Seq[F] = config.loadOrThrow[Seq[F]](classTag[Seq[F]], fieldConvert)
47   private def loadOptions(config: ConfigObjectSource): O = config.loadOrThrow[O](classTag[O], optsConvert)
48 
49   override def apply(sft: SimpleFeatureType, conf: Config): Option[SimpleFeatureConverter] = {
50     if (!conf.hasPath("type") || !conf.getString("type").equalsIgnoreCase(typeToProcess)) { None } else {
51       val (config, fields, opts) = try {
52         val c = ConfigSource.fromConfig(withDefaults(conf))
53         (loadConfig(c), loadFields(c), loadOptions(c))
54       } catch {
55         case NonFatal(e) => throw new IllegalArgumentException(s"Invalid configuration: ${e.getMessage}")
56       }
57       val args = Array(classOf[SimpleFeatureType], implicitly[ClassTag[C]].runtimeClass,
58         classOf[Seq[F]], implicitly[ClassTag[O]].runtimeClass)
59       val constructor = implicitly[ClassTag[S]].runtimeClass.getConstructor(args: _*)
60       try {
61         Some(constructor.newInstance(sft, config, fields, opts).asInstanceOf[SimpleFeatureConverter])
62       } catch {
63         case e: InvocationTargetException => throw e.getCause
64       }
65     }
66   }

113   def standardDefaults(conf: Config, logger: => Logger): Config = {
114     import scala.collection.JavaConverters._
115 
116     val updates = ArrayBuffer.empty[Config => Config]
117     if (conf.hasPath("options.validation-mode")) {
118       logger.warn(s"Using deprecated option 'validation-mode'. Prefer 'error-mode'")
119       updates.append(c => c.withValue("options.error-mode", conf.getValue("options.validation-mode")))
120     }
121     if (conf.hasPath("options.validating")) {
122       logger.warn(s"Using deprecated validation key 'validating'")
123       val validators = if (conf.getBoolean("options.validating")) {
124         ConfigValueFactory.fromIterable(Seq(HasGeoValidatorFactory.Name, HasDtgValidatorFactory.Name).asJava)
125       } else {
126         ConfigValueFactory.fromIterable(Collections.emptyList())
127       }
128       updates.append(c => c.withValue("options.validators", validators))
129     }
130 
131     updates.foldLeft(conf)((c, mod) => mod.apply(c)).withFallback(ConfigFactory.load("base-converter-defaults"))
132   }

186   /**
187     * Pureconfig convert that parses out basic config. Subclasses must implement `decodeConfig` and `encodeConfig`
188     * to read/write any custom config
189     *
190     * @tparam C config class
191     */
192   abstract class ConverterConfigConvert[C <: ConverterConfig]
193       extends ConfigConvert[C] with ExpressionConvert with ConfigMapConvert {
194 
195     protected def decodeConfig(
196         cur: ConfigObjectCursor,
197         `type`: String,
198         idField: Option[Expression],
199         caches: Map[String, Config],
200         userData: Map[String, Expression]): Either[ConfigReaderFailures, C]
201 
202     protected def encodeConfig(config: C, base: java.util.Map[String, AnyRef]): Unit
203 
204     override def from(cur: ConfigCursor): Either[ConfigReaderFailures, C] = {
205       for {
206         obj      <- cur.asObjectCursor.right
207         typ      <- obj.atKey("type").right.flatMap(_.asString).right
208         idField  <- idFieldFrom(obj.atKeyOrUndefined("id-field")).right
209         userData <- userDataFrom(obj.atKeyOrUndefined("user-data")).right
210         caches   <- configMapFrom(obj.atKeyOrUndefined("caches")).right
211         config   <- decodeConfig(obj, typ, idField, caches, userData).right
212       } yield {
213         config
214       }
215     }
216 
217     override def to(obj: C): ConfigObject = ConfigValueFactory.fromMap(configTo(obj))
218 
219     private def configTo(config: C): java.util.Map[String, AnyRef] = {
220       val map = new java.util.HashMap[String, AnyRef]
221       map.put("type", config.`type`)
222       config.idField.foreach(f => map.put("id-field", f.toString))
223       if (config.userData.nonEmpty) {
224         map.put("user-data", config.userData.map { case (k, v) => (k, v.toString.asInstanceOf[AnyRef]) }.asJava)
225       }
226       if (config.caches.nonEmpty) {
227         map.put("caches", config.caches.map { case (k, v) => (k, v.root().unwrapped()) })
228       }
229       encodeConfig(config, map)
230       map
231     }
232 
233     private def idFieldFrom(cur: ConfigCursor): Either[ConfigReaderFailures, Option[Expression]] = {
234       if (cur.isUndefined) { Right(None) } else {
235         for { expr <- exprFrom(cur).right } yield { Some(expr) }
236       }
237     }
238 
239     private def userDataFrom(cur: ConfigCursor): Either[ConfigReaderFailures, Map[String, Expression]] = {
240       import org.locationtech.geomesa.utils.conf.ConfConversions.RichConfig
241 
242       if (cur.isUndefined) { Right(Map.empty) } else {
243         def merge(cur: ConfigObjectCursor): Either[ConfigReaderFailures, Map[String, Expression]] = {
244           val map = cur.valueOpt.get.toConfig.toStringMap() // handles quoting keys
245           map.foldLeft[Either[ConfigReaderFailures, Map[String, Expression]]](Right(Map.empty)) {
246             case (map, (k, v)) =>
247               // convert back to a cursor for parsing the expression
248               val path = cur.pathElems ++ ConfigUtil.splitPath(k).asScala
249               val valueCursor = ConfigCursor(ConfigValueFactory.fromAnyRef(v), path)
250               for { m <- map.right; d <- exprFrom(valueCursor).right } yield { m + (k -> d) }
251           }
252         }
253         for { obj <- cur.asObjectCursor.right; data <- merge(obj).right } yield { data }
254       }
255     }
256   }

259     * Pureconfig convert that parses out basic fields. Subclasses must implement `decodeField` and `encodeField`
260     * to read/write any custom field values
261     *
262     * @tparam F field type
263     */
264   abstract class FieldConvert[F <: Field] extends ConfigConvert[Seq[F]] with ExpressionConvert {
265 
266     protected def decodeField(cur: ConfigObjectCursor,
267                               name: String,
268                               transform: Option[Expression]): Either[ConfigReaderFailures, F]
269 
270     protected def encodeField(field: F, base: java.util.Map[String, AnyRef]): Unit
271 
272     override def from(cur: ConfigCursor): Either[ConfigReaderFailures, Seq[F]] = {
273       for {
274         obj    <- cur.asObjectCursor.right
275         fields <- obj.atKey("fields").right.flatMap(_.asListCursor).right.flatMap(fieldsFrom).right
276       } yield {
277         fields
278       }
279     }
280 
281     override def to(obj: Seq[F]): ConfigObject = {
282       val map = new java.util.HashMap[String, AnyRef]
283       map.put("fields", obj.map(fieldTo).asJava)
284       ConfigValueFactory.fromMap(map)
285     }
286 
287     private def fieldsFrom(cur: ConfigListCursor): Either[ConfigReaderFailures, Seq[F]] = {
288       cur.list.foldLeft[Either[ConfigReaderFailures, Seq[F]]](Right(Seq.empty)) {
289         (list, field) => for { li <- list.right; f <- fieldFrom(field).right } yield { li :+ f }
290       }
291     }
292 
293     private def fieldFrom(cur: ConfigCursor): Either[ConfigReaderFailures, F] = {
294       def transformFrom(cur: ConfigCursor): Either[ConfigReaderFailures, Option[Expression]] = {
295         if (cur.isUndefined) { Right(None) } else {
296           exprFrom(cur).right.map(t => Some(t))
297         }
298       }
299 
300       for {
301         obj       <- cur.asObjectCursor.right
302         name      <- obj.atKey("name").right.flatMap(_.asString).right
303         transform <- transformFrom(obj.atKeyOrUndefined("transform")).right
304         field     <- decodeField(obj, name, transform).right
305       } yield {
306         field
307       }
308     }
309 
310     private def fieldTo(field: F): java.util.Map[String, AnyRef] = {
311       val map = new java.util.HashMap[String, AnyRef]
312       map.put("name", field.name)
313       field.transforms.foreach(t => map.put("transform", t.toString))
314       encodeField(field, map)
315       map
316     }
317   }

321     * to read/write any custom converter options
322     *
323     * @tparam O options class
324     */
325   abstract class ConverterOptionsConvert[O <: ConverterOptions]
326       extends ConfigConvert[O] with ConfigSeqConvert with ConfigMapConvert {
327 
328     protected def decodeOptions(
329         cur: ConfigObjectCursor,
330         validators: Seq[String],
331         reporters: Seq[Config],
332         parseMode: ParseMode,
333         errorMode: ErrorMode,
334         encoding: Charset): Either[ConfigReaderFailures, O]
335 
336     protected def encodeOptions(options: O, base: java.util.Map[String, AnyRef]): Unit
337 
338     override def from(cur: ConfigCursor): Either[ConfigReaderFailures, O] = {
339       for {
340         obj     <- cur.asObjectCursor.right
341         options <- obj.atKey("options").right.flatMap(_.asObjectCursor).right.flatMap(optionsFrom).right
342       } yield {
343         options
344       }
345     }
346 
347     override def to(obj: O): ConfigObject = {
348       val map = new java.util.HashMap[String, AnyRef]
349       map.put("options", optionsTo(obj))
350       ConfigValueFactory.fromMap(map)
351     }
352 
353     private def optionsFrom(cur: ConfigObjectCursor): Either[ConfigReaderFailures, O] = {
354 
355       def mergeValidators(cur: ConfigListCursor): Either[ConfigReaderFailures, Seq[String]] = {
356         cur.list.foldLeft[Either[ConfigReaderFailures, Seq[String]]](Right(Seq.empty)) {
357           case (seq, v) => for { s <- seq.right; string <- v.asString.right } yield { s :+ string }
358         }
359       }
360 
361       def parse[T](key: String, values: Iterable[T], fallback: Map[String, T] = Map.empty[String, T]): Either[ConfigReaderFailures, T] = {
362         cur.atKey(key).right.flatMap { value =>
363           value.asString.right.flatMap { string =>
364             val fb = fallback.collectFirst { case (k, v) if k.equalsIgnoreCase(string) => v }
365             values.find(_.toString.equalsIgnoreCase(string)).orElse(fb) match {
366               case Some(v) => Right(v)
367               case None => value.failed(CannotConvert(value.valueOpt.map(_.toString).orNull, values.head.getClass.getSimpleName, s"Must be one of: ${values.mkString(", ")}"))
368             }
369           }
370         }
371       }
372 
373       if (cur.atKey("verbose").isRight) {
374         logger.warn("'verbose' option is deprecated - please use logging levels instead")
375       }
376 
377       for {
378         validators <- cur.atKey("validators").right.flatMap(_.asListCursor).right.flatMap(mergeValidators).right
379         reporters  <- parseReporters(cur.atKeyOrUndefined("reporters")).right
380         parseMode  <- parse("parse-mode", ParseMode.values).right
381         errorMode  <- parse("error-mode", ErrorMode.values, Map("skip-bad-records" -> ErrorMode.LogErrors)).right
382         encoding   <- cur.atKey("encoding").right.flatMap(_.asString).right.map(Charset.forName).right
383         options    <- decodeOptions(cur, validators, reporters, parseMode, errorMode, encoding).right
384       } yield {
385         options
386       }
387     }
388 
389     /**
390       * Reads reporters as a config list, plus checks for back compatible config map
391       *
392       * @param cur cursor
393       * @return
394       */
395     private def parseReporters(cur: ConfigCursor): Either[ConfigReaderFailures, Seq[Config]] = {
396       if (cur.asObjectCursor.isRight) {
397         configMapFrom(cur).right.map(_.values.toList)
398       } else {
399         configSeqFrom(cur)
400       }
401     }
402 
403     private def optionsTo(options: O): java.util.Map[String, AnyRef] = {
404       val map = new java.util.HashMap[String, AnyRef]
405       map.put("parse-mode", options.parseMode.toString)
406       map.put("error-mode", options.errorMode.toString)
407       map.put("encoding", options.encoding.name)
408       map.put("validators", options.validators.asJava)
409       if (options.reporters.nonEmpty) {
410         map.put("reporters", options.reporters.map(_.root().unwrapped()))
411       }
412       encodeOptions(options, map)
413       map
414     }
415   }
