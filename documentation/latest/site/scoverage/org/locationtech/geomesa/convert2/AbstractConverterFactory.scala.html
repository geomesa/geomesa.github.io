33 /**
34   * Abstract converter factory implementation. Subclasses need to implement `typeToProcess` and make available
35   * pureconfig readers for the converter configuration
36  *
37  *  The converter to use is identified by the 'type' field in the config, e.g. 'xml' or 'json'
38   */
39 abstract class AbstractConverterFactory[S <: AbstractConverter[_, C, F, O] : ClassTag, C <: ConverterConfig : ClassTag, F <: Field : ClassTag, O <: ConverterOptions : ClassTag](
40     val typeToProcess: String,
41     protected val configConvert: ConverterConfigConvert[C],
42     protected val fieldConvert: FieldConvert[F],
43     protected val optsConvert: ConverterOptionsConvert[O]
44   ) extends SimpleFeatureConverterFactory {
45 
46   private def loadConfig(config: ConfigObjectSource): C = config.loadOrThrow[C](classTag[C], configConvert)
47   private def loadFields(config: ConfigObjectSource): Seq[F] = config.loadOrThrow[Seq[F]](classTag[Seq[F]], fieldConvert)
48   private def loadOptions(config: ConfigObjectSource): O = config.loadOrThrow[O](classTag[O], optsConvert)
49 
50   override def apply(sft: SimpleFeatureType, conf: Config): Option[SimpleFeatureConverter] = {
51     if (!conf.hasPath("type") || !conf.getString("type").equalsIgnoreCase(typeToProcess)) { None } else {
52       val (config, fields, opts) = try {
53         val c = ConfigSource.fromConfig(withDefaults(conf))
54         (loadConfig(c), loadFields(c), loadOptions(c))
55       } catch {
56         case NonFatal(e) => throw new IllegalArgumentException(s"Invalid configuration: ${e.getMessage}")
57       }
58       val args = Array(classOf[SimpleFeatureType], implicitly[ClassTag[C]].runtimeClass,
59         classOf[Seq[F]], implicitly[ClassTag[O]].runtimeClass)
60       val constructor = implicitly[ClassTag[S]].runtimeClass.getConstructor(args: _*)
61       try {
62         Some(constructor.newInstance(sft, config, fields, opts).asInstanceOf[SimpleFeatureConverter])
63       } catch {
64         case e: InvocationTargetException => throw e.getCause
65       }
66     }
67   }

75   protected def withDefaults(conf: Config): Config = AbstractConverterFactory.standardDefaults(conf, logger)

114   def standardDefaults(conf: Config, logger: => Logger): Config = {
115     import scala.collection.JavaConverters._
116 
117     val updates = ArrayBuffer.empty[Config => Config]
118     if (conf.hasPath("options.validation-mode")) {
119       logger.warn(s"Using deprecated option 'validation-mode'. Prefer 'error-mode'")
120       updates.append(c => c.withValue("options.error-mode", conf.getValue("options.validation-mode")))
121     }
122     if (conf.hasPath("options.validating")) {
123       logger.warn(s"Using deprecated validation key 'validating'")
124       val validators = if (conf.getBoolean("options.validating")) {
125         ConfigValueFactory.fromIterable(Seq(HasGeoValidatorFactory.Name, HasDtgValidatorFactory.Name).asJava)
126       } else {
127         ConfigValueFactory.fromIterable(Collections.emptyList())
128       }
129       updates.append(c => c.withValue("options.validators", validators))
130     }
131 
132     updates.foldLeft(conf)((c, mod) => mod.apply(c)).withFallback(ConfigFactory.load("base-converter-defaults"))
133   }

195   abstract class ConverterConfigConvert[C <: ConverterConfig]
196       extends ConfigConvert[C] with ExpressionConvert with ConfigMapConvert {
197 
198     protected def decodeConfig(
199         cur: ConfigObjectCursor,
200         `type`: String,
201         idField: Option[Expression],
202         caches: Map[String, Config],
203         userData: Map[String, Expression]): Either[ConfigReaderFailures, C]
204 
205     protected def encodeConfig(config: C, base: java.util.Map[String, AnyRef]): Unit
206 
207     override def from(cur: ConfigCursor): Either[ConfigReaderFailures, C] = {
208       for {
209         obj      <- cur.asObjectCursor.right
210         typ      <- obj.atKey("type").right.flatMap(_.asString).right
211         idField  <- idFieldFrom(obj.atKeyOrUndefined("id-field")).right
212         userData <- userDataFrom(obj.atKeyOrUndefined("user-data")).right
213         caches   <- configMapFrom(obj.atKeyOrUndefined("caches")).right
214         config   <- decodeConfig(obj, typ, idField, caches, userData).right
215       } yield {
216         config
217       }
218     }
219 
220     override def to(obj: C): ConfigObject = ConfigValueFactory.fromMap(configTo(obj))
221 
222     protected def converterName(cur: ConfigObjectCursor): Either[ConfigReaderFailures, Option[String]] =
223       optionalStringFrom(cur.atKeyOrUndefined(ConverterConfigLoader.ConverterNameKey))
224 
225     private def configTo(config: C): java.util.Map[String, AnyRef] = {
226       val map = new java.util.HashMap[String, AnyRef]
227       map.put("type", config.`type`)
228       config.idField.foreach(f => map.put("id-field", f.toString))
229       if (config.userData.nonEmpty) {
230         map.put("user-data", config.userData.map { case (k, v) => (k, v.toString.asInstanceOf[AnyRef]) }.asJava)
231       }
232       if (config.caches.nonEmpty) {
233         map.put("caches", config.caches.map { case (k, v) => (k, v.root().unwrapped()) })
234       }
235       encodeConfig(config, map)
236       map
237     }

239     private def optionalStringFrom(cur: ConfigCursor): Either[ConfigReaderFailures, Option[String]] =
240       if (cur.isUndefined) { Right(None) } else { cur.asString.right.map(s => Option(s)) }
241 
242     private def idFieldFrom(cur: ConfigCursor): Either[ConfigReaderFailures, Option[Expression]] = {
243       if (cur.isUndefined) { Right(None) } else {
244         for { expr <- exprFrom(cur).right } yield { Some(expr) }

248     private def userDataFrom(cur: ConfigCursor): Either[ConfigReaderFailures, Map[String, Expression]] = {
249       import org.locationtech.geomesa.utils.conf.ConfConversions.RichConfig

251       if (cur.isUndefined) { Right(Map.empty) } else {
252         def merge(cur: ConfigObjectCursor): Either[ConfigReaderFailures, Map[String, Expression]] = {
253           val map = cur.valueOpt.get.toConfig.toStringMap() // handles quoting keys
254           map.foldLeft[Either[ConfigReaderFailures, Map[String, Expression]]](Right(Map.empty)) {
255             case (map, (k, v)) =>
256               // convert back to a cursor for parsing the expression
257               val path = cur.pathElems ++ ConfigUtil.splitPath(k).asScala
258               val valueCursor = ConfigCursor(ConfigValueFactory.fromAnyRef(v), path)
259               for { m <- map.right; d <- exprFrom(valueCursor).right } yield { m + (k -> d) }
260           }
261         }
262         for { obj <- cur.asObjectCursor.right; data <- merge(obj).right } yield { data }
263       }
264     }

278 
279     protected def encodeField(field: F, base: java.util.Map[String, AnyRef]): Unit

281     override def from(cur: ConfigCursor): Either[ConfigReaderFailures, Seq[F]] = {
282       for {
283         obj    <- cur.asObjectCursor.right

296     private def fieldsFrom(cur: ConfigListCursor): Either[ConfigReaderFailures, Seq[F]] = {
297       cur.list.foldLeft[Either[ConfigReaderFailures, Seq[F]]](Right(Seq.empty)) {
298         (list, field) => for { li <- list.right; f <- fieldFrom(field).right } yield { li :+ f }
299       }
300     }
301 
302     private def fieldFrom(cur: ConfigCursor): Either[ConfigReaderFailures, F] = {
303       def transformFrom(cur: ConfigCursor): Either[ConfigReaderFailures, Option[Expression]] = {
304         if (cur.isUndefined) { Right(None) } else {
305           exprFrom(cur).right.map(t => Some(t))
306         }
307       }
308 
309       for {
310         obj       <- cur.asObjectCursor.right
311         name      <- obj.atKey("name").right.flatMap(_.asString).right
312         transform <- transformFrom(obj.atKeyOrUndefined("transform")).right
313         field     <- decodeField(obj, name, transform).right
314       } yield {
315         field
316       }
317     }
318

320       val map = new java.util.HashMap[String, AnyRef]
321       map.put("name", field.name)
322       field.transforms.foreach(t => map.put("transform", t.toString))
323       encodeField(field, map)
324       map
325     }

334   abstract class ConverterOptionsConvert[O <: ConverterOptions]
335       extends ConfigConvert[O] with ConfigSeqConvert with ConfigMapConvert {
336 
337     protected def decodeOptions(
338         cur: ConfigObjectCursor,
339         validators: Seq[String],
340         reporters: Seq[Config],
341         parseMode: ParseMode,
342         errorMode: ErrorMode,
343         encoding: Charset): Either[ConfigReaderFailures, O]
344 
345     protected def encodeOptions(options: O, base: java.util.Map[String, AnyRef]): Unit
346 
347     override def from(cur: ConfigCursor): Either[ConfigReaderFailures, O] = {
348       for {
349         obj     <- cur.asObjectCursor.right
350         options <- obj.atKey("options").right.flatMap(_.asObjectCursor).right.flatMap(optionsFrom).right
351       } yield {
352         options
353       }
354     }
355 
356     override def to(obj: O): ConfigObject = {
357       val map = new java.util.HashMap[String, AnyRef]
358       map.put("options", optionsTo(obj))
359       ConfigValueFactory.fromMap(map)
360     }
361 
362     private def optionsFrom(cur: ConfigObjectCursor): Either[ConfigReaderFailures, O] = {
363 
364       def mergeValidators(cur: ConfigListCursor): Either[ConfigReaderFailures, Seq[String]] = {
365         cur.list.foldLeft[Either[ConfigReaderFailures, Seq[String]]](Right(Seq.empty)) {
366           case (seq, v) => for { s <- seq.right; string <- v.asString.right } yield { s :+ string }
367         }
368       }
369 
370       def parse[T](key: String, values: Iterable[T], fallback: Map[String, T] = Map.empty[String, T]): Either[ConfigReaderFailures, T] = {
371         cur.atKey(key).right.flatMap { value =>
372           value.asString.right.flatMap { string =>
373             val fb = fallback.collectFirst { case (k, v) if k.equalsIgnoreCase(string) => v }
374             values.find(_.toString.equalsIgnoreCase(string)).orElse(fb) match {
375               case Some(v) => Right(v)
376               case None => value.failed(CannotConvert(value.valueOpt.map(_.toString).orNull, values.head.getClass.getSimpleName, s"Must be one of: ${values.mkString(", ")}"))
377             }
378           }
379         }
380       }
381 
382       if (cur.atKey("verbose").isRight) {
383         logger.warn("'verbose' option is deprecated - please use logging levels instead")
384       }
385 
386       for {
387         validators <- cur.atKey("validators").right.flatMap(_.asListCursor).right.flatMap(mergeValidators).right
388         reporters  <- parseReporters(cur.atKeyOrUndefined("reporters")).right
389         parseMode  <- parse("parse-mode", ParseMode.values).right
390         errorMode  <- parse("error-mode", ErrorMode.values, Map("skip-bad-records" -> ErrorMode.LogErrors)).right
391         encoding   <- cur.atKey("encoding").right.flatMap(_.asString).right.map(Charset.forName).right
392         options    <- decodeOptions(cur, validators, reporters, parseMode, errorMode, encoding).right
393       } yield {
394         options
395       }
396     }
397 
398     /**
399       * Reads reporters as a config list, plus checks for back compatible config map
400       *
401       * @param cur cursor
402       * @return
403       */
404     private def parseReporters(cur: ConfigCursor): Either[ConfigReaderFailures, Seq[Config]] = {
405       if (cur.asObjectCursor.isRight) {
406         configMapFrom(cur).right.map(_.values.toList)
407       } else {
408         configSeqFrom(cur)
409       }
410     }

412     private def optionsTo(options: O): java.util.Map[String, AnyRef] = {
413       val map = new java.util.HashMap[String, AnyRef]

415       map.put("error-mode", options.errorMode.toString)

420       }

422       map
423     }
