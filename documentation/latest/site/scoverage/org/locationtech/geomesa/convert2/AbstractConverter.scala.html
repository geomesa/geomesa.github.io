71     // compute only the input fields that we need to deal with to populate the simple feature

215   /**
216     * Convert input values into a simple feature with attributes.
217     *
218     * This method returns a CloseableIterator to simplify flatMapping over inputs, but it will always
219     * return either 0 or 1 feature.
220     *
221     * @param rawValues raw input values
222     * @param ec evaluation context
223     * @return
224     */
225   private def convert(rawValues: Array[AnyRef], ec: EvaluationContext): CloseableIterator[SimpleFeature] = {
226     val result = ec.evaluate(rawValues).right.flatMap { values =>
227       val sf = new ScalaSimpleFeature(sft, "")
228 
229       attributeIndices.foreach { case (i, j) =>
230         sf.setAttributeNoConvert(i, values(j))
231       }
232       // if no id builder, empty feature id will be replaced with an auto-gen one
233       if (idIndex != -1) {
234         values(idIndex) match {
235           case id: String =>
236             sf.setId(id)
237             sf.getUserData.put(Hints.USE_PROVIDED_FID, java.lang.Boolean.TRUE)
238 
239           case null => // no-op
240 
241           case id =>
242             // the missing feature id will be caught by the validator
243             logger.warn(s"Ignoring non-string feature ID: $id")
244         }
245       }
246       userDataIndices.foreach { case (k, j) =>
247         sf.getUserData.put(k, values(j))
248       }
249       val error = validators.validate(sf)
250       if (error == null) {
251         Right(sf)
252       } else {
253         Left(EvaluationError(null, ec.line, new IOException(s"Validation error: $error")))
254       }
255     }
256 
257     result match {
258       case Right(feature) =>
259         ec.stats.success()
260         latency.foreach(_.apply(feature))
261         CloseableIterator.single(feature)
262 
263       case Left(error) =>
264         ec.stats.failure()
265         def msg(verbose: Boolean): String = errorMessage(error, rawValues, verbose)
266         options.errorMode match {
267           case ErrorMode.LogErrors if logger.underlying.isDebugEnabled => logger.debug(msg(verbose = true), error.e)
268           case ErrorMode.LogErrors if logger.underlying.isInfoEnabled => logger.info(msg(verbose = false))
269           case ErrorMode.ReturnErrors => ec.errors.add(error.copy(e = new IOException(msg(verbose = true), error.e)))
270           case ErrorMode.RaiseErrors => throw new IOException(msg(verbose = true), error.e)
271           case _ => // no-op
272         }
273         CloseableIterator.empty
274     }
275   }

277   private def errorMessage(e: EvaluationError, input: Array[AnyRef], verbose: Boolean): String = {
278     val field = if (e.field == null) { "SimpleFeature" } else { s"field '${e.field}'" }
279     val terse = s"Failed to evaluate $field on line ${e.line}"
280     if (!verbose) { terse } else {
281       // head is the whole record
282       s"$terse using values:\n${input.headOption.orNull}\n[${input.drop(1).mkString(", ")}]"
283     }
284   }

399   private def addDependencies[F <: Field](field: Field, fieldMap: Map[String, F], dag: Dag): Unit = {
400     if (!dag.contains(field)) {
401       val deps = field.transforms.toSeq.flatMap(_.dependencies(Set(field), fieldMap)).toSet
402       dag.put(field, deps)
403       deps.foreach(addDependencies(_, fieldMap, dag))
404     }
405   }

407   /**
408     * Returns vertices in topological order.
409     *
410     * Note: will cause an infinite loop if there are circular dependencies
411     *
412     * @param dag graph
413     * @return ordered vertices
414     */
415   private def topologicalOrder(dag: Dag): Array[Field] = {
416     val res = ArrayBuffer.empty[Field]
417     val remaining = new scala.collection.mutable.Queue[Field]() ++ dag.keys
418     while (remaining.nonEmpty) {
419       val next = remaining.dequeue()
420       if (dag(next).forall(res.contains)) {
421         res.append(next)
422       } else {
423         remaining.enqueue(next) // put at the back of the queue
424       }
425     }
426     res.toArray
427   }
