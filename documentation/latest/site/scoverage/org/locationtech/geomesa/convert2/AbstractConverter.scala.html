212   /**
213     * Convert input values into a simple feature with attributes.
214     *
215     * This method returns a CloseableIterator to simplify flatMapping over inputs, but it will always
216     * return either 0 or 1 feature.
217     *
218     * @param rawValues raw input values
219     * @param ec evaluation context
220     * @return
221     */
222   private def convert(rawValues: Array[AnyRef], ec: EvaluationContext): CloseableIterator[SimpleFeature] = {
223     val result = ec.evaluate(rawValues).right.flatMap { values =>
224       val sf = new ScalaSimpleFeature(sft, "")
225 
226       attributeIndices.foreach { case (i, j) =>
227         sf.setAttributeNoConvert(i, values(j))
228       }
229       // if no id builder, empty feature id will be replaced with an auto-gen one
230       if (idIndex != -1) {
231         values(idIndex) match {
232           case id: String =>
233             sf.setId(id)
234             sf.getUserData.put(Hints.USE_PROVIDED_FID, java.lang.Boolean.TRUE)
235 
236           case null => // no-op
237 
238           case id =>
239             // the missing feature id will be caught by the validator
240             logger.warn(s"Ignoring non-string feature ID: $id")
241         }
242       }
243       userDataIndices.foreach { case (k, j) =>
244         sf.getUserData.put(k, values(j))
245       }
246       val error = validators.validate(sf)
247       if (error == null) {
248         Right(sf)
249       } else {
250         Left(EvaluationError(null, ec.line, new IOException(s"Validation error: $error")))
251       }
252     }
253 
254     result match {
255       case Right(feature) =>
256         ec.stats.success()
257         latency.foreach(_.apply(feature))
258         CloseableIterator.single(feature)
259 
260       case Left(error) =>
261         ec.stats.failure()
262         def msg(verbose: Boolean): String = errorMessage(error, rawValues, verbose)
263         options.errorMode match {
264           case ErrorMode.LogErrors if logger.underlying.isDebugEnabled => logger.debug(msg(verbose = true), error.e)
265           case ErrorMode.LogErrors if logger.underlying.isInfoEnabled => logger.info(msg(verbose = false))
266           case ErrorMode.ReturnErrors => ec.errors.add(error.copy(e = new IOException(msg(verbose = true), error.e)))
267           case ErrorMode.RaiseErrors => throw new IOException(msg(verbose = true), error.e)
268           case _ => // no-op
269         }
270         CloseableIterator.empty
271     }
272   }

274   private def errorMessage(e: EvaluationError, input: Array[AnyRef], verbose: Boolean): String = {
275     val field = if (e.field == null) { "SimpleFeature" } else { s"field '${e.field}'" }

390   private def addDependencies[F <: Field](field: Field, fieldMap: Map[String, F], dag: Dag): Unit = {
391     if (!dag.contains(field)) {
392       val deps = field.transforms.toSeq.flatMap(_.dependencies(Set(field), fieldMap)).toSet
393       dag.put(field, deps)

395     }

398   /**
399     * Returns vertices in topological order.
400     *
401     * Note: will cause an infinite loop if there are circular dependencies
402     *
403     * @param dag graph
404     * @return ordered vertices
405     */
406   private def topologicalOrder(dag: Dag): Array[Field] = {
407     val res = ArrayBuffer.empty[Field]
408     val remaining = new scala.collection.mutable.Queue[Field]() ++ dag.keys
409     while (remaining.nonEmpty) {
410       val next = remaining.dequeue()
411       if (dag(next).forall(res.contains)) {
412         res.append(next)
413       } else {
414         remaining.enqueue(next) // put at the back of the queue
415       }
416     }
417     res.toArray
418   }
