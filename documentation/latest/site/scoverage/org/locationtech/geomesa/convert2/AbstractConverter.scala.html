71     // compute only the input fields that we need to deal with to populate the simple feature

207   /**
208     * Convert input values into a simple feature with attributes.
209     *
210     * This method returns a CloseableIterator to simplify flatMapping over inputs, but it will always
211     * return either 0 or 1 feature.
212     *
213     * @param rawValues raw input values
214     * @param ec evaluation context
215     * @return
216     */
217   private def convert(rawValues: Array[AnyRef], ec: EvaluationContext): CloseableIterator[SimpleFeature] = {
218     val result = ec.evaluate(rawValues).right.flatMap { values =>
219       val sf = new ScalaSimpleFeature(sft, "")
220 
221       attributeIndices.foreach { case (i, j) =>
222         sf.setAttributeNoConvert(i, values(j))
223       }
224       // if no id builder, empty feature id will be replaced with an auto-gen one
225       if (idIndex != -1) {
226         values(idIndex) match {
227           case id: String =>
228             sf.setId(id)
229             sf.getUserData.put(Hints.USE_PROVIDED_FID, java.lang.Boolean.TRUE)
230 
231           case null => // no-op
232 
233           case id =>
234             // the missing feature id will be caught by the validator
235             logger.warn(s"Ignoring non-string feature ID: $id")
236         }
237       }
238       userDataIndices.foreach { case (k, j) =>
239         sf.getUserData.put(k, values(j))
240       }
241       val error = validators.validate(sf)
242       if (error == null) {
243         Right(sf)
244       } else {
245         Left(EvaluationError(null, ec.line, new IOException(s"Validation error: $error")))
246       }
247     }
248 
249     result match {
250       case Right(feature) =>
251         ec.stats.success()
252         latency.foreach(_.apply(feature))
253         CloseableIterator.single(feature)
254 
255       case Left(error) =>
256         ec.stats.failure()
257         def msg(verbose: Boolean): String = errorMessage(error, rawValues, verbose)
258         options.errorMode match {
259           case ErrorMode.LogErrors if logger.underlying.isDebugEnabled => logger.debug(msg(verbose = true), error.e)
260           case ErrorMode.LogErrors if logger.underlying.isInfoEnabled => logger.info(msg(verbose = false))
261           case ErrorMode.ReturnErrors => ec.errors.add(error.copy(e = new IOException(msg(verbose = true), error.e)))
262           case ErrorMode.RaiseErrors => throw new IOException(msg(verbose = true), error.e)
263           case _ => // no-op
264         }
265         CloseableIterator.empty
266     }
267   }

269   private def errorMessage(e: EvaluationError, input: Array[AnyRef], verbose: Boolean): String = {

385   private def addDependencies[F <: Field](field: Field, fieldMap: Map[String, F], dag: Dag): Unit = {
386     if (!dag.contains(field)) {
387       val deps = field.transforms.toSeq.flatMap(_.dependencies(Set(field), fieldMap)).toSet
388       dag.put(field, deps)
389       deps.foreach(addDependencies(_, fieldMap, dag))
390     }

393   /**
394     * Returns vertices in topological order.
395     *
396     * Note: will cause an infinite loop if there are circular dependencies
397     *
398     * @param dag graph
399     * @return ordered vertices
400     */
401   private def topologicalOrder(dag: Dag): Array[Field] = {
402     val res = ArrayBuffer.empty[Field]
403     val remaining = new scala.collection.mutable.Queue[Field]() ++ dag.keys
404     while (remaining.nonEmpty) {
405       val next = remaining.dequeue()
406       if (dag(next).forall(res.contains)) {
407         res.append(next)
408       } else {
409         remaining.enqueue(next) // put at the back of the queue
410       }
411     }
412     res.toArray
413   }
