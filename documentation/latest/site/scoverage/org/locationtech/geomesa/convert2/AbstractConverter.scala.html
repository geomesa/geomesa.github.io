70     // compute only the input fields that we need to deal with to populate the simple feature
71     sft.getAttributeDescriptors.asScala.foreach { ad =>

206   /**
207     * Convert input values into a simple feature with attributes.
208     *
209     * This method returns a CloseableIterator to simplify flatMapping over inputs, but it will always
210     * return either 0 or 1 feature.
211     *
212     * @param rawValues raw input values
213     * @param ec evaluation context
214     * @return
215     */
216   private def convert(rawValues: Array[AnyRef], ec: EvaluationContext): CloseableIterator[SimpleFeature] = {
217     val result = ec.evaluate(rawValues).right.flatMap { values =>
218       val sf = new ScalaSimpleFeature(sft, "")
219 
220       attributeIndices.foreach { case (i, j) =>
221         sf.setAttributeNoConvert(i, values(j))
222       }
223       // if no id builder, empty feature id will be replaced with an auto-gen one
224       if (idIndex != -1) {
225         values(idIndex) match {
226           case id: String =>
227             sf.setId(id)
228             sf.getUserData.put(Hints.USE_PROVIDED_FID, java.lang.Boolean.TRUE)
229 
230           case null => // no-op
231 
232           case id =>
233             // the missing feature id will be caught by the validator
234             logger.warn(s"Ignoring non-string feature ID: $id")
235         }
236       }
237       userDataIndices.foreach { case (k, j) =>
238         sf.getUserData.put(k, values(j))
239       }
240       val error = validators.validate(sf)
241       if (error == null) {
242         Right(sf)
243       } else {
244         Left(EvaluationError(null, ec.line, new IOException(s"Validation error: $error")))
245       }
246     }
247 
248     result match {
249       case Right(feature) =>
250         ec.stats.success()
251         latency.foreach(_.apply(feature))
252         CloseableIterator.single(feature)
253 
254       case Left(error) =>
255         ec.stats.failure()
256         def msg(verbose: Boolean): String = errorMessage(error, rawValues, verbose)
257         options.errorMode match {
258           case ErrorMode.LogErrors if logger.underlying.isDebugEnabled => logger.debug(msg(verbose = true), error.e)
259           case ErrorMode.LogErrors if logger.underlying.isInfoEnabled => logger.info(msg(verbose = false))
260           case ErrorMode.ReturnErrors => ec.errors.add(error.copy(e = new IOException(msg(verbose = true), error.e)))
261           case ErrorMode.RaiseErrors => throw new IOException(msg(verbose = true), error.e)
262           case _ => // no-op
263         }
264         CloseableIterator.empty
265     }
266   }

268   private def errorMessage(e: EvaluationError, input: Array[AnyRef], verbose: Boolean): String = {
269     val field = if (e.field == null) { "SimpleFeature" } else { s"field '${e.field}'" }

272       // head is the whole record
273       s"$terse using values:\n${input.headOption.orNull}\n[${input.drop(1).mkString(", ")}]"

384   private def addDependencies[F <: Field](field: Field, fieldMap: Map[String, F], dag: Dag): Unit = {
385     if (!dag.contains(field)) {
386       val deps = field.transforms.toSeq.flatMap(_.dependencies(Set(field), fieldMap)).toSet
387       dag.put(field, deps)
388       deps.foreach(addDependencies(_, fieldMap, dag))
389     }

392   /**
393     * Returns vertices in topological order.
394     *
395     * Note: will cause an infinite loop if there are circular dependencies
396     *
397     * @param dag graph
398     * @return ordered vertices
399     */
400   private def topologicalOrder(dag: Dag): Array[Field] = {
401     val res = ArrayBuffer.empty[Field]
402     val remaining = new scala.collection.mutable.Queue[Field]() ++ dag.keys
403     while (remaining.nonEmpty) {
404       val next = remaining.dequeue()
405       if (dag(next).forall(res.contains)) {
406         res.append(next)
407       } else {
408         remaining.enqueue(next) // put at the back of the queue
409       }
410     }
411     res.toArray
412   }
