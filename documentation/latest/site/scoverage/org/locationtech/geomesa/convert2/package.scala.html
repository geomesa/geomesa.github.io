28   /**
29    * Try a sequence of attempts. Return the first success. If there are no successes,
30    * Failures will be added as suppressed exceptions.
31    *
32    * @param attempts attempts
33    * @param failure umbrella exception in case of failure
34    * @tparam T attempt type
35    * @return
36    */
37   def multiTry[T](attempts: Iterator[Try[T]], failure: Throwable): Try[T] = {
38     var res: Try[T] = Failure(failure)
39     while (res.isFailure && attempts.hasNext) {
40       val attempt = attempts.next
41       if (attempt.isSuccess) {
42         res = attempt
43       } else {
44         val e = attempt.failed.get
45         // filter out UnsupportedOperationException, so we don't clutter up the logs with failed conversion attempts from
46         // converter factories that haven't implemented schema inference
47         if (!e.isInstanceOf[UnsupportedOperationException] && !Option(e.getCause).exists(_.isInstanceOf[UnsupportedOperationException])) {
48           failure.addSuppressed(e)
49         }
50       }
51     }
52     res
53   }

95   /**
96     * Handles errors from the underlying parsing of data, before converting to simple features
97     *
98     * @param underlying wrapped iterator
99     * @param mode error mode
100     * @param ec EvaluationContext for tracking errors
101     * @param timer timer for tracking convert times
102     */
103   class ErrorHandlingIterator[T](
104       underlying: CloseableIterator[T],
105       mode: ErrorMode,
106       ec: EvaluationContext,
107       timer: Timer
108     ) extends CloseableIterator[T] with LazyLogging {
109 
110     private var staged: T = _
111     private var error = false
112 
113     override final def hasNext: Boolean = staged != null || {
114       if (error) { false } else {
115         // make sure that we successfully read an underlying record, so that we can always return
116         // a valid record in `next`, otherwise failures will get double counted
117         val start = System.nanoTime()
118         try {
119           if (underlying.hasNext) {
120             staged = underlying.next
121             true
122           } else {
123             false
124           }
125         } catch {
126           case NonFatal(e) =>
127             ec.stats.failure()
128             mode match {
129               case ErrorMode.LogErrors => logger.warn("Failed parsing input: ", e)
130               case ErrorMode.ReturnErrors => ec.errors.add(EvaluationError(null, ec.line, new IOException("Failed parsing input: ", e)))
131               case ErrorMode.RaiseErrors => throw new IOException("Failed parsing input: ", e)
132             }
133             error = true
134             false // usually parsing can't continue if there is an exception in the underlying read
135         } finally {
136           timer.record(System.nanoTime() - start, TimeUnit.NANOSECONDS)
137         }
138       }
139     }
140 
141     override def next(): T = {
142       if (!hasNext) { Iterator.empty.next() } else {
143         val res = staged
144         staged = null.asInstanceOf[T]
145         res
146       }
147     }
148 
149     override def close(): Unit = underlying.close()
150   }
