27   /**
28    * Try a sequence of attempts. Return the first success. If there are no successes,
29    * Failures will be added as suppressed exceptions.
30    *
31    * @param attempts attempts
32    * @param failure umbrella exception in case of failure
33    * @tparam T attempt type
34    * @return
35    */
36   def multiTry[T](attempts: Iterator[Try[T]], failure: Throwable): Try[T] = {
37     var res: Try[T] = Failure(failure)
38     while (res.isFailure && attempts.hasNext) {
39       val attempt = attempts.next
40       if (attempt.isSuccess) {
41         res = attempt
42       } else {
43         val e = attempt.failed.get
44         // filter out NotImplementedErrors, so we don't clutter up the logs with failed conversion attempts from
45         // converter factories that haven't implemented schema inference
46         if (!e.isInstanceOf[NotImplementedError] && !Option(e.getCause).exists(_.isInstanceOf[NotImplementedError])) {
47           failure.addSuppressed(e)
48         }
49       }
50     }
51     res
52   }

83   /**
84     * Handles errors from the underlying parsing of data, before converting to simple features
85     *
86     * @param underlying wrapped iterator
87     * @param mode error mode
88     * @param ec EvaluationContext for tracking errors
89     * @param times histogram for tracking convert times
90     */
91   class ErrorHandlingIterator[T](
92       underlying: CloseableIterator[T],
93       mode: ErrorMode,
94       ec: EvaluationContext,
95       times: Histogram
96     ) extends CloseableIterator[T] with LazyLogging {
97 
98     private var staged: T = _
99     private var error = false
100 
101     override final def hasNext: Boolean = staged != null || {
102       if (error) { false } else {
103         // make sure that we successfully read an underlying record, so that we can always return
104         // a valid record in `next`, otherwise failures will get double counted
105         val start = System.nanoTime()
106         try {
107           if (underlying.hasNext) {
108             staged = underlying.next
109             true
110           } else {
111             false
112           }
113         } catch {
114           case NonFatal(e) =>
115             ec.failure.inc()
116             mode match {
117               case ErrorMode.LogErrors => logger.warn("Failed parsing input: ", e)
118               case ErrorMode.ReturnErrors => ec.errors.add(EvaluationError(null, ec.line, new IOException("Failed parsing input: ", e)))
119               case ErrorMode.RaiseErrors => throw new IOException("Failed parsing input: ", e)
120             }
121             error = true
122             false // usually parsing can't continue if there is an exception in the underlying read
123         } finally {
124           times.update(System.nanoTime() - start)
125         }
126       }
127     }
128 
129     override def next(): T = {
130       if (!hasNext) { Iterator.empty.next() } else {
131         val res = staged
132         staged = null.asInstanceOf[T]
133         res
134       }
135     }
136 
137     override def close(): Unit = underlying.close()
138   }
