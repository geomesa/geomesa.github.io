24   /**
25    * Try a sequence of attempts. Return the first success. If there are no successes,
26    * Failures will be added as suppressed exceptions.
27    *
28    * @param attempts attempts
29    * @param failure umbrella exception in case of failure
30    * @tparam T attempt type
31    * @return
32    */
33   def multiTry[T](attempts: Iterator[Try[T]], failure: Throwable): Try[T] = {
34     var res: Try[T] = Failure(failure)
35     while (res.isFailure && attempts.hasNext) {
36       val attempt = attempts.next
37       if (attempt.isSuccess) {
38         res = attempt
39       } else {
40         val e = attempt.failed.get
41         // filter out NotImplementedErrors, so we don't clutter up the logs with failed conversion attempts from
42         // converter factories that haven't implemented schema inference
43         if (!e.isInstanceOf[NotImplementedError] && !Option(e.getCause).exists(_.isInstanceOf[NotImplementedError])) {
44           failure.addSuppressed(e)
45         }
46       }
47     }
48     res
49   }

80   /**
81     * Handles errors from the underlying parsing of data, before converting to simple features
82     *
83     * @param underlying wrapped iterator
84     * @param mode error mode
85     * @param counter counter for failures
86     * @param times histogram for tracking convert times
87     */
88   class ErrorHandlingIterator[T](
89       underlying: CloseableIterator[T],
90       mode: ErrorMode,
91       counter: Counter,
92       times: Histogram
93     ) extends CloseableIterator[T] with LazyLogging {
94 
95     private var staged: T = _
96     private var error = false
97 
98     override final def hasNext: Boolean = staged != null || {
99       if (error) { false } else {
100         // make sure that we successfully read an underlying record, so that we can always return
101         // a valid record in `next`, otherwise failures will get double counted
102         val start = System.nanoTime()
103         try {
104           if (underlying.hasNext) {
105             staged = underlying.next
106             true
107           } else {
108             false
109           }
110         } catch {
111           case NonFatal(e) =>
112             counter.inc()
113             mode match {
114               case ErrorMode.SkipBadRecords => logger.warn("Failed parsing input: ", e)
115               case ErrorMode.RaiseErrors => throw e
116             }
117             error = true
118             false // usually parsing can't continue if there is an exception in the underlying read
119         } finally {
120           times.update(System.nanoTime() - start)
121         }
122       }
123     }
124 
125     override def next(): T = {
126       if (!hasNext) { Iterator.empty.next() } else {
127         val res = staged
128         staged = null.asInstanceOf[T]
129         res
130       }
131     }
132 
133     override def close(): Unit = underlying.close()
134   }
