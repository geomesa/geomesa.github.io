28   /**
29    * Try a sequence of attempts. Return the first success. If there are no successes,
30    * Failures will be added as suppressed exceptions.
31    *
32    * @param attempts attempts
33    * @param failure umbrella exception in case of failure
34    * @tparam T attempt type
35    * @return
36    */
37   def multiTry[T](attempts: Iterator[Try[T]], failure: Throwable): Try[T] = {
38     var res: Try[T] = Failure(failure)
39     while (res.isFailure && attempts.hasNext) {
40       val attempt = attempts.next
41       if (attempt.isSuccess) {
42         res = attempt
43       } else {
44         val e = attempt.failed.get
45         // filter out UnsupportedOperationException, so we don't clutter up the logs with failed conversion attempts from
46         // converter factories that haven't implemented schema inference
47         if (!e.isInstanceOf[UnsupportedOperationException] && !Option(e.getCause).exists(_.isInstanceOf[UnsupportedOperationException])) {
48           failure.addSuppressed(e)
49         }
50       }
51     }
52     res
53   }

97   /**
98     * Handles errors from the underlying parsing of data, before converting to simple features
99     *
100     * @param underlying wrapped iterator
101     * @param mode error mode
102     * @param ec EvaluationContext for tracking errors
103     * @param timer timer for tracking convert times
104     */
105   class ErrorHandlingIterator[T](
106       underlying: CloseableIterator[T],
107       mode: ErrorMode,
108       ec: EvaluationContext,
109       timer: Timer
110     ) extends CloseableIterator[T] with LazyLogging {
111 
112     private var staged: T = _
113     private var error = false
114 
115     override final def hasNext: Boolean = staged != null || {
116       if (error) { false } else {
117         // make sure that we successfully read an underlying record, so that we can always return
118         // a valid record in `next`, otherwise failures will get double counted
119         val start = System.nanoTime()
120         try {
121           if (underlying.hasNext) {
122             staged = underlying.next
123             true
124           } else {
125             false
126           }
127         } catch {
128           case NonFatal(e) =>
129             ec.stats.failure()
130             mode match {
131               case ErrorMode.LogErrors => logger.warn("Failed parsing input: ", e)
132               case ErrorMode.ReturnErrors => ec.errors.add(EvaluationError(null, ec.line, new IOException("Failed parsing input: ", e)))
133               case ErrorMode.RaiseErrors => throw new IOException("Failed parsing input: ", e)
134             }
135             error = true
136             false // usually parsing can't continue if there is an exception in the underlying read
137         } finally {
138           timer.record(System.nanoTime() - start, TimeUnit.NANOSECONDS)
139         }
140       }
141     }
142 
143     override def next(): T = {
144       if (!hasNext) { Iterator.empty.next() } else {
145         val res = staged
146         staged = null.asInstanceOf[T]
147         res
148       }
149     }
150 
151     override def close(): Unit = underlying.close()
152   }
