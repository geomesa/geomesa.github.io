20 /**
21   * Validator based on the indices used by the feature type. Currently only the x/z2 and x/z3 indices have
22   * input requirements. In addition, features that validate against the x/z3 index will also validate against
23   * the x/z2 index
24   */
25 class IndexValidatorFactory extends SimpleFeatureValidatorFactory {
26 
27   import IndexValidatorFactory._
28   import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
29 
30   override val name: String = IndexValidatorFactory.Name
31 
32   override def apply(sft: SimpleFeatureType, metrics: ConverterMetrics, config: Option[String]): SimpleFeatureValidator =
33     apply(sft, config, Tags.empty())
34 
35   override def apply(sft: SimpleFeatureType, config: Option[String], tags: Tags): SimpleFeatureValidator = {
36     val geom = sft.getGeomIndex
37     val dtg = sft.getDtgIndex.getOrElse(-1)
38     val enabled = sft.getIndices.collect { case id if id.mode.write => id.name.toLowerCase(Locale.US) }
39     if (enabled.contains("z3") || enabled.contains("xz3") || (enabled.isEmpty && geom != -1 && dtg != -1)) {
40       val minDate = Date.from(BinnedTime.ZMinDate.toInstant)
41       val maxDate = Date.from(BinnedTime.maxDate(sft.getZ3Interval).toInstant)
42       new Z3Validator(geom, dtg, minDate, maxDate, ErrorCounters("geom", tags), ErrorCounters("dtg", tags))
43     } else if (enabled.contains("z2") || enabled.contains("xz2") || (enabled.isEmpty && geom != -1)) {
44       new Z2Validator(geom, ErrorCounters("geom", tags))
45     } else {
46       NoValidator
47     }
48   }
49 }

58   private case class ErrorCounters(missing: Counter, bounds: Counter)
59 
60   private object ErrorCounters {
61     def apply(field: String, tags: Tags): ErrorCounters =
62       ErrorCounters(counter(s"$field.null", tags), counter(s"$field.bounds.invalid", tags))
63   }

65   /**
66    * Z2 validator
67    *
68    * @param geom geom index
69    * @param geomErrors geometry error counters
70    */
71   private class Z2Validator(geom: Int, geomErrors: ErrorCounters) extends SimpleFeatureValidator {
72     override def validate(sf: SimpleFeature): String = {
73       val g = sf.getAttribute(geom).asInstanceOf[Geometry]
74       if (g == null) {
75         geomErrors.missing.increment()
76         Errors.GeomNull
77       } else if (!WholeWorldEnvelope.contains(g.getEnvelopeInternal)) {
78         geomErrors.bounds.increment()
79         Errors.geomBounds(g)
80       } else {
81         null
82       }
83     }
84 
85     override def close(): Unit = {}
86   }

88   /**
89    * Z3 validator
90    *
91    * @param geom geom index
92    * @param dtg dtg index
93    * @param minDate min z3 date
94    * @param maxDate max z3 date
95    * @param geomErrors geometry error counters
96    * @param dtgErrors date error counters
97    */
98   private class Z3Validator(geom: Int, dtg: Int, minDate: Date, maxDate: Date, geomErrors: ErrorCounters, dtgErrors: ErrorCounters)
99       extends SimpleFeatureValidator {
100 
101     private val dateBefore = Errors.dateBoundsLow(minDate)
102     private val dateAfter = Errors.dateBoundsHigh(maxDate)
103 
104     override def validate(sf: SimpleFeature): String = {
105       val d = sf.getAttribute(dtg).asInstanceOf[Date]
106       val g = sf.getAttribute(geom).asInstanceOf[Geometry]
107       var error: String = null
108       if (g == null) {
109         geomErrors.missing.increment()
110         error =  s"$error, ${Errors.GeomNull}"
111       } else if (!WholeWorldEnvelope.contains(g.getEnvelopeInternal)) {
112         geomErrors.bounds.increment()
113         error =  s"$error, ${Errors.geomBounds(g)}"
114       }
115       if (d == null) {
116         dtgErrors.missing.increment()
117         error = s"$error, ${Errors.DateNull}"
118       } else if (d.before(minDate)) {
119         dtgErrors.bounds.increment()
120         error = s"$error, ${dateBefore(d)}"
121       } else if (d.after(maxDate)) {
122         dtgErrors.bounds.increment()
123         error = s"$error, ${dateAfter(d)}"
124       }
125       if (error == null) { null } else {
126         error.substring(6) // trim off leading 'null, '
127       }
128     }
129 
130     override def close(): Unit = {}
131   }
