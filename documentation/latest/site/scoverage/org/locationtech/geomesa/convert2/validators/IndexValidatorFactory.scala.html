20 /**
21   * Validator based on the indices used by the feature type. Currently only the x/z2 and x/z3 indices have
22   * input requirements. In addition, features that validate against the x/z3 index will also validate against
23   * the x/z2 index
24   */
25 class IndexValidatorFactory extends SimpleFeatureValidatorFactory {
26 
27   import IndexValidatorFactory._
28   import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
29 
30   override val name: String = IndexValidatorFactory.Name
31   override def apply(
32       sft: SimpleFeatureType,
33       metrics: ConverterMetrics,
34       config: Option[String]): SimpleFeatureValidator = {
35     val geom = sft.getGeomIndex
36     val dtg = sft.getDtgIndex.getOrElse(-1)
37     val enabled = sft.getIndices.collect { case id if id.mode.write => id.name.toLowerCase(Locale.US) }
38     if (enabled.contains("z3") || enabled.contains("xz3") || (enabled.isEmpty && geom != -1 && dtg != -1)) {
39       val minDate = Date.from(BinnedTime.ZMinDate.toInstant)
40       val maxDate = Date.from(BinnedTime.maxDate(sft.getZ3Interval).toInstant)
41       val counters = {
42         val geom = ErrorCounters(metrics.counter(GeomNullCounter), metrics.counter(GeomBoundsCounter))
43         val dtg = ErrorCounters(metrics.counter(DtgNullCounter), metrics.counter(DtgBoundsCounter))
44         Z3Counters(geom, dtg, metrics.counter(Z3TotalCounter))
45       }
46       new Z3Validator(geom, dtg, minDate, maxDate, counters)
47     } else if (enabled.contains("z2") || enabled.contains("xz2") || (enabled.isEmpty && geom != -1)) {
48       val counters = {
49         val geom = ErrorCounters(metrics.counter(GeomNullCounter), metrics.counter(GeomBoundsCounter))
50         Z2Counters(geom, metrics.counter(Z2TotalCounter))
51       }
52       new Z2Validator(geom, counters)
53     } else {
54       NoValidator
55     }
56   }
57 }

80   /**
81     * Z2 validator
82     *
83     * @param geom geom index
84     */
85   private class Z2Validator(geom: Int, counters: Z2Counters) extends SimpleFeatureValidator {
86     override def validate(sf: SimpleFeature): String = {
87       val g = sf.getAttribute(geom).asInstanceOf[Geometry]
88       if (g == null) {
89         counters.geom.missing.inc()
90         counters.total.inc()
91         Errors.GeomNull
92       } else if (!WholeWorldEnvelope.contains(g.getEnvelopeInternal)) {
93         counters.geom.bounds.inc()
94         counters.total.inc()
95         Errors.geomBounds(g)
96       } else {
97         null
98       }
99     }
100 
101     override def close(): Unit = {}
102   }

104   /**
105     * Z3 validator
106     *
107     * @param geom geom index
108     * @param dtg dtg index
109     * @param minDate min z3 date
110     * @param maxDate max z3 date
111     */
112   private class Z3Validator(geom: Int, dtg: Int, minDate: Date, maxDate: Date, counters: Z3Counters)
113       extends SimpleFeatureValidator {
114 
115     private val dateBefore = Errors.dateBoundsLow(minDate)
116     private val dateAfter = Errors.dateBoundsHigh(maxDate)
117 
118     override def validate(sf: SimpleFeature): String = {
119       val d = sf.getAttribute(dtg).asInstanceOf[Date]
120       val g = sf.getAttribute(geom).asInstanceOf[Geometry]
121       var error: String = null
122       if (g == null) {
123         counters.geom.missing.inc()
124         error =  s"$error, ${Errors.GeomNull}"
125       } else if (!WholeWorldEnvelope.contains(g.getEnvelopeInternal)) {
126         counters.geom.bounds.inc()
127         error =  s"$error, ${Errors.geomBounds(g)}"
128       }
129       if (d == null) {
130         counters.dtg.missing.inc()
131         error = s"$error, ${Errors.DateNull}"
132       } else if (d.before(minDate)) {
133         counters.dtg.bounds.inc()
134         error = s"$error, ${dateBefore(d)}"
135       } else if (d.after(maxDate)) {
136         counters.dtg.bounds.inc()
137         error = s"$error, ${dateAfter(d)}"
138       }
139       if (error == null) { null } else {
140         counters.total.inc()
141         error.substring(6) // trim off leading 'null, '
142       }
143     }
144 
145     override def close(): Unit = {}
146   }
