22 /**
23   * Validator based on the indices used by the feature type. Currently only the x/z2 and x/z3 indices have
24   * input requirements. In addition, features that validate against the x/z3 index will also validate against
25   * the x/z2 index
26   */
27 class IndexValidatorFactory extends SimpleFeatureValidatorFactory {
28 
29   import IndexValidatorFactory._
30   import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
31 
32   override val name: String = IndexValidatorFactory.Name
33 
34   override def apply(sft: SimpleFeatureType, config: Option[String], tags: Tags): SimpleFeatureValidator = {
35     val geom = sft.getGeomIndex
36     val dtg = sft.getDtgIndex.getOrElse(-1)
37     val enabled = sft.getIndices.collect { case id if id.mode.write => id.name.toLowerCase(Locale.US) }
38     if (enabled.contains("z3") || enabled.contains("xz3") || (enabled.isEmpty && geom != -1 && dtg != -1)) {
39       val minDate = Date.from(BinnedTime.ZMinDate.toInstant)
40       val maxDate = Date.from(BinnedTime.maxDate(sft.getZ3Interval).toInstant)
41       new Z3Validator(Attribute(sft, geom), Attribute(sft, dtg), minDate, maxDate, tags)
42     } else if (enabled.contains("z2") || enabled.contains("xz2") || (enabled.isEmpty && geom != -1)) {
43       new Z2Validator(Attribute(sft, geom), tags)
44     } else {
45       NoValidator
46     }
47   }
48 }

60   /**
61    * Z2 validator
62    *
63    * @param geom geom
64    * @param tags metrics tags
65    */
66   private class Z2Validator(geom: Attribute, tags: Tags) extends SimpleFeatureValidator {
67 
68     private val success = successCounter("z2", geom.name, tags)
69     private val nullFailure = failureCounter("z2", geom.name, "null", tags)
70     private val boundsFailure = failureCounter("z2", geom.name, "invalid.bounds", tags)
71 
72     override def validate(sf: SimpleFeature): String = {
73       val g = sf.getAttribute(geom.i).asInstanceOf[Geometry]
74       if (g == null) {
75         nullFailure.increment()
76         "geometry is null"
77       } else if (!WholeWorldEnvelope.contains(g.getEnvelopeInternal)) {
78         boundsFailure.increment()
79         geomBoundsError(g)
80       } else {
81         success.increment()
82         null
83       }
84     }
85 
86     override def close(): Unit = {}
87   }

89   /**
90    * Z3 validator
91    *
92    * @param geom geom
93    * @param dtg dtg
94    * @param minDate min z3 date
95    * @param maxDate max z3 date
96    * @param tags metrics tags
97    */
98   private class Z3Validator(geom: Attribute, dtg: Attribute, minDate: Date, maxDate: Date, tags: Tags)
99       extends SimpleFeatureValidator {
100 
101     private val success = successCounter("z3", s"${geom.name},${dtg.name}", tags)
102     private val geomNullFailure = failureCounter("z3", geom.name, "null", tags)
103     private val geomBoundsFailure = failureCounter("z3", geom.name, "invalid.bounds", tags)
104     private val dtgNullFailure = failureCounter("z3", dtg.name, "null", tags)
105     private val dtgBoundsFailure = failureCounter("z3", dtg.name, "invalid.bounds", tags)
106 
107     private val dateBefore = s"date is before minimum indexable date (${GeoToolsDateFormat.format(Instant.ofEpochMilli(minDate.getTime))}):"
108     private val dateAfter = s"date is after maximum indexable date (${GeoToolsDateFormat.format(Instant.ofEpochMilli(maxDate.getTime))}):"
109 
110     override def validate(sf: SimpleFeature): String = {
111       val d = sf.getAttribute(dtg.i).asInstanceOf[Date]
112       val g = sf.getAttribute(geom.i).asInstanceOf[Geometry]
113       var error: String = null
114       if (g == null) {
115         geomNullFailure.increment()
116         error =  s"$error, geometry is null"
117       } else if (!WholeWorldEnvelope.contains(g.getEnvelopeInternal)) {
118         geomBoundsFailure.increment()
119         error =  s"$error, ${geomBoundsError(g)}"
120       }
121       if (d == null) {
122         dtgNullFailure.increment()
123         error = s"$error, date is null"
124       } else if (d.before(minDate)) {
125         dtgBoundsFailure.increment()
126         error = s"$error, $dateBefore ${GeoToolsDateFormat.format(Instant.ofEpochMilli(d.getTime))}"
127       } else if (d.after(maxDate)) {
128         dtgBoundsFailure.increment()
129         error = s"$error, $dateAfter ${GeoToolsDateFormat.format(Instant.ofEpochMilli(d.getTime))}"
130       }
131       if (error == null) {
132         success.increment()
133         null
134       } else {
135         error.substring(6) // trim off leading 'null, '
136       }
137     }
138 
139     override def close(): Unit = {}
140   }
