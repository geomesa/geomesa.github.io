106   private val murmur3_128 =
107     TransformerFunction.pure("murmurHash3") { args =>
108       val bytes = args(0) match {
109         case s: String => s.getBytes(StandardCharsets.UTF_8)
110         case b: Array[Byte] => b
111         case a => throw new IllegalArgumentException(s"Expected String or byte[] but got: $a")
112       }
113       // mimic guava little-endian output
114       val sb = new StringBuilder(32)
115       MurmurHash3.hash128x64(bytes, 0, bytes.length, 0).foreach { hash =>
116         var i = 0
117         while (i < 64) {
118           sb.append(ByteArrays.toHex(((hash >> i) & 0xff).asInstanceOf[Byte]))
119           i += 8
120         }
121       }
122       sb.toString
123     }
