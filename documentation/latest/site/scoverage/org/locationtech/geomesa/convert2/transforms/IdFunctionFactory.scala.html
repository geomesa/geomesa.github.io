99   private val murmur3_128 =
100     TransformerFunction.pure("murmurHash3") { args =>
101       val bytes = args(0) match {
102         case s: String => s.getBytes(StandardCharsets.UTF_8)
103         case b: Array[Byte] => b
104         case a => throw new IllegalArgumentException(s"Expected String or byte[] but got: $a")
105       }
106       // mimic guava little-endian output
107       val sb = new StringBuilder(32)
108       MurmurHash3.hash128x64(bytes, 0, bytes.length, 0).foreach { hash =>
109         var i = 0
110         while (i < 64) {
111           sb.append(ByteArrays.toHex(((hash >> i) & 0xff).asInstanceOf[Byte]))
112           i += 8
113         }
114       }
115       sb.toString
116     }
