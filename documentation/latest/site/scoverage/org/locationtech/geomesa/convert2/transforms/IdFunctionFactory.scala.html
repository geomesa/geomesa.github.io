57   private val md5: TransformerFunction = new NamedTransformerFunction(Seq("md5"), pure = true) {
58     private val hashers = new ThreadLocal[MessageDigest]() {
59       override def initialValue(): MessageDigest = MessageDigest.getInstance("MD5")
60     }
61     override def apply(args: Array[AnyRef]): AnyRef = {
62       val bytes = args(0) match {
63         case s: String => s.getBytes(StandardCharsets.UTF_8)
64         case b: Array[Byte] => b
65         case a => throw new IllegalArgumentException(s"Expected String or byte[] but got: $a")
66       }
67       ByteArrays.toHex(hashers.get.digest(bytes))

82   // we've had some confusion around the names of these functions - the original function was murmur3_64,
83   // which was then incorrectly renamed to murmur3_128. currently both these functions only return the first 64
84   // bits of a 128 bit hash. the full 128-bit hash is now called murmurHash3 to avoid name conflicts
85   private val murmur3_64 =
86     TransformerFunction.pure("murmur3_128", "murmur3_64") { args =>
87       val bytes = args(0) match {
88         case s: String => s.getBytes(StandardCharsets.UTF_8)
89         case b: Array[Byte] => b
90         case a => throw new IllegalArgumentException(s"Expected String or byte[] but got: $a")
91       }
92       Long.box(MurmurHash3.hash128x64(bytes, 0, bytes.length, 0).head)
93     }
94 
95   private val murmur3_128 =
96     TransformerFunction.pure("murmurHash3") { args =>
97       val bytes = args(0) match {
98         case s: String => s.getBytes(StandardCharsets.UTF_8)
99         case b: Array[Byte] => b
100         case a => throw new IllegalArgumentException(s"Expected String or byte[] but got: $a")
101       }
102       // mimic guava little-endian output
103       val sb = new StringBuilder(32)
104       MurmurHash3.hash128x64(bytes, 0, bytes.length, 0).foreach { hash =>
105         var i = 0
106         while (i < 64) {
107           sb.append(ByteArrays.toHex(((hash >> i) & 0xff).asInstanceOf[Byte]))
108           i += 8
109         }
110       }
111       sb.toString
112     }
