65   private val md5: TransformerFunction = new NamedTransformerFunction(Seq("md5"), pure = true) {

90   // we've had some confusion around the names of these functions - the original function was murmur3_64,
91   // which was then incorrectly renamed to murmur3_128. currently both these functions only return the first 64
92   // bits of a 128 bit hash. the full 128-bit hash is now called murmurHash3 to avoid name conflicts
93   private val murmur3_64 =
94     TransformerFunction.pure("murmur3_128", "murmur3_64") { args =>

97         case b: Array[Byte] => b
98         case a => throw new IllegalArgumentException(s"Expected String or byte[] but got: $a")
99       }
100       Long.box(MurmurHash3.hash128x64(bytes, 0, bytes.length, 0).head)
101     }
102 
103   private val murmur3_128 =
104     TransformerFunction.pure("murmurHash3") { args =>
105       val bytes = args(0) match {
106         case s: String => s.getBytes(StandardCharsets.UTF_8)
107         case b: Array[Byte] => b
108         case a => throw new IllegalArgumentException(s"Expected String or byte[] but got: $a")
109       }
110       // mimic guava little-endian output
111       val sb = new StringBuilder(32)
112       MurmurHash3.hash128x64(bytes, 0, bytes.length, 0).foreach { hash =>
113         var i = 0
114         while (i < 64) {
115           sb.append(ByteArrays.toHex(((hash >> i) & 0xff).asInstanceOf[Byte]))
116           i += 8
117         }
118       }
119       sb.toString
120     }
