101   private val murmur3_128 =
102     TransformerFunction.pure("murmurHash3") { args =>
103       val bytes = args(0) match {
104         case s: String => s.getBytes(StandardCharsets.UTF_8)
105         case b: Array[Byte] => b
106         case a => throw new IllegalArgumentException(s"Expected String or byte[] but got: $a")
107       }
108       // mimic guava little-endian output
109       val sb = new StringBuilder(32)
110       MurmurHash3.hash128x64(bytes, 0, bytes.length, 0).foreach { hash =>
111         var i = 0
112         while (i < 64) {
113           sb.append(ByteArrays.toHex(((hash >> i) & 0xff).asInstanceOf[Byte]))
114           i += 8
115         }
116       }
117       sb.toString
118     }
