29 abstract class AbstractCompositeConverter[T <: AnyRef](
30     sft: SimpleFeatureType,
31     errorMode: ErrorMode,
32     delegates: Seq[(Predicate, ParsingConverter[T])]
33   ) extends SimpleFeatureConverter with LazyLogging {
34 
35   private val tags = ConverterMetrics.typeNameTag(sft)
36 
37   protected def parse(is: InputStream, ec: EvaluationContext): CloseableIterator[T]
38 
39   override def targetSft: SimpleFeatureType = sft
40 
41   override def createEvaluationContext(globalParams: Map[String, Any]): EvaluationContext =
42     createEvaluationContext(delegates.map(_._2.createEvaluationContext(globalParams)), Stats(tags))
43 
44   override def createEvaluationContext(globalParams: Map[String, Any], success: Counter, failure: Counter): EvaluationContext =
45     createEvaluationContext(delegates.map(_._2.createEvaluationContext(globalParams, success, failure)), Stats.wrap(success, failure, tags))
46 
47   private def createEvaluationContext(contexts: Seq[EvaluationContext], theseStats: Stats): EvaluationContext = {
48     val stats = new Stats() {
49       override def success(i: Int): Long = theseStats.success(i) + contexts.map(_.stats.success(0)).sum
50       override def failure(i: Int): Long = theseStats.failure(i) + contexts.map(_.stats.failure(0)).sum
51     }
52     CompositeEvaluationContext(contexts, stats)
53   }
54 
55   override def process(is: InputStream, ec: EvaluationContext): CloseableIterator[SimpleFeature] = {
56     val contexts = ec match {
57       case cec: CompositeEvaluationContext => cec.contexts.iterator
58       case _ => Iterator.continually(ec)
59     }
60     val predicates = delegates.map { case (predicate, converter) =>
61       if (!contexts.hasNext) {
62         throw new IllegalArgumentException(s"Invalid evaluation context doesn't match this converter: $ec")
63       }
64       val context = contexts.next()
65       PredicateContext(predicate.withContext(context), converter, context)
66     }
67 
68     val args = Array.ofDim[AnyRef](1)
69 
70     def matches(p: Predicate): Boolean = {
71       try { p.apply(args) } catch {
72         case NonFatal(e) => logger.warn(s"Error evaluating predicate $p: ", e); false
73       }
74     }
75 
76     val routing =
77       Timer.builder(ConverterMetrics.name("predicate.eval"))
78         .tags(tags)
79         .publishPercentileHistogram()
80         .minimumExpectedValue(Duration.ofNanos(1))
81         .maximumExpectedValue(Duration.ofMillis(2))
82         .register(Metrics.globalRegistry)
83 
84     def eval(element: T): CloseableIterator[SimpleFeature] = {
85       val start = System.nanoTime()
86       args(0) = element
87       val converted = predicates.collectFirst { case p if matches(p.predicate) =>
88         routing.record(System.nanoTime() - start, TimeUnit.NANOSECONDS) // note: invoke before executing the conversion
89         p.context.line = ec.line // update the line in the sub context
90         p.converter.convert(CloseableIterator.single(element), p.context)
91       }
92       converted.getOrElse {
93         routing.record(System.nanoTime() - start, TimeUnit.NANOSECONDS)
94         ec.stats.failure()
95         CloseableIterator.empty
96       }
97     }
98 
99     val parseTimer =
100       Timer.builder(ConverterMetrics.name("parse"))
101         .tags(tags)
102         .publishPercentileHistogram()
103         .minimumExpectedValue(Duration.ofNanos(1))
104         .maximumExpectedValue(Duration.ofMillis(10))
105         .register(Metrics.globalRegistry)
106     new ErrorHandlingIterator(parse(is, ec), errorMode, ec, parseTimer).flatMap(eval)
107   }
108 
109   override def close(): Unit = CloseWithLogging(delegates.map(_._2))
110 }

115     private val errs = new java.util.ArrayDeque[EvaluationError]()
116     override def errors: java.util.Queue[EvaluationError] = {
117       contexts.foreach { c =>
118         val iter = c.errors.iterator()
119         while (iter.hasNext) {
120           errs.add(iter.next())
121           iter.remove()
122         }
123       }
124       errs
125     }
126     override def withListener(listener: ContextListener): EvaluationContext =
127       CompositeEvaluationContext(contexts.map(_.withListener(listener)), StatListener(stats, listener))
128 
129     override def metrics: ConverterMetrics = contexts.head.metrics
130     override val success: com.codahale.metrics.Counter = new com.codahale.metrics.Counter() {

134     override val failure: com.codahale.metrics.Counter = new com.codahale.metrics.Counter() {
135       override def inc(n: Long): Unit = stats.failure(n.toInt)
136       override def getCount: Long = stats.failure(0)
137     }
138     override def cache: Map[String, EnrichmentCache] = throw new UnsupportedOperationException()
139     override def accessor(name: String): FieldAccessor = throw new UnsupportedOperationException()
140     override def evaluate(args: Array[AnyRef]): Either[EvaluationError, Array[AnyRef]] =
141       throw new UnsupportedOperationException()
142   }
