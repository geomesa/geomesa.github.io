30 abstract class AbstractCompositeConverter[T <: AnyRef](
31     sft: SimpleFeatureType,
32     errorMode: ErrorMode,
33     delegates: Seq[(Predicate, ParsingConverter[T])]
34   ) extends SimpleFeatureConverter with LazyLogging {
35 
36   private val tags = TagUtils.typeNameTag(sft.getTypeName)
37 
38   private val routingTimer =
39     Timer.builder(ConverterMetrics.name("predicate.eval"))
40       .tags(tags)
41       .publishPercentileHistogram()
42       .minimumExpectedValue(Duration.ofNanos(1))
43       .maximumExpectedValue(Duration.ofMillis(2))
44       .register(Metrics.globalRegistry)
45 
46   private val parseTimer =
47     Timer.builder(ConverterMetrics.name("parse"))
48       .tags(tags)
49       .publishPercentileHistogram()
50       .minimumExpectedValue(Duration.ofNanos(1))
51       .maximumExpectedValue(Duration.ofMillis(10))
52       .register(Metrics.globalRegistry)
53 
54   protected def parse(is: InputStream, ec: EvaluationContext): CloseableIterator[T]
55 
56   override def targetSft: SimpleFeatureType = sft
57 
58   override def createEvaluationContext(globalParams: Map[String, Any]): EvaluationContext =
59     createEvaluationContext(delegates.map(_._2.createEvaluationContext(globalParams)), Stats(tags))
60 
61   override def createEvaluationContext(globalParams: Map[String, Any], success: Counter, failure: Counter): EvaluationContext =
62     createEvaluationContext(delegates.map(_._2.createEvaluationContext(globalParams, success, failure)), Stats.wrap(success, failure, tags))
63 
64   private def createEvaluationContext(contexts: Seq[EvaluationContext], theseStats: Stats): EvaluationContext = {
65     val stats = new Stats() {
66       override def success(i: Int): Long = theseStats.success(i) + contexts.map(_.stats.success(0)).sum
67       override def failure(i: Int): Long = theseStats.failure(i) + contexts.map(_.stats.failure(0)).sum
68     }
69     CompositeEvaluationContext(contexts, stats)
70   }
71 
72   override def process(is: InputStream, ec: EvaluationContext): CloseableIterator[SimpleFeature] = {
73     val contexts = ec match {
74       case cec: CompositeEvaluationContext => cec.contexts.iterator
75       case _ => Iterator.continually(ec)
76     }
77     val predicates = delegates.map { case (predicate, converter) =>
78       if (!contexts.hasNext) {
79         throw new IllegalArgumentException(s"Invalid evaluation context doesn't match this converter: $ec")
80       }
81       val context = contexts.next()
82       PredicateContext(predicate.withContext(context), converter, context)
83     }
84 
85     val args = Array.ofDim[AnyRef](1)
86 
87     def matches(p: Predicate): Boolean = {
88       try { p.apply(args) } catch {
89         case NonFatal(e) => logger.warn(s"Error evaluating predicate $p: ", e); false
90       }
91     }
92 
93     def eval(element: T): CloseableIterator[SimpleFeature] = {
94       val start = System.nanoTime()
95       args(0) = element
96       val converted = predicates.collectFirst { case p if matches(p.predicate) =>
97         routingTimer.record(System.nanoTime() - start, TimeUnit.NANOSECONDS) // note: invoke before executing the conversion
98         p.context.line = ec.line // update the line in the sub context
99         p.converter.convert(CloseableIterator.single(element), p.context)
100       }
101       converted.getOrElse {
102         routingTimer.record(System.nanoTime() - start, TimeUnit.NANOSECONDS)
103         ec.stats.failure()
104         CloseableIterator.empty
105       }
106     }
107 
108     new ErrorHandlingIterator(parse(is, ec), errorMode, ec, parseTimer).flatMap(eval)
109   }
110 
111   override def close(): Unit = CloseWithLogging(delegates.map(_._2))
112 }
