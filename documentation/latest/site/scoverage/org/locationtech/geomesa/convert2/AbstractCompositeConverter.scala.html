29 abstract class AbstractCompositeConverter[T <: AnyRef](
30     sft: SimpleFeatureType,
31     errorMode: ErrorMode,

35   private val tags = TagUtils.typeNameTag(sft.getTypeName)

37   private val routingTimer =
38     Timer.builder(ConverterMetrics.name("predicate.eval"))
39       .tags(tags)

53   protected def parse(is: InputStream, ec: EvaluationContext): CloseableIterator[T]
54 
55   override def targetSft: SimpleFeatureType = sft
56 
57   override def createEvaluationContext(globalParams: Map[String, Any]): EvaluationContext =
58     createEvaluationContext(delegates.map(_._2.createEvaluationContext(globalParams)), Stats(tags))
59 
60   private def createEvaluationContext(contexts: Seq[EvaluationContext], theseStats: Stats): EvaluationContext = {
61     val stats = new Stats() {
62       override def success(i: Int): Long = theseStats.success(i) + contexts.map(_.stats.success(0)).sum
63       override def failure(i: Int): Long = theseStats.failure(i) + contexts.map(_.stats.failure(0)).sum
64     }
65     CompositeEvaluationContext(contexts, stats)
66   }
67 
68   override def process(is: InputStream, ec: EvaluationContext): CloseableIterator[SimpleFeature] = {
69     val contexts = ec match {
70       case cec: CompositeEvaluationContext => cec.contexts.iterator
71       case _ => Iterator.continually(ec)
72     }
73     val predicates = delegates.map { case (predicate, converter) =>
74       if (!contexts.hasNext) {
75         throw new IllegalArgumentException(s"Invalid evaluation context doesn't match this converter: $ec")
76       }
77       val context = contexts.next()
78       PredicateContext(predicate.withContext(context), converter, context)
79     }
80 
81     val args = Array.ofDim[AnyRef](1)
82 
83     def matches(p: Predicate): Boolean = {
84       try { p.apply(args) } catch {
85         case NonFatal(e) => logger.warn(s"Error evaluating predicate $p: ", e); false
86       }
87     }
88 
89     def eval(element: T): CloseableIterator[SimpleFeature] = {
90       val start = System.nanoTime()
91       args(0) = element
92       val converted = predicates.collectFirst { case p if matches(p.predicate) =>
93         routingTimer.record(System.nanoTime() - start, TimeUnit.NANOSECONDS) // note: invoke before executing the conversion
94         p.context.line = ec.line // update the line in the sub context
95         p.converter.convert(CloseableIterator.single(element), p.context)
96       }
97       converted.getOrElse {
98         routingTimer.record(System.nanoTime() - start, TimeUnit.NANOSECONDS)
99         ec.stats.failure()
100         CloseableIterator.empty
101       }
102     }
103 
104     new ErrorHandlingIterator(parse(is, ec), errorMode, ec, parseTimer).flatMap(eval)
105   }
106 
107   override def close(): Unit = CloseWithLogging(delegates.map(_._2))
108 }
