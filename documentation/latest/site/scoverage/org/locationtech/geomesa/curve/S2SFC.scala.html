23   override def index(x: Double, y: Double, lenient: Boolean): Long = {
24     if (lenient) {
25       val bx = if (x < LonMin) { LonMin } else if (x > LonMax) { LonMax } else { x }
26       val by = if (y < LatMin) { LatMin } else if (y > LatMax) { LatMax } else { y }
27       S2CellId.fromLatLng(S2LatLng.fromDegrees(by, bx)).id()
28     } else {
29       require(x >= LonMin && x <= LonMax && y >= LatMin && y <= LatMax,
30         s"Value(s) out of bounds ([$LonMin,$LonMax], [$LatMin,$LatMax]): $x, $y")
31       S2CellId.fromLatLng(S2LatLng.fromDegrees(y, x)).id()
32     }
33   }
34 
35   override def ranges(
36       xy: Seq[(Double, Double, Double, Double)],
37       precision: Int,
38       maxRanges: Option[Int]): Seq[IndexRange] = {
39 
40     xy.flatMap { case (xmin, ymin, xmax, ymax) =>
41       val lo = S2LatLng.fromDegrees(ymin, xmin)
42       val hi = S2LatLng.fromDegrees(ymax, xmax)
43       val rect = new S2LatLngRect(lo, hi)
44 
45       val cover = new S2RegionCoverer()
46       cover.setMinLevel(minLevel)
47       cover.setMaxLevel(maxLevel)
48       cover.setLevelMod(levelMod)
49       cover.setMaxCells(maxCells)
50 
51       val s2CellUnion = cover.getCovering(rect)
52 
53       val builder = Seq.newBuilder[IndexRange]
54       builder.sizeHint(s2CellUnion.cellIds().size())
55       s2CellUnion.cellIds().asScala.foreach(c => builder += IndexRange(c.rangeMin().id(), c.rangeMax().id(), contained = true))
56       builder.result()
57     }
58   }
59 
60   override def invert(i: Long): (Double, Double) = {
61     val latLon = new S2CellId(i).toLatLng
62     (latLon.lngDegrees(), latLon.latDegrees())
63   }
64 }
