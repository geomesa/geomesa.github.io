33 class FsIngestCommand extends IngestCommand[FileSystemDataStore] with FsDistributedCommand {
34 
35   override val params = new FsIngestParams
36 
37   override protected def setBackendSpecificOptions(sft: SimpleFeatureType): Unit =
38     FsCreateSchemaCommand.setOptions(sft, params)
39 
40   override protected def startIngest(
41       mode: RunMode,
42       ds: FileSystemDataStore,
43       sft: SimpleFeatureType,
44       converter: Config,
45       inputs: Inputs): Awaitable = {
46     mode match {
47       case RunModes.Local =>
48         super.startIngest(mode, ds, sft, converter, inputs)
49 
50       case RunModes.Distributed =>
51         Command.user.info(s"Running ingestion in distributed ${if (params.combineInputs) "combine " else "" }mode")
52         val reducers = Option(params.reducers).filter(_ > 0).getOrElse {
53           throw new ParameterException("Please specify --num-reducers for distributed ingest")
54         }
55         val storage = ds.storage(sft.getTypeName)
56         val tmpPath = Option(params.tempPath).map(d => storage.context.fs.makeQualified(new Path(d)))
57         val targetFileSize = storage.metadata.get(Metadata.TargetFileSize).map(_.toLong)
58 
59         tmpPath.foreach { tp =>
60           if (storage.context.fs.exists(tp)) {
61             Command.user.info(s"Deleting temp path $tp")
62             storage.context.fs.delete(tp, true)
63           }
64         }
65 
66         storage.metadata.encoding match {
67           case OrcFileSystemStorage.Encoding =>
68             new OrcConverterJob(
69               connection, sft, converter, inputs.paths, libjarsFiles, libjarsPaths, reducers,
70               storage.context.root, tmpPath, targetFileSize) {
71               override def configureJob(job: Job): Unit = {
72                 super.configureJob(job)
73                 if (params.combineInputs) {
74                   job.setInputFormatClass(classOf[ConverterCombineInputFormat])
75                   Option(params.maxSplitSize).foreach(s => FileInputFormat.setMaxInputSplitSize(job, s.toLong))
76                 }
77               }
78             }
79 
80           case ParquetFileSystemStorage.Encoding =>
81             new ParquetConverterJob(
82               connection, sft, converter, inputs.paths, libjarsFiles, libjarsPaths, reducers,
83               storage.context.root, tmpPath, targetFileSize) {
84               override def configureJob(job: Job): Unit = {
85                 super.configureJob(job)
86                 if (params.combineInputs) {
87                   job.setInputFormatClass(classOf[ConverterCombineInputFormat])
88                   Option(params.maxSplitSize).foreach(s => FileInputFormat.setMaxInputSplitSize(job, s.toLong))
89                 }
90               }
91             }
92 
93           case _ =>
94             throw new ParameterException(s"Ingestion is not supported for encoding '${params.encoding}'")
95         }
96 
97       case _ =>
98         throw new UnsupportedOperationException(s"Missing implementation for mode $mode")
99     }
100   }
101 }
