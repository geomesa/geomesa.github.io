31   private val expiry = PathCache.CacheDurationProperty.toDuration.get.toMillis

39   override def getPartition(name: String): Option[PartitionMetadata] = {
40     val path = new Path(context.root, name)
41     if (!PathCache.exists(context.fs, path)) { None } else {
42       val files = if (leafStorage) { Seq(StorageFile(name, 0L)) } else {
43         PathCache.list(context.fs, path).map(fs => StorageFile(fs.getPath.getName, 0L)).toList
44       }
45       Some(PartitionMetadata(name, files, None, -1L))
46     }
47   }
48 
49   override def getPartitions(prefix: Option[String]): Seq[PartitionMetadata] = {
50     buildPartitionList(prefix, dirty.compareAndSet(true, false)).map { name =>
51       val files = if (leafStorage) { Seq(StorageFile(name, 0L)) } else {
52         PathCache.list(context.fs, new Path(context.root, name)).map(fs => StorageFile(fs.getPath.getName, 0L)).toList
53       }
54       PartitionMetadata(name, files, None, -1L)
55     }
56   }

74   private def buildPartitionList(prefix: Option[String], invalidate: Boolean): List[String] = {
75     if (invalidate) {
76       PathCache.invalidate(context.fs, context.root)
77     }
78     val top = PathCache.list(context.fs, context.root)
79     top.flatMap(f => buildPartitionList(f.getPath, "", prefix, 1, invalidate)).toList
80   }
81 
82   private def buildPartitionList(
83       path: Path,
84       prefix: String,
85       filter: Option[String],
86       curDepth: Int,
87       invalidate: Boolean): List[String] = {
88     if (invalidate) {
89       PathCache.invalidate(context.fs, path)
90     }
91     if (curDepth > scheme.depth || !PathCache.status(context.fs, path).isDirectory) {
92       val file = s"$prefix${path.getName}"
93       if (filter.forall(file.startsWith)) { List(file) } else { List.empty }
94     } else {
95       val next = s"$prefix${path.getName}"
96       val continue = filter.forall { f =>
97         if (next.length >= f.length) { next.startsWith(f) } else { next == f.substring(0, next.length) }
98       }
99       if (continue) {
100         PathCache.list(context.fs, path).toList.flatMap { f =>
101           buildPartitionList(f.getPath, s"$next/", filter, curDepth + 1, invalidate)
102         }
103       } else {
104         List.empty
105       }
106     }
107   }
108 }
