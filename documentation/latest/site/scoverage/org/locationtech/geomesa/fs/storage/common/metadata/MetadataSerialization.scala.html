29   /**
30     * Serialize the metadata to the output stream as JSON
31     *
32     * @param out output stream
33     * @param metadata metadata
34     */
35   def serialize(out: OutputStream, metadata: Metadata): Unit = {
36     val sftConfig = SimpleFeatureTypes.toConfig(metadata.sft, includePrefix = false, includeUserData = true)
37     val schemeConfig = PartitionSchemeConfig(metadata.scheme.name, metadata.scheme.options)
38     val persistence = StoragePersistence(sftConfig, schemeConfig, metadata.config)
39 
40     val data = profile("Serialized storage configuration") {
41       ConfigWriter[StoragePersistence].to(persistence).render(RenderOptions)
42     }
43     profile("Wrote storage configuration") {
44       out.write(data.getBytes(StandardCharsets.UTF_8))
45     }
46   }

48   /**
49     * Deserialize metadata from an input stream
50     *
51     * @param in input stream
52     * @return
53     */
54   def deserialize(in: InputStream): Metadata = {
55     val persistence = profile("Parsed storage configuration") {
56       val config = ConfigFactory.parseReader(new InputStreamReader(in, StandardCharsets.UTF_8), ParseOptions)
57       try { ConfigSource.fromConfig(config).loadOrThrow[StoragePersistence] } catch {
58         case NonFatal(e) =>
59           def v1: Try[StoragePersistence] =
60             Try(ConfigSource.fromConfig(config).loadOrThrow[StoragePersistenceV1]).map { p =>
61               val leaf = p.partitionScheme.options.get("leaf-storage").forall(_.equalsIgnoreCase("true"))
62               val config = Map(Metadata.Encoding -> p.encoding, Metadata.LeafStorage -> s"$leaf")
63               StoragePersistence(p.featureType, p.partitionScheme, config)
64             }
65           def v2: Try[StoragePersistence] =
66             Try(ConfigSource.fromConfig(config).loadOrThrow[StoragePersistenceV2]).map { p =>
67               val leaf = java.lang.Boolean.toString(p.leafStorage)
68               val config = Map(Metadata.Encoding -> p.encoding, Metadata.LeafStorage -> leaf)
69               StoragePersistence(p.featureType, p.partitionScheme, config)
70             }
71           v2.orElse(v1).getOrElse(throw e)
72       }
73     }
74     val sft = profile("Parsed simple feature type") {
75       SimpleFeatureTypes.createType(persistence.featureType, path = None)
76     }
77     val scheme = NamedOptions(persistence.partitionScheme.scheme, persistence.partitionScheme.options)
78     Metadata(sft, scheme, persistence.config)
79   }
