27 /**
28   * An Input format that creates splits based on FSDS Partitions. This is used for compaction, when we want a single
29   * split per partition. Otherwise, use OrcSimpleFeatureInputFormat/ParquetSimpleFeatureInputFormat as those are
30   * more efficient
31   */
32 class PartitionInputFormat extends InputFormat[Void, SimpleFeature] {
33 
34   override def getSplits(context: JobContext): java.util.List[InputSplit] = {
35     val conf = context.getConfiguration
36 
37     val root = StorageConfiguration.getRootPath(conf)
38     val fsc = FileSystemContext(root, conf)
39     val fileSize = StorageConfiguration.getTargetFileSize(conf)
40 
41     val metadata = StorageMetadataFactory.load(fsc).getOrElse {
42       throw new IllegalArgumentException(s"No storage defined under path '$root'")
43     }
44     WithClose(metadata) { meta =>
45       WithClose(FileSystemStorageFactory(fsc, meta)) { storage =>
46         val sizeable = Option(storage).collect { case s: SizeableFileSystemStorage => s }
47         val sizeCheck = sizeable.flatMap(s => s.targetSize(fileSize).map(t => (p: Path) => s.fileIsSized(p, t)))
48         val splits = StorageConfiguration.getPartitions(conf).map { partition =>
49           var size = 0L
50           val files = storage.getFilePaths(partition).filter { f =>
51             if (sizeCheck.exists(_.apply(f.path))) { false } else {
52               size += PathCache.status(fsc.fs, f.path).getLen
53               true
54             }
55           }
56           new PartitionInputSplit(partition, files.map(_.file), size)
57         }
58         java.util.Arrays.asList(splits: _*)
59       }
60     }
61   }
62 
63   override def createRecordReader(split: InputSplit, context: TaskAttemptContext): RecordReader[Void, SimpleFeature] = {
64     val psplit = split.asInstanceOf[PartitionInputSplit]
65     new PartitionRecordReader(psplit.getName, psplit.getFiles)
66   }
67 }

71   /**
72     * InputSplit corresponding to a single FileSystemDataStore PartitionScheme partition
73     */
74   class PartitionInputSplit extends InputSplit with Writable {
75 
76     private var name: String = _
77     private var files: Seq[StorageFile] = _
78     private var length: java.lang.Long = _
79 
80     def this(name: String, files: Seq[StorageFile], length: Long) = {
81       this()
82       this.name = name
83       this.files = files
84       this.length = length
85     }
86 
87     /**
88       * @return the name of this partition
89       */
90     def getName: String = name
91 
92     def getFiles: Seq[StorageFile] = files
93 
94     override def getLength: Long = length
95 
96     // TODO attempt to optimize the locations where this should run in the case of HDFS
97     // With S3 this won't really matter
98     override def getLocations: Array[String] = Array.empty[String]
99 
100     override def write(out: DataOutput): Unit = {
101       out.writeUTF(name)
102       out.writeLong(length)
103       out.writeInt(files.length)
104       files.foreach { case StorageFile(file, ts, action, _ , _) =>
105         out.writeUTF(file)
106         out.writeLong(ts)
107         out.writeUTF(action.toString)
108       }
109     }
110 
111     override def readFields(in: DataInput): Unit = {
112       this.name = in.readUTF()
113       this.length = in.readLong()
114       this.files = Seq.fill(in.readInt) {
115         StorageFile(in.readUTF(), in.readLong, StorageFileAction.withName(in.readUTF()))
116       }
117     }
118   }
119 
120   class PartitionRecordReader(partition: String, files: Seq[StorageFile]) extends RecordReader[Void, SimpleFeature] {
121 
122     private var storage: FileSystemStorage = _
123     private var reader: CloseableFeatureIterator = _
124 
125     private var curValue: SimpleFeature = _
126 
127     override def initialize(split: InputSplit, context: TaskAttemptContext): Unit = {
128       val conf = context.getConfiguration
129       val root = StorageConfiguration.getRootPath(conf)
130       val fsc = FileSystemContext(root, conf)
131       val metadata = StorageMetadataFactory.load(fsc).getOrElse {
132         throw new IllegalArgumentException(s"No storage defined under path '$root'")
133       }
134       // use a cached metadata impl instead of reloading
135       val data = PartitionMetadata(partition, files, None, 0L)
136       val cached = new CachedMetadata(metadata.sft, metadata.encoding, metadata.scheme, metadata.leafStorage, data)
137       storage = FileSystemStorageFactory(fsc, cached)
138       reader = storage.getReader(new Query("", Filter.INCLUDE), Option(partition))
139       metadata.close()
140     }
141 
142     // TODO look at how the ParquetInputFormat provides progress and utilize something similar
143     override def getProgress: Float = 0.0f
144 
145     override def nextKeyValue(): Boolean = {
146       if (reader.hasNext) {
147         curValue = reader.next()
148         true
149       } else {
150         curValue = null
151         false
152       }
153     }
154 
155     override def getCurrentKey: Void = null
156     override def getCurrentValue: SimpleFeature = curValue
157 
158     override def close(): Unit = CloseWithLogging(reader, storage)
159   }
160 
161   class CachedMetadata(
162       val sft: SimpleFeatureType,
163       val encoding: String,
164       val scheme: PartitionScheme,
165       val leafStorage: Boolean,
166       partition: PartitionMetadata
167   ) extends StorageMetadata {
168     override def getPartition(name: String): Option[PartitionMetadata] =
169       if (partition.name == name) { Some(partition) } else { None }
170     override def getPartitions(prefix: Option[String]): Seq[PartitionMetadata] =
171       if (prefix.forall(partition.name.startsWith)) { Seq(partition) } else { Seq.empty }
172     override def addPartition(partition: PartitionMetadata): Unit = throw new NotImplementedError()
173     override def removePartition(partition: PartitionMetadata): Unit = throw new NotImplementedError()
174     override def setPartitions(partitions: Seq[PartitionMetadata]): Unit = throw new NotImplementedError()
175     override def compact(partition: Option[String], fileSize: Option[Long], threads: Int): Unit =
176       throw new NotImplementedError()
177     override def invalidate(): Unit = {}
178     override def close(): Unit = {}
179   }
