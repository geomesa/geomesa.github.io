21 abstract class SpatialScheme(bits: Int, geom: String) extends PartitionScheme {
22 
23   require(bits % 2 == 0, "Resolution must be an even number")
24 
25   protected val format = s"%0${digits(bits)}d"
26 
27   protected def digits(bits: Int): Int
28 
29   protected def generateRanges(xy: Seq[(Double, Double, Double, Double)]): Seq[IndexRange]
30 
31   override val depth: Int = 1
32 
33   override def getSimplifiedFilters(filter: Filter, partition: Option[String]): Option[Seq[SimplifiedFilter]] = {
34     getIntersectingPartitions(filter, partition).map { partitions =>
35       // note: we don't simplify the filter as usually we wouldn't be able to remove much
36       Seq(SimplifiedFilter(filter, partitions, partial = false))
37     }
38   }
39 
40   override def getIntersectingPartitions(filter: Filter): Option[Seq[String]] =
41     getIntersectingPartitions(filter, None)
42 
43   private def getIntersectingPartitions(filter: Filter, partition: Option[String]): Option[Seq[String]] = {
44     val geometries = FilterHelper.extractGeometries(filter, geom, intersect = true)
45     if (geometries.disjoint) {
46       Some(Seq.empty)
47     } else if (geometries.values.isEmpty) {
48       None
49     } else {
50       val partitions = partition.map(Seq(_)).getOrElse {
51         // there should be few enough partitions that we can enumerate them here and not exactly match the filter...
52         val ranges = generateRanges(geometries.values.map(GeometryUtils.bounds))
53         ranges.flatMap(r => r.lower to r.upper).distinct.map(_.formatted(format))
54       }
55       Some(partitions)
56     }
57   }
58 }

73     override def load(sft: SimpleFeatureType, config: NamedOptions): Option[PartitionScheme] = {
74       import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
75 
76       val matcher = namePattern.matcher(config.name)
77       if (!matcher.matches()) { None } else {
78         val geom = config.options.getOrElse(Config.GeomAttribute, sft.getGeomField)
79         val geomIndex = sft.indexOf(geom)
80         if (geomIndex == -1) {
81           throw new IllegalArgumentException(s"$name scheme requires valid geometry field '${Config.GeomAttribute}'")
82         }
83         val res =
84           Option(matcher.group(2))
85               .filterNot(_.isEmpty)
86               .orElse(config.options.get(resolution))
87               .map(Integer.parseInt)
88               .getOrElse(throw new IllegalArgumentException(s"$name scheme requires bit resolution '$resolution'"))
89         Some(buildPartitionScheme(res, geom, geomIndex))
90       }
91     }
92

94   }
