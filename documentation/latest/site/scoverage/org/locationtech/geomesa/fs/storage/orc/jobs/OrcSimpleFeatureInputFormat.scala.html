84   /**
85     * Abstract base class for reading Orc simple features without a key.
86     *
87     * This class is based on OrcInputFormat, but sets UTC reader options (which aren't exposed in OrcInputFormat)
88     * and doesn't use kryo serialization for configuration (which conflicts with spark kryo versions)
89     *
90     * @tparam T key type
91     */
92   abstract class OrcSimpleFeatureInputFormatBase[T] extends FileInputFormat[T, SimpleFeature] {
93 
94     override def createRecordReader(
95         inputSplit: InputSplit,
96         context: TaskAttemptContext): RecordReader[T, SimpleFeature] = {
97       val split = inputSplit.asInstanceOf[FileSplit]
98       val conf = context.getConfiguration
99 
100       val reader = {
101         val maxLength = OrcConf.MAX_FILE_LENGTH.getLong(conf)
102         val opts = OrcFile.readerOptions(conf).maxLength(maxLength).useUTCTimestamp(true)
103         OrcFile.createReader(split.getPath, opts)
104       }
105 
106       val options =
107         reader.options()
108             .range(split.getStart, split.getLength)
109             .useZeroCopy(OrcConf.USE_ZEROCOPY.getBoolean(conf))
110             .skipCorruptRecords(OrcConf.SKIP_CORRUPT_DATA.getBoolean(conf))
111             .tolerateMissingSchema(OrcConf.TOLERATE_MISSING_SCHEMA.getBoolean(conf))
112 
113       val sft = StorageConfiguration.getSft(conf)
114       val filter = StorageConfiguration.getFilter(conf, sft)
115       val transform = StorageConfiguration.getTransforms(conf)
116 
117       val OrcReadOptions(columns, pushDown) = OrcFileSystemReader.readOptions(sft, filter, transform)
118       columns.foreach(cols => options.include(OrcFileSystemReader.include(sft, cols)))
119       pushDown.foreach { case (sargs, names) => options.searchArgument(sargs, names) }
120 
121       val delegate = new OrcMapreduceRecordReader[OrcStruct](reader, options)
122 
123       createRecordReader(delegate, split, conf, sft, filter, transform, columns)
124     }
125 
126     override protected def listStatus(job: JobContext): util.List[FileStatus] = {
127       val complete = super.listStatus(job)
128       val result = new java.util.ArrayList[FileStatus](complete.size)
129       val iter = complete.iterator
130       while (iter.hasNext) {
131         val stat = iter.next
132         if (stat.getLen != 0) {
133           result.add(stat)
134         }
135       }
136       result
137     }
138 
139     protected def createRecordReader(
140         delegate: RecordReader[NullWritable, OrcStruct],
141         split: FileSplit,
142         conf: Configuration,
143         sft: SimpleFeatureType,
144         filter: Option[Filter],
145         transform: Option[(String, SimpleFeatureType)],
146         columns: Option[Set[Int]]): RecordReader[T, SimpleFeature]
147   }
148 
149   /**
150     * Abstract base class for reading orc simple feature records, without a key
151     *
152     * @param delegate primitive orc record reader
153     * @param sft simple feature type
154     * @param filter cql filter
155     * @param transform relational transform
156     * @param columns read columns
157     * @tparam T key type
158     */
159   abstract class OrcSimpleFeatureRecordReaderBase[T](
160       delegate: RecordReader[NullWritable, OrcStruct],
161       sft: SimpleFeatureType,
162       filter: Option[Filter],
163       transform: Option[(String, SimpleFeatureType)],
164       columns: Option[Set[Int]]
165     ) extends RecordReader[T, SimpleFeature] {
166 
167     private val feature = new ScalaSimpleFeature(sft, "")
168 
169     private val setAttributes = OrcInputFormatReader(sft, columns)
170 
171     private val current: SimpleFeature = transform match {
172       case Some((tdefs, tsft)) => TransformSimpleFeature(sft, tsft, tdefs).setFeature(feature)
173       case None => feature
174     }
175 
176     override def initialize(split: InputSplit, context: TaskAttemptContext): Unit =
177       delegate.initialize(split, context)
178 
179     override def nextKeyValue(): Boolean = {
180       while (delegate.nextKeyValue()) {
181         setAttributes(delegate.getCurrentValue, feature)
182         if (filter.forall(_.evaluate(feature))) {
183           return true
184         }
185       }
186       false
187     }
188 
189     override def getCurrentValue: SimpleFeature = current
190 
191     override def getProgress: Float = delegate.getProgress
192 
193     override def close(): Unit = delegate.close()
194   }
