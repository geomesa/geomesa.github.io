17 /**
18   * Caches file statuses to avoid repeated file system operations. Status expires after a
19   * configurable period, by default 10 minutes.
20   */
21 object PathCache {
22 
23   val CacheDurationProperty: SystemProperty = SystemProperty("geomesa.fs.file.cache.duration", "15 minutes")
24 
25   private val duration = CacheDurationProperty.toDuration.get.toMillis
26 
27   // cache for checking existence of files
28   private val pathCache =
29     Caffeine.newBuilder().expireAfterWrite(duration, TimeUnit.MILLISECONDS).build(
30       new CacheLoader[(FileSystem, Path), java.lang.Boolean]() {
31         override def load(key: (FileSystem, Path)): java.lang.Boolean = key._1.exists(key._2)
32       }
33     )
34 
35   // cache for individual file status
36   private val statusCache =
37     Caffeine.newBuilder().expireAfterWrite(duration, TimeUnit.MILLISECONDS).build(
38       new CacheLoader[(FileSystem, Path), FileStatus]() {
39         override def load(key: (FileSystem, Path)): FileStatus = key._1.getFileStatus(key._2)
40       }
41     )
42 
43   // cache for checking directory contents
44   private val listCache =
45     Caffeine.newBuilder().expireAfterWrite(duration, TimeUnit.MILLISECONDS).build(
46       new CacheLoader[(FileSystem, Path), Stream[FileStatus]]() {
47         override def load(key: (FileSystem, Path)): Stream[FileStatus] =
48           RemoteIterator(key._1.listStatusIterator(key._2)).toStream
49       }
50     )
51 
52   /**
53     * * Register a path as existing
54     *
55     * @param fs file system
56     * @param path path
57     */
58   def register(fs: FileSystem, path: Path): Unit = {
59     pathCache.put((fs, path), java.lang.Boolean.TRUE)
60     val status = statusCache.refresh((fs, path))
61     val parent = path.getParent
62     if (parent != null) {
63       listCache.getIfPresent((fs, parent)) match {
64         case null => // no-op
65         case list => listCache.put((fs, parent), list :+ status.get())
66       }
67     }
68   }
69 
70   /**
71     * Check to see if a path exists
72     *
73     * @param fs file system
74     * @param path path
75     * @param reload reload the file status from the underlying file system before checking
76     * @return
77     */
78   def exists(fs: FileSystem, path: Path, reload: Boolean = false): Boolean = {
79     if (reload) {
80       invalidate(fs, path)
81     }
82     pathCache.get((fs, path)).booleanValue()
83   }
84 
85   /**
86     * Gets the file status for a path
87     *
88     * @param fs file system
89     * @param path path
90     * @return
91     */
92   def status(fs: FileSystem, path: Path, reload: Boolean = false): FileStatus = {
93     if (reload) {
94       invalidate(fs, path)
95     }
96     statusCache.get((fs, path))
97   }
98 
99   /**
100     * List the children of a path
101     *
102     * @param fs file system
103     * @param dir directory path
104     * @return
105     */
106   def list(fs: FileSystem, dir: Path, reload: Boolean = false): Iterator[FileStatus] = {
107     if (reload) {
108       invalidate(fs, dir)
109     }
110     listCache.get((fs, dir)).iterator
111   }
112 
113   /**
114     * Invalidate any cached values for the path - they will be re-loaded on next access
115     *
116     * @param fs file system
117     * @param path path
118     */
119   def invalidate(fs: FileSystem, path: Path): Unit = Seq(pathCache, statusCache, listCache).foreach(_.invalidate((fs, path)))
120 
121   object RemoteIterator {
122     def apply[T](iter: RemoteIterator[T]): Iterator[T] = new Iterator[T] {
123       override def hasNext: Boolean = iter.hasNext
124       override def next(): T = iter.next
125     }
126   }
127 }
