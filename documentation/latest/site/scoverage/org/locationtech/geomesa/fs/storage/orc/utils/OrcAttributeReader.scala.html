34   /**
35     * Create a reader for an ORC batch
36     *
37     * @param sft simple feature type
38     * @param batch row batch
39     * @param columns columns to read, corresponding to simple feature attributes
40     * @param fid read feature id or not
41     * @return
42     */
43   def apply(sft: SimpleFeatureType,
44             batch: VectorizedRowBatch,
45             columns: Option[Set[Int]] = None,
46             fid: Boolean = true): OrcAttributeReader = {
47     val builder = Seq.newBuilder[OrcAttributeReader]
48     builder.sizeHint(columns.map(_.size).getOrElse(sft.getAttributeCount) + (if (fid) { 1 } else { 0 }))
49 
50     var i = 0
51     var col = 0
52     while (i < sft.getAttributeCount) {
53       val descriptor = sft.getDescriptor(i)
54       val bindings = ObjectType.selectType(descriptor)
55       if (columns.forall(_.contains(i))) {
56         val reader = bindings.head match {
57           case ObjectType.GEOMETRY => createGeometryReader(bindings(1), batch.cols, col, i)
58           case ObjectType.DATE     => new DateReader(batch.cols(col).asInstanceOf[TimestampColumnVector], i)
59           case ObjectType.STRING   => new StringReader(batch.cols(col).asInstanceOf[BytesColumnVector], i)
60           case ObjectType.INT      => new IntReader(batch.cols(col).asInstanceOf[LongColumnVector], i)
61           case ObjectType.LONG     => new LongReader(batch.cols(col).asInstanceOf[LongColumnVector], i)
62           case ObjectType.FLOAT    => new FloatReader(batch.cols(col).asInstanceOf[DoubleColumnVector], i)
63           case ObjectType.DOUBLE   => new DoubleReader(batch.cols(col).asInstanceOf[DoubleColumnVector], i)
64           case ObjectType.BOOLEAN  => new BooleanReader(batch.cols(col).asInstanceOf[LongColumnVector], i)
65           case ObjectType.BYTES    => new BytesReader(batch.cols(col).asInstanceOf[BytesColumnVector], i)
66           case ObjectType.UUID     => new UuidReader(batch.cols(col).asInstanceOf[BytesColumnVector], i)
67           case ObjectType.LIST     => new ListReader(batch.cols(col).asInstanceOf[ListColumnVector], i, bindings(1))
68           case ObjectType.MAP      => new MapReader(batch.cols(col).asInstanceOf[MapColumnVector], i, bindings(1), bindings(2))
69           case _ => throw new IllegalArgumentException(s"Unexpected object type ${bindings.head}")
70         }
71         builder += reader
72       }
73       i += 1
74       col += OrcFileSystemStorage.fieldCount(descriptor)
75     }
76 
77     if (fid) {
78       builder += new FidReader(batch.cols(col).asInstanceOf[BytesColumnVector])
79     }
80 
81     new SequenceReader(builder.result)
82   }

85   private def createGeometryReader(binding: ObjectType, cols: Array[ColumnVector], col: Int, i: Int): OrcAttributeReader = {
86     implicit def toDoubleColumnVector(vec: ColumnVector): DoubleColumnVector = vec.asInstanceOf[DoubleColumnVector]
87     implicit def toListColumnVector(vec: ColumnVector): ListColumnVector = vec.asInstanceOf[ListColumnVector]

93       case ObjectType.POINT           => new PointReader(x, y, i)
94       case ObjectType.LINESTRING      => new LineStringReader(x, y, i)
95       case ObjectType.MULTIPOINT      => new MultiPointReader(x, y, i)
96       case ObjectType.POLYGON         => new PolygonReader(x, y, i)
97       case ObjectType.MULTILINESTRING => new MultiLineStringReader(x, y, i)
98       case ObjectType.MULTIPOLYGON    => new MultiPolygonReader(x, y, i)
99       case ObjectType.GEOMETRY        => new GeometryWkbReader(cols(col).asInstanceOf[BytesColumnVector], i)
100       case _ => throw new IllegalArgumentException(s"Unexpected object type $binding")
101     }
