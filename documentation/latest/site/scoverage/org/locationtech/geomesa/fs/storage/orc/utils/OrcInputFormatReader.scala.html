32   def apply(sft: SimpleFeatureType, columns: Option[Set[Int]] = None, fid: Boolean = true): OrcInputFormatReader = {
33     val builder = Seq.newBuilder[OrcInputFormatReader]
34     builder.sizeHint(columns.map(_.size).getOrElse(sft.getAttributeCount) + (if (fid) { 1 } else { 0 }))
35 
36     var i = 0
37     var col = 0
38     while (i < sft.getAttributeCount) {
39       val descriptor = sft.getDescriptor(i)
40       val bindings = ObjectType.selectType(descriptor)
41       if (columns.forall(_.contains(i))) {
42         val reader = bindings.head match {
43           case ObjectType.GEOMETRY => createGeometryReader(bindings(1), col, i)
44           case ObjectType.DATE     => new DateInputFormatReader(col, i)
45           case ObjectType.STRING   => new StringInputFormatReader(col, i)
46           case ObjectType.INT      => new IntInputFormatReader(col, i)
47           case ObjectType.LONG     => new LongInputFormatReader(col, i)
48           case ObjectType.FLOAT    => new FloatInputFormatReader(col, i)
49           case ObjectType.DOUBLE   => new DoubleInputFormatReader(col, i)
50           case ObjectType.BOOLEAN  => new BooleanInputFormatReader(col, i)
51           case ObjectType.BYTES    => new BytesInputFormatReader(col, i)
52           case ObjectType.UUID     => new UuidInputFormatReader(col, i)
53           case ObjectType.LIST     => new ListInputFormatReader(col, i, bindings(1))
54           case ObjectType.MAP      => new MapInputFormatReader(col, i, bindings(1), bindings(2))
55           case _ => throw new IllegalArgumentException(s"Unexpected object type ${bindings.head}")
56         }
57         builder += reader
58       }
59       i += 1
60       col += OrcFileSystemStorage.fieldCount(descriptor)
61     }
62 
63     if (fid) {
64       builder += new FidInputFormatReader(col)
65     }
66 
67     new SequenceInputFormatReader(builder.result)
68   }

70   private def createGeometryReader(binding: ObjectType, col: Int, i: Int): OrcInputFormatReader = {
71     binding match {
72       case ObjectType.POINT           => new PointInputFormatReader(col, col + 1, i)
73       case ObjectType.LINESTRING      => new LineStringInputFormatReader(col, col + 1, i)
74       case ObjectType.MULTIPOINT      => new MultiPointInputFormatReader(col, col + 1, i)
75       case ObjectType.POLYGON         => new PolygonInputFormatReader(col, col + 1, i)
76       case ObjectType.MULTILINESTRING => new MultiLineStringInputFormatReader(col, col + 1, i)
77       case ObjectType.MULTIPOLYGON    => new MultiPolygonInputFormatReader(col, col + 1, i)
78       case ObjectType.GEOMETRY        => new GeometryWkbInputFormatReader(col, i)
79       case _ => throw new IllegalArgumentException(s"Unexpected geometry type $binding")
80     }

218       extends OrcInputFormatReader with ConvertInputFormatDouble {
219     override def apply(input: OrcStruct, sf: SimpleFeature): Unit = {
220       val xxList = input.getFieldValue(xCol).asInstanceOf[OrcList[OrcList[_ <: WritableComparable[_]]]]
221       val yyList = input.getFieldValue(yCol).asInstanceOf[OrcList[OrcList[_ <: WritableComparable[_]]]]
222       if (xxList == null || yyList == null) {
223         sf.setAttribute(attribute, null)
224       } else {
225         val lines = Array.ofDim[LinearRing](xxList.size)
226         var j = 0
227         while (j < lines.length) {
228           val xList = xxList.get(j)
229           val yList = yyList.get(j)
230           val coordinates = Array.ofDim[Coordinate](xList.size())
231           var i = 0
232           while (i < coordinates.length) {
233             coordinates(i) = new Coordinate(convertUnboxed(xList.get(i)), convertUnboxed(yList.get(i)))
234             i += 1
235           }
236           lines(j) = gf.createLinearRing(coordinates)
237           j += 1
238         }
239         val polygon = if (lines.length == 1) { gf.createPolygon(lines.head) } else { gf.createPolygon(lines.head, lines.tail) }
240         sf.setAttribute(attribute, polygon)
241       }
242     }
