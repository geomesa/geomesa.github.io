24   /**
25     * Creates a push-down predicate for Orc files based on a CQL filter
26     *
27     * @param sft simple feature type
28     * @param description orc file description
29     * @param filter cql filter
30     * @return
31     */
32   def apply(sft: SimpleFeatureType,
33             description: TypeDescription,
34             filter: Filter): Option[(SearchArgument, Array[String])] = {
35     import org.apache.orc.TypeDescription.Category._
36 
37     val predicates = FilterHelper.propertyNames(filter, sft).distinct.flatMap { prop =>
38       val binding = sft.getDescriptor(prop).getType.getBinding
39       val preds = if (classOf[Geometry].isAssignableFrom(binding)) {
40         if (classOf[Point] == binding) {
41           FilterHelper.extractGeometries(filter, prop).values.map(addPoint(prop, _))
42         } else {
43           // orc doesn't support push-down predicates against complex fields
44           Seq.empty
45         }
46       } else {
47         // count the columns before the property
48         val offset = sft.getAttributeDescriptors.asScala.take(sft.indexOf(prop)).foldLeft(0) { (sum, d) =>
49           sum + SimpleFeatureTypeDescription.fieldCount(d)
50         }
51 
52         val category = description.getChildren.get(offset).getCategory
53         val typeAndConversion = category match {
54           case BOOLEAN   => Some(PredicateLeaf.Type.BOOLEAN, (v: Any) => v)
55           case INT       => Some(PredicateLeaf.Type.LONG, (v: Any) => v.asInstanceOf[java.lang.Integer].longValue)
56           case LONG      => Some(PredicateLeaf.Type.LONG, (v: Any) => v)
57           case FLOAT     => Some(PredicateLeaf.Type.FLOAT, (v: Any) => v.asInstanceOf[java.lang.Float].doubleValue)
58           case DOUBLE    => Some(PredicateLeaf.Type.FLOAT, (v: Any) => v)
59           case STRING    => Some(PredicateLeaf.Type.STRING, (v: Any) => v)
60           case TIMESTAMP => Some(PredicateLeaf.Type.TIMESTAMP, (v: Any) => new Timestamp(v.asInstanceOf[java.util.Date].getTime))
61           case BINARY | LIST | MAP => None // orc doesn't support push-down predicates against complex fields
62           case BYTE | CHAR | SHORT | DATE | DECIMAL | VARCHAR | STRUCT =>
63             throw new IllegalArgumentException(s"SimpleFeature TypeDefinition should not have type '$category'")
64         }
65         typeAndConversion.toSeq.flatMap { case (typ, conversion) =>
66           FilterHelper.extractAttributeBounds(filter, prop, binding).values.flatMap(add(prop, _, typ, conversion))
67         }
68       }
69 
70       if (preds.isEmpty) {
71         Seq.empty
72       } else {
73         Seq(preds)
74       }
75     }
76 
77     if (predicates.isEmpty) { None } else {
78       val arg = SearchArgumentFactory.newBuilder
79       if (predicates.length > 1) {
80         arg.startAnd()
81       }
82       predicates.foreach { preds =>
83         if (preds.length == 1) {
84           preds.head.apply(arg)
85         } else {
86           arg.startOr()
87           preds.foreach(_.apply(arg))
88           arg.end()
89         }
90       }
91       if (predicates.length > 1) {
92         arg.end()
93       }
94       // note: column name array does not matter at all
95       Some((arg.build, Array.empty))
96     }
97   }

99   private def add(
100       prop: String,
101       bounds: Bounds[_],
102       typ: PredicateLeaf.Type,
103       convert: Any => Any): Option[SearchArgument.Builder => Unit] = {
104     if (bounds.isRange) {
105       if (bounds.isBoundedBothSides) {
106         // between seems to be endpoint inclusive, so should not have any false negatives regardless of bounds inclusiveness
107         Some(arg => arg.between(prop, typ, convert(bounds.lower.value.get), convert(bounds.upper.value.get)))
108       } else if (bounds.isBounded) {
109         if (bounds.upper.value.isDefined) {
110           if (bounds.upper.inclusive) {
111             Some(arg => arg.lessThanEquals(prop, typ, convert(bounds.upper.value.get)))
112           } else {
113             Some(arg => arg.lessThan(prop, typ, convert(bounds.upper.value.get)))
114           }
115         } else if (bounds.lower.inclusive) {
116           Some(arg => arg.startNot().lessThan(prop, typ, convert(bounds.lower.value.get)).end())
117         } else {
118           Some(arg => arg.startNot().lessThanEquals(prop, typ, convert(bounds.lower.value.get)).end())
119         }
120       } else {
121         None
122       }
123     } else {
124       Some(arg => arg.equals(prop, typ, convert(bounds.lower.value.get)))
125     }
126   }
127 
128   private def addPoint(prop: String, bounds: Geometry): SearchArgument.Builder => Unit = {
129     val x = SimpleFeatureTypeDescription.geometryXField(prop)
130     val y = SimpleFeatureTypeDescription.geometryYField(prop)
131     val envelope = bounds.getEnvelopeInternal
132 
133     arg => {
134       arg.startAnd()
135       if (envelope.getMinX == envelope.getMaxX) {
136         arg.equals(x, PredicateLeaf.Type.FLOAT, envelope.getMinX)
137       } else {
138         arg.between(x, PredicateLeaf.Type.FLOAT, envelope.getMinX, envelope.getMaxX)
139       }
140       if (envelope.getMinY == envelope.getMaxY) {
141         arg.equals(y, PredicateLeaf.Type.FLOAT, envelope.getMinY)
142       } else {
143         arg.between(y, PredicateLeaf.Type.FLOAT, envelope.getMinY, envelope.getMaxY)
144       }
145       arg.end()
146     }
147   }
