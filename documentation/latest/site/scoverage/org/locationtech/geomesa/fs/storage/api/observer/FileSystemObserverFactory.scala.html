18 /**
19  * Factory for observing file writes
20  */
21 trait FileSystemObserverFactory extends Closeable {
22 
23   /**
24    * Called once after instantiating the factory
25    *
26    * @param conf hadoop configuration
27    * @param root root path
28    * @param sft simple feature type
29    */
30   def init(conf: Configuration, root: Path, sft: SimpleFeatureType): Unit
31 
32   /**
33    * Create an observer for the given path
34    *
35    * @param path file path being written
36    * @return
37    */
38   def apply(path: Path): FileSystemObserver
39 }
40 
41 object FileSystemObserverFactory {
42 
43   object NoOpObserver extends FileSystemObserver {
44     override def apply(feature: SimpleFeature): Unit = {}
45     override def flush(): Unit = {}
46     override def close(): Unit = {}
47   }
48 
49   object CompositeObserver {
50     def apply(observers: Seq[FileSystemObserver]): FileSystemObserver = {
51       val seq = observers.flatMap {
52         case NoOpObserver => Seq.empty
53         case o: CompositeObserver => o.observers
54         case o => Seq(o)
55       }
56       if (seq.isEmpty) {
57         NoOpObserver
58       } else if (seq.lengthCompare(1) == 0) {
59         seq.head
60       } else {
61         new CompositeObserver(seq)
62       }
63     }
64   }
65 
66   /**
67    * Composite observer
68    *
69    * @param observers observers
70    */
71   class CompositeObserver(val observers: Seq[FileSystemObserver]) extends FileSystemObserver {
72 
73     override def apply(feature: SimpleFeature): Unit = observers.foreach(_.apply(feature))
74 
75     override def flush(): Unit = {
76       val errors = observers.flatMap(o => Try(o.flush()).failed.toOption)
77       if (errors.nonEmpty) {
78         throw collapseErrors(errors)
79       }
80     }
81 
82     override def close(): Unit = {
83       val errors = observers.flatMap(o => Try(o.close()).failed.toOption)
84       if (errors.nonEmpty) {
85         throw collapseErrors(errors)
86       }
87     }
88 
89     private def collapseErrors(errors: Seq[Throwable]): Throwable = {
90       val e = errors.head
91       errors.tail.foreach(e.addSuppressed)
92       e
93     }
94   }
