22 /**
23   * Manages the storages and associated simple feature types underneath a given path
24   *
25   * @param fs file system
26   * @param conf configuration
27   * @param root root path for the data store
28   */
29 class FileSystemStorageManager private (fs: FileSystem, conf: Configuration, root: Path, namespace: Option[String])
30     extends LazyLogging {
31 
32   import scala.collection.JavaConverters._
33 
34   private val cache = new ConcurrentHashMap[String, (Path, FileSystemStorage)]().asScala
35 
36   /**
37     * Gets the storage associated with the given simple feature type, if any
38     *
39     * @param typeName simple feature type name
40     * @return
41     */
42   def storage(typeName: String): Option[FileSystemStorage] = {
43     cache.get(typeName).map(_._2) // check cached values
44         .orElse(Some(defaultPath(typeName)).filter(PathCache.exists(fs, _)).flatMap(loadPath)) // check expected (default) path
45         .orElse(loadAll().find(_.metadata.sft.getTypeName == typeName)) // check other paths until we find it
46   }
47 
48   /**
49     * Gets all storages under the root path
50     *
51     * @return
52     */
53   def storages(): Seq[FileSystemStorage] = {
54     loadAll().foreach(_ => ()) // force loading of everything
55     cache.map { case (_, (_, storage)) => storage }.toSeq
56   }
57 
58   /**
59     * Caches a storage instance for future use. Avoids loading it a second time if referenced later.
60     *
61     * @param path path for the storage
62     * @param storage storage instance
63     */
64   def register(path: Path, storage: FileSystemStorage): Unit =
65     cache.put(storage.metadata.sft.getTypeName, (path, storage))
66 
67   /**
68     * Default path for a given simple feature type name. Generally the simple feature type will go under
69     * a folder with the type name, but this is not required
70     *
71     * @param typeName simple feature type name
72     * @return
73     */
74   def defaultPath(typeName: String): Path = new Path(root, typeName)
75 
76   /**
77     * Loads all storages under this root (if they aren't already loaded)
78     *
79     * @return
80     */
81   private def loadAll(): Iterator[FileSystemStorage] = {
82     if (!PathCache.exists(fs, root)) { Iterator.empty } else {
83       val dirs = PathCache.list(fs, root).filter(_.isDirectory).map(_.getPath)
84       dirs.filterNot(path => cache.exists { case (_, (p, _)) => p == path }).flatMap(loadPath)
85     }
86   }
87 
88   /**
89     * Attempt to load a storage under the given root path. Requires an appropriate storage implementation
90     * to be available on the classpath.
91     *
92     * @param path storage root path
93     * @return
94     */
95   private def loadPath(path: Path): Option[FileSystemStorage] = {
96     val start = System.nanoTime()
97     val context = FileSystemContext(fs, conf, path, namespace)
98     val storage = StorageMetadataFactory.load(context).map { meta =>
99       try {
100         val storage = FileSystemStorageFactory(context, meta)
101         register(path, storage)
102         storage
103       } catch {
104         case NonFatal(e) => CloseQuietly(meta).foreach(e.addSuppressed); throw e
105       }
106     }
107     logger.debug(
108       s"${ if (storage.isDefined) "Loaded" else "No" } storage at path '$path' in ${(System.nanoTime() - start) / 1000000L}ms")
109     storage
110   }
111 }
