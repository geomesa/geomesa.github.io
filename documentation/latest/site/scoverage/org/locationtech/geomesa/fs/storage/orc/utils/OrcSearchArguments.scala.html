25   /**
26     * Creates a push-down predicate for Orc files based on a CQL filter
27     *
28     * @param sft simple feature type
29     * @param description orc file description
30     * @param filter cql filter
31     * @return
32     */
33   def apply(sft: SimpleFeatureType,
34             description: TypeDescription,
35             filter: Filter): Option[(SearchArgument, Array[String])] = {
36     import org.apache.orc.TypeDescription.Category._
37 
38     val predicates = FilterHelper.propertyNames(filter, sft).distinct.flatMap { prop =>
39       val binding = sft.getDescriptor(prop).getType.getBinding
40       val preds = if (classOf[Geometry].isAssignableFrom(binding)) {
41         if (classOf[Point] == binding) {
42           FilterHelper.extractGeometries(filter, prop).values.map(addPoint(prop, _))
43         } else {
44           // orc doesn't support push-down predicates against complex fields
45           Seq.empty
46         }
47       } else {
48         // count the columns before the property
49         val offset = sft.getAttributeDescriptors.asScala.take(sft.indexOf(prop)).foldLeft(0) { (sum, d) =>
50           sum + OrcFileSystemStorage.fieldCount(d)
51         }
52 
53         val category = description.getChildren.get(offset).getCategory
54         val typeAndConversion = category match {
55           case BOOLEAN   => Some(PredicateLeaf.Type.BOOLEAN, (v: Any) => v)
56           case INT       => Some(PredicateLeaf.Type.LONG, (v: Any) => v.asInstanceOf[java.lang.Integer].longValue)
57           case LONG      => Some(PredicateLeaf.Type.LONG, (v: Any) => v)
58           case FLOAT     => Some(PredicateLeaf.Type.FLOAT, (v: Any) => v.asInstanceOf[java.lang.Float].doubleValue)
59           case DOUBLE    => Some(PredicateLeaf.Type.FLOAT, (v: Any) => v)
60           case STRING    => Some(PredicateLeaf.Type.STRING, (v: Any) => v)
61           case TIMESTAMP => Some(PredicateLeaf.Type.TIMESTAMP, (v: Any) => new Timestamp(v.asInstanceOf[java.util.Date].getTime))
62           case BINARY | LIST | MAP => None // orc doesn't support push-down predicates against complex fields
63           case BYTE | CHAR | SHORT | DATE | DECIMAL | VARCHAR | STRUCT =>
64             throw new IllegalArgumentException(s"SimpleFeature TypeDefinition should not have type '$category'")
65         }
66         typeAndConversion.toSeq.flatMap { case (typ, conversion) =>
67           FilterHelper.extractAttributeBounds(filter, prop, binding).values.flatMap(add(prop, _, typ, conversion))
68         }
69       }
70 
71       if (preds.isEmpty) {
72         Seq.empty
73       } else {
74         Seq(preds)
75       }
76     }
77 
78     if (predicates.isEmpty) { None } else {
79       val arg = SearchArgumentFactory.newBuilder
80       if (predicates.length > 1) {
81         arg.startAnd()
82       }
83       predicates.foreach { preds =>
84         if (preds.length == 1) {
85           preds.head.apply(arg)
86         } else {
87           arg.startOr()
88           preds.foreach(_.apply(arg))
89           arg.end()
90         }
91       }
92       if (predicates.length > 1) {
93         arg.end()
94       }
95       // note: column name array does not matter at all
96       Some((arg.build, Array.empty))
97     }
98   }

100   private def add(
101       prop: String,
102       bounds: Bounds[_],
103       typ: PredicateLeaf.Type,
104       convert: Any => Any): Option[SearchArgument.Builder => Unit] = {
105     if (bounds.isRange) {
106       if (bounds.isBoundedBothSides) {
107         // between seems to be endpoint inclusive, so should not have any false negatives regardless of bounds inclusiveness
108         Some(arg => arg.between(prop, typ, convert(bounds.lower.value.get), convert(bounds.upper.value.get)))
109       } else if (bounds.isBounded) {
110         if (bounds.upper.value.isDefined) {
111           if (bounds.upper.inclusive) {
112             Some(arg => arg.lessThanEquals(prop, typ, convert(bounds.upper.value.get)))
113           } else {
114             Some(arg => arg.lessThan(prop, typ, convert(bounds.upper.value.get)))
115           }
116         } else if (bounds.lower.inclusive) {
117           Some(arg => arg.startNot().lessThan(prop, typ, convert(bounds.lower.value.get)).end())
118         } else {
119           Some(arg => arg.startNot().lessThanEquals(prop, typ, convert(bounds.lower.value.get)).end())
120         }
121       } else {
122         None
123       }
124     } else {
125       Some(arg => arg.equals(prop, typ, convert(bounds.lower.value.get)))
126     }
127   }
128 
129   private def addPoint(prop: String, bounds: Geometry): SearchArgument.Builder => Unit = {
130     val x = OrcFileSystemStorage.geometryXField(prop)
131     val y = OrcFileSystemStorage.geometryYField(prop)
132     val envelope = bounds.getEnvelopeInternal
133 
134     arg => {
135       arg.startAnd()
136       if (envelope.getMinX == envelope.getMaxX) {
137         arg.equals(x, PredicateLeaf.Type.FLOAT, envelope.getMinX)
138       } else {
139         arg.between(x, PredicateLeaf.Type.FLOAT, envelope.getMinX, envelope.getMaxX)
140       }
141       if (envelope.getMinY == envelope.getMaxY) {
142         arg.equals(y, PredicateLeaf.Type.FLOAT, envelope.getMinY)
143       } else {
144         arg.between(y, PredicateLeaf.Type.FLOAT, envelope.getMinY, envelope.getMaxY)
145       }
146       arg.end()
147     }
148   }
