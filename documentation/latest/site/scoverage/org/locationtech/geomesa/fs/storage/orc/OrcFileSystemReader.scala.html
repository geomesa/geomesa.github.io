27 class OrcFileSystemReader(sft: SimpleFeatureType,
28                           config: Configuration,
29                           filter: Option[Filter],
30                           transform: Option[(String, SimpleFeatureType)]) extends FileSystemPathReader {
31 
32   private val (options, columns) = {
33     val options = new Reader.Options(config)
34     val readOptions = OrcFileSystemReader.readOptions(sft, filter, transform)
35     readOptions.columns.foreach(c => options.include(OrcFileSystemReader.include(sft, c)))
36     // note: push down will exclude whole record batches, but doesn't actually filter inside a batch
37     readOptions.pushDown.foreach { case (sarg, cols) => options.searchArgument(sarg, cols) }
38     (options, readOptions.columns)
39   }
40 
41   override def read(path: Path): CloseableIterator[SimpleFeature] = new PathReader(path)
42 
43   private class PathReader(file: Path) extends CloseableIterator[SimpleFeature] {
44     private val feature = new ScalaSimpleFeature(sft, "")
45     private val transformed = transform.map { case (tdefs, tsft) =>
46       val sf = TransformSimpleFeature(sft, tsft, tdefs)
47       sf.setFeature(feature)
48       sf
49     }
50     private val result = transformed.getOrElse(feature)
51 
52     private val reader = OrcFile.createReader(file, OrcFile.readerOptions(config).useUTCTimestamp(true))
53     private val rows = reader.rows(options)
54     private val batch = reader.getSchema.createRowBatch()
55     private val attributeReader = if (batch.cols.length > 0) { OrcAttributeReader(sft, batch, columns) } else { null }
56 
57     private var staged: Boolean = false
58     private var i = 0
59 
60     override def hasNext: Boolean = {
61       if (staged) { true } else {
62         stageNext()
63         staged
64       }
65     }
66 
67     /**
68       * Note: same feature is mutated and returned for each call to .next
69       *
70       * @return
71       */
72     override def next(): SimpleFeature = {
73       if (staged || hasNext) {
74         staged = false
75         result
76       } else {
77         Iterator.empty.next
78       }
79     }
80 
81     override def close(): Unit = rows.close()
82 
83     private def stageNext(): Unit = {
84       var loop = true
85       while (loop) {
86         if (i < batch.size) {
87           attributeReader.apply(feature, i)
88           i += 1
89           if (OrcFileSystemReader.this.filter.forall(_.evaluate(feature))) {
90             staged = true
91             loop = false
92           }
93         } else {
94           i = 0
95           loop = rows.nextBatch(batch)
96         }
97       }
98     }
99   }
100 }

106   /**
107     * Create low-level reading options used by ORC
108     *
109     * @param sft simple feature type
110     * @param filter ecql filter
111     * @param transform transform
112     * @return
113     */
114   def readOptions(
115       sft: SimpleFeatureType,
116       filter: Option[Filter],
117       transform: Option[(String, SimpleFeatureType)]): OrcReadOptions = {
118     val columns = transform.map { case (tdefs, _) =>
119       val fromFilter = filter.map(FilterHelper.propertyNames(_, sft)).getOrElse(Seq.empty)
120       val fromTransform = Transforms(sft, tdefs).flatMap(_.properties)
121       (fromFilter ++ fromTransform).toSet[String].map(sft.indexOf)
122     }
123     // push down will exclude whole record batches, but doesn't actually filter inside a batch
124     val pushDown = filter.flatMap { f =>
125       OrcSearchArguments(sft, OrcFileSystemStorage.createTypeDescription(sft), f)
126     }
127     OrcReadOptions(columns, pushDown)
128   }
129 
130   /**
131     * Construct the array used to include columns corresponding to the simple feature transform.
132     *
133     * Included columns don't just align with the top-level fields - complex fields need to have additional
134     * entries in the array for their children.
135     *
136     * @param sft simple feature type
137     * @param columns attributes to read
138     * @param fid read fid
139     * @return
140     */
141   def include(sft: SimpleFeatureType, columns: Set[Int], fid: Boolean = true): Array[Boolean] = {
142     val buffer = ArrayBuffer[Boolean](true) // outer struct
143 
144     var i = 0
145     while (i < sft.getAttributeCount) {
146       val bindings = ObjectType.selectType(sft.getDescriptor(i))
147       val count = bindings.head match {
148         case ObjectType.GEOMETRY =>
149           bindings(1) match {
150             case ObjectType.POINT => 2 // x + y
151             case ObjectType.LINESTRING | ObjectType.MULTIPOINT => 4 // list of x, list of y
152             case ObjectType.POLYGON | ObjectType.MULTILINESTRING => 6 // list of list of x, list of list of y
153             case ObjectType.MULTIPOLYGON => 8 // list of list of list of x, list of list of list of y
154             case ObjectType.GEOMETRY => 1 // wkb
155           }
156         case ObjectType.LIST => 2
157         case ObjectType.MAP => 3
158         case _ => 1
159       }
160       val include = columns.contains(i)
161       var j = 0
162       while (j < count) {
163         buffer += include
164         j += 1
165       }
166       i += 1
167     }
168 
169     buffer += fid
170 
171     buffer.toArray
172   }
