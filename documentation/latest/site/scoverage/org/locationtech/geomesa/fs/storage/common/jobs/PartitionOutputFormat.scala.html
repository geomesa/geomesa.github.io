28 /**
29   * Output format that writes to multiple partition files
30   *
31   * @param delegate underlying output format for a single file
32   */
33 class PartitionOutputFormat(delegate: SingleFileOutputFormat) extends OutputFormat[Void, SimpleFeature] {
34 
35   override def getRecordWriter(context: TaskAttemptContext): RecordWriter[Void, SimpleFeature] =
36     new PartitionSchemeRecordWriter(context)
37 
38   override def getOutputCommitter(context: TaskAttemptContext): OutputCommitter =
39     delegate.getOutputCommitter(context)
40 
41   // same as FileOutputFormat, but doesn't require that output directory doesn't exist
42   override def checkOutputSpecs(job: JobContext): Unit = {
43     // Ensure that the output directory is set
44     val outDir = FileOutputFormat.getOutputPath(job)
45     if (outDir == null) {
46       throw new InvalidJobConfException("Output directory not set")
47     }
48     // get delegation token for outDir's file system
49     TokenCache.obtainTokensForNamenodes(job.getCredentials, Array[Path](outDir), job.getConfiguration)
50   }

52   class PartitionSchemeRecordWriter(context: TaskAttemptContext)
53       extends RecordWriter[Void, SimpleFeature] with LazyLogging {
54 
55     import StorageConfiguration.Counters.{Features, Group}
56 
57     private val storage = {
58       val conf = context.getConfiguration
59       val root = StorageConfiguration.getRootPath(conf)
60       val fsc = FileSystemContext(root, conf)
61       val metadata = StorageMetadataFactory.load(fsc).getOrElse {
62         throw new IllegalArgumentException(s"No storage defined under path '$root'")
63       }
64       FileSystemStorageFactory(fsc, metadata)
65     }
66     private val encoding = storage.metadata.encoding
67     private val leaf = storage.metadata.leafStorage
68 
69     private val fileType = StorageConfiguration.getFileType(context.getConfiguration)
70     private val fileSize = StorageConfiguration.getTargetFileSize(context.getConfiguration)
71 
72     private val counter = context.getCounter(Group, Features)
73     private val cache = scala.collection.mutable.Map.empty[String, PartitionState]
74 
75     private val workPath = delegate.getOutputCommitter(context).asInstanceOf[FileOutputCommitter].getWorkPath
76 
77     private def newState(partition: String): PartitionState = {
78       val estimator = storage match {
79         case s: SizeableFileSystemStorage => s.targetSize(fileSize).map(s.estimator)
80         case _ => None
81       }
82       estimator match {
83         case None => new SinglePartitionState(partition)
84         case Some(e) => new ChunkedPartitionState(partition, e, context)
85       }
86     }
87 
88     override def write(key: Void, value: SimpleFeature): Unit = {
89       val partition = storage.metadata.scheme.getPartitionName(value)
90       val state = cache.getOrElseUpdate(partition, newState(partition))
91       state.write(key, value)
92       counter.increment(1)
93     }
94 
95     override def close(context: TaskAttemptContext): Unit = {
96       cache.foreach { case (partition, state) =>
97         logger.debug(s"Closing writer for $partition")
98         state.close(context)
99         storage.metadata.addPartition(state.meta)
100       }
101       CloseWithLogging(storage)
102     }
103 
104     sealed abstract class PartitionState(partition: String) {
105 
106       private var count = 0L
107       private val bounds = new Envelope()
108       private val files = ArrayBuffer.empty[String]
109 
110       def write(key: Void, value: SimpleFeature): Unit = {
111         val geom = value.getDefaultGeometry.asInstanceOf[Geometry]
112         if (geom != null) {
113           bounds.expandToInclude(geom.getEnvelopeInternal)
114         }
115         count += 1L
116       }
117 
118       def meta: PartitionMetadata = {
119         val millis = System.currentTimeMillis()
120         val f = files.map(StorageFile(_, millis))
121         PartitionMetadata(partition, f.toSeq, PartitionBounds(bounds), count)
122       }
123 
124       def close(context: TaskAttemptContext): Unit =
125         context.getCounter(Group, StorageConfiguration.Counters.partition(partition)).increment(count)
126 
127       protected def newWriter(): (Path, RecordWriter[Void, SimpleFeature]) = {
128         val file = StorageUtils.nextFile(workPath, partition, leaf, encoding, fileType)
129         logger.debug(s"Creating record writer at path $file")
130         files += file.getName
131         // noinspection LanguageFeature
132         (file, delegate.getRecordWriter(context, file))
133       }
134     }
135 
136     private class SinglePartitionState(partition: String) extends PartitionState(partition) {
137 
138       private val writer = newWriter()._2
139 
140       override def write(key: Void, value: SimpleFeature): Unit = {
141         writer.write(key, value)
142         super.write(key, value)
143       }
144 
145       override def close(context: TaskAttemptContext): Unit = {
146         writer.close(context)
147         super.close(context)
148       }
149     }
150 
151     private class ChunkedPartitionState(partition: String, estimator: FileSizeEstimator, context: TaskAttemptContext)
152         extends PartitionState(partition) {
153 
154       private var count = 0L // number of features written
155       private var total = 0L // sum size of all finished chunks
156       private var remaining = estimator.estimate(0L)
157 
158       private var path: Path = _
159       private var writer: RecordWriter[Void, SimpleFeature] = _
160 
161       override def write(key: Void, value: SimpleFeature): Unit = {
162         if (writer == null) {
163           val (p, w) = newWriter()
164           path = p
165           writer = w
166         }
167         writer.write(key, value)
168         count += 1
169         remaining -= 1
170         if (remaining == 0) {
171           writer.close(context)
172           writer = null
173           // adjust our estimate to account for the actual bytes written
174           total += storage.context.fs.getFileStatus(path).getLen
175           estimator.update(total, count)
176           remaining = estimator.estimate(0L)
177         }
178         super.write(key, value)
179       }
180 
181       override def close(context: TaskAttemptContext): Unit = {
182         if (writer != null) {
183           writer.close(context)
184         }
185         super.close(context)
186       }
187     }
188   }
189 }
