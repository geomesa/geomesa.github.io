19 /**
20   * Original parquet mapping - not versioned. Only supports points
21   *
22   * Of note, the FID field was marked as REPEATED, which seems to be an error and does not work with reading
23   * parquet files as avro GenericRecords (which is the main way to read an unknown parquet file)
24   */
25 object SimpleFeatureParquetSchemaV0 {
26 
27   import scala.collection.JavaConverters._
28 
29   def apply(sft: SimpleFeatureType): MessageType = {
30     val idField =
31       Types.primitive(PrimitiveTypeName.BINARY, Repetition.REPEATED)
32           .as(OriginalType.UTF8)
33           .named(SimpleFeatureParquetSchema.FeatureIdField)
34 
35     // NOTE: idField goes at the end of the record
36     val fields = sft.getAttributeDescriptors.asScala.map(convertField) :+ idField
37     new MessageType(sft.getTypeName, fields.asJava)
38   }
39 
40   private def convertField(ad: AttributeDescriptor): Type = {
41     val bindings = ObjectType.selectType(ad)
42     val builder = bindings.head match {
43       case ObjectType.GEOMETRY =>
44         Types.buildGroup(Repetition.REQUIRED)
45             .primitive(PrimitiveTypeName.DOUBLE, Repetition.REQUIRED).named(SimpleFeatureParquetSchema.GeometryColumnX)
46             .primitive(PrimitiveTypeName.DOUBLE, Repetition.REQUIRED).named(SimpleFeatureParquetSchema.GeometryColumnY)
47 
48       case ObjectType.DATE    => Types.primitive(PrimitiveTypeName.INT64, Repetition.OPTIONAL)
49       case ObjectType.STRING  => Types.primitive(PrimitiveTypeName.BINARY, Repetition.OPTIONAL).as(OriginalType.UTF8)
50       case ObjectType.INT     => Types.primitive(PrimitiveTypeName.INT32, Repetition.OPTIONAL)
51       case ObjectType.DOUBLE  => Types.primitive(PrimitiveTypeName.DOUBLE, Repetition.OPTIONAL)
52       case ObjectType.LONG    => Types.primitive(PrimitiveTypeName.INT64, Repetition.OPTIONAL)
53       case ObjectType.FLOAT   => Types.primitive(PrimitiveTypeName.FLOAT, Repetition.OPTIONAL)
54       case ObjectType.BOOLEAN => Types.primitive(PrimitiveTypeName.BOOLEAN, Repetition.OPTIONAL)
55       case ObjectType.BYTES   => Types.primitive(PrimitiveTypeName.BINARY, Repetition.OPTIONAL)
56       case ObjectType.UUID    => Types.primitive(PrimitiveTypeName.BINARY, Repetition.OPTIONAL)
57       case ObjectType.LIST    => Types.optionalList().optionalElement(matchType(bindings(1)))
58       case ObjectType.MAP     => Types.optionalMap().key(matchType(bindings(1))).optionalValue(matchType(bindings(2)))
59     }
60     builder.named(ad.getLocalName)
61   }
62 
63   private def matchType(objType: ObjectType): PrimitiveTypeName = {
64     objType match {
65       case ObjectType.DATE    => PrimitiveTypeName.INT64
66       case ObjectType.STRING  => PrimitiveTypeName.BINARY
67       case ObjectType.INT     => PrimitiveTypeName.INT32
68       case ObjectType.DOUBLE  => PrimitiveTypeName.DOUBLE
69       case ObjectType.LONG    => PrimitiveTypeName.INT64
70       case ObjectType.FLOAT   => PrimitiveTypeName.FLOAT
71       case ObjectType.BOOLEAN => PrimitiveTypeName.BOOLEAN
72       case ObjectType.BYTES   => PrimitiveTypeName.BINARY
73       case ObjectType.UUID    => PrimitiveTypeName.BINARY
74     }
75   }
