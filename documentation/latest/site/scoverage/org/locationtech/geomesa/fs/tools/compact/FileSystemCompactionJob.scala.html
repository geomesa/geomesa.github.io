36 trait FileSystemCompactionJob extends StorageConfiguration with JobWithLibJars {
37 
38   import FileSystemCompactionJob.{FailedCounter, MappedCounter}
39 
40   def run(
41       storage: FileSystemStorage,
42       partitions: Seq[PartitionMetadata],
43       targetFileSize: Option[Long],
44       tempPath: Option[Path],
45       libjarsFiles: Seq[String],
46       libjarsPaths: Iterator[() => Seq[File]],
47       statusCallback: StatusCallback): JobResult = {
48 
49     val job = Job.getInstance(new Configuration(storage.context.conf), "GeoMesa Storage Compaction")
50 
51     setLibJars(job, libjarsFiles, libjarsPaths)
52     job.setJarByClass(this.getClass)
53 
54     // InputFormat and Mappers
55     job.setInputFormatClass(classOf[PartitionInputFormat])
56     job.setMapperClass(classOf[CompactionMapper])
57 
58     // No reducers - Mapper will read/write its own things
59     job.setNumReduceTasks(0)
60 
61     job.setMapOutputKeyClass(classOf[Void])
62     job.setMapOutputValueClass(classOf[SimpleFeature])
63     job.setOutputKeyClass(classOf[Void])
64     job.setOutputValueClass(classOf[SimpleFeature])
65 
66     val qualifiedTempPath = tempPath.map(storage.context.fs.makeQualified)
67 
68     StorageConfiguration.setRootPath(job.getConfiguration, storage.context.root)
69     StorageConfiguration.setPartitions(job.getConfiguration, partitions.map(_.name).toArray)
70     StorageConfiguration.setFileType(job.getConfiguration, FileType.Compacted)
71     targetFileSize.foreach(StorageConfiguration.setTargetFileSize(job.getConfiguration, _))
72 
73     FileOutputFormat.setOutputPath(job, qualifiedTempPath.getOrElse(storage.context.root))
74 
75     // MapReduce options
76     job.getConfiguration.set("mapred.map.tasks.speculative.execution", "false")
77     job.getConfiguration.set("mapreduce.job.user.classpath.first", "true")
78 
79     configureOutput(storage.metadata.sft, job)
80 
81     // save the existing files so we can delete them afterwards
82     // mimic the filtering done in PartitionInputFormat
83     val sizeable = Option(storage).collect { case s: SizeableFileSystemStorage => s }
84     val sizeCheck = sizeable.flatMap(s => s.targetSize(targetFileSize).map(t => (p: Path) => s.fileIsSized(p, t)))
85     val existingDataFiles = partitions.toList.flatMap { p =>
86       val files = storage.getFilePaths(p.name).filterNot(f => sizeCheck.exists(_.apply(f.path)))
87       // TODO get counts right... use m/r counters?
88       if (files.isEmpty) { None } else { Some(p.copy(files = files.map(_.file)) -> files) }
89     }
90 
91     def mapCounters = Seq((MappedCounter, written(job)), (FailedCounter, failed(job)))
92 
93     val result = JobRunner.run(job, statusCallback, mapCounters, Seq.empty).merge {
94       qualifiedTempPath.map { tp =>
95         new DistributedCopy().copy(Seq(tp), storage.context.root, statusCallback)
96       }
97     }
98 
99     result match {
100       case JobSuccess(message, counts) =>
101         if (message.nonEmpty) {
102           Command.user.info(message)
103         }
104         Command.user.info("Removing old files")
105         existingDataFiles.foreach { case (partition, files) =>
106           val counter = StorageConfiguration.Counters.partition(partition.name)
107           val count = Option(job.getCounters.findCounter(StorageConfiguration.Counters.Group, counter)).map(_.getValue)
108           files.foreach(f => storage.context.fs.delete(f.path, false))
109           storage.metadata.removePartition(partition.copy(count = count.getOrElse(0L)))
110           val removed = count.map(c => s"containing $c features ").getOrElse("")
111           Command.user.info(s"Removed ${TextTools.getPlural(files.size, "file")} ${removed}in partition ${partition.name}")
112         }
113         Command.user.info("Compacting metadata")
114         storage.metadata.compact(None, None, threads = 4)
115         JobSuccess("", counts)
116 
117       case j => j
118     }
119   }
120 
121   private def written(job: Job): Long =
122     job.getCounters.findCounter(OutputCounters.Group, OutputCounters.Written).getValue
123 
124   private def failed(job: Job): Long =
125     job.getCounters.findCounter(OutputCounters.Group, OutputCounters.Failed).getValue
126 }

139     * is paired with the PartitionRecordReader which will feed all the features into a single map task
140     */
141   class CompactionMapper extends Mapper[Void, SimpleFeature, Void, SimpleFeature] with LazyLogging {
142 
143     type Context = Mapper[Void, SimpleFeature, Void, SimpleFeature]#Context
144 
145     private var written: Counter = _
146     private var mapped: Counter = _
147 
148     override def setup(context: Context): Unit = {
149       super.setup(context)
150       written = context.getCounter(OutputCounters.Group, OutputCounters.Written)
151       mapped = context.getCounter("org.locationtech.geomesa.fs.compaction", MappedCounter)
152     }
153 
154     override def map(key: Void, sf: SimpleFeature, context: Context): Unit = {
155       sf.getUserData.put(Hints.USE_PROVIDED_FID, java.lang.Boolean.TRUE)
156       mapped.increment(1)
157       context.write(null, sf)
158       written.increment(1)
159     }
