20 class FileBasedMetadataFactory extends StorageMetadataFactory {
21 
22   override def name: String = FileBasedMetadata.MetadataType
23 
24   /**
25     * Loads a metadata instance from an existing root. The metadata info is persisted in a `metadata.json`
26     * file under the root path.
27     *
28     * Will return a cached instance, if available. If a previous check was made to load a file from this root,
29     * and the file did not exist, will not re-attempt to load it until after a configurable timeout
30     *
31     * @see `org.locationtech.geomesa.fs.storage.common.utils.PathCache#CacheDurationProperty()`
32     * @param context file context
33     * @return
34     **/
35   override def load(context: FileSystemContext): Option[FileBasedMetadata] = {
36     val json = MetadataJson.readMetadata(context)
37     // note: do this after loading the json to allow for old metadata transition
38     val cached = FileBasedMetadataFactory.cached(context, json.getOrElse(FileBasedMetadata.LegacyOptions).options)
39     json match {
40       case Some(m) if m.name.equalsIgnoreCase(name) =>
41         cached.orElse(throw new IllegalArgumentException(s"Could not load metadata at root '${context.root.toUri}'"))
42 
43       case None if cached.isDefined =>
44         // a file-based metadata impl exists, but was created with an older version
45         // create a config file pointing to it, and register that
46         MetadataJson.writeMetadata(context, FileBasedMetadata.LegacyOptions)
47         cached
48 
49       case _ => None
50     }
51   }
52 
53   override def create(context: FileSystemContext, config: Map[String, String], meta: Metadata): FileBasedMetadata = {
54     val sft = namespaced(meta.sft, context.namespace)
55     // load the partition scheme first in case it fails
56     PartitionSchemeFactory.load(sft, meta.scheme)
57     val renderer = config.get(Config.RenderKey).map(MetadataConverter.apply).getOrElse(RenderCompact)
58     MetadataJson.writeMetadata(context, NamedOptions(name, config + (Config.RenderKey -> renderer.name)))
59     FileBasedMetadataFactory.write(context.fs, context.root, meta)
60     val directory = new Path(context.root, FileBasedMetadataFactory.MetadataDirectory)
61     val metadata = new FileBasedMetadata(context.fs, directory, sft, meta, renderer)
62     FileBasedMetadataFactory.cache.put(FileBasedMetadataFactory.key(context), metadata)
63     metadata
64   }
65 }

67 object FileBasedMetadataFactory extends LazyLogging {
68 
69   val MetadataDirectory = "metadata"
70   val StoragePath = s"$MetadataDirectory/storage.json"
71 
72   private val cache = new ConcurrentHashMap[String, FileBasedMetadata]()
73 
74   private def key(context: FileSystemContext): String =
75     context.namespace.map(ns => s"$ns:${context.root.toUri}").getOrElse(context.root.toUri.toString)
76 
77   private def cached(context: FileSystemContext, config: Map[String, String]): Option[FileBasedMetadata] = {
78     val loader = new java.util.function.Function[String, FileBasedMetadata]() {
79       override def apply(ignored: String): FileBasedMetadata = {
80         val file = new Path(context.root, StoragePath)
81         if (!PathCache.exists(context.fs, file)) { null } else {
82           val directory = new Path(context.root, MetadataDirectory)
83           val meta = WithClose(context.fs.open(file))(MetadataSerialization.deserialize)
84           val sft = namespaced(meta.sft, context.namespace)
85           val renderer = config.get(Config.RenderKey).map(MetadataConverter.apply).getOrElse(RenderPretty)
86           new FileBasedMetadata(context.fs, directory, sft, meta, renderer)
87         }
88       }
89     }
90     Option(cache.computeIfAbsent(key(context), loader))
91   }
92 
93   /**
94     * Write basic metadata to disk. This should be done once, when the storage is created
95     *
96     * @param fs file system
97     * @param root root path
98     * @param metadata simple feature type, file encoding, partition scheme, etc
99     */
100   private [metadata] def write(fs: FileSystem, root: Path, metadata: Metadata): Unit = {
101     val file = new Path(root, StoragePath)
102     WithClose(fs.create(file, true)) { out =>
103       MetadataSerialization.serialize(out, metadata)
104       out.hflush()
105       out.hsync()
106     }
107     PathCache.register(fs, file)
108   }
109 }
