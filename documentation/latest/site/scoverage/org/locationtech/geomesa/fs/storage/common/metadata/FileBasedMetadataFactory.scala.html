21 class FileBasedMetadataFactory extends StorageMetadataFactory {
22 
23   override def name: String = FileBasedMetadata.MetadataType
24 
25   /**
26     * Loads a metadata instance from an existing root. The metadata info is persisted in a `metadata.json`
27     * file under the root path.
28     *
29     * Will return a cached instance, if available. If a previous check was made to load a file from this root,
30     * and the file did not exist, will not re-attempt to load it until after a configurable timeout
31     *
32     * @see `org.locationtech.geomesa.fs.storage.common.utils.PathCache#CacheDurationProperty()`
33     * @param context file context
34     * @return
35     **/
36   override def load(context: FileSystemContext): Option[FileBasedMetadata] = {
37     val json = MetadataJson.readMetadata(context)
38     // note: do this after loading the json to allow for old metadata transition
39     val cached = FileBasedMetadataFactory.cached(context, json.getOrElse(FileBasedMetadata.LegacyOptions).options)
40     json match {
41       case Some(m) if m.name.equalsIgnoreCase(name) =>
42         cached.orElse(throw new IllegalArgumentException(s"Could not load metadata at root '${context.root.toUri}'"))
43 
44       case None if cached.isDefined =>
45         // a file-based metadata impl exists, but was created with an older version
46         // create a config file pointing to it, and register that
47         MetadataJson.writeMetadata(context, FileBasedMetadata.LegacyOptions)
48         cached
49 
50       case _ => None
51     }
52   }
53 
54   override def create(context: FileSystemContext, config: Map[String, String], meta: Metadata): FileBasedMetadata = {
55     val sft = namespaced(meta.sft, context.namespace)
56     // load the partition scheme first in case it fails
57     PartitionSchemeFactory.load(sft, meta.scheme)
58     val renderer = config.get(Config.RenderKey).map(MetadataConverter.apply).getOrElse(RenderCompact)
59     MetadataJson.writeMetadata(context, NamedOptions(name, config + (Config.RenderKey -> renderer.name)))
60     FileBasedMetadataFactory.write(context.fs, context.root, meta)
61     val directory = new Path(context.root, FileBasedMetadataFactory.MetadataDirectory)
62     val metadata = new FileBasedMetadata(context.fs, directory, sft, meta, renderer)
63     FileBasedMetadataFactory.cache.put(FileBasedMetadataFactory.key(context), metadata)
64     metadata
65   }
66 }

68 object FileBasedMetadataFactory extends MethodProfiling with LazyLogging {
69 
70   val MetadataDirectory = "metadata"
71   val StoragePath = s"$MetadataDirectory/storage.json"
72 
73   private val cache = new ConcurrentHashMap[String, FileBasedMetadata]()
74 
75   private def key(context: FileSystemContext): String =
76     context.namespace.map(ns => s"$ns:${context.root.toUri}").getOrElse(context.root.toUri.toString)
77 
78   private def cached(context: FileSystemContext, config: Map[String, String]): Option[FileBasedMetadata] = {
79     val loader = new java.util.function.Function[String, FileBasedMetadata]() {
80       override def apply(ignored: String): FileBasedMetadata = {
81         val file = new Path(context.root, StoragePath)
82         if (!PathCache.exists(context.fs, file)) { null } else {
83           val directory = new Path(context.root, MetadataDirectory)
84           val meta = WithClose(context.fs.open(file))(MetadataSerialization.deserialize)
85           val sft = namespaced(meta.sft, context.namespace)
86           val renderer = config.get(Config.RenderKey).map(MetadataConverter.apply).getOrElse(RenderPretty)
87           new FileBasedMetadata(context.fs, directory, sft, meta, renderer)
88         }
89       }
90     }
91     Option(cache.computeIfAbsent(key(context), loader))
92   }
93 
94   /**
95     * Write basic metadata to disk. This should be done once, when the storage is created
96     *
97     * @param fs file system
98     * @param root root path
99     * @param metadata simple feature type, file encoding, partition scheme, etc
100     */
101   private [metadata] def write(fs: FileSystem, root: Path, metadata: Metadata): Unit = {
102     val file = new Path(root, StoragePath)
103     WithClose(fs.create(file, true)) { out =>
104       MetadataSerialization.serialize(out, metadata)
105       out.hflush()
106       out.hsync()
107     }
108     PathCache.register(fs, file)
109   }
