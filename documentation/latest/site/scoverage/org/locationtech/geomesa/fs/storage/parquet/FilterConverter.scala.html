32   private def reduce(
33       sft: SimpleFeatureType
34     )(result: (Option[FilterPredicate], Option[Filter]),
35       name: String): (Option[FilterPredicate], Option[Filter]) = {
36     val (parquet, geotools) = result
37     val filter = geotools.orNull
38     if (filter == null) {
39       return result // no more filter to evaluate
40     }
41 
42     val bindings = ObjectType.selectType(sft.getDescriptor(name))
43     val col = StringSerialization.alphaNumericSafeString(name)
44 
45     val (predicate, remaining): (Option[FilterPredicate], Option[Filter]) = bindings.head match {
46       // note: non-points use repeated values, which aren't supported in parquet predicates
47       case ObjectType.GEOMETRY if bindings.last == ObjectType.POINT => spatial(sft, name, filter, col)
48       case ObjectType.DATE    => temporal(sft, name, filter, FilterApi.longColumn(col))
49       case ObjectType.STRING  => attribute(sft, name, filter, FilterApi.binaryColumn(col), Binary.fromString)
50       case ObjectType.INT     => attribute(sft, name, filter, FilterApi.intColumn(col), identity[java.lang.Integer])
51       case ObjectType.LONG    => attribute(sft, name, filter, FilterApi.longColumn(col), identity[java.lang.Long])
52       case ObjectType.FLOAT   => attribute(sft, name, filter, FilterApi.floatColumn(col), identity[java.lang.Float])
53       case ObjectType.DOUBLE  => attribute(sft, name, filter, FilterApi.doubleColumn(col), identity[java.lang.Double])
54       case ObjectType.BOOLEAN => boolean(sft, name, filter, FilterApi.booleanColumn(col))
55       case _ => (None, Some(filter))
56     }
57 
58     ((predicate.toSeq ++ parquet).reduceLeftOption(FilterApi.and), remaining)
59   }

61   private def spatial(
62       sft: SimpleFeatureType,
63       name: String,
64       filter: Filter,
65       col: String): (Option[FilterPredicate], Option[Filter]) = {
66     val (spatial, nonSpatial) = FilterExtractingVisitor(filter, name, sft, SpatialFilterStrategy.spatialCheck)
67     val predicate = spatial.map(FilterHelper.extractGeometries(_, name)).flatMap { extracted =>
68       Some(extracted).filter(e => e.nonEmpty && !e.disjoint).map { e =>
69         val xy = e.values.map(GeometryUtils.bounds).reduce { (a, b) =>
70           (math.min(a._1, b._1), math.min(a._2, b._2), math.max(a._3, b._3), math.max(a._4, b._4))
71         }
72         val xcol = FilterApi.doubleColumn(s"$col.x")
73         val ycol = FilterApi.doubleColumn(s"$col.y")
74         val filters = Seq[FilterPredicate](
75           FilterApi.gtEq(xcol, Double.box(xy._1)),
76           FilterApi.gtEq(ycol, Double.box(xy._2)),
77           FilterApi.ltEq(xcol, Double.box(xy._3)),
78           FilterApi.ltEq(ycol, Double.box(xy._4))
79         )
80         filters.reduce(FilterApi.and)
81       }
82     }
83     (predicate, nonSpatial)
84   }

86   private def temporal(
87       sft: SimpleFeatureType,
88       name: String,
89       filter: Filter,
90       col: LongColumn): (Option[FilterPredicate], Option[Filter]) = {
91     val (temporal, nonTemporal) = FilterExtractingVisitor(filter, name, sft)
92     val predicate = temporal.map(FilterHelper.extractIntervals(_, name)).flatMap { extracted =>
93       Some(extracted).filter(e => e.nonEmpty && !e.disjoint && e.forall(_.isBounded)).map { e =>
94         val filters = e.values.map { bounds =>
95           if (bounds.isEquals) {
96             FilterApi.eq(col, Long.box(bounds.lower.value.get.toInstant.toEpochMilli))
97           } else {
98             val lower: Option[FilterPredicate] = bounds.lower.value.map { value =>
99               val long = Long.box(value.toInstant.toEpochMilli)
100               if (bounds.lower.inclusive) { FilterApi.gtEq(col, long) } else { FilterApi.gt(col, long) }
101             }
102             val upper: Option[FilterPredicate] = bounds.upper.value.map { value =>
103               val long = Long.box(value.toInstant.toEpochMilli)
104               if (bounds.upper.inclusive) { FilterApi.ltEq(col, long) } else { FilterApi.lt(col, long) }
105             }
106             (lower, upper) match {
107               case (Some(lo), Some(hi)) => FilterApi.and(lo, hi)
108               case (Some(f), None) => f
109               case (None, Some(f)) => f
110               case (None, None) => throw new IllegalStateException() // shouldn't happen due to checks above
111             }
112           }
113         }
114         merge(filters)
115       }
116     }
117     (predicate, nonTemporal)
118   }

170   /**
171     * Merge OR'd filters
172     *
173     * Detect and re-write "not equals" filters to handle null values
174     *
175     * "foo != x" comes out as "foo < x OR foo > x"
176     * that won't return null values - not normally a concern since we don't index them in key-value dbs,
177     * but we want to handle that case here
178     *
179     * @param filters filters
180     * @return combined filter
181     */
182   private def merge[T <: Comparable[T]](filters: Seq[FilterPredicate]): FilterPredicate = {
183     lazy val values = filters.collect {
184       case f: Lt[T] => (f.getColumn.asInstanceOf[Column[T] with SupportsEqNotEq], f.getValue)
185       case f: Gt[T] => (f.getColumn.asInstanceOf[Column[T] with SupportsEqNotEq], f.getValue)
186     }.distinct
187 
188     if (filters.lengthCompare(2) == 0 &&
189         filters.exists(_.isInstanceOf[Lt[_]]) &&
190         filters.exists(_.isInstanceOf[Gt[_]]) &&
191         values.lengthCompare(1) == 0) {
192       FilterApi.notEq(values.head._1, values.head._2)
193     } else {
194       filters.reduce(FilterApi.or)
195     }
196   }
