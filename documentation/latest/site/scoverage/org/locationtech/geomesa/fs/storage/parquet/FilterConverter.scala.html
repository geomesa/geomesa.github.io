36   private def reduce(
37       sft: SimpleFeatureType
38     )(result: (Option[FilterPredicate], Option[Filter]),
39       name: String): (Option[FilterPredicate], Option[Filter]) = {
40     val (parquet, geotools) = result
41     val filter = geotools.orNull
42     if (filter == null) {
43       return result // no more filter to evaluate
44     }
45 
46     val bindings = ObjectType.selectType(sft.getDescriptor(name))
47     val col = StringSerialization.alphaNumericSafeString(name)
48 
49     val (predicate, remaining): (Option[FilterPredicate], Option[Filter]) = bindings.head match {
50       // note: non-points use repeated values, which aren't supported in parquet predicates
51       case ObjectType.GEOMETRY => spatial(sft, name, filter, col, bindings.last)
52       case ObjectType.DATE     => temporal(sft, name, filter, FilterApi.longColumn(col))
53       case ObjectType.STRING   => attribute(sft, name, filter, FilterApi.binaryColumn(col), Binary.fromString)
54       case ObjectType.INT      => attribute(sft, name, filter, FilterApi.intColumn(col), identity[java.lang.Integer])
55       case ObjectType.LONG     => attribute(sft, name, filter, FilterApi.longColumn(col), identity[java.lang.Long])
56       case ObjectType.FLOAT    => attribute(sft, name, filter, FilterApi.floatColumn(col), identity[java.lang.Float])
57       case ObjectType.DOUBLE   => attribute(sft, name, filter, FilterApi.doubleColumn(col), identity[java.lang.Double])
58       case ObjectType.BOOLEAN  => boolean(sft, name, filter, FilterApi.booleanColumn(col))
59       case _ => (None, Some(filter))
60     }
61 
62     ((predicate.toSeq ++ parquet).reduceLeftOption(FilterApi.and), remaining)
63   }

65   private def spatial(
66       sft: SimpleFeatureType,
67       name: String,
68       filter: Filter,
69       col: String,
70       typed: ObjectType): (Option[FilterPredicate], Option[Filter]) = {
71     val (spatial, _) = FilterExtractingVisitor(filter, name, sft, SpatialFilterStrategy.spatialCheck)
72     val xyBounds = spatial.map(FilterHelper.extractGeometries(_, name)).flatMap { extracted =>
73       Some(extracted).filter(e => e.nonEmpty && !e.disjoint).map { e =>
74         e.values.map(GeometryUtils.bounds).reduce { (a, b) =>
75           (math.min(a._1, b._1), math.min(a._2, b._2), math.max(a._3, b._3), math.max(a._4, b._4))
76         }
77       }
78     }
79 
80     val predicate = xyBounds.map { case (xmin, ymin, xmax, ymax) =>
81       // filter against the bbox field
82       val bboxGroup = BoundingBoxField(col, encoded = true).bbox
83       val bboxXminCol = FilterApi.floatColumn(s"$bboxGroup.${BoundingBoxField.XMin}")
84       val bbox = {
85         val yminCol = FilterApi.floatColumn(s"$bboxGroup.${BoundingBoxField.YMin}")
86         val xmaxCol = FilterApi.floatColumn(s"$bboxGroup.${BoundingBoxField.XMax}")
87         val ymaxCol = FilterApi.floatColumn(s"$bboxGroup.${BoundingBoxField.YMax}")
88         Seq[FilterPredicate](
89           FilterApi.ltEq(bboxXminCol, Float.box(xmax.toFloat)),
90           FilterApi.gtEq(xmaxCol, Float.box(xmin.toFloat)),
91           FilterApi.ltEq(yminCol, Float.box(ymax.toFloat)),
92           FilterApi.gtEq(ymaxCol, Float.box(ymin.toFloat))
93         ).reduce(FilterApi.and)
94       }
95       if (typed == ObjectType.POINT) {
96         // point types that are natively encoded don't have bbox fields, as we can filter on them directly
97         val xcol = FilterApi.doubleColumn(s"$col.x")
98         val ycol = FilterApi.doubleColumn(s"$col.y")
99         val xy = Seq[FilterPredicate](
100           FilterApi.gtEq(xcol, Double.box(xmin)),
101           FilterApi.gtEq(ycol, Double.box(ymin)),
102           FilterApi.ltEq(xcol, Double.box(xmax)),
103           FilterApi.ltEq(ycol, Double.box(ymax))
104         ).reduce(FilterApi.and)
105         // if x/y don't exist, then the bbox will (in WKB encoding)
106         FilterApi.or(bbox, xy)
107       } else {
108         // add null for back-compatibility with files that don't contain bbox cols
109         FilterApi.or(FilterApi.eq(bboxXminCol, null.asInstanceOf[java.lang.Float]), bbox)
110       }
111     }
112     // since we don't know what the actual file encoding is up front, we always have to evaluate the full predicate post-read
113     (predicate, Some(filter))
114   }

116   private def temporal(
117       sft: SimpleFeatureType,
118       name: String,
119       filter: Filter,
120       col: LongColumn): (Option[FilterPredicate], Option[Filter]) = {
121     val (predicate, remaining) = attribute[Date, java.lang.Long](sft, name, filter, col, _.getTime)
122     // note: we need to account for both millisecond and microsecond encoded dates
123     // since the reasonable overlap of the two is pretty non-existent, we just use an OR of both potential ranges
124     val millisOrMicros = predicate.map(p => FilterApi.or(p, p.accept(MillisToMicrosVisitor)))
125     (millisOrMicros, remaining)
126   }

128   private def attribute[T : ClassTag, U <: Comparable[U]](
129       sft: SimpleFeatureType,
130       name: String,
131       filter: Filter,
132       col: Column[U] with SupportsLtGt,
133       conversion: T => U): (Option[FilterPredicate], Option[Filter]) = {
134     val (attribute, nonAttribute) = FilterExtractingVisitor(filter, name, sft)
135     val binding = implicitly[ClassTag[T]].runtimeClass.asInstanceOf[Class[T]]
136     val bounds = attribute.map(FilterHelper.extractAttributeBounds(_, name, binding))
137     val predicate = bounds.flatMap { extracted =>
138       Some(extracted).filter(e => e.nonEmpty && !e.disjoint && e.values.forall(_.isBounded)).map { e =>
139         val filters = e.values.map { bounds =>
140           if (bounds.isEquals) {
141             FilterApi.eq(col, conversion(bounds.lower.value.get))
142           } else {
143             val lower: Option[FilterPredicate] = bounds.lower.value.map { value =>
144               val converted = conversion(value)
145               if (bounds.lower.inclusive) { FilterApi.gtEq(col, converted) } else { FilterApi.gt(col, converted) }
146             }
147             val upper: Option[FilterPredicate] = bounds.upper.value.map { value =>
148               val converted = conversion(value)
149               if (bounds.upper.inclusive) { FilterApi.ltEq(col, converted) } else { FilterApi.lt(col, converted) }
150             }
151             (lower, upper) match {
152               case (Some(lo), Some(hi)) => FilterApi.and(lo, hi)
153               case (Some(f), None) => f
154               case (None, Some(f)) => f
155               case (None, None) => throw new IllegalStateException() // shouldn't happen due to checks above
156             }
157           }
158         }
159         merge(filters)
160       }
161     }
162     val remaining = if (bounds.exists(_.precise)) { nonAttribute } else { Some(filter) }
163     (predicate, remaining)
164   }

182   /**
183     * Merge OR'd filters
184     *
185     * Detect and re-write "not equals" filters to handle null values
186     *
187     * FilterHelper methods make "foo != x" comes out as "foo < x OR foo > x"
188    *
189     * that won't return null values - not normally a concern since we don't index them in key-value dbs,
190     * but we want to handle that case here
191     *
192     * @param filters filters
193     * @return combined filter
194     */
195   private def merge[T <: Comparable[T]](filters: Seq[FilterPredicate]): FilterPredicate = {
196     lazy val values = filters.collect {
197       case f: Lt[T] => (f.getColumn.asInstanceOf[Column[T] with SupportsEqNotEq], f.getValue)
198       case f: Gt[T] => (f.getColumn.asInstanceOf[Column[T] with SupportsEqNotEq], f.getValue)
199     }.distinct
200 
201     if (filters.lengthCompare(2) == 0 &&
202         filters.exists(_.isInstanceOf[Lt[_]]) &&
203         filters.exists(_.isInstanceOf[Gt[_]]) &&
204         values.lengthCompare(1) == 0) {
205       FilterApi.notEq(values.head._1, values.head._2)
206     } else {
207       filters.reduce(FilterApi.or)
208     }
209   }

211   /**
212    * Visitor that replaces all millisecond-encoded date values with microsecond-encoded ones. Note that it is not
213    * robust, in that it only works if all values are Longs
214    */
215   private object MillisToMicrosVisitor extends Visitor[FilterPredicate] {
216 
217     import scala.collection.JavaConverters._
218 
219     private def millisToMicros[T <: Comparable[T]](value: T): T =
220       Long.box(value.asInstanceOf[java.lang.Long] * 1000L).asInstanceOf[T]
221 
222     override def visit[T <: Comparable[T]](eq: Eq[T]): FilterPredicate =
223       FilterApi.eq(eq.getColumn.asInstanceOf[Column[T] with SupportsEqNotEq], millisToMicros(eq.getValue))
224     override def visit[T <: Comparable[T]](notEq: NotEq[T]): FilterPredicate =
225       FilterApi.notEq(notEq.getColumn.asInstanceOf[Column[T] with SupportsEqNotEq], millisToMicros(notEq.getValue))
226     override def visit[T <: Comparable[T]](lt: Lt[T]): FilterPredicate =
227       FilterApi.lt(lt.getColumn.asInstanceOf[Column[T] with SupportsLtGt], millisToMicros(lt.getValue))
228     override def visit[T <: Comparable[T]](ltEq: LtEq[T]): FilterPredicate =
229       FilterApi.ltEq(ltEq.getColumn.asInstanceOf[Column[T] with SupportsLtGt], millisToMicros(ltEq.getValue))
230     override def visit[T <: Comparable[T]](gt: Gt[T]): FilterPredicate =
231       FilterApi.gt(gt.getColumn.asInstanceOf[Column[T] with SupportsLtGt], millisToMicros(gt.getValue))
232     override def visit[T <: Comparable[T]](gtEq: GtEq[T]): FilterPredicate =
233       FilterApi.gtEq(gtEq.getColumn.asInstanceOf[Column[T] with SupportsLtGt], millisToMicros(gtEq.getValue))
234     override def visit(and: And): FilterPredicate = FilterApi.and(and.getLeft.accept(this), and.getRight.accept(this))
235     override def visit(or: Or): FilterPredicate = FilterApi.or(or.getLeft.accept(this), or.getRight.accept(this))
236     override def visit(not: Not): FilterPredicate = FilterApi.not(not.getPredicate.accept(this))
237     override def visit[T <: Comparable[T]](in: In[T]): FilterPredicate =
238       FilterApi.in(in.getColumn.asInstanceOf[Column[T] with SupportsEqNotEq], in.getValues.asScala.map(millisToMicros).asJava)
239     override def visit[T <: Comparable[T]](notIn: NotIn[T]): FilterPredicate =
240       FilterApi.notIn(notIn.getColumn.asInstanceOf[Column[T] with SupportsEqNotEq], notIn.getValues.asScala.map(millisToMicros).asJava)
241     override def visit[T <: Comparable[T]](contains: Contains[T]): FilterPredicate =
242       throw new UnsupportedOperationException("visit Contains is not supported")
243     override def visit[T <: Comparable[T], U <: UserDefinedPredicate[T]](udp: UserDefined[T, U]): FilterPredicate = udp
244     override def visit[T <: Comparable[T], U <: UserDefinedPredicate[T]](udp: LogicalNotUserDefined[T, U]): FilterPredicate = udp
245   }
