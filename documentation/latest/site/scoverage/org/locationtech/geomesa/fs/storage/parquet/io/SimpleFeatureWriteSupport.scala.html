54   private def init(schema: SimpleFeatureParquetSchema): WriteContext = {
55     this.writer = new SimpleFeatureWriteSupport.SimpleFeatureWriter(schema)
56     this.geoParquetObserver = new GeoParquetObserver(schema)
57     this.baseMetadata = schema.metadata
58     new WriteContext(schema.schema, schema.metadata)
59   }
60 
61   // called per block
62   override def prepareForWrite(recordConsumer: RecordConsumer): Unit = consumer = recordConsumer
63 
64   // called per row
65   override def write(record: SimpleFeature): Unit = {
66     writer.write(consumer, record)
67     geoParquetObserver.write(record)
68   }
69 
70   // called once at the end
71   override def finalizeWrite(): FinalizedWriteContext = {
72     try {
73       val metadata = new java.util.HashMap[String, String]()
74       metadata.putAll(baseMetadata)
75       metadata.putAll(geoParquetObserver.metadata())
76       new FinalizedWriteContext(metadata)
77     } finally {
78       CloseWithLogging(geoParquetObserver)
79     }
80   }

87   private class SimpleFeatureWriter(schema: SimpleFeatureParquetSchema) {
88 
89     private val attributes =
90       Array.tabulate(schema.sft.getAttributeCount)(i => attribute(schema.sft.getDescriptor(i), i).asInstanceOf[AttributeWriter[AnyRef]])
91     private val fids = new FidWriter(attributes.length) // put the ID at the end of the record
92     private val vis = if (schema.hasVisibilities) { new VisibilityWriter(attributes.length + 1) } else { null }
93     private val bboxes = schema.boundingBoxes.zipWithIndex.map { case (field, bi) =>
94       val i = schema.sft.indexOf(StringSerialization.decodeAlphaNumericSafeString(field.geometry))
95       i -> new BBoxWriter(field.bbox, attributes.length + 1 + Option(vis).size + bi)
96     }
97 
98     def write(consumer: RecordConsumer, value: SimpleFeature): Unit = {
99       consumer.startMessage()
100       var i = 0
101       while (i < attributes.length) {
102         attributes(i).apply(consumer, value.getAttribute(i))
103         i += 1
104       }
105       fids.apply(consumer, value.getID)
106       if (vis != null) {
107         vis.apply(consumer, SecurityUtils.getVisibility(value))
108       }
109       bboxes.foreach { case (i, writer) =>
110         writer.apply(consumer, value.getAttribute(i).asInstanceOf[Geometry])
111       }
112       consumer.endMessage()
113     }
114 
115     private def attribute(descriptor: AttributeDescriptor, index: Int): AttributeWriter[_] =
116       attribute(alphaNumericSafeString(descriptor.getLocalName), index, ObjectType.selectType(descriptor))
117 
118     private def attribute(name: String, index: Int, bindings: Seq[ObjectType]): AttributeWriter[_] = {
119       bindings.head match {
120         case ObjectType.GEOMETRY => geometry(name, index, bindings.last)
121         case ObjectType.DATE     => date(name, index)
122         case ObjectType.STRING   => new StringWriter(name, index)
123         case ObjectType.INT      => new IntegerWriter(name, index)
124         case ObjectType.LONG     => new LongWriter(name, index)
125         case ObjectType.FLOAT    => new FloatWriter(name, index)
126         case ObjectType.DOUBLE   => new DoubleWriter(name, index)
127         case ObjectType.BYTES    => new BytesWriter(name, index)
128         case ObjectType.LIST     => list(name, index, attribute("element", 0, bindings.drop(1)))
129         case ObjectType.MAP      => new MapWriter(name, index, attribute("key", 0, bindings.slice(1, 2)), attribute("value", 1, bindings.slice(2, 3)))
130         case ObjectType.BOOLEAN  => new BooleanWriter(name, index)
131         case ObjectType.UUID     => new UuidWriter(name, index)
132         case _ => throw new IllegalArgumentException(s"Can't serialize field '$name' of type ${bindings.head}")
133       }
134     }
135 
136     // TODO support z/m
137     private def geometry(name: String, index: Int, binding: ObjectType): AttributeWriter[_] = {
138       if (schema.encodings.geometry == GeometryEncoding.GeoParquetWkb) {
139         new WkbWriter(name, index)
140       } else {
141         val native = schema.encodings.geometry == GeometryEncoding.GeoParquetNative
142         binding match {
143           case ObjectType.POINT                     => new PointWriter(name, index)
144           case ObjectType.LINESTRING      if native => new GeoParquetNativeLineStringWriter(name, index)
145           case ObjectType.LINESTRING                => new LineStringWriter(name, index)
146           case ObjectType.POLYGON         if native => new GeoParquetNativePolygonWriter(name, index)
147           case ObjectType.POLYGON                   => new PolygonWriter(name, index)
148           case ObjectType.MULTIPOINT      if native => new GeoParquetNativeMultiPointWriter(name, index)
149           case ObjectType.MULTIPOINT                => new MultiPointWriter(name, index)
150           case ObjectType.MULTILINESTRING if native => new GeoParquetNativeMultiLineStringWriter(name, index)
151           case ObjectType.MULTILINESTRING           => new MultiLineStringWriter(name, index)
152           case ObjectType.MULTIPOLYGON    if native => new GeoParquetNativeMultiPolygonWriter(name, index)
153           case ObjectType.MULTIPOLYGON              => new MultiPolygonWriter(name, index)
154           case ObjectType.GEOMETRY_COLLECTION       => new WkbWriter(name, index)
155           case ObjectType.GEOMETRY                  => new WkbWriter(name, index)
156           case _ => throw new IllegalArgumentException(s"Can't serialize field '$name' of type $binding")
157         }
158       }
159     }
160 
161     private def list(name: String, index: Int, elements: AttributeWriter[_]): AttributeWriter[_] = {
162       schema.encodings.list match {
163         case ListEncoding.ThreeLevel => new ListWriter(name, index, elements)
164         case ListEncoding.TwoLevel   => new TwoLevelListWriter(name, index, elements)
165         case encoding => throw new NotImplementedError(encoding.toString)
166       }
167     }
168 
169     private def date(name: String, index: Int): AttributeWriter[_] = {
170       schema.encodings.date match {
171         case DateEncoding.Millis => new DateMillisWriter(name, index)
172         case DateEncoding.Micros => new DateMicrosWriter(name, index)
173         case encoding => throw new NotImplementedError(encoding.toString)
174       }
175     }

256       extends AttributeWriter[java.util.List[T]](name, index) {
257 
258     override def writeFields(consumer: RecordConsumer, value: java.util.List[T]): Unit = {
259       consumer.startGroup()
260       if (!value.isEmpty) {
261         consumer.startField("list", 0)
262         consumer.startGroup()
263         consumer.startField("element", 0)
264         val iter = value.iterator
265         while (iter.hasNext) {
266           consumer.startGroup()
267           val item = iter.next
268           if (item != null) {
269             elements(consumer, item)
270           }
271           consumer.endGroup()
272         }
273         consumer.endField("element", 0)
274         consumer.endGroup()
275         consumer.endField("list", 0)
276       }
277       consumer.endGroup()
278     }
279   }

304       extends AttributeWriter[java.util.Map[U, V]](name, index) {
305     override def writeFields(consumer: RecordConsumer, value: java.util.Map[U, V]): Unit = {
306       consumer.startGroup()
307       if (!value.isEmpty) {
308         consumer.startField("key_value", 0)
309         val iter = value.entrySet().iterator
310         while (iter.hasNext) {
311           val entry = iter.next()
312           consumer.startGroup()
313           keys(consumer, entry.getKey)
314           val v = entry.getValue
315           if (v != null) {
316             values(consumer, v)
317           }
318           consumer.endGroup()
319         }
320         consumer.endField("key_value", 0)
321       }
322       consumer.endGroup()
323     }
