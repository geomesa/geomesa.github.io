180 
181     private val elementWriter = attribute("element", 0, Seq(valueType)).asInstanceOf[AttributeWriter[AnyRef]]
182 
183     override protected def write(consumer: RecordConsumer, value: java.util.List[AnyRef]): Unit = {
184       consumer.startGroup()
185       if (!value.isEmpty) {
186         consumer.startField("list", 0)
187         val iter = value.iterator
188         while (iter.hasNext) {
189           consumer.startGroup()
190           val item = iter.next
191           if (item != null) {
192             elementWriter(consumer, item)
193           }
194           consumer.endGroup()
195         }
196         consumer.endField("list", 0)
197       }
198       consumer.endGroup()

208     override protected def write(consumer: RecordConsumer, value: java.util.Map[AnyRef, AnyRef]): Unit = {
209       consumer.startGroup()
210       if (!value.isEmpty) {
211         consumer.startField("map", 0)
212         val iter = value.entrySet().iterator
213         while (iter.hasNext) {
214           val entry = iter.next()
215           consumer.startGroup()
216           keyWriter(consumer, entry.getKey)
217           val v = entry.getValue
218           if (v != null) {
219             valueWriter(consumer, v)
220           }
221           consumer.endGroup()
222         }
223         consumer.endField("map", 0)
224       }
225       consumer.endGroup()
226     }
