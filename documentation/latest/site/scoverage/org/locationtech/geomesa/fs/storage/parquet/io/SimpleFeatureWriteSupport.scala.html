53   private def init(schema: SimpleFeatureParquetSchema): WriteContext = {
54     this.writer = new SimpleFeatureWriteSupport.SimpleFeatureWriter(schema)
55     this.geoParquetObserver = new GeoParquetObserver(schema)
56     this.baseMetadata = schema.metadata
57     new WriteContext(schema.schema, schema.metadata)
58   }
59 
60   // called per block
61   override def prepareForWrite(recordConsumer: RecordConsumer): Unit = consumer = recordConsumer
62 
63   // called per row
64   override def write(record: SimpleFeature): Unit = {
65     writer.write(consumer, record)
66     geoParquetObserver(record)
67   }
68 
69   // called once at the end
70   override def finalizeWrite(): FinalizedWriteContext = {
71     try {
72       val metadata = new java.util.HashMap[String, String]()
73       metadata.putAll(baseMetadata)
74       metadata.putAll(geoParquetObserver.metadata())
75       new FinalizedWriteContext(metadata)
76     } finally {
77       CloseWithLogging(geoParquetObserver)
78     }
79   }

86   private class SimpleFeatureWriter(schema: SimpleFeatureParquetSchema) {
87 
88     private val attributes =
89       Array.tabulate(schema.sft.getAttributeCount)(i => attribute(schema.sft.getDescriptor(i), i).asInstanceOf[AttributeWriter[AnyRef]])
90     private val fids = new FidWriter(attributes.length) // put the ID at the end of the record
91     private val vis = if (schema.hasVisibilities) { new VisibilityWriter(attributes.length + 1) } else { null }
92     private val bboxes = schema.boundingBoxes.zipWithIndex.map { case (field, bi) =>
93       val i = schema.sft.indexOf(StringSerialization.decodeAlphaNumericSafeString(field.geometry))
94       i -> new BBoxWriter(field.bbox, attributes.length + 1 + Option(vis).size + bi)
95     }
96 
97     def write(consumer: RecordConsumer, value: SimpleFeature): Unit = {
98       consumer.startMessage()
99       var i = 0
100       while (i < attributes.length) {
101         attributes(i).apply(consumer, value.getAttribute(i))
102         i += 1
103       }
104       fids.apply(consumer, value.getID)
105       if (vis != null) {
106         vis.apply(consumer, value.getUserData.get("geomesa.feature.visibility").asInstanceOf[String])
107       }
108       bboxes.foreach { case (i, writer) =>
109         writer.apply(consumer, value.getAttribute(i).asInstanceOf[Geometry])
110       }
111       consumer.endMessage()
112     }
113 
114     private def attribute(descriptor: AttributeDescriptor, index: Int): AttributeWriter[_] =
115       attribute(alphaNumericSafeString(descriptor.getLocalName), index, ObjectType.selectType(descriptor))
116 
117     private def attribute(name: String, index: Int, bindings: Seq[ObjectType]): AttributeWriter[_] = {
118       bindings.head match {
119         case ObjectType.GEOMETRY => geometry(name, index, bindings.last)
120         case ObjectType.DATE     => date(name, index)
121         case ObjectType.STRING   => new StringWriter(name, index)
122         case ObjectType.INT      => new IntegerWriter(name, index)
123         case ObjectType.LONG     => new LongWriter(name, index)
124         case ObjectType.FLOAT    => new FloatWriter(name, index)
125         case ObjectType.DOUBLE   => new DoubleWriter(name, index)
126         case ObjectType.BYTES    => new BytesWriter(name, index)
127         case ObjectType.LIST     => list(name, index, attribute("element", 0, bindings.drop(1)))
128         case ObjectType.MAP      => new MapWriter(name, index, attribute("key", 0, bindings.slice(1, 2)), attribute("value", 1, bindings.slice(2, 3)))
129         case ObjectType.BOOLEAN  => new BooleanWriter(name, index)
130         case ObjectType.UUID     => new UuidWriter(name, index)
131         case _ => throw new IllegalArgumentException(s"Can't serialize field '$name' of type ${bindings.head}")
132       }
133     }
134 
135     // TODO support z/m
136     private def geometry(name: String, index: Int, binding: ObjectType): AttributeWriter[_] = {
137       if (schema.encodings.geometry == GeometryEncoding.GeoParquetWkb) {
138         new WkbWriter(name, index)
139       } else {
140         val native = schema.encodings.geometry == GeometryEncoding.GeoParquetNative
141         binding match {
142           case ObjectType.POINT                     => new PointWriter(name, index)
143           case ObjectType.LINESTRING      if native => new GeoParquetNativeLineStringWriter(name, index)
144           case ObjectType.LINESTRING                => new LineStringWriter(name, index)
145           case ObjectType.POLYGON         if native => new GeoParquetNativePolygonWriter(name, index)
146           case ObjectType.POLYGON                   => new PolygonWriter(name, index)
147           case ObjectType.MULTIPOINT      if native => new GeoParquetNativeMultiPointWriter(name, index)
148           case ObjectType.MULTIPOINT                => new MultiPointWriter(name, index)
149           case ObjectType.MULTILINESTRING if native => new GeoParquetNativeMultiLineStringWriter(name, index)
150           case ObjectType.MULTILINESTRING           => new MultiLineStringWriter(name, index)
151           case ObjectType.MULTIPOLYGON    if native => new GeoParquetNativeMultiPolygonWriter(name, index)
152           case ObjectType.MULTIPOLYGON              => new MultiPolygonWriter(name, index)
153           case ObjectType.GEOMETRY_COLLECTION       => new WkbWriter(name, index)
154           case ObjectType.GEOMETRY                  => new WkbWriter(name, index)
155           case _ => throw new IllegalArgumentException(s"Can't serialize field '$name' of type $binding")
156         }
157       }
158     }
159 
160     private def list(name: String, index: Int, elements: AttributeWriter[_]): AttributeWriter[_] = {
161       schema.encodings.list match {
162         case ListEncoding.ThreeLevel => new ListWriter(name, index, elements)
163         case ListEncoding.TwoLevel   => new TwoLevelListWriter(name, index, elements)
164         case encoding => throw new UnsupportedOperationException(encoding.toString)
165       }
166     }
167 
168     private def date(name: String, index: Int): AttributeWriter[_] = {
169       schema.encodings.date match {
170         case DateEncoding.Millis => new DateMillisWriter(name, index)
171         case DateEncoding.Micros => new DateMicrosWriter(name, index)
172         case encoding => throw new UnsupportedOperationException(encoding.toString)
173       }
174     }

255       extends AttributeWriter[java.util.List[T]](name, index) {
256 
257     override def writeFields(consumer: RecordConsumer, value: java.util.List[T]): Unit = {
258       consumer.startGroup()
259       if (!value.isEmpty) {
260         consumer.startField("list", 0)
261         consumer.startGroup()
262         consumer.startField("element", 0)
263         val iter = value.iterator
264         while (iter.hasNext) {
265           consumer.startGroup()
266           val item = iter.next
267           if (item != null) {
268             elements(consumer, item)
269           }
270           consumer.endGroup()
271         }
272         consumer.endField("element", 0)
273         consumer.endGroup()
274         consumer.endField("list", 0)
275       }
276       consumer.endGroup()
277     }

303       extends AttributeWriter[java.util.Map[U, V]](name, index) {
304     override def writeFields(consumer: RecordConsumer, value: java.util.Map[U, V]): Unit = {
305       consumer.startGroup()
306       if (!value.isEmpty) {
307         consumer.startField("key_value", 0)
308         val iter = value.entrySet().iterator
309         while (iter.hasNext) {
310           val entry = iter.next()
311           consumer.startGroup()
312           keys(consumer, entry.getKey)
313           val v = entry.getValue
314           if (v != null) {
315             values(consumer, v)
316           }
317           consumer.endGroup()
318         }
319         consumer.endField("key_value", 0)
320       }
321       consumer.endGroup()
322     }
