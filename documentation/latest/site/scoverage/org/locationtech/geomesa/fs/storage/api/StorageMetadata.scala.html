18 /**
19   * Metadata interface for managing storage partitions. Metadata implementations can be fairly expensive to
20   * instantiate, as they maintain all the partitions and files for a given storage instance. Generally,
21   * they may not load any partition state until `reload` is invoked - this allows for fast access in the cases
22   * where partition state is not required (e.g. access to partition scheme, blind writes, etc)
23   */
24 trait StorageMetadata extends Compactable with Closeable {
25 
26   /**
27     * The schema for SimpleFeatures stored in the file system storage
28     *
29     * @return schema
30     */
31   def sft: SimpleFeatureType
32 
33   /**
34     * The encoding of the underlying data files
35     *
36     * @return encoding
37     */
38   def encoding: String
39 
40   /**
41     * The partition scheme used to partition features for storage and querying
42     *
43     * @return partition scheme
44     */
45   def scheme: PartitionScheme
46 
47   /**
48     * Are partitions stored as leaves (multiple partitions in a single folder), or does each
49     * partition have a unique folder. Using leaf storage can reduce the level of nesting and make
50     * file system operations faster in some cases.
51     *
52     * @return leaf
53     */
54   def leafStorage: Boolean
55 
56   /**
57    * Get a previously set key-value pair
58    *
59    * @param key key
60    * @return
61    */
62   def get(key: String): Option[String] = None
63 
64   /**
65    * Set a key-value pair
66    *
67    * @param key key
68    * @param value value
69    */
70   def set(key: String, value: String): Unit = throw new NotImplementedError()
71 
72   /**
73     * Get a partition by name. Ensure that `reload` has been invoked at least once before calling this method
74     *
75     * @param name partition name
76     * @return partition metadata, if partition exists
77     */
78   def getPartition(name: String): Option[PartitionMetadata]
79 
80   /**
81     * Get all partitions, with an optional prefix filter. Ensure that `reload` has been invoked at least
82     * once before calling this method
83     *
84     * @param prefix prefix used to match partition names
85     * @return all partitions
86     */
87   def getPartitions(prefix: Option[String] = None): Seq[PartitionMetadata]
88 
89   /**
90     * Add (or update) metadata for a partition
91     *
92     * @param partition partition
93     */
94   def addPartition(partition: PartitionMetadata): Unit
95 
96   /**
97     * Update (or delete) metadata for a partition
98     *
99     * @param partition partition
100     */
101   def removePartition(partition: PartitionMetadata): Unit
102 
103   /**
104    * Overwrite any existing partitions
105    *
106    * @param partitions partitions
107    */
108   def setPartitions(partitions: Seq[PartitionMetadata]): Unit
109 
110   /**
111    * Invalidate any cached state
112    */
113   def invalidate(): Unit
114 }

116 object StorageMetadata {
117 
118   implicit val StorageFileOrdering: Ordering[StorageFile] = Ordering.by[StorageFile, Long](_.timestamp).reverse
119 
120   implicit val StorageFilePathOrdering: Ordering[StorageFilePath] =
121     Ordering.by[StorageFilePath, Long](_.file.timestamp).reverse
122 
123   /**
124     * Metadata for a given partition
125     *
126     * @param name partition name
127     * @param files list of files in the partition (relative to the root directory)
128     * @param bounds estimated spatial bounds for this partition, if known
129     * @param count estimated count of features in this partition
130     */
131   case class PartitionMetadata(name: String, files: Seq[StorageFile], bounds: Option[PartitionBounds], count: Long) {
132 
133     /**
134       * Combine two metadata instances for the same partition
135       *
136       * @param other metadata to combine
137       * @return
138       */
139     def +(other: PartitionMetadata): PartitionMetadata = {
140       val merged = bounds.map(b => other.bounds.map(_ + b).getOrElse(b)).orElse(other.bounds)
141       copy(files = files ++ other.files, bounds = merged, count = count + other.count)
142     }
143 
144     /**
145       * Remove some metadata for the same partition.
146       *
147       * Note that this is a lossy operation, as the reduced bounds aren't known
148       *
149       * @param other metadata to remove
150       * @return
151       */
152     def -(other: PartitionMetadata): PartitionMetadata =
153       copy(files = files.diff(other.files), count = math.max(0, count - other.count))
154   }
155 
156   /**
157    * Holds a storage file
158    *
159    * @param name file name (relative to the root path)
160    * @param timestamp timestamp for the file
161    * @param action type of file (append, modify, delete)
162    * @param sort sort fields, if any, as feature type attribute number
163    * @param bounds known bounds, if any, keyed by feature type attribute number
164    */
165   case class StorageFile(
166       name: String,
167       timestamp: Long,
168       action: StorageFileAction.StorageFileAction = StorageFileAction.Append,
169       sort: Seq[Int] = Seq.empty,
170       bounds: Seq[(Int, String, String)] = Seq.empty
171     )
172 
173   /**
174     * Holds a storage file path
175     *
176     * @param file storage file
177     * @param path full path to the file
178     */
179   case class StorageFilePath(file: StorageFile, path: Path)
180 
181   /**
182     * Action related to a storage file
183     */
184   object StorageFileAction extends Enumeration {
185     type StorageFileAction = Value
186     val Append, Modify, Delete = Value
187   }

189   /**
190     * Immutable representation of an envelope
191     *
192     * Note that conversions to/from 'null' envelopes should be handled carefully, as envelopes are considered
193     * null if xmin > xmax, however, when instantiating an envelope it will re-order the coordinates:
194     *
195     * {{{
196     *   val env = new Envelope()
197     *   val copy = new Envelope(env.getMinX, env.getMinY, env.getMaxX, env.getMaxY)
198     *   copy == env // false
199     * }}}
200     *
201     * Thus, ensure that 'null' envelopes are converted to `None` and not directly to a bounds object. See
202     * `PartitionBounds.apply`
203     *
204     * @param xmin min x dimension
205     * @param ymin min y dimension
206     * @param xmax max x dimension
207     * @param ymax max y dimension
208     */
209   case class PartitionBounds(xmin: Double, ymin: Double, xmax: Double, ymax: Double) {
210 
211     /**
212       * Calculate the minimal bounds encompassing both bounds
213       *
214       * @param b other bounds
215       * @return
216       */
217     def +(b: PartitionBounds): PartitionBounds =
218       PartitionBounds(math.min(xmin, b.xmin), math.min(ymin, b.ymin), math.max(xmax, b.xmax), math.max(ymax, b.ymax))
219 
220     /**
221       * Convert to a mutable envelope
222       *
223       * @return
224       */
225     def envelope: Envelope = new Envelope(xmin, xmax, ymin, ymax)
226   }
227 
228   object PartitionBounds {
229 
230     /**
231       * Converts an envelope to a bounds, handling 'null' (empty) envelopes
232       *
233       * @param env envelope
234       * @return
235       */
236     def apply(env: Envelope): Option[PartitionBounds] =
237       if (env.isNull) { None } else { Some(PartitionBounds(env.getMinX, env.getMinY, env.getMaxX, env.getMaxY)) }
238   }
