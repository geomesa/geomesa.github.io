30   def apply(sft: SimpleFeatureType, description: TypeDescription, fid: Boolean = true): OrcOutputFormatWriter = {
31     val builder = Seq.newBuilder[OrcOutputFormatWriter]
32     builder.sizeHint(sft.getAttributeCount + (if (fid) { 1 } else { 0 }))
33 
34     var i = 0
35     var col = 0
36     while (i < sft.getAttributeCount) {
37       val descriptor = sft.getDescriptor(i)
38       val bindings = ObjectType.selectType(descriptor)
39       val reader = bindings.head match {
40         case ObjectType.GEOMETRY => createGeometryWriter(bindings(1), col, i, description)
41         case ObjectType.DATE     => new DateOutputFormatWriter(col, i)
42         case ObjectType.STRING   => new StringOutputFormatWriter(col, i)
43         case ObjectType.INT      => new IntOutputFormatWriter(col, i)
44         case ObjectType.LONG     => new LongOutputFormatWriter(col, i)
45         case ObjectType.FLOAT    => new FloatOutputFormatWriter(col, i)
46         case ObjectType.DOUBLE   => new DoubleOutputFormatWriter(col, i)
47         case ObjectType.BOOLEAN  => new BooleanOutputFormatWriter(col, i)
48         case ObjectType.BYTES    => new BytesOutputFormatWriter(col, i)
49         case ObjectType.UUID     => new UuidOutputFormatWriter(col, i)
50         case ObjectType.LIST     => new ListOutputFormatWriter(col, i, bindings(1), description)
51         case ObjectType.MAP      => new MapOutputFormatWriter(col, i, bindings(1), bindings(2), description)
52         case _ => throw new IllegalArgumentException(s"Unexpected object type ${bindings.head}")
53       }
54       builder += reader
55       i += 1
56       col += OrcFileSystemStorage.fieldCount(descriptor)
57     }
58 
59     if (fid) {
60       builder += new FidOutputFormatWriter(col)
61     }
62 
63     new SequenceOutputFormatWriter(builder.result)
64   }

66   private def createGeometryWriter(
67       binding: ObjectType,
68       col: Int,
69       i: Int,
70       description: TypeDescription): OrcOutputFormatWriter = {
71     binding match {
72       case ObjectType.POINT           => new PointOutputFormatWriter(col, col + 1, i)
73       case ObjectType.LINESTRING      => new LineStringOutputFormatWriter(col, col + 1, i, description)
74       case ObjectType.MULTIPOINT      => new MultiPointOutputFormatWriter(col, col + 1, i, description)
75       case ObjectType.POLYGON         => new PolygonOutputFormatWriter(col, col + 1, i, description)
76       case ObjectType.MULTILINESTRING => new MultiLineStringOutputFormatWriter(col, col + 1, i, description)
77       case ObjectType.MULTIPOLYGON    => new MultiPolygonOutputFormatWriter(col, col + 1, i, description)
78       case ObjectType.GEOMETRY        => new GeometryWkbOutputFormatWriter(col, i)
79       case _ => throw new IllegalArgumentException(s"Unexpected geometry type $binding")
80     }
81   }

120     * Reads a point attribute from a simple feature and sets it in an output format

122     * @param xCol index of x field, containing points

126   class PointOutputFormatWriter(xCol: Int, yCol: Int, attribute: Int)
127       extends OrcOutputFormatWriter with SetOutputFormatDouble {
128     override def apply(sf: SimpleFeature, output: OrcStruct): Unit = {
129       val point = sf.getAttribute(attribute).asInstanceOf[Point]
130       if (point == null) {
131         output.setFieldValue(xCol, null)
132         output.setFieldValue(yCol, null)
133       } else {
134         var x = output.getFieldValue(xCol).asInstanceOf[DoubleWritable]
135         var y = output.getFieldValue(yCol).asInstanceOf[DoubleWritable]
136         if (x == null) {
137           x = new DoubleWritable
138           output.setFieldValue(xCol, x)
139         }
140         if (y == null) {
141           y = new DoubleWritable
142           output.setFieldValue(yCol, y)
143         }
144         x.set(point.getX)
145         y.set(point.getY)
146       }
147     }
148   }

164       if (line == null) {

169         var y = output.getFieldValue(yCol).asInstanceOf[OrcList[DoubleWritable]]
170         if (x == null) {

175         }

177           y = new OrcList[DoubleWritable](description.getChildren.get(yCol), line.getNumPoints)
178           output.setFieldValue(yCol, y)
179         } else {
180           y.clear()
181         }
182         var i = 0
183         while (i < line.getNumPoints) {
184           val pt = line.getCoordinateN(i)
185           x.add(new DoubleWritable(pt.x))
186           y.add(new DoubleWritable(pt.y))
187           i += 1
188         }
189       }
190     }

236   /**
237     * Reads a polygon attribute from a simple feature and sets it in an output format.
238     * A polygon is modeled as a list of lines, with the first value being the shell,
239     * and any subsequent values being interior holes.
240     *
241     * @see LineStringOutputFormatWriter
242     *
243     * @param xCol index of x field, containing a list of lists of points

247   class PolygonOutputFormatWriter(xCol: Int, yCol: Int, attribute: Int, description: TypeDescription)
248       extends OrcOutputFormatWriter with SetOutputFormatDouble {
249     override def apply(sf: SimpleFeature, output: OrcStruct): Unit = {
250       val polygon = sf.getAttribute(attribute).asInstanceOf[Polygon]
251       if (polygon == null) {
252         output.setFieldValue(xCol, null)
253         output.setFieldValue(yCol, null)
254       } else {
255         var xx = output.getFieldValue(xCol).asInstanceOf[OrcList[OrcList[DoubleWritable]]]
256         var yy = output.getFieldValue(yCol).asInstanceOf[OrcList[OrcList[DoubleWritable]]]
257         if (xx == null) {
258           xx = new OrcList[OrcList[DoubleWritable]](description.getChildren.get(xCol), polygon.getNumInteriorRing + 1)
259           output.setFieldValue(xCol, xx)
260         } else {
261           xx.clear()
262         }
263         if (yy == null) {
264           yy = new OrcList[OrcList[DoubleWritable]](description.getChildren.get(yCol), polygon.getNumInteriorRing + 1)
265           output.setFieldValue(yCol, yy)
266         } else {
267           yy.clear()
268         }
269         var j = 0
270         while (j < polygon.getNumInteriorRing + 1) {
271           val line = if (j == 0) { polygon.getExteriorRing } else { polygon.getInteriorRingN(j - 1) }
272           val x = new OrcList[DoubleWritable](description.getChildren.get(xCol).getChildren.get(0), line.getNumPoints)
273           val y = new OrcList[DoubleWritable](description.getChildren.get(yCol).getChildren.get(0), line.getNumPoints)
274           var i = 0
275           while (i < line.getNumPoints) {
276             val pt = line.getCoordinateN(i)
277             x.add(new DoubleWritable(pt.x))
278             y.add(new DoubleWritable(pt.y))
279             i += 1
280           }
281           xx.add(x)
282           yy.add(y)
283           j += 1
284         }
285       }
286     }
287   }
