33   /**
34     * Create a reader for an ORC batch
35     *
36     * @param sft simple feature type
37     * @param batch row batch
38     * @param columns columns to read, corresponding to simple feature attributes
39     * @param fid read feature id or not
40     * @return
41     */
42   def apply(sft: SimpleFeatureType,
43             batch: VectorizedRowBatch,
44             columns: Option[Set[Int]] = None,
45             fid: Boolean = true): OrcAttributeReader = {
46     val builder = Seq.newBuilder[OrcAttributeReader]
47     builder.sizeHint(columns.map(_.size).getOrElse(sft.getAttributeCount) + (if (fid) { 1 } else { 0 }))
48 
49     var i = 0
50     var col = 0
51     while (i < sft.getAttributeCount) {
52       val descriptor = sft.getDescriptor(i)
53       val bindings = ObjectType.selectType(descriptor)
54       if (columns.forall(_.contains(i))) {
55         val reader = bindings.head match {
56           case ObjectType.GEOMETRY => createGeometryReader(bindings(1), batch.cols, col, i)
57           case ObjectType.DATE     => new DateReader(batch.cols(col).asInstanceOf[TimestampColumnVector], i)
58           case ObjectType.STRING   => new StringReader(batch.cols(col).asInstanceOf[BytesColumnVector], i)
59           case ObjectType.INT      => new IntReader(batch.cols(col).asInstanceOf[LongColumnVector], i)
60           case ObjectType.LONG     => new LongReader(batch.cols(col).asInstanceOf[LongColumnVector], i)
61           case ObjectType.FLOAT    => new FloatReader(batch.cols(col).asInstanceOf[DoubleColumnVector], i)
62           case ObjectType.DOUBLE   => new DoubleReader(batch.cols(col).asInstanceOf[DoubleColumnVector], i)
63           case ObjectType.BOOLEAN  => new BooleanReader(batch.cols(col).asInstanceOf[LongColumnVector], i)
64           case ObjectType.BYTES    => new BytesReader(batch.cols(col).asInstanceOf[BytesColumnVector], i)
65           case ObjectType.UUID     => new UuidReader(batch.cols(col).asInstanceOf[BytesColumnVector], i)
66           case ObjectType.LIST     => new ListReader(batch.cols(col).asInstanceOf[ListColumnVector], i, bindings(1))
67           case ObjectType.MAP      => new MapReader(batch.cols(col).asInstanceOf[MapColumnVector], i, bindings(1), bindings(2))
68           case _ => throw new IllegalArgumentException(s"Unexpected object type ${bindings.head}")
69         }
70         builder += reader
71       }
72       i += 1
73       col += SimpleFeatureTypeDescription.fieldCount(descriptor)
74     }
75 
76     if (fid) {
77       builder += new FidReader(batch.cols(col).asInstanceOf[BytesColumnVector])
78     }
79 
80     new SequenceReader(builder.result)
81   }

84   private def createGeometryReader(binding: ObjectType, cols: Array[ColumnVector], col: Int, i: Int): OrcAttributeReader = {
85     implicit def toDoubleColumnVector(vec: ColumnVector): DoubleColumnVector = vec.asInstanceOf[DoubleColumnVector]
86     implicit def toListColumnVector(vec: ColumnVector): ListColumnVector = vec.asInstanceOf[ListColumnVector]

92       case ObjectType.POINT           => new PointReader(x, y, i)
93       case ObjectType.LINESTRING      => new LineStringReader(x, y, i)
94       case ObjectType.MULTIPOINT      => new MultiPointReader(x, y, i)
95       case ObjectType.POLYGON         => new PolygonReader(x, y, i)
96       case ObjectType.MULTILINESTRING => new MultiLineStringReader(x, y, i)
97       case ObjectType.MULTIPOLYGON    => new MultiPolygonReader(x, y, i)
98       case ObjectType.GEOMETRY        => new GeometryWkbReader(cols(col).asInstanceOf[BytesColumnVector], i)
99       case _ => throw new IllegalArgumentException(s"Unexpected object type $binding")
100     }

367     override def getValue(row: Int): AnyRef = {
368       if (vector.noNulls || !vector.isNull(row)) {
369         var bytes = vector.vector(row)
370         if (vector.start(row) != 0 || vector.length(row) != bytes.length) {
371           val tmp = Array.ofDim[Byte](vector.length(row))
372           System.arraycopy(bytes, vector.start(row), tmp, 0, tmp.length)
373           bytes = tmp
374         }
375         WKBUtils.read(bytes)
376       } else {
