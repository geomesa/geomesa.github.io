20 class JdbcMetadataFactory extends StorageMetadataFactory {
21 
22   override def name: String = JdbcMetadata.MetadataType
23 
24   /**
25     * Creates a metadata instance from an existing root. The metadata connection info is persisted in a
26     * `metadata.json` file under the root path.
27     *
28     * If a previous check was made to load a file from this root, and the file did not exist, will not
29     * re-attempt to load it until after a configurable timeout.
30     *
31     * @see `org.locationtech.geomesa.fs.storage.common.utils.PathCache#CacheDurationProperty()`
32     * @param context file context
33     * @return
34     **/
35   override def load(context: FileSystemContext): Option[JdbcMetadata] = {
36     MetadataJson.readMetadata(context).collect {
37       case NamedOptions(name, opts) if name.equalsIgnoreCase(this.name) =>
38         val root = context.root.toUri.toString
39         val source = JdbcMetadataFactory.createDataSource(opts)
40         try {
41           val metadata = JdbcMetadata.load(source, root).getOrElse {
42             throw new IllegalArgumentException(s"Could not load metadata at root '$root'")
43           }
44           val sft = namespaced(metadata.sft, context.namespace)
45           new JdbcMetadata(source, root, sft, metadata)
46         } catch {
47           case NonFatal(e) => CloseQuietly(source).foreach(e.addSuppressed); throw e
48         }
49     }
50   }
51 
52   override def create(context: FileSystemContext, config: Map[String, String], meta: Metadata): JdbcMetadata = {
53     // load the partition scheme first in case it fails
54     PartitionSchemeFactory.load(meta.sft, meta.scheme)
55     MetadataJson.writeMetadata(context, NamedOptions(name, config))
56     val root = context.root.toUri.toString
57     val sft = namespaced(meta.sft, context.namespace)
58     val source = JdbcMetadataFactory.createDataSource(config)
59     try {
60       JdbcMetadata.create(source, root, meta)
61       new JdbcMetadata(source, root, sft, meta)
62     } catch {
63       case NonFatal(e) => CloseQuietly(source).foreach(e.addSuppressed); throw e
64     }
65   }
66 }

68 object JdbcMetadataFactory extends LazyLogging {
69 
70   /**
71     * Create a jdbc data source based on a configuration
72     *
73     * @param config config
74     * @return
75     */
76   def createDataSource(config: Map[String, String]): PoolingDataSource[PoolableConnection] = {
77     import JdbcMetadata.Config._
78 
79     val url = config.getOrElse(UrlKey, throw new IllegalArgumentException(s"JdbcMetadata requires '$UrlKey'"))
80     config.get(DriverKey).foreach(Class.forName) // required for older drivers
81     val props = new Properties()
82     config.get(UserKey).foreach(props.put("user", _))
83     config.get(PasswordKey).foreach(props.put("password", _))
84 
85     val driver = new DriverManagerConnectionFactory(url, props)
86 
87     // validate the connection parameters
88     WithClose(driver.createConnection()) { connection =>
89       if (!connection.isValid(10)) {
90         throw new IllegalArgumentException(
91           s"Could not create valid connection using configuration ${config.mkString(", ")}")
92       }
93     }
94 
95     def setPoolConfig[T](key: String, conversion: String => T, method: T => Unit): Unit = {
96       config.get(key).foreach { v =>
97         try { method.apply(conversion(v)) } catch {
98           case NonFatal(e) => logger.warn(s"Invalid configuration value '$v' for key $key: $e")
99         }
100       }
101     }
102 
103     val poolConfig = new GenericObjectPoolConfig[PoolableConnection]()
104     setPoolConfig[Int](MaxIdleKey, _.toInt, poolConfig.setMaxIdle)
105     setPoolConfig[Int](MinIdleKey, _.toInt, poolConfig.setMinIdle)
106     setPoolConfig[Int](MaxSizeKey, _.toInt, poolConfig.setMaxTotal)
107     setPoolConfig[Boolean](FairnessKey, _.toBoolean, poolConfig.setFairness)
108     setPoolConfig[Boolean](TestOnBorrowKey, _.toBoolean, poolConfig.setTestOnBorrow)
109     setPoolConfig[Boolean](TestOnCreateKey, _.toBoolean, poolConfig.setTestOnCreate)
110     setPoolConfig[Boolean](TestWhileIdlKey, _.toBoolean, poolConfig.setTestWhileIdle)
111 
112     val factory = new PoolableConnectionFactory(driver, null)
113     val pool = new GenericObjectPool(factory, poolConfig)
114     factory.setPool(pool)
115 
116     new PoolingDataSource[PoolableConnection](pool)
117   }
