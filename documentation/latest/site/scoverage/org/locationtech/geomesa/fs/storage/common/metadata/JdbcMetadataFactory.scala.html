21 class JdbcMetadataFactory extends StorageMetadataFactory {
22 
23   override def name: String = JdbcMetadata.MetadataType
24 
25   /**
26     * Creates a metadata instance from an existing root. The metadata connection info is persisted in a
27     * `metadata.json` file under the root path.
28     *
29     * If a previous check was made to load a file from this root, and the file did not exist, will not
30     * re-attempt to load it until after a configurable timeout.
31     *
32     * @see `org.locationtech.geomesa.fs.storage.common.utils.PathCache#CacheDurationProperty()`
33     * @param context file context
34     * @return
35     **/
36   override def load(context: FileSystemContext): Option[JdbcMetadata] = {
37     MetadataJson.readMetadata(context).collect {
38       case NamedOptions(name, opts) if name.equalsIgnoreCase(this.name) =>
39         val root = context.root.toUri.toString
40         val source = JdbcMetadataFactory.createDataSource(opts)
41         try {
42           val metadata = JdbcMetadata.load(source, root).getOrElse {
43             throw new IllegalArgumentException(s"Could not load metadata at root '$root'")
44           }
45           val sft = namespaced(metadata.sft, context.namespace)
46           new JdbcMetadata(source, root, sft, metadata)
47         } catch {
48           case NonFatal(e) => CloseQuietly(source).foreach(e.addSuppressed); throw e
49         }
50     }
51   }
52 
53   override def create(context: FileSystemContext, config: Map[String, String], meta: Metadata): JdbcMetadata = {
54     // load the partition scheme first in case it fails
55     PartitionSchemeFactory.load(meta.sft, meta.scheme)
56     MetadataJson.writeMetadata(context, NamedOptions(name, config))
57     val root = context.root.toUri.toString
58     val sft = namespaced(meta.sft, context.namespace)
59     val source = JdbcMetadataFactory.createDataSource(config)
60     try {
61       JdbcMetadata.create(source, root, meta)
62       new JdbcMetadata(source, root, sft, meta)
63     } catch {
64       case NonFatal(e) => CloseQuietly(source).foreach(e.addSuppressed); throw e
65     }
66   }
67 }

71   /**
72     * Create a jdbc data source based on a configuration
73     *
74     * @param config config
75     * @return
76     */
77   def createDataSource(config: Map[String, String]): PoolingDataSource[PoolableConnection] = {
78     import JdbcMetadata.Config._
79 
80     val url = config.getOrElse(UrlKey, throw new IllegalArgumentException(s"JdbcMetadata requires '$UrlKey'"))
81     config.get(DriverKey).foreach(Class.forName) // required for older drivers
82     val props = new Properties()
83     config.get(UserKey).foreach(props.put("user", _))
84     config.get(PasswordKey).foreach(props.put("password", _))
85 
86     val driver = new DriverManagerConnectionFactory(url, props)
87 
88     // validate the connection parameters
89     WithClose(driver.createConnection()) { connection =>
90       if (!connection.isValid(10)) {
91         throw new IllegalArgumentException(
92           s"Could not create valid connection using configuration ${config.mkString(", ")}")
93       }
94     }
95 
96     def setPoolConfig[T](key: String, conversion: String => T, method: T => Unit): Unit = {
97       config.get(key).foreach { v =>
98         try { method.apply(conversion(v)) } catch {
99           case NonFatal(e) => logger.warn(s"Invalid configuration value '$v' for key $key: $e")
100         }
101       }
102     }
103 
104     val poolConfig = new GenericObjectPoolConfig[PoolableConnection]()
105     setPoolConfig[Int](MaxIdleKey, _.toInt, poolConfig.setMaxIdle)
106     setPoolConfig[Int](MinIdleKey, _.toInt, poolConfig.setMinIdle)
107     setPoolConfig[Int](MaxSizeKey, _.toInt, poolConfig.setMaxTotal)
108     setPoolConfig[Boolean](FairnessKey, _.toBoolean, poolConfig.setFairness)
109     setPoolConfig[Boolean](TestOnBorrowKey, _.toBoolean, poolConfig.setTestOnBorrow)
110     setPoolConfig[Boolean](TestOnCreateKey, _.toBoolean, poolConfig.setTestOnCreate)
111     setPoolConfig[Boolean](TestWhileIdlKey, _.toBoolean, poolConfig.setTestWhileIdle)
112 
113     val factory = new PoolableConnectionFactory(driver, null)
114     val pool = new GenericObjectPool(factory, poolConfig)
115     factory.setPool(pool)
116 
117     new PoolingDataSource[PoolableConnection](pool)
118   }
