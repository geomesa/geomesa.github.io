24 class DtgPathFiltering(attribute: String, pattern: Pattern, format: DateTimeFormatter, buffer: Duration)
25     extends PathFiltering with LazyLogging {
26 
27   def apply(filter: Filter): PathFilter = {
28     val filterIntervals = FilterHelper.extractIntervals(filter, attribute, handleExclusiveBounds = true)
29     path => try {
30       val time = parseDtg(path.getName).toInstant
31       val millis = buffer.toMillis
32       val lower = ZonedDateTime.ofInstant(time.minusMillis(millis), ZoneOffset.UTC)
33       val upper = ZonedDateTime.ofInstant(time.plusMillis(millis), ZoneOffset.UTC)
34       val buffered = Bounds(Bound.inclusive(lower), Bound.inclusive(upper))
35       val included = filterIntervals.exists(bounds => bounds.intersects(buffered))
36       logger.whenDebugEnabled {
37         if (included) {
38           logger.debug(s"Including path ${path.getName} for filter $filter")
39         } else {
40           logger.debug(s"Excluding path ${path.getName} for filter $filter")
41         }
42       }
43       included
44     } catch {
45       case NonFatal(ex) =>
46         logger.warn(s"Failed to evaluate filter for path '${path.getName}'", ex)
47         true
48     }
49   }
50 
51   private def parseDtg(name: String): ZonedDateTime = {
52     Option(name)
53       .map(pattern.matcher)
54       .filter(_.matches)
55       .filter(_.groupCount > 0)
56       .map(_.group(1))
57       .map(ZonedDateTime.parse(_, format))
58       .getOrElse {
59         throw new IllegalArgumentException(s"Failed to parse ${classOf[ZonedDateTime].getName} " +
60           s"from file name '$name' for pattern '$pattern' and format '$format'")
61       }
62   }
63 
64   override def toString: String = {
65     s"${this.getClass.getName}(attribute = $attribute, pattern = $pattern, format = $format, buffer = $buffer)"
66   }
