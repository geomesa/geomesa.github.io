19 case class CompositeScheme(schemes: Seq[PartitionScheme]) extends PartitionScheme {
20 
21   import org.locationtech.geomesa.filter.andFilters
22 
23   require(schemes.lengthCompare(1) > 0, "Must provide at least 2 schemes for a composite scheme")
24 
25   override val depth: Int = schemes.map(_.depth).sum
26 
27   override def pattern: String = schemes.map(_.pattern).mkString("/")
28 
29   override def getPartitionName(feature: SimpleFeature): String =
30     schemes.map(_.getPartitionName(feature)).mkString("/")
31 
32   override def getSimplifiedFilters(filter: Filter, partition: Option[String]): Option[Seq[SimplifiedFilter]] = {
33     var i = 0
34 
35     // get the partial path for our current partition scheme
36     def slice: Option[String] = partition.map(subPartition(_, i))
37 
38     schemes.head.getSimplifiedFilters(filter, slice).map { head =>
39       schemes.tail.foldLeft(head) { (results, scheme) =>
40         i += 1
41         results.flatMap { result =>
42           if (result.partial) {
43             // we haven't matched one of our tiers, so we can't keep building up partition paths
44             Seq(result)
45           } else {
46             val tiers = scheme.getSimplifiedFilters(result.filter, slice).orNull
47             if (tiers == null) {
48               Seq(SimplifiedFilter(result.filter, result.partitions, partial = true))
49             } else {
50               val prefixes = result.partitions
51               tiers.map { tier =>
52                 val paths = prefixes.flatMap(prefix => tier.partitions.map(t => s"$prefix/$t"))
53                 SimplifiedFilter(tier.filter, paths, partial = false)
54               }
55             }
56           }
57         }
58       }
59     }
60   }
61 
62   override def getIntersectingPartitions(filter: Filter): Option[Seq[String]] = {
63     val head = schemes.head.getIntersectingPartitions(filter)
64     schemes.tail.foldLeft(head) { case (result, scheme) =>
65       for { res <- result; partitions <- scheme.getIntersectingPartitions(filter) } yield {
66         res.flatMap(r => partitions.map(p => s"$r/$p"))
67       }
68     }
69   }
70 
71   override def getCoveringFilter(partition: String): Filter = {
72     val filters = schemes.zipWithIndex.map { case (scheme, i) =>
73       scheme.getCoveringFilter(subPartition(partition, i))
74     }
75     FilterHelper.flatten(andFilters(filters))
76   }
77 
78   /**
79    * Get the partition corresponding to one of the components of this composite scheme
80    *
81    * @param partition full composite partition name
82    * @param scheme the composite scheme to extract the partition for
83    * @return
84    */
85   private def subPartition(partition: String, scheme: Int): String = {
86     val splits = partition.split("/")
87     val from = schemes.take(scheme).map(_.depth).sum
88     val to = from + schemes(scheme).depth
89     splits.slice(from, to).mkString("/")
90   }
91 }

103         } catch {
104           case NonFatal(_) => None
105         }
106       }
107     }
