20 case class CompositeScheme(schemes: Seq[PartitionScheme]) extends PartitionScheme {
21 
22   import org.locationtech.geomesa.filter.andFilters
23 
24   require(schemes.lengthCompare(1) > 0, "Must provide at least 2 schemes for a composite scheme")
25 
26   override val depth: Int = schemes.map(_.depth).sum
27 
28   override def pattern: String = schemes.map(_.pattern).mkString("/")
29 
30   override def getPartitionName(feature: SimpleFeature): String =
31     schemes.map(_.getPartitionName(feature)).mkString("/")
32 
33   override def getSimplifiedFilters(filter: Filter, partition: Option[String]): Option[Seq[SimplifiedFilter]] = {
34     var i = 0
35 
36     // get the partial path for our current partition scheme
37     def slice: Option[String] = partition.map(subPartition(_, i))
38 
39     schemes.head.getSimplifiedFilters(filter, slice).map { head =>
40       schemes.tail.foldLeft(head) { (results, scheme) =>
41         i += 1
42         results.flatMap { result =>
43           if (result.partial) {
44             // we haven't matched one of our tiers, so we can't keep building up partition paths
45             Seq(result)
46           } else {
47             val tiers = scheme.getSimplifiedFilters(result.filter, slice).orNull
48             if (tiers == null) {
49               Seq(SimplifiedFilter(result.filter, result.partitions, partial = true))
50             } else {
51               val prefixes = result.partitions
52               tiers.map { tier =>
53                 val paths = prefixes.flatMap(prefix => tier.partitions.map(t => s"$prefix/$t"))
54                 SimplifiedFilter(tier.filter, paths, partial = false)
55               }
56             }
57           }
58         }
59       }
60     }
61   }
62 
63   override def getIntersectingPartitions(filter: Filter): Option[Seq[String]] = {
64     val head = schemes.head.getIntersectingPartitions(filter)
65     schemes.tail.foldLeft(head) { case (result, scheme) =>
66       for { res <- result; partitions <- scheme.getIntersectingPartitions(filter) } yield {
67         res.flatMap(r => partitions.map(p => s"$r/$p"))
68       }
69     }
70   }
71 
72   override def getCoveringFilter(partition: String): Filter = {
73     val filters = schemes.zipWithIndex.map { case (scheme, i) =>
74       scheme.getCoveringFilter(subPartition(partition, i))
75     }
76     FilterHelper.flatten(andFilters(filters))
77   }
78 
79   /**
80    * Get the partition corresponding to one of the components of this composite scheme
81    *
82    * @param partition full composite partition name
83    * @param scheme the composite scheme to extract the partition for
84    * @return
85    */
86   private def subPartition(partition: String, scheme: Int): String = {
87     val splits = partition.split("/")
88     val from = schemes.take(scheme).map(_.depth).sum
89     val to = from + schemes(scheme).depth
90     splits.slice(from, to).mkString("/")
91   }
92 }

100       if (!config.name.contains(SchemeSeparator)) { None } else {
101         val configs = config.name.split(SchemeSeparator).map(n => config.copy(name = n))
102         val composites = configs.map { c =>
103           try { PartitionSchemeFactory.load(sft, c) } catch {
104             case NonFatal(e) =>
105               throw new IllegalArgumentException(s"Error creating composite scheme '${c.name}' (from '${config.name}'):", e)
106           }
107         }
108         Some(CompositeScheme(composites))
109       }
110     }
111   }
