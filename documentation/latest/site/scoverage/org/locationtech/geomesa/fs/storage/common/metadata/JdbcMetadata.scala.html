25 /**
26   * Storage metadata implementation backed by a SQL database. Currently tested with H2 and Postgres - other
27   * databases may have incompatibilities in the SQL syntax.
28   *
29   * Scheme consists of three tables and one seqeuence:
30   *
31   * `storage_meta`
32   *
33   * * Holds the base metadata (simple feature type, partition scheme, encoding, leaf storage) as a JSON clob
34   *
35   * ** root varchar(256) not null
36   * ** value text not null
37   * ** primary key (root)
38   *
39   * `storage_partitions_id_seq`
40   *
41   * * Sequence for partition action IDs
42   *
43   * `storage_partitions`
44   *
45   * * Holds add/remove partition actions. Tracks the type, count (features) and bounds of each action
46   *
47   * ** root varchar(256) not null
48   * ** name varchar(256) not null
49   * ** id int not null
50   * ** action char(1) not null
51   * ** features bigint
52   * ** bounds_xmin double precision
53   * ** bounds_xmax double precision
54   * ** bounds_ymin double precision
55   * ** bounds_ymax double precision
56   * ** primary key (root, name, id)
57   *
58   * `storage_partition_files`
59   *
60   * * Holds the files associated with each partition action
61   *
62   * ** root varchar(256) not null
63   * ** name varchar(256) not null
64   * ** id int not null - foreign key to `storage_partitions`
65   * ** file varchar(256) not null
66   * ** primary key (root, name, id, file)
67   *
68   * @param pool connection pool
69   * @param root storage root path
70   * @param sft simple feature type
71   * @param meta basic metadata config
72   **/

339     private def write(connection: Connection, root: String, action: Char, partition: PartitionMetadata): Unit = {
340       // TODO insert as a single transaction
341       val id = WithClose(connection.prepareStatement(NextIdStatement)) { statement =>
342         WithClose(statement.executeQuery()) { rs => rs.next(); rs.getInt(1) }
343       }
344       WithClose(connection.prepareStatement(InsertStatement)) { statement =>
345         statement.setString(1, root)
346         statement.setString(2, partition.name)
347         statement.setInt(3, id)
348         statement.setString(4, action.toString)
349         statement.setLong(5, partition.count)
350         partition.bounds match {
351           case Some(b) =>
352             statement.setDouble(6, b.xmin)
353             statement.setDouble(7, b.ymin)
354             statement.setDouble(8, b.xmax)
355             statement.setDouble(9, b.ymax)
356 
357           case None =>
358             statement.setNull(6, java.sql.Types.DOUBLE)
359             statement.setNull(7, java.sql.Types.DOUBLE)
360             statement.setNull(8, java.sql.Types.DOUBLE)
361             statement.setNull(9, java.sql.Types.DOUBLE)
362         }
363         statement.executeUpdate()
364       }
365       FilesTable.insert(connection, root, partition.name, id, partition.files)
366     }

368     private def readConfigs(results: ResultSet): Seq[PartitionConfig] = {
369       if (!results.next()) {
370         return Seq.empty
371       }
372 
373       val partitions = Seq.newBuilder[PartitionConfig]
374 
375       var partition = results.partition()
376       var files = Seq.newBuilder[StorageFile] += results.file()
377 
378       while (results.next()) {
379         if (results.name() == partition.name && results.id() == partition.timestamp) {
380           files += results.file()
381         } else {
382           partitions += partition.copy(files = files.result)
383           partition = results.partition()
384           files = Seq.newBuilder[StorageFile] += results.file()
385         }
386       }
387 
388       partitions += partition.copy(files = files.result)
389       partitions.result
390     }

543     def updateSchema(connection: Connection): Unit = {
544       WithClose(connection.createStatement()) { statement =>
545         val cols = WithClose(statement.executeQuery(s"select * from $TableName limit 1")) { results =>
546           results.getMetaData.getColumnCount
547         }
548         def addTypeAndTime(): Unit = {
549           statement.executeUpdate(s"alter table $TableName add column $TypeCol char(1)")
550           statement.executeUpdate(s"alter table $TableName add column $TimeCol bigint")
551           statement.executeUpdate(s"update $TableName set $TypeCol = 'a', $TimeCol = 0")
552           statement.executeUpdate(s"alter table $TableName alter column $TypeCol set not null")
553         }
554         def addSortAndBounds(): Unit = {
555           statement.executeUpdate(s"alter table $TableName add column $SortCol varchar(256)")
556           statement.executeUpdate(s"alter table $TableName add column $BoundsCol varchar(256)")
557         }
558         if (cols == 4) {
559           addTypeAndTime()
560           addSortAndBounds()
561         } else if (cols == 6) {
562           addSortAndBounds()
563         } else if (cols != 8) {
564           throw new IllegalStateException(s"Unexpected schema detected for table $TableName: " +
