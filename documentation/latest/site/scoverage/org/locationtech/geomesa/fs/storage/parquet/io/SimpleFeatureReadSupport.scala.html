67   /**
68     * Zip x and y values into coordinates
69     *
70     * @param x x values
71     * @param y corresponding y values
72     * @return
73     */
74   def zip(x: Array[Double], y: Array[Double]): Array[Coordinate] = {
75     val result = Array.ofDim[Coordinate](x.length)
76     var i = 0
77     while (i < result.length) {
78       result(i) = new Coordinate(x(i), y(i))
79       i += 1
80     }
81     result
82   }

92     * Group converter that can create simple features
93     */
94   private class SimpleFeatureGroupConverter(schema: SimpleFeatureParquetSchema)
95       extends GroupConverter with ValueMaterializer[SimpleFeature] {
96 
97     private val idConverter = new StringConverter()
98 
99     private val visConverter = new StringConverter() {
100       override def materialize(): String = {
101         super.materialize() match {
102           case null => null
103           case s: String => s.intern()
104         }
105       }
106     }
107 
108     private val converters = {
109       val builder = Array.newBuilder[ValueMaterializer[_ <: AnyRef]]
110       var i = 0
111       while (i < schema.sft.getAttributeCount) {
112         builder += attribute(ObjectType.selectType(schema.sft.getDescriptor(i)), schema.encodings)
113         i += 1
114       }
115       builder += idConverter
116       if (schema.hasVisibilities) {
117         builder += visConverter
118       }
119       // note: we read bounding boxes since they're present, but we don't do anything with the result
120       schema.boundingBoxes.foreach(_ => builder += new BoundingBoxConverter())
121       builder.result()
122     }
123 
124     override def reset(): Unit = start()
125 
126     override def materialize(): SimpleFeature = {
127       val id = idConverter.materialize()
128       val vis = visConverter.materialize()
129       val values = Array.tabulate[AnyRef](schema.sft.getAttributeCount)(i => converters(i).materialize())
130       val userData = if (vis == null) { null } else {
131         val map = new java.util.HashMap[AnyRef, AnyRef](1)
132         map.put("geomesa.feature.visibility", vis)
133         map
134       }
135       new ScalaSimpleFeature(schema.sft, id, values, userData)
136     }
137 
138     override def start(): Unit = converters.foreach(_.reset())
139 
140     override def getConverter(fieldIndex: Int): Converter = converters(fieldIndex)
141 
142     override def end(): Unit = {}
143   }

145   /**
146    * Gets a reader for a simple feature attribute
147    *
148    * @param bindings bindings
149    * @param encodings schema encodings
150    * @return
151    */
152   def attribute(bindings: Seq[ObjectType], encodings: Encodings): ValueMaterializer[_ <: AnyRef] = {
153     bindings.head match {
154       case ObjectType.GEOMETRY => geometry(bindings.last, encodings.geometry)
155       case ObjectType.DATE     => date(encodings.date)
156       case ObjectType.STRING   => new StringConverter()
157       case ObjectType.INT      => new IntConverter()
158       case ObjectType.DOUBLE   => new DoubleConverter()
159       case ObjectType.LONG     => new LongConverter()
160       case ObjectType.FLOAT    => new FloatConverter()
161       case ObjectType.BOOLEAN  => new BooleanConverter()
162       case ObjectType.BYTES    => new BytesConverter()
163       case ObjectType.LIST     => list(attribute(bindings.drop(1), encodings), encodings.list)
164       case ObjectType.MAP      => new MapConverter(attribute(bindings.slice(1, 2), encodings), attribute(bindings.slice(2, 3), encodings))
165       case ObjectType.UUID     => new UuidConverter()
166       case _ => throw new IllegalArgumentException(s"Can't deserialize field of type ${bindings.head}")
167     }
168   }
169 
170   private def geometry(binding: ObjectType, encoding: GeometryEncoding): ValueMaterializer[_ <: Geometry] = {
171     if (encoding == GeometryEncoding.GeoParquetWkb) {
172       new WkbConverter()
173     } else if (encoding == GeometryEncoding.GeoParquetNative) {
174       binding match {
175         case ObjectType.POINT           => new PointConverter()
176         case ObjectType.LINESTRING      => new GeoParquetNativeLineStringConverter()
177         case ObjectType.POLYGON         => new GeoParquetNativePolygonConverter()
178         case ObjectType.MULTIPOINT      => new GeoParquetNativeMultiPointConverter()
179         case ObjectType.MULTILINESTRING => new GeoParquetNativeMultiLineStringConverter()
180         case ObjectType.MULTIPOLYGON    => new GeoParquetNativeMultiPolygonConverter()
181         case _                          => new WkbConverter()
182       }
183     } else {
184       // geomesa v1+v0
185       binding match {
186         case ObjectType.POINT           => new PointConverter()
187         case ObjectType.LINESTRING      => new LineStringConverter()
188         case ObjectType.POLYGON         => new PolygonConverter()
189         case ObjectType.MULTIPOINT      => new MultiPointConverter()
190         case ObjectType.MULTILINESTRING => new MultiLineStringConverter()
191         case ObjectType.MULTIPOLYGON    => new MultiPolygonConverter()
192         case _                          => new WkbConverter()
193       }
194     }
195   }
