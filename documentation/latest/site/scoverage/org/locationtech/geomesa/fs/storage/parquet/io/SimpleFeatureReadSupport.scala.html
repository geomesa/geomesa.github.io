68   /**
69     * Zip x and y values into coordinates
70     *
71     * @param x x values
72     * @param y corresponding y values
73     * @return
74     */
75   def zip(x: Array[Double], y: Array[Double]): Array[Coordinate] = {
76     val result = Array.ofDim[Coordinate](x.length)
77     var i = 0
78     while (i < result.length) {
79       result(i) = new Coordinate(x(i), y(i))
80       i += 1
81     }
82     result
83   }

93     * Group converter that can create simple features
94     */
95   private class SimpleFeatureGroupConverter(schema: SimpleFeatureParquetSchema)
96       extends GroupConverter with ValueMaterializer[SimpleFeature] {
97 
98     private val idConverter = new StringConverter()
99 
100     private val visConverter = new StringConverter() {
101       override def materialize(): String = {
102         super.materialize() match {
103           case null => null
104           case s: String => s.intern()
105         }
106       }
107     }
108 
109     private val converters = {
110       val builder = Array.newBuilder[ValueMaterializer[_ <: AnyRef]]
111       var i = 0
112       while (i < schema.sft.getAttributeCount) {
113         builder += attribute(ObjectType.selectType(schema.sft.getDescriptor(i)), schema.encodings)
114         i += 1
115       }
116       builder += idConverter
117       if (schema.hasVisibilities) {
118         builder += visConverter
119       }
120       // note: we read bounding boxes since they're present, but we don't do anything with the result
121       schema.boundingBoxes.foreach(_ => builder += new BoundingBoxConverter())
122       builder.result()
123     }
124 
125     override def reset(): Unit = start()
126 
127     override def materialize(): SimpleFeature = {
128       val id = idConverter.materialize()
129       val vis = visConverter.materialize()
130       val values = Array.tabulate[AnyRef](schema.sft.getAttributeCount)(i => converters(i).materialize())
131       val userData = if (vis == null) { null } else {
132         val map = new java.util.HashMap[AnyRef, AnyRef](1)
133         map.put(SecurityUtils.FEATURE_VISIBILITY, vis)
134         map
135       }
136       new ScalaSimpleFeature(schema.sft, id, values, userData)
137     }
138 
139     override def start(): Unit = converters.foreach(_.reset())
140 
141     override def getConverter(fieldIndex: Int): Converter = converters(fieldIndex)
142 
143     override def end(): Unit = {}
144   }

146   /**
147    * Gets a reader for a simple feature attribute
148    *
149    * @param bindings bindings
150    * @param encodings schema encodings
151    * @return
152    */
153   def attribute(bindings: Seq[ObjectType], encodings: Encodings): ValueMaterializer[_ <: AnyRef] = {
154     bindings.head match {
155       case ObjectType.GEOMETRY => geometry(bindings.last, encodings.geometry)
156       case ObjectType.DATE     => date(encodings.date)
157       case ObjectType.STRING   => new StringConverter()
158       case ObjectType.INT      => new IntConverter()
159       case ObjectType.DOUBLE   => new DoubleConverter()
160       case ObjectType.LONG     => new LongConverter()
161       case ObjectType.FLOAT    => new FloatConverter()
162       case ObjectType.BOOLEAN  => new BooleanConverter()
163       case ObjectType.BYTES    => new BytesConverter()
164       case ObjectType.LIST     => list(attribute(bindings.drop(1), encodings), encodings.list)
165       case ObjectType.MAP      => new MapConverter(attribute(bindings.slice(1, 2), encodings), attribute(bindings.slice(2, 3), encodings))
166       case ObjectType.UUID     => new UuidConverter()
167       case _ => throw new IllegalArgumentException(s"Can't deserialize field of type ${bindings.head}")
168     }
169   }
170 
171   private def geometry(binding: ObjectType, encoding: GeometryEncoding): ValueMaterializer[_ <: Geometry] = {
172     if (encoding == GeometryEncoding.GeoParquetWkb) {
173       new WkbConverter()
174     } else if (encoding == GeometryEncoding.GeoParquetNative) {
175       binding match {
176         case ObjectType.POINT           => new PointConverter()
177         case ObjectType.LINESTRING      => new GeoParquetNativeLineStringConverter()
178         case ObjectType.POLYGON         => new GeoParquetNativePolygonConverter()
179         case ObjectType.MULTIPOINT      => new GeoParquetNativeMultiPointConverter()
180         case ObjectType.MULTILINESTRING => new GeoParquetNativeMultiLineStringConverter()
181         case ObjectType.MULTIPOLYGON    => new GeoParquetNativeMultiPolygonConverter()
182         case _                          => new WkbConverter()
183       }
184     } else {
185       // geomesa v1+v0
186       binding match {
187         case ObjectType.POINT           => new PointConverter()
188         case ObjectType.LINESTRING      => new LineStringConverter()
189         case ObjectType.POLYGON         => new PolygonConverter()
190         case ObjectType.MULTIPOINT      => new MultiPointConverter()
191         case ObjectType.MULTILINESTRING => new MultiLineStringConverter()
192         case ObjectType.MULTIPOLYGON    => new MultiPolygonConverter()
193         case _                          => new WkbConverter()
194       }
195     }
196   }
