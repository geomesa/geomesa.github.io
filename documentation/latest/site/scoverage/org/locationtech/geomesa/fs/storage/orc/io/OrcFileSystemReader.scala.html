26 class OrcFileSystemReader(
27     sft: SimpleFeatureType,
28     config: Configuration,
29     filter: Option[Filter],
30     transform: Option[(String, SimpleFeatureType)]
31   ) extends FileSystemPathReader {
32 
33   private val (options, columns) = {
34     val options = new Reader.Options(config)
35     val readOptions = OrcFileSystemReader.readOptions(sft, filter, transform)
36     readOptions.columns.foreach(c => options.include(OrcFileSystemReader.include(sft, c)))
37     // note: push down will exclude whole record batches, but doesn't actually filter inside a batch
38     readOptions.pushDown.foreach { case (sarg, cols) => options.searchArgument(sarg, cols) }
39     (options, readOptions.columns)
40   }
41 
42   override def read(path: Path): CloseableIterator[SimpleFeature] = new PathReader(path)
43 
44   private class PathReader(file: Path) extends CloseableIterator[SimpleFeature] {
45     private val feature = new ScalaSimpleFeature(sft, "")
46     private val transformed = transform.map { case (tdefs, tsft) =>
47       val sf = TransformSimpleFeature(sft, tsft, tdefs)
48       sf.setFeature(feature)
49       sf
50     }
51     private val result = transformed.getOrElse(feature)
52 
53     private val reader = OrcFile.createReader(file, OrcFile.readerOptions(config).useUTCTimestamp(true))
54     private val rows = reader.rows(options)
55     private val batch = reader.getSchema.createRowBatch()
56     private val attributeReader = if (batch.cols.length > 0) { OrcAttributeReader(sft, batch, columns) } else { null }
57 
58     private var staged: Boolean = false
59     private var i = 0
60 
61     override def hasNext: Boolean = {
62       if (staged) { true } else {
63         stageNext()
64         staged
65       }
66     }
67 
68     /**
69       * Note: same feature is mutated and returned for each call to .next
70       *
71       * @return
72       */
73     override def next(): SimpleFeature = {
74       if (staged || hasNext) {
75         staged = false
76         result
77       } else {
78         Iterator.empty.next
79       }
80     }
81 
82     override def close(): Unit = rows.close()
83 
84     private def stageNext(): Unit = {
85       var loop = true
86       while (loop) {
87         if (i < batch.size) {
88           attributeReader.apply(feature, i)
89           i += 1
90           if (OrcFileSystemReader.this.filter.forall(_.evaluate(feature))) {
91             staged = true
92             loop = false
93           }
94         } else {
95           i = 0
96           loop = rows.nextBatch(batch)
97         }
98       }
99     }
100   }
101 }

107   /**
108     * Create low-level reading options used by ORC
109     *
110     * @param sft simple feature type
111     * @param filter ecql filter
112     * @param transform transform
113     * @return
114     */
115   def readOptions(
116       sft: SimpleFeatureType,
117       filter: Option[Filter],
118       transform: Option[(String, SimpleFeatureType)]): OrcReadOptions = {
119     val columns = transform.map { case (tdefs, _) =>
120       val fromFilter = filter.map(FilterHelper.propertyNames(_, sft)).getOrElse(Seq.empty)
121       val fromTransform = Transforms(sft, tdefs).flatMap(_.properties)
122       (fromFilter ++ fromTransform).toSet[String].map(sft.indexOf)
123     }
124     // push down will exclude whole record batches, but doesn't actually filter inside a batch
125     val pushDown = filter.flatMap { f =>
126       OrcSearchArguments(sft, SimpleFeatureTypeDescription(sft), f)
127     }
128     OrcReadOptions(columns, pushDown)
129   }
130 
131   /**
132     * Construct the array used to include columns corresponding to the simple feature transform.
133     *
134     * Included columns don't just align with the top-level fields - complex fields need to have additional
135     * entries in the array for their children.
136     *
137     * @param sft simple feature type
138     * @param columns attributes to read
139     * @param fid read fid
140     * @return
141     */
142   def include(sft: SimpleFeatureType, columns: Set[Int], fid: Boolean = true): Array[Boolean] = {
143     val buffer = ArrayBuffer[Boolean](true) // outer struct
144 
145     var i = 0
146     while (i < sft.getAttributeCount) {
147       val bindings = ObjectType.selectType(sft.getDescriptor(i))
148       val count = bindings.head match {
149         case ObjectType.GEOMETRY =>
150           bindings(1) match {
151             case ObjectType.POINT => 2 // x + y
152             case ObjectType.LINESTRING | ObjectType.MULTIPOINT => 4 // list of x, list of y
153             case ObjectType.POLYGON | ObjectType.MULTILINESTRING => 6 // list of list of x, list of list of y
154             case ObjectType.MULTIPOLYGON => 8 // list of list of list of x, list of list of list of y
155             case ObjectType.GEOMETRY => 1 // wkb
156           }
157         case ObjectType.LIST => 2
158         case ObjectType.MAP => 3
159         case _ => 1
160       }
161       val include = columns.contains(i)
162       var j = 0
163       while (j < count) {
164         buffer += include
165         j += 1
166       }
167       i += 1
168     }
169 
170     buffer += fid
171 
172     buffer.toArray
173   }
