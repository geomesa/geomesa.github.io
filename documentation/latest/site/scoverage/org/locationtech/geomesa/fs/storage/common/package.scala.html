33     /**
34       * Serialize configuration options as a typesafe config string
35       *
36       * @param options options
37       * @return
38       */
39     def serialize(options: NamedOptions): String = NamedOptionsConvert.to(options).render(RenderOptions)
40 
41     /**
42       * Deserialize configuration options, e.g. for partition schemes and metadata connections
43       *
44       * @param options options as a typesafe config string
45       * @return
46       */
47     def deserialize(options: String): NamedOptions = {
48       val config = ConfigFactory.parseString(options, ParseOptions)
49       try { ConfigSource.fromConfig(config).loadOrThrow[NamedOptions] } catch {
50         case NonFatal(e) => Try(deserializeOldScheme(config)).getOrElse(throw e)
51       }
52     }
53 
54     private def deserializeOldScheme(config: Config): NamedOptions = {
55       val parsed = ConfigSource.fromConfig(config).loadOrThrow[PartitionSchemeConfig]
56       NamedOptions(parsed.scheme, parsed.options)
57     }
58   }

88     def removeScheme(): Option[NamedOptions] =
89       remove(SchemeKey).map(deserialize).orElse(remove("geomesa.fs.partition-scheme.config").map(deserialize))
90 
91     def setMetadata(name: String, options: Map[String, String] = Map.empty): Unit =
92       sft.getUserData.put(MetadataKey, serialize(NamedOptions(name, options)))
93     def removeMetadata(): Option[NamedOptions] = remove(MetadataKey).map(deserialize)
94 
95     def setTargetFileSize(size: String): Unit = {
96       // validate input
97       Memory.bytes(size).failed.foreach(e => throw new IllegalArgumentException("Invalid file size", e))
98       sft.getUserData.put(FileSizeKey, size)
99     }
100     def removeTargetFileSize(): Option[Long] = {
101       remove(FileSizeKey).map { s =>
102         Memory.bytes(s) match {
103           case Success(b) => b
104           case Failure(e) => throw new IllegalArgumentException("Invalid file size", e)
105         }
106       }
107     }
108 
109     def setObservers(names: Seq[String]): Unit = sft.getUserData.put(ObserversKey, names.mkString(","))
110     def getObservers: Seq[String] = {
111       val obs = sft.getUserData.get(ObserversKey).asInstanceOf[String]
112       if (obs == null || obs.isEmpty) { Seq.empty } else { obs.split(",") }
113     }
114 
115     private def remove(key: String): Option[String] = Option(sft.getUserData.remove(key).asInstanceOf[String])
116   }
