25   /**
26    * Gets a count of the Orc fields created for a simple feature type.
27    *
28    * Geometry attributes create 2 columns, other attributes create 1, feature id creates 1
29    *
30    * @param sft simple feature type
31    * @param fid include feature ids or not
32    * @return
33    */
34   def fieldCount(sft: SimpleFeatureType, fid: Boolean = true): Int = {
35     val attributes = sft.getAttributeDescriptors.asScala.foldLeft(0) { case (sum, d) => sum + fieldCount(d) }
36     if (fid) { attributes + 1 } else { attributes }
37   }
38 
39   /**
40    * Gets a count of the Orc fields created for a given attribute
41    *
42    * @param descriptor descriptor
43    * @return
44    */
45   def fieldCount(descriptor: AttributeDescriptor): Int = {
46     descriptor.getType.getBinding match {
47       // plain Geometry bindings are encoded in a single WKB column, others as x + y
48       case b if classOf[Geometry].isAssignableFrom(b) && b != classOf[Geometry] => 2
49       case _ => 1
50     }
51   }

53   /**
54    * Create the Orc type description corresponding to the SimpleFeatureType. SimpleFeatureType is
55    * modeled as an Orc Struct, with nested fields for each attribute.
56    *
57    * @param sft simple feature type
58    * @return
59    */
60   def apply(sft: SimpleFeatureType, fid: Boolean = true): TypeDescription = {
61     val container = TypeDescription.createStruct()
62     var i = 0
63     while (i < sft.getAttributeCount) {
64       addTypeDescription(container, sft.getDescriptor(i))
65       i += 1
66     }
67     if (fid) {
68       container.addField("id", TypeDescription.createString())
69     }
70     container
71   }
72 
73   /**
74    * Add a type description for an attribute
75    *
76    * @param container top-level Orc struct corresponding to the SimpleFeatureType
77    * @param descriptor descriptor
78    */
79   private def addTypeDescription(container: TypeDescription, descriptor: AttributeDescriptor): Unit = {
80     val name = descriptor.getLocalName
81     val bindings = ObjectType.selectType(descriptor)
82     bindings.head match {
83       case ObjectType.GEOMETRY => addGeometryDescription(container, name, bindings(1))
84       case ObjectType.LIST     => container.addField(name, TypeDescription.createList(simple(bindings(1))))
85       case ObjectType.MAP      => container.addField(name, TypeDescription.createMap(simple(bindings(1)), simple(bindings(2))))
86       case binding             => container.addField(name, simple(binding))
87     }
88   }
89 
90   /**
91    * We create two separate fields, one for x-values and one for y-values. Orc doesn't support
92    * predicate push-down on nested fields, so we flatten them out. Note that Orc also doesn't
93    * support predicate push-down for complex fields, so we actually only benefit for Points.
94    */
95   private def addGeometryDescription(container: TypeDescription, name: String, binding: ObjectType): Unit = {
96     import TypeDescription.{createDouble, createList}
97 
98     def x: String = geometryXField(name)
99     def y: String = geometryYField(name)
100 
101     binding match {
102       case ObjectType.POINT =>
103         // point (x, y) pair
104         container.addField(x, createDouble())
105         container.addField(y, createDouble())
106 
107       case ObjectType.LINESTRING =>
108         // list of points
109         container.addField(x, createList(createDouble()))
110         container.addField(y, createList(createDouble()))
111 
112       case ObjectType.MULTIPOINT =>
113         // list of points
114         container.addField(x, createList(createDouble()))
115         container.addField(y, createList(createDouble()))
116 
117       case ObjectType.POLYGON =>
118         // list of lines (exterior ring + any holes)
119         container.addField(x, createList(createList(createDouble())))
120         container.addField(y, createList(createList(createDouble())))
121 
122       case ObjectType.MULTILINESTRING =>
123         // list of lines
124         container.addField(x, createList(createList(createDouble())))
125         container.addField(y, createList(createList(createDouble())))
126 
127       case ObjectType.MULTIPOLYGON =>
128         // list of polygons
129         container.addField(x, createList(createList(createList(createDouble()))))
130         container.addField(y, createList(createList(createList(createDouble()))))
131 
132       case ObjectType.GEOMETRY =>
133         // WKB
134         container.addField(name, TypeDescription.createBinary())
135 
136       case _ => throw new IllegalArgumentException(s"Unexpected geometry type $binding")
137     }
138   }
139 
140   /**
141    * Type description for simple types, e.g. int, float, date, etc
142    *
143    * @param binding type binding
144    * @return
145    */
146   private def simple(binding: ObjectType): TypeDescription = {
147     binding match {
148       case ObjectType.DATE    => TypeDescription.createTimestamp()
149       case ObjectType.STRING  => TypeDescription.createString()
150       case ObjectType.INT     => TypeDescription.createInt()
151       case ObjectType.LONG    => TypeDescription.createLong()
152       case ObjectType.FLOAT   => TypeDescription.createFloat()
153       case ObjectType.DOUBLE  => TypeDescription.createDouble()
154       case ObjectType.BOOLEAN => TypeDescription.createBoolean()
155       case ObjectType.BYTES   => TypeDescription.createBinary()
156       case ObjectType.UUID    => TypeDescription.createString()
157       case _ => throw new IllegalArgumentException(s"Unexpected simple object type $binding")
158     }
159   }
