62   import org.locationtech.geomesa.index.conf.QueryHints._
63 
64   /**
65     * Configure the input format
66     *
67     * @param conf conf
68     * @param sft simple feature type
69     * @param query query
70     */
71   def configure(conf: Configuration, sft: SimpleFeatureType, query: Query): Unit = {
72     val q = QueryRunner.configureDefaultQuery(sft, query)
73     val filter = Option(q.getFilter).filter(_ != Filter.INCLUDE)
74 
75     // Parquet read schema and final transform
76     val ReadSchema(readSft, readTransform) = ReadSchema(sft, filter, q.getHints.getTransform)
77     // push-down Parquet predicates and remaining gt-filter
78     val ReadFilter(parquetFilter, residualFilter) = ReadFilter(readSft, filter)
79 
80     // set the parquet push-down filter
81     parquetFilter.foreach(ParquetInputFormat.setFilterPredicate(conf, _))
82 
83     // set our residual filters and transforms
84     StorageConfiguration.setSft(conf, readSft)
85     residualFilter.foreach(StorageConfiguration.setFilter(conf, _))
86     readTransform.foreach(StorageConfiguration.setTransforms(conf, _))
87 
88     // need this for query planning
89     conf.set("parquet.filter.dictionary.enabled", "true")
90 
91     // @see org.apache.parquet.hadoop.ParquetInputFormat.setReadSupportClass(org.apache.hadoop.mapred.JobConf, java.lang.Class<?>)
92     conf.set(ParquetInputFormat.READ_SUPPORT_CLASS, classOf[SimpleFeatureReadSupport].getName)
93 
94     // replicates parquet input format strategy of always recursively listing directories
95     conf.set(FileInputFormat.INPUT_DIR_RECURSIVE, "true")
96   }

129   abstract class ParquetSimpleFeatureRecordReaderBase[T](
130       delegate: RecordReader[Void, SimpleFeature],
131       filter: Option[Filter]
132     ) extends RecordReader[T, SimpleFeature] {
133 
134     private var current: SimpleFeature = _
135 
136     override def initialize(split: InputSplit, context: TaskAttemptContext): Unit =
137       delegate.initialize(split, context)
138 
139     override def getProgress: Float = delegate.getProgress
140 
141     override def nextKeyValue(): Boolean = {
142       while (delegate.nextKeyValue()) {
143         current = delegate.getCurrentValue
144         if (filter.forall(_.evaluate(current))) {
145            return true
146         }
147       }
148       false
149     }
150 
151     override def getCurrentValue: SimpleFeature = current
152 
153     override def close(): Unit = delegate.close()
154   }

166     override def getCurrentValue: SimpleFeature = {
167       current.setFeature(super.getCurrentValue)
168       current
169     }
