52     def compact(ds: FileSystemDataStore): Unit = {
53       Command.user.info("Beginning compaction process...")
54 
55       val storage = ds.storage(params.featureName)
56 
57       val toCompact = if (params.partitions.isEmpty) { storage.getPartitions } else {
58         val filtered = params.partitions.asScala.flatMap(storage.metadata.getPartition)
59         if (filtered.lengthCompare(params.partitions.size()) != 0) {
60           val unmatched = params.partitions.asScala.filterNot(name => filtered.exists(_.name == name))
61           throw new ParameterException(s"Partition(s) ${unmatched.mkString(", ")} cannot be found in metadata")
62         }
63         filtered
64       }
65 
66       val mode = params.mode.getOrElse {
67         if (PathUtils.isRemote(storage.context.root.toString)) { RunModes.Distributed } else { RunModes.Local }
68       }
69       val fileSize = Option(params.targetFileSize).map(_.longValue)
70 
71       Command.user.info(s"Compacting ${toCompact.size} partitions in ${mode.toString.toLowerCase(Locale.US)} mode")
72 
73       val start = System.currentTimeMillis()
74       val status = TerminalCallback()
75 
76       mode match {
77         case RunModes.Local =>
78           val total = toCompact.length
79           val latch = new CountDownLatch(total)
80           val executor = Executors.newFixedThreadPool(math.max(1, math.min(params.threads, total)))
81 
82           try {
83             toCompact.foreach { p =>
84               executor.submit(
85                 new Runnable() {
86                   override def run(): Unit = {
87                     try {
88                       logger.info(s"Compacting ${p.name}")
89                       storage.compact(Some(p.name), fileSize)
90                     } catch {
91                       case NonFatal(e) => logger.error(s"Error processing partition '${p.name}':", e)
92                     } finally {
93                       latch.countDown()
94                     }
95                   }
96                 }
97               )
98             }
99           } finally {
100             executor.shutdown()
101           }
102 
103           while (latch.getCount > 0) {
104             Thread.sleep(1000)
105             status("", 1f - latch.getCount.toFloat / total, Seq.empty, done = false)
106           }
107           status("", 1f, Seq.empty, done = true)
108           Command.user.info("Compacting metadata")
109           storage.metadata.compact(None, None, math.max(1, params.threads))
110           Command.user.info(s"Local compaction complete in ${TextTools.getTime(start)}")
111 
112         case RunModes.Distributed =>
113           val encoding = storage.metadata.encoding
114           val job = if (ParquetFileSystemStorage.Encoding.equalsIgnoreCase(encoding)) {
115             new ParquetCompactionJob()
116           } else if (OrcFileSystemStorage.Encoding.equalsIgnoreCase(encoding)) {
117             new OrcCompactionJob()
118           } else {
119             throw new ParameterException(s"Compaction is not supported for encoding '$encoding'")
120           }
121           val tempDir = Option(params.tempPath).map(t => new Path(t))
122           job.run(storage, toCompact.toSeq, fileSize, tempDir, libjarsFiles, libjarsPaths, status) match {
123             case JobSuccess(message, counts) =>
124               Command.user.info(s"Distributed compaction complete in ${TextTools.getTime(start)}")
125               val success = counts(FileSystemCompactionJob.MappedCounter)
126               val failed = counts(FileSystemCompactionJob.FailedCounter)
127               Command.user.info(IngestCommand.getStatInfo(success, failed, "Compacted", message))
128 
129             case JobFailure(message) =>
130               Command.user.error(s"Distributed compaction failed in ${TextTools.getTime(start)}")
131               throw new CommandException(message)
132           }
133       }
134     }
135   }
