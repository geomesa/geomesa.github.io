39   private val truncateToPartitionStart: ZonedDateTime => ZonedDateTime = {
40     // note: `truncatedTo` is only valid up to DAYS, other units require additional steps
41     val truncate: ZonedDateTime => ZonedDateTime = stepUnit match {
42       case NANOS | MICROS | MILLIS | SECONDS | MINUTES | HOURS | DAYS =>
43         dt => dt.truncatedTo(stepUnit)
44 
45       case WEEKS =>
46         val adjuster = TemporalAdjusters.previousOrSame(DayOfWeek.MONDAY) // note: java day_of_week defines monday as the first day
47         dt => dt.`with`(adjuster).truncatedTo(DAYS)
48 
49       case MONTHS =>
50         val adjuster = TemporalAdjusters.firstDayOfMonth()
51         dt => dt.`with`(adjuster).truncatedTo(DAYS)
52 
53       case YEARS =>
54         val adjuster = TemporalAdjusters.firstDayOfYear()
55         dt => dt.`with`(adjuster).truncatedTo(DAYS)
56 
57       case _ =>
58         throw new IllegalArgumentException(s"${DateTimeScheme.Config.StepUnitOpt} $stepUnit is not supported")
59     }
60     if (step == 1) {
61       truncate
62     } else {
63       val chronoField = stepUnit match {
64         case NANOS   => ChronoField.NANO_OF_SECOND
65         case MICROS  => ChronoField.MICRO_OF_SECOND
66         case MILLIS  => ChronoField.MILLI_OF_SECOND
67         case SECONDS => ChronoField.SECOND_OF_MINUTE
68         case MINUTES => ChronoField.MINUTE_OF_HOUR
69         case HOURS   => ChronoField.HOUR_OF_DAY
70         case DAYS    => ChronoField.DAY_OF_YEAR
71         case WEEKS   => WeekFields.ISO.weekOfWeekBasedYear()
72         case MONTHS  => ChronoField.MONTH_OF_YEAR
73         case YEARS   => ChronoField.YEAR
74         case _ => throw new IllegalArgumentException(s"${DateTimeScheme.Config.StepUnitOpt} $stepUnit is not supported")
75       }
76       val min = math.max(chronoField.range().getMinimum, 0).toInt // account for 1-based fields like {month, week, day}_of_year
77       truncate.andThen { dt =>
78         val steps = dt.get(chronoField) - min
79         val remainder = steps % step
80         if (remainder == 0) { dt } else { dt.`with`(chronoField, steps + min - remainder ) }
81       }
82     }
83   }

94   override def getSimplifiedFilters(filter: Filter, partition: Option[String]): Option[Seq[SimplifiedFilter]] = {
95     getCoveringPartitions(filter).map { case (covered, intersecting) =>
96       val result = Seq.newBuilder[SimplifiedFilter]
97 
98       if (covered.nonEmpty) {
99         // remove the temporal filter that we've already accounted for in our covered partitions
100         val coveredFilter = andOption(partitionSubFilters(filter, isTemporalFilter(_, dtg))._2)
101         result += SimplifiedFilter(coveredFilter.getOrElse(Filter.INCLUDE), covered, partial = false)
102       }
103       if (intersecting.nonEmpty) {
104         result += SimplifiedFilter(filter, intersecting, partial = false)
105       }
106 
107       partition match {
108         case None => result.result
109         case Some(p) =>
110           val matched = result.result.find(_.partitions.contains(p))
111           matched.map(_.copy(partitions = Seq(p))).toSeq
112       }
113     }

119   override def getCoveringFilter(partition: String): Filter = {
120     val zdt = DateParsing.parse(partition, formatter)
121     val start = DateParsing.format(zdt)
122     val end = DateParsing.format(zdt.plus(step, stepUnit))
123     ff.and(ff.greaterOrEqual(ff.property(dtg), ff.literal(start)), ff.less(ff.property(dtg), ff.literal(end)))
124   }

126   private def getCoveringPartitions(filter: Filter): Option[(Seq[String], Seq[String])] = {
127     val bounds = FilterHelper.extractIntervals(filter, dtg, handleExclusiveBounds = false)
128     if (bounds.disjoint) {
129       Some((Seq.empty, Seq.empty))
130     } else if (bounds.isEmpty || !bounds.forall(_.isBoundedBothSides)) {
131       None
132     } else {
133       // there should be no duplicates in covered partitions, as our bounds will not overlap,
134       // but there may be multiple partial intersects with a given partition
135       val covered = ListBuffer.empty[String]
136       val intersecting = ListBuffer.empty[String]
137 
138       bounds.values.foreach { bound =>
139         // note: we verified both sides are bounded above
140         val lower = bound.lower.value.get
141         val upper = bound.upper.value.get
142         val start = truncateToPartitionStart(lower)
143         val end = truncateToPartitionStart(upper)
144 
145         // do our endpoints match the partition boundary, or do we need to apply a filter to the first/last partition?
146         val lowerBoundCovered = bound.lower.inclusive && lower == start
147 
148         // `stepUnit.between` claims to be upper endpoint exclusive, but doesn't seem to be...
149         val steps = stepUnit.between(start, end).toInt
150         if (steps < step) {
151           if (lowerBoundCovered &&
152               ((bound.upper.exclusive && upper == end) || (bound.upper.inclusive && upper == end.plus(step, stepUnit).minus(1, MILLIS)))) {
153             covered += formatter.format(start)
154           } else {
155             intersecting += formatter.format(start)
156           }
157         } else {
158           if (lowerBoundCovered) {
159             covered += formatter.format(start)
160           } else {
161             intersecting += formatter.format(start)
162           }
163           covered ++= Iterator.iterate(start)(_.plus(step, stepUnit)).drop(1).takeWhile(_.isBefore(end)).map(formatter.format)
164           if (bound.upper.inclusive || upper != end) {
165             intersecting += formatter.format(end)
166           }
167         }
168       }
169 
170       Some((covered.toSeq, intersecting.distinct.toSeq))
171     }
172   }
