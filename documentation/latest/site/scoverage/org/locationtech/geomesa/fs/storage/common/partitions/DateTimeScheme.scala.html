40   // note: `truncatedTo` is only valid up to DAYS, other units require additional steps
41   private val truncate: ZonedDateTime => ZonedDateTime = stepUnit match {
42     case NANOS | MICROS | MILLIS | SECONDS | MINUTES | HOURS | DAYS =>
43       dt => dt.truncatedTo(stepUnit)
44 
45     case MONTHS =>
46       val adjuster = TemporalAdjusters.firstDayOfMonth()
47       dt => dt.`with`(adjuster).truncatedTo(DAYS)
48 
49     case YEARS =>
50       val adjuster = TemporalAdjusters.firstDayOfYear()
51       dt => dt.`with`(adjuster).truncatedTo(DAYS)
52 
53     case _ =>
54       dt => DateParsing.parse(formatter.format(dt), formatter) // fall back to format and re-parse
55   }

64   override def getSimplifiedFilters(filter: Filter, partition: Option[String]): Option[Seq[SimplifiedFilter]] = {
65     getCoveringPartitions(filter).map { case (covered, intersecting) =>
66       val result = Seq.newBuilder[SimplifiedFilter]
67 
68       if (covered.nonEmpty) {
69         // remove the temporal filter that we've already accounted for in our covered partitions
70         val coveredFilter = andOption(partitionSubFilters(filter, isTemporalFilter(_, dtg))._2)
71         result += SimplifiedFilter(coveredFilter.getOrElse(Filter.INCLUDE), covered, partial = false)
72       }
73       if (intersecting.nonEmpty) {
74         result += SimplifiedFilter(filter, intersecting.distinct, partial = false)
75       }
76 
77       partition match {
78         case None => result.result
79         case Some(p) =>
80           val matched = result.result.find(_.partitions.contains(p))
81           matched.map(_.copy(partitions = Seq(p))).toSeq
82       }
83     }
84   }

89   override def getCoveringFilter(partition: String): Filter = {
90     val zdt = DateParsing.parse(partition, formatter)
91     val start = DateParsing.format(zdt)
92     val end = DateParsing.format(zdt.plus(1, stepUnit))
93     ff.and(ff.greaterOrEqual(ff.property(dtg), ff.literal(start)), ff.less(ff.property(dtg), ff.literal(end)))
94   }

96   private def getCoveringPartitions(filter: Filter): Option[(Seq[String], Seq[String])] = {
97     val bounds = FilterHelper.extractIntervals(filter, dtg, handleExclusiveBounds = false)
98     if (bounds.disjoint) {
99       Some((Seq.empty, Seq.empty))
100     } else if (bounds.isEmpty || !bounds.forall(_.isBoundedBothSides)) {
101       None
102     } else {
103       // there should be no duplicates in covered partitions, as our bounds will not overlap,
104       // but there may be multiple partial intersects with a given partition
105       val covered = ListBuffer.empty[String]
106       val intersecting = ListBuffer.empty[String]
107 
108       bounds.values.foreach { bound =>
109         // note: we verified both sides are bounded above
110         val lower = bound.lower.value.get
111         val upper = bound.upper.value.get
112         val start = truncate(lower)
113         // `stepUnit.between` claims to be upper endpoint exclusive, but doesn't seem to be...
114         // if exclusive end, subtract one milli so we don't search the whole next partition if it's on the endpoint
115         // note: we only support millisecond resolution
116         val end = truncate(if (bound.upper.inclusive) { upper } else { upper.minus(1, MILLIS) })
117         val steps = stepUnit.between(start, end).toInt
118 
119         // do our endpoints match the partition boundary, or do we need to apply a filter to the first/last partition?
120         val coveringFirst = bound.lower.inclusive && lower == start
121         val coveringLast =
122           (bound.upper.exclusive && upper == end.plus(1, stepUnit)) ||
123               (bound.upper.inclusive && !upper.isBefore(end.plus(1, stepUnit).minus(1, MILLIS)))
124 
125         if (steps == 0) {
126           if (coveringFirst && coveringLast) {
127             covered += formatter.format(start)
128           } else {
129             intersecting += formatter.format(start)
130           }
131         } else {
132           // add the first partition
133           if (coveringFirst) {
134             covered += formatter.format(start)
135           } else {
136             intersecting += formatter.format(start)
137           }
138 
139           @tailrec
140           def addSteps(step: Int, current: ZonedDateTime): Unit = {
141             if (step == steps) {
142               // last partition
143               if (coveringLast) {
144                 covered += formatter.format(current)
145               } else {
146                 intersecting += formatter.format(current)
147               }
148             } else {
149               covered += formatter.format(current)
150               addSteps(step + 1, current.plus(1, stepUnit))
151             }
152           }
153 
154           addSteps(1, start.plus(1, stepUnit))
155         }
156       }
157 
158       Some((covered.toSeq, intersecting.toSeq))
159     }
160   }
