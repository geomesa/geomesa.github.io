67   override def createSchema(original: SimpleFeatureType): Unit = {
68     import org.locationtech.geomesa.fs.storage.common.RichSimpleFeatureType
69 
70     manager.storage(original.getTypeName) match {
71       case Some(s) =>
72         logger.warn(s"Schema already exists: ${SimpleFeatureTypes.encodeType(s.metadata.sft, includeUserData = true)}")
73 
74       case None =>
75         // copy the feature type so that we don't affect it when removing user data, below
76         val sft = SimpleFeatureTypes.copy(original)
77 
78         GeoMesaSchemaValidator.validate(sft)
79 
80         // remove the configs in the user data as they're persisted separately
81         val meta = sft.removeMetadata().getOrElse(FileBasedMetadata.DefaultOptions)
82         val scheme = sft.removeScheme().getOrElse {
83           throw new IllegalArgumentException("Partition scheme must be specified in the SimpleFeatureType user data")
84         }
85         val encoding = sft.removeEncoding().orElse(config.defaultEncoding).getOrElse {
86           throw new IllegalArgumentException("Encoding type must be specified in either " +
87               "the SimpleFeatureType user data or the data store parameters")
88         }
89         val leafStorage = sft.removeLeafStorage().getOrElse {
90           val deprecated = scheme.options.get("leaf-storage").map { s =>
91             logger.warn("Using deprecated leaf-storage partition-scheme option. Please define leaf-storage using " +
92                 s"""`simpleFeatureType.getUserData.put("${StorageKeys.LeafStorageKey}", "$s")`""")
93             s.toBoolean
94           }
95           deprecated.getOrElse(true)
96         }
97         val fileSize = sft.removeTargetFileSize()
98 
99         val path = manager.defaultPath(sft.getTypeName)
100         val context = FileSystemContext(fs, config.conf, path, config.namespace)
101 
102         val metadata =
103           StorageMetadataFactory.create(context, meta, Metadata(sft, encoding, scheme, leafStorage, fileSize))
104         try { manager.register(path, FileSystemStorageFactory(context, metadata)) } catch {
105           case NonFatal(e) => CloseQuietly(metadata).foreach(e.addSuppressed); throw e
106         }
107         PathCache.register(fs, config.root)
108         PathCache.register(fs, path)
109     }
110   }
111 
112   override def createFeatureSource(entry: ContentEntry): ContentFeatureSource = {
113     val storage = this.storage(entry.getTypeName)
114     new FileSystemFeatureStore(storage, entry, Query.ALL, config.readThreads, config.writeTimeout, config.queryTimeout)
115   }
116 
117   /**
118     * Get a handle on the underlying storage instance for a given simple feature type
119     *
120     * @param typeName simple feature type name
121     * @return
122     */
123   def storage(typeName: String): FileSystemStorage = manager.storage(typeName).getOrElse {
124     throw new IllegalArgumentException(s"Schema '$typeName' doesn't exist or could not be loaded")
125   }
