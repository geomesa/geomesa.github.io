31 /**
32  * File system data store
33  *
34  * @param fs file system - note, this is expected to be a shared resource, and is not cleaned up on data store dispose
35  * @param conf conf
36  * @param root root path
37  * @param readThreads number of threads used for read ops
38  * @param writeTimeout timeout for write ops
39  * @param defaultEncoding default file encoding
40  * @param namespace geoserver namespace
41  */
42 class FileSystemDataStore(
43     fs: FileSystem,
44     conf: Configuration,
45     root: Path,
46     readThreads: Int,
47     writeTimeout: Duration,
48     defaultEncoding: Option[String],
49     namespace: Option[String]

70   override def createTypeNames(): java.util.List[Name] = {
71     val names = new java.util.ArrayList[Name]()
72     manager.storages().foreach(storage => names.add(storage.metadata.sft.getName))
73     names
74   }

76   override def createSchema(original: SimpleFeatureType): Unit = {
77     import org.locationtech.geomesa.fs.storage.common.RichSimpleFeatureType
78 
79     manager.storage(original.getTypeName) match {
80       case Some(s) =>
81         logger.warn(s"Schema already exists: ${SimpleFeatureTypes.encodeType(s.metadata.sft, includeUserData = true)}")
82 
83       case None =>
84         // copy the feature type so that we don't affect it when removing user data, below
85         val sft = SimpleFeatureTypes.copy(original)
86 
87         GeoMesaSchemaValidator.validate(sft)
88 
89         // remove the configs in the user data as they're persisted separately
90         val meta = sft.removeMetadata().getOrElse(FileBasedMetadata.DefaultOptions)
91         val scheme = sft.removeScheme().getOrElse {
92           throw new IllegalArgumentException("Partition scheme must be specified in the SimpleFeatureType user data")
93         }
94         val encoding = sft.removeEncoding().orElse(defaultEncoding).getOrElse {
95           throw new IllegalArgumentException("Encoding type must be specified in either " +
96               "the SimpleFeatureType user data or the data store parameters")
97         }
98         val leafStorage = sft.removeLeafStorage().getOrElse {
99           val deprecated = scheme.options.get("leaf-storage").map { s =>
100             logger.warn("Using deprecated leaf-storage partition-scheme option. Please define leaf-storage using " +
101                 s"""`simpleFeatureType.getUserData.put("${StorageKeys.LeafStorageKey}", "$s")`""")
102             s.toBoolean
103           }
104           deprecated.getOrElse(true)
105         }
106         val fileSize = sft.removeTargetFileSize()
107 
108         val path = manager.defaultPath(sft.getTypeName)
109         val context = FileSystemContext(fs, conf, path, namespace)
110 
111         val metadata =
112           StorageMetadataFactory.create(context, meta, Metadata(sft, encoding, scheme, leafStorage, fileSize))
113         try { manager.register(path, FileSystemStorageFactory(context, metadata)) } catch {
114           case NonFatal(e) => CloseQuietly(metadata).foreach(e.addSuppressed); throw e
115         }
116         PathCache.register(fs, root)
117         PathCache.register(fs, path)
118     }
119   }
120 
121   override def createFeatureSource(entry: ContentEntry): ContentFeatureSource =
122     new FileSystemFeatureStore(storage(entry.getTypeName), entry, Query.ALL, readThreads, writeTimeout)
123 
124   /**
125     * Get a handle on the underlying storage instance for a given simple feature type
126     *
127     * @param typeName simple feature type name
128     * @return
129     */
130   def storage(typeName: String): FileSystemStorage = manager.storage(typeName).getOrElse {
131     throw new IllegalArgumentException(s"Schema '$typeName' doesn't exist or could not be loaded")
132   }
