52   override def createSchema(original: SimpleFeatureType): Unit = {
53     import org.locationtech.geomesa.fs.storage.common.RichSimpleFeatureType
54 
55     manager.storage(original.getTypeName) match {
56       case Some(s) =>
57         logger.warn(s"Schema already exists: ${SimpleFeatureTypes.encodeType(s.metadata.sft, includeUserData = true)}")
58 
59       case None =>
60         // copy the feature type so that we don't affect it when removing user data, below
61         val sft = SimpleFeatureTypes.copy(original)
62 
63         GeoMesaSchemaValidator.validate(sft)
64 
65         // remove the configs in the user data as they're persisted separately
66         val meta = sft.removeMetadata().getOrElse(FileBasedMetadata.DefaultOptions)
67         val scheme = sft.removeScheme().getOrElse {
68           throw new IllegalArgumentException("Partition scheme must be specified in the SimpleFeatureType user data")
69         }
70         val encoding = sft.removeEncoding().orElse(defaultEncoding).getOrElse {
71           throw new IllegalArgumentException("Encoding type must be specified in either " +
72               "the SimpleFeatureType user data or the data store parameters")
73         }
74         val leafStorage = sft.removeLeafStorage().getOrElse {
75           val deprecated = scheme.options.get("leaf-storage").map { s =>
76             logger.warn("Using deprecated leaf-storage partition-scheme option. Please define leaf-storage using " +
77                 s"""`simpleFeatureType.getUserData.put("${StorageKeys.LeafStorageKey}", "$s")`""")
78             s.toBoolean
79           }
80           deprecated.getOrElse(true)
81         }
82         val fileSize = sft.removeTargetFileSize()
83 
84         val path = manager.defaultPath(sft.getTypeName)
85         val context = FileSystemContext(fc, conf, path, namespace)
86 
87         val metadata =
88           StorageMetadataFactory.create(context, meta, Metadata(sft, encoding, scheme, leafStorage, fileSize))
89         try { manager.register(path, FileSystemStorageFactory(context, metadata)) } catch {
90           case NonFatal(e) => CloseQuietly(metadata).foreach(e.addSuppressed); throw e
91         }
92     }
93   }
94 
95   override def createFeatureSource(entry: ContentEntry): ContentFeatureSource =
96     new FileSystemFeatureStore(storage(entry.getTypeName), entry, Query.ALL, readThreads, writeTimeout)

106   }
