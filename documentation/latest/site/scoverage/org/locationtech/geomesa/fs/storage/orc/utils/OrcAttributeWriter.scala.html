31   /**
32     * Create a writer for simple feature to the given vector batch
33     *
34     * @param sft simple feature type
35     * @param batch vector batch
36     * @param fid write feature id or not
37     * @return
38     */
39   def apply(sft: SimpleFeatureType, batch: VectorizedRowBatch, fid: Boolean = true): OrcAttributeWriter = {
40     require(batch.cols.length == OrcFileSystemStorage.fieldCount(sft, fid),
41       s"ORC schema does not match SimpleFeatureType: ${batch.cols.map(_.getClass.getName).mkString("\n\t", "\n\t", "")}")
42 
43     val builder = Seq.newBuilder[OrcAttributeWriter]
44     builder.sizeHint(sft.getAttributeCount + (if (fid) { 1 } else { 0 }))
45 
46     var i = 0
47     var col = 0
48     while (i < sft.getAttributeCount) {
49       val descriptor = sft.getDescriptor(i)
50       val bindings = ObjectType.selectType(descriptor)
51       val writer = bindings.head match {
52         case ObjectType.GEOMETRY => createGeometryWriter(bindings(1), batch.cols, col, i)
53         case ObjectType.DATE     => new DateWriter(batch.cols(col).asInstanceOf[TimestampColumnVector], i)
54         case ObjectType.STRING   => new StringWriter(batch.cols(col).asInstanceOf[BytesColumnVector], i)
55         case ObjectType.INT      => new IntWriter(batch.cols(col).asInstanceOf[LongColumnVector], i)
56         case ObjectType.LONG     => new LongWriter(batch.cols(col).asInstanceOf[LongColumnVector], i)
57         case ObjectType.FLOAT    => new FloatWriter(batch.cols(col).asInstanceOf[DoubleColumnVector], i)
58         case ObjectType.DOUBLE   => new DoubleWriter(batch.cols(col).asInstanceOf[DoubleColumnVector], i)
59         case ObjectType.BOOLEAN  => new BooleanWriter(batch.cols(col).asInstanceOf[LongColumnVector], i)
60         case ObjectType.BYTES    => new BytesWriter(batch.cols(col).asInstanceOf[BytesColumnVector], i)
61         case ObjectType.UUID     => new UuidWriter(batch.cols(col).asInstanceOf[BytesColumnVector], i)
62         case ObjectType.LIST     => new ListWriter(batch.cols(col).asInstanceOf[ListColumnVector], i, bindings(1))
63         case ObjectType.MAP      => new MapWriter(batch.cols(col).asInstanceOf[MapColumnVector], i, bindings(1), bindings(2))
64         case _ => throw new IllegalArgumentException(s"Unexpected object type ${bindings.head}")
65       }
66       builder += writer
67       i += 1
68       col += OrcFileSystemStorage.fieldCount(descriptor)
69     }
70 
71     if (fid) {
72       builder += new FidWriter(batch.cols(col).asInstanceOf[BytesColumnVector])
73     }
74 
75     new SequenceWriter(builder.result)
76   }

78   // noinspection LanguageFeature
79   private def createGeometryWriter(binding: ObjectType, cols: Array[ColumnVector], col: Int, i: Int): OrcAttributeWriter = {
80     implicit def toDoubleColumnVector(vec: ColumnVector): DoubleColumnVector = vec.asInstanceOf[DoubleColumnVector]
81     implicit def toListColumnVector(vec: ColumnVector): ListColumnVector = vec.asInstanceOf[ListColumnVector]
82 
83     def x: ColumnVector = cols(col)
84     def y: ColumnVector = cols(col + 1)
85 
86     binding match {
87       case ObjectType.POINT           => new PointWriter(x, y, i)
88       case ObjectType.LINESTRING      => new LineStringWriter(x, y, i)
89       case ObjectType.MULTIPOINT      => new MultiPointWriter(x, y, i)
90       case ObjectType.POLYGON         => new PolygonWriter(x, y, i)
91       case ObjectType.MULTILINESTRING => new MultiLineStringWriter(x, y, i)
92       case ObjectType.MULTIPOLYGON    => new MultiPolygonWriter(x, y, i)
93       case ObjectType.GEOMETRY        => new GeometryWkbWriter(cols(col).asInstanceOf[BytesColumnVector], i)
94       case _ => throw new IllegalArgumentException(s"Unexpected object type $binding")
95     }
96   }

169   /**
170     * Writes a linestring attribute to a vector. A linestring is modeled as a list of points.
171     *
172     * @see PointWriter
173     *
174     * @param xx outer list vector for x coordinates, containing a double vector for points
175     * @param yy outer list vector for y coordinates, containing a double vector for points
176     * @param attribute simple feature attribute index
177     */
178   class LineStringWriter(xx: ListColumnVector, yy: ListColumnVector, attribute: Int) extends OrcAttributeWriter {
179     private val x = xx.child.asInstanceOf[DoubleColumnVector]
180     private val y = yy.child.asInstanceOf[DoubleColumnVector]
181 
182     override def apply(sf: SimpleFeature, row: Int): Unit = {
183       val value = sf.getAttribute(attribute).asInstanceOf[LineString]
184       if (value != null) {
185         val length = value.getNumPoints
186         Seq(xx, yy).foreach { vector =>
187           vector.child.ensureSize(vector.childCount + length, true)
188           vector.offsets(row) = vector.childCount
189           vector.lengths(row) = length
190         }
191         var i = 0
192         while (i < length) {
193           val pt = value.getCoordinateN(i)
194           x.vector(xx.childCount + i) = pt.x
195           y.vector(yy.childCount + i) = pt.y
196           i += 1
197         }
198         xx.childCount += length
199         yy.childCount += length
200       } else {
201         xx.noNulls = false
202         yy.noNulls = false
203         xx.isNull(row) = true
204         yy.isNull(row) = true
205       }
206     }

249   /**
250     * Writes a polygon attribute to a vector. A polygon is modeled as a list of lines, with the first
251     * value being the shell, and any subsequent values being interior holes.
252     *
253     * @see LineStringWriter
254     *
255     * @param xxx outer list vector for x coordinates, containing a list vector for individual lines
256     * @param yyy outer list vector for y coordinates, containing a list vector for individual lines
257     * @param attribute simple feature attribute index
258     */
259   class PolygonWriter(xxx: ListColumnVector, yyy: ListColumnVector, attribute: Int) extends OrcAttributeWriter {
260     // list of points for each line
261     private val xx = xxx.child.asInstanceOf[ListColumnVector]
262     private val yy = yyy.child.asInstanceOf[ListColumnVector]
263     // points
264     private val x = xx.child.asInstanceOf[DoubleColumnVector]
265     private val y = yy.child.asInstanceOf[DoubleColumnVector]
266 
267     override def apply(sf: SimpleFeature, row: Int): Unit = {
268       val polygon = sf.getAttribute(attribute).asInstanceOf[Polygon]
269       if (polygon != null) {
270         val lineCount = polygon.getNumInteriorRing + 1
271         Seq(xxx, yyy).foreach { vector =>
272           vector.child.ensureSize(vector.childCount + lineCount, true)
273           vector.offsets(row) = vector.childCount
274           vector.lengths(row) = lineCount
275         }
276         var j = 0
277         while (j < lineCount) {
278           val line = if (j == 0) { polygon.getExteriorRing } else { polygon.getInteriorRingN(j - 1) }
279           val length = line.getNumPoints
280           Seq(xx, yy).foreach { vector =>
281             vector.child.ensureSize(vector.childCount + length, true)
282             vector.offsets(xxx.childCount + j) = vector.childCount
283             vector.lengths(xxx.childCount + j) = length
284           }
285           var i = 0
286           while (i < length) {
287             val pt = line.getCoordinateN(i)
288             x.vector(xx.childCount + i) = pt.x
289             y.vector(yy.childCount + i) = pt.y
290             i += 1
291           }
292           xx.childCount += length
293           yy.childCount += length
294           j += 1
295         }
296         xxx.childCount += lineCount
297         yyy.childCount += lineCount
298       } else {
299         xxx.noNulls = false
300         yyy.noNulls = false
301         xxx.isNull(row) = true
302         yyy.isNull(row) = true
303       }
304     }
305   }
