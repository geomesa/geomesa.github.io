33 /**
34  * StorageMetadata implementation. Saves changes as a series of timestamped changelogs to allow
35  * concurrent modifications. The current state is obtained by replaying all the logs.
36  *
37  * Note that state is not read off disk until 'reload' is called.
38  *
39  * When accessed through the standard factory methods, the state will periodically reload from disk
40  * in order to pick up external changes (every 10 minutes by default).
41  *
42  * Note that modifications made to the metadata may not be immediately available, if they occur
43  * simultaneously with a reload. For example, calling `getPartition` immediately after `addPartition` may
44  * not return anything. However, the change is always persisted to disk, and will be available after the next
45  * reload. In general this does not cause problems, as reads and writes happen in different JVMs (ingest
46  * vs query).
47  *
48  * @param fs file system
49  * @param directory metadata root path
50  * @param sft simple feature type
51  * @param meta basic metadata config
52  * @param converter file converter
53  */

107   override def addPartition(partition: PartitionMetadata): Unit = {
108     val config = {
109       val action = PartitionAction.Add
110       val envelope = partition.bounds.map(b => EnvelopeConfig(b.envelope)).getOrElse(Seq.empty)
111       PartitionConfig(partition.name, action, partition.files, partition.count, envelope, System.currentTimeMillis())
112     }

114     // if we have already loaded the partition, merge in the new value
115     if (metadata.getIfPresent(partition.name) != null) {
116       metadata.asMap.merge(partition.name, partition, addMetadata)
117     }
118     Option(partitions.getIfPresent(BoxedUnit.UNIT)).foreach { files =>
119       files.merge(partition.name, PartitionFiles(config = Seq(config), parsed = Seq(path)), addFiles)

140     val map = new ConcurrentHashMap[String, PartitionFiles]

158   override def compact(partition: Option[String], fileSize: Option[Long], threads: Int): Unit = {
159     require(threads > 0, "Threads must be a positive number")
160 
161     // in normal usage, we never pass in a partition to this method
162     partition.foreach(p => logger.warn(s"Ignoring requested partition '$p' and compacting all metadata"))
163 
164     val configs = ArrayBuffer.empty[PartitionConfig]
165     val paths = ArrayBuffer.empty[Path]
166 
167     readPartitionFiles(threads).asScala.foreach { case (name, f) =>
168       val config = readPartition(f, threads).filter(_.files.nonEmpty)
169       config match {
170         case None => metadata.invalidate(name)
171         case Some(c) =>
172           metadata.put(name, c.toMetadata)
173           configs += c
174       }
175       paths ++= f.unparsed
176       paths ++= f.parsed
177     }
178 
179     writeCompactedConfig(configs.toSeq)
180     delete(paths, threads)
181 
182     partitions.invalidate(BoxedUnit.UNIT)
183   }

198   private def writePartition(config: PartitionConfig): Path = {
199     val data = profile("Serialized partition configuration") {
200       converter.renderPartition(config)
201     }
202     profile("Persisted partition configuration") {
203       val encoded = StringSerialization.alphaNumericSafeString(config.name)
204       val name = s"$UpdatePartitionPrefix$encoded-${UUID.randomUUID()}${converter.suffix}"
205       val file = new Path(directory, name)
206       WithClose(fs.create(file, false)) { out =>
207         out.write(data.getBytes(StandardCharsets.UTF_8))
208         out.hflush()
209         out.hsync()
210       }
211       PathCache.register(fs, file)
212       file
213     }
214   }

226       val file = new Path(directory, CompactedPrefix + converter.suffix)

229         out.hflush()

231       }

234       val toRemove =

236       if (PathCache.exists(fs, toRemove, reload = true)) {

238         PathCache.invalidate(fs, toRemove)
239       }
240     }

244    * Reads all the metadata files and groups them by partition, parsing them only if needed
245    * to determine the partition name
246    *
247    * @param threads threads
248    * @return

250   private def readPartitionFiles(threads: Int): ConcurrentMap[String, PartitionFiles] = {
251     val result = new ConcurrentHashMap[String, PartitionFiles]()
252 
253     // list all the metadata files on disk
254     profile("Listed metadata files") {
255       val pool = new CachedThreadPool(threads)
256       // use a phaser to track worker thread completion
257       val phaser = new Phaser(2) // 1 for the initial directory worker + 1 for this thread
258       pool.submit(new DirectoryWorker(pool, phaser, fs.listStatusIterator(directory), result))
259       // wait for the worker threads to complete
260       try {
261         phaser.awaitAdvanceInterruptibly(phaser.arrive())
262       } finally {
263         pool.shutdown()
264       }
265     }
266 
267     result
268   }

271    * Parses and merges the config files for a given partition
272    *
273    * @param files files associated with the partition
274    * @param threads threads
275    * @return
276    */
277   private def readPartition(files: PartitionFiles, threads: Int): Option[PartitionConfig] = {
278     val updates = if (threads < 2) {
279       files.unparsed.flatMap(readPartitionConfig)
280     } else {
281       val ec = new CachedThreadPool(threads)
282       try {
283         val results = Seq.newBuilder[PartitionConfig]
284         def readOne(p: Path): Unit = {
285           readPartitionConfig(p).foreach { c =>
286             results.synchronized(results += c)
287           }
288         }
289         files.unparsed.toList.map(p => ec.submit(new Runnable() { override def run(): Unit = readOne(p)})).foreach(_.get)
290         results.result
291       } finally {
292         ec.shutdown()
293       }
294     }
295     mergePartitionConfigs(updates ++ files.config).filter(_.files.nonEmpty)
296   }

359   private class DirectoryWorker(
360       es: ExecutorService,
361       phaser: Phaser,
362       listDirectory: => RemoteIterator[FileStatus],
363       result: ConcurrentHashMap[String, PartitionFiles]
364     ) extends Runnable {
365 
366     override def run(): Unit = {
367       try {
368         var i = phaser.getRegisteredParties + 1
369         val iter = listDirectory
370         while (iter.hasNext && i < PhaserUtils.MaxParties) {
371           val status = iter.next
372           val path = status.getPath
373           lazy val name = path.getName
374           if (status.isDirectory) {
375             i += 1
376             // use a tiered phaser on each directory avoid the limit of 65535 registered parties
377             es.submit(new DirectoryWorker(es, new Phaser(phaser, 1), fs.listStatusIterator(path), result))
378           } else if (name.startsWith(UpdatePartitionPrefix)) {
379             // pull out the partition name but don't parse the file yet
380             val encoded = name.substring(8, name.length - 42) // strip out prefix and suffix
381             val partition = StringSerialization.decodeAlphaNumericSafeString(encoded)
382             result.merge(partition, PartitionFiles(unparsed = Seq(path)), addFiles)
383           } else if (name == CompactedHocon || name == CompactedJson) {
384             i += 1
385             phaser.register() // register the new worker thread
386             es.submit(new CompactedParser(phaser, path, result))
387           } else if (name.startsWith(UpdateFilePrefix) && name.endsWith(JsonPathSuffix)) {
388             // old update files - have to parse them to get the partition name
389             i += 1
390             phaser.register() // register the new worker thread
391             es.submit(new UpdateParser(phaser, path, result))
392           }
393         }
394         if (iter.hasNext) {
395           es.submit(new DirectoryWorker(es, new Phaser(phaser, 1), iter, result))
396         }
397       } catch {
398         case _: FileNotFoundException => // the partition dir was deleted... just return
399         case NonFatal(e) => logger.error("Error scanning metadata directory:", e)
400       } finally {
401         phaser.arrive() // notify that this thread is done
402       }
403     }
404   }
