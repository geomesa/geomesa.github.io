32 /**
33  * StorageMetadata implementation. Saves changes as a series of timestamped changelogs to allow
34  * concurrent modifications. The current state is obtained by replaying all the logs.
35  *
36  * Note that state is not read off disk until 'reload' is called.
37  *
38  * When accessed through the standard factory methods, the state will periodically reload from disk
39  * in order to pick up external changes (every 10 minutes by default).
40  *
41  * Note that modifications made to the metadata may not be immediately available, if they occur
42  * simultaneously with a reload. For example, calling `getPartition` immediately after `addPartition` may
43  * not return anything. However, the change is always persisted to disk, and will be available after the next
44  * reload. In general this does not cause problems, as reads and writes happen in different JVMs (ingest
45  * vs query).
46  *
47  * @param fs file system
48  * @param directory metadata root path
49  * @param sft simple feature type
50  * @param meta basic metadata config
51  * @param converter file converter
52  */

106   override def addPartition(partition: PartitionMetadata): Unit = {
107     val config = {
108       val action = PartitionAction.Add
109       val envelope = partition.bounds.map(b => EnvelopeConfig(b.envelope)).getOrElse(Seq.empty)
110       PartitionConfig(partition.name, action, partition.files, partition.count, envelope, System.currentTimeMillis())
111     }

113     // if we have already loaded the partition, merge in the new value
114     if (metadata.getIfPresent(partition.name) != null) {
115       metadata.asMap.merge(partition.name, partition, addMetadata)
116     }
117     Option(partitions.getIfPresent(BoxedUnit.UNIT)).foreach { files =>
118       files.merge(partition.name, PartitionFiles(config = Seq(config), parsed = Seq(path)), addFiles)

138   override def setPartitions(partitions: Seq[PartitionMetadata]): Unit = {
139     val map = new ConcurrentHashMap[String, PartitionFiles]
140     this.partitions.put(BoxedUnit.UNIT, map)
141     this.metadata.invalidateAll()

157   override def compact(partition: Option[String], fileSize: Option[Long], threads: Int): Unit = {
158     require(threads > 0, "Threads must be a positive number")
159 
160     // in normal usage, we never pass in a partition to this method
161     partition.foreach(p => logger.warn(s"Ignoring requested partition '$p' and compacting all metadata"))
162 
163     val configs = ArrayBuffer.empty[PartitionConfig]
164     val paths = ArrayBuffer.empty[Path]
165 
166     readPartitionFiles(threads).asScala.foreach { case (name, f) =>
167       val config = readPartition(f, threads).filter(_.files.nonEmpty)
168       config match {
169         case None => metadata.invalidate(name)
170         case Some(c) =>
171           metadata.put(name, c.toMetadata)
172           configs += c
173       }
174       paths ++= f.unparsed
175       paths ++= f.parsed
176     }
177 
178     writeCompactedConfig(configs.toSeq)
179     delete(paths, threads)
180 
181     partitions.invalidate(BoxedUnit.UNIT)
182   }

197   private def writePartition(config: PartitionConfig): Path = {
198     val data = profile("Serialized partition configuration") {
199       converter.renderPartition(config)
200     }
201     profile("Persisted partition configuration") {
202       val encoded = StringSerialization.alphaNumericSafeString(config.name)
203       val name = s"$UpdatePartitionPrefix$encoded-${UUID.randomUUID()}${converter.suffix}"
204       val file = new Path(directory, name)
205       WithClose(fs.create(file, false)) { out =>
206         out.write(data.getBytes(StandardCharsets.UTF_8))
207         out.hflush()
208         out.hsync()
209       }
210       PathCache.register(fs, file)
211       file
212     }
213   }

225       val file = new Path(directory, CompactedPrefix + converter.suffix)

239     }

243    * Reads all the metadata files and groups them by partition, parsing them only if needed
244    * to determine the partition name
245    *
246    * @param threads threads
247    * @return

249   private def readPartitionFiles(threads: Int): ConcurrentMap[String, PartitionFiles] = {
250     val result = new ConcurrentHashMap[String, PartitionFiles]()
251 
252     // list all the metadata files on disk
253     profile("Listed metadata files") {
254       val pool = new CachedThreadPool(threads)
255       // use a phaser to track worker thread completion
256       val phaser = new Phaser(2) // 1 for the initial directory worker + 1 for this thread
257       pool.submit(new DirectoryWorker(pool, phaser, fs.listStatusIterator(directory), result))
258       // wait for the worker threads to complete
259       try {
260         phaser.awaitAdvanceInterruptibly(phaser.arrive())
261       } finally {
262         pool.shutdown()
263       }
264     }
265 
266     result
267   }

269   /**
270    * Parses and merges the config files for a given partition
271    *
272    * @param files files associated with the partition
273    * @param threads threads
274    * @return
275    */
276   private def readPartition(files: PartitionFiles, threads: Int): Option[PartitionConfig] = {
277     val updates = if (threads < 2) {
278       files.unparsed.flatMap(readPartitionConfig)
279     } else {
280       val ec = new CachedThreadPool(threads)
281       try {
282         val results = Seq.newBuilder[PartitionConfig]
283         def readOne(p: Path): Unit = {
284           readPartitionConfig(p).foreach { c =>
285             results.synchronized(results += c)
286           }
287         }
288         files.unparsed.toList.map(p => ec.submit(new Runnable() { override def run(): Unit = readOne(p)})).foreach(_.get)
289         results.result
290       } finally {
291         ec.shutdown()
292       }
293     }
294     mergePartitionConfigs(updates ++ files.config).filter(_.files.nonEmpty)
295   }

358   private class DirectoryWorker(
359       es: ExecutorService,
360       phaser: Phaser,
361       listDirectory: => RemoteIterator[FileStatus],
362       result: ConcurrentHashMap[String, PartitionFiles]
363     ) extends Runnable {
364 
365     override def run(): Unit = {
366       try {
367         var i = phaser.getRegisteredParties + 1
368         val iter = listDirectory
369         while (iter.hasNext && i < PhaserUtils.MaxParties) {
370           val status = iter.next
371           val path = status.getPath
372           lazy val name = path.getName
373           if (status.isDirectory) {
374             i += 1
375             // use a tiered phaser on each directory avoid the limit of 65535 registered parties
376             es.submit(new DirectoryWorker(es, new Phaser(phaser, 1), fs.listStatusIterator(path), result))
377           } else if (name.startsWith(UpdatePartitionPrefix)) {
378             // pull out the partition name but don't parse the file yet
379             val encoded = name.substring(8, name.length - 42) // strip out prefix and suffix
380             val partition = StringSerialization.decodeAlphaNumericSafeString(encoded)
381             result.merge(partition, PartitionFiles(unparsed = Seq(path)), addFiles)
382           } else if (name == CompactedHocon || name == CompactedJson) {
383             i += 1
384             phaser.register() // register the new worker thread
385             es.submit(new CompactedParser(phaser, path, result))
386           } else if (name.startsWith(UpdateFilePrefix) && name.endsWith(JsonPathSuffix)) {
387             // old update files - have to parse them to get the partition name
388             i += 1
389             phaser.register() // register the new worker thread
390             es.submit(new UpdateParser(phaser, path, result))
391           }
392         }
393         if (iter.hasNext) {
394           es.submit(new DirectoryWorker(es, new Phaser(phaser, 1), iter, result))
395         }
396       } catch {
397         case _: FileNotFoundException => // the partition dir was deleted... just return
398         case NonFatal(e) => logger.error("Error scanning metadata directory:", e)
399       } finally {
400         phaser.arrive() // notify that this thread is done
401       }
402     }
403   }
