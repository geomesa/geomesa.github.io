30 /**

32  *
33  * @param sft simple feature type
34  * @param encodings type encoding
35  * @param hasVisibilities whether the schema encodes visibilities, or not
36  * @param boundingBoxes names of any bounding box fields, along with the original field name
37  * @param metadata file metadata

100   /**
101     * Extract the simple feature type from a parquet read context. The read context
102     * contains both file metadata and the provided read conf
103     *
104     * @param context parquet read context
105     * @return
106     */
107   def read(context: InitContext): Option[SimpleFeatureParquetSchema] = {
108     val metadata = new java.util.HashMap[String, String]()
109     // copy in the file level metadata - note, do this before copying in the conf so that transform schemas are applied correctly
110     context.getKeyValueMetadata.asScala.foreach { case (k, v) => if (!v.isEmpty) { metadata.put(k, v.iterator.next) }}
111     // noinspection ScalaDeprecation
112     Seq(SftNameKey, SftSpecKey, GeometryEncodingKey, SchemaVersionKey).foreach { key =>
113       val value =
114         try { context.getParquetConfiguration.get(key) } catch {
115           // noinspection ScalaDeprecation - use the older method for spark compatibility
116           case _: NoSuchMethodError => context.getConfiguration.get(key)
117         }
118       if (value != null) {
119         metadata.put(key, value)
120       }
121     }
122     read(context.getFileSchema, metadata)
123   }

155   /**
156    * Get the schema for a file being read
157    *
158    * @param fileSchema message type
159    * @param metadata file metadata
160    * @return
161    */
162   private def read(fileSchema: MessageType, metadata: java.util.Map[String, String]): Option[SimpleFeatureParquetSchema] = {
163     for {
164       name <- Option(metadata.get(SftNameKey))
165       spec <- Option(metadata.get(SftSpecKey))
166     } yield {
167       val sft = SimpleFeatureTypes.createType(name, spec)
168       val geometries = Option(metadata.get(GeometryEncodingKey)).map(GeometryEncoding.apply).getOrElse {
169         // noinspection ScalaDeprecation
170         if (metadata.get(SchemaVersionKey) == "1") { GeometryEncoding.GeoMesaV1 } else { GeometryEncoding.GeoMesaV0 }
171       }
172       val encodings = Encodings(geometries)
173       val bboxes = fileSchema.getFields.asScala.map(_.getName).flatMap(BoundingBoxField.fromBoundingBox).toSeq
174       val visibilities = fileSchema.containsField(VisibilitiesField)
175       SimpleFeatureParquetSchema(sft, encodings, visibilities, bboxes, Collections.unmodifiableMap(metadata))
176     }
177   }

179   /**
180    * Gets a schema for writing
181    *
182    * @param sftName sft name config
183    * @param sftSpec sft spec config
184    * @param geoms geometry encoding config
185    * @param bboxOpt bounding box encoding config
186    * @param vis visibility config
187    * @return
188    */
189   private def write(sftName: String, sftSpec: String, geoms: String, bboxOpt: String, vis: String): Option[SimpleFeatureParquetSchema] = {
190     for {
191       name <- Option(sftName)
192       spec <- Option(sftSpec)
193     } yield {
194       val sft = SimpleFeatureTypes.createType(name, spec)
195       val geometries = Option(geoms).map(GeometryEncoding.apply).getOrElse(GeometryEncoding.GeoParquetWkb)
196       // only include bboxes if they help with push-down filters
197       val bboxes = if (bboxOpt != null && !bboxOpt.toBoolean) { Seq.empty } else { BoundingBoxField(sft, geometries) }
198       val visibilities = Option(sft.getUserData.get(VisibilityEncodingKey)).orElse(Option(vis)).forall(_.toString.toBoolean)
199       val metadata = new java.util.HashMap[String, String]()
200       metadata.put(SftNameKey, name)
201       metadata.put(SftSpecKey, spec)
202       metadata.put(GeometryEncodingKey, geometries.toString)
203       if (geometries == GeometryEncoding.GeoMesaV1) {
204         // noinspection ScalaDeprecation
205         metadata.put(SchemaVersionKey, "1") // include schema version for back-compatibility when possible
206       }
207       val encodings = Encodings(geometries)
208       SimpleFeatureParquetSchema(sft, encodings, visibilities, bboxes, Collections.unmodifiableMap(metadata))
209     }
210   }

212   /**
213    * Get the message type for a simple feature type
214    *
215    * @param sft simple feature type
216    * @param encodings field type encoding
217    * @param bboxes include bounding boxes for each row
218    * @param visibilities include visibilities
219    * @return
220    */
221   private def schema(sft: SimpleFeatureType, encodings: Encodings, bboxes: Seq[String], visibilities: Boolean): MessageType = {
222     val isV0 = encodings.geometry == GeometryEncoding.GeoMesaV0
223     val id = {
224       val primitive = PrimitiveTypeName.BINARY
225       val builder = if (isV0) { Types.primitive(primitive, Repetition.REPEATED) } else { Types.required(primitive) }
226       builder.as(LogicalTypeAnnotation.stringType()).named(FeatureIdField)
227     }
228     val vis = if (!visibilities) { None } else {
229       Some(Types.optional(PrimitiveTypeName.BINARY).as(LogicalTypeAnnotation.stringType()).named(VisibilitiesField))
230     }
231     val boundingBoxes = bboxes.map(BoundingBoxField.schema)
232     // note: id field goes at the end of the record, then vis, then any bounding boxes
233     val fields = (sft.getAttributeDescriptors.asScala.map(schema(_, encodings)) :+ id) ++ vis ++ boundingBoxes
234     val name = if (isV0) { sft.getTypeName } else { alphaNumericSafeString(sft.getTypeName) }
235     new MessageType(name, fields.asJava)
236   }

238   /**
239    * Create a parquet field type from an attribute descriptor
240    *
241    * @param descriptor descriptor
242    * @param encodings field type encoding
243    * @return

245   private def schema(descriptor: AttributeDescriptor, encodings: Encodings): Type = {

251   /**
252    * Builds the schema type for an attribute
253    *
254    * @param name field name
255    * @param bindings object type
256    * @param encodings schema encodings
257    * @param repetition repetition
258    * @return
259    */
260   private def buildType(
261       name: String,
262       bindings: Seq[ObjectType],
263       encodings: Encodings,
264       repetition: Repetition = Repetition.OPTIONAL): Type = {
265     val builder = bindings.head match {
266       case ObjectType.STRING   => Types.primitive(PrimitiveTypeName.BINARY, repetition).as(LogicalTypeAnnotation.stringType())
267       case ObjectType.INT      => Types.primitive(PrimitiveTypeName.INT32, repetition)
268       case ObjectType.DOUBLE   => Types.primitive(PrimitiveTypeName.DOUBLE, repetition)
269       case ObjectType.LONG     => Types.primitive(PrimitiveTypeName.INT64, repetition)
270       case ObjectType.FLOAT    => Types.primitive(PrimitiveTypeName.FLOAT, repetition)
271       case ObjectType.BOOLEAN  => Types.primitive(PrimitiveTypeName.BOOLEAN, repetition)
272       case ObjectType.BYTES    => Types.primitive(PrimitiveTypeName.BINARY, repetition)
273 
274       case ObjectType.DATE =>
275         val unit = encodings.date match {
276           case DateEncoding.Micros => TimeUnit.MICROS
277           case DateEncoding.Millis => TimeUnit.MILLIS
278           case _ => throw new UnsupportedOperationException(encodings.date.toString)
279         }
280         Types.primitive(PrimitiveTypeName.INT64, repetition).as(LogicalTypeAnnotation.timestampType(true, unit))
281 
282       case ObjectType.UUID =>
283         encodings.uuid match {
284           case UuidEncoding.FixedLength =>
285             Types.primitive(PrimitiveTypeName.FIXED_LEN_BYTE_ARRAY, repetition).length(16).as(LogicalTypeAnnotation.uuidType())
286           case UuidEncoding.Bytes => Types.primitive(PrimitiveTypeName.BINARY, repetition)
287           case _ => throw new UnsupportedOperationException(encodings.uuid.toString)
288         }
289 
290       case ObjectType.LIST =>
291         encodings.list match {
292           case ListEncoding.ThreeLevel =>
293             Types.optionalList().requiredGroupElement().addField(buildType("element", bindings.drop(1), encodings, Repetition.REQUIRED))
294           case ListEncoding.TwoLevel => Types.optionalList().element(buildType("element", bindings.drop(1), encodings))
295           case _ => throw new UnsupportedOperationException(encodings.list.toString)
296         }
297 
298       case ObjectType.MAP =>
299         Types.optionalMap()
300           .key(buildType("key", bindings.slice(1, 2), encodings, Repetition.REQUIRED))
301           .value(buildType("value", bindings.slice(2, 3), encodings))
302 
303       case ObjectType.GEOMETRY =>
304         GeometrySchema(bindings(1), encodings.geometry)
305 
306       case binding =>
307         throw new UnsupportedOperationException(s"No mapping defined for type $binding")
308     }
309     builder.named(name)
310   }
