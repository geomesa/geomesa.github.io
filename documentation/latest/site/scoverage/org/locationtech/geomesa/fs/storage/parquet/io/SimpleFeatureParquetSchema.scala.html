57 object SimpleFeatureParquetSchema {
58 
59   import scala.collection.JavaConverters._
60 
61   val FeatureIdField = "__fid__"
62 
63   val SchemaVersionKey = "geomesa.parquet.version"
64 
65   val CurrentSchemaVersion = 1
66 
67   val GeometryColumnX = "x"
68   val GeometryColumnY = "y"
69 
70   /**
71     * Extract the simple feature type from a parquet read context. The read context
72     * contains both file metadata and the provided read conf
73     *
74     * @param context parquet read context
75     * @return
76     */
77   def read(context: InitContext): Option[SimpleFeatureParquetSchema] = {
78     val metadata = new java.util.HashMap[String, String]()
79     // copy in the file level metadata
80     context.getKeyValueMetadata.asScala.foreach { case (k, v) => if (!v.isEmpty) { metadata.put(k, v.iterator.next) }}
81     val conf = context.getConfiguration
82     // copy in the sft from the conf - overwrite the file level metadata as this has our transform schema
83     Seq(StorageConfiguration.SftNameKey, StorageConfiguration.SftSpecKey, SchemaVersionKey).foreach { key =>
84       val value = conf.get(key)
85       if (value != null) {
86         metadata.put(key, value)
87       }
88     }
89     apply(metadata)
90   }
91 
92   /**
93     * Extract the simple feature type from an existing parquet file, without any known read context
94     *
95     * @param footer parquet file footer
96     * @return
97     */
98   def read(footer: FileMetaData): Option[SimpleFeatureParquetSchema] = apply(footer.getKeyValueMetaData)
99 
100   /**
101     * Get a schema for writing. This will use the latest schema version
102     *
103     * @param conf write configuration, including the sft spec
104     * @return
105     */
106   def write(conf: Configuration): Option[SimpleFeatureParquetSchema] = {
107     val metadata = new java.util.HashMap[String, String]()
108     metadata.put(SchemaVersionKey, CurrentSchemaVersion.toString)
109     // copy in the sft from the conf
110     Seq(StorageConfiguration.SftNameKey, StorageConfiguration.SftSpecKey).foreach { key =>
111       val value = conf.get(key)
112       if (value != null) {
113         metadata.put(key, value)
114       }
115     }
116     apply(metadata)
117   }
118 
119   /**
120     * Determine the appropriate versioned schema
121     *
122     * @param metadata read metadata, which should include the projected simple feature type and version info
123     * @return
124     */
125   private def apply(metadata: java.util.Map[String, String]): Option[SimpleFeatureParquetSchema] = {
126     for {
127       name <- Option(metadata.get(StorageConfiguration.SftNameKey))
128       spec <- Option(metadata.get(StorageConfiguration.SftSpecKey))
129     } yield {
130       val sft = SimpleFeatureTypes.createType(name, spec)
131       Option(metadata.get(SchemaVersionKey)).map(_.toInt).getOrElse(0) match {
132         case 1 => new SimpleFeatureParquetSchema(sft, schema(sft))
133         case 0 => new SimpleFeatureParquetSchema(sft, SimpleFeatureParquetSchemaV0(sft))
134         case v => throw new IllegalArgumentException(s"Unknown SimpleFeatureParquetSchema version: $v")
135       }
136     }
137   }
138 
139   /**
140     * Get the message type for a simple feature type
141     *
142     * @param sft simple feature type
143     * @return
144     */
145   private def schema(sft: SimpleFeatureType): MessageType = {
146     val id = Types.required(PrimitiveTypeName.BINARY).as(OriginalType.UTF8).named(FeatureIdField)
147     // note: id field goes at the end of the record
148     val fields = sft.getAttributeDescriptors.asScala.map(schema) :+ id
149     // ensure that we use a valid name - for avro conversion, especially, names are very limited
150     new MessageType(StringSerialization.alphaNumericSafeString(sft.getTypeName), fields.asJava)
151   }
152 
153   /**
154     * Create a parquet field type from an attribute descriptor
155     *
156     * @param descriptor descriptor
157     * @return
158     */
159   private def schema(descriptor: AttributeDescriptor): Type = {
160     val bindings = ObjectType.selectType(descriptor)
161     val builder = bindings.head match {
162       case ObjectType.GEOMETRY => geometry(bindings(1))
163       case ObjectType.LIST     => Binding(bindings(1)).list()
164       case ObjectType.MAP      => Binding(bindings(1)).key(bindings(2))
165       case p                   => Binding(p).primitive()
166     }
167     builder.named(StringSerialization.alphaNumericSafeString(descriptor.getLocalName))
168   }
169 
170   /**
171     * Create a builder for a parquet geometry field
172     *
173     * @param binding geometry type
174     * @return
175     */
176   private def geometry(binding: ObjectType): Types.Builder[_, _ <: Type] = {
177     def group: Types.GroupBuilder[GroupType] = Types.buildGroup(Repetition.OPTIONAL)
178     binding match {
179       case ObjectType.POINT =>
180         group.id(GeometryBytes.TwkbPoint)
181             .required(PrimitiveTypeName.DOUBLE).named(GeometryColumnX)
182             .required(PrimitiveTypeName.DOUBLE).named(GeometryColumnY)
183 
184       case ObjectType.LINESTRING =>
185         group.id(GeometryBytes.TwkbLineString)
186             .repeated(PrimitiveTypeName.DOUBLE).named(GeometryColumnX)
187             .repeated(PrimitiveTypeName.DOUBLE).named(GeometryColumnY)
188 
189       case ObjectType.MULTIPOINT =>
190         group.id(GeometryBytes.TwkbMultiPoint)
191             .repeated(PrimitiveTypeName.DOUBLE).named(GeometryColumnX)
192             .repeated(PrimitiveTypeName.DOUBLE).named(GeometryColumnY)
193 
194       case ObjectType.POLYGON =>
195         group.id(GeometryBytes.TwkbPolygon)
196             .requiredList().element(PrimitiveTypeName.DOUBLE, Repetition.REPEATED).named(GeometryColumnX)
197             .requiredList().element(PrimitiveTypeName.DOUBLE, Repetition.REPEATED).named(GeometryColumnY)
198 
199       case ObjectType.MULTILINESTRING =>
200         group.id(GeometryBytes.TwkbMultiLineString)
201             .requiredList().element(PrimitiveTypeName.DOUBLE, Repetition.REPEATED).named(GeometryColumnX)
202             .requiredList().element(PrimitiveTypeName.DOUBLE, Repetition.REPEATED).named(GeometryColumnY)
203 
204       case ObjectType.MULTIPOLYGON =>
205         group.id(GeometryBytes.TwkbMultiPolygon)
206             .requiredList().requiredListElement().element(PrimitiveTypeName.DOUBLE, Repetition.REPEATED).named(GeometryColumnX)
207             .requiredList().requiredListElement().element(PrimitiveTypeName.DOUBLE, Repetition.REPEATED).named(GeometryColumnY)
208 
209       case ObjectType.GEOMETRY =>
210         Types.primitive(PrimitiveTypeName.BINARY, Repetition.OPTIONAL)
211 
212       case _ => throw new NotImplementedError(s"No mapping defined for geometry type $binding")
213     }
214   }

216   /**
217     * Defined parquet field, applicable as primitives, list elements, or map key/values
218     *
219     * @param name type
220     * @param as original type
221     * @param length fixed length
222     */
223   class Binding(private val name: PrimitiveTypeName, as: Option[OriginalType] = None, length: Option[Int] = None) {
224 
225     def primitive(): Types.Builder[_, _ <: Type] = opts(Types.primitive(name, Repetition.OPTIONAL))
226 
227     def list(): Types.Builder[_, _ <: Type] = opts(Types.optionalList().optionalElement(name))
228 
229     def key(value: ObjectType): Types.Builder[_, _ <: Type] = {
230       val key = opts(Types.optionalMap().key(name))
231       val values = Binding(value)
232       values.opts(key.optionalValue(values.name))
233     }
234 
235     private def opts[T <: BasePrimitiveBuilder[_ <: Type, _]](b: T): T = {
236       length.foreach(b.length)
237       as.foreach(b.as)
238       b
239     }
240   }
241 
242   object Binding {
243 
244     private val bindings = Map(
245       ObjectType.DATE    -> new Binding(PrimitiveTypeName.INT64,  Some(OriginalType.TIMESTAMP_MILLIS)),
246       ObjectType.STRING  -> new Binding(PrimitiveTypeName.BINARY, Some(OriginalType.UTF8)),
247       ObjectType.INT     -> new Binding(PrimitiveTypeName.INT32),
248       ObjectType.DOUBLE  -> new Binding(PrimitiveTypeName.DOUBLE),
249       ObjectType.LONG    -> new Binding(PrimitiveTypeName.INT64),
250       ObjectType.FLOAT   -> new Binding(PrimitiveTypeName.FLOAT),
251       ObjectType.BOOLEAN -> new Binding(PrimitiveTypeName.BOOLEAN),
252       ObjectType.BYTES   -> new Binding(PrimitiveTypeName.BINARY),
253       ObjectType.UUID    -> new Binding(PrimitiveTypeName.FIXED_LEN_BYTE_ARRAY, None, Some(16))
254     )
255 
256     def apply(binding: ObjectType): Binding =
257       bindings.getOrElse(binding, throw new NotImplementedError(s"No mapping defined for type $binding"))
258   }
