75   /**
76     * Extract the simple feature type from a parquet read context. The read context
77     * contains both file metadata and the provided read conf
78     *
79     * @param context parquet read context
80     * @return
81     */
82   def read(context: InitContext): Option[SimpleFeatureParquetSchema] = {
83     val metadata = new java.util.HashMap[String, String]()
84     // copy in the file level metadata
85     context.getKeyValueMetadata.asScala.foreach { case (k, v) => if (!v.isEmpty) { metadata.put(k, v.iterator.next) }}
86     val conf = context.getConfiguration
87     // copy in the sft from the conf - overwrite the file level metadata as this has our transform schema
88     Seq(StorageConfiguration.SftNameKey, StorageConfiguration.SftSpecKey, SchemaVersionKey).foreach { key =>
89       val value = conf.get(key)
90       if (value != null) {
91         metadata.put(key, value)
92       }
93     }
94     apply(metadata)
95   }

158   /**
159     * Create a parquet field type from an attribute descriptor
160     *
161     * @param descriptor descriptor
162     * @return
163     */
164   private def schema(descriptor: AttributeDescriptor): Type = {
165     val bindings = ObjectType.selectType(descriptor)
166     val builder = bindings.head match {
167       case ObjectType.GEOMETRY => geometry(bindings(1))
168       case ObjectType.LIST     => Binding(bindings(1)).list()
169       case ObjectType.MAP      => Binding(bindings(1)).key(bindings(2))
170       case p                   => Binding(p).primitive()
171     }
172     builder.named(StringSerialization.alphaNumericSafeString(descriptor.getLocalName))

175   /**
176     * Create a builder for a parquet geometry field
177     *
178     * @param binding geometry type
179     * @return
180     */
181   private def geometry(binding: ObjectType): Types.Builder[_, _ <: Type] = {
182     def group: Types.GroupBuilder[GroupType] = Types.buildGroup(Repetition.OPTIONAL)
183     binding match {
184       case ObjectType.POINT =>
185         group.id(GeometryBytes.TwkbPoint)
186             .required(PrimitiveTypeName.DOUBLE).named(GeometryColumnX)
187             .required(PrimitiveTypeName.DOUBLE).named(GeometryColumnY)
188 
189       case ObjectType.LINESTRING =>
190         group.id(GeometryBytes.TwkbLineString)
191             .repeated(PrimitiveTypeName.DOUBLE).named(GeometryColumnX)
192             .repeated(PrimitiveTypeName.DOUBLE).named(GeometryColumnY)
193 
194       case ObjectType.MULTIPOINT =>
195         group.id(GeometryBytes.TwkbMultiPoint)
196             .repeated(PrimitiveTypeName.DOUBLE).named(GeometryColumnX)
197             .repeated(PrimitiveTypeName.DOUBLE).named(GeometryColumnY)
198 
199       case ObjectType.POLYGON =>
200         group.id(GeometryBytes.TwkbPolygon)
201             .requiredList().element(PrimitiveTypeName.DOUBLE, Repetition.REPEATED).named(GeometryColumnX)
202             .requiredList().element(PrimitiveTypeName.DOUBLE, Repetition.REPEATED).named(GeometryColumnY)
203 
204       case ObjectType.MULTILINESTRING =>
205         group.id(GeometryBytes.TwkbMultiLineString)
206             .requiredList().element(PrimitiveTypeName.DOUBLE, Repetition.REPEATED).named(GeometryColumnX)
207             .requiredList().element(PrimitiveTypeName.DOUBLE, Repetition.REPEATED).named(GeometryColumnY)
208 
209       case ObjectType.MULTIPOLYGON =>
210         group.id(GeometryBytes.TwkbMultiPolygon)
211             .requiredList().requiredListElement().element(PrimitiveTypeName.DOUBLE, Repetition.REPEATED).named(GeometryColumnX)
212             .requiredList().requiredListElement().element(PrimitiveTypeName.DOUBLE, Repetition.REPEATED).named(GeometryColumnY)
213 
214       case ObjectType.GEOMETRY =>
215         Types.primitive(PrimitiveTypeName.BINARY, Repetition.OPTIONAL)
216 
217       case _ => throw new NotImplementedError(s"No mapping defined for geometry type $binding")
218     }

221   /**
222     * Defined parquet field, applicable as primitives, list elements, or map key/values
223     *
224     * @param name type
225     * @param as original type
226     * @param length fixed length
227     */
228   class Binding(private val name: PrimitiveTypeName, as: Option[OriginalType] = None, length: Option[Int] = None) {
229 
230     def primitive(): Types.Builder[_, _ <: Type] = opts(Types.primitive(name, Repetition.OPTIONAL))
231 
232     def list(): Types.Builder[_, _ <: Type] = opts(Types.optionalList().optionalElement(name))
233 
234     def key(value: ObjectType): Types.Builder[_, _ <: Type] = {
235       val key = opts(Types.optionalMap().key(name))
236       val values = Binding(value)
237       values.opts(key.optionalValue(values.name))
238     }
239 
240     private def opts[T <: BasePrimitiveBuilder[_ <: Type, _]](b: T): T = {
241       length.foreach(b.length)
242       as.foreach(b.as)
243       b
244     }
245   }

290     /**
291      * Generate json describing the GeoParquet file, conforming to the 1.1.0 GeoParquet schema
292      *
293      * @param primaryGeometry name of the primary geometry in the file
294      * @param geometries pairs of geometries and optional bounds
295      * @return
296      */
297     def apply(primaryGeometry: String, geometries: Seq[(GeometryDescriptor, Option[Envelope])]): String = {
298       val cols = geometries.map { case (descriptor, bounds) =>
299         val (encoding, types) = descriptor.getType.getBinding match {
300           case b if b == classOf[Geometry] => ("WKB", Collections.emptyList())
301           case b if b == classOf[GeometryCollection] => ("WKB", Collections.singletonList(classOf[GeometryCollection].getSimpleName))
302           case b => (b.getSimpleName.toLowerCase(Locale.US), Collections.singletonList(b.getSimpleName))
303         }
304         val metadata = new java.util.HashMap[String, AnyRef]
305         metadata.put("encoding", encoding)
306         metadata.put("geometry_types", types) // TODO add ' Z' for 3d points
307         bounds.filterNot(_.isNull).foreach { e =>
308           metadata.put("bbox", java.util.List.of(e.getMinX, e.getMinY, e.getMaxX, e.getMaxY)) // TODO add z for 3d points
309         }
310         StringSerialization.alphaNumericSafeString(descriptor.getLocalName) -> metadata
311       }
312 
313       val model = java.util.Map.of(
314         "version", "1.1.0",
315         "primary_column", primaryGeometry,
316         "columns", cols.toMap.asJava
317       )
318 
319       gson.toJson(model)
320     }

322     /**
323      * Observer class that generates GeoParquet metadata
324      *
325      * @param sft simple feature type
326      */
327     class GeoParquetObserver(sft: SimpleFeatureType) extends FileSystemObserver {
328 
329       import scala.collection.JavaConverters._
330 
331       private val bounds =
332         sft.getAttributeDescriptors.asScala.zipWithIndex.collect {
333           case (d: GeometryDescriptor, i) if supported(d) => (d, i, new Envelope())
334         }
335 
336       def metadata(): java.util.Map[String, String] = {
337         if (bounds.isEmpty) { Collections.emptyMap() } else {
338           val geoms = bounds.map { case (d, _, env) => (d, Some(env).filterNot(_.isNull)) }
339           val primary = bounds.find(_._1 == sft.getGeometryDescriptor).getOrElse(bounds.head)._1.getLocalName
340           Collections.singletonMap(GeoParquetMetadataKey, GeoParquetMetadata(primary, geoms.toSeq))
341         }
342       }
343 
344       override def write(feature: SimpleFeature): Unit = {
345         bounds.foreach { case (_, i, envelope) =>
346           val geom = feature.getAttribute(i).asInstanceOf[Geometry]
347           if (geom != null) {
348             envelope.expandToInclude(geom.getEnvelopeInternal)
349           }
350         }
351       }

354       override def close(): Unit = {}
355     }
