30 /**
31  * A paired simple feature type and parquet schema
32  *
33  * @param sft simple feature type
34  * @param encodings type encoding
35  * @param hasVisibilities whether the schema encodes visibilities, or not
36  * @param boundingBoxes names of any bounding box fields, along with the original field name
37  * @param metadata file metadata
38  */
39 case class SimpleFeatureParquetSchema(
40     sft: SimpleFeatureType,
41     encodings: Encodings,
42     hasVisibilities: Boolean,
43     boundingBoxes: Seq[BoundingBoxField],
44     metadata: java.util.Map[String, String]) {

51   /**
52     * Gets the name of the parquet field for the given simple feature type attribute
53     *
54     * @param i index of the sft attribute

101     * Extract the simple feature type from a parquet read context. The read context
102     * contains both file metadata and the provided read conf
103     *
104     * @param context parquet read context
105     * @return
106     */
107   def read(context: InitContext): Option[SimpleFeatureParquetSchema] = {
108     val metadata = new java.util.HashMap[String, String]()
109     // copy in the file level metadata - note, do this before copying in the conf so that transform schemas are applied correctly
110     context.getKeyValueMetadata.asScala.foreach { case (k, v) => if (!v.isEmpty) { metadata.put(k, v.iterator.next) }}
111     val conf = context.getParquetConfiguration
112     // noinspection ScalaDeprecation
113     Seq(SftNameKey, SftSpecKey, GeometryEncodingKey, SchemaVersionKey).foreach { key =>
114       Option(conf.get(key)).foreach(metadata.put(key, _))
115     }
116     read(context.getFileSchema, metadata)
117   }

149   /**
150    * Get the schema for a file being read
151    *
152    * @param fileSchema message type
153    * @param metadata file metadata
154    * @return
155    */
156   private def read(fileSchema: MessageType, metadata: java.util.Map[String, String]): Option[SimpleFeatureParquetSchema] = {
157     for {
158       name <- Option(metadata.get(SftNameKey))
159       spec <- Option(metadata.get(SftSpecKey))
160     } yield {
161       val sft = SimpleFeatureTypes.createType(name, spec)
162       val geometries = Option(metadata.get(GeometryEncodingKey)).map(GeometryEncoding.apply).getOrElse {
163         // noinspection ScalaDeprecation
164         if (metadata.get(SchemaVersionKey) == "1") { GeometryEncoding.GeoMesaV1 } else { GeometryEncoding.GeoMesaV0 }
165       }
166       val encodings = Encodings(geometries)
167       val bboxes = fileSchema.getFields.asScala.map(_.getName).flatMap(BoundingBoxField.fromBoundingBox).toSeq
168       val visibilities = fileSchema.containsField(VisibilitiesField)
169       SimpleFeatureParquetSchema(sft, encodings, visibilities, bboxes, Collections.unmodifiableMap(metadata))
170     }
171   }

173   /**
174    * Gets a schema for writing
175    *
176    * @param sftName sft name config
177    * @param sftSpec sft spec config
178    * @param geoms geometry encoding config
179    * @param bboxOpt bounding box encoding config
180    * @param vis visibility config
181    * @return
182    */
183   private def write(sftName: String, sftSpec: String, geoms: String, bboxOpt: String, vis: String): Option[SimpleFeatureParquetSchema] = {
184     for {
185       name <- Option(sftName)
186       spec <- Option(sftSpec)
187     } yield {
188       val sft = SimpleFeatureTypes.createType(name, spec)
189       val geometries = Option(geoms).map(GeometryEncoding.apply).getOrElse(GeometryEncoding.GeoParquetWkb)
190       // only include bboxes if they help with push-down filters
191       val bboxes = if (bboxOpt != null && !bboxOpt.toBoolean) { Seq.empty } else { BoundingBoxField(sft, geometries) }
192       val visibilities = Option(sft.getUserData.get(VisibilityEncodingKey)).orElse(Option(vis)).forall(_.toString.toBoolean)
193       val metadata = new java.util.HashMap[String, String]()
194       metadata.put(SftNameKey, name)
195       metadata.put(SftSpecKey, spec)
196       metadata.put(GeometryEncodingKey, geometries.toString)
197       if (geometries == GeometryEncoding.GeoMesaV1) {
198         // noinspection ScalaDeprecation
199         metadata.put(SchemaVersionKey, "1") // include schema version for back-compatibility when possible
200       }
201       val encodings = Encodings(geometries)
202       SimpleFeatureParquetSchema(sft, encodings, visibilities, bboxes, Collections.unmodifiableMap(metadata))
203     }
204   }

206   /**
207    * Get the message type for a simple feature type
208    *
209    * @param sft simple feature type
210    * @param encodings field type encoding
211    * @param bboxes include bounding boxes for each row
212    * @param visibilities include visibilities
213    * @return
214    */
215   private def schema(sft: SimpleFeatureType, encodings: Encodings, bboxes: Seq[String], visibilities: Boolean): MessageType = {
216     val isV0 = encodings.geometry == GeometryEncoding.GeoMesaV0
217     val id = {
218       val primitive = PrimitiveTypeName.BINARY
219       val builder = if (isV0) { Types.primitive(primitive, Repetition.REPEATED) } else { Types.required(primitive) }
220       builder.as(LogicalTypeAnnotation.stringType()).named(FeatureIdField)
221     }
222     val vis = if (!visibilities) { None } else {
223       Some(Types.optional(PrimitiveTypeName.BINARY).as(LogicalTypeAnnotation.stringType()).named(VisibilitiesField))
224     }
225     val boundingBoxes = bboxes.map(BoundingBoxField.schema)
226     // note: id field goes at the end of the record, then vis, then any bounding boxes
227     val fields = (sft.getAttributeDescriptors.asScala.map(schema(_, encodings)) :+ id) ++ vis ++ boundingBoxes
228     val name = if (isV0) { sft.getTypeName } else { alphaNumericSafeString(sft.getTypeName) }
229     new MessageType(name, fields.asJava)
230   }

232   /**
233    * Create a parquet field type from an attribute descriptor
234    *
235    * @param descriptor descriptor
236    * @param encodings field type encoding
237    * @return
238    */
239   private def schema(descriptor: AttributeDescriptor, encodings: Encodings): Type = {
240     val isV0 = encodings.geometry == GeometryEncoding.GeoMesaV0
241     val name = if (isV0) { descriptor.getLocalName } else { alphaNumericSafeString(descriptor.getLocalName) }

243   }

245   /**
246    * Builds the schema type for an attribute
247    *
248    * @param name field name
249    * @param bindings object type
250    * @param encodings schema encodings
251    * @param repetition repetition
252    * @return
253    */
254   private def buildType(
255       name: String,
256       bindings: Seq[ObjectType],
257       encodings: Encodings,
258       repetition: Repetition = Repetition.OPTIONAL): Type = {
259     val builder = bindings.head match {
260       case ObjectType.STRING   => Types.primitive(PrimitiveTypeName.BINARY, repetition).as(LogicalTypeAnnotation.stringType())
261       case ObjectType.INT      => Types.primitive(PrimitiveTypeName.INT32, repetition)
262       case ObjectType.DOUBLE   => Types.primitive(PrimitiveTypeName.DOUBLE, repetition)
263       case ObjectType.LONG     => Types.primitive(PrimitiveTypeName.INT64, repetition)
264       case ObjectType.FLOAT    => Types.primitive(PrimitiveTypeName.FLOAT, repetition)
265       case ObjectType.BOOLEAN  => Types.primitive(PrimitiveTypeName.BOOLEAN, repetition)
266       case ObjectType.BYTES    => Types.primitive(PrimitiveTypeName.BINARY, repetition)
267 
268       case ObjectType.DATE =>
269         val unit = encodings.date match {
270           case DateEncoding.Micros => TimeUnit.MICROS
271           case DateEncoding.Millis => TimeUnit.MILLIS
272           case _ => throw new UnsupportedOperationException(encodings.date.toString)
273         }
274         Types.primitive(PrimitiveTypeName.INT64, repetition).as(LogicalTypeAnnotation.timestampType(true, unit))
275 
276       case ObjectType.UUID =>
277         encodings.uuid match {
278           case UuidEncoding.FixedLength =>
279             Types.primitive(PrimitiveTypeName.FIXED_LEN_BYTE_ARRAY, repetition).length(16).as(LogicalTypeAnnotation.uuidType())
280           case UuidEncoding.Bytes => Types.primitive(PrimitiveTypeName.BINARY, repetition)
281           case _ => throw new UnsupportedOperationException(encodings.uuid.toString)
282         }
283 
284       case ObjectType.LIST =>
285         encodings.list match {
286           case ListEncoding.ThreeLevel =>
287             Types.optionalList().requiredGroupElement().addField(buildType("element", bindings.drop(1), encodings, Repetition.REQUIRED))
288           case ListEncoding.TwoLevel => Types.optionalList().element(buildType("element", bindings.drop(1), encodings))
289           case _ => throw new UnsupportedOperationException(encodings.list.toString)
290         }
291 
292       case ObjectType.MAP =>
293         Types.optionalMap()
294           .key(buildType("key", bindings.slice(1, 2), encodings, Repetition.REQUIRED))
295           .value(buildType("value", bindings.slice(2, 3), encodings))
296 
297       case ObjectType.GEOMETRY =>
298         GeometrySchema(bindings(1), encodings.geometry)
299 
300       case binding =>
301         throw new UnsupportedOperationException(s"No mapping defined for type $binding")
302     }
303     builder.named(name)
304   }
