31 /**
32  * A paired simple feature type and parquet schema
33  *
34  * @param sft simple feature type
35  * @param encodings type encoding
36  * @param hasVisibilities whether the schema encodes visibilities, or not
37  * @param boundingBoxes names of any bounding box fields, along with the original field name
38  * @param metadata file metadata
39  */
40 case class SimpleFeatureParquetSchema(
41     sft: SimpleFeatureType,
42     encodings: Encodings,
43     hasVisibilities: Boolean,
44     boundingBoxes: Seq[BoundingBoxField],
45     metadata: java.util.Map[String, String]) {

48    * Parquet message schema
49    */
50   val schema: MessageType = SimpleFeatureParquetSchema.schema(sft, encodings, boundingBoxes.map(_.bbox), hasVisibilities)

53     * Gets the name of the parquet field for the given simple feature type attribute
54     *
55     * @param i index of the sft attribute
56     * @return
57     */

89     * Extract the simple feature type from a parquet read context. The read context
90     * contains both file metadata and the provided read conf
91     *
92     * @param context parquet read context
93     * @return
94     */
95   def read(context: InitContext): Option[SimpleFeatureParquetSchema] = {
96     val metadata = new java.util.HashMap[String, String]()
97     // copy in the file level metadata - note, do this before copying in the conf so that transform schemas are applied correctly
98     context.getKeyValueMetadata.asScala.foreach { case (k, v) => if (!v.isEmpty) { metadata.put(k, v.iterator.next) }}
99     val conf = context.getParquetConfiguration
100     // noinspection ScalaDeprecation
101     Seq(SftNameKey, SftSpecKey, GeometryEncodingKey, SchemaVersionKey).foreach { key =>
102       Option(conf.get(key)).foreach(metadata.put(key, _))
103     }
104     read(context.getFileSchema, metadata)
105   }

137   /**
138    * Get the schema for a file being read
139    *
140    * @param fileSchema message type
141    * @param metadata file metadata
142    * @return
143    */
144   private def read(fileSchema: MessageType, metadata: java.util.Map[String, String]): Option[SimpleFeatureParquetSchema] = {
145     for {
146       name <- Option(metadata.get(SftNameKey))
147       spec <- Option(metadata.get(SftSpecKey))
148     } yield {
149       val sft = SimpleFeatureTypes.createType(name, spec)
150       val geometries = Option(metadata.get(GeometryEncodingKey)).map(GeometryEncoding.apply).getOrElse {
151         // noinspection ScalaDeprecation
152         if (metadata.get(SchemaVersionKey) == "1") { GeometryEncoding.GeoMesaV1 } else { GeometryEncoding.GeoMesaV0 }
153       }
154       val encodings = Encodings(geometries)
155       val bboxes = fileSchema.getFields.asScala.map(_.getName).flatMap(BoundingBoxField.fromBoundingBox).toSeq
156       val visibilities = fileSchema.containsField(VisibilitiesField)
157       SimpleFeatureParquetSchema(sft, encodings, visibilities, bboxes, Collections.unmodifiableMap(metadata))
158     }
159   }

161   /**
162    * Gets a schema for writing
163    *
164    * @param sftName sft name config
165    * @param sftSpec sft spec config
166    * @param geoms geometry encoding config
167    * @param bboxOpt bounding box encoding config
168    * @param vis visibility config
169    * @return
170    */
171   private def write(sftName: String, sftSpec: String, geoms: String, bboxOpt: String, vis: String): Option[SimpleFeatureParquetSchema] = {
172     for {
173       name <- Option(sftName)
174       spec <- Option(sftSpec)
175     } yield {
176       val sft = SimpleFeatureTypes.createType(name, spec)
177       val geometries = Option(geoms).map(GeometryEncoding.apply).getOrElse(GeometryEncoding.GeoParquetWkb)
178       // only include bboxes if they help with push-down filters
179       val bboxes = if (bboxOpt != null && !bboxOpt.toBoolean) { Seq.empty } else { BoundingBoxField(sft, geometries) }
180       val visibilities = Option(sft.getUserData.get(VisibilityEncodingKey)).orElse(Option(vis)).forall(_.toString.toBoolean)
181       val metadata = new java.util.HashMap[String, String]()
182       metadata.put(SftNameKey, name)
183       metadata.put(SftSpecKey, spec)
184       metadata.put(GeometryEncodingKey, geometries.toString)
185       if (geometries == GeometryEncoding.GeoMesaV1) {
186         // noinspection ScalaDeprecation
187         metadata.put(SchemaVersionKey, "1") // include schema version for back-compatibility when possible
188       }
189       val encodings = Encodings(geometries)
190       SimpleFeatureParquetSchema(sft, encodings, visibilities, bboxes, Collections.unmodifiableMap(metadata))
191     }
192   }

194   /**
195    * Get the message type for a simple feature type
196    *
197    * @param sft simple feature type
198    * @param encodings field type encoding
199    * @param bboxes include bounding boxes for each row
200    * @param visibilities include visibilities
201    * @return
202    */
203   private def schema(sft: SimpleFeatureType, encodings: Encodings, bboxes: Seq[String], visibilities: Boolean): MessageType = {
204     val isV0 = encodings.geometry == GeometryEncoding.GeoMesaV0
205     val id = {
206       val primitive = PrimitiveTypeName.BINARY
207       val builder = if (isV0) { Types.primitive(primitive, Repetition.REPEATED) } else { Types.required(primitive) }
208       builder.as(LogicalTypeAnnotation.stringType()).named(FeatureIdField)
209     }
210     val vis = if (!visibilities) { None } else {
211       Some(Types.optional(PrimitiveTypeName.BINARY).as(LogicalTypeAnnotation.stringType()).named(VisibilitiesField))
212     }
213     val boundingBoxes = bboxes.map(BoundingBoxField.schema)
214     // note: id field goes at the end of the record, then vis, then any bounding boxes
215     val fields = (sft.getAttributeDescriptors.asScala.map(schema(_, encodings)) :+ id) ++ vis ++ boundingBoxes
216     val name = if (isV0) { sft.getTypeName } else { alphaNumericSafeString(sft.getTypeName) }
217     new MessageType(name, fields.asJava)
218   }

220   /**

222    *
223    * @param descriptor descriptor
224    * @param encodings field type encoding

227   private def schema(descriptor: AttributeDescriptor, encodings: Encodings): Type = {

233   /**
234    * Builds the schema type for an attribute
235    *
236    * @param name field name
237    * @param bindings object type
238    * @param encodings schema encodings
239    * @param repetition repetition
240    * @return
241    */
242   private def buildType(
243       name: String,
244       bindings: Seq[ObjectType],
245       encodings: Encodings,
246       repetition: Repetition = Repetition.OPTIONAL): Type = {
247     val builder = bindings.head match {
248       case ObjectType.STRING   => Types.primitive(PrimitiveTypeName.BINARY, repetition).as(LogicalTypeAnnotation.stringType())
249       case ObjectType.INT      => Types.primitive(PrimitiveTypeName.INT32, repetition)
250       case ObjectType.DOUBLE   => Types.primitive(PrimitiveTypeName.DOUBLE, repetition)
251       case ObjectType.LONG     => Types.primitive(PrimitiveTypeName.INT64, repetition)
252       case ObjectType.FLOAT    => Types.primitive(PrimitiveTypeName.FLOAT, repetition)
253       case ObjectType.BOOLEAN  => Types.primitive(PrimitiveTypeName.BOOLEAN, repetition)
254       case ObjectType.BYTES    => Types.primitive(PrimitiveTypeName.BINARY, repetition)
255 
256       case ObjectType.DATE =>
257         val unit = encodings.date match {
258           case DateEncoding.Micros => TimeUnit.MICROS
259           case DateEncoding.Millis => TimeUnit.MILLIS
260           case _ => throw new NotImplementedError(encodings.date.toString)
261         }
262         Types.primitive(PrimitiveTypeName.INT64, repetition).as(LogicalTypeAnnotation.timestampType(true, unit))
263 
264       case ObjectType.UUID =>
265         encodings.uuid match {
266           case UuidEncoding.FixedLength =>
267             Types.primitive(PrimitiveTypeName.FIXED_LEN_BYTE_ARRAY, repetition).length(16).as(LogicalTypeAnnotation.uuidType())
268           case UuidEncoding.Bytes => Types.primitive(PrimitiveTypeName.BINARY, repetition)
269           case _ => throw new NotImplementedError(encodings.uuid.toString)
270         }
271 
272       case ObjectType.LIST =>
273         encodings.list match {
274           case ListEncoding.ThreeLevel =>
275             Types.optionalList().requiredGroupElement().addField(buildType("element", bindings.drop(1), encodings, Repetition.REQUIRED))
276           case ListEncoding.TwoLevel => Types.optionalList().element(buildType("element", bindings.drop(1), encodings))
277           case _ => throw new NotImplementedError(encodings.list.toString)
278         }
279 
280       case ObjectType.MAP =>
281         Types.optionalMap()
282           .key(buildType("key", bindings.slice(1, 2), encodings, Repetition.REQUIRED))
283           .value(buildType("value", bindings.slice(2, 3), encodings))
284 
285       case ObjectType.GEOMETRY =>
286         GeometrySchema(bindings(1), encodings.geometry)
287 
288       case binding =>
289         throw new NotImplementedError(s"No mapping defined for type $binding")
290     }
291     builder.named(name)
292   }
