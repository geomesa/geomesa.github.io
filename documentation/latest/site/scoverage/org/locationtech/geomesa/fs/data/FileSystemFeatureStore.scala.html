33 class FileSystemFeatureStore(
34     val storage: FileSystemStorage,
35     entry: ContentEntry,
36     query: Query,
37     readThreads: Int,
38     writeTimeout: Duration,
39     queryTimeout: Option[Duration],
40   ) extends ContentFeatureStore(entry, query) with LazyLogging {
41 
42   private val sft = storage.metadata.sft
43 
44   override def getWriterInternal(query: Query, flags: Int): FeatureWriter[SimpleFeatureType, SimpleFeature] = {
45     // note: check update first as sometimes we get ADD | UPDATE
46     if ((flags & WRITER_UPDATE) == WRITER_UPDATE) {
47       new FileSystemFeatureWriterModify(storage, sft, query.getFilter, readThreads)
48     } else if ((flags & WRITER_ADD) == WRITER_ADD) {
49       new FileSystemFeatureWriterAppend(storage, sft, writeTimeout)
50     } else {
51       throw new IllegalArgumentException(s"Expected one of $WRITER_ADD or $WRITER_UPDATE, but got: $flags")
52     }
53   }
54 
55   override def buildFeatureType(): SimpleFeatureType = sft
56 
57   override def getBoundsInternal(query: Query): ReferencedEnvelope = {
58     val envelope = new ReferencedEnvelope(org.locationtech.geomesa.utils.geotools.CRS_EPSG_4326)
59     storage.getPartitions(query.getFilter).foreach { partition =>
60       partition.bounds.foreach(b => envelope.expandToInclude(b.envelope))
61     }
62     envelope
63   }
64 
65   override def getCountInternal(query: Query): Int =
66     storage.getPartitions(query.getFilter).map(_.count).sum.toInt
67 
68   override def getReaderInternal(original: Query): FeatureReader[SimpleFeatureType, SimpleFeature] = {
69     import org.locationtech.geomesa.index.conf.QueryHints._
70 
71     val query = new Query(original)
72     // The type name can sometimes be empty such as Query.ALL
73     query.setTypeName(sft.getTypeName)
74 
75     val reader = queryTimeout match {
76       case None => storage.getReader(query, threads = readThreads)
77       case Some(timeout) =>
78         val filter = Option(query.getFilter).filter(_ != Filter.INCLUDE)
79         new ManagedScan(new FileSystemScanner(storage, query, readThreads), Timeout(timeout), query.getTypeName, filter)
80     }
81     // get a closeable java iterator that DelegateSimpleFeatureIterator will process correctly
82     val iter = new FileSystemFeatureIterator(reader)
83 
84     // transforms will be set after getting the iterator
85     val transformSft = query.getHints.getTransformSchema.getOrElse(sft)
86 
87     // note: DelegateSimpleFeatureIterator will close the iterator by checking that it implements Closeable
88     new DelegateSimpleFeatureReader(transformSft, new DelegateSimpleFeatureIterator(iter))
89   }

115   private class FileSystemScanner(storage: FileSystemStorage, val query: Query, threads: Int)
116       extends LowLevelScanner[SimpleFeature] {
117 
118     private var reader: CloseableFeatureIterator = _
119 
120     override def iterator: Iterator[SimpleFeature] = synchronized {
121       reader = storage.getReader(query, threads = threads)
122       reader
123     }
124     override def close(): Unit = synchronized {
125       if (reader != null) {
126         reader.close()
127       }
128     }
129   }

131   /**
132     * Iterator for querying file system storage
133     *
134     * Note: implements Closeable and not AutoCloseable so that DelegateFeatureIterator will close it properly
135     *
136     * @param iter delegate iterator
137     */
138   class FileSystemFeatureIterator(iter: CloseableFeatureIterator)
139       extends java.util.Iterator[SimpleFeature] with Closeable {
140     override def hasNext: Boolean = iter.hasNext
141     override def next(): SimpleFeature = iter.next()
142     override def close(): Unit = iter.close()
143   }
144 
145   /**
146     * Appending feature writer
147     *
148     * @param storage storage instance
149     * @param sft simple feature type
150     * @param timeout write timeout, for flushing partitions
151     */
152   class FileSystemFeatureWriterAppend(storage: FileSystemStorage, sft: SimpleFeatureType, timeout: Duration)
153       extends FastSettableFeatureWriter with LazyLogging {
154 
155     private val removalListener = new RemovalListener[String, Closeable with Flushable]() {
156       override def onRemoval(key: String, value: Closeable with Flushable, cause: RemovalCause): Unit = {
157         if (cause == RemovalCause.EXPIRED) {
158           logger.info(s"Flushing writer for partition: $key")
159           FlushWithLogging(value)
160           CloseWithLogging(value)
161         }
162       }
163     }
164 
165     private val writers =
166       Caffeine.newBuilder()
167         .expireAfterAccess(timeout.toMillis, TimeUnit.MILLISECONDS)
168         .removalListener[String, FileSystemWriter](removalListener)
169         .build(new CacheLoader[String, FileSystemWriter]() {
170           override def load(partition: String): FileSystemWriter = storage.getWriter(partition)
171         })
172 
173     private val featureIds = new AtomicLong(0)
174     private var feature: ScalaSimpleFeature = _
175 
176     override def getFeatureType: SimpleFeatureType = sft
177 
178     override def hasNext: Boolean = false
179 
180     override def next(): ScalaSimpleFeature = {
181       feature = new ScalaSimpleFeature(sft, featureIds.getAndIncrement().toString)
182       feature
183     }
184 
185     override def write(): Unit = {
186       val sf = GeoMesaFeatureWriter.featureWithFid(feature)
187       writers.get(storage.metadata.scheme.getPartitionName(sf)).write(sf)
188       feature = null
189     }
190 
191     override def remove(): Unit = throw new IllegalArgumentException("This writer is append only")
192 
193     override def close(): Unit = {
194       val values = writers.asMap().values().asScala.toSeq
195       val flush = FlushQuietly(values)
196       CloseQuietly(values) match {
197         case None => flush.foreach(e => throw e)
198         case Some(e) => flush.foreach(e.addSuppressed); throw e
199       }
200     }
201   }
202

227     override def close(): Unit = writer.close()
