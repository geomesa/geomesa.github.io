32 class FileSystemFeatureStore(
33     val storage: FileSystemStorage,
34     entry: ContentEntry,
35     query: Query,
36     readThreads: Int,
37     writeTimeout: Duration
38   ) extends ContentFeatureStore(entry, query) with LazyLogging {
39 
40   private val sft = storage.metadata.sft
41 
42   override def getWriterInternal(query: Query, flags: Int): FeatureWriter[SimpleFeatureType, SimpleFeature] = {
43     // note: check update first as sometimes we get ADD | UPDATE
44     if ((flags & WRITER_UPDATE) == WRITER_UPDATE) {
45       new FileSystemFeatureWriterModify(storage, sft, query.getFilter, readThreads)
46     } else if ((flags & WRITER_ADD) == WRITER_ADD) {
47       new FileSystemFeatureWriterAppend(storage, sft, writeTimeout)
48     } else {
49       throw new IllegalArgumentException(s"Expected one of $WRITER_ADD or $WRITER_UPDATE, but got: $flags")
50     }
51   }
52 
53   override def buildFeatureType(): SimpleFeatureType = sft
54 
55   override def getBoundsInternal(query: Query): ReferencedEnvelope = {
56     val envelope = new ReferencedEnvelope(org.locationtech.geomesa.utils.geotools.CRS_EPSG_4326)
57     storage.getPartitions(query.getFilter).foreach { partition =>
58       partition.bounds.foreach(b => envelope.expandToInclude(b.envelope))
59     }
60     envelope
61   }
62 
63   override def getCountInternal(query: Query): Int =
64     storage.getPartitions(query.getFilter).map(_.count).sum.toInt
65 
66   override def getReaderInternal(original: Query): FeatureReader[SimpleFeatureType, SimpleFeature] = {
67     import org.locationtech.geomesa.index.conf.QueryHints._
68 
69     val query = new Query(original)
70     // The type name can sometimes be empty such as Query.ALL
71     query.setTypeName(sft.getTypeName)
72 
73     // get a closeable java iterator that DelegateSimpleFeatureIterator will process correctly
74     val iter = new FileSystemFeatureIterator(storage.getReader(query, threads = readThreads))
75 
76     // transforms will be set after getting the iterator
77     val transformSft = query.getHints.getTransformSchema.getOrElse(sft)
78 
79     // note: DelegateSimpleFeatureIterator will close the iterator by checking that it implements Closeable
80     new DelegateSimpleFeatureReader(transformSft, new DelegateSimpleFeatureIterator(iter))
81   }
82

86   override def canEvent: Boolean = false
87   override def canReproject: Boolean = false

107   /**
108     * Iterator for querying file system storage
109     *
110     * Note: implements Closeable and not AutoCloseable so that DelegateFeatureIterator will close it properly
111     *
112     * @param iter delegate iterator
113     */
114   class FileSystemFeatureIterator(iter: CloseableFeatureIterator)
115       extends java.util.Iterator[SimpleFeature] with Closeable {
116     override def hasNext: Boolean = iter.hasNext
117     override def next(): SimpleFeature = iter.next()
118     override def close(): Unit = iter.close()
119   }
120 
121   /**
122     * Appending feature writer
123     *
124     * @param storage storage instance
125     * @param sft simple feature type
126     * @param timeout write timeout, for flushing partitions
127     */
128   class FileSystemFeatureWriterAppend(storage: FileSystemStorage, sft: SimpleFeatureType, timeout: Duration)
129       extends FeatureWriter[SimpleFeatureType, SimpleFeature] with LazyLogging {
130 
131     private val removalListener = new RemovalListener[String, Closeable with Flushable]() {
132       override def onRemoval(key: String, value: Closeable with Flushable, cause: RemovalCause): Unit = {
133         if (cause == RemovalCause.EXPIRED) {
134           logger.info(s"Flushing writer for partition: $key")
135           FlushWithLogging(value)
136           CloseWithLogging(value)
137         }
138       }
139     }
140 
141     private val writers =
142       Caffeine.newBuilder()
143         .expireAfterAccess(timeout.toMillis, TimeUnit.MILLISECONDS)
144         .removalListener[String, FileSystemWriter](removalListener)
145         .build(new CacheLoader[String, FileSystemWriter]() {
146           override def load(partition: String): FileSystemWriter = storage.getWriter(partition)
147         })
148 
149     private val featureIds = new AtomicLong(0)
150     private var feature: SimpleFeature = _
151 
152     override def getFeatureType: SimpleFeatureType = sft
153 
154     override def hasNext: Boolean = false
155 
156     override def next(): SimpleFeature = {
157       feature = new ScalaSimpleFeature(sft, featureIds.getAndIncrement().toString)
158       feature
159     }
160 
161     override def write(): Unit = {
162       val sf = GeoMesaFeatureWriter.featureWithFid(feature)
163       writers.get(storage.metadata.scheme.getPartitionName(sf)).write(sf)
164       feature = null
165     }
166 
167     override def remove(): Unit = throw new IllegalArgumentException("This writer is append only")
168 
169     override def close(): Unit = {
170       val values = writers.asMap().values().asScala.toSeq
171       val flush = FlushQuietly(values)
172       CloseQuietly(values) match {
173         case None => flush.foreach(e => throw e)
174         case Some(e) => flush.foreach(e.addSuppressed); throw e
175       }
176     }
177   }
178

200     override def write(): Unit = writer.write()
201     override def remove(): Unit = writer.remove()
202     override def close(): Unit = writer.close()
203   }
