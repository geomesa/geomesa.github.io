32 class FileSystemFeatureStore(
33     val storage: FileSystemStorage,
34     entry: ContentEntry,
35     query: Query,
36     readThreads: Int,
37     writeTimeout: Duration
38   ) extends ContentFeatureStore(entry, query) with LazyLogging {
39 
40   private val sft = storage.metadata.sft
41 
42   override def getWriterInternal(query: Query, flags: Int): FeatureWriter[SimpleFeatureType, SimpleFeature] = {
43     // note: check update first as sometimes we get ADD | UPDATE
44     if ((flags & WRITER_UPDATE) == WRITER_UPDATE) {
45       new FileSystemFeatureWriterModify(storage, sft, query.getFilter, readThreads)
46     } else if ((flags & WRITER_ADD) == WRITER_ADD) {
47       new FileSystemFeatureWriterAppend(storage, sft, writeTimeout)
48     } else {
49       throw new IllegalArgumentException(s"Expected one of $WRITER_ADD or $WRITER_UPDATE, but got: $flags")
50     }
51   }
52 
53   override def buildFeatureType(): SimpleFeatureType = sft
54 
55   override def getBoundsInternal(query: Query): ReferencedEnvelope = {
56     val envelope = new ReferencedEnvelope(org.locationtech.geomesa.utils.geotools.CRS_EPSG_4326)
57     storage.getPartitions(query.getFilter).foreach { partition =>
58       partition.bounds.foreach(b => envelope.expandToInclude(b.envelope))
59     }
60     envelope
61   }
62 
63   override def getCountInternal(query: Query): Int =
64     storage.getPartitions(query.getFilter).map(_.count).sum.toInt
65 
66   override def getReaderInternal(original: Query): FeatureReader[SimpleFeatureType, SimpleFeature] = {
67     import org.locationtech.geomesa.index.conf.QueryHints._
68 
69     val query = new Query(original)
70     // The type name can sometimes be empty such as Query.ALL
71     query.setTypeName(sft.getTypeName)
72 
73     // get a closeable java iterator that DelegateSimpleFeatureIterator will process correctly
74     val iter = new FileSystemFeatureIterator(storage.getReader(query, threads = readThreads))
75 
76     // transforms will be set after getting the iterator
77     val transformSft = query.getHints.getTransformSchema.getOrElse(sft)
78 
79     // note: DelegateSimpleFeatureIterator will close the iterator by checking that it implements Closeable
80     new DelegateSimpleFeatureReader(transformSft, new DelegateSimpleFeatureIterator(iter))
81   }
82 
83   override def canLimit: Boolean = false
84   override def canTransact: Boolean = false
85   override def canEvent: Boolean = false
86   override def canReproject: Boolean = false
87   override def canSort: Boolean = false
88 
89   override def canFilter: Boolean = true
90   override def canRetype: Boolean = true
91 
92   override protected def buildQueryCapabilities(): QueryCapabilities = FileSystemFeatureStore.capabilities
93 }

104   /**
105     * Iterator for querying file system storage
106     *
107     * Note: implements Closeable and not AutoCloseable so that DelegateFeatureIterator will close it properly
108     *
109     * @param iter delegate iterator
110     */
111   class FileSystemFeatureIterator(iter: CloseableFeatureIterator)
112       extends java.util.Iterator[SimpleFeature] with Closeable {
113     override def hasNext: Boolean = iter.hasNext
114     override def next(): SimpleFeature = iter.next()
115     override def close(): Unit = iter.close()
116   }
117 
118   /**
119     * Appending feature writer
120     *
121     * @param storage storage instance
122     * @param sft simple feature type
123     * @param timeout write timeout, for flushing partitions
124     */
125   class FileSystemFeatureWriterAppend(storage: FileSystemStorage, sft: SimpleFeatureType, timeout: Duration)
126       extends FeatureWriter[SimpleFeatureType, SimpleFeature] with LazyLogging {
127 
128     private val removalListener = new RemovalListener[String, Closeable with Flushable]() {
129       override def onRemoval(key: String, value: Closeable with Flushable, cause: RemovalCause): Unit = {
130         if (cause == RemovalCause.EXPIRED) {
131           logger.info(s"Flushing writer for partition: $key")
132           FlushWithLogging(value)
133           CloseWithLogging(value)
134         }
135       }
136     }
137 
138     private val writers =
139       Caffeine.newBuilder()
140         .expireAfterAccess(timeout.toMillis, TimeUnit.MILLISECONDS)
141         .removalListener[String, FileSystemWriter](removalListener)
142         .build(new CacheLoader[String, FileSystemWriter]() {
143           override def load(partition: String): FileSystemWriter = storage.getWriter(partition)
144         })
145 
146     private val featureIds = new AtomicLong(0)
147     private var feature: SimpleFeature = _
148 
149     override def getFeatureType: SimpleFeatureType = sft
150 
151     override def hasNext: Boolean = false
152 
153     override def next(): SimpleFeature = {
154       feature = new ScalaSimpleFeature(sft, featureIds.getAndIncrement().toString)
155       feature
156     }
157 
158     override def write(): Unit = {
159       val sf = GeoMesaFeatureWriter.featureWithFid(feature)
160       writers.get(storage.metadata.scheme.getPartitionName(sf)).write(sf)
161       feature = null
162     }
163 
164     override def remove(): Unit = throw new IllegalArgumentException("This writer is append only")
165 
166     override def close(): Unit = {
167       val values = writers.asMap().values().asScala.toSeq
168       val flush = FlushQuietly(values)
169       CloseQuietly(values) match {
170         case None => flush.foreach(e => throw e)
171         case Some(e) => flush.foreach(e.addSuppressed); throw e
172       }
173     }
174   }
175

198     override def remove(): Unit = writer.remove()
199     override def close(): Unit = writer.close()
200   }
