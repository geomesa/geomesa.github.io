104   /**
105     * Key used for merging feature updates.

107     * Implements hadoop writable for m/r, kryo serializable for spark, and comparable to sort in
108     * reverse chronological order
109     */
110   class SimpleFeatureAction extends Writable with KryoSerializable with Comparable[SimpleFeatureAction] {
111 
112     private var _id: String = _
113     private var _timestamp: Long = _
114     private var _action: StorageFileAction = _
115 
116     def this(id: String, timestamp: Long, action: StorageFileAction) = {
117       this()
118       this._id = id
119       this._timestamp = timestamp
120       this._action = action
121     }
122 
123     def id: String = _id
124     def timestamp: Long = _timestamp
125     def action: StorageFileAction = _action
126 
127     override def compareTo(o: SimpleFeatureAction): Int = {
128       var res = _id.compareTo(o.id)
129       if (res == 0) {
130         res = _timestamp.compareTo(o.timestamp) * -1 // note: reverse chronological sort
131         if (res == 0) {
132           res = _action.compareTo(o.action)
133         }
134       }
135       res
136     }
137 
138     override def write(out: DataOutput): Unit = {
139       out.writeUTF(_id)
140       out.writeLong(_timestamp)
141       out.writeUTF(_action.toString)
142     }
143

145       _id = in.readUTF()
