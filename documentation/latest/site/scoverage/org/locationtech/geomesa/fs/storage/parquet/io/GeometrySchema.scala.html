39   /**
40    * Holder for a bounding box field, along with a reference back to the original geometry field
41    *
42    * @param geometry name of the original geometry field being covered
43    * @param bbox name of the bounding box field
44    */
45   case class BoundingBoxField(geometry: String, bbox: String)
46 
47   object BoundingBoxField {
48 
49     import scala.collection.JavaConverters._
50 
51     val BoundingBoxFieldPrefix = "__"
52     val BoundingBoxFieldSuffix = "_bbox__"
53 
54     val XMin = "xmin"
55     val YMin = "ymin"
56     val XMax = "xmax"
57     val YMax = "ymax"
58 
59     def apply(geometry: String, encoded: Boolean = false): BoundingBoxField = {
60       val geom = if (encoded) { geometry } else { alphaNumericSafeString(geometry) }
61       val bbox = s"$BoundingBoxFieldPrefix$geom$BoundingBoxFieldSuffix"
62       BoundingBoxField(geom, bbox)
63     }
64 
65     /**
66      * Gets the fields of this schema that have per-row bounding-boxes. We only add bounding boxes when
67      * they help with predicate push-down, e.g. when a geometry is a non-point or WKB encoded
68      *
69      * @param sft simple feature type
70      * @param encoding geometry encoding
71      * @return
72      */
73     def apply(sft: SimpleFeatureType, encoding: GeometryEncoding): Seq[BoundingBoxField] = {
74       sft.getAttributeDescriptors.asScala.toSeq.flatMap { d =>
75         val binding = d.getType.getBinding
76         if (classOf[Geometry].isAssignableFrom(binding) &&
77             (encoding == GeometryEncoding.GeoParquetWkb || binding != classOf[Point])) {
78           Some(BoundingBoxField(d.getLocalName))
79         } else {
80           None
81         }
82       }
83     }
84 
85     /**
86      * Creates a field name based on a bounding box field
87      *
88      * @param bbox name of a potential bbox field
89      * @return
90      */
91     def fromBoundingBox(bbox: String): Option[BoundingBoxField] = {
92       if (bbox.startsWith(BoundingBoxFieldPrefix) && bbox.endsWith(BoundingBoxFieldSuffix)) {
93         Some(BoundingBoxField(bbox.substring(BoundingBoxFieldPrefix.length, bbox.length - BoundingBoxFieldSuffix.length), bbox))
94       } else {
95         None
96       }
97     }
98 
99     /**
100      * The parquet schema for a bbox field
101      *
102      * @param bbox field name
103      * @return
104      */
105     def schema(bbox: String): GroupType = {
106       Types.optionalGroup()
107         .required(PrimitiveTypeName.FLOAT).named(BoundingBoxField.XMin)
108         .required(PrimitiveTypeName.FLOAT).named(BoundingBoxField.YMin)
109         .required(PrimitiveTypeName.FLOAT).named(BoundingBoxField.XMax)
110         .required(PrimitiveTypeName.FLOAT).named(BoundingBoxField.YMax)
111         .named(bbox)
112     }
113   }
