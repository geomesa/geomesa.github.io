30   /**
31     * Create a writer for simple feature to the given vector batch
32     *
33     * @param sft simple feature type
34     * @param batch vector batch
35     * @param fid write feature id or not
36     * @return
37     */
38   def apply(sft: SimpleFeatureType, batch: VectorizedRowBatch, fid: Boolean = true): OrcAttributeWriter = {
39     require(batch.cols.length == SimpleFeatureTypeDescription.fieldCount(sft, fid),
40       s"ORC schema does not match SimpleFeatureType: ${batch.cols.map(_.getClass.getName).mkString("\n\t", "\n\t", "")}")
41 
42     val builder = Seq.newBuilder[OrcAttributeWriter]
43     builder.sizeHint(sft.getAttributeCount + (if (fid) { 1 } else { 0 }))
44 
45     var i = 0
46     var col = 0
47     while (i < sft.getAttributeCount) {
48       val descriptor = sft.getDescriptor(i)
49       val bindings = ObjectType.selectType(descriptor)
50       val writer = bindings.head match {
51         case ObjectType.GEOMETRY => createGeometryWriter(bindings(1), batch.cols, col, i)
52         case ObjectType.DATE     => new DateWriter(batch.cols(col).asInstanceOf[TimestampColumnVector], i)
53         case ObjectType.STRING   => new StringWriter(batch.cols(col).asInstanceOf[BytesColumnVector], i)
54         case ObjectType.INT      => new IntWriter(batch.cols(col).asInstanceOf[LongColumnVector], i)
55         case ObjectType.LONG     => new LongWriter(batch.cols(col).asInstanceOf[LongColumnVector], i)
56         case ObjectType.FLOAT    => new FloatWriter(batch.cols(col).asInstanceOf[DoubleColumnVector], i)
57         case ObjectType.DOUBLE   => new DoubleWriter(batch.cols(col).asInstanceOf[DoubleColumnVector], i)
58         case ObjectType.BOOLEAN  => new BooleanWriter(batch.cols(col).asInstanceOf[LongColumnVector], i)
59         case ObjectType.BYTES    => new BytesWriter(batch.cols(col).asInstanceOf[BytesColumnVector], i)
60         case ObjectType.UUID     => new UuidWriter(batch.cols(col).asInstanceOf[BytesColumnVector], i)
61         case ObjectType.LIST     => new ListWriter(batch.cols(col).asInstanceOf[ListColumnVector], i, bindings(1))
62         case ObjectType.MAP      => new MapWriter(batch.cols(col).asInstanceOf[MapColumnVector], i, bindings(1), bindings(2))
63         case _ => throw new IllegalArgumentException(s"Unexpected object type ${bindings.head}")
64       }
65       builder += writer
66       i += 1
67       col += SimpleFeatureTypeDescription.fieldCount(descriptor)
68     }
69 
70     if (fid) {
71       builder += new FidWriter(batch.cols(col).asInstanceOf[BytesColumnVector])
72     }
73 
74     new SequenceWriter(builder.result)
75   }

77   // noinspection LanguageFeature
78   private def createGeometryWriter(binding: ObjectType, cols: Array[ColumnVector], col: Int, i: Int): OrcAttributeWriter = {
79     implicit def toDoubleColumnVector(vec: ColumnVector): DoubleColumnVector = vec.asInstanceOf[DoubleColumnVector]
80     implicit def toListColumnVector(vec: ColumnVector): ListColumnVector = vec.asInstanceOf[ListColumnVector]
81 
82     def x: ColumnVector = cols(col)
83     def y: ColumnVector = cols(col + 1)
84 
85     binding match {
86       case ObjectType.POINT           => new PointWriter(x, y, i)
87       case ObjectType.LINESTRING      => new LineStringWriter(x, y, i)
88       case ObjectType.MULTIPOINT      => new MultiPointWriter(x, y, i)
89       case ObjectType.POLYGON         => new PolygonWriter(x, y, i)
90       case ObjectType.MULTILINESTRING => new MultiLineStringWriter(x, y, i)
91       case ObjectType.MULTIPOLYGON    => new MultiPolygonWriter(x, y, i)
92       case ObjectType.GEOMETRY        => new GeometryWkbWriter(cols(col).asInstanceOf[BytesColumnVector], i)
93       case _ => throw new IllegalArgumentException(s"Unexpected object type $binding")
94     }
95   }

168   /**
169     * Writes a linestring attribute to a vector. A linestring is modeled as a list of points.
170     *
171     * @see PointWriter
172     *
173     * @param xx outer list vector for x coordinates, containing a double vector for points
174     * @param yy outer list vector for y coordinates, containing a double vector for points
175     * @param attribute simple feature attribute index
176     */
177   class LineStringWriter(xx: ListColumnVector, yy: ListColumnVector, attribute: Int) extends OrcAttributeWriter {
178     private val x = xx.child.asInstanceOf[DoubleColumnVector]
179     private val y = yy.child.asInstanceOf[DoubleColumnVector]
180 
181     override def apply(sf: SimpleFeature, row: Int): Unit = {
182       val value = sf.getAttribute(attribute).asInstanceOf[LineString]
183       if (value != null) {
184         val length = value.getNumPoints
185         Seq(xx, yy).foreach { vector =>
186           vector.child.ensureSize(vector.childCount + length, true)
187           vector.offsets(row) = vector.childCount
188           vector.lengths(row) = length
189         }
190         var i = 0
191         while (i < length) {
192           val pt = value.getCoordinateN(i)
193           x.vector(xx.childCount + i) = pt.x
194           y.vector(yy.childCount + i) = pt.y
195           i += 1
196         }
197         xx.childCount += length
198         yy.childCount += length
199       } else {
200         xx.noNulls = false
201         yy.noNulls = false
202         xx.isNull(row) = true
203         yy.isNull(row) = true
204       }
205     }

248   /**
249     * Writes a polygon attribute to a vector. A polygon is modeled as a list of lines, with the first
250     * value being the shell, and any subsequent values being interior holes.
251     *
252     * @see LineStringWriter
253     *
254     * @param xxx outer list vector for x coordinates, containing a list vector for individual lines
255     * @param yyy outer list vector for y coordinates, containing a list vector for individual lines
256     * @param attribute simple feature attribute index
257     */
258   class PolygonWriter(xxx: ListColumnVector, yyy: ListColumnVector, attribute: Int) extends OrcAttributeWriter {
259     // list of points for each line
260     private val xx = xxx.child.asInstanceOf[ListColumnVector]
261     private val yy = yyy.child.asInstanceOf[ListColumnVector]
262     // points
263     private val x = xx.child.asInstanceOf[DoubleColumnVector]
264     private val y = yy.child.asInstanceOf[DoubleColumnVector]
265 
266     override def apply(sf: SimpleFeature, row: Int): Unit = {
267       val polygon = sf.getAttribute(attribute).asInstanceOf[Polygon]
268       if (polygon != null) {
269         val lineCount = polygon.getNumInteriorRing + 1
270         Seq(xxx, yyy).foreach { vector =>
271           vector.child.ensureSize(vector.childCount + lineCount, true)
272           vector.offsets(row) = vector.childCount
273           vector.lengths(row) = lineCount
274         }
275         var j = 0
276         while (j < lineCount) {
277           val line = if (j == 0) { polygon.getExteriorRing } else { polygon.getInteriorRingN(j - 1) }
278           val length = line.getNumPoints
279           Seq(xx, yy).foreach { vector =>
280             vector.child.ensureSize(vector.childCount + length, true)
281             vector.offsets(xxx.childCount + j) = vector.childCount
282             vector.lengths(xxx.childCount + j) = length
283           }
284           var i = 0
285           while (i < length) {
286             val pt = line.getCoordinateN(i)
287             x.vector(xx.childCount + i) = pt.x
288             y.vector(yy.childCount + i) = pt.y
289             i += 1
290           }
291           xx.childCount += length
292           yy.childCount += length
293           j += 1
294         }
295         xxx.childCount += lineCount
296         yyy.childCount += lineCount
297       } else {
298         xxx.noNulls = false
299         yyy.noNulls = false
300         xxx.isNull(row) = true
301         yyy.isNull(row) = true
302       }
303     }
304   }
