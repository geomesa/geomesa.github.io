41 class ParquetFileSystemStorage(context: FileSystemContext, metadata: StorageMetadata)
42     extends AbstractFileSystemStorage(context, metadata, ParquetFileSystemStorage.FileExtension) {
43 
44   private val authProvider: AuthorizationsProvider =
45     AuthUtils.getProvider(
46       Collections.emptyMap[String, String](),
47       context.conf.get(AuthsParam.key, "").split(",").toSeq.filter(_.nonEmpty)
48     )
49 
50   override protected def createWriter(file: Path, observer: FileSystemObserver): FileSystemWriter =
51     new ParquetFileSystemWriter(metadata.sft, context, file, observer)
52 
53   override protected def createReader(
54       filter: Option[Filter],
55       transform: Option[(String, SimpleFeatureType)]): FileSystemPathReader = {
56     // readSft has all the fields needed for filtering and return
57     val ReadSchema(readSft, readTransform) = ReadSchema(metadata.sft, filter, transform)
58     val ReadFilter(fc, residualFilter) = ReadFilter(readSft, filter)
59     val parquetFilter = fc.map(FilterCompat.get).getOrElse(FilterCompat.NOOP)
60     val gtFilter = residualFilter.map(FastFilterFactory.optimize(readSft, _))
61     val visFilter = VisibilityUtils.visible(Some(authProvider))
62 
63     logger.debug(s"Parquet filter: $parquetFilter and modified gt filter: ${gtFilter.getOrElse(Filter.INCLUDE)}")
64 
65     // WARNING it is important to create a new conf per query
66     // because we communicate the transform SFT set here
67     // with the init() method on SimpleFeatureReadSupport via
68     // the parquet api. Thus we need to deep copy conf objects
69     val conf = new Configuration(context.conf)
70     StorageConfiguration.setSft(conf, readSft)
71 
72     new ParquetPathReader(conf, readSft, parquetFilter, gtFilter, visFilter, readTransform)
73   }
74 }

83   class ParquetFileSystemWriter(
84       sft: SimpleFeatureType,
85       context: FileSystemContext,
86       file: Path,
87       observer: FileSystemObserver = NoOpObserver
88     ) extends FileSystemWriter {
89 
90     private val conf = {
91       val conf = new Configuration(context.conf)
92       StorageConfiguration.setSft(conf, sft)
93       conf
94     }
95 
96     private val writer = SimpleFeatureParquetWriter.builder(file, conf).build()
97 
98     override def write(f: SimpleFeature): Unit = {
99       writer.write(f)
100       observer.write(f)
101     }
102     override def flush(): Unit = observer.flush()
103     override def close(): Unit = {
104       CloseQuietly(Seq(writer, observer)).foreach(e => throw e)
105       PathCache.register(context.fs, file)
106       if (FileValidationEnabled.toBoolean.get) {
107         validateParquetFile(file)
108       }
109     }
110   }
111 
112   /**
113    * Validate a file by reading it back
114    *
115    * @param file file to validate
116    */
117   def validateParquetFile(file: Path): Unit = {
118     var reader: ParquetReader[Group] = null
119     try {
120       // read Parquet file content
121       reader = ParquetReader.builder(new GroupReadSupport(), file).build()
122       var record = reader.read()
123       while (record != null) {
124         // Process the record
125         record = reader.read()
126       }
127       logger.trace(s"$file is a valid Parquet file")
128     } catch {
129       case NonFatal(e) => throw new RuntimeException(s"Unable to validate $file: File may be corrupted", e)
130     } finally {
131       if (reader != null) {
132         reader.close()
133       }
134     }
135   }
