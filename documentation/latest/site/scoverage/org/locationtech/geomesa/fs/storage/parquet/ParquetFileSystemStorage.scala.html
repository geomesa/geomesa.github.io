40 class ParquetFileSystemStorage(context: FileSystemContext, metadata: StorageMetadata)
41     extends AbstractFileSystemStorage(context, metadata, ParquetFileSystemStorage.FileExtension) {
42 
43   import scala.collection.JavaConverters._
44 
45   private val authProvider: AuthorizationsProvider =
46     AuthUtils.getProvider(
47       Option(context.conf.get(AuthProviderParam.key)).map(p => AuthProviderParam.key -> p).toMap.asJava,
48       context.conf.get(AuthsParam.key, "").split(",").toSeq.filter(_.nonEmpty)
49     )
50 
51   override protected def createWriter(file: Path, observer: FileSystemObserver): FileSystemWriter = {
52     val conf = new Configuration(context.conf)
53     SimpleFeatureParquetSchema.setSft(conf, metadata.sft)
54     val observers =
55       if (FileValidationEnabled.toBoolean.get) {
56         CompositeObserver(Seq(observer, FileValidationObserver(file)))
57       } else {
58         observer
59       }
60     new ParquetFileSystemWriter(file, conf, observers)
61   }
62 
63   override protected def createReader(
64       filter: Option[Filter],
65       transform: Option[(String, SimpleFeatureType)]): FileSystemPathReader = {
66     // readSft has all the fields needed for filtering and return
67     val ReadSchema(readSft, readTransform) = ReadSchema(metadata.sft, filter, transform)
68     val ReadFilter(fc, residualFilter) = ReadFilter(readSft, filter)
69     val parquetFilter = fc.map(FilterCompat.get).getOrElse(FilterCompat.NOOP)
70     val gtFilter = residualFilter.map(FastFilterFactory.optimize(readSft, _))
71     val visFilter = VisibilityUtils.visible(authProvider)
72 
73     logger.debug(
74       s"Parquet filter: ${parquetFilter match { case f: FilterPredicateCompat => f.getFilterPredicate; case f => f }} " +
75         s"and modified gt filter: ${gtFilter.getOrElse(Filter.INCLUDE)}")
76 
77     // WARNING it is important to create a new conf per query
78     // because we communicate the transform SFT set here
79     // with the init() method on SimpleFeatureReadSupport via
80     // the parquet api. Thus we need to deep copy conf objects
81     val conf = new Configuration(context.conf)
82     StorageConfiguration.setSft(conf, readSft)
83 
84     new ParquetFileSystemReader(conf, readSft, parquetFilter, gtFilter, visFilter, readTransform)
85   }
86 }

95   /**
96    * Validate a file by reading it back
97    *
98    * @param file file to validate
99    */
100   case class FileValidationObserver(file: Path) extends FileSystemObserver {
101     override def apply(feature: SimpleFeature): Unit = {}
102     override def flush(): Unit = {}
103     override def close(): Unit = {
104       try {
105         WithClose(ParquetReader.builder(new GroupReadSupport(), file).build()) { reader =>
106           var record = reader.read()
107           while (record != null) {
108             // Process the record
109             record = reader.read()
110           }
111           logger.trace(s"$file is a valid Parquet file")
112         }
113       } catch {
114         case NonFatal(e) => throw new RuntimeException(s"File appears to be corrupted: $file", e)
115       }
116     }
117   }
