42 class ParquetFileSystemStorage(context: FileSystemContext, metadata: StorageMetadata)
43     extends AbstractFileSystemStorage(context, metadata, ParquetFileSystemStorage.FileExtension) {
44 
45   private val authProvider: AuthorizationsProvider =
46     AuthUtils.getProvider(
47       Collections.emptyMap[String, String](),
48       context.conf.get(AuthsParam.key, "").split(",").toSeq.filter(_.nonEmpty)
49     )
50 
51   override protected def createWriter(file: Path, observer: FileSystemObserver): FileSystemWriter =
52     new ParquetFileSystemWriter(metadata.sft, context, file, observer)
53 
54   override protected def createReader(
55       filter: Option[Filter],
56       transform: Option[(String, SimpleFeatureType)]): FileSystemPathReader = {
57     // readSft has all the fields needed for filtering and return
58     val ReadSchema(readSft, readTransform) = ReadSchema(metadata.sft, filter, transform)
59     val ReadFilter(fc, residualFilter) = ReadFilter(readSft, filter)
60     val parquetFilter = fc.map(FilterCompat.get).getOrElse(FilterCompat.NOOP)
61     val gtFilter = residualFilter.map(FastFilterFactory.optimize(readSft, _))
62     val visFilter = VisibilityUtils.visible(Some(authProvider))
63 
64     logger.debug(
65       s"Parquet filter: ${parquetFilter match { case f: FilterPredicateCompat => f.getFilterPredicate; case f => f }} " +
66         s"and modified gt filter: ${gtFilter.getOrElse(Filter.INCLUDE)}")
67 
68     // WARNING it is important to create a new conf per query
69     // because we communicate the transform SFT set here
70     // with the init() method on SimpleFeatureReadSupport via
71     // the parquet api. Thus we need to deep copy conf objects
72     val conf = new Configuration(context.conf)
73     StorageConfiguration.setSft(conf, readSft)
74 
75     new ParquetPathReader(conf, readSft, parquetFilter, gtFilter, visFilter, readTransform)
76   }
77 }

86   class ParquetFileSystemWriter(
87       sft: SimpleFeatureType,
88       context: FileSystemContext,
89       file: Path,
90       observer: FileSystemObserver = NoOpObserver
91     ) extends FileSystemWriter {
92 
93     private val conf = {
94       val conf = new Configuration(context.conf)
95       StorageConfiguration.setSft(conf, sft)
96       conf
97     }
98 
99     private val writer = SimpleFeatureParquetWriter.builder(file, conf).build()
100 
101     override def write(f: SimpleFeature): Unit = {
102       writer.write(f)
103       observer.write(f)
104     }
105     override def flush(): Unit = observer.flush()
106     override def close(): Unit = {
107       CloseQuietly(Seq(writer, observer)).foreach(e => throw e)
108       PathCache.register(context.fs, file)
109       if (FileValidationEnabled.toBoolean.get) {
110         validateParquetFile(file)
111       }
112     }
113   }
114 
115   /**
116    * Validate a file by reading it back
117    *
118    * @param file file to validate
119    */
120   def validateParquetFile(file: Path): Unit = {
121     var reader: ParquetReader[Group] = null
122     try {
123       // read Parquet file content
124       reader = ParquetReader.builder(new GroupReadSupport(), file).build()
125       var record = reader.read()
126       while (record != null) {
127         // Process the record
128         record = reader.read()
129       }
130       logger.trace(s"$file is a valid Parquet file")
131     } catch {
132       case NonFatal(e) => throw new RuntimeException(s"Unable to validate $file: File may be corrupted", e)
133     } finally {
134       if (reader != null) {
135         reader.close()
136       }
137     }
138   }
