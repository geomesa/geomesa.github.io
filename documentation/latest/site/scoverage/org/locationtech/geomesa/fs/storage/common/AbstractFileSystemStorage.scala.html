152   override def compact(partition: Option[String], fileSize: Option[Long], threads: Int): Unit = {
153     val target = targetSize(fileSize)
154     partition.map(Seq(_)).getOrElse(metadata.getPartitions().map(_.name)).foreach { partition =>
155       val paths = getFilePaths(partition)
156       val toCompact = target match {
157         case None => paths
158         case Some(t) =>
159           paths.filter { p =>
160             if (fileIsSized(p.path, t)) {
161               logger.debug(s"Skipping compaction for file [${p.path}] (already target size)")
162               false
163             } else {
164               true
165             }
166           }
167       }
168 
169       if (toCompact.isEmpty) {
170         logger.debug("Skipping compaction - no files to compact")
171       } else if (toCompact.lengthCompare(1) == 0 && target.forall(fileIsSized(toCompact.head.path, _))) {
172         logger.debug(s"Skipping compaction for single data file [${toCompact.mkString}]")
173       } else {
174         logger.debug(s"Compacting data files: [${toCompact.mkString(", ")}]")
175 
176         var written = 0L
177         val bounds = new Envelope()
178 
179         val reader = createReader(None, None)
180 
181         def writer: FileSystemWriter =
182           createWriter(partition, StorageFileAction.Append, FileType.Compacted, target)
183         def threaded: CloseableIterator[SimpleFeature] =
184           FileSystemThreadedReader(Iterator.single(reader -> toCompact), threads)
185 
186         WithClose(writer, threaded) { case (writer, features) =>
187           while (features.hasNext) {
188             val feature = features.next()
189             writer.write(feature)
190             written += 1
191             val geom = feature.getDefaultGeometry.asInstanceOf[Geometry]
192             if (geom != null) {
193               bounds.expandToInclude(geom.getEnvelopeInternal)
194             }
195           }
196         }
197 
198         logger.debug(s"Deleting old files [${toCompact.mkString(", ")}]")
199         metadata.removePartition(
200           PartitionMetadata(partition, toCompact.map(_.file), PartitionBounds(bounds), written))
201 
202         val failures = ListBuffer.empty[Path]
203         toCompact.foreach { file =>
204           if (!context.fs.delete(file.path, false)) {
205             failures.append(file.path)
206           }
207           PathCache.invalidate(context.fs, file.path)
208         }
209 
210         if (failures.nonEmpty) {
211           logger.error(s"Failed to delete some files: [${failures.mkString(", ")}]")
212         }
213 
214         logger.debug(s"Compacted $written records")
215       }
216     }

251   /**
252    * Writes files up to a given size, then starts a new file
253    *
254    * @param paths iterator of files to write
255    * @param estimator target file size estimator
256    */
257   class ChunkedFileSystemWriter(paths: Iterator[WriterConfig], estimator: FileSizeEstimator)
258       extends FileSystemWriter {
259 
260     private var count = 0L // number of features written
261     private var total = 0L // sum size of all finished chunks
262     private var remaining = estimator.estimate(0L)
263 
264     private var path: Path = _
265     private var writer: FileSystemWriter = _
266 
267     override def write(feature: SimpleFeature): Unit = {
268       if (writer == null) {
269         val config = paths.next
270         path = config.path
271         writer = createWriter(config)
272       }
273       writer.write(feature)
274       count += 1
275       remaining -= 1
276       if (remaining == 0) {
277         writer.close()
278         writer = null
279         // adjust our estimate to account for the actual bytes written
280         total += context.fs.getFileStatus(path).getLen
281         estimator.update(total, count)
282         remaining = estimator.estimate(0L)
283       }
284     }
285 
286     override def flush(): Unit = if (writer != null) { writer.flush() }
287 
288     override def close(): Unit = {
289       if (writer != null) {
290         writer.close()
291       }
292       updateFileSize(estimator)
293     }
294   }

297     * Update writer implementation
298     *
299     * @param reader reader for features to update
300     * @param readPartition read partition, if known
301     */
302   class FileSystemUpdateWriterImpl(reader: CloseableFeatureIterator, readPartition: Option[String])
303       extends FileSystemUpdateWriter {
304 
305     private val modifiers = scala.collection.mutable.Map.empty[String, FileSystemWriter]
306     private val deleters = scala.collection.mutable.Map.empty[String, FileSystemWriter]
307 
308     private var feature: SimpleFeature = _
309     private var partition: String = _
310 
311     override def write(): Unit = {
312       if (feature == null) {
313         throw new IllegalArgumentException("Must call 'next' before calling 'write'")
314       }
315       val update = metadata.scheme.getPartitionName(feature)
316       if (update != partition) {
317         // add a delete marker in the old partition, since we only track updates per-partition
318         deleters.getOrElseUpdate(partition, createWriter(partition, StorageFileAction.Delete, FileType.Deleted)).write(feature)
319       }
320       modifiers.getOrElseUpdate(update, createWriter(update, StorageFileAction.Modify, FileType.Modified)).write(feature)
321       feature = null
322     }
323 
324     override def remove(): Unit = {
325       if (feature == null) {
326         throw new IllegalArgumentException("Must call 'next' before calling 'remove'")
327       }
328       deleters.getOrElseUpdate(partition, createWriter(partition, StorageFileAction.Delete, FileType.Deleted)).write(feature)
329       feature = null
330     }
331 
332     override def hasNext: Boolean = reader.hasNext
333 
334     override def next(): SimpleFeature = {
335       feature = reader.next() // note: our reader returns a mutable copy of the feature
336       partition = readPartition.getOrElse(metadata.scheme.getPartitionName(feature))
337       feature
338     }
339 
340     override def flush(): Unit = FlushQuietly.raise(modifiers.values.toSeq ++ deleters.values)
341 
342     override def close(): Unit = CloseQuietly.raise(Seq(reader) ++ modifiers.values ++ deleters.values ++ observers)
343   }
