155   override def compact(partition: Option[String], fileSize: Option[Long], threads: Int): Unit = {
156     val target = targetSize(fileSize)
157     partition.map(Seq(_)).getOrElse(metadata.getPartitions().map(_.name)).foreach { partition =>
158       val paths = getFilePaths(partition)
159       val toCompact = target match {
160         case None => paths
161         case Some(t) =>
162           paths.filter { p =>
163             if (fileIsSized(p.path, t)) {
164               logger.debug(s"Skipping compaction for file [${p.path}] (already target size)")
165               false
166             } else {
167               true
168             }
169           }
170       }
171 
172       if (toCompact.isEmpty) {
173         logger.debug("Skipping compaction - no files to compact")
174       } else if (toCompact.lengthCompare(1) == 0 && target.forall(fileIsSized(toCompact.head.path, _))) {
175         logger.debug(s"Skipping compaction for single data file [${toCompact.mkString}]")
176       } else {
177         logger.debug(s"Compacting data files: [${toCompact.mkString(", ")}]")
178 
179         var written = 0L
180         val bounds = new Envelope()
181 
182         val reader = createReader(None, None)
183 
184         def writer: FileSystemWriter =
185           createWriter(partition, StorageFileAction.Append, FileType.Compacted, target)
186         def threaded: CloseableIterator[SimpleFeature] =
187           FileSystemThreadedReader(Iterator.single(reader -> toCompact), threads)
188 
189         WithClose(writer, threaded) { case (writer, features) =>
190           while (features.hasNext) {
191             val feature = features.next()
192             writer.write(feature)
193             written += 1
194             val geom = feature.getDefaultGeometry.asInstanceOf[Geometry]
195             if (geom != null) {
196               bounds.expandToInclude(geom.getEnvelopeInternal)
197             }
198           }
199         }
200 
201         logger.debug(s"Deleting old files [${toCompact.mkString(", ")}]")
202         metadata.removePartition(
203           PartitionMetadata(partition, toCompact.map(_.file), PartitionBounds(bounds), written))
204 
205         val failures = ListBuffer.empty[Path]
206         toCompact.foreach { file =>
207           if (!context.fs.delete(file.path, false)) {
208             failures.append(file.path)
209           }
210           PathCache.invalidate(context.fs, file.path)
211         }
212 
213         if (failures.nonEmpty) {
214           logger.error(s"Failed to delete some files: [${failures.mkString(", ")}]")
215         }
216 
217         logger.debug(s"Compacted $written records")
218       }
219     }

231   private def createWriter(
232       partition: String,
233       action: StorageFileAction,
234       fileType: FileType,
235       targetFileSize: Option[Long] = None): FileSystemWriter = {
236 
237     def pathAndObserver: WriterConfig = {
238       val path = StorageUtils.nextFile(context.root, partition, metadata.leafStorage, extension, fileType)
239       val updateObserver = new UpdateObserver(partition, path, action)
240       val observer = if (observers.isEmpty) { updateObserver } else {
241         new CompositeObserver(observers.map(_.apply(path)).+:(updateObserver))
242       }
243       WriterConfig(path, observer)
244     }
245 
246     targetSize(targetFileSize) match {
247       case None => createWriter(pathAndObserver)
248       case Some(s) => new ChunkedFileSystemWriter(Iterator.continually(pathAndObserver), estimator(s))
249     }
250   }

254   /**
255    * Writes files up to a given size, then starts a new file
256    *
257    * @param paths iterator of files to write
258    * @param estimator target file size estimator
259    */
260   class ChunkedFileSystemWriter(paths: Iterator[WriterConfig], estimator: FileSizeEstimator)
261       extends FileSystemWriter {
262 
263     private var count = 0L // number of features written
264     private var total = 0L // sum size of all finished chunks
265     private var remaining = estimator.estimate(0L)
266 
267     private var path: Path = _
268     private var writer: FileSystemWriter = _
269 
270     override def write(feature: SimpleFeature): Unit = {
271       if (writer == null) {
272         val config = paths.next
273         path = config.path
274         writer = createWriter(config)
275       }
276       writer.write(feature)
277       count += 1
278       remaining -= 1
279       if (remaining == 0) {
280         writer.close()
281         writer = null
282         // adjust our estimate to account for the actual bytes written
283         total += context.fs.getFileStatus(path).getLen
284         estimator.update(total, count)
285         remaining = estimator.estimate(0L)
286       }
287     }
288 
289     override def flush(): Unit = if (writer != null) { writer.flush() }
290 
291     override def close(): Unit = {
292       if (writer != null) {
293         writer.close()
294       }
295       updateFileSize(estimator)
296     }
297   }

300     * Update writer implementation
301     *
302     * @param reader reader for features to update
303     * @param readPartition read partition, if known
304     */
305   class FileSystemUpdateWriterImpl(reader: CloseableFeatureIterator, readPartition: Option[String])
306       extends FileSystemUpdateWriter {
307 
308     private val modifiers = scala.collection.mutable.Map.empty[String, FileSystemWriter]
309     private val deleters = scala.collection.mutable.Map.empty[String, FileSystemWriter]
310 
311     private var feature: SimpleFeature = _
312     private var partition: String = _
313 
314     override def write(): Unit = {
315       if (feature == null) {
316         throw new IllegalArgumentException("Must call 'next' before calling 'write'")
317       }
318       val update = metadata.scheme.getPartitionName(feature)
319       if (update != partition) {
320         // add a delete marker in the old partition, since we only track updates per-partition
321         deleters.getOrElseUpdate(partition, createWriter(partition, StorageFileAction.Delete, FileType.Deleted)).write(feature)
322       }
323       modifiers.getOrElseUpdate(update, createWriter(update, StorageFileAction.Modify, FileType.Modified)).write(feature)
324       feature = null
325     }
326 
327     override def remove(): Unit = {
328       if (feature == null) {
329         throw new IllegalArgumentException("Must call 'next' before calling 'remove'")
330       }
331       deleters.getOrElseUpdate(partition, createWriter(partition, StorageFileAction.Delete, FileType.Deleted)).write(feature)
332       feature = null
333     }
334 
335     override def hasNext: Boolean = reader.hasNext
336 
337     override def next(): SimpleFeature = {
338       feature = reader.next() // note: our reader returns a mutable copy of the feature
339       partition = readPartition.getOrElse(metadata.scheme.getPartitionName(feature))
340       feature
341     }
342 
343     override def flush(): Unit = FlushQuietly.raise(modifiers.values.toSeq ++ deleters.values)
344 
345     override def close(): Unit = CloseQuietly.raise(Seq(reader) ++ modifiers.values ++ deleters.values ++ observers)
