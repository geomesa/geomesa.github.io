156   override def compact(partition: Option[String], fileSize: Option[Long], threads: Int): Unit = {
157     val target = targetSize(fileSize)
158     partition.map(Seq(_)).getOrElse(metadata.getPartitions().map(_.name)).foreach { partition =>
159       val paths = getFilePaths(partition)
160       val toCompact = target match {
161         case None => paths
162         case Some(t) =>
163           paths.filter { p =>
164             if (fileIsSized(p.path, t)) {
165               logger.debug(s"Skipping compaction for file [${p.path}] (already target size)")
166               false
167             } else {
168               true
169             }
170           }
171       }
172 
173       if (toCompact.isEmpty) {
174         logger.debug("Skipping compaction - no files to compact")
175       } else if (toCompact.lengthCompare(1) == 0 && target.forall(fileIsSized(toCompact.head.path, _))) {
176         logger.debug(s"Skipping compaction for single data file [${toCompact.mkString}]")
177       } else {
178         logger.debug(s"Compacting data files: [${toCompact.mkString(", ")}]")
179 
180         var written = 0L
181         val bounds = new Envelope()
182 
183         val reader = createReader(None, None)
184 
185         def writer: FileSystemWriter =
186           createWriter(partition, StorageFileAction.Append, FileType.Compacted, target)
187         def threaded: CloseableIterator[SimpleFeature] =
188           FileSystemThreadedReader(Iterator.single(reader -> toCompact), threads)
189 
190         WithClose(writer, threaded) { case (writer, features) =>
191           while (features.hasNext) {
192             val feature = features.next()
193             writer.write(feature)
194             written += 1
195             val geom = feature.getDefaultGeometry.asInstanceOf[Geometry]
196             if (geom != null) {
197               bounds.expandToInclude(geom.getEnvelopeInternal)
198             }
199           }
200         }
201 
202         logger.debug(s"Deleting old files [${toCompact.mkString(", ")}]")
203         metadata.removePartition(
204           PartitionMetadata(partition, toCompact.map(_.file), PartitionBounds(bounds), written))
205 
206         val failures = ListBuffer.empty[Path]
207         toCompact.foreach { file =>
208           if (!context.fs.delete(file.path, false)) {
209             failures.append(file.path)
210           }
211           PathCache.invalidate(context.fs, file.path)
212         }
213 
214         if (failures.nonEmpty) {
215           logger.error(s"Failed to delete some files: [${failures.mkString(", ")}]")
216         }
217 
218         logger.debug(s"Compacted $written records")
219       }
220     }
221   }

232   private def createWriter(
233       partition: String,
234       action: StorageFileAction,
235       fileType: FileType,
236       targetFileSize: Option[Long] = None): FileSystemWriter = {
237 
238     def pathAndObserver: WriterConfig = {
239       val path = StorageUtils.nextFile(context.root, partition, metadata.leafStorage, extension, fileType)
240       val updateObserver = new UpdateObserver(partition, path, action)
241       val observer = if (observers.isEmpty) { updateObserver } else {
242         new CompositeObserver(observers.map(_.apply(path)).+:(updateObserver))
243       }
244       WriterConfig(path, observer)
245     }
246 
247     targetSize(targetFileSize) match {
248       case None => createWriter(pathAndObserver)
249       case Some(s) => new ChunkedFileSystemWriter(Iterator.continually(pathAndObserver), estimator(s))
250     }
251   }

255   /**
256    * Writes files up to a given size, then starts a new file
257    *
258    * @param paths iterator of files to write
259    * @param estimator target file size estimator
260    */
261   class ChunkedFileSystemWriter(paths: Iterator[WriterConfig], estimator: FileSizeEstimator)
262       extends FileSystemWriter {
263 
264     private var count = 0L // number of features written
265     private var total = 0L // sum size of all finished chunks
266     private var remaining = estimator.estimate(0L)
267 
268     private var path: Path = _
269     private var writer: FileSystemWriter = _
270 
271     override def write(feature: SimpleFeature): Unit = {
272       if (writer == null) {
273         val config = paths.next
274         path = config.path
275         writer = createWriter(config)
276       }
277       writer.write(feature)
278       count += 1
279       remaining -= 1
280       if (remaining == 0) {
281         writer.close()
282         writer = null
283         // adjust our estimate to account for the actual bytes written
284         total += context.fs.getFileStatus(path).getLen
285         estimator.update(total, count)
286         remaining = estimator.estimate(0L)
287       }
288     }
289 
290     override def flush(): Unit = if (writer != null) { writer.flush() }
291 
292     override def close(): Unit = {
293       if (writer != null) {
294         writer.close()
295       }
296       updateFileSize(estimator)
297     }
298   }

301     * Update writer implementation
302     *
303     * @param reader reader for features to update
304     * @param readPartition read partition, if known

306   class FileSystemUpdateWriterImpl(reader: CloseableFeatureIterator, readPartition: Option[String])
307       extends FileSystemUpdateWriter {
308 
309     private val modifiers = scala.collection.mutable.Map.empty[String, FileSystemWriter]
310     private val deleters = scala.collection.mutable.Map.empty[String, FileSystemWriter]
311 
312     private var feature: SimpleFeature = _
313     private var partition: String = _
314 
315     override def write(): Unit = {
316       if (feature == null) {
317         throw new IllegalArgumentException("Must call 'next' before calling 'write'")
318       }
319       val update = metadata.scheme.getPartitionName(feature)
320       if (update != partition) {
321         // add a delete marker in the old partition, since we only track updates per-partition
322         deleters.getOrElseUpdate(partition, createWriter(partition, StorageFileAction.Delete, FileType.Deleted)).write(feature)
323       }
324       modifiers.getOrElseUpdate(update, createWriter(update, StorageFileAction.Modify, FileType.Modified)).write(feature)
325       feature = null
326     }
327 
328     override def remove(): Unit = {
329       if (feature == null) {
330         throw new IllegalArgumentException("Must call 'next' before calling 'remove'")
331       }
332       deleters.getOrElseUpdate(partition, createWriter(partition, StorageFileAction.Delete, FileType.Deleted)).write(feature)
333       feature = null
334     }
335 
336     override def hasNext: Boolean = reader.hasNext
337 
338     override def next(): SimpleFeature = {
339       feature = reader.next() // note: our reader returns a mutable copy of the feature
340       partition = readPartition.getOrElse(metadata.scheme.getPartitionName(feature))
341       feature
342     }
343 
344     override def flush(): Unit = FlushQuietly.raise(modifiers.values.toSeq ++ deleters.values)
345 
346     override def close(): Unit = CloseQuietly.raise(Seq(reader) ++ modifiers.values ++ deleters.values ++ observers)
347   }
