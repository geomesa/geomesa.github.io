56   /**
57     * Create the Orc type description corresponding to the SimpleFeatureType. SimpleFeatureType is
58     * modeled as an Orc Struct, with nested fields for each attribute.
59     *
60     * @param sft simple feature type
61     * @return
62     */
63   def createTypeDescription(sft: SimpleFeatureType, fid: Boolean = true): TypeDescription = {
64     val container = TypeDescription.createStruct()
65     var i = 0
66     while (i < sft.getAttributeCount) {
67       addTypeDescription(container, sft.getDescriptor(i))
68       i += 1
69     }
70     if (fid) {
71       container.addField("id", TypeDescription.createString())
72     }
73     container
74   }

76   /**
77     * Gets a count of the Orc fields created for a simple feature type.
78     *
79     * Geometry attributes create 2 columns, other attributes create 1, feature id creates 1
80     *
81     * @param sft simple feature type
82     * @param fid include feature ids or not
83     * @return
84     */
85   def fieldCount(sft: SimpleFeatureType, fid: Boolean = true): Int = {
86     val attributes = sft.getAttributeDescriptors.asScala.foldLeft(0) { case (sum, d) => sum + fieldCount(d) }
87     if (fid) { attributes + 1 } else { attributes }
88   }
89 
90   /**
91    * Gets a count of the Orc fields created for a given attribute
92    *
93    * @param descriptor descriptor
94    * @return
95    */
96   def fieldCount(descriptor: AttributeDescriptor): Int = {
97     descriptor.getType.getBinding match {
98       // plain Geometry bindings are encoded in a single WKB column, others as x + y
99       case b if classOf[Geometry].isAssignableFrom(b) && b != classOf[Geometry] => 2
100       case _ => 1
101     }
102   }
103   /**
104     * Add a type description for an attribute
105     *
106     * @param container top-level Orc struct corresponding to the SimpleFeatureType
107     * @param descriptor descriptor
108     */
109   private def addTypeDescription(container: TypeDescription, descriptor: AttributeDescriptor): Unit = {
110     val name = descriptor.getLocalName
111     val bindings = ObjectType.selectType(descriptor)
112     bindings.head match {
113       case ObjectType.GEOMETRY => addGeometryDescription(container, name, bindings(1))
114       case ObjectType.LIST     => container.addField(name, TypeDescription.createList(simple(bindings(1))))
115       case ObjectType.MAP      => container.addField(name, TypeDescription.createMap(simple(bindings(1)), simple(bindings(2))))
116       case binding             => container.addField(name, simple(binding))
117     }
118   }
119 
120   /**
121     * We create two separate fields, one for x-values and one for y-values. Orc doesn't support
122     * predicate push-down on nested fields, so we flatten them out. Note that Orc also doesn't
123     * support predicate push-down for complex fields, so we actually only benefit for Points.
124     */
125   private def addGeometryDescription(container: TypeDescription, name: String, binding: ObjectType): Unit = {
126     import TypeDescription.{createDouble, createList}
127 
128     def x: String = geometryXField(name)
129     def y: String = geometryYField(name)
130 
131     binding match {
132       case ObjectType.POINT =>
133         // point (x, y) pair
134         container.addField(x, createDouble())
135         container.addField(y, createDouble())
136 
137       case ObjectType.LINESTRING =>
138         // list of points
139         container.addField(x, createList(createDouble()))
140         container.addField(y, createList(createDouble()))
141 
142       case ObjectType.MULTIPOINT =>
143         // list of points
144         container.addField(x, createList(createDouble()))
145         container.addField(y, createList(createDouble()))
146 
147       case ObjectType.POLYGON =>
148         // list of lines (exterior ring + any holes)
149         container.addField(x, createList(createList(createDouble())))
150         container.addField(y, createList(createList(createDouble())))
151 
152       case ObjectType.MULTILINESTRING =>
153         // list of lines
154         container.addField(x, createList(createList(createDouble())))
155         container.addField(y, createList(createList(createDouble())))
156 
157       case ObjectType.MULTIPOLYGON =>
158         // list of polygons
159         container.addField(x, createList(createList(createList(createDouble()))))
160         container.addField(y, createList(createList(createList(createDouble()))))
161 
162       case ObjectType.GEOMETRY =>
163         // WKB
164         container.addField(name, TypeDescription.createBinary())
165 
166       case _ => throw new IllegalArgumentException(s"Unexpected geometry type $binding")
167     }
168   }
169 
170   /**
171     * Type description for simple types, e.g. int, float, date, etc
172     *
173     * @param binding type binding
174     * @return
175     */
176   private def simple(binding: ObjectType): TypeDescription = {
177     binding match {
178       case ObjectType.DATE    => TypeDescription.createTimestamp()
179       case ObjectType.STRING  => TypeDescription.createString()
180       case ObjectType.INT     => TypeDescription.createInt()
181       case ObjectType.LONG    => TypeDescription.createLong()
182       case ObjectType.FLOAT   => TypeDescription.createFloat()
183       case ObjectType.DOUBLE  => TypeDescription.createDouble()
184       case ObjectType.BOOLEAN => TypeDescription.createBoolean()
185       case ObjectType.BYTES   => TypeDescription.createBinary()
186       case ObjectType.UUID    => TypeDescription.createString()
187       case _ => throw new IllegalArgumentException(s"Unexpected simple object type $binding")
188     }
189   }
