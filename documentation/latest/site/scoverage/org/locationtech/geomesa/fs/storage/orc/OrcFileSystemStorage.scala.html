22 /**
23   * Orc implementation of FileSystemStorage
24   *
25   * @param metadata metadata
26   */
27 class OrcFileSystemStorage(context: FileSystemContext, metadata: StorageMetadata)
28     extends AbstractFileSystemStorage(context, metadata, OrcFileSystemStorage.FileExtension) {
29 
30   override protected def createWriter(file: Path, observer: FileSystemObserver): FileSystemWriter =
31     new OrcFileSystemWriter(metadata.sft, context, file, observer)
32 
33   override protected def createReader(
34       filter: Option[Filter],
35       transform: Option[(String, SimpleFeatureType)]): FileSystemPathReader = {
36     val optimized = filter.map(FastFilterFactory.optimize(metadata.sft, _))
37     new OrcFileSystemReader(metadata.sft, context.conf, optimized, transform)
38   }
39 }
