19 /**
20   * Lexicoded attribute partitioning
21   *
22   * @param attribute attribute name
23   * @param index attribute index in the sft
24   * @param binding type binding
25   * @param allowedValues list of allowedValues to partition
26   */
27 case class AttributeScheme(attribute: String, index: Int, binding: Class[_], defaultPartition: String, allowedValues: Seq[String]) extends PartitionScheme {
28 
29   import FilterHelper.ff
30 
31   private val alias = AttributeIndexKey.alias(binding)
32 
33   override val depth: Int = 1
34 
35   override def pattern: String = s"<$attribute>"
36 
37   override def getPartitionName(feature: SimpleFeature): String = {
38     val value = feature.getAttribute(index)
39     if (value == null) {
40       return defaultPartition
41     }
42     val encodedValue = AttributeIndexKey.typeEncode(value)
43     if (allowedValues.isEmpty || allowedValues.contains(encodedValue)) {
44       encodedValue
45     } else {
46       defaultPartition
47     }
48   }
49 
50   override def getSimplifiedFilters(filter: Filter, partition: Option[String]): Option[Seq[SimplifiedFilter]] = {
51     getIntersectingPartitions(filter).map { covered =>
52       // remove the attribute filter that we've already accounted for in our covered partitions
53       val coveredFilter = FilterExtractingVisitor(filter, attribute, AttributeScheme.propertyIsEquals _)._2
54       val simplified = SimplifiedFilter(coveredFilter.getOrElse(Filter.INCLUDE), covered, partial = false)
55 
56       partition match {
57         case None => Seq(simplified)
58         case Some(p) if simplified.partitions.contains(p) => Seq(simplified.copy(partitions = Seq(p)))
59         case _ => Seq.empty
60       }
61     }
62   }
63 
64   override def getIntersectingPartitions(filter: Filter): Option[Seq[String]] = {
65     val bounds = FilterHelper.extractAttributeBounds(filter, attribute, binding)
66     if (bounds.disjoint) {
67       Some(Seq.empty)
68     } else if (bounds.isEmpty || !bounds.precise || !bounds.forall(_.isEquals)) {
69       if (allowedValues.nonEmpty) Some(allowedValues) else None
70     } else {
71       // note: we verified they are all single values above
72       val partitions = bounds.values.map(bound => AttributeIndexKey.encodeForQuery(bound.lower.value.get, binding))
73       if (allowedValues.nonEmpty) Some(partitions.filter(partition => allowedValues.contains(partition))) else Some(partitions)
74     }
75   }
76 
77   override def getCoveringFilter(partition: String): Filter =
78     ff.equals(ff.property(attribute), ff.literal(AttributeIndexKey.decode(alias, partition)))
79 
80 }

82 object AttributeScheme {
83 
84   val Name = "attribute"
85 
86   /**
87     * Check to extract only the equality filters that we can process with this partition scheme
88     *
89     * @param filter filter
90     * @return
91     */
92   def propertyIsEquals(filter: Filter): Boolean = {
93     filter match {
94       case _: And | _: Or => true // note: implies further processing of children
95       case _: PropertyIsEqualTo => true
96       case _ => false
97     }
98   }
99 
100   object Config {
101     val AttributeOpt: String = "partitioned-attribute"
102     val AllowedListOpt: String = "allow-list"
103     val DefaultPartitionOpt: String = "default-partition"
104   }
105 
106   class AttributePartitionSchemeFactory extends PartitionSchemeFactory {
107     override def load(sft: SimpleFeatureType, config: NamedOptions): Option[PartitionScheme] = {
108       if (config.name != Name) { None } else {
109         val attribute = config.options.getOrElse(Config.AttributeOpt, null)
110         require(attribute != null, s"Attribute scheme requires valid attribute name '${Config.AttributeOpt}'")
111         val index = sft.indexOf(attribute)
112         require(index != -1, s"Attribute '$attribute' does not exist in schema '${sft.getTypeName}'")
113         val binding = sft.getDescriptor(index).getType.getBinding
114         require(AttributeIndexKey.encodable(binding),
115           s"Invalid type binding '${binding.getName}' of attribute '$attribute'")
116         val allowedValues: Seq[String] = config.options.get(Config.AllowedListOpt).map(_.split(',').toSeq).getOrElse(Seq.empty).map(allowed => AttributeIndexKey.encodeForQuery(allowed.trim(),binding))
117         val defaultPartition = AttributeIndexKey.encodeForQuery(config.options.getOrElse(Config.DefaultPartitionOpt, allowedValues.headOption.getOrElse("")), binding)
118         require(allowedValues.isEmpty || allowedValues.contains(defaultPartition), "Default partition must be one of the allowed values")
119         Some(AttributeScheme(attribute, index, binding, defaultPartition, allowedValues))
120       }
121     }
122   }
