19 /**
20   * Lexicoded attribute partitioning
21   *
22   * @param attribute attribute name
23   * @param index attribute index in the sft
24   * @param binding type binding
25   */
26 case class AttributeScheme(attribute: String, index: Int, binding: Class[_]) extends PartitionScheme {
27 
28   import FilterHelper.ff
29 
30   private val alias = AttributeIndexKey.alias(binding)
31 
32   override val depth: Int = 1
33 
34   override def pattern: String = s"<$attribute>"
35 
36   override def getPartitionName(feature: SimpleFeature): String = {
37     val value = feature.getAttribute(index)
38     if (value == null) { "" } else { AttributeIndexKey.typeEncode(value) }
39   }
40 
41   override def getSimplifiedFilters(filter: Filter, partition: Option[String]): Option[Seq[SimplifiedFilter]] = {
42     getIntersectingPartitions(filter).map { covered =>
43       // remove the attribute filter that we've already accounted for in our covered partitions
44       val coveredFilter = FilterExtractingVisitor(filter, attribute, AttributeScheme.propertyIsEquals _)._2
45       val simplified = SimplifiedFilter(coveredFilter.getOrElse(Filter.INCLUDE), covered, partial = false)
46 
47       partition match {
48         case None => Seq(simplified)
49         case Some(p) if simplified.partitions.contains(p) => Seq(simplified.copy(partitions = Seq(p)))
50         case _ => Seq.empty
51       }
52     }
53   }
54 
55   override def getIntersectingPartitions(filter: Filter): Option[Seq[String]] = {
56     val bounds = FilterHelper.extractAttributeBounds(filter, attribute, binding)
57     if (bounds.disjoint) {
58       Some(Seq.empty)
59     } else if (bounds.isEmpty || !bounds.precise || !bounds.forall(_.isEquals)) {
60       None
61     } else {
62       // note: we verified they are all single values above
63       Some(bounds.values.map(bound => AttributeIndexKey.encodeForQuery(bound.lower.value.get, binding)))
64     }
65   }
66 
67   override def getCoveringFilter(partition: String): Filter =
68     ff.equals(ff.property(attribute), ff.literal(AttributeIndexKey.decode(alias, partition)))
69 
70 }

72 object AttributeScheme {
73 
74   val Name = "attribute"
75 
76   /**
77     * Check to extract only the equality filters that we can process with this partition scheme
78     *
79     * @param filter filter
80     * @return
81     */
82   def propertyIsEquals(filter: Filter): Boolean = {
83     filter match {
84       case _: And | _: Or => true // note: implies further processing of children
85       case _: PropertyIsEqualTo => true
86       case _ => false
87     }
88   }
89 
90   object Config {
91     val AttributeOpt: String = "partitioned-attribute"
92   }
93 
94   class AttributePartitionSchemeFactory extends PartitionSchemeFactory {
95     override def load(sft: SimpleFeatureType, config: NamedOptions): Option[PartitionScheme] = {
96       if (config.name != Name) { None } else {
97         val attribute = config.options.getOrElse(Config.AttributeOpt, null)
98         require(attribute != null, s"Attribute scheme requires valid attribute name '${Config.AttributeOpt}'")
99         val index = sft.indexOf(attribute)
100         require(index != -1, s"Attribute '$attribute' does not exist in schema '${sft.getTypeName}'")
101         val binding = sft.getDescriptor(index).getType.getBinding
102         require(AttributeIndexKey.encodable(binding),
103           s"Invalid type binding '${binding.getName}' of attribute '$attribute'")
104         Some(AttributeScheme(attribute, index, binding))
105       }
106     }
107   }
