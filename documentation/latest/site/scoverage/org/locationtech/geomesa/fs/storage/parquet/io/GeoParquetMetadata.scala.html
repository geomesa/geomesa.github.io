32   /**
33    * Indicates if an attribute is supported or not.
34    *
35    * Currently only mixed geometry (encoded as WKB) and point types are encoded correctly as GeoParquet.
36    *
37    * TODO GEOMESA-3259 support non-point geometries
38    *
39    * @param d descriptor
40    * @return
41    */
42   private[parquet] def supported(d: GeometryDescriptor): Boolean = {
43     ObjectType.selectType(d).last match {
44       case ObjectType.POINT | ObjectType.GEOMETRY => true
45       case _ => false
46     }
47   }
48 
49   /**
50    * Generate json describing the GeoParquet file, conforming to the 1.1.0 GeoParquet schema
51    *
52    * @param primaryGeometry name of the primary geometry in the file
53    * @param geometries pairs of geometries and optional bounds
54    * @return
55    */
56   def apply(primaryGeometry: String, geometries: Seq[(GeometryDescriptor, Option[Envelope])]): String = {
57     val cols = new JsonObject()
58     geometries.foreach { case (descriptor, bounds) =>
59       val types = new JsonArray(1)
60       val encoding = descriptor.getType.getBinding match {
61         case b if b == classOf[Geometry] => "WKB"
62         case b if b == classOf[GeometryCollection] => types.add(classOf[GeometryCollection].getSimpleName); "WKB"
63         case b => types.add(b.getSimpleName); b.getSimpleName.toLowerCase(Locale.US)
64       }
65       val metadata = new JsonObject()
66       metadata.addProperty("encoding", encoding)
67       metadata.add("geometry_types", types) // TODO add ' Z' for 3d points
68       bounds.filterNot(_.isNull).foreach { e =>
69         val bbox = new JsonArray(4)
70         bbox.add(e.getMinX)
71         bbox.add(e.getMinY)
72         bbox.add(e.getMaxX)
73         bbox.add(e.getMaxY)
74         metadata.add("bbox", bbox) // TODO add z for 3d points
75       }
76       cols.add(StringSerialization.alphaNumericSafeString(descriptor.getLocalName), metadata)
77     }
78 
79     val model = new JsonObject()
80     model.addProperty("version", "1.1.0")
81     model.addProperty("primary_column", primaryGeometry)
82     model.add("columns", cols)
83 
84     gson.toJson(model)
85   }
86 
87   /**
88    * Observer class that generates GeoParquet metadata
89    *
90    * @param sft simple feature type
91    */
92   class GeoParquetObserver(sft: SimpleFeatureType) extends FileSystemObserver {
93 
94     import scala.collection.JavaConverters._
95 
96     private val bounds =
97       sft.getAttributeDescriptors.asScala.zipWithIndex.collect {
98         case (d: GeometryDescriptor, i) if supported(d) => (d, i, new Envelope())
99       }
100 
101     def metadata(): java.util.Map[String, String] = {
102       if (bounds.isEmpty) { Collections.emptyMap() } else {
103         val geoms = bounds.map { case (d, _, env) => (d, Some(env).filterNot(_.isNull)) }
104         val primary = bounds.find(_._1 == sft.getGeometryDescriptor).getOrElse(bounds.head)._1.getLocalName
105         Collections.singletonMap(GeoParquetMetadataKey, GeoParquetMetadata(primary, geoms.toSeq))
106       }
107     }
108 
109     override def write(feature: SimpleFeature): Unit = {
110       bounds.foreach { case (_, i, envelope) =>
111         val geom = feature.getAttribute(i).asInstanceOf[Geometry]
112         if (geom != null) {
113           envelope.expandToInclude(geom.getEnvelopeInternal)
114         }
115       }
116     }
117 
118     override def flush(): Unit = {}
119     override def close(): Unit = {}
120   }
