80     /**
81      * Gets the geometry object type based on the GeoParquet column types
82      *
83      * @param colTypes column types from GeoParquet metadata
84      * @return
85      */
86     def toObjectType(colTypes: Seq[GeoParquetColumnType]): ObjectType = {
87       if (colTypes.lengthCompare(1) != 0) {
88         ObjectType.GEOMETRY // more than 1 geometry type
89       } else if (colTypes.head == GeometryCollection || colTypes.head == `GeometryCollection Z` ) {
90         ObjectType.GEOMETRY_COLLECTION // this is the one objectType whose name doesn't align with the columnType
91       } else {
92         ObjectType.withName(colTypes.head.toString.replace(" Z", "").toUpperCase(Locale.US))
93       }
94     }

97   /**
98    * Generate json describing the GeoParquet file, conforming to the 1.1.0 GeoParquet schema
99    *
100    * @param metadata metadata
101    * @return
102    */
103   def toJson(metadata: GeoParquetMetadata): String = {
104     val model = new JsonObject()
105     model.addProperty("version", "1.1.0")
106     model.addProperty("primary_column", metadata.primaryGeometry)
107     val cols = model.addObject("columns")
108     metadata.geometries.foreach { column =>
109       val metadata = cols.addObject(column.name)
110       metadata.addProperty("encoding", column.encoding.toString)
111       val types = metadata.addArray("geometry_types", 1)
112       column.types.foreach(t => types.add(t.toString))
113       if (!column.bounds.isNull) {
114         val bbox = metadata.addArray("bbox", 4)
115         bbox.add(column.bounds.getMinX)
116         bbox.add(column.bounds.getMinY)
117         bbox.add(column.bounds.getMaxX)
118         bbox.add(column.bounds.getMaxY)
119       }
120       column.covering.foreach { c =>
121         val bbox = metadata.addObject("covering").addObject("bbox")
122         Seq("xmin", "ymin", "xmax", "ymax").foreach { name =>
123           val bound = bbox.addArray(name, 2)
124           bound.add(c)
125           bound.add(name)
126         }
127       }
128     }
129 
130     gson.toJson(model)
131   }

133   /**
134    * Parse a GeoParquet json into a model object
135    *
136    * @param json json string
137    * @return
138    */
139   def fromJson(json: String): GeoParquetMetadata = {
140     val obj = gson.fromJson(json, classOf[JsonObject])
141     val primary = obj.get("primary_column").getAsString
142     val cols = obj.getAsJsonObject("columns").entrySet().asScala.map { entry =>
143       val name = entry.getKey
144       val obj = entry.getValue.getAsJsonObject
145       // required entries
146       val encoding = GeoParquetColumnEncoding.withName(obj.get("encoding").getAsString)
147       val types = obj.getAsJsonArray("geometry_types").asList().asScala.map(t => GeoParquetColumnType.withName(t.getAsString))
148       // optional entries
149       // note: all covering fields *must* refer to the same column, so we only pull out one
150       val covering =
151         Option(obj.getAsJsonObject("covering")).map(_.getAsJsonObject("bbox").getAsJsonArray("xmin").get(0).getAsString)
152       val bounds = new Envelope()
153       Option(obj.getAsJsonArray("bbox")).foreach { bbox =>
154         bounds.expandToInclude(bbox.get(0).getAsDouble, bbox.get(1).getAsDouble)
155         bounds.expandToInclude(bbox.get(2).getAsDouble, bbox.get(3).getAsDouble)
156       }
157       // TODO crs, orientation, edges, epoch
158       ColumnMetadata(name, encoding, types.toSeq, covering, bounds)
159     }
160     GeoParquetMetadata(primary, cols.toSeq)
161   }

181   /**
182    * Observer class that generates GeoParquet metadata
183    *
184    * @param schema simple feature schema
185    */
186   class GeoParquetObserver(schema: SimpleFeatureParquetSchema) extends FileSystemObserver {
187 
188     import scala.collection.JavaConverters._
189 
190     private val columns =
191       schema.sft.getAttributeDescriptors.asScala.collect { case d: GeometryDescriptor if isGeoParquet(d) => createColumnMetadata(d) }
192     private val boundsWithIndex = columns.map(c => c.bounds -> schema.sft.indexOf(decodeAlphaNumericSafeString(c.name)))
193 
194     /**
195      * Gets the file metadata for geoparquet
196      *
197      * @return
198      */
199     def metadata(): java.util.Map[String, String] = {
200       if (columns.isEmpty) { Collections.emptyMap() } else {
201         val primary = {
202           // there's a chance that the primary geom wasn't encoded in the file, but other geoms were
203           val expected = alphaNumericSafeString(schema.sft.getGeometryDescriptor.getLocalName)
204           columns.find(_.name == expected).getOrElse(columns.head).name
205         }
206         Collections.singletonMap(GeoParquetMetadataKey, GeoParquetMetadata(primary, columns.toSeq).toJson())
207       }
208     }
209 
210     override def write(feature: SimpleFeature): Unit = {
211       boundsWithIndex.foreach { case (bounds, i) =>
212         val geom = feature.getAttribute(i).asInstanceOf[Geometry]
213         if (geom != null) {
214           bounds.expandToInclude(geom.getEnvelopeInternal)
215         }
216       }
217     }
218 
219     /**
220      * Checks if a geometry is encoded as GeoParquet
221      *
222      * @param d descriptor
223      * @return
224      */
225     private def isGeoParquet(d: GeometryDescriptor): Boolean =
226       schema.encodings.geometry == GeometryEncoding.GeoParquetNative ||
227         schema.encodings.geometry == GeometryEncoding.GeoParquetWkb ||
228         d.getType.getBinding == classOf[Point] ||
229         d.getType.getBinding == classOf[GeometryCollection] ||
230         d.getType.getBinding == classOf[Geometry]
231 
232     /**
233      * Create metadata placeholder for geometry columns
234      *
235      * @param descriptor geometry descriptor
236      * @return
237      */
238     private def createColumnMetadata(descriptor: GeometryDescriptor): ColumnMetadata = {
239       val name = alphaNumericSafeString(descriptor.getLocalName)
240       val binding = descriptor.getType.getBinding
241       val encoding = {
242         // for non-wkb encoding schemes, we still use WKB for mixed-type geometries
243         if (schema.encodings.geometry == GeometryEncoding.GeoParquetWkb ||
244             binding == classOf[Geometry] || binding == classOf[GeometryCollection]) {
245           GeoParquetColumnEncoding.WKB
246         } else {
247           GeoParquetColumnEncoding.withName(binding.getSimpleName.toLowerCase(Locale.US))
248         }
249       }
250       // TODO add z for 3d points
251       val types = if (binding == classOf[Geometry]) { Seq.empty } else { Seq(GeoParquetColumnType.withName(binding.getSimpleName)) }
252       val covering = schema.boundingBoxes.collectFirst { case b if b.geometry == name => b.bbox }
253       ColumnMetadata(name, encoding, types, covering)
254     }
255 
256     override def flush(): Unit = {}
257     override def close(): Unit = {}
258   }
