40   /**
41    * Generate json describing the GeoParquet file, conforming to the 1.1.0 GeoParquet schema
42    *
43    * @param primaryGeometry name of the primary geometry in the file
44    * @param geometries pairs of geometries and optional bounds
45    * @return
46    */
47   private def apply(primaryGeometry: String, geometries: Seq[ColumnMetadata]): String = {
48     val model = new JsonObject()
49     model.addProperty("version", "1.1.0")
50     model.addProperty("primary_column", primaryGeometry)
51     val cols = model.addObject("columns")
52     geometries.foreach { column =>
53       val metadata = cols.addObject(column.name)
54       metadata.addProperty("encoding", column.encoding)
55       val types = metadata.addArray("geometry_types", 1) // TODO add ' Z' for 3d points
56       column.types.foreach(types.add)
57       if (!column.bounds.isNull) {
58         val bbox = metadata.addArray("bbox", 4) // TODO add z for 3d points
59         bbox.add(column.bounds.getMinX)
60         bbox.add(column.bounds.getMinY)
61         bbox.add(column.bounds.getMaxX)
62         bbox.add(column.bounds.getMaxY)
63       }
64       column.covering.foreach { c =>
65         val bbox = metadata.addObject("covering").addObject("bbox")
66         Seq("xmin", "ymin", "xmax", "ymax").foreach { name =>
67           val bound = bbox.addArray(name, 2)
68           bound.add(c)
69           bound.add(name)
70         }
71       }
72     }
73 
74     gson.toJson(model)
75   }

95   /**
96    * Observer class that generates GeoParquet metadata
97    *
98    * @param schema simple feature schema
99    */
100   class GeoParquetObserver(schema: SimpleFeatureParquetSchema) extends FileSystemObserver {
101 
102     import scala.collection.JavaConverters._
103 
104     private val columns =
105       schema.sft.getAttributeDescriptors.asScala.collect { case d: GeometryDescriptor if isGeoParquet(d) => createColumnMetadata(d) }
106     private val boundsWithIndex = columns.map(c => c.bounds -> schema.sft.indexOf(decodeAlphaNumericSafeString(c.name)))
107 
108     /**
109      * Gets the file metadata for geoparquet
110      *
111      * @return
112      */
113     def metadata(): java.util.Map[String, String] = {
114       if (columns.isEmpty) { Collections.emptyMap() } else {
115         val primary = {
116           // there's a chance that the primary geom wasn't encoded in the file, but other geoms were
117           val expected = alphaNumericSafeString(schema.sft.getGeometryDescriptor.getLocalName)
118           columns.find(_.name == expected).getOrElse(columns.head).name
119         }
120         Collections.singletonMap(GeoParquetMetadataKey, GeoParquetMetadata(primary, columns.toSeq))
121       }
122     }
123 
124     override def write(feature: SimpleFeature): Unit = {
125       boundsWithIndex.foreach { case (bounds, i) =>
126         val geom = feature.getAttribute(i).asInstanceOf[Geometry]
127         if (geom != null) {
128           bounds.expandToInclude(geom.getEnvelopeInternal)
129         }
130       }
131     }
132 
133     /**
134      * Checks if a geometry is encoded as GeoParquet
135      *
136      * @param d descriptor
137      * @return
138      */
139     private def isGeoParquet(d: GeometryDescriptor): Boolean =
140       schema.encodings.geometry == GeometryEncoding.GeoParquetNative ||
141         schema.encodings.geometry == GeometryEncoding.GeoParquetWkb ||
142         d.getType.getBinding == classOf[Point] ||
143         d.getType.getBinding == classOf[GeometryCollection] ||
144         d.getType.getBinding == classOf[Geometry]
145 
146     /**
147      * Create metadata placeholder for geometry columns
148      *
149      * @param descriptor geometry descriptor
150      * @return
151      */
152     private def createColumnMetadata(descriptor: GeometryDescriptor): ColumnMetadata = {
153       val name = alphaNumericSafeString(descriptor.getLocalName)
154       val binding = descriptor.getType.getBinding
155       val encoding = {
156         // for non-wkb encoding schemes, we still use WKB for mixed-type geometries
157         if (schema.encodings.geometry == GeometryEncoding.GeoParquetWkb ||
158             binding == classOf[Geometry] || binding == classOf[GeometryCollection]) {
159           "WKB"
160         } else {
161           binding.getSimpleName.toLowerCase(Locale.US)
162         }
163       }
164       val types = if (binding == classOf[Geometry]) { None } else { Some(binding.getSimpleName) }
165       val covering = schema.boundingBoxes.collectFirst { case b if b.geometry == name => b.bbox }
166       ColumnMetadata(name, encoding, types, covering)
167     }
168 
169     override def flush(): Unit = {}
170     override def close(): Unit = {}
171   }
