29   override def createDataStore(params: java.util.Map[String, _]): DataStore = {
30 
31     val xml = ConfigsParam.lookupOpt(params)
32     val resources = ConfigPathsParam.lookupOpt(params).toSeq.flatMap(_.split(',')).map(_.trim).filterNot(_.isEmpty)
33 
34     val conf = if (xml.isEmpty && resources.isEmpty) { FileSystemDataStoreFactory.configuration } else {
35       val conf = new Configuration(FileSystemDataStoreFactory.configuration)
36       // add the explicit props first, they may be needed for loading the path resources
37       xml.foreach(x => conf.addResource(new ByteArrayInputStream(x.getBytes(StandardCharsets.UTF_8))))
38       resources.foreach(HadoopUtils.addResource(conf, _))
39       conf
40     }
41 
42     val path = new Path(PathParam.lookup(params))
43     val encoding = EncodingParam.lookupOpt(params).filterNot(_.isEmpty)
44 
45     // Need to do more tuning here. On a local system 1 thread (so basic producer/consumer) was best
46     // because Parquet is also threading the reads underneath I think. using prod/cons pattern was
47     // about 30% faster but increasing beyond 1 thread slowed things down. This could be due to the
48     // cost of serializing simple features though. need to investigate more.
49     //
50     // However, if you are doing lots of filtering it appears that bumping the threads up high
51     // can be very useful. Seems possibly numcores/2 might is a good setting (which is a standard idea)
52 
53     val readThreads = ReadThreadsParam.lookup(params)
54     val writeTimeout = WriteTimeoutParam.lookup(params)
55     val queryTimeout = QueryTimeoutParam.lookupOpt(params).filter(_.isFinite)
56     AuthsParam.lookupOpt(params).foreach(conf.set(AuthsParam.key, _))
57     params.get(AuthProviderParam.key) match {
58       case null => // no-op
59       case clas: String => conf.set(AuthProviderParam.key, clas)
60       case p =>
61         throw new IllegalArgumentException(s"Parameter '${AuthProviderParam.key}' only supports a class name, but received: ${p.getClass}")
62     }
63 
64     val namespace = NamespaceParam.lookupOpt(params)
65 
66     val fs = FileSystem.get(path.toUri, conf)
67 
68     val config = FileSystemDataStoreConfig(conf, path, readThreads, writeTimeout, queryTimeout, encoding, namespace)
69 
70     new FileSystemDataStore(fs, config)
71   }
72
