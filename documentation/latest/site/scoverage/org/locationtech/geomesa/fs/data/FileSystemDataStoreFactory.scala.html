35   override def createDataStore(params: java.util.Map[String, _]): DataStore = {
36 
37     val xml = ConfigsParam.lookupOpt(params)
38     val resources = ConfigPathsParam.lookupOpt(params).toSeq.flatMap(_.split(',')).map(_.trim).filterNot(_.isEmpty)
39 
40     val conf = if (xml.isEmpty && resources.isEmpty) { FileSystemDataStoreFactory.configuration } else {
41       val conf = new Configuration(FileSystemDataStoreFactory.configuration)
42       // add the explicit props first, they may be needed for loading the path resources
43       xml.foreach(x => conf.addResource(new ByteArrayInputStream(x.getBytes(StandardCharsets.UTF_8))))
44       resources.foreach(HadoopUtils.addResource(conf, _))
45       conf
46     }
47 
48     val path = new Path(PathParam.lookup(params))
49     val encoding = EncodingParam.lookupOpt(params).filterNot(_.isEmpty)
50 
51     // Need to do more tuning here. On a local system 1 thread (so basic producer/consumer) was best
52     // because Parquet is also threading the reads underneath I think. using prod/cons pattern was
53     // about 30% faster but increasing beyond 1 thread slowed things down. This could be due to the
54     // cost of serializing simple features though. need to investigate more.
55     //
56     // However, if you are doing lots of filtering it appears that bumping the threads up high
57     // can be very useful. Seems possibly numcores/2 might is a good setting (which is a standard idea)
58 
59     val readThreads = ReadThreadsParam.lookup(params)
60     val writeTimeout = WriteTimeoutParam.lookup(params)
61     val queryTimeout = QueryTimeoutParam.lookupOpt(params).filter(_.isFinite)
62     AuthsParam.lookupOpt(params).foreach { auths =>
63       conf.set(AuthsParam.key, auths)
64     }
65 
66     val namespace = NamespaceParam.lookupOpt(params)
67 
68     val fs = FileSystem.get(path.toUri, conf)
69 
70     val config = FileSystemDataStoreConfig(conf, path, readThreads, writeTimeout, queryTimeout, encoding, namespace)
71 
72     new FileSystemDataStore(fs, config)
73   }
74
