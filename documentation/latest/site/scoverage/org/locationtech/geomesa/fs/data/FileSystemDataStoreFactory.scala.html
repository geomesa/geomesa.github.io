35   override def createDataStore(params: java.util.Map[String, _]): DataStore = {
36 
37     val xml = ConfigsParam.lookupOpt(params)
38     val resources = ConfigPathsParam.lookupOpt(params).toSeq.flatMap(_.split(',')).map(_.trim).filterNot(_.isEmpty)
39 
40     val conf = if (xml.isEmpty && resources.isEmpty) { FileSystemDataStoreFactory.configuration } else {
41       val conf = new Configuration(FileSystemDataStoreFactory.configuration)
42       // add the explicit props first, they may be needed for loading the path resources
43       xml.foreach(x => conf.addResource(new ByteArrayInputStream(x.getBytes(StandardCharsets.UTF_8))))
44       resources.foreach(HadoopUtils.addResource(conf, _))
45       conf
46     }
47 
48     val fc = fileContextCache.get(conf)
49 
50     val path = new Path(PathParam.lookup(params))
51     val encoding = EncodingParam.lookupOpt(params).filterNot(_.isEmpty)
52 
53     // Need to do more tuning here. On a local system 1 thread (so basic producer/consumer) was best
54     // because Parquet is also threading the reads underneath I think. using prod/cons pattern was
55     // about 30% faster but increasing beyond 1 thread slowed things down. This could be due to the
56     // cost of serializing simple features though. need to investigate more.
57     //
58     // However, if you are doing lots of filtering it appears that bumping the threads up high
59     // can be very useful. Seems possibly numcores/2 might is a good setting (which is a standard idea)
60 
61     val readThreads = ReadThreadsParam.lookup(params)
62     val writeTimeout = WriteTimeoutParam.lookup(params)
63 
64     val namespace = NamespaceParam.lookupOpt(params)
65 
66     new FileSystemDataStore(fc, conf, path, readThreads, writeTimeout, encoding, namespace)
67   }
