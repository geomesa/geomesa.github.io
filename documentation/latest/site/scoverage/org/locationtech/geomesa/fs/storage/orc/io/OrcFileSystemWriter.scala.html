22 class OrcFileSystemWriter(
23     sft: SimpleFeatureType,
24     context: FileSystemContext,
25     file: Path,
26     observer: FileSystemObserver = NoOpObserver
27   ) extends FileSystemWriter {
28 
29   private val schema = SimpleFeatureTypeDescription(sft)
30 
31   private val options = OrcFile.writerOptions(context.conf).setSchema(schema)
32   private val writer = OrcFile.createWriter(file, options)
33   private val batch = schema.createRowBatch()
34 
35   private val attributeWriter = OrcAttributeWriter(sft, batch)
36 
37   override def write(sf: SimpleFeature): Unit = {
38     attributeWriter.apply(sf, batch.size)
39     batch.size += 1
40     // If the batch is full, write it out and start over
41     if (batch.size == batch.getMaxSize) {
42       writer.addRowBatch(batch)
43       batch.reset()
44     }
45     observer(sf)
46   }
47 
48   override def flush(): Unit = {
49     flushBatch()
50     observer.flush()
51   }
52 
53   override def close(): Unit = {
54     try { flushBatch() } catch {
55       case NonFatal(e) => CloseQuietly(Seq(writer, observer)).foreach(e.addSuppressed); throw e
56     }
57     CloseQuietly.raise(Seq(writer, observer))
58   }
59 
60   private def flushBatch(): Unit = {
61     if (batch.size != 0) {
62       writer.addRowBatch(batch)
63       batch.reset()
64     }
65   }
66 }
