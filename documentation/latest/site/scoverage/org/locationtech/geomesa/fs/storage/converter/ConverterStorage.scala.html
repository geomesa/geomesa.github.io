23 class ConverterStorage(context: FileSystemContext,
24                        metadata: StorageMetadata,
25                        converter: SimpleFeatureConverter,
26                        pathFiltering: Option[PathFiltering])
27     extends AbstractFileSystemStorage(context, metadata, "") {
28 
29   // TODO close converter...
30   // the problem is that we aggressively cache storage instances for performance (in FileSystemStorageManager),
31   // so even if we wired a 'close' method through the entire storage api, we'd also have to implement a
32   // 'borrow/return' paradigm and expire idle instances. Since currently only converters with redis caches
33   // actually need to be closed, and since they will only open a single connection per converter, the
34   // impact should be low
35 
36   override protected def createWriter(file: Path, observer: FileSystemObserver): FileSystemWriter =
37     throw new UnsupportedOperationException()
38 
39   override protected def createReader(
40       filter: Option[Filter],
41       transform: Option[(String, SimpleFeatureType)]): FileSystemPathReader = {
42     new ConverterFileSystemReader(context.fs, converter, filter, transform, pathFiltering)
43   }
44 
45   override def getFilePaths(partition: String): Seq[StorageFilePath] = {
46     val path = new Path(context.root, partition)
47     if (metadata.leafStorage) { Seq(StorageFilePath(StorageFile(path.getName, 0L), path)) } else {
48       PathCache.list(context.fs, path).map(p => StorageFilePath(StorageFile(p.getPath.getName, 0L), p.getPath)).toList
49     }
50   }
51 
52   override def getWriter(partition: String): FileSystemWriter =
53     throw new UnsupportedOperationException("Converter storage does not support feature writing")
54 
55   override def compact(partition: Option[String], fileSize: Option[Long], threads: Int): Unit =
56     throw new UnsupportedOperationException("Converter storage does not support compactions")
57 }
