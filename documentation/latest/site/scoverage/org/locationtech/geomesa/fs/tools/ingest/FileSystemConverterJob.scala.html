38 abstract class FileSystemConverterJob(
39     dsParams: Map[String, String],
40     sft: SimpleFeatureType,
41     converterConfig: Config,
42     paths: Seq[String],
43     libjarsFiles: Seq[String],
44     libjarsPaths: Iterator[() => Seq[File]],
45     reducers: Int,
46     root: Path,
47     tmpPath: Option[Path],
48     targetFileSize: Option[Long]
49   ) extends ConverterIngestJob(dsParams, sft, converterConfig, paths, libjarsFiles, libjarsPaths)
50     with StorageConfiguration with LazyLogging {
51 
52   override protected def reduceCounters(job: Job): Seq[(String, Long)] =
53     Seq((IngestCounters.Persisted, job.getCounters.findCounter(OutputCounters.Group, "reduced").getValue))
54 
55   override def configureJob(job: Job): Unit = {
56     super.configureJob(job)
57 
58     job.setMapperClass(classOf[FsIngestMapper])
59     job.setMapOutputKeyClass(classOf[Text])
60     job.setMapOutputValueClass(classOf[BytesWritable])
61     job.setOutputKeyClass(classOf[Void])
62     job.setOutputValueClass(classOf[SimpleFeature])
63     job.setReducerClass(classOf[DummyReducer])
64     job.setNumReduceTasks(reducers)
65     // Ensure that the reducers don't start too early
66     // (default is at 0.05 which takes all the map slots and isn't needed)
67     job.getConfiguration.set("mapreduce.job.reduce.slowstart.completedmaps", ".90")
68 
69     StorageConfiguration.setRootPath(job.getConfiguration, root)
70     StorageConfiguration.setFileType(job.getConfiguration, FileType.Written)
71     targetFileSize.foreach(StorageConfiguration.setTargetFileSize(job.getConfiguration, _))
72 
73     FileOutputFormat.setOutputPath(job, tmpPath.getOrElse(root))
74 
75     configureOutput(sft, job)
76   }
77 
78   override def await(reporter: StatusCallback): JobResult = {
79     super.await(reporter).merge {
80       tmpPath.map { tp =>
81         reporter.reset()
82         new DistributedCopy().copy(Seq(tp), root, reporter) match {
83           case JobSuccess(message, counts) =>
84             Command.user.info(message)
85             JobSuccess("", counts)
86 
87           case j => j
88         }
89       }
90     }
91   }
92 }

126   class FsIngestMapper extends Mapper[LongWritable, SimpleFeature, Text, BytesWritable] with LazyLogging {
127 
128     type Context = Mapper[LongWritable, SimpleFeature, Text, BytesWritable]#Context
129 
130     private var serializer: KryoFeatureSerializer = _
131     private var metadata: StorageMetadata = _
132     private var scheme: PartitionScheme = _
133 
134     var mapped: Counter = _
135     var written: Counter = _
136     var failed: Counter = _
137 
138     override def setup(context: Context): Unit = {
139       val root = StorageConfiguration.getRootPath(context.getConfiguration)
140       val fc = FileContext.getFileContext(root.toUri, context.getConfiguration)
141       // note: we don't call `reload` (to get the partition metadata) as we aren't using it
142       metadata = StorageMetadataFactory.load(FileSystemContext(fc, context.getConfiguration, root)).getOrElse {
143         throw new IllegalArgumentException(s"Could not load storage instance at path $root")
144       }
145       serializer = KryoFeatureSerializer(metadata.sft, SerializationOptions.none)
146       scheme = metadata.scheme
147 
148       mapped = context.getCounter(OutputCounters.Group, "mapped")
149       written = context.getCounter(OutputCounters.Group, OutputCounters.Written)
150       failed = context.getCounter(OutputCounters.Group, OutputCounters.Failed)
151     }
152 
153     override def map(key: LongWritable, sf: SimpleFeature, context: Context): Unit = {
154       // partitionKey is important because this needs to be correct for the parquet file
155       try {
156         mapped.increment(1)
157         val sfWithFid = GeoMesaFeatureWriter.featureWithFid(sf)
158         val partitionKey = new Text(scheme.getPartitionName(sfWithFid))
159         context.write(partitionKey, new BytesWritable(serializer.serialize(sfWithFid)))
160         written.increment(1)
161       } catch {
162         case e: Throwable =>
163           logger.error(s"Failed to write '${DataUtilities.encodeFeature(sf)}'", e)
164           failed.increment(1)
165       }
166     }
167 
168     override def cleanup(context: Context): Unit = metadata.close()
169   }
170 
171   class DummyReducer extends Reducer[Text, BytesWritable, Void, SimpleFeature] {
172 
173     import scala.collection.JavaConverters._
174 
175     type Context = Reducer[Text, BytesWritable, Void, SimpleFeature]#Context
176 
177     private var serializer: KryoFeatureSerializer = _
178     private var reduced: Counter = _
179 
180     override def setup(context: Context): Unit = {
181       val root = StorageConfiguration.getRootPath(context.getConfiguration)
182       val fc = FileContext.getFileContext(root.toUri, context.getConfiguration)
183       // note: we don't call `reload` (to get the partition metadata) as we aren't using it
184       val metadata = StorageMetadataFactory.load(FileSystemContext(fc, context.getConfiguration, root)).getOrElse {
185         throw new IllegalArgumentException(s"Could not load storage instance at path $root")
186       }
187       serializer = KryoFeatureSerializer(metadata.sft, SerializationOptions.none)
188       reduced = context.getCounter(OutputCounters.Group, "reduced")
189       metadata.close()
190     }
191 
192     override def reduce(key: Text, values: java.lang.Iterable[BytesWritable], context: Context): Unit = {
193       values.asScala.foreach { bw =>
194         val sf = serializer.deserialize(bw.getBytes)
195         context.write(null, sf)
196         reduced.increment(1)
197       }
198     }
199   }
