32   val MetadataPath = "metadata.json"

34   private val cache = new ConcurrentHashMap[String, NamedOptions]()
35 
36   /**
37     * Read the metadata file at the given root path, if it exists
38     *
39     * @param context context
40     * @return
41     */
42   def readMetadata(context: FileSystemContext): Option[NamedOptions] = {
43     val key = context.root.toUri.toString
44     var cached = cache.get(key)
45     if (cached == null) {
46       // note: we may end up reading more than once, but doing it multiple times will only incur
47       // a slight performance penalty but not cause any incorrectness
48       // using an atomic operation or cache loader can cause problems, as we sometimes insert into the
49       // map during the load, which is not allowed
50       val file = new Path(context.root, MetadataPath)
51       if (PathCache.exists(context.fs, file)) {
52         val config = profile("Loaded metadata configuration") {
53           WithClose(new InputStreamReader(context.fs.open(file), StandardCharsets.UTF_8)) { in =>
54             ConfigFactory.load(ConfigFactory.parseReader(in, ParseOptions)) // call load to resolve sys props
55           }
56         }
57         if (config.hasPath("name")) {
58           cached = profile("Parsed metadata configuration") {
59             ConfigSource.fromConfig(config).loadOrThrow[NamedOptions]
60           }
61           cache.put(key, cached)
62         } else {
63           context.fs.rename(file, new Path(context.root, s"$MetadataPath.bak"))
64           PathCache.invalidate(context.fs, file)
65           transitionMetadata(context, config).foreach { meta =>
66             cached = meta // will be set in the cache in the transition code
67           }
68         }
69       }
70     }
71     Option(cached)
72   }
73 
74   /**
75     * Write a metadata file at the given root path
76     *
77     * @param context context
78     * @param metadata metadata configuration
79     */
80   def writeMetadata(context: FileSystemContext, metadata: NamedOptions): Unit = {
81     val file = new Path(context.root, MetadataPath)
82     if (PathCache.exists(context.fs, file, reload = true)) {
83       throw new IllegalArgumentException(
84         s"Trying to create a new storage instance but metadata already exists at '$file'")
85     }
86     val data = profile("Serialized metadata configuration") {
87       ConfigWriter[NamedOptions].to(metadata).render(RenderOptions)
88     }
89     // remove quotes around substitutions so that they resolve properly
90     // this logic relies on the fact that all strings will be quoted, and just puts another quote on
91     // either side of the expression (typesafe will concatenate them), i.e. "foo ${bar}" -> "foo "${bar}""
92     val interpolated = data.replaceAll("\\$\\{[a-zA-Z0-9_.]+}", "\"$0\"")
93     profile("Persisted metadata configuration") {
94       WithClose(context.fs.create(file, false)) { out =>
95         out.write(interpolated.getBytes(StandardCharsets.UTF_8))
96         out.hflush()
97         out.hsync()
98       }
99     }
100     val toCache = if (data == interpolated) { metadata } else {
101       // reload through ConfigFactory to resolve substitutions
102       ConfigSource.fromConfig(ConfigFactory.load(ConfigFactory.parseString(interpolated, ParseOptions)))
103           .loadOrThrow[NamedOptions]
104     }
105     cache.put(context.root.toUri.toString, toCache)
106     PathCache.register(context.fs, file)
107   }

110   /**
111     * Transition the old single-file metadata.json to the new append-log format
112     *
113     * @param context file system context
114     * @return
115     */
116   private def transitionMetadata(context: FileSystemContext, config: Config): Option[NamedOptions] = {
117     import scala.collection.JavaConverters._
118 
119     try {
120       val sft = SimpleFeatureTypes.createType(config.getConfig("featureType"), path = None)
121       val encoding = config.getString("encoding")
122       val scheme = {
123         val schemeConfig = config.getConfig("partitionScheme")
124         val schemeOpts = schemeConfig.getConfig("options")
125         NamedOptions(schemeConfig.getString("scheme"),
126           schemeOpts.entrySet().asScala.map(e => e.getKey -> schemeOpts.getString(e.getKey)).toMap)
127       }
128       val leafStorage = scheme.options.get("leaf-storage").forall(_.toBoolean)
129       val meta = Metadata(sft, encoding, scheme, leafStorage)
130       val partitionConfig = config.getConfig("partitions")
131 
132       val defaults = FileBasedMetadata.LegacyOptions
133       WithClose(new FileBasedMetadataFactory().create(context, defaults.options, meta)) { metadata =>
134         partitionConfig.root().entrySet().asScala.foreach { e =>
135           val name = e.getKey
136           val files = partitionConfig.getStringList(name).asScala.map(StorageFile(_, 0L))
137           metadata.addPartition(PartitionMetadata(name, files.toSeq, None, 0L))
138         }
139       }
140 
141       Some(defaults)
142     } catch {
143       case NonFatal(e) => logger.warn("Error transitioning old metadata format: ", e); None
144     }
145   }
