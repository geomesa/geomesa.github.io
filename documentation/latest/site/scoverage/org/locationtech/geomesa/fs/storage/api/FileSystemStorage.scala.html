20 /**
21   * Persists simple features to a file system and provides query access. Storage implementations are fairly
22   * 'lightweight', in that all state is captured in the metadata instance
23   */
24 trait FileSystemStorage extends Compactable with Closeable {
25 
26   /**
27     * Handle to the file context, root path and configuration
28     *
29     * @return
30     */
31   def context: FileSystemContext
32 
33   /**
34     * The metadata for this storage instance
35     *
36     * @return metadata
37     */
38   def metadata: StorageMetadata
39 
40   /**
41     * Pass through to get all partitions from the metadata
42     *
43     * @return
44     */
45   def getPartitions: Seq[PartitionMetadata] = metadata.getPartitions()
46 
47   /**
48     * Gets a list of partitions that match the given filter
49     *
50     * @return partitions
51     */
52   def getPartitions(filter: Filter): Seq[PartitionMetadata] = {
53     if (filter == Filter.INCLUDE) {
54       return metadata.getPartitions()
55     }
56 
57     val filters = metadata.scheme.getSimplifiedFilters(filter).orNull
58     if (filters == null) {
59       return metadata.getPartitions()
60     }
61 
62     filters.flatMap { f =>
63       if (f.partial) {
64         f.partitions.flatMap(p => metadata.getPartitions(Some(p)))
65       } else {
66         f.partitions.flatMap(metadata.getPartition)
67       }
68     }
69   }
70 
71   /**
72     * Get partitions that match a given filter. Each set of partitions will have a simplified
73     * filter that should be applied to that set
74     *
75     * If there are no partitions that match the filter, an empty list will be returned
76     *
77     * @return partitions and predicates for each partition
78     */
79   def getPartitionFilters(filter: Filter, partition: Option[String] = None): Seq[PartitionFilter] = {
80     val filters = metadata.scheme.getSimplifiedFilters(filter).orNull
81     if (filters == null) {
82       return Seq(PartitionFilter(filter, partition.map(Seq(_)).getOrElse(metadata.getPartitions().map(_.name))))
83     }
84 
85     partition match {
86       case None =>
87         filters.flatMap { f =>
88           val partitions = if (f.partial) {
89             f.partitions.flatMap(p => metadata.getPartitions(Some(p)))
90           } else {
91             f.partitions.flatMap(metadata.getPartition)
92           }
93           if (partitions.isEmpty) { Seq.empty } else {
94             Seq(PartitionFilter(f.filter, partitions.map(_.name)))
95           }
96         }
97 
98       case Some(p) =>
99         def matches(f: SimplifiedFilter): Boolean =
100           if (f.partial) { f.partitions.exists(p.startsWith) } else { f.partitions.contains(p) }
101         filters.collectFirst { case f if matches(f) => PartitionFilter(f.filter, Seq(p)) }.toSeq
102     }
103   }
104 
105   /**
106     * Get the full paths to any files contained in the partition
107     *
108     * @param partition partition
109     * @return file paths
110     */
111   def getFilePaths(partition: String): Seq[StorageFilePath]
112 
113   /**
114     * Get a reader for all relevant partitions
115     *
116     * @param query query
117     * @param partition restrict results to a single partition
118     * @param threads suggested threads used for reading data files
119     * @return reader
120     */
121   def getReader(query: Query, partition: Option[String] = None, threads: Int = 1): CloseableFeatureIterator
122 
123   /**
124     * Get a writer for a given partition. This method is thread-safe and can be called multiple times,
125     * although this can result in multiple data files.
126     *
127     * @param partition partition
128     * @return writer
129     */
130   def getWriter(partition: String): FileSystemWriter
131 
132   /**
133     * Gets a modifying writer. This method is thread-safe and can be called multiple times,
134     * although if a feature is modified multiple times concurrently, the last update 'wins'.
135     * There is no guarantee that any concurrent modifications will be reflected in the returned
136     * writer.
137     *
138     * @param filter the filter used to select features for modification
139     * @param partition restrict results to a single partition (writes will go to the appropriate partition, regardless)
140     * @param threads suggested threads used for reading data files
141     * @return
142     */
143   def getWriter(filter: Filter, partition: Option[String] = None, threads: Int = 1): FileSystemUpdateWriter
144 
145   override def close(): Unit = metadata.close()
146 }
147 
148 object FileSystemStorage {
149 
150   trait FileSystemWriter extends Closeable with Flushable {
151 
152     /**
153       * Write a feature
154       *
155       * @param feature feature
156       */
157     def write(feature: SimpleFeature): Unit
158   }
159 
160   trait FileSystemUpdateWriter extends Iterator[SimpleFeature] with Closeable with Flushable {
161 
162     /**
163       * Writes a modification to the last feature returned by `next`
164       */
165     def write(): Unit
166 
167     /**
168       * Deletes the last feature returned by `next`
169       */
170     def remove(): Unit
171   }
