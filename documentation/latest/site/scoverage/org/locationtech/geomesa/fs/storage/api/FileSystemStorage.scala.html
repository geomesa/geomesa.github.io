21 /**
22   * Persists simple features to a file system and provides query access. Storage implementations are fairly
23   * 'lightweight', in that all state is captured in the metadata instance
24   */
25 trait FileSystemStorage extends Compactable with Closeable {
26 
27   /**
28     * Handle to the file context, root path and configuration
29     *
30     * @return
31     */
32   def context: FileSystemContext
33 
34   /**
35     * The metadata for this storage instance
36     *
37     * @return metadata
38     */
39   def metadata: StorageMetadata
40 
41   /**
42     * Pass through to get all partitions from the metadata
43     *
44     * @return
45     */
46   def getPartitions: Seq[PartitionMetadata] = metadata.getPartitions()
47 
48   /**
49     * Gets a list of partitions that match the given filter
50     *
51     * @return partitions
52     */
53   def getPartitions(filter: Filter): Seq[PartitionMetadata] = {
54     if (filter == Filter.INCLUDE) {
55       return metadata.getPartitions()
56     }
57 
58     val filters = metadata.scheme.getSimplifiedFilters(filter).orNull
59     if (filters == null) {
60       return metadata.getPartitions()
61     }
62 
63     filters.flatMap { f =>
64       if (f.partial) {
65         f.partitions.flatMap(p => metadata.getPartitions(Some(p)))
66       } else {
67         f.partitions.flatMap(metadata.getPartition)
68       }
69     }
70   }
71 
72   /**
73     * Get partitions that match a given filter. Each set of partitions will have a simplified
74     * filter that should be applied to that set
75     *
76     * If there are no partitions that match the filter, an empty list will be returned
77     *
78     * @return partitions and predicates for each partition
79     */
80   def getPartitionFilters(filter: Filter, partition: Option[String] = None): Seq[PartitionFilter] = {
81     val filters = metadata.scheme.getSimplifiedFilters(filter).orNull
82     if (filters == null) {
83       return Seq(PartitionFilter(filter, partition.map(Seq(_)).getOrElse(metadata.getPartitions().map(_.name))))
84     }
85 
86     partition match {
87       case None =>
88         filters.flatMap { f =>
89           val partitions = if (f.partial) {
90             f.partitions.flatMap(p => metadata.getPartitions(Some(p)))
91           } else {
92             f.partitions.flatMap(metadata.getPartition)
93           }
94           if (partitions.isEmpty) { Seq.empty } else {
95             Seq(PartitionFilter(f.filter, partitions.map(_.name)))
96           }
97         }
98 
99       case Some(p) =>
100         def matches(f: SimplifiedFilter): Boolean =
101           if (f.partial) { f.partitions.exists(p.startsWith) } else { f.partitions.contains(p) }
102         filters.collectFirst { case f if matches(f) => PartitionFilter(f.filter, Seq(p)) }.toSeq
103     }
104   }
105 
106   /**
107     * Get the full paths to any files contained in the partition
108     *
109     * @param partition partition
110     * @return file paths
111     */
112   def getFilePaths(partition: String): Seq[StorageFilePath]
113 
114   /**
115     * Get a reader for all relevant partitions
116     *
117     * @param query query
118     * @param partition restrict results to a single partition
119     * @param threads suggested threads used for reading data files
120     * @return reader
121     */
122   def getReader(query: Query, partition: Option[String] = None, threads: Int = 1): CloseableFeatureIterator
123 
124   /**
125     * Get a writer for a given partition. This method is thread-safe and can be called multiple times,
126     * although this can result in multiple data files.
127     *
128     * @param partition partition
129     * @return writer
130     */
131   def getWriter(partition: String): FileSystemWriter
132 
133   /**
134     * Gets a modifying writer. This method is thread-safe and can be called multiple times,
135     * although if a feature is modified multiple times concurrently, the last update 'wins'.
136     * There is no guarantee that any concurrent modifications will be reflected in the returned
137     * writer.
138     *
139     * @param filter the filter used to select features for modification
140     * @param partition restrict results to a single partition (writes will go to the appropriate partition, regardless)
141     * @param threads suggested threads used for reading data files
142     * @return
143     */
144   def getWriter(filter: Filter, partition: Option[String] = None, threads: Int = 1): FileSystemUpdateWriter
145 
146   override def close(): Unit = metadata.close()
147 }

185   }
