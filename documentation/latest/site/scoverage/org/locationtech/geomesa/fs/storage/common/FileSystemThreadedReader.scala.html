24 /**
25   * Multi-threaded file reads
26   *
27   * @param es executor service used by the read tasks
28   * @param phaser phaser for tracking read task completion
29   * @param queue intermediate queue populated by the read tasks
30   */
31 class FileSystemThreadedReader private (es: ExecutorService, phaser: Phaser, queue: BlockingQueue[SimpleFeature])
32     extends CloseableIterator[SimpleFeature] {
33 
34   private val localQueue = new java.util.LinkedList[SimpleFeature]()
35 
36   private var current: SimpleFeature = _
37 
38   override def hasNext: Boolean = {
39     if (current != null) {
40       return true
41     }
42     current = localQueue.pollFirst
43     if (current != null) {
44       return true
45     }
46 
47     while (!phaser.isTerminated) {
48       current = queue.poll(100, TimeUnit.MILLISECONDS)
49       if (current != null) {
50         queue.drainTo(localQueue, 10000)
51         return true
52       }
53     }
54     // last check - if phaser.isTerminated, the queue should have whatever values are left
55     current = queue.poll()
56     if (current != null) {
57       queue.drainTo(localQueue, 10000)
58       true
59     } else {
60       false
61     }
62   }
63 
64   override def next(): SimpleFeature = {
65     if (hasNext) {
66       val ret = current
67       current = null
68       ret
69     } else {
70       Iterator.empty.next
71     }
72   }
73

83   def apply(
84       readers: Iterator[(FileSystemPathReader, Seq[StorageFilePath])],
85       threads: Int): CloseableIterator[SimpleFeature] = {
86 
87     if (threads < 2) {
88       CloseableIterator(readers).flatMap { case (reader, files) =>
89         val mods = scala.collection.mutable.HashSet.empty[String]
90         // ensure files are sorted in reverse chronological order
91         CloseableIterator(files.sorted.iterator).flatMap(f => read(reader, f, mods))
92       }
93     } else {
94       val queue = new LinkedBlockingQueue[SimpleFeature](2000000)
95       val es = Executors.newFixedThreadPool(threads)
96 
97       val phaser = new Phaser(1) {
98         override protected def onAdvance(phase: Int, registeredParties: Int): Boolean = {
99           // when all tasks have been completed, shutdown the executor service
100           es.shutdown()
101           true // return true to indicate the phaser should terminate
102         }
103       }
104 
105       try {
106         var child = new Phaser(phaser) // ensure that we don't register too many parties on this phaser
107         var parties = 0 // track registered parties
108         readers.foreach { case (reader, files) =>
109           // group our files by actions which can be parallelized
110           val groups = scala.collection.mutable.ListBuffer.empty[Seq[StorageFilePath]]
111           var group = scala.collection.mutable.ArrayBuffer.empty[StorageFilePath]
112           // ensure files are sorted in reverse chronological order
113           files.sorted.foreach { file =>
114             if (file.file.action == StorageFileAction.Append) {
115               group += file
116             } else {
117               if (group.nonEmpty) {
118                 groups += group.toSeq
119                 group = scala.collection.mutable.ArrayBuffer.empty[StorageFilePath]
120               }
121               groups += Seq(file)
122             }
123           }
124           if (group.nonEmpty) {
125             groups += group.toSeq // add the last group
126           }
127 
128           // each chained reader task will register at most groups.length parties
129           parties += groups.length
130           if (parties > PhaserUtils.MaxParties) {
131             parties = groups.length
132             child = new Phaser(phaser)
133           }
134           child.register() // register new task
135           es.submit(new ChainedReaderTask(es, child, reader, groups.head, groups.tail.toSeq, queue))
136         }
137       } catch {
138         case NonFatal(e) => es.shutdownNow(); throw e
139       } finally {
140         phaser.arriveAndDeregister()
141       }
142 
143       new FileSystemThreadedReader(es, phaser, queue)
144     }
145   }

147   /**
148     * Reads a file
149     *
150     * @param reader reader
151     * @param file file to read
152     * @param mods collection to track modifications
153     * @return
154     */
155   private def read(
156       reader: FileSystemPathReader,
157       file: StorageFilePath,
158       mods: scala.collection.mutable.Set[String]): CloseableIterator[SimpleFeature] = {
159     file.file.action match {
160       case StorageFileAction.Append => new AppendingReaderIterator(reader, file.path, mods)
161       case StorageFileAction.Modify => new ModifyingReaderIterator(reader, file.path, mods)
162       case StorageFileAction.Delete => new DeletingReaderIterator(reader, file.path, mods)
163       case _ => throw new UnsupportedOperationException(s"Unexpected storage action: ${file.file.action}")
164     }
165   }

167   /**
168     * Performs a set of chained (dependent) reads. The groups must be handled serially, but the files within a
169     * given group can be read in parallel.
170     *
171     * @param es executor service for submitting new read tasks
172     * @param phaser phaser to track run lifecycle
173     * @param reader reader
174     * @param group current group of files that can be read in parallel
175     * @param chain remaining groups of files that must be read sequentially
176     * @param queue result queue
177     * @param mods modifications/deletes for this group of files
178     */
179   private class ChainedReaderTask(
180       es: ExecutorService,
181       phaser: Phaser,
182       reader: FileSystemPathReader,
183       group: Seq[StorageFilePath],
184       chain: Seq[Seq[StorageFilePath]],
185       queue: BlockingQueue[SimpleFeature],
186       mods: scala.collection.mutable.Set[String] = scala.collection.mutable.HashSet.empty[String]
187     ) extends Runnable {
188 
189     override def run(): Unit = {
190       val child = new Phaser(1) {
191         override protected def onAdvance(phase: Int, registeredParties: Int): Boolean = {
192           // when this group is done, submit the next group for processing
193           try {
194             if (chain.nonEmpty) {
195               phaser.register() // register new task
196               es.submit(new ChainedReaderTask(es, phaser, reader, chain.head, chain.tail, queue, mods))
197             }
198             true // return true to indicate the phaser should terminate
199           } finally {
200             phaser.arriveAndDeregister()
201           }
202         }
203       }
204       try {
205         group.foreach { file =>
206           child.register() // register new task
207           es.submit(new ReaderTask(child, queue, file.path, read(reader, file, mods)))
208         }
209       } finally {
210         child.arriveAndDeregister()
211       }
212     }
213   }

241   /**
242     * Reads a file, skipping features marked as modified
243     *
244     * @param reader reader
245     * @param path file path
246     * @param mods set of modified feature IDs that shouldn't be returned
247     */
248   private class AppendingReaderIterator(
249       reader: FileSystemPathReader,
250       path: Path,
251       mods: scala.collection.Set[String]
252     ) extends CloseableIterator[SimpleFeature] {
253 
254     logger.debug(s"Reading file $path")
255 
256     private var count = 0
257 
258     private val delegate = {
259       val iter = CloseableIterator.wrap(reader.read(path))
260       if (mods.isEmpty) {
261         iter
262       } else {
263         iter.filter { f => if (mods.contains(f.getID)) { count += 1; false } else { true } }
264       }
265     }
266 
267     override def hasNext: Boolean = delegate.hasNext
268 
269     override def next(): SimpleFeature = {
270       count += 1
271       // need to copy the feature as it can be re-used
272       ScalaSimpleFeature.copy(delegate.next())
273     }
274 
275     override def close(): Unit = {
276       logger.debug(s"File $path produced $count records")
277       delegate.close()
278     }
279   }

318     * @param path file path

321   private class DeletingReaderIterator(
322       reader: FileSystemPathReader,
323       path: Path,
324       mods: scala.collection.mutable.Set[String]
325     ) extends CloseableIterator[SimpleFeature] {
326 
327     logger.debug(s"Reading deleting file $path")
328 
329     try {
330       var count = 0
331       WithClose(reader.read(path)) { features =>
332         while (features.hasNext) {
333           mods.add(features.next().getID)
334           count += 1
335         }
336       }
337       logger.debug(s"File $path produced $count deleted records")
338     } catch {
339       case NonFatal(e) => logger.error(s"Error reading deleting file $path", e)
340     }
341 
342     override def hasNext: Boolean = false
343 
344     override def next(): SimpleFeature = Iterator.empty.next()
