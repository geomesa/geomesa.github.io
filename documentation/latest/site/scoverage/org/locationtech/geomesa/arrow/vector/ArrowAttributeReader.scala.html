85   /**
86     * Creates a sequence of attribute readers based on the attributes of the simple feature type. There
87     * will be one reader per attribute.
88     *
89     * @param sft simple feature type
90     * @param vector simple feature vector to read from
91     * @param dictionaries dictionaries, if any
92     * @param encoding encoding options
93     * @return sequence of readers
94     */
95   def apply(sft: SimpleFeatureType,
96             vector: StructVector,
97             dictionaries: Map[String, ArrowDictionary],
98             encoding: SimpleFeatureEncoding = SimpleFeatureEncoding.Min): Seq[ArrowAttributeReader] = {
99     sft.getAttributeDescriptors.asScala.map { descriptor =>
100       val name = descriptor.getLocalName
101       val dictionary = dictionaries.get(name).orElse(dictionaries.get(descriptor.getLocalName))
102       apply(descriptor, vector.getChild(name), dictionary, encoding)
103     }.toSeq
104   }

107             vector: FieldVector,
108             dictionary: Option[ArrowDictionary],
109             encoding: SimpleFeatureEncoding): ArrowAttributeReader = {
110     apply(ObjectType.selectType(descriptor), vector, dictionary, encoding)
111   }

113   /**
114     * Creates an attribute reader for a single attribute
115     *
116     * @param bindings object bindings, the attribute type plus any subtypes (e.g. for lists or maps)
117     * @param vector the simple feature vector to read from
118     * @param dictionary the dictionary for the attribute, if any
119     * @param encoding encoding options
120     * @return reader
121     */
122   def apply(bindings: Seq[ObjectType],
123             vector: FieldVector,
124             dictionary: Option[ArrowDictionary],
125             encoding: SimpleFeatureEncoding): ArrowAttributeReader = {
126     dictionary match {
127       case None =>
128         bindings.head match {
129           case ObjectType.GEOMETRY => ArrowGeometryReader(vector, bindings(1), encoding.geometry)
130           case ObjectType.DATE     => ArrowDateReader(vector, encoding.date)
131           case ObjectType.STRING   => new ArrowStringReader(vector.asInstanceOf[VarCharVector])
132           case ObjectType.INT      => new ArrowIntReader(vector.asInstanceOf[IntVector])
133           case ObjectType.LONG     => new ArrowLongReader(vector.asInstanceOf[BigIntVector])
134           case ObjectType.FLOAT    => new ArrowFloatReader(vector.asInstanceOf[Float4Vector])
135           case ObjectType.DOUBLE   => new ArrowDoubleReader(vector.asInstanceOf[Float8Vector])
136           case ObjectType.BOOLEAN  => new ArrowBooleanReader(vector.asInstanceOf[BitVector])
137           case ObjectType.LIST     => new ArrowListReader(vector.asInstanceOf[ListVector], bindings(1), encoding)
138           case ObjectType.MAP      => new ArrowMapReader(vector.asInstanceOf[StructVector], bindings(1), bindings(2), encoding)
139           case ObjectType.BYTES    => new ArrowByteReader(vector.asInstanceOf[VarBinaryVector])
140           case ObjectType.UUID     => new ArrowUuidReader(vector.asInstanceOf[FixedSizeListVector])
141           case _ => throw new IllegalArgumentException(s"Unexpected object type ${bindings.head}")
142         }
143 
144       case Some(dict) =>
145         vector match {
146           case v: TinyIntVector  => new ArrowDictionaryByteReader(v, dict)
147           case v: SmallIntVector => new ArrowDictionaryShortReader(v, dict)
148           case v: IntVector      => new ArrowDictionaryIntReader(v, dict)
149 
150           case v: ListVector =>
151             val nested = v.getField.getChildren.get(0).getType
152             if (nested == MinorType.TINYINT.getType) {
153               new ArrowListDictionaryByteReader(v, dict)
154             } else if (nested == MinorType.SMALLINT.getType) {
155               new ArrowListDictionaryShortReader(v, dict)
156             } else if (nested == MinorType.INT.getType) {
157               new ArrowListDictionaryIntReader(v, dict)
158             } else {
159               throw new IllegalArgumentException(s"Unexpected dictionary vector: $vector")
160             }
161 
162           case _ => throw new IllegalArgumentException(s"Unexpected dictionary vector: $vector")
163         }
164     }
165   }

396   object ArrowGeometryReader {
397     def apply(vector: FieldVector, binding: ObjectType, encoding: Encoding): ArrowAttributeReader = {
398       if (binding == ObjectType.POINT) {
399         val delegate: AbstractPointVector[_] = encoding match {
400           case Encoding.Min => new PointFloatVector(vector.asInstanceOf[FixedSizeListVector])
401           case Encoding.Max => new PointVector(vector.asInstanceOf[FixedSizeListVector])
402         }
403         new ArrowPointReader(vector, delegate)
404       } else if (binding == ObjectType.LINESTRING) {
405         val delegate: AbstractLineStringVector[_] = encoding match {
406           case Encoding.Min => new LineStringFloatVector(vector.asInstanceOf[ListVector])
407           case Encoding.Max => new LineStringVector(vector.asInstanceOf[ListVector])
408         }
409         new ArrowLineStringReader(vector, delegate)
410       } else {
411         val delegate: GeometryVector[_ <: Geometry, _] = if (binding == ObjectType.POLYGON) {
412           encoding match {
413             case Encoding.Min => new PolygonFloatVector(vector.asInstanceOf[ListVector])
414             case Encoding.Max => new PolygonVector(vector.asInstanceOf[ListVector])
415           }
416         } else if (binding == ObjectType.MULTILINESTRING) {
417           encoding match {
418             case Encoding.Min => new MultiLineStringFloatVector(vector.asInstanceOf[ListVector])
419             case Encoding.Max => new MultiLineStringVector(vector.asInstanceOf[ListVector])
420           }
421         } else if (binding == ObjectType.MULTIPOLYGON) {
422           encoding match {
423             case Encoding.Min => new MultiPolygonFloatVector(vector.asInstanceOf[ListVector])
424             case Encoding.Max => new MultiPolygonVector(vector.asInstanceOf[ListVector])
425           }
426         } else if (binding == ObjectType.MULTIPOINT) {
427           encoding match {
428             case Encoding.Min => new MultiPointFloatVector(vector.asInstanceOf[ListVector])
429             case Encoding.Max => new MultiPointVector(vector.asInstanceOf[ListVector])
430           }
431         } else if (binding == ObjectType.GEOMETRY) {
432           new WKBGeometryVector(vector.asInstanceOf[VarBinaryVector])
433         } else if (binding == ObjectType.GEOMETRY_COLLECTION) {
434           new WKBGeometryVector(vector.asInstanceOf[VarBinaryVector])
435         } else {
436           throw new IllegalArgumentException(s"Expected geometry type, got $binding")
437         }
438         new ArrowGeometryReader(vector, delegate)
439       }
440     }
441   }
