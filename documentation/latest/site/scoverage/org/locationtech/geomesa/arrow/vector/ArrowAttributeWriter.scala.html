66   /**
67     * Writer for feature ID. The return FeatureWriter expects to be passed the entire SimpleFeature, not
68     * just the feature ID string (this is to support cached UUIDs).
69     *
70     * @param vector simple feature vector
71     * @param encoding actually write the feature ids, or omit them, in which case the writer is a no-op
72     * @return feature ID writer
73     */
74   def id(sft: SimpleFeatureType, encoding: SimpleFeatureEncoding, vector: StructVector): ArrowAttributeWriter = {
75     val name = SimpleFeatureVector.FeatureIdField
76     encoding.fids match {
77       case None => ArrowNoFidWriter
78       case Some(Encoding.Min) if sft.isUuid => new ArrowFeatureIdMinimalUuidWriter(name, VectorFactory(vector))
79       case Some(Encoding.Max) if sft.isUuid => new ArrowFeatureIdUuidWriter(name, VectorFactory(vector))
80       case Some(Encoding.Min) => new ArrowFeatureIdMinimalStringWriter(name, VectorFactory(vector))
81       case Some(Encoding.Max) => new ArrowFeatureIdStringWriter(name, VectorFactory(vector))
82     }
83   }

165   /**
166    * Low-level method to create a single attribute writer
167    *
168    * @param name attribute name
169    * @param bindings object bindings, the attribute type plus any subtypes (e.g. for lists or maps)
170    * @param dictionary the dictionary for the attribute, if any
171    * @param metadata vector metadata encoded in the field - generally the encoded attribute descriptor
172    * @param encoding encoding options
173    * @param factory parent vector or allocator
174    * @return
175    */
176   def apply(
177       name: String,
178       bindings: Seq[ObjectType],
179       dictionary: Option[ArrowDictionary],
180       metadata: Map[String, String],
181       encoding: SimpleFeatureEncoding,
182       factory: VectorFactory): ArrowAttributeWriter = {
183     dictionary match {
184       case Some(dict) =>
185         ArrowAttributeWriter.dictionary(name, dict, bindings.head == ObjectType.LIST, metadata, factory)
186 
187       case None =>
188         bindings.head match {
189           case ObjectType.STRING   => new ArrowStringWriter(name, metadata, factory)
190           case ObjectType.DATE     => date(name, encoding.date, metadata, factory)
191           case ObjectType.INT      => new ArrowIntWriter(name, metadata, factory)
192           case ObjectType.LONG     => new ArrowLongWriter(name, metadata, factory)
193           case ObjectType.FLOAT    => new ArrowFloatWriter(name, metadata, factory)
194           case ObjectType.DOUBLE   => new ArrowDoubleWriter(name, metadata, factory)
195           case ObjectType.GEOMETRY => geometry(name, bindings(1), encoding, metadata, factory)
196           case ObjectType.BOOLEAN  => new ArrowBooleanWriter(name, metadata, factory)
197           case ObjectType.LIST     => new ArrowListWriter(name, bindings(1), encoding, metadata, factory)
198           case ObjectType.MAP      => new ArrowMapWriter(name, bindings(1), bindings(2), encoding, metadata, factory)
199           case ObjectType.BYTES    => new ArrowBytesWriter(name, metadata, factory)
200           case ObjectType.UUID     => new ArrowUuidWriter(name, metadata, factory)
201           case _ => throw new IllegalArgumentException(s"Unexpected object type ${bindings.head}")
202         }
203     }
204   }

206   private def dictionary(
207       name: String,
208       dictionary: ArrowDictionary,
209       isList: Boolean,
210       metadata: Map[String, String],
211       factory: VectorFactory): ArrowAttributeWriter = {
212     (dictionary.encoding.getIndexType.getBitWidth, isList) match {
213       case (8,  false) => new ArrowDictionaryByteWriter(name, dictionary, metadata, factory)
214       case (16, false) => new ArrowDictionaryShortWriter(name, dictionary, metadata, factory)
215       case (32, false) => new ArrowDictionaryIntWriter(name, dictionary, metadata, factory)
216       case (8,  true)  => new ArrowListDictionaryByteWriter(name, dictionary, metadata, factory)
217       case (16, true)  => new ArrowListDictionaryShortWriter(name, dictionary, metadata, factory)
218       case (32, true)  => new ArrowListDictionaryIntWriter(name, dictionary, metadata, factory)
219       case (w, _)      => throw new IllegalArgumentException(s"Unsupported dictionary encoding width: $w")
220     }
221   }

235       name: String,
236       binding: ObjectType,
237       encoding: SimpleFeatureEncoding,

239       factory: VectorFactory): ArrowGeometryWriter = {
240     val m = metadata.asJava

243       case (ObjectType.POINT, Encoding.Min, FromAllocator(c))           => new PointFloatVector(name, c, m)
244       case (ObjectType.POINT, Encoding.Max, FromStruct(c))              => new PointVector(name, c, m)
245       case (ObjectType.POINT, Encoding.Max, FromAllocator(c))           => new PointVector(name, c, m)
246       case (ObjectType.LINESTRING, Encoding.Min, FromStruct(c))         => new LineStringFloatVector(name, c, m)
247       case (ObjectType.LINESTRING, Encoding.Min, FromAllocator(c))      => new LineStringFloatVector(name, c, m)
248       case (ObjectType.LINESTRING, Encoding.Max, FromStruct(c))         => new LineStringVector(name, c, m)
249       case (ObjectType.LINESTRING, Encoding.Max, FromAllocator(c))      => new LineStringVector(name, c, m)
250       case (ObjectType.POLYGON, Encoding.Min, FromStruct(c))            => new PolygonFloatVector(name, c, m)
251       case (ObjectType.POLYGON, Encoding.Min, FromAllocator(c))         => new PolygonFloatVector(name, c, m)
252       case (ObjectType.POLYGON, Encoding.Max, FromStruct(c))            => new PolygonVector(name, c, m)
253       case (ObjectType.POLYGON, Encoding.Max, FromAllocator(c))         => new PolygonVector(name, c, m)
254       case (ObjectType.MULTILINESTRING, Encoding.Min, FromStruct(c))    => new MultiLineStringFloatVector(name, c, m)
255       case (ObjectType.MULTILINESTRING, Encoding.Min, FromAllocator(c)) => new MultiLineStringFloatVector(name, c, m)
256       case (ObjectType.MULTILINESTRING, Encoding.Max, FromStruct(c))    => new MultiLineStringVector(name, c, m)
257       case (ObjectType.MULTILINESTRING, Encoding.Max, FromAllocator(c)) => new MultiLineStringVector(name, c, m)
258       case (ObjectType.MULTIPOLYGON, Encoding.Min, FromStruct(c))       => new MultiPolygonFloatVector(name, c, m)
259       case (ObjectType.MULTIPOLYGON, Encoding.Min, FromAllocator(c))    => new MultiPolygonFloatVector(name, c, m)
260       case (ObjectType.MULTIPOLYGON, Encoding.Max, FromStruct(c))       => new MultiPolygonVector(name, c, m)
261       case (ObjectType.MULTIPOLYGON, Encoding.Max, FromAllocator(c))    => new MultiPolygonVector(name, c, m)
262       case (ObjectType.MULTIPOINT, Encoding.Min, FromStruct(c))         => new MultiPointFloatVector(name, c, m)
263       case (ObjectType.MULTIPOINT, Encoding.Min, FromAllocator(c))      => new MultiPointFloatVector(name, c, m)
264       case (ObjectType.MULTIPOINT, Encoding.Max, FromStruct(c))         => new MultiPointVector(name, c, m)
265       case (ObjectType.MULTIPOINT, Encoding.Max, FromAllocator(c))      => new MultiPointVector(name, c, m)
266       case (ObjectType.GEOMETRY, _, FromStruct(c))                      => new WKBGeometryVector(name, c, m)
267       case (ObjectType.GEOMETRY, _, FromAllocator(c))                   => new WKBGeometryVector(name, c, m)
268       case (ObjectType.GEOMETRY_COLLECTION, _, FromStruct(c))           => new WKBGeometryVector(name, c, m)
269       case (ObjectType.GEOMETRY_COLLECTION, _, FromAllocator(c))        => new WKBGeometryVector(name, c, m)
270       case (_, _, FromList(_)) => throw new UnsupportedOperationException("Geometry lists are not supported")
271       case _ => throw new IllegalArgumentException(s"Unexpected geometry type $binding")
272     }
273     val geometryVector = vector.asInstanceOf[GeometryVector[Geometry, FieldVector]]
274     geometryVector.setFlipAxisOrder(encoding.flipAxisOrder)
275 
276     new ArrowGeometryWriter(name, geometryVector)
277   }

304   /**
305    * Converts a list value into a list of dictionary bytes and writes it
306    *
307    * @param name attribute/field name
308    * @param dictionary dictionary values
309    * @param metadata field metadata
310    * @param factory vector factory
311    */
312   class ArrowListDictionaryByteWriter(
313       val name: String,
314       val dictionary: ArrowDictionary,
315       metadata: Map[String, String],
316       factory: VectorFactory
317     ) extends ArrowDictionaryWriter {
318 
319     override val vector: ListVector = factory.apply(name, MinorType.LIST, metadata)
320 
321     private val inner: TinyIntVector =
322       FromList(vector).apply(null, MinorType.TINYINT, dictionary.encoding, Map.empty)
323 
324     override def apply(i: Int, value: AnyRef): Unit = {
325       if (vector.getLastSet >= i) {
326         vector.setLastSet(i - 1)
327       }
328       val start = vector.startNewValue(i)
329       // note: null gets converted to empty list
330       if (value == null) {
331         vector.endValue(i, 0)
332       } else {
333         val list = value.asInstanceOf[java.util.List[AnyRef]]
334         var offset = 0
335         while (offset < list.size()) {
336           // note: nulls get encoded in the dictionary
337           inner.setSafe(start + offset, dictionary.index(list.get(offset)).toByte)
338           offset += 1
339         }
340         vector.endValue(i, offset)
341       }
342     }
343   }

492   class ArrowUuidWriter(val name: String, metadata: Map[String, String], factory: VectorFactory)
493     extends ArrowAttributeWriter {
494     val fieldType: FieldType = new FieldType(true, new ArrowType.FixedSizeList(2), null, metadata.asJava)
495     override val vector: FixedSizeListVector = factory.apply(name, fieldType)
496 
497     private val bits = {
498       val result = vector.addOrGetVector[BigIntVector](FieldType.nullable(MinorType.BIGINT.getType))
499       if (result.isCreated) {
500         result.getVector.allocateNew()
501       }
502       result.getVector
503     }
504 
505     override def apply(i: Int, value: AnyRef): Unit = {
506       if (value == null) {
507         vector.setNull(i) // note: calls .setSafe internally
508       } else {
509         val uuid = value.asInstanceOf[UUID]
510         val (msb, lsb) = (uuid.getMostSignificantBits, uuid.getLeastSignificantBits)
511         vector.setNotNull(i)
512         bits.setSafe(i * 2, msb)
513         bits.setSafe(i * 2 + 1, lsb)
514       }
515     }
516   }
