23 /**
24   * Simple feature backed by an arrow vector. Attributes are lazily evaluated - this allows filters to only
25   * examine the relevant arrow vectors for optimized reads, but also means that they are tied to the underlying
26   * vectors
27   *
28   * @param sft simple feature type
29   * @param idReader id reader
30   * @param attributeReaders attribute readers
31   * @param index index of the feature in the arrow vector
32   */
33 class ArrowSimpleFeature(sft: SimpleFeatureType,
34                          idReader: ArrowAttributeReader,
35                          attributeReaders: Array[ArrowAttributeReader],
36                          private [arrow] var index: Int) extends SimpleFeature {
37   import scala.collection.JavaConverters._
38 
39   private lazy val geomIndex = sft.indexOf(sft.getGeometryDescriptor.getLocalName)
40 
41   override def getAttribute(i: Int): AnyRef = attributeReaders(i).apply(index)
42 
43   /**
44     * Gets the underlying arrow reader for this feature
45     *
46     * @param i attribute to be read
47     * @return reader
48     */
49   def getReader(i: Int): ArrowAttributeReader = attributeReaders(i)
50 
51   /**
52     * Gets the index of this feature, for use with the attribute reader
53     *
54     * @return
55     */
56   def getIndex: Int = index
57 
58   override def getID: String = idReader.apply(index).asInstanceOf[String]
59   override def getIdentifier: FeatureId = new ImmutableFeatureId(getID)
60 
61   override def getUserData: jMap[AnyRef, AnyRef] = Map.empty[AnyRef, AnyRef].asJava
62 
63   override def getType: SimpleFeatureType = sft
64   override def getFeatureType: SimpleFeatureType = sft
65   override def getName: Name = sft.getName
66 
67   override def getAttribute(name: Name): AnyRef = getAttribute(name.getLocalPart)
68   override def getAttribute(name: String): AnyRef = {
69     val index = sft.indexOf(name)
70     if (index == -1) { null } else { getAttribute(index) }
71   }
72 
73   override def getDefaultGeometry: AnyRef = getAttribute(geomIndex)
74   override def getAttributeCount: Int = sft.getAttributeCount
75 
76   override def getBounds: BoundingBox = getDefaultGeometry match {
77     case g: Geometry => new ReferencedEnvelope(g.getEnvelopeInternal, sft.getCoordinateReferenceSystem)
78     case _           => new ReferencedEnvelope(sft.getCoordinateReferenceSystem)
79   }
80 
81   override def getAttributes: jList[AnyRef] = {
82     val attributes = new java.util.ArrayList[AnyRef](sft.getAttributeCount)
83     var i = 0
84     while (i < sft.getAttributeCount) {
85       attributes.add(getAttribute(i))
86       i += 1
87     }
88     attributes
89   }

101   override def setAttribute(name: String, value: Object): Unit = throw new UnsupportedOperationException()

103   override def setAttributes(vals: jList[Object]): Unit = throw new UnsupportedOperationException()
104   override def setAttributes(vals: Array[Object]): Unit = throw new UnsupportedOperationException()

107   override def setValue(newValue: Object): Unit = throw new UnsupportedOperationException()
108   override def setValue(values: jCollection[Property]): Unit = throw new UnsupportedOperationException()
109 
110   override def isNillable: Boolean = true
111   override def validate(): Unit = throw new UnsupportedOperationException()
112 
113   override def toString: String = s"ArrowSimpleFeature:$getID:${getAttributes.asScala.mkString("|")}"
114 
115   override def hashCode: Int = Objects.hash(getID, getName, getAttributes)
116 
117   override def equals(obj: scala.Any): Boolean = obj match {
118     case other: SimpleFeature =>
119       getID == other.getID && getName == other.getName && getAttributes == other.getAttributes
120     case _ => false
121   }
