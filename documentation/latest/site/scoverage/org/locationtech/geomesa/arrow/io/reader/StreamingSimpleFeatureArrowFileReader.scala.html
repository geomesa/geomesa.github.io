30 /**
31   * Streams features from an input stream - note that a feature may not be valid after a call to `.next()`,
32   * as the underlying data may be reclaimed
33   *
34   * @param is creates a new input stream
35   */
36 class StreamingSimpleFeatureArrowFileReader(is: () => InputStream) extends SimpleFeatureArrowFileReader  {
37 
38   import scala.collection.JavaConverters._
39 
40   private val allocator = ArrowAllocator("streaming-file-reader")
41 
42   private val opened = new ConcurrentLinkedDeque[AutoCloseable](Collections.singletonList(allocator))
43 
44   lazy val (sft, sort) = {
45     var sft: SimpleFeatureType = null
46     var sort: Option[(String, Boolean)] = None
47     WithClose(new ArrowStreamReader(is(), allocator)) { reader =>
48       val root = reader.getVectorSchemaRoot
49       require(root.getFieldVectors.size() == 1 && root.getFieldVectors.get(0).isInstanceOf[StructVector], "Invalid file")
50       val underlying = root.getFieldVectors.get(0).asInstanceOf[StructVector]
51       sft = SimpleFeatureVector.getFeatureType(underlying)._1
52       sort = getSortFromMetadata(root.getSchema.getCustomMetadata)
53     }
54     (sft, sort)
55   }
56 
57   override lazy val dictionaries: Map[String, ArrowDictionary] = {
58     val dicts = WithClose(is()) { is =>
59       val reader = new ArrowStreamReader(is, allocator)
60       opened.addFirst(reader)
61       val root = reader.getVectorSchemaRoot
62       require(root.getFieldVectors.size() == 1 && root.getFieldVectors.get(0).isInstanceOf[StructVector], "Invalid file")
63       val underlying = root.getFieldVectors.get(0).asInstanceOf[StructVector]
64       reader.loadNextBatch() // load the first batch so we get any dictionaries
65       val encoding = SimpleFeatureVector.getFeatureType(underlying)._2
66       // load any dictionaries into memory
67       loadDictionaries(underlying.getField.getChildren.asScala.toSeq, reader, encoding)
68     }
69     dicts.values.foreach(opened.addFirst)
70     dicts
71   }
72 
73   override def vectors: Seq[SimpleFeatureVector] = throw new NotImplementedError()
74 
75   override def features(filter: Filter): CloseableIterator[ArrowSimpleFeature] = {
76     val stream = is()
77     val skip = new SkipIndicator
78     val nextBatch = SimpleFeatureArrowFileReader.features(sft, filter, skip, sort, dictionaries)
79 
80     // we track all the readers and close at the end to avoid closing the input stream prematurely
81     val readers = ArrayBuffer.empty[StreamingSingleFileReader]
82     // reader for current logical 'file'
83     var reader: StreamingSingleFileReader = null
84 
85     new CloseableIterator[ArrowSimpleFeature] {
86       private var done = false
87       private var batch: Iterator[ArrowSimpleFeature] = Iterator.empty
88 
89       @scala.annotation.tailrec
90       override def hasNext: Boolean = {
91         if (done) {
92           false
93         } else if (batch.hasNext) {
94           true
95         } else if (!skip.skip && stream.available() > 0) {
96           // new logical file
97           reader = new StreamingSingleFileReader(stream, allocator)
98           readers.append(reader)
99           batch = reader.features(nextBatch, skip)
100           hasNext
101         } else {
102           done = true
103           false
104         }
105       }
106 
107       override def next(): ArrowSimpleFeature = batch.next()
108 
109       override def close(): Unit = readers.foreach(_.close())
110     }
111   }
112 
113   override def close(): Unit = CloseWithLogging.raise(opened.asScala)
114 }

116 object StreamingSimpleFeatureArrowFileReader {
117 
118   import scala.collection.JavaConverters._
119 
120   /**
121    * Reads a single logical arrow 'file' from the stream, which may contain multiple record batches
122    */
123   private class StreamingSingleFileReader(is: InputStream, allocator: BufferAllocator) extends Closeable {
124 
125     private val reader = new ArrowStreamReader(is, allocator)
126     private val root = reader.getVectorSchemaRoot
127     require(root.getFieldVectors.size() == 1 && root.getFieldVectors.get(0).isInstanceOf[StructVector], "Invalid file")
128     private val underlying = root.getFieldVectors.get(0).asInstanceOf[StructVector]
129     private var done = !reader.loadNextBatch() // load the first batch so we get any dictionaries
130 
131     val (sft, encoding) = SimpleFeatureVector.getFeatureType(underlying)
132 
133     // load any dictionaries into memory
134     val dictionaries: Map[String, ArrowDictionary] =
135       SimpleFeatureArrowFileReader.loadDictionaries(underlying.getField.getChildren.asScala.toSeq, reader, encoding)
136     private val vector = new SimpleFeatureVector(sft, underlying, dictionaries, encoding, None)
137 
138     def metadata: java.util.Map[String, String] = reader.getVectorSchemaRoot.getSchema.getCustomMetadata
139 
140     // iterator of simple features read from the input stream
141     def features(nextBatch: VectorToIterator, skip: SkipIndicator): Iterator[ArrowSimpleFeature] = {
142       if (done) { Iterator.empty } else {
143         new Iterator[ArrowSimpleFeature] {
144           private var batch: Iterator[ArrowSimpleFeature] = nextBatch(vector)
145 
146           @scala.annotation.tailrec
147           override def hasNext: Boolean = {
148             if (done) {
149               false
150             } else if (batch.hasNext) {
151               true
152             } else if (!skip.skip && reader.loadNextBatch()) {
153               batch = nextBatch(vector)
154               hasNext
155             } else {
156               done = true
157               false
158             }
159           }
160 
161           override def next(): ArrowSimpleFeature = batch.next()
162         }
163       }
164     }
165 
166     override def close(): Unit = CloseWithLogging.raise(dictionaries.values ++ Seq(reader, vector))
167   }
