96   /**
97    * Create a transfer pair between two vectors. This handles geometry vectors correctly, which the underlying
98    * arrow transfer pairs do not.
99    *
100    * @param from from vector
101    * @param to to vector
102    * @return transfer(fromIndex, toIndex)
103    */
104   def createTransferPair(sft: SimpleFeatureType, from: FieldVector, to: FieldVector): (Int, Int) => Unit = {
105     val i = sft.indexOf(from.getField.getName)
106     lazy val binding = sft.getDescriptor(i).getType.getBinding
107     if (i != -1 && classOf[Geometry].isAssignableFrom(binding)) {
108       // geometry vectors use FixedSizeList vectors, for which transfer pairs aren't implemented
109       val fromGeom = GeometryFields.wrap(from, binding).asInstanceOf[GeometryVector[Geometry, FieldVector]]
110       val toGeom = GeometryFields.wrap(to, binding).asInstanceOf[GeometryVector[Geometry, FieldVector]]
111       (fromIndex: Int, toIndex: Int) => fromGeom.transfer(fromIndex, toIndex, toGeom)
112     } else {
113       val transfer = from.makeTransferPair(to)
114       (fromIndex: Int, toIndex: Int) => transfer.copyValueSafe(fromIndex, toIndex)
115     }
116   }

118   /**
119    * Write out the header, dictionaries, and first batch of an arrow streaming file
120    *
121    * @param result vector loaded with first batch
122    * @param dictionaries dictionaries
123    * @param sort sort
124    * @param count number of records in first batch
125    * @return
126    */
127   def writeHeaderAndFirstBatch(
128       result: SimpleFeatureVector,
129       dictionaries: Map[String, ArrowDictionary],
130       ipcOpts: IpcOption,
131       sort: Option[(String, Boolean)],
132       count: Int): Array[Byte] = {
133     val metadata = sort match {
134       case None => null
135       case Some((sortBy, reverse)) => getSortAsMetadata(sortBy, reverse)
136     }
137     // note: don't close root as it will close the underlying vector
138     val root = createRoot(result.underlying, metadata)
139     root.setRowCount(count)
140     val os = new ByteArrayOutputStream()
141     WithClose(SimpleFeatureArrowFileWriter.provider(dictionaries, result.encoding)) { provider =>
142       WithClose(new ArrowStreamWriter(root, provider, Channels.newChannel(os), ipcOpts)) { writer =>
143         writer.writeBatch()
144         os.toByteArray
145       }
146     }
147   }

149   /**
150    * Create an arrow file from record batches
151    *
152    * @param sft simple feature type
153    * @param dictionaries dictionaries
154    * @param encoding feature encoding
155    * @param sort sorting of the batches, if any
156    * @param batches batches
157    * @param firstBatchHasHeader does the first batch have the arrow file header or not
158    * @return
159    */
160   def createFileFromBatches(
161       sft: SimpleFeatureType,
162       dictionaries: Map[String, ArrowDictionary],
163       encoding: SimpleFeatureEncoding,
164       ipcOpts: IpcOption,
165       sort: Option[(String, Boolean)],
166       batches: CloseableIterator[Array[Byte]],
167       firstBatchHasHeader: Boolean): CloseableIterator[Array[Byte]] = {
168     val ft = if (ipcOpts.write_legacy_ipc_format) { legacyFooter } else { footer }
169     new ArrowFileIterator(sft, dictionaries, encoding, sort, ipcOpts, batches, firstBatchHasHeader, ft)
170   }

172   // per arrow streaming format footer is the encoded int -1, 0
173   private def footer: Array[Byte] = Array[Byte](-1, -1, -1, -1, 0, 0, 0, 0)
174   private def legacyFooter: Array[Byte] = Array[Byte](0, 0, 0, 0)

176   private class ArrowFileIterator(
177       sft: SimpleFeatureType,
178       dictionaries: Map[String, ArrowDictionary],
179       encoding: SimpleFeatureEncoding,
180       sort: Option[(String, Boolean)],
181       ipcOpts: IpcOption,
182       batches: CloseableIterator[Array[Byte]],
183       firstBatchHasHeader: Boolean,
184       footer: Array[Byte]
185     ) extends CloseableIterator[Array[Byte]] {
186 
187     private var seenBatch = false
188     private var seenFooter = false
189 
190     override def hasNext: Boolean = batches.hasNext || !seenBatch || !seenFooter
191 
192     override def next(): Array[Byte] = {
193       if (seenBatch) {
194         if (batches.hasNext) {
195           batches.next()
196         } else if (seenFooter) {
197           throw new NoSuchElementException("Next on an empty iterator")
198         } else {
199           seenFooter = true
200           footer
201         }
202       } else {
203         seenBatch = true
204         if (batches.hasNext) {
205           if (firstBatchHasHeader) { batches.next } else {
206             // add the file header and dictionaries
207             WithClose(SimpleFeatureVector.create(sft, dictionaries, encoding)) { vector =>
208               new RecordBatchLoader(vector.underlying).load(batches.next)
209               writeHeaderAndFirstBatch(vector, dictionaries, ipcOpts, sort, vector.reader.getValueCount)
210             }
211           }
212         } else {
213           // write out an empty batch so that we get the header and dictionaries
214           WithClose(SimpleFeatureVector.create(sft, dictionaries, encoding, 0)) { vector =>
215             writeHeaderAndFirstBatch(vector, dictionaries, ipcOpts, sort, 0)
216           }
217         }
218       }
219     }
220 
221     override def close(): Unit = batches.close()
222   }
