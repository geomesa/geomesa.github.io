95   /**
96    * Create a transfer pair between two vectors. This handles geometry vectors correctly, which the underlying
97    * arrow transfer pairs do not.
98    *
99    * @param from from vector
100    * @param to to vector
101    * @return transfer(fromIndex, toIndex)
102    */
103   def createTransferPair(sft: SimpleFeatureType, from: FieldVector, to: FieldVector): (Int, Int) => Unit = {
104     val i = sft.indexOf(from.getField.getName)
105     lazy val binding = sft.getDescriptor(i).getType.getBinding
106     if (i != -1 && classOf[Geometry].isAssignableFrom(binding)) {
107       // geometry vectors use FixedSizeList vectors, for which transfer pairs aren't implemented
108       val fromGeom = GeometryFields.wrap(from, binding).asInstanceOf[GeometryVector[Geometry, FieldVector]]
109       val toGeom = GeometryFields.wrap(to, binding).asInstanceOf[GeometryVector[Geometry, FieldVector]]
110       (fromIndex: Int, toIndex: Int) => fromGeom.transfer(fromIndex, toIndex, toGeom)
111     } else {
112       val transfer = from.makeTransferPair(to)
113       (fromIndex: Int, toIndex: Int) => transfer.copyValueSafe(fromIndex, toIndex)
114     }
115   }

117   /**
118    * Write out the header, dictionaries, and first batch of an arrow streaming file
119    *
120    * @param result vector loaded with first batch
121    * @param dictionaries dictionaries
122    * @param sort sort
123    * @param count number of records in first batch
124    * @return
125    */
126   def writeHeaderAndFirstBatch(
127       result: SimpleFeatureVector,
128       dictionaries: Map[String, ArrowDictionary],
129       ipcOpts: IpcOption,
130       sort: Option[(String, Boolean)],
131       count: Int): Array[Byte] = {
132     val metadata = sort match {
133       case None => null
134       case Some((sortBy, reverse)) => getSortAsMetadata(sortBy, reverse)
135     }
136     // note: don't close root as it will close the underlying vector
137     val root = createRoot(result.underlying, metadata)
138     root.setRowCount(count)
139     val os = new ByteArrayOutputStream()
140     WithClose(SimpleFeatureArrowFileWriter.provider(dictionaries, result.encoding)) { provider =>
141       WithClose(new ArrowStreamWriter(root, provider, Channels.newChannel(os), ipcOpts)) { writer =>
142         writer.writeBatch()
143         os.toByteArray
144       }
145     }
146   }

148   /**
149    * Create an arrow file from record batches
150    *
151    * @param sft simple feature type
152    * @param dictionaries dictionaries
153    * @param encoding feature encoding
154    * @param sort sorting of the batches, if any
155    * @param batches batches
156    * @param firstBatchHasHeader does the first batch have the arrow file header or not
157    * @return
158    */
159   def createFileFromBatches(
160       sft: SimpleFeatureType,
161       dictionaries: Map[String, ArrowDictionary],
162       encoding: SimpleFeatureEncoding,
163       ipcOpts: IpcOption,
164       sort: Option[(String, Boolean)],
165       batches: CloseableIterator[Array[Byte]],
166       firstBatchHasHeader: Boolean): CloseableIterator[Array[Byte]] = {
167     val ft = if (ipcOpts.write_legacy_ipc_format) { legacyFooter } else { footer }
168     new ArrowFileIterator(sft, dictionaries, encoding, sort, ipcOpts, batches, firstBatchHasHeader, ft)
169   }

171   // per arrow streaming format footer is the encoded int -1, 0
172   private def footer: Array[Byte] = Array[Byte](-1, -1, -1, -1, 0, 0, 0, 0)
173   private def legacyFooter: Array[Byte] = Array[Byte](0, 0, 0, 0)
174 
175   private class ArrowFileIterator(
176       sft: SimpleFeatureType,
177       dictionaries: Map[String, ArrowDictionary],
178       encoding: SimpleFeatureEncoding,
179       sort: Option[(String, Boolean)],
180       ipcOpts: IpcOption,
181       batches: CloseableIterator[Array[Byte]],
182       firstBatchHasHeader: Boolean,
183       footer: Array[Byte]
184     ) extends CloseableIterator[Array[Byte]] {
185 
186     private var seenBatch = false
187     private var seenFooter = false
188 
189     override def hasNext: Boolean = batches.hasNext || !seenBatch || !seenFooter
190 
191     override def next(): Array[Byte] = {
192       if (seenBatch) {
193         if (batches.hasNext) {
194           batches.next()
195         } else if (seenFooter) {
196           throw new NoSuchElementException("Next on an empty iterator")
197         } else {
198           seenFooter = true
199           footer
200         }
201       } else {
202         seenBatch = true
203         if (batches.hasNext) {
204           if (firstBatchHasHeader) { batches.next } else {
205             // add the file header and dictionaries
206             WithClose(SimpleFeatureVector.create(sft, dictionaries, encoding)) { vector =>
207               new RecordBatchLoader(vector.underlying).load(batches.next)
208               writeHeaderAndFirstBatch(vector, dictionaries, ipcOpts, sort, vector.reader.getValueCount)
209             }
210           }
211         } else {
212           // write out an empty batch so that we get the header and dictionaries
213           WithClose(SimpleFeatureVector.create(sft, dictionaries, encoding, 0)) { vector =>
214             writeHeaderAndFirstBatch(vector, dictionaries, ipcOpts, sort, 0)
215           }
216         }
217       }
218     }
219 
220     override def close(): Unit = batches.close()
221   }
