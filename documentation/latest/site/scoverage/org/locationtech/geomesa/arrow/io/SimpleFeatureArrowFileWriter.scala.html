24 /**
25   * For writing simple features to an arrow file.
26   *
27   * Uses arrow streaming format (no footer).
28   * Closing the SimpleFeatureArrowFileWriter closes the given os.
29   *
30   * @param vector simple feature vector
31   * @param provider dictionary provider
32   * @param os output stream
33   */
34 class SimpleFeatureArrowFileWriter private (
35     vector: SimpleFeatureVector,
36     provider: DictionaryProvider with Closeable,
37     os: OutputStream,
38     ipcOpts: IpcOption,
39     sort: Option[(String, Boolean)],
40     flattenStruct: Boolean = false
41   ) extends Closeable with Flushable with LazyLogging {
42   private val metadata = sort.map { case (field, reverse) => getSortAsMetadata(field, reverse) }.orNull
43   private val root = {
44     val potentialRoot = createRoot(vector.underlying, metadata)
45 
46     if (flattenStruct) {
47       new VectorSchemaRoot(potentialRoot.getVector(sft.getTypeName))
48     } else {
49       potentialRoot
50     }
51   }
52   private val writer = new ArrowStreamWriter(root, provider, Channels.newChannel(os), ipcOpts)
53 
54   private var index = 0
55 
56   def sft: SimpleFeatureType = vector.sft
57 
58   /**
59     * Buffer a feature to write
60     *
61     * @param sf simple feature
62     */
63   def add(sf: SimpleFeature): Unit = {
64     vector.writer.set(index, sf)
65     index += 1
66   }
67 
68   /**
69    * Writes any currently buffered features to disk. This will create an ArrowBatch
70    * containing the currently buffered features. Note that if there are no features,
71    * an empty record batch will be created
72    */
73   override def flush(): Unit = {
74     vector.writer.setValueCount(index)
75     root.setRowCount(index)
76     writer.writeBatch()
77     vector.clear()
78     index = 0
79   }
80 
81   /**
82     * Close the writer and flush any buffered features
83     */
84   override def close(): Unit = {
85     try {
86       if (index > 0) {
87         flush()
88       }
89       writer.end()
90     } finally {
91       // note: don't close the vector schema root as it closes the vector as well
92       CloseWithLogging.raise(Seq(writer, provider, vector))
93     }
94   }
95 }

101   /**
102    * For writing simple features to an arrow file.
103    *
104    * Uses arrow streaming format (no footer).
105    *
106    * @param os output stream
107    * @param sft simple feature type
108    * @param dictionaries map of field names to dictionary values, used for dictionary encoding fields.
109    *                     All values must be provided up front.
110    * @param encoding encoding options
111    */
112   def apply(
113       os: OutputStream,
114       sft: SimpleFeatureType,
115       dictionaries: Map[String, ArrowDictionary],
116       encoding: SimpleFeatureEncoding,
117       ipcOpts: IpcOption,
118       sort: Option[(String, Boolean)],
119       flattenStruct: Boolean = false): SimpleFeatureArrowFileWriter = {
120     val vector = SimpleFeatureVector.create(sft, dictionaries, encoding)
121     // convert the dictionary values into arrow vectors
122     // make sure we load dictionaries before instantiating the stream writer
123     val provider: DictionaryProvider with Closeable = new DictionaryProvider with Closeable {
124       private val dictionaries = vector.dictionaries.collect { case (_, d) => d.id -> d.toDictionary(vector.encoding) }
125       override def lookup(id: Long): Dictionary = dictionaries(id)
126       override def getDictionaryIds: java.util.Set[java.lang.Long] = dictionaries.keys.map(Long.box).toSet.asJava
127       override def close(): Unit = CloseWithLogging(dictionaries.values)
128     }
129     new SimpleFeatureArrowFileWriter(vector, provider, os, ipcOpts, sort, flattenStruct)
130   }
131 
132   // convert the dictionary values into arrow vectors
133   def provider(
134       dictionaries: Map[String, ArrowDictionary],
135       encoding: SimpleFeatureEncoding): DictionaryProvider with Closeable = {
136     new DictionaryProvider with Closeable {
137       private val dicts = dictionaries.collect { case (_, d) => d.id -> d.toDictionary(encoding) }
138       override def lookup(id: Long): Dictionary = dicts(id)
139       override def getDictionaryIds: java.util.Set[java.lang.Long] = dicts.keys.map(Long.box).toSet.asJava
140       override def close(): Unit = CloseWithLogging(dicts.values)
141     }
142   }
