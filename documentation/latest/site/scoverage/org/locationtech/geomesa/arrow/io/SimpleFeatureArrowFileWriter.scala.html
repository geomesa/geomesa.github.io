23 /**
24   * For writing simple features to an arrow file.
25   *
26   * Uses arrow streaming format (no footer).
27   * Closing the SimpleFeatureArrowFileWriter closes the given os.
28   *
29   * @param vector simple feature vector
30   * @param provider dictionary provider
31   * @param os output stream
32   */
33 class SimpleFeatureArrowFileWriter private (
34     vector: SimpleFeatureVector,
35     provider: DictionaryProvider with Closeable,
36     os: OutputStream,
37     ipcOpts: IpcOption,
38     sort: Option[(String, Boolean)]
39   ) extends Closeable with Flushable with LazyLogging {
40 
41   private val metadata = sort.map { case (field, reverse) => getSortAsMetadata(field, reverse) }.orNull
42   private val root = createRoot(vector.underlying, metadata)
43   private val writer = new ArrowStreamWriter(root, provider, Channels.newChannel(os), ipcOpts)
44 
45   private var index = 0
46 
47   def sft: SimpleFeatureType = vector.sft
48 
49   /**
50     * Buffer a feature to write
51     *
52     * @param sf simple feature
53     */
54   def add(sf: SimpleFeature): Unit = {
55     vector.writer.set(index, sf)
56     index += 1
57   }
58 
59   /**
60    * Writes any currently buffered features to disk. This will create an ArrowBatch
61    * containing the currently buffered features. Note that if there are no features,
62    * an empty record batch will be created
63    */
64   override def flush(): Unit = {
65     vector.writer.setValueCount(index)
66     root.setRowCount(index)
67     writer.writeBatch()
68     vector.clear()
69     index = 0
70   }
71 
72   /**
73     * Close the writer and flush any buffered features
74     */
75   override def close(): Unit = {
76     try {
77       if (index > 0) {
78         flush()
79       }
80       writer.end()
81     } finally {
82       // note: don't close the vector schema root as it closes the vector as well
83       CloseWithLogging.raise(Seq(writer, provider, vector))
84     }
85   }
86 }

92   /**
93    * For writing simple features to an arrow file.
94    *
95    * Uses arrow streaming format (no footer).
96    *
97    * @param os output stream
98    * @param sft simple feature type
99    * @param dictionaries map of field names to dictionary values, used for dictionary encoding fields.
100    *                     All values must be provided up front.
101    * @param encoding encoding options
102    */
103   def apply(
104       os: OutputStream,
105       sft: SimpleFeatureType,
106       dictionaries: Map[String, ArrowDictionary],
107       encoding: SimpleFeatureEncoding,
108       ipcOpts: IpcOption,
109       sort: Option[(String, Boolean)]): SimpleFeatureArrowFileWriter = {
110     val vector = SimpleFeatureVector.create(sft, dictionaries, encoding)
111     // convert the dictionary values into arrow vectors
112     // make sure we load dictionaries before instantiating the stream writer
113     val provider: DictionaryProvider with Closeable = new DictionaryProvider with Closeable {
114       private val dictionaries = vector.dictionaries.collect { case (_, d) => d.id -> d.toDictionary(vector.encoding) }
115       override def lookup(id: Long): Dictionary = dictionaries(id)
116       override def getDictionaryIds: java.util.Set[java.lang.Long] = dictionaries.keys.map(Long.box).toSet.asJava
117       override def close(): Unit = CloseWithLogging(dictionaries.values)
118     }
119     new SimpleFeatureArrowFileWriter(vector, provider, os, ipcOpts, sort)
120   }
121 
122   // convert the dictionary values into arrow vectors
123   def provider(
124       dictionaries: Map[String, ArrowDictionary],
125       encoding: SimpleFeatureEncoding): DictionaryProvider with Closeable = {
126     new DictionaryProvider with Closeable {
127       private val dicts = dictionaries.collect { case (_, d) => d.id -> d.toDictionary(encoding) }
128       override def lookup(id: Long): Dictionary = dicts(id)
129       override def getDictionaryIds: java.util.Set[java.lang.Long] = dicts.keys.map(Long.box).toSet.asJava
130       override def close(): Unit = CloseWithLogging(dicts.values)
131     }
132   }
