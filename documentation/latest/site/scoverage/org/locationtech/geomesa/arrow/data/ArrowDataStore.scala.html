31 class ArrowDataStore(val url: URL, caching: Boolean) extends ContentDataStore with FileDataStore
32     with HasGeoMesaMetadata[String] with HasGeoMesaStats {
33 
34   private var initialized = false
35 
36   // note: to avoid cache issues, don't allow writing if caching is enabled
37   private lazy val writable = !caching && Try(createOutputStream()).map(_.close()).isSuccess
38 
39   private lazy val ipcOpts = FormatVersion.options(FormatVersion.ArrowFormatVersion.get)
40 
41   private lazy val reader: SimpleFeatureArrowFileReader = {
42     initialized = true
43     if (caching) {
44       SimpleFeatureArrowFileReader.caching(createInputStream())
45     } else {
46       SimpleFeatureArrowFileReader.streaming(() => createInputStream())
47     }
48   }
49 
50   lazy val dictionaries: Map[String, ArrowDictionary] = reader.dictionaries
51 
52   override val metadata: GeoMesaMetadata[String] = GeoMesaMetadata.empty
53   override val stats: GeoMesaStats = new UnoptimizedRunnableStats(this)
54 
55   override def createFeatureSource(entry: ContentEntry): ContentFeatureSource = {
56     if (writable) {
57       new ArrowFeatureStore(entry, reader)
58     } else {
59       new ArrowFeatureSource(entry, reader)
60     }
61   }
62 
63   override def createTypeNames(): java.util.List[Name] = {
64     import scala.collection.JavaConverters._
65     Option(getSchema).map(s => (new NameImpl(namespaceURI, s.getTypeName)).asInstanceOf[Name]).toList.asJava
66   }
67 
68   override def createSchema(sft: SimpleFeatureType): Unit = {
69     if (!writable) {
70       throw new IllegalArgumentException("Can't write to the provided URL, or caching is enabled")
71     }
72     WithClose(createOutputStream(false)) { os =>
73       WithClose(SimpleFeatureArrowFileWriter(os, sft, Map.empty, SimpleFeatureEncoding.Max, ipcOpts, None)) { writer =>
74         // write an empty batch to write out the schema/metadata
75         writer.flush()
76       }
77     }
78   }
79 
80   override def dispose(): Unit = {
81     // avoid instantiating the lazy reader if it hasn't actually been used
82     if (initialized) {
83       CloseWithLogging(reader)
84     }
85   }
86 
87   // FileDataStore method
88 
89   override def getSchema: SimpleFeatureType = {
90     if (WithClose(createInputStream()) { _.available() < 1 }) { null } else {
91       try { reader.sft } catch {
92         case e: Exception => throw new IOException("Error reading schema", e)
93       }
94     }
95   }
96 
97   private def createInputStream(): InputStream = url.openStream()
98 
99   private [data] def createOutputStream(append: Boolean = true): OutputStream = {
100     Option(URLs.urlToFile(url)).map(new FileOutputStream(_, append)).getOrElse {
101       try {
102         val connection = url.openConnection()
103         connection.setDoOutput(true)
104         connection.getOutputStream
105       } catch {
106         case NonFatal(e) => throw new IOException(s"URL is not writable: $url", e)
107       }
108     }
109   }
110 
111   override def updateSchema(featureType: SimpleFeatureType): Unit = updateSchema(getSchema.getTypeName, featureType)
112 
113   override def getFeatureSource: SimpleFeatureSource = getFeatureSource(getSchema.getTypeName)
114 
115   override def getFeatureWriter(filter: Filter, transaction: Transaction): FeatureWriter[SimpleFeatureType, SimpleFeature] =
116     getFeatureWriter(getSchema.getTypeName, filter, transaction)
117 
118   override def getFeatureWriter(transaction: Transaction): FeatureWriter[SimpleFeatureType, SimpleFeature] =
119     getFeatureWriter(getSchema.getTypeName, transaction)
120 
121   override def getFeatureWriterAppend(transaction: Transaction): FeatureWriter[SimpleFeatureType, SimpleFeature] =
122     getFeatureWriterAppend(getSchema.getTypeName, transaction)
123 
124   override def getFeatureReader: FeatureReader[SimpleFeatureType, SimpleFeature] =
125     getFeatureReader(new Query(getSchema.getTypeName), Transaction.AUTO_COMMIT)
