52 class ArrowFeatureStore(entry: ContentEntry, reader: SimpleFeatureArrowFileReader)
53     extends ContentFeatureStore(entry, Query.ALL) {
54

59   override def getWriterInternal(query: Query, flags: Int): FeatureWriter[SimpleFeatureType, SimpleFeature] = {
60     require(flags != 0, "no write flags set")
61     require((flags | WRITER_ADD) == WRITER_ADD, "Only append supported")
62 
63     val sft = delegate.getSchema
64     val os = entry.getDataStore.asInstanceOf[ArrowDataStore].createOutputStream() // append = true
65 
66     val ipcOpts = FormatVersion.options(FormatVersion.ArrowFormatVersion.get)
67     val writer = SimpleFeatureArrowFileWriter(os, sft, Map.empty, SimpleFeatureEncoding.Max, ipcOpts, None)
68     val flushCount = ArrowProperties.BatchSize.get.toLong
69 
70     new FeatureWriter[SimpleFeatureType, SimpleFeature] {
71       private var count = 0L
72       private var feature: ScalaSimpleFeature = _
73 
74       override def getFeatureType: SimpleFeatureType = writer.sft
75 
76       override def hasNext: Boolean = false
77 
78       override def next(): SimpleFeature = {
79         feature = new ScalaSimpleFeature(writer.sft, featureIds.getAndIncrement().toString)
80         feature
81       }
82 
83       override def write(): Unit = {
84         writer.add(feature)
85         feature = null
86         count += 1
87         if (count % flushCount == 0) {
88           writer.flush()
89         }
90       }
91 
92       override def remove(): Unit = throw new NotImplementedError()
93 
94       override def close(): Unit = CloseWithLogging(writer)
95     }
96   }
97
