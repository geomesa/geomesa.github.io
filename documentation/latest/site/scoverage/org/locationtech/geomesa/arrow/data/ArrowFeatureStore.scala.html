60   override def getWriterInternal(query: Query, flags: Int): FeatureWriter[SimpleFeatureType, SimpleFeature] = {
61     require(flags != 0, "no write flags set")
62     require((flags | WRITER_ADD) == WRITER_ADD, "Only append supported")
63 
64     val sft = delegate.getSchema
65     val os = entry.getDataStore.asInstanceOf[ArrowDataStore].createOutputStream() // append = true
66 
67     val ipcOpts = FormatVersion.options(FormatVersion.ArrowFormatVersion.get)
68     val writer = SimpleFeatureArrowFileWriter(os, sft, Map.empty, SimpleFeatureEncoding.Max, ipcOpts, None)
69     val flushCount = ArrowProperties.BatchSize.get.toLong
70 
71     new FastSettableFeatureWriter {
72       private var count = 0L
73       private var feature: ScalaSimpleFeature = _
74 
75       override def getFeatureType: SimpleFeatureType = writer.sft
76 
77       override def hasNext: Boolean = false
78 
79       override def next(): FastSettableFeature = {
80         feature = new ScalaSimpleFeature(writer.sft, featureIds.getAndIncrement().toString)
81         feature
82       }
83 
84       override def write(): Unit = {
85         writer.add(feature)
86         feature = null
87         count += 1
88         if (count % flushCount == 0) {
89           writer.flush()
90         }
91       }
92 
93       override def remove(): Unit = throw new NotImplementedError()
94 
95       override def close(): Unit = CloseWithLogging(writer)
96     }
97   }
