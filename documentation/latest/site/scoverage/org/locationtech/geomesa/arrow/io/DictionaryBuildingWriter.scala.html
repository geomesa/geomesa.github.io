31 /**
32  * Writes an arrow file of simple features. Dictionaries will be built up as features are observed.
33  * Dictionaries are encoded based on `maxSize`. Values will not be correctly encoded if
34  * more than `maxSize` distinct values are encountered
35  *
36  * @param sft simple feature type
37  * @param dictionaries dictionary fields
38  * @param encoding vector encoding
39  * @param maxSize max size for dictionaries, will be used to pick the encoding for dictionary values
40  */
41 class DictionaryBuildingWriter(
42     sft: SimpleFeatureType,
43     dictionaries: Seq[String],
44     encoding: SimpleFeatureEncoding,
45     ipcOpts: IpcOption,
46     maxSize: Int = Short.MaxValue,
47     flattenStruct: Boolean = false
48   ) extends Closeable {
49 
50   import org.locationtech.geomesa.utils.geotools.RichAttributeDescriptors.RichAttributeDescriptor
51 
52   private val allocator = ArrowAllocator(s"dictionary-builder:${sft.getTypeName}")
53 
54   private val underlying = {
55     val struct = StructVector.empty(sft.getTypeName, allocator)
56     struct.allocateNew()
57     struct
58   }
59 
60   private val arrowWriter = underlying.getWriter
61 
62   private val idWriter = ArrowAttributeWriter.id(sft, encoding, underlying)
63   private val attributeWriters = DictionaryBuildingWriter.writers(sft, underlying, dictionaries, encoding, maxSize)
64 
65   private val root = {
66     val fields = Collections.singletonList[Field](underlying.getField)
67 
68     val potentialRoot = new VectorSchemaRoot(fields, Collections.singletonList[FieldVector](underlying), 0)
69     if(flattenStruct){
70       new VectorSchemaRoot(potentialRoot.getVector(sft.getTypeName))
71     } else {
72       potentialRoot
73     }
74   }
75 
76   private var index = 0
77 
78   def size: Int = index
79 
80   def add(feature: SimpleFeature): Unit = {
81     arrowWriter.setPosition(index)
82     arrowWriter.start()
83     idWriter.apply(index, feature)
84     var i = 0
85     while (i < attributeWriters.length) {
86       attributeWriters(i).apply(index, feature.getAttribute(i))
87       i += 1
88     }
89     arrowWriter.end()
90     index += 1
91   }
92 
93   /**
94     * Clear any simple features currently stored in the vector
95     */
96   def clear(): Unit = {
97     underlying.setValueCount(0)
98     index = 0
99     attributeWriters.foreach {
100       case w: ArrowDictionaryWriter => w.dictionary.asInstanceOf[ArrowDictionaryBuilder].clear()
101       case _ => // no-op
102     }
103   }
104 
105   /**
106     * Writes out as a stand-alone streaming arrow file, including current dictionary values
107     *
108     * @param os output stream to write to
109     */
110   def encode(os: OutputStream): Unit = {
111     arrowWriter.setValueCount(index)
112     idWriter.setValueCount(index)
113     attributeWriters.foreach(_.setValueCount(index))
114     root.setRowCount(index)
115 
116     val dictionaries = attributeWriters.collect { case w: ArrowDictionaryWriter =>
117       val name = s"dict-${w.dictionary.encoding.getId}"
118       val original = sft.getDescriptor(w.vector.getField.getName)
119       val descriptor = if (original.isList) {
120         SimpleFeatureTypes.createDescriptor(s"$name:${original.getListType().getSimpleName}")
121       } else {
122         SimpleFeatureTypes.renameDescriptor(original, name)
123       }
124       val writer = ArrowAttributeWriter(sft, descriptor, None, encoding, allocator)
125 
126       var i = 0
127       w.dictionary.foreach { value =>
128         writer.apply(i, value)
129         i += 1
130       }
131       writer.setValueCount(w.dictionary.length)
132 
133       new Dictionary(writer.vector, w.dictionary.encoding)
134     }
135 
136     val provider = new MapDictionaryProvider(dictionaries: _*)
137     WithClose(new ArrowStreamWriter(root, provider, Channels.newChannel(os), ipcOpts)) { writer =>
138       writer.start()
139       writer.writeBatch()
140     }
141 
142     dictionaries.foreach(d => CloseWithLogging(d.getVector))
143   }
144 
145   override def close(): Unit = CloseWithLogging.raise(Seq(arrowWriter, underlying, allocator))
146 }

152   /**
153     * Gets an attribute writer or a dictionary building writer, as appropriate
154     */
155   private def writers(
156       sft: SimpleFeatureType,
157       vector: StructVector,
158       dictionaries: Seq[String],
159       encoding: SimpleFeatureEncoding,
160       maxSize: Int): Array[ArrowAttributeWriter] = {
161     var dictionaryId = -1L
162     val writers = sft.getAttributeDescriptors.asScala.map { descriptor =>
163       val dictionary = dictionaries.collectFirst {
164         case name if name == descriptor.getLocalName =>
165           dictionaryId += 1
166           val bits = if (maxSize <= Byte.MaxValue) { 8 } else if (maxSize <= Short.MaxValue) { 16 } else { 32 }
167           new ArrowDictionaryBuilder(new DictionaryEncoding(dictionaryId, false, new ArrowType.Int(bits, true)))
168       }
169       ArrowAttributeWriter(sft, descriptor, dictionary, encoding, vector)
170     }
171     writers.toArray
172   }
