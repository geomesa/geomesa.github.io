33 /**
34   * Abstraction for using simple features in Arrow vectors
35   *
36   * @param sft simple feature type
37   * @param underlying underlying arrow vector
38   * @param dictionaries map of field names to dictionary values, used for dictionary encoding fields.
39   *                     All values must be provided up front.
40   * @param encoding options for encoding
41   */
42 class SimpleFeatureVector private [arrow] (
43     val sft: SimpleFeatureType,
44     val underlying: StructVector,
45     val dictionaries: Map[String, ArrowDictionary],
46     val encoding: SimpleFeatureEncoding,
47     allocator: Option[BufferAllocator]
48   ) extends Closeable {
49 
50   // note: writer creates the map child vectors based on the sft, and should be instantiated before the reader
51   val writer = new Writer()
52   val reader = new Reader()
53 
54   /**
55     * Clear any simple features currently stored in the vector
56     */
57   def clear(): Unit = underlying.setValueCount(0)
58 
59   override def close(): Unit = CloseWithLogging.raise(Seq(underlying) ++ allocator)

61   class Writer {
62 
63     private val idWriter = ArrowAttributeWriter.id(sft, encoding, underlying)
64     private val writers = ArrowAttributeWriter(sft, underlying, dictionaries, encoding).toArray
65 
66     /**
67      * Sets the feature at the given index
68      *
69      * @param index index to set
70      * @param feature feature to set
71      */
72     def set(index: Int, feature: SimpleFeature): Unit = {
73       underlying.getWriter.setPosition(index)
74       underlying.getWriter.start()
75       idWriter.apply(index, feature)
76       var i = 0
77       while (i < writers.length) {
78         writers(i).apply(index, feature.getAttribute(i))
79         i += 1
80       }
81       underlying.getWriter.end()
82     }
83 
84     /**
85      * Sets the value count, e.g. the number of features contained in this vector.
86      * Should only be called after all features have been written
87      *
88      * @param count count
89      */
90     def setValueCount(count: Int): Unit = {
91       underlying.getWriter.setValueCount(count)
92       writers.foreach(_.setValueCount(count))
93     }
94   }

97 
98     private val idReader = ArrowAttributeReader.id(sft, underlying, encoding)
99     private val readers = ArrowAttributeReader(sft, underlying, dictionaries, encoding).toArray

101     // feature that can be re-populated with calls to 'load'
102     val feature: ArrowSimpleFeature = new ArrowSimpleFeature(sft, idReader, readers, -1)

104     /**
105      * Gets the feature at a given index
106      *
107      * @param index index to get
108      * @return

112     /**
113      * Loads the re-usable `feature` with the given index
114      *
115      * @param index index to load
116      */
117     def load(index: Int): Unit = feature.index = index
118 
119     /**
120      * Gets the value count, e.g. the number of features contained in this vector
121      *
122      * @return
123      */
124     def getValueCount: Int = underlying.getValueCount
125   }

138

140     val Max: SimpleFeatureEncoding = SimpleFeatureEncoding(Some(Encoding.Max), Encoding.Max, Encoding.Max, flipAxisOrder = false)

142     def min(includeFids: Boolean, proxyFids: Boolean = false, flipAxisOrder: Boolean = false): SimpleFeatureEncoding = {
143       val fids = if (includeFids) { Some(if (proxyFids) { Encoding.Min } else { Encoding.Max }) } else { None }
144       SimpleFeatureEncoding(fids, Encoding.Min, Encoding.Min, flipAxisOrder)

153   /**
154     * Create a new simple feature vector
155     *
156     * @param sft simple feature type
157     * @param dictionaries map of field names to dictionary values, used for dictionary encoding fields.
158     *                     All values must be provided up front.
159     * @param encoding options for encoding
160     * @param capacity initial capacity for number of features able to be stored in vectors
161     * @return
162     */
163   def create(
164       sft: SimpleFeatureType,
165       dictionaries: Map[String, ArrowDictionary],
166       encoding: SimpleFeatureEncoding = SimpleFeatureEncoding.Min,
167       capacity: Int = DefaultCapacity): SimpleFeatureVector = {
168     val allocator = ArrowAllocator(s"simple-feature-vector:${sft.getTypeName}")
169     try {
170       val metadata = Collections.singletonMap(OptionsKey, SimpleFeatureTypes.encodeUserData(sft))
171       val fieldType = new FieldType(true, ArrowType.Struct.INSTANCE, null, metadata)
172       val underlying = new StructVector(sft.getTypeName, allocator, fieldType, null)
173       val vector = new SimpleFeatureVector(sft, underlying, dictionaries, encoding, Some(allocator))
174       // set capacity after all child vectors have been created by the writers, then allocate
175       underlying.setInitialCapacity(capacity)
176       underlying.allocateNew()
177       vector
178     } catch {
179       case NonFatal(e) => CloseQuietly(allocator).foreach(e.addSuppressed); throw e
180     }
181   }

206   /**
207     * Reads the feature type and feature encoding from an existing arrow vector
208     *
209     * @param vector vector
210     * @return
211     */
212   def getFeatureType(vector: StructVector): (SimpleFeatureType, SimpleFeatureEncoding) = {
213     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
214 
215     import scala.collection.JavaConverters._
216 
217     val attributes = ArrayBuffer.empty[String]
218     var fidEncoding: Option[Encoding] = None
219 
220     vector.getField.getChildren.asScala.foreach { field =>
221       if (field.getName == FeatureIdField) {
222         field.getType match {
223           case _: ArrowType.Int           => fidEncoding = Some(Encoding.Min) // proxy id encoded fids
224           case _: ArrowType.FixedSizeList => fidEncoding = Some(Encoding.Max) // uuid encoded fids
225           case _: ArrowType.Utf8          => fidEncoding = Some(Encoding.Max) // normal string fids
226           case _ => throw new IllegalArgumentException(s"Found feature ID vector field of unexpected type: $field")
227         }
228       } else {
229         attributes.append(field.getMetadata.get(DescriptorKey))
230       }
231     }
232     // add sft-level metadata
233     val options = Option(vector.getField.getMetadata.get(OptionsKey)).getOrElse("")
234 
235     val sft = SimpleFeatureTypes.createImmutableType(vector.getField.getName, attributes.mkString(",") + options)
236     val geomPrecision = {
237       val geomVector: Option[FieldVector] =
238         Option(sft.getGeomField).flatMap(d => Option(vector.getChild(d))).orElse(getNestedVector[Geometry](sft, vector))
239       val isDouble = geomVector.exists(v => GeometryFields.precisionFromField(v.getField) == FloatingPointPrecision.DOUBLE)
240       if (isDouble) { Encoding.Max } else { Encoding.Min }
241     }
242     val datePrecision = {
243       val dateVector: Option[FieldVector] =
244         sft.getDtgField.flatMap(d => Option(vector.getChild(d))).orElse(getNestedVector[Date](sft, vector))
245       val isLong = dateVector.exists(_.isInstanceOf[BigIntVector])
246       if (isLong) { Encoding.Max } else { Encoding.Min }
247     }
248     val encoding = SimpleFeatureEncoding(fidEncoding, geomPrecision, datePrecision, flipAxisOrder = false)
249 
250     (sft, encoding)
251   }
252 
253   def isGeometryVector(vector: FieldVector): Boolean = {
254     Option(vector.getField.getMetadata.get(DescriptorKey))
255         .map(SimpleFeatureTypes.createDescriptor)
256         .exists(d => classOf[Geometry].isAssignableFrom(d.getType.getBinding))
257   }

259   /**
260     * Checks nested vector types (lists and maps) for instances of the given type
261     *
262     * @param sft simple feature type
263     * @param vector simple feature vector
264     * @param ct class tag
265     *
266     * @return
267     */
268   private def getNestedVector[T](sft: SimpleFeatureType,
269                                  vector: StructVector)
270                                 (implicit ct: ClassTag[T]): Option[FieldVector] = {
271     import org.locationtech.geomesa.utils.geotools.RichAttributeDescriptors.RichAttributeDescriptor
272 
273     import scala.collection.JavaConverters._
274 
275     sft.getAttributeDescriptors.asScala.flatMap {
276       case d if d.isList && ct.runtimeClass.isAssignableFrom(d.getListType()) =>
277         Option(vector.getChild(d.getLocalName).asInstanceOf[ListVector]).map(_.getDataVector)
278       case d if d.isMap && ct.runtimeClass.isAssignableFrom(d.getMapTypes()._1) =>
279         Option(vector.getChild(d.getLocalName).asInstanceOf[StructVector]).map(_.getChildrenFromFields.get(0))
280       case d if d.isMap && ct.runtimeClass.isAssignableFrom(d.getMapTypes()._2) =>
281         Option(vector.getChild(d.getLocalName).asInstanceOf[StructVector]).map(_.getChildrenFromFields.get(1))
282       case _ => None
283     }.headOption
284   }
