70 object CachingSimpleFeatureArrowFileReader {
71 
72   import scala.collection.JavaConverters._
73 
74   /**
75    * Reads a single logical arrow 'file' from the stream, which may contain multiple record batches
76    */
77   private class CachingSingleFileReader(is: ReadableByteChannel, allocator: BufferAllocator) extends Closeable {
78 
79     import SimpleFeatureArrowFileReader.loadDictionaries
80 
81     private val reader = new ArrowStreamReader(is, allocator)
82 
83     private val opened = ArrayBuffer.empty[SimpleFeatureVector]
84 
85     val vectors: Stream[SimpleFeatureVector] = {
86       val hasMore = reader.loadNextBatch() // load dictionaries and the first batch
87       val root = reader.getVectorSchemaRoot
88       require(root.getFieldVectors.size() == 1 && root.getFieldVectors.get(0).isInstanceOf[StructVector], "Invalid file")
89       val underlying = root.getFieldVectors.get(0).asInstanceOf[StructVector]
90       val (_, encoding) = SimpleFeatureVector.getFeatureType(underlying)
91 
92       // load any dictionaries into memory
93       val dictionaries = loadDictionaries(underlying.getField.getChildren.asScala.toSeq, reader, encoding)
94 
95       // lazily evaluate batches as we need them
96       def createStream(current: SimpleFeatureVector): Stream[SimpleFeatureVector] = {
97         readIsolatedBatch(current, is, allocator) match {
98           case None       => current #:: Stream.empty
99           case Some(next) => opened.append(next); current #:: createStream(next)
100         }
101       }
102 
103       val head = SimpleFeatureVector.wrap(underlying, dictionaries)
104       opened.append(head)
105       if (hasMore) {
106         createStream(head)
107       } else {
108         head #:: Stream.empty
109       }
110     }
111 
112     def sft: SimpleFeatureType = vectors.head.sft
113     def dictionaries: Map[String, ArrowDictionary] = vectors.head.dictionaries
114     def metadata: java.util.Map[String, String] = reader.getVectorSchemaRoot.getSchema.getCustomMetadata
115 
116     // iterator of simple features read from the input stream
117     def features(nextBatch: VectorToIterator, skip: SkipIndicator): Iterator[ArrowSimpleFeature] with Closeable = {
118       new Iterator[ArrowSimpleFeature] with Closeable {
119         private var batch: Iterator[ArrowSimpleFeature] = Iterator.empty
120         private val batches = vectors.iterator
121 
122         @scala.annotation.tailrec
123         override def hasNext(): Boolean = {
124           if (batch.hasNext) {
125             true
126           } else if (batches.hasNext) {
127             if (skip.skip) {
128               // make sure we read the rest of the record batches so that our input stream is at the end of a 'file'
129               batches.foreach(_ => ())
130               false
131             } else {
132               batch = nextBatch(batches.next)
133               hasNext()
134             }
135           } else {
136             false
137           }
138         }
139 
140         override def next(): ArrowSimpleFeature = batch.next()
141 
142         override def close(): Unit = {}
143       }
144     }
145 
146     override def close(): Unit = {
147       reader.close()
148       opened.foreach(_.close())
149     }
150   }
151 
152   // read a batch into a new simple feature vector, so that the batch still is accessible after reading the next one
153   private def readIsolatedBatch(
154       original: SimpleFeatureVector,
155       is: ReadableByteChannel,
156       allocator: BufferAllocator): Option[SimpleFeatureVector] = {
157     WithClose(MessageSerializer.deserializeMessageBatch(new ReadChannel(is), allocator)) {
158       case null => None
159 
160       case b: ArrowRecordBatch =>
161         val fields = Seq(original.underlying.getField)
162         val vectors = fields.map(_.createVector(allocator))
163         val root = new VectorSchemaRoot(fields.asJava, vectors.asJava, 0)
164         new VectorLoader(root).load(b)
165         Some(SimpleFeatureVector.clone(original, vectors.head.asInstanceOf[StructVector]))
166 
167       case b => throw new IllegalArgumentException(s"Expected record batch but got $b")
168     }
169   }
