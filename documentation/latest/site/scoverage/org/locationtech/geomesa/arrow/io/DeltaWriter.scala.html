80   private val writers = sft.getAttributeDescriptors.asScala.map { descriptor =>
81     val name = descriptor.getLocalName
82     val isList = descriptor.isList
83     val bindings = ObjectType.selectType(descriptor)
84     val metadata = Map(SimpleFeatureVector.DescriptorKey -> SimpleFeatureTypes.encodeDescriptor(sft, descriptor))
85     if (dictionaryFields.contains(name)) {
86       // for list types, we encode the list items, not the whole list
87       val attribute = {
88         val desc = if (isList) { s"$name:List[Integer]" } else { s"$name:Integer" }
89         val encodedMetadata = Map(SimpleFeatureVector.DescriptorKey -> desc)
90         val encodedBindings = if (isList) { Seq(ObjectType.LIST, ObjectType.INT) } else { Seq(ObjectType.INT) }
91         ArrowAttributeWriter(name, encodedBindings, None, encodedMetadata, encoding, VectorFactory(vector))
92       }
93       // for list types, get the list item binding (which is the tail of the bindings)
94       val dictBindings = if (isList) { bindings.tail } else { bindings }
95       val ordering = AttributeOrdering(dictBindings)
96       val dict = ArrowAttributeWriter(name, dictBindings, None, metadata, encoding, VectorFactory(allocator))
97       dict.vector.setInitialCapacity(initialCapacity)
98       dict.vector.allocateNew()
99       if (isList) {
100         new DictionaryListFieldWriter(sft.indexOf(name), attribute, dict, ordering, ipcOpts)
101       } else {
102         new DictionaryFieldWriter(sft.indexOf(name), attribute, dict, ordering, ipcOpts)
103       }
104     } else {
105       val attribute = ArrowAttributeWriter(name, bindings, None, metadata, encoding, VectorFactory(vector))
106       new AttributeFieldWriter(sft.indexOf(name), attribute)
107     }
108   }

132   /**
133     * Writes out a record batch. Format is:
134     *
135     * 8 bytes long - threading key
136     * (foreach dictionaryField) -> {
137     *   4 byte int - length of dictionary batch
138     *   anyref vector batch with dictionary delta values
139     * }
140     * 4 byte int - length of record batch
141     * record batch (may be dictionary encodings)
142     *
143     * Note: will sort the feature array in place if sorting is defined
144     *
145     * @param features features to write
146     * @param count number of features to write, starting from array index 0
147     * @return serialized record batch
148     */
149   def encode(features: Array[SimpleFeature], count: Int): Array[Byte] = {
150 
151     result.reset()
152     result.write(ByteArrays.toBytes(threadingKey))
153 
154     ordering.foreach(java.util.Arrays.sort(features, 0, count, _))
155 
156     // write out the dictionaries
157 
158     dictionaryWriters.foreach { dictionary =>
159       var i = 0
160       while (i < count) {
161         dictionary.addDictionaryValue(features(i))
162         i += 1
163       }
164       val delta = dictionary.writeDictionaryDelta(result)
165       logger.trace(s"$threadingKey writing dictionary delta with $delta values")
166     }
167 
168     // set feature ids in the vector
169     if (encoding.fids.isDefined) {
170       var i = 0
171       while (i < count) {
172         idWriter.apply(i, features(i))
173         i += 1
174       }
175       idWriter.setValueCount(count)
176     }
177 
178     // set attributes in the vector
179     writers.foreach { writer =>
180       var i = 0
181       while (i < count) {
182         writer.write(i, features(i))
183         i += 1
184       }
185       writer.setValueCount(count)
186     }
187 
188     logger.trace(s"$threadingKey writing batch with $count values")
189 
190     // write out the vector batch
191     writer.writeBatch(count, result)
192 
193     result.toByteArray
194   }

201     CloseWithLogging(dictionaryWriters) // also closes dictionary vectors
202     CloseWithLogging(allocator)
203   }

291     val iter: CloseableIterator[Array[Byte]] = new CloseableIterator[Array[Byte]] {
292       private var writeHeader = true
293       private val toLoad = SimpleFeatureVector.create(sft, mergedDictionaries.dictionaries, encoding, batchSize)
294       private val result = SimpleFeatureVector.create(sft, mergedDictionaries.dictionaries, encoding, batchSize)
295       logger.trace(s"merge unsorted deltas - read schema ${result.underlying.getField}")
296       private val loader = new RecordBatchLoader(toLoad.underlying)
297       private val unloader = new RecordBatchUnloader(result, ipcOpts)

299       private val transfers: Seq[(String, (Int, Int, java.util.Map[Integer, Integer]) => Unit)] = {
300         toLoad.underlying.getChildrenFromFields.asScala.map { fromVector =>
301           val name = fromVector.getField.getName
302           val toVector = result.underlying.getChild(name)
303           val transfer: (Int, Int, java.util.Map[Integer, Integer]) => Unit =
304             if (mergedDictionaries.dictionaries.contains(name)) {
305               (fromVector, toVector) match {
306                 case (from: IntVector, to: IntVector) =>
307                   (fromIndex: Int, toIndex: Int, mapping: java.util.Map[Integer, Integer]) => {
308                     val n = from.getObject(fromIndex)
309                     if (n == null) {
310                       to.setNull(toIndex)
311                     } else {
312                       to.setSafe(toIndex, mapping.get(n))
313                     }
314                   }
315 
316                 case (from: ListVector, to: ListVector) =>
317                   val innerTo = to.getDataVector.asInstanceOf[IntVector]
318                   (fromIndex: Int, toIndex: Int, mapping: java.util.Map[Integer, Integer]) => {
319                     // note: should not be nulls as they are converted to empty list
320                     val list = from.getObject(fromIndex).asInstanceOf[java.util.List[Integer]]
321                     if (to.getLastSet >= toIndex) {
322                       to.setLastSet(toIndex - 1)
323                     }
324                     val start = to.startNewValue(toIndex)
325                     var offset = 0
326                     while (offset < list.size()) {
327                       // note: nulls get encoded in the dictionary
328                       innerTo.setSafe(start + offset, mapping.get(list.get(offset)))
329                       offset += 1
330                     }
331                     to.endValue(toIndex, offset)
332                   }
333 
334                 case _ =>
335                   throw new IllegalStateException(s"Encountered unexpected dictionary vector: $fromVector")
336               }
337             } else if (sft.indexOf(name) != -1 &&
338                 classOf[Geometry].isAssignableFrom(sft.getDescriptor(name).getType.getBinding)) {
339               // geometry vectors use FixedSizeList vectors, for which transfer pairs aren't implemented
340               val binding = sft.getDescriptor(name).getType.getBinding
341               val from = GeometryFields.wrap(fromVector, binding).asInstanceOf[GeometryVector[Geometry, FieldVector]]
342               val to = GeometryFields.wrap(toVector, binding).asInstanceOf[GeometryVector[Geometry, FieldVector]]
343               (fromIndex: Int, toIndex: Int, _: java.util.Map[Integer, Integer]) => {
344                 from.transfer(fromIndex, toIndex, to)
345               }
346             } else {
347               val pair = fromVector.makeTransferPair(toVector)
348               (fromIndex: Int, toIndex: Int, _: java.util.Map[Integer, Integer]) => {
349                 pair.copyValueSafe(fromIndex, toIndex)
350               }
351             }
352           (name, transfer)
353         }.toSeq
354       }

356       private val threadIterator = threadedBatches.iterator
357       private var threadIndex = -1
358       private var batches: Iterator[Array[Byte]] = Iterator.empty
359       private var mappings: Map[String, java.util.Map[Integer, Integer]] = _
360       private var count = 0 // records read in current batch
361 
362       override def hasNext: Boolean = count < toLoad.reader.getValueCount || loadNextBatch()
363 
364       override def next(): Array[Byte] = {
365         var total = 0
366         while (total < batchSize && hasNext) {
367           // read the rest of the current vector, up to the batch size
368           val toRead = math.min(batchSize - total, toLoad.reader.getValueCount - count)
369           transfers.foreach { case (name, transfer) =>
370             val mapping = mappings.get(name).orNull
371             logger.trace(s"dictionary mappings for $name: $mapping")
372             var i = 0
373             while (i < toRead) {
374               transfer(i + count, i + total, mapping)
375               i += 1
376             }
377           }
378           count += toRead
379           total += toRead
380         }
381 
382         if (writeHeader) {
383           // write the header in the first result, which includes the metadata and dictionaries
384           writeHeader = false
385           writeHeaderAndFirstBatch(result, mergedDictionaries.dictionaries, ipcOpts, sort, total)
386         } else {
387           unloader.unload(total)
388         }
389       }
390 
391       override def close(): Unit = CloseWithLogging.raise(Seq(toLoad, result, mergedDictionaries))

393       /**
394         * Read the next batch
395         *
396         * @return true if there was a batch to load, false if we've read all batches
397         */
398       @tailrec
399       private def loadNextBatch(): Boolean = {
400         if (batches.hasNext) {
401           val batch = batches.next
402           // skip the dictionary batches
403           var offset = 8 // initial threading key offset
404           dictionaryFields.foreach { _ =>
405             offset += ByteArrays.readInt(batch, offset) + 4
406           }
407           val messageLength = ByteArrays.readInt(batch, offset)
408           offset += 4 // skip over message length bytes
409           // load the record batch
410           loader.load(batch, offset, messageLength)
411           if (toLoad.reader.getValueCount > 0) {
412             count = 0 // reset count for this batch
413             true
414           } else {
415             loadNextBatch()
416           }
417         } else if (threadIterator.hasNext) {
418           threadIndex += 1
419           // set the mappings for this thread
420           mappings = mergedDictionaries.mappings.map { case (f, m) => (f, m(threadIndex)) }
421           batches = threadIterator.next.iterator
422           loadNextBatch()
423         } else {
424           false
425         }
426       }
427     }

432   /**
433     * Merge with sorting. Each batch is assumed to be already sorted
434     *
435     * @param sft simple feature type
436     * @param dictionaryFields dictionary fields
437     * @param encoding simple feature encoding
438     * @param mergedDictionaries merged dictionaries and batch mappings
439     * @param sortBy sort field
440     * @param reverse reverse sort or not
441     * @param batchSize record batch size
442     * @param threadedBatches record batches, grouped by threading key, internally sorted
443     * @return
444     */
445   private def reduceWithSort(
446       sft: SimpleFeatureType,
447       dictionaryFields: Seq[String],
448       encoding: SimpleFeatureEncoding,
449       ipcOpts: IpcOption,
450       mergedDictionaries: MergedDictionaries,
451       sortBy: String,
452       reverse: Boolean,
453       batchSize: Int,
454       threadedBatches: Array[Array[Array[Byte]]]): CloseableIterator[Array[Byte]] = {
455 
456     import org.locationtech.geomesa.utils.conversions.ScalaImplicits.RichArray
457 
458     val dictionaries = mergedDictionaries.dictionaries
459 
460     val result = SimpleFeatureVector.create(sft, dictionaries, encoding)
461     val unloader = new RecordBatchUnloader(result, ipcOpts)
462 
463     logger.trace(s"merging sorted deltas - read schema: ${result.underlying.getField}")
464 
465     // we do a merge sort of each batch
466     // queue sorted by current value in each batch
467     val queue = {
468       val ordering = if (reverse) { BatchMergerOrdering.reverse } else { BatchMergerOrdering }
469       new PriorityQueue[BatchMerger[Any]](ordering)
470     }
471 
472     // track our open vectors to close later
473     val cleanup = ArrayBuffer.empty[SimpleFeatureVector]
474     cleanup.sizeHint(threadedBatches.foldLeft(0)((sum, a) => sum + a.length))
475 
476     threadedBatches.foreachIndex { case (batches, batchIndex) =>
477       val mappings = mergedDictionaries.mappings.map { case (f, m) => (f, m(batchIndex)) }
478       logger.trace(s"loading ${batches.length} batch(es) from a single thread")
479 
480       batches.foreach { batch =>
481         val toLoad = SimpleFeatureVector.create(sft, dictionaries, encoding)
482         val loader = new RecordBatchLoader(toLoad.underlying)
483         cleanup += toLoad
484 
485         // skip the dictionary batches
486         var offset = 8
487         dictionaryFields.foreach { _ =>
488           offset += ByteArrays.readInt(batch, offset) + 4
489         }
490         val messageLength = ByteArrays.readInt(batch, offset)
491         offset += 4 // skip the length bytes
492         // load the record batch
493         loader.load(batch, offset, messageLength)
494         if (toLoad.reader.getValueCount > 0) {
495           val transfers: Seq[(Int, Int) => Unit] = toLoad.underlying.getChildrenFromFields.asScala.map { fromVector =>
496             val name = fromVector.getField.getName
497             val toVector = result.underlying.getChild(name)
498             if (dictionaries.contains(name)) {
499               val mapping = mappings(name)
500               (fromVector, toVector) match {
501                 case (from: IntVector, to: IntVector) =>
502                   (fromIndex: Int, toIndex: Int) => {
503                     val n = from.getObject(fromIndex)
504                     if (n == null) {
505                       to.setNull(toIndex)
506                     } else {
507                       to.setSafe(toIndex, mapping.get(n))
508                     }
509                   }
510 
511                 case (from: ListVector, to: ListVector) =>
512                   val innerTo = to.getDataVector.asInstanceOf[IntVector]
513                   (fromIndex: Int, toIndex: Int) => {
514                     // note: should not be nulls as they are converted to empty list
515                     val list = from.getObject(fromIndex).asInstanceOf[java.util.List[Integer]]
516                     if (to.getLastSet >= toIndex) {
517                       to.setLastSet(toIndex - 1)
518                     }
519                     val start = to.startNewValue(toIndex)
520                     var offset = 0
521                     while (offset < list.size()) {
522                       // note: nulls get encoded in the dictionary
523                       innerTo.setSafe(start + offset, mapping.get(list.get(offset)))
524                       offset += 1
525                     }
526                     to.endValue(toIndex, offset)
527                   }
528 
529                 case _ =>
530                   throw new IllegalStateException(s"Encountered unexpected dictionary vector: $fromVector")
531               }
532             } else if (sft.indexOf(name) != -1 &&
533                 classOf[Geometry].isAssignableFrom(sft.getDescriptor(name).getType.getBinding)) {
534               // geometry vectors use FixedSizeList vectors, for which transfer pairs aren't implemented
535               val binding = sft.getDescriptor(name).getType.getBinding
536               val from = GeometryFields.wrap(fromVector, binding).asInstanceOf[GeometryVector[Geometry, FieldVector]]
537               val to = GeometryFields.wrap(toVector, binding).asInstanceOf[GeometryVector[Geometry, FieldVector]]
538               (fromIndex: Int, toIndex: Int) => from.transfer(fromIndex, toIndex, to)
539             } else {
540               val transfer = fromVector.makeTransferPair(toVector)
541               (fromIndex: Int, toIndex: Int) => transfer.copyValueSafe(fromIndex, toIndex)
542             }
543           }.toSeq
544           val mapVector = toLoad.underlying
545           val dict = dictionaries.get(sortBy)
546           val descriptor = sft.getDescriptor(sortBy)
547           val merger = ArrowAttributeReader(descriptor, mapVector.getChild(sortBy), dict, encoding) match {
548             case r: ArrowDictionaryReader => new DictionaryBatchMerger(toLoad, transfers, r, mappings.get(sortBy).orNull)
549             case r: ArrowListDictionaryReader => new DictionaryListBatchMerger(toLoad, transfers, r, mappings.get(sortBy).orNull)
550             case r: ArrowDateReader => new DateBatchMerger(toLoad, transfers, r)
551             case r => new AttributeBatchMerger(toLoad, transfers, r, AttributeOrdering(descriptor))
552           }
553           queue.add(merger.asInstanceOf[BatchMerger[Any]])
554         }
555       }
556     }
557 
558     var writtenHeader = false
559 
560     // gets the next record batch to write - returns null if no further records
561     def nextBatch(): Array[Byte] = {
562       if (queue.isEmpty) { null } else {
563         result.clear()
564         var resultIndex = 0
565         // copy the next sorted value and then queue and sort the next element out of the batch we copied from
566         while ({
567           {
568             val next = queue.remove()
569             if (next.transfer(resultIndex)) {
570               queue.add(next)
571             }
572             result.underlying.setIndexDefined(resultIndex)
573             resultIndex += 1
574           }; !queue.isEmpty && resultIndex < batchSize
575         })()
576 
577         if (writtenHeader) {
578           unloader.unload(resultIndex)
579         } else {
580           // write the header in the first result, which includes the metadata and dictionaries
581           writtenHeader = true
582           writeHeaderAndFirstBatch(result, dictionaries, ipcOpts, Some(sortBy -> reverse), resultIndex)
583         }
584       }
585     }
