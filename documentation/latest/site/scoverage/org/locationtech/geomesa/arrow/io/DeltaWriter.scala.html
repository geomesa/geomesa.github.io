80   private val writers = sft.getAttributeDescriptors.asScala.map { descriptor =>
81     val name = descriptor.getLocalName
82     val isList = descriptor.isList
83     val bindings = ObjectType.selectType(descriptor)
84     val metadata = Map(SimpleFeatureVector.DescriptorKey -> SimpleFeatureTypes.encodeDescriptor(sft, descriptor))
85     if (dictionaryFields.contains(name)) {
86       // for list types, we encode the list items, not the whole list
87       val attribute = {
88         val desc = if (isList) { s"$name:List[Integer]" } else { s"$name:Integer" }
89         val encodedMetadata = Map(SimpleFeatureVector.DescriptorKey -> desc)
90         val encodedBindings = if (isList) { Seq(ObjectType.LIST, ObjectType.INT) } else { Seq(ObjectType.INT) }
91         ArrowAttributeWriter(name, encodedBindings, None, encodedMetadata, encoding, VectorFactory(vector))
92       }
93       // for list types, get the list item binding (which is the tail of the bindings)
94       val dictBindings = if (isList) { bindings.tail } else { bindings }
95       val ordering = AttributeOrdering(dictBindings)
96       val dict = ArrowAttributeWriter(name, dictBindings, None, metadata, encoding, VectorFactory(allocator))
97       dict.vector.setInitialCapacity(initialCapacity)
98       dict.vector.allocateNew()
99       if (isList) {
100         new DictionaryListFieldWriter(sft.indexOf(name), attribute, dict, ordering, ipcOpts)
101       } else {
102         new DictionaryFieldWriter(sft.indexOf(name), attribute, dict, ordering, ipcOpts)
103       }
104     } else {
105       val attribute = ArrowAttributeWriter(name, bindings, None, metadata, encoding, VectorFactory(vector))
106       new AttributeFieldWriter(sft.indexOf(name), attribute)
107     }
108   }

132   /**
133     * Writes out a record batch. Format is:
134     *
135     * 8 bytes long - threading key
136     * (foreach dictionaryField) -> {
137     *   4 byte int - length of dictionary batch
138     *   anyref vector batch with dictionary delta values
139     * }
140     * 4 byte int - length of record batch
141     * record batch (may be dictionary encodings)
142     *
143     * Note: will sort the feature array in place if sorting is defined
144     *
145     * @param features features to write
146     * @param count number of features to write, starting from array index 0
147     * @return serialized record batch
148     */
149   def encode(features: Array[SimpleFeature], count: Int): Array[Byte] = {
150 
151     result.reset()
152     result.write(ByteArrays.toBytes(threadingKey))
153 
154     ordering.foreach(java.util.Arrays.sort(features, 0, count, _))
155 
156     // write out the dictionaries
157 
158     dictionaryWriters.foreach { dictionary =>
159       var i = 0
160       while (i < count) {
161         dictionary.addDictionaryValue(features(i))
162         i += 1
163       }
164       val delta = dictionary.writeDictionaryDelta(result)
165       logger.trace(s"$threadingKey writing dictionary delta with $delta values")
166     }
167 
168     // set feature ids in the vector
169     if (encoding.fids.isDefined) {
170       var i = 0
171       while (i < count) {
172         idWriter.apply(i, features(i))
173         i += 1
174       }
175       idWriter.setValueCount(count)
176     }
177 
178     // set attributes in the vector
179     writers.foreach { writer =>
180       var i = 0
181       while (i < count) {
182         writer.write(i, features(i))
183         i += 1
184       }
185       writer.setValueCount(count)
186     }
187 
188     logger.trace(s"$threadingKey writing batch with $count values")
189 
190     // write out the vector batch
191     writer.writeBatch(count, result)
192 
193     result.toByteArray
194   }

200     CloseWithLogging(writer) // also closes `vector`
201     CloseWithLogging(dictionaryWriters) // also closes dictionary vectors
202     CloseWithLogging(allocator)
203   }

291     val iter: CloseableIterator[Array[Byte]] = new CloseableIterator[Array[Byte]] {
292       private var writeHeader = true
293       private val toLoad = SimpleFeatureVector.create(sft, mergedDictionaries.dictionaries, encoding, batchSize)
294       private val result = SimpleFeatureVector.create(sft, mergedDictionaries.dictionaries, encoding, batchSize)
295       logger.trace(s"merge unsorted deltas - read schema ${result.underlying.getField}")
296       private val loader = new RecordBatchLoader(toLoad.underlying)
297       private val unloader = new RecordBatchUnloader(result, ipcOpts)
298 
299       private val transfers: Seq[(String, (Int, Int, java.util.Map[Integer, Integer]) => Unit)] = {
300         toLoad.underlying.getChildrenFromFields.asScala.map { fromVector =>
301           val name = fromVector.getField.getName
302           val toVector = result.underlying.getChild(name)
303           val transfer: (Int, Int, java.util.Map[Integer, Integer]) => Unit =
304             if (mergedDictionaries.dictionaries.contains(name)) {
305               (fromVector, toVector) match {
306                 case (from: IntVector, to: IntVector) =>
307                   (fromIndex: Int, toIndex: Int, mapping: java.util.Map[Integer, Integer]) => {
308                     val n = from.getObject(fromIndex)
309                     if (n == null) {
310                       to.setNull(toIndex)
311                     } else {
312                       to.setSafe(toIndex, mapping.get(n))
313                     }
314                   }
315 
316                 case (from: ListVector, to: ListVector) =>
317                   val innerTo = to.getDataVector.asInstanceOf[IntVector]
318                   (fromIndex: Int, toIndex: Int, mapping: java.util.Map[Integer, Integer]) => {
319                     // note: should not be nulls as they are converted to empty list
320                     val list = from.getObject(fromIndex).asInstanceOf[java.util.List[Integer]]
321                     if (to.getLastSet >= toIndex) {
322                       to.setLastSet(toIndex - 1)
323                     }
324                     val start = to.startNewValue(toIndex)
325                     var offset = 0
326                     while (offset < list.size()) {
327                       // note: nulls get encoded in the dictionary
328                       innerTo.setSafe(start + offset, mapping.get(list.get(offset)))
329                       offset += 1
330                     }
331                     to.endValue(toIndex, offset)
332                   }
333 
334                 case _ =>
335                   throw new IllegalStateException(s"Encountered unexpected dictionary vector: $fromVector")
336               }
337             } else if (sft.indexOf(name) != -1 &&
338                 classOf[Geometry].isAssignableFrom(sft.getDescriptor(name).getType.getBinding)) {
339               // geometry vectors use FixedSizeList vectors, for which transfer pairs aren't implemented
340               val binding = sft.getDescriptor(name).getType.getBinding
341               val from = GeometryFields.wrap(fromVector, binding).asInstanceOf[GeometryVector[Geometry, FieldVector]]
342               val to = GeometryFields.wrap(toVector, binding).asInstanceOf[GeometryVector[Geometry, FieldVector]]
343               (fromIndex: Int, toIndex: Int, _: java.util.Map[Integer, Integer]) => {
344                 from.transfer(fromIndex, toIndex, to)
345               }
346             } else {
347               val pair = fromVector.makeTransferPair(toVector)
348               (fromIndex: Int, toIndex: Int, _: java.util.Map[Integer, Integer]) => {
349                 pair.copyValueSafe(fromIndex, toIndex)
350               }
351             }
352           (name, transfer)
353         }.toSeq
354       }

356       private val threadIterator = threadedBatches.iterator
357       private var threadIndex = -1
358       private var batches: Iterator[Array[Byte]] = Iterator.empty
359       private var mappings: Map[String, java.util.Map[Integer, Integer]] = _
360       private var count = 0 // records read in current batch
361 
362       override def hasNext: Boolean = count < toLoad.reader.getValueCount || loadNextBatch()
363 
364       override def next(): Array[Byte] = {
365         var total = 0
366         while (total < batchSize && hasNext) {
367           // read the rest of the current vector, up to the batch size
368           val toRead = math.min(batchSize - total, toLoad.reader.getValueCount - count)
369           transfers.foreach { case (name, transfer) =>
370             val mapping = mappings.get(name).orNull
371             logger.trace(s"dictionary mappings for $name: $mapping")
372             var i = 0
373             while (i < toRead) {
374               transfer(i + count, i + total, mapping)
375               i += 1
376             }
377           }
378           count += toRead
379           total += toRead
380         }
381         var i = 0
382         while (i < count) {
383           result.underlying.setIndexDefined(i)
384           i += 1
385         }
386 
387         if (writeHeader) {
388           // write the header in the first result, which includes the metadata and dictionaries
389           writeHeader = false
390           writeHeaderAndFirstBatch(result, mergedDictionaries.dictionaries, ipcOpts, sort, total)
391         } else {
392           unloader.unload(total)
393         }
394       }
395 
396       override def close(): Unit = CloseWithLogging.raise(Seq(toLoad, result, mergedDictionaries))
397 
398       /**
399         * Read the next batch
400         *
401         * @return true if there was a batch to load, false if we've read all batches
402         */
403       @tailrec
404       private def loadNextBatch(): Boolean = {
405         if (batches.hasNext) {
406           val batch = batches.next
407           // skip the dictionary batches
408           var offset = 8 // initial threading key offset
409           dictionaryFields.foreach { _ =>
410             offset += ByteArrays.readInt(batch, offset) + 4
411           }
412           val messageLength = ByteArrays.readInt(batch, offset)
413           offset += 4 // skip over message length bytes
414           // load the record batch
415           loader.load(batch, offset, messageLength)
416           if (toLoad.reader.getValueCount > 0) {
417             count = 0 // reset count for this batch
418             true
419           } else {
420             loadNextBatch()
421           }
422         } else if (threadIterator.hasNext) {
423           threadIndex += 1
424           // set the mappings for this thread
425           mappings = mergedDictionaries.mappings.map { case (f, m) => (f, m(threadIndex)) }
426           batches = threadIterator.next.iterator
427           loadNextBatch()
428         } else {
429           false
430         }
431       }
432     }

437   /**
438     * Merge with sorting. Each batch is assumed to be already sorted
439     *
440     * @param sft simple feature type
441     * @param dictionaryFields dictionary fields
442     * @param encoding simple feature encoding
443     * @param mergedDictionaries merged dictionaries and batch mappings
444     * @param sortBy sort field
445     * @param reverse reverse sort or not
446     * @param batchSize record batch size
447     * @param threadedBatches record batches, grouped by threading key, internally sorted
448     * @return
449     */
450   private def reduceWithSort(
451       sft: SimpleFeatureType,
452       dictionaryFields: Seq[String],
453       encoding: SimpleFeatureEncoding,
454       ipcOpts: IpcOption,
455       mergedDictionaries: MergedDictionaries,
456       sortBy: String,
457       reverse: Boolean,
458       batchSize: Int,
459       threadedBatches: Array[Array[Array[Byte]]]): CloseableIterator[Array[Byte]] = {
460 
461     import org.locationtech.geomesa.utils.conversions.ScalaImplicits.RichArray
462 
463     val dictionaries = mergedDictionaries.dictionaries
464 
465     val result = SimpleFeatureVector.create(sft, dictionaries, encoding)
466     val unloader = new RecordBatchUnloader(result, ipcOpts)
467 
468     logger.trace(s"merging sorted deltas - read schema: ${result.underlying.getField}")
469 
470     // we do a merge sort of each batch
471     // queue sorted by current value in each batch
472     val queue = {
473       val ordering = if (reverse) { BatchMergerOrdering.reverse } else { BatchMergerOrdering }
474       new PriorityQueue[BatchMerger[Any]](ordering)
475     }
476 
477     // track our open vectors to close later
478     val cleanup = ArrayBuffer.empty[SimpleFeatureVector]
479     cleanup.sizeHint(threadedBatches.foldLeft(0)((sum, a) => sum + a.length))
480 
481     threadedBatches.foreachIndex { case (batches, batchIndex) =>
482       val mappings = mergedDictionaries.mappings.map { case (f, m) => (f, m(batchIndex)) }
483       logger.trace(s"loading ${batches.length} batch(es) from a single thread")
484 
485       batches.foreach { batch =>
486         val toLoad = SimpleFeatureVector.create(sft, dictionaries, encoding)
487         val loader = new RecordBatchLoader(toLoad.underlying)
488         cleanup += toLoad
489 
490         // skip the dictionary batches
491         var offset = 8
492         dictionaryFields.foreach { _ =>
493           offset += ByteArrays.readInt(batch, offset) + 4
494         }
495         val messageLength = ByteArrays.readInt(batch, offset)
496         offset += 4 // skip the length bytes
497         // load the record batch
498         loader.load(batch, offset, messageLength)
499         if (toLoad.reader.getValueCount > 0) {
500           val transfers: Seq[(Int, Int) => Unit] = toLoad.underlying.getChildrenFromFields.asScala.map { fromVector =>
501             val name = fromVector.getField.getName
502             val toVector = result.underlying.getChild(name)
503             if (dictionaries.contains(name)) {
504               val mapping = mappings(name)
505               (fromVector, toVector) match {
506                 case (from: IntVector, to: IntVector) =>
507                   (fromIndex: Int, toIndex: Int) => {
508                     val n = from.getObject(fromIndex)
509                     if (n == null) {
510                       to.setNull(toIndex)
511                     } else {
512                       to.setSafe(toIndex, mapping.get(n))
513                     }
514                   }
515 
516                 case (from: ListVector, to: ListVector) =>
517                   val innerTo = to.getDataVector.asInstanceOf[IntVector]
518                   (fromIndex: Int, toIndex: Int) => {
519                     // note: should not be nulls as they are converted to empty list
520                     val list = from.getObject(fromIndex).asInstanceOf[java.util.List[Integer]]
521                     if (to.getLastSet >= toIndex) {
522                       to.setLastSet(toIndex - 1)
523                     }
524                     val start = to.startNewValue(toIndex)
525                     var offset = 0
526                     while (offset < list.size()) {
527                       // note: nulls get encoded in the dictionary
528                       innerTo.setSafe(start + offset, mapping.get(list.get(offset)))
529                       offset += 1
530                     }
531                     to.endValue(toIndex, offset)
532                   }
533 
534                 case _ =>
535                   throw new IllegalStateException(s"Encountered unexpected dictionary vector: $fromVector")
536               }
537             } else if (sft.indexOf(name) != -1 &&
538                 classOf[Geometry].isAssignableFrom(sft.getDescriptor(name).getType.getBinding)) {
539               // geometry vectors use FixedSizeList vectors, for which transfer pairs aren't implemented
540               val binding = sft.getDescriptor(name).getType.getBinding
541               val from = GeometryFields.wrap(fromVector, binding).asInstanceOf[GeometryVector[Geometry, FieldVector]]
542               val to = GeometryFields.wrap(toVector, binding).asInstanceOf[GeometryVector[Geometry, FieldVector]]
543               (fromIndex: Int, toIndex: Int) => from.transfer(fromIndex, toIndex, to)
544             } else {
545               val transfer = fromVector.makeTransferPair(toVector)
546               (fromIndex: Int, toIndex: Int) => transfer.copyValueSafe(fromIndex, toIndex)
547             }
548           }.toSeq
549           val mapVector = toLoad.underlying
550           val dict = dictionaries.get(sortBy)
551           val descriptor = sft.getDescriptor(sortBy)
552           val merger = ArrowAttributeReader(descriptor, mapVector.getChild(sortBy), dict, encoding) match {
553             case r: ArrowDictionaryReader => new DictionaryBatchMerger(toLoad, transfers, r, mappings.get(sortBy).orNull)
554             case r: ArrowListDictionaryReader => new DictionaryListBatchMerger(toLoad, transfers, r, mappings.get(sortBy).orNull)
555             case r: ArrowDateReader => new DateBatchMerger(toLoad, transfers, r)
556             case r => new AttributeBatchMerger(toLoad, transfers, r, AttributeOrdering(descriptor))
557           }
558           queue.add(merger.asInstanceOf[BatchMerger[Any]])
559         }
560       }
561     }
562 
563     var writtenHeader = false
564 
565     // gets the next record batch to write - returns null if no further records
566     def nextBatch(): Array[Byte] = {
567       if (queue.isEmpty) { null } else {
568         result.clear()
569         var resultIndex = 0
570         // copy the next sorted value and then queue and sort the next element out of the batch we copied from
571         while ({
572           {
573             val next = queue.remove()
574             if (next.transfer(resultIndex)) {
575               queue.add(next)
576             }
577             result.underlying.setIndexDefined(resultIndex)
578             resultIndex += 1
579           }; !queue.isEmpty && resultIndex < batchSize
580         })()
581 
582         if (writtenHeader) {
583           unloader.unload(resultIndex)
584         } else {
585           // write the header in the first result, which includes the metadata and dictionaries
586           writtenHeader = true
587           writeHeaderAndFirstBatch(result, dictionaries, ipcOpts, Some(sortBy -> reverse), resultIndex)
588         }
589       }
590     }

593

596       override def hasNext: Boolean = {
597         if (batch == null) {

599         }
600         batch != null
601       }
602 
603       override def next(): Array[Byte] = {
604         val res = batch
605         batch = null
606         res
607       }
608 
609       override def close(): Unit = {
610         CloseWithLogging(result)
611         CloseWithLogging(cleanup)
612         CloseWithLogging(mergedDictionaries)
613       }
614     }

617   }
