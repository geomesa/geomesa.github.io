44   def rewrite(filter: Filter, sft: SimpleFeatureType, dictionaries: Map[String, ArrowDictionary]): Filter = {
45     val bound = FastFilterFactory.optimize(sft, filter)
46     FastFilterFactory.sfts.set(sft)
47     try {
48       rewriteFilter(bound, sft, dictionaries)
49     } finally {
50       FastFilterFactory.sfts.remove()
51     }
52   }

54   private def rewriteFilter(filter: Filter, sft: SimpleFeatureType, dictionaries: Map[String, ArrowDictionary]): Filter = {
55     try {
56       filter match {
57         case f: BBOX              => rewriteBBox(f, sft)
58         case f: During            => rewriteDuring(f, sft)
59         case f: PropertyIsBetween => rewriteBetween(f, sft)
60         case f: PropertyIsEqualTo => rewritePropertyIsEqualTo(f, sft, dictionaries)
61         case a: And               => ff.and(a.getChildren.asScala.map(rewriteFilter(_, sft, dictionaries)).asJava)
62         case o: Or                => ff.or(o.getChildren.asScala.map(rewriteFilter(_, sft, dictionaries)).asJava)
63         case f: Not               => ff.not(rewriteFilter(f.getFilter, sft, dictionaries))
64         case _                    => filter
65       }
66     } catch {
67       case NonFatal(e) => logger.warn(s"Error re-writing filter $filter", e); filter
68     }
69   }

71   private def rewriteBBox(filter: BBOX, sft: SimpleFeatureType): Filter = {
72     if (sft.isPoints || sft.isLines) {
73       val props = checkOrderUnsafe(filter.getExpression1, filter.getExpression2)
74       val bbox = FastConverter.evaluate(props.literal, classOf[Polygon]).getEnvelopeInternal
75       val attrIndex = sft.indexOf(props.name)
76       if (sft.isPoints) {
77         ArrowPointBBox(attrIndex, bbox.getMinX, bbox.getMinY, bbox.getMaxX, bbox.getMaxY)
78       } else {
79         ArrowLineStringBBox(attrIndex, bbox.getMinX, bbox.getMinY, bbox.getMaxX, bbox.getMaxY)
80       }
81     } else {
82       filter
83     }
84   }

86   private def rewriteDuring(filter: During, sft: SimpleFeatureType): Filter = {
87     val props = checkOrderUnsafe(filter.getExpression1, filter.getExpression2)
88     val attrIndex = sft.indexOf(props.name)
89     val period = FastConverter.evaluate(props.literal, classOf[Period])
90     val lower = period.getBeginning.getPosition.getDate.getTime
91     val upper = period.getEnding.getPosition.getDate.getTime
92     ArrowDuring(attrIndex, lower, upper)
93   }

95   private def rewriteBetween(filter: PropertyIsBetween, sft: SimpleFeatureType): Filter = {
96     val attribute = filter.getExpression.asInstanceOf[PropertyName].getPropertyName
97     val attrIndex = sft.indexOf(attribute)
98     if (sft.getDescriptor(attrIndex).getType.getBinding != classOf[Date]) { filter } else {
99       val lower = FastConverter.evaluate(filter.getLowerBoundary, classOf[Date]).getTime
100       val upper = FastConverter.evaluate(filter.getUpperBoundary, classOf[Date]).getTime
101       ArrowBetweenDate(attrIndex, lower, upper)
102     }
103   }

105   private def rewritePropertyIsEqualTo(filter: PropertyIsEqualTo,
106                                        sft: SimpleFeatureType,
107                                        dictionaries: Map[String, ArrowDictionary]): Filter = {
108     val props = checkOrderUnsafe(filter.getExpression1, filter.getExpression2)
109     dictionaries.get(props.name) match {
110       case None => filter
111       case Some(dictionary) =>
112         val attrIndex = sft.indexOf(props.name)
113         val numericValue = dictionary.index(props.literal.evaluate(null))
114         if (sft.getDescriptor(attrIndex).isList) {
115           ArrowListDictionaryEquals(attrIndex, numericValue)
116         } else {
117           ArrowDictionaryEquals(attrIndex, numericValue)
118         }
119     }
120   }

122   case class ArrowPointBBox(i: Int, xmin: Double, ymin: Double, xmax: Double, ymax: Double) extends Filter {
123     override def accept(visitor: FilterVisitor, extraData: AnyRef): AnyRef = extraData
124     override def evaluate(o: AnyRef): Boolean = {
125       val arrow = o.asInstanceOf[ArrowSimpleFeature]
126       val reader = arrow.getReader(i).asInstanceOf[ArrowPointReader]
127       val index = arrow.getIndex
128       val y = reader.readPointY(index)
129       if (y < ymin || y > ymax) { false } else {
130         val x = reader.readPointX(index)
131         x >= xmin && x <= xmax
132       }
133     }
134   }

136   case class ArrowLineStringBBox(i: Int, xmin: Double, ymin: Double, xmax: Double, ymax: Double) extends Filter {
137     private val bboxEnvelope = new ReferencedEnvelope(xmin, xmax, ymin, ymax, CRS_EPSG_4326)
138     private val bbox = GeometryVector.factory.toGeometry(bboxEnvelope)
139 
140     override def accept(visitor: FilterVisitor, extraData: AnyRef): AnyRef = extraData
141     override def evaluate(o: AnyRef): Boolean = {
142       val arrow = o.asInstanceOf[ArrowSimpleFeature]
143       val reader = arrow.getReader(i).asInstanceOf[ArrowLineStringReader]
144       val (start, end) = reader.readOffsets(arrow.getIndex)
145       var offset = start
146       val points = Array.ofDim[Coordinate](2)
147       while (offset < end) {
148         val y = reader.readPointY(offset)
149         if (y >= ymin && y <= ymax) {
150           val x = reader.readPointX(offset)
151           if (x >= xmin && x <= xmax) {
152             // we have a point in the bbox, short-circuit and return
153             return true
154           }
155           // check for intersection even if the points aren't contained in the bbox
156           points(1) = points(0)
157           points(0) = new Coordinate(x, y)
158           if (offset > start) {
159             val line = GeometryVector.factory.createLineString(points)
160             if (line.getEnvelopeInternal.intersects(bboxEnvelope) && line.intersects(bbox)) {
161               // found an intersection, short-circuit and return
162               return true
163             }
164           }
165         }
166         offset += 1
167       }
168       false
169     }
170   }
