27 /**
28   * For reading simple features from an arrow file written by SimpleFeatureArrowFileWriter.
29   *
30   * Expects arrow streaming format (no footer). Can handle multiple 'files' in a single input stream
31   */
32 trait SimpleFeatureArrowFileReader extends Closeable {
33 
34   /**
35     * The simple feature type for the file. Note: this may change as features are read,
36     * if there are multiple logical 'files' in the input stream. By convention, we keep
37     * a single file with a single sft, but that is not enforced.
38     *
39     * @return current simple feature type
40     */

43   /**
44     * Dictionaries from the file. Note: this may change as features are read, if there are
45     * multiple logical 'files' in the input stream. This method is exposed for completeness,
46     * but generally would not be needed since dictionary values are automatically decoded
47     * into the returned simple features.
48     *
49     * @return current dictionaries, keyed by attribute
50     */
51   def dictionaries: Map[String, ArrowDictionary]
52 
53   def vectors: Seq[SimpleFeatureVector]
54 
55   /**
56     * Reads features from the underlying arrow file
57     *
58     * @param filter filter to apply
59     * @return
60     */

90   /**
91     *
92     * @param fields dictionary encoded fields
93     * @param provider dictionary provider
94     * @return
95     */
96   private [io] def loadDictionaries(
97       fields: Seq[Field],
98       provider: DictionaryProvider,
99       precision: SimpleFeatureEncoding): Map[String, ArrowDictionary] = {
100     fields.flatMap { field =>
101       // check top-level dictionaries plus nested (i.e. for list-type attributes)
102       val encodings = Seq(field.getDictionary) ++ field.getChildren.asScala.map(_.getDictionary)
103       encodings.collect { case encoding if encoding != null =>
104         val descriptor = SimpleFeatureTypes.createDescriptor(field.getMetadata.get(DescriptorKey))
105         val bindings = {
106           val main = ObjectType.selectType(descriptor)
107           // for list types, get the list item binding (which is the tail of the bindings)
108           if (descriptor.isList) { main.tail } else { main }
109         }
110         val vector = provider.lookup(encoding.getId).getVector
111         field.getName -> ArrowDictionary.create(encoding, vector, bindings, precision)
112       }
113     }.toMap
114   }
115 
116   /**
117     * Reads features from simple feature vectors based on a filter
118     *
119     * @param sft simple feature type
120     * @param filter filter
121     * @param skip indicator that we should skip any further batches
122     * @param sort sort for the file being read, if any
123     * @param dictionaries dictionaries
124     * @return
125     */
126   private [io] def features(sft: SimpleFeatureType,
127                             filter: Filter,
128                             skip: SkipIndicator,
129                             sort: Option[(String, Boolean)],
130                             dictionaries: Map[String, ArrowDictionary]): VectorToIterator = {
131     val optimized = ArrowFilterOptimizer.rewrite(filter, sft, dictionaries)
132     sort match {
133       case None => features(_, optimized)
134       case Some((field, reverse)) =>
135         val i = sft.indexOf(field)
136         val binding = sft.getDescriptor(i).getType.getBinding
137         val bounds = FilterHelper.extractAttributeBounds(filter, field, binding).values
138         if (bounds.isEmpty) {
139           features(_, optimized)
140         } else {
141           sortedFeatures(_, optimized, skip, bounds.asInstanceOf[Seq[Bounds[Comparable[Any]]]], i, reverse)
142         }
143     }
144   }
145 
146   /**
147     * Reads features from a simple feature vector
148     *
149     * @param vector simple feature vector
150     * @param filter filter
151     * @return
152     */
153   private def features(vector: SimpleFeatureVector, filter: Filter): Iterator[ArrowSimpleFeature] = {
154     val total = vector.reader.getValueCount
155     if (total == 0) { Iterator.empty } else {
156       // re-use the same feature object
157       val feature = vector.reader.feature
158       val all = Iterator.range(0, total).map { i => vector.reader.load(i); feature }
159       if (filter == Filter.INCLUDE) { all } else {
160         all.filter(filter.evaluate)
161       }
162     }
163   }
164 
165   /**
166     * Reads features from a simple feature vector. The underlying features are assumed to be sorted
167     *
168     * @param vector simple feature vector
169     * @param filter filter
170     * @param skip will be toggled if no further vectors need to be queried due to sort order and filter bounds
171     * @param filterBounds bounds for the sort field, extracted from the filter
172     * @param sortField field that the features are sorted by
173     * @param reverse if the sort order is reversed or not
174     * @return
175     */
176   private def sortedFeatures(vector: SimpleFeatureVector,
177                              filter: Filter,
178                              skip: SkipIndicator,
179                              filterBounds: Seq[Bounds[Comparable[Any]]],
180                              sortField: Int,
181                              reverse: Boolean): Iterator[ArrowSimpleFeature] = {
182     val total = vector.reader.getValueCount
183 
184     if (total == 0 || skip.skip) { Iterator.empty } else {
185       // re-use the same feature object
186       val feature = vector.reader.feature
187 
188       // bounds for the current batch
189       val currentBatchBounds = {
190         vector.reader.load(0)
191         val lo = Bound(Option(feature.getAttribute(sortField).asInstanceOf[Comparable[Any]]), inclusive = true)
192         vector.reader.load(total - 1)
193         val hi = Bound(Option(feature.getAttribute(sortField).asInstanceOf[Comparable[Any]]), inclusive = true)
194         if (reverse) { Bounds(hi, lo) } else { Bounds(lo, hi) }
195       }
196 
197       if (filterBounds.exists(Bounds.intersection(_, currentBatchBounds).isDefined)) {
198         // we have a match in this batch
199         val all = Iterator.range(0, vector.reader.getValueCount).map { i => vector.reader.load(i); feature }
200         all.filter(filter.evaluate)
201       } else {
202         // nothing from this batch matches, check to see if any further batches could match
203         val hasMore = if (reverse) {
204           filterBounds.exists(fb => Bounds.smallerLowerBound(fb.lower, currentBatchBounds.lower).eq(fb.lower))
205         } else {
206           filterBounds.exists(fb => Bounds.largerUpperBound(fb.upper, currentBatchBounds.upper).eq(fb.upper))
207         }
208         // toggle the skip indicator if there are no further batches that could match
209         skip.skip = !hasMore
210         Iterator.empty
211       }
212     }
213   }
214 
215   // holder for a skip indicator - this will be toggled if we ever determine there can be no more results
216   private [io] class SkipIndicator(var skip: Boolean = false)
