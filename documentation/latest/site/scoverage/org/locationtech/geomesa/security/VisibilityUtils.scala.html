16 object VisibilityUtils {
17 
18   type IsVisible = SimpleFeature => Boolean
19 
20   /**
21    * Return a local function that will check features for visibility, based on the user's current authorizations.
22    * Not thread-safe or re-usable in a subsequent request.
23    *
24    * @param provider auth provider
25    * @return
26    */
27   def visible(provider: AuthorizationsProvider): IsVisible = new AuthVisibilityCheck(provider.getAuthorizations)
28 
29   /**
30    * Parses any visibilities in the feature and compares with the user's authorizations
31    *
32    * @param auths authorizations for the current user
33    */
34   private class AuthVisibilityCheck(auths: java.util.List[String]) extends (SimpleFeature => Boolean) {
35 
36     private val access = AccessEvaluator.of(Authorizations.of(auths))
37     private val cache = scala.collection.mutable.Map.empty[String, Boolean]
38 
39     /**
40      * Checks auths against the feature's visibility
41      *
42      * @param f feature
43      * @return true if feature is visible to the current user, otherwise false
44      */
45     override def apply(f: SimpleFeature): Boolean = {
46       val vis = SecurityUtils.getVisibility(f)
47       vis == null || cache.getOrElseUpdate(vis, try { access.canAccess(vis) } catch { case NonFatal(_) => false })
48     }
49   }
