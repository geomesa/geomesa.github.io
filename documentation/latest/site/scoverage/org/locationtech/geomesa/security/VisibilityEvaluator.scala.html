20   * Evaluates visibilities against authorizations. Abstracted from Accumulo visibility code

22 @deprecated("Replaced with org.apache.accumulo:accumulo-access")

27   private val cache = new ConcurrentHashMap[String, VisibilityExpression]

29   // copied from org.apache.accumulo.core.security.Authorizations
30   private val validAuthChars = {
31     val chars = Array.fill[Boolean](256)(false)
32     Range.inclusive('a', 'z').foreach(chars(_) = true)
33     Range.inclusive('A', 'Z').foreach(chars(_) = true)
34     Range.inclusive('0', '9').foreach(chars(_) = true)
35     Seq('_', '-', ':', '.', '/').foreach(chars(_) = true)
36     chars
37   }
38 
39   /**
40     * Parses a visibility from a string. Results are cached for repeated calls
41     *
42     * Per standard operator precedence, the binary `&` has higher precedence than `|`, i.e.
43     * `user|admin&test` is the same as `user|(admin&test)` and `user&admin|test` is the same as `(user&admin)|test`
44     *
45     * @param visibility visibility string, e.g. 'admin|user'
46     * @param report provide detailed reporting on errors, or not
47     * @throws org.parboiled.errors.ParsingException if visibility is not valid
48     * @return parsed visibility expression
49     */
50   @throws(classOf[ParsingException])
51   def parse(visibility: String, report: Boolean = false): VisibilityExpression = {
52     if (visibility == null || visibility.isEmpty) { VisibilityNone } else {
53       var parsed = cache.get(visibility)
54       if (parsed == null) {
55         val runner = if (report) { ReportingParseRunner(Parser.visibility) } else { BasicParseRunner(Parser.visibility) }
56         val parsing = runner.run(visibility)
57         parsed = parsing.result.getOrElse(throw new ParsingException(ErrorUtils.printParseErrors(parsing)))
58         cache.put(visibility, parsed)
59       }
60       parsed
61     }
62   }
63 
64   /**
65     * Parsed visibility that can be evaluated
66     */
67   @deprecated("Replaced with org.apache.accumulo:accumulo-access")
68   sealed trait VisibilityExpression {
69 
70     /**
71       * Checks if the data tagged with this visibility can be seen or not
72       *
73       * @param authorizations authorizations of the user attempting to access the tagged data
74       * @return true if can see, otherwise false
75       */
76     def evaluate(authorizations: Seq[Array[Byte]]): Boolean
77 
78     /**
79       * Converts back to a string. The result can be converted back to a Visibility Expression again by
80       * calling `VisbilityEvaluator.parse`
81       *
82       * @return the expression as a string
83       */
84     def expression: String
85   }
86 
87   object VisibilityExpression {
88     def apply(visibility: String): VisibilityExpression = parse(visibility)
89     def unapply(visibility: VisibilityExpression): Option[String] = Some(visibility.expression)
90   }
91 
92   /**
93     * No visibility restrictions, can be seen by anyone
94     */
95   case object VisibilityNone extends VisibilityExpression {
96     override def evaluate(authorizations: Seq[Array[Byte]]): Boolean = true
97     override def expression: String = ""
98   }
99 
100   /**
101     * A specific visibility tag, which can only be seen by the equivalent authorization
102     *
103     * @param value visibility tag
104     */
105   case class VisibilityValue(value: Array[Byte]) extends VisibilityExpression {
106 
107     override def evaluate(authorizations: Seq[Array[Byte]]): Boolean =
108       authorizations.exists(java.util.Arrays.equals(value, _))
109 
110     override def expression: String = {
111       val string = new String(value, StandardCharsets.UTF_8)
112       if (value.forall(isValidAuthChar)) { string } else { '"' + StringEscapeUtils.escapeJava(string) + '"' }
113     }
114 
115     override def equals(o: Any): Boolean = o match {
116       case VisibilityValue(v) => java.util.Arrays.equals(value, v)
117       case _ => false
118     }
119 
120     override def hashCode(): Int = java.util.Arrays.hashCode(value)
121   }
122 
123   /**
124     * Boolean AND of visibilities
125     *
126     * @param expressions visibilities
127     */
128   case class VisibilityAnd(expressions: Seq[VisibilityExpression]) extends VisibilityExpression {
129     override def evaluate(authorizations: Seq[Array[Byte]]): Boolean = expressions.forall(_.evaluate(authorizations))
130     override def expression: String = {
131       val clauses = expressions.map {
132         case e: VisibilityOr => s"(${e.expression})"
133         case e => e.expression
134       }
135       clauses.mkString("&")
136     }
137   }
138 
139   /**
140     * Boolean OR of visibilities
141     *
142     * @param expressions visibilities
143     */
144   case class VisibilityOr(expressions: Seq[VisibilityExpression]) extends VisibilityExpression {
145     override def evaluate(authorizations: Seq[Array[Byte]]): Boolean = expressions.exists(_.evaluate(authorizations))
146     override def expression: String = {
147       val clauses = expressions.map {
148         case e: VisibilityAnd => s"(${e.expression})"
149         case e => e.expression
150       }
151       clauses.mkString("|")
152     }
153   }
154 
155   private def isValidAuthChar(b: Byte): Boolean = validAuthChars(0xff & b)
156 }

168   private def expression: Rule1[VisibilityExpression] = rule {
169     oneOrMore(term, "|") ~~> (a => if (a.length == 1) { a.head } else { VisibilityOr(a) })
170   }
171 
172   private def term: Rule1[VisibilityExpression] = rule {
173     oneOrMore(factor, "&") ~~> (a => if (a.length == 1) { a.head } else { VisibilityAnd(a) })
174   }
175 
176   private def factor: Rule1[VisibilityExpression] = rule { value | parens }
177 
178   private def parens: Rule1[VisibilityExpression] = rule { "(" ~ expression ~ ")" }
179 
180   private def authChar: Rule0 = rule { char | "-" | ":" | "." | "/" }
181 
182   private def unquotedAuthString: Rule1[String] = rule { oneOrMore(authChar) ~> { c => c } }
183 
184   private def authString: Rule1[String] = rule { quotedString | singleQuotedString | unquotedAuthString }
185 
186   private def value: Rule1[VisibilityExpression] = rule {
187     authString ~~> { s => VisibilityValue(s.getBytes(StandardCharsets.UTF_8))}
188   }
