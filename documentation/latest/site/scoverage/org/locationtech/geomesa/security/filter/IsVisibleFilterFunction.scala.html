25  * Function to compare auths with vis. Note that this function "fails open", in that if visibilities are not found
26  * it will count as visible

33   override def setParameters(params: java.util.List[Expression]): Unit = {
34     super.setParameters(params)

42   override def evaluate(obj: Object): Object = {
43     val vis = expression.evaluate(obj).asInstanceOf[String]

45       java.lang.Boolean.TRUE

51   }
