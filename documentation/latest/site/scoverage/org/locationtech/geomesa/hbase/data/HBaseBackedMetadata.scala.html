21 class HBaseBackedMetadata[T](connection: Connection, catalog: TableName, val serializer: MetadataSerializer[T])
22     extends KeyValueStoreMetadata[T] {
23 
24   import HBaseBackedMetadata._
25   import state.table
26 
27   // state object to allow table val to be instantiated before superclass initializes
28   private object state {
29     val table: Table = connection.getTable(catalog)
30   }
31 
32   override protected def checkIfTableExists: Boolean = WithClose(connection.getAdmin)(_.tableExists(catalog))
33 
34   override protected def createTable(): Unit = {
35     WithClose(connection.getAdmin) { admin =>
36       if (!admin.tableExists(catalog)) {
37         HBaseVersions.createTableAsync(admin, catalog, Seq(ColumnFamily), None, None, None, Some(true), None, Seq.empty)
38         HBaseIndexAdapter.waitForTable(admin, catalog)
39       }
40     }
41   }
42 
43   override protected def createEmptyBackup(timestamp: String): HBaseBackedMetadata[T] =
44     new HBaseBackedMetadata(connection, TableName.valueOf(s"${catalog}_${timestamp}_bak"), serializer)
45 
46   override protected def write(rows: Seq[(Array[Byte], Array[Byte])]): Unit =
47     table.put(rows.map { case (r, v) => new Put(r).addColumn(ColumnFamily, ColumnQualifier, v) }.toList.asJava)
48 
49   override protected def delete(rows: Seq[Array[Byte]]): Unit =
50     // note: list passed in must be mutable
51     table.delete(rows.map(r => new Delete(r)).toBuffer.asJava)
52 
53   override protected def scanValue(row: Array[Byte]): Option[Array[Byte]] = {
54     val result = table.get(new Get(row).addColumn(ColumnFamily, ColumnQualifier))
55     if (result.isEmpty) { None } else { Option(result.getValue(ColumnFamily, ColumnQualifier)) }
56   }
57 
58   override protected def scanRows(prefix: Option[Array[Byte]]): CloseableIterator[(Array[Byte], Array[Byte])] = {
59     val scan = new Scan().addColumn(ColumnFamily, ColumnQualifier)
60     prefix.foreach(scan.setRowPrefixFilter)
61     val scanner = table.getScanner(scan)
62     val results = scanner.iterator.asScala.map(s => (s.getRow, s.getValue(ColumnFamily, ColumnQualifier)))
63     CloseableIterator(results, scanner.close())
64   }
65 
66   override def close(): Unit = table.close()
67 }
68

70   val ColumnFamily: Array[Byte] = Bytes.toBytes("m")
71   val ColumnQualifier: Array[Byte] = Bytes.toBytes("v")
