19 object IndexFilters {
20 
21   /**
22    * Configure z/s curve filters, based on the index
23    *
24    * @param index feature index
25    * @param values query values
26    * @return
27    */
28   def apply(index: GeoMesaFeatureIndex[_, _], values: Any): Option[(Int, FilterBase)] = {
29     // TODO pull this out to be SPI loaded so that new indices can be added seamlessly
30     index match {
31       case _: Z3Index =>
32         Some(Z3HBaseFilter.Priority, Z3HBaseFilter(Z3Filter(values.asInstanceOf[Z3IndexValues]), index.keySpace.sharding.length))
33 
34       case _: Z2Index =>
35         Some(Z2HBaseFilter.Priority, Z2HBaseFilter(Z2Filter(values.asInstanceOf[Z2IndexValues]), index.keySpace.sharding.length))
36 
37       case _: S2Index =>
38         Some(S2HBaseFilter.Priority, S2HBaseFilter(S2Filter(values.asInstanceOf[S2IndexValues]), index.keySpace.sharding.length))
39 
40       case _: S3Index =>
41         Some(S3HBaseFilter.Priority, S3HBaseFilter(S3Filter(values.asInstanceOf[S3IndexValues]), index.keySpace.sharding.length))
42 
43       // TODO GEOMESA-1807 deal with non-points in a pushdown XZ filter
44       case _ => None
45     }
46   }
47 }
