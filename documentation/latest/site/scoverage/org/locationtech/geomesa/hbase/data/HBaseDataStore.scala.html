101   override protected def transitionIndices(sft: SimpleFeatureType): Unit = {
102     val dtg = sft.getDtgField.toSeq
103     val geom = Option(sft.getGeomField).toSeq
104 
105     val indices = Seq.newBuilder[IndexId]
106     val tableNameKeys = Seq.newBuilder[(String, String)]
107 
108     sft.getIndices.foreach {
109       case id if id.name == IdIndex.name =>
110         require(id.version == 1, s"Expected index version of 1 but got: $id")
111         indices += id.copy(version = 3)
112         tableNameKeys += ((s"table.${IdIndex.name}.v1", s"table.${IdIndex.name}.v3"))
113 
114       case id if id.name == Z3Index.name =>
115         require(id.version <= 2, s"Expected index version of 1 or 2 but got: $id")
116         indices += id.copy(attributes = geom ++ dtg, version = id.version + 3)
117         tableNameKeys += ((s"table.${Z3Index.name}.v${id.version}", s"table.${Z3Index.name}.v${id.version + 3}"))
118 
119       case id if id.name == XZ3Index.name =>
120         require(id.version == 1, s"Expected index version of 1 but got: $id")
121         indices += id.copy(attributes = geom ++ dtg)
122 
123       case id if id.name == Z2Index.name =>
124         require(id.version <= 2, s"Expected index version of 1 or 2 but got: $id")
125         indices += id.copy(attributes = geom, version = id.version + 2)
126         tableNameKeys += ((s"table.${Z2Index.name}.v${id.version}", s"table.${Z2Index.name}.v${id.version + 2}"))
127 
128       case id if id.name == XZ2Index.name =>
129         require(id.version == 1, s"Expected index version of 1 but got: $id")
130         indices += id.copy(attributes = geom)
131 
132       case id if id.name == AttributeIndex.name =>
133         require(id.version <= 5, s"Expected index version of 1-5 but got: $id")
134         lazy val fields = if (id.version == 1) { dtg } else { geom ++ dtg }
135         sft.getAttributeDescriptors.asScala.foreach { d =>
136           val index = d.getUserData.remove(AttributeOptions.OptIndex).asInstanceOf[String]
137           if (index == null || index.equalsIgnoreCase("false") || index.equalsIgnoreCase(IndexCoverage.NONE.toString)) {
138             // no-op
139           } else if (java.lang.Boolean.valueOf(index) || index.equalsIgnoreCase(IndexCoverage.FULL.toString) ||
140               index.equalsIgnoreCase(IndexCoverage.JOIN.toString)) {
141             indices += id.copy(attributes = Seq(d.getLocalName) ++ fields, version = id.version + 2)
142           } else {
143             throw new IllegalStateException(s"Expected an index coverage or boolean but got: $index")
144           }
145         }
146         tableNameKeys ++=
147             Seq(s"table.${AttributeIndex.name}.v${id.version}", "tables.idx.attr.name")
148                 .map((_, s"table.${AttributeIndex.name}.v${id.version + 2}"))
149     }
150 
151     sft.setIndices(indices.result)
152 
153     // update metadata keys for tables
154     tableNameKeys.result.foreach { case (from, to) =>
155       metadata.scan(sft.getTypeName, from, cache = false).foreach { case (key, name) =>
156         metadata.insert(sft.getTypeName, to + key.substring(from.length), name)
157         metadata.remove(sft.getTypeName, key)
158       }
159     }
160   }
