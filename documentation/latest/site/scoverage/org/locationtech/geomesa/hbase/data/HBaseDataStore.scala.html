103   override protected def transitionIndices(sft: SimpleFeatureType): Unit = {
104     val dtg = sft.getDtgField.toSeq
105     val geom = Option(sft.getGeomField).toSeq
106 
107     val indices = Seq.newBuilder[IndexId]
108     val tableNameKeys = Seq.newBuilder[(String, String)]
109 
110     sft.getIndices.foreach {
111       case id if id.name == IdIndex.name =>
112         require(id.version == 1, s"Expected index version of 1 but got: $id")
113         indices += id.copy(version = 3)
114         tableNameKeys += ((s"table.${IdIndex.name}.v1", s"table.${IdIndex.name}.v3"))
115 
116       case id if id.name == Z3Index.name =>
117         require(id.version <= 2, s"Expected index version of 1 or 2 but got: $id")
118         indices += id.copy(attributes = geom ++ dtg, version = id.version + 3)
119         tableNameKeys += ((s"table.${Z3Index.name}.v${id.version}", s"table.${Z3Index.name}.v${id.version + 3}"))
120 
121       case id if id.name == XZ3Index.name =>
122         require(id.version == 1, s"Expected index version of 1 but got: $id")
123         indices += id.copy(attributes = geom ++ dtg)
124 
125       case id if id.name == Z2Index.name =>
126         require(id.version <= 2, s"Expected index version of 1 or 2 but got: $id")
127         indices += id.copy(attributes = geom, version = id.version + 2)
128         tableNameKeys += ((s"table.${Z2Index.name}.v${id.version}", s"table.${Z2Index.name}.v${id.version + 2}"))
129 
130       case id if id.name == XZ2Index.name =>
131         require(id.version == 1, s"Expected index version of 1 but got: $id")
132         indices += id.copy(attributes = geom)
133 
134       case id if id.name == AttributeIndex.name =>
135         require(id.version <= 5, s"Expected index version of 1-5 but got: $id")
136         lazy val fields = if (id.version == 1) { dtg } else { geom ++ dtg }
137         sft.getAttributeDescriptors.asScala.foreach { d =>
138           val index = d.getUserData.remove(AttributeOptions.OptIndex).asInstanceOf[String]
139           if (index == null || index.equalsIgnoreCase("false") || index.equalsIgnoreCase(IndexCoverage.NONE.toString)) {
140             // no-op
141           } else if (java.lang.Boolean.valueOf(index) || index.equalsIgnoreCase(IndexCoverage.FULL.toString) ||
142               index.equalsIgnoreCase(IndexCoverage.JOIN.toString)) {
143             indices += id.copy(attributes = Seq(d.getLocalName) ++ fields, version = id.version + 2)
144           } else {
145             throw new IllegalStateException(s"Expected an index coverage or boolean but got: $index")
146           }
147         }
148         tableNameKeys ++=
149             Seq(s"table.${AttributeIndex.name}.v${id.version}", "tables.idx.attr.name")
150                 .map((_, s"table.${AttributeIndex.name}.v${id.version + 2}"))
151     }
152 
153     sft.setIndices(indices.result)
154 
155     // update metadata keys for tables
156     tableNameKeys.result.foreach { case (from, to) =>
157       metadata.scan(sft.getTypeName, from, cache = false).foreach { case (key, name) =>
158         metadata.insert(sft.getTypeName, to + key.substring(from.length), name)
159         metadata.remove(sft.getTypeName, key)
160       }
161     }
162   }
