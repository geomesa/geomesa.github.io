24 /**
25  * Callback class for invoking htable services

27  * @param scan initial scan (warning: may be mutated)
28  * @param options coprocessor configuration options
29  * @param result queue for storing results

36   val callable: Call[GeoMesaCoprocessorService, GeoMesaCoprocessorResponse] =
37     new Call[GeoMesaCoprocessorService, GeoMesaCoprocessorResponse]() {
38       override def call(instance: GeoMesaCoprocessorService): GeoMesaCoprocessorResponse = {
39         val controller = new GeoMesaHBaseRpcController()
40         val callback = new RpcCallbackImpl()
41         var cancelled = false
42         // note: synchronous call
43         try { instance.getResult(controller, buildRequest, callback) } catch {
44           case _: InterruptedException | _: InterruptedIOException | _: CancellationException =>
45             logger.warn("Cancelling remote coprocessor call")

48         }

51           logger.error(s"Controller failed with error:\n${controller.errorText()}")
52         }

55       }
56     }

58   // in `update` (below), we use scan.startRow as a sentinel to indicate we're done
59   // we won't ever start with a null row, as we use an empty byte array to indicate no start row
60   override def hasNext: Boolean = scan.getStartRow != null
61 
62   override def next: Scan = {

65   }

67   override def update(region: Array[Byte], row: Array[Byte], response: GeoMesaCoprocessorResponse): Unit = {
68     logger.trace(s"In update for region ${ByteArrays.printable(region)} and row ${ByteArrays.printable(row)}")
69 
70     val opt = Option(response)
71     opt.collect { case r if r.hasVersion => r.getVersion }.foreach { i =>
72       logger.warn(s"Coprocessors responded with incompatible version: $i")
73     }
74 
75     val lastScanned = opt.collect { case r if r.hasLastScanned => r.getLastScanned }.orNull
76     if (lastScanned != null && !lastScanned.isEmpty) {
77       if (updated != null) {
78         logger.error(
79           s"Last row was not null for region ${ByteArrays.printable(region)} and row ${ByteArrays.printable(row)}\n" +
80               "This indicates that one range spanned multiple regions - results might be incorrect.")
81       }
82       updated = lastScanned.toByteArray
83       scan.setStartRow(ByteArrays.rowFollowingRow(updated))
84     } else {
85       scan.setStartRow(null)
86     }
87 
88     val result = opt.map(_.getPayloadList).orNull
89     if (result != null) {
90       this.result.addAll(result)
91     }
92   }
93 
94   // since we mutate the scan start row, we need to rebuild the request for each call
95   private def buildRequest: GeoMesaCoprocessorRequest = {
96     val opts = options ++ Map(ScanOpt -> Base64.getEncoder.encodeToString(ProtobufUtil.toScan(scan).toByteArray))
97     GeoMesaCoprocessorRequest.newBuilder()
98         .setVersion(GeoMesaHBaseRequestVersion)
99         .setOptions(ByteString.copyFrom(GeoMesaCoprocessor.serializeOptions(opts)))
100         .build()
101   }

111 
112     override def run(parameter: GeoMesaCoprocessorResponse): Unit = {
113       response = parameter
114     }
115   }
