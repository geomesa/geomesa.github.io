96   private class RpcIterator(
97       connection: Connection,
98       table: TableName,
99       scan: Scan,
100       options: Map[String, String],
101       pool: ExecutorService
102     ) extends CloseableIterator[ByteString] {
103 
104     private var staged: ByteString = _
105 
106     private val htable = connection.getTable(table, pool)
107     private val resultQueue = new LinkedBlockingQueue[ByteString]()
108     private val call = CachedThreadPool.submit(new CoprocessorInvoker())
109 
110     override def hasNext: Boolean = {
111       if (staged != null) { true } else {
112         try {
113           staged = resultQueue.take()
114         } catch {
115           case _ :InterruptedException =>
116             // propagate the interruption through to the rpc call
117             call.cancel(true)
118             return false
119         }
120         if (terminator eq staged) {
121           resultQueue.add(staged)
122           staged = null
123           false
124         } else {
125           true
126         }
127       }
128     }
129 
130     override def next(): ByteString = {
131       val res = staged
132       staged = null
133       res
134     }
135 
136     override def close(): Unit = htable.close()
137 
138     class CoprocessorInvoker extends Runnable() {
139       override def run(): Unit = {
140         try {
141           val callback = new GeoMesaHBaseCallBack(scan, options, resultQueue)
142           var i = 0
143           // if the scan hasn't been killed and we are not done, issue a new request
144           while (callback.hasNext) {
145             val next = callback.next
146             logger.trace(
147               s"Calling htable.coprocessorService (call ${i += 1; i}): " +
148                   s"${ByteArrays.printable(next.getStartRow)} to ${ByteArrays.printable(next.getStopRow)}")
149             htable.coprocessorService(service, next.getStartRow, next.getStopRow, callback.callable, callback)
150           }
151         } catch {
152           case e @ (_ :InterruptedException | _ :InterruptedIOException) =>
153             logger.warn("Interrupted executing coprocessor query:", e)
154         } finally {
155           resultQueue.add(terminator)
156         }
157       }
158     }
159   }
