38 class HBaseIndexFileMapper extends Mapper[Writable, SimpleFeature, ImmutableBytesWritable, Put] with LazyLogging {
39 
40   private var sft: SimpleFeatureType = _
41   private var wrapper: FeatureWrapper[WritableFeature] = _
42   private var writer: WriteConverter[_] = _
43 
44   private var features: Counter = _
45   private var entries: Counter = _
46   private var failed: Counter = _
47 
48   private val bytes = new ImmutableBytesWritable
49 
50   override def setup(context: HBaseIndexFileMapper.MapContext): Unit = {
51     WithStore[HBaseDataStore](GeoMesaConfigurator.getDataStoreOutParams(context.getConfiguration)) { ds =>
52       require(ds != null, "Could not find data store - check your configuration and hbase-site.xml")
53       sft = ds.getSchema(HBaseIndexFileMapper.getTypeName(context.getConfiguration))
54       require(sft != null, "Could not find schema - check your configuration")
55       wrapper = WritableFeature.wrapper(sft, ds.adapter.groups)
56       writer = GeoMesaConfigurator.getIndicesOut(context.getConfiguration) match {
57         case Some(Seq(idx)) => ds.manager.index(sft, idx, IndexMode.Write).createConverter()
58         case _ => throw new IllegalArgumentException("Could not find write index - check your configuration")
59       }
60     }
61 
62     features = context.getCounter(OutputCounters.Group, OutputCounters.Written)
63     entries = context.getCounter(OutputCounters.Group, "entries")
64     failed = context.getCounter(OutputCounters.Group, OutputCounters.Failed)
65   }
66 
67   override def cleanup(context: HBaseIndexFileMapper.MapContext): Unit = {}
68 
69   override def map(key: Writable, value: SimpleFeature, context: HBaseIndexFileMapper.MapContext): Unit = {
70     // TODO create a common writer that will create mutations without writing them
71     try {
72       val feature = wrapper.wrap(value)
73       writer.convert(feature) match {
74         case kv: SingleRowKeyValue[_] =>
75           kv.values.foreach { value =>
76             val put = new Put(kv.row)
77             put.addImmutable(value.cf, value.cq, value.value)
78             if (!value.vis.isEmpty) {
79               put.setCellVisibility(new CellVisibility(new String(value.vis, StandardCharsets.UTF_8)))
80             }
81             put.setDurability(HBaseIndexAdapter.durability)
82 
83             bytes.set(put.getRow)
84             context.write(bytes, put)
85             entries.increment(1L)
86           }
87 
88         case mkv: MultiRowKeyValue[_] =>
89           mkv.rows.foreach { row =>
90             mkv.values.foreach { value =>
91               val put = new Put(row)
92               put.addImmutable(value.cf, value.cq, value.value)
93               if (!value.vis.isEmpty) {
94                 put.setCellVisibility(new CellVisibility(new String(value.vis, StandardCharsets.UTF_8)))
95               }
96               put.setDurability(HBaseIndexAdapter.durability)
97 
98               bytes.set(put.getRow)
99               context.write(bytes, put)
100               entries.increment(1L)
101             }
102           }
103       }
104       features.increment(1L)
105     } catch {
106       case NonFatal(e) =>
107         logger.error(s"Error writing feature ${Option(value).orNull}", e)
108         failed.increment(1L)
109     }
110   }
111 }

115   type MapContext = Mapper[Writable, SimpleFeature, ImmutableBytesWritable, Put]#Context
116 
117   private val TypeNameKey   = "org.locationtech.geomesa.hbase.type"
118 
119   private def setTypeName(conf: Configuration, typeName: String): Unit = conf.set(TypeNameKey, typeName)
120   private def getTypeName(conf: Configuration): String = conf.get(TypeNameKey)
121 
122   /**
123     * Sets mapper class, reducer class, output format and associated options
124     *
125     * @param job job
126     * @param params data store params for output data
127     * @param typeName feature type name to write (schema must exist already)
128     * @param index index table to write
129     * @param output output path for HFiles
130     */
131   def configure(
132       job: Job,
133       params: Map[String, String],
134       typeName: String,
135       index: String,
136       output: Path): Unit = {
137     WithStore[HBaseDataStore](params) { ds =>
138       require(ds != null, s"Could not find data store with provided parameters ${params.mkString(",")}")
139       val sft = ds.getSchema(typeName)
140       require(sft != null, s"Schema $typeName does not exist, please create it first")
141       require(!TablePartition.partitioned(sft), "Writing to partitioned tables is not currently supported")
142       val idx = ds.manager.index(sft, index, IndexMode.Write)
143       val tableName = idx.getTableNames(None) match {
144         case Seq(t) => TableName.valueOf(t) // should always be writing to a single table here
145         case tables => throw new IllegalStateException(s"Expected a single table but got: ${tables.mkString(", ")}")
146       }
147       val table = ds.connection.getTable(tableName)
148 
149       GeoMesaConfigurator.setDataStoreOutParams(job.getConfiguration, params)
150       GeoMesaConfigurator.setIndicesOut(job.getConfiguration, Seq(idx.identifier))
151       GeoMesaConfigurator.setSerialization(job.getConfiguration, sft)
152       setTypeName(job.getConfiguration, sft.getTypeName)
153       FileOutputFormat.setOutputPath(job, output)
154 
155       // this defaults to /user/<user>/hbase-staging, which generally doesn't exist...
156       job.getConfiguration.set(HConstants.TEMPORARY_FS_DIRECTORY_KEY,
157         s"${System.getProperty("java.io.tmpdir")}/hbase-staging")
158       // this requires a connection to hbase, which we don't always have
159       // TODO allow this as an option
160       job.getConfiguration.set(HFileOutputFormat2.LOCALITY_SENSITIVE_CONF_KEY, "false")
161 
162       job.setMapperClass(classOf[HBaseIndexFileMapper])
163       job.setMapOutputKeyClass(classOf[ImmutableBytesWritable])
164       job.setMapOutputValueClass(classOf[Put])
165 
166       // set hbase config
167       HBaseConfiguration.merge(job.getConfiguration, HBaseConfiguration.create(job.getConfiguration))
168       HBaseConnectionPool.configureSecurity(job.getConfiguration)
169 
170       // sets reducer, output classes and num-reducers
171       // override the libjars hbase tries to set, as they end up conflicting with the ones we set
172       val libjars = job.getConfiguration.get("tmpjars")
173       HFileOutputFormat2.configureIncrementalLoad(job, table, ds.connection.getRegionLocator(tableName))
174       job.getConfiguration.set("tmpjars", libjars)
175     }
176   }
