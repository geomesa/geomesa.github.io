29 /**
30   * Input format that allows processing of simple features from GeoMesa based on a CQL query
31   */
32 class GeoMesaHBaseInputFormat extends InputFormat[Text, SimpleFeature] with Configurable with LazyLogging {
33 
34   private val delegate = new MultiTableInputFormat
35 
36   /**
37     * Gets splits for a job.
38     */
39   override def getSplits(context: JobContext): java.util.List[InputSplit] = {
40     val splits = Security.doAuthorized(context.getConfiguration) {
41       delegate.getSplits(context)
42     }
43     logger.debug(s"Got ${splits.size()} splits")
44     splits
45   }
46 
47   override def createRecordReader(
48       split: InputSplit,
49       context: TaskAttemptContext
50     ): RecordReader[Text, SimpleFeature] = {
51     val toFeatures = GeoMesaConfigurator.getResultsToFeatures[Result](context.getConfiguration)
52     val reducer = GeoMesaConfigurator.getReducer(context.getConfiguration)
53     Security.doAuthorized(context.getConfiguration) {
54       new GeoMesaHBaseRecordReader(toFeatures, reducer, delegate.createRecordReader(split, context))
55     }
56   }
57 
58   override def setConf(conf: Configuration): Unit = {
59     delegate.setConf(conf)
60     // configurations aren't thread safe - if multiple input formats are configured at once,
61     // updating it could cause ConcurrentModificationExceptions
62     conf.synchronized {
63       // see TableMapReduceUtil.java
64       HBaseConfiguration.merge(conf, HBaseConfiguration.create(conf))
65       HBaseConnectionPool.configureSecurity(conf)
66     }
67   }
68 
69   override def getConf: Configuration = delegate.getConf
70 }

75     * Configure the input format based on a query
76     *
77     * @param job job to configure
78     * @param params data store parameters
79     * @param query query
80     */
81   def configure(job: Job, params: java.util.Map[String, _], query: Query): Unit = {
82     // get the query plan to set up the iterators, ranges, etc
83     val plan = WithStore[HBaseDataStore](params) { ds =>
84       assert(ds != null, "Invalid data store parameters")
85       HBaseJobUtils.getSingleScanPlan(ds, query)
86     }
87     configure(job, plan)
88   }
89 
90   /**
91     * Configure the input format based on a query plan
92     *
93     * @param job job to configure
94     * @param plan query plan
95     */
96   def configure(job: Job, plan: ScanPlan): Unit = {
97     job.setInputFormatClass(classOf[GeoMesaHBaseInputFormat])
98     configure(job.getConfiguration, plan)
99   }
100 
101   /**
102     * Configure the input format based on a query plan
103     *
104     * @param conf conf
105     * @param plan query plan
106     */
107   def configure(conf: Configuration, plan: ScanPlan): Unit = {
108     if (plan.scans.lengthCompare(1) != 0) {
109       throw new IllegalArgumentException(s"Query requires multiple tables: ${plan.scans.map(_.table).mkString(", ")}")
110     }
111     conf.set(TableInputFormat.INPUT_TABLE, plan.scans.head.table.getNameAsString)
112     // note: secondary filter is handled by scan push-down filter
113     val scans = plan.scans.head.scans.map { scan =>
114       // need to set the table name in each scan
115       scan.setAttribute(Scan.SCAN_ATTRIBUTES_TABLE_NAME, plan.scans.head.table.getName)
116       TableMapReduceUtil.convertScanToString(scan)
117     }
118     conf.setStrings(MultiTableInputFormat.SCANS, scans: _*)
119 
120     GeoMesaConfigurator.setResultsToFeatures(conf, plan.resultsToFeatures)
121     // note: reduce and sorting have to be handled in the job reducers
122     plan.reducer.foreach(GeoMesaConfigurator.setReducer(conf, _))
123     plan.sort.foreach(GeoMesaConfigurator.setSorting(conf, _))
124     plan.projection.foreach(GeoMesaConfigurator.setProjection(conf, _))
125   }
126 
127   /**
128     * Record reader for simple features
129     *
130     * @param toFeatures converts results to features
131     * @param reducer feature reducer, if any
132     * @param reader underlying hbase reader
133     */
134   class GeoMesaHBaseRecordReader(
135       toFeatures: ResultsToFeatures[Result],
136       reducer: Option[FeatureReducer],
137       reader: RecordReader[ImmutableBytesWritable, Result]
138     ) extends RecordReader[Text, SimpleFeature] with LazyLogging {
139 
140     private val features = {
141       val base = new RecordReaderIterator(reader, toFeatures)
142       reducer match {
143         case None => base
144         case Some(reduce) => reduce(base)
145       }
146     }
147 
148     private val key = new Text()
149     private var value: SimpleFeature = _
150 
151     override def initialize(split: InputSplit, context: TaskAttemptContext): Unit = reader.initialize(split, context)
152 
153     override def getProgress: Float = reader.getProgress
154 
155     override def nextKeyValue(): Boolean = {
156       if (features.hasNext) {
157         value = features.next
158         key.set(value.getID)
159         true
160       } else {
161         false
162       }
163     }
164 
165     override def getCurrentKey: Text = key
166 
167     override def getCurrentValue: SimpleFeature = value
168 
169     override def close(): Unit = features.close()
170   }
171 
172   private class RecordReaderIterator(
173       reader: RecordReader[ImmutableBytesWritable, Result],
174       toFeatures: ResultsToFeatures[Result]
175     ) extends CloseableIterator[SimpleFeature] {
176 
177     private var staged: SimpleFeature = _
178 
179     override def hasNext: Boolean = {
180       staged != null || {
181         if (reader.nextKeyValue()) {
182           staged = toFeatures(reader.getCurrentValue)
183           true
184         } else {
185           false
186         }
187       }
188     }
189 
190     override def next(): SimpleFeature = {
191       val res = staged
192       staged = null
193       res
194     }
195 
196     override def close(): Unit = reader.close()
197   }
