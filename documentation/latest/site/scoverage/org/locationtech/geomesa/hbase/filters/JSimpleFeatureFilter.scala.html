23   override def toByteArray: Array[Byte] = {

26     val transform = delegate.transform.map(_._1).orNull

31     def getLengthArray(s: String): Array[Byte] = {

35     }

37     val arrays =
38       Array(
39         getLengthArray(sftString),
40         getLengthArray(filterString),
41         getLengthArray(transform),

45   }

79     }
80     new JSimpleFeatureFilter(delegate)
