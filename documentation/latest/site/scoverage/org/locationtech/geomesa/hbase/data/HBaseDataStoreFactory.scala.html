40   override def createDataStore(params: java.util.Map[String, _]): DataStore = {
41     val connection = HBaseConnectionPool.getConnection(params, validateConnection)
42 
43     val remoteFilters = RemoteFilteringParam.lookup(params).booleanValue
44 
45     val audit = if (!AuditQueriesParam.lookup(params)) { None } else {
46       Some(new AuditLogger("hbase", AuditProvider.Loader.loadOrNone(params)))
47     }
48     val auths = if (!EnableSecurityParam.lookup(params)) { NoAuthsProvider } else {
49       HBaseDataStoreFactory.buildAuthsProvider(connection.connection, params)
50     }
51     val queries = HBaseQueryConfig(
52       threads = QueryThreadsParam.lookup(params),
53       timeout = QueryTimeoutParam.lookupOpt(params).map(_.toMillis),
54       looseBBox = LooseBBoxParam.lookup(params),
55       parallelPartitionScans = PartitionParallelScansParam.lookup(params),
56       maxRangesPerExtendedScan = MaxRangesPerExtendedScanParam.lookup(params)
57     )
58     val enabledCoprocessors = EnabledCoprocessors(
59       arrow = ArrowCoprocessorParam.lookup(params),
60       bin = BinCoprocessorParam.lookup(params),
61       density = DensityCoprocessorParam.lookup(params),
62       stats = StatsCoprocessorParam.lookup(params)
63     )
64     val coprocessors = CoprocessorConfig(
65       enabled = enabledCoprocessors,
66       threads = CoprocessorThreadsParam.lookup(params),
67       yieldPartialResults = YieldPartialResultsParam.lookup(params),
68       maxRangesPerExtendedScan = MaxRangesPerCoprocessorScanParam.lookup(params),
69       url = CoprocessorUrlParam.lookupOpt(params)
70     )
71     val config = HBaseDataStoreConfig(
72       catalog = HBaseCatalogParam.lookup(params),
73       remoteFilter = remoteFilters,
74       generateStats = GenerateStatsParam.lookup(params),
75       queries = queries,
76       coprocessors = coprocessors,
77       authProvider = auths,
78       audit = audit,
79       namespace = NamespaceParam.lookupOpt(params)
80     )
81 
82     logger.debug(s"Using ${if (remoteFilters) "remote" else "local" } filtering")
83     lazy val enabled =
84       Seq(ArrowCoprocessorParam, BinCoprocessorParam, DensityCoprocessorParam, StatsCoprocessorParam).collect {
85         case p if p.exists(params) && p.lookup(params).booleanValue() => p.key
86       }
87     if (!remoteFilters && enabled.nonEmpty) {
88       logger.warn(s"Ignoring configs '${enabled.mkString("', '")}' due to remote filtering being disabled")
89     }
90 
91     val ds = new HBaseDataStore(connection, config)
92     GeoMesaDataStore.initRemoteVersion(ds)
93     ds
94   }

204   def buildAuthsProvider(connection: Connection, params: java.util.Map[String, _]): AuthorizationsProvider = {
205     val forceEmptyOpt: Option[java.lang.Boolean] = ForceEmptyAuthsParam.lookupOpt(params)
206     val forceEmptyAuths = forceEmptyOpt.getOrElse(java.lang.Boolean.FALSE).asInstanceOf[Boolean]
207 
208     if (!VisibilityClient.isCellVisibilityEnabled(connection)) {
209       throw new IllegalArgumentException("HBase cell visibility is not enabled on cluster")
210     }
211 
212     // master auths is the superset of auths this connector/user can support
213     val userName = UserGroupInformation.getLoginUser.getUserName
214     val masterAuths = VisibilityClient.getAuths(connection, userName).getAuthList.asScala.map(_.toStringUtf8)
215 
216     // get the auth params passed in as a comma-delimited string
217     val configuredAuths = AuthsParam.lookupOpt(params).getOrElse("").split(",").filterNot(_.isEmpty)
218 
219     // verify that the configured auths are valid for the connector we are using (fail-fast)
220     val invalidAuths = configuredAuths.filterNot(masterAuths.contains)
221     if (invalidAuths.nonEmpty) {
222       val msg = s"The authorizations '${invalidAuths.mkString("', '")}' are not valid for the HBase user '$userName'"
223       if (masterAuths.isEmpty) {
224         // looking up auths requires a system-level user - likely the user does not have permission
225         logger.warn(s"$msg. This may be due to the user not having permissions" +
226             " to read its own authorizations, in which case this warning can be ignored.")
227       } else {
228         throw new IllegalArgumentException(s"$msg. Available authorizations are: ${masterAuths.mkString(", ")}")
229       }
230     }
231 
232     // if the caller provided any non-null string for authorizations, use it;
233     // otherwise, grab all authorizations to which the user is entitled
234     if (configuredAuths.length != 0 && forceEmptyAuths) {
235       throw new IllegalArgumentException("Forcing empty auths is checked, but explicit auths are provided")
236     }
237     val auths = if (forceEmptyAuths || configuredAuths.nonEmpty) { configuredAuths.toList } else { masterAuths.toList }
238 
239     AuthUtils.getProvider(params, auths)
240   }
