38   override def createDataStore(params: java.util.Map[String, _]): DataStore = {
39     val connection = HBaseConnectionPool.getConnection(params, validateConnection)
40 
41     val remoteFilters = RemoteFilteringParam.lookup(params).booleanValue
42 
43     val audit = if (!AuditQueriesParam.lookup(params)) { None } else {
44       Some(new AuditLogger("hbase", AuditProvider.Loader.loadOrNone(params)))
45     }
46     val metrics = MetricsRegistryParam.lookupRegistry(params)
47     val auths = if (!EnableSecurityParam.lookup(params)) { NoAuthsProvider } else {
48       HBaseDataStoreFactory.buildAuthsProvider(connection.connection, params)
49     }
50     val queries = HBaseQueryConfig(
51       threads = QueryThreadsParam.lookup(params),
52       timeout = QueryTimeoutParam.lookupOpt(params).map(_.toMillis),
53       looseBBox = LooseBBoxParam.lookup(params),
54       parallelPartitionScans = PartitionParallelScansParam.lookup(params),
55       maxRangesPerExtendedScan = MaxRangesPerExtendedScanParam.lookup(params)
56     )
57     val enabledCoprocessors = EnabledCoprocessors(
58       arrow = ArrowCoprocessorParam.lookup(params),
59       bin = BinCoprocessorParam.lookup(params),
60       density = DensityCoprocessorParam.lookup(params),
61       stats = StatsCoprocessorParam.lookup(params)
62     )
63     val coprocessors = CoprocessorConfig(
64       enabled = enabledCoprocessors,
65       threads = CoprocessorThreadsParam.lookup(params),
66       yieldPartialResults = YieldPartialResultsParam.lookup(params),
67       maxRangesPerExtendedScan = MaxRangesPerCoprocessorScanParam.lookup(params),
68       url = CoprocessorUrlParam.lookupOpt(params)
69     )
70     val config = HBaseDataStoreConfig(
71       catalog = HBaseCatalogParam.lookup(params),
72       remoteFilter = remoteFilters,
73       generateStats = GenerateStatsParam.lookup(params),
74       queries = queries,
75       coprocessors = coprocessors,
76       authProvider = auths,
77       audit = audit,
78       metrics = metrics,
79       namespace = NamespaceParam.lookupOpt(params)
80     )
81 
82     logger.debug(s"Using ${if (remoteFilters) "remote" else "local" } filtering")
83     lazy val enabled =
84       Seq(ArrowCoprocessorParam, BinCoprocessorParam, DensityCoprocessorParam, StatsCoprocessorParam).collect {
85         case p if p.exists(params) && p.lookup(params).booleanValue() => p.key
86       }
87     if (!remoteFilters && enabled.nonEmpty) {
88       logger.warn(s"Ignoring configs '${enabled.mkString("', '")}' due to remote filtering being disabled")
89     }
90 
91     val ds = new HBaseDataStore(connection, config)
92     GeoMesaDataStore.initRemoteVersion(ds)
93     ds
94   }

193   def buildAuthsProvider(connection: Connection, params: java.util.Map[String, _]): AuthorizationsProvider = {
194     val forceEmptyOpt: Option[java.lang.Boolean] = ForceEmptyAuthsParam.lookupOpt(params)
195     val forceEmptyAuths = forceEmptyOpt.getOrElse(java.lang.Boolean.FALSE).asInstanceOf[Boolean]
196 
197     if (!VisibilityClient.isCellVisibilityEnabled(connection)) {
198       throw new IllegalArgumentException("HBase cell visibility is not enabled on cluster")
199     }
200 
201     // master auths is the superset of auths this connector/user can support
202     val userName = UserGroupInformation.getLoginUser.getUserName
203     val masterAuths = VisibilityClient.getAuths(connection, userName).getAuthList.asScala.map(_.toStringUtf8)
204 
205     // get the auth params passed in as a comma-delimited string
206     val configuredAuths = AuthsParam.lookupOpt(params).getOrElse("").split(",").filterNot(_.isEmpty)
207 
208     // verify that the configured auths are valid for the connector we are using (fail-fast)
209     val invalidAuths = configuredAuths.filterNot(masterAuths.contains)
210     if (invalidAuths.nonEmpty) {
211       val msg = s"The authorizations '${invalidAuths.mkString("', '")}' are not valid for the HBase user '$userName'"
212       if (masterAuths.isEmpty) {
213         // looking up auths requires a system-level user - likely the user does not have permission
214         logger.warn(s"$msg. This may be due to the user not having permissions" +
215             " to read its own authorizations, in which case this warning can be ignored.")
216       } else {
217         throw new IllegalArgumentException(s"$msg. Available authorizations are: ${masterAuths.mkString(", ")}")
218       }
219     }
220 
221     // if the caller provided any non-null string for authorizations, use it;
222     // otherwise, grab all authorizations to which the user is entitled
223     if (configuredAuths.length != 0 && forceEmptyAuths) {
224       throw new IllegalArgumentException("Forcing empty auths is checked, but explicit auths are provided")
225     }
226     val auths = if (forceEmptyAuths || configuredAuths.nonEmpty) { configuredAuths.toList } else { masterAuths.toList }
227 
228     AuthUtils.getProvider(params, auths)
229   }
