44     * Scans to be executed
45     *
46     * @return

48   def scans: Seq[TableScan]

50   override def scan(): CloseableIterator[Results] = {
51     // query guard hook - also handles full table scan checks
52     ds.interceptors.runGuards(strategy)
53     // convert the relative timeout to an absolute timeout up front
54     val timeout = ds.config.queries.timeout.map(Timeout.apply)
55     val iter = scans.iterator.map(singleTableScan(_, ds.connection, threads, timeout))
56     if (ds.config.queries.parallelPartitionScans) {
57       // kick off all the scans at once
58       iter.foldLeft(CloseableIterator.empty[Results])(_ concat _)
59     } else {
60       // kick off the scans sequentially as they finish
61       SelfClosingIterator(iter).flatMap(s => s)
62     }
63   }
64 
65   override def explain(explainer: Explainer): Unit = {
66     explainer.pushLevel(s"Plan: ${getClass.getSimpleName}")
67     explainer(s"Tables: ${scans.map(_.table).mkString(", ")}")
68     explainer(s"Ranges (${ranges.size}): ${ranges.take(5).map(rangeToString).mkString(", ")}")
69     explainer(s"Scans (${scans.headOption.map(_.scans.size).getOrElse(0)}): ${scans.headOption.toSeq.flatMap(_.scans.take(5)).map(scanToString).mkString(", ")}")
70     explainer(s"Column families: ${scans.headOption.flatMap(_.scans.headOption).flatMap(r => Option(r.getFamilies)).getOrElse(Array.empty).map(Bytes.toString).mkString(",")}")
71     explainer(s"Remote filters: ${scans.headOption.flatMap(_.scans.headOption).flatMap(r => Option(r.getFilter)).map(filterToString).getOrElse("none")}")
72     moreExplaining(explainer)
73     explainer(s"Reduce: ${reducer.getOrElse("none")}")
74     explainer.popLevel()
75   }
