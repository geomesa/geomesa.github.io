38     *
39     * @return

41   def scans: Seq[TableScan]

43   override def scan(ds: HBaseDataStore): CloseableIterator[Results] = {
44     // convert the relative timeout to an absolute timeout up front
45     val timeout = ds.config.queries.timeout.map(Timeout.apply)
46     val iter = scans.iterator.map(singleTableScan(_, ds.connection, threads(ds), timeout))
47     if (ds.config.queries.parallelPartitionScans) {
48       // kick off all the scans at once
49       iter.foldLeft(CloseableIterator.empty[Results])(_ concat _)
50     } else {
51       // kick off the scans sequentially as they finish
52       SelfClosingIterator(iter).flatMap(s => s)
53     }
54   }
55 
56   override def explain(explainer: Explainer, prefix: String = ""): Unit = {
57     explainer.pushLevel(s"${prefix}Plan: ${getClass.getSimpleName}")
58     explainer(s"Tables: ${scans.map(_.table).mkString(", ")}")
59     explainer(s"Ranges (${ranges.size}): ${ranges.take(5).map(rangeToString).mkString(", ")}")
60     explainer(s"Scans (${scans.headOption.map(_.scans.size).getOrElse(0)}): ${scans.headOption.toSeq.flatMap(_.scans.take(5)).map(scanToString).mkString(", ")}")
61     explainer(s"Column families: ${scans.headOption.flatMap(_.scans.headOption).flatMap(r => Option(r.getFamilies)).getOrElse(Array.empty).map(Bytes.toString).mkString(",")}")
62     explainer(s"Remote filters: ${scans.headOption.flatMap(_.scans.headOption).flatMap(r => Option(r.getFilter)).map(filterToString).getOrElse("none")}")
63     explain(explainer)
64     explainer(s"Reduce: ${reducer.getOrElse("none")}")
65     explainer.popLevel()
66   }
