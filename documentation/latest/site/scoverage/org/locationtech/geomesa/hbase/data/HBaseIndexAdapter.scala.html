85   override def createTable(
86       index: GeoMesaFeatureIndex[_, _],
87       partition: Option[String],
88       splits: => Seq[Array[Byte]]): Unit = {
89     // write table name to metadata
90     val name = TableName.valueOf(index.configureTableName(partition, tableNameLimit))
91 
92     WithClose(ds.connection.getAdmin) { admin =>
93       if (!admin.tableExists(name)) {
94         logger.debug(s"Creating table $name")
95 
96         val conf = admin.getConfiguration
97 
98         val compression = index.sft.getCompression.map { alg =>
99           logger.debug(s"Setting compression '$alg' on table $name for feature ${index.sft.getTypeName}")
100           // note: all compression types in HBase are case-sensitive and lower-cased
101           Compression.getCompressionAlgorithmByName(alg.toLowerCase(Locale.US))
102         }
103 
104         val cols = groups.apply(index.sft).map(_._1)
105         val bloom = Some(BloomType.NONE)
106         val encoding = if (index.name == IdIndex.name) { None } else { Some(DataBlockEncoding.FAST_DIFF) }
107 
108         val coprocessor = if (!ds.config.remoteFilter) { None } else {
109           // if the coprocessors are installed site-wide don't register them in the table descriptor.
110           // this key is CoprocessorHost.USER_REGION_COPROCESSOR_CONF_KEY - but don't want to pull in
111           // a dependency on hbase-server just for this constant
112           val installed = Option(conf.get("hbase.coprocessor.user.region.classes"))
113           val names = installed.map(_.split(":").toSet).getOrElse(Set.empty[String])
114           if (names.contains(CoprocessorClass)) { None } else {
115             // noinspection ScalaDeprecation
116             def urlFromSysProp: Option[Path] = CoprocessorUrl.option.orElse(CoprocessorPath.option).map(new Path(_))
117             val coprocessorUrl = ds.config.coprocessors.url.orElse(urlFromSysProp).orElse(dynamicJarPath)
118             logger.debug(s"Using coprocessor path ${coprocessorUrl.orNull}")
119             // TODO: Warn if the path given is different from paths registered in other coprocessors
120             // if so, other tables would need updating
121             Some(CoprocessorClass -> coprocessorUrl)
122           }
123         }
124 
125         try {
126           HBaseVersions.createTableAsync(admin, name, cols, bloom, compression, encoding, None, coprocessor, splits)
127         } catch {
128           case _: org.apache.hadoop.hbase.TableExistsException => // ignore, another thread created it for us
129         }
130       }
131 
132       waitForTable(admin, name)
133     }
134   }

136   override def renameTable(from: String, to: String): Unit = {
137     WithClose(ds.connection.getAdmin) { admin =>
138       val existing = TableName.valueOf(from)
139       val renamed = TableName.valueOf(to)
140       if (admin.tableExists(existing)) {
141         // renaming in hbase requires creating a snapshot and using that to create the new table
142         val snapshot = StringSerialization.alphaNumericSafeString(UUID.randomUUID().toString)
143         admin.disableTable(existing)
144         admin.snapshot(snapshot, existing)
145         admin.cloneSnapshot(snapshot, renamed)
146         admin.deleteSnapshot(snapshot)
147         admin.deleteTable(existing)
148         waitForTable(admin, renamed)
149       }
150     }
151   }

153   override def deleteTables(tables: Seq[String]): Unit = {
154     WithClose(ds.connection.getAdmin) { admin =>
155       def deleteOne(name: String): Unit = {
156         val table = TableName.valueOf(name)
157         if (admin.tableExists(table)) {
158           HBaseVersions.disableTableAsync(admin, table)
159           val timeout = TableAvailabilityTimeout.toUnboundedDuration.filter(_.isFinite)
160           logger.debug(s"Waiting for table '$table' to be disabled with " +
161               s"${timeout.map(t => s"a timeout of $t").getOrElse("no timeout")}")
162           val stop = timeout.map(t => System.currentTimeMillis() + t.toMillis)
163           while (!admin.isTableDisabled(table) && stop.forall(_ > System.currentTimeMillis())) {
164             Thread.sleep(1000)
165           }
166           // no async operation, but timeout can be controlled through hbase-site.xml "hbase.client.sync.wait.timeout.msec"
167           admin.deleteTable(table)
168         }
169       }
170       tables.toList.map(t => CachedThreadPool.submit(() => deleteOne(t))).foreach(_.get)
171     }
172   }

174   override def clearTables(tables: Seq[String], prefix: Option[Array[Byte]]): Unit = {
175     def clearOne(name: String): Unit = {
176       val tableName = TableName.valueOf(name)
177       WithClose(ds.connection.getTable(tableName)) { table =>
178         val scan = new Scan().setFilter(new KeyOnlyFilter)
179         prefix.foreach(scan.setRowPrefixFilter)
180         ds.applySecurity(scan)
181         val mutateParams = new BufferedMutatorParams(tableName)
182         WithClose(table.getScanner(scan), ds.connection.getBufferedMutator(mutateParams)) { case (scanner, mutator) =>
183           scanner.iterator.asScala.grouped(10000).foreach { result =>
184             // TODO GEOMESA-2546 set delete visibilities
185             val deletes = result.map(r => new Delete(r.getRow))
186             mutator.mutate(deletes.asJava)
187           }
188         }
189       }
190     }
191     tables.toList.map(t => CachedThreadPool.submit(() => clearOne(t))).foreach(_.get)
192   }

194   override def createQueryPlan(strategy: QueryStrategy): HBaseQueryPlan = {
195 
196     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
197 
198     val QueryStrategy(filter, byteRanges, _, _, ecql, hints, _) = strategy
199     val index = filter.index
200 
201     // index api defines empty start/end for open-ended range
202     // index api defines start row inclusive, end row exclusive
203     // both these conventions match the conventions for hbase scan objects
204     val ranges = byteRanges.map {
205       case BoundedByteRange(start, stop) => new RowRange(start, true, stop, false)
206       case SingleRowByteRange(row)       => new RowRange(row, true, ByteArrays.rowFollowingRow(row), false)
207     }
208     val small = byteRanges.headOption.exists(_.isInstanceOf[SingleRowByteRange])
209 
210     val tables = index.getTablesForQuery(filter.filter).map(TableName.valueOf)
211     val (colFamily, schema) = groups.group(index.sft, hints.getTransformDefinition, ecql)
212 
213     val transform: Option[(String, SimpleFeatureType)] = hints.getTransform
214 
215     // check for an empty query plan, if there are no tables or ranges to scan
216     def empty(reducer: Option[FeatureReducer]): Option[HBaseQueryPlan] =
217       if (tables.isEmpty || ranges.isEmpty) { Some(EmptyPlan(filter, reducer)) } else { None }
218 
219     if (!ds.config.remoteFilter) {
220       // everything is done client side
221       // note: we assume visibility filtering is still done server-side as it's part of core hbase
222       // note: we use the full filter here, since we can't use the z3 server-side filter
223       // for some attribute queries we wouldn't need the full filter...
224       val reducer = Some(new LocalTransformReducer(schema, filter.filter, None, transform, hints))
225       empty(reducer).getOrElse {
226         val scans = configureScans(tables, ranges, small, colFamily, Seq.empty, coprocessor = false)
227         val resultsToFeatures = new HBaseResultsToFeatures(index, schema)
228         val sort = hints.getSortFields
229         val max = hints.getMaxFeatures
230         val project = hints.getProjection
231         ScanPlan(filter, ranges, scans, resultsToFeatures, reducer, sort, max, project)
232       }
233     } else {
234       // TODO pull this out to be SPI loaded so that new indices can be added seamlessly
235       val indexFilter = strategy.index match {
236         case _: Z3Index =>
237           strategy.values.map { case v: Z3IndexValues =>
238             (Z3HBaseFilter.Priority, Z3HBaseFilter(Z3Filter(v), index.keySpace.sharding.length))
239           }
240 
241         case _: Z2Index =>
242           strategy.values.map { case v: Z2IndexValues =>
243             (Z2HBaseFilter.Priority, Z2HBaseFilter(Z2Filter(v), index.keySpace.sharding.length))
244           }
245 
246         case _: S2Index =>
247           strategy.values.map { case v: S2IndexValues =>
248             (S2HBaseFilter.Priority, S2HBaseFilter(S2Filter(v), index.keySpace.sharding.length))
249           }
250 
251         case _: S3Index =>
252           strategy.values.map { case v: S3IndexValues =>
253             (S3HBaseFilter.Priority, S3HBaseFilter(S3Filter(v), index.keySpace.sharding.length))
254           }
255         // TODO GEOMESA-1807 deal with non-points in a pushdown XZ filter
256 
257         case _ => None
258       }
259 
260       val max = hints.getMaxFeatures
261       val projection = hints.getProjection
262       lazy val returnSchema = transform.map(_._2).getOrElse(schema)
263       lazy val filters = {
264         val cqlFilter = if (ecql.isEmpty && transform.isEmpty && hints.getSampling.isEmpty) { Seq.empty } else {
265           Seq((CqlTransformFilter.Priority, CqlTransformFilter(schema, strategy.index, ecql, transform, hints)))
266         }
267         (cqlFilter ++ indexFilter).sortBy(_._1).map(_._2)
268       }
269       lazy val coprocessorOptions =
270         Map(GeoMesaCoprocessor.YieldOpt -> String.valueOf(ds.config.coprocessors.yieldPartialResults))
271       lazy val scans = configureScans(tables, ranges, small, colFamily, filters, coprocessor = false)
272       lazy val coprocessorScans =
273         configureScans(tables, ranges, small, colFamily, indexFilter.toSeq.map(_._2), coprocessor = true)
274       lazy val resultsToFeatures = new HBaseResultsToFeatures(index, returnSchema)
275       lazy val localReducer = Some(new LocalTransformReducer(returnSchema, None, None, None, hints))
276 
277       if (hints.isDensityQuery) {
278         empty(None).getOrElse {
279           if (ds.config.coprocessors.enabled.density) {
280             val options = HBaseDensityAggregator.configure(schema, index, ecql, hints) ++ coprocessorOptions
281             val results = new HBaseDensityResultsToFeatures()
282             CoprocessorPlan(filter, ranges, coprocessorScans, options, results, None, max, projection)
283           } else {
284             if (hints.isSkipReduce) {
285               // override the return sft to reflect what we're actually returning,
286               // since the density sft is only created in the local reduce step
287               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
288             }
289             ScanPlan(filter, ranges, scans, resultsToFeatures, localReducer, None, max, projection)
290           }
291         }
292       } else if (hints.isArrowQuery) {
293         val config = HBaseArrowAggregator.configure(schema, index, ds.stats, filter.filter, ecql, hints)
294         val reducer = Some(config.reduce)
295         empty(reducer).getOrElse {
296           if (ds.config.coprocessors.enabled.arrow) {
297             val options = config.config ++ coprocessorOptions
298             val results = new HBaseArrowResultsToFeatures()
299             CoprocessorPlan(filter, ranges, coprocessorScans, options, results, reducer, max, projection)
300           } else {
301             if (hints.isSkipReduce) {
302               // override the return sft to reflect what we're actually returning,
303               // since the arrow sft is only created in the local reduce step
304               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
305             }
306             ScanPlan(filter, ranges, scans, resultsToFeatures, localReducer, None, max, projection)
307           }
308         }
309       } else if (hints.isStatsQuery) {
310         val reducer = Some(StatsScan.StatsReducer(returnSchema, hints))
311         empty(reducer).getOrElse {
312           if (ds.config.coprocessors.enabled.stats) {
313             val options = HBaseStatsAggregator.configure(schema, index, ecql, hints) ++ coprocessorOptions
314             val results = new HBaseStatsResultsToFeatures()
315             CoprocessorPlan(filter, ranges, coprocessorScans, options, results, reducer, max, projection)
316           } else {
317             if (hints.isSkipReduce) {
318               // override the return sft to reflect what we're actually returning,
319               // since the stats sft is only created in the local reduce step
320               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
321             }
322             ScanPlan(filter, ranges, scans, resultsToFeatures, localReducer, None, max, projection)
323           }
324         }
325       } else if (hints.isBinQuery) {
326         empty(None).getOrElse {
327           if (ds.config.coprocessors.enabled.bin) {
328             val options = HBaseBinAggregator.configure(schema, index, ecql, hints) ++ coprocessorOptions
329             val results = new HBaseBinResultsToFeatures()
330             CoprocessorPlan(filter, ranges, coprocessorScans, options , results, None, max, projection)
331           } else {
332             if (hints.isSkipReduce) {
333               // override the return sft to reflect what we're actually returning,
334               // since the bin sft is only created in the local reduce step
335               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
336             }
337             ScanPlan(filter, ranges, scans, resultsToFeatures, localReducer, None, max, projection)
338           }
339         }
340       } else {
341         empty(None).getOrElse {
342           ScanPlan(filter, ranges, scans, resultsToFeatures, None, hints.getSortFields, max, projection)
343         }
344       }
345     }
346   }

362   /**
363    * Configure the hbase scan
364    *
365    * @param tables tables being scanned, used for region location information
366    * @param ranges ranges to scan, non-empty. needs to be mutable as we will sort it in place
367    * @param small whether 'small' ranges (i.e. gets)
368    * @param colFamily col family to scan
369    * @param filters scan filters
370    * @param coprocessor is this a coprocessor scan or not
371    * @return
372    */
373   protected def configureScans(
374       tables: Seq[TableName],
375       ranges: Seq[RowRange],
376       small: Boolean,
377       colFamily: Array[Byte],
378       filters: Seq[HFilter],
379       coprocessor: Boolean): Seq[TableScan] = {
380     val cacheBlocks = HBaseSystemProperties.ScannerBlockCaching.toBoolean.get // has a default value so .get is safe
381     val cacheSize = HBaseSystemProperties.ScannerCaching.toInt
382 
383     logger.debug(s"HBase client scanner: block caching: $cacheBlocks, caching: $cacheSize")
384 
385     if (small && !coprocessor) {
386       val filter = filters match {
387         case Nil    => None
388         case Seq(f) => Some(f)
389         case f      => Some(new FilterList(f: _*))
390       }
391       // note: we have to copy the ranges for each table scan
392       tables.map { table =>
393         val scans = ranges.map { r =>
394           val scan = new Scan(r.getStartRow, r.getStopRow)
395           scan.addFamily(colFamily).setCacheBlocks(cacheBlocks).setSmall(true)
396           filter.foreach(scan.setFilter)
397           cacheSize.foreach(scan.setCaching)
398           ds.applySecurity(scan)
399           scan
400         }
401         TableScan(table, scans)
402       }
403     } else {
404       // split and group ranges by region server
405       // note: we have to copy the ranges for each table scan anyway
406       val rangesPerTable: Seq[(TableName, collection.Map[String, java.util.List[RowRange]])] =
407         tables.map(t => t -> groupRangesByRegion(t, ranges))
408 
409       def createGroup(group: java.util.List[RowRange]): Scan = {
410         val scan = new Scan(group.get(0).getStartRow, group.get(group.size() - 1).getStopRow)
411         val mrrf = if (group.size() < 2) { filters } else {
412           // TODO GEOMESA-1806
413           // currently, the MultiRowRangeFilter constructor will call sortAndMerge a second time
414           // this is unnecessary as we have already sorted and merged
415           // note: mrrf first priority
416           filters.+:(new MultiRowRangeFilter(group))
417         }
418         scan.setFilter(if (mrrf.lengthCompare(1) > 0) { new FilterList(mrrf: _*) } else { mrrf.headOption.orNull })
419         scan.addFamily(colFamily).setCacheBlocks(cacheBlocks)
420         cacheSize.foreach(scan.setCaching)
421 
422         // apply visibilities
423         ds.applySecurity(scan)
424 
425         scan
426       }
427 
428       rangesPerTable.map { case (table, rangesPerRegion) =>
429         val maxRangesPerGroup = {
430           def calcMax(maxPerGroup: Int, threads: Int): Int = {
431             val totalRanges = rangesPerRegion.values.map(_.size).sum
432             math.min(maxPerGroup, math.max(1, math.ceil(totalRanges.toDouble / threads).toInt))
433           }
434           if (coprocessor) {
435             calcMax(ds.config.coprocessors.maxRangesPerExtendedScan, ds.config.coprocessors.threads)
436           } else {
437             calcMax(ds.config.queries.maxRangesPerExtendedScan, ds.config.queries.threads)
438           }
439         }
440 
441         val groupedScans = Seq.newBuilder[Scan]
442 
443         rangesPerRegion.foreach { case (_, list) =>
444           // our ranges are non-overlapping, so just sort them but don't bother merging them
445           Collections.sort(list)
446 
447           var i = 0
448           while (i < list.size()) {
449             val groupSize = math.min(maxRangesPerGroup, list.size() - i)
450             groupedScans += createGroup(list.subList(i, i + groupSize))
451             i += groupSize
452           }
453         }
454 
455         // shuffle the ranges, otherwise our threads will tend to all hit the same region server at once
456         TableScan(table, Random.shuffle(groupedScans.result))
457       }
458     }
459   }

478   /**
479    * Group the range based on the region server hosting it. Splits ranges as needed if they span
480    * more than one region
481    *
482    * @param locator region locator
483    * @param range range to group
484    * @param result collected results
485    */
486   @scala.annotation.tailrec
487   private def groupRange(
488       locator: RegionLocator,
489       range: RowRange,
490       result: scala.collection.mutable.Map[String, java.util.List[RowRange]]): Unit = {
491     var encodedName: String = null
492     var split: Array[Byte] = null
493     try {
494       val regionInfo = locator.getRegionLocation(range.getStartRow).getRegionInfo
495       encodedName = regionInfo.getEncodedName
496       val regionEndKey = regionInfo.getEndKey // note: this is exclusive
497       if (regionEndKey.nonEmpty &&
498           (range.getStopRow.isEmpty || ByteArrays.ByteOrdering.compare(regionEndKey, range.getStopRow) <= 0)) {
499         if (ByteArrays.ByteOrdering.compare(range.getStartRow, regionEndKey) < 0) {
500           split = regionEndKey
501         } else {
502           logger.warn(s"HBase region location does not correspond to requested range:\n" +
503               s"  requested row: ${ByteArrays.toHex(range.getStartRow)}\n" +
504               s"  region: $encodedName ${ByteArrays.toHex(regionInfo.getStartKey)} :: ${ByteArrays.toHex(regionEndKey)}")
505         }
506       }
507     } catch {
508       case NonFatal(e) => logger.warn(s"Error checking range location for '$range''", e)
509     }
510     val buffer = result.getOrElseUpdate(encodedName, new java.util.ArrayList())
511     if (split == null) {
512       buffer.add(range)
513     } else {
514       // split the range based on the current region
515       buffer.add(new RowRange(range.getStartRow, true, split, false))
516       groupRange(locator, new RowRange(split, true, range.getStopRow, false), result)
517     }
518   }
