83   override def createTable(
84       index: GeoMesaFeatureIndex[_, _],
85       partition: Option[String],
86       splits: => Seq[Array[Byte]]): Unit = {
87     // write table name to metadata
88     val name = TableName.valueOf(index.configureTableName(partition, tableNameLimit))
89 
90     WithClose(ds.connection.getAdmin) { admin =>
91       if (!admin.tableExists(name)) {
92         logger.debug(s"Creating table $name")
93 
94         val conf = admin.getConfiguration
95 
96         val compression = index.sft.getCompression.map { alg =>
97           logger.debug(s"Setting compression '$alg' on table $name for feature ${index.sft.getTypeName}")
98           // note: all compression types in HBase are case-sensitive and lower-cased
99           Compression.getCompressionAlgorithmByName(alg.toLowerCase(Locale.US))
100         }
101 
102         val cols = groups.apply(index.sft).map(_._1)
103         val bloom = Some(BloomType.NONE)
104         val encoding = if (index.name == IdIndex.name) { None } else { Some(DataBlockEncoding.FAST_DIFF) }
105 
106         val coprocessor = if (!ds.config.remoteFilter) { None } else {
107           // if the coprocessors are installed site-wide don't register them in the table descriptor.
108           // this key is CoprocessorHost.USER_REGION_COPROCESSOR_CONF_KEY - but don't want to pull in
109           // a dependency on hbase-server just for this constant
110           val installed = Option(conf.get("hbase.coprocessor.user.region.classes"))
111           val names = installed.map(_.split(":").toSet).getOrElse(Set.empty[String])
112           if (names.contains(CoprocessorClass)) { None } else {
113             // noinspection ScalaDeprecation
114             def urlFromSysProp: Option[Path] = CoprocessorUrl.option.orElse(CoprocessorPath.option).map(new Path(_))
115             val coprocessorUrl = ds.config.coprocessors.url.orElse(urlFromSysProp).orElse(dynamicJarPath)
116             logger.debug(s"Using coprocessor path ${coprocessorUrl.orNull}")
117             // TODO: Warn if the path given is different from paths registered in other coprocessors
118             // if so, other tables would need updating
119             Some(CoprocessorClass -> coprocessorUrl)
120           }
121         }
122 
123         try {
124           HBaseVersions.createTableAsync(admin, name, cols, bloom, compression, encoding, None, coprocessor, splits)
125         } catch {
126           case _: org.apache.hadoop.hbase.TableExistsException => // ignore, another thread created it for us
127         }
128       }
129 
130       waitForTable(admin, name)
131     }
132   }

134   override def renameTable(from: String, to: String): Unit = {
135     WithClose(ds.connection.getAdmin) { admin =>
136       val existing = TableName.valueOf(from)
137       val renamed = TableName.valueOf(to)
138       if (admin.tableExists(existing)) {
139         // renaming in hbase requires creating a snapshot and using that to create the new table
140         val snapshot = StringSerialization.alphaNumericSafeString(UUID.randomUUID().toString)
141         admin.disableTable(existing)
142         admin.snapshot(snapshot, existing)
143         admin.cloneSnapshot(snapshot, renamed)
144         admin.deleteSnapshot(snapshot)
145         admin.deleteTable(existing)
146         waitForTable(admin, renamed)
147       }
148     }

151   override def deleteTables(tables: Seq[String]): Unit = {
152     WithClose(ds.connection.getAdmin) { admin =>
153       def deleteOne(name: String): Unit = {
154         val table = TableName.valueOf(name)
155         if (admin.tableExists(table)) {
156           HBaseVersions.disableTableAsync(admin, table)
157           val timeout = TableAvailabilityTimeout.toUnboundedDuration.filter(_.isFinite)
158           logger.debug(s"Waiting for table '$table' to be disabled with " +
159               s"${timeout.map(t => s"a timeout of $t").getOrElse("no timeout")}")
160           val stop = timeout.map(t => System.currentTimeMillis() + t.toMillis)
161           while (!admin.isTableDisabled(table) && stop.forall(_ > System.currentTimeMillis())) {
162             Thread.sleep(1000)
163           }
164           // no async operation, but timeout can be controlled through hbase-site.xml "hbase.client.sync.wait.timeout.msec"
165           admin.deleteTable(table)
166         }
167       }
168       tables.toList.map(t => CachedThreadPool.submit(() => deleteOne(t))).foreach(_.get)
169     }
170   }

173     def clearOne(name: String): Unit = {
174       val tableName = TableName.valueOf(name)
175       WithClose(ds.connection.getTable(tableName)) { table =>
176         val scan = new Scan().setFilter(new KeyOnlyFilter)
177         prefix.foreach(scan.setStartStopRowForPrefixScan)
178         ds.applySecurity(scan)
179         val mutateParams = new BufferedMutatorParams(tableName)
180         WithClose(table.getScanner(scan), ds.connection.getBufferedMutator(mutateParams)) { case (scanner, mutator) =>

182             // TODO GEOMESA-2546 set delete visibilities

184             mutator.mutate(deletes.asJava)
185           }
186         }
187       }
188     }
189     tables.toList.map(t => CachedThreadPool.submit(() => clearOne(t))).foreach(_.get)
190   }

192   override def createQueryPlan(strategy: QueryStrategy): HBaseQueryPlan = {
193 
194     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
195 
196     val byteRanges = strategy.ranges
197     val hints = strategy.hints
198     val index = strategy.index
199 
200     // index api defines empty start/end for open-ended range
201     // index api defines start row inclusive, end row exclusive
202     // both these conventions match the conventions for hbase scan objects
203     val ranges = byteRanges.map {
204       case BoundedByteRange(start, stop) => new RowRange(start, true, stop, false)
205       case SingleRowByteRange(row)       => new RowRange(row, true, ByteArrays.rowFollowingRow(row), false)
206     }
207     val small = byteRanges.headOption.exists(_.isInstanceOf[SingleRowByteRange])
208 
209     val tables = index.getTablesForQuery(strategy.filter.filter).map(TableName.valueOf)
210 
211     // check for an empty query plan, if there are no tables or ranges to scan
212     def empty(reducer: Option[FeatureReducer]): Option[HBaseQueryPlan] =
213       if (tables.isEmpty || ranges.isEmpty) { Some(EmptyPlan(ds, strategy, reducer)) } else { None }
214 
215     if (!ds.config.remoteFilter) {
216       // everything is done client side
217       // note: we use the full filter here, since we can't use the z3 server-side filter
218       // for some attribute queries we wouldn't need the full filter...
219       val (colFamily, schema) = groups.group(index.sft, hints.getTransformDefinition, strategy.filter.filter)
220       // re-optimize the filter to account for the colFamily/schema being used
221       val ecql = {
222         val f = strategy.filter.filter
223         if (colFamily.eq(ColumnGroups.Default)) { f } else { f.map(FastFilterFactory.optimize(schema, _)) }
224       }
225       // note: we assume visibility filtering is still done server-side as it's part of core hbase
226       val processor = LocalProcessor(schema, hints, None)
227       empty(processor.reducer).getOrElse {
228         val scans = configureScans(tables, ranges, small, colFamily, Seq.empty, coprocessor = false)
229         LocalProcessorScanPlan(ds, strategy, ranges, scans, ecql, processor, hints.getProjection)
230       }
231     } else {
232       val ecql = strategy.ecql
233       val (colFamily, schema) = groups.group(index.sft, hints.getTransformDefinition, ecql)
234       val projection = hints.getProjection
235       val indexFilter = strategy.values.flatMap(IndexFilters(index, _))
236       lazy val returnSchema = hints.getTransformSchema.getOrElse(schema)
237       lazy val scans = {
238         val transform = hints.getTransform
239         val cqlFilter = if (ecql.isEmpty && transform.isEmpty && hints.getSampling.isEmpty) { Seq.empty } else {
240           Seq((CqlTransformFilter.Priority, CqlTransformFilter(schema, index, ecql, transform, hints)))
241         }
242         val filters = (cqlFilter ++ indexFilter).sortBy(_._1).map(_._2)
243         configureScans(tables, ranges, small, colFamily, filters, coprocessor = false)
244       }
245 
246       def coprocessorPlan(options: Map[String, String], toFeatures: ResultsToFeatures[Array[Byte]], reducer: Option[FeatureReducer]): CoprocessorPlan = {
247         val coprocessorOptions = Map(GeoMesaCoprocessor.YieldOpt -> String.valueOf(ds.config.coprocessors.yieldPartialResults))
248         val scans = configureScans(tables, ranges, small, colFamily, indexFilter.toSeq.map(_._2), coprocessor = true)
249         CoprocessorPlan(ds, strategy, ranges, scans, options ++ coprocessorOptions, toFeatures, reducer, hints.getMaxFeatures, projection)
250       }
251 
252       def semiLocalPlan(): LocalProcessorScanPlan = {
253         // note: transforms are handled in the cqlTransformFilter
254         val processor = LocalProcessor(returnSchema, QueryHints.Internal.clearTransforms(hints), None)
255         LocalProcessorScanPlan(ds, strategy, ranges, scans, None, processor, projection)
256       }
257 
258       if (hints.isDensityQuery) {
259         empty(None).getOrElse {
260           if (ds.config.coprocessors.enabled.density) {
261             val options = HBaseDensityAggregator.configure(schema, index, ecql, hints)
262             val results = new HBaseDensityResultsToFeatures()
263             coprocessorPlan(options, results, None)
264           } else {
265             semiLocalPlan()
266           }
267         }
268       } else if (hints.isArrowQuery) {
269         val config = HBaseArrowAggregator.configure(schema, index, ds.stats, strategy.filter.filter, ecql, hints)
270         val reducer = Some(config.reduce)
271         empty(reducer).getOrElse {
272           if (ds.config.coprocessors.enabled.arrow) {
273             val options = config.config
274             val results = new HBaseArrowResultsToFeatures()
275             coprocessorPlan(options, results, reducer)
276           } else {
277             semiLocalPlan()
278           }
279         }
280       } else if (hints.isStatsQuery) {
281         val reducer = Some(StatsScan.StatsReducer(returnSchema, hints))
282         empty(reducer).getOrElse {
283           if (ds.config.coprocessors.enabled.stats) {
284             val options = HBaseStatsAggregator.configure(schema, index, ecql, hints)
285             val results = new HBaseStatsResultsToFeatures()
286             coprocessorPlan(options, results, reducer)
287           } else {
288             semiLocalPlan()
289           }
290         }
291       } else if (hints.isBinQuery) {
292         empty(None).getOrElse {
293           if (ds.config.coprocessors.enabled.bin) {
294             val options = HBaseBinAggregator.configure(schema, index, ecql, hints)
295             val results = new HBaseBinResultsToFeatures()
296             coprocessorPlan(options, results, None)
297           } else {
298             semiLocalPlan()
299           }
300         }
301       } else {
302         empty(None).getOrElse {
303           val resultsToFeatures = new HBaseResultsToFeatures(index, returnSchema)
304           ScanPlan(ds, strategy, ranges, scans, resultsToFeatures, hints.getSortFields, hints.getMaxFeatures, projection)
305         }
306       }
307     }
308   }

327   /**
328    * Configure the hbase scan
329    *
330    * @param tables tables being scanned, used for region location information
331    * @param ranges ranges to scan, non-empty. needs to be mutable as we will sort it in place
332    * @param small whether 'small' ranges (i.e. gets)
333    * @param colFamily col family to scan
334    * @param filters scan filters
335    * @param coprocessor is this a coprocessor scan or not
336    * @return
337    */
338   private def configureScans(
339       tables: Seq[TableName],
340       ranges: Seq[RowRange],
341       small: Boolean,
342       colFamily: Array[Byte],
343       filters: Seq[HFilter],
344       coprocessor: Boolean): Seq[TableScan] = {
345     val cacheBlocks = HBaseSystemProperties.ScannerBlockCaching.toBoolean.get // has a default value so .get is safe
346     val cacheSize = HBaseSystemProperties.ScannerCaching.toInt
347 
348     logger.debug(s"HBase client scanner: block caching: $cacheBlocks, caching: $cacheSize")
349 
350     if (small && !coprocessor) {
351       val filter = filters match {
352         case Nil    => None
353         case Seq(f) => Some(f)
354         case f      => Some(new FilterList(f: _*))
355       }
356       // note: we have to copy the ranges for each table scan
357       tables.map { table =>
358         val scans = ranges.map { r =>
359           val scan = new Scan().withStartRow(r.getStartRow).withStopRow(r.getStopRow)
360           scan.addFamily(colFamily).setCacheBlocks(cacheBlocks).setSmall(true)
361           filter.foreach(scan.setFilter)
362           cacheSize.foreach(scan.setCaching)
363           ds.applySecurity(scan)
364           scan
365         }
366         TableScan(table, scans)
367       }
368     } else {
369       // split and group ranges by region server
370       // note: we have to copy the ranges for each table scan anyway
371       val rangesPerTable: Seq[(TableName, collection.Map[String, java.util.List[RowRange]])] =
372         tables.map(t => t -> groupRangesByRegion(t, ranges))
373 
374       def createGroup(group: java.util.List[RowRange]): Scan = {
375         val scan = new Scan().withStartRow(group.get(0).getStartRow).withStopRow(group.get(group.size() - 1).getStopRow)
376         val mrrf = if (group.size() < 2) { filters } else {
377           // TODO GEOMESA-1806
378           // currently, the MultiRowRangeFilter constructor will call sortAndMerge a second time
379           // this is unnecessary as we have already sorted and merged
380           // note: mrrf first priority
381           filters.+:(new MultiRowRangeFilter(group))
382         }
383         scan.setFilter(if (mrrf.lengthCompare(1) > 0) { new FilterList(mrrf: _*) } else { mrrf.headOption.orNull })
384         scan.addFamily(colFamily).setCacheBlocks(cacheBlocks)
385         cacheSize.foreach(scan.setCaching)
386 
387         // apply visibilities
388         ds.applySecurity(scan)
389 
390         scan
391       }
392 
393       rangesPerTable.map { case (table, rangesPerRegion) =>
394         val maxRangesPerGroup = {
395           def calcMax(maxPerGroup: Int, threads: Int): Int = {
396             val totalRanges = rangesPerRegion.values.map(_.size).sum
397             math.min(maxPerGroup, math.max(1, math.ceil(totalRanges.toDouble / threads).toInt))
398           }
399           if (coprocessor) {
400             calcMax(ds.config.coprocessors.maxRangesPerExtendedScan, ds.config.coprocessors.threads)
401           } else {
402             calcMax(ds.config.queries.maxRangesPerExtendedScan, ds.config.queries.threads)
403           }
404         }
405 
406         val groupedScans = Seq.newBuilder[Scan]
407 
408         rangesPerRegion.foreach { case (_, list) =>
409           // our ranges are non-overlapping, so just sort them but don't bother merging them
410           Collections.sort(list)
411 
412           var i = 0
413           while (i < list.size()) {
414             val groupSize = math.min(maxRangesPerGroup, list.size() - i)
415             groupedScans += createGroup(list.subList(i, i + groupSize))
416             i += groupSize
417           }
418         }
419 
420         // shuffle the ranges, otherwise our threads will tend to all hit the same region server at once
421         TableScan(table, Random.shuffle(groupedScans.result))
422       }
423     }
424   }

443   /**
444    * Group the range based on the region server hosting it. Splits ranges as needed if they span
445    * more than one region
446    *
447    * @param locator region locator
448    * @param range range to group
449    * @param result collected results
450    */
451   @scala.annotation.tailrec
452   private def groupRange(
453       locator: RegionLocator,
454       range: RowRange,
455       result: scala.collection.mutable.Map[String, java.util.List[RowRange]]): Unit = {
456     var encodedName: String = null
457     var split: Array[Byte] = null
458     try {
459       val regionInfo = locator.getRegionLocation(range.getStartRow).getRegionInfo
460       encodedName = regionInfo.getEncodedName
461       val regionEndKey = regionInfo.getEndKey // note: this is exclusive
462       if (regionEndKey.nonEmpty &&
463           (range.getStopRow.isEmpty || ByteArrays.ByteOrdering.compare(regionEndKey, range.getStopRow) <= 0)) {
464         if (ByteArrays.ByteOrdering.compare(range.getStartRow, regionEndKey) < 0) {
465           split = regionEndKey
466         } else {
467           logger.warn(s"HBase region location does not correspond to requested range:\n" +
468               s"  requested row: ${ByteArrays.toHex(range.getStartRow)}\n" +
469               s"  region: $encodedName ${ByteArrays.toHex(regionInfo.getStartKey)} :: ${ByteArrays.toHex(regionEndKey)}")
470         }
471       }
472     } catch {
473       case NonFatal(e) => logger.warn(s"Error checking range location for '$range''", e)
474     }
475     val buffer = result.getOrElseUpdate(encodedName, new java.util.ArrayList())
476     if (split == null) {
477       buffer.add(range)
478     } else {
479       // split the range based on the current region
480       buffer.add(new RowRange(range.getStartRow, true, split, false))
481       groupRange(locator, new RowRange(split, true, range.getStopRow, false), result)
482     }
483   }
