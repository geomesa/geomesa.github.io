87   override def createTable(
88       index: GeoMesaFeatureIndex[_, _],
89       partition: Option[String],
90       splits: => Seq[Array[Byte]]): Unit = {
91     // write table name to metadata
92     val name = TableName.valueOf(index.configureTableName(partition, tableNameLimit))
93 
94     WithClose(ds.connection.getAdmin) { admin =>
95       if (!admin.tableExists(name)) {
96         logger.debug(s"Creating table $name")
97 
98         val conf = admin.getConfiguration
99 
100         val compression = index.sft.getCompression.map { alg =>
101           logger.debug(s"Setting compression '$alg' on table $name for feature ${index.sft.getTypeName}")
102           // note: all compression types in HBase are case-sensitive and lower-cased
103           Compression.getCompressionAlgorithmByName(alg.toLowerCase(Locale.US))
104         }
105 
106         val cols = groups.apply(index.sft).map(_._1)
107         val bloom = Some(BloomType.NONE)
108         val encoding = if (index.name == IdIndex.name) { None } else { Some(DataBlockEncoding.FAST_DIFF) }
109 
110         val coprocessor = if (!ds.config.remoteFilter) { None } else {
111           // if the coprocessors are installed site-wide don't register them in the table descriptor.
112           // this key is CoprocessorHost.USER_REGION_COPROCESSOR_CONF_KEY - but don't want to pull in
113           // a dependency on hbase-server just for this constant
114           val installed = Option(conf.get("hbase.coprocessor.user.region.classes"))
115           val names = installed.map(_.split(":").toSet).getOrElse(Set.empty[String])
116           if (names.contains(CoprocessorClass)) { None } else {
117             // noinspection ScalaDeprecation
118             def urlFromSysProp: Option[Path] = CoprocessorUrl.option.orElse(CoprocessorPath.option).map(new Path(_))
119             val coprocessorUrl = ds.config.coprocessors.url.orElse(urlFromSysProp).orElse(dynamicJarPath)
120             logger.debug(s"Using coprocessor path ${coprocessorUrl.orNull}")
121             // TODO: Warn if the path given is different from paths registered in other coprocessors
122             // if so, other tables would need updating
123             Some(CoprocessorClass -> coprocessorUrl)
124           }
125         }
126 
127         try {
128           HBaseVersions.createTableAsync(admin, name, cols, bloom, compression, encoding, None, coprocessor, splits)
129         } catch {
130           case _: org.apache.hadoop.hbase.TableExistsException => // ignore, another thread created it for us
131         }
132       }
133 
134       waitForTable(admin, name)
135     }
136   }

138   override def renameTable(from: String, to: String): Unit = {
139     WithClose(ds.connection.getAdmin) { admin =>
140       val existing = TableName.valueOf(from)
141       val renamed = TableName.valueOf(to)
142       if (admin.tableExists(existing)) {
143         // renaming in hbase requires creating a snapshot and using that to create the new table
144         val snapshot = StringSerialization.alphaNumericSafeString(UUID.randomUUID().toString)
145         admin.disableTable(existing)
146         admin.snapshot(snapshot, existing)
147         admin.cloneSnapshot(snapshot, renamed)
148         admin.deleteSnapshot(snapshot)
149         admin.deleteTable(existing)
150         waitForTable(admin, renamed)
151       }
152     }
153   }

155   override def deleteTables(tables: Seq[String]): Unit = {
156     WithClose(ds.connection.getAdmin) { admin =>
157       def deleteOne(name: String): Unit = {
158         val table = TableName.valueOf(name)
159         if (admin.tableExists(table)) {
160           HBaseVersions.disableTableAsync(admin, table)
161           val timeout = TableAvailabilityTimeout.toUnboundedDuration.filter(_.isFinite)
162           logger.debug(s"Waiting for table '$table' to be disabled with " +
163               s"${timeout.map(t => s"a timeout of $t").getOrElse("no timeout")}")
164           val stop = timeout.map(t => System.currentTimeMillis() + t.toMillis)
165           while (!admin.isTableDisabled(table) && stop.forall(_ > System.currentTimeMillis())) {
166             Thread.sleep(1000)
167           }
168           // no async operation, but timeout can be controlled through hbase-site.xml "hbase.client.sync.wait.timeout.msec"
169           admin.deleteTable(table)
170         }
171       }
172       tables.toList.map(t => CachedThreadPool.submit(() => deleteOne(t))).foreach(_.get)
173     }
174   }

176   override def clearTables(tables: Seq[String], prefix: Option[Array[Byte]]): Unit = {
177     def clearOne(name: String): Unit = {
178       val tableName = TableName.valueOf(name)
179       WithClose(ds.connection.getTable(tableName)) { table =>
180         val scan = new Scan().setFilter(new KeyOnlyFilter)
181         prefix.foreach(scan.setRowPrefixFilter)
182         ds.applySecurity(scan)
183         val mutateParams = new BufferedMutatorParams(tableName)
184         WithClose(table.getScanner(scan), ds.connection.getBufferedMutator(mutateParams)) { case (scanner, mutator) =>
185           scanner.iterator.asScala.grouped(10000).foreach { result =>
186             // TODO GEOMESA-2546 set delete visibilities
187             val deletes = result.map(r => new Delete(r.getRow))
188             mutator.mutate(deletes.asJava)
189           }
190         }
191       }
192     }
193     tables.toList.map(t => CachedThreadPool.submit(() => clearOne(t))).foreach(_.get)
194   }

196   override def createQueryPlan(strategy: QueryStrategy): HBaseQueryPlan = {
197 
198     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
199 
200     val byteRanges = strategy.ranges
201     val ecql = strategy.ecql
202     val hints = strategy.hints
203     val index = strategy.index
204 
205     // index api defines empty start/end for open-ended range
206     // index api defines start row inclusive, end row exclusive
207     // both these conventions match the conventions for hbase scan objects
208     val ranges = byteRanges.map {
209       case BoundedByteRange(start, stop) => new RowRange(start, true, stop, false)
210       case SingleRowByteRange(row)       => new RowRange(row, true, ByteArrays.rowFollowingRow(row), false)
211     }
212     val small = byteRanges.headOption.exists(_.isInstanceOf[SingleRowByteRange])
213 
214     val tables = index.getTablesForQuery(strategy.filter.filter).map(TableName.valueOf)
215     val (colFamily, schema) = groups.group(index.sft, hints.getTransformDefinition, ecql)
216 
217     val transform: Option[(String, SimpleFeatureType)] = hints.getTransform
218 
219     // check for an empty query plan, if there are no tables or ranges to scan
220     def empty(reducer: Option[FeatureReducer]): Option[HBaseQueryPlan] =
221       if (tables.isEmpty || ranges.isEmpty) { Some(EmptyPlan(strategy, reducer)) } else { None }
222 
223     if (!ds.config.remoteFilter) {
224       // everything is done client side
225       // note: we assume visibility filtering is still done server-side as it's part of core hbase
226       // note: we use the full filter here, since we can't use the z3 server-side filter
227       // for some attribute queries we wouldn't need the full filter...
228       val reducer = Some(new LocalTransformReducer(schema, strategy.filter.filter, None, transform, hints))
229       empty(reducer).getOrElse {
230         val scans = configureScans(tables, ranges, small, colFamily, Seq.empty, coprocessor = false)
231         val resultsToFeatures = new HBaseResultsToFeatures(index, schema)
232         val sort = hints.getSortFields
233         val max = hints.getMaxFeatures
234         val project = hints.getProjection
235         ScanPlan(strategy, ranges, scans, resultsToFeatures, reducer, sort, max, project)
236       }
237     } else {
238       // TODO pull this out to be SPI loaded so that new indices can be added seamlessly
239       val indexFilter = strategy.index match {
240         case _: Z3Index =>
241           strategy.values.map { case v: Z3IndexValues =>
242             (Z3HBaseFilter.Priority, Z3HBaseFilter(Z3Filter(v), index.keySpace.sharding.length))
243           }
244 
245         case _: Z2Index =>
246           strategy.values.map { case v: Z2IndexValues =>
247             (Z2HBaseFilter.Priority, Z2HBaseFilter(Z2Filter(v), index.keySpace.sharding.length))
248           }
249 
250         case _: S2Index =>
251           strategy.values.map { case v: S2IndexValues =>
252             (S2HBaseFilter.Priority, S2HBaseFilter(S2Filter(v), index.keySpace.sharding.length))
253           }
254 
255         case _: S3Index =>
256           strategy.values.map { case v: S3IndexValues =>
257             (S3HBaseFilter.Priority, S3HBaseFilter(S3Filter(v), index.keySpace.sharding.length))
258           }
259         // TODO GEOMESA-1807 deal with non-points in a pushdown XZ filter
260 
261         case _ => None
262       }
263 
264       val max = hints.getMaxFeatures
265       val projection = hints.getProjection
266       lazy val returnSchema = transform.map(_._2).getOrElse(schema)
267       lazy val filters = {
268         val cqlFilter = if (ecql.isEmpty && transform.isEmpty && hints.getSampling.isEmpty) { Seq.empty } else {
269           Seq((CqlTransformFilter.Priority, CqlTransformFilter(schema, strategy.index, ecql, transform, hints)))
270         }
271         (cqlFilter ++ indexFilter).sortBy(_._1).map(_._2)
272       }
273       lazy val coprocessorOptions =
274         Map(GeoMesaCoprocessor.YieldOpt -> String.valueOf(ds.config.coprocessors.yieldPartialResults))
275       lazy val scans = configureScans(tables, ranges, small, colFamily, filters, coprocessor = false)
276       lazy val coprocessorScans =
277         configureScans(tables, ranges, small, colFamily, indexFilter.toSeq.map(_._2), coprocessor = true)
278       lazy val resultsToFeatures = new HBaseResultsToFeatures(index, returnSchema)
279       lazy val localReducer = Some(new LocalTransformReducer(returnSchema, None, None, None, hints))
280 
281       if (hints.isDensityQuery) {
282         empty(None).getOrElse {
283           if (ds.config.coprocessors.enabled.density) {
284             val options = HBaseDensityAggregator.configure(schema, index, ecql, hints) ++ coprocessorOptions
285             val results = new HBaseDensityResultsToFeatures()
286             CoprocessorPlan(strategy, ranges, coprocessorScans, options, results, None, max, projection)
287           } else {
288             if (hints.isSkipReduce) {
289               // override the return sft to reflect what we're actually returning,
290               // since the density sft is only created in the local reduce step
291               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
292             }
293             ScanPlan(strategy, ranges, scans, resultsToFeatures, localReducer, None, max, projection)
294           }
295         }
296       } else if (hints.isArrowQuery) {
297         val config = HBaseArrowAggregator.configure(schema, index, ds.stats, strategy.filter.filter, ecql, hints)
298         val reducer = Some(config.reduce)
299         empty(reducer).getOrElse {
300           if (ds.config.coprocessors.enabled.arrow) {
301             val options = config.config ++ coprocessorOptions
302             val results = new HBaseArrowResultsToFeatures()
303             CoprocessorPlan(strategy, ranges, coprocessorScans, options, results, reducer, max, projection)
304           } else {
305             if (hints.isSkipReduce) {
306               // override the return sft to reflect what we're actually returning,
307               // since the arrow sft is only created in the local reduce step
308               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
309             }
310             ScanPlan(strategy, ranges, scans, resultsToFeatures, localReducer, None, max, projection)
311           }
312         }
313       } else if (hints.isStatsQuery) {
314         val reducer = Some(StatsScan.StatsReducer(returnSchema, hints))
315         empty(reducer).getOrElse {
316           if (ds.config.coprocessors.enabled.stats) {
317             val options = HBaseStatsAggregator.configure(schema, index, ecql, hints) ++ coprocessorOptions
318             val results = new HBaseStatsResultsToFeatures()
319             CoprocessorPlan(strategy, ranges, coprocessorScans, options, results, reducer, max, projection)
320           } else {
321             if (hints.isSkipReduce) {
322               // override the return sft to reflect what we're actually returning,
323               // since the stats sft is only created in the local reduce step
324               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
325             }
326             ScanPlan(strategy, ranges, scans, resultsToFeatures, localReducer, None, max, projection)
327           }
328         }
329       } else if (hints.isBinQuery) {
330         empty(None).getOrElse {
331           if (ds.config.coprocessors.enabled.bin) {
332             val options = HBaseBinAggregator.configure(schema, index, ecql, hints) ++ coprocessorOptions
333             val results = new HBaseBinResultsToFeatures()
334             CoprocessorPlan(strategy, ranges, coprocessorScans, options , results, None, max, projection)
335           } else {
336             if (hints.isSkipReduce) {
337               // override the return sft to reflect what we're actually returning,
338               // since the bin sft is only created in the local reduce step
339               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
340             }
341             ScanPlan(strategy, ranges, scans, resultsToFeatures, localReducer, None, max, projection)
342           }
343         }
344       } else {
345         empty(None).getOrElse {
346           ScanPlan(strategy, ranges, scans, resultsToFeatures, None, hints.getSortFields, max, projection)
347         }
348       }
349     }
350   }

369   /**
370    * Configure the hbase scan
371    *
372    * @param tables tables being scanned, used for region location information
373    * @param ranges ranges to scan, non-empty. needs to be mutable as we will sort it in place
374    * @param small whether 'small' ranges (i.e. gets)
375    * @param colFamily col family to scan
376    * @param filters scan filters
377    * @param coprocessor is this a coprocessor scan or not
378    * @return
379    */
380   protected def configureScans(
381       tables: Seq[TableName],
382       ranges: Seq[RowRange],
383       small: Boolean,
384       colFamily: Array[Byte],
385       filters: Seq[HFilter],
386       coprocessor: Boolean): Seq[TableScan] = {
387     val cacheBlocks = HBaseSystemProperties.ScannerBlockCaching.toBoolean.get // has a default value so .get is safe
388     val cacheSize = HBaseSystemProperties.ScannerCaching.toInt
389 
390     logger.debug(s"HBase client scanner: block caching: $cacheBlocks, caching: $cacheSize")
391 
392     if (small && !coprocessor) {
393       val filter = filters match {
394         case Nil    => None
395         case Seq(f) => Some(f)
396         case f      => Some(new FilterList(f: _*))
397       }
398       // note: we have to copy the ranges for each table scan
399       tables.map { table =>
400         val scans = ranges.map { r =>
401           val scan = new Scan(r.getStartRow, r.getStopRow)
402           scan.addFamily(colFamily).setCacheBlocks(cacheBlocks).setSmall(true)
403           filter.foreach(scan.setFilter)
404           cacheSize.foreach(scan.setCaching)
405           ds.applySecurity(scan)
406           scan
407         }
408         TableScan(table, scans)
409       }
410     } else {
411       // split and group ranges by region server
412       // note: we have to copy the ranges for each table scan anyway
413       val rangesPerTable: Seq[(TableName, collection.Map[String, java.util.List[RowRange]])] =
414         tables.map(t => t -> groupRangesByRegion(t, ranges))
415 
416       def createGroup(group: java.util.List[RowRange]): Scan = {
417         val scan = new Scan(group.get(0).getStartRow, group.get(group.size() - 1).getStopRow)
418         val mrrf = if (group.size() < 2) { filters } else {
419           // TODO GEOMESA-1806
420           // currently, the MultiRowRangeFilter constructor will call sortAndMerge a second time
421           // this is unnecessary as we have already sorted and merged
422           // note: mrrf first priority
423           filters.+:(new MultiRowRangeFilter(group))
424         }
425         scan.setFilter(if (mrrf.lengthCompare(1) > 0) { new FilterList(mrrf: _*) } else { mrrf.headOption.orNull })
426         scan.addFamily(colFamily).setCacheBlocks(cacheBlocks)
427         cacheSize.foreach(scan.setCaching)
428 
429         // apply visibilities
430         ds.applySecurity(scan)
431 
432         scan
433       }
434 
435       rangesPerTable.map { case (table, rangesPerRegion) =>
436         val maxRangesPerGroup = {
437           def calcMax(maxPerGroup: Int, threads: Int): Int = {
438             val totalRanges = rangesPerRegion.values.map(_.size).sum
439             math.min(maxPerGroup, math.max(1, math.ceil(totalRanges.toDouble / threads).toInt))
440           }
441           if (coprocessor) {
442             calcMax(ds.config.coprocessors.maxRangesPerExtendedScan, ds.config.coprocessors.threads)
443           } else {
444             calcMax(ds.config.queries.maxRangesPerExtendedScan, ds.config.queries.threads)
445           }
446         }
447 
448         val groupedScans = Seq.newBuilder[Scan]
449 
450         rangesPerRegion.foreach { case (_, list) =>
451           // our ranges are non-overlapping, so just sort them but don't bother merging them
452           Collections.sort(list)
453 
454           var i = 0
455           while (i < list.size()) {
456             val groupSize = math.min(maxRangesPerGroup, list.size() - i)
457             groupedScans += createGroup(list.subList(i, i + groupSize))
458             i += groupSize
459           }
460         }
461 
462         // shuffle the ranges, otherwise our threads will tend to all hit the same region server at once
463         TableScan(table, Random.shuffle(groupedScans.result))
464       }
465     }
466   }

485   /**
486    * Group the range based on the region server hosting it. Splits ranges as needed if they span
487    * more than one region
488    *
489    * @param locator region locator
490    * @param range range to group
491    * @param result collected results
492    */
493   @scala.annotation.tailrec
494   private def groupRange(
495       locator: RegionLocator,
496       range: RowRange,
497       result: scala.collection.mutable.Map[String, java.util.List[RowRange]]): Unit = {
498     var encodedName: String = null
499     var split: Array[Byte] = null
500     try {
501       val regionInfo = locator.getRegionLocation(range.getStartRow).getRegionInfo
502       encodedName = regionInfo.getEncodedName
503       val regionEndKey = regionInfo.getEndKey // note: this is exclusive
504       if (regionEndKey.nonEmpty &&
505           (range.getStopRow.isEmpty || ByteArrays.ByteOrdering.compare(regionEndKey, range.getStopRow) <= 0)) {
506         if (ByteArrays.ByteOrdering.compare(range.getStartRow, regionEndKey) < 0) {
507           split = regionEndKey
508         } else {
509           logger.warn(s"HBase region location does not correspond to requested range:\n" +
510               s"  requested row: ${ByteArrays.toHex(range.getStartRow)}\n" +
511               s"  region: $encodedName ${ByteArrays.toHex(regionInfo.getStartKey)} :: ${ByteArrays.toHex(regionEndKey)}")
512         }
513       }
514     } catch {
515       case NonFatal(e) => logger.warn(s"Error checking range location for '$range''", e)
516     }
517     val buffer = result.getOrElseUpdate(encodedName, new java.util.ArrayList())
518     if (split == null) {
519       buffer.add(range)
