70   override def createTable(
71       index: GeoMesaFeatureIndex[_, _],
72       partition: Option[String],
73       splits: => Seq[Array[Byte]]): Unit = {
74     // write table name to metadata
75     val name = TableName.valueOf(index.configureTableName(partition, tableNameLimit))
76 
77     WithClose(ds.connection.getAdmin) { admin =>
78       if (!admin.tableExists(name)) {
79         logger.debug(s"Creating table $name")
80 
81         val conf = admin.getConfiguration
82 
83         val compression = index.sft.getCompression.map { alg =>
84           logger.debug(s"Setting compression '$alg' on table $name for feature ${index.sft.getTypeName}")
85           // note: all compression types in HBase are case-sensitive and lower-cased
86           Compression.getCompressionAlgorithmByName(alg.toLowerCase(Locale.US))
87         }
88 
89         val cols = groups.apply(index.sft).map(_._1)
90         val bloom = Some(BloomType.NONE)
91         val encoding = if (index.name == IdIndex.name) { None } else { Some(DataBlockEncoding.FAST_DIFF) }
92 
93         // noinspection ScalaDeprecation
94         val coprocessor = if (!ds.config.remoteFilter) { None } else {
95           def urlFromSysProp: Option[Path] = CoprocessorUrl.option.orElse(CoprocessorPath.option).map(new Path(_))
96           lazy val coprocessorUrl = ds.config.coprocessors.url.orElse(urlFromSysProp).orElse {
97             try {
98               // the jar should be under hbase.dynamic.jars.dir to enable filters, so look there
99               val dir = new Path(conf.get("hbase.dynamic.jars.dir"))
100               WithClose(dir.getFileSystem(conf)) { fs =>
101                 if (!fs.isDirectory(dir)) { None } else {
102                   fs.listStatus(dir).collectFirst {
103                     case s if distributedJarNamePattern.matcher(s.getPath.getName).matches() => s.getPath
104                   }
105                 }
106               }
107             } catch {
108               case NonFatal(e) => logger.warn("Error checking dynamic jar path:", e); None
109             }
110           }
111           // if the coprocessors are installed site-wide don't register them in the table descriptor.
112           // this key is CoprocessorHost.USER_REGION_COPROCESSOR_CONF_KEY - but don't want to pull in
113           // a dependency on hbase-server just for this constant
114           val installed = Option(conf.get("hbase.coprocessor.user.region.classes"))
115           val names = installed.map(_.split(":").toSet).getOrElse(Set.empty[String])
116           if (names.contains(CoprocessorClass)) { None } else {
117             logger.debug(s"Using coprocessor path ${coprocessorUrl.orNull}")
118             // TODO: Warn if the path given is different from paths registered in other coprocessors
119             // if so, other tables would need updating
120             Some(CoprocessorClass -> coprocessorUrl)
121           }
122         }
123 
124         try {
125           HBaseVersions.createTableAsync(admin, name, cols, bloom, compression, encoding, None, coprocessor, splits)
126         } catch {
127           case _: org.apache.hadoop.hbase.TableExistsException => // ignore, another thread created it for us
128         }
129       }
130 
131       waitForTable(admin, name)
132     }
133   }

135   override def renameTable(from: String, to: String): Unit = {
136     WithClose(ds.connection.getAdmin) { admin =>
137       val existing = TableName.valueOf(from)
138       val renamed = TableName.valueOf(to)
139       if (admin.tableExists(existing)) {
140         // renaming in hbase requires creating a snapshot and using that to create the new table
141         val snapshot = StringSerialization.alphaNumericSafeString(UUID.randomUUID().toString)
142         admin.disableTable(existing)
143         admin.snapshot(snapshot, existing)
144         admin.cloneSnapshot(snapshot, renamed)
145         admin.deleteSnapshot(snapshot)
146         admin.deleteTable(existing)
147         waitForTable(admin, renamed)
148       }
149     }

152   override def deleteTables(tables: Seq[String]): Unit = {
153     WithClose(ds.connection.getAdmin) { admin =>
154       def deleteOne(name: String): Unit = {
155         val table = TableName.valueOf(name)
156         if (admin.tableExists(table)) {
157           HBaseVersions.disableTableAsync(admin, table)
158           val timeout = TableAvailabilityTimeout.toDuration.filter(_.isFinite)
159           logger.debug(s"Waiting for table '$table' to be disabled with " +
160               s"${timeout.map(t => s"a timeout of $t").getOrElse("no timeout")}")
161           val stop = timeout.map(t => System.currentTimeMillis() + t.toMillis)
162           while (!admin.isTableDisabled(table) && stop.forall(_ > System.currentTimeMillis())) {
163             Thread.sleep(1000)
164           }
165           // no async operation, but timeout can be controlled through hbase-site.xml "hbase.client.sync.wait.timeout.msec"
166           admin.deleteTable(table)
167         }
168       }
169       tables.toList.map(t => CachedThreadPool.submit(() => deleteOne(t))).foreach(_.get)
170     }
171   }

173   override def clearTables(tables: Seq[String], prefix: Option[Array[Byte]]): Unit = {
174     def clearOne(name: String): Unit = {
175       val tableName = TableName.valueOf(name)
176       WithClose(ds.connection.getTable(tableName)) { table =>
177         val scan = new Scan().setFilter(new KeyOnlyFilter)
178         prefix.foreach(scan.setRowPrefixFilter)
179         ds.applySecurity(scan)
180         val mutateParams = new BufferedMutatorParams(tableName)
181         WithClose(table.getScanner(scan), ds.connection.getBufferedMutator(mutateParams)) { case (scanner, mutator) =>
182           scanner.iterator.asScala.grouped(10000).foreach { result =>
183             // TODO GEOMESA-2546 set delete visibilities
184             val deletes = result.map(r => new Delete(r.getRow))
185             mutator.mutate(deletes.asJava)
186           }
187         }
188       }
189     }
190     tables.toList.map(t => CachedThreadPool.submit(() => clearOne(t))).foreach(_.get)
191   }

193   override def createQueryPlan(strategy: QueryStrategy): HBaseQueryPlan = {
194 
195     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
196 
197     val QueryStrategy(filter, byteRanges, _, _, ecql, hints, _) = strategy
198     val index = filter.index
199 
200     // index api defines empty start/end for open-ended range
201     // index api defines start row inclusive, end row exclusive
202     // both these conventions match the conventions for hbase scan objects
203     val ranges = byteRanges.map {
204       case BoundedByteRange(start, stop) => new RowRange(start, true, stop, false)
205       case SingleRowByteRange(row)       => new RowRange(row, true, ByteArrays.rowFollowingRow(row), false)
206     }
207     val small = byteRanges.headOption.exists(_.isInstanceOf[SingleRowByteRange])
208 
209     val tables = index.getTablesForQuery(filter.filter).map(TableName.valueOf)
210     val (colFamily, schema) = groups.group(index.sft, hints.getTransformDefinition, ecql)
211 
212     val transform: Option[(String, SimpleFeatureType)] = hints.getTransform
213 
214     // check for an empty query plan, if there are no tables or ranges to scan
215     def empty(reducer: Option[FeatureReducer]): Option[HBaseQueryPlan] =
216       if (tables.isEmpty || ranges.isEmpty) { Some(EmptyPlan(filter, reducer)) } else { None }
217 
218     if (!ds.config.remoteFilter) {
219       // everything is done client side
220       val arrowHook = Some(ArrowDictionaryHook(ds.stats, filter.filter))
221       // note: we assume visibility filtering is still done server-side as it's part of core hbase
222       // note: we use the full filter here, since we can't use the z3 server-side filter
223       // for some attribute queries we wouldn't need the full filter...
224       val reducer = Some(new LocalTransformReducer(schema, filter.filter, None, transform, hints, arrowHook))
225       empty(reducer).getOrElse {
226         val scans = configureScans(tables, ranges, small, colFamily, Seq.empty, coprocessor = false)
227         val resultsToFeatures = new HBaseResultsToFeatures(index, schema)
228         val sort = hints.getSortFields
229         val max = hints.getMaxFeatures
230         val project = hints.getProjection
231         ScanPlan(filter, ranges, scans, resultsToFeatures, reducer, sort, max, project)
232       }
233     } else {
234       // TODO pull this out to be SPI loaded so that new indices can be added seamlessly
235       val indexFilter = strategy.index match {
236         case _: Z3Index =>
237           strategy.values.map { case v: Z3IndexValues =>
238             (Z3HBaseFilter.Priority, Z3HBaseFilter(Z3Filter(v), index.keySpace.sharding.length))
239           }
240 
241         case _: Z2Index =>
242           strategy.values.map { case v: Z2IndexValues =>
243             (Z2HBaseFilter.Priority, Z2HBaseFilter(Z2Filter(v), index.keySpace.sharding.length))
244           }
245 
246         case _: S2Index =>
247           strategy.values.map { case v: S2IndexValues =>
248             (S2HBaseFilter.Priority, S2HBaseFilter(S2Filter(v), index.keySpace.sharding.length))
249           }
250 
251         case _: S3Index =>
252           strategy.values.map { case v: S3IndexValues =>
253             (S3HBaseFilter.Priority, S3HBaseFilter(S3Filter(v), index.keySpace.sharding.length))
254           }
255         // TODO GEOMESA-1807 deal with non-points in a pushdown XZ filter
256 
257         case _ => None
258       }
259 
260       val max = hints.getMaxFeatures
261       val projection = hints.getProjection
262       lazy val returnSchema = transform.map(_._2).getOrElse(schema)
263       lazy val filters = {
264         val cqlFilter = if (ecql.isEmpty && transform.isEmpty && hints.getSampling.isEmpty) { Seq.empty } else {
265           Seq((CqlTransformFilter.Priority, CqlTransformFilter(schema, strategy.index, ecql, transform, hints)))
266         }
267         (cqlFilter ++ indexFilter).sortBy(_._1).map(_._2)
268       }
269       lazy val coprocessorOptions =
270         Map(GeoMesaCoprocessor.YieldOpt -> String.valueOf(ds.config.coprocessors.yieldPartialResults))
271       lazy val scans = configureScans(tables, ranges, small, colFamily, filters, coprocessor = false)
272       lazy val coprocessorScans =
273         configureScans(tables, ranges, small, colFamily, indexFilter.toSeq.map(_._2), coprocessor = true)
274       lazy val resultsToFeatures = new HBaseResultsToFeatures(index, returnSchema)
275       lazy val localReducer = {
276         val arrowHook = Some(ArrowDictionaryHook(ds.stats, filter.filter))
277         Some(new LocalTransformReducer(returnSchema, None, None, None, hints, arrowHook))
278       }
279 
280       if (hints.isDensityQuery) {
281         empty(None).getOrElse {
282           if (ds.config.coprocessors.enabled.density) {
283             val options = HBaseDensityAggregator.configure(schema, index, ecql, hints) ++ coprocessorOptions
284             val results = new HBaseDensityResultsToFeatures()
285             CoprocessorPlan(filter, ranges, coprocessorScans, options, results, None, max, projection)
286           } else {
287             if (hints.isSkipReduce) {
288               // override the return sft to reflect what we're actually returning,
289               // since the density sft is only created in the local reduce step
290               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
291             }
292             ScanPlan(filter, ranges, scans, resultsToFeatures, localReducer, None, max, projection)
293           }
294         }
295       } else if (hints.isArrowQuery) {
296         val config = HBaseArrowAggregator.configure(schema, index, ds.stats, filter.filter, ecql, hints)
297         val reducer = Some(config.reduce)
298         empty(reducer).getOrElse {
299           if (ds.config.coprocessors.enabled.arrow) {
300             val options = config.config ++ coprocessorOptions
301             val results = new HBaseArrowResultsToFeatures()
302             CoprocessorPlan(filter, ranges, coprocessorScans, options, results, reducer, max, projection)
303           } else {
304             if (hints.isSkipReduce) {
305               // override the return sft to reflect what we're actually returning,
306               // since the arrow sft is only created in the local reduce step
307               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
308             }
309             ScanPlan(filter, ranges, scans, resultsToFeatures, localReducer, None, max, projection)
310           }
311         }
312       } else if (hints.isStatsQuery) {
313         val reducer = Some(StatsScan.StatsReducer(returnSchema, hints))
314         empty(reducer).getOrElse {
315           if (ds.config.coprocessors.enabled.stats) {
316             val options = HBaseStatsAggregator.configure(schema, index, ecql, hints) ++ coprocessorOptions
317             val results = new HBaseStatsResultsToFeatures()
318             CoprocessorPlan(filter, ranges, coprocessorScans, options, results, reducer, max, projection)
319           } else {
320             if (hints.isSkipReduce) {
321               // override the return sft to reflect what we're actually returning,
322               // since the stats sft is only created in the local reduce step
323               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
324             }
325             ScanPlan(filter, ranges, scans, resultsToFeatures, localReducer, None, max, projection)
326           }
327         }
328       } else if (hints.isBinQuery) {
329         empty(None).getOrElse {
330           if (ds.config.coprocessors.enabled.bin) {
331             val options = HBaseBinAggregator.configure(schema, index, ecql, hints) ++ coprocessorOptions
332             val results = new HBaseBinResultsToFeatures()
333             CoprocessorPlan(filter, ranges, coprocessorScans, options , results, None, max, projection)
334           } else {
335             if (hints.isSkipReduce) {
336               // override the return sft to reflect what we're actually returning,
337               // since the bin sft is only created in the local reduce step
338               hints.hints.put(QueryHints.Internal.RETURN_SFT, returnSchema)
339             }
340             ScanPlan(filter, ranges, scans, resultsToFeatures, localReducer, None, max, projection)
341           }
342         }
343       } else {
344         empty(None).getOrElse {
345           ScanPlan(filter, ranges, scans, resultsToFeatures, None, hints.getSortFields, max, projection)
346         }
347       }
348     }
349   }

365   /**
366    * Configure the hbase scan
367    *
368    * @param tables tables being scanned, used for region location information
369    * @param ranges ranges to scan, non-empty. needs to be mutable as we will sort it in place
370    * @param small whether 'small' ranges (i.e. gets)
371    * @param colFamily col family to scan
372    * @param filters scan filters
373    * @param coprocessor is this a coprocessor scan or not
374    * @return
375    */
376   protected def configureScans(
377       tables: Seq[TableName],
378       ranges: Seq[RowRange],
379       small: Boolean,
380       colFamily: Array[Byte],
381       filters: Seq[HFilter],
382       coprocessor: Boolean): Seq[TableScan] = {
383     val cacheBlocks = HBaseSystemProperties.ScannerBlockCaching.toBoolean.get // has a default value so .get is safe
384     val cacheSize = HBaseSystemProperties.ScannerCaching.toInt
385 
386     logger.debug(s"HBase client scanner: block caching: $cacheBlocks, caching: $cacheSize")
387 
388     if (small && !coprocessor) {
389       val filter = filters match {
390         case Nil    => None
391         case Seq(f) => Some(f)
392         case f      => Some(new FilterList(f: _*))
393       }
394       // note: we have to copy the ranges for each table scan
395       tables.map { table =>
396         val scans = ranges.map { r =>
397           val scan = new Scan(r.getStartRow, r.getStopRow)
398           scan.addFamily(colFamily).setCacheBlocks(cacheBlocks).setSmall(true)
399           filter.foreach(scan.setFilter)
400           cacheSize.foreach(scan.setCaching)
401           ds.applySecurity(scan)
402           scan
403         }
404         TableScan(table, scans)
405       }
406     } else {
407       // split and group ranges by region server
408       // note: we have to copy the ranges for each table scan anyway
409       val rangesPerTable: Seq[(TableName, collection.Map[String, java.util.List[RowRange]])] =
410         tables.map(t => t -> groupRangesByRegion(t, ranges))
411 
412       def createGroup(group: java.util.List[RowRange]): Scan = {
413         val scan = new Scan(group.get(0).getStartRow, group.get(group.size() - 1).getStopRow)
414         val mrrf = if (group.size() < 2) { filters } else {
415           // TODO GEOMESA-1806
416           // currently, the MultiRowRangeFilter constructor will call sortAndMerge a second time
417           // this is unnecessary as we have already sorted and merged
418           // note: mrrf first priority
419           filters.+:(new MultiRowRangeFilter(group))
420         }
421         scan.setFilter(if (mrrf.lengthCompare(1) > 0) { new FilterList(mrrf: _*) } else { mrrf.headOption.orNull })
422         scan.addFamily(colFamily).setCacheBlocks(cacheBlocks)
423         cacheSize.foreach(scan.setCaching)
424 
425         // apply visibilities
426         ds.applySecurity(scan)
427 
428         scan
429       }
430 
431       rangesPerTable.map { case (table, rangesPerRegion) =>
432         val maxRangesPerGroup = {
433           def calcMax(maxPerGroup: Int, threads: Int): Int = {
434             val totalRanges = rangesPerRegion.values.map(_.size).sum
435             math.min(maxPerGroup, math.max(1, math.ceil(totalRanges.toDouble / threads).toInt))
436           }
437           if (coprocessor) {
438             calcMax(ds.config.coprocessors.maxRangesPerExtendedScan, ds.config.coprocessors.threads)
439           } else {
440             calcMax(ds.config.queries.maxRangesPerExtendedScan, ds.config.queries.threads)
441           }
442         }
443 
444         val groupedScans = Seq.newBuilder[Scan]
445 
446         rangesPerRegion.foreach { case (_, list) =>
447           // our ranges are non-overlapping, so just sort them but don't bother merging them
448           Collections.sort(list)
449 
450           var i = 0
451           while (i < list.size()) {
452             val groupSize = math.min(maxRangesPerGroup, list.size() - i)
453             groupedScans += createGroup(list.subList(i, i + groupSize))
454             i += groupSize
455           }
456         }
457 
458         // shuffle the ranges, otherwise our threads will tend to all hit the same region server at once
459         TableScan(table, Random.shuffle(groupedScans.result))
460       }
461     }
462   }

481   /**
482    * Group the range based on the region server hosting it. Splits ranges as needed if they span
483    * more than one region
484    *
485    * @param locator region locator
486    * @param range range to group
487    * @param result collected results
488    */
489   @scala.annotation.tailrec
490   private def groupRange(
491       locator: RegionLocator,
492       range: RowRange,
493       result: scala.collection.mutable.Map[String, java.util.List[RowRange]]): Unit = {
494     var encodedName: String = null
495     var split: Array[Byte] = null
496     try {
497       val regionInfo = locator.getRegionLocation(range.getStartRow).getRegionInfo
498       encodedName = regionInfo.getEncodedName
499       val regionEndKey = regionInfo.getEndKey // note: this is exclusive
500       if (regionEndKey.nonEmpty &&
501           (range.getStopRow.isEmpty || ByteArrays.ByteOrdering.compare(regionEndKey, range.getStopRow) <= 0)) {
502         if (ByteArrays.ByteOrdering.compare(range.getStartRow, regionEndKey) < 0) {
503           split = regionEndKey
504         } else {
505           logger.warn(s"HBase region location does not correspond to requested range:\n" +
506               s"  requested row: ${ByteArrays.toHex(range.getStartRow)}\n" +
507               s"  region: $encodedName ${ByteArrays.toHex(regionInfo.getStartKey)} :: ${ByteArrays.toHex(regionEndKey)}")
508         }
509       }
510     } catch {
511       case NonFatal(e) => logger.warn(s"Error checking range location for '$range''", e)
512     }
513     val buffer = result.getOrElseUpdate(encodedName, new java.util.ArrayList())
514     if (split == null) {
515       buffer.add(range)
516     } else {
517       // split the range based on the current region
518       buffer.add(new RowRange(range.getStartRow, true, split, false))
519       groupRange(locator, new RowRange(split, true, range.getStopRow, false), result)
520     }
