48    * Executes the job
49    *
50    * @param opts opts

54   private def copy(opts: DistCpOptions, statusCallback: StatusCallback): JobResult = {

63     }
64   }

71   private def distCpOptions(sources: Either[Seq[Path], Path], dest: Path): DistCpOptions =
72     try { distCpOptions3(sources, dest) } catch { case _: ClassNotFoundException => distCpOptions2(sources, dest) }

74   // hadoop 3 API
75   private def distCpOptions3(sources: Either[Seq[Path], Path], dest: Path): DistCpOptions = {
76     val builder = sources match {
77       case Right(file) => new DistCpOptions.Builder(file, dest)
78       case Left(dirs)  => new DistCpOptions.Builder(dirs.asJava, dest)
79     }
80     builder.withAppend(false).withOverwrite(true).withBlocking(false).withCopyStrategy("dynamic").build()
81   }
82 
83   // hadoop 2 API
84   private def distCpOptions2(sources: Either[Seq[Path], Path], dest: Path): DistCpOptions = {
85     val opts = sources match {
86       case Right(file) =>
87         classOf[DistCpOptions].getConstructor(classOf[Path], classOf[Path]).newInstance(file, dest)
88       case Left(dirs) =>
89         classOf[DistCpOptions].getConstructor(classOf[java.util.List[Path]], classOf[Path]).newInstance(dirs.asJava, dest)
90     }
91     classOf[DistCpOptions].getMethod("setAppend", classOf[Boolean]).invoke(opts, java.lang.Boolean.FALSE)
92     classOf[DistCpOptions].getMethod("setOverwrite", classOf[Boolean]).invoke(opts, java.lang.Boolean.TRUE)
93     classOf[DistCpOptions].getMethod("setCopyStrategy", classOf[String]).invoke(opts, "dynamic")
94     opts
95   }
