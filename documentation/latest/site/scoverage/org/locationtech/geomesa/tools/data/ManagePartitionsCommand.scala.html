53     * List existing partitions
54     */
55   trait ListPartitionsCommand[DS <: GeoMesaDataStore[DS]] extends PartitionsCommand[DS] {
56 
57     override val name = "list"
58 
59     override protected def execute(ds: DS, sft: SimpleFeatureType, partition: TablePartition): Unit = {
60       Command.user.info(s"Partitions for schema ${params.featureName}:")
61       val partitions = ArrayBuffer.empty[String]
62       def loadSingle(index: GeoMesaFeatureIndex[_, _]): Unit = {
63         val p = index.getPartitions
64         partitions.synchronized(partitions ++= p)
65       }
66       ds.manager.indices(sft).toList.map(i => CachedThreadPool.submit(() => loadSingle(i))).foreach(_.get)
67       partitions.distinct.sorted.foreach(p => Command.output.info(p))
68     }
69   }

88   /**
89     * Adopt an existing set of index tables as a new partitions
90     */
91   trait AdoptPartitionCommand[DS <: GeoMesaDataStore[DS]] extends PartitionsCommand[DS] {
92 
93     override val name = "adopt"
94     override def params: AdoptPartitionParam
95 
96     override protected def execute(ds: DS, sft: SimpleFeatureType, partition: TablePartition): Unit = {
97       val time = Option(partition).collect { case p: TimePartition => p }.getOrElse {
98         throw new UnsupportedOperationException(s"Unsupported partition implementation: ${partition.getClass.getName}")
99       }
100       val (start, end) = {
101         val (s, e) = new IntervalConverter("value").convert(params.value)
102         (ZonedDateTime.ofInstant(s.toInstant, ZoneOffset.UTC), ZonedDateTime.ofInstant(e.toInstant, ZoneOffset.UTC))
103       }
104       val tables = params.tables.asScala
105 
106       Command.user.info(s"Adopting tables ${tables.mkString(", ")} as partition ${params.partition}")
107 
108       val indices = ds.manager.indices(sft)
109       if (indices.lengthCompare(tables.length) != 0) {
110         throw new IllegalArgumentException(s"Expected an index table for each index: ${indices.map(_.name).mkString(", ")}")
111       }
112       // match tables first, to fail fast if there is no match
113       val indexNames = indices.map { index =>
114         val name = StringSerialization.alphaNumericSafeString(index.name)
115         val attributes = index.attributes.map(StringSerialization.alphaNumericSafeString).mkString("_")
116         val version = s"v${index.version}"
117         (index, name, attributes, version)
118       }
119       val tableToIndexName = indexNames
120         .toList
121         .sortWith(_._3.length > _._3.length) // Sorted so we get best match first
122         .foldLeft(Map[String,(GeoMesaFeatureIndex[_,_],String,String,String)]()){
123           (mappedTables,indexName) =>
124           val fullMatches = tables.filter(table =>
125             !mappedTables.contains(table) // Don't match again
126               && table.contains(indexName._2) // Correct type name as in z3, attr, id...
127               && table.contains(indexName._3) // Correct attributes
128               && table.contains(indexName._4) // Correct version
129           )
130           if (fullMatches.lengthCompare(1) >= 0) {
131             mappedTables + (fullMatches.head -> indexName)
132           } else {
133            throw new IllegalArgumentException(s"Could not match a table to an index of '${indexName._3}'")
134           }
135         }
136       tableToIndexName.foreach { case (table, indexName) =>
137         ds.metadata.insert(sft.getTypeName, indexName._1.tableNameKey(Some(params.partition)), table)
138       }
139       time.register(params.partition, start, end)
140 
141       Command.user.info("Done")
142     }
143   }

148   trait DeletePartitionsCommand[DS <: GeoMesaDataStore[DS]] extends ModifyPartitionsCommand[DS] {
149 
150     override val name = "delete"
151     override def params: PartitionsParam with OptionalForceParam
152 
153     override protected def execute(ds: DS, sft: SimpleFeatureType, partition: TablePartition): Unit = {
154       lazy val prompt = s"Deleting partition(s) '${params.partitions.asScala.mkString("', '")}'. Continue (y/n)? "
155       if (params.force || Prompt.confirm(prompt)) {
156         super.execute(ds, sft, partition)
157       } else {
158         Command.user.info("Delete cancelled")
159       }
160     }
161 
162     override protected def modify(ds: DS, sft: SimpleFeatureType, partition: TablePartition, p: String): Unit = {
163       Command.user.info(s"Deleting partition '$p'")
164       def deleteSingle(index: GeoMesaFeatureIndex[_, _]): Unit =
165         ds.adapter.deleteTables(index.deleteTableNames(Some(p)))
166       ds.manager.indices(sft).toList.map(i => CachedThreadPool.submit(() => deleteSingle(i))).foreach(_.get)
167     }
168   }

212 
213     override def params: RequiredTypeNameParam
214 
215     override def execute(): Unit = {
216       withDataStore { ds =>
217         val sft = ds.getSchema(params.featureName)
218         if (sft == null) {
219           throw new ParameterException(s"Schema '${params.featureName}' does not exist in the data store. " +
220               s"Available types: ${ds.getTypeNames.mkString(", ")}")
221         }
222         TablePartition(ds, sft) match {
223           case None => Command.user.error(s"Schema is not partitioned")
224           case Some(p) => execute(ds, sft, p)
225         }
226       }
227     }
