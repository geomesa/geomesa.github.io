55     */
56   trait ListPartitionsCommand[DS <: GeoMesaDataStore[DS]] extends PartitionsCommand[DS] {
57 
58     override val name = "list"
59 
60     override protected def execute(ds: DS, sft: SimpleFeatureType, partition: TablePartition): Unit = {
61       Command.user.info(s"Partitions for schema ${params.featureName}:")
62       val partitions = ArrayBuffer.empty[String]
63       def loadSingle(index: GeoMesaFeatureIndex[_, _]): Unit = {
64         val p = index.getPartitions
65         partitions.synchronized(partitions ++= p)
66       }
67       ds.manager.indices(sft).toList.map(i => CachedThreadPool.submit(() => loadSingle(i))).foreach(_.get)
68       partitions.distinct.sorted.foreach(p => Command.output.info(p))
69     }
70   }

90     * Adopt an existing set of index tables as a new partitions
91     */
92   trait AdoptPartitionCommand[DS <: GeoMesaDataStore[DS]] extends PartitionsCommand[DS] {
93 
94     override val name = "adopt"
95     override def params: AdoptPartitionParam
96 
97     override protected def execute(ds: DS, sft: SimpleFeatureType, partition: TablePartition): Unit = {
98       val time = Option(partition).collect { case p: TimePartition => p }.getOrElse {
99         throw new NotImplementedError(s"Unsupported partition implementation: ${partition.getClass.getName}")
100       }
101       val (start, end) = {
102         val (s, e) = new IntervalConverter("value").convert(params.value)
103         (ZonedDateTime.ofInstant(toInstant(s), ZoneOffset.UTC), ZonedDateTime.ofInstant(toInstant(e), ZoneOffset.UTC))
104       }
105       val tables = params.tables.asScala
106 
107       Command.user.info(s"Adopting tables ${tables.mkString(", ")} as partition ${params.partition}")
108 
109       val indices = ds.manager.indices(sft)
110       if (indices.lengthCompare(tables.length) != 0) {
111         throw new IllegalArgumentException(s"Expected an index table for each index: ${indices.map(_.name).mkString(", ")}")
112       }
113       // match tables first, to fail fast if there is no match
114       val indexNames = indices.map { index =>
115         val name = StringSerialization.alphaNumericSafeString(index.name)
116         val attributes = index.attributes.map(StringSerialization.alphaNumericSafeString).mkString("_")
117         val version = s"v${index.version}"
118         (index, name, attributes, version)
119       }
120       val tableToIndexName = indexNames
121         .toList
122         .sortWith(_._3.length > _._3.length) // Sorted so we get best match first
123         .foldLeft(Map[String,(GeoMesaFeatureIndex[_,_],String,String,String)]()){
124           (mappedTables,indexName) =>
125           val fullMatches = tables.filter(table =>
126             !mappedTables.contains(table) // Don't match again
127               && table.contains(indexName._2) // Correct type name as in z3, attr, id...
128               && table.contains(indexName._3) // Correct attributes
129               && table.contains(indexName._4) // Correct version
130           )
131           if (fullMatches.lengthCompare(1) >= 0) {
132             mappedTables + (fullMatches.head -> indexName)
133           } else {
134            throw new IllegalArgumentException(s"Could not match a table to an index of '${indexName._3}'")
135           }
136         }
137       tableToIndexName.foreach { case (table, indexName) =>
138         ds.metadata.insert(sft.getTypeName, indexName._1.tableNameKey(Some(params.partition)), table)
139       }
140       time.register(params.partition, start, end)
141 
142       Command.user.info("Done")
143     }
144   }

149   trait DeletePartitionsCommand[DS <: GeoMesaDataStore[DS]] extends ModifyPartitionsCommand[DS] {
150 
151     override val name = "delete"
152     override def params: PartitionsParam with OptionalForceParam
153 
154     override protected def execute(ds: DS, sft: SimpleFeatureType, partition: TablePartition): Unit = {
155       lazy val prompt = s"Deleting partition(s) '${params.partitions.asScala.mkString("', '")}'. Continue (y/n)? "
156       if (params.force || Prompt.confirm(prompt)) {
157         super.execute(ds, sft, partition)
158       } else {
159         Command.user.info("Delete cancelled")
160       }
161     }
162 
163     override protected def modify(ds: DS, sft: SimpleFeatureType, partition: TablePartition, p: String): Unit = {
164       Command.user.info(s"Deleting partition '$p'")
165       def deleteSingle(index: GeoMesaFeatureIndex[_, _]): Unit =
166         ds.adapter.deleteTables(index.deleteTableNames(Some(p)))
167       ds.manager.indices(sft).toList.map(i => CachedThreadPool.submit(() => deleteSingle(i))).foreach(_.get)
168     }
169   }

213 
214     override def params: RequiredTypeNameParam
215 
216     override def execute(): Unit = {
217       withDataStore { ds =>
218         val sft = ds.getSchema(params.featureName)

220           throw new ParameterException(s"Schema '${params.featureName}' does not exist in the data store. " +

222         }
223         TablePartition(ds, sft) match {
224           case None => Command.user.error(s"Schema is not partitioned")
225           case Some(p) => execute(ds, sft, p)

227       }
228     }
