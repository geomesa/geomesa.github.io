222   class SortKeyMapper extends Mapper[Text, SimpleFeature, Text, SimpleFeature] with StrictLogging {
223 
224     type Context = Mapper[Text, SimpleFeature, Text, SimpleFeature]#Context
225 
226     private val text = new Text
227     private val builder = new StringBuilder()
228 
229     private var sortFields: Seq[Int] = _
230     private var counter: Counter = _
231 
232     override protected def setup(context: Context): Unit = {
233       val sft = GeoMesaConfigurator.getResultsToFeatures(context.getConfiguration).schema
234       val sorting = GeoMesaConfigurator.getSorting(context.getConfiguration).getOrElse {
235         throw new IllegalStateException("No sorting defined in configuration")
236       }
237       sortFields = sorting.map { case (f, _) => sft.indexOf(f) }
238       counter = context.getCounter(Counters.Group, Counters.Loaded)
239     }
240 
241     override protected def map(key: Text, sf: SimpleFeature, context: Context): Unit = {
242       logger.debug(s"map key ${key.toString}, map value ${DataUtilities.encodeFeature(sf)}")
243       builder.clear()
244       sortFields.foreach { i =>
245         val attribute = sf.getAttribute(i)
246         val encoded = if (attribute == null) { "" } else { AttributeIndexKey.typeEncode(attribute) }
247         builder.append(encoded).append(ByteArrays.ZeroByte)
248       }
249       text.set(builder.toString)
250       context.write(text, sf)
251       counter.increment(1L)
252     }

324     * @param files source for file names

329   class ExportChunkedRecordWriter(
330       context: TaskAttemptContext,
331       files: Iterator[String],
332       opts: ExportOptions,
333       hints: Hints,
334       chunks: Long
335     ) extends RecordWriter[Text, SimpleFeature] with LazyLogging {

338

346         if (estimator == null) {

356       if (estimator.done(exporter.bytes)) {
357         exporter.close()
358         exporter = null
359       }
360     }

363   }

365   /**
366     * Record writer that wraps a chunked non-countable FeatureExporter. Because the exporter is not
367     * countable, the bytes written will generally only be accurate after the exporter is closed
368     *
369     * @param context task context
370     * @param files source for file names
371     * @param opts options
372     * @param hints query hints
373     * @param chunks chunk size, in bytes
374     */
375   class ExportUncountableChunkedRecordWriter(
376       context: TaskAttemptContext,
377       files: Iterator[String],
378       opts: ExportOptions,
379       hints: Hints,
380       chunks: Long
381     ) extends RecordWriter[Text, SimpleFeature] with LazyLogging {
382 
383     private val counter = context.getCounter(Counters.Group, Counters.Written)
384 
385     private var exporter: Exporter = _
386     private var estimator: FileSizeEstimator = _
387     private var estimate = 0L // estimated number of features to write to hit our chunk size
388     private var count = 0L // current number of features written since the last estimate
389 
390     override def write(key: Text, value: SimpleFeature): Unit = {
391       if (exporter == null) {
392         exporter = new Exporter(opts.copy(file = Some(files.next)), hints)
393         exporter.start(value.getFeatureType)
394         if (estimator == null) {
395           val bytesPerFeature = opts.format.bytesPerFeature(value.getFeatureType)
396           estimator = new FileSizeEstimator(chunks, 0.05f, bytesPerFeature) // 5% error threshold
397         }
398         estimate = estimator.estimate(0L)
399       }
400 
401       exporter.export(Iterator.single(value))
402       counter.increment(1L) // since aggregation is disabled, this will always be 1 to 1 with features
403       count += 1
404 
405       if (count >= estimate) {
406         // since the bytes aren't generally available until after closing the writer,
407         // we have to go with our initial estimate and adjust after the first chunk
408         exporter.close()
409         // update the estimator so that it's more accurate on the next run
410         estimator.update(exporter.bytes, count)
411         exporter = null
412         count = 0
413       }
414     }
415 
416     override def close(context: TaskAttemptContext): Unit = if (exporter != null) { exporter.close() }
417   }
