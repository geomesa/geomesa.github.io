20 /**
21   * Exports features in various formats. Usage pattern is:
22   *
23   *   start()
24   *   export() - 0 to n times
25   *   close()
26   */

29   /**
30     * Start the export

32     * @param sft simple feature type
33     */

36   /**
37     * Export a batch of features
38     *
39     * @param features features to export
40     * @return count of features exported, if available
41     */

78   /**
79    * Export output stream, lazily instantiated
80    *
81    * @param name file name
82    * @param gzip gzip
83    */
84   class LazyExportStream(name: Option[String], gzip: Option[Int]) extends ExportStream {
85 
86     // lowest level - keep track of the bytes we write
87     // do this before any compression, buffering, etc so we get an accurate count
88     private var counter: CountingOutputStream = _
89     private var stream: OutputStream = _
90 
91     override def os: OutputStream = {
92       if (stream == null) {
93         val base = name match {
94           case None    => System.out
95           case Some(n) => PathUtils.getHandle(n).write(CreateMode.Create, createParents = true)
96         }
97         counter = new CountingOutputStream(base)
98         val compressed = gzip match {
99           case None => counter
100           case Some(c) => new GZIPOutputStream(counter) { `def`.setLevel(c) } // hack to access the protected deflate level
101         }
102         stream = new BufferedOutputStream(compressed)
103       }
104       stream
105     }
106 
107     override def bytes: Long = if (counter == null) { 0L } else { counter.getBytesWritten }
108 
109     override def close(): Unit = if (stream != null) { stream.close() }
110   }
111 
112   /**
113    * Byte output stream, mainly for testing
114    */
115   class ByteExportStream extends ExportStream {
116     override val os: ByteArrayOutputStream = new ByteArrayOutputStream()
117     override def bytes: Long = os.size()
118     override def close(): Unit = {}
119     def toByteArray: Array[Byte] = os.toByteArray
120   }
