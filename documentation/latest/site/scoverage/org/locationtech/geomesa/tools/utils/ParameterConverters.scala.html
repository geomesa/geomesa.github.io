75     override def convert(value: String): (String, String) = {

102   class IntervalConverter(name: String) extends BaseConverter[(Date, Date)](name) {
103     override def convert(value: String): (Date, Date) = {
104       try {
105         val i = value.indexOf('/')
106         if (i == -1 || value.indexOf('/', i + 1) != -1) {
107           throw new IllegalArgumentException("Interval from/to must be separated by a single '/'")
108         }
109         val start = FastConverter.convert(value.substring(0, i), classOf[Date])
110         val end = FastConverter.convert(value.substring(i + 1), classOf[Date])
111         if (start == null || end == null) {
112           throw new IllegalArgumentException(s"Could not convert $value to date interval")
113         }
114         (start, end)
115       } catch {
116         case NonFatal(e) => throw new ParameterException(getErrorString(value, s"format: $e"))
117       }
118     }
119   }
