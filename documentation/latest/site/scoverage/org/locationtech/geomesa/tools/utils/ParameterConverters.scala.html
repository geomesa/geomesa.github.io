98   class IntervalConverter(name: String) extends BaseConverter[(Date, Date)](name) {
99     override def convert(value: String): (Date, Date) = {
100       try {
101         val i = value.indexOf('/')
102         if (i == -1 || value.indexOf('/', i + 1) != -1) {
103           throw new IllegalArgumentException("Interval from/to must be separated by a single '/'")
104         }
105         val start = FastConverter.convert(value.substring(0, i), classOf[Date])
106         val end = FastConverter.convert(value.substring(i + 1), classOf[Date])
107         if (start == null || end == null) {
108           throw new IllegalArgumentException(s"Could not convert $value to date interval")
109         }
110         (start, end)
111       } catch {
112         case NonFatal(e) => throw new ParameterException(getErrorString(value, s"format: $e"))
113       }
114     }
115   }
