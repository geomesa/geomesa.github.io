74   class KeyValueConverter(name: String) extends BaseConverter[(String, String)](name) {
75     override def convert(value: String): (String, String) = {
76       try {
77         val i = value.indexOf('=')
78         if (i == -1 || value.indexOf('=', i + 1) != -1) {
79           throw new IllegalArgumentException("key-value pairs must be separated by a single '='")
80         }
81         (value.substring(0, i), value.substring(i + 1))
82       } catch {
83         case NonFatal(e) => throw new ParameterException(getErrorString(value, s"format: $e"))
84       }
85     }

88   class IntervalConverter(name: String) extends BaseConverter[(Date, Date)](name) {
89     override def convert(value: String): (Date, Date) = {
90       try {
91         val i = value.indexOf('/')
92         if (i == -1 || value.indexOf('/', i + 1) != -1) {
93           throw new IllegalArgumentException("Interval from/to must be separated by a single '/'")
94         }
95         val start = FastConverter.convert(value.substring(0, i), classOf[Date])
96         val end = FastConverter.convert(value.substring(i + 1), classOf[Date])
97         if (start == null || end == null) {
98           throw new IllegalArgumentException(s"Could not convert $value to date interval")
99         }
100         (start, end)
101       } catch {
102         case NonFatal(e) => throw new ParameterException(getErrorString(value, s"format: $e"))
103       }
104     }
105   }
