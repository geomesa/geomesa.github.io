24 class AutoCompleteCommand(runner: Runner, jc: JCommander) extends Command {
25 
26   import scala.collection.JavaConverters._
27 
28   override val name = "autocomplete"
29   override val params = new AutoCompleteParameters()
30 
31   override def execute(): Unit = {
32     if (params.force ||
33         Prompt.confirm(s"Register auto-complete for ${runner.name} commands to ${params.file.getAbsolutePath} (y/n)? ", default = "y")) {
34       val exists = params.file.exists() &&
35         WithClose(Source.fromFile(params.file)(StandardCharsets.UTF_8)) { source =>
36           source.getLines().exists(_.startsWith(s"_${runner.name}()"))
37         }
38       if (exists) {
39         Command.user.info("Auto-complete function already installed")
40       } else {
41         if (params.file.getParentFile != null && !params.file.getParentFile.exists()) {
42           params.file.getParentFile.mkdirs()
43         }
44         WithClose(new FileWriter(params.file, StandardCharsets.UTF_8, true)) { writer =>
45           val commands = jc.getCommands.asScala.keys.toSeq.sorted
46           writer.write(
47             s"""# begin added by ${runner.name}
48                |_${runner.name}(){
49                |  local cur prev;
50                |  COMPREPLY=();
51                |  cur="$${COMP_WORDS[COMP_CWORD]}";
52                |  prev="$${COMP_WORDS[COMP_CWORD-1]}";
53                |
54                |  case $${COMP_CWORD} in
55                |    1)
56                |      COMPREPLY=( $$(compgen -W "${commands.mkString(" ")}" $${cur}));
57                |      ;;
58                |    *)
59                |      case $${COMP_WORDS[1]} in
60                |""".stripMargin)
61           commands.foreach { command =>
62             writer.write(
63               s"          $command)\n")
64             if (command == "help") {
65               writer.write(
66             s"""            COMPREPLY=( $$(compgen -W "${commands.mkString(" ")}" $${cur}));
67                |""".stripMargin)
68             } else {
69               val params = jc.getCommands.get(command).getParameters.asScala.filterNot(_.getParameter.hidden())
70               val flags = params.flatMap(_.getParameter.names().filterNot(_.length == 2)/* filter out short flag variants */)
71               // note: this isn't universally true, but it is for all our commands
72               val hasFiles = jc.getCommands.get(command).getMainParameter != null
73               // TODO we could compare `prev` and check the arity of the command to know whether an arg is expected
74               // TODO we could also do some kind of check or override to determine if the arg is a file or not
75               if (hasFiles) {
76           writer.write(
77             s"""            if [[ "$${cur}" =~ ^-[a-zA-Z-]?+$$ ]]; then
78                |              COMPREPLY=( $$(compgen -W "${flags.mkString(" ").replaceAll("[,\\s]+", " ")}" -- $${cur}));
79                |            else
80                |              compopt -o filenames -o nospace;
81                |              COMPREPLY=( $$(compgen -f "$$2") );
82                |            fi
83                |""".stripMargin)
84               } else {
85           writer.write(
86             s"""            if [[ -z "$${cur}" ]] || [[ "$${cur}" =~ ^-[a-zA-Z-]?+$$ ]]; then
87                |              COMPREPLY=( $$(compgen -W "${flags.mkString(" ").replaceAll("[,\\s]+", " ")}" -- $${cur}));
88                |            else
89                |              compopt -o filenames -o nospace;
90                |              COMPREPLY=( $$(compgen -f "$$2") );
91                |            fi
92                |""".stripMargin)
93               }
94             }
95             writer.write(
96             s"""            return 0;
97                |            ;;
98                |
99                |""".stripMargin)
100           }
101           writer.write(
102             s"""      esac;
103                |      ;;
104                |  esac;
105                |};
106                |complete -F _${runner.name} ${runner.name};
107                |complete -F _${runner.name} bin/${runner.name};
108                |# end added by ${runner.name}
109                |""".stripMargin)
110         }
111         Command.user.info("Auto-complete installed, to use now run:")
112         Command.user.info(s"  source ${params.file.getAbsolutePath}")
113       }
114     }
115   }
116 }
