35 trait PlaybackCommand[DS <: DataStore] extends ExportCommand[DS] {
36 
37   import org.locationtech.geomesa.filter.ff
38   import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
39 
40   override val name = "playback"
41   override def params: PlaybackParams
42 
43   override protected def export(
44       ds: DS,
45       query: Query,
46       exporter: FeatureExporter,
47       writeEmptyFiles: Boolean): Option[Long] = {
48     val features: SimpleFeatureCollection = new DataFeatureCollection(GeoMesaFeatureCollection.nextId) {
49 
50       private val fs = ds.getFeatureSource(query.getTypeName)
51       private val transform = query.getPropertyNames
52       private val dtg = Option(params.dtg)
53       private val filter = Option(query.getFilter).filter(_ != Filter.INCLUDE)
54       private val window = Option(params.window)
55       private val rate = Option(params.rate).map(_.floatValue()).getOrElse(1f)
56       private val live = Option(params.live).exists(_.booleanValue())
57 
58       private lazy val queryWithInterval = {
59         val dtg = Option(params.dtg).orElse(ds.getSchema(query.getTypeName).getDtgField).getOrElse {
60           throw new IllegalArgumentException("Schema does not have a default date field")
61         }
62         val period = new DefaultPeriod(
63           new DefaultInstant(new DefaultPosition(params.interval._1)),
64           new DefaultInstant(new DefaultPosition(params.interval._2))
65         )
66         val during = ff.during(ff.property(dtg), ff.literal(period))
67         val filterWithInterval = query.getFilter match {
68           case null | Filter.INCLUDE => during
69           case f => ff.and(f, during)
70         }
71         new Query(query.getTypeName, filterWithInterval, query.getMaxFeatures, query.getProperties, query.getHandle)
72       }
73 
74       override def getSchema: SimpleFeatureType = fs.getFeatures(query).getSchema
75 
76       override def getBounds: ReferencedEnvelope = fs.getBounds(queryWithInterval)
77 
78       override def getCount: Int = fs.getCount(queryWithInterval)
79 
80       override protected def openIterator(): java.util.Iterator[SimpleFeature] = {
81         val iter = new PlaybackIterator(ds, query.getTypeName, params.interval, dtg, filter, transform, window, rate, live)
82 
83         // note: result needs to implement Closeable in order to be closed by the DataFeatureCollection
84         if (params.maxFeatures != null) {
85           var count = 0
86           new java.util.Iterator[SimpleFeature] with Closeable {
87             override def hasNext: Boolean = count < params.maxFeatures && iter.hasNext
88             override def next(): SimpleFeature = { count += 1; iter.next() }
89             override def close(): Unit = iter.close()
90           }
91         } else {
92           new java.util.Iterator[SimpleFeature] with Closeable {
93             override def hasNext: Boolean = iter.hasNext
94             override def next(): SimpleFeature = iter.next()
95             override def close(): Unit = iter.close()
96           }
97         }
98       }
99     }
100 
101     try {
102       WithClose(CloseableIterator(features.features())) { iter =>
103         if (writeEmptyFiles || iter.hasNext) {
104           exporter.start(features.getSchema)
105           var count: Option[Long] = None
106           while (iter.hasNext) {
107             val res = exporter.export(Iterator.single(iter.next))
108             count = count match {
109               case None => res
110               case Some(c) => res.map(_ + c).orElse(count)
111             }
112           }
113           count
114         } else {
115           Some(0L)
116         }
117       }
118     } catch {
119       case NonFatal(e) =>
120         throw new RuntimeException("Could not execute export query. Please ensure " +
121             "that all arguments are correct", e)
122     }
123   }
124 
125   override final protected def configure(job: Job, ds: DS, query: Query): Unit =
126     throw new ParameterException("Distributed playback is not supported, please use --run-mode local")
127 }
