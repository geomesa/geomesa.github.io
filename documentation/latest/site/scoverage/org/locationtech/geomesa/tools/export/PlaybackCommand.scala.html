49   override protected def export(
50       ds: DS,
51       query: Query,
52       exporter: FeatureExporter,
53       writeEmptyFiles: Boolean): Option[Long] = {
54     val features: SimpleFeatureCollection = new DataFeatureCollection(GeoMesaFeatureCollection.nextId) {
55 
56       private val fs = ds.getFeatureSource(query.getTypeName)
57       private val transform = query.getPropertyNames
58       private val dtg = Option(params.dtg)
59       private val filter = Option(query.getFilter).filter(_ != Filter.INCLUDE)
60       private val window = Option(params.window)
61       private val rate = Option(params.rate).map(_.floatValue()).getOrElse(1f)
62       private val live = Option(params.live).exists(_.booleanValue())
63 
64       private lazy val queryWithInterval = {
65         val dtg = Option(params.dtg).orElse(ds.getSchema(query.getTypeName).getDtgField).getOrElse {
66           throw new IllegalArgumentException("Schema does not have a default date field")
67         }
68         val period = new DefaultPeriod(
69           new DefaultInstant(new DefaultPosition(params.interval._1)),
70           new DefaultInstant(new DefaultPosition(params.interval._2))
71         )
72         val during = ff.during(ff.property(dtg), ff.literal(period))
73         val filterWithInterval = query.getFilter match {
74           case null | Filter.INCLUDE => during
75           case f => ff.and(f, during)
76         }
77         new Query(query.getTypeName, filterWithInterval, query.getMaxFeatures, query.getProperties, query.getHandle)
78       }
79 
80       override def getSchema: SimpleFeatureType = fs.getFeatures(query).getSchema
81 
82       override def getBounds: ReferencedEnvelope = fs.getBounds(queryWithInterval)
83 
84       override def getCount: Int = fs.getCount(queryWithInterval)
85 
86       override protected def openIterator(): java.util.Iterator[SimpleFeature] = {
87         val iter = new PlaybackIterator(ds, query.getTypeName, params.interval, dtg, filter, transform, window, rate, live)
88 
89         // note: result needs to implement Closeable in order to be closed by the DataFeatureCollection
90         if (params.maxFeatures != null) {
91           var count = 0
92           new java.util.Iterator[SimpleFeature] with Closeable {
93             override def hasNext: Boolean = count < params.maxFeatures && iter.hasNext
94             override def next(): SimpleFeature = { count += 1; iter.next() }
95             override def close(): Unit = iter.close()
96           }
97         } else {
98           new java.util.Iterator[SimpleFeature] with Closeable {
99             override def hasNext: Boolean = iter.hasNext
100             override def next(): SimpleFeature = iter.next()
101             override def close(): Unit = iter.close()
102           }
103         }
104       }
105     }
106 
107     try {
108       WithClose(CloseableIterator(features.features())) { iter =>
109         if (writeEmptyFiles || iter.hasNext) {
110           exporter.start(features.getSchema)
111           var count: Option[Long] = None
112           while (iter.hasNext) {
113             val res = exporter.export(Iterator.single(iter.next))
114             count = count match {
115               case None => res
116               case Some(c) => res.map(_ + c).orElse(count)
117             }
118           }
119           count
120         } else {
121           Some(0L)
122         }
123       }
124     } catch {
125       case NonFatal(e) =>
126         throw new RuntimeException("Could not execute export query. Please ensure " +
127             "that all arguments are correct", e)
128     }
129   }

141     @Parameter(names = Array("--interval"), description = "Date interval to query, in the format yyyy-MM-dd'T'HH:mm:ss.SSSZ/yyyy-MM-dd'T'HH:mm:ss.SSSZ", required = true, converter = classOf[IntervalConverter])

148     var window: Duration = _
149

151     var rate: java.lang.Float = _

158   /**
159    * Query over a time frame and return the features in sorted order, delayed based on the date of each feature
160    * to simulate the original ingestion stream
161    *
162    * @param ds data store
163    * @param typeName simple feature type name
164    * @param interval interval to query
165    * @param dtg date attribute to sort by
166    * @param filter additional filter predicate, if any
167    * @param transforms query transforms, if any
168    * @param window length of a single query window, used to chunk up the total features
169    * @param rate multiplier for the rate of returning features, applied to the original delay between features
170    * @param live project dates to current time
171    * @param readAhead size of the read-ahead queue used for holding features before returning them
172    */
173   class PlaybackIterator(
174       ds: DataStore,
175       typeName: String,
176       interval: (Date, Date),
177       dtg: Option[String] = None,
178       filter: Option[Filter] = None,
179       transforms: Array[String] = null,
180       window: Option[Duration] = None,
181       rate: Float = 10f,
182       live: Boolean = false,
183       readAhead: Int = 10000
184     ) extends CloseableIterator[SimpleFeature] with StrictLogging {
185 
186     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
187 
188     require(interval._2.after(interval._1), s"Interval is not ordered correctly: ${interval._1}/${interval._2}")
189 
190     private val sft = ds.getSchema(typeName)
191     private val dtgName = dtg.orElse(sft.getDtgField).getOrElse {
192       throw new IllegalArgumentException("Schema does not have a default date field")
193     }
194     private val tdefs = transforms match {
195       case null => null
196       case t if t.indexOf(dtgName) == -1 => t :+ dtgName
197       case t => t
198     }
199     private val dtgIndex = tdefs match {
200       case null => sft.indexOf(dtgName)
201       case t => t.indexOf(dtgName)
202     }
203     require(dtgIndex != -1, "Invalid date field")
204     private val dtgProp = ff.property(dtgName)
205     private val sort = ff.sort(dtgName, SortOrder.ASCENDING)
206 
207     private val windowMillis = window.map(_.toMillis).getOrElse(interval._2.getTime - interval._1.getTime + 1)
208 
209     private var start: Long = -1
210     private var eventStart: Long = -1
211 
212     private val features = new LinkedBlockingQueue[SimpleFeature](readAhead)
213     private var staged: SimpleFeature = _
214 
215     private val executor = Executors.newSingleThreadExecutor()
216     executor.submit(new QueryRunnable())
217 
218     override def hasNext: Boolean = {
219       if (staged != null) {
220         true
221       } else {
222         staged = features.take()
223         if (!terminal.eq(staged)) {
224           true
225         } else {
226           features.put(staged) // re-queue the terminal value to keep this method idempotent
227           staged = null
228           false
229         }
230       }
231     }
232 
233     override def next(): SimpleFeature = {
234       val feature = staged
235       staged = null
236       val featureTime = feature.getAttribute(dtgIndex).asInstanceOf[Date].getTime
237       if (start == -1L) {
238         // emit the first feature as soon as it's available, and set the clock to start timing from here
239         logger.debug("Starting replay clock")
240         start = System.currentTimeMillis()
241         eventStart = featureTime
242       }
243       val featureRelativeTime = start + ((featureTime - eventStart) / rate).toLong
244       val sleep = featureRelativeTime - System.currentTimeMillis()
245       if (sleep > 0) {
246         Thread.sleep(sleep)
247       }
248       if (live) {
249         feature.setAttribute(dtgIndex, new Date(featureRelativeTime))
250       }
251       feature
252     }
253 
254     override def close(): Unit = executor.shutdownNow()
255 
256     private class QueryRunnable extends Runnable {
257       override def run(): Unit = {
258         try {
259           var from = interval._1
260           var to = new Date(from.getTime + windowMillis)
261           var loop = true
262 
263           while (loop && !Thread.currentThread().isInterrupted) {
264             if (interval._2.before(to)) {
265               // this query will finish the last window
266               to = interval._2
267               loop = false
268             }
269 
270             logger.debug(s"Running query window $from to $to")
271 
272             val during = {
273               val period = new DefaultPeriod(
274                 new DefaultInstant(new DefaultPosition(from)),
275                 new DefaultInstant(new DefaultPosition(to))
276               )
277               ff.during(dtgProp, ff.literal(period))
278             }
279             val query = new Query(typeName, filter.map(ff.and(_, during)).getOrElse(during), tdefs: _*)
280             query.setSortBy(sort)
281             // prevent ContentDataStore from sorting on disk
282             query.getHints.put(Hints.MAX_MEMORY_SORT, java.lang.Integer.MAX_VALUE)
283 
284             var count = 0L
285 
286             // populate the queue - this will block if we get too far ahead
287             CloseableIterator(ds.getFeatureReader(query, Transaction.AUTO_COMMIT)).foreach { f =>
288               features.put(f)
289               count += 1
290             }
291 
292             logger.debug(s"Returned $count features from query window $from to $to")
293 
294             // increment time window
295             from = to
296             to = new Date(from.getTime + windowMillis)
297           }
298         } catch {
299           case NonFatal(e) => logger.error("Error querying playback:", e)
300         } finally {
301           features.put(terminal)
302         }
303       }
304     }
305   }
