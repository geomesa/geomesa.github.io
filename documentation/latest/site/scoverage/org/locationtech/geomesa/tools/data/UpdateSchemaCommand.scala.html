36   override val name = "update-schema"
37   override def params: UpdateSchemaParams
38 
39   override def execute(): Unit = withDataStore(update)

41   protected def update(ds: DS): Unit = {
42     // ensure we have an operation
43     params.validate().foreach(e => throw e)
44 
45     val sft = try { ds.getSchema(params.featureName) } catch { case _: IOException => null }
46     if (sft == null) {
47       throw new ParameterException(s"Schema '${params.featureName}' does not exist in the data store")
48     }
49 
50     var n = 0
51     // numbering for our prompts
52     def number: Int = { n += 1; n }
53     val prompts = new StringBuilder()
54     var canRenameTables = false
55 
56     val builder = new SimpleFeatureTypeBuilder()
57     builder.init(sft)
58 
59     Option(params.rename).foreach { rename =>
60       rename.indexOf(':') match {
61         case -1 => builder.setName(rename)
62         case i  => builder.setNamespaceURI(rename.substring(0, i)); builder.setName(rename.substring(i + 1))
63       }
64       prompts.append(s"\n  $number: Renaming schema to '$rename'")
65       canRenameTables = true
66     }
67 
68     params.renameAttributes.asScala.grouped(2).foreach { case Seq(from, to) =>
69       val i = sft.indexOf(from.asInstanceOf[String])
70       if (i == -1) {
71         throw new ParameterException(s"Attribute '$from' does not exist in the schema")
72       }
73       val attribute = new AttributeTypeBuilder()
74       attribute.init(sft.getDescriptor(i))
75       builder.set(i, attribute.buildDescriptor(to.asInstanceOf[String]))
76       if (sft.getGeomField == from) {
77         builder.setDefaultGeometry(to.asInstanceOf[String])
78       }
79       prompts.append(s"\n  $number: Renaming attribute '$from' to '$to'")
80       canRenameTables = canRenameTables || sft.getIndices.exists(_.attributes.contains(from))
81     }
82 
83     if (canRenameTables && params.renameTables) {
84       prompts.append(s"\n  $number: Renaming index tables (WARNING may be expensive)")
85     }
86 
87     params.attributes.asScala.foreach { attribute =>
88       val spec = try { SimpleFeatureSpecParser.parseAttribute(attribute) } catch {
89         case NonFatal(e) => throw new ParameterException(s"Invalid attribute spec: $attribute", e)
90       }
91       builder.add(spec.toDescriptor)
92       prompts.append(s"\n  $number: Adding attribute '${spec.name}' of type ${spec.clazz.getName}")
93     }
94 
95     val updated = builder.buildFeatureType()
96     updated.getUserData.putAll(sft.getUserData)
97 
98     if (!params.plusKeywords.isEmpty) {
99       val keywords = params.plusKeywords.asScala
100       updated.addKeywords(keywords.toSet)
101       prompts.append(s"\n  $number: Adding keywords: '${keywords.mkString("', '")}'")
102     }
103     if (!params.minusKeywords.isEmpty) {
104       val keywords = params.minusKeywords.asScala
105       updated.removeKeywords(keywords.toSet)
106       prompts.append(s"\n  $number: Removing keywords: '${keywords.mkString("', '")}'")
107     }
108     if (!params.userData.isEmpty) {
109       params.userData.asScala.foreach { ud =>
110         ud.split(":", 2) match {
111           case Array(k, v) =>
112             updated.getUserData.put(k, v) match {
113               case null => prompts.append(s"\n  $number: Adding user data: '$k=$v'")
114               case old  => prompts.append(s"\n  $number: Updating user data: '$k=$v' (was '$old')")
115             }
116           case _ => throw new ParameterException(s"Invalid user data entry - expected 'key:value': $ud")
117         }
118       }
119     }
120 
121     Option(params.enableStats).map(_.booleanValue()).foreach { enable =>
122       sft.setStatsEnabled(enable)
123       prompts.append(s"\n  $number: ${if (enable) { "En" } else { "Dis" }}abling stats")
124     }
125 
126     if (params.renameTables) {
127       updated.getUserData.put(SimpleFeatureTypes.Configs.UpdateRenameTables, java.lang.Boolean.TRUE)
128     }
129     if (params.noBackup) {
130       updated.getUserData.put(SimpleFeatureTypes.Configs.UpdateBackupMetadata, java.lang.Boolean.FALSE)
131     }
132 
133     Command.user.info(s"Preparing to update schema '${sft.getTypeName}':$prompts")
134     if (params.force || Prompt.confirm("Continue (y/n)? ")) {
135       Command.user.info("Updating, please wait...")
136       ds.updateSchema(sft.getTypeName, updated)
137       Command.user.info("Update complete")
138     }
139   }
140 }
