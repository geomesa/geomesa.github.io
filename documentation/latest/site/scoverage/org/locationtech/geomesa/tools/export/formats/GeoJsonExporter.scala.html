30   override def export(features: Iterator[SimpleFeature]): Option[Long] = {
31     var count = 0L
32     while (features.hasNext) {
33       serializer.write(writer, features.next)
34       count += 1L
35     }
36     writer.flush()
37     Some(count)
38   }
39 
40   override def close(): Unit = {
41     try {
42       if (serializer != null) {

46       }
47     } finally {
48       if (writer != null) {

50       }
51     }
52   }
