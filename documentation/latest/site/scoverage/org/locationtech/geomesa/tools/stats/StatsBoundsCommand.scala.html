20 trait StatsBoundsCommand[DS <: DataStore with HasGeoMesaStats] extends DataStoreCommand[DS] {
21 
22   override val name = "stats-bounds"
23   override def params: StatsBoundsParams
24 
25   override def execute(): Unit = withDataStore(calculateBounds)
26 
27   protected def calculateBounds(ds: DS): Unit = {
28     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
29 
30     for {
31       p <- Option(params).collect { case p: ProvidedTypeNameParam => p }
32       f <- Option(ds).collect { case f: FileDataStore => f }
33     } { p.featureName = f.getSchema.getTypeName }
34 
35     val sft = ds.getSchema(params.featureName)
36     if (sft == null) {
37       throw new ParameterException(s"Schema '${params.featureName}' does not exist")
38     }
39     val attributes = getAttributesFromParams(sft, params)
40     val filter = Option(params.cqlFilter).getOrElse(Filter.INCLUDE)
41 
42     if (params.exact) {
43       Command.user.info("Running stat query...")
44     } else if (filter != Filter.INCLUDE) {
45       Command.user.warn("Non-exact stat queries may not fully account for the specified CQL filter")
46     }
47 
48     val allBounds = ds.stats.getSeqStat[MinMax[Any]](sft, attributes.map(Stat.MinMax), filter, params.exact)
49 
50     attributes.foreach { attribute =>
51       val out = allBounds.find(_.property == attribute) match {
52         case None => "[ unavailable ]"
53         case Some(mm) if mm.isEmpty => "[ no matching data ]"
54         case Some(mm) =>
55           if (attribute == sft.getGeomField) {
56             val e = mm.min.asInstanceOf[Geometry].getEnvelopeInternal
57             e.expandToInclude(mm.max.asInstanceOf[Geometry].getEnvelopeInternal)
58             s"[ ${e.getMinX}, ${e.getMinY}, ${e.getMaxX}, ${e.getMaxY} ] cardinality: ${mm.cardinality}"
59           } else {
60             val stringify = Stat.stringifier(sft.getDescriptor(attribute).getType.getBinding)
61             s"[ ${stringify(mm.min)} to ${stringify(mm.max)} ] cardinality: ${mm.cardinality}"
62           }
63       }
64       Command.output.info(s"  $attribute $out")
65     }
66   }
67 }
