73   private def export(ds: DS): JobResult = {
74     // for file data stores, handle setting the default type name so the user doesn't have to
75     for {
76       p <- Option(params).collect { case p: ProvidedTypeNameParam => p }
77       f <- Option(ds).collect { case f: FileDataStore => f }
78     } { p.featureName = f.getSchema.getTypeName }
79 
80     val options = ExportOptions(params)
81     val remote = options.file.exists(PathUtils.isRemote)
82     val reducers = Option(params.reducers).map(_.intValue()).getOrElse(-1)
83     val mode = params.mode.getOrElse(if (reducers == -1) { RunModes.Local } else { RunModes.Distributed })
84     val chunks = Option(params.chunkSize).map(_.longValue())
85 
86     // do some validation up front
87     if (options.file.isEmpty && !options.format.streaming) {
88       throw new ParameterException(s"Format '${options.format}' requires file-based output, please use --output")
89     } else if (remote && options.format == ExportFormat.Shp) {
90       throw new ParameterException("Shape file export is not supported for distributed file systems")
91     } else if (!remote && mode == RunModes.Distributed) {
92       throw new ParameterException("Distributed export requires an output file in a distributed file system")
93     } else if (mode == RunModes.Distributed && params.maxFeatures != null) {
94       throw new ParameterException("Distributed export does not support --max-features")
95     }
96 
97     val sft = ds.getSchema(params.featureName)
98     if (sft == null) {
99       throw new ParameterException(s"Schema '${params.featureName}' does not exist in the store")
100     }
101 
102     val query = ExportCommand.createQuery(sft, params)
103 
104     mode match {
105       case RunModes.Local =>
106         options.file.map(PathUtils.getHandle).foreach { file =>
107           if (file.exists) {
108             if (params.force) {
109               Command.user.warn(s"Output file '${file.path}' exists - deleting it")
110             } else if (!Prompt.confirm(s"WARNING Output file '${file.path}' exists, delete it and continue (y/n)? ")) {
111               throw new ParameterException(s"Output file '${file.path}' exists")
112             }
113             file.delete()
114           }
115         }
116         val exporter = chunks match {
117           case None    => new Exporter(options, query.getHints)
118           case Some(c) => new ChunkedExporter(options, query.getHints, c)
119         }
120         val count = try { export(ds, query, exporter, !params.suppressEmpty) } finally { exporter.close() }
121         val outFile = options.file match {
122           case None => "standard out"
123           case Some(f) if chunks.isDefined => PathUtils.getBaseNameAndExtension(f).productIterator.mkString("_*")
124           case Some(f) => f
125         }
126 
127         JobSuccess(s"Feature export complete to $outFile", count.map(CountKey -> _).toMap)
128 
129       case RunModes.Distributed =>
130         val job = Job.getInstance(new Configuration, "GeoMesa Tools Export")
131 
132         // for remote jobs, don't push down format transforms, to enable counting and global sorting
133         val hints = new Hints(query.getHints)
134         ExportCommand.disableAggregation(sft, query.getHints)
135 
136         configure(job, ds, query) // note: do this first so that input format is set for the TotalOrderPartitioner
137 
138         // note: these confs should be set by the input format
139         val reduce = Seq(GeoMesaConfigurator.Keys.FeatureReducer, GeoMesaConfigurator.Keys.Sorting).filter { key =>
140           job.getConfiguration.get(key) != null
141         }
142 
143         if (reducers < 1 && reduce.nonEmpty) {
144           if (reduce.contains(GeoMesaConfigurator.Keys.Sorting)) {
145             throw new ParameterException("Distributed export sorting requires --num-reducers")
146           } else {
147             throw new ParameterException(s"Distributed export format '${options.format}' requires --num-reducers")
148           }
149         }
150 
151         // must be some due to our remote check
152         val file = options.file.getOrElse(throw new IllegalStateException("file should be Some"))
153         val output = new Path(PathUtils.getUrl(file).toURI).getParent
154 
155         // file output format doesn't generally let you write to an existing directory
156         val context = FileContext.getFileContext(output.toUri, job.getConfiguration)
157         if (context.util.exists(output)) {
158           val warning = s"Output directory '$output' exists - files may be overwritten"
159           if (params.force) {
160             Command.user.warn(warning)
161           } else if (!Prompt.confirm(s"WARNING $warning. Continue anyway (y/n)? ")) {
162             if (Prompt.confirm("WARNING DATA MAY BE LOST - delete directory and proceed with export (y/n)? ")) {
163               context.delete(output, true)
164             } else {
165               throw new ParameterException(s"Output directory '$output' exists")
166             }
167           }
168         }
169 
170         val filename = FilenameUtils.getName(file)
171 
172         // note: use our original hints, which have the aggregating keys
173         ExportJob.configure(job, connection, sft, hints, filename, output, options.format, options.headers,
174           chunks, options.gzip, reducers, libjars(options.format), libjarsPaths)
175 
176         val reporter = TerminalCallback()
177         JobRunner.run(job, reporter, ExportJob.Counters.mapping(job), ExportJob.Counters.reducing(job)).merge {
178           Some(JobSuccess(s"Feature export complete to $output", Map(CountKey -> ExportJob.Counters.count(job))))
179         }
180 
181       case _ => throw new UnsupportedOperationException() // someone added a run mode and didn't implement it here...
182     }
183   }

205       }

209             "that all arguments are correct", e)
210     }

243   /**
244     * Create the query to execute
245     *
246     * @param sft simple feature type
247     * @param params parameters
248     * @return
249     */
250   def createQuery(sft: SimpleFeatureType, params: ExportParams): Query = {
251     val typeName = Option(params).collect { case p: TypeNameParam => p.featureName }.orNull
252     val filter = Option(params.cqlFilter).getOrElse(Filter.INCLUDE)
253 
254     val query = new Query(typeName, filter)
255     val hints = query.getHints
256     Option(params.maxFeatures).map(Int.unbox).foreach(query.setMaxFeatures)
257     Option(params).collect { case p: OptionalIndexParam => p }.foreach { p =>
258       Option(p.index).foreach { index =>
259         logger.debug(s"Using index $index")
260         hints.put(QueryHints.QUERY_INDEX, index)
261       }
262     }
263 
264     Option(params.hints).foreach { hintStrings =>
265       hints.put(Hints.VIRTUAL_TABLE_PARAMETERS, hintStrings)
266       ViewParams.setHints(query)
267     }
268 
269     if (params.outputFormat == ExportFormat.Arrow) {
270       hints.put(QueryHints.ARROW_ENCODE, java.lang.Boolean.TRUE)
271     } else if (params.outputFormat == ExportFormat.Bin) {
272       // if not specified in hints, set it here to trigger the bin query
273       if (!hints.containsKey(QueryHints.BIN_TRACK)) {
274         hints.put(QueryHints.BIN_TRACK, "id")
275       }
276       if (!hints.containsKey(QueryHints.BIN_DTG)) {
277         sft.getDtgField.foreach(hints.put(QueryHints.BIN_DTG, _))
278       }
279     } else if (params.outputFormat == ExportFormat.Leaflet) {
280       configureLeafletExport(params)
281     }
282 
283     val attributes: Array[String] = {
284       val combined = params.attributes.asScala ++ params.transforms.asScala
285       if (combined.nonEmpty) {
286         val (id, attributes) = combined.partition(_.equalsIgnoreCase("id"))
287         if (id.isEmpty && !hints.containsKey(QueryHints.ARROW_INCLUDE_FID)) {
288           // note: we also use this hint for delimited text export
289           hints.put(QueryHints.ARROW_INCLUDE_FID, java.lang.Boolean.FALSE)
290         }
291         attributes.toArray
292       } else if (params.outputFormat == ExportFormat.Bin) {
293         BinAggregatingScan.propertyNames(hints, sft).toArray
294       } else {
295         null // all props
296       }
297     }
298     query.setPropertyNames(attributes: _*)
299 
300     if (!params.sortFields.isEmpty) {
301       val fields = params.sortFields.asScala.toSeq
302       if (fields.exists(a => sft.indexOf(a) == -1)) {
303         val errors = fields.filter(a => sft.indexOf(a) == -1)
304         throw new ParameterException(s"Invalid sort attribute${if (errors.lengthCompare(1) == 0) "" else "s"}: " +
305             errors.mkString(", "))
306       }
307       val order = if (params.sortDescending) { SortOrder.DESCENDING } else { SortOrder.ASCENDING }
308       query.setSortBy(fields.map(f => org.locationtech.geomesa.filter.ff.sort(f, order)): _*)
309     } else if (hints.isArrowQuery) {
310       hints.getArrowSort.foreach { case (f, r) =>
311         val order = if (r) { SortOrder.DESCENDING } else { SortOrder.ASCENDING }
312         query.setSortBy(org.locationtech.geomesa.filter.ff.sort(f, order))
313       }
314     }
315 
316     logger.debug(s"Applying CQL filter ${ECQL.toCQL(filter)}")
317     logger.debug(s"Applying transform ${Option(attributes).map(_.mkString(",")).orNull}")
318 
319     QueryRunner.configureDefaultQuery(sft, query)
320   }

369   /**
370     * Disable hints for aggregating scans by removing them, and moving any aggregating sort hints to
371     * regular sort hints
372     *
373     * @param sft simple feature type
374     * @param hints hints
375     * @return
376     */
377   private def disableAggregation(sft: SimpleFeatureType, hints: Hints): Unit = {
378     if (hints.isArrowQuery) {
379       hints.remove(QueryHints.ARROW_ENCODE)
380       val sort = hints.remove(QueryHints.ARROW_SORT_FIELD).asInstanceOf[String]
381       if (sort != null) {
382         val reverse = Option(hints.remove(QueryHints.ARROW_SORT_REVERSE).asInstanceOf[java.lang.Boolean])
383         val order = if (reverse.exists(_.booleanValue)) { SortOrder.DESCENDING } else { SortOrder.ASCENDING }
384         val hint = org.locationtech.geomesa.filter.ff.sort(sort, order)
385         hints.put(QueryHints.Internal.SORT_FIELDS, QueryHints.Internal.toSortHint(Array(hint)))
386       }
387     } else if (hints.isBinQuery) {
388       hints.remove(QueryHints.BIN_TRACK)
389       if (hints.isBinSorting) {
390         hints.getBinDtgField.orElse(sft.getDtgField).foreach { dtg =>
391           val hint = org.locationtech.geomesa.filter.ff.sort(dtg, SortOrder.ASCENDING)
392           hints.put(QueryHints.Internal.SORT_FIELDS, QueryHints.Internal.toSortHint(Array(hint)))
393         }
394       }
395     }

421     * Single exporter that handles the command options and delegates to the correct implementation
422     *
423     * @param options options
424     * @param hints query hints
425     */
426   class Exporter(options: ExportOptions, hints: Hints) extends FeatureExporter {
427 
428     // used only for streaming export formats
429     private lazy val stream: ByteCounterStream = {
430       // avro compression is handled differently, see AvroExporter below
431       val gzip = options.gzip.filter(_ => options.format != ExportFormat.Avro && options.format != ExportFormat.AvroNative)
432       val handle = options.file match {
433         case None => StdInHandle.get() // writes to std out
434         case Some(f) => PathUtils.getHandle(f)
435       }
436       if (options.writeEmptyFiles) {
437         new ExportStream(handle, gzip)
438       } else {
439         new LazyExportStream(handle, gzip)
440       }
441     }
442 
443     // used only for file-based export formats
444     private lazy val name = options.file.getOrElse {
445       // should have been validated already...
446       throw new IllegalStateException("Export format requires a file but none was specified")
447     }
448 
449     // noinspection ComparingUnrelatedTypes
450     private lazy val fids = !Option(hints.get(QueryHints.ARROW_INCLUDE_FID)).contains(java.lang.Boolean.FALSE)
451 
452     private val exporter = options.format match {
453       case ExportFormat.Arrow      => new ArrowExporter(stream, hints)
454       case ExportFormat.Avro       => new AvroExporter(stream, options.gzip)
455       case ExportFormat.AvroNative => new AvroExporter(stream, options.gzip, Set(SerializationOption.NativeCollections))
456       case ExportFormat.Bin        => new BinExporter(stream, hints)
457       case ExportFormat.Csv        => DelimitedExporter.csv(stream, options.headers, fids)
458       case ExportFormat.Gml2       => GmlExporter.gml2(stream)
459       case ExportFormat.Gml3       => GmlExporter(stream)
460       case ExportFormat.Json       => new GeoJsonExporter(stream)
461       case ExportFormat.Leaflet    => new LeafletMapExporter(stream)
462       case ExportFormat.Null       => NullExporter
463       case ExportFormat.Orc        => new OrcFileSystemExporter(name)
464       case ExportFormat.Parquet    => new ParquetFileSystemExporter(name)
465       case ExportFormat.Shp        => new ShapefileExporter(new File(name))
466       case ExportFormat.Tsv        => DelimitedExporter.tsv(stream, options.headers, fids)
467       // shouldn't happen unless someone adds a new format and doesn't implement it here
468       case _ => throw new UnsupportedOperationException(s"Export for '${options.format}' is not implemented")
469     }
470 
471     override def start(sft: SimpleFeatureType): Unit = exporter.start(sft)
472 
473     override def export(features: Iterator[SimpleFeature]): Option[Long] = exporter.export(features)
474 
475     def bytes: Long = {
476       if (options.format.streaming) {
477         stream.bytes
478       } else {
479         exporter match {
480           case e: ShapefileExporter => e.bytes
481           case _ => PathUtils.getHandle(name).length
482         }
483       }
484     }
485 
486     override def close(): Unit = exporter.close()
487   }

489   /**
490     * Feature exporter that handles chunking output into multiple files
491     *
492     * @param options export options
493     * @param hints query hints
494     * @param chunks number of bytes to write per file
495     */
496   class ChunkedExporter(options: ExportOptions, hints: Hints, chunks: Long)
497       extends FeatureExporter with LazyLogging {
498 
499     private val names = options.file match {
500       case None    => Iterator.continually(None)
501       case Some(f) => new IncrementingFileName(f).map(Option.apply)
502     }
503 
504     private var sft: SimpleFeatureType = _
505     private var exporter: Exporter = _
506     private var estimator: FileSizeEstimator = _
507     private var count = 0L // number of features written
508     private var total = 0L // sum size of all finished chunks
509 
510     override def start(sft: SimpleFeatureType): Unit = {
511       this.sft = sft
512       estimator = new FileSizeEstimator(chunks, 0.05f, options.format.bytesPerFeature(sft)) // 5% error threshold
513       nextChunk()
514     }
515 
516     override def export(features: Iterator[SimpleFeature]): Option[Long] = export(features, None)
517 
518     override def close(): Unit = if (exporter != null) { exporter.close() }
519 
520     private def nextChunk(): Unit = {
521       if (exporter != null) {
522         exporter.close()
523         // adjust our estimate to account for the actual bytes written
524         // do this after closing the exporter to account for footers, batching, etc
525         val written = exporter.bytes
526         estimator.update(written, count)
527         total += written
528       }
529       exporter = new Exporter(options.copy(file = names.next), hints)
530       exporter.start(sft)
531       count = 0L
532     }
533 
534     @tailrec
535     private def export(features: Iterator[SimpleFeature], result: Option[Long]): Option[Long] = {
536       var estimate = estimator.estimate(exporter.bytes)
537       val counter = features.takeWhile { _ => count += 1; estimate -= 1; estimate >= 0 }
538       val exported = exporter.export(counter) match {
539         case None    => result
540         case Some(c) => result.map(_ + c).orElse(Some(c))
541       }
542       if (features.isEmpty) {
543         exported
544       } else {
545         // if it's a countable format, the bytes should be available now and we can compare to our chunk size
546         // otherwise, the bytes aren't generally available until after closing the writer,
547         // so we have to go with our initial estimate and adjust after the first chunk
548         if (options.format.countable) {
549           val bytes = exporter.bytes
550           if (estimator.done(bytes)) {
551             nextChunk()
552           } else {
553             estimator.update(bytes, count)
554           }
555         } else {
556           nextChunk()
557         }
558         export(features, exported)
559       }
560     }
561   }
