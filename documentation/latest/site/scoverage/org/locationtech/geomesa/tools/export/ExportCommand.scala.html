77   private def export(ds: DS): JobResult = {
78     // for file data stores, handle setting the default type name so the user doesn't have to
79     for {
80       p <- Option(params).collect { case p: ProvidedTypeNameParam => p }
81       f <- Option(ds).collect { case f: FileDataStore => f }
82     } { p.featureName = f.getSchema.getTypeName }
83 
84     val options = ExportOptions(params)
85     val remote = options.file.exists(PathUtils.isRemote)
86     val reducers = Option(params.reducers).map(_.intValue()).getOrElse(-1)
87     val mode = params.mode.getOrElse(if (reducers == -1) { RunModes.Local } else { RunModes.Distributed })
88     val chunks = Option(params.chunkSize).map(_.longValue())
89 
90     // do some validation up front
91     if (options.file.isEmpty && !options.format.streaming) {
92       throw new ParameterException(s"Format '${options.format}' requires file-based output, please use --output")
93     } else if (remote && options.format == ExportFormat.Shp) {
94       throw new ParameterException("Shape file export is not supported for distributed file systems")
95     } else if (!remote && mode == RunModes.Distributed) {
96       throw new ParameterException("Distributed export requires an output file in a distributed file system")
97     } else if (mode == RunModes.Distributed && params.maxFeatures != null) {
98       throw new ParameterException("Distributed export does not support --max-features")
99     }
100 
101     val sft = ds.getSchema(params.featureName)
102     if (sft == null) {
103       throw new ParameterException(s"Schema '${params.featureName}' does not exist in the store")
104     }
105 
106     val query = ExportCommand.createQuery(sft, params)
107 
108     mode match {
109       case RunModes.Local =>
110         options.file.map(PathUtils.getHandle).foreach { file =>
111           if (file.exists) {
112             if (params.force) {
113               Command.user.warn(s"Output file '${file.path}' exists - deleting it")
114             } else if (!Prompt.confirm(s"WARNING Output file '${file.path}' exists, delete it and continue (y/n)? ")) {
115               throw new ParameterException(s"Output file '${file.path}' exists")
116             }
117             file.delete()
118           }
119         }
120         val exporter = chunks match {
121           case None    => new Exporter(options, query.getHints)
122           case Some(c) => new ChunkedExporter(options, query.getHints, c)
123         }
124         val count = try { export(ds, query, exporter, !params.suppressEmpty) } finally { exporter.close() }
125         val outFile = options.file match {
126           case None => "standard out"
127           case Some(f) if chunks.isDefined => PathUtils.getBaseNameAndExtension(f).productIterator.mkString("_*")
128           case Some(f) => f
129         }
130 
131         JobSuccess(s"Feature export complete to $outFile", count.map(CountKey -> _).toMap)
132 
133       case RunModes.Distributed =>
134         val job = Job.getInstance(new Configuration, "GeoMesa Tools Export")
135 
136         // for remote jobs, don't push down format transforms, to enable counting and global sorting
137         val hints = new Hints(query.getHints)
138         ExportCommand.disableAggregation(sft, query.getHints)
139 
140         configure(job, ds, query) // note: do this first so that input format is set for the TotalOrderPartitioner
141 
142         // note: these confs should be set by the input format
143         val reduce = Seq(GeoMesaConfigurator.Keys.FeatureReducer, GeoMesaConfigurator.Keys.Sorting).filter { key =>
144           job.getConfiguration.get(key) != null
145         }
146 
147         if (reducers < 1 && reduce.nonEmpty) {
148           if (reduce.contains(GeoMesaConfigurator.Keys.Sorting)) {
149             throw new ParameterException("Distributed export sorting requires --num-reducers")
150           } else {
151             throw new ParameterException(s"Distributed export format '${options.format}' requires --num-reducers")
152           }
153         }
154 
155         // must be some due to our remote check
156         val file = options.file.getOrElse(throw new IllegalStateException("file should be Some"))
157         val output = new Path(PathUtils.getUrl(file).toURI).getParent
158 
159         // file output format doesn't generally let you write to an existing directory
160         val context = FileContext.getFileContext(output.toUri, job.getConfiguration)
161         if (context.util.exists(output)) {
162           val warning = s"Output directory '$output' exists - files may be overwritten"
163           if (params.force) {
164             Command.user.warn(warning)
165           } else if (!Prompt.confirm(s"WARNING $warning. Continue anyway (y/n)? ")) {
166             if (Prompt.confirm("WARNING DATA MAY BE LOST - delete directory and proceed with export (y/n)? ")) {
167               context.delete(output, true)
168             } else {
169               throw new ParameterException(s"Output directory '$output' exists")
170             }
171           }
172         }
173 
174         val filename = FilenameUtils.getName(file)
175 
176         // note: use our original hints, which have the aggregating keys
177         ExportJob.configure(job, connection, sft, hints, filename, output, options.format, options.headers,
178           chunks, options.gzip, reducers, libjars(options.format), libjarsPaths)
179 
180         val reporter = TerminalCallback()
181         JobRunner.run(job, reporter, ExportJob.Counters.mapping(job), ExportJob.Counters.reducing(job)).merge {
182           Some(JobSuccess(s"Feature export complete to $output", Map(CountKey -> ExportJob.Counters.count(job))))
183         }
184 
185       case _ => throw new UnsupportedOperationException() // someone added a run mode and didn't implement it here...
186     }
187   }

214     }

247   /**
248     * Create the query to execute
249     *
250     * @param sft simple feature type
251     * @param params parameters
252     * @return
253     */
254   def createQuery(sft: SimpleFeatureType, params: ExportParams): Query = {
255     val typeName = Option(params).collect { case p: TypeNameParam => p.featureName }.orNull
256     val filter = Option(params.cqlFilter).getOrElse(Filter.INCLUDE)
257 
258     val query = new Query(typeName, filter)
259     val hints = query.getHints
260     Option(params.maxFeatures).map(Int.unbox).foreach(query.setMaxFeatures)
261     Option(params).collect { case p: OptionalIndexParam => p }.foreach { p =>
262       Option(p.index).foreach { index =>
263         logger.debug(s"Using index $index")
264         hints.put(QueryHints.QUERY_INDEX, index)
265       }
266     }
267 
268     Option(params.hints).foreach { hintStrings =>
269       hints.put(Hints.VIRTUAL_TABLE_PARAMETERS, hintStrings)
270       ViewParams.setHints(query)
271     }
272 
273     if (params.outputFormat == ExportFormat.Arrow) {
274       hints.put(QueryHints.ARROW_ENCODE, java.lang.Boolean.TRUE)
275     } else if (params.outputFormat == ExportFormat.Bin) {
276       // if not specified in hints, set it here to trigger the bin query
277       if (!hints.containsKey(QueryHints.BIN_TRACK)) {
278         hints.put(QueryHints.BIN_TRACK, "id")
279       }
280       if (!hints.containsKey(QueryHints.BIN_DTG)) {
281         sft.getDtgField.foreach(hints.put(QueryHints.BIN_DTG, _))
282       }
283     } else if (params.outputFormat == ExportFormat.Leaflet) {
284       configureLeafletExport(params)
285     }
286 
287     val attributes: Array[String] = {
288       val combined = params.attributes.asScala ++ params.transforms.asScala
289       if (combined.nonEmpty) {
290         val (id, attributes) = combined.partition(_.equalsIgnoreCase("id"))
291         if (id.isEmpty && !hints.containsKey(QueryHints.ARROW_INCLUDE_FID)) {
292           // note: we also use this hint for delimited text export
293           hints.put(QueryHints.ARROW_INCLUDE_FID, java.lang.Boolean.FALSE)
294         }
295         attributes.toArray
296       } else if (params.outputFormat == ExportFormat.Bin) {
297         BinAggregatingScan.propertyNames(hints, sft).toArray
298       } else {
299         null // all props
300       }
301     }
302     query.setPropertyNames(attributes: _*)
303 
304     if (!params.sortFields.isEmpty) {
305       val fields = params.sortFields.asScala.toSeq
306       if (fields.exists(a => sft.indexOf(a) == -1)) {
307         val errors = fields.filter(a => sft.indexOf(a) == -1)
308         throw new ParameterException(s"Invalid sort attribute${if (errors.lengthCompare(1) == 0) "" else "s"}: " +
309             errors.mkString(", "))
310       }
311       val order = if (params.sortDescending) { SortOrder.DESCENDING } else { SortOrder.ASCENDING }
312       query.setSortBy(fields.map(f => org.locationtech.geomesa.filter.ff.sort(f, order)): _*)
313     } else if (hints.isArrowQuery) {
314       hints.getArrowSort.foreach { case (f, r) =>
315         val order = if (r) { SortOrder.DESCENDING } else { SortOrder.ASCENDING }
316         query.setSortBy(org.locationtech.geomesa.filter.ff.sort(f, order))
317       }
318     }
319 
320     logger.debug(s"Applying CQL filter ${ECQL.toCQL(filter)}")
321     logger.debug(s"Applying transform ${Option(attributes).map(_.mkString(",")).orNull}")
322 
323     QueryRunner.configureDefaultQuery(sft, query)
324   }

375     * regular sort hints

377     * @param sft simple feature type

381   private def disableAggregation(sft: SimpleFeatureType, hints: Hints): Unit = {
382     if (hints.isArrowQuery) {
383       hints.remove(QueryHints.ARROW_ENCODE)
384       val sort = hints.remove(QueryHints.ARROW_SORT_FIELD).asInstanceOf[String]
385       if (sort != null) {
386         val reverse = Option(hints.remove(QueryHints.ARROW_SORT_REVERSE).asInstanceOf[java.lang.Boolean])
387         val order = if (reverse.exists(_.booleanValue)) { SortOrder.DESCENDING } else { SortOrder.ASCENDING }
388         val hint = org.locationtech.geomesa.filter.ff.sort(sort, order)
389         hints.put(QueryHints.Internal.SORT_FIELDS, QueryHints.Internal.toSortHint(Array(hint)))
390       }
391     } else if (hints.isBinQuery) {
392       hints.remove(QueryHints.BIN_TRACK)
393       if (hints.isBinSorting) {
394         hints.getBinDtgField.orElse(sft.getDtgField).foreach { dtg =>
395           val hint = org.locationtech.geomesa.filter.ff.sort(dtg, SortOrder.ASCENDING)
396           hints.put(QueryHints.Internal.SORT_FIELDS, QueryHints.Internal.toSortHint(Array(hint)))
397         }
398       }
399     }

425     * Single exporter that handles the command options and delegates to the correct implementation
426     *
427     * @param options options
428     * @param hints query hints
429     */
430   class Exporter(options: ExportOptions, hints: Hints) extends FeatureExporter {
431 
432     // used only for streaming export formats
433     private lazy val stream: ByteCounterStream = {
434       // avro compression is handled differently, see AvroExporter below
435       val gzip = options.gzip.filter(_ => options.format != ExportFormat.Avro && options.format != ExportFormat.AvroNative)
436       val handle = options.file match {
437         case None => StdInHandle.get() // writes to std out
438         case Some(f) => PathUtils.getHandle(f)
439       }
440       if (options.writeEmptyFiles) {
441         new ExportStream(handle, gzip)
442       } else {
443         new LazyExportStream(handle, gzip)
444       }
445     }
446 
447     // used only for file-based export formats
448     private lazy val name = options.file.getOrElse {
449       // should have been validated already...
450       throw new IllegalStateException("Export format requires a file but none was specified")
451     }
452 
453     // noinspection ComparingUnrelatedTypes
454     private lazy val fids = !Option(hints.get(QueryHints.ARROW_INCLUDE_FID)).contains(java.lang.Boolean.FALSE)
455 
456     private val exporter = options.format match {
457       case ExportFormat.Arrow      => new ArrowExporter(stream, hints)
458       case ExportFormat.Avro       => new AvroExporter(stream, options.gzip)
459       case ExportFormat.AvroNative => new AvroExporter(stream, options.gzip, Set(SerializationOption.NativeCollections))
460       case ExportFormat.Bin        => new BinExporter(stream, hints)
461       case ExportFormat.Csv        => DelimitedExporter.csv(stream, options.headers, fids)
462       case ExportFormat.Gml2       => GmlExporter.gml2(stream)
463       case ExportFormat.Gml3       => GmlExporter(stream)
464       case ExportFormat.Json       => new GeoJsonExporter(stream)
465       case ExportFormat.Leaflet    => new LeafletMapExporter(stream)
466       case ExportFormat.Null       => NullExporter
467       case ExportFormat.Orc        => new OrcFileSystemExporter(name)
468       case ExportFormat.Parquet    => new ParquetFileSystemExporter(name)
469       case ExportFormat.Shp        => new ShapefileExporter(new File(name))
470       case ExportFormat.Tsv        => DelimitedExporter.tsv(stream, options.headers, fids)
471       // shouldn't happen unless someone adds a new format and doesn't implement it here
472       case _ => throw new UnsupportedOperationException(s"Export for '${options.format}' is not implemented")
473     }
474 
475     override def start(sft: SimpleFeatureType): Unit = exporter.start(sft)
476 
477     override def export(features: Iterator[SimpleFeature]): Option[Long] = exporter.export(features)
478 
479     def bytes: Long = {
480       if (options.format.streaming) {
481         stream.bytes
482       } else {
483         exporter match {
484           case e: ShapefileExporter => e.bytes
485           case _ => PathUtils.getHandle(name).length
486         }
487       }
488     }
489 
490     override def close(): Unit = exporter.close()
491   }

493   /**
494     * Feature exporter that handles chunking output into multiple files
495     *
496     * @param options export options
497     * @param hints query hints
498     * @param chunks number of bytes to write per file
499     */
500   class ChunkedExporter(options: ExportOptions, hints: Hints, chunks: Long)
501       extends FeatureExporter with LazyLogging {
502 
503     private val names = options.file match {
504       case None    => Iterator.continually(None)
505       case Some(f) => new IncrementingFileName(f).map(Option.apply)
506     }
507 
508     private var sft: SimpleFeatureType = _
509     private var exporter: Exporter = _
510     private var estimator: FileSizeEstimator = _
511     private var count = 0L // number of features written
512     private var total = 0L // sum size of all finished chunks
513 
514     override def start(sft: SimpleFeatureType): Unit = {
515       this.sft = sft
516       estimator = new FileSizeEstimator(chunks, 0.05f, options.format.bytesPerFeature(sft)) // 5% error threshold
517       nextChunk()
518     }
519 
520     override def export(features: Iterator[SimpleFeature]): Option[Long] = export(features, None)
521 
522     override def close(): Unit = if (exporter != null) { exporter.close() }
523 
524     private def nextChunk(): Unit = {
525       if (exporter != null) {
526         exporter.close()
527         // adjust our estimate to account for the actual bytes written
528         // do this after closing the exporter to account for footers, batching, etc
529         val written = exporter.bytes
530         estimator.update(written, count)
531         total += written
532       }
533       exporter = new Exporter(options.copy(file = names.next), hints)
534       exporter.start(sft)
535       count = 0L
536     }
537 
538     @tailrec
539     private def export(features: Iterator[SimpleFeature], result: Option[Long]): Option[Long] = {
540       var estimate = estimator.estimate(exporter.bytes)
541       val counter = features.takeWhile { _ => count += 1; estimate -= 1; estimate >= 0 }
542       val exported = exporter.export(counter) match {
543         case None    => result
544         case Some(c) => result.map(_ + c).orElse(Some(c))
545       }
546       if (features.isEmpty) {
547         exported
548       } else {
549         // if it's a countable format, the bytes should be available now and we can compare to our chunk size
550         // otherwise, the bytes aren't generally available until after closing the writer,
551         // so we have to go with our initial estimate and adjust after the first chunk
552         if (options.format.countable) {
553           val bytes = exporter.bytes
554           if (estimator.done(bytes)) {
555             nextChunk()
556           } else {
557             estimator.update(bytes, count)
558           }
559         } else {
560           nextChunk()
561         }
562         export(features, exported)
563       }
564     }
565   }
