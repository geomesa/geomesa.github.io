29   override def execute(): Unit = withDataStore { ds =>
30     val sft = getSchema(ds)
31     if (sft == null) {
32       val msg = params match {
33         case p: TypeNameParam => s"Feature '${p.featureName}' not found"
34         case _ => s"Feature type not found"
35       }

38     Command.user.info(s"Describing attributes of feature '${sft.getTypeName}'")
39     describe(ds, sft, Command.output.info)
40   }

42   protected def getSchema(ds: DS): SimpleFeatureType = params match {
43     case p: TypeNameParam => ds.getSchema(p.featureName)
44   }

46   protected def describe(ds: DS, sft: SimpleFeatureType, output: String => Unit): Unit = {
47     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType._
48 
49     val indices = ds match {
50       case gmds: GeoMesaDataStore[_] => gmds.manager.indices(sft)
51       case _ => Seq.empty
52     }
53 
54     val namesAndDescriptions = sft.getAttributeDescriptors.asScala.map { descriptor =>
55       val name = descriptor.getLocalName
56       val description = ArrayBuffer.empty[String]
57 
58       indices.foreach {
59         case i if (i.name == Z3Index.name || i.name == XZ3Index.name) && i.attributes.take(2).contains(name) =>
60           description.append("(Spatio-temporally indexed)")
61 
62         case i if (i.name == Z2Index.name || i.name == XZ2Index.name) && i.attributes.headOption.contains(name) =>
63           description.append("(Spatially indexed)")
64 
65         case i if i.name == AttributeIndex.name && i.attributes.headOption.contains(name) =>
66           description.append("(Attribute indexed)")
67 
68         case i if i.name == AttributeIndex.JoinIndexName && i.attributes.headOption.contains(name) =>
69           description.append("(Attribute indexed - join)")
70 
71         case _ => // no-op
72       }
73 
74       Option(descriptor.getDefaultValue).foreach(v => description.append(s"Default Value: $v"))
75       (name, descriptor.getType.getBinding.getSimpleName, description.mkString(" "))
76     }
77 
78     val maxName = namesAndDescriptions.map(_._1.length).max
79     val maxType = namesAndDescriptions.map(_._2.length).max
80     namesAndDescriptions.foreach { case (n, t, d) =>
81       output(s"${n.padTo(maxName, ' ')} | ${t.padTo(maxType, ' ')} $d")
82     }
83 
84     val userData = sft.getUserData
85     if (!userData.isEmpty) {
86       output("\nUser data:")
87       val namesAndValues = userData.asScala.toSeq.map { case (k, v) =>
88         if (k == SimpleFeatureTypes.Configs.Keywords) {
89           (SimpleFeatureTypes.Configs.Keywords, sft.getKeywords.mkString("'", "', '", "'"))
90         } else {
91           (s"$k", s"$v")
92         }
93       }
94       val maxName = namesAndValues.map(_._1.length).max
95       namesAndValues.sortBy(_._1).foreach { case (n, v) =>
96         output(s"  ${n.padTo(maxName, ' ')} | $v")
97       }
98     }
99   }
