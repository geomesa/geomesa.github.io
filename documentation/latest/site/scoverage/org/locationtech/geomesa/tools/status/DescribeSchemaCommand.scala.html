31     val sft = getSchema(ds)

33       val msg = params match {

47   protected def describe(ds: DS, sft: SimpleFeatureType, logger: Logger): Unit = {
48     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType._
49 
50     val indices = ds match {
51       case gmds: GeoMesaDataStore[_] => gmds.manager.indices(sft)

54 
55     val namesAndDescriptions = sft.getAttributeDescriptors.asScala.map { descriptor =>
56       val name = descriptor.getLocalName
57       val description = ArrayBuffer.empty[String]
58 
59       indices.foreach {
60         case i if (i.name == Z3Index.name || i.name == XZ3Index.name) && i.attributes.take(2).contains(name) =>
61           description.append("(Spatio-temporally indexed)")
62 
63         case i if (i.name == Z2Index.name || i.name == XZ2Index.name) && i.attributes.headOption.contains(name) =>
64           description.append("(Spatially indexed)")
65 
66         case i if i.name == AttributeIndex.name && i.attributes.headOption.contains(name) =>
67           description.append("(Attribute indexed)")
68 
69         case i if i.name == AttributeIndex.JoinIndexName && i.attributes.headOption.contains(name) =>
70           description.append("(Attribute indexed - join)")
71 
72         case _ => // no-op
73       }
74 
75       Option(descriptor.getDefaultValue).foreach(v => description.append(s"Default Value: $v"))
76       (name, descriptor.getType.getBinding.getSimpleName, description.mkString(" "))
77     }
78 
79     val maxName = namesAndDescriptions.map(_._1.length).max
80     val maxType = namesAndDescriptions.map(_._2.length).max
81     namesAndDescriptions.foreach { case (n, t, d) =>
82       logger.info(s"${n.padTo(maxName, ' ')} | ${t.padTo(maxType, ' ')} $d")
83     }
84 
85     val userData = sft.getUserData
86     if (!userData.isEmpty) {
87       logger.info("\nUser data:")
88       val namesAndValues = userData.asScala.toSeq.map { case (k, v) =>
89         if (k == SimpleFeatureTypes.Configs.Keywords) {
90           (SimpleFeatureTypes.Configs.Keywords, sft.getKeywords.mkString("'", "', '", "'"))
91         } else {
92           (s"$k", s"$v")
93         }
94       }
95       val maxName = namesAndValues.map(_._1.length).max
96       namesAndValues.sortBy(_._1).foreach { case (n, v) =>
97         logger.info(s"  ${n.padTo(maxName, ' ')} | $v")
98       }
99     }
100   }
