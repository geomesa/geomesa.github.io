47   private def convertAndExport(): Option[Long] = {
48     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
49 
50     import scala.collection.JavaConverters._
51 
52     val inputs: Inputs = {
53       val files = Inputs(params.files.asScala.toSeq)
54       if (files.stdin && !StdInHandle.isAvailable) {
55         if (files.paths.isEmpty) {
56           throw new ParameterException("Missing option: <files>... is required, or use `-` to ingest from standard in")
57         } else {
58           Command.user.info("Waiting for input...")
59           while (!StdInHandle.isAvailable) {
60             Thread.sleep(10)
61           }
62         }
63       }
64       if (params.srcList) { files.asSourceList } else { files }
65     }
66 
67     val format = IngestCommand.getDataFormat(params, inputs.paths)
68 
69     // use .get to re-throw the exception if we fail
70     IngestCommand.getSftAndConverter(params, inputs, format, None).get.flatMap { case (sft: SimpleFeatureType, config: com.typesafe.config.Config) =>
71       val files = inputs.handles.iterator
72 
73       WithClose(SimpleFeatureConverter(sft, config)) { converter =>
74         val ec = converter.createEvaluationContext()
75         val query = ExportCommand.createQuery(sft, params)
76         val exporter = Option(params.chunkSize) match {
77           case None    => new Exporter(ExportOptions(params), query.getHints)
78           case Some(c) => new ChunkedExporter(ExportOptions(params), query.getHints, c)
79         }
80         try {
81           val count = WithClose(ConvertCommand.convertFeatures(files, converter, ec, query)) { iter =>
82             if (!params.suppressEmpty || iter.hasNext) {
83               exporter.start(query.getHints.getReturnSft)
84               exporter.export(iter)
85             } else {
86               Some(0L)
87             }
88           }
89           val records = ec.line - (if (params.noHeader) { 0 } else { params.files.size })
90           Command.user.info(s"Converted ${getPlural(records, "line")} "
91               + s"with ${getPlural(ec.success.getCount, "success", "successes")} "
92               + s"and ${getPlural(ec.failure.getCount, "failure")}")
93           count
94         } finally {
95           exporter.close()
96         }
97       }
98     }
99   }

104   /**
105     * Convert features
106     *
107     * @param files inputs
108     * @param converter converter
109     * @param ec evaluation context
110     * @param query query used to filter/transform inputs
111     * @return
112     */
113   def convertFeatures(
114       files: Iterator[FileHandle],
115       converter: SimpleFeatureConverter,
116       ec: EvaluationContext,
117       query: Query): CloseableIterator[SimpleFeature] = {
118 
119     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
120 
121     def convert(): CloseableIterator[SimpleFeature] = CloseableIterator(files).flatMap { file =>
122       file.open.flatMap { case (name, is) =>
123         val params = EvaluationContext.inputFileParam(name.getOrElse(file.path))
124         val context = converter.createEvaluationContext(params, ec.success, ec.failure)
125         val features = converter.process(is, context)
126         CloseableIterator(features, { features.close(); ec.line += context.line })
127       }
128     }
129 
130     def filter(iter: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] =
131       if (query.getFilter == Filter.INCLUDE) { iter } else { iter.filter(query.getFilter.evaluate) }
132 
133     def limit(iter: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = {
134       query.getHints.getMaxFeatures match {
135         case None    => iter
136         case Some(m) => iter.take(m)
137       }
138     }
139 
140     def transform(iter: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = {
141       import org.locationtech.geomesa.filter.filterToString
142 
143       val stats = new RunnableStats(null) {
144         override protected def query[T <: Stat](sft: SimpleFeatureType, ignored: Filter, stats: String, queryHints: Hints) : Option[T] = {
145           val stat = Stat(sft, stats).asInstanceOf[T]
146           try {
147             WithClose(limit(filter(convert())))(_.foreach(stat.observe))
148             Some(stat)
149           } catch {
150             case e: Exception =>
151               logger.error(s"Error running stats query with stats '$stats' and filter '${filterToString(ignored)}'", e)
152               None
153           }
154         }
155       }
156       val hook = Some(ArrowDictionaryHook(stats, Option(query.getFilter).filter(_ != Filter.INCLUDE)))
157       LocalQueryRunner.transform(converter.targetSft, iter, query.getHints.getTransform, query.getHints, hook)
158     }
159 
160     transform(limit(filter(convert())))
161   }
