46   private def convertAndExport(): Option[Long] = {
47     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
48 
49     import scala.collection.JavaConverters._
50 
51     val inputs: Inputs = {
52       val files = Inputs(params.files.asScala.toSeq)
53       if (files.stdin && !StdInHandle.isAvailable) {
54         if (files.paths.isEmpty) {
55           throw new ParameterException("Missing option: <files>... is required, or use `-` to ingest from standard in")
56         } else {
57           Command.user.info("Waiting for input...")
58           while (!StdInHandle.isAvailable) {
59             Thread.sleep(10)
60           }
61         }
62       }
63       if (params.srcList) { files.asSourceList } else { files }
64     }
65 
66     val format = IngestCommand.getDataFormat(params, inputs.paths)
67 
68     // use .get to re-throw the exception if we fail
69     IngestCommand.getSftAndConverter(params, inputs, format, None).get.flatMap { case (sft: SimpleFeatureType, config: com.typesafe.config.Config) =>
70       val files = inputs.handles.iterator
71 
72       WithClose(SimpleFeatureConverter(sft, config)) { converter =>
73         val ec = converter.createEvaluationContext()
74         val query = ExportCommand.createQuery(sft, params)
75         val exporter = Option(params.chunkSize) match {
76           case None    => new Exporter(ExportOptions(params), query.getHints)
77           case Some(c) => new ChunkedExporter(ExportOptions(params), query.getHints, c)
78         }
79         try {
80           val count = WithClose(ConvertCommand.convertFeatures(files, converter, ec, query)) { iter =>
81             if (!params.suppressEmpty || iter.hasNext) {
82               exporter.start(query.getHints.getReturnSft)
83               exporter.export(iter)
84             } else {
85               Some(0L)
86             }
87           }
88           val records = ec.line - (if (params.noHeader) { 0 } else { params.files.size })
89           Command.user.info(s"Converted ${getPlural(records, "line")} "
90               + s"with ${getPlural(ec.success.getCount, "success", "successes")} "
91               + s"and ${getPlural(ec.failure.getCount, "failure")}")
92           count
93         } finally {
94           exporter.close()
95         }
96       }
97     }
98   }

103   /**
104     * Convert features

106     * @param files inputs
107     * @param converter converter
108     * @param ec evaluation context
109     * @param query query used to filter/transform inputs
110     * @return
111     */
112   def convertFeatures(
113       files: Iterator[FileHandle],
114       converter: SimpleFeatureConverter,
115       ec: EvaluationContext,
116       query: Query): CloseableIterator[SimpleFeature] = {
117 
118     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
119 
120     def convert(): CloseableIterator[SimpleFeature] = CloseableIterator(files).flatMap { file =>
121       file.open.flatMap { case (name, is) =>
122         val params = EvaluationContext.inputFileParam(name.getOrElse(file.path))
123         val context = converter.createEvaluationContext(params, ec.success, ec.failure)
124         val features = converter.process(is, context)
125         CloseableIterator(features, { features.close(); ec.line += context.line })
126       }
127     }
128 
129     def filter(iter: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] =
130       if (query.getFilter == Filter.INCLUDE) { iter } else { iter.filter(query.getFilter.evaluate) }
131 
132     def limit(iter: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = {
133       query.getHints.getMaxFeatures match {
134         case None    => iter
135         case Some(m) => iter.take(m)
136       }
137     }
138 
139     def transform(iter: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = {
140       import org.locationtech.geomesa.filter.filterToString
141 
142       val stats = new RunnableStats(null) {
143         override protected def query[T <: Stat](sft: SimpleFeatureType, ignored: Filter, stats: String, queryHints: Hints) : Option[T] = {
144           val stat = Stat(sft, stats).asInstanceOf[T]
145           try {
146             WithClose(limit(filter(convert())))(_.foreach(stat.observe))
147             Some(stat)
148           } catch {
149             case e: Exception =>
150               logger.error(s"Error running stats query with stats '$stats' and filter '${filterToString(ignored)}'", e)
151               None
152           }
153         }
154       }
155       LocalQueryRunner.transform(converter.targetSft, iter, query.getHints.getTransform, query.getHints)
156     }
157 
158     transform(limit(filter(convert())))
159   }
