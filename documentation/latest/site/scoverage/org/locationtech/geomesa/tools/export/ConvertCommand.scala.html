45   private def convertAndExport(): Option[Long] = {
46     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
47 
48     import scala.collection.JavaConverters._
49 
50     val inputs: Inputs = {
51       val files = Inputs(params.files.asScala.toSeq)
52       if (files.stdin && !StdInHandle.isAvailable) {
53         if (files.paths.isEmpty) {
54           throw new ParameterException("Missing option: <files>... is required, or use `-` to ingest from standard in")
55         } else {
56           Command.user.info("Waiting for input...")
57           while (!StdInHandle.isAvailable) {
58             Thread.sleep(10)
59           }
60         }
61       }
62       if (params.srcList) { files.asSourceList } else { files }
63     }
64 
65     val format = IngestCommand.getDataFormat(params, inputs.paths)
66 
67     // use .get to re-throw the exception if we fail
68     IngestCommand.getSftAndConverter(params, inputs, format, None).get.flatMap { case (sft: SimpleFeatureType, config: com.typesafe.config.Config) =>
69       val files = inputs.handles.iterator
70 
71       WithClose(SimpleFeatureConverter(sft, config)) { converter =>
72         val results = new ResultTracker()
73         val query = ExportCommand.createQuery(sft, params)
74         val exporter = Option(params.chunkSize) match {
75           case None    => new Exporter(ExportOptions(params), query.getHints)
76           case Some(c) => new ChunkedExporter(ExportOptions(params), query.getHints, c)
77         }
78         try {
79           val count = WithClose(ConvertCommand.convertFeatures(files, converter, results, query)) { iter =>
80             if (!params.suppressEmpty || iter.hasNext) {
81               exporter.start(query.getHints.getReturnSft)
82               exporter.export(iter)
83             } else {
84               Some(0L)
85             }
86           }
87           Command.user.info(s"Converted ${getPlural(results.lines, "line")} "
88               + s"with ${getPlural(results.successes, "success", "successes")} "
89               + s"and ${getPlural(results.failures, "failure")}")
90           count
91         } finally {
92           exporter.close()
93         }
94       }
95     }
96   }

101   /**
102     * Convert features
103     *
104     * @param files inputs
105     * @param converter converter
106     * @param tracker holder for returning conversion results
107     * @param query query used to filter/transform inputs
108     * @return
109     */
110   private def convertFeatures(
111       files: Iterator[FileHandle],
112       converter: SimpleFeatureConverter,
113       tracker: ResultTracker,
114       query: Query): CloseableIterator[SimpleFeature] = {
115 
116     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
117 
118     def convert(): CloseableIterator[SimpleFeature] = CloseableIterator(files).flatMap { file =>
119       file.open.flatMap { case (name, is) =>
120         val params = EvaluationContext.inputFileParam(name.getOrElse(file.path))
121         val context = converter.createEvaluationContext(params)
122         val features = converter.process(is, context)
123         CloseableIterator(features, { features.close(); tracker.update(context) })
124       }
125     }
126 
127     def filter(iter: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] =
128       if (query.getFilter == Filter.INCLUDE) { iter } else { iter.filter(query.getFilter.evaluate) }
129 
130     def limit(iter: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = {
131       query.getHints.getMaxFeatures match {
132         case None    => iter
133         case Some(m) => iter.take(m)
134       }
135     }
136 
137     def transform(iter: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] =
138       LocalQueryRunner.transform(converter.targetSft, iter, query.getHints.getTransform, query.getHints)
139 
140     transform(limit(filter(convert())))
141   }
142 
143   private class ResultTracker {
144     var failures: Long = 0
145     var successes: Long = 0
146     var lines: Long = 0
147 
148     def update(context: EvaluationContext): Unit = {
149       failures += context.stats.failure(0)
150       successes += context.stats.success(0)
151       lines += context.line
152     }
153   }
