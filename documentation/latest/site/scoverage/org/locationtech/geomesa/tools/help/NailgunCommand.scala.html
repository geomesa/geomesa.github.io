38   class NailgunStatsCommand extends Command with NailgunAware {
39 
40     import NailgunStatsCommand.{Header, toSeconds}
41 
42     override val name: String = "stats"
43     override val params: NailgunStatsParams = new NailgunStatsParams()
44 
45     override def execute(): Unit = {
46       nailgun match {
47         case None => Command.output.info("Nailgun server not enabled")
48         case Some(ng) =>
49           val widths = Header.map(_.length).toArray
50           val stats = ng.stats.map { stat =>
51               val strings =
52                 Seq(
53                   stat.name,
54                   s"${stat.active}",
55                   s"${stat.complete}",
56                   f"${toSeconds(stat.mean)}%2.2f",
57                   f"${toSeconds(stat.median)}%2.2f",
58                   f"${toSeconds(stat.n95)}%2.2f",
59                   f"${stat.rate * 60}%2.2f"
60                 )
61             var i = 0
62             while (i < widths.length) {
63               widths(i) = math.max(widths(i), strings(i).length)
64               i += 1
65             }
66             strings
67           }
68 
69           def pad(seq: Seq[String]): String = {
70             var i = 0
71             val padded = seq.map { s =>
72               val p = s.padTo(widths(i), ' ')
73               i += 1
74               p
75             }
76             padded.mkString(" ")
77           }
78 
79           Command.output.info(s"Uptime ${TextTools.getTime(ng.started)}\n")
80           Command.output.info(pad(Header))
81           stats.sortBy(_.head).foreach(s => Command.output.info(pad(s)))
82       }
83     }
84   }
