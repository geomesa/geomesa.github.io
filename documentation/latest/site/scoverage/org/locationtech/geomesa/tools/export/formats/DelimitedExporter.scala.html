34   override def start(sft: SimpleFeatureType): Unit = {
35     printer = format.print(new OutputStreamWriter(stream.os, StandardCharsets.UTF_8))
36     // write out a header line
37     if (withHeader) {
38       if (includeIds) {
39         printer.print("id")
40       }
41       sft.getAttributeDescriptors.asScala.foreach { descriptor =>
42         printer.print(SimpleFeatureTypes.encodeDescriptor(sft, descriptor))
43       }
44       printer.println()
45       printer.flush()
46     }
47   }
48 
49   override def export(features: Iterator[SimpleFeature]): Option[Long] = {
50     var count = 0L
51     features.foreach { sf =>
52       if (includeIds) {
53         printer.print(sf.getID)
54       }
55       var i = 0
56       while (i < sf.getAttributeCount) {
57         printer.print(stringify(sf.getAttribute(i)))
58         i += 1
59       }
60       printer.println()
61 
62       count += 1
63       if (count % 10000 == 0) {
64         logger.debug(s"wrote $count features")
65       }
66     }
67 
68     printer.flush()
69 
70     logger.debug(s"Exported $count features")
71     Some(count)
72   }
73 
74   override def close(): Unit =  if (printer != null) { printer.close() }
75 
76   private def stringify(obj: Any): String = obj match {
77     case null                   => ""
78     case g: Geometry            => WKTUtils.write(g)
79     case d: Date                => GeoToolsDateFormat.format(Instant.ofEpochMilli(d.getTime).atZone(ZoneOffset.UTC))
80     case l: java.util.List[_]   => l.asScala.map(stringify).mkString(",")
81     case m: java.util.Map[_, _] => m.asScala.map { case (k, v) => s"${stringify(k)}->${stringify(v)}"}.mkString(",")
82     case _                      => obj.toString
83   }
84 }
