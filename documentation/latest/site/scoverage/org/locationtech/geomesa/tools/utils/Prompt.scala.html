13 object Prompt {

15   /**
16     * Abstraction over system console, to allow for unit testing
17     */
18   type SystemConsole = Any {
19     def readLine(): String
20     def readPassword(): Array[Char]
21   }
22 
23   lazy val SystemConsole: SystemConsole = {
24     val console = System.console()
25     if (console == null) {
26       throw new IllegalStateException("Unable to access console..." +
27           "Please ensure stdout is not redirected or --force flag is set")
28     }
29     console
30   }
31 
32   def confirm(
33       msg: String,
34       confirmStrings: List[String] = List("yes", "y"),
35       default: String = "")
36      (implicit console: SystemConsole = SystemConsole): Boolean = {
37     print(msg)
38     var response = console.readLine().toLowerCase.trim
39     if (response.isEmpty) {
40       response = default
41     }
42     confirmStrings.map(_.toLowerCase).contains(response)
43   }
44 
45   // note: user must press enter as java does not support reading single chars from the console
46   def acknowledge(msg: String)(implicit console: SystemConsole = SystemConsole): Unit = {
47     print(msg)
48     console.readLine()
49   }
50 
51   def read(msg: String)(implicit console: SystemConsole = SystemConsole): String = {
52     print(msg)
53     console.readLine().trim
54   }
55 
56   def readPassword()(implicit console: SystemConsole = SystemConsole): String = {
57     print("Password (mask enabled)> ")
58     console.readPassword().mkString
59   }
