26 class NailgunServer(addr: InetAddress, port: Int, sessionPoolSize: Int, timeoutMillis: Int, idleTimeoutMillis: Long)
27     extends NGServer(addr, port, sessionPoolSize, timeoutMillis) {

31   private val registry = new MetricRegistry()
32   private val requests = new Phaser(1) // 1 for the idle timer
33 
34   private val start = System.currentTimeMillis()
35   private val lastRequest = new AtomicLong(System.currentTimeMillis())
36 
37   private val timers: LoadingCache[String, (AtomicInteger, Timer)] = Caffeine.newBuilder().build(
38     new CacheLoader[String, (AtomicInteger, Timer)]() {
39       override def load(k: String): (AtomicInteger, Timer) = (new AtomicInteger(0), registry.timer(k))
40     }
41   )
42 
43   private val idle: Future[_] = CachedThreadPool.submit(new IdleCheck())
44 
45   def execute(command: Command): Unit = {
46     requests.register()
47     lastRequest.set(System.currentTimeMillis())
48     try {
49       command match {
50         case c: NailgunAware => c.nailgun = Some(this)
51         case _ => // no-op
52       }
53       val (active, timer) = timers.get(command.name)
54       active.incrementAndGet()
55       try {
56         timer.time(command)
57       } finally {
58         active.decrementAndGet()
59       }
60     } finally {
61       requests.arriveAndDeregister()
62     }
63   }
64

97   /**
98    * Shuts down the nailgun server if it's idle
99    */
100   class IdleCheck extends Runnable {
101     override def run(): Unit = {
102       var loop = true
103       try {
104         Thread.sleep(idleTimeoutMillis) // wait the initial timeout period before starting to check idle
105       } catch {
106         case _: InterruptedException => loop = false
107       }
108       while (loop) {
109         try {
110           val phase = requests.arrive()
111           if (requests.getPhase == phase) {
112             // there are currently executing requests - wait for them to finish
113             requests.awaitAdvanceInterruptibly(phase)
114             // pause 30 seconds before checking for idle timeout, to allow the client to make another request
115             Thread.sleep(30000)
116           }
117 
118           val remaining = idleTimeoutMillis - (System.currentTimeMillis() - lastRequest.get)
119           if (remaining <= 0) {
120             loop = false
121             NailgunServer.this.shutdown()
122           } else {
123             Thread.sleep(remaining)
124           }
125         } catch {
126           case _: InterruptedException => loop = false
127         }
128       }
129     }
130   }
