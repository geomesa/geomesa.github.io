33   protected def histogram(ds: DS): Unit = {
34     for {
35       p <- Option(params).collect { case p: ProvidedTypeNameParam => p }
36       f <- Option(ds).collect { case f: FileDataStore => f }
37     } { p.featureName = f.getSchema.getTypeName }
38

40     if (sft == null) {
41       throw new ParameterException(s"Schema '${params.featureName}' does not exist")
42     }
43 
44     val attributes = getAttributesFromParams(sft, params)
45     val filter = Option(params.cqlFilter).getOrElse(Filter.INCLUDE)
46     val bins = Option(params.bins).map(_.intValue)
47 
48     val histograms = if (params.exact) {
49       val bounds = scala.collection.mutable.Map.empty[String, (Any, Any)]
50       attributes.foreach { attribute =>
51         ds.stats.getMinMax[Any](sft, attribute).foreach { b =>
52           bounds.put(attribute, if (b.min == b.max) Histogram.buffer(b.min) else b.bounds)
53         }
54       }
55 
56       if (bounds.size != attributes.size) {
57         val noBounds = attributes.filterNot(bounds.contains)
58         Command.user.warn(s"Initial bounds are not available for attributes ${noBounds.mkString(", ")}.")
59         var response: Integer = null
60         Command.user.info("Would you like to:\n" +
61                 "  1. Calculate bounds (may be slow)\n" +
62                 "  2. Use default bounds (may be less accurate)\n" +
63                 "  3. Manually enter bounds\n" +
64                 "  4. Cancel operation\n")
65         while (response == null) {
66           val in = Prompt.read("Please enter the number of your choice: ")
67           response = Try(in.toInt.asInstanceOf[Integer]).filter(r => r > 0 && r < 5).getOrElse(null)
68           if (response == null) {
69             Command.user.error("Invalid input. Please enter 1-4.")
70           }
71         }
72         if (response == 1) {
73           Command.user.info("Running bounds query...")
74           ds.stats.getSeqStat[MinMax[Any]](sft, noBounds.map(Stat.MinMax), filter, exact = true).foreach { mm =>
75             bounds.put(mm.property, mm.bounds)
76           }
77         } else if (response == 2) {
78           noBounds.foreach { attribute =>
79             val binding = sft.getDescriptor(attribute).getType.getBinding
80             bounds.put(attribute, GeoMesaStats.defaultBounds(binding))
81           }
82         } else if (response == 3) {
83           noBounds.foreach { attribute =>
84             val ct = sft.getDescriptor(attribute).getType.getBinding
85             var lower: Any = null
86             var upper: Any = null
87             while (lower == null) {
88               lower = FastConverter.convert(Prompt.read(s"Enter initial lower bound for '$attribute': "), ct)
89               if (lower == null) {
90                 Command.user.error(s"Couldn't convert input to appropriate type: ${ct.getSimpleName}")
91               }
92             }
93             while (upper == null) {
94               upper = FastConverter.convert(Prompt.read(s"Enter initial upper bound for '$attribute': "), ct)
95               if (upper == null) {
96                 Command.user.error(s"Couldn't convert input to appropriate type: ${ct.getSimpleName}")
97               }
98             }
99             if (lower == upper) {
100               bounds.put(attribute, Histogram.buffer(lower))
101             } else {
102               bounds.put(attribute, (lower, upper))
103             }
104           }
105         } else {
106           Command.user.info("Operation cancelled.")
107           return // cancel operation
108         }
109       }
110 
111       Command.user.info("Running stat query...")
112       val length = bins.getOrElse(GeoMesaStats.DefaultHistogramSize)
113       val queries = attributes.map { attribute =>
114         val ct = ClassTag[Any](sft.getDescriptor(attribute).getType.getBinding)
115         val (lower, upper) = bounds(attribute)
116         Stat.Histogram[Any](attribute, length, lower, upper)(ct)
117       }
118       ds.stats.getSeqStat[Histogram[Any]](sft, queries, filter, exact = true)
119     } else {
120       if (filter != Filter.INCLUDE) {
121         Command.user.warn("Non-exact stat queries may not fully account for the specified CQL filter")
122       }
123       val queries = attributes.map(attribute => Stat.Histogram[AnyRef](attribute, 0, null, null))
124       ds.stats.getSeqStat[Histogram[Any]](sft, queries, filter).map {
125         case histogram: Histogram[Any] if bins.forall(_ == histogram.length) => histogram
126         case histogram: Histogram[Any] =>
127           val descriptor = sft.getDescriptor(histogram.property)
128           val ct = ClassTag[Any](descriptor.getType.getBinding)
129           val statString = Stat.Histogram[Any](histogram.property, bins.get, histogram.min, histogram.max)(ct)
130           val binned = Stat(sft, statString).asInstanceOf[Histogram[Any]]
131           binned.addCountsFrom(histogram)
132           binned
133       }
134     }
135 
136     attributes.foreach { attribute =>
137       histograms.find(_.property == attribute) match {
138         case None => Command.user.info(s"No histogram available for attribute '$attribute'")
139         case Some(hist) =>
140           if (classOf[Geometry].isAssignableFrom(sft.getDescriptor(attribute).getType.getBinding)) {
141             Command.output.info(StatsHistogramCommand.geomHistToString(attribute, hist.asInstanceOf[Histogram[Geometry]]))
142           } else {
143             StatsHistogramCommand.printHist(hist, sft, attribute)
144           }
145       }
146     }
147   }

163     Command.output.info(s"Binned histogram for '$attribute':")
164     if (stat.isEmpty) {
165       Command.output.info("  No values")
166     } else {

168       (0 until stat.length).foreach { i =>
169         val (min, max) = stat.bounds(i)
170         Command.output.info(s"  [ ${stringify(min)} to ${stringify(max)} ] ${stat.count(i)}")

176     * Creates string containing an ASCII, color-coded map of densities.
177     */
178   def geomHistToString(attribute: String, stat: Histogram[Geometry]): String = {
179     // grid of counts, corresponds to our world map dimensions
180     val counts = Array.fill[Array[Long]](AsciiWorldMapHeight)(Array.fill[Long](AsciiWorldMapLength)(0))
181 
182     // translate histogram values into the grid and also calculate min/max for normalization
183     def putCountsInGrid(): Unit = {
184       var i = 0
185       while (i < stat.length) {
186         val count = stat.count(i)
187         if (count > 0) {
188           val point = stat.medianValue(i).asInstanceOf[Point]
189           val (x, y) = (point.getX, point.getY)
190           val xOffset = (x + 180) / 360 // normalize to 0-1
191           val yOffset = 1 - (y + 90) / 180  // normalize to 0-1 and invert axis
192           // index into map string
193           val j = math.floor(yOffset * AsciiWorldMapHeight).toInt
194           val k = math.floor(xOffset * AsciiWorldMapLength).toInt
195           counts(j)(k) += count
196         }
197         i += 1
198       }
199     }
200 
201     putCountsInGrid()
202 
203     // min/max to normalize our densities
204     val min = counts.map(_.min).min
205     val max = counts.map(_.max).max
206 
207     // normalize a count to 0-1 based on our min/max values
208     def normalize(count: Long): Float = (count - min).toFloat / (max - min)
209     // reverse a normalized percent
210     def denormalize(percent: Float): Long = (percent * (max - min)).toLong + min
211 
212     val sb = new StringBuilder
213 
214     // build up our string - if we have data in a given cell, put a sized circle, else put the map char
215     // in addition to size of circle, use console ANSI colors to mark our densities
216     var i = 0
217     var currentColor: String = null
218     while (i < AsciiWorldMapHeight) {
219       val row = counts(i)
220       var j = 0
221       while (j < AsciiWorldMapLength) {
222         val normalized = normalize(row(j))
223         val (color, char) =
224           if (normalized < .1f) {
225             (Console.RESET, AsciiWorldMap(i)(j))
226           } else if (normalized < .3f) {
227             Threshold1
228           } else if (normalized < .5f) {
229             Threshold2
230           } else if (normalized < .8f) {
231             Threshold3
232           } else {
233             Threshold4
234           }
235         if (color != currentColor) {
236           sb.append(color)
237           currentColor = color
238         }
239         sb.append(char)
240         j += 1
241       }
242       sb.append('\n')
243       i += 1
244     }
245 
246     // write out a key of the actual count ranges represented by each marker
247     sb.append(s"${Console.RESET}\nKey: ")
248     sb.append(s"[${Threshold1._1}${Threshold1._2}${Console.RESET} ${denormalize(0.1f)} to ${denormalize(0.3f)}] ")
249     sb.append(s"[${Threshold2._1}${Threshold2._2}${Console.RESET} ${denormalize(0.3f)} to ${denormalize(0.5f)}] ")
250     sb.append(s"[${Threshold3._1}${Threshold3._2}${Console.RESET} ${denormalize(0.5f)} to ${denormalize(0.8f)}] ")
251     sb.append(s"[${Threshold4._1}${Threshold4._2}${Console.RESET}  ${denormalize(0.8f)} to $max]")
252 
253     sb.toString
254   }
