155 object IndexParam {
156 
157   @throws[ParameterException]
158   def loadIndex[DS <: GeoMesaDataStore[DS]](ds: DS,
159                                             typeName: String,
160                                             index: String,
161                                             mode: IndexMode): GeoMesaFeatureIndex[_, _] = {
162     val sft = ds.getSchema(typeName)
163     val indices = ds.manager.indices(sft, mode)
164 
165     lazy val available = {
166       val names = scala.collection.mutable.Map.empty[String, Int].withDefaultValue(0)
167       indices.foreach(i => names.put(i.name, names(i.name) + 1))
168       (indices.map(_.identifier) ++ names.collect { case (n, 1) => n }).distinct.sorted.mkString(", ")
169     }
170 
171     def single(indices: Seq[GeoMesaFeatureIndex[_, _]]): Option[GeoMesaFeatureIndex[_, _]] = indices match {
172       case Nil => None
173       case Seq(i) => Some(i)
174       case _ => throw new ParameterException(s"Specified index '$index' is ambiguous. Available indices are: $available")
175     }
176 
177     def byId: Option[GeoMesaFeatureIndex[_, _]] = indices.find(_.identifier.equalsIgnoreCase(index))
178     def byName: Option[GeoMesaFeatureIndex[_, _]] = single(indices.filter(_.name.equalsIgnoreCase(index)))
179     // check for attr vs join index name
180     def byJoin: Option[GeoMesaFeatureIndex[_, _]] = if (!index.equalsIgnoreCase(AttributeIndex.name)) { None } else {
181       single(indices.filter(_.name == AttributeIndex.JoinIndexName))
182     }
183 
184     byId.orElse(byName).orElse(byJoin).getOrElse {
185       throw new ParameterException(s"Specified index '$index' not found. Available indices are: $available")
186     }
187   }
188 }

226 trait DistributedRunParam {
227 
228   @Parameter(names = Array("--run-mode"), description = "Run locally or on a cluster", required = false)
229   var runMode: String = _
230 
231   lazy val mode: Option[RunModes.RunMode] = {
232     Option(runMode).map {
233       case m if m.equalsIgnoreCase(RunModes.Local.toString) => RunModes.Local
234       case m if m.equalsIgnoreCase(RunModes.Distributed.toString) => RunModes.Distributed
235       case m if m.equalsIgnoreCase("distributedcombine") =>
236         DistributedRunParam.this match {
237           case p: DistributedCombineParam =>
238             Command.user.warn("Using deprecated run-mode 'DistributedCombine' - please use --combine-inputs instead")
239             p.combineInputs = true
240             RunModes.Distributed
241 
242           case _ => throw invalid()
243         }
244 
245       case _ => throw invalid()
246     }
247   }
248 
249   private def invalid(): ParameterException =
250     new ParameterException(s"Invalid --run-mode '$runMode': valid values are 'local' or 'distributed'")
251 }
