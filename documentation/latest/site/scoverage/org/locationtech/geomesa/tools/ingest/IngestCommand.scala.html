58   override def execute(): Unit = {
59     val inputs: Inputs = {
60       val files = Inputs(params.files.asScala.toSeq)
61       if (files.stdin && !StdInHandle.isAvailable) {
62         if (files.paths.isEmpty) {
63           throw new ParameterException("Missing option: <files>... is required, or use `-` to ingest from standard in")
64         } else {
65           Command.user.info("Waiting for input...")
66           while (!StdInHandle.isAvailable) {
67             Thread.sleep(10)
68           }
69         }
70       }
71       if (params.srcList) { files.asSourceList } else { files }
72     }
73 
74     val format = IngestCommand.getDataFormat(params, inputs.paths)
75     val remote = inputs.paths.exists(PathUtils.isRemote)
76 
77     if (remote) {
78       // If we have a remote file, make sure they are all the same FS
79       val prefix = inputs.paths.head.split("/")(0).toLowerCase
80       if (!inputs.paths.drop(1).forall(_.toLowerCase.startsWith(prefix))) {
81         throw new ParameterException(s"Files must all be on the same file system: ($prefix) or all be local")
82       }
83     }
84 
85     val mode = if (format.contains("shp")) {
86       // shapefiles have to be ingested locally, as we need access to the related files
87       if (params.mode.exists(_ != RunModes.Local)) {
88         Command.user.warn("Forcing run mode to local for shapefile ingestion")
89       }
90       RunModes.Local
91     } else if (remote) {
92       params.mode.getOrElse(RunModes.Distributed)
93     } else {
94       if (params.mode.exists(_ != RunModes.Local)) {
95         throw new ParameterException("Input files must be in a distributed file system to run in distributed mode")
96       }
97       RunModes.Local
98     }
99 
100     if (mode == RunModes.Local) {
101       if (!params.waitForCompletion) {
102         throw new ParameterException("Tracking must be enabled when running in local mode")
103       }
104     } else if (params.threads != 1) {
105       throw new ParameterException("Threads can only be specified in local mode")
106     }
107     if (params.maxSplitSize != null && !params.combineInputs) {
108       throw new ParameterException("--split-max-size can only be used with --combine-inputs")
109     }
110 
111     withDataStore { ds =>
112       // use .get to re-throw the exception if we fail
113       IngestCommand.getSftAndConverter(params, inputs, format, Some(ds)).get.foreach { case (sft, converter) =>
114         val start = System.currentTimeMillis()
115         // create schema for the feature prior to ingest
116         var existing = Try(ds.getSchema(sft.getTypeName)).getOrElse(null)
117         if (existing == null) {
118           Command.user.info(s"Creating schema '${sft.getTypeName}'")
119           setBackendSpecificOptions(sft)
120           ds.createSchema(sft)
121           existing = ds.getSchema(sft.getTypeName)
122         } else {
123           // note: sft will have been loaded from the datastore if it already exists, so will match existing
124           Command.user.info(s"Schema '${sft.getTypeName}' exists")
125         }
126         val result = startIngest(mode, ds, existing, converter, inputs)
127         if (params.waitForCompletion) {
128           result.await(TerminalCallback()) match {
129             case JobSuccess(message, counts) =>
130               Command.user.info(s"Ingestion complete in ${TextTools.getTime(start)}")
131               val count = counts.getOrElse(IngestCounters.Persisted, counts.getOrElse(IngestCounters.Ingested, 0L))
132               val failed = counts.getOrElse(IngestCounters.Failed, 0L)
133               Command.user.info(IngestCommand.getStatInfo(count, failed, input = message))
134 
135             case JobFailure(message) =>
136               Command.user.error(s"Ingestion failed in ${TextTools.getTime(start)}")
137               // propagate out and return an exit code error
138               throw new CommandException(message)
139           }
140         } else {
141           Command.user.info("Job submitted, check tracking for progress and result")
142         }
143       }
144     }
145   }

150    * @param mode run mode
151    * @param ds data store

153    * @param converter converter config

157   protected def startIngest(
158       mode: RunMode,
159       ds: DS,
160       sft: SimpleFeatureType,
161       converter: Config,
162       inputs: Inputs): Awaitable = {
163     mode match {
164       case RunModes.Local =>
165         Command.user.info("Running ingestion in local mode")
166         new LocalConverterIngest(ds, connection.asJava, sft, converter, inputs, params.threads)
167 
168       case RunModes.Distributed =>
169         Command.user.info(s"Running ingestion in distributed ${if (params.combineInputs) "combine " else "" }mode")
170         new ConverterIngestJob(connection, sft, converter, inputs.paths, libjarsFiles, libjarsPaths) {
171           override def configureJob(job: Job): Unit = {
172             super.configureJob(job)
173             if (params.combineInputs) {
174               job.setInputFormatClass(classOf[ConverterCombineInputFormat])
175               Option(params.maxSplitSize).foreach(s => FileInputFormat.setMaxInputSplitSize(job, s.toLong))
176             }
177           }
178         }
179 
180       case _ =>
181         throw new UnsupportedOperationException(s"Missing implementation for mode $mode")
182     }
183   }

225   /**
226     * Loads and/or infers a SimpleFeatureType and Converter config based on the inputs. If not possible to load
227     * or infer the values, will return a failure. If `interactive` and the user declines to use an
228     * inferred result, will return a Success(None). Otherwise will return a Success(Some)
229     *
230     * @param params params
231     * @param inputs input files
232     * @param format input format
233     * @param ds data store for loading schemas by name
234     * @return None if user declines inferred result, otherwise the loaded/inferred result
235     */
236   def getSftAndConverter(
237       params: TypeNameParam with FeatureSpecParam with ConverterConfigParam with OptionalForceParam,
238       inputs: Inputs,
239       format: Option[String],
240       ds: Option[_ <: DataStore]): Try[Option[(SimpleFeatureType, Config)]] = Try {
241     // try to load the sft, first check for an existing schema, then load from the params/environment
242     var sft: SimpleFeatureType = loadSft(params, ds).orNull
243 
244     var converter: Config = Option(params.config).map(CLArgResolver.getConfig).orNull
245 
246     // if there is no converter passed in
247     if (converter == null && (inputs.stdin || inputs.paths.nonEmpty)) {
248       val errorMsg = if (inputs.stdin) { "Unable to read data from stdin" } else {
249         "Parameter <files> did not evaluate to something that could be read"
250       }
251       // try to infer the schema from the inputs
252       Command.user.info(
253         s"No converter defined - will attempt to detect schema from ${if (inputs.stdin) "stdin" else "input files"}")
254       val file = inputs.handles.headOption.getOrElse(throw new ParameterException(errorMsg))
255       val opened = ListBuffer.empty[CloseableIterator[InputStream]]
256       def open(): InputStream = {
257         val is = try {
258           val streams = file.open.map(_._2)
259           opened += streams
260           if (streams.hasNext) { streams.next } else { null }
261         } catch {
262           case NonFatal(e) => throw new RuntimeException(errorMsg, e)
263         }
264         if (is == null) {
265           throw new ParameterException(errorMsg)
266         } else {
267           is
268         }
269       }
270 
271       val path = if (inputs.stdin) { Map.empty[String, AnyRef] } else { EvaluationContext.inputFileParam(file.path) }
272       val opt = format match {
273         case None => SimpleFeatureConverter.infer(open _, Option(sft), path)
274         case Some(fmt) => TypeAwareInference.infer(fmt, open _, Option(sft), path)
275       }
276 
277       val (inferredSft, inferredConverter) = opt match {
278         case Success(o) => o
279         case Failure(e) =>
280           throw new ParameterException("Could not determine converter from inputs - please specify a converter", e)
281       }
282 
283       val renderOptions = ConfigRenderOptions.concise().setFormatted(true)
284       var inferredSftString: Option[String] = None
285 
286       if (sft == null) {
287         val typeName = if (inputs.stdin) {
288           // Throw an error if the user doesn't specify an SFT name
289           Option(params.featureName).getOrElse {
290             throw new ParameterException(
291               "SimpleFeatureType name not specified. Please ensure the -f or --feature-name flag is set.")
292           }
293         } else {
294           Option(params.featureName).getOrElse {
295             val existing = ds.toSet[DataStore].flatMap(_.getTypeNames)
296             val fileName = Option(FilenameUtils.getBaseName(file.path))
297             val base = fileName.map(_.trim.replaceAll("[^A-Za-z0-9]+", "_")).filterNot(_.isEmpty).getOrElse {
298               throw new RuntimeException("Unable to infer SimpleFeatureType name from file name. " +
299                   "Please specify a name manually by setting the -f or --feature-name flag.")
300             }
301             var name = base
302             var i = 0
303             while (existing.contains(name)) {
304               name = s"${base}_$i"
305               i += 1
306             }
307             name
308           }
309         }
310 
311         sft = SimpleFeatureTypes.renameSft(inferredSft, typeName)
312         inferredSftString = Some(SimpleFeatureTypes.toConfig(sft, includePrefix = false).root().render(renderOptions))
313         if (!params.force) {
314           Command.user.info(s"Inferred schema: $typeName identified ${SimpleFeatureTypes.encodeType(sft)}")
315         }
316       }
317       converter = inferredConverter
318 
319       if (!params.force) {
320         val converterString = inferredConverter.root().render(renderOptions)
321 
322         def persist(): Unit = if (Prompt.confirm("Persist this converter for future use (y/n)? ")) {
323           writeInferredConverter(sft.getTypeName, converterString, inferredSftString)
324         }
325 
326         Command.user.info(s"Inferred converter:\n$converterString")
327         if (Prompt.confirm("Use inferred converter (y/n)? ")) {
328           persist()
329         } else {
330           Command.user.info("Please re-run with a valid converter")
331           persist()
332           return Success(None)
333         }
334       }
335     }
336 
337     if (sft == null) {
338       throw new ParameterException("SimpleFeatureType name and/or specification argument is required")
339     } else if (converter == null) {
340       throw new ParameterException("Converter config argument is required")
341     }
342 
343     if (params.errorMode != null) {
344       val opts = ConfigValueFactory.fromMap(Collections.singletonMap("error-mode", params.errorMode.toString))
345       converter = ConfigFactory.empty().withValue("options", opts).withFallback(converter)
346     }
347 
348     Some((sft, converter))
349   }

411     * Tries to load a feature type, first from the data store then from the params/environment
412     *

414     * @param ds data store access
415     * @return
416     */
417   private def loadSft(
418       params: TypeNameParam with FeatureSpecParam,
419       ds: Option[_ <: DataStore]): Option[SimpleFeatureType] = {
420     val fromStore = for {
421       d    <- ds
422       name <- Option(params.featureName)
423       sft  <- Try(d.getSchema(name)).filter(_ != null).toOption
424     } yield {
425       sft
426     }
427     lazy val fromEnv = Option(params.spec).map(CLArgResolver.getSft(_, params.featureName)).orElse {
428       Option(params.featureName).flatMap(name => Try(CLArgResolver.getSft(name)).toOption)
429     }
430 
431     if (logger.underlying.isWarnEnabled()) {
432       for { fs <- fromStore; fe <- fromEnv } {
433         SimpleFeatureTypeComparator.compare(fs, fe) match {
434           case TypeComparison.Compatible(false, false, _) if fs.getTypeName == fe.getTypeName => // ok
435           case _ =>
436             logger.warn(
437               "Schema from data store does not match schema from environment." +
438                 s"\n  From data store:  ${fs.getTypeName} identified ${SimpleFeatureTypes.encodeType(fs)}" +
439                 s"\n  From environment: ${fe.getTypeName} identified ${SimpleFeatureTypes.encodeType(fe)}")
440         }
441       }
442     }
443 
444     fromStore.orElse(fromEnv)
445   }

447   private def writeInferredConverter(typeName: String, converterString: String, schemaString: Option[String]): Unit = {
448     import scala.collection.JavaConverters._
449 
450     try {
451       val conf = this.getClass.getClassLoader.getResources("reference.conf").asScala.find { u =>
452         "file".equalsIgnoreCase(u.getProtocol) && u.getPath.endsWith("/conf/reference.conf")
453       }
454       conf match {
455         case None => Command.user.error("Could not persist converter: could not find 'conf/reference.conf'")
456         case Some(r) =>
457           val reference = new File(r.toURI)
458           val folder = reference.getParentFile
459           val baseName = typeName.replaceAll("[^A-Za-z0-9_]+", "_")
460           var convert = new File(folder, s"$baseName.conf")
461           var i = 1
462           while (convert.exists()) {
463             convert = new File(folder, s"${baseName}_$i.conf")
464             i += 1
465           }
466           WithClose(new PrintWriter(new FileWriter(convert))) { writer =>
467             writer.println(s"${ConverterConfigLoader.path}.$baseName : $converterString")
468             schemaString.foreach(s => writer.println(s"${ConfigSftParsing.path}.$baseName : $s"))
469           }
470           WithClose(new PrintWriter(new FileWriter(reference, true))) { writer =>
471             writer.println(s"""include "${convert.getName}"""")
472           }
473           val (names, refs) = if (schemaString.isDefined) {
474             ("schema and converter", s"'--spec $baseName' and '--converter $baseName'")
475           } else {
476             ("converter", s"'--converter $baseName'")
477           }
478           Command.user.info(s"Added import in reference.conf and saved inferred $names to ${convert.getAbsolutePath}")
479           Command.user.info(s"In future commands, the $names may be invoked with $refs")
480       }
481     } catch {
482       case NonFatal(e) =>
483         logger.error("Error trying to persist inferred schema", e)
484         Command.user.error(s"Error trying to persist inferred schema: $e")
485     }
