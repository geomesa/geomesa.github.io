58   override def execute(): Unit = {
59     val inputs: Inputs = {
60       val files = Inputs(params.files.asScala.toSeq)
61       if (files.stdin && !StdInHandle.isAvailable) {
62         if (files.paths.isEmpty) {
63           throw new ParameterException("Missing option: <files>... is required, or use `-` to ingest from standard in")
64         } else {
65           Command.user.info("Waiting for input...")
66           while (!StdInHandle.isAvailable) {
67             Thread.sleep(10)
68           }
69         }
70       }
71       if (params.srcList) { files.asSourceList } else { files }
72     }
73 
74     val format = IngestCommand.getDataFormat(params, inputs.paths)
75     val remote = inputs.paths.exists(PathUtils.isRemote)
76 
77     if (remote) {
78       // If we have a remote file, make sure they are all the same FS
79       val prefix = inputs.paths.head.split("/")(0).toLowerCase
80       if (!inputs.paths.drop(1).forall(_.toLowerCase.startsWith(prefix))) {
81         throw new ParameterException(s"Files must all be on the same file system: ($prefix) or all be local")
82       }
83     }
84 
85     val mode = if (format.contains("shp")) {
86       // shapefiles have to be ingested locally, as we need access to the related files
87       if (params.mode.exists(_ != RunModes.Local)) {
88         Command.user.warn("Forcing run mode to local for shapefile ingestion")
89       }
90       RunModes.Local
91     } else if (remote) {
92       params.mode.getOrElse(RunModes.Distributed)
93     } else {
94       if (params.mode.exists(_ != RunModes.Local)) {
95         throw new ParameterException("Input files must be in a distributed file system to run in distributed mode")
96       }
97       RunModes.Local
98     }
99 
100     if (mode == RunModes.Local) {
101       if (!params.waitForCompletion) {
102         throw new ParameterException("Tracking must be enabled when running in local mode")
103       }
104     } else if (params.threads != 1) {
105       throw new ParameterException("Threads can only be specified in local mode")
106     }
107     if (params.maxSplitSize != null && !params.combineInputs) {
108       throw new ParameterException("--split-max-size can only be used with --combine-inputs")
109     }
110 
111     withDataStore { ds =>
112       // use .get to re-throw the exception if we fail
113       IngestCommand.getSftAndConverter(params, inputs, format, Some(ds)).get.foreach { case (sft, converter) =>
114         val start = System.currentTimeMillis()
115         // create schema for the feature prior to ingest
116         val existing = Try(ds.getSchema(sft.getTypeName)).getOrElse(null)
117         if (existing == null) {
118           Command.user.info(s"Creating schema '${sft.getTypeName}'")
119           setBackendSpecificOptions(sft)
120           ds.createSchema(sft)
121         } else {
122           // note: sft will have been loaded from the datastore if it already exists, so will match existing
123           Command.user.info(s"Schema '${sft.getTypeName}' exists")
124         }
125         val result = startIngest(mode, ds, sft, converter, inputs)
126         if (params.waitForCompletion) {
127           result.await(TerminalCallback()) match {
128             case JobSuccess(message, counts) =>
129               Command.user.info(s"Ingestion complete in ${TextTools.getTime(start)}")
130               val count = counts.getOrElse(IngestCounters.Persisted, counts.getOrElse(IngestCounters.Ingested, 0L))
131               val failed = counts.getOrElse(IngestCounters.Failed, 0L)
132               Command.user.info(IngestCommand.getStatInfo(count, failed, input = message))
133 
134             case JobFailure(message) =>
135               Command.user.error(s"Ingestion failed in ${TextTools.getTime(start)}")
136               // propagate out and return an exit code error
137               throw new CommandException(message)
138           }
139         } else {
140           Command.user.info("Job submitted, check tracking for progress and result")
141         }
142       }
143     }
144   }

156   protected def startIngest(

159       sft: SimpleFeatureType,

161       inputs: Inputs): Awaitable = {
162     mode match {
163       case RunModes.Local =>

167       case RunModes.Distributed =>

175             }
176           }
177         }
178 
179       case _ =>
180         throw new NotImplementedError(s"Missing implementation for mode $mode")
181     }

224   /**
225     * Loads and/or infers a SimpleFeatureType and Converter config based on the inputs. If not possible to load
226     * or infer the values, will return a failure. If `interactive` and the user declines to use an
227     * inferred result, will return a Success(None). Otherwise will return a Success(Some)
228     *
229     * @param params params
230     * @param inputs input files
231     * @param format input format
232     * @param ds data store for loading schemas by name
233     * @return None if user declines inferred result, otherwise the loaded/inferred result
234     */
235   def getSftAndConverter(
236       params: TypeNameParam with FeatureSpecParam with ConverterConfigParam with OptionalForceParam,
237       inputs: Inputs,
238       format: Option[String],
239       ds: Option[_ <: DataStore]): Try[Option[(SimpleFeatureType, Config)]] = Try {
240     // try to load the sft, first check for an existing schema, then load from the params/environment
241     var sft: SimpleFeatureType = loadSft(params, ds).orNull
242 
243     var converter: Config = Option(params.config).map(CLArgResolver.getConfig).orNull
244 
245     // if there is no converter passed in
246     if (converter == null && (inputs.stdin || inputs.paths.nonEmpty)) {
247       val errorMsg = if (inputs.stdin) { "Unable to read data from stdin" } else {
248         "Parameter <files> did not evaluate to something that could be read"
249       }
250       // try to infer the schema from the inputs
251       Command.user.info(
252         s"No converter defined - will attempt to detect schema from ${if (inputs.stdin) "stdin" else "input files"}")
253       val file = inputs.handles.headOption.getOrElse(throw new ParameterException(errorMsg))
254       val opened = ListBuffer.empty[CloseableIterator[InputStream]]
255       def open(): InputStream = {
256         val is = try {
257           val streams = file.open.map(_._2)
258           opened += streams
259           if (streams.hasNext) { streams.next } else { null }
260         } catch {
261           case NonFatal(e) => throw new RuntimeException(errorMsg, e)
262         }
263         if (is == null) {
264           throw new ParameterException(errorMsg)
265         } else {
266           is
267         }
268       }
269 
270       val path = if (inputs.stdin) { Map.empty[String, AnyRef] } else { EvaluationContext.inputFileParam(file.path) }
271       val opt = format match {
272         case None => SimpleFeatureConverter.infer(open _, Option(sft), path)
273         case Some(fmt) => TypeAwareInference.infer(fmt, open _, Option(sft), path)
274       }
275 
276       val (inferredSft, inferredConverter) = opt match {
277         case Success(o) => o
278         case Failure(e) =>
279           throw new ParameterException("Could not determine converter from inputs - please specify a converter", e)
280       }
281 
282       val renderOptions = ConfigRenderOptions.concise().setFormatted(true)
283       var inferredSftString: Option[String] = None
284 
285       if (sft == null) {
286         val typeName = if (inputs.stdin) {
287           // Throw an error if the user doesn't specify an SFT name
288           Option(params.featureName).getOrElse {
289             throw new ParameterException(
290               "SimpleFeatureType name not specified. Please ensure the -f or --feature-name flag is set.")
291           }
292         } else {
293           Option(params.featureName).getOrElse {
294             val existing = ds.toSet[DataStore].flatMap(_.getTypeNames)
295             val fileName = Option(FilenameUtils.getBaseName(file.path))
296             val base = fileName.map(_.trim.replaceAll("[^A-Za-z0-9]+", "_")).filterNot(_.isEmpty).getOrElse {
297               throw new RuntimeException("Unable to infer SimpleFeatureType name from file name. " +
298                   "Please specify a name manually by setting the -f or --feature-name flag.")
299             }
300             var name = base
301             var i = 0
302             while (existing.contains(name)) {
303               name = s"${base}_$i"
304               i += 1
305             }
306             name
307           }
308         }
309 
310         sft = SimpleFeatureTypes.renameSft(inferredSft, typeName)
311         inferredSftString = Some(SimpleFeatureTypes.toConfig(sft, includePrefix = false).root().render(renderOptions))
312         if (!params.force) {
313           Command.user.info(s"Inferred schema: $typeName identified ${SimpleFeatureTypes.encodeType(sft)}")
314         }
315       }
316       converter = inferredConverter
317 
318       if (!params.force) {
319         val converterString = inferredConverter.root().render(renderOptions)
320 
321         def persist(): Unit = if (Prompt.confirm("Persist this converter for future use (y/n)? ")) {
322           writeInferredConverter(sft.getTypeName, converterString, inferredSftString)
323         }
324 
325         Command.user.info(s"Inferred converter:\n$converterString")
326         if (Prompt.confirm("Use inferred converter (y/n)? ")) {
327           persist()
328         } else {
329           Command.user.info("Please re-run with a valid converter")
330           persist()
331           return Success(None)
332         }
333       }
334     }
335 
336     if (sft == null) {
337       throw new ParameterException("SimpleFeatureType name and/or specification argument is required")
338     } else if (converter == null) {
339       throw new ParameterException("Converter config argument is required")
340     }
341 
342     if (params.errorMode != null) {
343       val opts = ConfigValueFactory.fromMap(Collections.singletonMap("error-mode", params.errorMode.toString))
344       converter = ConfigFactory.empty().withValue("options", opts).withFallback(converter)
345     }
346 
347     Some((sft, converter))
348   }

410     * Tries to load a feature type, first from the data store then from the params/environment
411     *
412     * @param params params
413     * @param ds data store access
414     * @return
415     */
416   private def loadSft(
417       params: TypeNameParam with FeatureSpecParam,
418       ds: Option[_ <: DataStore]): Option[SimpleFeatureType] = {
419     val fromStore = for {
420       d    <- ds
421       name <- Option(params.featureName)
422       sft  <- Try(d.getSchema(name)).filter(_ != null).toOption
423     } yield {
424       sft
425     }
426     lazy val fromEnv = Option(params.spec).map(CLArgResolver.getSft(_, params.featureName)).orElse {
427       Option(params.featureName).flatMap(name => Try(CLArgResolver.getSft(name)).toOption)
428     }
429 
430     if (logger.underlying.isWarnEnabled()) {
431       for { fs <- fromStore; fe <- fromEnv } {
432         SimpleFeatureTypeComparator.compare(fs, fe) match {
433           case TypeComparison.Compatible(false, false, _) if fs.getTypeName == fe.getTypeName => // ok
434           case _ =>
435             logger.warn(
436               "Schema from data store does not match schema from environment." +
437                 s"\n  From data store:  ${fs.getTypeName} identified ${SimpleFeatureTypes.encodeType(fs)}" +
438                 s"\n  From environment: ${fe.getTypeName} identified ${SimpleFeatureTypes.encodeType(fe)}")
439         }
440       }
441     }
442 
443     fromStore.orElse(fromEnv)
444   }

446   private def writeInferredConverter(typeName: String, converterString: String, schemaString: Option[String]): Unit = {
447     import scala.collection.JavaConverters._
448 
449     try {
450       val conf = this.getClass.getClassLoader.getResources("reference.conf").asScala.find { u =>
451         "file".equalsIgnoreCase(u.getProtocol) && u.getPath.endsWith("/conf/reference.conf")
452       }
453       conf match {
454         case None => Command.user.error("Could not persist converter: could not find 'conf/reference.conf'")
455         case Some(r) =>
456           val reference = new File(r.toURI)
457           val folder = reference.getParentFile
458           val baseName = typeName.replaceAll("[^A-Za-z0-9_]+", "_")
459           var convert = new File(folder, s"$baseName.conf")
460           var i = 1
461           while (convert.exists()) {
462             convert = new File(folder, s"${baseName}_$i.conf")
463             i += 1
464           }
465           WithClose(new PrintWriter(new FileWriter(convert))) { writer =>
466             writer.println(s"${ConverterConfigLoader.path}.$baseName : $converterString")
467             schemaString.foreach(s => writer.println(s"${ConfigSftParsing.path}.$baseName : $s"))
468           }
469           WithClose(new PrintWriter(new FileWriter(reference, true))) { writer =>
470             writer.println(s"""include "${convert.getName}"""")
471           }
472           val (names, refs) = if (schemaString.isDefined) {
473             ("schema and converter", s"'--spec $baseName' and '--converter $baseName'")
474           } else {
475             ("converter", s"'--converter $baseName'")
476           }
477           Command.user.info(s"Added import in reference.conf and saved inferred $names to ${convert.getAbsolutePath}")
478           Command.user.info(s"In future commands, the $names may be invoked with $refs")
479       }
480     } catch {
481       case NonFatal(e) =>
482         logger.error("Error trying to persist inferred schema", e)
483         Command.user.error(s"Error trying to persist inferred schema: $e")
484     }
