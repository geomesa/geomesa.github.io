46   private def execute(executor: Executor): Unit = {
47     val result = try { executor.execute(); CommandResult(0) } catch {
48       case e @ (_: ClassNotFoundException | _: NoClassDefFoundError) =>
49         // log the underling exception to the log file, but don't show to the user
50         val msg = s"Warning: Missing dependency for command execution: ${e.getMessage}"
51         logger.error(msg, e)
52         CommandResult(1, Seq(Left(msg)) ++ getEnvironmentErrors.map(Left.apply[String, Throwable]))
53       case e @ (_: ParameterException | _: CommandException) =>
54         logger.error("Error executing command:", e)
55         CommandResult(1, failureMessages(e))
56       case NonFatal(e) =>
57         logger.error("Error executing command:", e)
58         CommandResult(1, Seq(Right(e)))
59     }
60 
61     result.errors.foreach {
62       case Left(msg) => Command.user.error(msg)
63       case Right(e)  => Command.user.error(e.getMessage, e)
64     }
65 
66     sys.exit(result.code)
67   }

69   private def failureMessages(e: Throwable, indent: String = "", prefix: String = ""): Seq[Left[String, Throwable]] = {
70     val messages = ArrayBuffer.empty[Left[String, Throwable]]
71     val message = if (indent == "") { e.getMessage } else { s"$indent$prefix${e.toString}" }
72     messages += Left(message)
73     if (e.getCause != null) {
74       messages ++= failureMessages(e.getCause, indent + "  ", "Caused by: ")
75     }
76     e.getSuppressed.foreach { e =>
77       messages ++= failureMessages(e, indent + "  ", "Suppressed: ")
78     }
79     messages.toSeq
80   }

82   def parseCommand(args: Array[String]): Command = {
83     val jc =
84       JCommander.newBuilder()
85           .programName(name)
86           .addConverterFactory(new GeoMesaIStringConverterFactory)
87           .build()
88 
89     val commands = this.commands :+ new HelpCommand(this, jc)
90     commands.foreach { command =>
91       jc.addCommand(command.name, command.params)
92       command.subCommands.foreach(sub => jc.getCommands.get(command.name).addCommand(sub.name, sub.params))
93     }
94 
95     def logAndThrowError(e: ParameterException): Unit = {
96       Command.user.error(s"Error parsing arguments: ${e.getMessage}")
97       Command.user.info(usage(jc, jc.getParsedCommand))
98       throw e
99     }
100 
101     try { jc.parse(args: _*) } catch {
102       case e: ParameterException => logAndThrowError(e)
103     }
104 
105     val parsed = commands.find(_.name == jc.getParsedCommand).getOrElse(new DefaultCommand(jc))
106     resolveEnvironment(parsed)
107     val command = if (parsed.subCommands.isEmpty) { parsed } else {
108       lazy val available =
109         s"Use '${parsed.name} <sub-command>', where sub-command is one of: " +
110             parsed.subCommands.map(_.name).mkString(", ")
111       val sub = Option(jc.getCommands.get(parsed.name).getParsedCommand).getOrElse {
112         throw new ParameterException(s"No sub-command specified. $available")
113       }
114       parsed.subCommands.find(_.name == sub).getOrElse {
115         throw new ParameterException(s"Sub-command '$sub' not found. $available")
116       }
117     }
118     command.validate().foreach(logAndThrowError)
119     command
120   }

122   def usage(jc: JCommander): String = {
123     val out = new StringBuilder()
124     out.append(s"Usage: $name [command] [command options]\n")
125     val commands = jc.getCommands.asScala.keys.toSeq.sorted
126     out.append("  Commands:\n")
127     val maxLen = commands.map(_.length).max + 4
128     commands.foreach { name =>
129       val spaces = " " * (maxLen - name.length)
130       out.append(s"    $name$spaces${jc.getUsageFormatter.getCommandDescription(name)}\n")
131     }
132     out.toString()
133   }

135   def usage(jc: JCommander, name: String): String = {
136     Option(name).flatMap(n => Option(jc.getCommands.get(n))) match {
137       case None => usage(jc)
138       case Some(command) =>
139         val out = new java.lang.StringBuilder()
140         command.getUsageFormatter.usage(out)
141         out.toString
142     }

145   def autocompleteUsage(jc: JCommander, autocompleteInfo: AutocompleteInfo): Unit = {
146     val file = new File(autocompleteInfo.path)
147     val commands = jc.getCommands.asScala.keys.toSeq
148     val out = new StringBuilder
149     out.append(

151          |  local cur prev;
152          |  COMPREPLY=();
153          |  cur="$${COMP_WORDS[COMP_CWORD]}";
154          |  prev="$${COMP_WORDS[COMP_CWORD-1]}";
155          |
156          |  if [[ "$${COMP_WORDS[1]}" == "help" ]]; then
157          |    COMPREPLY=( $$(compgen -W "${commands.mkString(" ")}" $${cur}));
158          |    return 0;
159          |  fi;
160          |
161          |  case $${COMP_CWORD} in
162          |    1)
163          |      COMPREPLY=( $$(compgen -W "${commands.mkString(" ")}" $${cur}));
164          |      ;;
165          |    [2-9] | [1-9][0-9])
166          |      if [[ "$${cur}" =~ ^-[a-zA-Z-]?+$$ ]]; then
167          |        case $${COMP_WORDS[1]} in
168         """.stripMargin)
169     commands.foreach { command =>
170       val params = jc.getCommands.get(command).getParameters.asScala.filter(!_.getParameter.hidden()).flatMap(_.getParameter.names().filter(_.length != 2))
171         out.append(
172       s"""            $command)
173          |              COMPREPLY=( $$(compgen -W "${params.mkString(" ").replaceAll("[,\\s]+", " ")}" -- $${cur}));
174          |              return 0;
175          |              ;;
176       """.stripMargin)
177     }
178     out.append(
179       s"""        esac;
180          |      else
181          |        compopt -o filenames -o nospace;
182          |        COMPREPLY=( $$(compgen -f "$$2") );
183          |      fi;
184          |      return 0;
185          |      ;;
186          |    *)
187          |      COMPREPLY=();
188          |      ;;
189          |  esac;
190          |};
191          |complete -F _${autocompleteInfo.commandName} ${autocompleteInfo.commandName};
192          |complete -F _${autocompleteInfo.commandName} bin/${autocompleteInfo.commandName};
193          |
194          |
195        """.stripMargin)
196     FileUtils.writeStringToFile(file, out.toString(), StandardCharsets.UTF_8)
197   }

220   private def getEnvironmentErrors: Option[String] = {
221     val env = classpathEnvironments
222     if (env.forall(sys.env.contains)) { None } else {
223       val envMsg = if (env.size == 1) { "is not set as an environment variable" } else { "are not set as environment variables" }
224       val types = env.map(e => e.take(1) ++ e.drop(1).toLowerCase(Locale.US).replace("_home", "")).mkString(" and ")
225       val msg =
226         s"\nWarning: ${env.mkString(" and/or ")} $envMsg." +
227             s"\nGeoMesa tools will not run without the appropriate $types JARs in the tools classpath." +
228             s"\nPlease ensure that those JARs are present in the classpath by running '$name classpath'." +
229             "\nTo take corrective action, copy the necessary JAR files in the GeoMesa tools lib directory " +
230             "using the provided 'install-dependencies.sh' and 'install-*-support.sh' scripts."
231       Some(msg)
232     }
233   }
