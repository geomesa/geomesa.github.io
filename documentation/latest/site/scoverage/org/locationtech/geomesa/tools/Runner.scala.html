52   private def execute(executor: Executor): Unit = {
53     val result = try { executor.execute(); CommandResult(0) } catch {
54       case e @ (_: ClassNotFoundException | _: NoClassDefFoundError) =>
55         // log the underling exception to the log file, but don't show to the user
56         val msg = s"Warning: Missing dependency for command execution: ${e.getMessage}"
57         logger.error(msg, e)
58         CommandResult(1, Seq(Left(msg)) ++ getEnvironmentErrors.map(Left.apply[String, Throwable]))
59       case e @ (_: ParameterException | _: CommandException) =>
60         logger.error("Error executing command:", e)
61         CommandResult(1, failureMessages(e))
62       case NonFatal(e) =>
63         logger.error("Error executing command:", e)
64         CommandResult(1, Seq(Right(e)))
65     }
66 
67     result.errors.foreach {
68       case Left(msg) => Command.user.error(msg)
69       case Right(e)  => Command.user.error(e.getMessage, e)
70     }
71 
72     sys.exit(result.code)
73   }

75   private def failureMessages(e: Throwable, indent: String = "", prefix: String = ""): Seq[Left[String, Throwable]] = {
76     val messages = ArrayBuffer.empty[Left[String, Throwable]]
77     val message = if (indent == "") { e.getMessage } else { s"$indent$prefix${e.toString}" }
78     messages += Left(message)
79     if (e.getCause != null) {
80       messages ++= failureMessages(e.getCause, indent + "  ", "Caused by: ")
81     }
82     e.getSuppressed.foreach { e =>
83       messages ++= failureMessages(e, indent + "  ", "Suppressed: ")
84     }
85     messages.toSeq
86   }

88   def parseCommand(args: Array[String]): Command = {
89     val jc =
90       JCommander.newBuilder()
91           .programName(name)
92           .addConverterFactory(new GeoMesaIStringConverterFactory)
93           .build()
94 
95     val commands = this.commands ++ Seq(new HelpCommand(this, jc), new AutoCompleteCommand(this, jc))
96     commands.foreach { command =>
97       jc.addCommand(command.name, command.params)
98       command.subCommands.foreach(sub => jc.getCommands.get(command.name).addCommand(sub.name, sub.params))
99     }
100 
101     def logAndThrowError(e: ParameterException): Unit = {
102       Command.user.error(s"Error parsing arguments: ${e.getMessage}")
103       Command.user.info(usage(jc, jc.getParsedCommand))
104       throw e
105     }
106 
107     try { jc.parse(args: _*) } catch {
108       case e: ParameterException => logAndThrowError(e)
109     }
110 
111     val parsed = commands.find(_.name == jc.getParsedCommand).getOrElse(new DefaultCommand(jc))
112     resolveEnvironment(parsed)
113     val command = if (parsed.subCommands.isEmpty) { parsed } else {
114       lazy val available =
115         s"Use '${parsed.name} <sub-command>', where sub-command is one of: " +
116             parsed.subCommands.map(_.name).mkString(", ")
117       val sub = Option(jc.getCommands.get(parsed.name).getParsedCommand).getOrElse {
118         throw new ParameterException(s"No sub-command specified. $available")
119       }
120       parsed.subCommands.find(_.name == sub).getOrElse {
121         throw new ParameterException(s"Sub-command '$sub' not found. $available")
122       }
123     }
124     command.validate().foreach(logAndThrowError)
125     command
126   }

128   def usage(jc: JCommander): String = {
129     val out = new StringBuilder()
130     out.append(s"Usage: $name [command] [command options]\n")
131     val commands = jc.getCommands.asScala.keys.toSeq.sorted
132     out.append("  Commands:\n")
133     val maxLen = commands.map(_.length).max + 4
134     commands.foreach { name =>
135       val spaces = " " * (maxLen - name.length)
136       out.append(s"    $name$spaces${jc.getUsageFormatter.getCommandDescription(name)}\n")
137     }
138     out.toString()
139   }

141   def usage(jc: JCommander, name: String): String = {
142     Option(name).flatMap(n => Option(jc.getCommands.get(n))) match {
143       case None => usage(jc)
144       case Some(command) =>
145         val out = new java.lang.StringBuilder()
146         command.getUsageFormatter.usage(out)
147         out.toString
148     }

172   private def getEnvironmentErrors: Option[String] = {
173     val env = classpathEnvironments
174     if (env.forall(sys.env.contains)) { None } else {
175       val envMsg = if (env.size == 1) { "is not set as an environment variable" } else { "are not set as environment variables" }
176       val types = env.map(e => e.take(1) ++ e.drop(1).toLowerCase(Locale.US).replace("_home", "")).mkString(" and ")
177       val msg =
178         s"\nWarning: ${env.mkString(" and/or ")} $envMsg." +
179             s"\nGeoMesa tools will not run without the appropriate $types JARs in the tools classpath." +
180             s"\nPlease ensure that those JARs are present in the classpath by running '$name classpath'." +
181             "\nTo take corrective action, copy the necessary JAR files in the GeoMesa tools lib directory " +
182             "using the provided 'install-dependencies.sh' and 'install-*-support.sh' scripts."
183       Some(msg)
184     }
185   }
