53   private def execute(executor: Executor): Unit = {
54     val result = try { executor.execute(); CommandResult(0) } catch {
55       case e @ (_: ClassNotFoundException | _: NoClassDefFoundError) =>
56         // log the underling exception to the log file, but don't show to the user
57         val msg = s"Warning: Missing dependency for command execution: ${e.getMessage}"
58         logger.error(msg, e)
59         CommandResult(1, Seq(Left(msg)) ++ getEnvironmentErrors.map(Left.apply[String, Throwable]))
60       case e @ (_: ParameterException | _: CommandException) =>
61         logger.error("Error executing command:", e)
62         CommandResult(1, failureMessages(e))
63       case NonFatal(e) =>
64         logger.error("Error executing command:", e)
65         CommandResult(1, Seq(Right(e)))
66     }
67 
68     result.errors.foreach {
69       case Left(msg) => Command.user.error(msg)
70       case Right(e)  => Command.user.error(e.getMessage, e)
71     }
72 
73     sys.exit(result.code)
74   }
75 
76   private def failureMessages(e: Throwable, indent: String = "", prefix: String = ""): Seq[Left[String, Throwable]] = {
77     val messages = ArrayBuffer.empty[Left[String, Throwable]]
78     val message = if (indent == "") { e.getMessage } else { s"$indent$prefix${e.toString}" }
79     messages += Left(message)
80     if (e.getCause != null) {
81       messages ++= failureMessages(e.getCause, indent + "  ", "Caused by: ")
82     }
83     e.getSuppressed.foreach { e =>
84       messages ++= failureMessages(e, indent + "  ", "Suppressed: ")
85     }
86     messages.toSeq
87   }

89   def parseCommand(args: Array[String]): Command = {
90     val jc =
91       JCommander.newBuilder()
92           .programName(name)
93           .addConverterFactory(new GeoMesaIStringConverterFactory)
94           .build()
95 
96     val commands = this.commands ++ Seq(new HelpCommand(this, jc), new AutoCompleteCommand(this, jc))
97     commands.foreach { command =>
98       jc.addCommand(command.name, command.params)
99       command.subCommands.foreach(sub => jc.getCommands.get(command.name).addCommand(sub.name, sub.params))
100     }
101 
102     def logAndThrowError(e: ParameterException): Unit = {
103       Command.user.error(s"Error parsing arguments: ${e.getMessage}")
104       Command.user.info(usage(jc, jc.getParsedCommand))
105       throw e
106     }
107 
108     try { jc.parse(args: _*) } catch {
109       case e: ParameterException => logAndThrowError(e)
110     }
111 
112     val parsed = commands.find(_.name == jc.getParsedCommand).getOrElse(new DefaultCommand(jc))
113     resolveEnvironment(parsed)
114     val command = if (parsed.subCommands.isEmpty) { parsed } else {
115       lazy val available =
116         s"Use '${parsed.name} <sub-command>', where sub-command is one of: " +
117             parsed.subCommands.map(_.name).mkString(", ")
118       val sub = Option(jc.getCommands.get(parsed.name).getParsedCommand).getOrElse {
119         throw new ParameterException(s"No sub-command specified. $available")
120       }
121       parsed.subCommands.find(_.name == sub).getOrElse {
122         throw new ParameterException(s"Sub-command '$sub' not found. $available")
123       }
124     }
125     command.validate().foreach(logAndThrowError)
126     command
127   }

129   def usage(jc: JCommander): String = {
130     val out = new StringBuilder()
131     out.append(s"Usage: $name [command] [command options]\n")
132     val commands = jc.getCommands.asScala.keys.toSeq.sorted
133     out.append("  Commands:\n")
134     val maxLen = commands.map(_.length).max + 4
135     commands.foreach { name =>
136       val spaces = " " * (maxLen - name.length)
137       out.append(s"    $name$spaces${jc.getUsageFormatter.getCommandDescription(name)}\n")
138     }
139     out.toString()
140   }

142   def usage(jc: JCommander, name: String): String = {
143     Option(name).flatMap(n => Option(jc.getCommands.get(n))) match {
144       case None => usage(jc)
145       case Some(command) =>
146         val out = new java.lang.StringBuilder()
147         command.getUsageFormatter.usage(out)
148         out.toString
149     }
150   }

173   private def getEnvironmentErrors: Option[String] = {
174     val env = classpathEnvironments
175     if (env.forall(sys.env.contains)) { None } else {
176       val envMsg = if (env.size == 1) { "is not set as an environment variable" } else { "are not set as environment variables" }
177       val types = env.map(e => e.take(1) ++ e.drop(1).toLowerCase(Locale.US).replace("_home", "")).mkString(" and ")
178       val msg =
179         s"\nWarning: ${env.mkString(" and/or ")} $envMsg." +
180             s"\nGeoMesa tools will not run without the appropriate $types JARs in the tools classpath." +
181             s"\nPlease ensure that those JARs are present in the classpath by running '$name classpath'." +
182             "\nTo take corrective action, copy the necessary JAR files in the GeoMesa tools lib directory " +
183             "using the provided 'install-dependencies.sh' and 'install-*-support.sh' scripts."
184       Some(msg)
185     }
186   }
