22 class ShapefileExporter(file: File) extends FeatureExporter {
23 
24   import scala.collection.JavaConverters._
25 
26   private val url = URLs.fileToUrl(file)
27   private var ds: ShapefileDataStore = _
28   private var mappings: Map[Int, Int] = _
29 
30   override def start(sft: SimpleFeatureType): Unit = {
31     // ensure that the parent directory exists, otherwise the data store will error out
32     Option(file.getParentFile).filterNot(_.exists()).foreach(_.mkdirs())
33     ds = new ShapefileDataStoreFactory().createDataStore(url).asInstanceOf[ShapefileDataStore]
34     ds.createSchema(sft)
35 
36     var i = -1
37     var j = 0
38     // map attributes according to the shapefile data store
39     // default geometry goes to attribute 0, other geometries are dropped
40     // byte arrays are dropped, but everything else is kept or transformed
41     // see: `org.geotools.data.shapefile.ShapefileDataStore.createDbaseHeader()`
42     val attributes = sft.getAttributeDescriptors.asScala.flatMap { d =>
43       i += 1
44       val binding = d.getType.getBinding
45       if (classOf[Geometry].isAssignableFrom(binding) || binding == classOf[Array[Byte]] ) { None } else {
46         j += 1
47         Some(i -> j)
48       }
49     }
50     mappings = attributes.toMap + (sft.indexOf(sft.getGeometryDescriptor.getLocalName) -> 0)
51   }
52 
53   override def export(features: Iterator[SimpleFeature]): Option[Long] = {
54     var count = 0L
55 
56     WithClose(ds.getFeatureWriterAppend(Transaction.AUTO_COMMIT)) { writer =>
57       features.foreach { feature =>
58         val toWrite = writer.next()
59         mappings.foreach { case (from, to) => toWrite.setAttribute(to, feature.getAttribute(from)) }
60         // copy over the user data
61         // note: shapefile doesn't support provided fid
62         toWrite.getUserData.putAll(feature.getUserData)
63         writer.write()
64         count += 1L
65       }
66     }
67 
68     Some(count)
69   }
70 
71   override def bytes: Long = {
72     val files = new ShpFiles(url)
73     try {
74       var sum = 0L
75       files.getFileNames.asScala.values.foreach { file =>
76         sum += URLs.urlToFile(new URL(file)).length()
77       }
78       sum
79     } finally {
80       files.dispose()
81     }
82   }
83 
84   override def close(): Unit = Option(ds).foreach(_.dispose)
85 }
