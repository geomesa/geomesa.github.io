19 trait StatsTopKCommand[DS <: DataStore with HasGeoMesaStats] extends DataStoreCommand[DS] {
20 
21   override val name = "stats-top-k"
22   override val params: StatsTopKParams
23 
24   override def execute(): Unit = withDataStore(topK)
25 
26   protected def topK(ds: DS): Unit = {
27     for {
28       p <- Option(params).collect { case p: ProvidedTypeNameParam => p }
29       f <- Option(ds).collect { case f: FileDataStore => f }
30     } { p.featureName = f.getSchema.getTypeName }
31 
32     val sft = ds.getSchema(params.featureName)
33     if (sft == null) {
34       throw new ParameterException(s"Schema '${params.featureName}' does not exist")
35     }
36 
37     val attributes = getAttributesFromParams(sft, params)
38     val filter = Option(params.cqlFilter).getOrElse(Filter.INCLUDE)
39     val k = Option(params.k).map(_.intValue)
40 
41     if (params.exact) {
42       Command.user.info("Running stat query...")
43     } else if (filter != Filter.INCLUDE) {
44       Command.user.warn("Non-exact stat queries may not fully account for the specified CQL filter")
45     }
46 
47     val results = ds.stats.getSeqStat[TopK[Any]](sft, attributes.map(Stat.TopK), filter, params.exact)
48 
49     attributes.foreach { attribute =>
50       Command.output.info(s"Top values for '$attribute':")
51       val stat = results.find(_.property == attribute)
52       stat match {
53         case None => Command.output.info("  unavailable")
54         case Some(s) =>
55           val stringify = Stat.stringifier(sft.getDescriptor(attribute).getType.getBinding)
56           s.topK(k.getOrElse(s.size)).foreach { case (value, count) =>
57             Command.output.info(s"  ${stringify(value)} ($count)")
58           }
59       }
60     }
61   }
62 }
