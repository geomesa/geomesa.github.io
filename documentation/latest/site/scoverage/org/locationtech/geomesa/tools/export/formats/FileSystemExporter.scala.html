22   * Export to a FileSystem data store format

24 abstract class FileSystemExporter(path: String) extends FeatureExporter {
25 
26   private lazy val handle = PathUtils.getHandle(path)
27 
28   private var writer: FileSystemWriter = _
29 
30   protected def createWriter(sft: SimpleFeatureType): FileSystemWriter
31 
32   override def start(sft: SimpleFeatureType): Unit = {
33     if (writer != null) {
34       writer.close()
35     }
36     writer = createWriter(sft)
37   }
38 
39   override def export(features: Iterator[SimpleFeature]): Option[Long] = {
40     require(writer != null, "Must call 'start' before 'export'")
41     var i = 0L
42     features.foreach { f => writer.write(f); i += 1 }
43     Some(i)
44   }
45 
46   override def bytes: Long = handle.length
47 
48   override def close(): Unit = {
49     if (writer != null) {
50       writer.close()
51     }
52   }
53 }

57   class ParquetFileSystemExporter(path: String) extends FileSystemExporter(path) {
58     override protected def createWriter(sft: SimpleFeatureType): FileSystemWriter = {
59       val conf = new Configuration()
60       StorageConfiguration.setSft(conf, sft)
61       try { Class.forName("org.xerial.snappy.Snappy") } catch {
62         case _: ClassNotFoundException =>
63           logger.warn("SNAPPY compression is not available on the classpath - falling back to GZIP")
64           conf.set("parquet.compression", "GZIP")
65       }
66       // use PathUtils.getUrl to handle local files, otherwise default can be in hdfs
67       new ParquetFileSystemWriter(sft, new Path(PathUtils.getUrl(path).toURI), conf)
68     }
69   }
