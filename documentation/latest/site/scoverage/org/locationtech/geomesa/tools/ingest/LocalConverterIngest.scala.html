36 /**
37  * Ingestion that uses geomesa converters to process input files
38  *
39  * @param ds data store
40  * @param sft simple feature type
41  * @param converterConfig converter definition
42  * @param inputs paths to ingest
43  * @param numThreads how many threads to use
44  */
45 class LocalConverterIngest(
46     ds: DataStore,
47     dsParams: java.util.Map[String, _],
48     sft: SimpleFeatureType,
49     converterConfig: Config,
50     inputs: Inputs,
51     numThreads: Int
52   ) extends Awaitable with LazyLogging {
53 
54   private val files = inputs.handles
55   private val latch = new CountDownLatch(files.length)
56 
57   private val threads = if (numThreads <= files.length) { numThreads } else {
58     Command.user.warn("Can't use more threads than there are input files - reducing thread count")
59     files.length
60   }
61 
62   private val batch = IngestCommand.LocalBatchSize.toInt.getOrElse {
63     throw new IllegalArgumentException(
64       s"Invalid batch size for property ${IngestCommand.LocalBatchSize.property}: " +
65           IngestCommand.LocalBatchSize.get)
66   }
67 
68   private val es = Executors.newFixedThreadPool(threads)
69 
70   // global counts shared among threads
71   private val written = new AtomicLong(0)
72   private val failed = new AtomicLong(0)
73   private val errors = new AtomicInteger(0)
74 
75   private val bytesRead = new AtomicLong(0L)
76 
77   private val batches = new ConcurrentHashMap[FeatureWriter[SimpleFeatureType, SimpleFeature], AtomicInteger](threads)
78 
79   // keep track of failure at a global level, keep line counts and success local
80   private val globalFailures = new com.codahale.metrics.Counter {
81     override def inc(): Unit = failed.incrementAndGet()
82     override def inc(n: Long): Unit = failed.addAndGet(n)
83     override def dec(): Unit = failed.decrementAndGet()
84     override def dec(n: Long): Unit = failed.addAndGet(-1 * n)
85     override def getCount: Long = failed.get()
86   }
87 
88   private val progress: () => Float =
89     if (inputs.stdin) {
90       () => .99f // we don't know how many bytes are actually available
91     } else {
92       val length = files.map(_.length).sum.toFloat // only evaluate once
93       () => bytesRead.get / length
94     }
95 
96   Command.user.info(s"Ingesting ${if (inputs.stdin) { "from stdin" } else { TextTools.getPlural(files.length, "file") }} " +
97       s"with ${TextTools.getPlural(threads, "thread")}")
98 
99   private val converters = CloseablePool(SimpleFeatureConverter(sft, converterConfig), threads)
100   private val writers = {
101     def factory: FeatureWriter[SimpleFeatureType, SimpleFeature] =
102       if (threads > 1 && ds.getClass.getSimpleName.equals("JDBCDataStore")) {
103         // creates a new data store for each writer to avoid synchronized blocks in JDBCDataStore.
104         // the synchronization is to allow for generated fids from the database.
105         // generally, this shouldn't be an issue since we use provided fids,
106         // but running with 1 thread  would restore the old behavior
107         new DataStoreWriter(dsParams, sft.getTypeName)
108       } else {
109         ds.getFeatureWriterAppend(sft.getTypeName, Transaction.AUTO_COMMIT)
110       }
111     CloseablePool(factory, threads)
112   }
113 
114   private val closer = new Runnable() {
115     override def run(): Unit = {
116       latch.await()
117       CloseWithLogging(converters)
118       CloseWithLogging(writers).foreach(_ => errors.incrementAndGet())
119     }
120   }
121 
122   private val futures = files.map(f => es.submit(new LocalIngestWorker(f))) :+ CachedThreadPool.submit(closer)
123 
124   es.shutdown()
125 
126   override def await(reporter: StatusCallback): JobResult = {
127     while (!es.isTerminated) {
128       Thread.sleep(500)
129       reporter("", progress(), counters, done = false)
130     }
131     reporter("", progress(), counters, done = true)
132 
133     // Get all futures so that we can propagate the logging up to the top level for handling
134     // in org.locationtech.geomesa.tools.Runner to catch missing dependencies
135     futures.foreach(_.get)
136 
137     if (errors.get > 0) {
138       JobFailure("Some files caused errors, check logs for details")
139     } else {
140       val message =
141         if (inputs.stdin) { "from stdin" } else if (files.lengthCompare(1) == 0) { s"for file ${files.head.path}" } else { "" }
142       JobSuccess(message, counters.toMap)
143     }
144   }
145 
146   /**
147    * Hook to allow modification of the feature returned by the converter
148    *
149    * @param iter features
150    * @return
151    */
152   protected def features(iter: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = iter
153 
154   private def counters: Seq[(String, Long)] =
155     Seq((IngestCounters.Ingested, written.get()), (IngestCounters.Failed, failed.get()))
156 
157   class LocalIngestWorker(file: FileHandle) extends Runnable {
158     override def run(): Unit = {
159       try {
160         converters.borrow { converter =>
161           WithClose(file.open) { streams =>
162             streams.foreach { case (name, is) =>
163               val params = EvaluationContext.inputFileParam(name.getOrElse(file.path))
164               val success = converter.createEvaluationContext().success
165               val ec = converter.createEvaluationContext(params, success, globalFailures)
166               WithClose(LocalConverterIngest.this.features(converter.process(is, ec))) { features =>
167                 writers.borrow { writer =>
168                   var count = batches.get(writer)
169                   if (count == null) {
170                     count = new AtomicInteger(0)
171                     batches.put(writer, count)
172                   }
173                   val helper = FeatureWriterHelper(writer)
174                   features.foreach { sf =>
175                     try {
176                       helper.write(sf)
177                       written.incrementAndGet()
178                       count.incrementAndGet()
179                     } catch {
180                       case NonFatal(e) =>
181                         logger.error(s"Failed to write '${DataUtilities.encodeFeature(sf)}'", e)
182                         failed.incrementAndGet()
183                     }
184                     if (count.get % batch == 0) {
185                       count.set(0)
186                       writer match {
187                         case f: Flushable => f.flush()
188                         case _ => // no-op
189                       }
190                     }
191                   }
192                 }
193               }
194             }
195           }
196         }
197       } catch {
198         case e @ (_: ClassNotFoundException | _: NoClassDefFoundError) =>
199           // Rethrow exception so it can be caught by getting the future of this runnable in the main thread
200           // which will in turn cause the exception to be handled by org.locationtech.geomesa.tools.Runner
201           // Likely all threads will fail if a dependency is missing so it will terminate quickly
202           throw e
203 
204         case NonFatal(e) =>
205           // Don't kill the entire program b/c this thread was bad! use outer try/catch
206           val msg = s"Fatal error running local ingest worker on ${file.path}"
207           Command.user.error(msg)
208           logger.error(msg, e)
209           errors.incrementAndGet()
210       } finally {
211         latch.countDown()
212         bytesRead.addAndGet(file.length)
213       }
214     }
215   }
216 }
