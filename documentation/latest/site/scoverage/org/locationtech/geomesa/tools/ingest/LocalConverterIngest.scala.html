35 /**
36  * Ingestion that uses geomesa converters to process input files
37  *
38  * @param ds data store
39  * @param sft simple feature type
40  * @param converterConfig converter definition
41  * @param inputs paths to ingest
42  * @param numThreads how many threads to use
43  */
44 class LocalConverterIngest(
45     ds: DataStore,
46     dsParams: java.util.Map[String, _],
47     sft: SimpleFeatureType,
48     converterConfig: Config,
49     inputs: Inputs,
50     numThreads: Int
51   ) extends Awaitable with LazyLogging {
52 
53   private val files = inputs.handles
54   private val latch = new CountDownLatch(files.length)
55 
56   private val threads = if (numThreads <= files.length) { numThreads } else {
57     Command.user.warn("Can't use more threads than there are input files - reducing thread count")
58     files.length
59   }
60 
61   private val batch = IngestCommand.LocalBatchSize.toInt.getOrElse {
62     throw new IllegalArgumentException(
63       s"Invalid batch size for property ${IngestCommand.LocalBatchSize.property}: " +
64           IngestCommand.LocalBatchSize.get)
65   }
66 
67   private val es = Executors.newFixedThreadPool(threads)
68 
69   // global counts shared among threads
70   private val written = new AtomicLong(0)
71   private val failed = new AtomicLong(0)
72   private val errors = new AtomicInteger(0)
73 
74   private val bytesRead = new AtomicLong(0L)
75 
76   private val batches = new ConcurrentHashMap[FeatureWriter[SimpleFeatureType, SimpleFeature], AtomicInteger](threads)
77 
78   // keep track of failure at a global level, keep line counts and success local
79   private val globalFailures = new com.codahale.metrics.Counter {
80     override def inc(): Unit = failed.incrementAndGet()
81     override def inc(n: Long): Unit = failed.addAndGet(n)
82     override def dec(): Unit = failed.decrementAndGet()
83     override def dec(n: Long): Unit = failed.addAndGet(-1 * n)
84     override def getCount: Long = failed.get()
85   }
86 
87   private val progress: () => Float =
88     if (inputs.stdin) {
89       () => .99f // we don't know how many bytes are actually available
90     } else {
91       val length = files.map(_.length).sum.toFloat // only evaluate once
92       () => bytesRead.get / length
93     }
94 
95   Command.user.info(s"Ingesting ${if (inputs.stdin) { "from stdin" } else { TextTools.getPlural(files.length, "file") }} " +
96       s"with ${TextTools.getPlural(threads, "thread")}")
97 
98   private val converters = CloseablePool(SimpleFeatureConverter(sft, converterConfig), threads)
99   private val writers = {
100     def factory: FeatureWriter[SimpleFeatureType, SimpleFeature] =
101       if (threads > 1 && ds.getClass.getSimpleName.equals("JDBCDataStore")) {
102         // creates a new data store for each writer to avoid synchronized blocks in JDBCDataStore.
103         // the synchronization is to allow for generated fids from the database.
104         // generally, this shouldn't be an issue since we use provided fids,
105         // but running with 1 thread  would restore the old behavior
106         new DataStoreWriter(dsParams, sft.getTypeName)
107       } else {
108         ds.getFeatureWriterAppend(sft.getTypeName, Transaction.AUTO_COMMIT)
109       }
110     CloseablePool(factory, threads)
111   }
112 
113   private val closer = new Runnable() {
114     override def run(): Unit = {
115       latch.await()
116       CloseWithLogging(converters)
117       CloseWithLogging(writers).foreach(_ => errors.incrementAndGet())
118     }
119   }
120 
121   private val futures = files.map(f => es.submit(new LocalIngestWorker(f))) :+ CachedThreadPool.submit(closer)
122 
123   es.shutdown()
124 
125   override def await(reporter: StatusCallback): JobResult = {
126     while (!es.isTerminated) {
127       Thread.sleep(500)
128       reporter("", progress(), counters, done = false)
129     }
130     reporter("", progress(), counters, done = true)
131 
132     // Get all futures so that we can propagate the logging up to the top level for handling
133     // in org.locationtech.geomesa.tools.Runner to catch missing dependencies
134     futures.foreach(_.get)
135 
136     if (errors.get > 0) {
137       JobFailure("Some files caused errors, check logs for details")
138     } else {
139       val message =
140         if (inputs.stdin) { "from stdin" } else if (files.lengthCompare(1) == 0) { s"for file ${files.head.path}" } else { "" }
141       JobSuccess(message, counters.toMap)
142     }
143   }
144 
145   /**
146    * Hook to allow modification of the feature returned by the converter
147    *
148    * @param iter features
149    * @return
150    */
151   protected def features(iter: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = iter
152 
153   private def counters: Seq[(String, Long)] =
154     Seq((IngestCounters.Ingested, written.get()), (IngestCounters.Failed, failed.get()))
155 
156   class LocalIngestWorker(file: FileHandle) extends Runnable {
157     override def run(): Unit = {
158       try {
159         converters.borrow { converter =>
160           WithClose(file.open) { streams =>
161             streams.foreach { case (name, is) =>
162               val params = EvaluationContext.inputFileParam(name.getOrElse(file.path))
163               val success = converter.createEvaluationContext().success
164               val ec = converter.createEvaluationContext(params, success, globalFailures)
165               WithClose(LocalConverterIngest.this.features(converter.process(is, ec))) { features =>
166                 writers.borrow { writer =>
167                   var count = batches.get(writer)
168                   if (count == null) {
169                     count = new AtomicInteger(0)
170                     batches.put(writer, count)
171                   }
172                   features.foreach { sf =>
173                     try {
174                       FeatureUtils.write(writer, sf)
175                       written.incrementAndGet()
176                       count.incrementAndGet()
177                     } catch {
178                       case NonFatal(e) =>
179                         logger.error(s"Failed to write '${DataUtilities.encodeFeature(sf)}'", e)
180                         failed.incrementAndGet()
181                     }
182                     if (count.get % batch == 0) {
183                       count.set(0)
184                       writer match {
185                         case f: Flushable => f.flush()
186                         case _ => // no-op
187                       }
188                     }
189                   }
190                 }
191               }
192             }
193           }
194         }
195       } catch {
196         case e @ (_: ClassNotFoundException | _: NoClassDefFoundError) =>
197           // Rethrow exception so it can be caught by getting the future of this runnable in the main thread
198           // which will in turn cause the exception to be handled by org.locationtech.geomesa.tools.Runner
199           // Likely all threads will fail if a dependency is missing so it will terminate quickly
200           throw e
201 
202         case NonFatal(e) =>
203           // Don't kill the entire program b/c this thread was bad! use outer try/catch
204           val msg = s"Fatal error running local ingest worker on ${file.path}"
205           Command.user.error(msg)
206           logger.error(msg, e)
207           errors.incrementAndGet()
208       } finally {
209         latch.countDown()
210         bytesRead.addAndGet(file.length)
211       }
212     }
213   }
214 }
