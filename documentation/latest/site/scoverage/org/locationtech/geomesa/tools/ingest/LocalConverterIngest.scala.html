35 /**
36  * Ingestion that uses geomesa converters to process input files
37  *
38  * @param ds data store
39  * @param sft simple feature type
40  * @param converterConfig converter definition
41  * @param inputs paths to ingest
42  * @param numThreads how many threads to use
43  */
44 class LocalConverterIngest(
45     ds: DataStore,
46     dsParams: java.util.Map[String, _],
47     sft: SimpleFeatureType,
48     converterConfig: Config,
49     inputs: Inputs,
50     numThreads: Int
51   ) extends Awaitable with LazyLogging {
52 
53   private val files = inputs.handles
54   private val latch = new CountDownLatch(files.length)
55 
56   private val threads = if (numThreads <= files.length) { numThreads } else {
57     Command.user.warn("Can't use more threads than there are input files - reducing thread count")
58     files.length
59   }
60 
61   private val batch = IngestCommand.LocalBatchSize.toInt.getOrElse {
62     throw new IllegalArgumentException(
63       s"Invalid batch size for property ${IngestCommand.LocalBatchSize.property}: " +
64           IngestCommand.LocalBatchSize.get)
65   }
66 
67   private val es = Executors.newFixedThreadPool(threads)
68 
69   // global counts shared among threads
70   private val written = new AtomicLong(0)
71   private val failed = new AtomicLong(0)
72   private val errors = new AtomicInteger(0)
73 
74   private val bytesRead = new AtomicLong(0L)
75 
76   private val batches = new ConcurrentHashMap[FeatureWriter[SimpleFeatureType, SimpleFeature], AtomicInteger](threads)
77 
78   // keep track of failure at a global level, but we don't count successes until they're written to the store
79   private val listener = new EvaluationContext.ContextListener {
80     override def onSuccess(i: Int): Unit = {}
81     override def onFailure(i: Int): Unit = failed.addAndGet(i)
82   }
83 
84   private val progress: () => Float =
85     if (inputs.stdin) {
86       () => .99f // we don't know how many bytes are actually available
87     } else {
88       val length = files.map(_.length).sum.toFloat // only evaluate once
89       () => bytesRead.get / length
90     }
91 
92   Command.user.info(s"Ingesting ${if (inputs.stdin) { "from stdin" } else { TextTools.getPlural(files.length, "file") }} " +
93       s"with ${TextTools.getPlural(threads, "thread")}")
94 
95   private val converters = CloseablePool(SimpleFeatureConverter(sft, converterConfig), threads)
96   private val writers = {
97     def factory: FeatureWriter[SimpleFeatureType, SimpleFeature] =
98       if (threads > 1 && ds.getClass.getSimpleName.equals("JDBCDataStore")) {
99         // creates a new data store for each writer to avoid synchronized blocks in JDBCDataStore.
100         // the synchronization is to allow for generated fids from the database.
101         // generally, this shouldn't be an issue since we use provided fids,
102         // but running with 1 thread  would restore the old behavior
103         new DataStoreWriter(dsParams, sft.getTypeName)
104       } else {
105         ds.getFeatureWriterAppend(sft.getTypeName, Transaction.AUTO_COMMIT)
106       }
107     CloseablePool(factory, threads)
108   }
109 
110   private val closer = new Runnable() {
111     override def run(): Unit = {
112       latch.await()
113       CloseWithLogging(converters)
114       CloseWithLogging(writers).foreach(_ => errors.incrementAndGet())
115     }
116   }
117 
118   private val futures = files.map(f => es.submit(new LocalIngestWorker(f))) :+ CachedThreadPool.submit(closer)
119 
120   es.shutdown()
121 
122   override def await(reporter: StatusCallback): JobResult = {
123     while (!es.isTerminated) {
124       Thread.sleep(500)
125       reporter("", progress(), counters, done = false)
126     }
127     reporter("", progress(), counters, done = true)
128 
129     // Get all futures so that we can propagate the logging up to the top level for handling
130     // in org.locationtech.geomesa.tools.Runner to catch missing dependencies
131     futures.foreach(_.get)
132 
133     if (errors.get > 0) {
134       JobFailure("Some files caused errors, check logs for details")
135     } else {
136       val message =
137         if (inputs.stdin) { "from stdin" } else if (files.lengthCompare(1) == 0) { s"for file ${files.head.path}" } else { "" }
138       JobSuccess(message, counters.toMap)
139     }
140   }
141 
142   /**
143    * Hook to allow modification of the feature returned by the converter
144    *
145    * @param iter features
146    * @return
147    */
148   protected def features(iter: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = iter
149 
150   private def counters: Seq[(String, Long)] =
151     Seq((IngestCounters.Ingested, written.get()), (IngestCounters.Failed, failed.get()))
152 
153   class LocalIngestWorker(file: FileHandle) extends Runnable {
154     override def run(): Unit = {
155       try {
156         converters.borrow { converter =>
157           WithClose(file.open) { streams =>
158             streams.foreach { case (name, is) =>
159               val params = EvaluationContext.inputFileParam(name.getOrElse(file.path))
160               val ec = converter.createEvaluationContext(params).withListener(listener)
161               WithClose(LocalConverterIngest.this.features(converter.process(is, ec))) { features =>
162                 writers.borrow { writer =>
163                   var count = batches.get(writer)
164                   if (count == null) {
165                     count = new AtomicInteger(0)
166                     batches.put(writer, count)
167                   }
168                   val helper = FeatureWriterHelper(writer)
169                   features.foreach { sf =>
170                     try {
171                       helper.write(sf)
172                       written.incrementAndGet()
173                       count.incrementAndGet()
174                     } catch {
175                       case NonFatal(e) =>
176                         logger.error(s"Failed to write '${DataUtilities.encodeFeature(sf)}'", e)
177                         failed.incrementAndGet()
178                     }
179                     if (count.get % batch == 0) {
180                       count.set(0)
181                       writer match {
182                         case f: Flushable => f.flush()
183                         case _ => // no-op
184                       }
185                     }
186                   }
187                 }
188               }
189             }
190           }
191         }
192       } catch {
193         case e @ (_: ClassNotFoundException | _: NoClassDefFoundError) =>
194           // Rethrow exception so it can be caught by getting the future of this runnable in the main thread
195           // which will in turn cause the exception to be handled by org.locationtech.geomesa.tools.Runner
196           // Likely all threads will fail if a dependency is missing so it will terminate quickly
197           throw e
198 
199         case NonFatal(e) =>
200           // Don't kill the entire program b/c this thread was bad! use outer try/catch
201           val msg = s"Fatal error running local ingest worker on ${file.path}"
202           Command.user.error(msg)
203           logger.error(msg, e)
204           errors.incrementAndGet()
205       } finally {
206         latch.countDown()
207         bytesRead.addAndGet(file.length)
208       }
209     }
210   }
211 }
