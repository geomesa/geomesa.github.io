27 class RedisEnrichmentCache(jedisPool: RedisConnectionBuilder,
28                            expiration: Long = -1,
29                            localCache: Boolean) extends EnrichmentCache {
30   type KV = java.util.Map[String, String]
31 
32   private val builder =
33     if (expiration > 0) {
34       Caffeine.newBuilder().expireAfterWrite(expiration, TimeUnit.MILLISECONDS)
35     } else {
36       if (!localCache) {
37         Caffeine.newBuilder().expireAfterWrite(0, TimeUnit.MILLISECONDS).maximumSize(0)
38       } else {
39         // TODO we shouldn't allow non-expiring caches :/
40         Caffeine.newBuilder()
41       }
42     }
43 
44   private val cache: LoadingCache[String, KV] =
45     builder
46       .build(new CacheLoader[String, KV] {
47         override def load(k: String): KV = {
48           val conn = jedisPool.getConn
49           try {
50             conn.hgetAll(k)
51           } finally {
52             // Note: for a JedisPool this only returns it to the pool instead of actionally
53             // closing the connection (so it's safe to call close() on the conn)
54             conn.close()
55           }
56         }
57       })
58 
59   override def get(args: Array[String]): Any = cache.get(args(0)).get(args(1))
60   override def put(args: Array[String], value: Any): Unit = throw new UnsupportedOperationException()
61   override def clear(): Unit = throw new UnsupportedOperationException()
62   override def close(): Unit = jedisPool.close()
63 }
64 
65 class RedisEnrichmentCacheFactory extends EnrichmentCacheFactory {
66   override def canProcess(conf: Config): Boolean = conf.hasPath("type") && conf.getString("type").equals("redis")
67 
68   override def build(conf: Config): EnrichmentCache = {
69     val redisUrl = {
70       val url = conf.getString("redis-url")
71       Some(url).filter(u => Try(new URI(u)).toOption.exists(JedisURIHelper.isValid)).getOrElse {
72         if (url.indexOf(":") == -1) { url } else { s"redis://$url" }
73       }
74     }
75     val timeout = if (conf.hasPath("expiration")) conf.getLong("expiration") else -1
76     val connBuilder: RedisConnectionBuilder = new RedisConnectionBuilder {
77       private val pool = new JedisPool(redisUrl)
78       override def getConn: Jedis = pool.getResource
79       override def close(): Unit = pool.close()
80     }
81 
82     val localCache = Try(conf.getBoolean("local-cache")).getOrElse(true)
83     new RedisEnrichmentCache(connBuilder, timeout, localCache)
84   }
