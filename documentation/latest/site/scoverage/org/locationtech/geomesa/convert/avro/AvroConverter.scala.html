32   private val schema = config.schema match {
33     case SchemaEmbedded => None
34     case SchemaString(s) => Some(new Parser().parse(s))
35     case SchemaFile(s) =>
36       val loader = Option(Thread.currentThread.getContextClassLoader).getOrElse(getClass.getClassLoader)
37       val res = Option(loader.getResourceAsStream(s)).orElse(Option(getClass.getResourceAsStream(s))).getOrElse {
38         throw new IllegalArgumentException(s"Could not load schema resource at $s")
39       }
40       Some(new Parser().parse(res))
41   }

43   // if required, set the raw bytes in the result array
44   private val requiresBytes = {
45     val expressions = config.idField.toSeq ++ fields.flatMap(_.transforms) ++ config.userData.values
46     Expression.flatten(expressions).contains(Column(0))
47   }
48 
49   override protected def parse(is: InputStream, ec: EvaluationContext): CloseableIterator[GenericRecord] = {
50     schema match {
51       case Some(s) if requiresBytes => new GenericRecordBytesIterator(new CopyingInputStream(is), s, ec)
52       case Some(s)                  => new GenericRecordIterator(is, s, ec)
53       case None    if requiresBytes => new FileStreamBytesIterator(is, ec)
54       case None                     => new FileStreamIterator(is, ec)
55     }
56   }
57 
58   override protected def values(parsed: CloseableIterator[GenericRecord],
59                                 ec: EvaluationContext): CloseableIterator[Array[Any]] = {
60     val array = Array.ofDim[Any](2)
61     if (requiresBytes) {
62       parsed.map { record => array(0) = record.get(BytesField); array(1) = record; array }
63     } else {
64       parsed.map { record => array(1) = record; array }
65     }
66   }

73   val BytesField = "__bytes__"
74 
75   /**
76     * Add a `__bytes__` field to the schema, for storing the raw bytes
77     *
78     * @param schema schema
79     * @return
80     */
81   def addBytes(schema: Schema): Schema = {
82     schema.getType match {
83       case Schema.Type.RECORD =>
84         val fields = new java.util.ArrayList[Schema.Field](schema.getFields.size() + 1)
85         schema.getFields.asScala.foreach { field =>
86           fields.add(new Schema.Field(field.name, field.schema, field.doc, field.defaultVal()))
87         }
88         fields.add(new Schema.Field(BytesField, Schema.create(Schema.Type.BYTES), "raw bytes", ""))
89 
90         val updated = Schema.createRecord(schema.getName, schema.getDoc, schema.getNamespace, schema.isError)
91         updated.setFields(fields)
92         updated
93 
94       case Schema.Type.UNION =>
95         Schema.createUnion(schema.getTypes.asScala.map(s => addBytes(s)).toSeq: _*)
96 
97       case _ =>
98         throw new UnsupportedOperationException(
99           s"Raw Avro bytes (i.e. $$0) is not implemented for schema type ${schema.getType}")
100     }
101   }

145   /**
146     * Reads avro records using a pre-defined schema, setting the bytes for each record in a
147     * special `__bytes__` field
148     *
149     * @param is input stream
150     * @param schema schema
151     * @param ec evaluation context
152     */
153   class GenericRecordBytesIterator private [AvroConverter] (is: CopyingInputStream, schema: Schema, ec: EvaluationContext)
154       extends CloseableIterator[GenericRecord] {
155 
156     private val reader = new GenericDatumReader[GenericRecord](schema, addBytes(schema))
157     private val decoder = DecoderFactory.get.binaryDecoder(is, null)
158     private var record: GenericRecord = _
159 
160     override def hasNext: Boolean = !decoder.isEnd
161 
162     override def next(): GenericRecord = {
163       ec.line += 1
164       record = reader.read(record, decoder)
165       // parse out the bytes read and set them in the record
166       // check to see if the decoder buffered some bytes that weren't actually used
167       val buffered = decoder.inputStream().available()
168       record.put(BytesField, is.replay(is.copied - buffered))
169       record
170     }
171 
172     override def close(): Unit = is.close()
173   }

198   /**
199     * Reads avro records from an avro file, with the schema embedded, setting the bytes for
200     * each record in a special `__bytes__` field
201     *
202     * @param is input
203     * @param ec evaluation context
204     */
205   class FileStreamBytesIterator private [AvroConverter] (is: InputStream, ec: EvaluationContext)
206       extends CloseableIterator[GenericRecord] {
207 
208     private val reader = new GenericDatumReader[GenericRecord]()
209     private val stream = new DataFileStream(is, reader)
210     private var record: GenericRecord = _
211 
212     reader.setExpected(addBytes(reader.getSchema))
213 
214     // we can't tell which bytes correspond to which feature (due to buffering). if we could access the
215     // underlying avro encoder we could figure it out, but it is not exposed through DataFileStream. instead,
216     // re-serialize each record to get the raw bytes
217     private val out = new ByteArrayOutputStream()
218     private val writer = new GenericDatumWriter[GenericRecord](stream.getSchema)
219     private val encoder = EncoderFactory.get.binaryEncoder(out, null)
220 
221     override def hasNext: Boolean = stream.hasNext
222 
223     override def next(): GenericRecord = {
224       ec.line += 1
225       record = stream.next(record)
226       // regenerate the bytes read and set them in the record
227       out.reset()
228       writer.write(record, encoder)
229       encoder.flush()
230       record.put(BytesField, out.toByteArray)
231       record
232     }
233 
234     override def close(): Unit = stream.close()
235   }
