20 /**
21  * Attempts to resolve Converter config from arguments as either a string or
22  * as a filename containing the converter config
23  */
24 object ConverterConfigResolver extends ArgResolver[Config, ConfArgs] with LazyLogging {
25 
26   import org.locationtech.geomesa.utils.conf.ArgResolver.ArgTypes._
27 
28   private val fileNameReg = """([^.]*)\.([^.]*)""" // e.g. "foo.bar"
29   private val confStrings = Seq("geomesa{", "geomesa {", "geomesa.converters")
30 
31   override def argType(args: ConfArgs): ArgTypes = {
32     // guess the type we are trying to parse, to determine which error we show for failures
33     // order is important here
34     if (confStrings.exists(args.config.contains)) {
35       CONFSTR
36     } else if (args.config.matches(fileNameReg) || args.config.contains("/")) {
37       PATH
38     } else {
39       NAME
40     }
41   }
42 
43   override val parseMethodList: List[ConfArgs => ResEither] = List[ConfArgs => ResEither](
44     getLoadedConf,
45     parseFile,
46     parseString
47   )
48 
49   private [ConverterConfigResolver] def getLoadedConf(args: ConfArgs): ResEither = {
50     ConverterConfigLoader.confs.get(args.config) match {
51       case Some(conf) => Right(conf)
52       case None =>
53         val error =
54           new RuntimeException(s"Unable to find config with name: ${args.config}\n  " +
55               s"Available configs: ${ConverterConfigLoader.confs.keys.mkString(", ")}")
56         Left((s"Unable to get loaded conf: ${args.config}", error, NAME))
57     }
58   }
59 
60   private [ConverterConfigResolver] def parseString(args: ConfArgs): ResEither = {
61     try {
62       val confs = SimpleConverterConfigParser.parseConf(ConfigFactory.parseString(args.config, parseOpts).resolve())
63       if (confs.size > 1) {
64         logger.warn(s"Found more than one SFT conf in arg '${args.config}'")
65       }
66       Right(confs.values.head)
67     } catch {
68       case NonFatal(e) => Left((s"Unable to parse config from string: ${args.config}", e, CONFSTR))
69     }
70   }
71 
72   private [ConverterConfigResolver] def parseFile(args: ConfArgs): ResEither = {
73     try {
74       val handle = PathUtils.interpretPath(args.config).headOption.getOrElse {
75         throw new RuntimeException(s"Could not read file: ${args.config}")
76       }
77       WithClose(handle.open) { is =>
78         if (is.hasNext) {
79           val reader = new InputStreamReader(is.next._2, StandardCharsets.UTF_8)
80           val confs = SimpleConverterConfigParser.parseConf(ConfigFactory.parseReader(reader, parseOpts).resolve())
81           if (confs.size > 1) {
82             logger.warn(s"Found more than one SFT conf in arg '${args.config}'")
83           }
84           Right(confs.values.head)
85         } else {
86           throw new RuntimeException(s"Could not read file: ${args.config}")
87         }
88       }
89 
90     } catch {
91       case NonFatal(e) => Left((s"Unable to parse config from file: ${args.config}", e, PATH))
92     }
93   }
94 }
