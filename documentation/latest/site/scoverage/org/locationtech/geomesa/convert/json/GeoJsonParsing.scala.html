57   /**
58     * Parse a geojson feature element
59     *
60     * @see `isFeature` to determine if this is likely to succeed
61     *
62     * @param el element
63     * @return
64     */
65   def parseFeature(el: JsonElement): GeoJsonFeature = {
66     val obj = el.getAsJsonObject
67     val id = obj.get(IdKey) match {
68       case s: JsonPrimitive if s.isString => Some(s.getAsString)
69       case _ => None
70     }
71     val geometry = parseGeometry(obj.get(GeometryKey))
72     val props: Map[String, Any] = obj.get(PropertiesKey) match {
73       case o: JsonObject => parseElement(o, s"$$['$PropertiesKey']").toMap
74       case _ => Map.empty
75     }
76     GeoJsonFeature(id, geometry, props)
77   }
78 
79   /**
80     * Parse a geometry element
81     *
82     * @param el element
83     * @return
84     */
85   def parseGeometry(el: JsonElement): Geometry = el match {
86     case o: JsonObject    => parseGeometryObject(o).get
87     case o: JsonPrimitive => WKTUtils.read(o.getAsString)
88     case _: JsonNull      => null.asInstanceOf[Geometry]
89     case _ => throw new IllegalArgumentException(s"Unknown geometry type: $el")
90   }

93   /**
94    * Parse a json element
95    *
96    * @param elem element
97    * @param path json path to the object
98    * @return map of key is json path to value, value is a string, boolean or list[string]
99    */
100   def parseElement(elem: JsonElement, path: String): Seq[(String, Any)] = {
101     elem match {
102       case e: JsonPrimitive =>
103         if (e.isBoolean) {
104           Seq(path -> e.getAsBoolean)
105         } else {
106           // note: gson numbers don't have a defined type so type checking doesn't work
107           Seq(path -> e.getAsString)
108         }
109 
110       case e: JsonObject =>
111         parseGeometryObject(e) match {
112           case Success(geom) => Seq(path -> geom)
113           case Failure(_) =>
114             val builder = Seq.newBuilder[(String, Any)]
115             e.entrySet().asScala.foreach { entry =>
116               builder ++= parseElement(entry.getValue, s"$path['${entry.getKey}']")
117             }
118             builder.result
119         }
120 
121       case e: JsonArray =>
122         val list = new java.util.ArrayList[String](e.size())
123         var i = 0
124         while (i < e.size()) {
125           list.add(e.get(i).toString)
126           i += 1
127         }
128         Seq(path -> Collections.unmodifiableList(list))
129 
130       case _ =>
131         Seq.empty // no-op
132     }
133   }
134 
135   /**
136     * Parse a geometry object
137     *
138     * @param obj object
139     * @return
140     */
141   private def parseGeometryObject(obj: JsonObject): Try[Geometry] = {
142     val geomType = obj.get(TypeKey)
143 
144     if (geomType == null || !geomType.isJsonPrimitive) {
145       return Failure(new RuntimeException("Not a Geometry object"))
146     }
147 
148     geomType.getAsString.toLowerCase(Locale.US) match {
149       case "point" =>
150         Try(factory.createPoint(toPointCoords(obj.get(CoordinatesKey))))
151 
152       case "linestring" =>
153         Try(factory.createLineString(toCoordSeq(obj.get(CoordinatesKey))))
154 
155       case "polygon" =>
156         Try(toPolygon(obj.get(CoordinatesKey)))
157 
158       case "multipoint" =>
159         Try(factory.createMultiPoint(toCoordSeq(obj.get(CoordinatesKey))))
160 
161       case "multilinestring" =>
162         Try {
163           val coords =
164             obj.get(CoordinatesKey).getAsJsonArray.asScala
165               .map(c => factory.createLineString(toCoordSeq(c))).toArray
166           factory.createMultiLineString(coords)
167         }
168 
169       case "multipolygon" =>
170         Try(factory.createMultiPolygon(obj.get(CoordinatesKey).getAsJsonArray.asScala.map(toPolygon).toArray))
171 
172       case "geometrycollection" =>
173         Try(factory.createGeometryCollection(obj.get(GeometriesKey).getAsJsonArray.asScala.map(parseGeometry).toArray))
174 
175       case unknown =>
176         Failure(new NotImplementedError(s"Can't parse geometry type of $unknown"))
177     }
178   }
179 
180   private def toPointCoords(el: JsonElement): Coordinate = {
181     el.getAsJsonArray.asScala.map(_.getAsDouble).toSeq match {
182       case Seq(x, y)    => new Coordinate(x, y)
183       case Seq(x, y, z) => new Coordinate(x, y, z)
184       case s => throw new IllegalArgumentException(s"Invalid point - expected 2 or 3 values, got ${s.mkString(", ")}")
185     }
186   }
187 
188   private def toCoordSeq(el: JsonElement): CoordinateSequence =
189     new CoordinateArraySequence(el.getAsJsonArray.asScala.map(_.getAsJsonArray).map(toPointCoords).toArray)
190 
191   private def toPolygon(el: JsonElement): Polygon = {
192     val rings = el.getAsJsonArray.iterator.asScala.map(c => factory.createLinearRing(toCoordSeq(c)))
193     val shell = rings.next
194     if (rings.hasNext) {
195       factory.createPolygon(shell, rings.toArray)
196     } else {
197       factory.createPolygon(shell)
198     }
199   }
200 
201   private def isType(el: JsonElement, t: String): Boolean = el match {
202     case o: JsonObject => Option(o.get(TypeKey)).exists(e => e.isJsonPrimitive && e.getAsString == t)
203     case _ => false
204   }
205 }
