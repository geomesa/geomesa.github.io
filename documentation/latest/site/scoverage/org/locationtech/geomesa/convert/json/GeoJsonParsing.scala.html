56   /**
57     * Parse a geojson feature element
58     *
59     * @see `isFeature` to determine if this is likely to succeed
60     *
61     * @param el element
62     * @return
63     */
64   def parseFeature(el: JsonElement): GeoJsonFeature = {
65     val obj = el.getAsJsonObject
66     val id = obj.get(IdKey) match {
67       case s: JsonPrimitive if s.isString => Some(s.getAsString)
68       case _ => None
69     }
70     val geometry = parseGeometry(obj.get(GeometryKey))
71     val props: Map[String, Any] = obj.get(PropertiesKey) match {
72       case o: JsonObject => parseElement(o, s"$$['$PropertiesKey']").toMap
73       case _ => Map.empty
74     }
75     GeoJsonFeature(id, geometry, props)
76   }
77 
78   /**
79     * Parse a geometry element
80     *
81     * @param el element
82     * @return
83     */
84   def parseGeometry(el: JsonElement): Geometry = el match {
85     case o: JsonObject    => parseGeometryObject(o)
86     case o: JsonPrimitive => WKTUtils.read(o.getAsString)
87     case _: JsonNull      => null.asInstanceOf[Geometry]
88     case _ => throw new IllegalArgumentException(s"Unknown geometry type: $el")
89   }
90 
91   /**
92     * Parse a geometry object
93     *
94     * @param obj object
95     * @return
96     */
97   private def parseGeometryObject(obj: JsonObject): Geometry = {
98     obj.get(TypeKey).getAsString.toLowerCase match {
99       case "point" =>
100         factory.createPoint(toPointCoords(obj.get(CoordinatesKey)))
101 
102       case "linestring" =>
103         factory.createLineString(toCoordSeq(obj.get(CoordinatesKey)))
104 
105       case "polygon" =>
106         toPolygon(obj.get(CoordinatesKey))
107 
108       case "multipoint" =>
109         factory.createMultiPoint(toCoordSeq(obj.get(CoordinatesKey)))
110 
111       case "multilinestring" =>
112         val coords = obj.get(CoordinatesKey).getAsJsonArray.asScala
113             .map(c => factory.createLineString(toCoordSeq(c))).toArray
114         factory.createMultiLineString(coords)
115 
116       case "multipolygon" =>
117         factory.createMultiPolygon(obj.get(CoordinatesKey).getAsJsonArray.asScala.map(toPolygon).toArray)
118 
119       case "geometrycollection" =>
120         factory.createGeometryCollection(obj.get(GeometriesKey).getAsJsonArray.asScala.map(parseGeometry).toArray)
121 
122       case unknown =>
123         throw new NotImplementedError(s"Can't parse geometry type of $unknown")
124     }
125   }
126 
127   private def toPointCoords(el: JsonElement): Coordinate = {
128     el.getAsJsonArray.asScala.map(_.getAsDouble).toSeq match {
129       case Seq(x, y)    => new Coordinate(x, y)
130       case Seq(x, y, z) => new Coordinate(x, y, z)
131       case s => throw new IllegalArgumentException(s"Invalid point - expected 2 or 3 values, got ${s.mkString(", ")}")
132     }
133   }
134 
135   private def toCoordSeq(el: JsonElement): CoordinateSequence =
136     new CoordinateArraySequence(el.getAsJsonArray.asScala.map(_.getAsJsonArray).map(toPointCoords).toArray)
137 
138   private def toPolygon(el: JsonElement): Polygon = {
139     val rings = el.getAsJsonArray.iterator.asScala.map(c => factory.createLinearRing(toCoordSeq(c)))
140     val shell = rings.next
141     if (rings.hasNext) {
142       factory.createPolygon(shell, rings.toArray)
143     } else {
144       factory.createPolygon(shell)
145     }
146   }
147 
148   private def isType(el: JsonElement, t: String): Boolean = el match {
149     case o: JsonObject => Option(o.get(TypeKey)).exists(e => e.isJsonPrimitive && e.getAsString == t)
150     case _ => false
151   }
152 }

158   /**
159    * Parsed geojson feature element
160    *
161    * @param id id, if present
162    * @param geom geometry
163    * @param properties 'properties' values - key is json path to value, value is a string, boolean or list[string].
164    *                   nested elements will be flattened out, with a path pointing into the element
165    */
166   case class GeoJsonFeature(id: Option[String], geom: Geometry, properties: Map[String, Any])
167 
168   private val FeatureType = "Feature"
169   private val FeatureCollectionType = "FeatureCollection"
170 
171   private val TypeKey = "type"
172   private val FeaturesKey = "features"
173   private val CoordinatesKey = "coordinates"
174   private val PropertiesKey = "properties"
175   private val GeometryKey = "geometry"
176   private val GeometriesKey = "geometries"
177   private val IdKey = "id"
178 
179   /**
180    * Parse a json element
181    *
182    * @param elem element
183    * @param path json path to the object
184    * @return map of key is json path to value, value is a string, boolean or list[string]
185    */
186   def parseElement(elem: JsonElement, path: String): Seq[(String, Any)] = {
187     elem match {
188       case e: JsonPrimitive =>
189         if (e.isBoolean) {
190           Seq(path -> e.getAsBoolean)
191         } else {
192           // note: gson numbers don't have a defined type so type checking doesn't work
193           Seq(path -> e.getAsString)
194         }
195 
196       case e: JsonObject =>
197         val builder = Seq.newBuilder[(String, Any)]
198         e.entrySet().asScala.foreach { entry =>
199           builder ++= parseElement(entry.getValue, s"$path['${entry.getKey}']")
200         }
201         builder.result
202 
203       case e: JsonArray =>
204         val list = new java.util.ArrayList[String](e.size())
205         var i = 0
206         while (i < e.size()) {
207           list.add(e.get(i).toString)
208           i += 1
209         }
210         Seq(path -> Collections.unmodifiableList(list))
211 
212       case _ =>
213         Seq.empty // no-op
214     }
215   }
