31   @throws(classOf[ParsingException])
32   def parse(path: String, report: Boolean = true): AvroPath = {
33     if (path == null) {
34       throw new IllegalArgumentException("Invalid path string: null")
35     }
36     val runner = if (report) { ReportingParseRunner(Parser.path) } else { BasicParseRunner(Parser.path) }
37     val parsing = runner.run(path)
38     parsing.result.getOrElse(throw new ParsingException(s"Error parsing avro path: $path"))
39   }

41   private def convert(record: Any): Any = {
42     record match {
43       case x: Utf8                       => x.toString
44       case x: ByteBuffer                 => convertBytes(x)
45       case x: GenericFixed               => x.bytes()
46       case x: GenericEnumSymbol[_]       => x.toString
47       case x: GenericArray[Any]          => convertList(x)
48       case x: java.util.Map[String, Any] => convertMap(x)
49       case x                             => x
50     }
51   }
52 
53   private def convertBytes(x: ByteBuffer): Array[Byte] = {
54     val start = x.position()
55     val length = x.limit() - start
56     val bytes = Array.ofDim[Byte](length)
57     x.get(bytes, 0, length)
58     x.position(start)
59     bytes
60   }
61 
62   private def convertList(list: java.util.List[Any]): java.util.List[Any] = {
63     val result = new java.util.ArrayList[Any](list.size())
64     val iter = list.iterator()
65     while (iter.hasNext) {
66       result.add(convert(iter.next()))
67     }
68     result
69   }
70 
71   // note: maps get re-used, so we need to copy to a new structure
72   private def convertMap(map: java.util.Map[String, Any]): java.util.Map[String, Any] = {
73     val result = new java.util.HashMap[String, Any](map.size())
74     val iter = map.entrySet().iterator()
75     while (iter.hasNext) {
76       val entry = iter.next
77       result.put(convert(entry.getKey).asInstanceOf[String], convert(entry.getValue))
78     }
79     result
80   }

82   case class PathExpr(field: String, predicate: AvroPredicate) extends AvroPath {
83     override def eval(record: Any): Option[Any] = {
84       record match {
85         // schema null check supports our parquet support
86         case gr: GenericRecord if gr.getSchema == null || gr.getSchema.getField(field) != null =>
87           gr.get(field) match {
88             case x: GenericRecord => Some(x).filter(predicate)
89             case x                => Option(convert(x))
90           }
91 
92         case _ => None
93       }
94     }
95   }
96 
97   case class ArrayRecordExpr(field: String, matched: String) extends AvroPath {
98 
99     import scala.collection.JavaConverters._
100 
101     override def eval(r: Any): Option[Any] = r match {
102       case a: java.util.List[GenericRecord] => a.asScala.find(predicate)
103       case _ => None
104     }
105 
106     private def predicate(record: GenericRecord): Boolean = {
107       record.get(field) match {
108         case x: Utf8 => x.toString == matched
109         case x       => x == matched
110       }
111     }
112   }
113 
114   case class CompositeExpr(se: Seq[AvroPath]) extends AvroPath {
115     override def eval(r: Any): Option[Any] = r match {
116       case gr: GenericRecord => se.foldLeft[Option[Any]](Some(gr))((result, current) => result.flatMap(current.eval))
117       case _ => None
118     }
119   }
120 
121   case class UnionTypeFilter(n: String) extends AvroPredicate {
122     override def apply(v1: GenericRecord): Boolean = v1.getSchema.getName == n
123   }

125   class AvroPathParser extends BasicParser {
126 
127     import org.parboiled.scala._
128 
129     // full simple feature spec
130     def path: Rule1[AvroPath] = rule("Path") {
131       oneOrMore(pathExpression | arrayRecord) ~ EOI ~~> {
132         paths => if (paths.lengthCompare(1) == 0) { paths.head } else { CompositeExpr(paths) }
133       }
134     }
135 
136     private def pathExpression: Rule1[PathExpr] = rule("PathExpression") {
137       "/" ~ identifier ~ optional("$type=" ~ identifier) ~~> {
138         (field, typed) => PathExpr(field, typed.map(UnionTypeFilter.apply).getOrElse(_ => true))
139       }
140     }
141 
142     private def arrayRecord: Rule1[ArrayRecordExpr] = rule("ArrayRecord") {
143       ("[$" ~ identifier ~ "=" ~ identifier ~ "]") ~~> {
144         (field, matched) => ArrayRecordExpr(field, matched)
145       }
146     }
147 
148     private def identifier: Rule1[String] = rule("Identifier") {
149       oneOrMore(char | anyOf(".-")) ~> { s => s }
150     }
151   }
