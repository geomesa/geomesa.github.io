19 object ConverterConfigLoader extends LazyLogging {
20 
21   val ConfigPathProperty: SystemProperty = SystemProperty("org.locationtech.geomesa.converter.config.path", "geomesa.converters")
22 
23   val ConverterNameKey = "converter-name"
24 
25   private val configProviders = {
26     val pList = ServiceLoader.load(classOf[ConverterConfigProvider]).asScala.toList
27     logger.debug(s"Found ${pList.size} SPI providers for ${classOf[ConverterConfigProvider].getName}" +
28       s": ${pList.map(_.getClass.getName).mkString(", ")}")
29     pList
30   }
31 
32   def path: String = ConfigPathProperty.get
33 
34   // this is intentionally a method to allow reloading by the providers
35   def confs: Map[String, Config] =
36     configProviders.map(_.loadConfigs.asScala).reduce( _ ++ _).toMap.map { case (k, v) =>
37       // add the key used to identify the converter (used for metrics)
38       k -> (if (v.hasPath(ConverterNameKey)) { v } else { v.withValue(ConverterNameKey, ConfigValueFactory.fromAnyRef(k)) })
39     }
40 
41   // Public API
42   def listConverterNames: List[String] = confs.keys.toList
43   def getAllConfigs: Map[String, Config] = confs
44   def configForName(name: String): Option[Config] = confs.get(name)
45 
46   // Rebase a config to the converter root...allows standalone
47   // configurations to start with "converter", "input-converter"
48   // or optional other prefix string
49   @deprecated("unused")
50   def rebaseConfig(conf: Config, pathOverride: Option[String] = None): Config = {
51     import org.locationtech.geomesa.utils.conf.ConfConversions._
52     (pathOverride.toSeq ++ Seq(path, "converter", "input-converter"))
53       .foldLeft(conf)( (c, p) => c.getConfigOpt(p).map(c.withFallback).getOrElse(c))
54   }

58 trait GeoMesaConvertParser extends LazyLogging {
59   def parseConf(config: Config): Map[String, Config] = {
60     logger.trace(s"Attempting to load Converters from path ${ConverterConfigLoader.path}")
61     if (!config.hasPath(ConverterConfigLoader.path)) {
62       Map.empty[String, Config]
63     } else {
64       val confs = config.getConfig(ConverterConfigLoader.path)
65       confs.root.keySet.asScala.map { k =>
66         logger.trace(s"Found conf block $k")
67         k -> confs.getConfig(k)
68       }.toMap[String, Config]
69     }
70   }
71 }

86 /** Load Config from URLs */
87 class URLConfigProvider extends ConverterConfigProvider with GeoMesaConvertParser {
88   import URLConfigProvider._
89 
90   override def loadConfigs(): java.util.Map[String, Config] = {
91     val confs = configURLs.flatMap { url =>
92       try {
93         Some(ConfigFactory.parseURL(url).resolve())
94       } catch {
95         case e: Throwable =>
96           logger.warn(s"Unable to load converter config from url $url")
97           logger.trace(s"Unable to load converter config from url $url", e)
98           None
99       }
100     }
101     confs.reduceLeftOption(_.withFallback(_)).map(parseConf).getOrElse(Map.empty[String, Config]).asJava
102   }
103 
104   // Will also pick things up from the SystemProperties
105   def configURLs: Seq[URL] = {
106     val config = ConfigFactory.load()
107     if (config.hasPath(ConverterConfigURLs)) {
108       config.getAnyRef(ConverterConfigURLs) match {
109         case s: String          => s.split(',').map(_.trim).map(new URL(_))
110         case lst: JList[String] => lst.asScala.map(new URL(_)).toSeq
111       }
112     } else {
113       Seq.empty[URL]
114     }
115   }
116 
117 }
