19 object ConverterConfigLoader extends LazyLogging {

23   val ConverterNameKey = "converter-name"

25   private val configProviders = {
26     val pList = ServiceLoader.load(classOf[ConverterConfigProvider]).asScala.toList
27     logger.debug(s"Found ${pList.size} SPI providers for ${classOf[ConverterConfigProvider].getName}" +
28       s": ${pList.map(_.getClass.getName).mkString(", ")}")

30   }

32   def path: String = ConfigPathProperty.get
33 
34   // this is intentionally a method to allow reloading by the providers
35   def confs: Map[String, Config] =
36     configProviders.map(_.loadConfigs.asScala).reduce( _ ++ _).toMap.map { case (k, v) =>
37       // add the key used to identify the converter (used for metrics)
38       k -> (if (v.hasPath(ConverterNameKey)) { v } else { v.withValue(ConverterNameKey, ConfigValueFactory.fromAnyRef(k)) })
39     }
40 
41   // Public API
42   def listConverterNames: List[String] = confs.keys.toList
43   def getAllConfigs: Map[String, Config] = confs
44   def configForName(name: String): Option[Config] = confs.get(name)

47 trait GeoMesaConvertParser extends LazyLogging {
48   def parseConf(config: Config): Map[String, Config] = {
49     logger.trace(s"Attempting to load Converters from path ${ConverterConfigLoader.path}")
50     if (!config.hasPath(ConverterConfigLoader.path)) {
51       Map.empty[String, Config]
52     } else {
53       val confs = config.getConfig(ConverterConfigLoader.path)
54       confs.root.keySet.asScala.map { k =>
55         logger.trace(s"Found conf block $k")
56         k -> confs.getConfig(k)
57       }.toMap[String, Config]
58     }
59   }
60 }

75 /** Load Config from URLs */
76 class URLConfigProvider extends ConverterConfigProvider with GeoMesaConvertParser {
77   import URLConfigProvider._
78 
79   override def loadConfigs(): java.util.Map[String, Config] = {
80     val confs = configURLs.flatMap { url =>
81       try {
82         Some(ConfigFactory.parseURL(url).resolve())
83       } catch {
84         case e: Throwable =>
85           logger.warn(s"Unable to load converter config from url $url")
86           logger.trace(s"Unable to load converter config from url $url", e)
87           None
88       }
89     }
90     confs.reduceLeftOption(_.withFallback(_)).map(parseConf).getOrElse(Map.empty[String, Config]).asJava
91   }
92 
93   // Will also pick things up from the SystemProperties
94   def configURLs: Seq[URL] = {
95     val config = ConfigFactory.load()
96     if (config.hasPath(ConverterConfigURLs)) {
97       config.getAnyRef(ConverterConfigURLs) match {
98         case s: String          => s.split(',').map(_.trim).map(new URL(_))
99         case lst: JList[String] => lst.asScala.map(new URL(_)).toSeq
100       }
101     } else {
102       Seq.empty[URL]
103     }
104   }
105 
106 }
