40   // Rebase a config to to the converter root...allows standalone

44     import org.locationtech.geomesa.utils.conf.ConfConversions._

51 trait GeoMesaConvertParser extends LazyLogging {
52   def parseConf(config: Config): Map[String, Config] = {
53     logger.trace(s"Attempting to load Converters from path ${ConverterConfigLoader.path}")
54     if (!config.hasPath(ConverterConfigLoader.path)) {
55       Map.empty[String, Config]
56     } else {
57       val confs = config.getConfig(ConverterConfigLoader.path)
58       confs.root.keySet.asScala.map { k =>
59         logger.trace(s"Found conf block $k")
60         k -> confs.getConfig(k)
61       }.toMap[String, Config]
62     }
63   }
64 }

79 /** Load Config from URLs */
80 class URLConfigProvider extends ConverterConfigProvider with GeoMesaConvertParser {
81   import URLConfigProvider._
82 
83   override def loadConfigs(): java.util.Map[String, Config] = {
84     val confs = configURLs.flatMap { url =>
85       try {
86         Some(ConfigFactory.parseURL(url).resolve())
87       } catch {
88         case e: Throwable =>
89           logger.warn(s"Unable to load converter config from url $url")
90           logger.trace(s"Unable to load converter config from url $url", e)
91           None
92       }
93     }
94     confs.reduceLeftOption(_.withFallback(_)).map(parseConf).getOrElse(Map.empty[String, Config]).asJava
95   }
96 
97   // Will also pick things up from the SystemProperties
98   def configURLs: Seq[URL] = {
99     val config = ConfigFactory.load()
100     if (config.hasPath(ConverterConfigURLs)) {
101       config.getAnyRef(ConverterConfigURLs) match {
102         case s: String          => s.split(',').map(_.trim).map(new URL(_))
103         case lst: JList[String] => lst.asScala.map(new URL(_)).toSeq
104       }
105     } else {
106       Seq.empty[URL]
107     }
108   }
109 
110 }
