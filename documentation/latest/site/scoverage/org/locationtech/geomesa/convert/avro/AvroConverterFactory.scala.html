57   override def infer(
58       is: InputStream,
59       sft: Option[SimpleFeatureType],
60       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = {
61     Try {
62       WithClose(new DataFileStream[GenericRecord](is, new GenericDatumReader[GenericRecord]())) { dfs =>
63         val (schema, id, fields, userData) = if (AvroDataFile.canParse(dfs)) {
64           // this is a file written in the geomesa avro format
65           val records = dfs.iterator.asScala.take(AbstractConverterFactory.inferSampleSize)
66           // get the version from the first record
67           val version =
68             records.headOption
69                 .flatMap(r => Option(r.get(VersionField.name)).map(_.asInstanceOf[Int]))
70                 .getOrElse(SerializationVersions.DefaultVersion)
71           val nameEncoder = new FieldNameEncoder(version)
72           val dataSft = AvroDataFile.getSft(dfs)
73           val native = AvroSerialization.usesNativeCollections(dfs.getSchema)
74 
75           val fields = dataSft.getAttributeDescriptors.asScala.map { descriptor =>
76             // some types need a function applied to the underlying avro value
77             val fn = ObjectType.selectType(descriptor).head match {
78               case ObjectType.DATE            => Some("millisToDate")
79               case ObjectType.UUID            => Some("avroBinaryUuid")
80               case ObjectType.GEOMETRY        => Some("geometry") // note: handles both wkt (v1) and wkb (v2)
81               case ObjectType.LIST if !native => Some("avroBinaryList")
82               case ObjectType.MAP if !native  => Some("avroBinaryMap")
83               case _ => None
84             }
85 
86             val path = s"avroPath($$1, '/${nameEncoder.encode(descriptor.getLocalName)}')"
87             val expression = fn.map(f => s"$f($path)").getOrElse(path)
88             BasicField(descriptor.getLocalName, Some(Expression(expression)))
89           }
90 
91           val id = Expression(s"avroPath($$1, '/${FidField.name}')")
92           val userData: Map[String, Expression] =
93             if (dfs.getSchema.getField(UserDataField.name) == null) { Map.empty } else {
94               // avro user data is stored as an array of 'key', 'keyClass', 'value', and 'valueClass'
95               // our converters require global key->expression, so pull out the unique keys
96               val kvs = scala.collection.mutable.Map.empty[String, Expression]
97               records.foreach { record =>
98                 val ud = record.get(UserDataField.name).asInstanceOf[java.util.Collection[GenericRecord]]
99                 ud.asScala.foreach { rec =>
100                   if (rec.getSchema.getField("key") != null) {
101                     Option(rec.get("key")).map(_.toString).foreach { key =>
102                       kvs.getOrElseUpdate(key, {
103                         var expression = s"avroPath($$1, '/${UserDataField.name}[$$key=$key]/value')"
104                         if (rec.getSchema.getField("valueClass") != null &&
105                             Option(rec.get("valueClass")).map(_.toString).contains("java.util.Date")) {
106                           // dates have to be converted from millis
107                           expression = s"millisToDate($expression)"
108                         }
109                         Expression(expression)
110                       })
111                     }
112                   }
113                 }
114               }
115               kvs.toMap
116             }
117 
118           (dataSft, id, fields, userData)
119         } else {
120           // this is an arbitrary avro file, create fields based on the schema
121           val types = AvroConverterFactory.schemaTypes(dfs.getSchema)
122           val dataSft = TypeInference.schema("inferred-avro", types)
123           // note: avro values are always stored at index 1
124           val id = Expression("md5(string2bytes($1::string))")
125           val fields = types.map(t => BasicField(t.name, Some(Expression(t.transform.apply(1)))))
126 
127           (dataSft, id, fields, Map.empty[String, Expression])
128         }
129 
130         // validate the existing schema, if any
131         sft.foreach { existing =>
132           val inferred = schema.getAttributeDescriptors.asScala
133           val matched = existing.getAttributeDescriptors.asScala.count { d =>
134             inferred.exists { i =>
135               i.getLocalName == d.getLocalName && d.getType.getBinding.isAssignableFrom(i.getType.getBinding)
136             }
137           }
138           if (matched == 0) {
139             throw new IllegalArgumentException("Inferred schema does not match existing schema")
140           } else if (matched < existing.getAttributeCount) {
141             logger.warn(s"Inferred schema only matched $matched out of ${existing.getAttributeCount} attributes")
142           }
143         }
144 
145         val converterConfig = AvroConfig(typeToProcess, SchemaEmbedded, Some(id), Map.empty, userData)
146 
147         val config = configConvert.to(converterConfig)
148             .withFallback(fieldConvert.to(fields.toSeq))
149             .withFallback(optsConvert.to(BasicOptions.default))
150             .toConfig
151 
152         (schema, config)
153       }
154     }
155   }

162   /**
163     * Take an avro schema and return the simple feature type bindings for it
164     *
165     * @param schema avro schema
166     * @return
167     */
168   def schemaTypes(schema: Schema): Seq[InferredType] = {
169     val namer = new Namer()
170     val types = scala.collection.mutable.ArrayBuffer.empty[InferredType]
171 
172     def mapField(field: Schema.Field, path: String = ""): Unit = {
173       // get a valid attribute name
174       val name = namer(field.name())
175 
176       // checks for nested array/map types we can handle
177       def isSimpleArray: Boolean = isSimple(field.schema().getElementType)
178       def isSimpleMap: Boolean = isSimple(field.schema().getValueType)
179 
180       val transform = FunctionTransform("avroPath(", s",'$path/${field.name}')")
181       field.schema().getType match {
182         case Schema.Type.STRING  => types += InferredType(name, ObjectType.STRING, transform)
183         case Schema.Type.BYTES   => types += InferredType(name, ObjectType.BYTES, transform)
184         case Schema.Type.INT     => types += InferredType(name, ObjectType.INT, transform)
185         case Schema.Type.LONG    => types += InferredType(name, ObjectType.LONG, transform)
186         case Schema.Type.FLOAT   => types += InferredType(name, ObjectType.FLOAT, transform)
187         case Schema.Type.DOUBLE  => types += InferredType(name, ObjectType.DOUBLE, transform)
188         case Schema.Type.BOOLEAN => types += InferredType(name, ObjectType.BOOLEAN, transform)
189         case Schema.Type.ARRAY   => if (isSimpleArray) { types += InferredType(name, ObjectType.LIST, transform) }
190         case Schema.Type.MAP     => if (isSimpleMap) { types += InferredType(name, ObjectType.MAP, transform) }
191         case Schema.Type.FIXED   => types += InferredType(name, ObjectType.BYTES, transform)
192         case Schema.Type.ENUM    => types += InferredType(name, ObjectType.STRING, transform.copy(suffix = transform.suffix + "::string"))
193         case Schema.Type.UNION   => types += InferredType(name, ObjectType.STRING, transform.copy(suffix = transform.suffix + "::string"))
194         case Schema.Type.RECORD  => field.schema().getFields.asScala.foreach(mapField(_, s"$path/${field.name}"))
195         case _ => // no-op
196       }
197     }
198 
199     schema.getFields.asScala.foreach(mapField(_))
200 
201     // check if we can derive a geometry field
202     TypeInference.deriveGeometry(types.toSeq, namer).foreach(g => types += g)
203 
204     types.toSeq
205   }

207   private def isSimple(s: Schema): Boolean = s.getType match {
208     case Schema.Type.STRING  => true
209     case Schema.Type.INT     => true
210     case Schema.Type.LONG    => true
211     case Schema.Type.FLOAT   => true
212     case Schema.Type.DOUBLE  => true
213     case Schema.Type.BOOLEAN => true
214     case _ => false
215   }

229         (schema, schemaFile) match {
230           case (Some(s), None) if s.equalsIgnoreCase(SchemaEmbedded.name) => Right(SchemaEmbedded)

232           case (None, Some(s)) => Right(SchemaFile(s))

238         }
239       }
240 
241       for {
242         schema     <- optional(cur, "schema").right
243         schemaFile <- optional(cur, "schema-file").right
244         either     <- schemaOrFile(schema, schemaFile).right
245       } yield {
246         AvroConfig(`type`, either, idField, caches, userData)
247       }
248     }
