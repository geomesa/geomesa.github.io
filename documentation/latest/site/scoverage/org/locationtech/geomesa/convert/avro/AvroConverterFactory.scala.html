40   /**
41     * Note: only works on Avro files with embedded schemas
42     *
43     * @param is input
44     * @param sft simple feature type, if known ahead of time
45     * @param hints hints
46     * @return
47     */
48   override def infer(
49       is: InputStream,
50       sft: Option[SimpleFeatureType],
51       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = {
52     Try {
53       WithClose(new DataFileStream[GenericRecord](is, new GenericDatumReader[GenericRecord]())) { dfs =>
54         val (schema, id, fields, userData) = if (AvroDataFile.canParse(dfs)) {
55           // this is a file written in the geomesa avro format
56           val records = dfs.iterator.asScala.take(AbstractConverterFactory.inferSampleSize)
57           // get the version from the first record
58           val version = records.flatMap(r => Option(r.get(VersionField.name))).collectFirst { case i: Integer => i.intValue() }
59           val nameEncoder = new FieldNameEncoder(version.getOrElse(SerializationVersions.DefaultVersion))
60           val dataSft = AvroDataFile.getSft(dfs)
61           val native = AvroSerialization.usesNativeCollections(dfs.getSchema)
62 
63           val fields = dataSft.getAttributeDescriptors.asScala.map { descriptor =>
64             // some types need a function applied to the underlying avro value
65             val fn = ObjectType.selectType(descriptor).head match {
66               case ObjectType.DATE            => Some("millisToDate")
67               case ObjectType.UUID            => Some("avroBinaryUuid")
68               case ObjectType.GEOMETRY        => Some("geometry") // note: handles both wkt (v1) and wkb (v2)
69               case ObjectType.LIST if !native => Some("avroBinaryList")
70               case ObjectType.MAP if !native  => Some("avroBinaryMap")
71               case _ => None
72             }
73 
74             val path = s"avroPath($$1, '/${nameEncoder.encode(descriptor.getLocalName)}')"
75             val expression = fn.map(f => s"$f($path)").getOrElse(path)
76             BasicField(descriptor.getLocalName, Some(Expression(expression)))
77           }
78 
79           val id = Expression(s"avroPath($$1, '/${FidField.name}')")
80           val userData: Map[String, Expression] =
81             if (dfs.getSchema.getField(UserDataField.name) == null) { Map.empty } else {
82               // avro user data is stored as an array of 'key', 'keyClass', 'value', and 'valueClass'
83               // our converters require global key->expression, so pull out the unique keys
84               val kvs = scala.collection.mutable.Map.empty[String, Expression]
85               records.foreach { record =>
86                 val ud = record.get(UserDataField.name).asInstanceOf[java.util.Collection[GenericRecord]]
87                 ud.asScala.foreach { rec =>
88                   if (rec.getSchema.getField("key") != null) {
89                     Option(rec.get("key")).map(_.toString).foreach { key =>
90                       kvs.getOrElseUpdate(key, {
91                         var expression = s"avroPath($$1, '/${UserDataField.name}[$$key=$key]/value')"
92                         if (rec.getSchema.getField("valueClass") != null &&
93                             Option(rec.get("valueClass")).map(_.toString).contains("java.util.Date")) {
94                           // dates have to be converted from millis
95                           expression = s"millisToDate($expression)"
96                         }
97                         Expression(expression)
98                       })
99                     }
100                   }
101                 }
102               }
103               kvs.toMap
104             }
105 
106           (dataSft, id, fields, userData)
107         } else {
108           // this is an arbitrary avro file, create fields based on the schema
109           val types = AvroConverterFactory.schemaTypes(dfs.getSchema)
110           val dataSft = TypeInference.schema("inferred-avro", types)
111           // note: avro values are always stored at index 1
112           val id = Expression("md5(string2bytes($1::string))")
113           val fields = types.map(t => BasicField(t.name, Some(Expression(t.transform.apply(1)))))
114 
115           (dataSft, id, fields, Map.empty[String, Expression])
116         }
117 
118         // validate the existing schema, if any
119         sft.foreach { existing =>
120           val inferred = schema.getAttributeDescriptors.asScala
121           val matched = existing.getAttributeDescriptors.asScala.count { d =>
122             inferred.exists { i =>
123               i.getLocalName == d.getLocalName && d.getType.getBinding.isAssignableFrom(i.getType.getBinding)
124             }
125           }
126           if (matched == 0) {
127             throw new IllegalArgumentException("Inferred schema does not match existing schema")
128           } else if (matched < existing.getAttributeCount) {
129             logger.warn(s"Inferred schema only matched $matched out of ${existing.getAttributeCount} attributes")
130           }
131         }
132 
133         val converterConfig = AvroConfig(typeToProcess, None, SchemaEmbedded, Some(id), Map.empty, userData)
134 
135         val config = configConvert.to(converterConfig)
136             .withFallback(fieldConvert.to(fields.toSeq))
137             .withFallback(optsConvert.to(BasicOptions.default))
138             .toConfig
139 
140         (schema, config)
141       }
142     }
143   }

151     * Take an avro schema and return the simple feature type bindings for it
152     *
153     * @param schema avro schema
154     * @return
155     */
156   def schemaTypes(schema: Schema): Seq[InferredType] = {
157     val namer = new Namer()
158     val types = scala.collection.mutable.ArrayBuffer.empty[InferredType]
159 
160     def mapField(field: Schema.Field, path: String = ""): Unit = {
161       // get a valid attribute name
162       val name = namer(field.name())
163 
164       // checks for nested array/map types we can handle
165       def isSimpleArray: Boolean = isSimple(field.schema().getElementType)
166       def isSimpleMap: Boolean = isSimple(field.schema().getValueType)
167 
168       val transform = FunctionTransform("avroPath(", s",'$path/${field.name}')")
169       field.schema().getType match {
170         case Schema.Type.STRING  => types += InferredType(name, ObjectType.STRING, transform)
171         case Schema.Type.BYTES   => types += InferredType(name, ObjectType.BYTES, transform)
172         case Schema.Type.INT     => types += InferredType(name, ObjectType.INT, transform)
173         case Schema.Type.LONG    => types += InferredType(name, ObjectType.LONG, transform)
174         case Schema.Type.FLOAT   => types += InferredType(name, ObjectType.FLOAT, transform)
175         case Schema.Type.DOUBLE  => types += InferredType(name, ObjectType.DOUBLE, transform)
176         case Schema.Type.BOOLEAN => types += InferredType(name, ObjectType.BOOLEAN, transform)
177         case Schema.Type.ARRAY   => if (isSimpleArray) { types += InferredType(name, ObjectType.LIST, transform) }
178         case Schema.Type.MAP     => if (isSimpleMap) { types += InferredType(name, ObjectType.MAP, transform) }
179         case Schema.Type.FIXED   => types += InferredType(name, ObjectType.BYTES, transform)
180         case Schema.Type.ENUM    => types += InferredType(name, ObjectType.STRING, transform.copy(suffix = transform.suffix + "::string"))
181         case Schema.Type.UNION   => types += InferredType(name, ObjectType.STRING, transform.copy(suffix = transform.suffix + "::string"))
182         case Schema.Type.RECORD  => field.schema().getFields.asScala.foreach(mapField(_, s"$path/${field.name}"))
183         case _ => // no-op
184       }
185     }
186 
187     schema.getFields.asScala.foreach(mapField(_))
188 
189     // check if we can derive a geometry field
190     TypeInference.deriveGeometry(types.toSeq, namer).foreach(g => types += g)
191 
192     types.toSeq
193   }

199     case Schema.Type.FLOAT   => true
200     case Schema.Type.DOUBLE  => true
201     case Schema.Type.BOOLEAN => true
202     case _ => false
203   }

215           schema: Option[String],
216           schemaFile: Option[String]): Either[ConfigReaderFailures, SchemaConfig] = {
217         (schema, schemaFile) match {
218           case (Some(s), None) if s.equalsIgnoreCase(SchemaEmbedded.name) => Right(SchemaEmbedded)

227       }

230         name       <- converterName(cur).right
231         schema     <- optional(cur, "schema").right
232         schemaFile <- optional(cur, "schema-file").right
233         either     <- schemaOrFile(schema, schemaFile).right
234       } yield {
235         AvroConfig(`type`, name, either, idField, caches, userData)
236       }
237     }

244       }
245     }
