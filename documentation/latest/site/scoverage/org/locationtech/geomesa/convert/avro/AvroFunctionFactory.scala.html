81   private def toJson(arg: Any): JsonElement = {
82     arg match {
83       case s: String => new JsonPrimitive(s)
84       case n: Number => new JsonPrimitive(n)
85       case b: Boolean => new JsonPrimitive(b)
86       case r: GenericRecord =>
87         val obj = new JsonObject()
88         var i = 0
89         r.getSchema.getFields.asScala.foreach { field =>
90           val value = toJson(r.get(i))
91           if (!value.isJsonNull) {
92             obj.add(field.name(), value)
93           }
94           i += 1
95         }
96         obj
97       case list: java.util.List[_] =>
98         val array = new JsonArray(list.size())
99         list.asScala.foreach { elem =>
100           val json = toJson(elem)
101           if (!json.isJsonNull) {
102             array.add(json)
103           }
104         }
105         array
106       case m: java.util.Map[_, _] =>
107         val obj = new JsonObject()
108         m.asScala.foreach { case (k, v) =>
109           if (k != null) {
110             val value = toJson(v)
111             if (!value.isJsonNull) {
112               obj.add(k.toString, value)
113             }
114           }
115         }
116         obj
117       case null => JsonNull.INSTANCE
118       case _ => new JsonPrimitive(arg.toString)
119     }
120   }
