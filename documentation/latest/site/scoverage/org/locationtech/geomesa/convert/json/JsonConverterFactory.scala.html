57   /**
58    * Infer a configuration and simple feature type from an input stream, if possible
59    *
60    * Available hints:
61    *  - `featurePath` - json path expression pointing to the feature element
62    *
63    * @param is input
64    * @param sft simple feature type, if known ahead of time
65    * @param hints implementation specific hints about the input
66    * @return
67    */
68   override def infer(
69       is: InputStream,
70       sft: Option[SimpleFeatureType],
71       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = {
72 
73     val tryElements = Try {
74       WithClose(new JsonReader(new InputStreamReader(is, StandardCharsets.UTF_8))) { reader =>
75         reader.setLenient(true)
76         val iter: Iterator[JsonElement] = new Iterator[JsonElement] {
77           override def hasNext: Boolean = reader.peek() != JsonToken.END_DOCUMENT
78           override def next(): JsonElement = JsonParser.parseReader(reader)
79         }
80         iter.take(AbstractConverterFactory.inferSampleSize).toList
81       }
82     }
83     tryElements.flatMap { elements =>
84       lazy val featurePath = hints.get(JsonConverterFactory.FeaturePathKey).map(_.toString)
85       if (elements.isEmpty) {
86         Failure(new RuntimeException("Could not parse the input as JSON"))
87       } else if (JsonConverter.isFeature(elements.head)) {
88         Try(elements.map(JsonConverter.parseFeature))
89             .flatMap(inferGeoJson(_, None, sft))
90       } else if (JsonConverter.isFeatureCollection(elements.head)) {
91         Try(elements.flatMap(JsonConverter.parseFeatureCollection))
92             .flatMap(inferGeoJson(_, Some("$.features[*]"), sft))
93       } else if (elements.head.isJsonObject) {
94         inferJson(elements, featurePath, sft)
95       } else if (elements.head.isJsonArray) {
96         inferJson(elements, featurePath.orElse(Some("$.[*]")), sft)
97       } else {
98         Failure(new RuntimeException("Could not parse the input as a JSON object or array"))
99       }
100     }
101   }

103   private def inferGeoJson(
104       features: Seq[GeoJsonFeature],
105       featurePath: Option[String],
106       sft: Option[SimpleFeatureType]): Try[(SimpleFeatureType, Config)] = Try {
107     // track the 'properties', geometry type and 'id' in each feature
108     // use linkedHashMap to retain insertion order
109     val props = scala.collection.mutable.LinkedHashMap.empty[String, ListBuffer[Any]]
110     val geoms = ListBuffer.empty[Any]
111     var hasId = true
112 
113     features.take(AbstractConverterFactory.inferSampleSize).foreach { feature =>
114       feature.properties.foreach { case (k, v) => props.getOrElseUpdate(k, ListBuffer.empty) += v }
115       geoms += feature.geom
116       hasId = hasId && feature.id.isDefined
117     }
118 
119     val pathsAndValues =
120       props.toSeq.map { case (path, values) => PathWithValues(path, values) } :+
121           PathWithValues(JsonConverterFactory.GeoJsonGeometryPath, geoms)
122     val inferredTypes = TypeInference.infer(pathsAndValues, sft.toRight("inferred-json"), new PropNamer())
123 
124     val idJsonField = if (hasId) { Some(StringJsonField("id", "$.id", pathIsRoot = false, None)) } else { None }
125     val idField = idJsonField match {
126       case None    => Some(Expression("md5(stringToBytes(toString($0)))"))
127       case Some(f) => Some(Expression(s"$$${f.name}"))
128     }
129     val fieldConfig = idJsonField.toSeq ++ inferredTypes.types.map(createFieldConfig)
130 
131     val jsonConfig = JsonConfig(typeToProcess, None, featurePath, idField, Map.empty, Map.empty)
132 
133     val config = configConvert.to(jsonConfig)
134         .withFallback(fieldConvert.to(fieldConfig))
135         .withFallback(optsConvert.to(BasicOptions.default))
136         .toConfig
137 
138     (inferredTypes.sft, config)
139   }

141   private def inferJson(
142       elements: Seq[JsonElement],
143       featurePath: Option[String],
144       sft: Option[SimpleFeatureType]): Try[(SimpleFeatureType, Config)] = {
145     val tryFeatures = Try {
146       val features = featurePath match {
147         case None => elements
148         case Some(p) =>
149           val path = JsonPath.compile(p)
150           elements.flatMap { el =>
151             val res = try { path.read[JsonElement](el, JsonConverter.JsonConfiguration) } catch {
152               case _: PathNotFoundException => JsonNull.INSTANCE
153             }
154             if (res.isJsonArray) {
155               res.getAsJsonArray.asList().asScala
156             } else {
157               Seq(res)
158             }
159           }
160       }
161       features.collect { case r if r.isJsonObject => r.getAsJsonObject }
162     }
163 
164     tryFeatures.flatMap { features =>
165       if (features.isEmpty) {
166         Failure(new RuntimeException("Could not parse input as JSON"))
167       } else {
168         Try {
169           // track the properties in each feature
170           // use linkedHashMap to retain insertion order
171           val props = scala.collection.mutable.LinkedHashMap.empty[String, ListBuffer[Any]]
172 
173           features.take(AbstractConverterFactory.inferSampleSize).foreach { feature =>
174             GeoJsonParsing.parseElement(feature, "").foreach { case (k, v) =>
175               props.getOrElseUpdate(k, ListBuffer.empty) += v
176             }
177           }
178 
179           val pathsAndValues = props.toSeq.map { case (path, values) => PathWithValues(path, values) }
180           val inferredTypes = TypeInference.infer(pathsAndValues, sft.toRight("inferred-json"))
181 
182           val idField = Some(Expression("md5(stringToBytes(toString($0)))"))
183           val fieldConfig = inferredTypes.types.map(createFieldConfig)
184 
185           val jsonConfig = JsonConfig(typeToProcess, None, featurePath, idField, Map.empty, Map.empty)
186 
187           val config =
188             configConvert.to(jsonConfig)
189                 .withFallback(fieldConvert.to(fieldConfig))
190                 .withFallback(optsConvert.to(BasicOptions.default))
191                 .toConfig
192 
193           (inferredTypes.sft, config)
194         }
195       }
196     }
197   }

199   private def createFieldConfig(typed: TypeWithPath): JsonField = {
200     val TypeWithPath(path, inferredType) = typed
201     val transform = inferredType.transform match {
202       case IdentityTransform => None
203       // account for optional nodes by wrapping transform with a try/null
204       case t => Some(Expression(s"try(${t.apply(0)},null)"))
205     }
206     if (path.isEmpty) {
207       DerivedField(inferredType.name, transform)
208     } else {
209       inferredType.typed match {
210         case ObjectType.BOOLEAN =>
211           BooleanJsonField(inferredType.name, path, pathIsRoot = false, transform)
212         case ObjectType.LIST =>
213           // if type is list, that means the transform is 'identity', but we need to replace it with jsonList.
214           // this is due to GeoJsonParsing decoding the json array for us, above
215           ArrayJsonField(inferredType.name, path, pathIsRoot = false, Some(Expression("try(jsonList('string',$0),null)")))
216         case t if transform.isEmpty && ObjectType.GeometrySubtypes.contains(t) =>
217           GeometryJsonField(inferredType.name, path, pathIsRoot = false, None)
218         case _ =>
219           // all other types will be parsed as strings with appropriate transforms
220           StringJsonField(inferredType.name, path, pathIsRoot = false, transform)
221       }
222     }
223   }

255   object JsonFieldConvert extends FieldConvert[JsonField] with OptionConvert {
256     override protected def decodeField(cur: ConfigObjectCursor,
257                                        name: String,
258                                        transform: Option[Expression]): Either[ConfigReaderFailures, JsonField] = {
259       val jsonTypeCur =  cur.atKeyOrUndefined("json-type")
260       val jsonType = if (jsonTypeCur.isUndefined) { Right(None) } else { jsonTypeCur.asString.right.map(Option.apply) }
261 
262       val config = for {
263         jType    <- jsonType.right
264         path     <- optional(cur, "path").right
265         rootPath <- optional(cur, "root-path").right
266       } yield {
267         (jType, path, rootPath)
268       }
269       config.right.flatMap { case (jType, path, rootPath) =>
270         if (path.isDefined && rootPath.isDefined) {
271           cur.failed(CannotConvert(cur.toString, "JsonField", "Json fields must define only one of 'path' or 'root-path'"))
272         } else if (jType.isDefined && path.isEmpty && rootPath.isEmpty) {
273           cur.failed(CannotConvert(cur.toString, "JsonField", "Json fields must define a 'path' or 'root-path'"))
274         } else if (jType.isEmpty) {
275           Right(DerivedField(name, transform))
276         } else {
277           val (jsonPath, pathIsRoot) = (path, rootPath) match {
278             case (Some(p), None) => (p, false)
279             case (None, Some(p)) => (p, true)
280           }
281           jType.get.toLowerCase(Locale.US) match {
282             case "string"           => Right(StringJsonField(name, jsonPath, pathIsRoot, transform))
283             case "float"            => Right(FloatJsonField(name, jsonPath, pathIsRoot, transform))
284             case "double"           => Right(DoubleJsonField(name, jsonPath, pathIsRoot, transform))
285             case "integer" | "int"  => Right(IntJsonField(name, jsonPath, pathIsRoot, transform))
286             case "boolean" | "bool" => Right(BooleanJsonField(name, jsonPath, pathIsRoot, transform))
287             case "long"             => Right(LongJsonField(name, jsonPath, pathIsRoot, transform))
288             case "geometry"         => Right(GeometryJsonField(name, jsonPath, pathIsRoot, transform))
289             case "array" | "list"   => Right(ArrayJsonField(name, jsonPath, pathIsRoot, transform))
290             case "object" | "map"   => Right(ObjectJsonField(name, jsonPath, pathIsRoot, transform))
291             case t => cur.failed(CannotConvert(cur.toString, "JsonField", s"Invalid json-type '$t'"))
292           }
293         }
294       }
295     }
296 
297     override protected def encodeField(field: JsonField, base: java.util.Map[String, AnyRef]): Unit = {
298       field match {
299         case f: TypedJsonField =>
300           base.put("json-type", f.jsonType)
301           base.put(if (f.pathIsRoot) { "root-path" } else { "path" }, f.path)
302 
303         case _ => // no-op
304       }
305     }
