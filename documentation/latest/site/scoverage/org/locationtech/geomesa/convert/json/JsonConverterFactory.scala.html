64   /**
65    * Infer a configuration and simple feature type from an input stream, if possible
66    *
67    * Available hints:
68    *  - `featurePath` - json path expression pointing to the feature element
69    *
70    * @param is input
71    * @param sft simple feature type, if known ahead of time
72    * @param hints implementation specific hints about the input
73    * @return
74    */
75   override def infer(
76       is: InputStream,
77       sft: Option[SimpleFeatureType],
78       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = {
79 
80     val tryElements = Try {
81       WithClose(new JsonReader(new InputStreamReader(is, StandardCharsets.UTF_8))) { reader =>
82         reader.setLenient(true)
83         val iter: Iterator[JsonElement] = new Iterator[JsonElement] {
84           override def hasNext: Boolean = reader.peek() != JsonToken.END_DOCUMENT
85           override def next(): JsonElement = JsonParser.parseReader(reader)
86         }
87         iter.take(AbstractConverterFactory.inferSampleSize).toList
88       }
89     }
90     tryElements.flatMap { elements =>
91       lazy val featurePath = hints.get(JsonConverterFactory.FeaturePathKey).map(_.toString)
92       if (elements.isEmpty) {
93         Failure(new RuntimeException("Could not parse the input as JSON"))
94       } else if (JsonConverter.isFeature(elements.head)) {
95         Try(elements.map(JsonConverter.parseFeature))
96             .flatMap(inferGeoJson(_, None, sft))
97       } else if (JsonConverter.isFeatureCollection(elements.head)) {
98         Try(elements.flatMap(JsonConverter.parseFeatureCollection))
99             .flatMap(inferGeoJson(_, Some("$.features[*]"), sft))
100       } else if (elements.head.isJsonObject) {
101         inferJson(elements, featurePath, sft)
102       } else if (elements.head.isJsonArray) {
103         inferJson(elements, featurePath.orElse(Some("$.[*]")), sft)
104       } else {
105         Failure(new RuntimeException("Could not parse the input as a JSON object or array"))
106       }
107     }
108   }

110   private def inferGeoJson(
111       features: Seq[GeoJsonFeature],
112       featurePath: Option[String],
113       sft: Option[SimpleFeatureType]): Try[(SimpleFeatureType, Config)] = Try {
114     // track the 'properties', geometry type and 'id' in each feature
115     // use linkedHashMap to retain insertion order
116     val props = scala.collection.mutable.LinkedHashMap.empty[String, ListBuffer[Any]]
117     val geoms = ListBuffer.empty[Any]
118     var hasId = true
119 
120     features.take(AbstractConverterFactory.inferSampleSize).foreach { feature =>
121       feature.properties.foreach { case (k, v) => props.getOrElseUpdate(k, ListBuffer.empty) += v }
122       geoms += feature.geom
123       hasId = hasId && feature.id.isDefined
124     }
125 
126     val pathsAndValues =
127       props.toSeq.map { case (path, values) => PathWithValues(path, values) } :+
128           PathWithValues(JsonConverterFactory.GeoJsonGeometryPath, geoms)
129     val inferredTypes = TypeInference.infer(pathsAndValues, sft.toRight("inferred-json"), new PropNamer())
130 
131     val idJsonField = if (hasId) { Some(StringJsonField("id", "$.id", pathIsRoot = false, None)) } else { None }
132     val idField = idJsonField match {
133       case None    => Some(Expression("md5(stringToBytes(toString($0)))"))
134       case Some(f) => Some(Expression(s"$$${f.name}"))
135     }
136     val fieldConfig = idJsonField.toSeq ++ inferredTypes.types.map(createFieldConfig)
137 
138     val jsonConfig = JsonConfig(typeToProcess, featurePath, idField, Map.empty, Map.empty)
139 
140     val config = configConvert.to(jsonConfig)
141         .withFallback(fieldConvert.to(fieldConfig))
142         .withFallback(optsConvert.to(BasicOptions.default))
143         .toConfig
144 
145     (inferredTypes.sft, config)
146   }

148   private def inferJson(
149       elements: Seq[JsonElement],
150       featurePath: Option[String],
151       sft: Option[SimpleFeatureType]): Try[(SimpleFeatureType, Config)] = {
152     val tryFeatures = Try {
153       val features = featurePath match {
154         case None => elements
155         case Some(p) =>
156           val path = JsonPath.compile(p)
157           elements.flatMap { el =>
158             val res = try { path.read[JsonElement](el, JsonConverter.JsonConfiguration) } catch {
159               case _: PathNotFoundException => JsonNull.INSTANCE
160             }
161             if (res.isJsonArray) {
162               res.getAsJsonArray.asList().asScala
163             } else {
164               Seq(res)
165             }
166           }
167       }
168       features.collect { case r if r.isJsonObject => r.getAsJsonObject }
169     }
170 
171     tryFeatures.flatMap { features =>
172       if (features.isEmpty) {
173         Failure(new RuntimeException("Could not parse input as JSON"))
174       } else {
175         Try {
176           // track the properties in each feature
177           // use linkedHashMap to retain insertion order
178           val props = scala.collection.mutable.LinkedHashMap.empty[String, ListBuffer[Any]]
179 
180           features.take(AbstractConverterFactory.inferSampleSize).foreach { feature =>
181             GeoJsonParsing.parseElement(feature, "").foreach { case (k, v) =>
182               props.getOrElseUpdate(k, ListBuffer.empty) += v
183             }
184           }
185 
186           val pathsAndValues = props.toSeq.map { case (path, values) => PathWithValues(path, values) }
187           val inferredTypes = TypeInference.infer(pathsAndValues, sft.toRight("inferred-json"))
188 
189           val idField = Some(Expression("md5(stringToBytes(toString($0)))"))
190           val fieldConfig = inferredTypes.types.map(createFieldConfig)
191 
192           val jsonConfig = JsonConfig(typeToProcess, featurePath, idField, Map.empty, Map.empty)
193 
194           val config =
195             configConvert.to(jsonConfig)
196                 .withFallback(fieldConvert.to(fieldConfig))
197                 .withFallback(optsConvert.to(BasicOptions.default))
198                 .toConfig
199 
200           (inferredTypes.sft, config)
201         }
202       }
203     }
204   }

206   private def createFieldConfig(typed: TypeWithPath): JsonField = {
207     val TypeWithPath(path, inferredType) = typed
208     val transform = inferredType.transform match {
209       case IdentityTransform => None
210       // account for optional nodes by wrapping transform with a try/null
211       case t => Some(Expression(s"try(${t.apply(0)},null)"))
212     }
213     if (path.isEmpty) {
214       DerivedField(inferredType.name, transform)
215     } else {
216       inferredType.typed match {
217         case ObjectType.BOOLEAN =>
218           BooleanJsonField(inferredType.name, path, pathIsRoot = false, transform)
219         case ObjectType.LIST =>
220           // if type is list, that means the transform is 'identity', but we need to replace it with jsonList.
221           // this is due to GeoJsonParsing decoding the json array for us, above
222           ArrayJsonField(inferredType.name, path, pathIsRoot = false, Some(Expression("try(jsonList('string',$0),null)")))
223         case t if transform.isEmpty && (t == ObjectType.GEOMETRY || ObjectType.GeometrySubtypes.contains(t)) =>
224           GeometryJsonField(inferredType.name, path, pathIsRoot = false, None)
225         case _ =>
226           // all other types will be parsed as strings with appropriate transforms
227           StringJsonField(inferredType.name, path, pathIsRoot = false, transform)
228       }
229     }
230   }

259   object JsonFieldConvert extends FieldConvert[JsonField] with OptionConvert {
260     override protected def decodeField(cur: ConfigObjectCursor,
261                                        name: String,
262                                        transform: Option[Expression]): Either[ConfigReaderFailures, JsonField] = {
263       val jsonTypeCur =  cur.atKeyOrUndefined("json-type")
264       val jsonType = if (jsonTypeCur.isUndefined) { Right(None) } else { jsonTypeCur.asString.right.map(Option.apply) }
265 
266       val config = for {
267         jType    <- jsonType.right
268         path     <- optional(cur, "path").right
269         rootPath <- optional(cur, "root-path").right
270       } yield {
271         (jType, path, rootPath)
272       }
273       config.right.flatMap { case (jType, path, rootPath) =>
274         if (path.isDefined && rootPath.isDefined) {
275           cur.failed(CannotConvert(cur.toString, "JsonField", "Json fields must define only one of 'path' or 'root-path'"))
276         } else if (jType.isDefined && path.isEmpty && rootPath.isEmpty) {
277           cur.failed(CannotConvert(cur.toString, "JsonField", "Json fields must define a 'path' or 'root-path'"))
278         } else if (jType.isEmpty) {
279           Right(DerivedField(name, transform))
280         } else {
281           val (jsonPath, pathIsRoot) = (path, rootPath) match {
282             case (Some(p), None) => (p, false)
283             case (None, Some(p)) => (p, true)
284           }
285           jType.get.toLowerCase(Locale.US) match {
286             case "string"           => Right(StringJsonField(name, jsonPath, pathIsRoot, transform))
287             case "float"            => Right(FloatJsonField(name, jsonPath, pathIsRoot, transform))
288             case "double"           => Right(DoubleJsonField(name, jsonPath, pathIsRoot, transform))
289             case "integer" | "int"  => Right(IntJsonField(name, jsonPath, pathIsRoot, transform))
290             case "boolean" | "bool" => Right(BooleanJsonField(name, jsonPath, pathIsRoot, transform))
291             case "long"             => Right(LongJsonField(name, jsonPath, pathIsRoot, transform))
292             case "geometry"         => Right(GeometryJsonField(name, jsonPath, pathIsRoot, transform))
293             case "array" | "list"   => Right(ArrayJsonField(name, jsonPath, pathIsRoot, transform))
294             case "object" | "map"   => Right(ObjectJsonField(name, jsonPath, pathIsRoot, transform))
295             case t => cur.failed(CannotConvert(cur.toString, "JsonField", s"Invalid json-type '$t'"))
296           }
297         }
298       }
299     }
300 
301     override protected def encodeField(field: JsonField, base: java.util.Map[String, AnyRef]): Unit = {
302       field match {
303         case f: TypedJsonField =>
304           base.put("json-type", f.jsonType)
305           base.put(if (f.pathIsRoot) { "root-path" } else { "path" }, f.path)
306 
307         case _ => // no-op
308       }
309     }
