30 class ShapefileConverter(sft: SimpleFeatureType, config: BasicConfig, fields: Seq[BasicField], options: BasicOptions)
31     extends AbstractConverter[SimpleFeature, BasicConfig, BasicField, BasicOptions](sft, config, fields, options)  {

33   import org.locationtech.geomesa.convert.shp.ShapefileFunctionFactory.{InputSchemaKey, InputValuesKey}
34 
35   override def createEvaluationContext(
36       globalParams: Map[String, Any],
37       success: Counter,
38       failure: Counter): EvaluationContext = {
39     // inject placeholders for shapefile attributes into the evaluation context
40     // used for accessing shapefile properties by name in ShapefileFunctionFactory
41     val shpParams = Map(InputSchemaKey -> ArrayBuffer.empty[String], InputValuesKey -> ArrayBuffer.empty[AnyRef])
42     super.createEvaluationContext(globalParams ++ shpParams, success, failure)
43   }
44 
45   override protected def parse(is: InputStream, ec: EvaluationContext): CloseableIterator[SimpleFeature] = {
46     CloseWithLogging(is) // we don't use the input stream, just close it
47 
48     val path = ec.getInputFilePath.getOrElse {
49       throw new IllegalArgumentException(s"Shapefile converter requires '${EvaluationContext.InputFilePathKey}' " +
50           "to be set in the evaluation context")
51     }
52     val ds = ShapefileConverter.getDataStore(path, options.encoding)
53     val schema = ds.getSchema()
54 
55     (ec.accessor(InputSchemaKey).apply(), ec.accessor(InputValuesKey).apply()) match {
56       case (n: ArrayBuffer[String], v: ArrayBuffer[AnyRef]) =>
57         n.clear()
58         n ++= Array.tabulate(schema.getAttributeCount)(i => schema.getDescriptor(i).getLocalName)
59         v.clear()
60         v ++= Array.fill[AnyRef](n.length + 1)(null)
61       case _ =>
62         logger.warn("Input schema not found in evaluation context, shapefile functions " +
63             s"${TextTools.wordList(new ShapefileFunctionFactory().functions.map(_.names.head))} will not be available")
64     }
65 
66     val q = new Query
67     // Only ask to reproject if the Shapefile has a non-4326 CRS
68     if (ds.getSchema.getCoordinateReferenceSystem == null) {
69       logger.warn(s"Shapefile does not have CRS info")
70     } else if (!CRS.equalsIgnoreMetadata(ds.getSchema.getCoordinateReferenceSystem, CRS_EPSG_4326)) {
71       q.setCoordinateSystemReproject(CRS_EPSG_4326)
72     }
73 
74     val reader = CloseableIterator(ds.getFeatureSource.getReader(q)).map { f => ec.line += 1; f }
75 
76     CloseableIterator(reader, { CloseWithLogging(reader); ds.dispose() })
77   }
78 
79   override protected def values(parsed: CloseableIterator[SimpleFeature],
80                                 ec: EvaluationContext): CloseableIterator[Array[Any]] = {
81     (ec.accessor(InputSchemaKey).apply(), ec.accessor(InputValuesKey).apply()) match {
82       case (_: ArrayBuffer[String], v: ArrayBuffer[AnyRef]) =>
83         parsed.map { feature =>
84           var i = 1
85           while (i < v.length) {
86             v(i) = feature.getAttribute(i - 1)
87             i += 1
88           }
89           v(0) = feature.getID
90           v.toArray
91         }
92 
93       case _ =>
94         logger.warn("Input schema not found in evaluation context, shapefile functions " +
95             s"${TextTools.wordList(new ShapefileFunctionFactory().functions.map(_.names.head))} will not be available")
96         var array: Array[Any] = null
97         parsed.map { feature =>
98           if (array == null) {
99             array = Array.ofDim(feature.getAttributeCount + 1)
100           }
101           var i = 1
102           while (i < array.length) {
103             array(i) = feature.getAttribute(i - 1)
104             i += 1
105           }
106           array(0) = feature.getID
107           array
108         }
109     }
110   }
111 }

113 object ShapefileConverter extends LazyLogging {

115   /**
116     * Creates a URL, needed for the shapefile data store
117     *
118     * @param path input path
119     * @return
120     */
121   def getDataStore(path: String, charset: Charset): ShapefileDataStore = {
122     val params = java.util.Map.of(
123       ShapefileDataStoreFactory.URLP.key, PathUtils.getUrl(path),
124       ShapefileDataStoreFactory.DBFCHARSET.key, charset
125     )
126     val ds = DataStoreFinder.getDataStore(params).asInstanceOf[ShapefileDataStore]
127     if (ds == null) {
128       throw new IllegalArgumentException(s"Could not read shapefile using path '$path'")
129     }
130     tryInferCharsetFromCPG(path).foreach(ds.setCharset)
131     ds
132   }
133 
134   // Infer charset to decode strings in DBF file by inspecting the content of the CPG file. 
135   private def tryInferCharsetFromCPG(path: String): Option[Charset] = {
136     val shpDirPath = Paths.get(path).getParent
137     val (baseName, _) = PathUtils.getBaseNameAndExtension(path)
138     val cpgPath = shpDirPath.resolve(baseName + ".cpg")
139     if (!Files.isRegularFile(cpgPath)) { None } else {
140       val source = scala.io.Source.fromFile(cpgPath.toFile)
141       try {
142         source.getLines.take(1).toList match {
143           case Nil => None
144           case charsetName :: _ => Some(Charset.forName(charsetName.trim))
145         }
146       } catch {
147         case _: Exception =>
148           logger.warn("Can't figure out charset from cpg file, will use default charset")
149           None
150       } finally {
151         source.close()
152       }
153     }
154   }
