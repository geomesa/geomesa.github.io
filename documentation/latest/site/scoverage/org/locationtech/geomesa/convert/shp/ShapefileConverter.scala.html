30 class ShapefileConverter(sft: SimpleFeatureType, config: BasicConfig, fields: Seq[BasicField], options: BasicOptions)
31     extends AbstractConverter[SimpleFeature, BasicConfig, BasicField, BasicOptions](sft, config, fields, options)  {
32 
33   import org.locationtech.geomesa.convert.shp.ShapefileFunctionFactory.{InputSchemaKey, InputValuesKey}
34 
35   // inject placeholders for shapefile attributes into the evaluation context
36   // used for accessing shapefile properties by name in ShapefileFunctionFactory
37   private def shpParams: Map[String, Any] =
38     Map(InputSchemaKey -> ArrayBuffer.empty[String], InputValuesKey -> ArrayBuffer.empty[AnyRef])
39 
40   override def createEvaluationContext(globalParams: Map[String, Any]): EvaluationContext =
41     super.createEvaluationContext(globalParams ++ shpParams)
42 
43   override protected def parse(is: InputStream, ec: EvaluationContext): CloseableIterator[SimpleFeature] = {
44     CloseWithLogging(is) // we don't use the input stream, just close it
45 
46     val path = ec.getInputFilePath.getOrElse {
47       throw new IllegalArgumentException(s"Shapefile converter requires '${EvaluationContext.InputFilePathKey}' " +
48           "to be set in the evaluation context")
49     }
50     val ds = ShapefileConverter.getDataStore(path, options.encoding)
51     val schema = ds.getSchema()
52 
53     (ec.accessor(InputSchemaKey).apply(), ec.accessor(InputValuesKey).apply()) match {
54       case (n: ArrayBuffer[String], v: ArrayBuffer[AnyRef]) =>
55         n.clear()
56         n ++= Array.tabulate(schema.getAttributeCount)(i => schema.getDescriptor(i).getLocalName)
57         v.clear()
58         v ++= Array.fill[AnyRef](n.length + 1)(null)
59       case _ =>
60         logger.warn("Input schema not found in evaluation context, shapefile functions " +
61             s"${TextTools.wordList(new ShapefileFunctionFactory().functions.map(_.names.head))} will not be available")
62     }
63 
64     val q = new Query
65     // Only ask to reproject if the Shapefile has a non-4326 CRS
66     if (ds.getSchema.getCoordinateReferenceSystem == null) {
67       logger.warn(s"Shapefile does not have CRS info")
68     } else if (!CRS.equalsIgnoreMetadata(ds.getSchema.getCoordinateReferenceSystem, CRS_EPSG_4326)) {
69       q.setCoordinateSystemReproject(CRS_EPSG_4326)
70     }
71 
72     val reader = CloseableIterator(ds.getFeatureSource.getReader(q)).map { f => ec.line += 1; f }
73 
74     CloseableIterator(reader, { CloseWithLogging(reader); ds.dispose() })
75   }
76 
77   override protected def values(parsed: CloseableIterator[SimpleFeature],
78                                 ec: EvaluationContext): CloseableIterator[Array[Any]] = {
79     (ec.accessor(InputSchemaKey).apply(), ec.accessor(InputValuesKey).apply()) match {
80       case (_: ArrayBuffer[String], v: ArrayBuffer[AnyRef]) =>
81         parsed.map { feature =>
82           var i = 1
83           while (i < v.length) {
84             v(i) = feature.getAttribute(i - 1)
85             i += 1
86           }
87           v(0) = feature.getID
88           v.toArray
89         }
90 
91       case _ =>
92         logger.warn("Input schema not found in evaluation context, shapefile functions " +
93             s"${TextTools.wordList(new ShapefileFunctionFactory().functions.map(_.names.head))} will not be available")
94         var array: Array[Any] = null
95         parsed.map { feature =>
96           if (array == null) {
97             array = Array.ofDim(feature.getAttributeCount + 1)
98           }
99           var i = 1
100           while (i < array.length) {
101             array(i) = feature.getAttribute(i - 1)
102             i += 1
103           }
104           array(0) = feature.getID
105           array
106         }
107     }
108   }
109 }

111 object ShapefileConverter extends LazyLogging {
112 
113   /**
114     * Creates a URL, needed for the shapefile data store
115     *
116     * @param path input path
117     * @return
118     */
119   def getDataStore(path: String, charset: Charset): ShapefileDataStore = {
120     val params = java.util.Map.of(
121       ShapefileDataStoreFactory.URLP.key, PathUtils.getUrl(path),
122       ShapefileDataStoreFactory.DBFCHARSET.key, charset
123     )
124     val ds = DataStoreFinder.getDataStore(params).asInstanceOf[ShapefileDataStore]
125     if (ds == null) {
126       throw new IllegalArgumentException(s"Could not read shapefile using path '$path'")
127     }
128     tryInferCharsetFromCPG(path).foreach(ds.setCharset)
129     ds
130   }
131 
132   // Infer charset to decode strings in DBF file by inspecting the content of the CPG file. 
133   private def tryInferCharsetFromCPG(path: String): Option[Charset] = {
134     val shpDirPath = Paths.get(path).getParent
135     val (baseName, _) = PathUtils.getBaseNameAndExtension(path)
136     val cpgPath = shpDirPath.resolve(baseName + ".cpg")
137     if (!Files.isRegularFile(cpgPath)) { None } else {
138       val source = scala.io.Source.fromFile(cpgPath.toFile)
139       try {
140         source.getLines.take(1).toList match {
141           case Nil => None
142           case charsetName :: _ => Some(Charset.forName(charsetName.trim))
143         }
144       } catch {
145         case _: Exception =>
146           logger.warn("Can't figure out charset from cpg file, will use default charset")
147           None
148       } finally {
149         source.close()
150       }
151     }
152   }
