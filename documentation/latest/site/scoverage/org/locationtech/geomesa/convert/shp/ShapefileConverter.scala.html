30 class ShapefileConverter(sft: SimpleFeatureType, config: BasicConfig, fields: Seq[BasicField], options: BasicOptions)
31     extends AbstractConverter[SimpleFeature, BasicConfig, BasicField, BasicOptions](sft, config, fields, options)  {

35   // inject placeholders for shapefile attributes into the evaluation context
36   // used for accessing shapefile properties by name in ShapefileFunctionFactory
37   private def shpParams: Map[String, Any] =
38     Map(InputSchemaKey -> ArrayBuffer.empty[String], InputValuesKey -> ArrayBuffer.empty[AnyRef])
39 
40   override def createEvaluationContext(globalParams: Map[String, Any]): EvaluationContext =
41     super.createEvaluationContext(globalParams ++ shpParams)
42 
43   override def createEvaluationContext(globalParams: Map[String, Any], success: Counter, failure: Counter): EvaluationContext =
44     super.createEvaluationContext(globalParams ++ shpParams, success, failure)

46   override protected def parse(is: InputStream, ec: EvaluationContext): CloseableIterator[SimpleFeature] = {
47     CloseWithLogging(is) // we don't use the input stream, just close it
48 
49     val path = ec.getInputFilePath.getOrElse {
50       throw new IllegalArgumentException(s"Shapefile converter requires '${EvaluationContext.InputFilePathKey}' " +
51           "to be set in the evaluation context")
52     }
53     val ds = ShapefileConverter.getDataStore(path, options.encoding)
54     val schema = ds.getSchema()
55 
56     (ec.accessor(InputSchemaKey).apply(), ec.accessor(InputValuesKey).apply()) match {
57       case (n: ArrayBuffer[String], v: ArrayBuffer[AnyRef]) =>
58         n.clear()
59         n ++= Array.tabulate(schema.getAttributeCount)(i => schema.getDescriptor(i).getLocalName)
60         v.clear()
61         v ++= Array.fill[AnyRef](n.length + 1)(null)
62       case _ =>
63         logger.warn("Input schema not found in evaluation context, shapefile functions " +
64             s"${TextTools.wordList(new ShapefileFunctionFactory().functions.map(_.names.head))} will not be available")
65     }
66 
67     val q = new Query
68     // Only ask to reproject if the Shapefile has a non-4326 CRS
69     if (ds.getSchema.getCoordinateReferenceSystem == null) {
70       logger.warn(s"Shapefile does not have CRS info")
71     } else if (!CRS.equalsIgnoreMetadata(ds.getSchema.getCoordinateReferenceSystem, CRS_EPSG_4326)) {
72       q.setCoordinateSystemReproject(CRS_EPSG_4326)
73     }
74 
75     val reader = CloseableIterator(ds.getFeatureSource.getReader(q)).map { f => ec.line += 1; f }
76 
77     CloseableIterator(reader, { CloseWithLogging(reader); ds.dispose() })
78   }
79 
80   override protected def values(parsed: CloseableIterator[SimpleFeature],
81                                 ec: EvaluationContext): CloseableIterator[Array[Any]] = {
82     (ec.accessor(InputSchemaKey).apply(), ec.accessor(InputValuesKey).apply()) match {
83       case (_: ArrayBuffer[String], v: ArrayBuffer[AnyRef]) =>
84         parsed.map { feature =>
85           var i = 1
86           while (i < v.length) {
87             v(i) = feature.getAttribute(i - 1)
88             i += 1
89           }
90           v(0) = feature.getID
91           v.toArray
92         }
93 
94       case _ =>
95         logger.warn("Input schema not found in evaluation context, shapefile functions " +
96             s"${TextTools.wordList(new ShapefileFunctionFactory().functions.map(_.names.head))} will not be available")
97         var array: Array[Any] = null
98         parsed.map { feature =>
99           if (array == null) {
100             array = Array.ofDim(feature.getAttributeCount + 1)
101           }
102           var i = 1
103           while (i < array.length) {
104             array(i) = feature.getAttribute(i - 1)
105             i += 1
106           }
107           array(0) = feature.getID
108           array
109         }
110     }
111   }
112 }

114 object ShapefileConverter extends LazyLogging {
115 
116   /**
117     * Creates a URL, needed for the shapefile data store
118     *
119     * @param path input path
120     * @return
121     */
122   def getDataStore(path: String, charset: Charset): ShapefileDataStore = {
123     val params = java.util.Map.of(
124       ShapefileDataStoreFactory.URLP.key, PathUtils.getUrl(path),
125       ShapefileDataStoreFactory.DBFCHARSET.key, charset
126     )
127     val ds = DataStoreFinder.getDataStore(params).asInstanceOf[ShapefileDataStore]
128     if (ds == null) {
129       throw new IllegalArgumentException(s"Could not read shapefile using path '$path'")
130     }
131     tryInferCharsetFromCPG(path).foreach(ds.setCharset)
132     ds
133   }
134 
135   // Infer charset to decode strings in DBF file by inspecting the content of the CPG file. 
136   private def tryInferCharsetFromCPG(path: String): Option[Charset] = {
137     val shpDirPath = Paths.get(path).getParent
138     val (baseName, _) = PathUtils.getBaseNameAndExtension(path)
139     val cpgPath = shpDirPath.resolve(baseName + ".cpg")
140     if (!Files.isRegularFile(cpgPath)) { None } else {
141       val source = scala.io.Source.fromFile(cpgPath.toFile)
142       try {
143         source.getLines.take(1).toList match {
144           case Nil => None
145           case charsetName :: _ => Some(Charset.forName(charsetName.trim))
146         }
147       } catch {
148         case _: Exception =>
149           logger.warn("Can't figure out charset from cpg file, will use default charset")
150           None
151       } finally {
152         source.close()
153       }
154     }
155   }
