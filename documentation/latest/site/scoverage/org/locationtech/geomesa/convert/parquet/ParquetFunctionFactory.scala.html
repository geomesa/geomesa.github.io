35   /**
36    * Zip x and y values into coordinates
37    *
38    * @param x x values
39    * @param y corresponding y values
40    * @return
41    */
42   private def zip(x: java.util.List[Double], y: java.util.List[Double]): Array[Coordinate] = {
43     val result = Array.ofDim[Coordinate](x.size)
44     var i = 0
45     while (i < result.length) {
46       result(i) = new Coordinate(x.get(i), y.get(i))
47       i += 1
48     }
49     result
50   }

52   /**
53    * Base class for handling parquet-encoded geometries
54    *
55    * @param name function name
56    * @param path optional avro path
57    * @tparam T geometry type
58    * @tparam U column type
59    */
60   abstract class ParquetGeometryFn[T <: Geometry, U](name: String, path: Option[AvroPath])
61       extends NamedTransformerFunction(Seq(name), pure = true) {
62 
63     import org.locationtech.geomesa.fs.storage.parquet.io.GeometrySchema.{GeometryColumnX, GeometryColumnY}
64 
65     override def apply(args: Array[AnyRef]): AnyRef = {
66       val attribute = path match {
67         case None => args.headOption
68         case Some(p) => p.eval(args(0).asInstanceOf[GenericRecord])
69       }
70       attribute match {
71         case Some(b: Array[Byte]) => WKBUtils.read(b).asInstanceOf[T]
72         case Some(r: GenericRecord) => eval(r.get(GeometryColumnX).asInstanceOf[U], r.get(GeometryColumnY).asInstanceOf[U])
73         case Some(g: Geometry) => g
74         case _ => null
75       }
76     }
77 
78     protected def eval(x: U, y: U): T
79   }
