30                                        transform: Option[Expression]): Either[ConfigReaderFailures, FixedWidthField] = {
31       val startCur = cur.atKeyOrUndefined("start")
32       val widthCur = cur.atKeyOrUndefined("width")
33       if (startCur.isUndefined && widthCur.isUndefined) { Right(DerivedField(name, transform)) } else {
34         for {
35           start <- PrimitiveConvert.intConfigReader.from(startCur).right
36           width <- PrimitiveConvert.intConfigReader.from(widthCur).right

38           OffsetField(name, transform, start, width)
39         }
40       }
41     }

47       }
