79     override final def hasNext: Boolean = results != null || {
80       CloseWithLogging(Option(statement))

114     private var array: Array[Any] = _
115     private var hasNextResult = false
116 
117     @tailrec
118     override final def hasNext: Boolean = hasNextResult || {
119       CloseWithLogging(Option(results))
120       if (!iter.hasNext) {
121         results = null
122         false
123       } else {
124         results = iter.next
125         array = Array.ofDim[Any](results.getMetaData.getColumnCount + 1)
126         hasNextResult = results.next()
127         hasNext
128       }
129     }
130 
131     override def next(): Array[Any] = {
132       if (!hasNext) { Iterator.empty.next() } else {
133         ec.line += 1
134         // the first column will hold the entire row, but set it empty here to
135         // avoid the previous row being captured in mkString, below
136         array(0) = ""
137         var i = 1
138         while (i < array.length) {
139           array(i) = results.getObject(i) // note: results are 1-indexed
140           i += 1
141         }
142         array(0) = array.mkString // set the whole row value for reference
143         hasNextResult = results.next()
144         array
145       }
146     }
147 
148     override def close(): Unit = CloseWithLogging(Option(results), iter)
149   }
