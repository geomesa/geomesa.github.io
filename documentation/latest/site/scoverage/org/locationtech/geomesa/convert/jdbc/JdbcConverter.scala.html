71       extends CloseableIterator[ResultSet] {

75     private var statement: PreparedStatement = _
76     private var results: ResultSet = _

78     override final def hasNext: Boolean = results != null || {
79       CloseWithLogging(Option(statement))
80       if (!statements.hasNext) {

98     }

110       extends CloseableIterator[Array[Any]] {
111 
112     private var results: ResultSet = _
113     private var array: Array[Any] = _
114     private var hasNextResult = false
115 
116     @tailrec
117     override final def hasNext: Boolean = hasNextResult || {
118       CloseWithLogging(Option(results))
119       if (!iter.hasNext) {
120         results = null
121         false
122       } else {
123         results = iter.next
124         array = Array.ofDim[Any](results.getMetaData.getColumnCount + 1)
125         hasNextResult = results.next()
126         hasNext
127       }
128     }
129 
130     override def next(): Array[Any] = {
131       if (!hasNext) { Iterator.empty.next() } else {
132         ec.line += 1
133         // the first column will hold the entire row, but set it empty here to
134         // avoid the previous row being captured in mkString, below
135         array(0) = ""
136         var i = 1
137         while (i < array.length) {
138           array(i) = results.getObject(i) // note: results are 1-indexed
139           i += 1
140         }
141         array(0) = array.mkString // set the whole row value for reference
142         hasNextResult = results.next()
143         array
144       }
145     }
146 
147     override def close(): Unit = CloseWithLogging(Option(results), iter)
148   }
