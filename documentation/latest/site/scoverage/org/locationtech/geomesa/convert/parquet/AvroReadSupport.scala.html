60   /**
61     * Schema-less implementation of GenericRecord
62     *
63     * @param fields field names
64     */
65   class AvroRecord(fields: IndexedSeq[String]) extends GenericRecord {
66 
67     private val values = Array.ofDim[AnyRef](fields.length)
68 
69     override def put(key: String, v: AnyRef): Unit = values(fields.indexOf(key)) = v
70     override def get(key: String): AnyRef = values(fields.indexOf(key))
71     override def put(i: Int, v: AnyRef): Unit = values(i) = v
72     override def get(i: Int): AnyRef = values(i)
73 
74     override def getSchema: Schema = null
75 
76     override def toString: String = {
77       val builder = new StringBuilder("AvroRecord[")
78       var i = 0
79       while (i < fields.length) {
80         if (i != 0) {
81           builder.append(',')
82         }
83         builder.append(fields(i)).append(':').append(values(i))
84         i += 1
85       }
86       builder.append(']')
87       builder.toString
88     }
89   }

97   /**
98     * Group converter for a record
99     *
100     * @param fields fields
101     */
102   class GenericGroupConverter(fields: Seq[Type]) extends GroupConverter with ValueMaterializer {
103 
104     private val names = fields.map(_.getName).toIndexedSeq
105     private val converters = Array.tabulate(fields.length)(i => converter(fields(i)))
106 
107     private var rec: AvroRecord = _
108 
109     def record: GenericRecord = rec
110 
111     override def getConverter(fieldIndex: Int): Converter = converters(fieldIndex)
112     override def start(): Unit = {
113       rec = new AvroRecord(names)
114       converters.foreach(_.reset())
115     }
116     override def end(): Unit = {
117       var i = 0
118       while (i < fields.length) {
119         rec.put(i, converters(i).materialize())
120         i += 1
121       }
122     }
123     override def reset(): Unit = rec = null
124     override def materialize(): AnyRef = rec
125   }

127   /**
128     * Get a converter for a field type
129     *
130     * @param field field
131     * @return
132     */
133   private def converter(field: Type): Converter with ValueMaterializer = {
134     val logical = field.getLogicalTypeAnnotation
135     if (field.isPrimitive) {
136       lazy val stringOrBytes = logical match {
137         case _: StringLogicalTypeAnnotation => new StringConverter()
138         case _ => new BytesConverter()
139       }
140       lazy val intOrDate = logical match {
141         case _: DateLogicalTypeAnnotation => new DaysConverter()
142         case _ => new IntConverter()
143       }
144       lazy val longOrTimestamp = logical match {
145         case t: TimestampLogicalTypeAnnotation if t.getUnit == LogicalTypeAnnotation.TimeUnit.MILLIS => new DateConverter()
146         case t: TimestampLogicalTypeAnnotation if t.getUnit == LogicalTypeAnnotation.TimeUnit.MICROS => new MicrosConverter()
147         case _ => new LongConverter()
148       }
149       val convert = field.asPrimitiveType().getPrimitiveTypeName match {
150         case PrimitiveTypeName.BINARY               => stringOrBytes
151         case PrimitiveTypeName.INT32                => intOrDate
152         case PrimitiveTypeName.INT64                => longOrTimestamp
153         case PrimitiveTypeName.DOUBLE               => new DoubleConverter()
154         case PrimitiveTypeName.FLOAT                => new FloatConverter()
155         case PrimitiveTypeName.BOOLEAN              => new BooleanConverter()
156         case PrimitiveTypeName.FIXED_LEN_BYTE_ARRAY => new BytesConverter()
157         case _                                      => NullPrimitiveConverter
158       }
159       if (field.isRepetition(Repetition.REPEATED)) {
160         new RepeatedPrimitiveConverter(convert)
161       } else {
162         convert
163       }
164     } else {
165       val group = field.asGroupType()
166       logical match {
167         case _: ListLogicalTypeAnnotation =>
168           require(group.getFieldCount == 1 && !group.getType(0).isPrimitive, s"Invalid list type: $group")
169           val list = group.getType(0).asGroupType()
170           require(list.getFieldCount == 1 && list.isRepetition(Repetition.REPEATED), s"Invalid list type: $group")
171           new GenericListConverter(list.getType(0))
172 
173         case _: MapLogicalTypeAnnotation =>
174           require(group.getFieldCount == 1 && !group.getType(0).isPrimitive, s"Invalid map type: $group")
175           val map = group.getType(0).asGroupType()
176           require(map.getFieldCount == 2 && map.isRepetition(Repetition.REPEATED), s"Invalid map type: $group")
177           new GenericMapConverter(map.getType(0), map.getType(1))
178 
179         case _ =>
180           new GenericGroupConverter(group.getFields.asScala.toSeq)
181       }
182     }
183   }

216     private var map: java.util.Map[AnyRef, AnyRef] = _

223         if (fieldIndex == 0) { keyConverter } else { valueConverter }

226       override def end(): Unit = map.put(keyConverter.materialize(), valueConverter.materialize())

230     override def start(): Unit = map = new java.util.HashMap[AnyRef, AnyRef]()

236   /**
237    * Converter for primitive fields with repetition of 'repeated'
238    *
239    * @param delegate single value converter
240    */
241   private class RepeatedPrimitiveConverter(delegate: PrimitiveConverter with ValueMaterializer)
242     extends PrimitiveConverter with ValueMaterializer {
243 
244     private var list: java.util.List[AnyRef] = _
245 
246     override def addBinary(value: Binary): Unit = { delegate.asPrimitiveConverter().addBinary(value); addElement() }
247     override def addBoolean(value: Boolean): Unit = { delegate.asPrimitiveConverter().addBoolean(value); addElement() }
248     override def addInt(value: Int): Unit = { delegate.asPrimitiveConverter().addInt(value); addElement() }
249     override def addFloat(value: Float): Unit = { delegate.asPrimitiveConverter().addFloat(value); addElement() }
250     override def addLong(value: Long): Unit = { delegate.asPrimitiveConverter().addLong(value); addElement() }
251     override def addDouble(value: Double): Unit = { delegate.asPrimitiveConverter().addDouble(value); addElement() }
252 
253     private def addElement(): Unit = {
254       if (list == null) {
255         list = new java.util.ArrayList[AnyRef]()
256       }
257       list.add(delegate.materialize())
258       delegate.reset()
259     }
260     override def reset(): Unit = list = null
261     override def materialize(): AnyRef = list
262   }
