64   class AvroRecordMaterializer(fileSchema: MessageType, schema: Option[SimpleFeatureParquetSchema])
65       extends RecordMaterializer[GenericRecord] {
66     private val root = new GenericGroupConverter(fileSchema, schema)
67     override def getCurrentRecord: GenericRecord = root.materialize()
68     override def getRootConverter: GroupConverter = root
69   }

71   /**
72    * Group converter for a record
73    *
74    * @param fileSchema parquet schema
75    * @param schema geomesa schema encodings, if it's a geomesa file
76    */
77   private class GenericGroupConverter(fileSchema: MessageType, schema: Option[SimpleFeatureParquetSchema])
78       extends GroupConverter with ValueMaterializer[GenericRecord] {
79 
80     private val avroSchema = new AvroSchemaConverter().convert(fileSchema)
81     private val fields = fileSchema.getFields.asScala.toSeq
82     private val converters = schema match {
83       case None => Array.tabulate(fields.length)(i => converter(fields(i)))
84       case Some(s) =>
85         // for attributes, we re-use our parquet read support so that they get parsed into standard simple feature attribute
86         // types (including geometries) instead of generic avro types
87         val attributes =
88           s.sft.getAttributeDescriptors.asScala.map(d => SimpleFeatureReadSupport.attribute(ObjectType.selectType(d), s.encodings))
89         // the remaining non-attribute fields are added as generic types (currently fid, vis, and bboxes)
90         val remaining = fields.drop(s.sft.getAttributeCount).map(converter).toArray
91         attributes.toArray ++ remaining
92     }
93 
94     private var rec: GenericRecord = _
95 
96     override def getConverter(fieldIndex: Int): Converter = converters(fieldIndex)
97     override def start(): Unit = {
98       rec = new GenericData.Record(avroSchema)
99       converters.foreach(_.reset())
100     }
101     override def end(): Unit = {
102       var i = 0
103       while (i < fields.length) {
104         rec.put(i, converters(i).materialize())
105         i += 1
106       }
107     }
108     override def reset(): Unit = rec = null
109     override def materialize(): GenericRecord = rec
110   }
111 
112   /**
113     * Get a converter for a field type
114     *
115     * @param field field
116     * @return
117     */
118   private def converter(field: Type): ValueMaterializer[_ <: AnyRef] = {
119     val logical = field.getLogicalTypeAnnotation
120     if (field.isPrimitive) {
121       lazy val logicalConverter = logical match {
122         case _: StringLogicalTypeAnnotation => Some(new StringConverter())
123         case _: DateLogicalTypeAnnotation => Some(new DaysConverter())
124         case t: TimestampLogicalTypeAnnotation if t.getUnit == LogicalTypeAnnotation.TimeUnit.MILLIS => Some(new DateMillisConverter())
125         case t: TimestampLogicalTypeAnnotation if t.getUnit == LogicalTypeAnnotation.TimeUnit.MICROS => Some(new DateMicrosConverter())
126         case t: TimestampLogicalTypeAnnotation if t.getUnit == LogicalTypeAnnotation.TimeUnit.NANOS => Some(new NanosConverter())
127         case _: UUIDLogicalTypeAnnotation => Some(new UuidConverter())
128         case _ => None
129       }
130       val convert = field.asPrimitiveType().getPrimitiveTypeName match {
131         case PrimitiveTypeName.BINARY               => logicalConverter.getOrElse(new GeneralPrimitiveConverter())
132         case PrimitiveTypeName.INT32                => logicalConverter.getOrElse(new GeneralPrimitiveConverter())
133         case PrimitiveTypeName.INT64                => logicalConverter.getOrElse(new GeneralPrimitiveConverter())
134         case PrimitiveTypeName.DOUBLE               => new GeneralPrimitiveConverter()
135         case PrimitiveTypeName.FLOAT                => new GeneralPrimitiveConverter()
136         case PrimitiveTypeName.BOOLEAN              => new GeneralPrimitiveConverter()
137         case PrimitiveTypeName.FIXED_LEN_BYTE_ARRAY => logicalConverter.getOrElse(new GeneralPrimitiveConverter())
138         case _                                      => NullPrimitiveConverter
139       }
140       if (field.isRepetition(Repetition.REPEATED)) {
141         new RepeatedPrimitiveConverter(convert)
142       } else {
143         convert
144       }
145     } else {
146       val group = field.asGroupType()
147       def genericGroupConverter(): GenericGroupConverter =
148         new GenericGroupConverter(new MessageType(group.getName, group.getFields), None)
149       logical match {
150         case _: ListLogicalTypeAnnotation =>
151           ParquetConverterFactory.getListElementType(group) match {
152             case Some(items) => new ListConverter(converter(items))
153             case _ => genericGroupConverter()
154           }
155 
156         case _: MapLogicalTypeAnnotation =>
157           ParquetConverterFactory.getMapKeyValueTypes(group) match {
158             case Some((k, v)) => new MapConverter(converter(k), converter(v))
159             case _ => genericGroupConverter()
160           }
161 
162         case _ =>
163           genericGroupConverter()
164       }
165     }
166   }

168   /**
169    * Converter for primitive fields with repetition of 'repeated'
170    *
171    * @param delegate single value converter
172    */
173   private class RepeatedPrimitiveConverter(delegate: PrimitiveConverter with ValueMaterializer[_ <: AnyRef])
174     extends PrimitiveConverter with ValueMaterializer[java.util.List[AnyRef]] {
175 
176     private var list: java.util.List[AnyRef] = _
177 
178     override def addBinary(value: Binary): Unit = { delegate.addBinary(value); addElement() }
179     override def addBoolean(value: Boolean): Unit = { delegate.addBoolean(value); addElement() }
180     override def addInt(value: Int): Unit = { delegate.addInt(value); addElement() }
181     override def addFloat(value: Float): Unit = { delegate.addFloat(value); addElement() }
182     override def addLong(value: Long): Unit = { delegate.addLong(value); addElement() }
183     override def addDouble(value: Double): Unit = { delegate.addDouble(value); addElement() }
184 
185     private def addElement(): Unit = {
186       if (list == null) {
187         list = new java.util.ArrayList[AnyRef]()
188       }
189       list.add(delegate.materialize())
190       delegate.reset()
191     }
192     override def reset(): Unit = list = null
193     override def materialize(): java.util.List[AnyRef] = list
194   }
