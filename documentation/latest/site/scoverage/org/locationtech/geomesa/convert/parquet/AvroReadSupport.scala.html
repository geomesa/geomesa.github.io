56   class AvroRecord(fields: IndexedSeq[String]) extends GenericRecord {
57 
58     private val values = Array.ofDim[AnyRef](fields.length)
59 
60     override def put(key: String, v: AnyRef): Unit = values(fields.indexOf(key)) = v
61     override def get(key: String): AnyRef = values(fields.indexOf(key))
62     override def put(i: Int, v: AnyRef): Unit = values(i) = v
63     override def get(i: Int): AnyRef = values(i)
64 
65     override def getSchema: Schema = null
66 
67     override def toString: String = {
68       val builder = new StringBuilder("AvroRecord[")
69       var i = 0
70       while (i < fields.length) {
71         if (i != 0) {
72           builder.append(',')
73         }
74         builder.append(fields(i)).append(':').append(values(i))
75         i += 1
76       }
77       builder.append(']')
78       builder.toString
79     }
80   }

109   /**
110     * Get a converter for a field type
111     *
112     * @param field field
113     * @param i callback index to set
114     * @param callback callback to set result
115     * @return
116     */
117   private def converter(field: Type, i: Int, callback: Settable): Converter = {
118     val original = field.getOriginalType
119     if (field.isPrimitive) {
120       lazy val string = original match {
121         case OriginalType.UTF8 => new StringConverter(i, callback)
122         case _ => new BytesConverter(i, callback)
123       }
124       lazy val int32 = original match {
125         case OriginalType.DATE => new DaysConverter(i, callback)
126         case _ => new IntConverter(i, callback)
127       }
128       lazy val int64 = original match {
129         case OriginalType.TIMESTAMP_MILLIS => new DateConverter(i, callback)
130         case OriginalType.TIMESTAMP_MICROS => new MicrosConverter(i, callback)
131         case _ => new LongConverter(i, callback)
132       }
133       field.asPrimitiveType().getPrimitiveTypeName match {
134         case PrimitiveTypeName.BINARY               => string
135         case PrimitiveTypeName.INT32                => int32
136         case PrimitiveTypeName.INT64                => int64
137         case PrimitiveTypeName.DOUBLE               => new DoubleConverter(i, callback)
138         case PrimitiveTypeName.FLOAT                => new FloatConverter(i, callback)
139         case PrimitiveTypeName.BOOLEAN              => new BooleanConverter(i, callback)
140         case PrimitiveTypeName.FIXED_LEN_BYTE_ARRAY => new BytesConverter(i, callback)
141         case _                                      => NullPrimitiveConverter
142       }
143     } else {
144       val group = field.asGroupType()
145       original match {
146         case OriginalType.LIST =>
147           require(group.getFieldCount == 1 && !group.getType(0).isPrimitive, s"Invalid list type: $group")
148           val list = group.getType(0).asGroupType()
149           require(list.getFieldCount == 1 && list.isRepetition(Repetition.REPEATED), s"Invalid list type: $group")
150           new GenericListConverter(list.getType(0), i, callback)
151 
152         case OriginalType.MAP =>
153           require(group.getFieldCount == 1 && !group.getType(0).isPrimitive, s"Invalid map type: $group")
154           val map = group.getType(0).asGroupType()
155           require(map.getFieldCount == 2 && map.isRepetition(Repetition.REPEATED), s"Invalid map type: $group")
156           new GenericMapConverter(map.getType(0), map.getType(1), i, callback)
157 
158         case _ =>
159           if (group.getFields.asScala.forall(t => t.isPrimitive && t.isRepetition(Repetition.REPEATED))) {
160             new GroupedRepeatedConverter(group.getFields.asScala.map(_.asPrimitiveType).toSeq, i, callback)
161           } else {
162             new GenericGroupConverter(group.getFields.asScala.toSeq) {
163               override def end(): Unit = callback.set(i, record)
164             }
165           }
166       }
167     }
168   }

202     * @param keys key type

207   class GenericMapConverter(keys: Type, values: Type, index: Int, callback: Settable) extends GroupConverter {

213     private val group: GroupConverter = new GroupConverter {
214       private var k: AnyRef = _

239   class RepeatedConverter(field: PrimitiveType, index: Int, callback: Settable)
240       extends GroupConverter with Settable {
241 
242     private val array = ArrayBuffer.empty[Any]
243 
244     private val converter = AvroReadSupport.converter(field, 0, this)
245 
246     override def getConverter(fieldIndex: Int): Converter = converter
247 
248     override def start(): Unit = array.clear()
249 
250     override def end(): Unit = {
251       val list = new java.util.ArrayList[Any](array.length)
252       array.foreach(list.add)
253       callback.set(index, list)
254     }
255 
256     override def set(ignored: Int, value: AnyRef): Unit = array += value
257   }

259   /**
260     * Group converter for handling groups that consist solely of REPEATED fields. Returns a GenericRecord
261     * containing the repeated fields
262     *
263     * @param fields repeated field types
264     * @param index callback index to set
265     * @param callback callback to set result

267   class GroupedRepeatedConverter(fields: Seq[PrimitiveType], index: Int, callback: Settable)
268       extends GroupConverter with Settable {
269 
270     private val names = fields.map(_.getName).toIndexedSeq
271 
272     private val arrays = Array.fill(fields.length)(ArrayBuffer.empty[Any])
273 
274     private val converters = Array.tabulate(fields.length)(i => AvroReadSupport.converter(fields(i), i, this))
275 
276     override def getConverter(fieldIndex: Int): Converter = converters(fieldIndex)
277 
278     override def start(): Unit = arrays.foreach(_.clear())
279 
280     override def end(): Unit = {
281       val rec = new AvroRecord(names)
282       var i = 0
283       while (i < arrays.length) {
284         val list = new java.util.ArrayList[Any](arrays(i).length)
285         arrays(i).foreach(list.add)
286         rec.put(i, list)
287         i += 1
288       }
289       callback.set(index, rec)
290     }
291 
292     override def set(i: Int, value: AnyRef): Unit = arrays(i) += value
293   }
