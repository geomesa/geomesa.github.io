19 /**
20  * Holds the state associated with a conversion attempt. Evaluation contexts are not thread safe.
21  */
22 trait EvaluationContext {
23 
24   /**
25     * The current line being processed.
26     *
27     * This may be an actual line (e.g. a csv row), or a logical line (e.g. an avro record)
28     *
29     * @return
30     */
31   var line: Long = 0
32 
33   /**
34     * Enrichment caches
35     *
36     * @return
37     */
38   def cache: Map[String, EnrichmentCache]
39 
40   /**
41     * Metrics registry, accessible for tracking any custom values
42     *
43     * @return
44     */
45   def metrics: ConverterMetrics
46 
47   /**
48     * Counter for tracking successes
49     *
50     * @return
51     */
52   def success: com.codahale.metrics.Counter
53 
54   /**
55     * Counter for tracking failures
56     *
57     * @return
58     */
59   def failure: com.codahale.metrics.Counter
60 
61   /**
62    * Access to any errors that have occurred - note that errors will generally only be kept if the converter
63    * error mode is set to `ReturnErrors`
64    *
65    * @return
66    */
67   def errors: java.util.Queue[EvaluationError]
68 
69   /**
70    * Gets a references to a field's value
71    *
72    * @param name field name
73    * @return
74    */
75   def accessor(name: String): FieldAccessor
76 
77   /**
78    * Evaluate all values using the given arguments. The returned array may be mutated on subsequent calls to
79    * `evaluate`, so shouldn't be kept long-term
80    *
81    * @param args single row of input
82    */
83   def evaluate(args: Array[AnyRef]): Either[EvaluationError, Array[AnyRef]]
84 }

86 object EvaluationContext extends LazyLogging {
87 
88   val InputFilePathKey = "inputFilePath"
89   val FilterKey = "filter"
90 
91   /**
92     * Creates a new, empty evaluation context
93     *
94     * @return
95     */
96   def empty: EvaluationContext = {
97     val metrics = ConverterMetrics.empty
98     val success = metrics.counter("success")
99     val failures = metrics.counter("failure")
100     new StatefulEvaluationContext(Array.empty, Map.empty, Map.empty, metrics, success, failures)
101   }
102 
103   /**
104    * Creates a new evaluation context with the given state
105    *
106    * @param fields converter fields, in topological dependency order
107    * @param globalValues global values
108    * @param caches enrichment caches
109    * @param metrics metrics
110    * @param success success counter
111    * @param failure failure counter
112    * @return
113    */
114   def apply(
115       fields: Seq[Field],
116       globalValues: Map[String, _ <: AnyRef],
117       caches: Map[String, EnrichmentCache],
118       metrics: ConverterMetrics,
119       success: Counter,
120       failure: Counter): EvaluationContext = {
121     new StatefulEvaluationContext(fields.toArray, globalValues, caches, metrics, success, failure)
122   }
123 
124   /**
125     * Gets a global parameter map containing the input file path
126     *
127     * @param file input file path
128     * @return
129     */
130   def inputFileParam(file: String): Map[String, AnyRef] = Map(InputFilePathKey -> file)
131 
132   /**
133    * Trait for reading a field from an evaluation context
134    */
135   sealed trait FieldAccessor {
136     def apply(): AnyRef
137   }
138 
139   case object NullFieldAccessor extends FieldAccessor {
140     override def apply(): AnyRef = null
141   }
142 
143   class FieldValueAccessor(values: Array[AnyRef], i: Int) extends FieldAccessor {
144     override def apply(): AnyRef = values(i)
145   }
146 
147   class GlobalFieldAccessor(value: AnyRef) extends FieldAccessor {
148     override def apply(): AnyRef = value
149   }
150 
151   /**
152    * Marker trait for resources that are dependent on the evaluation context state
153    *
154    * @tparam T type
155    */
156   trait ContextDependent[T <: ContextDependent[T]] {
157 
158     /**
159      * Return a copy of the instance tied to the given evaluation context.
160      *
161      * If the instance does not use the evaluation context, it should return itself instead of a copy
162      *
163      * @param ec evaluation context
164      * @return
165      */
166     def withContext(ec: EvaluationContext): T
167   }
168 
169   /**
170    * Evaluation error
171    *
172    * @param field field name that had an error
173    * @param line line number of the input being evaluated
174    * @param e error
175    */
176   case class EvaluationError(field: String, line: Long, e: Throwable)
177 
178   /**
179     * Evaluation context accessors
180     *
181     * @param ec context
182     */
183   implicit class RichEvaluationContext(val ec: EvaluationContext) extends AnyVal {
184     def getInputFilePath: Option[String] = Option(ec.accessor(InputFilePathKey).apply().asInstanceOf[String])
185   }
186 
187   /**
188     * Evaluation context implementation
189     *
190     * @param fields fields to evaluate, in topological dependency order
191     * @param globalValues global variable name/values
192     * @param cache enrichment caches
193     * @param metrics metrics
194     */
195   class StatefulEvaluationContext(
196       fields: Array[Field],
197       globalValues: Map[String, _ <: AnyRef],
198       val cache: Map[String, EnrichmentCache],
199       val metrics: ConverterMetrics,
200       val success: Counter,
201       val failure: Counter,
202       val errors: java.util.Queue[EvaluationError] = new java.util.ArrayDeque[EvaluationError]()
203     ) extends EvaluationContext {
204 
205     // holder for results from evaluating each row
206     private val values = Array.ofDim[AnyRef](fields.length)
207     // temp array for holding the arguments for a field
208     private val fieldArray = Array.ofDim[AnyRef](1)
209     // copy the transforms and tie them to the context
210     // note: the class isn't fully instantiated yet, but this statement is last in the initialization
211     private val transforms = fields.map(_.transforms.map(_.withContext(this)))
212 
213     override def accessor(name: String): FieldAccessor = {
214       val i = fields.indexWhere(_.name == name)
215       if (i >= 0) { new FieldValueAccessor(values, i) } else {
216         globalValues.get(name) match {
217           case Some(value) => new GlobalFieldAccessor(value)
218           case None => NullFieldAccessor
219         }
220       }
221     }
222 
223     override def evaluate(args: Array[AnyRef]): Either[EvaluationError, Array[AnyRef]] = {
224       var i = 0
225       // note: since fields are in topological order we don't need to clear them
226       while (i < values.length) {
227         values(i) = try {
228           val fieldArgs = fields(i).fieldArg match {
229             case None => args
230             case Some(f) => fieldArray(0) = f.apply(args); fieldArray
231           }
232           transforms(i) match {
233             case Some(t) => t.apply(fieldArgs)
234             case None    => fieldArgs(0)
235           }
236         } catch {
237           case NonFatal(e) => return Left(EvaluationError(fields(i).name, line, e))
238         }
239         i += 1
240       }
241       Right(values)
242     }
243   }
