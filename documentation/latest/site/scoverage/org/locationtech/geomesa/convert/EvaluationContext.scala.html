19 /**
20  * Holds the state associated with a conversion attempt. Evaluation contexts are not thread safe.
21  */
22 trait EvaluationContext {
23 
24   /**
25     * The current line being processed.
26     *
27     * This may be an actual line (e.g. a csv row), or a logical line (e.g. an avro record)
28     *
29     * @return
30     */
31   var line: Long = 0
32 
33   /**
34     * Enrichment caches
35     *
36     * @return
37     */
38   def cache: Map[String, EnrichmentCache]
39 
40   /**
41     * Metrics registry, accessible for tracking any custom values
42     *
43     * @return
44     */
45   def metrics: ConverterMetrics
46 
47   /**
48     * Counter for tracking successes
49     *
50     * @return
51     */
52   def success: com.codahale.metrics.Counter
53 
54   /**
55     * Counter for tracking failures
56     *
57     * @return
58     */
59   def failure: com.codahale.metrics.Counter
60 
61   /**
62    * Gets a references to a field's value
63    *
64    * @param name field name
65    * @return
66    */
67   def accessor(name: String): FieldAccessor
68 
69   /**
70    * Evaluate all values using the given arguments. The returned array may be mutated on subsequent calls to
71    * `evaluate`, so shouldn't be kept long-term
72    *
73    * @param args single row of input
74    */
75   def evaluate(args: Array[AnyRef]): Either[EvaluationError, Array[AnyRef]]
76 }
77 
78 object EvaluationContext extends LazyLogging {
79 
80   val InputFilePathKey = "inputFilePath"
81 
82   /**
83     * Creates a new, empty evaluation context
84     *
85     * @return
86     */
87   def empty: EvaluationContext = {
88     val metrics = ConverterMetrics.empty
89     val success = metrics.counter("success")
90     val failures = metrics.counter("failure")
91     new StatefulEvaluationContext(Array.empty, Map.empty, Map.empty, metrics, success, failures)
92   }
93 
94   /**
95    * Creates a new evaluation context with the given state
96    *
97    * @param fields converter fields, in topological dependency order
98    * @param globalValues global values
99    * @param caches enrichment caches
100    * @param metrics metrics
101    * @param success success counter
102    * @param failure failure counter
103    * @return
104    */
105   def apply(
106       fields: Seq[Field],
107       globalValues: Map[String, _ <: AnyRef],
108       caches: Map[String, EnrichmentCache],
109       metrics: ConverterMetrics,
110       success: Counter,
111       failure: Counter): EvaluationContext = {
112     new StatefulEvaluationContext(fields.toArray, globalValues, caches, metrics, success, failure)
113   }
114 
115   /**
116     * Gets a global parameter map containing the input file path
117     *
118     * @param file input file path
119     * @return
120     */
121   def inputFileParam(file: String): Map[String, AnyRef] = Map(InputFilePathKey -> file)
122 
123   /**
124    * Trait for reading a field from an evaluation context
125    */
126   sealed trait FieldAccessor {
127     def apply(): AnyRef
128   }
129 
130   case object NullFieldAccessor extends FieldAccessor {
131     override def apply(): AnyRef = null
132   }
133 
134   class FieldValueAccessor(values: Array[AnyRef], i: Int) extends FieldAccessor {
135     override def apply(): AnyRef = values(i)
136   }
137 
138   class GlobalFieldAccessor(value: AnyRef) extends FieldAccessor {
139     override def apply(): AnyRef = value
140   }
141 
142   /**
143    * Marker trait for resources that are dependent on the evaluation context state
144    *
145    * @tparam T type
146    */
147   trait ContextDependent[T <: ContextDependent[T]] {
148 
149     /**
150      * Return a copy of the instance tied to the given evaluation context.
151      *
152      * If the instance does not use the evaluation context, it should return itself instead of a copy
153      *
154      * @param ec evaluation context
155      * @return
156      */
157     def withContext(ec: EvaluationContext): T
158   }
159 
160   /**
161    * Evaluation error
162    *
163    * @param field field name that had an error
164    * @param line line number of the input being evaluated
165    * @param e error
166    */
167   case class EvaluationError(field: String, line: Long, e: Throwable)
168 
169   /**
170     * Evaluation context accessors
171     *
172     * @param ec context
173     */
174   implicit class RichEvaluationContext(val ec: EvaluationContext) extends AnyVal {
175     def getInputFilePath: Option[String] = Option(ec.accessor(InputFilePathKey).apply().asInstanceOf[String])
176   }
177 
178   /**
179     * Evaluation context implementation
180     *
181     * @param fields fields to evaluate, in topological dependency order
182     * @param globalValues global variable name/values
183     * @param cache enrichment caches
184     * @param metrics metrics
185     */
186   class StatefulEvaluationContext(
187       fields: Array[Field],
188       globalValues: Map[String, _ <: AnyRef],
189       val cache: Map[String, EnrichmentCache],
190       val metrics: ConverterMetrics,
191       val success: Counter,
192       val failure: Counter
193     ) extends EvaluationContext {
194 
195     // holder for results from evaluating each row
196     private val values = Array.ofDim[AnyRef](fields.length)
197     // temp array for holding the arguments for a field
198     private val fieldArray = Array.ofDim[AnyRef](1)
199     // copy the transforms and tie them to the context
200     // note: the class isn't fully instantiated yet, but this statement is last in the initialization
201     private val transforms = fields.map(_.transforms.map(_.withContext(this)))
202 
203     override def accessor(name: String): FieldAccessor = {
204       val i = fields.indexWhere(_.name == name)
205       if (i >= 0) { new FieldValueAccessor(values, i) } else {
206         globalValues.get(name) match {
207           case Some(value) => new GlobalFieldAccessor(value)
208           case None => NullFieldAccessor
209         }
210       }
211     }
212 
213     override def evaluate(args: Array[AnyRef]): Either[EvaluationError, Array[AnyRef]] = {
214       var i = 0
215       // note: since fields are in topological order we don't need to clear them
216       while (i < values.length) {
217         values(i) = try {
218           val fieldArgs = fields(i).fieldArg match {
219             case None => args
220             case Some(f) => fieldArray(0) = f.apply(args); fieldArray
221           }
222           transforms(i) match {
223             case Some(t) => t.apply(fieldArgs)
224             case None    => fieldArgs(0)
225           }
226         } catch {
227           case NonFatal(e) => return Left(EvaluationError(fields(i).name, line, e))
228         }
229         i += 1
230       }
231       Right(values)
232     }
233   }
