20 /**
21  * Holds the state associated with a conversion attempt. Evaluation contexts are not thread safe.
22  */
23 trait EvaluationContext {
24 
25   /**

37     * @return

39   def cache: Map[String, EnrichmentCache]

43     *
44     * @return
45     */
46   @deprecated("Use micrometer global registry for metrics")

49   /**

51     *
52     * @return

225       new Stats {

227         override def success(i: Int): Long = { delegate.success(i); dwSuccess.inc(i); dwSuccess.getCount }

232     /**
233      * Tracks success and failures in the conversion process. Counters are globally shared, so
234      * we track counts locally, while updating the global counters at the same time.
235      *
236      * Note that micrometer meters (including counters) will not actually store anything unless a registry has been configured.
237      *
238      * @param success success counter
239      * @param failure failure counter
240      */
241     private case class MicrometerStats(success: Counter, failure: Counter) extends Stats {
242 
243       private val localSuccessCount = new AtomicLong(0)
244       private val localFailureCount = new AtomicLong(0)
245 
246       override def success(i: Int = 1): Long = {
247         if (i > 0) {
248           success.increment(i)
249           localSuccessCount.addAndGet(i)
250         } else {
251           localSuccessCount.get()
252         }
253       }
254 
255       override def failure(i: Int = 1): Long = {
256         if (i > 0) {
257           failure.increment(i)
258           localFailureCount.addAndGet(i)
259         } else {
260           localFailureCount.get()
261         }
262       }
263     }
264   }

304   /**
305    * Evaluation context implementation
306    *
307    * @param fields fields to evaluate, in topological dependency order
308    * @param globalValues global variable name/values
309    * @param cache enrichment caches
310    * @param metrics deprecated metrics
311    * @param stats metrics
312    * @param errors error tracker
313    */
314   class StatefulEvaluationContext(
315       fields: Array[Field],
316       globalValues: Map[String, _ <: AnyRef],
317       val cache: Map[String, EnrichmentCache],
318       val metrics: ConverterMetrics,
319       val stats: Stats,
320       val errors: java.util.Queue[EvaluationError] = new java.util.ArrayDeque[EvaluationError]()
321     ) extends EvaluationContext {
322 
323     // holder for results from evaluating each row
324     private val values = Array.ofDim[AnyRef](fields.length)
325     // temp array for holding the arguments for a field
326     private val fieldArray = Array.ofDim[AnyRef](1)
327     // copy the transforms and tie them to the context
328     // note: the class isn't fully instantiated yet, but this statement is last in the initialization
329     private val transforms = fields.map(_.transforms.map(_.withContext(this)))
330 
331     override lazy val success: com.codahale.metrics.Counter = new com.codahale.metrics.Counter() {
332       override def inc(n: Long): Unit = stats.success(n.toInt)
333       override def getCount: Long = stats.success(0)
334     }
335     override lazy val failure: com.codahale.metrics.Counter = new com.codahale.metrics.Counter() {
336       override def inc(n: Long): Unit = stats.failure(n.toInt)
337       override def getCount: Long = stats.failure(0)
338     }
339 
340     override def accessor(name: String): FieldAccessor = {
341       val i = fields.indexWhere(_.name == name)
342       if (i >= 0) { new FieldValueAccessor(values, i) } else {
343         globalValues.get(name) match {
344           case Some(value) => new GlobalFieldAccessor(value)
345           case None => NullFieldAccessor
346         }
347       }
348     }
349 
350     override def evaluate(args: Array[AnyRef]): Either[EvaluationError, Array[AnyRef]] = {
351       var i = 0
352       // note: since fields are in topological order we don't need to clear them
353       while (i < values.length) {
354         values(i) = try {
355           val fieldArgs = fields(i).fieldArg match {
356             case None => args
357             case Some(f) => fieldArray(0) = f.apply(args); fieldArray
358           }
359           transforms(i) match {
360             case Some(t) => t.apply(fieldArgs)
361             case None    => fieldArgs(0)
362           }
363         } catch {
364           case NonFatal(e) => return Left(EvaluationError(fields(i).name, line, e))
365         }
366         i += 1
367       }
368       Right(values)
369     }
370 
371     override def withListener(listener: ContextListener): EvaluationContext =
372       new StatefulEvaluationContext(fields, globalValues, cache, metrics, StatListener(stats, listener))
373   }
374 
375   /**
376    * Stats implementation that adds a listener.
377    *
378    * Note that this implementation can chain additional listeners, but it's not very efficient doing so. The
379    * typical usage of listeners is reporting out counts, so usually there will only be one.
380    *
381    * @param delegate delegate stats
382    * @param listener listener
383    */
384   case class StatListener(delegate: Stats, listener: ContextListener) extends Stats {
385     override def success(i: Int): Long = { if (i > 0) { listener.onSuccess(i) }; delegate.success(i) }
386     override def failure(i: Int): Long = { if (i > 0) { listener.onFailure(i) }; delegate.failure(i) }
387   }
