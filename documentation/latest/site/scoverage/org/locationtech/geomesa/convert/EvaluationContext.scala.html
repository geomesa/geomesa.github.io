20 /**
21  * Holds the state associated with a conversion attempt. Evaluation contexts are not thread safe.
22  */
23 trait EvaluationContext {

229         override def success(i: Int): Long = { delegate.success(i); dwSuccess.inc(i); dwSuccess.getCount }

234     /**
235      * Tracks success and failures in the conversion process. Counters are globally shared, so
236      * we track counts locally, while updating the global counters at the same time.
237      *
238      * Note that micrometer meters (including counters) will not actually store anything unless a registry has been configured.
239      *
240      * @param success success counter
241      * @param failure failure counter
242      */
243     private case class MicrometerStats(success: Counter, failure: Counter) extends Stats {
244 
245       private val localSuccessCount = new AtomicLong(0)
246       private val localFailureCount = new AtomicLong(0)
247 
248       override def success(i: Int = 1): Long = {
249         if (i > 0) {
250           success.increment(i)
251           localSuccessCount.addAndGet(i)
252         } else {
253           localSuccessCount.get()
254         }
255       }
256 
257       override def failure(i: Int = 1): Long = {
258         if (i > 0) {
259           failure.increment(i)
260           localFailureCount.addAndGet(i)
261         } else {
262           localFailureCount.get()
263         }
264       }
265     }

306   /**
307    * Evaluation context implementation
308    *
309    * @param fields fields to evaluate, in topological dependency order
310    * @param globalValues global variable name/values
311    * @param cache enrichment caches
312    * @param metrics deprecated metrics
313    * @param stats metrics
314    * @param errors error tracker
315    */
316   class StatefulEvaluationContext(
317       fields: Array[Field],
318       globalValues: Map[String, _ <: AnyRef],
319       val cache: Map[String, EnrichmentCache],
320       val metrics: ConverterMetrics,
321       val stats: Stats,
322       val errors: java.util.Queue[EvaluationError] = new java.util.ArrayDeque[EvaluationError]()
323     ) extends EvaluationContext {
324 
325     // holder for results from evaluating each row
326     private val values = Array.ofDim[AnyRef](fields.length)
327     // temp array for holding the arguments for a field
328     private val fieldArray = Array.ofDim[AnyRef](1)
329     // copy the transforms and tie them to the context
330     // note: the class isn't fully instantiated yet, but this statement is last in the initialization
331     private val transforms = fields.map(_.transforms.map(_.withContext(this)))
332 
333     override lazy val success: com.codahale.metrics.Counter = new com.codahale.metrics.Counter() {
334       override def inc(n: Long): Unit = stats.success(n.toInt)
335       override def getCount: Long = stats.success(0)
336     }
337     override lazy val failure: com.codahale.metrics.Counter = new com.codahale.metrics.Counter() {
338       override def inc(n: Long): Unit = stats.failure(n.toInt)
339       override def getCount: Long = stats.failure(0)
340     }
341 
342     override def accessor(name: String): FieldAccessor = {
343       val i = fields.indexWhere(_.name == name)
344       if (i >= 0) { new FieldValueAccessor(values, i) } else {
345         globalValues.get(name) match {
346           case Some(value) => new GlobalFieldAccessor(value)
347           case None => NullFieldAccessor
348         }
349       }
350     }
351 
352     override def evaluate(args: Array[AnyRef]): Either[EvaluationError, Array[AnyRef]] = {
353       var i = 0
354       // note: since fields are in topological order we don't need to clear them
355       while (i < values.length) {
356         values(i) = try {
357           val fieldArgs = fields(i).fieldArg match {
358             case None => args
359             case Some(f) => fieldArray(0) = f.apply(args); fieldArray
360           }
361           transforms(i) match {
362             case Some(t) => t.apply(fieldArgs)
363             case None    => fieldArgs(0)
364           }
365         } catch {
366           case NonFatal(e) => return Left(EvaluationError(fields(i).name, line, e))
367         }
368         i += 1
369       }
370       Right(values)
371     }
372 
373     override def withListener(listener: ContextListener): EvaluationContext =
374       new StatefulEvaluationContext(fields, globalValues, cache, metrics, StatListener(stats, listener))
375   }
376 
377   /**
378    * Stats implementation that adds a listener.
379    *
380    * Note that this implementation can chain additional listeners, but it's not very efficient doing so. The
381    * typical usage of listeners is reporting out counts, so usually there will only be one.
382    *
383    * @param delegate delegate stats
384    * @param listener listener
385    */
386   case class StatListener(delegate: Stats, listener: ContextListener) extends Stats {
387     override def success(i: Int): Long = { if (i > 0) { listener.onSuccess(i) }; delegate.success(i) }
388     override def failure(i: Int): Long = { if (i > 0) { listener.onFailure(i) }; delegate.failure(i) }
389   }
