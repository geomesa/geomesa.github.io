41   /**
42     * Handles parquet files (including those produced by the FSDS and CLI export)
43     *
44     * @param is input

46     * @param path file path, if there is a file available
47     * @return

50       is: InputStream,
51       sft: Option[SimpleFeatureType],
52       path: Option[String]): Option[(SimpleFeatureType, Config)] =
53     infer(is, sft, path.fold(Map.empty[String, AnyRef])(EvaluationContext.inputFileParam)).toOption

55   override def infer(
56       is: InputStream,
57       sft: Option[SimpleFeatureType],
58       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = {
59     val path = hints.get(EvaluationContext.InputFilePathKey) match {
60       case None => Failure(new RuntimeException("No file path specified to the input data"))
61       case Some(p) => Success(p.toString)
62     }
63     path.flatMap { p =>
64       Try {
65         // note: get the path as a URI so that we handle local files appropriately
66         val filePath = new Path(PathUtils.getUrl(p).toURI)
67         val footer = ParquetFileReader.readFooter(new Configuration(), filePath, ParquetMetadataConverter.NO_FILTER)
68         val (schema, fields, id, userData) = SimpleFeatureParquetSchema.read(footer.getFileMetaData) match {
69           case Some(parquet) =>
70             // this is a geomesa encoded parquet file
71             val fields = parquet.sft.getAttributeDescriptors.asScala.map { descriptor =>
72               val name = parquet.field(parquet.sft.indexOf(descriptor.getLocalName))
73               // note: parquet converter stores the generic record under index 0
74               val path = s"avroPath($$0, '/$name')"
75               // some types need a function applied to the underlying avro value
76               val expression = ObjectType.selectType(descriptor) match {
77                 case Seq(ObjectType.GEOMETRY, ObjectType.POINT)           => s"parquetPoint($$0, '/$name')"
78                 case Seq(ObjectType.GEOMETRY, ObjectType.MULTIPOINT)      => s"parquetMultiPoint($$0, '/$name')"
79                 case Seq(ObjectType.GEOMETRY, ObjectType.LINESTRING)      => s"parquetLineString($$0, '/$name')"
80                 case Seq(ObjectType.GEOMETRY, ObjectType.MULTILINESTRING) => s"parquetMultiLineString($$0, '/$name')"
81                 case Seq(ObjectType.GEOMETRY, ObjectType.POLYGON)         => s"parquetPolygon($$0, '/$name')"
82                 case Seq(ObjectType.GEOMETRY, ObjectType.MULTIPOLYGON)    => s"parquetMultiPolygon($$0, '/$name')"
83                 case Seq(ObjectType.UUID)                                 => s"avroBinaryUuid($path)"
84                 case _                                                    => path
85               }
86               BasicField(descriptor.getLocalName, Some(Expression(expression)))
87             }
88             val id = Expression(s"avroPath($$0, '/${SimpleFeatureParquetSchema.FeatureIdField}')")
89             val userData =
90               if (parquet.hasVisibilities) {
91                 Map(SecurityUtils.FEATURE_VISIBILITY -> Expression(s"avroPath($$0, '/${SimpleFeatureParquetSchema.VisibilitiesField}')"))
92               } else {
93                 Map.empty[String, Expression]
94               }
95 
96             // validate the existing schema, if any
97             if (sft.exists(_.getAttributeDescriptors.asScala != parquet.sft.getAttributeDescriptors.asScala)) {
98               throw new IllegalArgumentException("Inferred schema does not match existing schema")
99             }
100             (parquet.sft, fields, Some(id), userData)
101 
102           case _ =>
103             // this is an arbitrary parquet file, create fields based on the schema
104             val types = ParquetConverterFactory.schemaTypes(footer.getFileMetaData.getSchema)
105             val dataSft = TypeInference.schema("inferred-parquet", types)
106             // note: parquet converter stores the generic record under index 0
107             val fields = types.map(t => BasicField(t.name, Some(Expression(t.transform.apply(0)))))
108 
109             // validate the existing schema, if any
110             sft.foreach(AbstractConverterFactory.validateInferredType(_, types.map(_.typed)))
111 
112             (dataSft, fields, None, Map.empty[String, Expression])
113         }
114 
115         val converterConfig = BasicConfig(typeToProcess, id, Map.empty, userData)
116 
117         val config = configConvert.to(converterConfig)
118             .withFallback(fieldConvert.to(fields.toSeq))
119             .withFallback(optsConvert.to(BasicOptions.default))
120             .toConfig
121 
122         (schema, config)
123       }
124     }
125   }
126 }

134   /**
135     * Take an avro schema and return the simple feature type bindings for it
136     *
137     * @param schema avro schema
138     * @return
139     */
140   private def schemaTypes(schema: MessageType): Seq[InferredType] = {
141     val namer = new Namer()
142     val types = scala.collection.mutable.ArrayBuffer.empty[InferredType]
143 
144     def mapField(field: Type, path: String = ""): Unit = {
145       // get a valid attribute name
146       val name = namer(field.getName)
147       val transform = FunctionTransform("avroPath(", s",'$path/${field.getName}')")
148 
149       val logical = field.getLogicalTypeAnnotation
150       if (field.isPrimitive) {
151         // note: date field transforms are handled by AvroReadSupport
152         lazy val intOrDate: InferredType = logical match {
153           case _: DateLogicalTypeAnnotation => InferredType(name, ObjectType.DATE, transform)
154           case _ => InferredType(name, ObjectType.INT, transform)
155         }
156         lazy val longOrTimestamp: InferredType = logical match {
157           case t: TimestampLogicalTypeAnnotation if t.getUnit == LogicalTypeAnnotation.TimeUnit.MILLIS => InferredType(name, ObjectType.DATE, transform)
158           case t: TimestampLogicalTypeAnnotation if t.getUnit == LogicalTypeAnnotation.TimeUnit.MICROS => InferredType(name, ObjectType.DATE, transform)
159           case _ => InferredType(name, ObjectType.LONG, transform)
160         }
161         lazy val stringOrBytes: InferredType = logical match {
162           case _: StringLogicalTypeAnnotation => InferredType(name, ObjectType.STRING, transform)
163           case _ => InferredType(name, ObjectType.BYTES, transform)
164         }
165         field.asPrimitiveType().getPrimitiveTypeName match {
166           case PrimitiveTypeName.BINARY               => types += stringOrBytes
167           case PrimitiveTypeName.INT32                => types += intOrDate
168           case PrimitiveTypeName.INT64                => types += longOrTimestamp
169           case PrimitiveTypeName.FLOAT                => types += InferredType(name, ObjectType.FLOAT, transform)
170           case PrimitiveTypeName.DOUBLE               => types += InferredType(name, ObjectType.DOUBLE, transform)
171           case PrimitiveTypeName.BOOLEAN              => types += InferredType(name, ObjectType.BOOLEAN, transform)
172           case PrimitiveTypeName.FIXED_LEN_BYTE_ARRAY => types += InferredType(name, ObjectType.BYTES, transform)
173           case _ => // no-op
174         }
175       } else {
176         val group = field.asGroupType()
177         logical match {
178           case _: ListLogicalTypeAnnotation =>
179             if (group.getFieldCount == 1 && !group.getType(0).isPrimitive) {
180               val list = group.getType(0).asGroupType()
181               if (list.getFieldCount == 1 && list.isRepetition(Repetition.REPEATED) && list.getType(0).isPrimitive) {
182                 types += InferredType(name, ObjectType.LIST, transform)
183               }
184             }
185 
186           case _: MapLogicalTypeAnnotation =>
187             if (group.getFieldCount == 1 && !group.getType(0).isPrimitive) {
188               val map = group.getType(0).asGroupType()
189               if (map.getFieldCount == 2 && map.isRepetition(Repetition.REPEATED) &&
190                   map.getFields.asScala.forall(_.isPrimitive)) {
191                 types += InferredType(name, ObjectType.MAP, transform)
192               }
193             }
194 
195           case _ =>
196             if (group.getFieldCount == 1 && group.getType(0).isPrimitive &&
197                 group.getType(0).isRepetition(Repetition.REPEATED)) {
198               types += InferredType(name, ObjectType.LIST, transform)
199             } else {
200               group.getFields.asScala.foreach(mapField(_, s"$path/${field.getName}"))
201             }
202         }
203       }
204     }
205 
206     schema.getFields.asScala.foreach(mapField(_))
207 
208     // check if we can derive a geometry field
209     TypeInference.deriveGeometry(types.toSeq, namer).foreach(g => types += g)
210 
211     types.toSeq
212   }
