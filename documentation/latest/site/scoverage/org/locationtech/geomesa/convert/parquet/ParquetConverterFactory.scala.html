45   /**
46     * Handles parquet files (including those produced by the FSDS and CLI export)
47     *
48     * @param is input
49     * @param sft simple feature type, if known ahead of time
50     * @param hints hints, including the file path, if there is a file available
51     * @return
52     */
53   override def infer(
54       is: InputStream,
55       sft: Option[SimpleFeatureType],
56       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = {
57     val path = hints.get(EvaluationContext.InputFilePathKey) match {
58       case None => Failure(new RuntimeException("No file path specified to the input data"))
59       case Some(p) => Success(p.toString)
60     }
61     path.flatMap { p =>
62       Try {
63         // note: get the path as a URI so that we handle local files appropriately
64         val filePath = new Path(PathUtils.getUrl(p).toURI)
65         val footer = ParquetFileReader.readFooter(new Configuration(), filePath, ParquetMetadataConverter.NO_FILTER)
66         val (schema, fields, id, userData) = SimpleFeatureParquetSchema.read(footer.getFileMetaData) match {
67           case Some(parquet) =>
68             // this is a geomesa encoded parquet file
69             val fields = parquet.sft.getAttributeDescriptors.asScala.map { descriptor =>
70               val name = parquet.field(parquet.sft.indexOf(descriptor.getLocalName))
71               // note: parquet converter stores the generic record under index 0
72               BasicField(descriptor.getLocalName, Some(Expression(s"avroPath($$0, '/$name')")))
73             }
74             val id = Expression(s"avroPath($$0, '/${SimpleFeatureParquetSchema.FeatureIdField}')")
75             val userData =
76               if (parquet.hasVisibilities) {
77                 Map("geomesa.feature.visibility" -> Expression(s"avroPath($$0, '/${SimpleFeatureParquetSchema.VisibilitiesField}')"))
78               } else {
79                 Map.empty[String, Expression]
80               }
81 
82             // validate the existing schema, if any
83             if (sft.exists(_.getAttributeDescriptors.asScala != parquet.sft.getAttributeDescriptors.asScala)) {
84               throw new IllegalArgumentException("Inferred schema does not match existing schema")
85             }
86             (parquet.sft, fields, Some(id), userData)
87 
88           case _ =>
89             // this is an arbitrary parquet file, create fields based on the schema
90             val types = ParquetConverterFactory.schemaTypes(footer.getFileMetaData)
91             val dataSft = TypeInference.schema("inferred-parquet", types)
92             // note: parquet converter stores the generic record under index 0
93             val fields = types.map(t => BasicField(t.name, Some(Expression(t.transform.apply(0)))))
94 
95             // validate the existing schema, if any
96             sft.foreach(AbstractConverterFactory.validateInferredType(_, types.map(_.typed)))
97 
98             (dataSft, fields, None, Map.empty[String, Expression])
99         }
100 
101         val converterConfig = BasicConfig(typeToProcess, None, id, Map.empty, userData)
102 
103         val config = configConvert.to(converterConfig)
104             .withFallback(fieldConvert.to(fields.toSeq))
105             .withFallback(optsConvert.to(BasicOptions.default))
106             .toConfig
107 
108         (schema, config)
109       }
110     }
111   }

120   /**
121    * Gets the type of elements in this list, assuming that this is a standard, 3-level parquet list type
122    *
123    * @param group list type
124    * @return element type, if input is a properly formatted 3-level parquet list
125    */
126   def getListElementType(group: GroupType): Option[GroupType] = {
127     for {
128       g <- Some(group) // optional group foo (LIST) {
129       if g.getFieldCount == 1 && !g.getType(0).isPrimitive
130       list = g.getType(0).asGroupType() // repeated group list {
131       if list.getFieldCount == 1 && list.isRepetition(Repetition.REPEATED) && !list.getType(0).isPrimitive
132     } yield {
133       list.getType(0).asGroupType() // optional group element {
134     }
135   }
136 
137   /**
138    * Gets the type of keys and values in this map, assuming that this is a standard, 3-level parquet map type
139    *
140    * @param map map type
141    * @return (key type, value type), if input is a properly formatted 3-level parquet map
142    */
143   def getMapKeyValueTypes(map: GroupType): Option[(Type, Type)] = {
144     for {
145       mapGroup <- Some(map)
146       if mapGroup.getFieldCount == 1
147       tuples = mapGroup.getType(0)
148       if !tuples.isPrimitive
149       tuplesGroup = tuples.asGroupType()
150       if tuplesGroup.getFieldCount == 2 && tuplesGroup.isRepetition(Repetition.REPEATED)
151     } yield {
152       (tuplesGroup.getType(0), tuplesGroup.getType(1))
153     }
154   }

156   /**
157     * Takes parquet file info returns the simple feature type bindings for it
158     *
159     * @param fileMetaData parquet file footer
160     * @return
161     */
162   private def schemaTypes(fileMetaData: FileMetaData): Seq[InferredType] = {
163     val namer = new Namer()
164     val geos: Option[GeoParquetMetadata] =
165       Option(fileMetaData.getKeyValueMetaData.get(GeoParquetMetadata.GeoParquetMetadataKey)).map(GeoParquetMetadata.fromJson)
166 
167     val types = fileMetaData.getSchema.getFields.asScala.flatMap(mapField(_, namer, geos)).toSeq
168     // check if we can derive a geometry field
169     val geom = TypeInference.deriveGeometry(types, namer)
170     types ++ geom
171   }

173   /**
174    * Infer a type from a top-level parquet field
175    *
176    * @param field field
177    * @param namer name helper
178    * @param geos extracted GeoParquet metadata from the parquet file
179    * @return
180    */
181   private def mapField(field: Type, namer: Namer, geos: Option[GeoParquetMetadata]): Seq[InferredType] = {
182     if (geos.exists(_.geometries.exists(_.covering.contains(field.getName)))) {
183       return Seq.empty // ignore bbox fields
184     }
185     // note: geometries *must* be at the root level, so we don't need to track them for nested fields
186     geos.flatMap(_.geometries.find(_.name == field.getName)) match {
187       case None => mapNormalField(field, namer)
188       case Some(geo) => mapGeoField(field, namer, geo)
189     }
190   }
191 
192   /**
193    * Maps a non-geometry type field
194    *
195    * @param field field
196    * @param namer name helper
197    * @param path avro path to the parent of the field
198    * @return
199    */
200   private def mapNormalField(field: Type, namer: Namer, path: String = ""): Seq[InferredType] = {
201     val fieldPath = s"$path/${field.getName}"
202     lazy val name = namer(fieldPath)
203     val transform = FunctionTransform("avroPath(", s",'$fieldPath')")
204     val logical = field.getLogicalTypeAnnotation
205     if (field.isPrimitive) {
206       // note: date field transforms are handled by AvroReadSupport
207       lazy val intOrDate: InferredType = logical match {
208         case _: DateLogicalTypeAnnotation => InferredType(name, ObjectType.DATE, transform)
209         case _ => InferredType(name, ObjectType.INT, transform)
210       }
211       lazy val longOrTimestamp: InferredType = logical match {
212         case t: TimestampLogicalTypeAnnotation if t.getUnit == LogicalTypeAnnotation.TimeUnit.MILLIS => InferredType(name, ObjectType.DATE, transform)
213         case t: TimestampLogicalTypeAnnotation if t.getUnit == LogicalTypeAnnotation.TimeUnit.MICROS => InferredType(name, ObjectType.DATE, transform)
214         case _ => InferredType(name, ObjectType.LONG, transform)
215       }
216       lazy val stringOrBytes: InferredType = logical match {
217         case _: StringLogicalTypeAnnotation => InferredType(name, ObjectType.STRING, transform)
218         case _ => InferredType(name, ObjectType.BYTES, transform)
219       }
220       field.asPrimitiveType().getPrimitiveTypeName match {
221         case PrimitiveTypeName.BINARY               => Seq(stringOrBytes)
222         case PrimitiveTypeName.INT32                => Seq(intOrDate)
223         case PrimitiveTypeName.INT64                => Seq(longOrTimestamp)
224         case PrimitiveTypeName.FLOAT                => Seq(InferredType(name, ObjectType.FLOAT, transform))
225         case PrimitiveTypeName.DOUBLE               => Seq(InferredType(name, ObjectType.DOUBLE, transform))
226         case PrimitiveTypeName.BOOLEAN              => Seq(InferredType(name, ObjectType.BOOLEAN, transform))
227         case PrimitiveTypeName.FIXED_LEN_BYTE_ARRAY => Seq(InferredType(name, ObjectType.BYTES, transform))
228         case _ => Seq.empty
229       }
230     } else {
231       val group = field.asGroupType()
232       def jsonTransform(): FunctionTransform = FunctionTransform(s"avroToJson(${transform.prefix}", s"${transform.suffix})")
233       logical match {
234         case _: ListLogicalTypeAnnotation =>
235           if (getListElementType(group).exists(e => e.getFieldCount == 1 && e.getType(0).isPrimitive)) {
236             Seq(InferredType(name, ObjectType.LIST, transform))
237           } else {
238             // for non-standard or complex-element lists, we just convert to a json string
239             Seq(InferredType(name, ObjectType.JSON, jsonTransform()))
240           }
241 
242         case _: MapLogicalTypeAnnotation =>
243           if (getMapKeyValueTypes(group).exists { case (k, v) => k.isPrimitive && v.isPrimitive }) {
244             Seq(InferredType(name, ObjectType.MAP, transform))
245           } else {
246             // for non-standard or complex-value maps, we just convert to a json string
247             Seq(InferredType(name, ObjectType.JSON, jsonTransform()))
248           }
249 
250         case _ =>
251           group.getFields.asScala.flatMap(mapNormalField(_, namer, fieldPath)).toSeq
252       }
253     }
254   }

256   /**
257    * Maps a GeoParquet geometry field
258    *
259    * @param field field
260    * @param namer name helper
261    * @param geo GeoParquet metadata
262    * @return
263    */
264   private def mapGeoField(field: Type, namer: Namer, geo: ColumnMetadata): Seq[InferredType] = {
265     lazy val name = namer(field.getName)
266 
267     def geomTransform(fn: String): FunctionTransform = FunctionTransform(s"$fn(avroPath(", s",'/${field.getName}'))")
268 
269     def mismatch(): Seq[InferredType] = {
270       logger.warn(s"Found GeoParquet metadata for column '${field.getName}' but the schema does not match the expected type")
271       mapNormalField(field, namer)
272     }
273 
274     if (geo.encoding == GeoParquetColumnEncoding.WKB) {
275       if (isWkbType(field)) {
276         Seq(InferredType(name, GeoParquetColumnType.toObjectType(geo.types), geomTransform("geometry")))
277       } else {
278         mismatch()
279       }
280     } else {
281       val objectType = GeoParquetColumnEncoding.toObjectType(geo.encoding)
282       if ((geo.types.isEmpty || GeoParquetColumnType.toObjectType(geo.types) == objectType) && isNativeType(field, objectType)) {
283         Seq(InferredType(name, objectType, geomTransform(geo.encoding.toString)))
284       } else {
285         mismatch()
286       }
287     }
288   }

290   /**
291    * Checks if a field corresponds to the GeoParquet schema for a WKB-encoded geometry
292    *
293    * @param field field
294    * @return
295    */
296   private def isWkbType(field: Type): Boolean =
297     !field.isRepetition(Repetition.REPEATED) && field.isPrimitive &&
298       field.asPrimitiveType().getPrimitiveTypeName == PrimitiveTypeName.BINARY
299 
300   /**
301    * Checks if the field corresponds to the GeoParquet schema for a native-encoded geometry
302    *
303    * @param field field
304    * @param objectType native type
305    * @return
306    */
307   private def isNativeType(field: Type, objectType: ObjectType): Boolean = {
308     // compare child fields, so we don't consider optional vs required, id, name, logicalTypes, etc
309     !field.isRepetition(Repetition.REPEATED) &&
310       field.asGroupType().getFields == GeometrySchema(objectType, GeoParquetNative).named(field.getName).asGroupType().getFields
311   }
