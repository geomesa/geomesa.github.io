41   /**
42     * Handles parquet files (including those produced by the FSDS and CLI export)
43     *
44     * @param is input
45     * @param sft simple feature type, if known ahead of time
46     * @param path file path, if there is a file available
47     * @return
48     */
49   override def infer(
50       is: InputStream,
51       sft: Option[SimpleFeatureType],
52       path: Option[String]): Option[(SimpleFeatureType, Config)] =
53     infer(is, sft, path.fold(Map.empty[String, AnyRef])(EvaluationContext.inputFileParam)).toOption

55   override def infer(
56       is: InputStream,
57       sft: Option[SimpleFeatureType],
58       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = {
59     val path = hints.get(EvaluationContext.InputFilePathKey) match {
60       case None => Failure(new RuntimeException("No file path specified to the input data"))
61       case Some(p) => Success(p.toString)
62     }
63     path.flatMap { p =>
64       Try {
65         // note: get the path as a URI so that we handle local files appropriately
66         val filePath = new Path(PathUtils.getUrl(p).toURI)
67         val footer = ParquetFileReader.readFooter(new Configuration(), filePath, ParquetMetadataConverter.NO_FILTER)
68         val (schema, fields, id, userData) = SimpleFeatureParquetSchema.read(footer.getFileMetaData) match {
69           case Some(parquet) =>
70             // this is a geomesa encoded parquet file
71             val fields = parquet.sft.getAttributeDescriptors.asScala.map { descriptor =>
72               val name = parquet.field(parquet.sft.indexOf(descriptor.getLocalName))
73               // note: parquet converter stores the generic record under index 0
74               BasicField(descriptor.getLocalName, Some(Expression(s"avroPath($$0, '/$name')")))
75             }
76             val id = Expression(s"avroPath($$0, '/${SimpleFeatureParquetSchema.FeatureIdField}')")
77             val userData =
78               if (parquet.hasVisibilities) {
79                 Map(SecurityUtils.FEATURE_VISIBILITY -> Expression(s"avroPath($$0, '/${SimpleFeatureParquetSchema.VisibilitiesField}')"))
80               } else {
81                 Map.empty[String, Expression]
82               }
83 
84             // validate the existing schema, if any
85             if (sft.exists(_.getAttributeDescriptors.asScala != parquet.sft.getAttributeDescriptors.asScala)) {
86               throw new IllegalArgumentException("Inferred schema does not match existing schema")
87             }
88             (parquet.sft, fields, Some(id), userData)
89 
90           case _ =>
91             // this is an arbitrary parquet file, create fields based on the schema
92             val types = ParquetConverterFactory.schemaTypes(footer.getFileMetaData.getSchema)
93             val dataSft = TypeInference.schema("inferred-parquet", types)
94             // note: parquet converter stores the generic record under index 0
95             val fields = types.map(t => BasicField(t.name, Some(Expression(t.transform.apply(0)))))
96 
97             // validate the existing schema, if any
98             sft.foreach(AbstractConverterFactory.validateInferredType(_, types.map(_.typed)))
99 
100             (dataSft, fields, None, Map.empty[String, Expression])
101         }
102 
103         val converterConfig = BasicConfig(typeToProcess, id, Map.empty, userData)
104 
105         val config = configConvert.to(converterConfig)
106             .withFallback(fieldConvert.to(fields.toSeq))
107             .withFallback(optsConvert.to(BasicOptions.default))
108             .toConfig
109 
110         (schema, config)
111       }
112     }
113   }
114 }

122   /**
123     * Take an avro schema and return the simple feature type bindings for it
124     *
125     * @param schema avro schema
126     * @return
127     */
128   private def schemaTypes(schema: MessageType): Seq[InferredType] = {
129     val namer = new Namer()
130     val types = scala.collection.mutable.ArrayBuffer.empty[InferredType]
131 
132     def mapField(field: Type, path: String = ""): Unit = {
133       // get a valid attribute name
134       val name = namer(field.getName)
135       val transform = FunctionTransform("avroPath(", s",'$path/${field.getName}')")
136 
137       val logical = field.getLogicalTypeAnnotation
138       if (field.isPrimitive) {
139         // note: date field transforms are handled by AvroReadSupport
140         lazy val intOrDate: InferredType = logical match {
141           case _: DateLogicalTypeAnnotation => InferredType(name, ObjectType.DATE, transform)
142           case _ => InferredType(name, ObjectType.INT, transform)
143         }
144         lazy val longOrTimestamp: InferredType = logical match {
145           case t: TimestampLogicalTypeAnnotation if t.getUnit == LogicalTypeAnnotation.TimeUnit.MILLIS => InferredType(name, ObjectType.DATE, transform)
146           case t: TimestampLogicalTypeAnnotation if t.getUnit == LogicalTypeAnnotation.TimeUnit.MICROS => InferredType(name, ObjectType.DATE, transform)
147           case _ => InferredType(name, ObjectType.LONG, transform)
148         }
149         lazy val stringOrBytes: InferredType = logical match {
150           case _: StringLogicalTypeAnnotation => InferredType(name, ObjectType.STRING, transform)
151           case _ => InferredType(name, ObjectType.BYTES, transform)
152         }
153         field.asPrimitiveType().getPrimitiveTypeName match {
154           case PrimitiveTypeName.BINARY               => types += stringOrBytes
155           case PrimitiveTypeName.INT32                => types += intOrDate
156           case PrimitiveTypeName.INT64                => types += longOrTimestamp
157           case PrimitiveTypeName.FLOAT                => types += InferredType(name, ObjectType.FLOAT, transform)
158           case PrimitiveTypeName.DOUBLE               => types += InferredType(name, ObjectType.DOUBLE, transform)
159           case PrimitiveTypeName.BOOLEAN              => types += InferredType(name, ObjectType.BOOLEAN, transform)
160           case PrimitiveTypeName.FIXED_LEN_BYTE_ARRAY => types += InferredType(name, ObjectType.BYTES, transform)
161           case _ => // no-op
162         }
163       } else {
164         val group = field.asGroupType()
165         logical match {
166           case _: ListLogicalTypeAnnotation =>
167             if (group.getFieldCount == 1 && !group.getType(0).isPrimitive) {
168               val list = group.getType(0).asGroupType()
169               if (list.getFieldCount == 1 && list.isRepetition(Repetition.REPEATED) && list.getType(0).isPrimitive) {
170                 types += InferredType(name, ObjectType.LIST, transform)
171               }
172             }
173 
174           case _: MapLogicalTypeAnnotation =>
175             if (group.getFieldCount == 1 && !group.getType(0).isPrimitive) {
176               val map = group.getType(0).asGroupType()
177               if (map.getFieldCount == 2 && map.isRepetition(Repetition.REPEATED) &&
178                   map.getFields.asScala.forall(_.isPrimitive)) {
179                 types += InferredType(name, ObjectType.MAP, transform)
180               }
181             }
182 
183           case _ =>
184             if (group.getFieldCount == 1 && group.getType(0).isPrimitive &&
185                 group.getType(0).isRepetition(Repetition.REPEATED)) {
186               types += InferredType(name, ObjectType.LIST, transform)
187             } else {
188               group.getFields.asScala.foreach(mapField(_, s"$path/${field.getName}"))
189             }
190         }
191       }
192     }
193 
194     schema.getFields.asScala.foreach(mapField(_))
195 
196     // check if we can derive a geometry field
197     TypeInference.deriveGeometry(types.toSeq, namer).foreach(g => types += g)
198 
199     types.toSeq
200   }
