59   override def infer(
60       is: InputStream,
61       sft: Option[SimpleFeatureType],
62       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = {
63     val path = hints.get(EvaluationContext.InputFilePathKey) match {
64       case None => Failure(new RuntimeException("No file path specified to the input data"))
65       case Some(p) => Success(p.toString)
66     }
67     path.flatMap { p =>
68       Try {
69         // note: get the path as a URI so that we handle local files appropriately
70         val filePath = new Path(PathUtils.getUrl(p).toURI)
71         val footer = ParquetFileReader.readFooter(new Configuration(), filePath, ParquetMetadataConverter.NO_FILTER)
72         val (schema, fields, id, userData) = SimpleFeatureParquetSchema.read(footer.getFileMetaData) match {
73           case Some(parquet) =>
74             // this is a geomesa encoded parquet file
75             val fields = parquet.sft.getAttributeDescriptors.asScala.map { descriptor =>
76               val name = parquet.field(parquet.sft.indexOf(descriptor.getLocalName))
77               // note: parquet converter stores the generic record under index 0
78               BasicField(descriptor.getLocalName, Some(Expression(s"avroPath($$0, '/$name')")))
79             }
80             val id = Expression(s"avroPath($$0, '/${SimpleFeatureParquetSchema.FeatureIdField}')")
81             val userData =
82               if (parquet.hasVisibilities) {
83                 Map("geomesa.feature.visibility" -> Expression(s"avroPath($$0, '/${SimpleFeatureParquetSchema.VisibilitiesField}')"))
84               } else {
85                 Map.empty[String, Expression]
86               }
87 
88             // validate the existing schema, if any
89             if (sft.exists(_.getAttributeDescriptors.asScala != parquet.sft.getAttributeDescriptors.asScala)) {
90               throw new IllegalArgumentException("Inferred schema does not match existing schema")
91             }
92             (parquet.sft, fields, Some(id), userData)
93 
94           case _ =>
95             // this is an arbitrary parquet file, create fields based on the schema
96             val types = ParquetConverterFactory.schemaTypes(footer.getFileMetaData)
97             val dataSft = TypeInference.schema("inferred-parquet", types)
98             // note: parquet converter stores the generic record under index 0
99             val fields = types.map(t => BasicField(t.name, Some(Expression(t.transform.apply(0)))))
100 
101             // validate the existing schema, if any
102             sft.foreach(AbstractConverterFactory.validateInferredType(_, types.map(_.typed)))
103 
104             (dataSft, fields, None, Map.empty[String, Expression])
105         }
106 
107         val converterConfig = BasicConfig(typeToProcess, None, id, Map.empty, userData)
108 
109         val config = configConvert.to(converterConfig)
110             .withFallback(fieldConvert.to(fields.toSeq))
111             .withFallback(optsConvert.to(BasicOptions.default))
112             .toConfig
113 
114         (schema, config)
115       }
116     }
117   }

126   /**
127    * Gets the type of elements in this list, assuming that this is a standard, 3-level parquet list type
128    *
129    * @param group list type
130    * @return element type, if input is a properly formatted 3-level parquet list
131    */
132   def getListElementType(group: GroupType): Option[GroupType] = {
133     for {
134       g <- Some(group) // optional group foo (LIST) {
135       if g.getFieldCount == 1 && !g.getType(0).isPrimitive
136       list = g.getType(0).asGroupType() // repeated group list {
137       if list.getFieldCount == 1 && list.isRepetition(Repetition.REPEATED) && !list.getType(0).isPrimitive
138     } yield {
139       list.getType(0).asGroupType() // optional group element {
140     }
141   }
142 
143   /**
144    * Gets the type of keys and values in this map, assuming that this is a standard, 3-level parquet map type
145    *
146    * @param map map type
147    * @return (key type, value type), if input is a properly formatted 3-level parquet map
148    */
149   def getMapKeyValueTypes(map: GroupType): Option[(Type, Type)] = {
150     for {
151       mapGroup <- Some(map)
152       if mapGroup.getFieldCount == 1
153       tuples = mapGroup.getType(0)
154       if !tuples.isPrimitive
155       tuplesGroup = tuples.asGroupType()
156       if tuplesGroup.getFieldCount == 2 && tuplesGroup.isRepetition(Repetition.REPEATED)
157     } yield {
158       (tuplesGroup.getType(0), tuplesGroup.getType(1))
159     }
160   }

179   /**
180    * Infer a type from a top-level parquet field
181    *
182    * @param field field
183    * @param namer name helper
184    * @param geos extracted GeoParquet metadata from the parquet file
185    * @return
186    */
187   private def mapField(field: Type, namer: Namer, geos: Option[GeoParquetMetadata]): Seq[InferredType] = {
188     if (geos.exists(_.geometries.exists(_.covering.contains(field.getName)))) {
189       return Seq.empty // ignore bbox fields
190     }
191     // note: geometries *must* be at the root level, so we don't need to track them for nested fields
192     geos.flatMap(_.geometries.find(_.name == field.getName)) match {
193       case None => mapNormalField(field, namer)
194       case Some(geo) => mapGeoField(field, namer, geo)
195     }
196   }

198   /**
199    * Maps a non-geometry type field
200    *
201    * @param field field
202    * @param namer name helper
203    * @param path avro path to the parent of the field
204    * @return
205    */
206   private def mapNormalField(field: Type, namer: Namer, path: String = ""): Seq[InferredType] = {
207     val fieldPath = s"$path/${field.getName}"
208     lazy val name = namer(fieldPath)
209     val transform = FunctionTransform("avroPath(", s",'$fieldPath')")
210     val logical = field.getLogicalTypeAnnotation
211     if (field.isPrimitive) {
212       // note: date field transforms are handled by AvroReadSupport
213       lazy val intOrDate: InferredType = logical match {
214         case _: DateLogicalTypeAnnotation => InferredType(name, ObjectType.DATE, transform)
215         case _ => InferredType(name, ObjectType.INT, transform)
216       }
217       lazy val longOrTimestamp: InferredType = logical match {
218         case t: TimestampLogicalTypeAnnotation if t.getUnit == LogicalTypeAnnotation.TimeUnit.MILLIS => InferredType(name, ObjectType.DATE, transform)
219         case t: TimestampLogicalTypeAnnotation if t.getUnit == LogicalTypeAnnotation.TimeUnit.MICROS => InferredType(name, ObjectType.DATE, transform)
220         case _ => InferredType(name, ObjectType.LONG, transform)
221       }
222       lazy val stringOrBytes: InferredType = logical match {
223         case _: StringLogicalTypeAnnotation => InferredType(name, ObjectType.STRING, transform)
224         case _ => InferredType(name, ObjectType.BYTES, transform)
225       }
226       field.asPrimitiveType().getPrimitiveTypeName match {
227         case PrimitiveTypeName.BINARY               => Seq(stringOrBytes)
228         case PrimitiveTypeName.INT32                => Seq(intOrDate)
229         case PrimitiveTypeName.INT64                => Seq(longOrTimestamp)
230         case PrimitiveTypeName.FLOAT                => Seq(InferredType(name, ObjectType.FLOAT, transform))
231         case PrimitiveTypeName.DOUBLE               => Seq(InferredType(name, ObjectType.DOUBLE, transform))
232         case PrimitiveTypeName.BOOLEAN              => Seq(InferredType(name, ObjectType.BOOLEAN, transform))
233         case PrimitiveTypeName.FIXED_LEN_BYTE_ARRAY => Seq(InferredType(name, ObjectType.BYTES, transform))
234         case _ => Seq.empty
235       }
236     } else {
237       val group = field.asGroupType()
238       def jsonTransform(): FunctionTransform = FunctionTransform(s"avroToJson(${transform.prefix}", s"${transform.suffix})")
239       logical match {
240         case _: ListLogicalTypeAnnotation =>
241           if (getListElementType(group).exists(e => e.getFieldCount == 1 && e.getType(0).isPrimitive)) {
242             Seq(InferredType(name, ObjectType.LIST, transform))
243           } else {
244             // for non-standard or complex-element lists, we just convert to a json string
245             Seq(InferredType(name, ObjectType.JSON, jsonTransform()))
246           }
247 
248         case _: MapLogicalTypeAnnotation =>
249           if (getMapKeyValueTypes(group).exists { case (k, v) => k.isPrimitive && v.isPrimitive }) {
250             Seq(InferredType(name, ObjectType.MAP, transform))
251           } else {
252             // for non-standard or complex-value maps, we just convert to a json string
253             Seq(InferredType(name, ObjectType.JSON, jsonTransform()))
254           }
255 
256         case _ =>
257           group.getFields.asScala.flatMap(mapNormalField(_, namer, fieldPath)).toSeq
258       }
259     }
260   }

262   /**
263    * Maps a GeoParquet geometry field
264    *
265    * @param field field
266    * @param namer name helper
267    * @param geo GeoParquet metadata
268    * @return
269    */
270   private def mapGeoField(field: Type, namer: Namer, geo: ColumnMetadata): Seq[InferredType] = {
271     lazy val name = namer(field.getName)
272 
273     def geomTransform(fn: String): FunctionTransform = FunctionTransform(s"$fn(avroPath(", s",'/${field.getName}'))")
274 
275     def mismatch(): Seq[InferredType] = {
276       logger.warn(s"Found GeoParquet metadata for column '${field.getName}' but the schema does not match the expected type")
277       mapNormalField(field, namer)
278     }
279 
280     if (geo.encoding == GeoParquetColumnEncoding.WKB) {
281       if (isWkbType(field)) {
282         Seq(InferredType(name, GeoParquetColumnType.toObjectType(geo.types), geomTransform("geometry")))
283       } else {
284         mismatch()
285       }
286     } else {
287       val objectType = GeoParquetColumnEncoding.toObjectType(geo.encoding)
288       if ((geo.types.isEmpty || GeoParquetColumnType.toObjectType(geo.types) == objectType) && isNativeType(field, objectType)) {
289         Seq(InferredType(name, objectType, geomTransform(geo.encoding.toString)))
290       } else {
291         mismatch()
292       }
293     }
294   }

296   /**
297    * Checks if a field corresponds to the GeoParquet schema for a WKB-encoded geometry
298    *
299    * @param field field
300    * @return
301    */
302   private def isWkbType(field: Type): Boolean =
303     !field.isRepetition(Repetition.REPEATED) && field.isPrimitive &&
304       field.asPrimitiveType().getPrimitiveTypeName == PrimitiveTypeName.BINARY

306   /**
307    * Checks if the field corresponds to the GeoParquet schema for a native-encoded geometry
308    *
309    * @param field field
310    * @param objectType native type
311    * @return
312    */
313   private def isNativeType(field: Type, objectType: ObjectType): Boolean = {
314     // compare child fields, so we don't consider optional vs required, id, name, logicalTypes, etc
315     !field.isRepetition(Repetition.REPEATED) &&
316       field.asGroupType().getFields == GeometrySchema(objectType, GeoParquetNative).named(field.getName).asGroupType().getFields
317   }
