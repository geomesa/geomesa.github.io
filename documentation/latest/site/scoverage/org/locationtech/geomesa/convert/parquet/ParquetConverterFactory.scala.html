60   override def infer(
61       is: InputStream,
62       sft: Option[SimpleFeatureType],
63       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = {
64     val path = hints.get(EvaluationContext.InputFilePathKey) match {
65       case None => Failure(new RuntimeException("No file path specified to the input data"))
66       case Some(p) => Success(p.toString)
67     }
68     path.flatMap { p =>
69       Try {
70         // note: get the path as a URI so that we handle local files appropriately
71         val filePath = new Path(PathUtils.getUrl(p).toURI)
72         val footer = ParquetFileReader.readFooter(new Configuration(), filePath, ParquetMetadataConverter.NO_FILTER)
73         val (schema, fields, id, userData) = SimpleFeatureParquetSchema.read(footer.getFileMetaData) match {
74           case Some(parquet) =>
75             // this is a geomesa encoded parquet file
76             val fields = parquet.sft.getAttributeDescriptors.asScala.map { descriptor =>
77               val name = parquet.field(parquet.sft.indexOf(descriptor.getLocalName))
78               // note: parquet converter stores the generic record under index 0
79               BasicField(descriptor.getLocalName, Some(Expression(s"avroPath($$0, '/$name')")))
80             }
81             val id = Expression(s"avroPath($$0, '/${SimpleFeatureParquetSchema.FeatureIdField}')")
82             val userData =
83               if (parquet.hasVisibilities) {
84                 Map(SecurityUtils.FEATURE_VISIBILITY -> Expression(s"avroPath($$0, '/${SimpleFeatureParquetSchema.VisibilitiesField}')"))
85               } else {
86                 Map.empty[String, Expression]
87               }
88 
89             // validate the existing schema, if any
90             if (sft.exists(_.getAttributeDescriptors.asScala != parquet.sft.getAttributeDescriptors.asScala)) {
91               throw new IllegalArgumentException("Inferred schema does not match existing schema")
92             }
93             (parquet.sft, fields, Some(id), userData)
94 
95           case _ =>
96             // this is an arbitrary parquet file, create fields based on the schema
97             val types = ParquetConverterFactory.schemaTypes(footer.getFileMetaData)
98             val dataSft = TypeInference.schema("inferred-parquet", types)
99             // note: parquet converter stores the generic record under index 0
100             val fields = types.map(t => BasicField(t.name, Some(Expression(t.transform.apply(0)))))
101 
102             // validate the existing schema, if any
103             sft.foreach(AbstractConverterFactory.validateInferredType(_, types.map(_.typed)))
104 
105             (dataSft, fields, None, Map.empty[String, Expression])
106         }
107 
108         val converterConfig = BasicConfig(typeToProcess, None, id, Map.empty, userData)
109 
110         val config = configConvert.to(converterConfig)
111             .withFallback(fieldConvert.to(fields.toSeq))
112             .withFallback(optsConvert.to(BasicOptions.default))
113             .toConfig
114 
115         (schema, config)
116       }
117     }
118   }

127   /**
128    * Gets the type of elements in this list, assuming that this is a standard, 3-level parquet list type
129    *
130    * @param group list type
131    * @return element type, if input is a properly formatted 3-level parquet list
132    */
133   def getListElementType(group: GroupType): Option[GroupType] = {
134     for {
135       g <- Some(group) // optional group foo (LIST) {
136       if g.getFieldCount == 1 && !g.getType(0).isPrimitive
137       list = g.getType(0).asGroupType() // repeated group list {
138       if list.getFieldCount == 1 && list.isRepetition(Repetition.REPEATED) && !list.getType(0).isPrimitive
139     } yield {
140       list.getType(0).asGroupType() // optional group element {
141     }
142   }
143 
144   /**
145    * Gets the type of keys and values in this map, assuming that this is a standard, 3-level parquet map type
146    *
147    * @param map map type
148    * @return (key type, value type), if input is a properly formatted 3-level parquet map
149    */
150   def getMapKeyValueTypes(map: GroupType): Option[(Type, Type)] = {
151     for {
152       mapGroup <- Some(map)
153       if mapGroup.getFieldCount == 1
154       tuples = mapGroup.getType(0)
155       if !tuples.isPrimitive
156       tuplesGroup = tuples.asGroupType()
157       if tuplesGroup.getFieldCount == 2 && tuplesGroup.isRepetition(Repetition.REPEATED)
158     } yield {
159       (tuplesGroup.getType(0), tuplesGroup.getType(1))
160     }
161   }

164     * Takes parquet file info returns the simple feature type bindings for it
165     *
166     * @param fileMetaData parquet file footer
167     * @return

169   private def schemaTypes(fileMetaData: FileMetaData): Seq[InferredType] = {
170     val namer = new Namer()
171     val geos: Option[GeoParquetMetadata] =
172       Option(fileMetaData.getKeyValueMetaData.get(GeoParquetMetadata.GeoParquetMetadataKey)).map(GeoParquetMetadata.fromJson)

174     val types = fileMetaData.getSchema.getFields.asScala.flatMap(mapField(_, namer, geos)).toSeq
175     // check if we can derive a geometry field
176     val geom = TypeInference.deriveGeometry(types, namer)
177     types ++ geom

180   /**
181    * Infer a type from a top-level parquet field
182    *
183    * @param field field
184    * @param namer name helper
185    * @param geos extracted GeoParquet metadata from the parquet file
186    * @return
187    */
188   private def mapField(field: Type, namer: Namer, geos: Option[GeoParquetMetadata]): Seq[InferredType] = {
189     if (geos.exists(_.geometries.exists(_.covering.contains(field.getName)))) {
190       return Seq.empty // ignore bbox fields
191     }
192     // note: geometries *must* be at the root level, so we don't need to track them for nested fields
193     geos.flatMap(_.geometries.find(_.name == field.getName)) match {
194       case None => mapNormalField(field, namer)
195       case Some(geo) => mapGeoField(field, namer, geo)
196     }
197   }

199   /**
200    * Maps a non-geometry type field
201    *
202    * @param field field
203    * @param namer name helper
204    * @param path avro path to the parent of the field
205    * @return
206    */
207   private def mapNormalField(field: Type, namer: Namer, path: String = ""): Seq[InferredType] = {
208     val fieldPath = s"$path/${field.getName}"
209     lazy val name = namer(fieldPath)
210     val transform = FunctionTransform("avroPath(", s",'$fieldPath')")
211     val logical = field.getLogicalTypeAnnotation
212     if (field.isPrimitive) {
213       // note: date field transforms are handled by AvroReadSupport
214       lazy val intOrDate: InferredType = logical match {
215         case _: DateLogicalTypeAnnotation => InferredType(name, ObjectType.DATE, transform)
216         case _ => InferredType(name, ObjectType.INT, transform)
217       }
218       lazy val longOrTimestamp: InferredType = logical match {
219         case t: TimestampLogicalTypeAnnotation if t.getUnit == LogicalTypeAnnotation.TimeUnit.MILLIS => InferredType(name, ObjectType.DATE, transform)
220         case t: TimestampLogicalTypeAnnotation if t.getUnit == LogicalTypeAnnotation.TimeUnit.MICROS => InferredType(name, ObjectType.DATE, transform)
221         case _ => InferredType(name, ObjectType.LONG, transform)
222       }
223       lazy val stringOrBytes: InferredType = logical match {
224         case _: StringLogicalTypeAnnotation => InferredType(name, ObjectType.STRING, transform)
225         case _ => InferredType(name, ObjectType.BYTES, transform)
226       }
227       field.asPrimitiveType().getPrimitiveTypeName match {
228         case PrimitiveTypeName.BINARY               => Seq(stringOrBytes)
229         case PrimitiveTypeName.INT32                => Seq(intOrDate)
230         case PrimitiveTypeName.INT64                => Seq(longOrTimestamp)
231         case PrimitiveTypeName.FLOAT                => Seq(InferredType(name, ObjectType.FLOAT, transform))
232         case PrimitiveTypeName.DOUBLE               => Seq(InferredType(name, ObjectType.DOUBLE, transform))
233         case PrimitiveTypeName.BOOLEAN              => Seq(InferredType(name, ObjectType.BOOLEAN, transform))
234         case PrimitiveTypeName.FIXED_LEN_BYTE_ARRAY => Seq(InferredType(name, ObjectType.BYTES, transform))
235         case _ => Seq.empty
236       }
237     } else {
238       val group = field.asGroupType()
239       def jsonTransform(): FunctionTransform = FunctionTransform(s"avroToJson(${transform.prefix}", s"${transform.suffix})")
240       logical match {
241         case _: ListLogicalTypeAnnotation =>
242           if (getListElementType(group).exists(e => e.getFieldCount == 1 && e.getType(0).isPrimitive)) {
243             Seq(InferredType(name, ObjectType.LIST, transform))
244           } else {
245             // for non-standard or complex-element lists, we just convert to a json string
246             Seq(InferredType(name, ObjectType.JSON, jsonTransform()))
247           }
248 
249         case _: MapLogicalTypeAnnotation =>
250           if (getMapKeyValueTypes(group).exists { case (k, v) => k.isPrimitive && v.isPrimitive }) {
251             Seq(InferredType(name, ObjectType.MAP, transform))
252           } else {
253             // for non-standard or complex-value maps, we just convert to a json string
254             Seq(InferredType(name, ObjectType.JSON, jsonTransform()))
255           }
256 
257         case _ =>
258           group.getFields.asScala.flatMap(mapNormalField(_, namer, fieldPath)).toSeq
259       }
260     }
261   }

263   /**
264    * Maps a GeoParquet geometry field
265    *
266    * @param field field
267    * @param namer name helper
268    * @param geo GeoParquet metadata
269    * @return
270    */
271   private def mapGeoField(field: Type, namer: Namer, geo: ColumnMetadata): Seq[InferredType] = {
272     lazy val name = namer(field.getName)
273 
274     def geomTransform(fn: String): FunctionTransform = FunctionTransform(s"$fn(avroPath(", s",'/${field.getName}'))")
275 
276     def mismatch(): Seq[InferredType] = {
277       logger.warn(s"Found GeoParquet metadata for column '${field.getName}' but the schema does not match the expected type")
278       mapNormalField(field, namer)
279     }
280 
281     if (geo.encoding == GeoParquetColumnEncoding.WKB) {
282       if (isWkbType(field)) {
283         Seq(InferredType(name, GeoParquetColumnType.toObjectType(geo.types), geomTransform("geometry")))
284       } else {
285         mismatch()
286       }
287     } else {
288       val objectType = GeoParquetColumnEncoding.toObjectType(geo.encoding)
289       if ((geo.types.isEmpty || GeoParquetColumnType.toObjectType(geo.types) == objectType) && isNativeType(field, objectType)) {
290         Seq(InferredType(name, objectType, geomTransform(geo.encoding.toString)))
291       } else {
292         mismatch()
293       }
294     }
295   }

297   /**
298    * Checks if a field corresponds to the GeoParquet schema for a WKB-encoded geometry
299    *
300    * @param field field
301    * @return
302    */
303   private def isWkbType(field: Type): Boolean =
304     !field.isRepetition(Repetition.REPEATED) && field.isPrimitive &&
305       field.asPrimitiveType().getPrimitiveTypeName == PrimitiveTypeName.BINARY
306 
307   /**
308    * Checks if the field corresponds to the GeoParquet schema for a native-encoded geometry
309    *
310    * @param field field
311    * @param objectType native type
312    * @return
313    */
314   private def isNativeType(field: Type, objectType: ObjectType): Boolean = {
315     // compare child fields, so we don't consider optional vs required, id, name, logicalTypes, etc
316     !field.isRepetition(Repetition.REPEATED) &&
317       field.asGroupType().getFields == GeometrySchema(objectType, GeoParquetNative).named(field.getName).asGroupType().getFields
318   }
