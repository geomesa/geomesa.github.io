39   /**
40     * Handles parquet files (including those produced by the FSDS and CLI export)
41     *
42     * @param is input
43     * @param sft simple feature type, if known ahead of time
44     * @param path file path, if there is a file available
45     * @return
46     */
47   override def infer(
48       is: InputStream,
49       sft: Option[SimpleFeatureType],
50       path: Option[String]): Option[(SimpleFeatureType, Config)] =
51     infer(is, sft, path.fold(Map.empty[String, AnyRef])(EvaluationContext.inputFileParam)).toOption
52 
53   override def infer(
54       is: InputStream,
55       sft: Option[SimpleFeatureType],
56       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = {
57     val path = hints.get(EvaluationContext.InputFilePathKey) match {
58       case None => Failure(new RuntimeException("No file path specified to the input data"))
59       case Some(p) => Success(p.toString)
60     }
61     path.flatMap { p =>
62       Try {
63         // note: get the path as a URI so that we handle local files appropriately
64         val filePath = new Path(PathUtils.getUrl(p).toURI)
65         val footer = ParquetFileReader.readFooter(new Configuration(), filePath, ParquetMetadataConverter.NO_FILTER)
66         val (schema, fields, id) = SimpleFeatureParquetSchema.read(footer.getFileMetaData) match {
67           case Some(parquet) =>
68             // this is a geomesa encoded parquet file
69             val fields = parquet.sft.getAttributeDescriptors.asScala.map { descriptor =>
70               val name = parquet.field(parquet.sft.indexOf(descriptor.getLocalName))
71               // note: parquet converter stores the generic record under index 0
72               val path = s"avroPath($$0, '/$name')"
73               // some types need a function applied to the underlying avro value
74               val expression = ObjectType.selectType(descriptor) match {
75                 case Seq(ObjectType.GEOMETRY, ObjectType.POINT)           => s"parquetPoint($$0, '/$name')"
76                 case Seq(ObjectType.GEOMETRY, ObjectType.MULTIPOINT)      => s"parquetMultiPoint($$0, '/$name')"
77                 case Seq(ObjectType.GEOMETRY, ObjectType.LINESTRING)      => s"parquetLineString($$0, '/$name')"
78                 case Seq(ObjectType.GEOMETRY, ObjectType.MULTILINESTRING) => s"parquetMultiLineString($$0, '/$name')"
79                 case Seq(ObjectType.GEOMETRY, ObjectType.POLYGON)         => s"parquetPolygon($$0, '/$name')"
80                 case Seq(ObjectType.GEOMETRY, ObjectType.MULTIPOLYGON)    => s"parquetMultiPolygon($$0, '/$name')"
81                 case Seq(ObjectType.UUID)                                 => s"avroBinaryUuid($path)"
82                 case _                                                    => path
83               }
84               BasicField(descriptor.getLocalName, Some(Expression(expression)))
85             }
86             val id = Expression(s"avroPath($$0, '/${SimpleFeatureParquetSchema.FeatureIdField}')")
87 
88             // validate the existing schema, if any
89             if (sft.exists(_.getAttributeDescriptors.asScala != parquet.sft.getAttributeDescriptors.asScala)) {
90               throw new IllegalArgumentException("Inferred schema does not match existing schema")
91             }
92             (parquet.sft, fields, Some(id))
93 
94           case _ =>
95             // this is an arbitrary parquet file, create fields based on the schema
96             val types = ParquetConverterFactory.schemaTypes(footer.getFileMetaData.getSchema)
97             val dataSft = TypeInference.schema("inferred-parquet", types)
98             // note: parquet converter stores the generic record under index 0
99             val fields = types.map(t => BasicField(t.name, Some(Expression(t.transform.apply(0)))))
100 
101             // validate the existing schema, if any
102             sft.foreach(AbstractConverterFactory.validateInferredType(_, types.map(_.typed)))
103 
104             (dataSft, fields, None)
105         }
106 
107         val converterConfig = BasicConfig(typeToProcess, id, Map.empty, Map.empty)
108 
109         val config = configConvert.to(converterConfig)
110             .withFallback(fieldConvert.to(fields.toSeq))
111             .withFallback(optsConvert.to(BasicOptions.default))
112             .toConfig
113 
114         (schema, config)
115       }
116     }
117   }
118 }

126   /**
127     * Take an avro schema and return the simple feature type bindings for it
128     *
129     * @param schema avro schema
130     * @return
131     */
132   def schemaTypes(schema: MessageType): Seq[InferredType] = {
133     val namer = new Namer()
134     val types = scala.collection.mutable.ArrayBuffer.empty[InferredType]
135 
136     def mapField(field: Type, path: String = ""): Unit = {
137       // get a valid attribute name
138       val name = namer(field.getName)
139       val transform = FunctionTransform("avroPath(", s",'$path/${field.getName}')")
140 
141       val original = field.getOriginalType
142       if (field.isPrimitive) {
143         // note: date field transforms are handled by AvroReadSupport
144         lazy val int32: InferredType = original match {
145           case OriginalType.DATE => InferredType(name, ObjectType.DATE, transform)
146           case _ => InferredType(name, ObjectType.INT, transform)
147         }
148         lazy val int64: InferredType = original match {
149           case OriginalType.TIMESTAMP_MILLIS => InferredType(name, ObjectType.DATE, transform)
150           case OriginalType.TIMESTAMP_MICROS => InferredType(name, ObjectType.DATE, transform)
151           case _ => InferredType(name, ObjectType.LONG, transform)
152         }
153         lazy val binary: InferredType = original match {
154           case OriginalType.UTF8 => InferredType(name, ObjectType.STRING, transform)
155           case _ => InferredType(name, ObjectType.BYTES, transform)
156         }
157 
158         field.asPrimitiveType().getPrimitiveTypeName match {
159           case PrimitiveTypeName.BINARY               => types += binary
160           case PrimitiveTypeName.INT32                => types += int32
161           case PrimitiveTypeName.INT64                => types += int64
162           case PrimitiveTypeName.FLOAT                => types += InferredType(name, ObjectType.FLOAT, transform)
163           case PrimitiveTypeName.DOUBLE               => types += InferredType(name, ObjectType.DOUBLE, transform)
164           case PrimitiveTypeName.BOOLEAN              => types += InferredType(name, ObjectType.BOOLEAN, transform)
165           case PrimitiveTypeName.FIXED_LEN_BYTE_ARRAY => types += InferredType(name, ObjectType.BYTES, transform)
166           case _ => // no-op
167         }
168       } else {
169         val group = field.asGroupType()
170         original match {
171           case OriginalType.LIST =>
172             if (group.getFieldCount == 1 && !group.getType(0).isPrimitive) {
173               val list = group.getType(0).asGroupType()
174               if (list.getFieldCount == 1 && list.isRepetition(Repetition.REPEATED) && list.getType(0).isPrimitive) {
175                 types += InferredType(name, ObjectType.LIST, transform)
176               }
177             }
178 
179           case OriginalType.MAP =>
180             if (group.getFieldCount == 1 && !group.getType(0).isPrimitive) {
181               val map = group.getType(0).asGroupType()
182               if (map.getFieldCount == 2 && map.isRepetition(Repetition.REPEATED) &&
183                   map.getFields.asScala.forall(_.isPrimitive)) {
184                 types += InferredType(name, ObjectType.MAP, transform)
185               }
186             }
187 
188           case _ =>
189             if (group.getFieldCount == 1 && group.getType(0).isPrimitive &&
190                 group.getType(0).isRepetition(Repetition.REPEATED)) {
191               types += InferredType(name, ObjectType.LIST, transform)
192             } else {
193               group.getFields.asScala.foreach(mapField(_, s"$path/${field.getName}"))
194             }
195         }
196       }
197     }
198 
199     schema.getFields.asScala.foreach(mapField(_))
200 
201     // check if we can derive a geometry field
202     TypeInference.deriveGeometry(types.toSeq, namer).foreach(g => types += g)
203 
204     types.toSeq
205   }
