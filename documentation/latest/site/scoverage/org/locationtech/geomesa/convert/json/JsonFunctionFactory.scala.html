44   private val jsonPath: TransformerFunction = new NamedTransformerFunction(Array("jsonPath"), pure = true) {
45     private val cache = new ConcurrentHashMap[Any, JsonPath]()
46     override def apply(args: Array[AnyRef]): AnyRef = {
47       var path = cache.get(args(0))
48       if (path == null) {
49         path = JsonPath.compile(args(0).asInstanceOf[String])
50         cache.put(args(0), path)
51       }
52       val elem = path.read[JsonElement](args(1), JsonConverter.JsonConfiguration)
53       // unwrap primitive and null elements
54       if (elem.isJsonNull) {
55         null
56       } else if (elem.isJsonPrimitive) {
57         val p = elem.getAsJsonPrimitive
58         if (p.isString) {
59           p.getAsString
60         } else if (p.isNumber) {
61           p.getAsNumber
62         } else if (p.isBoolean) {
63           Boolean.box(p.getAsBoolean)
64         } else {
65           p.getAsString // this shouldn't really ever happen...
66         }
67       } else {
68         elem
69       }
70     }
71   }

137   private val newJsonObject = TransformerFunction.pure("newJsonObject") { args =>
138     val obj = new JsonObject()
139     var i = 1
140     while (i < args.length) {
141       val key = args(i -1).toString
142       args(i) match {
143         case null => // skip nulls
144         case j: JsonElement => obj.add(key, j)
145         case j: String  => obj.add(key, new JsonPrimitive(j))
146         case j: Number  => obj.add(key, new JsonPrimitive(j))
147         case j: Boolean => obj.add(key, new JsonPrimitive(j))
148         case j: Date    => obj.add(key, new JsonPrimitive(DateParsing.formatDate(j)))
149         case j          => obj.add(key, gson.toJsonTree(j))
150       }
151 
152       i += 2
153     }
154     obj
155   }

157   private val emptyToNull = TransformerFunction.pure("emptyJsonToNull") { args =>
158     args(0) match {
159       case JsonNull.INSTANCE => null

161       case j: JsonObject if j.entrySet().asScala.forall(_.getValue == JsonNull.INSTANCE) => null
162       case j: JsonArray if j.size() == 0 => null
163       case j => j
164     }
