44   private val jsonPath: TransformerFunction = new NamedTransformerFunction(Array("jsonPath"), pure = true) {
45     private val cache = new ConcurrentHashMap[Any, JsonPath]()
46     override def apply(args: Array[AnyRef]): AnyRef = {
47       var path = cache.get(args(0))
48       if (path == null) {
49         path = JsonPath.compile(args(0).asInstanceOf[String])
50         cache.put(args(0), path)
51       }
52       val elem = path.read[JsonElement](args(1), JsonConverter.JsonConfiguration)
53       // unwrap primitive and null elements
54       if (elem.isJsonNull) {
55         null
56       } else if (elem.isJsonPrimitive) {
57         val p = elem.getAsJsonPrimitive
58         if (p.isString) {
59           p.getAsString
60         } else if (p.isNumber) {
61           p.getAsNumber
62         } else if (p.isBoolean) {
63           Boolean.box(p.getAsBoolean)
64         } else {
65           p.getAsString // this shouldn't really ever happen...
66         }
67       } else {
68         elem
69       }
70     }
71   }

122   private val newJsonObject = TransformerFunction.pure("newJsonObject") { args =>
123     val obj = new JsonObject()
124     var i = 1
125     while (i < args.length) {
126       val key = args(i -1).toString
127       args(i) match {
128         case null => // skip nulls
129         case j: JsonElement => obj.add(key, j)
130         case j: String  => obj.add(key, new JsonPrimitive(j))
131         case j: Number  => obj.add(key, new JsonPrimitive(j))
132         case j: Boolean => obj.add(key, new JsonPrimitive(j))
133         case j: Date    => obj.add(key, new JsonPrimitive(DateParsing.formatDate(j)))
134         case j          => obj.add(key, gson.toJsonTree(j))
135       }
136 
137       i += 2
138     }
139     obj
140   }

142   private val emptyToNull = TransformerFunction.pure("emptyJsonToNull") { args =>
143     args(0) match {
144       case JsonNull.INSTANCE => null
145       case j: JsonObject if j.size() == 0 => null
146       case j: JsonObject if j.entrySet().asScala.forall(_.getValue == JsonNull.INSTANCE) => null
147       case j: JsonArray if j.size() == 0 => null
148       case j => j
149     }
