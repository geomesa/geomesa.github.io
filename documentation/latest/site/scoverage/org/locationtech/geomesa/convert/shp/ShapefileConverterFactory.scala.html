39   override def infer(
40       is: InputStream,
41       sft: Option[SimpleFeatureType],
42       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = {
43     val url = hints.get(EvaluationContext.InputFilePathKey) match {
44       case Some(p) => p.toString
45       case None => return Failure(new RuntimeException("No file path specified to the input data"))
46 
47     }
48     Try {
49       WithClose(ShapefileConverter.getDataStore(url, ShapefileConverterFactory.DefaultCharset)) { ds =>
50         val fields = sft match {
51           case None =>
52             var i = 0
53             ds.getSchema.getAttributeDescriptors.asScala.map { d =>
54               i += 1
55               BasicField(d.getLocalName, Some(Column(i)))
56             }
57 
58           case Some(s) =>
59             // map the attributes whose name match the shapefile
60             val descriptors = ds.getSchema.getAttributeDescriptors.asScala
61             s.getAttributeDescriptors.asScala.flatMap { d =>
62               val name = d.getLocalName
63               val i = descriptors.indexWhere(_.getLocalName.equalsIgnoreCase(name))
64               if (i == -1) { Seq.empty } else {
65                 Seq(BasicField(name, Some(Column(i + 1))))
66               }
67             }
68         }
69 
70         val shpConfig = BasicConfig(TypeToProcess, Some(Column(0)), Map.empty, Map.empty)
71 
72         val config = BasicConfigConvert.to(shpConfig)
73             .withFallback(BasicFieldConvert.to(fields.toSeq))
74             .withFallback(BasicOptionsConvert.to(BasicOptions.default))
75             .toConfig
76 
77         (sft.getOrElse(ds.getSchema), config)
78       }
79     }
80   }
81 
82   override protected def withDefaults(conf: Config): Config =
83     super.withDefaults(conf.withFallback(ShapefileConverterFactory.ShpConfigDefaults))
84 }
