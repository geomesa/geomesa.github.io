29     ShapefileConverterFactory.TypeToProcess, BasicConfigConvert, BasicFieldConvert, BasicOptionsConvert) {

33   override def infer(
34       is: InputStream,
35       sft: Option[SimpleFeatureType],
36       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = {
37     val url = hints.get(EvaluationContext.InputFilePathKey) match {
38       case Some(p) => p.toString
39       case None => return Failure(new RuntimeException("No file path specified to the input data"))
40 
41     }
42     Try {
43       WithClose(ShapefileConverter.getDataStore(url, ShapefileConverterFactory.DefaultCharset)) { ds =>
44         val fields = sft match {
45           case None =>
46             var i = 0
47             ds.getSchema.getAttributeDescriptors.asScala.map { d =>
48               i += 1
49               BasicField(d.getLocalName, Some(Column(i)))
50             }
51 
52           case Some(s) =>
53             // map the attributes whose name match the shapefile
54             val descriptors = ds.getSchema.getAttributeDescriptors.asScala
55             s.getAttributeDescriptors.asScala.flatMap { d =>
56               val name = d.getLocalName
57               val i = descriptors.indexWhere(_.getLocalName.equalsIgnoreCase(name))
58               if (i == -1) { Seq.empty } else {
59                 Seq(BasicField(name, Some(Column(i + 1))))
60               }
61             }
62         }
63 
64         val shpConfig = BasicConfig(TypeToProcess, None, Some(Column(0)), Map.empty, Map.empty)
65 
66         val config = BasicConfigConvert.to(shpConfig)
67             .withFallback(BasicFieldConvert.to(fields.toSeq))
68             .withFallback(BasicOptionsConvert.to(BasicOptions.default))
69             .toConfig
70 
71         (sft.getOrElse(ds.getSchema), config)
72       }
73     }
74   }
75 
76   override protected def withDefaults(conf: Config): Config =
77     super.withDefaults(conf.withFallback(ShapefileConverterFactory.ShpConfigDefaults))
78 }
