25 class ShapefileConverterFactory
26   extends AbstractConverterFactory[ShapefileConverter, BasicConfig, BasicField, BasicOptions](
27     ShapefileConverterFactory.TypeToProcess, BasicConfigConvert, BasicFieldConvert, BasicOptionsConvert) {
28 
29   import scala.collection.JavaConverters._
30 
31   override def infer(
32       is: InputStream,
33       sft: Option[SimpleFeatureType],
34       path: Option[String]): Option[(SimpleFeatureType, Config)] =
35     infer(is, sft, path.fold(Map.empty[String, AnyRef])(EvaluationContext.inputFileParam)).toOption
36 
37   override def infer(
38       is: InputStream,
39       sft: Option[SimpleFeatureType],
40       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = {
41     val url = hints.get(EvaluationContext.InputFilePathKey) match {
42       case Some(p) => p.toString
43       case None => return Failure(new RuntimeException("No file path specified to the input data"))
44 
45     }
46     Try {
47       WithClose(ShapefileConverter.getDataStore(url)) { ds =>
48         val fields = sft match {
49           case None =>
50             var i = 0
51             ds.getSchema.getAttributeDescriptors.asScala.map { d =>
52               i += 1
53               BasicField(d.getLocalName, Some(Column(i)))
54             }
55 
56           case Some(s) =>
57             // map the attributes whose name match the shapefile
58             val descriptors = ds.getSchema.getAttributeDescriptors.asScala
59             s.getAttributeDescriptors.asScala.flatMap { d =>
60               val name = d.getLocalName
61               val i = descriptors.indexWhere(_.getLocalName.equalsIgnoreCase(name))
62               if (i == -1) { Seq.empty } else {
63                 Seq(BasicField(name, Some(Column(i + 1))))
64               }
65             }
66         }
67 
68         val shpConfig = BasicConfig(TypeToProcess, Some(Column(0)), Map.empty, Map.empty)
69 
70         val config = BasicConfigConvert.to(shpConfig)
71             .withFallback(BasicFieldConvert.to(fields.toSeq))
72             .withFallback(BasicOptionsConvert.to(BasicOptions.default))
73             .toConfig
74 
75         (sft.getOrElse(ds.getSchema), config)
76       }
77     }
78   }
79 }
