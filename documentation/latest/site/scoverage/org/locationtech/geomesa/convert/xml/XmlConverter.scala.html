83   def createXPath(factory: String, namespaces: Map[String, String] = Map.empty): XPath = {
84     val fact = try {
85       val res = XPathFactory.newInstance(XPathFactory.DEFAULT_OBJECT_MODEL_URI, factory, getClass.getClassLoader)
86       logger.info(s"Loaded xpath factory ${res.getClass}")
87       res
88     } catch {
89       case NonFatal(e) =>
90         logger.warn(s"Unable to load xpath provider '$factory': ${e.toString}. " +
91             "Xpath queries may be slower - check your classpath")
92         XPathFactory.newInstance(XPathFactory.DEFAULT_OBJECT_MODEL_URI)
93     }
94     val xpath = fact.newXPath()
95     if (namespaces.nonEmpty) {
96       xpath.setNamespaceContext(new NamespaceContext() {
97         override def getPrefix(namespaceURI: String): String = null
98         override def getPrefixes(namespaceURI: String): java.util.Iterator[String] = null
99         override def getNamespaceURI(prefix: String): String = namespaces.getOrElse(prefix, null)
100       })
101     }
102     xpath
103   }

105   def iterator(
106       parser: DocParser,
107       is: InputStream,
108       encoding: Charset,
109       mode: LineMode,
110       ec: EvaluationContext): CloseableIterator[Element] = {
111 
112     // detect and exclude the BOM if it exists
113     val bis = new BOMInputStream(is)
114     if (mode == LineMode.Single) {
115       val lines = IOUtils.lineIterator(bis, encoding)
116       val elements = lines.asScala.flatMap { line =>
117         ec.line += 1
118         if (line.isBlank) { Iterator.empty } else {
119           Iterator.single(parser.parse(new StringReader(line)))
120         }
121       }
122       CloseableIterator(elements, lines.close())
123     } else {
124       val reader = new InputStreamReader(bis, encoding)
125       CloseableIterator.fill(1, reader.close()) { ec.line += 1; parser.parse(reader) }
126     }
127   }

177   /**
178    * XML parser helper - holds non-thread-safe classes
179    *
180    * @param xpathFactory class name to use for creating xpaths
181    * @param namespaces xml namespaces
182    * @param xsd xsd path to an xsd used to validate parsed documents
183    * @param featurePath path to features in the xml doc
184    */
185   class XmlHelper(
186       xpathFactory: String,
187       namespaces: Map[String, String],
188       xsd: Option[String],
189       featurePath: Option[String]) {
190     val parser = new DocParser(xsd)
191     val xpath: XPath = createXPath(xpathFactory, namespaces)
192     val rootPath: Option[XPathExpression] = featurePath.map(xpath.compile)
193   }
194 
195   /**
196    * Document parser helper
197    *
198    * @param xsd xsd path to an xsd used to validate parsed documents
199    */
200   class DocParser(xsd: Option[String]) {
201 
202     private val builder = {
203       val factory = DocumentBuilderFactory.newInstance()
204       factory.setNamespaceAware(true)
205       val builder = factory.newDocumentBuilder()
206       // override default error handler which prints to stdout/stderr
207       builder.setErrorHandler(new ErrorHandler() {
208         override def warning(e: SAXParseException): Unit = logger.warn("XML parsing error:", e)
209         override def error(e: SAXParseException): Unit = throw e
210         override def fatalError(e: SAXParseException): Unit = throw e
211       })
212       builder
213     }
214 
215     private val validator = xsd.map { path =>
216       val schemaFactory = SchemaFactory.newInstance(W3C_XML_SCHEMA_NS_URI)
217       WithClose(getClass.getClassLoader.getResourceAsStream(path)) { xsdStream =>
218         schemaFactory.newSchema(new StreamSource(xsdStream)).newValidator()
219       }
220     }
221 
222     def parse(reader: Reader): Element = {
223       // parse the document once, then extract each feature node and operate on it
224       val document = builder.parse(new InputSource(reader))
225       // if a schema is defined, validate it - this will throw an exception on failure
226       validator.foreach(_.validate(new DOMSource(document)))
227       document.getDocumentElement
228     }
229   }
