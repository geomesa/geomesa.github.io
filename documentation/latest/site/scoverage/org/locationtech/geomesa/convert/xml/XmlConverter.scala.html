84   def createXPath(factory: String, namespaces: Map[String, String] = Map.empty): XPath = {
85     val fact = try {
86       val res = XPathFactory.newInstance(XPathFactory.DEFAULT_OBJECT_MODEL_URI, factory, getClass.getClassLoader)
87       logger.info(s"Loaded xpath factory ${res.getClass}")
88       res
89     } catch {
90       case NonFatal(e) =>
91         logger.warn(s"Unable to load xpath provider '$factory': ${e.toString}. " +
92             "Xpath queries may be slower - check your classpath")
93         XPathFactory.newInstance(XPathFactory.DEFAULT_OBJECT_MODEL_URI)
94     }
95     val xpath = fact.newXPath()
96     if (namespaces.nonEmpty) {
97       xpath.setNamespaceContext(new NamespaceContext() {
98         override def getPrefix(namespaceURI: String): String = null
99         override def getPrefixes(namespaceURI: String): java.util.Iterator[String] = null
100         override def getNamespaceURI(prefix: String): String = namespaces.getOrElse(prefix, null)
101       })
102     }
103     xpath
104   }

106   def iterator(
107       parser: DocParser,
108       is: InputStream,
109       encoding: Charset,
110       mode: LineMode,
111       ec: EvaluationContext): CloseableIterator[Element] = {
112 
113     // detect and exclude the BOM if it exists
114     val bis = new BOMInputStream(is)
115     if (mode == LineMode.Single) {
116       val lines = IOUtils.lineIterator(bis, encoding)
117       val elements = lines.asScala.flatMap { line =>
118         ec.line += 1
119         if (TextTools.isWhitespace(line)) { Iterator.empty } else {
120           Iterator.single(parser.parse(new StringReader(line)))
121         }
122       }
123       CloseableIterator(elements, lines.close())
124     } else {
125       val reader = new InputStreamReader(bis, encoding)
126       CloseableIterator.fill(1, reader.close()) { ec.line += 1; parser.parse(reader) }
127     }
128   }

179   /**
180    * XML parser helper - holds non-thread-safe classes
181    *
182    * @param xpathFactory class name to use for creating xpaths
183    * @param namespaces xml namespaces
184    * @param xsd xsd path to an xsd used to validate parsed documents
185    * @param featurePath path to features in the xml doc
186    */
187   class XmlHelper(
188       xpathFactory: String,
189       namespaces: Map[String, String],
190       xsd: Option[String],
191       featurePath: Option[String]) {
192     val parser = new DocParser(xsd)
193     val xpath: XPath = createXPath(xpathFactory, namespaces)
194     val rootPath: Option[XPathExpression] = featurePath.map(xpath.compile)
195   }
196 
197   /**
198    * Document parser helper
199    *
200    * @param xsd xsd path to an xsd used to validate parsed documents
201    */
202   class DocParser(xsd: Option[String]) {
203 
204     private val builder = {
205       val factory = DocumentBuilderFactory.newInstance()
206       factory.setNamespaceAware(true)
207       val builder = factory.newDocumentBuilder()
208       // override default error handler which prints to stdout/stderr
209       builder.setErrorHandler(new ErrorHandler() {
210         override def warning(e: SAXParseException): Unit = logger.warn("XML parsing error:", e)
211         override def error(e: SAXParseException): Unit = throw e
212         override def fatalError(e: SAXParseException): Unit = throw e
213       })
214       builder
215     }
216 
217     private val validator = xsd.map { path =>
218       val schemaFactory = SchemaFactory.newInstance(W3C_XML_SCHEMA_NS_URI)
219       WithClose(getClass.getClassLoader.getResourceAsStream(path)) { xsdStream =>
220         schemaFactory.newSchema(new StreamSource(xsdStream)).newValidator()
221       }
222     }
223 
224     def parse(reader: Reader): Element = {
225       // parse the document once, then extract each feature node and operate on it
226       val document = builder.parse(new InputSource(reader))
227       // if a schema is defined, validate it - this will throw an exception on failure
228       validator.foreach(_.validate(new DOMSource(document)))
229       document.getDocumentElement
230     }
231   }
