84   def createXPath(factory: String, namespaces: Map[String, String] = Map.empty): XPath = {
85     val fact = try {
86       val res = XPathFactory.newInstance(XPathFactory.DEFAULT_OBJECT_MODEL_URI, factory, getClass.getClassLoader)
87       logger.info(s"Loaded xpath factory ${res.getClass}")
88       res
89     } catch {
90       case NonFatal(e) =>
91         logger.warn(s"Unable to load xpath provider '$factory': ${e.toString}. " +
92             "Xpath queries may be slower - check your classpath")
93         XPathFactory.newInstance(XPathFactory.DEFAULT_OBJECT_MODEL_URI)
94     }
95     val xpath = fact.newXPath()
96     if (namespaces.nonEmpty) {
97       xpath.setNamespaceContext(new NamespaceContext() {
98         override def getPrefix(namespaceURI: String): String = null
99         override def getPrefixes(namespaceURI: String): java.util.Iterator[String] = null
100         override def getNamespaceURI(prefix: String): String = namespaces.getOrElse(prefix, null)
101       })
102     }
103     xpath
104   }

106   def iterator(
107       parser: DocParser,
108       is: InputStream,
109       encoding: Charset,
110       mode: LineMode,
111       ec: EvaluationContext): CloseableIterator[Element] = {
112 
113     // detect and exclude the BOM if it exists
114     val bis = new BOMInputStream(is)
115     if (mode == LineMode.Single) {
116       val lines = IOUtils.lineIterator(bis, encoding)
117       val elements = lines.asScala.flatMap { line =>
118         ec.line += 1
119         if (TextTools.isWhitespace(line)) { Iterator.empty } else {
120           Iterator.single(parser.parse(new StringReader(line)))
121         }
122       }
123       CloseableIterator(elements, lines.close())
124     } else {
125       val reader = new InputStreamReader(bis, encoding)
126       CloseableIterator.fill(1, reader.close()) { ec.line += 1; parser.parse(reader) }
127     }
128   }

178   /**
179    * XML parser helper - holds non-thread-safe classes
180    *
181    * @param xpathFactory class name to use for creating xpaths
182    * @param namespaces xml namespaces
183    * @param xsd xsd path to an xsd used to validate parsed documents
184    * @param featurePath path to features in the xml doc
185    */
186   class XmlHelper(
187       xpathFactory: String,
188       namespaces: Map[String, String],
189       xsd: Option[String],
190       featurePath: Option[String]) {
191     val parser = new DocParser(xsd)
192     val xpath: XPath = createXPath(xpathFactory, namespaces)
193     val rootPath: Option[XPathExpression] = featurePath.map(xpath.compile)
194   }
195 
196   /**
197    * Document parser helper
198    *
199    * @param xsd xsd path to an xsd used to validate parsed documents
200    */
201   class DocParser(xsd: Option[String]) {
202 
203     private val builder = {
204       val factory = DocumentBuilderFactory.newInstance()
205       factory.setNamespaceAware(true)
206       val builder = factory.newDocumentBuilder()
207       // override default error handler which prints to stdout/stderr
208       builder.setErrorHandler(new ErrorHandler() {
209         override def warning(e: SAXParseException): Unit = logger.warn("XML parsing error:", e)
210         override def error(e: SAXParseException): Unit = throw e
211         override def fatalError(e: SAXParseException): Unit = throw e
212       })
213       builder
214     }
215 
216     private val validator = xsd.map { path =>
217       val schemaFactory = SchemaFactory.newInstance(W3C_XML_SCHEMA_NS_URI)
218       WithClose(getClass.getClassLoader.getResourceAsStream(path)) { xsdStream =>
219         schemaFactory.newSchema(new StreamSource(xsdStream)).newValidator()
220       }
221     }
222 
223     def parse(reader: Reader): Element = {
224       // parse the document once, then extract each feature node and operate on it
225       val document = builder.parse(new InputSource(reader))
226       // if a schema is defined, validate it - this will throw an exception on failure
227       validator.foreach(_.validate(new DOMSource(document)))
228       document.getDocumentElement
229     }
230   }
