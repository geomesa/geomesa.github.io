48   /**
49    * Infer a configuration and simple feature type from an input stream, if possible
50    *
51    * Available hints:
52    *  - `featurePath` - xpath expression pointing to the feature element
53    *
54    * @param is input
55    * @param sft simple feature type, if known ahead of time
56    * @param hints implementation specific hints about the input
57    * @return
58    */
59   override def infer(
60       is: InputStream,
61       sft: Option[SimpleFeatureType],
62       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = {
63     val tryElements = Try {
64       val parser = new DocParser(None)
65       WithClose(XmlConverter.iterator(parser, is, StandardCharsets.UTF_8, LineMode.Multi, EvaluationContext.empty)) { iter =>
66         iter.take(AbstractConverterFactory.inferSampleSize).toList
67       }
68     }
69 
70     tryElements.flatMap { elements =>
71       val featurePath = hints.get(XmlConverterFactory.FeaturePathKey).map(_.toString)
72       val namespaces = elements.headOption.map(XmlConverterFactory.getNamespaces).getOrElse(Map.empty)
73       val xpathFactory =
74         ConfigSource.fromConfig(withDefaults(XmlConverterFactory.TypeConfig))
75             .loadOrThrow[XmlConfig](classTag[XmlConfig], XmlConfigConvert)
76             .xpathFactory
77 
78       val tryFeatures = Try {
79         featurePath match {
80           case None => elements
81           case Some(p) =>
82             val xpath = XmlConverter.createXPath(xpathFactory, namespaces).compile(p)
83             val features = elements.flatMap { element =>
84               val nodeList = xpath.evaluate(element, XPathConstants.NODESET).asInstanceOf[NodeList]
85               Seq.tabulate(nodeList.getLength)(i => nodeList.item(i).asInstanceOf[Element])
86             }
87             features.take(AbstractConverterFactory.inferSampleSize)
88         }
89       }
90 
91       tryFeatures.flatMap { features =>
92         if (features.isEmpty) {
93           Failure(new RuntimeException("Could not parse input as XML"))
94         } else {
95           Try {
96             // track the properties in each feature
97             // use linkedHashMap to retain insertion order
98             val props = scala.collection.mutable.LinkedHashMap.empty[String, ListBuffer[Any]]
99 
100             val namer = new XmlNamer()
101             features.foreach { feature =>
102               namer.addRootElement(feature)
103               XmlConverterFactory.parseNode(feature, "").foreach { case (k, v) =>
104                 props.getOrElseUpdate(k, ListBuffer.empty) += v
105               }
106             }
107 
108             val pathsAndValues = props.toSeq.map { case (path, values) => PathWithValues(path, values) }
109             val inferredTypes = TypeInference.infer(pathsAndValues, sft.toRight("inferred-xml"), namer)
110 
111             val fieldConfig = inferredTypes.types.map { case TypeWithPath(path, inferredType) =>
112               val transform = inferredType.transform match {
113                 case IdentityTransform => None
114                 // account for optional nodes by wrapping transform with a try/null
115                 case t => Some(Expression(s"try(${t.apply(0)},null)"))
116               }
117               if (path.isEmpty) {
118                 DerivedField(inferredType.name, transform)
119               } else {
120                 XmlPathField(inferredType.name, path, transform)
121               }
122             }
123 
124             val idField = Some(Expression("md5(stringToBytes(toString($0)))"))
125 
126             val xmlConfig =
127               XmlConfig(typeToProcess, xpathFactory, namespaces, None, featurePath, idField, Map.empty, Map.empty)
128 
129             val config =
130               configConvert.to(xmlConfig)
131                   .withFallback(fieldConvert.to(fieldConfig))
132                   .toConfig
133 
134             (inferredTypes.sft, config)
135           }
136         }
137       }
138     }
139   }

235    * Attribute namer for xml elements
236    */
237   private class XmlNamer extends Namer {
238 
239     private val rootPrefixes = scala.collection.mutable.HashSet.empty[String]
240     private val namespaceMatcher = Pattern.compile("(/@?)[^:/@]*:")
241 
242     /**
243      * Add a root element, whose tag will be removed from any derived attribute names
244      *
245      * @param elem element
246      */
247     def addRootElement(elem: Element): Unit =
248       rootPrefixes += s"/${elem.getTagName}/"
249 
250     override def apply(key: String): String = {
251       val withoutRoot = rootPrefixes.find(key.startsWith) match {
252         case None    => key
253         case Some(p) => key.substring(p.length - 1)
254       }
255       // remove namespace prefixes from the attribute names
256       val withoutPrefixes = namespaceMatcher.matcher(withoutRoot).replaceAll("$1")
257       super.apply(withoutPrefixes)
258     }
259   }

261   /**
262    * Extract xmlns declarations from this element
263    *
264    * @param doc element
265    * @return
266    */
267   private def getNamespaces(doc: Element): Map[String, String] = {
268     val namespaces = Map.newBuilder[String, String]
269     val attributes = doc.getAttributes
270     var i = 0
271     while (i < attributes.getLength) {
272       val attr = attributes.item(i).asInstanceOf[Attr]
273       if (attr.getName.startsWith("xmlns:")) {
274         namespaces += attr.getName.substring(6) -> attr.getValue
275       }
276       i += 1
277     }
278     namespaces.result
279   }

281   /**
282    * Parse an xml node
283    *
284    * @param node node
285    * @param path xpath to the parent of the object
286    * @return map of key is xpath to value, value is a string
287    */
288   private def parseNode(node: Node, path: String): Seq[(String, String)] = {
289     // use linkedHashMap to preserve insertion order
290     val builder = scala.collection.mutable.LinkedHashMap.empty[String, String]
291     node match {
292       case n: Element =>
293         val p = s"$path/${n.getTagName}"
294         Seq.tabulate(n.getAttributes.getLength)(n.getAttributes.item(_).asInstanceOf[Attr])
295             .filterNot(_.getName.startsWith("xmlns:"))
296             .sortBy(_.getLocalName)
297             .foreach(attr => builder += s"$p/@${attr.getName}" -> attr.getValue)
298         Seq.tabulate(n.getChildNodes.getLength)(n.getChildNodes.item)
299             .foreach(child => builder ++= parseNode(child, p))
300 
301       case n: Text if !TextTools.isWhitespace(n.getData) =>
302         builder += path -> n.getData
303 
304       case _ => // no-op
305     }
306     builder.toSeq
307   }
