48   /**
49    * Infer a configuration and simple feature type from an input stream, if possible
50    *
51    * Available hints:
52    *  - `featurePath` - xpath expression pointing to the feature element
53    *
54    * @param is input
55    * @param sft simple feature type, if known ahead of time
56    * @param hints implementation specific hints about the input
57    * @return
58    */
59   override def infer(
60       is: InputStream,
61       sft: Option[SimpleFeatureType],
62       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = {
63     val tryElements = Try {
64       val parser = new DocParser(None)
65       WithClose(XmlConverter.iterator(parser, is, StandardCharsets.UTF_8, LineMode.Multi, EvaluationContext.empty)) { iter =>
66         iter.take(AbstractConverterFactory.inferSampleSize).toList
67       }
68     }
69 
70     tryElements.flatMap { elements =>
71       val featurePath = hints.get(XmlConverterFactory.FeaturePathKey).map(_.toString)
72       val (defaultNs, namespaces) = elements.headOption match {
73         case None => (None, Map.empty[String, String])
74         case Some(e) =>
75           val ns = XmlConverterFactory.getNamespaces(e)
76           ns.get("") match {
77             case None => (None, ns)
78             case Some(default) =>
79               val alphas = Random.alphanumeric.filter(_.isLower)
80               var key: String = null
81               while ({ key = alphas.take(4).mkString(""); ns.contains(key) }) {
82                 // loop
83               }
84               (Some(s"$key:"), ns - "" + (key -> default))
85           }
86       }
87 
88       val xpathFactory =
89         ConfigSource.fromConfig(withDefaults(XmlConverterFactory.TypeConfig))
90             .loadOrThrow[XmlConfig](classTag[XmlConfig], XmlConfigConvert)
91             .xpathFactory
92 
93       val tryFeatures = Try {
94         featurePath match {
95           case None => elements
96           case Some(p) =>
97             val xpath = XmlConverter.createXPath(xpathFactory, namespaces).compile(p)
98             val features = elements.flatMap { element =>
99               val nodeList = xpath.evaluate(element, XPathConstants.NODESET).asInstanceOf[NodeList]
100               Seq.tabulate(nodeList.getLength)(i => nodeList.item(i).asInstanceOf[Element])
101             }
102             features.take(AbstractConverterFactory.inferSampleSize)
103         }
104       }
105 
106       tryFeatures.flatMap { features =>
107         if (features.isEmpty) {
108           Failure(new RuntimeException("Could not parse input as XML"))
109         } else {
110           Try {
111             // track the properties in each feature
112             // use linkedHashMap to retain insertion order
113             val props = scala.collection.mutable.LinkedHashMap.empty[String, ListBuffer[Any]]
114 
115             val namer = new XmlNamer(defaultNs)
116             features.foreach { feature =>
117               namer.addRootElement(feature)
118               XmlConverterFactory.parseNode(feature, "", defaultNs.getOrElse("")).foreach { case (k, v) =>
119                 props.getOrElseUpdate(k, ListBuffer.empty) += v
120               }
121             }
122 
123             val pathsAndValues = props.toSeq.map { case (path, values) => PathWithValues(path, values) }
124             val inferredTypes = TypeInference.infer(pathsAndValues, sft.toRight("inferred-xml"), namer)
125 
126             val fieldConfig = inferredTypes.types.map { case TypeWithPath(path, inferredType) =>
127               val transform = inferredType.transform match {
128                 case IdentityTransform => None
129                 // account for optional nodes by wrapping transform with a try/null
130                 case t => Some(Expression(s"try(${t.apply(0)},null)"))
131               }
132               if (path.isEmpty) {
133                 DerivedField(inferredType.name, transform)
134               } else {
135                 XmlPathField(inferredType.name, path, transform)
136               }
137             }
138 
139             val idField = Some(Expression("md5(stringToBytes(toString($0)))"))
140 
141             val xmlConfig =
142               XmlConfig(typeToProcess, xpathFactory, namespaces, None, featurePath, idField, Map.empty, Map.empty)
143 
144             val config =
145               configConvert.to(xmlConfig)
146                   .withFallback(fieldConvert.to(fieldConfig))
147                   .toConfig
148 
149             (inferredTypes.sft, config)
150           }
151         }
152       }
153     }
154   }

249   /**
250    * Attribute namer for xml elements
251    */
252   private class XmlNamer(defaultPrefix: Option[String]) extends Namer {
253 
254     private val rootPrefixes = scala.collection.mutable.HashSet.empty[String]
255     private val namespaceMatcher = Pattern.compile("(/@?)[^:/@]*:")
256 
257     /**
258      * Add a root element, whose tag will be removed from any derived attribute names
259      *
260      * @param elem element
261      */
262     def addRootElement(elem: Element): Unit = {
263       rootPrefixes += {
264         defaultPrefix match {
265           case Some(p) if elem.getPrefix == null || elem.getPrefix.isEmpty => s"/$p${elem.getTagName}/"
266           case _ => s"/${elem.getTagName}/"
267         }
268       }
269     }
270 
271     override def apply(key: String): String = {
272       val withoutRoot = rootPrefixes.find(key.startsWith) match {
273         case None    => key
274         case Some(p) => key.substring(p.length - 1)
275       }
276       // remove namespace prefixes from the attribute names
277       val withoutPrefixes = namespaceMatcher.matcher(withoutRoot).replaceAll("$1")
278       super.apply(withoutPrefixes)
279     }
280   }

282   /**
283    * Extract xmlns declarations from this element
284    *
285    * @param doc element
286    * @return
287    */
288   private def getNamespaces(doc: Element): Map[String, String] = {
289     val namespaces = Map.newBuilder[String, String]
290     val attributes = doc.getAttributes
291     var i = 0
292     while (i < attributes.getLength) {
293       val attr = attributes.item(i).asInstanceOf[Attr]
294       if (attr.getName.startsWith("xmlns:")) {
295         namespaces += attr.getName.substring(6) -> attr.getValue
296       } else if (attr.getName == "xmlns") {
297         namespaces += "" -> attr.getValue
298       }
299       i += 1
300     }
301     namespaces.result
302   }

304   /**
305    * Parse an xml node
306    *
307    * @param node node
308    * @param path xpath to the parent of the object
309    * @param defaultNamespacePrefix path prefix for elements without an explicit namespace
310    * @return map of key is xpath to value, value is a string
311    */
312   private def parseNode(node: Node, path: String, defaultNamespacePrefix: String): Seq[(String, String)] = {
313     // use linkedHashMap to preserve insertion order
314     val builder = scala.collection.mutable.LinkedHashMap.empty[String, String]
315     node match {
316       case n: Element =>
317         val elemPath = {
318           val prefix = if (n.getPrefix == null || n.getPrefix.isEmpty) { defaultNamespacePrefix } else { "" }
319           s"$path/$prefix${n.getTagName}"
320         }
321         Seq.tabulate(n.getAttributes.getLength)(n.getAttributes.item(_).asInstanceOf[Attr])
322             .filterNot(a => a.getName.startsWith("xmlns:") || a.getName == "xmlns")
323             .sortBy(_.getLocalName)
324             .foreach(attr => builder += s"$elemPath/@${attr.getName}" -> attr.getValue)
325         Seq.tabulate(n.getChildNodes.getLength)(n.getChildNodes.item)
326             .foreach(child => builder ++= parseNode(child, elemPath, defaultNamespacePrefix))
327 
328       case n: Text if !TextTools.isWhitespace(n.getData) =>
329         builder += path -> n.getData
330 
331       case _ => // no-op
332     }
333     builder.toSeq
334   }
