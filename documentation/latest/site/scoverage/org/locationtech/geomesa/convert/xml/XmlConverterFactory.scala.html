41   /**
42    * Infer a configuration and simple feature type from an input stream, if possible
43    *
44    * Available hints:
45    *  - `featurePath` - xpath expression pointing to the feature element
46    *
47    * @param is input
48    * @param sft simple feature type, if known ahead of time
49    * @param hints implementation specific hints about the input
50    * @return
51    */
52   override def infer(
53       is: InputStream,
54       sft: Option[SimpleFeatureType],
55       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = {
56     val tryElements = Try {
57       val parser = new DocParser(None)
58       WithClose(XmlConverter.iterator(parser, is, StandardCharsets.UTF_8, LineMode.Multi, EvaluationContext.empty)) { iter =>
59         iter.take(AbstractConverterFactory.inferSampleSize).toList
60       }
61     }
62 
63     tryElements.flatMap { elements =>
64       val featurePath = hints.get(XmlConverterFactory.FeaturePathKey).map(_.toString)
65       val (defaultNs, namespaces) = elements.headOption match {
66         case None => (None, Map.empty[String, String])
67         case Some(e) =>
68           val ns = XmlConverterFactory.getNamespaces(e)
69           ns.get("") match {
70             case None => (None, ns)
71             case Some(default) =>
72               val alphas = Random.alphanumeric.filter(_.isLower)
73               var key: String = null
74               while ({ key = alphas.take(4).mkString(""); ns.contains(key) }) {
75                 // loop
76               }
77               (Some(s"$key:"), ns - "" + (key -> default))
78           }
79       }
80 
81       val xpathFactory =
82         ConfigSource.fromConfig(withDefaults(XmlConverterFactory.TypeConfig))
83             .loadOrThrow[XmlConfig](classTag[XmlConfig], XmlConfigConvert)
84             .xpathFactory
85 
86       val tryFeatures = Try {
87         featurePath match {
88           case None => elements
89           case Some(p) =>
90             val xpath = XmlConverter.createXPath(xpathFactory, namespaces).compile(p)
91             val features = elements.flatMap { element =>
92               val nodeList = xpath.evaluate(element, XPathConstants.NODESET).asInstanceOf[NodeList]
93               Seq.tabulate(nodeList.getLength)(i => nodeList.item(i).asInstanceOf[Element])
94             }
95             features.take(AbstractConverterFactory.inferSampleSize)
96         }
97       }
98 
99       tryFeatures.flatMap { features =>
100         if (features.isEmpty) {
101           Failure(new RuntimeException("Could not parse input as XML"))
102         } else {
103           Try {
104             // track the properties in each feature
105             // use linkedHashMap to retain insertion order
106             val props = scala.collection.mutable.LinkedHashMap.empty[String, ListBuffer[Any]]
107 
108             val namer = new XmlNamer(defaultNs)
109             features.foreach { feature =>
110               namer.addRootElement(feature)
111               XmlConverterFactory.parseNode(feature, "", defaultNs.getOrElse("")).foreach { case (k, v) =>
112                 props.getOrElseUpdate(k, ListBuffer.empty) += v
113               }
114             }
115 
116             val pathsAndValues = props.toSeq.map { case (path, values) => PathWithValues(path, values) }
117             val inferredTypes = TypeInference.infer(pathsAndValues, sft.toRight("inferred-xml"), namer)
118 
119             val fieldConfig = inferredTypes.types.map { case TypeWithPath(path, inferredType) =>
120               val transform = inferredType.transform match {
121                 case IdentityTransform => None
122                 // account for optional nodes by wrapping transform with a try/null
123                 case t => Some(Expression(s"try(${t.apply(0)},null)"))
124               }
125               if (path.isEmpty) {
126                 DerivedField(inferredType.name, transform)
127               } else {
128                 XmlPathField(inferredType.name, path, transform)
129               }
130             }
131 
132             val idField = Some(Expression("md5(stringToBytes(toString($0)))"))
133 
134             val xmlConfig =
135               XmlConfig(typeToProcess, None, xpathFactory, namespaces, None, featurePath, idField, Map.empty, Map.empty)
136 
137             val config =
138               configConvert.to(xmlConfig)
139                   .withFallback(fieldConvert.to(fieldConfig))
140                   .toConfig
141 
142             (inferredTypes.sft, config)
143           }
144         }
145       }
146     }
147   }

242   /**
243    * Attribute namer for xml elements
244    */
245   private class XmlNamer(defaultPrefix: Option[String]) extends Namer {
246 
247     private val rootPrefixes = scala.collection.mutable.HashSet.empty[String]
248     private val namespaceMatcher = Pattern.compile("(/@?)[^:/@]*:")
249 
250     /**
251      * Add a root element, whose tag will be removed from any derived attribute names
252      *
253      * @param elem element
254      */
255     def addRootElement(elem: Element): Unit = {
256       rootPrefixes += {
257         defaultPrefix match {
258           case Some(p) if elem.getPrefix == null || elem.getPrefix.isEmpty => s"/$p${elem.getTagName}/"
259           case _ => s"/${elem.getTagName}/"
260         }
261       }
262     }
263 
264     override def apply(key: String): String = {
265       val withoutRoot = rootPrefixes.find(key.startsWith) match {
266         case None    => key
267         case Some(p) => key.substring(p.length - 1)
268       }
269       // remove namespace prefixes from the attribute names
270       val withoutPrefixes = namespaceMatcher.matcher(withoutRoot).replaceAll("$1")
271       super.apply(withoutPrefixes)
272     }
273   }

275   /**
276    * Extract xmlns declarations from this element
277    *
278    * @param doc element
279    * @return
280    */
281   private def getNamespaces(doc: Element): Map[String, String] = {
282     val namespaces = Map.newBuilder[String, String]
283     val attributes = doc.getAttributes
284     var i = 0
285     while (i < attributes.getLength) {
286       val attr = attributes.item(i).asInstanceOf[Attr]
287       if (attr.getName.startsWith("xmlns:")) {
288         namespaces += attr.getName.substring(6) -> attr.getValue
289       } else if (attr.getName == "xmlns") {
290         namespaces += "" -> attr.getValue
291       }
292       i += 1
293     }
294     namespaces.result
295   }

297   /**
298    * Parse an xml node
299    *
300    * @param node node
301    * @param path xpath to the parent of the object
302    * @param defaultNamespacePrefix path prefix for elements without an explicit namespace
303    * @return map of key is xpath to value, value is a string
304    */
305   private def parseNode(node: Node, path: String, defaultNamespacePrefix: String): Seq[(String, String)] = {
306     // use linkedHashMap to preserve insertion order
307     val builder = scala.collection.mutable.LinkedHashMap.empty[String, String]
308     node match {
309       case n: Element =>
310         val elemPath = {
311           val prefix = if (n.getPrefix == null || n.getPrefix.isEmpty) { defaultNamespacePrefix } else { "" }
312           s"$path/$prefix${n.getTagName}"
313         }
314         Seq.tabulate(n.getAttributes.getLength)(n.getAttributes.item(_).asInstanceOf[Attr])
315             .filterNot(a => a.getName.startsWith("xmlns:") || a.getName == "xmlns")
316             .sortBy(_.getLocalName)
317             .foreach(attr => builder += s"$elemPath/@${attr.getName}" -> attr.getValue)
318         Seq.tabulate(n.getChildNodes.getLength)(n.getChildNodes.item)
319             .foreach(child => builder ++= parseNode(child, elemPath, defaultNamespacePrefix))
320 
321       case n: Text if !n.getData.isBlank =>
322         builder += path -> n.getData
323 
324       case _ => // no-op
325     }
326     builder.toSeq
327   }
