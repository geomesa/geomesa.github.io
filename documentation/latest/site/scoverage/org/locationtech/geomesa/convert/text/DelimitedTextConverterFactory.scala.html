67   private def infer(
68       format: CSVFormat,
69       lines: Seq[String],
70       sft: Option[SimpleFeatureType]): Try[(SimpleFeatureType, Config)] = {
71     // : Seq[List[String]]
72     val rows = lines.flatMap { line =>
73       if (TextTools.isWhitespace(line)) { None } else {
74         Try(format.parse(new StringReader(line)).iterator().next.iterator.asScala.toList).toOption
75       }
76     }
77     val counts = rows.map(_.length).distinct
78     // try to verify that we actually have a delimited file
79     // ensure that some lines parsed, that there were at most 3 different col counts, and that there were at least 2 cols
80     if (counts.isEmpty || counts.lengthCompare(3) > 0 || counts.max < 2) {
81       val f = if (format == Formats.Tabs) { "tsv" } else if (format == Formats.Quoted) { "quoted csv" }  else { "csv" }
82       return Failure(new RuntimeException(s"Not a valid delimited text file using format: $f"))
83     }
84     Try {
85       val names = sft match {
86         case Some(s) =>
87           s.getAttributeDescriptors.asScala.map(_.getLocalName)
88 
89         case None =>
90           val values = rows.head.map(v => PathWithValues("", Seq(v)))
91           if (TypeInference.infer(values, Left("")).types.forall(_.inferredType.typed == ObjectType.STRING)) {
92             // assume the first row is headers
93             rows.head
94           } else {
95             Seq.empty
96           }
97       }
98       val nameIter = names.iterator
99       val pathsAndValues = Seq.tabulate(counts.max) { col =>
100         val values = rows.drop(1).map(vals => if (vals.lengthCompare(col) > 0) { vals(col) } else { null })
101         PathWithValues(if (nameIter.hasNext) { nameIter.next } else { "" }, values)
102       }
103       val inferred =
104         TypeInference.infer(pathsAndValues,
105           sft.toRight(s"inferred-${if (format == Formats.Tabs) { "tsv" } else { "csv" }}"))
106       val converterConfig = DelimitedTextConfig(typeToProcess, formats.find(_._2 == format).get._1,
107         Some(Expression("md5(string2bytes($0))")), Map.empty, Map.empty)
108 
109       val fields = {
110         var i = 0 // 0 is the whole record
111         inferred.types.map { case TypeWithPath(_, inferredType) =>
112           i += 1
113           BasicField(inferredType.name, Some(Expression(inferredType.transform(i))))
114         }
115       }
116 
117       val options = DelimitedTextOptions(None, CharNotSpecified, CharNotSpecified, None,
118         SimpleFeatureValidator.default, Seq.empty, ParseMode.Default, ErrorMode(), StandardCharsets.UTF_8)
119 
120       val config = configConvert.to(converterConfig)
121           .withFallback(fieldConvert.to(fields.toSeq))
122           .withFallback(optsConvert.to(options))
123           .toConfig
124 
125       (inferred.sft, config)
126     }
127   }

129   private def magic(lines: Seq[String], sft: Option[SimpleFeatureType]): Try[(SimpleFeatureType, Config)] = {
130     if (!lines.head.startsWith("id,")) { Failure(new RuntimeException("Not a geomesa delimited export")) } else {
131       Try {
132         val spec = WithClose(Formats.QuotedMinimal.parse(new StringReader(lines.head.drop(3)))) { result =>
133           result.iterator().next.asScala.mkString(",")
134         }
135         val schema = SimpleFeatureTypes.createType("", spec)
136 
137         val converterConfig = DelimitedTextConfig(typeToProcess,
138           formats.find(_._2 == Formats.QuotedMinimal).get._1, Some(Expression("$1")), Map.empty, Map.empty)
139 
140         var i = 1 // 0 is the whole record, 1 is the id
141         val fields = schema.getAttributeDescriptors.asScala.map { d =>
142           val bindings = ObjectType.selectType(d)
143           val transform = bindings.head match {
144             case ObjectType.STRING   => TypeInference.IdentityTransform
145             case ObjectType.INT      => TypeInference.CastToInt
146             case ObjectType.LONG     => TypeInference.CastToLong
147             case ObjectType.FLOAT    => TypeInference.CastToFloat
148             case ObjectType.DOUBLE   => TypeInference.CastToDouble
149             case ObjectType.BOOLEAN  => TypeInference.CastToBoolean
150             case ObjectType.DATE     => FunctionTransform("datetime(", ")")
151             case ObjectType.UUID     => TypeInference.IdentityTransform
152             case ObjectType.LIST     => FunctionTransform(s"parseList('${bindings(1)}',", ")")
153             case ObjectType.MAP      => FunctionTransform(s"parseMap('${bindings(1)}->${bindings(2)}',", ")")
154             case ObjectType.BYTES    => TypeInference.IdentityTransform
155             case ObjectType.GEOMETRY =>
156               bindings.drop(1).headOption.getOrElse(ObjectType.GEOMETRY) match {
157                 case ObjectType.POINT               => FunctionTransform("point(", ")")
158                 case ObjectType.LINESTRING          => FunctionTransform("linestring(", ")")
159                 case ObjectType.POLYGON             => FunctionTransform("polygon(", ")")
160                 case ObjectType.MULTIPOINT          => FunctionTransform("multipoint(", ")")
161                 case ObjectType.MULTILINESTRING     => FunctionTransform("multilinestring(", ")")
162                 case ObjectType.MULTIPOLYGON        => FunctionTransform("multipolygon(", ")")
163                 case ObjectType.GEOMETRY_COLLECTION => FunctionTransform("geometrycollection(", ")")
164                 case _                              => FunctionTransform("geometry(", ")")
165               }
166 
167             case _ => throw new IllegalStateException(s"Unexpected binding: $bindings")
168           }
169           i += 1
170           BasicField(d.getLocalName, Some(TryExpression(Expression(transform.apply(i)), LiteralNull)))
171         }
172 
173         val options = DelimitedTextOptions(Some(1), CharNotSpecified, CharNotSpecified, None,
174           SimpleFeatureValidator.default, Seq.empty, ParseMode.Default, ErrorMode(), StandardCharsets.UTF_8)
175 
176         val config = configConvert.to(converterConfig)
177             .withFallback(fieldConvert.to(fields.toSeq))
178             .withFallback(optsConvert.to(options))
179             .toConfig
180 
181         (sft.getOrElse(schema), config)
182       }
183     }
184   }

219       def option[T](key: String, reader: ConfigReader[T]): Either[ConfigReaderFailures, Option[T]] = {
220         val value = cur.atKeyOrUndefined(key)

224       def optionalChar(key: String): Either[ConfigReaderFailures, OptionalChar] = {
225         val value = cur.atKeyOrUndefined(key)
226         if (value.isUndefined) { Right(CharNotSpecified) } else {
227           PrimitiveConvert.charConfigReader.from(value) match {
228             case Right(c) => Right(CharEnabled(c))
229             case Left(failures) =>
230               if (PrimitiveConvert.stringConfigReader.from(value).right.exists(_.isEmpty)) {
231                 Right(CharDisabled)
232               } else {
233                 Left(failures)
234               }
235           }
236         }
237       }
238 
239       for {
240         skipLines <- option("skip-lines", PrimitiveConvert.intConfigReader).right
241         quote     <- optionalChar("quote").right
242         escape    <- optionalChar("escape").right
243         delimiter <- option("delimiter", PrimitiveConvert.charConfigReader).right
244       } yield {
245         DelimitedTextOptions(skipLines, quote, escape, delimiter, validators, reporters, parseMode, errorMode, encoding)
246       }
247     }
