59   private def infer(
60       format: CSVFormat,
61       lines: Seq[String],
62       sft: Option[SimpleFeatureType]): Try[(SimpleFeatureType, Config)] = {
63     // : Seq[List[String]]
64     val rows = lines.flatMap { line =>
65       if (line.isBlank) { None } else {
66         Try(format.parse(new StringReader(line)).iterator().next.iterator.asScala.toList).toOption
67       }
68     }
69     val counts = rows.map(_.length).distinct
70     // try to verify that we actually have a delimited file
71     // ensure that some lines parsed, that there were at most 3 different col counts, and that there were at least 2 cols
72     if (counts.isEmpty || counts.lengthCompare(3) > 0 || counts.max < 2) {
73       val f = if (format == Formats.Tabs) { "tsv" } else if (format == Formats.Quoted) { "quoted csv" }  else { "csv" }
74       return Failure(new RuntimeException(s"Not a valid delimited text file using format: $f"))
75     }
76     Try {
77       val names = sft match {
78         case Some(s) =>
79           s.getAttributeDescriptors.asScala.map(_.getLocalName)
80 
81         case None =>
82           val values = rows.head.map(v => PathWithValues("", Seq(v)))
83           if (TypeInference.infer(values, Left("")).types.forall(_.inferredType.typed == ObjectType.STRING)) {
84             // assume the first row is headers
85             rows.head
86           } else {
87             Seq.empty
88           }
89       }
90       val nameIter = names.iterator
91       val pathsAndValues = Seq.tabulate(counts.max) { col =>
92         val values = rows.drop(1).map(vals => if (vals.lengthCompare(col) > 0) { vals(col) } else { null })
93         PathWithValues(if (nameIter.hasNext) { nameIter.next } else { "" }, values)
94       }
95       val inferred =
96         TypeInference.infer(pathsAndValues,
97           sft.toRight(s"inferred-${if (format == Formats.Tabs) { "tsv" } else { "csv" }}"))
98       val converterConfig = DelimitedTextConfig(typeToProcess, None, formats.find(_._2 == format).get._1,
99         Some(Expression("md5(string2bytes($0))")), Map.empty, Map.empty)
100 
101       val fields = {
102         var i = 0 // 0 is the whole record
103         inferred.types.map { case TypeWithPath(_, inferredType) =>
104           i += 1
105           BasicField(inferredType.name, Some(Expression(inferredType.transform(i))))
106         }
107       }
108 
109       val options = DelimitedTextOptions(None, CharNotSpecified, CharNotSpecified, None,
110         SimpleFeatureValidator.default, ParseMode.Default, ErrorMode(), StandardCharsets.UTF_8)
111 
112       val config = configConvert.to(converterConfig)
113           .withFallback(fieldConvert.to(fields.toSeq))
114           .withFallback(optsConvert.to(options))
115           .toConfig
116 
117       (inferred.sft, config)
118     }
119   }

121   private def magic(lines: Seq[String], sft: Option[SimpleFeatureType]): Try[(SimpleFeatureType, Config)] = {
122     if (!lines.head.startsWith("id,")) { Failure(new RuntimeException("Not a geomesa delimited export")) } else {
123       Try {
124         val spec = WithClose(Formats.QuotedMinimal.parse(new StringReader(lines.head.drop(3)))) { result =>
125           result.iterator().next.asScala.mkString(",")
126         }
127         val schema = SimpleFeatureTypes.createType("", spec)
128 
129         val converterConfig =
130           DelimitedTextConfig(typeToProcess, None, formats.find(_._2 == Formats.QuotedMinimal).get._1,
131             Some(Expression("$1")), Map.empty, Map.empty)
132 
133         var i = 1 // 0 is the whole record, 1 is the id
134         val fields = schema.getAttributeDescriptors.asScala.map { d =>
135           val bindings = ObjectType.selectType(d)
136           val transform = bindings.head match {
137             case ObjectType.STRING   => TypeInference.IdentityTransform
138             case ObjectType.INT      => TypeInference.CastToInt
139             case ObjectType.LONG     => TypeInference.CastToLong
140             case ObjectType.FLOAT    => TypeInference.CastToFloat
141             case ObjectType.DOUBLE   => TypeInference.CastToDouble
142             case ObjectType.BOOLEAN  => TypeInference.CastToBoolean
143             case ObjectType.DATE     => FunctionTransform("datetime(", ")")
144             case ObjectType.UUID     => TypeInference.IdentityTransform
145             case ObjectType.LIST     => FunctionTransform(s"parseList('${bindings(1)}',", ")")
146             case ObjectType.MAP      => FunctionTransform(s"parseMap('${bindings(1)}->${bindings(2)}',", ")")
147             case ObjectType.BYTES    => TypeInference.IdentityTransform
148             case ObjectType.GEOMETRY =>
149               bindings.drop(1).headOption.getOrElse(ObjectType.GEOMETRY) match {
150                 case ObjectType.POINT               => FunctionTransform("point(", ")")
151                 case ObjectType.LINESTRING          => FunctionTransform("linestring(", ")")
152                 case ObjectType.POLYGON             => FunctionTransform("polygon(", ")")
153                 case ObjectType.MULTIPOINT          => FunctionTransform("multipoint(", ")")
154                 case ObjectType.MULTILINESTRING     => FunctionTransform("multilinestring(", ")")
155                 case ObjectType.MULTIPOLYGON        => FunctionTransform("multipolygon(", ")")
156                 case ObjectType.GEOMETRY_COLLECTION => FunctionTransform("geometrycollection(", ")")
157                 case _                              => FunctionTransform("geometry(", ")")
158               }
159 
160             case _ => throw new IllegalStateException(s"Unexpected binding: $bindings")
161           }
162           i += 1
163           BasicField(d.getLocalName, Some(TryExpression(Expression(transform.apply(i)), LiteralNull)))
164         }
165 
166         val options = DelimitedTextOptions(Some(1), CharNotSpecified, CharNotSpecified, None,
167           SimpleFeatureValidator.default, ParseMode.Default, ErrorMode(), StandardCharsets.UTF_8)
168 
169         val config = configConvert.to(converterConfig)
170             .withFallback(fieldConvert.to(fields.toSeq))
171             .withFallback(optsConvert.to(options))
172             .toConfig
173 
174         (sft.getOrElse(schema), config)
175       }
176     }
177   }
