24 class XmlCompositeConverterFactory extends SimpleFeatureConverterFactory with LazyLogging {
25 
26   import XmlCompositeConverterFactory.TypeToProcess
27 
28   import scala.collection.JavaConverters._
29 
30   override def apply(sft: SimpleFeatureType, conf: Config): Option[SimpleFeatureConverter] = {
31     if (!conf.hasPath("type") || conf.getString("type") != TypeToProcess) { None } else {
32       val defaults =
33         AbstractConverterFactory.standardDefaults(conf, logger)
34             .withFallback(ConfigFactory.load("xml-converter-defaults"))
35       try {
36         implicit val configConvert: ConverterConfigConvert[XmlConfig] = XmlConfigConvert
37         implicit val optionsConvert: ConfigConvert[XmlOptions] = XmlConverterFactory.XmlOptionsConvert
38         val xsd = ConfigSource.fromConfig(defaults).loadOrThrow[XmlConfig].xsd
39         val options = ConfigSource.fromConfig(defaults).loadOrThrow[XmlOptions]
40         val typeToProcess = ConfigValueFactory.fromAnyRef(XmlConverterFactory.TypeToProcess)
41         val delegates = defaults.getConfigList("converters").asScala.map { base =>
42           val conf = base.withFallback(defaults).withValue("type", typeToProcess)
43           val converter = SimpleFeatureConverter(sft, conf) match {
44             case c: ParsingConverter[Element] => c
45             case c => throw new IllegalArgumentException(s"Expected XmlConverter but got: $c")
46           }
47           val predicate = Predicate(conf.getString("predicate"))
48           (predicate, converter)
49         }
50         Some(new XmlCompositeConverter(sft, xsd, options.encoding, options.lineMode, options.errorMode, delegates.toSeq))
51       } catch {
52         case NonFatal(e) => throw new IllegalArgumentException(s"Invalid configuration: ${e.getMessage}")
53       }
54     }
55   }
56 }
