18 trait SimpleFeatureSpatialIndex extends SpatialIndex[SimpleFeature] {
19 
20   import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
21 
22   def sft: SimpleFeatureType
23 
24   /**
25    * Query based on a geotools filter
26    *
27    * @param filter filter
28    * @return
29    */
30   def query(filter: Filter): Iterator[SimpleFeature] = {
31     if (filter == Filter.INCLUDE) { query() } else {
32       val geometries = FilterHelper.extractGeometries(filter, sft.getGeomField, intersect = false)
33       if (geometries.isEmpty) { query().filter(filter.evaluate) } else {
34         val env = geometries.values.head.getEnvelopeInternal
35         geometries.values.tail.foreach(g => env.expandToInclude(g.getEnvelopeInternal))
36         query(env.getMinX, env.getMinY, env.getMaxX, env.getMaxY).filter(filter.evaluate)
37       }
38     }
39   }
40 }
