41   override def visit(filter: And, data: scala.Any): AnyRef = {
42     val children = filter.getChildren
43 
44     val query = children.asScala.map { f =>
45       f.accept(this, null) match {

50     if (query.exists(_.isInstanceOf[All[_]])) {
51       new cqquery.simple.All(classOf[SimpleFeature])
52     } else {
53       new cqquery.logical.And[SimpleFeature](query.asJava)

79   override def visit(filter: Not, data: scala.Any): AnyRef = {

83       case q: Query[SimpleFeature] => q

91     } else {
92       new cqquery.logical.Not[SimpleFeature](subquery)
93     }
94   }

289     val name = getAttribute(filter)
290     val attribute: Attribute[SimpleFeature, Any] = lookup.lookup[Any](name)
291     val literals = Iterator(filter.getExpression1, filter.getExpression2).collect {
292       case lit: Literal => FastConverter.evaluate(lit, attribute.getAttributeType)
293     }
294     if (literals.hasNext) {
295       val value = literals.next()
296       // TODO: could this be done better?
297       // may not be as big an issue as PropertyIsNull, as I'm not
298       // even sure how to build this filter in (E)CQL
299       new cqquery.logical.Not(new cqquery.simple.Equal(attribute, value))
300     } else {
301       throw new RuntimeException(s"Can't parse not equal to values: ${FilterHelper.toString(filter)}")
302     }
303   }
