20 /**
21   * Spatial index that breaks up space into discrete buckets to index points
22   *
23   * Does not support non-point inserts
24   *
25   * @param xBuckets number of x buckets
26   * @param yBuckets number of y buckets
27   * @param extents area to be indexed
28   * @tparam T index value binding
29   */
30 class BucketIndex[T](
31     val xBuckets: Int = 360,
32     val yBuckets: Int = 180,
33     val extents: Envelope = new Envelope(-180.0, 180.0, -90.0, 90.0)
34   ) extends SpatialIndex[T] with LazyLogging {
35 
36   // create the buckets up front to avoid having to synchronize the whole array
37   // we use a ConcurrentHashMap, which gives us iterators that aren't affected by modifications to the backing map
38   private val buckets = Array.fill(xBuckets, yBuckets)(new ConcurrentHashMap[String, T]())
39 
40   private val gridSnap = new GridSnap(extents, xBuckets, yBuckets)
41 
42   override def insert(geom: Geometry, key: String, value: T): Unit = {
43     val pt = geom.asInstanceOf[Point]
44     val i = snapX(pt.getX)
45     val j = snapY(pt.getY)
46     buckets(i)(j).put(key, value)
47   }
48 
49   override def remove(geom: Geometry, key: String): T = {
50     val pt = geom.asInstanceOf[Point]
51     val i = snapX(pt.getX)
52     val j = snapY(pt.getY)
53     buckets(i)(j).remove(key)
54   }
55 
56   override def get(geom: Geometry, key: String): T = {
57     val pt = geom.asInstanceOf[Point]
58     val i = snapX(pt.getX)
59     val j = snapY(pt.getY)
60     buckets(i)(j).get(key)
61   }
62 
63   override def query(xmin: Double, ymin: Double, xmax: Double, ymax: Double): Iterator[T] =
64     new BucketIterator(snapX(xmin), snapX(xmax), snapY(ymin), snapY(ymax))
65 
66   override def query(): Iterator[T] = new BucketIterator(0, xBuckets - 1, 0, yBuckets - 1)
67 
68   override def size(): Int = {
69     var size = 0
70     var i = 0
71     while (i < xBuckets) {
72       var j = 0
73       while (j < yBuckets) {
74         size += buckets(i)(j).size()
75         j += 1
76       }
77       i += 1
78     }
79     size
80   }
81 
82   override def clear(): Unit = {
83     var i = 0
84     while (i < xBuckets) {
85       var j = 0
86       while (j < yBuckets) {
87         buckets(i)(j).clear()
88         j += 1
89       }
90       i += 1
91     }
92   }
93 
94   private def snapX(x: Double): Int = {
95     val i = gridSnap.i(x)
96     if (i != -1) { i } else if (x < extents.getMinX) { 0 } else { xBuckets - 1 }
97   }
98 
99   private def snapY(y: Double): Int = {
100     val j = gridSnap.j(y)
101     if (j != -1) { j } else if (y < extents.getMinY) { 0 } else { yBuckets - 1 }
102   }
103 
104   /**
105     * Iterator over a range of buckets
106     */
107   private class BucketIterator(mini: Int, maxi: Int, minj: Int, maxj: Int) extends Iterator[T] {
108 
109     private var i = mini
110     private var j = minj
111     private var iter = buckets(i)(j).values.iterator() // note: `.values` is a cached view
112 
113     @tailrec
114     override final def hasNext: Boolean = iter.hasNext || {
115       if (i == maxi && j == maxj) { false } else {
116         if (j < maxj) {
117           j += 1
118         } else {
119           j = minj
120           i += 1
121         }
122         iter = buckets(i)(j).values.iterator() // note: `.values` is a cached view
123         hasNext
124       }
125     }
126 
127     override def next(): T = iter.next()
128   }
129 }
