52   override def createSchema(featureType: SimpleFeatureType): Unit = {
53     namesToEngine.computeIfAbsent(featureType.getTypeName, _ => {
54       val geo = if (!useGeoIndex) { Seq.empty } else {
55         Seq((featureType.getGeometryDescriptor.getLocalName, CQIndexType.GEOMETRY))
56       }
57       val attributes = CQIndexType.getDefinedAttributes(featureType) ++ geo
58       new GeoCQEngine(featureType, attributes)
59     })
60   }
