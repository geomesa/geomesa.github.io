18 trait CompQueryBuilder {
19   // The type the product queries process, and its class object
20   type VALUE <: Comparable[VALUE]
21   val valueClass: Class[VALUE]    // needed to interact with gt's Java API.
22   type ATTRIBUTE = Attribute[SimpleFeature, VALUE]
23   // The data needed to build a query, often expressed in terms of VALUE
24   type DATA
25   // The type of query to produce
26   type QUERY <: SimpleQuery[SimpleFeature, VALUE]
27 
28   protected [this] def mkQuery(attr: ATTRIBUTE, data: DATA): QUERY
29 
30   def apply(name: String, data: DATA)(implicit lookup: SFTAttributes): QUERY = {
31     val attr = lookup.lookup[VALUE](name)
32     mkQuery(attr, data)
33   }
34 }

36 trait GTQueryBuilder extends CompQueryBuilder {

55 trait BetweenQueryBuilder extends CompQueryBuilder {
56 
57   type QUERY = Between[SimpleFeature, VALUE]
58   type DATA = (VALUE, VALUE)
59   // GeoTools PropertyIsBetween is inclusive at both ends
60   val lowerInclusive = true
61   val upperInclusive = true
62 
63   protected [this] def mkQuery(attr: ATTRIBUTE, data: DATA): QUERY =
64     new Between(attr, data._1, lowerInclusive, data._2, upperInclusive)
65 }
