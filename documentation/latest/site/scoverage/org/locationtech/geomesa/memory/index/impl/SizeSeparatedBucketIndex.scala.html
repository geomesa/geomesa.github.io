20 /**
21   * Creates an index suitable for geometries with extents. The index is broken up into tiers, where each
22   * geometry is assigned a tier based on its envelope size. When querying, each tier must be evaluated, so
23   * if possible tiers should be matched closely to the envelopes of the entries.
24   *
25   * Values are indexed by the centroid of their envelope. When querying, the bounding box is expanded based
26   * on the max envelope size of the tier, to ensure that all potential results are found. Thus, the false
27   * positive rate tends to go up with larger tiers, and post-filtering is recommended.
28   *
29   * @param sizes (x, y) max envelope size for each tier
30   * @param xBucketMultiplier multiplier for number of x buckets to create per tier
31   * @param yBucketMultiplier multiplier for number of y buckets to create per tier
32   * @param extents total area being indexed
33   * @tparam T item type
34   */
35 class SizeSeparatedBucketIndex[T](sizes: Seq[(Double, Double)] = SizeSeparatedBucketIndex.DefaultTiers,
36                                   xBucketMultiplier: Double = 1,
37                                   yBucketMultiplier: Double = 1,
38                                   extents: Envelope = new Envelope(-180.0, 180.0, -90.0, 90.0))
39     extends SpatialIndex[T] with StrictLogging {
40 
41   // TODO https://geomesa.atlassian.net/browse/GEOMESA-2323 better anti-meridian handling
42 
43   require(sizes.nonEmpty, "No valid tier sizes specified")
44   require(sizes.lengthCompare(1) == 0 ||
45       sizes.sliding(2).forall { case Seq((x1, y1), (x2, y2)) => x1 <= x2 && y1 <= y2 },
46     "Tiers must be ordered by increasing size")
47 
48   // note: for point ops, we always use the first (smallest) tier
49 
50   private val tiers = sizes.map { case (width, height) =>
51     val xSize = math.ceil(extents.getWidth * xBucketMultiplier / width).toInt
52     val ySize = math.ceil(extents.getHeight * yBucketMultiplier / height).toInt
53     // create the buckets up front to avoid having to synchronize the whole array
54     // we use a ConcurrentHashMap, which gives us iterators that aren't affected by modifications to the backing map
55     val buckets = Array.fill(xSize, ySize)(new ConcurrentHashMap[String, T]())
56     logger.debug(s"Creating tier for size ($width $height) with buckets [${xSize}x$ySize]")
57     new Tier(width, height, buckets, new GridSnap(extents, xSize, ySize))
58   }
59

61   override def insert(geom: Geometry, key: String, value: T): Unit = {
62     val envelope = geom.getEnvelopeInternal
63     val tier = selectTier(envelope)
64     // volatile reads should be cheaper than writes, so only update the variable if necessary
65     if (tier.empty) {
66       tier.empty = false
67     }
68     tier.bucket(envelope).put(key, value)
69   }
70 
71   override def remove(geom: Geometry, key: String): T = {
72     val envelope = geom.getEnvelopeInternal
73     selectTier(envelope).bucket(envelope).remove(key)
74   }
75 
76   override def get(geom: Geometry, key: String): T = {
77     val envelope = geom.getEnvelopeInternal
78     selectTier(envelope).bucket(envelope).get(key)
79   }
80 
81   override def query(xmin: Double, ymin: Double, xmax: Double, ymax: Double): Iterator[T] =
82     tiers.iterator.flatMap(_.iterator(xmin, ymin, xmax, ymax))
83 
84   override def query(): Iterator[T] = query(extents.getMinX, extents.getMinY, extents.getMaxX, extents.getMaxY)
85 
86   override def size(): Int = {
87     var size = 0
88     tiers.foreach(tier => size += tier.size())
89     size
90   }
91 
92   override def clear(): Unit = tiers.foreach(_.clear())
93 
94   private def selectTier(envelope: Envelope): Tier = {
95     val width = envelope.getWidth
96     val height = envelope.getHeight
97     tiers.find(t => t.maxSizeX >= width && t.maxSizeY >= height).getOrElse {
98       throw new IllegalArgumentException(s"Envelope $envelope exceeds the max tier size ${sizes.last}")
99     }
100   }
101 
102   private class Tier(val maxSizeX: Double,
103                      val maxSizeY: Double,
104                      buckets: Array[Array[ConcurrentHashMap[String, T]]],
105                      gridSnap: GridSnap) {
106 
107     // we can safely use volatile instead of synchronized here, as this is a primitive boolean whose
108     // state doesn't depend on its own value
109     @volatile
110     var empty: Boolean = true
111 
112     private val maxX = buckets.length - 1
113     private val maxY = buckets(0).length - 1
114 
115     def bucket(x: Double, y: Double): ConcurrentHashMap[String, T] = buckets(snapX(x))(snapY(y))
116 
117     // the bucket is selected based on the envelope centroid
118     def bucket(envelope: Envelope): ConcurrentHashMap[String, T] =
119       buckets(snapX((envelope.getMinX + envelope.getMaxX) / 2.0))(snapY((envelope.getMinY + envelope.getMaxY) / 2.0))
120 
121     def iterator(xmin: Double, ymin: Double, xmax: Double, ymax: Double): Iterator[T] =
122       if (empty) { Iterator.empty } else { new TierIterator(xmin, ymin, xmax, ymax) }
123 
124     def size(): Int = {
125       if (empty) { 0 } else {
126         var size = 0
127         var i = 0
128         while (i <= maxX) {
129           var j = 0
130           while (j <= maxY) {
131             size += buckets(i)(j).size()
132             j += 1
133           }
134           i += 1
135         }
136         size
137       }
138     }
139 
140     def clear(): Unit = {
141       var i = 0
142       while (i <= maxX) {
143         var j = 0
144         while (j <= maxY) {
145           buckets(i)(j).clear()
146           j += 1
147         }
148         i += 1
149       }
150     }
151 
152     private def snapX(x: Double): Int = {
153       val i = gridSnap.i(x)
154       if (i != -1) { i } else if (x < extents.getMinX) { 0 } else { maxX }
155     }
156 
157     private def snapY(y: Double): Int = {
158       val j = gridSnap.j(y)
159       if (j != -1) { j } else if (y < extents.getMinY) { 0 } else { maxY }
160     }
161 
162     /**
163       * Iterator over a range of buckets
164       */
165     class TierIterator (xmin: Double, ymin: Double, xmax: Double, ymax: Double) extends Iterator[T] {
166 
167       private val maxi = snapX(xmax + maxSizeX)
168       private val minj = snapY(ymin - maxSizeY)
169       private val maxj = snapY(ymax + maxSizeY)
170 
171       private var i = snapX(xmin - maxSizeX)
172       private var j = minj
173       private var iter = buckets(i)(j).values.iterator() // note: `.values` is a cached view
174 
175       @tailrec
176       override final def hasNext: Boolean = iter.hasNext || {
177         if (i == maxi && j == maxj) { false } else {
178           if (j < maxj) {
179             j += 1
180           } else {
181             j = minj
182             i += 1
183           }
184           iter = buckets(i)(j).values.iterator() // note: `.values` is a cached view
185           hasNext
186         }
187       }
188 
189       override def next(): T = iter.next()
190     }
191   }
