57 
58     private val tempFeatureIds = new AtomicLong(0)
59     private val currentFeature: ScalaSimpleFeature = new ScalaSimpleFeature(engine.sft, "")
60     private var queued = false
61 
62     override def getFeatureType: SimpleFeatureType = engine.sft

64     override def hasNext: Boolean = false

66     override def next(): SimpleFeature = {
67       var i = 0
68       while (i < engine.sft.getAttributeCount) {
69         currentFeature.setAttributeNoConvert(i, null)
70         i += 1
71       }
72       currentFeature.getUserData.clear()
73       currentFeature.setId(tempFeatureIds.getAndIncrement().toString)
74       queued = true
75       currentFeature
76     }
77 
78     override def write(): Unit = {
79       if (!queued) {
80         throw new IllegalStateException("Must call 'next' before 'write'")
81       }
82       val values = Array.tabulate(engine.sft.getAttributeCount)(currentFeature.getAttribute)
83       val id = if (currentFeature.getUserData.containsKey(Hints.PROVIDED_FID)) {
84         currentFeature.getUserData.get(Hints.PROVIDED_FID).toString
85       } else if (currentFeature.getUserData.containsKey(Hints.USE_PROVIDED_FID) &&
86           currentFeature.getUserData.get(Hints.USE_PROVIDED_FID).asInstanceOf[Boolean]) {
87         currentFeature.getID
88       } else {
89         UUID.randomUUID.toString
90       }
91       val userData = new java.util.HashMap[AnyRef, AnyRef](currentFeature.getUserData)
92       engine.update(new ImmutableSimpleFeature(engine.sft, id, values, userData))
93       queued = false
94     }
95 
96     override def remove(): Unit = throw new NotImplementedError()
97 
98     override def close(): Unit = {}
99   }
