84           currentFeature.getUserData.get(Hints.USE_PROVIDED_FID).asInstanceOf[Boolean]) {
85         currentFeature.getID

88       }
89       val userData = new java.util.HashMap[AnyRef, AnyRef](currentFeature.getUserData)
90       engine.update(new ImmutableSimpleFeature(engine.sft, id, values, userData))
91       queued = false
92     }
