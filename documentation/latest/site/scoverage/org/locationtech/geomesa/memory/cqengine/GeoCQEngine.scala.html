33 class GeoCQEngine(val sft: SimpleFeatureType,
34                   attributes: Seq[(String, CQIndexType)],
35                   enableFidIndex: Boolean = false,
36                   geoIndexType: GeoIndexType = GeoIndexType.Bucket,
37                   geoIndexParam: Option[_ <: GeoIndexParams] = Option.empty,
38                   dedupe: Boolean = true) extends LazyLogging {
39 
40   protected val cqcache: IndexedCollection[SimpleFeature] = new ConcurrentIndexedCollection[SimpleFeature]()
41 
42   def this(sft: SimpleFeatureType,
43            attributes: Seq[(String, CQIndexType)],
44            enableFidIndex: Boolean,
45            geomResolution: (Int, Int),
46            dedupe: Boolean) = {
47     this(sft, attributes, enableFidIndex, GeoIndexType.Bucket, Option.apply(new BucketIndexParam(geomResolution._1, geomResolution._2).asInstanceOf[GeoIndexParams]), dedupe)
48   }
49 
50 
51   def this(sft: SimpleFeatureType,
52            attributes: Seq[(String, CQIndexType)],
53            enableFidIndex: Boolean,
54            geomResolution: (Int, Int)) = {
55     this(sft, attributes, enableFidIndex, geomResolution, true)
56   }
57 
58   addIndices()
59 
60   /**
61    * Insert a simple feature into the index
62    *
63    * @param feature feature
64    */
65   def insert(feature: SimpleFeature): Unit = cqcache.add(feature)
66 
67   /**
68    * Insert multiple features into the index
69    *
70    * @param features features
71    */
72   def insert(features: Iterable[SimpleFeature]): Unit = cqcache.addAll(features.asJavaCollection)
73 
74   /**
75    * Update an existing feature
76    *
77    * @param feature new feature
78    * @return old feature, if any
79    */
80   def update(feature: SimpleFeature): SimpleFeature = {
81     val existing = remove(feature.getID)
82     cqcache.add(feature)
83     existing
84   }
85 
86   /**
87    * Remove an existing feature
88    *
89    * @param id feature ID
90    * @return removed feature, if any
91    */
92   def remove(id: String): SimpleFeature = {
93     val existing = get(id)
94     if (existing != null) {
95       cqcache.remove(existing)
96     }
97     existing
98   }
99 
100   /**
101    * Return a feature by feature ID
102    *
103    * @param id feature id
104    * @return feature, if it exists
105    */
106   def get(id: String): SimpleFeature = {
107     // if this gets used, set enableFidIndex=true
108     cqcache.retrieve(new Equal(SFTAttributes.fidAttribute, id)).asScala.headOption.orNull
109   }
110 
111   /**
112    * Query features
113    *
114    * @param filter filter
115    * @return
116    */
117   def query(filter: Filter): Iterator[SimpleFeature] = {
118     val query = filter.accept(new CQEngineQueryVisitor(sft), null).asInstanceOf[Query[SimpleFeature]]
119     val iter = if (dedupe) {
120       val dedupeOpt = QueryFactory.deduplicate(DeduplicationStrategy.LOGICAL_ELIMINATION)
121       cqcache.retrieve(query, QueryFactory.queryOptions(dedupeOpt)).iterator()
122     } else {
123       cqcache.retrieve(query).iterator()
124     }
125     if (query.isInstanceOf[All[_]]) { iter.asScala.filter(filter.evaluate) } else { iter.asScala }
126   }
127 
128   def size(): Int = cqcache.size()
129   def clear(): Unit = cqcache.clear()
130 
131   private def addIndices(): Unit = {
132 
133     import CQIndexType._
134 
135     if (enableFidIndex) {
136       cqcache.addIndex(HashIndex.onAttribute(SFTAttributes.fidAttribute))
137     }
138 
139     // track attribute names in case there are duplicates...
140     val names = scala.collection.mutable.Set.empty[String]
141 
142     // works around casting to T <: Comparable[T]
143     def navigableIndex[T <: Comparable[T]](name: String): NavigableIndex[T, SimpleFeature] = {
144       val attribute = new SimpleFeatureAttribute(classOf[Comparable[_]], sft, name)
145       NavigableIndex.onAttribute(attribute.asInstanceOf[Attribute[SimpleFeature, T]])
146     }
147 
148     attributes.foreach { case (name, indexType) =>
149       if (indexType != NONE && names.add(name)) {
150         val descriptor = sft.getDescriptor(name)
151         require(descriptor != null, s"Could not find descriptor for $name in schema ${sft.getTypeName}")
152         val binding = descriptor.getType.getBinding
153         val index = indexType match {
154           case RADIX | DEFAULT if classOf[String].isAssignableFrom(binding) =>
155               RadixTreeIndex.onAttribute(new SimpleFeatureAttribute(classOf[String], sft, name))
156 
157           case GEOMETRY | DEFAULT if classOf[Geometry].isAssignableFrom(binding) =>
158               val attribute = new SimpleFeatureAttribute(binding.asInstanceOf[Class[Geometry]], sft, name)
159               GeoIndexFactory.onAttribute(sft, attribute, geoIndexType, geoIndexParam);
160 
161           case DEFAULT if classOf[UUID].isAssignableFrom(binding) =>
162               UniqueIndex.onAttribute(new SimpleFeatureAttribute(classOf[UUID], sft, name))
163 
164           case DEFAULT if classOf[java.lang.Boolean].isAssignableFrom(binding) =>
165               HashIndex.onAttribute(new SimpleFeatureAttribute(classOf[java.lang.Boolean], sft, name))
166 
167           case NAVIGABLE | DEFAULT if classOf[Comparable[_]].isAssignableFrom(binding) =>
168             navigableIndex(name)
169 
170           case UNIQUE =>
171             UniqueIndex.onAttribute(new SimpleFeatureAttribute(classOf[AnyRef], sft, name))
172 
173           case HASH =>
174             HashIndex.onAttribute(new SimpleFeatureAttribute(classOf[AnyRef], sft, name))
175 
176           case t =>
177               throw new IllegalArgumentException(s"No CQEngine binding available for type $t and class $binding")
178         }
179         cqcache.addIndex(index)
180       }
181     }
182   }
