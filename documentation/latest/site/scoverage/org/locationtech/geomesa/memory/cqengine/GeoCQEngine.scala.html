34 class GeoCQEngine(val sft: SimpleFeatureType,
35                   attributes: Seq[(String, CQIndexType)],
36                   enableFidIndex: Boolean = false,
37                   geoIndexType: GeoIndexType = GeoIndexType.Bucket,
38                   geoIndexParam: Option[_ <: GeoIndexParams] = Option.empty,
39                   dedupe: Boolean = true) extends SimpleFeatureIndex with LazyLogging {
40 
41   protected val cqcache: IndexedCollection[SimpleFeature] = new ConcurrentIndexedCollection[SimpleFeature]()
42 
43   def this(sft: SimpleFeatureType,
44            attributes: Seq[(String, CQIndexType)],
45            enableFidIndex: Boolean,
46            geomResolution: (Int, Int),
47            dedupe: Boolean) = {
48     this(sft, attributes, enableFidIndex, GeoIndexType.Bucket, Option.apply(new BucketIndexParam(geomResolution._1, geomResolution._2).asInstanceOf[GeoIndexParams]), dedupe)
49   }
50 
51 
52   def this(sft: SimpleFeatureType,
53            attributes: Seq[(String, CQIndexType)],
54            enableFidIndex: Boolean,
55            geomResolution: (Int, Int)) = {
56     this(sft, attributes, enableFidIndex, geomResolution, true)
57   }
58 
59   addIndices()
60 
61   // methods from SimpleFeatureIndex
62 
63   override def insert(feature: SimpleFeature): Unit = cqcache.add(feature)
64 
65   override def insert(features: Iterable[SimpleFeature]): Unit = cqcache.addAll(features.asJavaCollection)
66 
67   override def update(feature: SimpleFeature): SimpleFeature = {
68     val existing = remove(feature.getID)
69     cqcache.add(feature)
70     existing
71   }
72 
73   override def remove(id: String): SimpleFeature = {
74     val existing = get(id)
75     if (existing != null) {
76       cqcache.remove(existing)
77     }
78     existing
79   }
80 
81   override def get(id: String): SimpleFeature = {
82     // if this gets used, set enableFidIndex=true
83     cqcache.retrieve(new Equal(SFTAttributes.fidAttribute, id)).asScala.headOption.orNull
84   }
85 
86   override def query(filter: Filter): Iterator[SimpleFeature] = {
87     val query = filter.accept(new CQEngineQueryVisitor(sft), null).asInstanceOf[Query[SimpleFeature]]
88     val iter = if (dedupe) {
89       val dedupeOpt = QueryFactory.deduplicate(DeduplicationStrategy.LOGICAL_ELIMINATION)
90       cqcache.retrieve(query, QueryFactory.queryOptions(dedupeOpt)).iterator()
91     } else {
92       cqcache.retrieve(query).iterator()
93     }
94     if (query.isInstanceOf[All[_]]) { iter.asScala.filter(filter.evaluate) } else { iter.asScala }
95   }
96 
97   def size(): Int = cqcache.size()
98   def clear(): Unit = cqcache.clear()
99 
100   private def addIndices(): Unit = {
101 
102     import CQIndexType._
103 
104     if (enableFidIndex) {
105       cqcache.addIndex(HashIndex.onAttribute(SFTAttributes.fidAttribute))
106     }
107 
108     // track attribute names in case there are duplicates...
109     val names = scala.collection.mutable.Set.empty[String]
110 
111     // works around casting to T <: Comparable[T]
112     def navigableIndex[T <: Comparable[T]](name: String): NavigableIndex[T, SimpleFeature] = {
113       val attribute = new SimpleFeatureAttribute(classOf[Comparable[_]], sft, name)
114       NavigableIndex.onAttribute(attribute.asInstanceOf[Attribute[SimpleFeature, T]])
115     }
116 
117     attributes.foreach { case (name, indexType) =>
118       if (indexType != NONE && names.add(name)) {
119         val descriptor = sft.getDescriptor(name)
120         require(descriptor != null, s"Could not find descriptor for $name in schema ${sft.getTypeName}")
121         val binding = descriptor.getType.getBinding
122         val index = indexType match {
123           case RADIX | DEFAULT if classOf[String].isAssignableFrom(binding) =>
124               RadixTreeIndex.onAttribute(new SimpleFeatureAttribute(classOf[String], sft, name))
125 
126           case GEOMETRY | DEFAULT if classOf[Geometry].isAssignableFrom(binding) =>
127               val attribute = new SimpleFeatureAttribute(binding.asInstanceOf[Class[Geometry]], sft, name)
128               GeoIndexFactory.onAttribute(sft, attribute, geoIndexType, geoIndexParam);
129 
130           case DEFAULT if classOf[UUID].isAssignableFrom(binding) =>
131               UniqueIndex.onAttribute(new SimpleFeatureAttribute(classOf[UUID], sft, name))
132 
133           case DEFAULT if classOf[java.lang.Boolean].isAssignableFrom(binding) =>
134               HashIndex.onAttribute(new SimpleFeatureAttribute(classOf[java.lang.Boolean], sft, name))
135 
136           case NAVIGABLE | DEFAULT if classOf[Comparable[_]].isAssignableFrom(binding) =>
137             navigableIndex(name)
138 
139           case UNIQUE =>
140             UniqueIndex.onAttribute(new SimpleFeatureAttribute(classOf[AnyRef], sft, name))
141 
142           case HASH =>
143             HashIndex.onAttribute(new SimpleFeatureAttribute(classOf[AnyRef], sft, name))
144 
145           case t =>
146               throw new IllegalArgumentException(s"No CQEngine binding available for type $t and class $binding")
147         }
148         cqcache.addIndex(index)
149       }
150     }
151   }
