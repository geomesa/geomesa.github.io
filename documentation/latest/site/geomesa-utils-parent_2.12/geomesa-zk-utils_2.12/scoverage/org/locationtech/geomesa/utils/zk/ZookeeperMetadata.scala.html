17 class ZookeeperMetadata[T](val namespace: String, val zookeepers: String, val serializer: MetadataSerializer[T])
18     extends KeyValueStoreMetadata[T] {
19 
20   import org.locationtech.geomesa.utils.zk.ZookeeperMetadata.Root
21 
22   private val client = CuratorHelper.client(zookeepers).namespace(namespace).build()
23   client.start()
24 
25   override protected def checkIfTableExists: Boolean = true
26 
27   override protected def createTable(): Unit = {}
28 
29   override protected def createEmptyBackup(timestamp: String): ZookeeperMetadata[T] =
30     new ZookeeperMetadata(s"${namespace}_${timestamp}_bak", zookeepers, serializer)
31 
32   override protected def write(rows: Seq[(Array[Byte], Array[Byte])]): Unit = {
33     rows.foreach { case (row, value) =>
34       val path = toPath(row)
35       if (client.checkExists().forPath(path) == null) {
36         client.create().creatingParentsIfNeeded().forPath(path)
37       }
38       client.setData().forPath(path, value)
39     }
40   }
41 
42   override protected def delete(rows: Seq[Array[Byte]]): Unit = {
43     rows.foreach { row =>
44       val path = toPath(row)
45       if (client.checkExists().forPath(path) != null) {
46         client.delete().deletingChildrenIfNeeded().forPath(path)
47       }
48     }
49   }
50 
51   override protected def scanValue(row: Array[Byte]): Option[Array[Byte]] = {
52     val path = toPath(row)
53     if (client.checkExists().forPath(path) == null) { None } else {
54       Option(client.getData.forPath(path))
55     }
56   }
57 
58   override protected def scanRows(prefix: Option[Array[Byte]]): CloseableIterator[(Array[Byte], Array[Byte])] = {
59     val path = prefix.map(toPath(_, withSlash = false))
60     if (client.checkExists().forPath(Root) == null) { CloseableIterator.empty } else {
61       val all = CloseableIterator(client.getChildren.forPath(Root).iterator)
62       val filtered = path match {
63         case None => all
64         case Some(p) => all.filter(_.startsWith(p))
65       }
66       filtered.map { f =>
67         val bytes = f.getBytes(StandardCharsets.UTF_8)
68         (bytes, client.getData.forPath(toPath(bytes)))
69       }
70     }
71   }
72 
73   override def close(): Unit = {
74     CloseWithLogging(client)
75     super.close()
76   }
77 
78   private def toPath(row: Array[Byte], withSlash: Boolean = true): String = {
79     val string = new String(row, StandardCharsets.UTF_8) // TODO escape?
80     if (withSlash) { "/" + string } else { string }
81   }
82 }
