47   override def getHandle(path: String): FileHandle = {
48     val p = new Path(path)
49     val fs = FileSystem.get(p.toUri, conf)
50     PathUtils.getUncompressedExtension(p.getName).toLowerCase(Locale.US) match {
51       case TAR       => new HadoopTarHandle(fs, p)
52       case ZIP | JAR => new HadoopZipHandle(fs, p)
53       case _         => new HadoopFileHandle(fs, p)
54     }
55   }
56 
57   // based on logic from hadoop FileInputFormat
58   override def interpretPath(path: String): Seq[FileHandle] = {
59     val p = new Path(path)
60     val fs = FileSystem.get(p.toUri, conf)
61     val files = fs.globStatus(p, HiddenFileFilter)
62 
63     if (files == null) {
64       throw new IllegalArgumentException(s"Input path does not exist: $path")
65     } else if (files.isEmpty) {
66       throw new IllegalArgumentException(s"Input path does not match any files: $path")
67     }
68 
69     val remaining = scala.collection.mutable.Queue(files: _*)
70     val result = ListBuffer.empty[FileHandle]
71 
72     while (remaining.nonEmpty) {
73       val file = remaining.dequeue()
74       if (file.isDirectory) {
75         if (recursive) {
76           val children = fs.listLocatedStatus(file.getPath)
77           val iter = new Iterator[LocatedFileStatus] {
78             override def hasNext: Boolean = children.hasNext
79             override def next(): LocatedFileStatus = children.next
80           }
81           remaining ++= iter.filter(f => HiddenFileFilter.accept(f.getPath))
82         }
83       } else {
84         PathUtils.getUncompressedExtension(file.getPath.getName).toLowerCase(Locale.US) match {
85           case TAR       => result += new HadoopTarHandle(fs, file.getPath)
86           case ZIP | JAR => result += new HadoopZipHandle(fs, file.getPath)
87           case _         => result += new HadoopFileHandle(fs, file.getPath)
88         }
89       }
90     }
91 
92     result.result
93   }

118   private def configureURLFactory(): Unit = synchronized {
119     if (setUrlFactory) {
120       setUrlFactory = false

131   class HadoopFileHandle(fs: FileSystem, file: Path) extends FileHandle {
132 
133     override def path: String = file.toString
134 
135     override def exists: Boolean = fs.exists(file)
136 
137     override def length: Long = if (exists) { fs.getFileStatus(file).getLen } else { 0L }
138 
139     override def open: CloseableIterator[(Option[String], InputStream)] = {
140       val is = PathUtils.handleCompression(fs.open(file), file.getName)
141       CloseableIterator.single(None -> is, is.close())
142     }
143 
144     override def write(mode: CreateMode): OutputStream = {
145       mode.validate()
146       if (mode.append) {
147         fs.append(file)
148       } else {
149         fs.create(file, mode.overwrite) // TODO do we need to hsync/hflush?
150       }
151     }
152 
153     override def delete(recursive: Boolean): Unit = {
154       if (!fs.delete(file, recursive)) {
155         throw new IOException(s"Could not delete file: $path")
156       }
157     }
158   }

160   class HadoopZipHandle(fs: FileSystem, file: Path) extends HadoopFileHandle(fs, file) {
161     override def open: CloseableIterator[(Option[String], InputStream)] = {
162       // we have to read the bytes into memory to get random access reads
163       val bytes = WithClose(PathUtils.handleCompression(fs.open(file), file.getName)) { is =>
164         IOUtils.toByteArray(is)
165       }
166       new ZipFileIterator(new ZipFile(new SeekableInMemoryByteChannel(bytes)), file.toString)
167     }
168 
169     override def write(mode: CreateMode): OutputStream =
170       factory.createArchiveOutputStream(ArchiveStreamFactory.ZIP, super.write(mode))
171   }
172 
173   class HadoopTarHandle(fs: FileSystem, file: Path) extends HadoopFileHandle(fs, file) {
174     override def open: CloseableIterator[(Option[String], InputStream)] = {
175       val uncompressed = PathUtils.handleCompression(fs.open(file), file.getName)
176       val archive: ArchiveInputStream[_ <: ArchiveEntry] =
177         factory.createArchiveInputStream(ArchiveStreamFactory.TAR, uncompressed)
178       new ArchiveFileIterator(archive, file.toString)
179     }
180 
181     override def write(mode: CreateMode): OutputStream =
182       factory.createArchiveOutputStream(ArchiveStreamFactory.TAR, super.write(mode))
183   }
