29   /**
30     * Add a resource to the given conf
31     *
32     * @param conf conf
33     * @param path resource path
34     */
35   def addResource(conf: Configuration, path: String): Unit = {
36     // use our path handling logic, which is more robust than just passing paths to the config
37     val delegate = if (PathUtils.isRemote(path)) { new HadoopDelegate(conf) } else { PathUtils }
38     val handle = delegate.getHandle(path)
39     if (!handle.exists) {
40       logger.warn(s"Could not load configuration file at: $path")
41     } else {
42       // note: foreach closes the iterator in a finally block
43       handle.open.foreach { case (name, is) =>
44         name match {
45           case None => conf.addResource(is)
46           case Some(name) => conf.addResource(is, name)
47         }
48         conf.size() // this forces a loading of the resource files, before we close our file handle
49       }
50     }
51   }
52 
53   /**
54    * Checks for a secured cluster and creates a thread to periodically renew the kerberos ticket
55    *
56    * @return
57    */
58   def kerberosTicketRenewer(): Closeable = synchronized {
59     if (krbRenewer == null) {
60       krbRenewer = new TicketLogin()
61     }
62     krbRenewer.registrations += 1
63     new KrbRegistration()
64   }
65 
66   /**
67    * Deregister a reference to the singleton kerberos ticket renewer
68    */
69   private def deregister(): Unit = synchronized {
70     krbRenewer.registrations -= 1
71     if (krbRenewer.registrations == 0) {
72       CloseWithLogging(krbRenewer)
73       krbRenewer = null
74     }
75   }
76 
77   /**
78    * Runnable class to reload tickets
79    */
80   private class TicketLogin extends Runnable with Closeable with LazyLogging {
81 
82     private val executor = ExitingExecutor(new ScheduledThreadPoolExecutor(1))
83     executor.scheduleAtFixedRate(this, 0, 10, TimeUnit.MINUTES)
84 
85     var registrations = 0
86 
87     override def run(): Unit = {
88       try {
89         logger.debug(s"Checking whether TGT needs renewing for ${UserGroupInformation.getLoginUser}")
90         logger.debug(s"Logged in from keytab? ${UserGroupInformation.getLoginUser.isFromKeytab}")
91         UserGroupInformation.getLoginUser.checkTGTAndReloginFromKeytab()
92       } catch {
93         case NonFatal(e) => logger.warn("Error checking and renewing TGT", e)
94       }
95     }
96 
97     override def close(): Unit = executor.shutdown()
98   }
99 
100   /**
101    * Ensures that each registration is only closed at most once
102    */
103   private class KrbRegistration extends Closeable {
104     private val closed = new AtomicBoolean(false)
105     override def close(): Unit = if (closed.compareAndSet(false, true)) { deregister() }
106   }
