21 class JavaTimeConverterFactory extends ConverterFactory {
22   def createConverter(source: Class[_], target: Class[_], hints: Hints): Converter = {
23     if (classOf[Date].isAssignableFrom(source) && classOf[String] == target) {
24       JavaTimeConverterFactory.DateToStringConverter
25     } else if (classOf[Date].isAssignableFrom(target)) {
26       val toDate =
27         if (classOf[String].isAssignableFrom(source)) {
28           JavaTimeConverterFactory.StringToDateConverter
29         } else if (classOf[ZonedDateTime].isAssignableFrom(source) || classOf[Instant].isAssignableFrom(source)
30             || classOf[OffsetDateTime].isAssignableFrom(source)) {
31           JavaTimeConverterFactory.TemporalToDateConverter
32         } else if (classOf[LocalDateTime].isAssignableFrom(source)) {
33           JavaTimeConverterFactory.LocalDateTimeToDateConverter
34         } else if (classOf[LocalDate].isAssignableFrom(source)) {
35           JavaTimeConverterFactory.LocalDateToDateConverter
36         } else {
37           null
38         }
39       if (toDate == null || classOf[Date] == target) {
40         toDate
41       } else {
42         val toTarget = new TemporalConverterFactory().createConverter(classOf[Date], target, hints)
43         if (toTarget == null) {
44           null
45         } else {
46           new TwoStageTemporalConverter(toDate, toTarget)
47         }
48       }
49     } else {
50       null
51     }
52   }
53 }

57   private trait DateConverter extends Converter {
58 
59     override def convert[T](source: AnyRef, target: Class[T]): T = convert(source).asInstanceOf[T]
60 
61     def convert(source: AnyRef): Date
62   }
63 
64   private object DateToStringConverter extends Converter {
65     override def convert[T](source: AnyRef, target: Class[T]): T =
66       DateParsing.formatDate(source.asInstanceOf[Date]).asInstanceOf[T]
67   }
68 
69   private object StringToDateConverter extends DateConverter {
70     override def convert(source: AnyRef): Date = DateParsing.parseDate(source.asInstanceOf[String])
71   }
72 
73   private object TemporalToDateConverter extends DateConverter {
74     override def convert(source: AnyRef): Date = {
75       val accessor = source.asInstanceOf[TemporalAccessor]
76       val millis = accessor.getLong(ChronoField.INSTANT_SECONDS) * 1000 +
77           accessor.getLong(ChronoField.MILLI_OF_SECOND)
78       new Date(millis)
79     }
80   }
81 
82   private object LocalDateToDateConverter extends DateConverter {
83     override def convert(source: AnyRef): Date =
84       TemporalToDateConverter.convert(source.asInstanceOf[LocalDate].atStartOfDay(ZoneOffset.UTC))
85   }
86 
87   private object LocalDateTimeToDateConverter extends DateConverter {
88     override def convert(source: AnyRef): Date =
89       TemporalToDateConverter.convert(source.asInstanceOf[LocalDateTime].atZone(ZoneOffset.UTC))
90   }
91 
92   private class TwoStageTemporalConverter(toDate: DateConverter, toTarget: Converter) extends Converter {
93     override def convert[T](source: AnyRef, target: Class[T]): T =
94       toTarget.convert(toDate.convert(source), target)
95   }
