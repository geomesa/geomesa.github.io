11 /**
12   * BitSet for indicating read/write mode of an index
13   */
14 object IndexMode {
15 
16   private val ReadBit  = 1 // 01
17   private val WriteBit = 2 // 10
18 
19   val Any       = new IndexMode(0)
20   val Read      = new IndexMode(ReadBit)
21   val Write     = new IndexMode(WriteBit)
22   val ReadWrite = new IndexMode(ReadBit | WriteBit)
23 
24   class IndexMode(val flag: Int) extends AnyRef {
25     def read: Boolean =  (flag & ReadBit)  != 0
26     def write: Boolean = (flag & WriteBit) != 0
27     def supports(m: IndexMode): Boolean = (this.read || !m.read) && (this.write || !m.write)
28   }
29 
30   def apply(flag: Int): IndexMode = new IndexMode(flag)
31 }
