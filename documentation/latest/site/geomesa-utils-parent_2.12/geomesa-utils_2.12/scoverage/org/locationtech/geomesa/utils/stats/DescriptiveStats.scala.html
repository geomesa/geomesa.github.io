131   override def observe(sf: SimpleFeature): Unit = {
132     val values = Array.newBuilder[Double]
133     values.sizeHint(indices.length)
134 
135     var i = 0
136     while (i < indices.length) {
137       sf.getAttribute(indices(i)) match {
138         case n: Number =>
139           val double = n.doubleValue()
140           if (double != double) {
141             return // NaN, short-circuit evaluation
142           }
143           values += double
144 
145         case null => return // short-circuit evaluation
146 
147         case n => throw new IllegalArgumentException(s"Not a number: $n")
148       }
149       i += 1
150     }
151 
152     val values_v = new SimpleMatrix(size, 1, true, values.result())
153 
154     if (_count < 1) {
155       _count = 1
156       _min.insertIntoThis(0, 0, values_v)
157       _max.insertIntoThis(0, 0, values_v)
158       _sum.insertIntoThis(0, 0, values_v)
159       _mean.insertIntoThis(0, 0, values_v)
160     } else {
161 
162       _sum += values_v
163 
164       val r = _count
165       val n = { _count += 1; _count }
166       val n_i = 1d / n
167 
168       val delta = values_v - _mean
169 
170       val A = delta * n_i
171       _mean += A
172 
173       _m4n += A * (A * A * delta * r * (n * (n - 3d) + 3d) + A * _m2n * 6d - _m3n * 4d)
174 
175       val B = values_v - _mean
176       _m3n += A * (B * delta * (n - 2d) - _m2n * 3d)
177       _m2n += delta * B
178 
179 
180       /* optimize original code (below) by special handling of diagonal and reflection about it
181        * _c2 += (delta |*| delta.T * (n_i * r))
182        */
183       val coef = n_i * r
184       var ri = 0
185       while (ri < size) {
186         _c2.set(ri, ri, _m2n.get(ri)) // c2 diagonal is equal to m2n
187         val rd = delta.get(ri)
188         var ci = ri + 1  // traverse upper diagonal
189         while (ci < size) {
190           val c2 = _c2.get(ri, ci) + rd * delta.get(ci) * coef
191           _c2.set(ri, ci, c2) // set upper diagonal
192           _c2.set(ci, ri, c2) // set lower diagonal
193           ci += 1
194         }
195         ri += 1
196       } // c2 update
197 
198       var i = 0
199       while (i < size) {
200         val v = values_v.get(i)
201         if (v > _max.get(i)) {
202           _max.set(i, v)
203         } else if (v < _min.get(i)) { // 'else if' optimization due to how min/max set when _count == 1 (below)
204           _min.set(i, v)
205         }
206         i += 1
207       } // min/max update
208 
209     }
210   }

227   override def +=(that: DescriptiveStats): Unit = {
228     if (this == that)
229       return
230     if (that.isEmpty)
231       return
232     if (this.isEmpty) {
233       copyFrom(that)
234     } else {
235 
236       val n1 = this._count
237       val n2 = that._count
238       val n1_squared = n1 * n1
239       val n2_squared = n2 * n2
240       val n_product = n1 * n2
241       val n = n1 + n2
242       val n_i = 1d / n
243 
244       val delta = that._mean - this._mean
245 
246       val A = delta * n_i
247       val A_squared = A * A
248 
249       _m4n += that._m4n +
250         n_product * (n1_squared - n_product + n2_squared) * delta * A * A_squared +
251         (that._m2n * n1_squared + _m2n * n2_squared) * A_squared * 6d +
252         (that._m3n * n1 - _m3n * n2) * A * 4d
253 
254       _m3n += that._m3n +
255         n_product * (n1 - n2) * delta * A_squared +
256         (that._m2n * n1 - _m2n * n2) * A * 3d
257 
258       _m2n += that._m2n +
259         delta * A * n_product
260 
261       /* optimize original code (below) by special handling of diagonal and reflection about it
262        * _c2 += (that._c2 + (delta |*| delta.T) * (n_product * n_i))
263        */
264       val coef = n_product * n_i
265       var ri = 0
266       while (ri < size) {
267         _c2.set(ri, ri, _m2n.get(ri)) // c2 diagonal is equal to m2n
268         val rd = delta.get(ri)
269         var ci = ri + 1 // traverse upper diagonal
270         while (ci < size) {
271           val c2 = _c2.get(ri, ci) + that._c2.get(ri,ci) + rd * delta.get(ci) * coef
272           _c2.set(ri, ci, c2) // set upper diagonal
273           _c2.set(ci, ri, c2) // set lower diagonal
274           ci += 1
275         }
276         ri += 1
277       } // c2 update
278 
279       _mean += A * n2
280 
281       _sum += that._sum
282 
283       var i = 0
284       while (i < size) {
285         val min = that._min.get(i)
286         val max = that._max.get(i)
287         if (min < _min.get(i)) {
288           _min.set(i, min)
289         }
290         if (max > _max.get(i)) {
291           _max.set(i, max)
292         }
293         i += 1
294       }  // min/max update
295 
296       _count += that._count
297     }
