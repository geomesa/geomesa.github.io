18  * Ordering of simple feature attributes

20 object AttributeOrdering {
21 
22   /**
23    * An ordering for a particular attribute. Note that although the signature is AnyRef for ease of use
24    * with the simple feature API, the actual values being sorted must correspond to the type of the attribute
25    *
26    * @param descriptor descriptor
27    * @return
28    */
29   def apply(descriptor: AttributeDescriptor): Ordering[AnyRef] = apply(ObjectType.selectType(descriptor))
30 
31   /**
32    * An ordering for a particular attribute. Note that although the signature is AnyRef for ease of use
33    * with the simple feature API, the actual values being sorted must correspond to the type of the attribute
34    *
35    * @param bindings type bindings
36    * @return
37    */
38   def apply(bindings: Seq[ObjectType]): Ordering[AnyRef] = {
39     val ordering = bindings.head match {
40       case ObjectType.STRING   => StringOrdering
41       case ObjectType.INT      => IntOrdering
42       case ObjectType.LONG     => LongOrdering
43       case ObjectType.FLOAT    => FloatOrdering
44       case ObjectType.DOUBLE   => DoubleOrdering
45       case ObjectType.BOOLEAN  => BooleanOrdering
46       case ObjectType.DATE     => DateOrdering
47       case ObjectType.UUID     => UuidOrdering
48       case ObjectType.GEOMETRY => GeometryOrdering
49       case ObjectType.BYTES    => BytesOrdering
50       case ObjectType.LIST     => list(bindings.last)
51       case ObjectType.MAP      => throw new UnsupportedOperationException("Ordering for Map-type attributes is not supported")
52       case b                   => throw new UnsupportedOperationException(s"Unexpected attribute type: $b")
53     }
54     ordering.asInstanceOf[Ordering[AnyRef]]
55   }
56 
57   private def list(binding: ObjectType): Ordering[AnyRef] = {
58     val ordering = binding match {
59       case ObjectType.STRING   => StringListOrdering
60       case ObjectType.INT      => IntListOrdering
61       case ObjectType.LONG     => LongListOrdering
62       case ObjectType.FLOAT    => FloatListOrdering
63       case ObjectType.DOUBLE   => DoubleListOrdering
64       case ObjectType.BOOLEAN  => BooleanListOrdering
65       case ObjectType.DATE     => DateListOrdering
66       case ObjectType.UUID     => UuidListOrdering
67       case ObjectType.GEOMETRY => GeometryListOrdering
68       case ObjectType.BYTES    => BytesListOrdering
69       case b                   => throw new UnsupportedOperationException(s"Unexpected attribute type: List[$b]")
70     }
71     ordering.asInstanceOf[Ordering[AnyRef]]
72   }

96   private class GeometryOrdering extends Ordering[Geometry] {
97     override def compare(x: Geometry, y: Geometry): Int = x.compareTo(y)
98   }
99 
100   private class BytesOrdering extends Ordering[Array[Byte]] {
101     override def compare(x: Array[Byte], y: Array[Byte]): Int = {
102       val len = math.min(x.length, y.length)
103       var i = 0
104       while (i < len) {
105         val res = java.lang.Byte.compare(x(i), y(i))
106         if (res != 0) {
107           return res
108         }
109         i += 1
110       }
111       Integer.compare(x.length, y.length)
112     }
113   }
114 
115   private object ListOrdering {
116     def apply[T](delegate: Ordering[T]): Ordering[java.util.List[T]] = new NullOrdering(new ListOrdering(delegate))
117   }
118 
119   private class ListOrdering[T](delegate: Ordering[T]) extends Ordering[java.util.List[T]] {
120     override def compare(x: java.util.List[T], y: java.util.List[T]): Int = {
121       val len = math.min(x.size, y.size)
122       var i = 0
123       while (i < len) {
124         val res = delegate.compare(x.get(i), y.get(i))
125         if (res != 0) {
126           return res
127         }
128         i += 1
129       }
130       Integer.compare(x.size, y.size)
131     }
132   }
133 
134   private class NullOrdering[T](delegate: Ordering[T]) extends Ordering[T] {
135     override def compare(x: T, y: T): Int = {
136       if (x == null) {
137         if (y == null) { 0 } else { -1 }
138       } else if (y == null) {
139         1
140       } else {
141         delegate.compare(x, y)
142       }
143     }
144   }
