45    * Utility method to return the bit-string as a full binary string.
46    *
47    * NB:  We align our bits to the RIGHT by default, not to the LEFT.
48    * This means that the most significant bit in the GeoHash (as it is
49    * interpreted to a geometry) is actually the least significant bit
50    * in the bit-set.  In other words:  Interpret the bit-set from
51    * right to left.

60   /**
61     * Converts the geohash into an int.
62     *
63     * Note: geohashes with more than 6 base32 digits will not fit in an int
64     *
65     * @return the geohash as an int
66     */
67   def toInt: Int = {
68     var i = 0
69     var asInt = 0
70     while (i < prec) {
71       if (bitset(i)) {
72         asInt += 1 << prec - 1 - i
73       }
74       i += 1
75     }
76     asInt
77   }

303   /**
304    * There is no visible difference between "t4bt" as a 20-bit GeoHash and
305    * the same string as a 17-bit GeoHash.  Unless otherwise specified, assume
306    * that the string represents a full complement of bits.
307    *
308    * If the call specifies a precision, then there is some additional work:
309    * 1.  the full-characters (those each representing a full 5 bits) should
310    *     be interpreted as they were before
311    * 2.  the partial-character (zero or one; representing the remainder bits,
312    *     guaranteed to be fewer than 5) must be appended
313    *
314    * @param string the base-32 encoded string to decode
315    * @param precisionOption the desired precision as an option
316    * @return the decoded GeoHash
317    */
318   private def decode(string: String, precisionOption: Option[Int] = None): GeoHash = {
319     // figure out what precision we should use
320     val prec = precisionOption.getOrElse(5*string.length)
321     checkPrecision(prec)
322 
323     // compute bit-sets for both the full and partial characters
324     val bitsets = string.zipWithIndex.map { case (c: Char, i: Int) => shift(i*5, characterMap(c)) }
325 
326     // OR all of these bit-sets together
327     val finalBitset: BitSet = bitsets.size match {
328       case 0 => BitSet()
329       case 1 => bitsets(0)
330       case _ => bitsets.reduce(_|_)
331     }
332 
333     val (lonIndex, latIndex) = extractReverseBits(finalBitset.toBitMask.head, prec)
334     val (lonDelta, latDelta) = (lonDeltaMap(prec), latDeltaMap(prec))
335     val bbox = bboxFromIndiciesDeltas(lonIndex, latIndex, lonDelta, latDelta)
336 
337     GeoHash(bbox.midLon, bbox.midLat, bbox, finalBitset, prec, Some(string))
338   }

347   /**
348    * Reverses and interleaves every other bits of two longs into one long. The two longs must be
349    * same size or first can be one bit longer than second.
350    * @param first first long
351    * @param second second long
352    * @param numBits The total number of bits of the interleaved & reversed result
353    * @return long with a total of numBits bits of first and second interleaved & reversed
354    */
355   private def interleaveReverseBits(first: Long, second: Long, numBits: Int): Long = {
356     /* We start with the first value of the interleaved long, coming from first if
357        numBits is odd or from second if numBits is even */
358     val even = (numBits & 0x01) == 0
359     val (actualFirst, actualSecond) = if(even) (second, first) else (first, second)
360     val numPairs = numBits >> 1
361     var result = 0L
362     (0 until numPairs).foreach{ pairNum =>
363       result = (result << 1) | ((actualFirst >> pairNum) & 1L)
364       result = (result << 1) | ((actualSecond >> pairNum) & 1L)
365     }
366     if (!even) (result << 1) | ((actualFirst >> numPairs) & 1L) else result
367   }

369   /**
370    * Reverses and extracts every other bit up to numBits and extracts them into two longs.
371    * @param value the long
372    * @param numBits the number of bits to extract from the long
373    * @return (first long, second long)  The first long will be one bit longer than the second if
374    *         numBits is odd
375    */
376   private def extractReverseBits(value: Long, numBits: Int): (Long, Long) = {
377     val numPairs = numBits >> 1
378     var first = 0L
379     var second = 0L
380     (0 until numPairs).foreach{ pairNum =>
381       first = (first << 1) | ((value >> pairNum * 2) & 1L)
382       second = (second << 1) | ((value >> (pairNum * 2 + 1)) & 1L)
383     }
384     ((if((numBits & 0x01) == 1) ((first << 1) | ((value >> (numBits - 1)) & 1L)) else first), second)
385   }

395   private def toBase32(bitset: BitSet, prec: Int): String = {
396     // compute the precision padded to the next 5-bit boundary
397     val numLeftoverBits = prec % 5
398     val precision = prec + (numLeftoverBits match {
399       case 0 => 0
400       case _ => (5-numLeftoverBits)
401     })
402 
403     // take the bit positions in groups of 5, and map each set to a character
404     // (based on the current bit-set); this works for off-5 precisions, because
405     // the additional bits will simply not be there (assumed to be zero)
406     (0 until precision).grouped(5).map(i=>ch(i, bitset)).mkString
407   }
