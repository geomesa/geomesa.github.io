18 /**
19   * Spatial index that breaks up space into discrete buckets to index points
20   *
21   * Does not support non-point inserts
22   *
23   * @param xBuckets number of x buckets
24   * @param yBuckets number of y buckets
25   * @param extents area to be indexed
26   * @tparam T index value binding
27   */
28 class BucketIndex[T](
29     val xBuckets: Int = 360,
30     val yBuckets: Int = 180,
31     val extents: Envelope = new Envelope(-180.0, 180.0, -90.0, 90.0)
32   ) extends SpatialIndex[T] with LazyLogging {
33 
34   // create the buckets up front to avoid having to synchronize the whole array
35   // we use a ConcurrentHashMap, which gives us iterators that aren't affected by modifications to the backing map
36   private val buckets = Array.fill(xBuckets, yBuckets)(new ConcurrentHashMap[String, T]())
37 
38   private val gridSnap = new GridSnap(extents, xBuckets, yBuckets)
39 
40   override def insert(geom: Geometry, key: String, value: T): Unit = {
41     val pt = geom.asInstanceOf[Point]
42     val i = snapX(pt.getX)
43     val j = snapY(pt.getY)
44     buckets(i)(j).put(key, value)
45   }
46 
47   override def remove(geom: Geometry, key: String): T = {
48     val pt = geom.asInstanceOf[Point]
49     val i = snapX(pt.getX)
50     val j = snapY(pt.getY)
51     buckets(i)(j).remove(key)
52   }
53 
54   override def get(geom: Geometry, key: String): T = {
55     val pt = geom.asInstanceOf[Point]
56     val i = snapX(pt.getX)
57     val j = snapY(pt.getY)
58     buckets(i)(j).get(key)
59   }
60 
61   override def query(xmin: Double, ymin: Double, xmax: Double, ymax: Double): Iterator[T] =
62     new BucketIterator(snapX(xmin), snapX(xmax), snapY(ymin), snapY(ymax))
63 
64   override def query(): Iterator[T] = new BucketIterator(0, xBuckets - 1, 0, yBuckets - 1)
65 
66   override def size(): Int = {
67     var size = 0
68     var i = 0
69     while (i < xBuckets) {
70       var j = 0
71       while (j < yBuckets) {
72         size += buckets(i)(j).size()
73         j += 1
74       }
75       i += 1
76     }
77     size
78   }
79 
80   override def clear(): Unit = {
81     var i = 0
82     while (i < xBuckets) {
83       var j = 0
84       while (j < yBuckets) {
85         buckets(i)(j).clear()
86         j += 1
87       }
88       i += 1
89     }
90   }
91 
92   private def snapX(x: Double): Int = {
93     val i = gridSnap.i(x)
94     if (i != -1) { i } else if (x < extents.getMinX) { 0 } else { xBuckets - 1 }
95   }
96 
97   private def snapY(y: Double): Int = {
98     val j = gridSnap.j(y)
99     if (j != -1) { j } else if (y < extents.getMinY) { 0 } else { yBuckets - 1 }
100   }
101 
102   /**
103     * Iterator over a range of buckets
104     */
105   class BucketIterator private [BucketIndex] (mini: Int, maxi: Int, minj: Int, maxj: Int) extends Iterator[T] {
106 
107     private var i = mini
108     private var j = minj
109     private var iter = buckets(i)(j).values.iterator() // note: `.values` is a cached view
110 
111     @tailrec
112     override final def hasNext: Boolean = iter.hasNext || {
113       if (i == maxi && j == maxj) { false } else {
114         if (j < maxj) {
115           j += 1
116         } else {
117           j = minj
118           i += 1
119         }
120         iter = buckets(i)(j).values.iterator() // note: `.values` is a cached view
121         hasNext
122       }
123     }
124 
125     override def next(): T = iter.next()
126   }
127 }
