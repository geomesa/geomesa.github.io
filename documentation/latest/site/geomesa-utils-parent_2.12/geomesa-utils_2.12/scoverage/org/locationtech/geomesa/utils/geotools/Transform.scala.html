27

30    *
31    * @return

51 /**
52   * Convenience methods for working with relational query projections (aka transforms)
53   */
54 object Transform {
55 
56   val DefinitionDelimiter = ";"
57 
58   /**
59    * A simple relational projection (selection)
60    *
61    * @param name name
62    * @param binding binding
63    * @param i index of the original attribute being projected
64    */
65   case class PropertyTransform(name: String, binding: Class[_], i: Int) extends Transform {
66     override def evaluate(feature: SimpleFeature): AnyRef = feature.getAttribute(i)
67   }
68 
69   /**
70    * A simple relational projection, but also renamed
71    *
72    * @param name name
73    * @param binding binding
74    * @param original name of the original attribute being projected
75    * @param i index of the original attribute being projected
76    */
77   case class RenameTransform(name: String, binding: Class[_], original: String, i: Int) extends Transform {
78     override def evaluate(feature: SimpleFeature): AnyRef = feature.getAttribute(i)
79   }
80 
81   /**
82    * An attribute comprised of an expression (function, math transform, json-path, etc)
83    *
84    * @param name name
85    * @param binding binding
86    * @param expression expression
87    */
88   case class ExpressionTransform(name: String, binding: Class[_], expression: Expression) extends Transform {
89     override def evaluate(feature: SimpleFeature): AnyRef = expression.evaluate(feature)
90   }
91 
92   object Transforms {
93 
94     /**
95      * Create transform definitions for a delimited expression string
96      *
97      * @param sft original simple feature type
98      * @param transforms semi-colon delimited transform expressions
99      * @return
100      */
101     def apply(sft: SimpleFeatureType, transforms: String): Seq[Transform] =
102       if (transforms.isEmpty) { Seq.empty } else { apply(sft, transforms.split(DefinitionDelimiter)) }
103 
104     /**
105      * Create transform definitions for a sequence of expression strings
106      *
107      * @param sft original simple feature type
108      * @param transforms transform expressions
109      * @return
110      */
111     def apply(sft: SimpleFeatureType, transforms: Seq[String]): Seq[Transform] =
112       transforms.map(definition(sft, _))
113 
114     /**
115      * Create a transform definition from an expression
116      *
117      * @param sft simple feature type
118      * @param transform transform expression
119      * @return
120      */
121     private def definition(sft: SimpleFeatureType, transform: String): Transform = {
122       val equals = transform.indexOf('=')
123       if (equals == -1) {
124         val name = transform.trim()
125         val i = sft.indexOf(name)
126         if (i == -1) {
127           attributeExpression(sft, name, name)
128         } else {
129           PropertyTransform(name, sft.getDescriptor(i).getType.getBinding, i)
130         }
131       } else {
132         val name = transform.substring(0, equals).trim()
133         val exp = transform.substring(equals + 1).trim()
134         val expression = try { ECQL.toExpression(exp) } catch {
135           case NonFatal(e) => throw new IllegalArgumentException(s"Unable to parse expression $transform:", e)
136         }
137         expression match {
138           case f: Function =>
139             ExpressionTransform(name, f.getFunctionName.getReturn.getType, f)
140 
141           case f: MathExpressionImpl =>
142             ExpressionTransform(name, classOf[java.lang.Double], f) // math ops always return doubles?
143 
144           case p: PropertyName =>
145             val orig = p.getPropertyName
146             val i = sft.indexOf(orig)
147             if (i == -1) {
148               attributeExpression(sft, name, p)
149             } else if (name == orig) {
150               PropertyTransform(name, sft.getDescriptor(i).getType.getBinding, i)
151             } else {
152               RenameTransform(name, sft.getDescriptor(i).getType.getBinding, orig, i)
153             }
154 
155           // TODO: Add support for LiteralExpressionImpl and/or ClassificationFunction?
156           case _ =>
157             throw new IllegalArgumentException(s"Unable to handle transform expression: $expression")
158         }
159       }
160     }
161 
162     /**
163      * This handles custom attribute accessors (e.g. json-path) and transforms that don't correspond
164      * to any attribute (e.g. in AttributeKeyPlusValueIterator)
165      *
166      * @param sft simple feature type
167      * @param name transform name
168      * @param e expression
169      * @return
170      */
171     private def attributeExpression(sft: SimpleFeatureType, name: String, e: String): ExpressionTransform = {
172       val expression = try { ECQL.toExpression(e) } catch {
173         case NonFatal(e) => throw new IllegalArgumentException(s"Unable to parse expression '$e':", e)
174       }
175       attributeExpression(sft, name, expression)
176     }
177 
178     /**
179      * This handles custom attribute accessors (e.g. json-path) and transforms that don't correspond
180      * to any attribute (e.g. in AttributeKeyPlusValueIterator)
181      *
182      * @param sft simple feature type
183      * @param name transform name
184      * @param expression expression
185      * @return
186      */
187     private def attributeExpression(
188         sft: SimpleFeatureType,
189         name: String,
190         expression: Expression): ExpressionTransform = {
191       val descriptor = expression.evaluate(sft).asInstanceOf[AttributeDescriptor]
192       val binding = if (descriptor == null) { classOf[String] } else { descriptor.getType.getBinding }
193       ExpressionTransform(name, binding, expression)
194     }
195 
196     /**
197      * Create the feature type corresponding a transform
198      *
199      * @param sft original simple feature type
200      * @param transforms transforms
201      * @return
202      */
203     def schema(sft: SimpleFeatureType, transforms: Seq[Transform]): SimpleFeatureType = {
204       val schema = new java.util.ArrayList[AttributeDescriptor]()
205       var geom: GeometryDescriptor = null
206 
207       lazy val typeBuilder = new AttributeTypeBuilder()
208 
209       transforms.foreach { t =>
210         val descriptor = t match {
211           case t: PropertyTransform =>
212             val d = sft.getDescriptor(t.i)
213             val im = SimpleFeatureTypes.immutable(d)
214             if (d == sft.getGeometryDescriptor) {
215               geom = im.asInstanceOf[GeometryDescriptor]
216             }
217             im
218 
219           case t: RenameTransform =>
220             sft.getDescriptor(t.i) match {
221               case d: GeometryDescriptor =>
222                 val im = new ImmutableGeometryDescriptor(d.getType, new NameImpl(t.name), d.getMinOccurs,
223                   d.getMaxOccurs, d.isNillable, d.getDefaultValue, d.getUserData)
224                 if (d == sft.getGeometryDescriptor) {
225                   geom = im
226                 }
227                 im
228               case d: AttributeDescriptor =>
229                 new ImmutableAttributeDescriptor(d.getType, new NameImpl(t.name), d.getMinOccurs, d.getMaxOccurs,
230                   d.isNillable, d.getDefaultValue, d.getUserData)
231             }
232 
233           case t: ExpressionTransform =>
234             typeBuilder.setBinding(t.binding)
235             if (classOf[Geometry].isAssignableFrom(t.binding)) {
236               typeBuilder.crs(CRS_EPSG_4326)
237               val typ = typeBuilder.buildGeometryType()
238               new ImmutableGeometryDescriptor(typ, new NameImpl(t.name), 0, 1, true, null, Collections.emptyMap())
239             } else {
240               val typ = typeBuilder.buildType()
241               new ImmutableAttributeDescriptor(typ, new NameImpl(t.name), 0, 1, true, null, Collections.emptyMap())
242             }
243         }
244 
245         schema.add(descriptor)
246       }
247 
248       var i = 0
249       while (geom == null && i < schema.size) {
250         schema.get(i) match {
251           case d: ImmutableGeometryDescriptor => geom = d
252           case _ => // no-op
253         }
254         i += 1
255       }
256 
257       // TODO reconsider default field user data?
258       new ImmutableSimpleFeatureType(sft.getName, schema, geom, sft.isAbstract, sft.getRestrictions, sft.getSuper,
259         sft.getDescription, sft.getUserData)
260     }
261   }
