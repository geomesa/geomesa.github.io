13 import scala.reflect.ClassTag
14 
15 class GroupBy[T](val sft: SimpleFeatureType, val property: String, val stat: String)(implicit val ct: ClassTag[T])
16     extends Stat {
17 
18   override type S = GroupBy[T]
19 
20   private val i = sft.indexOf(property)
21   private [stats] val groups = scala.collection.mutable.Map.empty[T, Stat]
22 
23   def size: Int = groups.size
24   def get(key: T): Option[Stat] = groups.get(key)
25   def getOrElse[U >: Stat](key: T, default: => U): U = groups.getOrElse(key, default)
26   def iterator: Iterator[(T, Stat)] = groups.iterator
27 
28   private def buildNewStat: Stat = StatParser.parse(sft, stat)
29 
30   /**
31     * Compute statistics based upon the given simple feature.
32     * This method will be called for every SimpleFeature a query returns.
33     *
34     * @param sf feature to evaluate
35     */
36   override def observe(sf: SimpleFeature): Unit = {
37     val key = sf.getAttribute(i).asInstanceOf[T]
38     if (key != null) {
39       groups.getOrElseUpdate(key, buildNewStat).observe(sf)
40     }
41   }
42 
43   /**
44     * Tries to remove the given simple feature from the compiled statistics.
45     * Note: may not be possible to un-observe a feature, in which case this method will
46     * have no effect.
47     *
48     * @param sf feature to un-evaluate
49     */
50   override def unobserve(sf: SimpleFeature): Unit = {
51     val key = sf.getAttribute(i).asInstanceOf[T]
52     if (key != null) {
53       groups.get(key).foreach(groupedStat => groupedStat.unobserve(sf))
54     }
55   }
56 
57   /**
58     * Add another stat to this stat. Avoids allocating another object.
59     *
60     * @param other the other stat to add
61     */
62   override def +=(other: GroupBy[T]): Unit = {
63     other.groups.foreach { case (key, s) =>
64       groups.getOrElseUpdate(key, buildNewStat) += s
65     }
66   }
67 
68   /**
69     * Combine two stats into a new stat
70     *
71     * @param other the other stat to add
72     */
73   override def +(other: GroupBy[T]): GroupBy[T] = {
74     val sum = new GroupBy[T](sft, property, stat)
75     sum += this
76     sum += other
77     sum
78   }
79 
80   override def toJsonObject: Map[T, Any] = {
81     val builder = collection.immutable.ListMap.newBuilder[T, Any]
82     val keyClass = groups.headOption.map(_._1.getClass).getOrElse(ct.runtimeClass)
83     val ordered: Iterable[(T, Stat)] = if (classOf[Comparable[_]].isAssignableFrom(keyClass)) {
84       val ordering = new Ordering[T] {
85         override def compare(x: T, y: T): Int = x.asInstanceOf[Comparable[Any]].compareTo(y)
86       }
87       groups.toSeq.sortBy(_._1)(ordering)
88     } else {
89       groups
90     }
91     ordered.foreach { case (k, v) => builder += k -> v.toJsonObject }
92     builder.result
93   }
94 
95   /**
96     * Necessary method used by the StatIterator. Indicates if the stat has any values or not
97     *
98     * @return true if stat contains values
99     */
100   override def isEmpty: Boolean = groups.values.forall(_.isEmpty)
101 
102   /**
103     * Compares the two stats for equivalence. We don't use standard 'equals' as it gets messy with
104     * mutable state and hash codes
105     *
106     * @param other other stat to compare
107     * @return true if equals
108     */
109   override def isEquivalent(other: Stat): Boolean = {
110     other match {
111       case other: GroupBy[T] =>
112         groups.keys == other.groups.keys &&
113           groups.forall { case (key, s) => other.groups(key).isEquivalent(s) }
114       case _ => false
115     }
116   }
117 
118   /**
119     * Clears the stat to its original state when first initialized.
120     * Necessary method used by the StatIterator.
121     */
122   override def clear(): Unit = groups.clear()
123 }
