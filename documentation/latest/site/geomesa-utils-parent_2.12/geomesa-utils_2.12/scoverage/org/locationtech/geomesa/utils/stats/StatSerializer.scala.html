29 /**
30   * Serialize and deserialize stats
31   */
32 trait StatSerializer {
33   def serialize(stat: Stat): Array[Byte]
34   def deserialize(bytes: Array[Byte], immutable: Boolean = false): Stat =
35     deserialize(bytes, 0, bytes.length, immutable)
36   def deserialize(bytes: Array[Byte], offset: Int, length: Int, immutable: Boolean): Stat
37 }
38 
39 object StatSerializer {
40 
41   private val serializers = scala.collection.mutable.Map.empty[String, StatSerializer]
42 
43   def apply(sft: SimpleFeatureType): StatSerializer = serializers.synchronized {
44     serializers.getOrElseUpdate(CacheKeyGenerator.cacheKey(sft), new KryoStatSerializer(sft))
45   }

47   /**
48     * Kryo implementation of stat serializer. Thread-safe.
49     *
50     * @param sft simple feature type
51     */
52   class KryoStatSerializer(sft: SimpleFeatureType) extends StatSerializer {
53 
54     override def serialize(stat: Stat): Array[Byte] = {
55       val output = KryoStatSerializer.outputs.getOrElseUpdate(new Output(1024, -1))
56       output.setOutputStream(null) // resets the buffer
57       KryoStatSerializer.write(output, sft, stat)
58       output.toBytes
59     }
60 
61     override def deserialize(bytes: Array[Byte], offset: Int, length: Int, immutable: Boolean): Stat = {
62       val input = KryoStatSerializer.inputs.getOrElseUpdate(new NonMutatingInput())
63       input.setBuffer(bytes, offset, length)
64       KryoStatSerializer.read(input, sft, immutable)
65     }
66   }

111     private def write(output: Output, sft: SimpleFeatureType, stat: Stat): Unit = {
112       stat match {
113         case s: CountStat           => output.writeByte(CountByte);           writeCount(output, s)
114         case s: MinMax[_]           => output.writeByte(MinMaxByte);          writeMinMax(output, sft, s)
115         case s: EnumerationStat[_]  => output.writeByte(EnumerationByte);     writeEnumeration(output, sft, s)
116         case s: TopK[_]             => output.writeByte(TopKByte);            writeTopK(output, sft, s)
117         case s: Histogram[_]        => output.writeByte(HistogramByte);       writeHistogram(output, sft, s)
118         case s: Frequency[_]        => output.writeByte(FrequencyByte);       writeFrequency(output, sft, s)
119         case s: Z3Histogram         => output.writeByte(Z3HistogramByte);     writeZ3Histogram(output, sft, s)
120         case s: Z3Frequency         => output.writeByte(Z3FrequencyByte);     writeZ3Frequency(output, sft, s)
121         case s: IteratorStackCount  => output.writeByte(IteratorStackByte);   writeIteratorStackCount(output, s)
122         case s: SeqStat             => output.writeByte(SeqStatByte);         writeSeqStat(output, sft, s)
123         case s: DescriptiveStats    => output.writeByte(DescriptiveStatByte); writeDescriptiveStats(output, sft, s)
124         case s: GroupBy[_]          => output.writeByte(GroupByByte);         writeGroupBy(output, sft, s)
125         case _ => throw new UnsupportedOperationException(s"Unhandled stat $stat")
126       }
127     }

129     private def read(input: Input, sft: SimpleFeatureType, immutable: Boolean): Stat = {
130       input.readByte() match {
131         case CountByte             => readCount(input, sft, immutable)
132         case MinMaxByte            => readMinMax(input, sft, immutable, 3)
133         case EnumerationByte       => readEnumeration(input, sft, immutable, 2)
134         case TopKByte              => readTopK(input, sft, immutable, 3)
135         case HistogramByte         => readHistogram(input, sft, immutable, 2)
136         case FrequencyByte         => readFrequency(input, sft, immutable, 3)
137         case Z3HistogramByte       => readZ3Histogram(input, sft, immutable, 3)
138         case Z3FrequencyByte       => readZ3Frequency(input, sft, immutable, 3)
139         case IteratorStackByte     => readIteratorStackCount(input, sft, immutable)
140         case SeqStatByte           => readSeqStat(input, sft, immutable)
141         case DescriptiveStatByte   => readDescriptiveStat(input, sft, immutable, 2)
142         case GroupByByte           => readGroupBy(input, sft, immutable, 2)
143         case EnumerationByteV1     => readEnumeration(input, sft, immutable, 1)

172     private def readGroupBy(input: Input, sft: SimpleFeatureType, immutable: Boolean, version: Int): GroupBy[_] = {
173       val attribute = version match {
174         case 2 => input.readString
175         case 1 => sft.getDescriptor(input.readInt(true)).getLocalName
176         case _ => throw new IllegalArgumentException(s"Invalid group by serialization version: $version")
177       }
178       val exampleStat = input.readString()
179       val keyLength   = input.readInt(true)
180 
181       val classTag = ClassTag[Any](sft.getDescriptor(attribute).getType.getBinding)
182       val stat = if (immutable) {
183         new GroupBy(sft, attribute, exampleStat)(classTag) with ImmutableStat
184       } else {
185         new GroupBy(sft, attribute, exampleStat)(classTag)
186       }
187 
188       val keyReader = reader(input, sft.getDescriptor(attribute).getType.getBinding)
189 
190       var i = 0
191       while (i < keyLength) {
192         val key = keyReader.apply()
193         val groupedStat = read(input, sft, immutable)
194         stat.groups.put(key, groupedStat)
195         i += 1
196       }
197       stat
198     }

289     private def readMinMax(input: Input, sft: SimpleFeatureType, immutable: Boolean, version: Int): MinMax[_] = {
290       val attribute = version match {
291         case 3     => input.readString()
292         case 1 | 2 => sft.getDescriptor(input.readInt(true)).getLocalName
293         case _ => throw new IllegalArgumentException(s"Invalid min/max serialization version: $version")
294       }
295       val hpp = if (version > 1) {
296         val log2m = input.readInt(true)
297         val size = input.readInt(true)
298         val bytes = Array.fill(size)(input.readInt)
299         HyperLogLog(log2m, bytes)
300       } else {
301         val hppBytes = Array.ofDim[Byte](input.readInt(true))
302         input.read(hppBytes)
303         val clearspring = com.clearspring.analytics.stream.cardinality.HyperLogLog.Builder.build(hppBytes)
304         // use reflection to access private variables
305         def getField[T](name: String): T = {
306           val field = clearspring.getClass.getDeclaredField(name)
307           field.setAccessible(true)
308           field.get(clearspring).asInstanceOf[T]
309         }
310         val log2m = getField[Int]("log2m")
311         val registerSet = getField[RegisterSet]("registerSet").bits
312         HyperLogLog(log2m, registerSet)
313       }
314 
315       val binding = sft.getDescriptor(attribute).getType.getBinding
316       val read = reader(input, binding)
317       val min = read()
318       val max = read()
319 
320       val defaults = MinMaxDefaults[Any](binding)
321 
322       if (immutable) {
323         new MinMax[Any](sft, attribute, min, max, hpp)(defaults) with ImmutableStat
324       } else {
325         new MinMax[Any](sft, attribute, min, max, hpp)(defaults)
326       }
327     }

509     private def readFrequency(input: Input, sft: SimpleFeatureType, immutable: Boolean, version: Int): Frequency[_] = {
510       def name(i: Int): String = sft.getDescriptor(i).getLocalName
511       val (attribute, dtg) = version match {
512         case 3     => (input.readString(), Option(input.readString()))
513         case 1 | 2 => (name(input.readInt(true)), Some(input.readInt(true)).filter(_ != -1).map(name))
514         case _ => throw new IllegalArgumentException(s"Invalid frequency serialization version: $version")
515       }
516       val period = if (version > 1) { TimePeriod.withName(input.readString()) } else { TimePeriod.Week }
517       val precision = input.readInt(true)
518       val eps = input.readDouble()
519       val confidence = input.readDouble()
520 
521       val binding = sft.getDescriptor(attribute).getType.getBinding
522       val stat = if (immutable) {
523         new Frequency[Any](sft, attribute, dtg, period, precision, eps, confidence)(ClassTag[Any](binding)) with ImmutableStat
524       } else {
525         new Frequency[Any](sft, attribute, dtg, period, precision, eps, confidence)(ClassTag[Any](binding))
526       }
527 
528       val sketchCount = input.readInt(true)
529       var c = 0
530       while (c < sketchCount) {
531         val week = input.readShort
532         val sketch = stat.newSketch
533         stat.sketchMap.put(week, sketch)
534         var i = 0
535         while (i < sketch.table.length) {
536           readCountArray(input, sketch.table(i))
537           i += 1
538         }
539         sketch._size = input.readLong(true)
540         c += 1
541       }
542 
543       stat
544     }
