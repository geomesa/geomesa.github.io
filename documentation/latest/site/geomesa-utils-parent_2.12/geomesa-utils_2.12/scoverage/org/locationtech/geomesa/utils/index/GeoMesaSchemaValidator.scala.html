35 /**
36   * Utility object that ensures that none of the (local portion of the) property
37   * names is a reserved word in ECQL.  Using those reserved words in a simple
38   * feature type will cause queries to fail.
39   */
40 object ReservedWordCheck extends LazyLogging {
41 
42   // ensure that no attribute names are reserved words within GeoTools that will cause query problems
43   def validateAttributeNames(sft: SimpleFeatureType): Unit = {
44     val reservedWords = FeatureUtils.sftReservedWords(sft)
45     if (reservedWords.nonEmpty) {
46       val msg = "The simple feature type contains attribute name(s) that are reserved words: " +
47           s"${reservedWords.mkString(", ")}. You may override this check by putting Boolean.TRUE into the " +
48           s"SimpleFeatureType user data under the key '$OverrideReservedWords' before calling createSchema, or by " +
49           s"setting the system property '$OverrideReservedWords' to 'true', however it may cause errors with some functionality."
50       if (GeoMesaSchemaValidator.declared(sft, OverrideReservedWords)) {
51         logger.warn(msg)
52       } else {
53         throw new IllegalArgumentException(msg)
54       }
55     }
56   }
57 }

59 /**
60  * Utility object for emitting a warning to the user if a SimpleFeatureType contains a temporal attribute, but
61  * none is used in the index.
62  *
63  * Furthermore, this object presents a candidate to be used in this case.
64  *
65  * This is useful since the only symptom of this mistake is slower than normal queries on temporal ranges.
66  */
67 object TemporalIndexCheck extends LazyLogging {
68 
69   def validateDtgField(sft: SimpleFeatureType): Unit = {
70     val dtgCandidates = scanForTemporalAttributes(sft) // all attributes which may be used
71     // validate that we have an ok field, and replace it if not
72     if (!sft.getDtgField.exists(dtgCandidates.contains)) {
73       sft.getDtgField.foreach { dtg => // clear out any existing invalid field
74         logger.warn(s"Invalid date field '$dtg' specified for schema $sft")
75         sft.clearDtgField()
76       }
77       // if there are valid fields, warn and set to the first available
78       if (!GeoMesaSchemaValidator.declared(sft, IndexIgnoreDtg)) {
79         dtgCandidates.headOption.foreach { candidate =>
80           lazy val theWarning = s"$DefaultDtgField is not valid or defined for simple feature type $sft. " +
81               "However, the following attribute(s) can be used in GeoMesa's temporal index: " +
82               s"${dtgCandidates.mkString(", ")}. To disable temporal indexing, put Boolean.TRUE into the " +
83               s"SimpleFeatureType user data under the key '$IndexIgnoreDtg' before calling createSchema, or " +
84               s"set the system property '$IndexIgnoreDtg' to 'true'. GeoMesa will now point $DefaultDtgField " +
85               s"to the first temporal attribute found: $candidate"
86           logger.warn(theWarning)
87           sft.setDtgField(candidate)
88         }
89       }
90     }
91   }
92 
93   private def scanForTemporalAttributes(sft: SimpleFeatureType): Seq[String] =
94     sft.getAttributeDescriptors.asScala.toList
95       .withFilter { classOf[java.util.Date] isAssignableFrom _.getType.getBinding } .map { _.getLocalName }
96 }

98 object MixedGeometryCheck extends LazyLogging {
99 
100   def validateGeometryType(sft: SimpleFeatureType): Unit = {
101     val gd = sft.getGeometryDescriptor
102     if (gd != null && gd.getType.getBinding == classOf[Geometry]) {
103       if (!GeoMesaSchemaValidator.declared(sft, MixedGeometries)) {
104         throw new IllegalArgumentException("Trying to create a schema with mixed geometry type " +
105             s"'${gd.getLocalName}:Geometry'. Queries may be slower when using mixed geometries. " +
106             "If this is intentional, you may override this check by putting Boolean.TRUE into the " +
107             s"SimpleFeatureType user data under the key '$MixedGeometries' before calling createSchema, or by " +
108             s"setting the system property '$MixedGeometries' to 'true'. Otherwise, please specify a single " +
109             s"geometry type (e.g. Point, LineString, Polygon, etc).")
110       }
111     }
112   }
113 }

115 object IndexConfigurationCheck {
116 
117   def validateIndices(sft: SimpleFeatureType): Unit = {
118     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
119     require(sft.getZ2Shards > 0 && sft.getZ2Shards < 128, "Z shards must be between 1 and 127")
120     require(sft.getZ3Shards > 0 && sft.getZ3Shards < 128, "Z shards must be between 1 and 127")
121     require(sft.getAttributeShards > 0 && sft.getAttributeShards < 128, "Attribute shards must be between 1 and 127")
122   }
123 }

125 object PropertyCheck {
126   def validateDuplicateProperty(sft: SimpleFeatureType):Unit = {
127     val names = sft.getAttributeDescriptors.asScala.map(descriptor => descriptor.getLocalName)
128     val duplicateProperties = names.diff(names.distinct)
129     if ( duplicateProperties.nonEmpty ){
130       throw new IllegalArgumentException(s"Schema contains duplicate attribute names: '${duplicateProperties.mkString(", ")}' ")
131     }
132   }
