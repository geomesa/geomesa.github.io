36 /**
37   * Utility object that ensures that none of the (local portion of the) property
38   * names is a reserved word in ECQL.  Using those reserved words in a simple
39   * feature type will cause queries to fail.
40   */
41 object ReservedWordCheck extends LazyLogging {
42 
43   // ensure that no attribute names are reserved words within GeoTools that will cause query problems
44   def validateAttributeNames(sft: SimpleFeatureType): Unit = {
45     val reservedWords = FeatureUtils.sftReservedWords(sft)
46     if (reservedWords.nonEmpty) {
47       val msg = "The simple feature type contains attribute name(s) that are reserved words: " +
48           s"${reservedWords.mkString(", ")}. You may override this check by putting Boolean.TRUE into the " +
49           s"SimpleFeatureType user data under the key '$OverrideReservedWords' before calling createSchema, or by " +
50           s"setting the system property '$OverrideReservedWords' to 'true', however it may cause errors with some functionality."
51       if (GeoMesaSchemaValidator.declared(sft, OverrideReservedWords)) {
52         logger.warn(msg)
53       } else {
54         throw new IllegalArgumentException(msg)
55       }
56     }
57   }
58 }

60 /**
61  * Utility object for emitting a warning to the user if a SimpleFeatureType contains a temporal attribute, but
62  * none is used in the index.
63  *
64  * Furthermore, this object presents a candidate to be used in this case.
65  *
66  * This is useful since the only symptom of this mistake is slower than normal queries on temporal ranges.
67  */
68 object TemporalIndexCheck extends LazyLogging {
69 
70   def validateDtgField(sft: SimpleFeatureType): Unit = {
71     val dtgCandidates = scanForTemporalAttributes(sft) // all attributes which may be used
72     // validate that we have an ok field, and replace it if not
73     if (!sft.getDtgField.exists(dtgCandidates.contains)) {
74       sft.getDtgField.foreach { dtg => // clear out any existing invalid field
75         logger.warn(s"Invalid date field '$dtg' specified for schema $sft")
76         sft.clearDtgField()
77       }
78       // if there are valid fields, warn and set to the first available
79       if (!GeoMesaSchemaValidator.declared(sft, IndexIgnoreDtg)) {
80         dtgCandidates.headOption.foreach { candidate =>
81           lazy val theWarning = s"$DefaultDtgField is not valid or defined for simple feature type $sft. " +
82               "However, the following attribute(s) can be used in GeoMesa's temporal index: " +
83               s"${dtgCandidates.mkString(", ")}. To disable temporal indexing, put Boolean.TRUE into the " +
84               s"SimpleFeatureType user data under the key '$IndexIgnoreDtg' before calling createSchema, or " +
85               s"set the system property '$IndexIgnoreDtg' to 'true'. GeoMesa will now point $DefaultDtgField " +
86               s"to the first temporal attribute found: $candidate"
87           logger.warn(theWarning)
88           sft.setDtgField(candidate)
89         }
90       }
91     }
92   }
93 
94   private def scanForTemporalAttributes(sft: SimpleFeatureType): Seq[String] =
95     sft.getAttributeDescriptors.asScala.toList
96       .withFilter { classOf[java.util.Date] isAssignableFrom _.getType.getBinding } .map { _.getLocalName }
97 }

99 object MixedGeometryCheck extends LazyLogging {
100 
101   def validateGeometryType(sft: SimpleFeatureType): Unit = {
102     val gd = sft.getGeometryDescriptor
103     if (gd != null && gd.getType.getBinding == classOf[Geometry]) {
104       if (!GeoMesaSchemaValidator.declared(sft, MixedGeometries)) {
105         throw new IllegalArgumentException("Trying to create a schema with mixed geometry type " +
106             s"'${gd.getLocalName}:Geometry'. Queries may be slower when using mixed geometries. " +
107             "If this is intentional, you may override this check by putting Boolean.TRUE into the " +
108             s"SimpleFeatureType user data under the key '$MixedGeometries' before calling createSchema, or by " +
109             s"setting the system property '$MixedGeometries' to 'true'. Otherwise, please specify a single " +
110             s"geometry type (e.g. Point, LineString, Polygon, etc).")
111       }
112     }
113   }
114 }

116 object IndexConfigurationCheck {
117 
118   def validateIndices(sft: SimpleFeatureType): Unit = {
119     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
120     require(sft.getZ2Shards > 0 && sft.getZ2Shards < 128, "Z shards must be between 1 and 127")
121     require(sft.getZ3Shards > 0 && sft.getZ3Shards < 128, "Z shards must be between 1 and 127")
122     require(sft.getAttributeShards > 0 && sft.getAttributeShards < 128, "Attribute shards must be between 1 and 127")
123   }
124 }

126 object PropertyCheck {
127   def validateDuplicateProperty(sft: SimpleFeatureType):Unit = {
128     val names = sft.getAttributeDescriptors.asScala.map(descriptor => descriptor.getLocalName)
129     val duplicateProperties = names.diff(names.distinct)
130     if ( duplicateProperties.nonEmpty ){
131       throw new IllegalArgumentException(s"Schema contains duplicate attribute names: '${duplicateProperties.mkString(", ")}' ")
132     }
133   }
