19   case class SystemProperty(property: String, default: String = null) {
20 
21     val threadLocalValue = new InheritableThreadLocal[String]()
22 
23     def set(value: String): Unit = System.setProperty(property, value)
24 
25     def clear(): Unit = System.clearProperty(property)
26 
27     def get: String = ConfigLoader.Config.get(property) match {
28       case Some((value, true))  => value // final value - can't be overridden
29       case Some((value, false)) => fromSysProps.getOrElse(value)
30       case None => fromSysProps.getOrElse(default)
31     }
32 
33     def option: Option[String] = Option(get)
34 
35     def toDuration: Option[Duration] = option.flatMap { value =>
36       Try(DurationParsing.caseInsensitive(value)) match {
37         case Success(v) if v.isFinite && v > Duration.Zero => Some(v)
38         case _ =>
39           logger.warn(s"Invalid duration for property $property: $value")
40           Option(default).map(Duration.apply)
41       }
42     }
43 
44     def toJavaDuration: Option[java.time.Duration] = toDuration.map(d => java.time.Duration.ofMillis(d.toMillis))
45 
46     def toMillis: Option[Long] = toDuration.map(_.toMillis)
47 
48     def toUnboundedDuration: Option[Duration] = option.flatMap { value =>
49       Try(DurationParsing.caseInsensitive(value)) match {
50         case Success(v) => Some(v)
51         case _ =>
52           logger.warn(s"Invalid duration for property $property: $value")
53           Option(default).map(Duration.apply)
54       }
55     }
56 
57     def toBytes: Option[Long] = option.flatMap { value =>
58       Suffixes.Memory.bytes(value) match {
59         case Success(b) => Some(b)
60         case Failure(e) =>
61           logger.warn(s"Invalid bytes for property $property=$value: ${e.toString}")
62           Option(default).flatMap(Suffixes.Memory.bytes(_).toOption)
63       }
64     }
65 
66     def toInt: Option[Int] = option.flatMap { value =>
67       Try(value.toInt) match {

73     }

100     }
101 
102     def toBoolean: Option[Boolean] = option.flatMap { value =>
103       Try(value.toBoolean) match {
104         case Success(v) => Some(v)
105         case Failure(_) =>
106           logger.warn(s"Invalid Boolean for property $property: $value")
107           Option(default).map(java.lang.Boolean.parseBoolean)
108       }
109     }
110 
111     private def fromSysProps: Option[String] =
112       Option(threadLocalValue.get).orElse(sys.props.get(property).filterNot(_.isEmpty))
