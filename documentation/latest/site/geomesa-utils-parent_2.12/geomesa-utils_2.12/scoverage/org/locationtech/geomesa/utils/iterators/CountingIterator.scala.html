15 /**
16  * Class for gathering a count for an iterator
17  *
18  * @param delegate iterator to count
19  * @param counter counting function
20  * @param callback count callback
21  * @tparam T type binding of iterator
22  */
23 class CountingIterator[T](delegate: CloseableIterator[T], counter: T => Int, callback: Long => Unit)
24     extends CloseableIterator[T] {
25 
26   private var total = 0L
27 
28   private val callbacks = ArrayBuffer(callback)
29 
30   /**
31    * Add another callback
32    *
33    * @param callback callback to be executed when iterator is closed, will be passed nanoseconds spent running the iterator
34    * @return this iterator, for method chaining
35    */
36   def addCallback(callback: Long => Unit): CountingIterator[T] = {
37     callbacks += callback
38     this
39   }
40 
41   override def hasNext: Boolean = delegate.hasNext
42 
43   override def next(): T = {
44     val result = delegate.next()
45     total += counter(result)
46     result
47   }
48 
49   override def close(): Unit = {
50     try { delegate.close() } finally {
51       callbacks.foreach(_.apply(total))
52     }
53   }
54 }
