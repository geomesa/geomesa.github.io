66   /**
67     * Render a line string
68     *
69     * @param lineString geometry
70     * @param weight weight
71     */
72   def render(lineString: LineString, weight: Double): Unit = {
73     if (lineString.getNumPoints > 0) {
74       var iN, jN = -1 // track the last pixel we've written to avoid double-counting
75 
76       // our working coordinates
77       var p0: Coordinate = null
78       var i0: Seq[Int] = null
79       var j0: Int = -1
80       var p1 = lineString.getCoordinateN(0)
81       var i1 = translate(p1.x)
82       var j1 = grid.j(p1.y)
83 
84       var n = 1
85       while (n < lineString.getNumPoints) {
86         // increment to the next pair of points
87         p0 = p1
88         i0 = i1
89         j0 = j1
90         p1 = lineString.getCoordinateN(n)
91         i1 = translate(p1.x)
92         j1 = grid.j(p1.y)
93 
94         if (i0.isEmpty || j0 == -1 || i1.isEmpty || j1 == -1) {
95           // line is not entirely contained in the grid region
96           // find the intersection of the line segment with the grid region
97           try {
98             val intersection = GeometryUtils.geoFactory.createLineString(Array(p0, p1)).intersection(grid.envelope)
99             if (!intersection.isEmpty) {
100               render(intersection, weight)
101               count -= 1 // don't double count
102             }
103           } catch {
104             case NonFatal(e) => logger.error(s"Error intersecting line string [$p0 $p1] with ${grid.envelope}", e)
105           }
106         } else {
107           val bresenham = grid.bresenhamLine(i0.head, j0, i1.head, j1)
108           // check the first point for overlap with last line segment
109           val (iF, jF) = bresenham.next
110           if (iF != iN || jF != jN) {
111             pixels(iF, jF) += weight
112             i0.tail.foreach { i0N =>
113               pixels(iF - i0.head + i0N, jF) += weight
114             }
115           }
116           if (!bresenham.hasNext) {
117             iN = iF
118             jN = jF
119           } else {
120             @tailrec
121             def writeNext(): Unit = {
122               val (i, j) = bresenham.next
123               pixels(i, j) += weight
124               i0.tail.foreach { i0N =>
125                 pixels(i - i0.head + i0N, j) += weight
126               }
127               if (bresenham.hasNext) {
128                 writeNext()
129               } else {
130                 iN = i
131                 jN = j
132               }
133             }
134             writeNext()
135           }
136         }
137 
138         n += 1
139       }
140     }
141     count += 1
142   }

159   /**
160     * Render a polygon
161     *
162     * @param polygon geometry
163     * @param weight weight
164     */
165   def render(polygon: Polygon, weight: Double): Unit = {
166     val envelope = polygon.getEnvelopeInternal
167     val imins = translate(envelope.getMinX)
168     val imaxes = translate(envelope.getMaxX)
169     val jmin = grid.j(envelope.getMinY)
170     val jmax = grid.j(envelope.getMaxY)
171 
172     if (imins.isEmpty || imaxes.isEmpty || jmin == -1 || jmax == -1) {
173       // polygon is not entirely contained in the grid region
174       // find the intersection of the polygon with the grid region
175       try {
176         val intersection = polygon.intersection(grid.envelope)
177         if (!intersection.isEmpty) {
178           render(intersection, weight)
179           count -= 1 // don't double count
180         }
181       } catch {
182         case NonFatal(e) => logger.error(s"Error intersecting polygon [$polygon] with ${grid.envelope}", e)
183       }
184     } else {
185       val imin = imins.head
186       val iLength = imaxes.head - imin + 1
187       val jLength = jmax - jmin + 1
188 
189       val raster = {
190         // use java awt graphics to draw our polygon on the grid
191         val image = new BufferedImage(iLength, jLength, BufferedImage.TYPE_BYTE_BINARY)
192         val graphics = image.createGraphics()
193 
194         val border = polygon.getExteriorRing
195         val xPoints = Array.ofDim[Int](border.getNumPoints)
196         val yPoints = Array.ofDim[Int](border.getNumPoints)
197         var i = 0
198         while (i < xPoints.length) {
199           val coord = border.getCoordinateN(i)
200           xPoints(i) = translate(coord.x).head - imin
201           yPoints(i) = grid.j(coord.y) - jmin
202           i += 1
203         }
204         graphics.fillPolygon(xPoints, yPoints, xPoints.length)
205         image.getRaster
206       }
207 
208       var i, j = 0
209       while (i < iLength) {
210         while (j < jLength) {
211           if (raster.getSample(i, j, 0) != 0) {
212             imins.foreach(im => pixels(i + im, j + jmin) += weight)
213           }
214           j += 1
215         }
216         j = 0
217         i += 1
218       }
219     }
220     count += 1
221   }

239     * Render an arbitrary geometry
240     *
241     * @param geometry geometry
242     * @param weight weight
243     */
244   def render(geometry: Geometry, weight: Double): Unit = {
245     geometry match {
246       case g: Point              => render(g, weight)
247       case g: LineString         => render(g, weight)
248       case g: Polygon            => render(g, weight)
249       case g: MultiPoint         => render(g, weight)

251       case g: MultiPolygon       => render(g, weight)
252       case g: GeometryCollection =>
253         var i = 0
254         while (i < g.getNumGeometries) {
255           render(g.getGeometryN(i), weight)
256           i += 1
257         }
258         count += (1 - i)
259 
260       case _ => throw new UnsupportedOperationException(s"Unexpected geometry type: $geometry")
261     }

292   /**
293     * Translate a point into the output envelope. If the envelope is larger than 360 degrees, points may
294     * be rendered more than once
295     *
296     * @param x longitude
297     * @return
298     */
299   private def translate(x: Double): Seq[Int] = {
300     if (x < xMin) {
301       val xt = x + (360d * math.ceil((xMin - x) / 360))
302       if (xt > xMax) { Seq.empty } else { widen(xt) }
303     } else if (x > xMax) {
304       val xt = x - (360d * math.ceil((x - xMax) / 360))
305       if (xt < xMin) { Seq.empty } else { widen(xt) }
306     } else {
307       widen(x)
308     }
309   }
310 
311   /**
312     * Returns any pixels that should be rendered for the given point
313     *
314     * @param x longitude
315     * @return
316     */
317   private def widen(x: Double): Seq[Int] = {
318     if (wide) {
319       val seq = Seq.newBuilder[Int]
320       // start with the smallest x value greater than xMin
321       var xup = x - 360d * math.floor((x - xMin) / 360)
322       while (xup <= xMax) {
323         seq += grid.i(xup)
324         xup += 360d
325       }
326       seq.result
327     } else {
328       Seq(grid.i(x))
329     }
330   }
