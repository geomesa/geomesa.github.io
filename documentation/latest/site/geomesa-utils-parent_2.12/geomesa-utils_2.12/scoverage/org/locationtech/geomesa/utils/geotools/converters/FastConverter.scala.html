24 /**
25   * Replacement for geotools `Converters`, which caches the converters for each pair of (from -> to) instead of
26   * re-creating them each time.
27   */

58   /**
59     * Convert the value into the given type
60     *
61     * @param value value to convert
62     * @param binding type to convert to
63     * @tparam T type binding
64     * @return converted value, or null if it could not be converted
65     */
66   def convert[T](value: Any, binding: Class[T]): T = {
67     if (value == null || binding.isAssignableFrom(value.getClass)) {
68       return value.asInstanceOf[T]
69     }
70 
71     val converters = cache.get((value.getClass, binding))
72 
73     val result = tryConvert(value, binding, converters)
74     if (result == null) {
75       val msg =
76         s"Could not convert '$value' (of type ${value.getClass.getName}) to ${binding.getName} " +
77           s"using ${converters.map(_.getClass.getName).mkString(", ")}"
78       logger.warn(msg)
79       logger.debug(msg, new RuntimeException())
80     }
81     result
82   }
83 
84   /**
85    * Convert the value into one of the given type
86    *
87    * @param value value to convert
88    * @param bindings type to convert to, in order of preference
89    * @tparam T type binding
90    * @return converted value, or null if it could not be converted
91    */
92   def convertFirst[T](value: Any, bindings: Iterator[Class[_ <: T]]): T = {
93     if (value == null) {
94       return null.asInstanceOf[T]
95     }
96 
97     val clas = value.getClass
98     val errors = ArrayBuffer.empty[(Class[_], Array[Converter])]
99 
100     while (bindings.hasNext) {
101       val binding = bindings.next
102       if (binding.isAssignableFrom(clas)) {
103         return value.asInstanceOf[T]
104       }
105 
106       val converters = cache.get((clas, binding))
107 
108       val result = tryConvert(value, binding, converters)
109       if (result != null) {
110         return result
111       } else {
112         errors += binding -> converters
113       }
114     }
115 
116     val msg =
117       s"Could not convert '$value' (of type ${value.getClass.getName}) to any of:" +
118         errors.map { case (b, c) => s"${b.getClass.getName} using ${c.map(_.getClass.getName).mkString(", ")}"}.mkString("\n  ", "\n  ", "")
119 
120     logger.warn(msg)
121     logger.debug(msg, new RuntimeException())
122 
123     null.asInstanceOf[T]
124   }

126   /**
127     * Convert the value into the given type, returning the default if it could not be converted or is null
128     *
129     * @param value value to convert
130     * @param default value to return if convert results in null
131     * @param ct class tag
132     * @tparam T type to convert to
133     * @return
134     */
135   def convertOrElse[T <: AnyRef](value: Any, default: => T)(implicit ct: ClassTag[T]): T = {
136     val attempt = convert(value, ct.runtimeClass.asInstanceOf[Class[T]])
137     if (attempt == null) { default } else { attempt }
138   }

140   /**
141     * Evaluate and convert an expression
142     *
143     * @param expression expression to evaluate
144     * @param binding type to convert to
145     * @tparam T type binding
146     * @return converted value, or null if it could not be converted
147     */
148   def evaluate[T](expression: Expression, binding: Class[T]): T = convert(expression.evaluate(null), binding)
149 
150   /**
151    * Try to convert the value
152    *
153    * @param value value
154    * @param binding expected return type
155    * @param converters converters to use
156    * @tparam T expected return type
157    * @return the converted value as type T, or null if could not convert
158    */
159   private def tryConvert[T](value: Any, binding: Class[T], converters: Array[Converter]): T = {
160     var i = 0
161     while (i < converters.length) {
162       try {
163         val result = converters(i).convert(value, binding)
164         if (result != null) {
165           return result
166         }
167       } catch {
168         case NonFatal(e) =>
169           logger.trace(s"Error converting $value (of type ${value.getClass.getName}) " +
170             s"to ${binding.getName} using converter ${converters(i).getClass.getName}:", e)
171       }
172       i += 1
173     }
174     null.asInstanceOf[T]
175   }
176 
177   private object ToStringConverter extends Converter {
178     override def convert[T](source: Any, target: Class[T]): T = source.toString.asInstanceOf[T]
179   }
