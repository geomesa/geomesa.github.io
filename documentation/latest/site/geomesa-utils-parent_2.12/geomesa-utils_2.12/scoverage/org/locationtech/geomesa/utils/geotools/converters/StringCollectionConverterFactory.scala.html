15 /**
16   * Converts between strings and collections (maps and lists).
17   *
18   * The collection subtype(s) need to be passed into the hints using keys from ConverterFactories.
19   * This is required, as we can't use reflection to figure out subtypes due to type erasure.
20   * If the subtypes were incorrect, we could get serialization exceptions.
21   */
22 class StringCollectionConverterFactory extends ConverterFactory {
23 
24   import StringCollectionConverterFactory._
25 
26   def createConverter(source: Class[_], target: Class[_], hints: Hints): Converter = {
27     if (source != classOf[String] || hints == null || hints.isEmpty) {
28       null
29     } else if (classOf[java.util.List[_]].isAssignableFrom(target)) {
30       val listType = hints.get(ListTypeKey).asInstanceOf[Class[_]]
31       if (listType == null) {
32         null
33       } else {
34         new StringToListConverter(listType)
35       }
36     } else if (classOf[java.util.Map[_, _]].isAssignableFrom(target)) {
37       val keyType   = hints.get(MapKeyTypeKey).asInstanceOf[Class[_]]
38       val valueType = hints.get(MapValueTypeKey).asInstanceOf[Class[_]]
39       if (keyType == null || valueType == null) {
40         null
41       } else {
42         new StringToMapConverter(keyType, valueType)
43       }
44     } else {
45       null
46     }
47   }
48 }

51 
52   val ListTypeKey     = new Key(classOf[Class[_]])
53   val MapKeyTypeKey   = new Key(classOf[Class[_]])
54   val MapValueTypeKey = new Key(classOf[Class[_]])

56   /**
57     * Converts a specially formatted string to a java list.
58     *
59     * Examples:
60     *   "[value1, value2]" // from java List.toString
61     *   "value1,value2" // from geomesa feature exporter conversion
62     *
63     * @param listType type of list elements
64     */
65   class StringToListConverter(listType: Class[_]) extends Converter {
66 
67     override def convert[T](source: scala.Any, target: Class[T]): T = {
68       val string = source.toString.trim
69       // strip outer designators if any
70       // java .toString looks like: [value1, value2]
71       // geomesa looks like: value1,value2
72       val isToString = string.startsWith("[") && string.endsWith("]")
73       val stripped = if (isToString) string.substring(1, string.length - 1) else string
74       if (string.isEmpty) {
75         new java.util.ArrayList[Any](0).asInstanceOf[T]
76       } else {
77         val result = new java.util.ArrayList[Any]
78         stripped.split(",").foreach { e =>
79           val converted = FastConverter.convert(e.trim, listType)
80           if (converted != null) {
81             result.add(converted)
82           }
83         }
84         if (result.isEmpty) {
85           // we failed to convert anything... return null and let someone else try
86           null.asInstanceOf[T]
87         } else {
88           result.asInstanceOf[T]
89         }
90       }
91     }
92   }
93 
94   /**
95     * Converts a specially formatted string to a java map.
96     *
97     * Examples:
98     *   "{key1=value1, key2=value2}" // from java Map.toString
99     *   "key1->value1,key2->value2" // from geomesa feature exporter conversion
100     *
101     * @param keyType map key type
102     * @param valueType map value type
103     */
104   class StringToMapConverter(keyType: Class[_], valueType: Class[_]) extends Converter {
105 
106     override def convert[T](source: scala.Any, target: Class[T]): T = {
107       val string = source.toString.trim
108       // pick our delimiter and strip outer designators if any
109       // java .toString looks like: {key1=value1, key2=value2}
110       // geomesa looks like: key1->value1,key2->value2
111       val isToString = string.startsWith("{") && string.endsWith("}")
112       val (stripped, split) = if (isToString) (string.substring(1, string.length - 1), "=") else (string, "->")
113       if (stripped.isEmpty) {
114         new java.util.HashMap[Any, Any](0).asInstanceOf[T]
115       } else {
116         val result = new java.util.HashMap[Any, Any]
117         stripped.split(",").map(_.split(split)).filter(_.length == 2).foreach { case Array(k, v) =>
118           val convertedKey = FastConverter.convert(k.trim, keyType)
119           if (convertedKey != null) {
120             val convertedValue = FastConverter.convert(v.trim, valueType)
121             if (convertedValue != null) {
122               result.put(convertedKey, convertedValue)
123             }
124           }
125         }
126         if (result.isEmpty) {
127           // we failed to convert anything... return null and let someone else try
128           null.asInstanceOf[T]
129         } else {
130           result.asInstanceOf[T]
131         }
132       }
133     }
134   }
