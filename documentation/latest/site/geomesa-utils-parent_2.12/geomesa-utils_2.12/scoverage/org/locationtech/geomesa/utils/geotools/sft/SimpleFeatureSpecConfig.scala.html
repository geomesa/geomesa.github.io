35   /**
36     * Parse a SimpleFeatureType spec from a typesafe Config
37     *
38     * @param conf config
39     * @param path instead of parsing the root config, parse the nested config at this path
40     * @return
41     */
42   def parse(conf: Config, path: Option[String]): (Option[String], SimpleFeatureSpec) = {
43     import org.locationtech.geomesa.utils.conf.ConfConversions._
44 
45     val toParse = path match {
46       case Some(p) => conf.getConfigOpt(p).map(conf.withFallback).getOrElse(conf)
47       case None    => conf
48     }
49     parse(toParse)
50   }

52   /**
53     * Convert a simple feature type to a typesafe config
54     *
55     * @param sft simple feature type
56     * @param includeUserData include user data
57     * @param includePrefix include the geomesa.sfts.XXX prefix
58     * @return
59     */
60   def toConfig(sft: SimpleFeatureType, includeUserData: Boolean, includePrefix: Boolean): Config = {
61     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
62 
63     // Update "default" options (dtg and geom)
64     val defaults = sft.getDtgField.toSeq ++ Option(sft.getGeomField)
65     val attributes = sft.getAttributeDescriptors.asScala.map { ad =>
66       val config = SimpleFeatureSpec.attribute(sft, ad).toConfigMap
67       if (defaults.contains(ad.getLocalName)) {
68         config.updated("default", "true").asJava
69       } else {
70         config.asJava
71       }
72     }
73 
74     val base = ConfigFactory.empty()
75       .withValue(TypeNamePath, ConfigValueFactory.fromAnyRef(sft.getTypeName))
76       .withValue(AttributesPath, ConfigValueFactory.fromIterable(attributes.asJava))
77 
78     val updated = if (includeUserData) {
79       val prefixes = sft.getUserDataPrefixes
80       // special handling for keywords delimiter
81       val keywords = Map(Keywords -> sft.getKeywords.asJava).filterNot(_._2.isEmpty)
82       val toConvert = keywords ++ sft.getUserData.asScala.collect {
83         case (k, v) if v != null && prefixes.exists(k.toString.startsWith) && k != Keywords => (k.toString, v)
84       }
85       val userData = ConfigValueFactory.fromMap(toConvert.asJava)
86       base.withValue(UserDataPath, userData)
87     } else {
88       base
89     }
90 
91     if (includePrefix) {
92       updated.atPath(s"${ConfigSftParsing.path}.${sft.getTypeName}")
93     } else {
94       updated
95     }
96   }
97 
98   /**
99     * Convert a simple feature type to a typesafe config and renders it as a string
100     *
101     * @param sft simple feature type
102     * @param includeUserData include user data
103     * @param concise concise or verbose string
104     * @return
105     */
106   def toConfigString(sft: SimpleFeatureType,
107                      includeUserData: Boolean,
108                      concise: Boolean,
109                      includePrefix: Boolean,
110                      json: Boolean): String = {
111     val opts = if (concise) {
112       ConfigRenderOptions.concise.setJson(json)
113     } else {
114       ConfigRenderOptions.defaults().setFormatted(true).setComments(false).setOriginComments(false).setJson(json)
115     }
116     toConfig(sft, includeUserData, includePrefix).root().render(opts)
117   }

119   private def parse(conf: Config): (Option[String], SimpleFeatureSpec) = {
120     import org.locationtech.geomesa.utils.conf.ConfConversions.RichConfig
121 
122     val name = conf.getStringOpt(TypeNamePath)
123     val attributes = conf.getConfigListOpt("fields").getOrElse(conf.getConfigList(AttributesPath)).asScala.map(buildField).toSeq
124     val opts = {
125       val userDataConfig = conf.getConfigOpt(UserDataPath).getOrElse(ConfigFactory.empty)
126       val base = userDataConfig.toStringMap()
127       if (!base.contains(Keywords)) { base } else {
128         // special case to handle keywords
129         base ++ userDataConfig.withOnlyPath(Keywords).toStringMap(KeywordsDelimiter)
130       }
131     }
132 
133     (name, SimpleFeatureSpec(attributes, opts))
134   }
135 
136   private def buildField(conf: Config): AttributeSpec = {
137     import org.locationtech.geomesa.utils.conf.ConfConversions.RichConfig
138 
139     val attribute = SimpleFeatureSpecParser.parseAttribute(s"${conf.getString(NamePath)}:${conf.getString(TypePath)}")
140     val options = conf.toStringMap().filterNot { case (k, _) => NonOptions.contains(k) }
141 
142     attribute match {
143       case s: SimpleAttributeSpec => s.copy(options = options)
144       case s: GeomAttributeSpec   => s.copy(options = options)
145       case s: ListAttributeSpec   => s.copy(options = options)
146       case s: MapAttributeSpec    => s.copy(options = options)
147     }
148   }
