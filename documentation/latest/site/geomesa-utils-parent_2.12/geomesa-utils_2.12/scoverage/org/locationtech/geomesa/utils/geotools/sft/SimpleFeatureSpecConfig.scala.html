35   /**
36     * Parse a SimpleFeatureType spec from a typesafe Config
37     *
38     * @param conf config
39     * @param path instead of parsing the root config, parse the nested config at this path
40     * @return
41     */
42   def parse(conf: Config, path: Option[String]): (Option[String], SimpleFeatureSpec) = {
43     val toParse = path.filter(conf.hasPath) match {
44       case Some(p) => conf.withFallback(conf.getConfig(p))
45       case None    => conf
46     }
47     parse(toParse)
48   }

50   /**
51     * Convert a simple feature type to a typesafe config
52     *
53     * @param sft simple feature type
54     * @param includeUserData include user data
55     * @param includePrefix include the geomesa.sfts.XXX prefix
56     * @return
57     */
58   def toConfig(sft: SimpleFeatureType, includeUserData: Boolean, includePrefix: Boolean): Config = {
59     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
60 
61     // Update "default" options (dtg and geom)
62     val defaults = sft.getDtgField.toSeq ++ Option(sft.getGeomField)
63     val attributes = sft.getAttributeDescriptors.asScala.map { ad =>
64       val config = SimpleFeatureSpec.attribute(sft, ad).toConfigMap
65       if (defaults.contains(ad.getLocalName)) {
66         config.updated("default", "true").asJava
67       } else {
68         config.asJava
69       }
70     }
71 
72     val base = ConfigFactory.empty()
73       .withValue(TypeNamePath, ConfigValueFactory.fromAnyRef(sft.getTypeName))
74       .withValue(AttributesPath, ConfigValueFactory.fromIterable(attributes.asJava))
75 
76     val updated = if (includeUserData) {
77       val prefixes = sft.getUserDataPrefixes
78       // special handling for keywords delimiter
79       val keywords = Map(Keywords -> sft.getKeywords.asJava).filterNot(_._2.isEmpty)
80       val toConvert = keywords ++ sft.getUserData.asScala.collect {
81         case (k, v) if v != null && prefixes.exists(k.toString.startsWith) && k != Keywords => (k.toString, v)
82       }
83       val userData = ConfigValueFactory.fromMap(toConvert.asJava)
84       base.withValue(UserDataPath, userData)
85     } else {
86       base
87     }
88 
89     if (includePrefix) {
90       updated.atPath(s"${ConfigSftParsing.path}.${sft.getTypeName}")
91     } else {
92       updated
93     }
94   }
95 
96   /**
97     * Convert a simple feature type to a typesafe config and renders it as a string
98     *
99     * @param sft simple feature type
100     * @param includeUserData include user data
101     * @param concise concise or verbose string
102     * @return
103     */
104   def toConfigString(sft: SimpleFeatureType,
105                      includeUserData: Boolean,
106                      concise: Boolean,
107                      includePrefix: Boolean,
108                      json: Boolean): String = {
109     val opts = if (concise) {
110       ConfigRenderOptions.concise.setJson(json)
111     } else {
112       ConfigRenderOptions.defaults().setFormatted(true).setComments(false).setOriginComments(false).setJson(json)
113     }
114     toConfig(sft, includeUserData, includePrefix).root().render(opts)
115   }

117   private def parse(conf: Config): (Option[String], SimpleFeatureSpec) = {
118     import org.locationtech.geomesa.utils.conf.ConfConversions.RichConfig
119 
120     val name = if (conf.hasPath(TypeNamePath)) { Option(conf.getString(TypeNamePath)) } else { None }
121     val attributes = {
122       val fields = if (conf.hasPath("fields")) { conf.getConfigList("fields") } else { conf.getConfigList(AttributesPath) }
123       fields.asScala.map(buildField).toSeq
124     }
125     val opts = {
126       val userDataConfig = if (conf.hasPath(UserDataPath)) { conf.getConfig(UserDataPath) } else { ConfigFactory.empty }
127       val base = userDataConfig.toStringMap()
128       if (!base.contains(Keywords)) { base } else {
129         // special case to handle keywords
130         base ++ userDataConfig.withOnlyPath(Keywords).toStringMap(KeywordsDelimiter)
131       }
132     }
133 
134     (name, SimpleFeatureSpec(attributes, opts))
135   }
136 
137   private def buildField(conf: Config): AttributeSpec = {
138     import org.locationtech.geomesa.utils.conf.ConfConversions.RichConfig
139 
140     val attribute = SimpleFeatureSpecParser.parseAttribute(s"${conf.getString(NamePath)}:${conf.getString(TypePath)}")
141     val options = conf.toStringMap().filterNot { case (k, _) => NonOptions.contains(k) }
142 
143     attribute match {
144       case s: SimpleAttributeSpec => s.copy(options = options)
145       case s: GeomAttributeSpec   => s.copy(options = options)
146       case s: ListAttributeSpec   => s.copy(options = options)
147       case s: MapAttributeSpec    => s.copy(options = options)
148     }
149   }
