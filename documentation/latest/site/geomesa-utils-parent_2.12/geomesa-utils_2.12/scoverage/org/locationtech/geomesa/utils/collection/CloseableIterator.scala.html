23 // A CloseableIterator is one which involves some kind of close function which should be called at the end of use.

86   private final class CloseableSingleIterator[A](elem: => A, closeIter: => Unit) extends CloseableIterator[A] {
87     private var result = true
88     override def hasNext: Boolean = result
89     override def next(): A = if (result) { result = false; elem } else { empty.next() }
90     override def close(): Unit = closeIter

93   /**
94     * Based on scala's ++ iterator implementation
95     *
96     * Avoid stack overflows when applying ++ to lots of iterators by flattening the unevaluated
97     * iterators out into a vector of closures.
98     */
99   private final class ConcatCloseableIterator[+A](queue: scala.collection.mutable.Queue[() => CloseableIterator[A]])
100       extends CloseableIterator[A] {
101 
102     private [this] var current: CloseableIterator[A] = queue.dequeue()()
103 
104     // Advance current to the next non-empty iterator
105     // current is set to empty when all iterators are exhausted
106     @tailrec
107     private [this] def advance(): Boolean = {
108       current.close()
109       if (queue.isEmpty) {
110         current = CloseableIterator.empty
111         false
112       } else {
113         current = queue.dequeue()()
114         current.hasNext || advance()
115       }
116     }
117 
118     override def hasNext: Boolean = current.hasNext || advance()
119     override def next(): A = current.next
120 
121     override def close(): Unit = {
122       current.close()
123       queue.foreach(_.apply().close())
124       queue.clear()
125     }
126   }
127 
128   private final class FlatMapCloseableIterator[A, B](source: CloseableIterator[A], f: A => GenTraversableOnce[B])
129       extends CloseableIterator[B] {
130 
131     private var cur: CloseableIterator[B] = empty
132 
133     @tailrec
134     override def hasNext: Boolean = cur.hasNext || {
135       cur.close()
136       if (source.hasNext) {
137         cur = CloseableIterator.wrap(f(source.next()))
138         hasNext
139       } else {
140         cur = empty
141         false
142       }
143     }
144 
145     override def next(): B = if (hasNext) { cur.next() } else { empty.next() }
146 
147     override def close(): Unit = CloseQuietly(cur, source).foreach(f => throw f)
148   }
149 }

168   def concat[B >: A](xs: => GenTraversableOnce[B]): CloseableIterator[B] = {
169     lazy val applied = CloseableIterator.wrap(xs)
170     val queue = new scala.collection.mutable.Queue[() => CloseableIterator[B]]
171     queue.enqueue(() => this)
172     queue.enqueue(() => applied)
173     new ConcatCloseableIterator[B](queue)
174   }
175 
176   // in scala 2.13 this method is final, and can cause resource leaks due to not returning a closeable iterator
177   override def ++[B >: A](that: => GenTraversableOnce[B]): CloseableIterator[B] =
178     throw new UnsupportedOperationException("Not safe for cross-scala usage")
179 
180   override def flatMap[B](f: A => GenTraversableOnce[B]): CloseableIterator[B] =
181     new FlatMapCloseableIterator(this, f)
