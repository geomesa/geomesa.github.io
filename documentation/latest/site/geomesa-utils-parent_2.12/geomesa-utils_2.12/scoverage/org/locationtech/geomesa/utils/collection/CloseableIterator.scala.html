93   /**
94     * Based on scala's ++ iterator implementation
95     *
96     * Avoid stack overflows when applying ++ to lots of iterators by flattening the unevaluated
97     * iterators out into a vector of closures.
98     */
99   private final class ConcatCloseableIterator[+A](queue: scala.collection.mutable.Queue[() => CloseableIterator[A]])
100       extends CloseableIterator[A] {
101 
102     private [this] var current: CloseableIterator[A] = queue.dequeue()()
103 
104     // Advance current to the next non-empty iterator
105     // current is set to empty when all iterators are exhausted
106     @tailrec
107     private [this] def advance(): Boolean = {
108       current.close()
109       if (queue.isEmpty) {
110         current = CloseableIterator.empty
111         false
112       } else {
113         current = queue.dequeue()()
114         current.hasNext || advance()
115       }
116     }
117 
118     override def hasNext: Boolean = current.hasNext || advance()
119     override def next(): A = current.next
120 
121     override def close(): Unit = {
122       current.close()
123       queue.foreach(_.apply().close())
124       queue.clear()
125     }
126   }
127 
128   private final class FlatMapCloseableIterator[A, B](source: CloseableIterator[A], f: A => GenTraversableOnce[B])
129       extends CloseableIterator[B] {
130 
131     private var cur: CloseableIterator[B] = empty
132 
133     @tailrec
134     override def hasNext: Boolean = cur.hasNext || {
135       cur.close()
136       if (source.hasNext) {
137         cur = CloseableIterator.wrap(f(source.next()))
138         hasNext
139       } else {
140         cur = empty
141         false
142       }
143     }
144 
145     override def next(): B = if (hasNext) { cur.next() } else { empty.next() }
146 
147     override def close(): Unit = CloseQuietly(cur, source).foreach(f => throw f)
148   }

151 /**
152  * An iterator that should be closed after use.
153  *
154  * For convenience, some methods that consume the iterator will automatically close the iterator when invoked:
155  *
156  * <ul>
157  * <li>`toList`</li>
158  * <li>`foreach`</li>
159  * <li>`size`</li>
160  * </ul>
161  *
162  * Note that other methods (for example, `length`) may end up invoking these methods, due to the internal implementation of
163  * collections.
164  *
165  * @tparam A iterator values
166  */
167 trait CloseableIterator[+A] extends Iterator[A] with Closeable {
168 
169   // note: in scala 2.12, toList ends up calling foreach, which closes the iterator
170   // override def toList: List[A] = try { super.toList } finally { close() }
171 
172   override def foreach[U](f: A => U): Unit = try { super.foreach(f) } finally { close() }
173 
174   override def size: Int =  try { super.size } finally { close() }
175 
176   override def map[B](f: A => B): CloseableIterator[B] = new CloseableIteratorImpl(super.map(f), close())
177 
178   override def filter(p: A => Boolean): CloseableIterator[A] = new CloseableIteratorImpl(super.filter(p), close())
179 
180   override def withFilter(p: A => Boolean): CloseableIterator[A] = filter(p)
181 
182   override def filterNot(p: A => Boolean): CloseableIterator[A] = new CloseableIteratorImpl(super.filterNot(p), close())
183 
184   override def take(n: Int): CloseableIterator[A] = new CloseableIteratorImpl(super.take(n), close())
185 
186   override def takeWhile(p: A => Boolean): CloseableIterator[A] = new CloseableIteratorImpl(super.takeWhile(p), close())
187 
188   override def drop(n: Int): CloseableIterator[A] = new CloseableIteratorImpl(super.drop(n), close())
189 
190   override def dropWhile(p: A => Boolean): CloseableIterator[A] = new CloseableIteratorImpl(super.dropWhile(p), close())
191 
192   override def collect[B](pf: PartialFunction[A, B]): CloseableIterator[B] = new CloseableIteratorImpl(super.collect(pf), close())
193 
194   def concat[B >: A](xs: => GenTraversableOnce[B]): CloseableIterator[B] = {
195     lazy val applied = CloseableIterator.wrap(xs)
196     val queue = new scala.collection.mutable.Queue[() => CloseableIterator[B]]
197     queue.enqueue(() => this)
198     queue.enqueue(() => applied)
199     new ConcatCloseableIterator[B](queue)
200   }
201 
202   override def flatMap[B](f: A => GenTraversableOnce[B]): CloseableIterator[B] = new FlatMapCloseableIterator(this, f)
203 
204   // in scala 2.13 this method is final, and can cause resource leaks due to not returning a closeable iterator
205   override def ++[B >: A](that: => GenTraversableOnce[B]): CloseableIterator[B] =
206     throw new UnsupportedOperationException("Not safe for cross-scala usage")
207 }
