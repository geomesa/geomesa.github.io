41   val Epoch: ZonedDateTime = ZonedDateTime.ofInstant(Instant.EPOCH, ZoneOffset.UTC)

43   object TemporalQueries {
44     val ZonedQuery: TemporalQuery[ZonedDateTime] = new TemporalQuery[ZonedDateTime] {
45       override def queryFrom(temporal: TemporalAccessor): ZonedDateTime = ZonedDateTime.from(temporal)
46     }
47     val LocalQuery: TemporalQuery[LocalDateTime] = new TemporalQuery[LocalDateTime] {
48       override def queryFrom(temporal: TemporalAccessor): LocalDateTime = LocalDateTime.from(temporal)
49     }
50     val LocalDateQuery: TemporalQuery[LocalDate] = new TemporalQuery[LocalDate] {
51       override def queryFrom(temporal: TemporalAccessor): LocalDate = LocalDate.from(temporal)
52     }
53     val YearMonthQuery: TemporalQuery[YearMonth] = new TemporalQuery[YearMonth] {
54       override def queryFrom(temporal: TemporalAccessor): YearMonth = YearMonth.from(temporal)
55     }
56     val YearQuery: TemporalQuery[Year] = new TemporalQuery[Year] {
57       override def queryFrom(temporal: TemporalAccessor): Year = Year.from(temporal)
58     }

61   /**
62     * Parses a date string, with optional time and zone
63     *
64     * @param value date string
65     * @param format date formatter, default ISO format with optional time and zone
66     * @return
67     */
68   def parse(value: String, format: DateTimeFormatter = format): ZonedDateTime = {
69     import TemporalQueries._
70     format.parseBest(value, ZonedQuery, LocalQuery, LocalDateQuery, YearMonthQuery, YearQuery) match {
71       case d: ZonedDateTime => d
72       case d: LocalDateTime => d.atZone(ZoneOffset.UTC)
73       case d: LocalDate     => d.atTime(LocalTime.MIN).atZone(ZoneOffset.UTC)
74       case d: YearMonth     => d.atDay(1).atTime(LocalTime.MIN).atZone(ZoneOffset.UTC)
75       case d: Year          => d.atMonth(1).atDay(1).atTime(LocalTime.MIN).atZone(ZoneOffset.UTC)
76     }
77   }
78 
79   /**
80     * Parses a date string, with optional time and zone
81     *
82     * @param value date string
83     * @param format date formatter, default ISO format with optional time and zone
84     * @return
85     */
86   def parseInstant(value: String, format: DateTimeFormatter = format): Instant = {
87     import TemporalQueries.{LocalDateQuery, LocalQuery, ZonedQuery}
88     format.parseBest(value, ZonedQuery, LocalQuery, LocalDateQuery) match {
89       case d: ZonedDateTime => d.toInstant
90       case d: LocalDateTime => d.toInstant(ZoneOffset.UTC)
91       case d: LocalDate     => d.atTime(LocalTime.MIN).toInstant(ZoneOffset.UTC)
92     }
93   }
94 
95   /**
96     * Parses a date string, with optional time and zone
97     *
98     * @param value date string
99     * @param format date formatter, default ISO format with optional time and zone
100     * @return
101     */
102   def parseDate(value: String, format: DateTimeFormatter = format): Date = {
103     import TemporalQueries.{LocalDateQuery, LocalQuery, ZonedQuery}
104     format.parseBest(value, ZonedQuery, LocalQuery, LocalDateQuery) match {
105       case d: ZonedDateTime => Date.from(d.toInstant)
106       case d: LocalDateTime => Date.from(d.toInstant(ZoneOffset.UTC))
107       case d: LocalDate     => Date.from(d.atTime(LocalTime.MIN).toInstant(ZoneOffset.UTC))
108     }
109   }
110 
111   /**
112     * Parses a date string, with optional time and zone
113     *
114     * @param value date string
115     * @param format date formatter, default ISO format with optional time and zone
116     * @return
117     */
118   def parseMillis(value: String, format: DateTimeFormatter = format): Long = {
119     import TemporalQueries.{LocalDateQuery, LocalQuery, ZonedQuery}
120     format.parseBest(value, ZonedQuery, LocalQuery, LocalDateQuery) match {
121       case d: ZonedDateTime => d.toInstant.toEpochMilli
122       case d: LocalDateTime => d.toInstant(ZoneOffset.UTC).toEpochMilli
123       case d: LocalDate     => d.atTime(LocalTime.MIN).toInstant(ZoneOffset.UTC).toEpochMilli
124     }
125   }
126 
127   def format(value: ZonedDateTime, format: DateTimeFormatter = format): String = value.format(format)
128 
129   def formatDate(value: Date, format: DateTimeFormatter = format): String =
130     ZonedDateTime.ofInstant(value.toInstant, ZoneOffset.UTC).format(format)
131 
132   def formatInstant(value: Instant, format: DateTimeFormatter = format): String =
133     ZonedDateTime.ofInstant(value, ZoneOffset.UTC).format(format)
134 
135   def formatMillis(value: Long, format: DateTimeFormatter = format): String =
136     ZonedDateTime.ofInstant(Instant.ofEpochMilli(value), ZoneOffset.UTC).format(format)
