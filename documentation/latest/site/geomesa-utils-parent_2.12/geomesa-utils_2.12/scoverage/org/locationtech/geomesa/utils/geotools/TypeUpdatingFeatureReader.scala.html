15 /** A [[DelegatingFeatureReader]] that re-types simple features.  Unlike [[ReTypeFeatureReader]] this
16   * feature reader will preserve user data.
17   *
18   * @param delegate the delegate reader
19   * @param featureType the projected type
20   */
21 class TypeUpdatingFeatureReader(delegate: FeatureReader[SimpleFeatureType, SimpleFeature],
22                                 featureType: SimpleFeatureType)
23   extends DelegatingFeatureReader[SimpleFeatureType, SimpleFeature] {
24 
25   override val getDelegate: FeatureReader[SimpleFeatureType, SimpleFeature] = delegate
26 
27   override def next(): SimpleFeature = FeatureUtils.retype(delegate.next(), featureType)
28 
29   override def hasNext: Boolean = delegate.hasNext
30   override def getFeatureType: SimpleFeatureType = featureType
31   override def close(): Unit = delegate.close()
32 }
