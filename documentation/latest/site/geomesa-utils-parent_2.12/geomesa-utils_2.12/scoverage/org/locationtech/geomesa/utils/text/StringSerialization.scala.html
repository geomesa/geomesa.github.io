32   /**
33     * Encode a sequence of string values to a single string
34     *
35     * @param values values
36     * @return
37     */
38   def encodeSeq(values: Seq[String]): String = {
39     if (values.isEmpty) { "" } else {
40       val sb = new java.lang.StringBuilder()
41       val printer = new CSVPrinter(sb, CSVFormat.DEFAULT)
42       values.foreach(printer.print)
43       sb.toString
44     }
45   }

47   /**
48     * Recover a sequence of string values encoded with `encodedSeq`
49     *
50     * @param values encoded string
51     * @return
52     */
53   def decodeSeq(values: String): Seq[String] = {
54     if (values.isEmpty) { Seq.empty } else {
55       WithClose(CSVParser.parse(values, CSVFormat.DEFAULT))(_.iterator.next.iterator.asScala.toList)
56     }
57   }

77   /**
78     * Encode a map of sequences as a string
79     *
80     * @param map map of keys to sequences of values
81     * @return
82     */
83   def encodeSeqMap(map: Map[String, Seq[AnyRef]]): String = {
84     val sb = new java.lang.StringBuilder
85     val printer = new CSVPrinter(sb, CSVFormat.DEFAULT)
86     map.foreach { case (k, v) =>
87       val strings = v.headOption match {
88         case Some(_: Date) => v.map(d => ZonedDateTime.ofInstant(d.asInstanceOf[Date].toInstant, ZoneOffset.UTC).format(dateFormat))
89         case _ => v
90       }
91       printer.print(k)
92       strings.foreach(printer.print)
93       printer.println()
94     }
95     sb.toString
96   }

110   /**
111     * Decode a map of sequences from a string encoded by @see encodeSeqMap
112     *
113     * @param encoded encoded map
114     * @return decoded map
115     */
116   def decodeSeqMap(encoded: String, bindings: Map[String, Class[_]]): Map[String, Array[AnyRef]] = {
117     // encoded as CSV, first element of each row is key, rest is value
118     WithClose(CSVParser.parse(encoded, CSVFormat.DEFAULT)) { parser =>
119       parser.iterator.asScala.map { record =>
120         val iter = record.iterator.asScala
121         val key = iter.next
122         val values = bindings.get(key) match {
123           case Some(c) if c == classOf[String]              => iter.toArray[AnyRef]
124           case Some(c) if c == classOf[Integer]             => iter.map(Integer.valueOf).toArray[AnyRef]
125           case Some(c) if c == classOf[java.lang.Long]      => iter.map(java.lang.Long.valueOf).toArray[AnyRef]
126           case Some(c) if c == classOf[java.lang.Float]     => iter.map(java.lang.Float.valueOf).toArray[AnyRef]
127           case Some(c) if c == classOf[java.lang.Double]    => iter.map(java.lang.Double.valueOf).toArray[AnyRef]
128           case Some(c) if classOf[Date].isAssignableFrom(c) => iter.map(v => Date.from(ZonedDateTime.parse(v, dateFormat).toInstant)).toArray[AnyRef]
129           case Some(c) if c == classOf[java.lang.Boolean]   => iter.map(java.lang.Boolean.valueOf).toArray[AnyRef]
130           case c => logger.warn(s"No conversion defined for encoded attribute '$key' of type ${c.orNull}"); iter.toArray[AnyRef]
131         }
132         key -> values
133       }.toMap
134     }
135   }

137   /**
138     * Encode non-alphanumeric characters in a string with
139     * underscore plus hex digits representing the bytes. Note
140     * that multibyte characters will be represented with multiple
141     * underscores and bytes...e.g. _8a_2f_3b
142     */
143   def alphaNumericSafeString(input: String): String = {
144     if (AlphaNumericPattern.matcher(input).matches()) { input } else {
145       val sb = new StringBuilder
146       input.foreach { c =>
147         if (AlphaNumeric.contains(c)) { sb.append(c) } else {
148           val hex = Hex.encodeHex(c.toString.getBytes(StandardCharsets.UTF_8))
149           val encoded = hex.grouped(2).map(arr => "_" + arr(0) + arr(1)).mkString.toLowerCase(Locale.US)
150           sb.append(encoded)
151         }
152       }
153       sb.toString()
154     }
155   }
156 
157   def decodeAlphaNumericSafeString(input: String): String = {
158     if (AlphaNumericPattern.matcher(input).matches()) { input } else {
159       val sb = new StringBuilder
160       var i = 0
161       while (i < input.length) {
162         val c = input.charAt(i)
163         if (c != '_') { sb.append(c) } else {
164           i += 2
165           sb.append(new String(Hex.decodeHex(Array(input.charAt(i - 1), input.charAt(i))), StandardCharsets.UTF_8))
166         }
167         i += 1
168       }
169       sb.toString()
170     }
171   }
