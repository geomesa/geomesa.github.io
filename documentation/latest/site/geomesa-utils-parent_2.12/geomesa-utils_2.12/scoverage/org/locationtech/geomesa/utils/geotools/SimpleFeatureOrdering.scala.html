14 /**
15   * Ordering for simple features
16   */
17 object SimpleFeatureOrdering {
18 
19   /**
20    * Sort on an attribute by name. `null`, `"id"` or an empty string can be used to indicate
21    * 'natural' ordering by feature ID
22    *
23    * @param sft simple feature type
24    * @param sortBy attribute to sort by
25    * @return
26    */
27   def apply(sft: SimpleFeatureType, sortBy: String): Ordering[SimpleFeature] = apply(sft, sortBy, reverse = false)
28 
29   /**
30    * Sort on an attribute by name. `null`, `"id"` or an empty string can be used to indicate
31    * * 'natural' ordering by feature ID
32    *
33    * @param sft simple feature type
34    * @param sortBy attribute to sort by
35    * @param reverse reverse the sort (from ascending to descending)
36    * @return
37    */
38   def apply(sft: SimpleFeatureType, sortBy: String, reverse: Boolean): Ordering[SimpleFeature] = {
39     val sort = if (sortBy == null || sortBy.isEmpty || sortBy.equalsIgnoreCase("id")) { FidOrdering } else {
40       val i = sft.indexOf(sortBy)
41       if (i == -1) {
42         throw new IllegalArgumentException(s"Trying to sort by an attribute that is not in the schema: $sortBy")
43       }
44       new SimpleFeatureOrdering(i, AttributeOrdering(sft.getDescriptor(i)))
45     }
46     if (reverse) { sort.reverse } else { sort }
47   }
48 
49   /**
50    * Sort by multiple attributes by name
51    *
52    * @param sft simple feature type
53    * @param sortBy pairs of (attribute name, reverse ordering)
54    * @return
55    */
56   def apply(sft: SimpleFeatureType, sortBy: Seq[(String, Boolean)]): Ordering[SimpleFeature] = {
57     if (sortBy.lengthCompare(1) == 0) {
58       apply(sft, sortBy.head._1, sortBy.head._2)
59     } else {
60       new TieredOrdering(sortBy.map { case (field, reverse) => apply(sft, field, reverse) })
61     }
62   }
63 
64   /**
65    * Sort on a geotools SortBy instance
66    *
67    * @param sft simple feature type
68    * @param sortBy sort by
69    * @return
70    */
71   def apply(sft: SimpleFeatureType, sortBy: SortBy): Ordering[SimpleFeature] = {
72     val name = Option(sortBy.getPropertyName).map(_.getPropertyName).orNull
73     apply(sft, name, sortBy.getSortOrder == SortOrder.DESCENDING)
74   }
75 
76   /**
77    * Sort on a geotools SortBy array
78    *
79    * @param sft simple feature type
80    * @param sortBy sort by
81    * @return
82    */
83   def apply(sft: SimpleFeatureType, sortBy: Array[SortBy]): Ordering[SimpleFeature] = {
84     if (sortBy.length == 1) {
85       apply(sft, sortBy.head)
86     } else {
87       new TieredOrdering(sortBy.map(apply(sft, _)))
88     }
89   }
90 
91   def fid: Ordering[SimpleFeature] = FidOrdering
92 
93   object FidOrdering extends Ordering[SimpleFeature] {
94     override def compare(x: SimpleFeature, y: SimpleFeature): Int = x.getID.compareTo(y.getID)
95   }
96 
97   private class SimpleFeatureOrdering(i: Int, delegate: Ordering[AnyRef]) extends Ordering[SimpleFeature] {
98     override def compare(x: SimpleFeature, y: SimpleFeature): Int =
99       delegate.compare(x.getAttribute(i), y.getAttribute(i))
100   }
101 
102   private class TieredOrdering[T](orderings: Seq[Ordering[T]]) extends Ordering[T] {
103     override def compare(x: T, y: T): Int = {
104       orderings.foreach { o =>
105         val i = o.compare(x, y)
106         if (i != 0) {
107           return i
108         }
109       }
110       0
111     }
112   }
113 }
