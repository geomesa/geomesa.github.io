45       }
46     }

49   def buildSingleClass(sft: SimpleFeatureType, name: String, pkg: String, out: Writer): Unit = {
50     out.append(s"package $pkg\n\n")
51     out.append("import org.geotools.api.feature.simple.SimpleFeature\n\n")
52     out.append(s"// this class was generated by ${GenerateRichFeatureModels.getClass.getName} for feature type ${sft.getTypeName}\n")
53     out.append(s"object $name {\n\n")
54     out.append(s"  implicit class RichFeature(val sf: SimpleFeature) extends AnyVal {\n")
55     var i = 0
56     while (i < sft.getAttributeCount) {
57       val attr = AttributeMethods(sft, i)
58       out.append("\n")
59       out.append(s"    ${attr.getter}\n")
60       out.append(s"    ${attr.optionGetter}\n")
61       out.append(s"    ${attr.setter}\n")
62       i += 1
63     }
64     out.append(
65       s"""
66          |    def debug(): String = {
67          |      import scala.collection.JavaConverters._
68          |      val sb = new StringBuilder(s"$${sf.getType.getTypeName}:$${sf.getID}")
69          |      sf.getProperties.asScala.foreach(p => sb.append(s"|$${p.getName.getLocalPart}=$${p.getValue}"))
70          |      sb.toString()
71          |    }
72          |""".stripMargin)
73     out.append("  }\n}")
74   }

76   private[geotools] def getAttributeBinding(descriptor: AttributeDescriptor): String = {
77     import org.locationtech.geomesa.utils.geotools.RichAttributeDescriptors._
78 
79     def subtype(c: Class[_]): String = c match {
80       case null => "String"
81       case c => c.getCanonicalName
82     }
83 
84     val binding = descriptor.getType.getBinding.getCanonicalName
85     if (descriptor.isList) {
86       s"$binding[${subtype(descriptor.getListType())}]"
87     } else if (descriptor.isMap) {
88       val (keyType, valueType) = descriptor.getMapTypes()
89       s"$binding[${subtype(keyType)},${subtype(valueType)}]"
90     } else {
91       binding
92     }
93   }

95   private def safeIdentifier(name: String): String = {
96     val safe = new StringBuilder()
97     var i = 0
98     var upper = true // camel case
99     while (i < name.length) {
100       val char = name.charAt(i)
101       if (char == '_'||
102           (i == 0 && !Character.isJavaIdentifierStart(char)) ||
103           (i != 0 && !Character.isJavaIdentifierPart(char))) {
104         upper = true
105       } else if (upper) {
106         safe.append(char.toUpper)
107         upper = false
108       } else {
109         safe.append(char)
110       }
111       i += 1
112     }
113     if (safe.isEmpty) {
114       throw new IllegalArgumentException(s"Can't create valid Java identifier from $name")
115     }
116     safe.toString
117   }

119   private class AttributeMethods(name: String, index: Int, clazz: String) {
120     def getter: String = s"def get$name: $clazz = sf.getAttribute($index).asInstanceOf[$clazz]"
121     def optionGetter: String = s"def opt$name: Option[$clazz] = Option(get$name)"
122     def setter: String = s"def set$name(x: $clazz): Unit = sf.setAttribute($index, x)"
123   }
124 
125   private object AttributeMethods {
126     def apply(sft: SimpleFeatureType, i: Int): AttributeMethods = {
127       val descriptor = sft.getDescriptor(i)
128       val binding = getAttributeBinding(descriptor)
129       new AttributeMethods(safeIdentifier(descriptor.getLocalName), i, binding)
130     }
131   }
