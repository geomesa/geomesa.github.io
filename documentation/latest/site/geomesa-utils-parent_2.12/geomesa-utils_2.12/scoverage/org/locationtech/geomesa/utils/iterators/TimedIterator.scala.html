13 /**
14  * Class for gathering timing metrics on an iterator
15  *
16  * @param delegate iterator to time
17  * @param callback timer callback, will be passed nanoseconds spent running the iterator
18  * @tparam T type binding of iterator
19  */

21 
22   private var total = 0L
23 
24   override def hasNext: Boolean = {
25     val start = System.nanoTime()

29   }
30

39     val start = System.nanoTime()
40     try { delegate.close() } finally {

43     }
44   }
