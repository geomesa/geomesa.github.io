338   /**
339     * Creates an immutable copy of the simple feature type
340     *
341     * Note that immutable types will never be `equals` to a mutable type, due to specific class checking
342     * in the geotools implementation
343     *
344     * Note that some parts of the feature type may still be mutable - in particular AttributeType,
345     * GeometryType and SuperType are not used by geomesa so we don't bother with them. In addition,
346     * user data keys and values may be mutable objects, so while the user data map will not change,
347     * the values inside may
348     *
349     * @param sft simple feature type
350     * @param extraData additional user data to add to the simple feature type
351     * @return immutable copy of the simple feature type
352     */
353   def immutable(
354       sft: SimpleFeatureType,
355       extraData: java.util.Map[_ <: AnyRef, _ <: AnyRef] = null): SimpleFeatureType = {
356     sft match {
357       case immutable: ImmutableSimpleFeatureType if extraData == null || extraData.isEmpty => immutable
358       case _ =>
359         val schema = new java.util.ArrayList[AttributeDescriptor](sft.getAttributeCount)
360         var geom: GeometryDescriptor = null
361         var i = 0
362         while (i < sft.getAttributeCount) {
363           val descriptor = sft.getDescriptor(i)
364           val im = immutable(descriptor)
365           if (descriptor == sft.getGeometryDescriptor) {
366             geom = im.asInstanceOf[GeometryDescriptor]
367           }
368           schema.add(im)
369           i += 1
370         }
371         val userData = Option(extraData).filterNot(_.isEmpty).map { data =>
372           val map = new java.util.HashMap[AnyRef, AnyRef](data.size() + sft.getUserData.size)
373           map.putAll(sft.getUserData)
374           map.putAll(data)
375           map
376         }
377 
378         new ImmutableSimpleFeatureType(sft.getName, schema, geom, sft.isAbstract, sft.getRestrictions, sft.getSuper,
379           sft.getDescription, userData.getOrElse(sft.getUserData))
380     }
381   }

460   /**
461     * Compare two feature types. This method compares the the schemas of each feature type, i.e. the
462     * names and type bindings of the attributes. The result will be:
463     *
464     * <ul>
465     *   <li>a positive int if type A is a sub type or reorder of type B</li>
466     *   <li>zero if type A has the same schema as type B</li>
467     *   <li>a negative int if none of the above</li>
468     * </ul>
469     *
470     * Note that this shouldn't be used for sorting.
471     *
472     * Note: GeoTools DataUtilities has a similar method that does not work correctly
473     *
474     * @param a schema A
475     * @param b schema B
476     * @return comparison
477     */
478   def compare(a: SimpleFeatureType, b: SimpleFeatureType): Int = {
479     if (a.getAttributeCount > b.getAttributeCount) {
480       return -1 // not equals, subtype or reorder
481     }
482 
483     var exact = a.getAttributeCount == b.getAttributeCount
484 
485     var i = 0
486     while (i < a.getAttributeCount) {
487       val ad = a.getDescriptor(i)
488       val bd = b.getDescriptor(i)
489       if (ad.getLocalName == bd.getLocalName) {
490         if (ad.getType.getBinding != bd.getType.getBinding) {
491           if (bd.getType.getBinding.isAssignableFrom(ad.getType.getBinding)) {
492             exact = false // we've seen a subtyping
493           } else {
494             return -1 // not a subtype/reorder
495           }
496         }
497       } else {
498         val reorder = b.getDescriptor(ad.getLocalName)
499         if (reorder == null || !reorder.getType.getBinding.isAssignableFrom(ad.getType.getBinding)) {
500           return -1 // not a subtype/reorder
501         } else {
502           exact = false // we've seen a reordering
503         }
504       }
505       i += 1
506     }
507 
508     if (exact) { 0 } else {
509       1  // we haven't hit an absolute inequality, but we've found a reordering or subtype
510     }
511   }

514    * Compares that index tables are the same format (key and value) between the two feature types
515    *
516    * @param a first type

518    * @return 0 if indices are the same, non-zero otherwise

520   def compareIndexConfigs(a: SimpleFeatureType, b: SimpleFeatureType): Int = {
521     def compareAttributes(i: Int): Boolean = {
522       val (ad, bd) = (a.getDescriptor(i), b.getDescriptor(i))
523       ad.getType.getBinding == bd.getType.getBinding && AttributeChecks.forall(c => c(ad) == c(bd))
524     }
525     if (a.getAttributeCount == b.getAttributeCount &&
526         IndexChecks.forall(c => c(a) == c(b)) &&
527         Range(0, a.getAttributeCount).forall(compareAttributes)) {
528       0
529     } else {
530       1
531     }

534   private def createFeatureType(namespace: String,
535                                 name: String,
536                                 spec: SimpleFeatureSpec,
537                                 factory: Option[FeatureTypeFactory] = None): SimpleFeatureType = {
538     import AttributeOptions.OptDefault
539 
540     val defaultGeom = {
541       val geomAttributes = spec.attributes.collect { case g: GeomAttributeSpec => g }
542       geomAttributes.find(_.options.get(OptDefault).exists(_.toBoolean))
543           .orElse(geomAttributes.headOption)
544           .map(_.name)
545     }
546     val defaultDate = if (spec.options.get(IndexIgnoreDtg).exists(toBoolean)) { None } else {
547       val dateAttributes = spec.attributes.filter(a => classOf[Date].isAssignableFrom(a.clazz))
548       spec.options.get(DefaultDtgField).flatMap(dtg => dateAttributes.find(_.name == dtg))
549           .orElse(dateAttributes.find(_.options.get(OptDefault).exists(_.toBoolean)))
550           .orElse(dateAttributes.headOption)
551           .map(_.name)
552     }
553 
554     val b = factory.map(new SimpleFeatureTypeBuilder(_)).getOrElse(new SimpleFeatureTypeBuilder())
555     b.setNamespaceURI(namespace)
556     b.setName(name)
557     b.addAll(spec.attributes.map(_.toDescriptor).asJava)
558     defaultGeom.foreach(b.setDefaultGeometry)
559 
560     val sft = b.buildFeatureType()
561     sft.getUserData.putAll(spec.options.asJava)
562     defaultDate.foreach(sft.setDtgField)
563     sft
564   }
