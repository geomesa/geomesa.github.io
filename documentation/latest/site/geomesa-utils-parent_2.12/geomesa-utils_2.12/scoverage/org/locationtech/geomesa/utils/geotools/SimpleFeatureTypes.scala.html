293   /**
294     * Creates an immutable copy of the simple feature type
295     *
296     * Note that immutable types will never be `equals` to a mutable type, due to specific class checking
297     * in the geotools implementation
298     *
299     * Note that some parts of the feature type may still be mutable - in particular AttributeType,
300     * GeometryType and SuperType are not used by geomesa so we don't bother with them. In addition,
301     * user data keys and values may be mutable objects, so while the user data map will not change,
302     * the values inside may
303     *
304     * @param sft simple feature type
305     * @param extraData additional user data to add to the simple feature type
306     * @return immutable copy of the simple feature type
307     */
308   def immutable(
309       sft: SimpleFeatureType,
310       extraData: java.util.Map[_ <: AnyRef, _ <: AnyRef] = null): SimpleFeatureType = {
311     sft match {
312       case immutable: ImmutableSimpleFeatureType if extraData == null || extraData.isEmpty => immutable
313       case _ =>
314         val schema = new java.util.ArrayList[AttributeDescriptor](sft.getAttributeCount)
315         var geom: GeometryDescriptor = null
316         var i = 0
317         while (i < sft.getAttributeCount) {
318           val descriptor = sft.getDescriptor(i)
319           val im = immutable(descriptor)
320           if (descriptor == sft.getGeometryDescriptor) {
321             geom = im.asInstanceOf[GeometryDescriptor]
322           }
323           schema.add(im)
324           i += 1
325         }
326         val userData = Option(extraData).filterNot(_.isEmpty).map { data =>
327           val map = new java.util.HashMap[AnyRef, AnyRef](data.size() + sft.getUserData.size)
328           map.putAll(sft.getUserData)
329           map.putAll(data)
330           map
331         }
332 
333         new ImmutableSimpleFeatureType(sft.getName, schema, geom, sft.isAbstract, sft.getRestrictions, sft.getSuper,
334           sft.getDescription, userData.getOrElse(sft.getUserData))
335     }
336   }

415   /**
416     * Compare two feature types. This method compares the the schemas of each feature type, i.e. the
417     * names and type bindings of the attributes. The result will be:
418     *
419     * <ul>
420     *   <li>a positive int if type A is a sub type or reorder of type B</li>
421     *   <li>zero if type A has the same schema as type B</li>
422     *   <li>a negative int if none of the above</li>
423     * </ul>
424     *
425     * Note that this shouldn't be used for sorting.
426     *
427     * Note: GeoTools DataUtilities has a similar method that does not work correctly
428     *
429     * @param a schema A
430     * @param b schema B
431     * @return comparison
432     */
433   def compare(a: SimpleFeatureType, b: SimpleFeatureType): Int = {
434     if (a.getAttributeCount > b.getAttributeCount) {
435       return -1 // not equals, subtype or reorder
436     }
437 
438     var exact = a.getAttributeCount == b.getAttributeCount
439 
440     var i = 0
441     while (i < a.getAttributeCount) {
442       val ad = a.getDescriptor(i)
443       val bd = b.getDescriptor(i)
444       if (ad.getLocalName == bd.getLocalName) {
445         if (ad.getType.getBinding != bd.getType.getBinding) {
446           if (bd.getType.getBinding.isAssignableFrom(ad.getType.getBinding)) {
447             exact = false // we've seen a subtyping
448           } else {
449             return -1 // not a subtype/reorder
450           }
451         }
452       } else {
453         val reorder = b.getDescriptor(ad.getLocalName)
454         if (reorder == null || !reorder.getType.getBinding.isAssignableFrom(ad.getType.getBinding)) {
455           return -1 // not a subtype/reorder
456         } else {
457           exact = false // we've seen a reordering
458         }
459       }
460       i += 1
461     }
462 
463     if (exact) { 0 } else {
464       1  // we haven't hit an absolute inequality, but we've found a reordering or subtype
465     }
466   }

468   private def createFeatureType(namespace: String,
469                                 name: String,
470                                 spec: SimpleFeatureSpec,
471                                 factory: Option[FeatureTypeFactory] = None): SimpleFeatureType = {
472     import AttributeOptions.OptDefault
473 
474     val defaultGeom = {
475       val geomAttributes = spec.attributes.collect { case g: GeomAttributeSpec => g }
476       geomAttributes.find(_.options.get(OptDefault).exists(_.toBoolean))
477           .orElse(geomAttributes.headOption)
478           .map(_.name)
479     }
480     val defaultDate = if (spec.options.get(IndexIgnoreDtg).exists(toBoolean)) { None } else {
481       val dateAttributes = spec.attributes.filter(a => classOf[Date].isAssignableFrom(a.clazz))
482       spec.options.get(DefaultDtgField).flatMap(dtg => dateAttributes.find(_.name == dtg))
483           .orElse(dateAttributes.find(_.options.get(OptDefault).exists(_.toBoolean)))
484           .orElse(dateAttributes.headOption)
485           .map(_.name)
486     }
487 
488     val b = factory.map(new SimpleFeatureTypeBuilder(_)).getOrElse(new SimpleFeatureTypeBuilder())
489     b.setNamespaceURI(namespace)
490     b.setName(name)
491     b.addAll(spec.attributes.map(_.toDescriptor).asJava)
492     defaultGeom.foreach(b.setDefaultGeometry)
493 
494     val sft = b.buildFeatureType()
495     sft.getUserData.putAll(spec.options.asJava)
496     defaultDate.foreach(sft.setDtgField)
497     sft
498   }

501     val nsIndex = name.lastIndexOf(':')

505       (name.substring(0, nsIndex), name.substring(nsIndex + 1))
