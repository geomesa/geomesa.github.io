328   /**
329     * Creates an immutable copy of the simple feature type
330     *
331     * Note that immutable types will never be `equals` to a mutable type, due to specific class checking
332     * in the geotools implementation
333     *
334     * Note that some parts of the feature type may still be mutable - in particular AttributeType,
335     * GeometryType and SuperType are not used by geomesa so we don't bother with them. In addition,
336     * user data keys and values may be mutable objects, so while the user data map will not change,
337     * the values inside may
338     *
339     * @param sft simple feature type
340     * @param extraData additional user data to add to the simple feature type
341     * @return immutable copy of the simple feature type
342     */
343   def immutable(
344       sft: SimpleFeatureType,
345       extraData: java.util.Map[_ <: AnyRef, _ <: AnyRef] = null): SimpleFeatureType = {
346     sft match {
347       case immutable: ImmutableSimpleFeatureType if extraData == null || extraData.isEmpty => immutable
348       case _ =>
349         val schema = new java.util.ArrayList[AttributeDescriptor](sft.getAttributeCount)
350         var geom: GeometryDescriptor = null
351         var i = 0
352         while (i < sft.getAttributeCount) {
353           val descriptor = sft.getDescriptor(i)
354           val im = immutable(descriptor)
355           if (descriptor == sft.getGeometryDescriptor) {
356             geom = im.asInstanceOf[GeometryDescriptor]
357           }
358           schema.add(im)
359           i += 1
360         }
361         val userData = Option(extraData).filterNot(_.isEmpty).map { data =>
362           val map = new java.util.HashMap[AnyRef, AnyRef](data.size() + sft.getUserData.size)
363           map.putAll(sft.getUserData)
364           map.putAll(data)
365           map
366         }
367 
368         new ImmutableSimpleFeatureType(sft.getName, schema, geom, sft.isAbstract, sft.getRestrictions, sft.getSuper,
369           sft.getDescription, userData.getOrElse(sft.getUserData))
370     }
371   }

450   /**
451     * Compare two feature types. This method compares the the schemas of each feature type, i.e. the
452     * names and type bindings of the attributes. The result will be:
453     *
454     * <ul>
455     *   <li>a positive int if type A is a sub type or reorder of type B</li>
456     *   <li>zero if type A has the same schema as type B</li>
457     *   <li>a negative int if none of the above</li>
458     * </ul>
459     *
460     * Note that this shouldn't be used for sorting.
461     *
462     * Note: GeoTools DataUtilities has a similar method that does not work correctly
463     *
464     * @param a schema A
465     * @param b schema B
466     * @return comparison
467     */
468   def compare(a: SimpleFeatureType, b: SimpleFeatureType): Int = {
469     if (a.getAttributeCount > b.getAttributeCount) {
470       return -1 // not equals, subtype or reorder
471     }
472 
473     var exact = a.getAttributeCount == b.getAttributeCount
474 
475     var i = 0
476     while (i < a.getAttributeCount) {
477       val ad = a.getDescriptor(i)
478       val bd = b.getDescriptor(i)
479       if (ad.getLocalName == bd.getLocalName) {
480         if (ad.getType.getBinding != bd.getType.getBinding) {
481           if (bd.getType.getBinding.isAssignableFrom(ad.getType.getBinding)) {
482             exact = false // we've seen a subtyping
483           } else {
484             return -1 // not a subtype/reorder
485           }
486         }
487       } else {
488         val reorder = b.getDescriptor(ad.getLocalName)
489         if (reorder == null || !reorder.getType.getBinding.isAssignableFrom(ad.getType.getBinding)) {
490           return -1 // not a subtype/reorder
491         } else {
492           exact = false // we've seen a reordering
493         }
494       }
495       i += 1
496     }
497 
498     if (exact) { 0 } else {
499       1  // we haven't hit an absolute inequality, but we've found a reordering or subtype
500     }
501   }

503   /**
504    * Compares that index tables are the same format (key and value) between the two feature types
505    *
506    * @param a first type
507    * @param b second type
508    * @return 0 if indices are the same, non-zero otherwise
509    */
510   def compareIndexConfigs(a: SimpleFeatureType, b: SimpleFeatureType): Int = {
511     def compareAttributes(i: Int): Boolean = {
512       val (ad, bd) = (a.getDescriptor(i), b.getDescriptor(i))
513       ad.getType.getBinding == bd.getType.getBinding && AttributeChecks.forall(c => c(ad) == c(bd))
514     }
515     if (a.getAttributeCount == b.getAttributeCount &&
516         IndexChecks.forall(c => c(a) == c(b)) &&
517         Range(0, a.getAttributeCount).forall(compareAttributes)) {
518       0
519     } else {
520       1
521     }
522   }

524   private def createFeatureType(namespace: String,
525                                 name: String,
526                                 spec: SimpleFeatureSpec,
527                                 factory: Option[FeatureTypeFactory] = None): SimpleFeatureType = {
528     import AttributeOptions.OptDefault
529 
530     val defaultGeom = {
531       val geomAttributes = spec.attributes.collect { case g: GeomAttributeSpec => g }
532       geomAttributes.find(_.options.get(OptDefault).exists(_.toBoolean))
533           .orElse(geomAttributes.headOption)
534           .map(_.name)
535     }
536     val defaultDate = if (spec.options.get(IndexIgnoreDtg).exists(toBoolean)) { None } else {
537       val dateAttributes = spec.attributes.filter(a => classOf[Date].isAssignableFrom(a.clazz))
538       spec.options.get(DefaultDtgField).flatMap(dtg => dateAttributes.find(_.name == dtg))
539           .orElse(dateAttributes.find(_.options.get(OptDefault).exists(_.toBoolean)))
540           .orElse(dateAttributes.headOption)
541           .map(_.name)
542     }
543 
544     val b = factory.map(new SimpleFeatureTypeBuilder(_)).getOrElse(new SimpleFeatureTypeBuilder())
545     b.setNamespaceURI(namespace)
546     b.setName(name)
547     b.addAll(spec.attributes.map(_.toDescriptor).asJava)
548     defaultGeom.foreach(b.setDefaultGeometry)
549 
550     val sft = b.buildFeatureType()
551     sft.getUserData.putAll(spec.options.asJava)
552     defaultDate.foreach(sft.setDtgField)
553     sft
554   }
