14 /**
15  *   A bounding box is a lat-lon rectangle defined by a "lower left"
16  *    and an "upper right" point.
17  *
18  *   We do make the assumption that the lower left corner's latitude and longitude are less
19  *   than or equal to the upper right corner's.
20  *
21  *   NB: Lat-lon rectangles which cross the {+|-}180 longitude line cannot be represented.
22  */
23 case class BoundingBox(ll: Point, ur: Point) {
24   
25   import org.locationtech.geomesa.utils.geohash.BoundingBox._
26 
27   require(ll.getX <= ur.getX, s"Bounding box lower left X: ${ll.getX} > upper right X: ${ur.getX}")
28   require(ll.getY <= ur.getY, s"Bounding box lower left Y: ${ll.getY} > upper right Y: ${ur.getY}")
29 
30   lazy val envelope: Envelope = new Envelope(ll.getX, ur.getX, ll.getY, ur.getY)
31   lazy val geom: Geometry = latLonGeoFactory.toGeometry(envelope)
32 
33   lazy val ul: Point = latLonGeoFactory.createPoint(new Coordinate(ll.getX, ur.getY))
34   lazy val lr: Point = latLonGeoFactory.createPoint(new Coordinate(ur.getX, ll.getY))
35 
36   def intersects(bbox: BoundingBox): Boolean = covers(bbox.ll) || covers(bbox.ul) ||
37                                                covers(bbox.lr) || covers(bbox.ur)
38 
39   /**
40    * This bounding box covers bbox iff no points of bbox lie in the exterior of this bounding box.
41    * @param bbox
42    * @return
43    */
44   def covers(bbox: BoundingBox):Boolean = covers(bbox.ll) && covers(bbox.ul) &&
45                                           covers(bbox.lr) && covers(bbox.ur)
46 
47   /**
48    * This bounding box covers pt iff pt does not line in the exterior of this bounding box.
49    * @param pt
50    * @return
51    */
52   def covers(pt: Point): Boolean = (ll.getX <= pt.getX && pt.getX <= ur.getX) &&
53                                    (ll.getY <= pt.getY && pt.getY <= ur.getY)
54 
55   /**
56    * This bounding box contains geom iff no points of goem lie in the exterior of this bounding box,
57    * and at least one point of the interior of goem lies in the interior of this bounding box.
58    * @param otherGeom
59    * @return
60    */
61   def contains(otherGeom: Geometry): Boolean = geom.contains(otherGeom)
62 
63   lazy val minLon: Double = ll.getX
64   lazy val minLat: Double = ll.getY
65   lazy val maxLon: Double = ur.getX
66   lazy val maxLat: Double = ur.getY
67 
68   // these very simple calculations work because this class assumes min{x|y} < max{x|y}
69   // (no date line or pole wrap-arounds)
70   lazy val midLon: Double = (minLon + maxLon) / 2
71   lazy val midLat: Double = (minLat + maxLat) / 2
72 }
73 
74 object BoundingBox {
75 
76   lazy val latLonGeoFactory = new GeometryFactory(new PrecisionModel(PrecisionModel.FLOATING), 4326)
77 
78   def apply(x1: Double, x2: Double, y1: Double, y2: Double): BoundingBox =
79      apply(Bounds(Math.min(x1, x2), Math.max(x1, x2)),
80            Bounds(Math.min(y1, y2), Math.max(y1, y2)))
81 
82   def apply(lons: Bounds, lats: Bounds): BoundingBox = {
83     val Bounds(minLat, maxLat) = lats
84     val Bounds(minLon, maxLon) = lons
85     new BoundingBox(
86       latLonGeoFactory.createPoint(new Coordinate(minLon, minLat)),
87       latLonGeoFactory.createPoint(new Coordinate(maxLon, maxLat)))
88   }
89 
90   def apply(env: Envelope): BoundingBox = {
91     apply(env.getMinX, env.getMaxX, env.getMinY, env.getMaxY)
92   }
93 
94   def intersects(l: BoundingBox, r: BoundingBox): Boolean = l.geom.intersects(r.geom)
95 
96   /**
97    * get geohash that covers the bounding box to the given precision 
98    * @param bbox
99    * @param precision
100    * @return
101    */
102   def getCoveringGeoHash(bbox: BoundingBox, precision: Int): GeoHash = {
103     val ll = GeoHash(bbox.ll, precision).hash
104     val ur = GeoHash(bbox.ur.getX-1e-12, bbox.ur.getY-1e-12, precision).hash
105     GeoHash(ll.zip(ur).takeWhile(Function.tupled(_ == _)).map(_._1).mkString)
106   }
107 }
