18 /**
19  *   A bounding box is a lat-lon rectangle defined by a "lower left"
20  *    and an "upper right" point.
21  *
22  *   We do make the assumption that the lower left corner's latitude and longitude are less
23  *   than or equal to the upper right corner's.
24  *
25  *   NB: Lat-lon rectangles which cross the {+|-}180 longitude line cannot be represented.
26  */
27 case class BoundingBox(ll: Point, ur: Point) {
28   
29   import org.locationtech.geomesa.utils.geohash.BoundingBox._
30 
31   require(ll.getX <= ur.getX, s"Bounding box lower left X: ${ll.getX} > upper right X: ${ur.getX}")
32   require(ll.getY <= ur.getY, s"Bounding box lower left Y: ${ll.getY} > upper right Y: ${ur.getY}")
33 
34   lazy val envelope: Envelope = new Envelope(ll.getX, ur.getX, ll.getY, ur.getY)
35   lazy val geom: Geometry = latLonGeoFactory.toGeometry(envelope)
36 
37   lazy val ul: Point = latLonGeoFactory.createPoint(new Coordinate(ll.getX, ur.getY))
38   lazy val lr: Point = latLonGeoFactory.createPoint(new Coordinate(ur.getX, ll.getY))
39 
40   def intersects(bbox: BoundingBox): Boolean = covers(bbox.ll) || covers(bbox.ul) ||
41                                                covers(bbox.lr) || covers(bbox.ur)
42 
43   /**
44    * This bounding box covers bbox iff no points of bbox lie in the exterior of this bounding box.
45    * @param bbox
46    * @return
47    */
48   def covers(bbox: BoundingBox):Boolean = covers(bbox.ll) && covers(bbox.ul) &&
49                                           covers(bbox.lr) && covers(bbox.ur)
50 
51   /**
52    * This bounding box covers pt iff pt does not line in the exterior of this bounding box.
53    * @param pt
54    * @return
55    */
56   def covers(pt: Point): Boolean = (ll.getX <= pt.getX && pt.getX <= ur.getX) &&
57                                    (ll.getY <= pt.getY && pt.getY <= ur.getY)
58 
59   /**
60    * This bounding box contains geom iff no points of goem lie in the exterior of this bounding box,
61    * and at least one point of the interior of goem lies in the interior of this bounding box.
62    * @param otherGeom
63    * @return
64    */
65   def contains(otherGeom: Geometry): Boolean = geom.contains(otherGeom)
66 
67   lazy val longitudeSize = ur.getX - ll.getX
68   lazy val latitudeSize = ur.getY - ll.getY
69 
70   lazy val minLon = ll.getX
71   lazy val minLat = ll.getY
72   lazy val maxLon = ur.getX
73   lazy val maxLat = ur.getY
74 
75   // these very simple calculations work because this class assumes min{x|y} < max{x|y}
76   // (no date line or pole wrap-arounds)
77   lazy val midLon = (minLon + maxLon) / 2
78   lazy val midLat = (minLat + maxLat) / 2
79 
80   lazy val centerPoint = latLonGeoFactory.createPoint(new Coordinate(midLon,midLat))
81 
82   def getExpandedBoundingBox(that: BoundingBox): BoundingBox =
83     BoundingBox.getCoveringBoundingBox(this, that)
84 }

195   /**
196    * get geohash that covers the bounding box to the given precision 
197    * @param bbox
198    * @param precision
199    * @return
200    */
201   def getCoveringGeoHash(bbox: BoundingBox, precision: Int) = {
202     val ll = GeoHash(bbox.ll, precision).hash
203     val ur = GeoHash(bbox.ur.getX-1e-12, bbox.ur.getY-1e-12, precision).hash
204     GeoHash(ll.zip(ur).takeWhile(Function.tupled(_ == _)).unzip._1.mkString)
205   }
206 
207   def generateSubGeoHashes(hash: GeoHash): Seq[GeoHash] =
208     (0 to 31).map { i =>
209       val oneBits = padLongToString(i)
210                     .zipWithIndex
211                     .filter { case (ch, idx) => ch != '0' }
212                     .map { case (ch, idx) => hash.prec + idx }
213       GeoHash(hash.bitset | BitSet(oneBits: _*), hash.prec + 5)
214     }
215 
216   private def padLongToString(i: Long) = String.format("%5s", i.toBinaryString).replace(' ', '0')
217 }
