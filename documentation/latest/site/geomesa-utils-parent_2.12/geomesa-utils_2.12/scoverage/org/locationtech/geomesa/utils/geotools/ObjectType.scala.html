24   // geometry sub-types
25   val POINT, LINESTRING, POLYGON, MULTIPOINT, MULTILINESTRING, MULTIPOLYGON, GEOMETRY_COLLECTION = Value
26 
27   // string sub-types
28   val JSON: Value = Value

30   val GeometrySubtypes: Seq[Value] =

33   /**
34    * Turns a SimpleFeatureType attribute class binding into an enumeration.
35    *
36    * The first element in the result will be the primary binding. For geometries, lists and maps,
37    * the result will also contain secondary types.
38    *
39    * Lists will contain the type of the list elements.
40    * Maps will contain the type of the map keys, then the type of the map values.
41    * Geometries will contain the specific geometry type.
42    *
43    * Note: geometries will always return GEOMETRY as the primary type to allow for generic matching.
44    *
45    * @param descriptor attribute descriptor
46    * @return binding
47    */
48   def selectType(descriptor: AttributeDescriptor): Seq[ObjectType] =
49     selectType(descriptor.getType.getBinding, descriptor.getUserData, Option(descriptor.getLocalName))
50 
51   /**
52    * Used if there is no descriptor available. Note that this will throw an error for list or map-type attributes.
53    *
54    * @see selectType(descriptor: AttributeDescriptor)
55    *
56    * @param clazz class, must be valid for a SimpleFeatureType attribute
57    * @return binding
58    */
59   def selectType(clazz: Class[_]): Seq[ObjectType] = selectType(clazz, Collections.emptyMap(), None)
60 
61   /**
62    * Get the type enumeration
63    *
64    * @param clazz class, must be valid for a SimpleFeatureType attribute
65    * @param metadata attribute metadata (user data)
66    * @param fieldName name of the field
67    * @return binding
68    */
69   private def selectType(clazz: Class[_], metadata: jMap[_, _], fieldName: Option[String]): Seq[ObjectType] = {
70     clazz match {
71       case c if classOf[java.lang.String].isAssignableFrom(c) =>
72         if (metadata.get(OptJson) == "true") { Seq(STRING, JSON) } else { Seq(STRING) }
73       case c if classOf[java.lang.Integer].isAssignableFrom(c) => Seq(INT)
74       case c if classOf[java.lang.Long].isAssignableFrom(c) => Seq(LONG)
75       case c if classOf[java.lang.Float].isAssignableFrom(c) => Seq(FLOAT)
76       case c if classOf[java.lang.Double].isAssignableFrom(c) => Seq(DOUBLE)
77       case c if classOf[java.lang.Boolean].isAssignableFrom(c) => Seq(BOOLEAN)
78       case c if classOf[java.util.Date].isAssignableFrom(c) => Seq(DATE)
79       case c if classOf[UUID].isAssignableFrom(c) => Seq(UUID)
80       case c if classOf[Geometry].isAssignableFrom(c) => geometryType(c.asInstanceOf[Class[_ <: Geometry]])
81       case c if classOf[Array[Byte]].isAssignableFrom(c) => Seq(BYTES)
82       case c if classOf[jList[_]].isAssignableFrom(c) => listType(metadata, fieldName)
83       case c if classOf[jMap[_, _]].isAssignableFrom(c) => mapType(metadata, fieldName)
84 
85       case _ =>
86         throw new IllegalArgumentException(s"${fieldName.fold("Type")(f => s"Field $f of type")} $clazz can't be serialized")
87     }
88   }
89 
90   private def geometryType(clazz: Class[_ <: Geometry]): Seq[ObjectType] = {
91     val subtype = clazz match {
92       case c if c == classOf[Point]              => POINT
93       case c if c == classOf[LineString]         => LINESTRING
94       case c if c == classOf[Polygon]            => POLYGON
95       case c if c == classOf[MultiLineString]    => MULTILINESTRING
96       case c if c == classOf[MultiPolygon]       => MULTIPOLYGON
97       case c if c == classOf[MultiPoint]         => MULTIPOINT
98       case c if c == classOf[GeometryCollection] => GEOMETRY_COLLECTION
99       case _                                     => GEOMETRY
100     }
101     Seq(GEOMETRY, subtype)
102   }
103 
104   private def listType(metadata: jMap[_, _], fieldName: Option[String]): Seq[ObjectType] = {
105     val elementType = getSubType(metadata, UserDataListType, fieldName)
106     Seq(LIST, elementType)
107   }
108 
109   private def mapType(metadata: jMap[_, _], fieldName: Option[String]): Seq[ObjectType] = {
110     val keyType = getSubType(metadata, UserDataMapKeyType, fieldName)
111     val valueType = getSubType(metadata, UserDataMapValueType, fieldName)
112     Seq(MAP, keyType, valueType)
113   }
114 
115   private def getSubType(metadata: jMap[_, _], typeKey: String, fieldName: Option[String]): ObjectType = {
116     def fieldError: String = s"for collection-type field${fieldName.fold("")(f => s" '$f'")}"
117     metadata.get(typeKey) match {
118       case s: String =>
119         val clazz = Class.forName(s)
120         selectType(clazz) match {
121           case Seq(binding) => binding
122           case _ => throw new IllegalArgumentException(s"Can't serialize sub-type of ${clazz.getName} $fieldError")
123         }
124 
125       case null => throw new IllegalArgumentException(s"Missing user data key '$typeKey' $fieldError")
126       case t => throw new IllegalArgumentException(s"Unexpected user data '$typeKey' $fieldError: $t")
127     }
128   }
