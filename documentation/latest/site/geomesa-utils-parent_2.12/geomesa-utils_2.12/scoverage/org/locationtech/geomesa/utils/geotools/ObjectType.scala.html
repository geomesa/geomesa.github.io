41   /**
42    * Turns a SimpleFeatureType attribute class binding into an enumeration.
43    *
44    * The first element in the result will be the primary binding. For geometries, lists and maps,
45    * the result will also contain secondary types.
46    *
47    * Lists will contain the type of the list elements.
48    * Maps will contain the type of the map keys, then the type of the map values.
49    * Geometries will contain the specific geometry type.
50    *
51    * Note: geometries will always return GEOMETRY as the primary type to allow for generic matching.
52    *
53    * @param clazz class, must be valid for a SimpleFeatureType attribute
54    * @param metadata attribute metadata (user data)
55    * @return binding
56    */
57   def selectType(clazz: Class[_], metadata: jMap[_, _] = jCollections.emptyMap()): Seq[ObjectType] = {
58     clazz match {
59       case c if classOf[java.lang.String].isAssignableFrom(c) =>
60         if (metadata.get(OptJson) == "true") { Seq(STRING, JSON) } else { Seq(STRING) }
61       case c if classOf[java.lang.Integer].isAssignableFrom(c) => Seq(INT)
62       case c if classOf[java.lang.Long].isAssignableFrom(c) => Seq(LONG)
63       case c if classOf[java.lang.Float].isAssignableFrom(c) => Seq(FLOAT)
64       case c if classOf[java.lang.Double].isAssignableFrom(c) => Seq(DOUBLE)
65       case c if classOf[java.lang.Boolean].isAssignableFrom(c) => Seq(BOOLEAN)
66       case c if classOf[java.util.Date].isAssignableFrom(c) => Seq(DATE)
67       case c if classOf[UUID].isAssignableFrom(c) => Seq(UUID)
68       case c if classOf[Geometry].isAssignableFrom(c) => geometryType(c.asInstanceOf[Class[_ <: Geometry]])
69       case c if classOf[Array[Byte]].isAssignableFrom(c) => Seq(BYTES)
70       case c if classOf[jList[_]].isAssignableFrom(c) => listType(metadata)
71       case c if classOf[jMap[_, _]].isAssignableFrom(c) => mapType(metadata)
72 
73       case _ => throw new IllegalArgumentException(s"Class $clazz can't be serialized")
74     }
75   }
76 
77   private def geometryType(clazz: Class[_ <: Geometry]): Seq[ObjectType] = {
78     val subtype = clazz match {
79       case c if c == classOf[Point]              => POINT
80       case c if c == classOf[LineString]         => LINESTRING
81       case c if c == classOf[Polygon]            => POLYGON
82       case c if c == classOf[MultiLineString]    => MULTILINESTRING
83       case c if c == classOf[MultiPolygon]       => MULTIPOLYGON
84       case c if c == classOf[MultiPoint]         => MULTIPOINT
85       case c if c == classOf[GeometryCollection] => GEOMETRY_COLLECTION
86       case _                                     => GEOMETRY
87     }
88     Seq(GEOMETRY, subtype)
89   }
90 
91   private def listType(metadata: jMap[_, _]): Seq[ObjectType] = {
92     val clazz = Class.forName(metadata.get(UserDataListType).asInstanceOf[String])
93     selectType(clazz) match {
94       case Seq(binding) => Seq(LIST, binding)
95       case _ => throw new IllegalArgumentException(s"Can't serialize list sub-type of ${clazz.getName}")
96     }
97   }
98 
99   private def mapType(metadata: jMap[_, _]): Seq[ObjectType] = {
100     val keyClass = Class.forName(metadata.get(UserDataMapKeyType).asInstanceOf[String])
101     val keyType = selectType(keyClass) match {
102       case Seq(binding) => binding
103       case _ => throw new IllegalArgumentException(s"Can't serialize map key type of ${keyClass.getName}")
104     }
105     val valueClass = Class.forName(metadata.get(UserDataMapValueType).asInstanceOf[String])
106     val valueType = selectType(valueClass) match {
107       case Seq(binding) => binding
108       case _ => throw new IllegalArgumentException(s"Can't serialize map value type of ${valueClass.getName}")
109     }
110     Seq(MAP, keyType, valueType)
111   }
