19   private val jarFileFilter = new FilenameFilter() {
20     override def accept(dir: File, name: String): Boolean =
21       name.endsWith(".jar") && !name.endsWith("-sources.jar") && !name.endsWith("-javadoc.jar")
22   }

24   private val folderFileFilter = new FileFilter() {
25     override def accept(pathname: File): Boolean = pathname.isDirectory
26   }
27 
28   private val fileFilter = new FileFilter() {
29     override def accept(pathname: File): Boolean = pathname.isFile
30   }

32   def findJars(jars: Seq[String], searchPath: Iterator[() => Seq[File]]): Seq[File] = {
33     val foundJars = ArrayBuffer.empty[File]
34     var remaining = jars
35     // search each path in order until we've found all our jars
36     while (remaining.nonEmpty && searchPath.hasNext) {
37       val files = searchPath.next()()
38       remaining = remaining.filter { jarPrefix =>
39         val matched = files.filter(_.getName.startsWith(jarPrefix))
40         foundJars ++= matched
41         matched.isEmpty
42       }
43     }
44 
45     if (remaining.nonEmpty) {
46       logger.warn(s"Could not find requested jars: ${remaining.mkString(", ")}")
47     }
48 
49     foundJars.distinct.toSeq
50   }

71   /**
72    * Finds URLs of jar files based on the current classpath
73    *
74    * @param clas
75    * @return
76    */
77   def getJarsFromClasspath(clas: Class[_]): Seq[File] = {
78     clas.getClassLoader match {
79       case cl: URLClassLoader => cl.getURLs.map(u => new File(cleanClassPathURL(u.getFile)))
80       case cl =>
81         logger.warn(s"Can't load jars from classloader of type ${cl.getClass.getCanonicalName}")
82         Seq.empty
83     }
84   }

86   /**
87    * Recursively searches folders for jar files
88    *
89    * @param dir
90    * @return
91    */
92   def loadJarsFromFolder(dir: File): Seq[File] = {
93     val files = Option(dir.listFiles(jarFileFilter)).toSeq.flatten
94     val children = Option(dir.listFiles(folderFileFilter)).toSeq.flatten.flatMap(loadJarsFromFolder)
95     files ++ children
96   }

98   /**
99     * Finds URLs of files based on a system property
100     *
101     * @param prop
102     * @return
103     */
104   def getFilesFromSystemProperty(prop: String): Seq[File] = {
105     Option(System.getProperty(prop)) match {
106       case Some(path) => path.split(":").map(new File(_)).toSeq.flatMap(loadFiles)
107       case None =>
108         logger.debug(s"No files loaded onto classpath from system property: ${prop}")
109         Seq.empty
110     }
111   }
112 
113   /**
114     * Recursively searches file for all files. Accepts file or dir.
115     *
116     * @param file
117     * @return
118     */
119   private def loadFiles(file: File): Seq[File] = {
120     if (file.isDirectory) {
121       val files = Option(file.listFiles(fileFilter)).toSeq.flatten
122       val childDirs = Option(file.listFiles(folderFileFilter)).toSeq.flatten.flatMap(loadFiles)
123       files ++ childDirs
124     } else {
125       Option(file).toSeq
126     }
127   }
128 
129   private def cleanClassPathURL(url: String): String =
130     URLDecoder.decode(url, "UTF-8").replace("file:", "").replace("!", "")
131 
132   /**
133     * <p>Load files (jars, resources, configuration, etc) from a classpath defined by an environmental
134     * variable following these rules:
135     * <ul>
136     *   <li>Entries are colon (:) separated</li>
137     *   <li>If the entry ends with "&#47;*", treat it as a directory, and list jars in that
138     *   directory...no recursion</li>
139     *   <li>If the entry is a file then add it</li>
140     *   <li>If the entry is a directory list all files (jars and files) in the directory</li>
141     * </ul>
142     * </p>
143     *
144     * @param prop - environmental variable
145     * @return a list of files found in the classpath
146     */
147   def loadClassPathFromEnv(prop: String): Seq[File] = {
148     val files = sys.env.get(prop).toSeq.flatMap(_.split(':').toSeq).flatMap { entry =>
149       if (entry.endsWith("/*")) {
150         new File(entry.dropRight(2)).listFiles(jarFileFilter)
151       } else {
152         val f = new File(entry)
153         if (f.isDirectory) {
154           Option(f.listFiles).toSeq.flatten
155         } else {
156           Seq(f)
157         }
158       }
159     }
160     logger.debug(s"Loaded env classpath '$prop': ${files.map(_.getAbsolutePath).mkString(":")}")
161     files
162   }
