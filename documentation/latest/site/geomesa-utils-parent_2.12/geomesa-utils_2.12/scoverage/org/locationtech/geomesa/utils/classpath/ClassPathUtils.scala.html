19   private val jarFileFilter = new FilenameFilter() {
20     override def accept(dir: File, name: String) =
21       name.endsWith(".jar") && !name.endsWith("-sources.jar") && !name.endsWith("-javadoc.jar")
22   }

24   private val folderFileFilter = new FileFilter() {
25     override def accept(pathname: File) = pathname.isDirectory
26   }

28   private val fileFilter = new FileFilter() {
29     override def accept(pathname: File) = pathname.isFile()
30   }

32   def findJars(jars: Seq[String], searchPath: Iterator[() => Seq[File]]): Seq[File] = {
33     val foundJars = ArrayBuffer.empty[File]
34     var remaining = jars
35     // search each path in order until we've found all our jars
36     while (remaining.nonEmpty && searchPath.hasNext) {
37       val files = searchPath.next()()
38       remaining = remaining.filter { jarPrefix =>
39         val matched = files.filter(_.getName.startsWith(jarPrefix))
40         foundJars ++= matched
41         matched.isEmpty
42       }
43     }
44 
45     if (remaining.nonEmpty) {
46       logger.warn(s"Could not find requested jars: ${remaining.mkString(", ")}")
47     }
48 
49     foundJars.distinct.toSeq
50   }

71   /**
72    * Finds URLs of jar files based on the current classpath
73    *
74    * @param clas
75    * @return

77   def getJarsFromClasspath(clas: Class[_]): Seq[File] = {
78     clas.getClassLoader match {
79       case cl: URLClassLoader => cl.getURLs.map(u => new File(cleanClassPathURL(u.getFile)))
80       case cl =>
81         logger.warn(s"Can't load jars from classloader of type ${cl.getClass.getCanonicalName}")
82         Seq.empty
83     }
84   }

98   def loadJarsFromFolder(dir: File): Seq[File] = {
99     val files = Option(dir.listFiles(jarFileFilter)).toSeq.flatten

110   def getFilesFromSystemProperty(prop: String): Seq[File] = {
111     Option(System.getProperty(prop)) match {
112       case Some(path) => path.toString().split(":").map(new File(_)).toSeq.flatMap(loadFiles)
113       case None =>
114         logger.debug(s"No files loaded onto classpath from system property: ${prop}")
115         Seq.empty
116     }
117   }

119   /**
120     * Recursively searches file for all files. Accepts file or dir.
121     *
122     * @param file
123     * @return
124     */
125   def loadFiles(file: File): Seq[File] = {
126     if (file.isDirectory) {
127       val files = Option(file.listFiles(fileFilter)).toSeq.flatten
128       val childDirs = Option(file.listFiles(folderFileFilter)).toSeq.flatten.flatMap(loadFiles)
129       files ++ childDirs
130     } else {
131       Option(file).toSeq
132     }
133   }

135   def cleanClassPathURL(url: String): String =
136     URLDecoder.decode(url, "UTF-8").replace("file:", "").replace("!", "")
137 
138   /**
139     * <p>Load files (jars, resources, configuration, etc) from a classpath defined by an environmental
140     * variable following these rules:
141     * <ul>
142     *   <li>Entries are colon (:) separated</li>
143     *   <li>If the entry ends with "&#47;*", treat it as a directory, and list jars in that
144     *   directory...no recursion</li>
145     *   <li>If the entry is a file then add it</li>
146     *   <li>If the entry is a directory list all files (jars and files) in the directory</li>
147     * </ul>
148     * </p>
149     *
150     * @param prop - environmental variable
151     * @return a list of files found in the classpath
152     */
153   def loadClassPathFromEnv(prop: String): Seq[File] = {
154     val files = sys.env.get(prop).toSeq.flatMap(_.split(':').toSeq).flatMap { entry =>
155       if (entry.endsWith("/*")) {
156         new File(entry.dropRight(2)).listFiles(jarFileFilter)
157       } else {
158         val f = new File(entry)
159         if (f.isDirectory) {
160           Option(f.listFiles).toSeq.flatten
161         } else {
162           Seq(f)
163         }
164       }
165     }
166     logger.debug(s"Loaded env classpath '$prop': ${files.map(_.getAbsolutePath).mkString(":")}")
167     files
168   }
