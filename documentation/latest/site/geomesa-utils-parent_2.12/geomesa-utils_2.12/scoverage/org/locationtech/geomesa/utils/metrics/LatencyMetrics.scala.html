19 /**
20  * Latency metrics tracker
21  *
22  * @param dtgIndex index of the date attribute to track in the feature type
23  * @param prefix prefix for metric names
24  * @param tags metrics tags
25  * @param maxExpectedLatency max expected latency for histogramming
26  */
27 class LatencyMetrics(
28     dtgIndex: Int,
29     prefix: String,
30     tags: java.lang.Iterable[Tag] = java.util.List.of(),
31     maxExpectedLatency: FiniteDuration = Duration(1, TimeUnit.DAYS)) {
32 
33   private val date = GaugeUtils.timeGauge(s"$prefix.dtg.latest", tags)
34 
35   private val latency =
36       DistributionSummary.builder(s"$prefix.dtg.latency")
37         .tags(tags)
38         .publishPercentileHistogram()
39         .baseUnit("milliseconds")
40         .minimumExpectedValue(1d)
41         .maximumExpectedValue(maxExpectedLatency.toMillis.toDouble)
42         .register(Metrics.globalRegistry)
43 
44   /**
45    * Record latency for a feature
46    *
47    * @param feature feature
48    */
49   def apply(feature: SimpleFeature): Unit = {
50     val dtg = feature.getAttribute(dtgIndex).asInstanceOf[Date]
51     if (dtg != null) {
52       date.set(dtg.getTime)
53       latency.record(System.currentTimeMillis() - dtg.getTime)
54     }
55   }
56 }
