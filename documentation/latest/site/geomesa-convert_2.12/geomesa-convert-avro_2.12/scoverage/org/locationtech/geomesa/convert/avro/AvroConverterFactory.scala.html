55   override def infer(
56       is: InputStream,
57       sft: Option[SimpleFeatureType],
58       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = {
59     Try {
60       WithClose(new DataFileStream[GenericRecord](is, new GenericDatumReader[GenericRecord]())) { dfs =>
61         val (schema, id, fields, userData) = if (AvroDataFile.canParse(dfs)) {
62           // this is a file written in the geomesa avro format
63           val records = dfs.iterator.asScala.take(AbstractConverterFactory.inferSampleSize)
64           // get the version from the first record
65           val version = records.flatMap(r => Option(r.get(VersionField.name))).collectFirst { case i: Integer => i.intValue() }
66           val nameEncoder = new FieldNameEncoder(version.getOrElse(SerializationVersions.DefaultVersion))
67           val dataSft = AvroDataFile.getSft(dfs)
68           val native = AvroSerialization.usesNativeCollections(dfs.getSchema)
69 
70           val fields = dataSft.getAttributeDescriptors.asScala.map { descriptor =>
71             // some types need a function applied to the underlying avro value
72             val fn = ObjectType.selectType(descriptor).head match {
73               case ObjectType.DATE            => Some("millisToDate")
74               case ObjectType.UUID            => Some("avroBinaryUuid")
75               case ObjectType.GEOMETRY        => Some("geometry") // note: handles both wkt (v1) and wkb (v2)
76               case ObjectType.LIST if !native => Some("avroBinaryList")
77               case ObjectType.MAP if !native  => Some("avroBinaryMap")
78               case _ => None
79             }
80 
81             val path = s"avroPath($$1, '/${nameEncoder.encode(descriptor.getLocalName)}')"
82             val expression = fn.map(f => s"$f($path)").getOrElse(path)
83             BasicField(descriptor.getLocalName, Some(Expression(expression)))
84           }
85 
86           val id = Expression(s"avroPath($$1, '/${FidField.name}')")
87           val userData: Map[String, Expression] =
88             if (dfs.getSchema.getField(UserDataField.name) == null) { Map.empty } else {
89               // avro user data is stored as an array of 'key', 'keyClass', 'value', and 'valueClass'
90               // our converters require global key->expression, so pull out the unique keys
91               val kvs = scala.collection.mutable.Map.empty[String, Expression]
92               records.foreach { record =>
93                 val ud = record.get(UserDataField.name).asInstanceOf[java.util.Collection[GenericRecord]]
94                 ud.asScala.foreach { rec =>
95                   if (rec.getSchema.getField("key") != null) {
96                     Option(rec.get("key")).map(_.toString).foreach { key =>
97                       kvs.getOrElseUpdate(key, {
98                         var expression = s"avroPath($$1, '/${UserDataField.name}[$$key=$key]/value')"
99                         if (rec.getSchema.getField("valueClass") != null &&
100                             Option(rec.get("valueClass")).map(_.toString).contains("java.util.Date")) {
101                           // dates have to be converted from millis
102                           expression = s"millisToDate($expression)"
103                         }
104                         Expression(expression)
105                       })
106                     }
107                   }
108                 }
109               }
110               kvs.toMap
111             }
112 
113           (dataSft, id, fields, userData)
114         } else {
115           // this is an arbitrary avro file, create fields based on the schema
116           val types = AvroConverterFactory.schemaTypes(dfs.getSchema)
117           val dataSft = TypeInference.schema("inferred-avro", types)
118           // note: avro values are always stored at index 1
119           val id = Expression("md5(string2bytes($1::string))")
120           val fields = types.map(t => BasicField(t.name, Some(Expression(t.transform.apply(1)))))
121 
122           (dataSft, id, fields, Map.empty[String, Expression])
123         }
124 
125         // validate the existing schema, if any
126         sft.foreach { existing =>
127           val inferred = schema.getAttributeDescriptors.asScala
128           val matched = existing.getAttributeDescriptors.asScala.count { d =>
129             inferred.exists { i =>
130               i.getLocalName == d.getLocalName && d.getType.getBinding.isAssignableFrom(i.getType.getBinding)
131             }
132           }
133           if (matched == 0) {
134             throw new IllegalArgumentException("Inferred schema does not match existing schema")
135           } else if (matched < existing.getAttributeCount) {
136             logger.warn(s"Inferred schema only matched $matched out of ${existing.getAttributeCount} attributes")
137           }
138         }
139 
140         val converterConfig = AvroConfig(typeToProcess, None, SchemaEmbedded, Some(id), Map.empty, userData)
141 
142         val config = configConvert.to(converterConfig)
143             .withFallback(fieldConvert.to(fields.toSeq))
144             .withFallback(optsConvert.to(BasicOptions.default))
145             .toConfig
146 
147         (schema, config)
148       }
149     }
150   }

157   /**
158     * Take an avro schema and return the simple feature type bindings for it
159     *
160     * @param schema avro schema
161     * @return
162     */
163   def schemaTypes(schema: Schema): Seq[InferredType] = {
164     val namer = new Namer()
165     val types = scala.collection.mutable.ArrayBuffer.empty[InferredType]
166 
167     def mapField(field: Schema.Field, path: String = ""): Unit = {
168       // get a valid attribute name
169       val name = namer(field.name())
170 
171       // checks for nested array/map types we can handle
172       def isSimpleArray: Boolean = isSimple(field.schema().getElementType)
173       def isSimpleMap: Boolean = isSimple(field.schema().getValueType)
174 
175       val transform = FunctionTransform("avroPath(", s",'$path/${field.name}')")
176       field.schema().getType match {
177         case Schema.Type.STRING  => types += InferredType(name, ObjectType.STRING, transform)
178         case Schema.Type.BYTES   => types += InferredType(name, ObjectType.BYTES, transform)
179         case Schema.Type.INT     => types += InferredType(name, ObjectType.INT, transform)
180         case Schema.Type.LONG    => types += InferredType(name, ObjectType.LONG, transform)
181         case Schema.Type.FLOAT   => types += InferredType(name, ObjectType.FLOAT, transform)
182         case Schema.Type.DOUBLE  => types += InferredType(name, ObjectType.DOUBLE, transform)
183         case Schema.Type.BOOLEAN => types += InferredType(name, ObjectType.BOOLEAN, transform)
184         case Schema.Type.ARRAY   => if (isSimpleArray) { types += InferredType(name, ObjectType.LIST, transform) }
185         case Schema.Type.MAP     => if (isSimpleMap) { types += InferredType(name, ObjectType.MAP, transform) }
186         case Schema.Type.FIXED   => types += InferredType(name, ObjectType.BYTES, transform)
187         case Schema.Type.ENUM    => types += InferredType(name, ObjectType.STRING, transform.copy(suffix = transform.suffix + "::string"))
188         case Schema.Type.UNION   => types += InferredType(name, ObjectType.STRING, transform.copy(suffix = transform.suffix + "::string"))
189         case Schema.Type.RECORD  => field.schema().getFields.asScala.foreach(mapField(_, s"$path/${field.name}"))
190         case _ => // no-op
191       }
192     }
193 
194     schema.getFields.asScala.foreach(mapField(_))
195 
196     // check if we can derive a geometry field
197     TypeInference.deriveGeometry(types.toSeq, namer).foreach(g => types += g)
198 
199     types.toSeq
200   }

202   private def isSimple(s: Schema): Boolean = s.getType match {
203     case Schema.Type.STRING  => true
204     case Schema.Type.INT     => true
205     case Schema.Type.LONG    => true
206     case Schema.Type.FLOAT   => true
207     case Schema.Type.DOUBLE  => true
208     case Schema.Type.BOOLEAN => true
209     case _ => false
210   }

214     override protected def decodeConfig(
215         cur: ConfigObjectCursor,
216         `type`: String,
217         idField: Option[Expression],
218         caches: Map[String, Config],
219         userData: Map[String, Expression]): Either[ConfigReaderFailures, AvroConfig] = {
220 
221       def schemaOrFile(
222           schema: Option[String],
223           schemaFile: Option[String]): Either[ConfigReaderFailures, SchemaConfig] = {
224         (schema, schemaFile) match {
225           case (Some(s), None) if s.equalsIgnoreCase(SchemaEmbedded.name) => Right(SchemaEmbedded)
226           case (Some(s), None) => Right(SchemaString(s))
227           case (None, Some(s)) => Right(SchemaFile(s))
228           case _ =>
229             val reason: FailureReason = new FailureReason {
230               override val description: String = "Exactly one of 'schema' or 'schema-file' must be defined"
231             }
232             cur.failed(reason)
233         }
234       }
235 
236       for {
237         name       <- converterName(cur).right
238         schema     <- optional(cur, "schema").right
239         schemaFile <- optional(cur, "schema-file").right
240         either     <- schemaOrFile(schema, schemaFile).right
241       } yield {
242         AvroConfig(`type`, name, either, idField, caches, userData)
243       }
244     }
245 
246     override protected def encodeConfig(config: AvroConfig, base: java.util.Map[String, AnyRef]): Unit = {
247       config.schema match {
248         case SchemaEmbedded  => base.put("schema", SchemaEmbedded.name)
249         case SchemaString(s) => base.put("schema", s)
250         case SchemaFile(s)   => base.put("schema-file", s)
251       }
252     }
