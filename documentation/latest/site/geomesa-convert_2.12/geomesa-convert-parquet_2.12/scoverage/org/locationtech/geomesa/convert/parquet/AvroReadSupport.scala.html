76     * @param fields field names
77     */
78   private class AvroRecord(fields: IndexedSeq[String]) extends GenericRecord {
79 
80     private val values = Array.ofDim[AnyRef](fields.length)
81 
82     override def put(key: String, v: AnyRef): Unit = values(fields.indexOf(key)) = v
83     override def get(key: String): AnyRef = values(fields.indexOf(key))
84     override def put(i: Int, v: AnyRef): Unit = values(i) = v
85     override def get(i: Int): AnyRef = values(i)
86 
87     override def getSchema: Schema = null
88 
89     override def toString: String = {
90       val builder = new StringBuilder("AvroRecord[")
91       var i = 0
92       while (i < fields.length) {
93         if (i != 0) {
94           builder.append(',')
95         }
96         builder.append(fields(i)).append(':').append(values(i))
97         i += 1
98       }
99       builder.append(']')
100       builder.toString
101     }
102   }

104   /**
105    * Group converter for a record
106    *
107    * @param fields fields
108    * @param schema geomesa schema encodings, if it's a geomesa file
109    */
110   private class GenericGroupConverter(fields: Seq[Type], schema: Option[SimpleFeatureParquetSchema])
111       extends GroupConverter with ValueMaterializer[GenericRecord] {
112 
113     private val names = fields.map(_.getName).toIndexedSeq
114     private val converters = schema match {
115       case None => Array.tabulate(fields.length)(i => converter(fields(i)))
116       case Some(s) =>
117         // for attributes, we re-use our parquet read support so that they get parsed into standard simple feature attribute
118         // types (including geometries) instead of generic avro types
119         val attributes =
120           s.sft.getAttributeDescriptors.asScala.map(d => SimpleFeatureReadSupport.attribute(ObjectType.selectType(d), s.encodings))
121         // the remaining non-attribute fields are added as generic types (currently fid, vis, and bboxes)
122         val remaining = fields.drop(s.sft.getAttributeCount).map(converter).toArray
123         attributes.toArray ++ remaining
124     }
125 
126     private var rec: AvroRecord = _
127 
128     override def getConverter(fieldIndex: Int): Converter = converters(fieldIndex)
129     override def start(): Unit = {
130       rec = new AvroRecord(names)
131       converters.foreach(_.reset())
132     }
133     override def end(): Unit = {
134       var i = 0
135       while (i < fields.length) {
136         rec.put(i, converters(i).materialize())
137         i += 1
138       }
139     }
140     override def reset(): Unit = rec = null
141     override def materialize(): GenericRecord = rec
142   }

144   /**
145     * Get a converter for a field type
146     *
147     * @param field field
148     * @return
149     */
150   private def converter(field: Type): ValueMaterializer[_ <: AnyRef] = {
151     val logical = field.getLogicalTypeAnnotation
152     if (field.isPrimitive) {
153       lazy val logicalConverter = logical match {
154         case _: StringLogicalTypeAnnotation => Some(new StringConverter())
155         case _: DateLogicalTypeAnnotation => Some(new DaysConverter())
156         case t: TimestampLogicalTypeAnnotation if t.getUnit == LogicalTypeAnnotation.TimeUnit.MILLIS => Some(new DateMillisConverter())
157         case t: TimestampLogicalTypeAnnotation if t.getUnit == LogicalTypeAnnotation.TimeUnit.MICROS => Some(new DateMicrosConverter())
158         case t: TimestampLogicalTypeAnnotation if t.getUnit == LogicalTypeAnnotation.TimeUnit.NANOS => Some(new NanosConverter())
159         case _: UUIDLogicalTypeAnnotation => Some(new UuidConverter())
160         case _ => None
161       }
162       val convert = field.asPrimitiveType().getPrimitiveTypeName match {
163         case PrimitiveTypeName.BINARY               => logicalConverter.getOrElse(new GeneralPrimitiveConverter())
164         case PrimitiveTypeName.INT32                => logicalConverter.getOrElse(new GeneralPrimitiveConverter())
165         case PrimitiveTypeName.INT64                => logicalConverter.getOrElse(new GeneralPrimitiveConverter())
166         case PrimitiveTypeName.DOUBLE               => new GeneralPrimitiveConverter()
167         case PrimitiveTypeName.FLOAT                => new GeneralPrimitiveConverter()
168         case PrimitiveTypeName.BOOLEAN              => new GeneralPrimitiveConverter()
169         case PrimitiveTypeName.FIXED_LEN_BYTE_ARRAY => logicalConverter.getOrElse(new GeneralPrimitiveConverter())
170         case _                                      => NullPrimitiveConverter
171       }
172       if (field.isRepetition(Repetition.REPEATED)) {
173         new RepeatedPrimitiveConverter(convert)
174       } else {
175         convert
176       }
177     } else {
178       val group = field.asGroupType()
179       logical match {
180         case _: ListLogicalTypeAnnotation =>
181           require(group.getFieldCount == 1 && !group.getType(0).isPrimitive, s"Invalid list type: $group")
182           val list = group.getType(0).asGroupType()
183           require(list.getFieldCount == 1 && list.isRepetition(Repetition.REPEATED), s"Invalid list type: $group")
184           val elements = list.getType(0)
185           require(!elements.isPrimitive, s"Invalid list type: $group")
186           val elementGroup = elements.asGroupType()
187           require(elementGroup.getFieldCount == 1 && elementGroup.getType(0).isPrimitive, s"Invalid list type: $group")
188           new ListConverter(converter(list.getType(0)))
189 
190         case _: MapLogicalTypeAnnotation =>
191           require(group.getFieldCount == 1 && !group.getType(0).isPrimitive, s"Invalid map type: $group")
192           val map = group.getType(0).asGroupType()
193           require(map.getFieldCount == 2 && map.isRepetition(Repetition.REPEATED), s"Invalid map type: $group")
194           new MapConverter(converter(map.getType(0)), converter(map.getType(1)))
195 
196         case _ =>
197           new GenericGroupConverter(group.getFields.asScala.toSeq, None)
198       }
199     }
200   }

202   /**
203    * Converter for primitive fields with repetition of 'repeated'
204    *
205    * @param delegate single value converter
206    */
207   private class RepeatedPrimitiveConverter(delegate: PrimitiveConverter with ValueMaterializer[_ <: AnyRef])
208     extends PrimitiveConverter with ValueMaterializer[java.util.List[AnyRef]] {
209 
210     private var list: java.util.List[AnyRef] = _
211 
212     override def addBinary(value: Binary): Unit = { delegate.addBinary(value); addElement() }
213     override def addBoolean(value: Boolean): Unit = { delegate.addBoolean(value); addElement() }
214     override def addInt(value: Int): Unit = { delegate.addInt(value); addElement() }
215     override def addFloat(value: Float): Unit = { delegate.addFloat(value); addElement() }
216     override def addLong(value: Long): Unit = { delegate.addLong(value); addElement() }
217     override def addDouble(value: Double): Unit = { delegate.addDouble(value); addElement() }
218 
219     private def addElement(): Unit = {
220       if (list == null) {
221         list = new java.util.ArrayList[AnyRef]()
222       }
223       list.add(delegate.materialize())
224       delegate.reset()
225     }
226     override def reset(): Unit = list = null
227     override def materialize(): java.util.List[AnyRef] = list
228   }
