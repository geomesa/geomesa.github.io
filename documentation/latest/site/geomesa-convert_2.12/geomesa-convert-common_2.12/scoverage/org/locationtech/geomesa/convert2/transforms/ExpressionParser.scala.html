24   @throws(classOf[ParsingException])
25   def parse(expression: String, report: Boolean = true): Expression = {
26     logger.trace(s"Parsing expression: $expression")
27     if (expression == null) {
28       throw new IllegalArgumentException("Invalid expression: null")
29     }
30     val runner = if (report) { ReportingParseRunner(Parser.expression) } else { BasicParseRunner(Parser.expression) }
31     val parsing = runner.run(expression)
32     parsing.result.getOrElse {
33       throw new ParsingException(s"Invalid expression: ${ErrorUtils.printParseErrors(parsing)}")
34     }
35   }
