28 class CompositeConverter(val targetSft: SimpleFeatureType, delegates: Seq[(Predicate, SimpleFeatureConverter)])
29     extends SimpleFeatureConverter {

31   private val tags = TagUtils.typeNameTag(targetSft.getTypeName)
32 
33   override def createEvaluationContext(globalParams: Map[String, Any]): EvaluationContext =
34     createEvaluationContext(delegates.map(_._2.createEvaluationContext(globalParams)), Stats(tags))

36   override def createEvaluationContext(globalParams: Map[String, Any], success: Counter, failure: Counter): EvaluationContext =
37     createEvaluationContext(delegates.map(_._2.createEvaluationContext(globalParams, success, failure)), Stats.wrap(success, failure, tags))
38 
39   private def createEvaluationContext(contexts: Seq[EvaluationContext], theseStats: Stats): EvaluationContext = {
40     val stats = new Stats() {
41       override def success(i: Int): Long = theseStats.success(i) + contexts.map(_.stats.success(0)).sum
42       override def failure(i: Int): Long = theseStats.failure(i) + contexts.map(_.stats.failure(0)).sum
43     }
44     CompositeEvaluationContext(contexts, stats)
45   }
46 
47   override def process(is: InputStream, ec: EvaluationContext): CloseableIterator[SimpleFeature] = {
48     val contexts = ec match {
49       case cec: CompositeEvaluationContext => cec.contexts.iterator
50       case _ => Iterator.continually(ec)
51     }
52     val predicates = delegates.map { case (predicate, converter) =>
53       if (!contexts.hasNext) {
54         throw new IllegalArgumentException(s"Invalid evaluation context doesn't match this converter: $ec")
55       }
56       val context = contexts.next()
57       PredicateContext(predicate.withContext(context), converter, context)
58     }
59 
60     val toEval = Array.ofDim[AnyRef](1)
61 
62     val lines = IOUtils.lineIterator(is, StandardCharsets.UTF_8)
63 
64     new CloseableIterator[SimpleFeature] {
65 
66       private var delegate: CloseableIterator[SimpleFeature] = CloseableIterator.empty
67 
68       @tailrec
69       override def hasNext: Boolean = delegate.hasNext || {
70         if (!lines.hasNext) {
71           false
72         } else {
73           toEval(0) = lines.next()
74           ec.line += 1
75           delegate.close()
76           delegate = null
77           lazy val in = new ByteArrayInputStream(toEval(0).asInstanceOf[String].getBytes(StandardCharsets.UTF_8))
78           predicates.foreach { p =>
79             if (delegate == null && Try(p.predicate(toEval)).getOrElse(false)) {
80               delegate = p.converter.process(in, p.context)
81             } else {
82               p.context.line += 1
83             }
84           }
85           if (delegate == null) {
86             ec.stats.failure()
87             delegate = CloseableIterator.empty
88           }
89           hasNext
90         }
91       }
92 
93       override def next(): SimpleFeature = if (hasNext) { delegate.next } else { Iterator.empty.next }
94 
95       override def close(): Unit = {
96         CloseWithLogging(delegate)
97         is.close()
98       }
99     }
100   }
101 
102   override def close(): Unit = CloseWithLogging(delegates.map(_._2))
