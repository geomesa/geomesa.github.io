20 trait SimpleFeatureValidator extends Closeable {
21 
22   /**
23     * Validate a feature
24     *
25     * @param sf simple feature
26     * @return validation error message, or null if valid
27     */
28   def validate(sf: SimpleFeature): String
29 }

31 object SimpleFeatureValidator extends LazyLogging {
32 
33   private lazy val factories = ServiceLoader.load[SimpleFeatureValidatorFactory]()
34 
35   val DefaultValidators: SystemProperty = SystemProperty("geomesa.converter.validators", IndexValidatorFactory.Name)
36 
37   /**
38     * Default validator names
39     *
40     * @return
41     */
42   def default: Seq[String] = DefaultValidators.get.split(",")
43 
44   /**
45    * Create validators for the given feature type
46    *
47    * @param sft simple feature type
48    * @param names validator names and options
49    * @param tags tags used for metrics
50    * @return
51    */
52   def apply(sft: SimpleFeatureType, names: Seq[String], tags: Tags = Tags.empty()): SimpleFeatureValidator = {
53     val validators = names.filterNot(_.equalsIgnoreCase(NoneValidatorFactory.Name)).map { full =>
54       val i = full.indexOf('(')
55       val (name, options) = if (i == -1) { (full, None) } else {
56         require(full.last == ')', s"Invalid option parentheses: $full")
57         (full.substring(0, i), Some(full.substring(i + 1, full.length - 1)))
58       }
59       val factory = factories.find(_.name.equalsIgnoreCase(name)).getOrElse {
60         throw new IllegalArgumentException(s"No factory found for name '$name'. " +
61           s"Available factories: ${factories.map(_.name).mkString(", ")}")
62       }
63       factory.apply(sft, options, tags)
64     }
65 
66     if (validators.lengthCompare(2) < 0) {
67       validators.headOption.getOrElse(NoValidator)
68     } else {
69       new CompositeValidator(validators)
70     }
71   }
72 
73   /**
74     * Evaluates multiple validators
75     *
76     * @param validators validators
77     */
78   private class CompositeValidator(validators: Seq[SimpleFeatureValidator]) extends SimpleFeatureValidator {
79     override def validate(sf: SimpleFeature): String = {
80       var error: String = null
81       validators.foreach { validator =>
82         val e = validator.validate(sf)
83         if (e != null) {
84           error = if (error == null) { e } else { s"$error, $e" }
85         }
86       }
87       error
88     }
89     override def close(): Unit = CloseWithLogging(validators)
90   }
