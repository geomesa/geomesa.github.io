40   /**
41    * Infer types
42    *
43    * @param pathsAndValues paths (xpath, jsonpath, etc) to values
44    * @param sft simple feature type, or name for inferred type
45    * @param namer naming for inferred attributes
46    * @param failureRate allowed failure rate for conversion to a given type
47    * @return
48    */
49   def infer(
50       pathsAndValues: Seq[PathWithValues],
51       sft: Either[String, SimpleFeatureType],
52       namer: String => String = new Namer(),
53       failureRate: Float = 0.1f): InferredTypesWithPaths = {
54     val inferred = pathsAndValues.map { case PathWithValues(path, values) =>
55       val rawTypes = values.map(InferredType.infer)
56       // merge the types for this column from each row
57       val typed = merge(rawTypes.toSeq, failureRate)
58       val baseName = if (path.nonEmpty) { path } else { String.valueOf(typed.typed).toLowerCase(Locale.US) }
59       TypeWithPath(path, typed.copy(name = namer(baseName)))
60     }
61     val geom = deriveGeometry(inferred.map(_.inferredType), namer).map(TypeWithPath("", _))
62 
63     val typesWithPaths = inferred ++ geom
64 
65     sft match {
66       case Left(name) =>
67         val schema = TypeInference.schema(name, typesWithPaths.map(_.inferredType))
68         InferredTypesWithPaths(schema, typesWithPaths)
69 
70       case Right(sft) =>
71         // validate the existing schema
72         AbstractConverterFactory.validateInferredType(sft, typesWithPaths.map(_.inferredType.typed))
73         val renamed = typesWithPaths.zip(sft.getAttributeDescriptors.asScala.map(_.getLocalName)).map {
74           case (inferred, name) => inferred.copy(inferredType = inferred.inferredType.copy(name = name))
75         }
76         InferredTypesWithPaths(sft, renamed)
77     }
78   }

80   /**
81     * Try to derive a geometry field, if one does not already exist
82     *
83     * @param types known types
84     * @return
85     */
86   def deriveGeometry(types: Seq[InferredType], namer: String => String): Option[InferredType] = {
87     // if there is no geometry field, see if we can derive one
88     if (types.map(_.typed).exists(ObjectType.GeometrySubtypes.contains)) { None } else {
89       val nums = types.filter(t => t.typed == ObjectType.DOUBLE || t.typed == ObjectType.FLOAT)
90       if (nums.lengthCompare(2) < 0) { None } else {
91         def findBestMatch(names: Seq[String]): Option[InferredType] = {
92           // determine priority order
93           val potentials = nums.flatMap { t =>
94             val name = t.name.toLowerCase(Locale.US)
95             val i = names.indexOf(name)
96             val j = names.indexWhere(n => name.endsWith(s"_$n"))
97             if (i != -1) {
98               Some(i -> t)
99             } else if (j != -1) {
100               Some(j + names.length -> t) // add names.length to make exact matches sort first
101             } else {
102               None
103             }
104           }
105           potentials.sortBy(_._1).collectFirst { case (_, t) => t }
106         }
107         for {
108           lat <- findBestMatch(latitudeNames)
109           lon <- findBestMatch(longitudeNames)
110         } yield {
111           InferredType(namer("geom"), POINT, DerivedTransform("point", lon.name, lat.name))
112         }
113       }
114     }
115   }

138   /**
139     * Merge the values from a given column into a single type
140     *
141     * @param types types extracted from the column
142     * @param failureRate max percentage of values that can fail a conversion to a particular type,
143     *                    and still consider the column to be that type
144     * @return
145     */
146   private def merge(types: Seq[InferredType], failureRate: Float): InferredType = {
147 
148     val typeCounts = scala.collection.mutable.Map.empty[InferredType, Int].withDefaultValue(0)
149     types.foreach(t => typeCounts(t) += 1)
150 
151     if (typeCounts.size == 1) {
152       // if only one type, we're good
153       return types.head
154     }
155 
156     var (mostFrequentType, mostFrequentTypeCount) = typeCounts.maxBy(_._2)
157 
158     if (1f - mostFrequentTypeCount.toFloat / types.length < failureRate) {
159       // if the most frequent type passes the error threshold, use it
160       return mostFrequentType
161     }
162 
163     // try to combine types up into more general ones
164     val mergedTypeCounts = typeCounts.foldLeft(Map.empty[InferredType, Int]) { case (counts, (typ, count)) =>
165       val merged = scala.collection.mutable.Map.empty[InferredType, (InferredType, Int)] // from -> to
166       counts.foreach { case (mergedType, mergedCount) =>
167         merge(typ, mergedType).foreach(t => merged += mergedType -> (t, mergedCount + count))
168       }
169       if (merged.isEmpty) {
170         counts + (typ -> count)
171       } else {
172         (counts -- merged.keys) ++ merged.values
173       }
174     }
175 
176     mergedTypeCounts.maxBy(_._2) match {
177       case (t, c) => mostFrequentType = t; mostFrequentTypeCount = c
178     }
179 
180     if (1f - mostFrequentTypeCount.toFloat / types.length < failureRate) {
181       // if the most frequent merged type passes the error threshold, use it
182       return mostFrequentType
183     }
184 
185     // if nothing else, fall back to string
186     InferredType("", STRING, CastToString)
187   }

189   /**
190     * Merge two types into a single super-type, if one exists
191     *
192     * for example:
193     *
194     *   merge(float, double) == double
195     *   merge(point, linestring) == geometry
196     *
197     * @param left first type
198     * @param right second type
199     * @return
200     */
201   private def merge(left: InferredType, right: InferredType): Option[InferredType] = {
202     // string + foo => string, null + foo => foo
203     if (left == right || left.typed == STRING || right.typed == null) {
204       Some(left)
205     } else if (left.typed == null || right.typed == STRING) {
206       Some(right)
207     } else if (ObjectType.GeometrySubtypes.contains(left.typed) && ObjectType.GeometrySubtypes.contains(right.typed)) {
208       Some(InferredType("", GEOMETRY, IdentityTransform))
209     } else {
210       for {
211         (leftPriority, leftTransform) <- mergeUpNumbers.get(left.typed)
212         (rightPriority, rightTransform) <- mergeUpNumbers.get(right.typed)
213       } yield {
214         if (leftPriority <= rightPriority) {
215           left.copy(transform = leftTransform)
216         } else {
217           right.copy(transform = rightTransform)
218         }
219       }
220     }
221   }

257     // track the names we use for each column to ensure no duplicates

260     /**
261      * Get a valid attribute name based on the element name
262      *
263      * @param key json key
264      * @return
265      */
266     override def apply(key: String): String = {
267       val base = key.replaceAll("[^A-Za-z0-9]+", "_").replaceAll("^_|_$", "")
268       var candidate = base
269       var i = 0
270       while (FeatureUtils.ReservedWords.contains(candidate.toUpperCase(Locale.US)) || !uniqueNames.add(candidate)) {
271         candidate = s"${base}_$i"
272         i += 1
273       }
274       candidate
275     }
276   }

355     private val dateParsers = TransformerFunction.functions.values.collect { case f: StandardDateParser => f }.toArray

357     /**
358       * Infer a type from a value. Returned type will have an empty name
359       *
360       * @param value value
361       * @return
362       */
363     def infer(value: Any): InferredType = {
364       value match {
365         case null | ""                => InferredType("", null, IdentityTransform)
366         case _: Int | _: Integer      => InferredType("", INT, IdentityTransform)
367         case _: Long | _: jLong       => InferredType("", LONG, IdentityTransform)
368         case _: Float | _: jFloat     => InferredType("", FLOAT, IdentityTransform)
369         case _: Double | _: jDouble   => InferredType("", DOUBLE, IdentityTransform)
370         case _: Boolean | _: jBoolean => InferredType("", BOOLEAN, IdentityTransform)
371         case _: Date                  => InferredType("", DATE, IdentityTransform)
372         case _: Array[Byte]           => InferredType("", BYTES, IdentityTransform)
373         case _: java.util.UUID        => InferredType("", UUID, IdentityTransform)
374         case _: Point                 => InferredType("", POINT, IdentityTransform)
375         case _: LineString            => InferredType("", LINESTRING, IdentityTransform)
376         case _: Polygon               => InferredType("", POLYGON, IdentityTransform)
377         case _: MultiPoint            => InferredType("", MULTIPOINT, IdentityTransform)
378         case _: MultiLineString       => InferredType("", MULTILINESTRING, IdentityTransform)
379         case _: MultiPolygon          => InferredType("", MULTIPOLYGON, IdentityTransform)
380         case _: GeometryCollection    => InferredType("", GEOMETRY_COLLECTION, IdentityTransform)
381         case _: java.util.List[_]     => InferredType("", LIST, IdentityTransform)
382 
383         case s: String =>
384           val trimmed = s.trim
385           tryNumberParsing(trimmed)
386               .orElse(tryDateParsing(trimmed))
387               .orElse(tryGeometryParsing(trimmed))
388               .orElse(tryBooleanParsing(trimmed))
389               .orElse(tryUuidParsing(trimmed))
390               .getOrElse(InferredType("", STRING, IdentityTransform))
391 
392         case _ => InferredType("", STRING, CastToString)
393       }
394     }
395 
396     private def tryNumberParsing(s: String): Option[InferredType] = {
397       Try(BigDecimal(s)).toOption.collect {
398         case n if s.indexOf('.') == -1 && n.isValidInt  => InferredType("", INT, CastToInt)
399         case n if s.indexOf('.') == -1 && n.isValidLong => InferredType("", LONG, CastToLong)
400         // don't consider floats - even valid floats cause rounding changes when using as doubles in geometries
401         case n if n.isDecimalDouble                     => InferredType("", DOUBLE, CastToDouble)
402       }
403     }
404 
405     private def tryDateParsing(s: String): Option[InferredType] = {
406       var i = 0
407       while (i < dateParsers.length) {
408         val p = dateParsers(i)
409         if (Try(DateParsing.parseDate(s, p.format)).isSuccess) {
410           return Some(InferredType("", DATE, FunctionTransform(s"${p.names.head}(", ")")))
411         }
412         i += 1
413       }
414       None
415     }
416 
417     private def tryGeometryParsing(s: String): Option[InferredType] = {
418       Try(WKTUtils.read(s)).toOption.map {
419         case _: Point              => InferredType("", POINT, FunctionTransform("point(", ")"))
420         case _: LineString         => InferredType("", LINESTRING, FunctionTransform("linestring(", ")"))
421         case _: Polygon            => InferredType("", POLYGON, FunctionTransform("polygon(", ")"))
422         case _: MultiPoint         => InferredType("", MULTIPOINT, FunctionTransform("multipoint(", ")"))
423         case _: MultiLineString    => InferredType("", MULTILINESTRING, FunctionTransform("multilinestring(", ")"))
424         case _: MultiPolygon       => InferredType("", MULTIPOLYGON, FunctionTransform("multipolygon(", ")"))
425         case _: GeometryCollection => InferredType("", GEOMETRY_COLLECTION, FunctionTransform("geometrycollection(", ")"))
426         case _                     => InferredType("", GEOMETRY, FunctionTransform("geometry(", ")"))
427       }
428     }
429 
430     private def tryBooleanParsing(s: String): Option[InferredType] =
431       Try(s.toBoolean).toOption.map(_ => InferredType("", BOOLEAN, CastToBoolean))
432 
433     private def tryUuidParsing(s: String): Option[InferredType] =
434       Try(java.util.UUID.fromString(s)).toOption.map(_ => InferredType("", UUID, IdentityTransform))
435   }
