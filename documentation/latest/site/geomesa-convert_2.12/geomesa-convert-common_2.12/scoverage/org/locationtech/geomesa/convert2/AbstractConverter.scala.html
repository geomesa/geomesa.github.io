71     // compute only the input fields that we need to deal with to populate the simple feature
72     sft.getAttributeDescriptors.asScala.foreach { ad =>
73       fieldNameMap.get(ad.getLocalName).foreach(addDependencies(_, fieldNameMap, dag))
74     }

78       addDependencies(BasicField(IdFieldName, Some(expression)), fieldNameMap, dag)

226   /**
227     * Convert input values into a simple feature with attributes.
228     *
229     * This method returns a CloseableIterator to simplify flatMapping over inputs, but it will always
230     * return either 0 or 1 feature.
231     *
232     * @param rawValues raw input values
233     * @param ec evaluation context
234     * @return
235     */
236   private def convert(
237       rawValues: Array[AnyRef],
238       ec: EvaluationContext,
239       dtgMetrics: Option[(Int, AtomicLong, Timer)]): CloseableIterator[SimpleFeature] = {
240 
241     val result = ec.evaluate(rawValues).right.flatMap { values =>
242       val sf = new ScalaSimpleFeature(sft, "")
243 
244       attributeIndices.foreach { case (i, j) =>
245         sf.setAttributeNoConvert(i, values(j))
246       }
247       // if no id builder, empty feature id will be replaced with an auto-gen one
248       if (idIndex != -1) {
249         values(idIndex) match {
250           case id: String =>
251             sf.setId(id)
252             sf.getUserData.put(Hints.USE_PROVIDED_FID, java.lang.Boolean.TRUE)
253 
254           case null => // no-op
255 
256           case id =>
257             // the missing feature id will be caught by the validator
258             logger.warn(s"Ignoring non-string feature ID: $id")
259         }
260       }
261       userDataIndices.foreach { case (k, j) =>
262         sf.getUserData.put(k, values(j))
263       }
264       val error = validators.validate(sf)
265       if (error == null) {
266         Right(sf)
267       } else {
268         Left(EvaluationError(null, ec.line, new IOException(s"Validation error: $error")))
269       }
270     }
271 
272     result match {
273       case Right(feature) =>
274         ec.stats.success()
275         dtgMetrics.foreach { case (index, dtg, latency) =>
276           val date = feature.getAttribute(index).asInstanceOf[Date]
277           if (date != null) {
278             dtg.set(date.getTime)
279             latency.record(System.currentTimeMillis() - date.getTime, TimeUnit.MILLISECONDS)
280           }
281         }
282         CloseableIterator.single(feature)
283 
284       case Left(error) =>
285         ec.stats.failure()
286         def msg(verbose: Boolean): String = errorMessage(error, rawValues, verbose)
287         options.errorMode match {
288           case ErrorMode.LogErrors if logger.underlying.isDebugEnabled => logger.debug(msg(verbose = true), error.e)
289           case ErrorMode.LogErrors if logger.underlying.isInfoEnabled => logger.info(msg(verbose = false))
290           case ErrorMode.ReturnErrors => ec.errors.add(error.copy(e = new IOException(msg(verbose = true), error.e)))
291           case ErrorMode.RaiseErrors => throw new IOException(msg(verbose = true), error.e)
292           case _ => // no-op
293         }
294         CloseableIterator.empty
295     }
296   }

298   private def errorMessage(e: EvaluationError, input: Array[AnyRef], verbose: Boolean): String = {
299     val field = if (e.field == null) { "SimpleFeature" } else { s"field '${e.field}'" }

303       s"$terse using values:\n${input.headOption.orNull}\n[${input.drop(1).mkString(", ")}]"

387   private def addDependencies[F <: Field](field: Field, fieldMap: Map[String, F], dag: Dag): Unit = {
388     if (!dag.contains(field)) {
389       val deps = field.transforms.toSeq.flatMap(_.dependencies(Set(field), fieldMap)).toSet
390       dag.put(field, deps)
391       deps.foreach(addDependencies(_, fieldMap, dag))
392     }

395   /**
396     * Returns vertices in topological order.
397     *
398     * Note: will cause an infinite loop if there are circular dependencies
399     *
400     * @param dag graph
401     * @return ordered vertices
402     */
403   private def topologicalOrder(dag: Dag): Array[Field] = {
404     val res = ArrayBuffer.empty[Field]
405     val remaining = new scala.collection.mutable.Queue[Field]() ++ dag.keys
406     while (remaining.nonEmpty) {
407       val next = remaining.dequeue()
408       if (dag(next).forall(res.contains)) {
409         res.append(next)
410       } else {
411         remaining.enqueue(next) // put at the back of the queue
412       }
413     }
414     res.toArray
415   }
