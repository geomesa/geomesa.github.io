203     /**
204      * Tracks success and failures in the conversion process. Counters are globally shared, so
205      * we track counts locally, while updating the global counters at the same time.
206      *
207      * Note that micrometer meters (including counters) will not actually store anything unless a registry has been configured.
208      *
209      * @param success success counter
210      * @param failure failure counter
211      */
212     private case class MicrometerStats(success: Counter, failure: Counter) extends Stats {
213 
214       private val localSuccessCount = new AtomicLong(0)
215       private val localFailureCount = new AtomicLong(0)
216 
217       override def success(i: Int = 1): Long = {
218         if (i > 0) {
219           success.increment(i)
220           localSuccessCount.addAndGet(i)
221         } else {
222           localSuccessCount.get()
223         }
224       }
225 
226       override def failure(i: Int = 1): Long = {
227         if (i > 0) {
228           failure.increment(i)
229           localFailureCount.addAndGet(i)
230         } else {

232         }
233       }

275   /**
276    * Evaluation context implementation
277    *
278    * @param fields fields to evaluate, in topological dependency order
279    * @param globalValues global variable name/values
280    * @param cache enrichment caches
281    * @param stats metrics
282    * @param errors error tracker
283    */
284   class StatefulEvaluationContext(
285       fields: Array[Field],
286       globalValues: Map[String, _ <: AnyRef],
287       val cache: Map[String, EnrichmentCache],
288       val stats: Stats,
289       val errors: java.util.Queue[EvaluationError] = new java.util.ArrayDeque[EvaluationError]()
290     ) extends EvaluationContext {
291 
292     // holder for results from evaluating each row
293     private val values = Array.ofDim[AnyRef](fields.length)
294     // temp array for holding the arguments for a field
295     private val fieldArray = Array.ofDim[AnyRef](1)
296     // copy the transforms and tie them to the context
297     // note: the class isn't fully instantiated yet, but this statement is last in the initialization
298     private val transforms = fields.map(_.transforms.map(_.withContext(this)))
299 
300     override lazy val success: com.codahale.metrics.Counter = new com.codahale.metrics.Counter() {
301       override def inc(n: Long): Unit = stats.success(n.toInt)
302       override def getCount: Long = stats.success(0)
303     }
304     override lazy val failure: com.codahale.metrics.Counter = new com.codahale.metrics.Counter() {
305       override def inc(n: Long): Unit = stats.failure(n.toInt)
306       override def getCount: Long = stats.failure(0)
307     }
308 
309     override def accessor(name: String): FieldAccessor = {
310       val i = fields.indexWhere(_.name == name)
311       if (i >= 0) { new FieldValueAccessor(values, i) } else {
312         globalValues.get(name) match {
313           case Some(value) => new GlobalFieldAccessor(value)
314           case None => NullFieldAccessor
315         }
316       }
317     }
318 
319     override def evaluate(args: Array[AnyRef]): Either[EvaluationError, Array[AnyRef]] = {
320       var i = 0
321       // note: since fields are in topological order we don't need to clear them
322       while (i < values.length) {
323         values(i) = try {
324           val fieldArgs = fields(i).fieldArg match {
325             case None => args
326             case Some(f) => fieldArray(0) = f.apply(args); fieldArray
327           }
328           transforms(i) match {
329             case Some(t) => t.apply(fieldArgs)
330             case None    => fieldArgs(0)
331           }
332         } catch {
333           case NonFatal(e) => return Left(EvaluationError(fields(i).name, line, e))
334         }
335         i += 1
336       }
337       Right(values)
338     }
339 
340     override def withListener(listener: ContextListener): EvaluationContext =
341       new StatefulEvaluationContext(fields, globalValues, cache, StatListener(stats, listener))
342   }
343 
344   /**
345    * Stats implementation that adds a listener.
346    *
347    * Note that this implementation can chain additional listeners, but it's not very efficient doing so. The
348    * typical usage of listeners is reporting out counts, so usually there will only be one.
349    *
350    * @param delegate delegate stats
351    * @param listener listener
352    */
353   case class StatListener(delegate: Stats, listener: ContextListener) extends Stats {
354     override def success(i: Int): Long = { if (i > 0) { listener.onSuccess(i) }; delegate.success(i) }
355     override def failure(i: Int): Long = { if (i > 0) { listener.onFailure(i) }; delegate.failure(i) }
356   }
