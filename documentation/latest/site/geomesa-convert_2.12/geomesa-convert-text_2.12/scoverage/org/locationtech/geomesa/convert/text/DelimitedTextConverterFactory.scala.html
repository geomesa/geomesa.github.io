66   private def infer(
67       format: CSVFormat,
68       lines: Seq[String],
69       sft: Option[SimpleFeatureType]): Try[(SimpleFeatureType, Config)] = {
70     // : Seq[List[String]]
71     val rows = lines.flatMap { line =>
72       if (line.isBlank) { None } else {
73         Try(format.parse(new StringReader(line)).iterator().next.iterator.asScala.toList).toOption
74       }
75     }
76     val counts = rows.map(_.length).distinct
77     // try to verify that we actually have a delimited file
78     // ensure that some lines parsed, that there were at most 3 different col counts, and that there were at least 2 cols
79     if (counts.isEmpty || counts.lengthCompare(3) > 0 || counts.max < 2) {
80       val f = if (format == Formats.Tabs) { "tsv" } else if (format == Formats.Quoted) { "quoted csv" }  else { "csv" }
81       return Failure(new RuntimeException(s"Not a valid delimited text file using format: $f"))
82     }
83     Try {
84       val names = sft match {
85         case Some(s) =>
86           s.getAttributeDescriptors.asScala.map(_.getLocalName)
87 
88         case None =>
89           val values = rows.head.map(v => PathWithValues("", Seq(v)))
90           if (TypeInference.infer(values, Left("")).types.forall(_.inferredType.typed == ObjectType.STRING)) {
91             // assume the first row is headers
92             rows.head
93           } else {
94             Seq.empty
95           }
96       }
97       val nameIter = names.iterator
98       val pathsAndValues = Seq.tabulate(counts.max) { col =>
99         val values = rows.drop(1).map(vals => if (vals.lengthCompare(col) > 0) { vals(col) } else { null })
100         PathWithValues(if (nameIter.hasNext) { nameIter.next } else { "" }, values)
101       }
102       val inferred =
103         TypeInference.infer(pathsAndValues,
104           sft.toRight(s"inferred-${if (format == Formats.Tabs) { "tsv" } else { "csv" }}"))
105       val converterConfig = DelimitedTextConfig(typeToProcess, None, formats.find(_._2 == format).get._1,
106         Some(Expression("md5(string2bytes($0))")), Map.empty, Map.empty)
107 
108       val fields = {
109         var i = 0 // 0 is the whole record
110         inferred.types.map { case TypeWithPath(_, inferredType) =>
111           i += 1
112           BasicField(inferredType.name, Some(Expression(inferredType.transform(i))))
113         }
114       }
115 
116       val options = DelimitedTextOptions(None, CharNotSpecified, CharNotSpecified, None,
117         SimpleFeatureValidator.default, ParseMode.Default, ErrorMode(), StandardCharsets.UTF_8)
118 
119       val config = configConvert.to(converterConfig)
120           .withFallback(fieldConvert.to(fields.toSeq))
121           .withFallback(optsConvert.to(options))
122           .toConfig
123 
124       (inferred.sft, config)
125     }
126   }

128   private def magic(lines: Seq[String], sft: Option[SimpleFeatureType]): Try[(SimpleFeatureType, Config)] = {
129     if (!lines.head.startsWith("id,")) { Failure(new RuntimeException("Not a geomesa delimited export")) } else {
130       Try {
131         val spec = WithClose(Formats.QuotedMinimal.parse(new StringReader(lines.head.drop(3)))) { result =>
132           result.iterator().next.asScala.mkString(",")
133         }
134         val schema = SimpleFeatureTypes.createType("", spec)
135 
136         val converterConfig =
137           DelimitedTextConfig(typeToProcess, None, formats.find(_._2 == Formats.QuotedMinimal).get._1,
138             Some(Expression("$1")), Map.empty, Map.empty)
139 
140         var i = 1 // 0 is the whole record, 1 is the id
141         val fields = schema.getAttributeDescriptors.asScala.map { d =>
142           val bindings = ObjectType.selectType(d)
143           val transform = bindings.head match {
144             case ObjectType.STRING   => TypeInference.IdentityTransform
145             case ObjectType.INT      => TypeInference.CastToInt
146             case ObjectType.LONG     => TypeInference.CastToLong
147             case ObjectType.FLOAT    => TypeInference.CastToFloat
148             case ObjectType.DOUBLE   => TypeInference.CastToDouble
149             case ObjectType.BOOLEAN  => TypeInference.CastToBoolean
150             case ObjectType.DATE     => FunctionTransform("datetime(", ")")
151             case ObjectType.UUID     => TypeInference.IdentityTransform
152             case ObjectType.LIST     => FunctionTransform(s"parseList('${bindings(1)}',", ")")
153             case ObjectType.MAP      => FunctionTransform(s"parseMap('${bindings(1)}->${bindings(2)}',", ")")
154             case ObjectType.BYTES    => TypeInference.IdentityTransform
155             case ObjectType.GEOMETRY =>
156               bindings.drop(1).headOption.getOrElse(ObjectType.GEOMETRY) match {
157                 case ObjectType.POINT               => FunctionTransform("point(", ")")
158                 case ObjectType.LINESTRING          => FunctionTransform("linestring(", ")")
159                 case ObjectType.POLYGON             => FunctionTransform("polygon(", ")")
160                 case ObjectType.MULTIPOINT          => FunctionTransform("multipoint(", ")")
161                 case ObjectType.MULTILINESTRING     => FunctionTransform("multilinestring(", ")")
162                 case ObjectType.MULTIPOLYGON        => FunctionTransform("multipolygon(", ")")
163                 case ObjectType.GEOMETRY_COLLECTION => FunctionTransform("geometrycollection(", ")")
164                 case _                              => FunctionTransform("geometry(", ")")
165               }
166 
167             case _ => throw new IllegalStateException(s"Unexpected binding: $bindings")
168           }
169           i += 1
170           BasicField(d.getLocalName, Some(TryExpression(Expression(transform.apply(i)), LiteralNull)))
171         }
172 
173         val options = DelimitedTextOptions(Some(1), CharNotSpecified, CharNotSpecified, None,
174           SimpleFeatureValidator.default, ParseMode.Default, ErrorMode(), StandardCharsets.UTF_8)
175 
176         val config = configConvert.to(converterConfig)
177             .withFallback(fieldConvert.to(fields.toSeq))
178             .withFallback(optsConvert.to(options))
179             .toConfig
180 
181         (sft.getOrElse(schema), config)
182       }
183     }
184   }
