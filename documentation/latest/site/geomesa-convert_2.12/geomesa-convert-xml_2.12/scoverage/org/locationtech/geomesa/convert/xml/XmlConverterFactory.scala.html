47   /**
48    * Infer a configuration and simple feature type from an input stream, if possible
49    *
50    * Available hints:
51    *  - `featurePath` - xpath expression pointing to the feature element
52    *
53    * @param is input
54    * @param sft simple feature type, if known ahead of time
55    * @param hints implementation specific hints about the input
56    * @return
57    */
58   override def infer(
59       is: InputStream,
60       sft: Option[SimpleFeatureType],
61       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = {
62     val tryElements = Try {
63       val parser = new DocParser(None)
64       WithClose(XmlConverter.iterator(parser, is, StandardCharsets.UTF_8, LineMode.Multi, EvaluationContext.empty)) { iter =>
65         iter.take(AbstractConverterFactory.inferSampleSize).toList
66       }
67     }
68 
69     tryElements.flatMap { elements =>
70       val featurePath = hints.get(XmlConverterFactory.FeaturePathKey).map(_.toString)
71       val (defaultNs, namespaces) = elements.headOption match {
72         case None => (None, Map.empty[String, String])
73         case Some(e) =>
74           val ns = XmlConverterFactory.getNamespaces(e)
75           ns.get("") match {
76             case None => (None, ns)
77             case Some(default) =>
78               val alphas = Random.alphanumeric.filter(_.isLower)
79               var key: String = null
80               while ({ key = alphas.take(4).mkString(""); ns.contains(key) }) {
81                 // loop
82               }
83               (Some(s"$key:"), ns - "" + (key -> default))
84           }
85       }
86 
87       val xpathFactory =
88         ConfigSource.fromConfig(withDefaults(XmlConverterFactory.TypeConfig))
89             .loadOrThrow[XmlConfig](classTag[XmlConfig], XmlConfigConvert)
90             .xpathFactory
91 
92       val tryFeatures = Try {
93         featurePath match {
94           case None => elements
95           case Some(p) =>
96             val xpath = XmlConverter.createXPath(xpathFactory, namespaces).compile(p)
97             val features = elements.flatMap { element =>
98               val nodeList = xpath.evaluate(element, XPathConstants.NODESET).asInstanceOf[NodeList]
99               Seq.tabulate(nodeList.getLength)(i => nodeList.item(i).asInstanceOf[Element])
100             }
101             features.take(AbstractConverterFactory.inferSampleSize)
102         }
103       }
104 
105       tryFeatures.flatMap { features =>
106         if (features.isEmpty) {
107           Failure(new RuntimeException("Could not parse input as XML"))
108         } else {
109           Try {
110             // track the properties in each feature
111             // use linkedHashMap to retain insertion order
112             val props = scala.collection.mutable.LinkedHashMap.empty[String, ListBuffer[Any]]
113 
114             val namer = new XmlNamer(defaultNs)
115             features.foreach { feature =>
116               namer.addRootElement(feature)
117               XmlConverterFactory.parseNode(feature, "", defaultNs.getOrElse("")).foreach { case (k, v) =>
118                 props.getOrElseUpdate(k, ListBuffer.empty) += v
119               }
120             }
121 
122             val pathsAndValues = props.toSeq.map { case (path, values) => PathWithValues(path, values) }
123             val inferredTypes = TypeInference.infer(pathsAndValues, sft.toRight("inferred-xml"), namer)
124 
125             val fieldConfig = inferredTypes.types.map { case TypeWithPath(path, inferredType) =>
126               val transform = inferredType.transform match {
127                 case IdentityTransform => None
128                 // account for optional nodes by wrapping transform with a try/null
129                 case t => Some(Expression(s"try(${t.apply(0)},null)"))
130               }
131               if (path.isEmpty) {
132                 DerivedField(inferredType.name, transform)
133               } else {
134                 XmlPathField(inferredType.name, path, transform)
135               }
136             }
137 
138             val idField = Some(Expression("md5(stringToBytes(toString($0)))"))
139 
140             val xmlConfig =
141               XmlConfig(typeToProcess, None, xpathFactory, namespaces, None, featurePath, idField, Map.empty, Map.empty)
142 
143             val config =
144               configConvert.to(xmlConfig)
145                   .withFallback(fieldConvert.to(fieldConfig))
146                   .toConfig
147 
148             (inferredTypes.sft, config)
149           }
150         }
151       }
152     }
153   }

248   /**
249    * Attribute namer for xml elements
250    */
251   private class XmlNamer(defaultPrefix: Option[String]) extends Namer {
252 
253     private val rootPrefixes = scala.collection.mutable.HashSet.empty[String]
254     private val namespaceMatcher = Pattern.compile("(/@?)[^:/@]*:")
255 
256     /**
257      * Add a root element, whose tag will be removed from any derived attribute names
258      *
259      * @param elem element
260      */
261     def addRootElement(elem: Element): Unit = {
262       rootPrefixes += {
263         defaultPrefix match {
264           case Some(p) if elem.getPrefix == null || elem.getPrefix.isEmpty => s"/$p${elem.getTagName}/"
265           case _ => s"/${elem.getTagName}/"
266         }
267       }
268     }
269 
270     override def apply(key: String): String = {
271       val withoutRoot = rootPrefixes.find(key.startsWith) match {
272         case None    => key
273         case Some(p) => key.substring(p.length - 1)
274       }
275       // remove namespace prefixes from the attribute names
276       val withoutPrefixes = namespaceMatcher.matcher(withoutRoot).replaceAll("$1")
277       super.apply(withoutPrefixes)
278     }
279   }

281   /**
282    * Extract xmlns declarations from this element
283    *
284    * @param doc element
285    * @return
286    */
287   private def getNamespaces(doc: Element): Map[String, String] = {
288     val namespaces = Map.newBuilder[String, String]
289     val attributes = doc.getAttributes
290     var i = 0
291     while (i < attributes.getLength) {
292       val attr = attributes.item(i).asInstanceOf[Attr]
293       if (attr.getName.startsWith("xmlns:")) {
294         namespaces += attr.getName.substring(6) -> attr.getValue
295       } else if (attr.getName == "xmlns") {
296         namespaces += "" -> attr.getValue
297       }
298       i += 1
299     }
300     namespaces.result
301   }

303   /**
304    * Parse an xml node
305    *
306    * @param node node
307    * @param path xpath to the parent of the object
308    * @param defaultNamespacePrefix path prefix for elements without an explicit namespace
309    * @return map of key is xpath to value, value is a string
310    */
311   private def parseNode(node: Node, path: String, defaultNamespacePrefix: String): Seq[(String, String)] = {
312     // use linkedHashMap to preserve insertion order
313     val builder = scala.collection.mutable.LinkedHashMap.empty[String, String]
314     node match {
315       case n: Element =>
316         val elemPath = {
317           val prefix = if (n.getPrefix == null || n.getPrefix.isEmpty) { defaultNamespacePrefix } else { "" }
318           s"$path/$prefix${n.getTagName}"
319         }
320         Seq.tabulate(n.getAttributes.getLength)(n.getAttributes.item(_).asInstanceOf[Attr])
321             .filterNot(a => a.getName.startsWith("xmlns:") || a.getName == "xmlns")
322             .sortBy(_.getLocalName)
323             .foreach(attr => builder += s"$elemPath/@${attr.getName}" -> attr.getValue)
324         Seq.tabulate(n.getChildNodes.getLength)(n.getChildNodes.item)
325             .foreach(child => builder ++= parseNode(child, elemPath, defaultNamespacePrefix))
326 
327       case n: Text if !n.getData.isBlank =>
328         builder += path -> n.getData
329 
330       case _ => // no-op
331     }
332     builder.toSeq
333   }
