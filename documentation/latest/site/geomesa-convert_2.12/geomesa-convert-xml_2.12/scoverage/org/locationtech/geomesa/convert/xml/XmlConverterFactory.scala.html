48   /**
49    * Infer a configuration and simple feature type from an input stream, if possible
50    *
51    * Available hints:
52    *  - `featurePath` - xpath expression pointing to the feature element
53    *
54    * @param is input
55    * @param sft simple feature type, if known ahead of time
56    * @param hints implementation specific hints about the input
57    * @return
58    */
59   override def infer(
60       is: InputStream,
61       sft: Option[SimpleFeatureType],
62       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = {
63     val tryElements = Try {
64       val parser = new DocParser(None)
65       WithClose(XmlConverter.iterator(parser, is, StandardCharsets.UTF_8, LineMode.Multi, EvaluationContext.empty)) { iter =>
66         iter.take(AbstractConverterFactory.inferSampleSize).toList
67       }
68     }
69 
70     tryElements.flatMap { elements =>
71       val featurePath = hints.get(XmlConverterFactory.FeaturePathKey).map(_.toString)
72       val (defaultNs, namespaces) = elements.headOption match {
73         case None => (None, Map.empty[String, String])
74         case Some(e) =>
75           val ns = XmlConverterFactory.getNamespaces(e)
76           ns.get("") match {
77             case None => (None, ns)
78             case Some(default) =>
79               val alphas = Random.alphanumeric.filter(_.isLower)
80               var key: String = null
81               while ({ key = alphas.take(4).mkString(""); ns.contains(key) }) {
82                 // loop
83               }
84               (Some(s"$key:"), ns - "" + (key -> default))
85           }
86       }
87 
88       val xpathFactory =
89         ConfigSource.fromConfig(withDefaults(XmlConverterFactory.TypeConfig))
90             .loadOrThrow[XmlConfig](classTag[XmlConfig], XmlConfigConvert)
91             .xpathFactory
92 
93       val tryFeatures = Try {
94         featurePath match {
95           case None => elements
96           case Some(p) =>
97             val xpath = XmlConverter.createXPath(xpathFactory, namespaces).compile(p)
98             val features = elements.flatMap { element =>
99               val nodeList = xpath.evaluate(element, XPathConstants.NODESET).asInstanceOf[NodeList]
100               Seq.tabulate(nodeList.getLength)(i => nodeList.item(i).asInstanceOf[Element])
101             }
102             features.take(AbstractConverterFactory.inferSampleSize)
103         }
104       }
105 
106       tryFeatures.flatMap { features =>
107         if (features.isEmpty) {
108           Failure(new RuntimeException("Could not parse input as XML"))
109         } else {
110           Try {
111             // track the properties in each feature
112             // use linkedHashMap to retain insertion order
113             val props = scala.collection.mutable.LinkedHashMap.empty[String, ListBuffer[Any]]
114 
115             val namer = new XmlNamer(defaultNs)
116             features.foreach { feature =>
117               namer.addRootElement(feature)
118               XmlConverterFactory.parseNode(feature, "", defaultNs.getOrElse("")).foreach { case (k, v) =>
119                 props.getOrElseUpdate(k, ListBuffer.empty) += v
120               }
121             }
122 
123             val pathsAndValues = props.toSeq.map { case (path, values) => PathWithValues(path, values) }
124             val inferredTypes = TypeInference.infer(pathsAndValues, sft.toRight("inferred-xml"), namer)
125 
126             val fieldConfig = inferredTypes.types.map { case TypeWithPath(path, inferredType) =>
127               val transform = inferredType.transform match {
128                 case IdentityTransform => None
129                 // account for optional nodes by wrapping transform with a try/null
130                 case t => Some(Expression(s"try(${t.apply(0)},null)"))
131               }
132               if (path.isEmpty) {
133                 DerivedField(inferredType.name, transform)
134               } else {
135                 XmlPathField(inferredType.name, path, transform)
136               }
137             }
138 
139             val idField = Some(Expression("md5(stringToBytes(toString($0)))"))
140 
141             val xmlConfig =
142               XmlConfig(typeToProcess, None, xpathFactory, namespaces, None, featurePath, idField, Map.empty, Map.empty)
143 
144             val config =
145               configConvert.to(xmlConfig)
146                   .withFallback(fieldConvert.to(fieldConfig))
147                   .toConfig
148 
149             (inferredTypes.sft, config)
150           }
151         }
152       }
153     }
154   }

250   /**
251    * Attribute namer for xml elements
252    */
253   private class XmlNamer(defaultPrefix: Option[String]) extends Namer {
254 
255     private val rootPrefixes = scala.collection.mutable.HashSet.empty[String]
256     private val namespaceMatcher = Pattern.compile("(/@?)[^:/@]*:")
257 
258     /**
259      * Add a root element, whose tag will be removed from any derived attribute names
260      *
261      * @param elem element
262      */
263     def addRootElement(elem: Element): Unit = {
264       rootPrefixes += {
265         defaultPrefix match {
266           case Some(p) if elem.getPrefix == null || elem.getPrefix.isEmpty => s"/$p${elem.getTagName}/"
267           case _ => s"/${elem.getTagName}/"
268         }
269       }
270     }
271 
272     override def apply(key: String): String = {
273       val withoutRoot = rootPrefixes.find(key.startsWith) match {
274         case None    => key
275         case Some(p) => key.substring(p.length - 1)
276       }
277       // remove namespace prefixes from the attribute names
278       val withoutPrefixes = namespaceMatcher.matcher(withoutRoot).replaceAll("$1")
279       super.apply(withoutPrefixes)
280     }
281   }

283   /**
284    * Extract xmlns declarations from this element
285    *
286    * @param doc element
287    * @return
288    */
289   private def getNamespaces(doc: Element): Map[String, String] = {
290     val namespaces = Map.newBuilder[String, String]
291     val attributes = doc.getAttributes
292     var i = 0
293     while (i < attributes.getLength) {
294       val attr = attributes.item(i).asInstanceOf[Attr]
295       if (attr.getName.startsWith("xmlns:")) {
296         namespaces += attr.getName.substring(6) -> attr.getValue
297       } else if (attr.getName == "xmlns") {
298         namespaces += "" -> attr.getValue
299       }
300       i += 1
301     }
302     namespaces.result
303   }

305   /**
306    * Parse an xml node
307    *
308    * @param node node
309    * @param path xpath to the parent of the object
310    * @param defaultNamespacePrefix path prefix for elements without an explicit namespace
311    * @return map of key is xpath to value, value is a string
312    */
313   private def parseNode(node: Node, path: String, defaultNamespacePrefix: String): Seq[(String, String)] = {
314     // use linkedHashMap to preserve insertion order
315     val builder = scala.collection.mutable.LinkedHashMap.empty[String, String]
316     node match {
317       case n: Element =>
318         val elemPath = {
319           val prefix = if (n.getPrefix == null || n.getPrefix.isEmpty) { defaultNamespacePrefix } else { "" }
320           s"$path/$prefix${n.getTagName}"
321         }
322         Seq.tabulate(n.getAttributes.getLength)(n.getAttributes.item(_).asInstanceOf[Attr])
323             .filterNot(a => a.getName.startsWith("xmlns:") || a.getName == "xmlns")
324             .sortBy(_.getLocalName)
325             .foreach(attr => builder += s"$elemPath/@${attr.getName}" -> attr.getValue)
326         Seq.tabulate(n.getChildNodes.getLength)(n.getChildNodes.item)
327             .foreach(child => builder ++= parseNode(child, elemPath, defaultNamespacePrefix))
328 
329       case n: Text if !TextTools.isWhitespace(n.getData) =>
330         builder += path -> n.getData
331 
332       case _ => // no-op
333     }
334     builder.toSeq
335   }
