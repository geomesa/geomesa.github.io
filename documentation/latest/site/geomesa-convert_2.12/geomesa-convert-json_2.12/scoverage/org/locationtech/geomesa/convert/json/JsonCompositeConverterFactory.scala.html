28   override def apply(sft: SimpleFeatureType, conf: Config): Option[SimpleFeatureConverter] = {
29     if (!conf.hasPath("type") || conf.getString("type") != TypeToProcess) { None } else {
30       val defaults = AbstractConverterFactory.standardDefaults(conf, logger)
31       try {
32         implicit val convert: ConfigConvert[BasicOptions] = AbstractConverterFactory.BasicOptionsConvert
33         val options = ConfigSource.fromConfig(defaults).loadOrThrow[BasicOptions]
34         val typeToProcess = ConfigValueFactory.fromAnyRef(JsonConverterFactory.TypeToProcess)
35         val delegates = defaults.getConfigList("converters").asScala.map { base =>
36           val conf = base.withFallback(defaults).withValue("type", typeToProcess)
37           val converter = SimpleFeatureConverter(sft, conf) match {
38             case c: ParsingConverter[JsonElement] => c
39             case c => throw new IllegalArgumentException(s"Expected JsonConverter but got: $c")
40           }
41           val predicate = Predicate(conf.getString("predicate"))
42           (predicate, converter)
43         }
44         Some(new JsonCompositeConverter(sft, options.encoding, options.errorMode, delegates.toSeq))
45       } catch {
46         case NonFatal(e) => throw new IllegalArgumentException(s"Invalid configuration: ${e.getMessage}")
47       }
48     }
49   }
50 }
