19 /**
20  * Factory for observing file writes
21  */
22 trait FileSystemObserverFactory extends Closeable {
23 
24   /**
25    * Called once after instantiating the factory
26    *
27    * @param conf hadoop configuration
28    * @param root root path
29    * @param sft simple feature type
30    */
31   def init(conf: Configuration, root: Path, sft: SimpleFeatureType): Unit
32 
33   /**
34    * Create an observer for the given path
35    *
36    * @param path file path being written
37    * @return
38    */
39   def apply(path: Path): FileSystemObserver
40 }

42 object FileSystemObserverFactory {
43 
44   object NoOpObserver extends FileSystemObserver {
45     override def apply(feature: SimpleFeature): Unit = {}
46     override def flush(): Unit = {}
47     override def close(): Unit = {}
48   }
49 
50   object CompositeObserver {
51     def apply(observers: Seq[FileSystemObserver]): FileSystemObserver = {
52       val seq = observers.flatMap {
53         case NoOpObserver => Seq.empty
54         case o: CompositeObserver => o.observers
55         case o => Seq(o)
56       }
57       if (seq.isEmpty) {
58         NoOpObserver
59       } else if (seq.lengthCompare(1) == 0) {
60         seq.head
61       } else {
62         new CompositeObserver(seq)
63       }
64     }
65   }
66 
67   /**
68    * Composite observer
69    *
70    * @param observers observers
71    */
72   class CompositeObserver(val observers: Seq[FileSystemObserver]) extends FileSystemObserver {
73 
74     override def apply(feature: SimpleFeature): Unit = observers.foreach(_.apply(feature))
75 
76     override def flush(): Unit = {
77       val errors = ArrayBuffer.empty[Throwable]
78       observers.foreach { o =>
79         Try(o.flush()).failed.foreach(errors.+=)
80       }
81       if (errors.nonEmpty) {
82         val e = errors.head
83         errors.tail.foreach(e.addSuppressed)
84         throw e
85       }
86     }
87 
88     override def close(): Unit = {
89       val errors = ArrayBuffer.empty[Throwable]
90       observers.foreach { o =>
91         Try(o.close()).failed.foreach(errors.+=)
92       }
93       if (errors.nonEmpty) {
94         val e = errors.head
95         errors.tail.foreach(e.addSuppressed)
96         throw e
97       }
98     }
99   }
