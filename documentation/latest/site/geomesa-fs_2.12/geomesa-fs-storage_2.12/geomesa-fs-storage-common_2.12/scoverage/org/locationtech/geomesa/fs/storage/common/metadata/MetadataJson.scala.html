33   val MetadataPath = "metadata.json"

35   private val cache = new ConcurrentHashMap[String, NamedOptions]()

37   /**
38     * Read the metadata file at the given root path, if it exists
39     *
40     * @param context context
41     * @return
42     */
43   def readMetadata(context: FileSystemContext): Option[NamedOptions] = {
44     val key = context.root.toUri.toString
45     var cached = cache.get(key)
46     if (cached == null) {
47       // note: we may end up reading more than once, but doing it multiple times will only incur
48       // a slight performance penalty but not cause any incorrectness
49       // using an atomic operation or cache loader can cause problems, as we sometimes insert into the
50       // map during the load, which is not allowed
51       val file = new Path(context.root, MetadataPath)
52       if (PathCache.exists(context.fc, file)) {
53         val config = profile("Loaded metadata configuration") {
54           WithClose(new InputStreamReader(context.fc.open(file), StandardCharsets.UTF_8)) { in =>
55             ConfigFactory.load(ConfigFactory.parseReader(in, ParseOptions)) // call load to resolve sys props
56           }
57         }
58         if (config.hasPath("name")) {
59           cached = profile("Parsed metadata configuration") {
60             ConfigSource.fromConfig(config).loadOrThrow[NamedOptions]
61           }
62           cache.put(key, cached)
63         } else {
64           context.fc.rename(file, new Path(context.root, s"$MetadataPath.bak"))
65           PathCache.invalidate(context.fc, file)
66           transitionMetadata(context, config).foreach { meta =>
67             cached = meta // will be set in the cache in the transition code
68           }
69         }
70       }
71     }
72     Option(cached)
73   }
74 
75   /**
76     * Write a metadata file at the given root path
77     *
78     * @param context context
79     * @param metadata metadata configuration
80     */
81   def writeMetadata(context: FileSystemContext, metadata: NamedOptions): Unit = {
82     val file = new Path(context.root, MetadataPath)
83     if (PathCache.exists(context.fc, file, reload = true)) {
84       throw new IllegalArgumentException(
85         s"Trying to create a new storage instance but metadata already exists at '$file'")
86     }
87     val data = profile("Serialized metadata configuration") {
88       ConfigWriter[NamedOptions].to(metadata).render(RenderOptions)
89     }
90     // remove quotes around substitutions so that they resolve properly
91     // this logic relies on the fact that all strings will be quoted, and just puts another quote on
92     // either side of the expression (typesafe will concatenate them), i.e. "foo ${bar}" -> "foo "${bar}""
93     val interpolated = data.replaceAll("\\$\\{[a-zA-Z0-9_.]+}", "\"$0\"")
94     profile("Persisted metadata configuration") {
95       WithClose(context.fc.create(file, java.util.EnumSet.of(CreateFlag.CREATE), CreateOpts.createParent)) { out =>
96         out.write(interpolated.getBytes(StandardCharsets.UTF_8))
97         out.hflush()
98         out.hsync()
99       }
100     }
101     val toCache = if (data == interpolated) { metadata } else {
102       // reload through ConfigFactory to resolve substitutions
103       ConfigSource.fromConfig(ConfigFactory.load(ConfigFactory.parseString(interpolated, ParseOptions)))
104           .loadOrThrow[NamedOptions]
105     }
106     cache.put(context.root.toUri.toString, toCache)
107     PathCache.register(context.fc, file)
108   }

111   /**
112     * Transition the old single-file metadata.json to the new append-log format
113     *
114     * @param context file system context
115     * @return
116     */
117   private def transitionMetadata(context: FileSystemContext, config: Config): Option[NamedOptions] = {
118     import scala.collection.JavaConverters._
119 
120     try {
121       val sft = SimpleFeatureTypes.createType(config.getConfig("featureType"), path = None)
122       val encoding = config.getString("encoding")
123       val scheme = {
124         val schemeConfig = config.getConfig("partitionScheme")
125         val schemeOpts = schemeConfig.getConfig("options")
126         NamedOptions(schemeConfig.getString("scheme"),
127           schemeOpts.entrySet().asScala.map(e => e.getKey -> schemeOpts.getString(e.getKey)).toMap)
128       }
129       val leafStorage = scheme.options.get("leaf-storage").forall(_.toBoolean)
130       val meta = Metadata(sft, encoding, scheme, leafStorage)
131       val partitionConfig = config.getConfig("partitions")
132 
133       val defaults = FileBasedMetadata.LegacyOptions
134       WithClose(new FileBasedMetadataFactory().create(context, defaults.options, meta)) { metadata =>
135         partitionConfig.root().entrySet().asScala.foreach { e =>
136           val name = e.getKey
137           val files = partitionConfig.getStringList(name).asScala.map(StorageFile(_, 0L))
138           metadata.addPartition(PartitionMetadata(name, files.toSeq, None, 0L))
139         }
140       }
141 
142       Some(defaults)
143     } catch {
144       case NonFatal(e) => logger.warn("Error transitioning old metadata format: ", e); None
145     }
146   }
