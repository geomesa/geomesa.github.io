27 /**
28  * Scheme for partitioning based on "receipt time", i.e. when a message is received. Generally this is useful
29  * only for reading existing data that may have been aggregated and stored by an external process.
30  *
31  * @param delegate delegate date time scheme options
32  * @param buffer amount of time to buffer queries by, in order to match a feature date to a receipt time date -
33  *               i.e. the amount of latency in the ingest process
34  */
35 case class ReceiptTimeScheme(delegate: DateTimeScheme, buffer: Duration) extends PartitionScheme {
36 
37   override val depth: Int = delegate.depth
38 
39   override val pattern: String = delegate.pattern
40 
41   override def getPartitionName(feature: SimpleFeature): String = delegate.getPartitionName(feature)
42 
43   override def getSimplifiedFilters(filter: Filter, partition: Option[String]): Option[Seq[SimplifiedFilter]] = {
44     delegate.getSimplifiedFilters(buffered(filter), partition).map { filters =>
45       // always use the full filter since our dates are not guaranteed to match the partition bounds
46       filters.map(f => f.copy(filter = filter))
47     }
48   }
49 
50   override def getIntersectingPartitions(filter: Filter): Option[Seq[String]] =
51     delegate.getIntersectingPartitions(buffered(filter))
52 
53   override def getCoveringFilter(partition: String): Filter =
54     throw new UnsupportedOperationException("Dates may overlap in multiple partitions")
55 
56   private def buffered(filter: Filter): Filter =
57     filter.accept(new BufferingFilterVisitor(buffer, delegate.dtg), null).asInstanceOf[Filter]
58 }

83   /**

89   class BufferingFilterVisitor(buffer: Duration, dtg: String) extends DuplicatingFilterVisitor {

94     override def visit(filter: PropertyIsBetween, extraData: AnyRef): AnyRef = {
95       val factory = getFactory(extraData)
96       def buffer(p: PropertyName, lower: Literal, upper: Literal): Option[Filter] = {
97         for {
98           lo <- Option(FastConverter.evaluate(lower, classOf[Date]))
99           up <- Option(FastConverter.evaluate(upper, classOf[Date]))
100         } yield {
101           val bufferedLo = bufferDown(lo)
102           val bufferedUp = bufferUp(up)
103           // account for inverted filters that may result in invalid clauses after buffering
104           if (bufferedLo.before(bufferedUp)) {
105             factory.between(p, factory.literal(bufferedLo), factory.literal(bufferedUp), filter.getMatchAction)
106           } else {
107             Filter.EXCLUDE
108           }
109         }
110       }
111       val prop = visit(filter.getExpression, extraData)
112       val lowerBoundary = visit(filter.getLowerBoundary, extraData)
113       val upperBoundary = visit(filter.getUpperBoundary, extraData)
114       val buffered = (prop, lowerBoundary, upperBoundary) match {
115         case (p: PropertyName, lower: Literal, upper: Literal) if p.getPropertyName == dtg => buffer(p, lower, upper)
116         case _ => None
117       }
118 
119       buffered.getOrElse(super.visit(filter, extraData))
120     }

273     private def bufferUp(lit: Literal, extraData: AnyRef): Expression =
274       buffer(lit, if (inverted) { -1L * millis } else { millis }, extraData)

276     private def bufferUp(date: Date): Date =

279     private def bufferDown(lit: Literal, extraData: AnyRef): Expression =

281 
282     private def bufferDown(date: Date): Date =
283       buffer(date, if (inverted) { millis } else { -1L * millis })
284 
285     private def buffer(lit: Literal, offset: Long, extraData: AnyRef): Expression = {
286       FastConverter.convert(lit.evaluate(null), classOf[Date]) match {
287         case null => lit
288         case date => getFactory(extraData).literal(buffer(date, offset))
289       }
290     }
