152   override def compact(partition: Option[String], fileSize: Option[Long], threads: Int): Unit = {
153     val target = targetSize(fileSize)
154     partition.map(Seq(_)).getOrElse(metadata.getPartitions().map(_.name)).foreach { partition =>
155       val paths = getFilePaths(partition)
156       val toCompact = target match {
157         case None => paths
158         case Some(t) =>
159           paths.filter { p =>
160             if (fileIsSized(p.path, t)) {
161               logger.debug(s"Skipping compaction for file [${p.path}] (already target size)")
162               false
163             } else {
164               true
165             }
166           }
167       }
168 
169       if (toCompact.isEmpty) {
170         logger.debug("Skipping compaction - no files to compact")
171       } else if (toCompact.lengthCompare(1) == 0 && target.forall(fileIsSized(toCompact.head.path, _))) {
172         logger.debug(s"Skipping compaction for single data file [${toCompact.mkString}]")
173       } else {
174         logger.debug(s"Compacting data files: [${toCompact.mkString(", ")}]")
175 
176         var written = 0L
177         val bounds = new Envelope()
178 
179         val reader = createReader(None, None)
180 
181         def writer: FileSystemWriter =
182           createWriter(partition, StorageFileAction.Append, FileType.Compacted, target)
183         def threaded: CloseableIterator[SimpleFeature] =
184           FileSystemThreadedReader(Iterator.single(reader -> toCompact), threads)
185 
186         WithClose(writer, threaded) { case (writer, features) =>
187           while (features.hasNext) {
188             val feature = features.next()
189             writer.write(feature)
190             written += 1
191             val geom = feature.getDefaultGeometry.asInstanceOf[Geometry]
192             if (geom != null) {
193               bounds.expandToInclude(geom.getEnvelopeInternal)
194             }
195           }
196         }
197 
198         logger.debug(s"Deleting old files [${toCompact.mkString(", ")}]")
199         metadata.removePartition(
200           PartitionMetadata(partition, toCompact.map(_.file), PartitionBounds(bounds), written))
201 
202         val failures = ListBuffer.empty[Path]
203         toCompact.foreach { file =>
204           if (!context.fs.delete(file.path, false)) {
205             failures.append(file.path)
206           }
207           PathCache.invalidate(context.fs, file.path)
208         }
209 
210         if (failures.nonEmpty) {
211           logger.error(s"Failed to delete some files: [${failures.mkString(", ")}]")
212         }
213 
214         logger.debug(s"Compacted $written records")
215       }
216     }
217   }

230       action: StorageFileAction,
231       fileType: FileType,
232       targetFileSize: Option[Long] = None): FileSystemWriter = {

234     def pathAndObserver: WriterConfig = {
235       val path = StorageUtils.nextFile(context.root, partition, metadata.leafStorage, extension, fileType)
236       PathCache.register(context.fs, path)
237       val updateObserver = new UpdateObserver(partition, path, action)
238       val observer = if (observers.isEmpty) { updateObserver } else {
239         new CompositeObserver(observers.map(_.apply(path)).+:(updateObserver))
240       }
241       WriterConfig(path, observer)
242     }

244     targetSize(targetFileSize) match {
245       case None => createWriter(pathAndObserver)
246       case Some(s) => new ChunkedFileSystemWriter(Iterator.continually(pathAndObserver), estimator(s))
247     }
248   }

252   /**
253    * Writes files up to a given size, then starts a new file
254    *
255    * @param paths iterator of files to write
256    * @param estimator target file size estimator
257    */
258   class ChunkedFileSystemWriter(paths: Iterator[WriterConfig], estimator: FileSizeEstimator)
259       extends FileSystemWriter {
260 
261     private var count = 0L // number of features written
262     private var total = 0L // sum size of all finished chunks
263     private var remaining = estimator.estimate(0L)
264 
265     private var path: Path = _
266     private var writer: FileSystemWriter = _
267 
268     override def write(feature: SimpleFeature): Unit = {
269       if (writer == null) {
270         val config = paths.next
271         path = config.path
272         writer = createWriter(config)
273       }
274       writer.write(feature)
275       count += 1
276       remaining -= 1
277       if (remaining == 0) {
278         writer.close()
279         writer = null
280         // adjust our estimate to account for the actual bytes written
281         total += context.fs.getFileStatus(path).getLen
282         estimator.update(total, count)
283         remaining = estimator.estimate(0L)
284       }
285     }
286 
287     override def flush(): Unit = if (writer != null) { writer.flush() }
288 
289     override def close(): Unit = {
290       if (writer != null) {
291         writer.close()
292       }
293       updateFileSize(estimator)
294     }
295   }

303   class FileSystemUpdateWriterImpl(reader: CloseableFeatureIterator, readPartition: Option[String])
304       extends FileSystemUpdateWriter {
305 
306     private val modifiers = scala.collection.mutable.Map.empty[String, FileSystemWriter]
307     private val deleters = scala.collection.mutable.Map.empty[String, FileSystemWriter]
308 
309     private var feature: SimpleFeature = _
310     private var partition: String = _
311 
312     override def write(): Unit = {
313       if (feature == null) {
314         throw new IllegalArgumentException("Must call 'next' before calling 'write'")
315       }
316       val update = metadata.scheme.getPartitionName(feature)
317       if (update != partition) {
318         // add a delete marker in the old partition, since we only track updates per-partition
319         deleters.getOrElseUpdate(partition, createWriter(partition, StorageFileAction.Delete, FileType.Deleted)).write(feature)
320       }
321       modifiers.getOrElseUpdate(update, createWriter(update, StorageFileAction.Modify, FileType.Modified)).write(feature)
322       feature = null
323     }
324 
325     override def remove(): Unit = {
326       if (feature == null) {
327         throw new IllegalArgumentException("Must call 'next' before calling 'remove'")
328       }
329       deleters.getOrElseUpdate(partition, createWriter(partition, StorageFileAction.Delete, FileType.Deleted)).write(feature)
330       feature = null
331     }
332 
333     override def hasNext: Boolean = reader.hasNext
334 
335     override def next(): SimpleFeature = {
336       feature = reader.next() // note: our reader returns a mutable copy of the feature
337       partition = readPartition.getOrElse(metadata.scheme.getPartitionName(feature))
338       feature
339     }
340 
341     override def flush(): Unit = FlushQuietly.raise(modifiers.values.toSeq ++ deleters.values)
342 
343     override def close(): Unit = CloseQuietly.raise(Seq(reader) ++ modifiers.values ++ deleters.values ++ observers)
344   }
