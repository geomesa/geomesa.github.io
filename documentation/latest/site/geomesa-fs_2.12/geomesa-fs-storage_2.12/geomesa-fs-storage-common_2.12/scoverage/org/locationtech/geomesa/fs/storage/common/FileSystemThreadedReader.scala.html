24 /**
25   * Multi-threaded file reads
26   *
27   * @param es executor service used by the read tasks
28   * @param phaser phaser for tracking read task completion
29   * @param queue intermediate queue populated by the read tasks
30   */
31 class FileSystemThreadedReader private (es: ExecutorService, phaser: Phaser, queue: BlockingQueue[SimpleFeature])
32     extends CloseableIterator[SimpleFeature] {
33 
34   private val localQueue = new java.util.LinkedList[SimpleFeature]()
35 
36   private var current: SimpleFeature = _
37 
38   override def hasNext: Boolean = {
39     if (current != null) {
40       return true
41     }
42     current = localQueue.pollFirst
43     if (current != null) {
44       return true
45     }
46 
47     while (!phaser.isTerminated) {
48       current = queue.poll(100, TimeUnit.MILLISECONDS)
49       if (current != null) {
50         queue.drainTo(localQueue, 10000)
51         return true
52       }
53     }
54     // last check - if phaser.isTerminated, the queue should have whatever values are left
55     current = queue.poll()
56     if (current != null) {
57       queue.drainTo(localQueue, 10000)
58       true
59     } else {
60       false
61     }
62   }
63 
64   override def next(): SimpleFeature = {
65     if (hasNext) {
66       val ret = current
67       current = null
68       ret
69     } else {
70       Iterator.empty.next
71     }
72   }
73 
74   override def close(): Unit = es.shutdownNow()

77 object FileSystemThreadedReader extends StrictLogging {
78 
79   def apply(
80       readers: Iterator[(FileSystemPathReader, Seq[StorageFilePath])],
81       threads: Int): CloseableIterator[SimpleFeature] = {
82 
83     if (threads < 2) {
84       readers.flatMap { case (reader, files) =>
85         val mods = scala.collection.mutable.HashSet.empty[String]
86         // ensure files are sorted in reverse chronological order
87         CloseableIterator(files.sorted.iterator).flatMap(f => read(reader, f, mods))
88       }
89     } else {
90       val queue = new LinkedBlockingQueue[SimpleFeature](2000000)
91       val es = Executors.newFixedThreadPool(threads)
92 
93       val phaser = new Phaser(1) {
94         override protected def onAdvance(phase: Int, registeredParties: Int): Boolean = {
95           // when all tasks have been completed, shutdown the executor service
96           es.shutdown()
97           true // return true to indicate the phaser should terminate
98         }
99       }
100 
101       try {
102         var child = new Phaser(phaser) // ensure that we don't register too many parties on this phaser
103         var parties = 0 // track registered parties
104         readers.foreach { case (reader, files) =>
105           // group our files by actions which can be parallelized
106           val groups = scala.collection.mutable.ListBuffer.empty[Seq[StorageFilePath]]
107           var group = scala.collection.mutable.ArrayBuffer.empty[StorageFilePath]
108           // ensure files are sorted in reverse chronological order
109           files.sorted.foreach { file =>
110             if (file.file.action == StorageFileAction.Append) {
111               group += file
112             } else {
113               if (group.nonEmpty) {
114                 groups += group.toSeq
115                 group = scala.collection.mutable.ArrayBuffer.empty[StorageFilePath]
116               }
117               groups += Seq(file)
118             }
119           }
120           if (group.nonEmpty) {
121             groups += group.toSeq // add the last group
122           }
123 
124           // each chained reader task will register at most groups.length parties
125           parties += groups.length
126           if (parties > PhaserUtils.MaxParties) {
127             parties = groups.length
128             child = new Phaser(phaser)
129           }
130           child.register() // register new task
131           es.submit(new ChainedReaderTask(es, child, reader, groups.head, groups.tail.toSeq, queue))
132         }
133       } catch {
134         case NonFatal(e) => es.shutdownNow(); throw e
135       } finally {
136         phaser.arriveAndDeregister()
137       }
138 
139       new FileSystemThreadedReader(es, phaser, queue)
140     }
141   }

143   /**
144     * Reads a file
145     *
146     * @param reader reader
147     * @param file file to read
148     * @param mods collection to track modifications
149     * @return
150     */
151   private def read(
152       reader: FileSystemPathReader,
153       file: StorageFilePath,
154       mods: scala.collection.mutable.Set[String]): CloseableIterator[SimpleFeature] = {
155     file.file.action match {
156       case StorageFileAction.Append => new AppendingReaderIterator(reader, file.path, mods)
157       case StorageFileAction.Modify => new ModifyingReaderIterator(reader, file.path, mods)
158       case StorageFileAction.Delete => new DeletingReaderIterator(reader, file.path, mods)
159       case _ => throw new NotImplementedError(s"Unexpected storage action: ${file.file.action}")
160     }
161   }
162 
163   /**
164     * Performs a set of chained (dependent) reads. The groups must be handled serially, but the files within a
165     * given group can be read in parallel.
166     *
167     * @param es executor service for submitting new read tasks
168     * @param phaser phaser to track run lifecycle
169     * @param reader reader
170     * @param group current group of files that can be read in parallel
171     * @param chain remaining groups of files that must be read sequentially
172     * @param queue result queue
173     * @param mods modifications/deletes for this group of files
174     */
175   private class ChainedReaderTask(
176       es: ExecutorService,
177       phaser: Phaser,
178       reader: FileSystemPathReader,
179       group: Seq[StorageFilePath],
180       chain: Seq[Seq[StorageFilePath]],
181       queue: BlockingQueue[SimpleFeature],
182       mods: scala.collection.mutable.Set[String] = scala.collection.mutable.HashSet.empty[String]
183     ) extends Runnable {
184 
185     override def run(): Unit = {
186       val child = new Phaser(1) {
187         override protected def onAdvance(phase: Int, registeredParties: Int): Boolean = {
188           // when this group is done, submit the next group for processing
189           try {
190             if (chain.nonEmpty) {
191               phaser.register() // register new task
192               es.submit(new ChainedReaderTask(es, phaser, reader, chain.head, chain.tail, queue, mods))
193             }
194             true // return true to indicate the phaser should terminate
195           } finally {
196             phaser.arriveAndDeregister()
197           }
198         }
199       }
200       try {
201         group.foreach { file =>
202           child.register() // register new task
203           es.submit(new ReaderTask(child, queue, file.path, read(reader, file, mods)))
204         }
205       } finally {
206         child.arriveAndDeregister()
207       }
208     }
209   }

237   /**
238     * Reads a file, skipping features marked as modified
239     *
240     * @param reader reader
241     * @param path file path
242     * @param mods set of modified feature IDs that shouldn't be returned
243     */
244   private class AppendingReaderIterator(
245       reader: FileSystemPathReader,
246       path: Path,
247       mods: scala.collection.Set[String]
248     ) extends CloseableIterator[SimpleFeature] {
249 
250     logger.debug(s"Reading file $path")
251 
252     private var count = 0
253 
254     private val delegate = {
255       if (mods.isEmpty) {
256         reader.read(path)
257       } else {
258         reader.read(path).filter { f => if (mods.contains(f.getID)) { count += 1; false } else { true } }
259       }
260     }
261 
262     override def hasNext: Boolean = delegate.hasNext
263 
264     override def next(): SimpleFeature = {
265       count += 1
266       // need to copy the feature as it can be re-used
267       ScalaSimpleFeature.copy(delegate.next())
268     }
269 
270     override def close(): Unit = {
271       logger.debug(s"File $path produced $count records")
272       delegate.close()
273     }
274   }

309   /**
310     * Reads a file, tracking deletes
311     *
312     * @param reader reader
313     * @param path file path
314     * @param mods deleted feature ids will be added here
315     */
316   private class DeletingReaderIterator(
317       reader: FileSystemPathReader,
318       path: Path,
319       mods: scala.collection.mutable.Set[String]
320     ) extends CloseableIterator[SimpleFeature] {
321 
322     logger.debug(s"Reading deleting file $path")
323 
324     try {
325       var count = 0
326       WithClose(reader.read(path)) { features =>
327         while (features.hasNext) {
328           mods.add(features.next().getID)
329           count += 1
330         }
331       }
332       logger.debug(s"File $path produced $count deleted records")
333     } catch {
334       case NonFatal(e) => logger.error(s"Error reading deleting file $path", e)
335     }
336 
337     override def hasNext: Boolean = false
338 
339     override def next(): SimpleFeature = Iterator.empty.next()
340 
341     override def close(): Unit = {}
342   }
