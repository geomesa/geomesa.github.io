84   class SimpleFeatureWriter(attributes: Array[AttributeWriter[AnyRef]], includeVisibilities: Boolean) {
85 
86     private val fids = new FidWriter(attributes.length) // put the ID at the end of the record
87     private val vis = new VisibilityWriter(attributes.length + 1)
88 
89     def write(consumer: RecordConsumer, value: SimpleFeature): Unit = {
90       consumer.startMessage()
91       var i = 0
92       while (i < attributes.length) {
93         attributes(i).apply(consumer, value.getAttribute(i))
94         i += 1
95       }
96       fids.apply(consumer, value.getID)
97       if (includeVisibilities) {
98         vis.apply(consumer, SecurityUtils.getVisibility(value))
99       }
100       consumer.endMessage()
101     }
102   }
