40 class ParquetFileSystemStorage(context: FileSystemContext, metadata: StorageMetadata)
41     extends AbstractFileSystemStorage(context, metadata, ParquetFileSystemStorage.FileExtension) {
42 
43   private val authProvider: AuthorizationsProvider =
44     AuthUtils.getProvider(
45       Collections.emptyMap[String, String](),
46       context.conf.get(AuthsParam.key, "").split(",").toSeq.filter(_.nonEmpty)
47     )
48 
49   override protected def createWriter(file: Path, observer: FileSystemObserver): FileSystemWriter = {
50     val conf = new Configuration(context.conf)
51     SimpleFeatureParquetSchema.setSft(conf, metadata.sft)
52     val observers =
53       if (FileValidationEnabled.toBoolean.get) {
54         CompositeObserver(Seq(observer, FileValidationObserver(file)))
55       } else {
56         observer
57       }
58     new ParquetFileSystemWriter(file, conf, observers)
59   }
60 
61   override protected def createReader(
62       filter: Option[Filter],
63       transform: Option[(String, SimpleFeatureType)]): FileSystemPathReader = {
64     // readSft has all the fields needed for filtering and return
65     val ReadSchema(readSft, readTransform) = ReadSchema(metadata.sft, filter, transform)
66     val ReadFilter(fc, residualFilter) = ReadFilter(readSft, filter)
67     val parquetFilter = fc.map(FilterCompat.get).getOrElse(FilterCompat.NOOP)
68     val gtFilter = residualFilter.map(FastFilterFactory.optimize(readSft, _))
69     val visFilter = VisibilityUtils.visible(Some(authProvider))
70 
71     logger.debug(
72       s"Parquet filter: ${parquetFilter match { case f: FilterPredicateCompat => f.getFilterPredicate; case f => f }} " +
73         s"and modified gt filter: ${gtFilter.getOrElse(Filter.INCLUDE)}")
74 
75     // WARNING it is important to create a new conf per query
76     // because we communicate the transform SFT set here
77     // with the init() method on SimpleFeatureReadSupport via
78     // the parquet api. Thus we need to deep copy conf objects
79     val conf = new Configuration(context.conf)
80     StorageConfiguration.setSft(conf, readSft)
81 
82     new ParquetFileSystemReader(conf, readSft, parquetFilter, gtFilter, visFilter, readTransform)
83   }
84 }

93   /**
94    * Validate a file by reading it back
95    *
96    * @param file file to validate
97    */
98   case class FileValidationObserver(file: Path) extends FileSystemObserver {
99     override def apply(feature: SimpleFeature): Unit = {}
100     override def flush(): Unit = {}
101     override def close(): Unit = {
102       try {
103         WithClose(ParquetReader.builder(new GroupReadSupport(), file).build()) { reader =>
104           var record = reader.read()
105           while (record != null) {
106             // Process the record
107             record = reader.read()
108           }
109           logger.trace(s"$file is a valid Parquet file")
110         }
111       } catch {
112         case NonFatal(e) => throw new RuntimeException(s"File appears to be corrupted: $file", e)
113       }
114     }
115   }
