39 class ParquetFileSystemStorage(context: FileSystemContext, metadata: StorageMetadata)
40     extends AbstractFileSystemStorage(context, metadata, ParquetFileSystemStorage.FileExtension) {
41 
42   override protected def createWriter(file: Path, observer: FileSystemObserver): FileSystemWriter =
43     new ParquetFileSystemWriter(metadata.sft, context, file, observer)
44 
45   override protected def createReader(
46       filter: Option[Filter],
47       transform: Option[(String, SimpleFeatureType)]): FileSystemPathReader = {
48     // readSft has all the fields needed for filtering and return
49     val ReadSchema(readSft, readTransform) = ReadSchema(metadata.sft, filter, transform)
50     val ReadFilter(fc, residualFilter) = ReadFilter(readSft, filter)
51     val parquetFilter = fc.map(FilterCompat.get).getOrElse(FilterCompat.NOOP)
52     val gtFilter = residualFilter.map(FastFilterFactory.optimize(readSft, _))
53 
54     logger.debug(s"Parquet filter: $parquetFilter and modified gt filter: ${gtFilter.getOrElse(Filter.INCLUDE)}")
55 
56     // WARNING it is important to create a new conf per query
57     // because we communicate the transform SFT set here
58     // with the init() method on SimpleFeatureReadSupport via
59     // the parquet api. Thus we need to deep copy conf objects
60     val conf = new Configuration(context.conf)
61     StorageConfiguration.setSft(conf, readSft)
62 
63     new ParquetPathReader(conf, readSft, parquetFilter, gtFilter, readTransform)
64   }
65 }

74   class ParquetFileSystemWriter(
75       sft: SimpleFeatureType,
76       context: FileSystemContext,
77       file: Path,
78       observer: FileSystemObserver = NoOpObserver
79     ) extends FileSystemWriter {
80 
81     private val conf = {
82       val conf = new Configuration(context.conf)
83       StorageConfiguration.setSft(conf, sft)
84       conf
85     }
86 
87     private val writer = SimpleFeatureParquetWriter.builder(file, conf).build()
88 
89     override def write(f: SimpleFeature): Unit = {
90       writer.write(f)
91       observer.write(f)
92     }
93     override def flush(): Unit = observer.flush()
94     override def close(): Unit = {
95       CloseQuietly(Seq(writer, observer)).foreach(e => throw e)
96       PathCache.register(context.fs, file)
97       if (FileValidationEnabled.toBoolean.get) {
98         validateParquetFile(file)
99       }
100     }
101   }
102 
103   /**
104    * Validate a file by reading it back
105    *
106    * @param file file to validate
107    */
108   def validateParquetFile(file: Path): Unit = {
109     var reader: ParquetReader[Group] = null
110     try {
111       // read Parquet file content
112       reader = ParquetReader.builder(new GroupReadSupport(), file).build()
113       var record = reader.read()
114       while (record != null) {
115         // Process the record
116         record = reader.read()
117       }
118       logger.trace(s"$file is a valid Parquet file")
119     } catch {
120       case NonFatal(e) => throw new RuntimeException(s"Unable to validate $file: File may be corrupted", e)
121     } finally {
122       if (reader != null) {
123         reader.close()
124       }
125     }
126   }
