86     * @return

88   def read(context: InitContext): Option[SimpleFeatureParquetSchema] = {
89     val metadata = new java.util.HashMap[String, String]()
90     // copy in the file level metadata - note, do this before copying in the conf so that transform schemas are applied correctly
91     context.getKeyValueMetadata.asScala.foreach { case (k, v) => if (!v.isEmpty) { metadata.put(k, v.iterator.next) }}
92     val conf = context.getParquetConfiguration
93     // noinspection ScalaDeprecation

95       Option(conf.get(key)).foreach(metadata.put(key, _))
96     }

108   private def read(fileSchema: MessageType, metadata: java.util.Map[String, String]): Option[SimpleFeatureParquetSchema] = {
109     for {
110       name <- Option(metadata.get(SftNameKey))
111       spec <- Option(metadata.get(SftSpecKey))
112     } yield {
113       val sft = SimpleFeatureTypes.createType(name, spec)
114       // noinspection ScalaDeprecation
115       lazy val schemaVersion = Option(metadata.get(SchemaVersionKey)).map(_.toInt).getOrElse(0)
116       val geometryEncoding = Option(metadata.get(GeometryEncodingKey)).map(GeometryEncoding.apply).getOrElse {
117         if (schemaVersion == 1) { GeometryEncoding.GeoMesaV1 } else { GeometryEncoding.GeoMesaV0 }
118       }
119       val visibilities = fileSchema.containsField(VisibilitiesField)
120       SimpleFeatureParquetSchema(sft, schema(sft, geometryEncoding, visibilities), Collections.unmodifiableMap(metadata))
121     }

171   /**
172    * Get the message type for a simple feature type
173    *
174    * @param sft simple feature type
175    * @param geometries geometry encoding
176    * @param visibilities include visibilities
177    * @return
178    */
179   private def schema(sft: SimpleFeatureType, geometries: GeometryEncoding, visibilities: Boolean): MessageType = {
180     val isV0 = geometries == GeometryEncoding.GeoMesaV0
181     val id = {
182       val primitive = PrimitiveTypeName.BINARY
183       val builder = if (isV0) { Types.primitive(primitive, Repetition.REPEATED) } else { Types.required(primitive) }
184       builder.as(LogicalTypeAnnotation.stringType()).named(FeatureIdField)
185     }
186     val vis =
187       if (visibilities) {
188         Some(Types.optional(PrimitiveTypeName.BINARY).as(LogicalTypeAnnotation.stringType()).named(VisibilitiesField))
189       } else {
190         None
191       }
192     // note: id field goes at the end of the record, then vis
193     val fields = (sft.getAttributeDescriptors.asScala.map(schema(_, geometries)) :+ id) ++ vis
194     val name = if (isV0) { sft.getTypeName } else { alphaNumericSafeString(sft.getTypeName) }
195     new MessageType(name, fields.asJava)
196   }

198   /**
199    * Create a parquet field type from an attribute descriptor
200    *
201    * @param descriptor descriptor
202    * @param geometries geometry encoding
203    * @return
204    */
205   private def schema(descriptor: AttributeDescriptor, geometries: GeometryEncoding): Type = {
206     val isV0 = geometries == GeometryEncoding.GeoMesaV0
207     val bindings = ObjectType.selectType(descriptor)
208     val builder = bindings.head match {
209       case ObjectType.GEOMETRY => GeometrySchema(bindings(1), geometries)
210       case ObjectType.LIST     => Binding(bindings(1), isV0).list()
211       case ObjectType.MAP      => Binding(bindings(1), isV0).map(Binding(bindings(2), isV0))
212       case p                   => Binding(p, isV0).primitive()
213     }
214     builder.named(if (isV0) { descriptor.getLocalName } else { alphaNumericSafeString(descriptor.getLocalName) })
215   }

217   /**
218     * Defined parquet field, applicable as primitives, list elements, or map key/values
219     *
220     * @param name type
221     * @param as original type
222     * @param length fixed length
223     */
224   private class Binding(val name: PrimitiveTypeName, as: Option[LogicalTypeAnnotation] = None, length: Option[Int] = None) {
225 
226     def primitive(): Types.Builder[_, _ <: Type] = opts(Types.primitive(name, Repetition.OPTIONAL))
227 
228     def list(): Types.Builder[_, _ <: Type] = opts(Types.optionalList().optionalElement(name))
229 
230     def map(value: Binding): Types.Builder[_, _ <: Type] = {
231       val key = opts(Types.optionalMap().key(name))
232       value.opts(key.optionalValue(value.name))
233     }
234 
235     private def opts[T <: BasePrimitiveBuilder[_ <: Type, _]](b: T): T = {
236       length.foreach(b.length)
237       as.foreach(b.as)
238       b
239     }
240   }

242   private object Binding {
243 
244     private val bindings = Map(
245       ObjectType.DATE    -> new Binding(PrimitiveTypeName.INT64,  Some(LogicalTypeAnnotation.timestampType(true, TimeUnit.MILLIS))),
246       ObjectType.STRING  -> new Binding(PrimitiveTypeName.BINARY, Some(LogicalTypeAnnotation.stringType())),
247       ObjectType.INT     -> new Binding(PrimitiveTypeName.INT32),
248       ObjectType.DOUBLE  -> new Binding(PrimitiveTypeName.DOUBLE),
249       ObjectType.LONG    -> new Binding(PrimitiveTypeName.INT64),
250       ObjectType.FLOAT   -> new Binding(PrimitiveTypeName.FLOAT),
251       ObjectType.BOOLEAN -> new Binding(PrimitiveTypeName.BOOLEAN),
252       ObjectType.BYTES   -> new Binding(PrimitiveTypeName.BINARY),
253       ObjectType.UUID    -> new Binding(PrimitiveTypeName.FIXED_LEN_BYTE_ARRAY, None, Some(16))
254     )
255 
256     def apply(binding: ObjectType, isV0: Boolean): Binding = {
257       if (isV0 && binding == ObjectType.UUID) {
258         new Binding(PrimitiveTypeName.BINARY)
259       } else {
260         bindings.getOrElse(binding, throw new NotImplementedError(s"No mapping defined for type $binding"))
261       }
262     }
263   }
