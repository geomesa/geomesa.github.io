21   /**
22    * Create a builder for a parquet geometry field
23    *
24    * @param binding geometry type
25    * @param encoding encoding scheme
26    * @return
27    */
28   def apply(binding: ObjectType, encoding: GeometryEncoding): Types.Builder[_, _ <: Type] = {
29     encoding.fn.lift(binding).getOrElse {
30       throw new NotImplementedError(s"No mapping defined for geometry type $binding with encoding $encoding")
31     }
32   }

34   /**
35    * Enumeration of supported geometry encodings
36    */
37   sealed trait GeometryEncoding {

39   }
