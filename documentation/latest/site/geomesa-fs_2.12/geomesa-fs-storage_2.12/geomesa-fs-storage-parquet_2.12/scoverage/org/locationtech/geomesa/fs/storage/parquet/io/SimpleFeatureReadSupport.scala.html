111   class SimpleFeatureGroupConverter(schema: SimpleFeatureParquetSchema) extends GroupConverter {
112 
113     private val idConverter = new PrimitiveConverter with ValueMaterializer {
114       private var id: Binary = _
115       override def materialize(): AnyRef = if (id == null) { null } else { id.toStringUsingUTF8 }
116       override def reset(): Unit = id = null
117       override def addBinary(value: Binary): Unit = id = value
118     }
119     private val visConverter = new PrimitiveConverter with ValueMaterializer {
120       private var vis: Binary = _
121       override def materialize(): AnyRef = if (vis == null) { null } else { vis.toStringUsingUTF8.intern() }
122       override def reset(): Unit = vis = null
123       override def addBinary(value: Binary): Unit = vis = value
124     }
125 
126     private val converters = {
127       val builder = Array.newBuilder[Converter with ValueMaterializer]
128       var i = 0
129       while (i < schema.sft.getAttributeCount) {
130         builder += SimpleFeatureReadSupport.attribute(ObjectType.selectType(schema.sft.getDescriptor(i)))
131         i += 1
132       }
133       builder += idConverter
134       if (schema.hasVisibilities) {
135         builder += visConverter
136       }
137       builder.result()
138     }
139 
140     // don't materialize unless we have to
141     def materialize: SimpleFeature = {
142       val id = idConverter.materialize().asInstanceOf[String]
143       val vis = visConverter.materialize().asInstanceOf[String]
144       val values = Array.tabulate[AnyRef](schema.sft.getAttributeCount)(i => converters(i).materialize())
145       val userData = if (vis == null) { null } else {
146         val map = new java.util.HashMap[AnyRef, AnyRef](1)
147         map.put(SecurityUtils.FEATURE_VISIBILITY, vis)
148         map
149       }
150       new ScalaSimpleFeature(schema.sft, id, values, userData)
151     }
152 
153     override def start(): Unit = converters.foreach(_.reset())
154 
155     override def getConverter(fieldIndex: Int): Converter = converters(fieldIndex)
156 
157     override def end(): Unit = {}
158   }
