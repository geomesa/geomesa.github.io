33   override def createDataStore(params: java.util.Map[String, _]): DataStore = {
34 
35     val xml = ConfigsParam.lookupOpt(params)
36     val resources = ConfigPathsParam.lookupOpt(params).toSeq.flatMap(_.split(',')).map(_.trim).filterNot(_.isEmpty)
37 
38     val conf = if (xml.isEmpty && resources.isEmpty) { FileSystemDataStoreFactory.configuration } else {
39       val conf = new Configuration(FileSystemDataStoreFactory.configuration)
40       // add the explicit props first, they may be needed for loading the path resources
41       xml.foreach(x => conf.addResource(new ByteArrayInputStream(x.getBytes(StandardCharsets.UTF_8))))
42       resources.foreach(HadoopUtils.addResource(conf, _))
43       conf
44     }
45 
46     val path = new Path(PathParam.lookup(params))
47     val encoding = EncodingParam.lookupOpt(params).filterNot(_.isEmpty)
48 
49     // Need to do more tuning here. On a local system 1 thread (so basic producer/consumer) was best
50     // because Parquet is also threading the reads underneath I think. using prod/cons pattern was
51     // about 30% faster but increasing beyond 1 thread slowed things down. This could be due to the
52     // cost of serializing simple features though. need to investigate more.
53     //
54     // However, if you are doing lots of filtering it appears that bumping the threads up high
55     // can be very useful. Seems possibly numcores/2 might is a good setting (which is a standard idea)
56 
57     val readThreads = ReadThreadsParam.lookup(params)
58     val writeTimeout = WriteTimeoutParam.lookup(params)
59 
60     val namespace = NamespaceParam.lookupOpt(params)
61 
62     val fs = FileSystem.get(path.toUri, conf)
63 
64     new FileSystemDataStore(fs, conf, path, readThreads, writeTimeout, encoding, namespace)
65   }
