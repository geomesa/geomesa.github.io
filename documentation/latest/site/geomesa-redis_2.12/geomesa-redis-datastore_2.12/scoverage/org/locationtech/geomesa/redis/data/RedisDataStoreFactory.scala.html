89   /**
90     * Builds a redis connection from the data store parameters
91     *
92     * @param params params
93     * @return
94     */
95   def buildConnection(params: java.util.Map[String, _]): Pool[_ <: CloseableJedisCommands] = {
96     ConnectionPoolParam.lookupOpt(params).getOrElse {
97       RedisClusterParam.lookup(params).booleanValue() match {
98         case false =>
99           val uri = {
100             val urls = RedisUrlParam.lookup(params).split(",")
101             if (urls.length > 1) {
102               throw new IllegalArgumentException(
103                 "Multiple Redis URLs can only be used in cluster mode. Set 'redis.clusterMode' to true.")
104             }
105             val url = urls.headOption.getOrElse {
106               throw new IllegalArgumentException("No Redis URL provided. Please set 'redis.url'")
107             }
108             // if there is no protocol/port, or the url is a valid redis url, use as is
109             // else use the redis:// protocol to support databases, etc
110             val parsed =
111               if (url.indexOf(":") == -1) {
112                 Try(new URI(url))
113               } else {
114                 parse(url).orElse(parse(s"redis://$url"))
115               }
116             parsed match {
117               case Success(uri) => uri
118               case Failure(e) =>
119                 throw new IllegalArgumentException(s"Could not create valid Redis connection URI from: $url", e)
120             }
121           }
122           val config = new GenericObjectPoolConfig[Jedis]()
123           PoolSizeParam.lookupOpt(params).foreach(s => config.setMaxTotal(s.intValue()))
124           config.setTestOnBorrow(TestConnectionParam.lookup(params))
125           val timeout = SocketTimeoutParam.lookup(params).toMillis.toInt
126 
127           new JedisPool(config, uri, timeout)
128         case true =>
129           val urls = RedisUrlParam.lookup(params).split(",")
130           if (urls.isEmpty) {
131             throw new IllegalArgumentException("No Redis URLs provided. Please set 'redis.url'")
132           }
133           val uri = URI.create(urls.head)
134           val objectPoolConfig = new GenericObjectPoolConfig[Connection]()
135           PoolSizeParam.lookupOpt(params).foreach(s => objectPoolConfig.setMaxTotal(s.intValue()))
136           objectPoolConfig.setTestOnBorrow(TestConnectionParam.lookup(params))
137           val timeout = SocketTimeoutParam.lookup(params).toMillis.toInt
138 
139           val clusterNodes: java.util.Set[HostAndPort] =
140             urls.map { url =>
141               val parsed = parse(url).getOrElse {
142                 throw new IllegalArgumentException(s"Could not create valid Redis connection URI from: $url")
143               }
144               JedisURIHelper.getHostAndPort(parsed)
145             }.toSet.asJava
146           val jedisClientConfig: JedisClientConfig =
147             DefaultJedisClientConfig.builder.connectionTimeoutMillis(timeout)
148               .socketTimeoutMillis(timeout)
149               .blockingSocketTimeoutMillis(0)
150               .user(JedisURIHelper.getUser(uri))
151               .password(JedisURIHelper.getPassword(uri))
152               .database(JedisURIHelper.getDBIndex(uri))
153               .clientName(null)
154               .build
155           new SingletonJedisClusterPool(clusterNodes, jedisClientConfig, objectPoolConfig)
156       }
157     }
158   }

160   /**
161     * Builds configuration from data store parameters
162     *
163     * @param params params
164     * @return
165     */
166   def buildConfig(params: java.util.Map[String, _]): RedisDataStoreConfig = {
167     val catalog = RedisCatalogParam.lookup(params)
168     val generateStats = GenerateStatsParam.lookup(params)
169     val pipeline = PipelineParam.lookup(params)
170 
171     val audit = if (!AuditQueriesParam.lookup(params)) { None } else {
172       Some(new AuditLogger("redis", AuditProvider.Loader.loadOrNone(params)))
173     }
174     // get the auth params passed in as a comma-delimited string
175     val authProvider = AuthUtils.getProvider(params,
176       AuthsParam.lookupOpt(params).map(_.split(",").toSeq.filterNot(_.isEmpty)).getOrElse(Seq.empty))
177 
178     val queries = RedisQueryConfig(
179       threads = QueryThreadsParam.lookup(params),
180       timeout = QueryTimeoutParam.lookupOpt(params).map(_.toMillis),
181       looseBBox = LooseBBoxParam.lookup(params).booleanValue(),
182       parallelPartitionScans = PartitionParallelScansParam.lookup(params)
183     )
184 
185     val ns = Option(NamespaceParam.lookUp(params).asInstanceOf[String])
186 
187     RedisDataStoreConfig(catalog, generateStats, audit, authProvider, queries, pipeline, ns)
188   }
