106   val MinRange: Array[Byte] = "-".getBytes(StandardCharsets.UTF_8)

112   /**
113     * Convert a byte range to a redis zlex range
114     *
115     * @param byteRange geomesa range
116     * @return
117     */
118   private def toRedisRange(byteRange: ByteRange): BoundedByteRange = byteRange match {
119     case BoundedByteRange(start, end) =>
120       val rangeStart = if (start.isEmpty) { RedisIndexAdapter.MinRange } else {
121         val range = Array.ofDim[Byte](start.length + 1)
122         System.arraycopy(start, 0, range, 1, start.length)
123         range(0) = RedisIndexAdapter.InclusiveRangePrefix
124         range
125       }
126       val rangeEnd = if (end.isEmpty) { RedisIndexAdapter.MaxRange } else {
127         val range = Array.ofDim[Byte](end.length + 1)
128         System.arraycopy(end, 0, range, 1, end.length)
129         range(0) = RedisIndexAdapter.ExclusiveRangePrefix
130         range
131       }
132       BoundedByteRange(rangeStart, rangeEnd)
133 
134     case SingleRowByteRange(row) =>
135       val rangeStart = Array.ofDim[Byte](row.length + 1)
136       System.arraycopy(row, 0, rangeStart, 1, row.length)
137       rangeStart(0) = RedisIndexAdapter.InclusiveRangePrefix
138       // since the value is appended to the row, we have to add a suffix
139       val rangeEnd = Array.ofDim[Byte](row.length + 1 + ByteRange.UnboundedUpperRange.length)
140       System.arraycopy(row, 0, rangeEnd, 1, row.length)
141       System.arraycopy(ByteRange.UnboundedUpperRange, 0, rangeEnd, row.length + 1, ByteRange.UnboundedUpperRange.length)
142       rangeEnd(0) = RedisIndexAdapter.ExclusiveRangePrefix
143       BoundedByteRange(rangeStart, rangeEnd)
144   }
145 
146   /**
147     * Convert a byte range into a zlex range, specifically for the id index. Since we store the length
148     * of the id in the row key, we have to prepend that to our ranges
149     *
150     * @param byteRange geomesa range
151     * @return
152     */
153   private def toRedisIdRange(byteRange: ByteRange): BoundedByteRange = byteRange match {
154     case BoundedByteRange(start, end) =>
155       val rangeStart = if (start.isEmpty) { RedisIndexAdapter.MinRange } else {
156         // add the two byte length prefix
157         val range = Array.ofDim[Byte](start.length + 3)
158         System.arraycopy(start, 0, range, 3, start.length)
159         ByteArrays.writeShort(start.length.toShort, range, 1)
160         range(0) = RedisIndexAdapter.InclusiveRangePrefix
161         range
162       }
163       val rangeEnd = if (end.isEmpty) { RedisIndexAdapter.MaxRange } else {
164         // add the two byte length prefix
165         val range = Array.ofDim[Byte](end.length + 3)
166         System.arraycopy(end, 0, range, 3, end.length)
167         ByteArrays.writeShort(end.length.toShort, range, 1)
168         range(0) = RedisIndexAdapter.ExclusiveRangePrefix
169         range
170       }
171       BoundedByteRange(rangeStart, rangeEnd)
172 
173     case SingleRowByteRange(row) =>
174       // add the two byte length prefix
175       val rangeStart = Array.ofDim[Byte](row.length + 3)
176       System.arraycopy(row, 0, rangeStart, 3, row.length)
177       ByteArrays.writeShort(row.length.toShort, rangeStart, 1)
178       rangeStart(0) = RedisIndexAdapter.InclusiveRangePrefix
179       // since the value is appended to the row, we have to add a suffix
180       val rangeEnd = Array.ofDim[Byte](row.length + 3 + ByteRange.UnboundedUpperRange.length)
181       System.arraycopy(row, 0, rangeEnd, 3, row.length)
182       System.arraycopy(ByteRange.UnboundedUpperRange, 0, rangeEnd, row.length + 3, ByteRange.UnboundedUpperRange.length)
183       ByteArrays.writeShort(row.length.toShort, rangeEnd, 1)
184       rangeEnd(0) = RedisIndexAdapter.ExclusiveRangePrefix
185       BoundedByteRange(rangeStart, rangeEnd)
186   }

191     private var idSerializer: (Array[Byte], Int, Int, SimpleFeature) => String = _
192 
193     override def apply(result: Array[Byte]): SimpleFeature = {
194       // parse out the feature id and the serialized value from the concatenated row + value
195       val idStart = index.getIdOffset(result, 0, result.length)
196       val idLength = ByteArrays.readShort(result, idStart)
197       val id = idSerializer(result, idStart + 2, idLength, null)
198       val valueStart = idStart + idLength + 2
199       serializer.deserialize(id, result, valueStart, result.length - valueStart)
200     }

203       idSerializer = GeoMesaFeatureIndex.idFromBytes(index.sft)
204       KryoFeatureSerializer.builder(index.sft).`lazy`.withUserData.withoutId.build()
205     }

208   /**
209     * Writer for redis
210     *
211     * @param jedis connection
212     * @param indices indices to write to
213     * @param partition partition to write to
214     * @param aging age-off writer
215     * @param wrapper feature wrapper
216     */
217   class RedisIndexWriter(
218       jedis: JedisPool,
219       indices: Seq[GeoMesaFeatureIndex[_, _]],
220       partition: Option[String],
221       aging: Option[AgeOffWriter],
222       wrapper: FeatureWrapper[RedisWritableFeature]
223     ) extends BaseIndexWriter[RedisWritableFeature](indices, wrapper) {
224 
225     private val batchSize = RedisSystemProperties.WriteBatchSize.toInt match {
226       case Some(s) if s > 0 => s - 1
227       case _ =>
228         throw new IllegalStateException(s"Value '${RedisSystemProperties.WriteBatchSize.get}' for " +
229             s"'${RedisSystemProperties.WriteBatchSize.property}' is not a positive int")
230     }
231 
232     private val tables = indices.toArray.map { index =>
233       index.getTableNames(partition) match {
234         case Seq(t) => t.getBytes(StandardCharsets.UTF_8) // should always be writing to a single table here
235         case names => throw new IllegalStateException(s"Expected a single table but got: ${names.mkString(", ")}")
236       }
237     }
238 
239     private val inserts = Array.fill[java.util.Map[Array[Byte], java.lang.Double]](tables.length)(new java.util.HashMap[Array[Byte], java.lang.Double]())
240     private val deletes = Array.fill[ArrayBuffer[Array[Byte]]](tables.length)(ArrayBuffer.empty[Array[Byte]])
241 
242     private var i = 0
243     private var batch = 0
244 
245     private val errors = ArrayBuffer.empty[Throwable]
246 
247     override protected def append(feature: RedisWritableFeature, values: Array[RowKeyValue[_]]): Unit = {
248       i = 0
249       while (i < values.length) {
250         val insert = inserts(i)
251         values(i) match {
252           case kv: SingleRowKeyValue[_] =>
253             kv.values.foreach(v => insert.put(ByteArrays.concat(kv.row, v.value), 0d))
254           case kv: MultiRowKeyValue[_] =>
255             kv.rows.foreach(row => kv.values.foreach(v => insert.put(ByteArrays.concat(row, v.value), 0d)))
256         }
257         i += 1
258       }
259 
260       aging.foreach(_.write(feature))
261 
262       if (batch < batchSize) {
263         batch += 1
264       } else {
265         flush()
266       }
267     }
268 
269     override protected def update(
270         feature: RedisWritableFeature,
271         values: Array[RowKeyValue[_]],
272         previous: RedisWritableFeature,
273         previousValues: Array[RowKeyValue[_]]): Unit = {
274       delete(previous, previousValues)
275       append(feature, values)
276     }
277 
278     override protected def delete(feature: RedisWritableFeature, values: Array[RowKeyValue[_]]): Unit = {
279       i = 0
280       while (i < values.length) {
281         val buffer = deletes(i)
282         values(i) match {
283           case kv: SingleRowKeyValue[_] =>
284             kv.values.foreach(v => buffer.append(ByteArrays.concat(kv.row, v.value)))
285           case kv: MultiRowKeyValue[_] =>
286             kv.rows.foreach(row => kv.values.foreach(v => buffer.append(ByteArrays.concat(row, v.value))))
287         }
288         i += 1
289       }
290 
291       aging.foreach(_.delete(feature))
292 
293       if (batch < batchSize) {
294         batch += 1
295       } else {
296         flush()
297       }
298     }
299 
300     override def flush(): Unit = {
301       i = 0
302       while (i < tables.length) {
303         if (deletes(i).nonEmpty) {
304           try { WithClose(jedis.getResource)(_.zrem(tables(i), deletes(i).toSeq: _*)) } catch {
305             case NonFatal(e) => errors.append(e)
306           }
307           deletes(i).clear()
308         }
309         if (!inserts(i).isEmpty) {
310           try { WithClose(jedis.getResource)(_.zadd(tables(i), inserts(i))) } catch {
311             case NonFatal(e) => errors.append(e)
312           }
313           inserts(i).clear()
314         }
315         i += 1
316       }
317       batch = 0
318 
319       try { aging.foreach(_.flush()) } catch {
320         case NonFatal(e) => errors.append(e)
321       }
322 
323       throwErrors()
324     }
325 
326     override def close(): Unit = {
327       try { flush() } catch {
328         case NonFatal(e) => errors.append(e)
329       }
330       try { aging.foreach(_.close()) } catch {
331         case NonFatal(e) => errors.append(e)
332       }
333       throwErrors()
334     }
335 
336     private def throwErrors(): Unit = {
337       if (errors.nonEmpty) {
338         val error = errors.head
339         errors.tail.foreach(error.addSuppressed)
340         errors.clear()
341         throw error
342       }
343     }
344   }
