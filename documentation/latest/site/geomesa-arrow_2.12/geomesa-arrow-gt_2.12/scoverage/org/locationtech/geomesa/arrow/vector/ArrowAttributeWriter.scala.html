66   /**
67     * Writer for feature ID. The return FeatureWriter expects to be passed the entire SimpleFeature, not
68     * just the feature ID string (this is to support cached UUIDs).
69     *
70     * @param vector simple feature vector
71     * @param encoding actually write the feature ids, or omit them, in which case the writer is a no-op
72     * @return feature ID writer
73     */
74   def id(sft: SimpleFeatureType, encoding: SimpleFeatureEncoding, vector: StructVector): ArrowAttributeWriter = {
75     val name = SimpleFeatureVector.FeatureIdField
76     encoding.fids match {
77       case None => ArrowNoFidWriter
78       case Some(Encoding.Min) if sft.isUuid => new ArrowFeatureIdMinimalUuidWriter(name, VectorFactory(vector))
79       case Some(Encoding.Max) if sft.isUuid => new ArrowFeatureIdUuidWriter(name, VectorFactory(vector))
80       case Some(Encoding.Min) => new ArrowFeatureIdMinimalStringWriter(name, VectorFactory(vector))
81       case Some(Encoding.Max) => new ArrowFeatureIdStringWriter(name, VectorFactory(vector))
82     }
83   }

165   /**
166    * Low-level method to create a single attribute writer
167    *
168    * @param name attribute name
169    * @param bindings object bindings, the attribute type plus any subtypes (e.g. for lists or maps)
170    * @param dictionary the dictionary for the attribute, if any
171    * @param metadata vector metadata encoded in the field - generally the encoded attribute descriptor
172    * @param encoding encoding options
173    * @param factory parent vector or allocator
174    * @return
175    */
176   def apply(
177       name: String,
178       bindings: Seq[ObjectType],
179       dictionary: Option[ArrowDictionary],
180       metadata: Map[String, String],
181       encoding: SimpleFeatureEncoding,
182       factory: VectorFactory): ArrowAttributeWriter = {
183     dictionary match {
184       case Some(dict) =>
185         ArrowAttributeWriter.dictionary(name, dict, bindings.head == ObjectType.LIST, metadata, factory)
186 
187       case None =>
188         bindings.head match {
189           case ObjectType.STRING   => new ArrowStringWriter(name, metadata, factory)
190           case ObjectType.DATE     => date(name, encoding.date, metadata, factory)
191           case ObjectType.INT      => new ArrowIntWriter(name, metadata, factory)
192           case ObjectType.LONG     => new ArrowLongWriter(name, metadata, factory)
193           case ObjectType.FLOAT    => new ArrowFloatWriter(name, metadata, factory)
194           case ObjectType.DOUBLE   => new ArrowDoubleWriter(name, metadata, factory)
195           case ObjectType.GEOMETRY => geometry(name, bindings(1), encoding, metadata, factory)
196           case ObjectType.BOOLEAN  => new ArrowBooleanWriter(name, metadata, factory)
197           case ObjectType.LIST     => new ArrowListWriter(name, bindings(1), encoding, metadata, factory)
198           case ObjectType.MAP      => new ArrowMapWriter(name, bindings(1), bindings(2), encoding, metadata, factory)
199           case ObjectType.BYTES    => new ArrowBytesWriter(name, metadata, factory)
200           case ObjectType.UUID     => new ArrowUuidWriter(name, metadata, factory)
201           case _ => throw new IllegalArgumentException(s"Unexpected object type ${bindings.head}")
202         }
203     }
204   }

206   private def dictionary(
207       name: String,
208       dictionary: ArrowDictionary,
209       isList: Boolean,
210       metadata: Map[String, String],
211       factory: VectorFactory): ArrowAttributeWriter = {
212     (dictionary.encoding.getIndexType.getBitWidth, isList) match {
213       case (8,  false) => new ArrowDictionaryByteWriter(name, dictionary, metadata, factory)
214       case (16, false) => new ArrowDictionaryShortWriter(name, dictionary, metadata, factory)
215       case (32, false) => new ArrowDictionaryIntWriter(name, dictionary, metadata, factory)
216       case (8,  true)  => new ArrowListDictionaryByteWriter(name, dictionary, metadata, factory)
217       case (16, true)  => new ArrowListDictionaryShortWriter(name, dictionary, metadata, factory)
218       case (32, true)  => new ArrowListDictionaryIntWriter(name, dictionary, metadata, factory)
219       case (w, _)      => throw new IllegalArgumentException(s"Unsupported dictionary encoding width: $w")
220     }
221   }

255       case (ObjectType.MULTILINESTRING, Encoding.Min, FromAllocator(c)) => new MultiLineStringFloatVector(name, c, m)

257       case (ObjectType.MULTILINESTRING, Encoding.Max, FromAllocator(c)) => new MultiLineStringVector(name, c, m)
258       case (ObjectType.MULTIPOLYGON, Encoding.Min, FromStruct(c))       => new MultiPolygonFloatVector(name, c, m)
259       case (ObjectType.MULTIPOLYGON, Encoding.Min, FromAllocator(c))    => new MultiPolygonFloatVector(name, c, m)
260       case (ObjectType.MULTIPOLYGON, Encoding.Max, FromStruct(c))       => new MultiPolygonVector(name, c, m)
261       case (ObjectType.MULTIPOLYGON, Encoding.Max, FromAllocator(c))    => new MultiPolygonVector(name, c, m)
262       case (ObjectType.MULTIPOINT, Encoding.Min, FromStruct(c))         => new MultiPointFloatVector(name, c, m)
263       case (ObjectType.MULTIPOINT, Encoding.Min, FromAllocator(c))      => new MultiPointFloatVector(name, c, m)

303   /**
304    * Converts a list value into a list of dictionary bytes and writes it
305    *
306    * @param name attribute/field name
307    * @param dictionary dictionary values
308    * @param metadata field metadata
309    * @param factory vector factory
310    */
311   class ArrowListDictionaryByteWriter(
312       val name: String,
313       val dictionary: ArrowDictionary,
314       metadata: Map[String, String],
315       factory: VectorFactory
316     ) extends ArrowDictionaryWriter {
317 
318     override val vector: ListVector = factory.apply(name, MinorType.LIST, metadata)
319 
320     private val inner: TinyIntVector =
321       FromList(vector).apply(null, MinorType.TINYINT, dictionary.encoding, Map.empty)
322 
323     override def apply(i: Int, value: AnyRef): Unit = {
324       if (vector.getLastSet >= i) {
325         vector.setLastSet(i - 1)
326       }
327       val start = vector.startNewValue(i)
328       // note: null gets converted to empty list
329       if (value == null) {
330         vector.endValue(i, 0)
331       } else {
332         val list = value.asInstanceOf[java.util.List[AnyRef]]
333         var offset = 0
334         while (offset < list.size()) {
335           // note: nulls get encoded in the dictionary
336           inner.setSafe(start + offset, dictionary.index(list.get(offset)).toByte)
337           offset += 1
338         }
339         vector.endValue(i, offset)
340       }
341     }
342   }

491   class ArrowUuidWriter(val name: String, metadata: Map[String, String], factory: VectorFactory)
492     extends ArrowAttributeWriter {
493     val fieldType: FieldType = new FieldType(true, new ArrowType.FixedSizeList(2), null, metadata.asJava)
494     override val vector: FixedSizeListVector = factory.apply(name, fieldType)
495 
496     private val bits = {
497       val result = vector.addOrGetVector[BigIntVector](FieldType.nullable(MinorType.BIGINT.getType))
498       if (result.isCreated) {
499         result.getVector.allocateNew()
500       }
501       result.getVector
502     }
503 
504     override def apply(i: Int, value: AnyRef): Unit = {
505       if (value == null) {
506         vector.setNull(i) // note: calls .setSafe internally
507       } else {
508         val uuid = value.asInstanceOf[UUID]
509         val (msb, lsb) = (uuid.getMostSignificantBits, uuid.getLeastSignificantBits)
510         vector.setNotNull(i)
511         bits.setSafe(i * 2, msb)
512         bits.setSafe(i * 2 + 1, lsb)
513       }
514     }
515   }
