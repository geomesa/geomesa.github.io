30 /**
31   * Streams features from an input stream - note that a feature may not be valid after a call to `.next()`,
32   * as the underlying data may be reclaimed
33   *
34   * @param is input stream
35   */
36 class StreamingSimpleFeatureArrowFileReader(is: InputStream) extends SimpleFeatureArrowFileReader  {
37 
38   require(is.available() > 0, "Empty input stream")
39 
40   private val allocator = ArrowAllocator("streaming-file-reader")
41 
42   // reader for the first logical 'file'
43   private val reader = Try(new StreamingSingleFileReader(is, allocator))
44 
45   override def sft: SimpleFeatureType = reader.get.sft
46   override def dictionaries: Map[String, ArrowDictionary] = reader.get.dictionaries
47 
48   override def vectors: Seq[SimpleFeatureVector] = throw new UnsupportedOperationException()
49 
50   override def features(filter: Filter): CloseableIterator[ArrowSimpleFeature] = {
51     // if the read failed, throw an exception up front
52     val reader = this.reader match {
53       case Success(r) => r
54       case Failure(e) => throw e
55     }
56     new StreamingFeatureIterator(is, reader, filter, allocator)
57   }
58 
59   override def close(): Unit = CloseWithLogging.raise(reader.toOption.toSeq ++ Seq(allocator))
60 }

62 object StreamingSimpleFeatureArrowFileReader {
63 
64   import scala.collection.JavaConverters._
65 
66   /**
67    * Creates an iterator from an arrow file input stream
68    *
69    * @param is input stream
70    * @param starter first file reader, created from the input stream, for dictionaries and metadata
71    * @param filter filter for feature reading
72    * @param allocator allocator
73    */
74   private class StreamingFeatureIterator(is: InputStream, starter: StreamingSingleFileReader, filter: Filter, allocator: BufferAllocator)
75       extends CloseableIterator[ArrowSimpleFeature] {
76 
77     // note: we track all the readers and close at the end to avoid closing the input stream prematurely
78     private val opened = ArrayBuffer.empty[AutoCloseable]
79     private val skip = new SkipIndicator
80 
81     private var done = false
82     private var batch: Iterator[ArrowSimpleFeature] = starter.features(filter, skip)
83 
84     @tailrec
85     override final def hasNext: Boolean = {
86       if (done) {
87         false
88       } else if (batch.hasNext) {
89         true
90       } else if (!skip.skip && is.available() > 0) {
91         // new logical file
92         val nextReader = new StreamingSingleFileReader(is, allocator)
93         opened += nextReader
94         batch = nextReader.features(filter, skip)
95         hasNext
96       } else {
97         done = true
98         false
99       }
100     }
101 
102     override def next(): ArrowSimpleFeature = batch.next()
103 
104     override def close(): Unit = CloseWithLogging.raise(opened)
105   }
106 
107   /**
108    * Reads a single logical arrow 'file' from the stream, which may contain multiple record batches
109    */
110   private class StreamingSingleFileReader(is: InputStream, allocator: BufferAllocator) extends Closeable {
111 
112     private val reader = new ArrowStreamReader(is, allocator)
113     private val root = reader.getVectorSchemaRoot
114     require(root.getFieldVectors.size() == 1 && root.getFieldVectors.get(0).isInstanceOf[StructVector], "Invalid file")
115     private val underlying = root.getFieldVectors.get(0).asInstanceOf[StructVector]
116     private var done = !reader.loadNextBatch() // load the first batch so we get any dictionaries
117 
118     val (sft, encoding) = SimpleFeatureVector.getFeatureType(underlying)
119 
120     // load any dictionaries into memory
121     val dictionaries: Map[String, ArrowDictionary] =
122       SimpleFeatureArrowFileReader.loadDictionaries(underlying.getField.getChildren.asScala.toSeq, reader, encoding)
123     private val vector = new SimpleFeatureVector(sft, underlying, dictionaries, encoding, None)
124 
125     private def metadata: java.util.Map[String, String] = reader.getVectorSchemaRoot.getSchema.getCustomMetadata
126 
127     // iterator of simple features read from the input stream
128     def features(filter: Filter, skip: SkipIndicator): Iterator[ArrowSimpleFeature] = {
129       if (done) { Iterator.empty } else {
130         val nextBatch = SimpleFeatureArrowFileReader.features(sft, filter, skip, getSortFromMetadata(metadata), dictionaries)
131 
132         new Iterator[ArrowSimpleFeature] {
133           private var batch: Iterator[ArrowSimpleFeature] = nextBatch(vector)
134 
135           @scala.annotation.tailrec
136           override def hasNext: Boolean = {
137             if (done) {
138               false
139             } else if (batch.hasNext) {
140               true
141             } else if (!skip.skip && reader.loadNextBatch()) {
142               batch = nextBatch(vector)
143               hasNext
144             } else {
145               done = true
146               false
147             }
148           }
149 
150           override def next(): ArrowSimpleFeature = batch.next()
151         }
152       }
153     }
154 
155     override def close(): Unit = CloseWithLogging.raise(dictionaries.values ++ Seq(reader, vector))
156   }
