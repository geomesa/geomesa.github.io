27 class ZookeeperOffsetManager(zookeepers: String, namespace: String = "geomesa") extends OffsetManager {
28 
29   import ZookeeperOffsetManager.offsetsPath
30 
31   private val client =
32     CuratorFrameworkFactory.builder()
33       .connectString(zookeepers)
34       .retryPolicy(new ExponentialBackoffRetry(1000, 3))
35       .dontUseContainerParents()
36       .namespace(namespace)
37       .build()
38   client.start()
39 
40   private val listeners = scala.collection.mutable.Map.empty[String, CuratorOffsetListener]
41 
42   override def getOffset(topic: String, partition: Int): Long = {
43     val path = ZookeeperOffsetManager.offsetsPath(topic, partition)
44     if (client.checkExists().forPath(path) == null) { -1L } else {
45       ZookeeperOffsetManager.deserializeOffsets(client.getData.forPath(path))
46     }
47   }
48 
49   override def setOffset(topic: String, partition: Int, offset: Long): Unit = {
50     val path = ZookeeperOffsetManager.offsetsPath(topic, partition)
51     if (client.checkExists().forPath(path) == null) {
52       client.create().creatingParentsIfNeeded().forPath(path)
53     }
54     client.setData().forPath(path, ZookeeperOffsetManager.serializeOffset(offset))
55   }
56 
57   override def deleteOffsets(topic: String): Unit = {
58     val path = ZookeeperOffsetManager.offsetsPath(topic)
59     if (client.checkExists().forPath(path) != null) {
60       client.delete().deletingChildrenIfNeeded().forPath(path)
61     }
62   }
63 
64   override def addOffsetListener(topic: String, listener: OffsetListener): Unit = synchronized {
65     listeners.getOrElseUpdate(topic, new CuratorOffsetListener(client, offsetsPath(topic))).addListener(listener)
66   }
67 
68   override def removeOffsetListener(topic: String, listener: OffsetListener): Unit = synchronized {
69     listeners.get(topic).foreach(_.removeListener(listener))
70   }
71 
72   override def close(): Unit = synchronized {
73     CloseWithLogging(listeners.values)
74     listeners.clear()
75     CloseWithLogging(client)
76   }
77 
78   override protected def acquireDistributedLock(path: String): Closeable =
79     acquireLock(path, lock => { lock.acquire(); true })
80 
81   override protected def acquireDistributedLock(path: String, timeOut: Long): Option[Closeable] =
82     Option(acquireLock(path, lock => lock.acquire(timeOut, TimeUnit.MILLISECONDS)))
83 
84   private def acquireLock(path: String, acquire: InterProcessSemaphoreMutex => Boolean): Closeable = {
85     val lock = new InterProcessSemaphoreMutex(client, s"/$path/locks")
86     if (acquire(lock)) {
87       () => lock.release()
88     } else {
89       null
90     }
91   }

94 object ZookeeperOffsetManager {
95 
96   def serializeOffset(offset: Long): Array[Byte] = offset.toString.getBytes(StandardCharsets.UTF_8)
97   def deserializeOffsets(bytes: Array[Byte]): Long = new String(bytes, StandardCharsets.UTF_8).toLong
98 
99   private def offsetsPath(topic: String): String = s"/$topic/offsets"
100   private def offsetsPath(topic: String, partition: Int): String = s"${offsetsPath(topic)}/$partition"
101   private def partitionFromPath(path: String): Int = path.substring(path.lastIndexOf("/") + 1).toInt
102 
103   private class CuratorOffsetListener(client: CuratorFramework, path: String)
104       extends PathChildrenCacheListener with Closeable with LazyLogging {
105 
106     private val executor = Executors.newCachedThreadPool()
107 
108     private val listeners = scala.collection.mutable.Set.empty[OffsetListener]
109 
110     private var cache: PathChildrenCache = _
111 
112     def addListener(listener: OffsetListener): Unit = {
113       closeCache()
114       listeners += listener
115       cache = new PathChildrenCache(client, path, true) {
116         // override ensure path to avoid using containers, which aren't supported in zk 3.4
117         override protected def ensurePath(): Unit = {
118           if (client.checkExists().forPath(path) == null) {
119             client.create().creatingParentsIfNeeded().forPath(path)
120           }
121         }
122       }
123       cache.getListenable.addListener(this)
124       cache.start()
125     }
126 
127     def removeListener(listener: OffsetListener): Unit = {
128       listeners -= listener
129       if (listeners.isEmpty) {
130         closeCache()
131       }
132     }
133 
134     override def childEvent(client: CuratorFramework, event: PathChildrenCacheEvent): Unit = {
135       import PathChildrenCacheEvent.Type.{CHILD_ADDED, CHILD_UPDATED}
136       try {
137         val eventPath = Option(event.getData).map(_.getPath).getOrElse("")
138         if ((event.getType == CHILD_ADDED || event.getType == CHILD_UPDATED) && eventPath.startsWith(path)) {
139           logger.trace(s"ZK event triggered for: $eventPath")
140           val partition = partitionFromPath(eventPath)
141           val data = event.getData.getData
142           Try(ZookeeperOffsetManager.deserializeOffsets(data)) match {
143             case Success(offset) =>
144               listeners.foreach { listener =>
145                 executor.execute(() => {
146                   try { listener.offsetChanged(partition, offset) } catch {
147                     case NonFatal(e) => logger.warn("Error calling offset listener", e)
148                   }
149                 })
150               }
151 
152             case Failure(e) =>
153               // for some reason we get IP addresses sometimes... seems to be something curator is doing.
154               // we generally get the correct data immediately afterwards, so it appears to be harmless
155               logger.warn(
156                 s"Error deserializing offset data " +
157                   s"'${Try(new String(data, StandardCharsets.UTF_8)).getOrElse(ByteArrays.toHex(data))}': ${e.getClass.getName}")
158           }
159         }
160       } catch {
161         case NonFatal(e) => logger.warn("Error handling ZK event", e)
162       }
163     }
164 
165     override def close(): Unit = {
166       executor.shutdown()
167       closeCache()
168     }
169 
170     private def closeCache(): Unit = synchronized {
171       if (cache != null) {
172         cache.getListenable.removeListener(this)
173         CloseWithLogging(cache)
174         cache = null
175       }
176     }
177   }
