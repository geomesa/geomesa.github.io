104   override def offsetChanged(partition: Int, offset: Long): Unit = {
105     if (offsets.length <= partition) {
106       logger.debug(s"Offsets changed for [$topic:$partition]: (unassigned) -> $offset")
107       ensurePartition(partition, offset)

113     logger.debug(s"Offsets changed for [$topic:$partition]: $last -> $offset")
114     while (last < offset && !offsets(partition).compareAndSet(last, offset)) {
115       last = offsets(partition).get

164   private class Persistence(expiryMillis: Long, batchSize: Int) extends Closeable {
165 
166     private val frequency = KafkaFeatureCache.PersistInterval.toMillis.get
167     private val executor = ExitingExecutor(new ScheduledThreadPoolExecutor(1))
168     private val schedule = executor.scheduleWithFixedDelay(() => persist(), frequency, frequency, TimeUnit.MILLISECONDS)

170     /**
171      * Persist any expired features
172      *
173      * @return list of last persisted offsets per partition
174      */
175     def persist(): Seq[PartitionOffset] = {
176       logger.debug(s"Running persistence for [$topic]")
177       // lock per-partition to allow for multiple write threads
178       // randomly access the partitions to avoid contention if multiple data stores are all on the same schedule
179       Random.shuffle(offsets.indices.toList).flatMap { partition =>
180         // if we don't get the lock just try again next run
181         logger.trace(s"Acquiring lock for [$topic:$partition]")
182         offsetManager.acquireLock(topic, partition, lockTimeout) match {
183           case None =>
184             logger.trace(s"Could not acquire lock for [$topic:$partition] within ${lockTimeout}ms")
185             None
186           case Some(lock) =>
187             try {
188               logger.trace(s"Acquired lock for [$topic:$partition]")
189               persist(partition).map(PartitionOffset(partition, _))
190             } finally {
191               lock.close()
192               logger.trace(s"Released lock for [$topic:$partition]")
193             }
194         }
195       }
196     }

216     /**
217      * Persist any expired features that haven't yet been persisted
218      *
219      * @param partition partition to persist
220      * @param lastPersistedOffset result from previous recursive run, if any
221      * @return offset of latest persisted feature
222      */
223     @tailrec
224     private def persist(partition: Int, lastPersistedOffset: Option[Long] = None): Option[Long] = {
225       val expiry = clock.millis() - expiryMillis
226 
227       val lastOffset = offsetManager.getOffset(topic, partition)
228       logger.debug(s"Last persisted offsets for [$topic:$partition]: $lastOffset")
229 
230       var nextOffset = -1L
231       // note: copy to a new collection so that any updates don't affect our persistence here
232       val expired = {
233         val builder = Map.newBuilder[String, FeatureReference]
234         features.asScala.foreach { case (id, f) =>
235           // note: offset > lastOffset ensures we don't operate on features that have already been processed
236           if (f.partition == partition && f.created <= expiry && f.offset > lastOffset) {
237             nextOffset = math.max(nextOffset, f.offset)
238             builder += id -> f
239           }
240         }
241         builder.result()
242       }
243       logger.debug(s"Found ${expired.size} expired entries for [$topic:$partition]")
244 
245       if (expired.isEmpty) {
246         lastPersistedOffset
247       } else {
248         logger.trace(expired.values.map(e => s"offset ${e.offset}: ${e.feature}").mkString("Expired entries:\n\t", "\n\t", ""))
249         val batch =
250           if (expired.size <= batchSize) {
251             expired
252           } else {
253             logger.trace(s"Skipping persistence for ${expired.size - batchSize} features based on batch size of $batchSize")
254             // sort by offset since we track persistence based on offset
255             val sorted = expired.toList.sortBy(_._2.offset).take(batchSize)
256             nextOffset = sorted.last._2.offset
257             sorted.toMap
258           }
259         persist(partition, batch)
260 
261         logger.debug(s"Committing offset [$topic:$partition:$nextOffset]")
262         // this will trigger our listener and cause the feature to be removed from the cache
263         offsetManager.setOffset(topic, partition, nextOffset)
264         if (batch.size < expired.size) {
265           // we skipped some expired features, so run again immediately
266           persist(partition, Some(nextOffset))
267         } else {
268           Some(nextOffset)
269         }
270       }
271     }

273     /**
274      * Persist expired features
275      *
276      * @param partition partition
277      * @param batch expired features
278      */
279     private def persist(partition: Int, batch: Map[String, FeatureReference]): Unit = {
280       // do an update query first
281       val remaining = persistUpdates(partition, batch)
282       // if any weren't updates, add them as inserts
283       if (remaining.nonEmpty) {
284         persistAppends(partition, remaining)
285       }
286     }

288     /**
289      * Attempt to persist expired features through modifying writes
290      *
291      * @param partition partition being persisted
292      * @param batch expired feature
293      * @return any features that were not persisted
294      */
295     private def persistUpdates(partition: Int, batch: Map[String, FeatureReference]): Iterable[FeatureReference] = {
296       logger.debug(s"Attempting persistent updates on ${batch.size} features")
297       val persistedKeys = scala.collection.mutable.Set.empty[String]
298       profile((c: Int, time: Long) => logger.debug(s"Wrote $c updated feature(s) to persistent storage in ${time}ms")) {
299         val filter = ff.id(batch.keys.map(ff.featureId).toSeq: _*)
300         WithClose(ds.getFeatureWriter(sft.getTypeName, filter, Transaction.AUTO_COMMIT)) { writer =>
301           var count = 0
302           while (writer.hasNext) {
303             val next = writer.next()
304             batch.get(next.getID).foreach { p =>
305               logger.trace(s"Persistent store modify [$topic:$partition:${p.offset}] ${p.feature}")
306               persistedKeys += next.getID
307               FeatureUtils.copyToFeature(next, p.feature, useProvidedFid = true)
308               try {
309                 writer.write()
310                 count += 1
311               } catch {
312                 case NonFatal(e) => logger.error(s"Error persisting feature: ${p.feature}", e)
313               }
314             }
315           }
316           count
317         }
318       }
319       batch.collect { case (k, v) if !persistedKeys.contains(k) => v }
320     }

322     /**
323      * Attempt to persist expired features through appending writes
324      *
325      * @param partition partition being persisted
326      * @param batch expired feature
327      */
328     private def persistAppends(partition: Int, batch: Iterable[FeatureReference]): Unit = {
329       logger.debug(s"Attempting persistent appends on ${batch.size} features")
330       profile((c: Int, time: Long) => logger.debug(s"Wrote $c new feature(s) to persistent storage in ${time}ms")) {
331         WithClose(ds.getFeatureWriterAppend(sft.getTypeName, Transaction.AUTO_COMMIT)) { writer =>
332           var count = 0
333           val helper = FeatureWriterHelper(writer, useProvidedFids = true)
334           batch.foreach { p =>
335             logger.trace(s"Persistent store append [$topic:$partition:${p.offset}] ${p.feature}")
336             try {
337               helper.write(p.feature)
338               count += 1
339             } catch {
340               case NonFatal(e) => logger.error(s"Error persisting feature: ${p.feature}", e)
341             }
342           }
343           count
344         }
345       }
346     }
