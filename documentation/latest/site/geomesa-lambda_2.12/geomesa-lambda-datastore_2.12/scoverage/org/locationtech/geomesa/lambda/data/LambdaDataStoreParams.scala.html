118   def parse(params: java.util.Map[String, _], namespace: String): LambdaConfig = {
119     val brokers = BrokersParam.lookup(params)
120     val persistence = if (!PersistParam.lookup(params).booleanValue) { None } else {
121       ExpiryParam.lookupOpt(params).collect { case d: FiniteDuration =>
122         PersistenceConfig(d, BatchSizeParam.lookup(params).intValue())
123       }
124     }
125 
126     val partitions = PartitionsParam.lookup(params).intValue
127     val consumers = ConsumersParam.lookup(params).intValue
128 
129     val bootstrap = Map("bootstrap.servers" -> brokers)
130     val consumerConfig = ConsumerOptsParam.lookupOpt(params).map(_.asScala.toMap).getOrElse(Map.empty) ++ bootstrap
131     val producerConfig = ProducerOptsParam.lookupOpt(params).map(_.asScala.toMap).getOrElse(Map.empty) ++ bootstrap
132 
133     val zk = ZookeepersParam.lookupOpt(params).orElse(Params.Accumulo.ZookeepersParam.lookupOpt(params)).getOrElse {
134       throw new IllegalArgumentException(s"Missing required key: ${ZookeepersParam.key}/${Params.Accumulo.ZookeepersParam.key}")
135     }
136     val zkNamespace = s"gm_lambda_$namespace"
137 
138     val offsetCommitInterval = ConsumerOffsetCommitInterval.lookup(params)
139 
140     LambdaConfig(zk, zkNamespace, producerConfig, consumerConfig, partitions, consumers, persistence, offsetCommitInterval)
141   }
