117   def parse(params: java.util.Map[String, _], namespace: String): LambdaConfig = {
118     val brokers = BrokersParam.lookup(params)
119     val persistence = if (!PersistParam.lookup(params).booleanValue) { None } else {
120       ExpiryParam.lookupOpt(params).collect { case d: FiniteDuration =>
121         PersistenceConfig(d, BatchSizeParam.lookup(params).intValue())
122       }
123     }
124 
125     val partitions = PartitionsParam.lookup(params).intValue
126     val consumers = ConsumersParam.lookup(params).intValue
127 
128     val bootstrap = Map("bootstrap.servers" -> brokers)
129     val consumerConfig = ConsumerOptsParam.lookupOpt(params).map(_.asScala.toMap).getOrElse(Map.empty) ++ bootstrap
130     val producerConfig = ProducerOptsParam.lookupOpt(params).map(_.asScala.toMap).getOrElse(Map.empty) ++ bootstrap
131 
132     val zk = ZookeepersParam.lookup(params)
133     val zkNamespace = s"gm_lambda_$namespace"
134 
135     val offsetCommitInterval = ConsumerOffsetCommitInterval.lookup(params)
136 
137     LambdaConfig(zk, zkNamespace, producerConfig, consumerConfig, partitions, consumers, persistence, offsetCommitInterval)
138   }
