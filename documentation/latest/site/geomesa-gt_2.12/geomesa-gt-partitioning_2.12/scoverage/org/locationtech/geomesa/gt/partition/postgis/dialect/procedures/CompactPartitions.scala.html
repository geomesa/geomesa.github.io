15 /**
16  * Re-sorts a partition table. Can be used for maintenance if time-latent data comes in and degrades performance
17  * of the BRIN index
18  */
19 object CompactPartitions extends SqlProcedure {
20 
21   override def name(info: TypeInfo): FunctionName = FunctionName(s"${info.typeName}_compact_partitions")

23   override protected def createStatements(info: TypeInfo): Seq[String] = Seq(proc(info))

25   private def proc(info: TypeInfo): String = {
26     s"""CREATE OR REPLACE PROCEDURE ${info.schema.quoted}.${name(info).quoted}(for_date timestamp without time zone) LANGUAGE plpgsql AS
27        |  $$BODY$$
28        |    DECLARE
29        |      partition_name text;
30        |      spill_partition text;
31        |      min_dtg timestamp without time zone;         -- min date in our partitioned tables
32        |      partition_start timestamp without time zone; -- start bounds for the partition we're writing
33        |      partition_end timestamp without time zone;   -- end bounds for the partition we're writing
34        |      partition_size int;                          -- size of the partition, in hours
35        |      partition_tablespace text;
36        |      index_tablespace text;                       -- index tablespace
37        |      table_wa_logging text;                       -- wa log options
38        |      unsorted_count bigint;
39        |      pexists boolean;                             -- table exists check
40        |    BEGIN
41        |      IF for_date IS NULL THEN
42        |        RAISE EXCEPTION 'date is null' USING HINT = 'Please use a valid date';
43        |      END IF;
44        |
45        |      SELECT COALESCE(
46        |        (SELECT value::int FROM ${info.schema.quoted}.${UserDataTable.Name.quoted}
47        |          WHERE type_name = ${literal(info.typeName)} AND key = ${literal(SftUserData.IntervalHours.key)}),
48        |          ${SftUserData.IntervalHours.default})
49        |        INTO partition_size;
50        |      partition_name := ${info.tables.mainPartitions.name.asLiteral} || '_' ||
51        |        to_char(${info.schema.quoted}.truncate_to_partition(for_date, partition_size), 'YYYY_MM_DD_HH24');
52        |
53        |      SELECT value FROM ${info.schema.quoted}.${UserDataTable.Name.quoted}
54        |        WHERE type_name = ${literal(info.typeName)} AND key = ${literal(SftUserData.MainTableSpace.key)}
55        |        INTO partition_tablespace;
56        |      IF partition_tablespace IS NULL THEN
57        |        index_tablespace := '';
58        |        partition_tablespace := '';
59        |      ELSE
60        |        index_tablespace := ' USING INDEX TABLESPACE '|| quote_ident(partition_tablespace);
61        |        partition_tablespace := ' TABLESPACE ' || quote_ident(partition_tablespace);
62        |      END IF;
63        |
64        |      SELECT COALESCE(
65        |        (SELECT value FROM ${info.schema.quoted}.${UserDataTable.Name.quoted}
66        |          WHERE type_name = ${literal(info.typeName)} AND key = ${literal(SftUserData.WalLogEnabled.key)}),
67        |          '${SftUserData.WalLogEnabled.default}'
68        |        ) INTO table_wa_logging;
69        |      IF table_wa_logging IS NULL OR NOT table_wa_logging::boolean THEN
70        |        table_wa_logging := '';
71        |      ELSE
72        |        table_wa_logging := 'UNLOGGED ';
73        |      END IF;
74        |
75        |      RAISE INFO '% Compacting partition table %', timeofday()::timestamp, partition_name;
76        |      LOCK TABLE ONLY ${info.tables.mainPartitions.name.qualified} IN SHARE UPDATE EXCLUSIVE MODE;
77        |      -- lock the child table to prevent any inserts that would be lost
78        |      EXECUTE 'LOCK TABLE ${info.schema.quoted}.' || quote_ident(partition_name) ||
79        |        ' IN SHARE ROW EXCLUSIVE MODE';
80        |      EXECUTE 'SELECT ${info.cols.dtg.quoted} FROM ${info.schema.quoted}.' || quote_ident(partition_name) ||
81        |        ' LIMIT 1' INTO min_dtg;
82        |      partition_start := ${info.schema.quoted}.truncate_to_partition(min_dtg, partition_size);
83        |      partition_end := partition_start + make_interval(hours => partition_size);
84        |      spill_partition := ${info.tables.spillPartitions.name.asLiteral} || '_' || to_char(partition_start, 'YYYY_MM_DD_HH24');
85        |
86        |      SELECT EXISTS(SELECT FROM pg_tables WHERE schemaname = ${info.schema.asLiteral} AND tablename = spill_partition)
87        |        INTO pexists;
88        |
89        |      -- use "create table as" (vs create then insert) for performance benefits related to WAL skipping
90        |      IF pexists THEN
91        |        -- lock the child table to prevent any inserts that would be lost
92        |        EXECUTE 'LOCK TABLE ${info.schema.quoted}.' || quote_ident(spill_partition) ||
93        |          ' IN SHARE ROW EXCLUSIVE MODE';
94        |        EXECUTE 'CREATE ' || table_wa_logging || 'TABLE ${info.schema.quoted}.' || quote_ident(partition_name || '_tmp_sort') ||
95        |          partition_tablespace || ' AS SELECT * FROM' ||
96        |          ' (SELECT * FROM ${info.schema.quoted}.' || quote_ident(partition_name) ||
97        |          ' UNION ALL SELECT * FROM ${info.schema.quoted}.' || quote_ident(spill_partition) ||
98        |          ') results' ||
99        |          ' ORDER BY _st_sortablehash(${info.cols.geom.quoted})';
100        |        GET DIAGNOSTICS unsorted_count := ROW_COUNT;
101        |      ELSE
102        |        EXECUTE 'CREATE ' || table_wa_logging || 'TABLE ${info.schema.quoted}.' || quote_ident(partition_name || '_tmp_sort') ||
103        |          partition_tablespace || ' AS SELECT * FROM ' || quote_ident(partition_name) ||
104        |          ' ORDER BY _st_sortablehash(${info.cols.geom.quoted})';
105        |        GET DIAGNOSTICS unsorted_count := ROW_COUNT;
106        |      END IF;
107        |
108        |      EXECUTE 'ALTER TABLE ${info.schema.quoted}.' || quote_ident(partition_name || '_tmp_sort') ||
109        |        ' ADD CONSTRAINT ' || quote_ident(partition_name || '_pkey_tmp_sort') ||
110        |        ' PRIMARY KEY (fid, ${info.cols.dtg.quoted})' || index_tablespace;
111        |      -- creating a constraint allows it to be attached to the parent without any additional checks
112        |      EXECUTE 'ALTER TABLE  ${info.schema.quoted}.' || quote_ident(partition_name || '_tmp_sort') ||
113        |        ' ADD CONSTRAINT ' || quote_ident(partition_name || '_constraint_tmp_sort') ||
114        |        ' CHECK ( ${info.cols.dtg.quoted} >= ' || quote_literal(partition_start) ||
115        |        ' AND ${info.cols.dtg.quoted} < ' || quote_literal(partition_end) || ' )';
116        |
117        |      -- create indices before attaching to minimize time to attach, copied from PartitionTables code
118        |      EXECUTE 'CREATE INDEX IF NOT EXISTS ' || quote_ident(partition_name || '_${info.cols.geom.raw}_tmp_sort') ||
119        |        ' ON ${info.schema.quoted}.' || quote_ident(partition_name || '_tmp_sort') ||
120        |        ' USING BRIN(${info.cols.geom.quoted})' || partition_tablespace;
121        |      EXECUTE 'CREATE INDEX IF NOT EXISTS ' || quote_ident(partition_name || '_${info.cols.dtg.raw}_tmp_sort') ||
122        |        ' ON ${info.schema.quoted}.' || quote_ident(partition_name || '_tmp_sort') ||
123        |        ' (${info.cols.dtg.quoted})' || partition_tablespace;
124        |${info.cols.indexed.map { col =>
125     s"""      EXECUTE 'CREATE INDEX IF NOT EXISTS ' || quote_ident(partition_name || '_${col.raw}_tmp_sort') ||
126        |        ' ON ${info.schema.quoted}.' || quote_ident(partition_name || '_tmp_sort') ||
127        |        ' (${col.quoted})' || partition_tablespace;""".stripMargin}.mkString("\n")}
128        |
129        |      RAISE INFO '% Dropping old partition table (queries will be blocked) %', timeofday()::timestamp, partition_name;
130        |      EXECUTE 'DROP TABLE IF EXISTS ${info.schema.quoted}.' || quote_ident(partition_name);
131        |      IF pexists THEN
132        |        EXECUTE 'DROP TABLE IF EXISTS ${info.schema.quoted}.' || quote_ident(spill_partition);
133        |      END IF;
134        |
135        |      RAISE INFO '% Renaming newly sorted partition table %', timeofday()::timestamp, partition_name;
136        |      EXECUTE 'ALTER TABLE ${info.schema.quoted}.' || quote_ident(partition_name || '_tmp_sort') ||
137        |        ' RENAME TO ' || quote_ident(partition_name);
138        |      EXECUTE 'ALTER TABLE ${info.schema.quoted}.' || quote_ident(partition_name) ||
139        |        ' RENAME CONSTRAINT ' || quote_ident(partition_name || '_pkey_tmp_sort') ||
140        |        ' TO ' || quote_ident(partition_name || '_pkey');
141        |      EXECUTE 'ALTER INDEX ' || quote_ident(partition_name || '_${info.cols.geom.raw}_tmp_sort') ||
142        |        ' RENAME TO ' || quote_ident(partition_name || '_${info.cols.geom.raw}');
143        |      EXECUTE 'ALTER INDEX ' || quote_ident(partition_name || '_${info.cols.dtg.raw}_tmp_sort') ||
144        |        ' RENAME TO ' || quote_ident(partition_name || '_${info.cols.dtg.raw}');
145        |${info.cols.indexed.map { col =>
146     s"""      EXECUTE 'ALTER INDEX ' || quote_ident(partition_name || '_${col.raw}_tmp_sort') ||
147        |        ' RENAME TO ' || quote_ident(partition_name || '_${col.raw}');""".stripMargin}.mkString("\n")}
148        |
149        |      RAISE INFO '% Attaching newly sorted partition table %', timeofday()::timestamp, partition_name;
150        |      EXECUTE 'ALTER TABLE ${info.tables.mainPartitions.name.qualified}' ||
151        |        ' ATTACH PARTITION ${info.schema.quoted}.' || quote_ident(partition_name) ||
152        |        ' FOR VALUES FROM (' || quote_literal(partition_start) || ') TO (' ||
153        |        quote_literal(partition_end) || ' );';
154        |      RAISE INFO '% Done compacting partition table %', timeofday()::timestamp, partition_name;
155        |
156        |      -- now that we've attached the table we can drop the redundant constraint
157        |      RAISE INFO '% Dropping constraint for partition table %', timeofday()::timestamp, partition_name;
158        |      EXECUTE 'ALTER TABLE ${info.schema.quoted}.' || quote_ident(partition_name) ||
159        |        ' DROP CONSTRAINT ' || quote_ident(partition_name || '_constraint_tmp_sort');
160        |
161        |      -- mark the partition to be analyzed in a separate thread
162        |      INSERT INTO ${info.tables.analyzeQueue.name.qualified}(partition_name, enqueued)
163        |        VALUES (partition_name, now());
164        |
165        |      -- commit to release our locks
166        |      COMMIT;
167        |      RAISE INFO '% Done compacting % rows in partition %', timeofday()::timestamp, unsorted_count, partition_name;
168        |
169        |    END;
170        |  $$BODY$$;
171        |""".stripMargin
172   }
