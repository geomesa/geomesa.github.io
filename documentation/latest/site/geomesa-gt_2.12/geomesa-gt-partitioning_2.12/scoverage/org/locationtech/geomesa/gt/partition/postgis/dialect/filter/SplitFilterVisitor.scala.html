19 /**
20  * Filter visitor we use for processing filters in the dialect's `splitFilter`method.
21  *
22  * Currently the filter does the following:
23  * <ul>
24  *   <li>replaces constant functions with their literal values</li>
25  *   <li>removes "whole-world" geometry filters</li>
26  * </ul>
27  */
28 class SplitFilterVisitor(removeWholeWorldFilters: Boolean = true) extends DuplicatingFilterVisitor {
29 
30   // replaces 'constant' functions with their literal values
31   override def visit(expression: Function, extraData: Any): AnyRef = {
32     val extractor = new FilterAttributeExtractor()
33     expression.accept(extractor, null)
34     if (extractor.isConstantExpression) {
35       val literal = expression.evaluate(null)
36       SplitFilterVisitor.ff.literal(literal)
37     } else {
38       super.visit(expression, extraData)
39     }
40   }
41 
42   override def visit(f: DWithin, data: AnyRef): AnyRef =
43     if (removeWholeWorldFilters && FilterHelper.isFilterWholeWorld(f)) { Filter.INCLUDE } else { super.visit(f, data) }
44 
45   override def visit(f: BBOX, data: AnyRef): AnyRef =
46     if (removeWholeWorldFilters && FilterHelper.isFilterWholeWorld(f)) { Filter.INCLUDE } else { super.visit(f, data) }
47 
48   override def visit(f: Within, data: AnyRef): AnyRef =
49     if (removeWholeWorldFilters && FilterHelper.isFilterWholeWorld(f)) { Filter.INCLUDE } else { super.visit(f, data) }
50 
51   override def visit(f: Intersects, data: AnyRef): AnyRef =
52     if (removeWholeWorldFilters && FilterHelper.isFilterWholeWorld(f)) { Filter.INCLUDE } else { super.visit(f, data) }
53 
54   override def visit(f: Overlaps, data: AnyRef): AnyRef =
55     if (removeWholeWorldFilters && FilterHelper.isFilterWholeWorld(f)) { Filter.INCLUDE } else { super.visit(f, data) }
56 
57   override def visit(f: Contains, data: AnyRef): AnyRef =
58     if (removeWholeWorldFilters && FilterHelper.isFilterWholeWorld(f)) { Filter.INCLUDE } else { super.visit(f, data) }
59 
60   override def visit(f: Or, data: AnyRef): AnyRef = {
61     val children = new java.util.ArrayList[Filter](f.getChildren.size)
62     var i = 0
63     while (i < f.getChildren.size) {
64       val child = f.getChildren.get(i).accept(this, data).asInstanceOf[Filter]
65       if (child == Filter.INCLUDE) {
66         // INCLUDE OR foo == INCLUDE
67         return Filter.INCLUDE
68       } else if (child != Filter.EXCLUDE) {
69         // EXCLUDE OR foo == foo
70         children.add(child)
71       }
72       i += 1
73     }
74 
75     children.size() match {
76       case 0 => Filter.EXCLUDE
77       case 1 => children.get(0)
78       case _ => getFactory(data).or(children)
79     }
80   }
81 
82   override def visit(f: And, data: AnyRef): AnyRef = {
83     val children = new java.util.ArrayList[Filter](f.getChildren.size)
84     var i = 0
85     while (i < f.getChildren.size) {
86       val child = f.getChildren.get(i).accept(this, data).asInstanceOf[Filter]
87       if (child == Filter.EXCLUDE) {
88         // EXCLUDE AND foo == EXCLUDE
89         return Filter.EXCLUDE
90       } else if (child != Filter.INCLUDE) {
91         // INCLUDE AND foo == foo
92         children.add(child)
93       }
94       i += 1
95     }
96     children.size() match {
97       case 0 => Filter.INCLUDE
98       case 1 => children.get(0)
99       case _ => getFactory(data).and(children)
100     }
101   }
102 }
