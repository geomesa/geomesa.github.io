65   override def setValue(
66       value: AnyRef,
67       binding: Class[_],
68       att: AttributeDescriptor,
69       ps: PreparedStatement,
70       column: Int,
71       cx: Connection): Unit = {
72     // json columns are string type in geotools, but we have to use setObject or else we get a binding error
73     if (binding == classOf[String] && jsonColumns.get(new PreparedStatementKey(ps, column))) {
74       ps.setObject(column, value, Types.OTHER)
75     } else if (binding.isArray || binding == classOf[java.util.List[_]]) {
76       // handle bug in jdbc store not calling setArrayValue in update statements
77       value match {
78         case null =>
79           ps.setNull(column, Types.ARRAY)
80 
81         case list: java.util.Collection[_] =>
82           if (list.isEmpty) {
83             ps.setNull(column, Types.ARRAY)
84           } else {
85             setArrayValue(list.toArray(), att, ps, column, cx)
86           }
87 
88         case array: Array[_] =>
89           if (array.isEmpty) {
90             ps.setNull(column, Types.ARRAY)
91           } else {
92             setArrayValue(array, att, ps, column, cx)
93           }
94 
95         case singleton =>
96           setArrayValue(Array(singleton), att, ps, column, cx)
97       }
98     } else {
99       super.setValue(value, binding, att, ps, column, cx)
100     }
101   }

187     override def setFeatureType(featureType: SimpleFeatureType): Unit = {
188       // convert List-type attributes to Array-types so that prepared statement bindings work correctly
189       if (featureType.getAttributeDescriptors.asScala.exists(_.getType.getBinding == classOf[java.util.List[_]])) {
190         val builder = new SimpleFeatureTypeBuilder() {
191           override def init(`type`: SimpleFeatureType): Unit = {
192             super.init(`type`)
193             attributes().clear()
194           }
195         }
196         builder.init(featureType)
197         featureType.getAttributeDescriptors.asScala.foreach { descriptor =>
198           val ab = new AttributeTypeBuilder(builder.getFeatureTypeFactory)
199           ab.init(descriptor)
200           if (descriptor.getType.getBinding == classOf[java.util.List[_]]) {
201             ab.setBinding(java.lang.reflect.Array.newInstance(Option(descriptor.getListType()).getOrElse(classOf[String]), 0).getClass)
202           }
203           builder.add(ab.buildDescriptor(descriptor.getLocalName))
204         }
205         this.featureType = builder.buildFeatureType()
206         this.featureType.getUserData.putAll(featureType.getUserData)
207       } else {
208         this.featureType = featureType
209       }
210     }

212     // note: this would be a cleaner solution, but it doesn't get invoked due to explicit calls to
213     // super.getExpressionType in PostgisPSFilterToSql :/
214     override def getExpressionType(expression: Expression): Class[_] = {
215       val result = Option(expression).collect { case p: PropertyName => p }.flatMap { p =>
216         Option(p.evaluate(featureType).asInstanceOf[AttributeDescriptor]).map { descriptor =>
217           val binding = descriptor.getType.getBinding
218           if (binding == classOf[java.util.List[_]]) {
219             val listType = descriptor.getListType()
220             if (listType == null) {
221               classOf[Array[String]]
222             } else {
223               java.lang.reflect.Array.newInstance(listType, 0).getClass
224             }
225           } else {
226             binding
227           }
228         }
229       }
230 
231       result.getOrElse(super.getExpressionType(expression))
232     }

234     override def visit(filter: Or, extraData: AnyRef): AnyRef = {
235       // for array-types, skip the super-class implementation, which merges ORs into INs, as it breaks array OR queries
236       // for other types, keep the super handling as INs may be more efficient that ORs
237       if (FilterHelper.propertyNames(filter).flatMap(name => Option(featureType.getDescriptor(name))).exists(_.getType.getBinding.isArray)) {
238         visit(filter.asInstanceOf[BinaryLogicOperator], "OR")
239       } else {
240         super.visit(filter, extraData)
241       }
242     }
