50   override def setValue(
51       value: AnyRef,
52       binding: Class[_],
53       att: AttributeDescriptor,
54       ps: PreparedStatement,
55       column: Int,
56       cx: Connection): Unit = {
57     // json columns are string type in geotools, but we have to use setObject or else we get a binding error
58     if (binding == classOf[String] && jsonColumns.get(new PreparedStatementKey(ps, column))) {
59       ps.setObject(column, value, Types.OTHER)
60     } else if (binding == classOf[java.util.List[_]]) {
61       // handle bug in jdbc store not calling setArrayValue in update statements
62       value match {
63         case null =>
64           ps.setNull(column, Types.ARRAY)
65 
66         case list: java.util.Collection[_] =>
67           if (list.isEmpty) {
68             ps.setNull(column, Types.ARRAY)
69           } else {
70             setArray(list.toArray(), ps, column, cx)
71           }
72 
73         case array: Array[_] =>
74           if (array.isEmpty) {
75             ps.setNull(column, Types.ARRAY)
76           } else {
77             setArray(array, ps, column, cx)
78           }
79 
80         case _ =>
81           // this will almost certainly fail...
82           super.setValue(value, binding, att, ps, column, cx)
83       }
84     } else {
85       super.setValue(value, binding, att, ps, column, cx)
86     }
87   }

124   // based on setArrayValue, but we don't have the attribute descriptor to use
125   private def setArray(array: Array[_], ps: PreparedStatement, column: Int, cx: Connection): Unit = {
126     val componentType = array(0).getClass
127     val sqlType = dataStore.getSqlTypeNameToClassMappings.asScala.collectFirst { case (k, v) if v == componentType => k }
128     val componentTypeName = sqlType.getOrElse {
129       throw new java.sql.SQLException(s"Failed to find a SQL type for $componentType")
130     }
131     ps.setArray(column, super.convertToArray(array, componentTypeName, componentType, cx))
132   }
