20   override val name = "env"
21   override val params = new EnvironmentParameters()

23   override def execute(): Unit = {
24     import scala.collection.JavaConverters._
25 
26     if (params.sfts == null && params.converters == null && !params.listSfts &&
27       !params.listConverters && !params.describeSfts && !params.describeConverters) {
28       Command.output.info("No flags given; displaying list of SFTs and Converters.")
29       Command.output.info(s"Use 'help $name' to see complete command options.")
30 
31       listSftsNames()
32       listConverterNames()
33     } else {
34       if (params.listSfts) {
35         listSftsNames()
36       }
37       if (params.listConverters){
38         listConverterNames()
39       }
40       if (params.describeSfts) {
41         listSfts()
42       }
43       if (params.describeConverters) {
44         listConverters()
45       }
46       if (params.sfts != null) {
47         listSfts(params.sfts.asScala.toList)
48       }
49       if (params.converters != null) {
50         listConverters(params.converters.asScala.toList)
51       }
52     }
53   }
54 
55   def listSfts(names: List[String] = List.empty): Unit = {
56     val all = SimpleFeatureTypeLoader.sfts
57     val filtered = if (names.isEmpty) all else names.flatMap(n => all.find(_.getTypeName == n))
58     Command.output.info("Simple Feature Types:")
59     if (filtered.isEmpty) {
60       Command.output.info("None available")
61     } else {
62       val paramsLower = params.format.toLowerCase
63       if (paramsLower == "typesafe" || paramsLower == "spec") {
64         val formatFn = paramsLower match {
65           case "typesafe" =>
66             (sft: SimpleFeatureType) => s"${SimpleFeatureTypes.toConfigString(sft, !params.excludeUserData, params.concise)}"
67           case "spec" =>
68             (sft: SimpleFeatureType) => s"${SimpleFeatureTypes.encodeType(sft, !params.excludeUserData)}"
69         }
70         filtered.sortBy(_.getTypeName).map(s => s"${s.getTypeName} = ${formatFn(s)}").foreach(m => Command.output.info(m))
71       } else {
72         throw new ParameterException(s"Unknown format '${params.format}'. Valid values are 'typesafe' or 'spec'")
73       }
74     }
75   }
76 
77   def listConverters(names: List[String] = List.empty): Unit = {
78     val all = ConverterConfigLoader.confs
79     val filtered = if (names.isEmpty) all else names.flatMap(n => all.find(_._1 == n))
80     Command.output.info("Simple Feature Type Converters:")
81     if (filtered.isEmpty) {
82       Command.output.info("None available")
83     } else {
84       val options = ConfigRenderOptions.defaults().setJson(false).setOriginComments(false)
85       def render(c: Config) = c.root().render(options)
86       val strings = filtered.map { case (cname, conf)=> s"converter-name=$cname\n${render(conf)}\n" }
87       strings.toArray.sorted.foreach(m => Command.output.info(m))
88     }
89   }
90 
91   def listSftsNames(): Unit = {
92     Command.output.info("Simple Feature Types:")
93     val all = SimpleFeatureTypeLoader.sfts
94     all.sortBy(_.getTypeName).map(s => s"${s.getTypeName}").foreach(m => Command.output.info(m))
95   }
96   def listConverterNames(): Unit = {
97     Command.output.info("Simple Feature Type Converters:")
98     val all = ConverterConfigLoader.confs
99     all.map { case (cname, conf) => s"$cname"}.toArray.sorted.foreach(m => Command.output.info(m))
100   }
101 }
