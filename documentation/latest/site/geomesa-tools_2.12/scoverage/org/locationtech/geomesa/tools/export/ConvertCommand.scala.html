43   private def convertAndExport(): Option[Long] = {
44     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
45 
46     import scala.collection.JavaConverters._
47 
48     val inputs: Inputs = {
49       val files = Inputs(params.files.asScala.toSeq)
50       if (files.stdin && !StdInHandle.isAvailable) {
51         if (files.paths.isEmpty) {
52           throw new ParameterException("Missing option: <files>... is required, or use `-` to ingest from standard in")
53         } else {
54           Command.user.info("Waiting for input...")
55           while (!StdInHandle.isAvailable) {
56             Thread.sleep(10)
57           }
58         }
59       }
60       if (params.srcList) { files.asSourceList } else { files }
61     }
62 
63     val format = IngestCommand.getDataFormat(params, inputs.paths)
64 
65     // use .get to re-throw the exception if we fail
66     IngestCommand.getSftAndConverter(params, inputs, format, None).get.flatMap { case (sft: SimpleFeatureType, config: com.typesafe.config.Config) =>
67       val files = inputs.handles.iterator
68 
69       WithClose(SimpleFeatureConverter(sft, config)) { converter =>
70         val results = new ResultTracker()
71         val query = ExportCommand.createQuery(sft, params)
72         val exporter = Option(params.chunkSize) match {
73           case None    => new Exporter(ExportOptions(params), query.getHints)
74           case Some(c) => new ChunkedExporter(ExportOptions(params), query.getHints, c)
75         }
76         try {
77           val count = WithClose(ConvertCommand.convertFeatures(files, converter, results, query)) { iter =>
78             if (!params.suppressEmpty || iter.hasNext) {
79               exporter.start(query.getHints.getReturnSft)
80               exporter.export(iter)
81             } else {
82               Some(0L)
83             }
84           }
85           Command.user.info(s"Converted ${getPlural(results.lines, "line")} "
86               + s"with ${getPlural(results.successes, "success", "successes")} "
87               + s"and ${getPlural(results.failures, "failure")}")
88           count
89         } finally {
90           exporter.close()
91         }
92       }
93     }
94   }

99   /**
100     * Convert features
101     *
102     * @param files inputs
103     * @param converter converter
104     * @param tracker holder for returning conversion results
105     * @param query query used to filter/transform inputs
106     * @return
107     */
108   private def convertFeatures(
109       files: Iterator[FileHandle],
110       converter: SimpleFeatureConverter,
111       tracker: ResultTracker,
112       query: Query): CloseableIterator[SimpleFeature] = {
113 
114     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
115 
116     def convert(): CloseableIterator[SimpleFeature] = CloseableIterator(files).flatMap { file =>
117       file.open.flatMap { case (name, is) =>
118         val params = EvaluationContext.inputFileParam(name.getOrElse(file.path))
119         val context = converter.createEvaluationContext(params)
120         val features = converter.process(is, context)
121         CloseableIterator(features, { features.close(); tracker.update(context) })
122       }
123     }
124 
125     def filter(iter: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] =
126       if (query.getFilter == Filter.INCLUDE) { iter } else { iter.filter(query.getFilter.evaluate) }
127 
128     def limit(iter: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = {
129       query.getHints.getMaxFeatures match {
130         case None    => iter
131         case Some(m) => iter.take(m)
132       }
133     }
134 
135     def transform(iter: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] =
136       LocalQueryRunner.transform(converter.targetSft, iter, query.getHints.getTransform, query.getHints)
137 
138     transform(limit(filter(convert())))
139   }
140 
141   private class ResultTracker {
142     var failures: Long = 0
143     var successes: Long = 0
144     var lines: Long = 0
145 
146     def update(context: EvaluationContext): Unit = {
147       failures += context.stats.failure(0)
148       successes += context.stats.success(0)
149       lines += context.line
150     }
151   }
