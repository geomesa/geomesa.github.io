31   private val coordMap = scala.collection.mutable.Map.empty[Coordinate, Int].withDefaultValue(0)

34   private var jsonWriter: JsonWriter = _
35   private var jsonSerializer: GeoJsonSerializer = _

37   private var featureInfo = ""
38 
39   override def start(sft: SimpleFeatureType): Unit = {
40     featureInfo = LeafletMapExporter.getFeatureInfo(sft)
41     writer = new OutputStreamWriter(stream.os, StandardCharsets.UTF_8)
42     jsonWriter = GeoJsonSerializer.writer(writer)
43     writer.write(LeafletMapExporter.IndexHead)
44     writer.write("var points = ")
45     writer.flush()
46     jsonSerializer = new GeoJsonSerializer(sft)
47     jsonSerializer.startFeatureCollection(jsonWriter)
48   }
49 
50   override def export(features: Iterator[SimpleFeature]): Option[Long] = {
51     var count = 0L
52     while (features.hasNext) {
53       val feature = features.next
54       jsonSerializer.write(jsonWriter, feature)
55       val geom = feature.getDefaultGeometry.asInstanceOf[Geometry]
56       if (geom != null) {
57         geom.getCoordinates.foreach(c => coordMap(c) += 1)
58       }
59       count += 1L
60     }
61     jsonWriter.flush()
62     Some(count)
63   }
64 
65   override def close(): Unit  = {
66     // finish writing GeoJson
67     if (writer != null) {
68       try {
69         jsonSerializer.endFeatureCollection(jsonWriter)
70         jsonWriter.flush()
71 
72         writer.write(";\n\n")
73         writer.write(featureInfo)
74         // Write Heatmap Data
75         writer.write("var heat = L.heatLayer([\n")
76         if (coordMap.isEmpty) {
77           Command.user.warn("No features were exported - the map will not render correctly")
78         } else {
79           val max = coordMap.maxBy(_._2)._2
80           val iter = coordMap.iterator
81           iter.take(1).foreach { case (coord, weight) =>
82             writer.write(s"        [${coord.y}, ${coord.x}, ${weight / max}]")
83           }
84           iter.foreach { case (coord, weight) =>
85             writer.write(s",\n        [${coord.y}, ${coord.x}, ${weight / max}]")
86           }
87         }
88         writer.write("\n    ], { radius: 25 });\n\n")
89         writer.write(LeafletMapExporter.IndexTail)
90       } finally {
91         jsonWriter.close() // also closes writer and output stream
92       }
93     }
94   }

117   def configure(params: ExportParams): Boolean = {
118     val large = "The Leaflet map may exhibit performance issues when displaying large numbers of features. For a " +
119         "more robust solution, consider using GeoServer."
120     if (params.maxFeatures == null) {
121       Command.user.warn(large)

123           "please use --max-features")
124       params.maxFeatures = LeafletMapExporter.MaxFeatures
125     } else if (params.maxFeatures > LeafletMapExporter.MaxFeatures) {
126       if (params.force) {
127         Command.user.warn(large)

129         return false
130       }
131     }
132 
133     if (params.gzip != null) {
134       Command.user.warn("Ignoring gzip parameter for Leaflet export")
135       params.gzip = null
136     }
137 
138     if (params.file == null) {
139       params.file = sys.props("user.dir")
140     }
141     if (PathUtils.isRemote(params.file)) {
142       if (params.file.endsWith("/")) {
143         params.file = s"${params.file}index.html"
144       } else if (FilenameUtils.indexOfExtension(params.file) == -1) {
145         params.file = s"${params.file}/index.html"
146       }
147     } else {
148       val file = new File(params.file)
149       val destination = if (file.isDirectory || (!file.exists && file.getName.indexOf(".") == -1)) {
150         new File(file, "index.html")
151       } else {
152         file
153       }
154       params.file = destination.getAbsolutePath
155     }
156 
157     true
158   }

160   def getFeatureInfo(sft: SimpleFeatureType): String = {
161     val str = new StringBuilder()
162     str.append("    function onEachFeature(feature, layer) {\n")
163     if (sft == null) {
164       str.append("\n    };\n\n").toString
165     } else {
166       val attributes = sft.getAttributeDescriptors.asScala.map(_.getLocalName). collect {
167         case name if name != "geom" => s""""$name: " + feature.properties.$name"""
168       }
169       str.append("""        layer.bindPopup("ID: " + feature.id + "<br>" + """)
170       str.append(""""GEOM: " + feature.geometry.type + "[" + feature.geometry.coordinates + "]<br>" + """)
171       str.append(attributes.mkString(""" + "<br>" + """))
172     }
173     str.append(");\n    }\n\n")
174     str.toString()
175   }
