25 
26   import scala.collection.JavaConverters._

28   override val name = "update-features"
29   override def params: UpdateFeaturesParams
30 
31   override def execute(): Unit = {
32     val attributes = params.attributes.asScala.toMap
33     val vis = Option(params.visibility).filterNot(_.isBlank)
34     if (attributes.isEmpty && vis.isEmpty) {
35       throw new ParameterException("Must specify at least one --set or --set-visibility")
36     }
37     withDataStore { ds =>
38       val sft = Try(ds.getSchema(params.featureName)).getOrElse(null)
39       if (sft == null) {
40         throw new ParameterException(s"Schema '${params.featureName}' does not exist in the data store")
41       } else if (attributes.keys.exists(a => sft.indexOf(a) == -1)) {
42         val missing = attributes.keys.filter(a => sft.indexOf(a) == -1).mkString("'", "', '", "'")
43         throw new ParameterException(
44           s"Attribute(s) $missing do not exist in schema '${params.featureName}'. Available attributes: " +
45             sft.getAttributeDescriptors.asScala.map(_.getLocalName).mkString(", "))
46       }
47       val filter = Option(params.cqlFilter).getOrElse(Filter.INCLUDE)
48       val msg = new StringBuilder(s"Updating schema '${params.featureName}', setting ")
49       if (attributes.nonEmpty) {
50         msg.append(attributes.map { case (k, v) => s"'$k' to '$v'" }.mkString("attribute(s) ", ", ", ""))
51         if (vis.nonEmpty) {
52           msg.append(" and ")
53         }
54       }
55       vis.foreach(v => msg.append(s"visibility to '$v'"))
56       if (filter == Filter.INCLUDE) {
57         msg.append(" for all features")
58       } else {
59         msg.append(s""" for features matching filter "${ECQL.toCQL(filter)}"""")
60       }
61       Command.user.info(msg.toString())
62       if (params.force || Prompt.confirm("Continue (y/n)? ")) {
63         Command.user.info(s"Updating features, please wait...")
64         var count = 0L
65         WithClose(ds.getFeatureWriter(params.featureName, filter, Transaction.AUTO_COMMIT)) { writer =>
66           while (writer.hasNext) {
67             val next = writer.next()
68             attributes.foreach { case (k, v) => next.setAttribute(k, v) }
69             vis.foreach(next.getUserData.put(SecurityUtils.FEATURE_VISIBILITY, _))
70             writer.write()
71             count += 1
72             if (count % 10000 == 0) {
73               print('.')
74             }
75           }
76           println()
77         }
78         Command.user.info(s"Complete - updated $count total features")
79       }
80     }
81   }
82 }
