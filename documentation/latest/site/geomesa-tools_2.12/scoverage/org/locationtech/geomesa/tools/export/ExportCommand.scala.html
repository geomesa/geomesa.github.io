74   private def export(ds: DS): JobResult = {
75     // for file data stores, handle setting the default type name so the user doesn't have to
76     for {
77       p <- Option(params).collect { case p: ProvidedTypeNameParam => p }
78       f <- Option(ds).collect { case f: FileDataStore => f }
79     } { p.featureName = f.getSchema.getTypeName }
80 
81     val options = ExportOptions(params)
82     val remote = options.file.exists(PathUtils.isRemote)
83     val reducers = Option(params.reducers).map(_.intValue()).getOrElse(-1)
84     val mode = params.mode.getOrElse(if (reducers == -1) { RunModes.Local } else { RunModes.Distributed })
85     val chunks = Option(params.chunkSize).map(_.longValue())
86 
87     // do some validation up front
88     if (options.file.isEmpty && !options.format.streaming) {
89       throw new ParameterException(s"Format '${options.format}' requires file-based output, please use --output")
90     } else if (remote && options.format == ExportFormat.Shp) {
91       throw new ParameterException("Shape file export is not supported for distributed file systems")
92     } else if (!remote && mode == RunModes.Distributed) {
93       throw new ParameterException("Distributed export requires an output file in a distributed file system")
94     } else if (mode == RunModes.Distributed && params.maxFeatures != null) {
95       throw new ParameterException("Distributed export does not support --max-features")
96     }
97 
98     val sft = ds.getSchema(params.featureName)
99     if (sft == null) {
100       throw new ParameterException(s"Schema '${params.featureName}' does not exist in the store")
101     }
102 
103     val query = ExportCommand.createQuery(sft, params)
104 
105     mode match {
106       case RunModes.Local =>
107         options.file.map(PathUtils.getHandle).foreach { file =>
108           if (file.exists) {
109             if (params.force) {
110               Command.user.warn(s"Output file '${file.path}' exists - deleting it")
111             } else if (!Prompt.confirm(s"WARNING Output file '${file.path}' exists, delete it and continue (y/n)? ")) {
112               throw new ParameterException(s"Output file '${file.path}' exists")
113             }
114             file.delete()
115           }
116         }
117         val exporter = chunks match {
118           case None    => new Exporter(options, query.getHints)
119           case Some(c) => new ChunkedExporter(options, query.getHints, c)
120         }
121         val count = try { export(ds, query, exporter, !params.suppressEmpty) } finally { exporter.close() }
122         val outFile = options.file match {
123           case None => "standard out"
124           case Some(f) if chunks.isDefined => PathUtils.getBaseNameAndExtension(f).productIterator.mkString("_*")
125           case Some(f) => f
126         }
127 
128         JobSuccess(s"Feature export complete to $outFile", count.map(CountKey -> _).toMap)
129 
130       case RunModes.Distributed =>
131         val job = Job.getInstance(new Configuration, "GeoMesa Tools Export")
132 
133         // for remote jobs, don't push down format transforms, to enable counting and global sorting
134         val hints = new Hints(query.getHints)
135         ExportCommand.disableAggregation(sft, query.getHints)
136 
137         configure(job, ds, query) // note: do this first so that input format is set for the TotalOrderPartitioner
138 
139         // note: these confs should be set by the input format
140         val reduce = Seq(GeoMesaConfigurator.Keys.FeatureReducer, GeoMesaConfigurator.Keys.Sorting).filter { key =>
141           job.getConfiguration.get(key) != null
142         }
143 
144         if (reducers < 1 && reduce.nonEmpty) {
145           if (reduce.contains(GeoMesaConfigurator.Keys.Sorting)) {
146             throw new ParameterException("Distributed export sorting requires --num-reducers")
147           } else {
148             throw new ParameterException(s"Distributed export format '${options.format}' requires --num-reducers")
149           }
150         }
151 
152         // must be some due to our remote check
153         val file = options.file.getOrElse(throw new IllegalStateException("file should be Some"))
154         val output = new Path(PathUtils.getUrl(file).toURI).getParent
155 
156         // file output format doesn't generally let you write to an existing directory
157         val context = FileContext.getFileContext(output.toUri, job.getConfiguration)
158         if (context.util.exists(output)) {
159           val warning = s"Output directory '$output' exists - files may be overwritten"
160           if (params.force) {
161             Command.user.warn(warning)
162           } else if (!Prompt.confirm(s"WARNING $warning. Continue anyway (y/n)? ")) {
163             if (Prompt.confirm("WARNING DATA MAY BE LOST - delete directory and proceed with export (y/n)? ")) {
164               context.delete(output, true)
165             } else {
166               throw new ParameterException(s"Output directory '$output' exists")
167             }
168           }
169         }
170 
171         val filename = FilenameUtils.getName(file)
172 
173         // note: use our original hints, which have the aggregating keys
174         ExportJob.configure(job, connection, sft, hints, filename, output, options.format, options.headers,
175           chunks, options.gzip, reducers, libjars(options.format), libjarsPaths)
176 
177         val reporter = TerminalCallback()
178         JobRunner.run(job, reporter, ExportJob.Counters.mapping(job), ExportJob.Counters.reducing(job)).merge {
179           Some(JobSuccess(s"Feature export complete to $output", Map(CountKey -> ExportJob.Counters.count(job))))
180         }
181 
182       case _ => throw new NotImplementedError() // someone added a run mode and didn't implement it here...
183     }
184   }

196     try {
197       Command.user.info("Running export - please wait...")
198       val features = ds.getFeatureSource(query.getTypeName).getFeatures(query)
199       WithClose(CloseableIterator(features.features())) { iter =>
200         if (writeEmptyFiles || iter.hasNext) {
201           exporter.start(features.getSchema)
202           exporter.export(iter)
203         } else {
204           Some(0L)
205         }
206       }
207     } catch {
208       case NonFatal(e) =>
209         throw new RuntimeException("Could not execute export query. Please ensure " +
210             "that all arguments are correct", e)
211     }

251   def createQuery(sft: SimpleFeatureType, params: ExportParams): Query = {
252     val typeName = Option(params).collect { case p: TypeNameParam => p.featureName }.orNull
253     val filter = Option(params.cqlFilter).getOrElse(Filter.INCLUDE)
254 
255     val query = new Query(typeName, filter)
256     val hints = query.getHints
257     Option(params.maxFeatures).map(Int.unbox).foreach(query.setMaxFeatures)
258     Option(params).collect { case p: OptionalIndexParam => p }.foreach { p =>
259       Option(p.index).foreach { index =>
260         logger.debug(s"Using index $index")
261         hints.put(QueryHints.QUERY_INDEX, index)
262       }
263     }
264 
265     Option(params.hints).foreach { hintStrings =>
266       hints.put(Hints.VIRTUAL_TABLE_PARAMETERS, hintStrings)
267       ViewParams.setHints(query)
268     }
269 
270     if (params.outputFormat == ExportFormat.Arrow) {
271       hints.put(QueryHints.ARROW_ENCODE, java.lang.Boolean.TRUE)
272     } else if (params.outputFormat == ExportFormat.Bin) {
273       // if not specified in hints, set it here to trigger the bin query
274       if (!hints.containsKey(QueryHints.BIN_TRACK)) {
275         hints.put(QueryHints.BIN_TRACK, "id")
276       }
277       if (!hints.containsKey(QueryHints.BIN_DTG)) {
278         sft.getDtgField.foreach(hints.put(QueryHints.BIN_DTG, _))
279       }
280     } else if (params.outputFormat == ExportFormat.Leaflet) {

282         throw new ParameterException("Terminating execution")
283       }
284     }
285 
286     val attributes: Array[String] = {
287       val combined = params.attributes.asScala ++ params.transforms.asScala
288       if (combined.nonEmpty) {
289         val (id, attributes) = combined.partition(_.equalsIgnoreCase("id"))
290         if (id.isEmpty && !hints.containsKey(QueryHints.ARROW_INCLUDE_FID)) {
291           // note: we also use this hint for delimited text export
292           hints.put(QueryHints.ARROW_INCLUDE_FID, java.lang.Boolean.FALSE)
293         }
294         attributes.toArray
295       } else if (params.outputFormat == ExportFormat.Bin) {
296         BinAggregatingScan.propertyNames(hints, sft).toArray
297       } else {
298         null // all props
299       }
300     }
301     query.setPropertyNames(attributes: _*)
302 
303     if (!params.sortFields.isEmpty) {
304       val fields = params.sortFields.asScala.toSeq
305       if (fields.exists(a => sft.indexOf(a) == -1)) {
306         val errors = fields.filter(a => sft.indexOf(a) == -1)
307         throw new ParameterException(s"Invalid sort attribute${if (errors.lengthCompare(1) == 0) "" else "s"}: " +
308             errors.mkString(", "))
309       }
310       val order = if (params.sortDescending) { SortOrder.DESCENDING } else { SortOrder.ASCENDING }
311       query.setSortBy(fields.map(f => org.locationtech.geomesa.filter.ff.sort(f, order)): _*)
312     } else if (hints.isArrowQuery) {
313       hints.getArrowSort.foreach { case (f, r) =>
314         val order = if (r) { SortOrder.DESCENDING } else { SortOrder.ASCENDING }
315         query.setSortBy(org.locationtech.geomesa.filter.ff.sort(f, order))
316       }
317     }
318 
319     logger.debug(s"Applying CQL filter ${ECQL.toCQL(filter)}")
320     logger.debug(s"Applying transform ${Option(attributes).map(_.mkString(",")).orNull}")
321 
322     QueryRunner.configureDefaultQuery(sft, query)
323   }

422   /**
423     * Feature exporter that handles chunking output into multiple files
424     *
425     * @param options export options
426     * @param hints query hints
427     * @param dictionaries arrow dictionaries (lazily evaluated)
428     * @param chunks number of bytes to write per file
429     */
430   class ChunkedExporter(options: ExportOptions, hints: Hints, chunks: Long)
431       extends FeatureExporter with LazyLogging {
432 
433     private val names = options.file match {
434       case None    => Iterator.continually(None)
435       case Some(f) => new IncrementingFileName(f).map(Option.apply)
436     }
437 
438     private var sft: SimpleFeatureType = _
439     private var exporter: FeatureExporter = _
440     private var estimator: FileSizeEstimator = _
441     private var count = 0L // number of features written
442     private var total = 0L // sum size of all finished chunks
443 
444     override def start(sft: SimpleFeatureType): Unit = {
445       this.sft = sft
446       estimator = new FileSizeEstimator(chunks, 0.05f, options.format.bytesPerFeature(sft)) // 5% error threshold
447       nextChunk()
448     }
449 
450     override def export(features: Iterator[SimpleFeature]): Option[Long] = export(features, None)
451 
452     override def bytes: Long = if (exporter == null) { total } else { total + exporter.bytes }
453 
454     override def close(): Unit = if (exporter != null) { exporter.close() }
455 
456     private def nextChunk(): Unit = {
457       if (exporter != null) {
458         exporter.close()
459         // adjust our estimate to account for the actual bytes written
460         // do this after closing the exporter to account for footers, batching, etc
461         val written = exporter.bytes
462         estimator.update(written, count)
463         total += written
464       }
465       exporter = new Exporter(options.copy(file = names.next), hints)
466       exporter.start(sft)
467       count = 0L
468     }
469 
470     @tailrec
471     private def export(features: Iterator[SimpleFeature], result: Option[Long]): Option[Long] = {
472       var estimate = estimator.estimate(exporter.bytes)
473       val counter = features.takeWhile { _ => count += 1; estimate -= 1; estimate >= 0 }
474       val exported = exporter.export(counter) match {
475         case None    => result
476         case Some(c) => result.map(_ + c).orElse(Some(c))
477       }
478       if (features.isEmpty) {
479         exported
480       } else {
481         // if it's a countable format, the bytes should be available now and we can compare to our chunk size
482         // otherwise, the bytes aren't generally available until after closing the writer,
483         // so we have to go with our initial estimate and adjust after the first chunk
484         if (options.format.countable) {
485           val bytes = exporter.bytes
486           if (estimator.done(bytes)) {
487             nextChunk()
488           } else {
489             estimator.update(bytes, count)
490           }
491         } else {
492           nextChunk()
493         }
494         export(features, exported)
495       }
496     }
497   }
