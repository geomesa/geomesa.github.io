25 class ArrowExporter(stream: ExportStream, hints: Hints) extends ByteCounterExporter(stream) {
26 
27   import org.locationtech.geomesa.index.conf.QueryHints.RichHints
28 
29   private lazy val sort = hints.getArrowSort
30   private lazy val encoding = SimpleFeatureEncoding.min(hints.isArrowIncludeFid, hints.isArrowProxyFid)
31   private lazy val ipc = hints.getArrowFormatVersion.getOrElse(FormatVersion.ArrowFormatVersion.get)
32   private lazy val batchSize = hints.getArrowBatchSize.getOrElse(ArrowProperties.BatchSize.get.toInt)
33   private lazy val dictionaryFields = hints.getArrowDictionaryFields
34 
35   private var delegate: FeatureExporter = _
36 
37   override def start(sft: SimpleFeatureType): Unit = {
38     delegate = if (sft == org.locationtech.geomesa.arrow.ArrowEncodedSft) {
39       new EncodedDelegate(stream.os)
40     } else if (dictionaryFields.isEmpty) {
41       // note: features should be sorted already, even if arrow encoding wasn't performed
42       new BatchDelegate(stream.os, encoding, FormatVersion.options(ipc), sort, batchSize, Map.empty)
43     } else {
44       if (sort.isDefined) {
45         throw new NotImplementedError("Sorting and calculating dictionaries at the same time is not supported")
46       }
47       new DictionaryDelegate(stream.os, dictionaryFields, encoding, FormatVersion.options(ipc), batchSize)
48     }
49     delegate.start(sft)
50   }
51 
52   override def export(features: Iterator[SimpleFeature]): Option[Long] = delegate.export(features)
53 
54   override def close(): Unit = {
55     CloseWithLogging(Option(delegate))
56     stream.close()
57   }
58 }

73   private class DictionaryDelegate(
74       os: OutputStream,
75       dictionaryFields: Seq[String],
76       encoding: SimpleFeatureEncoding,
77       ipcOpts: IpcOption,
78       batchSize: Int
79     ) extends FeatureExporter {
80 
81     private var writer: DictionaryBuildingWriter = _
82     private var count = 0L
83 
84     override def start(sft: SimpleFeatureType): Unit = {
85       writer = new DictionaryBuildingWriter(sft, dictionaryFields, encoding, ipcOpts)
86     }
87 
88     override def export(features: Iterator[SimpleFeature]): Option[Long] = {
89       val start = count
90       features.foreach { f =>
91         writer.add(f)
92         count += 1
93         if (count % batchSize == 0) {
94           writer.encode(os)
95           writer.clear()
96         }
97       }
98       Some(count - start)
99     }
100 
101     override def bytes: Long = 0L
102 
103     override def close(): Unit = {
104       if (writer != null) {
105         if (count % batchSize != 0) {
106           writer.encode(os)
107           writer.clear()
108         }
109         CloseWithLogging(writer)
110       }
111     }
112   }
