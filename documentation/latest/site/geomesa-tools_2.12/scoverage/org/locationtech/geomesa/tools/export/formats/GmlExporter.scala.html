29 /**
30   * GML exporter implementation.
31   *
32   * The geotools GML export classes only support encoding a feature collection. To support our usage
33   * pattern (start, export n times, end), we create an asynchronous feature collection and do the actual
34   * encoding in a separate thread. The encoder thread will block until there are more features to export,
35   * so that we only get a single feature collection in the xml.
36   *
37   * @param stream output stream
38   * @param configuration wfs configuration (gml3 vs gml2)
39   */
40 class GmlExporter private (stream: ExportStream, configuration: WFSConfiguration)
41     extends ByteCounterExporter(stream) {
42 
43   private val encoder: Encoder = {
44     val props = configuration.getProperties.asInstanceOf[java.util.Set[QName]]
45     props.add(org.geotools.gml2.GMLConfiguration.OPTIMIZED_ENCODING)
46     props.add(org.geotools.gml2.GMLConfiguration.NO_FEATURE_BOUNDS)
47     val e = new Encoder(configuration)
48     e.getNamespaces.declarePrefix("geomesa", "http://geomesa.org")
49     e.setEncoding(StandardCharsets.UTF_8)
50     e.setIndenting(true)
51     e
52   }
53 
54   private val es = Executors.newSingleThreadExecutor()
55   private var fc: AsyncFeatureCollection = _

57   override def start(sft: SimpleFeatureType): Unit = {
58     fc = new AsyncFeatureCollection(sft)
59     val features = if (sft.getName.getNamespaceURI != null) { fc } else {
60       val builder = new SimpleFeatureTypeBuilder()
61       builder.init(sft)
62       builder.setNamespaceURI("http://geomesa.org")
63       new ReTypingFeatureCollection(fc, builder.buildFeatureType())
64     }
65     val collection = WfsFactory.eINSTANCE.createFeatureCollectionType()
66     collection.getFeature.asInstanceOf[java.util.List[SimpleFeatureCollection]].add(features)
67 
68     def encode(): Unit = encoder.encode(collection, org.geotools.wfs.WFS.FeatureCollection, stream.os)
69 
70     val runnable = new Runnable() {
71       override def run(): Unit = {
72         if (System.getProperty(GmlExporter.TransformerProperty) != null) { encode() } else {
73           // explicitly set the default java transformer, to avoid picking up saxon (which causes errors)
74           // the default class is hard-coded in javax.xml.transform.TransformerFactory.newInstance() ...
75           System.setProperty(GmlExporter.TransformerProperty,
76             "com.sun.org.apache.xalan.internal.xsltc.trax.TransformerFactoryImpl")
77           try { encode() } finally {

79           }
80         }
81       }
82     }
83 
84     es.execute(runnable)
85   }
86 
87   override def export(features: Iterator[SimpleFeature]): Option[Long] = {
88     var count = 0L
89     val counting = features.map { f => count += 1; f }
90     while (counting.nonEmpty) {
91       // export in chunks of 100 so that the exporter thread gets notified and doesn't keep blocking
92       fc.addAsync(counting.take(100))
93     }
94     Some(count)
95   }
96 
97   override def close(): Unit = {
98     try {
99       if (fc != null) {
100         fc.endAsync()
101       }
102       es.shutdown()
103       es.awaitTermination(Long.MaxValue, TimeUnit.MILLISECONDS)
104     } finally {
105       stream.close()
106     }
107   }

132   /**
133     * Feature collection that lets us add additional features in an asynchronous fashion. The consumer
134     * thread will be blocked on calls to 'hasNext' until the producer thread adds features or indicates
135     * completion
136     *
137     * @param sft simple feature type
138     */
139   private class AsyncFeatureCollection(sft: SimpleFeatureType) extends DataFeatureCollection(null, sft) {
140 
141     private val buffer = new ConcurrentLinkedQueue[SimpleFeature]()
142     private val done = new AtomicBoolean(false)
143     private val lock = new ReentrantLock()
144     private val condition = lock.newCondition()
145 
146     private val iter: SimpleFeatureIterator = new SimpleFeatureIterator() {
147 
148       private var current: SimpleFeature = _
149 
150       override def hasNext: Boolean = {
151         if (current != null) {
152           return true
153         }
154         lock.lock()
155         try {
156           current = buffer.poll()
157           // note: we need to loop here to skip 'spurious wake-ups'
158           while (current == null) {
159             if (done.get) {
160               return false
161             }
162             condition.await()
163             current = buffer.poll()
164           }
165           true
166         } finally {
167           lock.unlock()
168         }
169       }
170 
171       override def next(): SimpleFeature = {
172         // note: we shouldn't need to synchronize this as next/hasNext should be a single caller thread
173         val result = current
174         current = null
175         result
176       }
177 
178       override def close(): Unit = endAsync()
179     }
180 
181     /**
182       * Add features to be returned from this feature collection
183       *
184       * @param features features
185       */
186     def addAsync(features: Iterator[SimpleFeature]): Unit = {
187       lock.lock()
188       try {
189         features.foreach(buffer.add)
190         condition.signal()
191       } finally {
192         lock.unlock()
193       }
194     }
195 
196     /**
197       * Signal that there are no more features that will be added
198       */
199     def endAsync(): Unit = {
200       lock.lock()
201       try {
202         done.set(true)
203         condition.signal()
204       } finally {
205         lock.unlock()
206       }
207     }
208
