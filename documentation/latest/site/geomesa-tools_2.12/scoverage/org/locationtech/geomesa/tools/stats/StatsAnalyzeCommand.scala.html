31   private def analyze(ds: DS): Unit = {
32     val sft = ds.getSchema(params.featureName)
33     if (sft == null) {
34       throw new ParameterException(s"Schema '${params.featureName}' does not exist")
35     }
36 
37     def noStats = ds.metadata.read(sft.getTypeName, GeoMesaMetadata.StatsGenerationKey, cache = false).isEmpty
38 
39     Command.user.info(s"Running stat analysis for feature type ${sft.getTypeName}...")
40     val stats = if (noStats) {
41       // this is the first time anyone has accessed this data store since stats were implemented
42       // stats should be kicked off and run in the background, we just have to wait for it to finish
43       // note: if something goes wrong this will just hang... but this is kind of a fringe case
44       while (noStats) {
45         Thread.sleep(1000)
46       }
47       // sleep an additional bit so that the stat table gets configured
48       Thread.sleep(1000)
49       val queries = sft.getAttributeDescriptors.asScala.map(d => Stat.MinMax(d.getLocalName))
50       ds.stats.getStat[CountStat](sft, Stat.Count()) ++ ds.stats.getSeqStat[MinMax[Any]](sft, queries.toSeq)
51     } else {
52       ds.stats.writer.analyze(sft)
53     }
54 
55     val strings = stats.collect {
56       case s: CountStat   => s"Total features: ${s.count}"
57       case s: MinMax[Any] =>
58         val bounds = if (s.isEmpty) {
59           "[ no matching data ]"
60         } else if (sft.getGeomField == s.property) {
61           val e = s.min.asInstanceOf[Geometry].getEnvelopeInternal
62           e.expandToInclude(s.max.asInstanceOf[Geometry].getEnvelopeInternal)
63           s"[ ${e.getMinX}, ${e.getMinY}, ${e.getMaxX}, ${e.getMaxY} ] cardinality: ${s.cardinality}"
64         } else {
65           val stringify = Stat.stringifier(sft.getDescriptor(s.property).getType.getBinding)
66           s"[ ${stringify(s.min)} to ${stringify(s.max)} ] cardinality: ${s.cardinality}"
67         }
68         s"Bounds for ${s.property}: $bounds"
69     }
70 
71     Command.user.info("Stats analyzed:")
72     Command.output.info(strings.mkString("  ", "\n  ", ""))
73     Command.user.info("Use 'stats-histogram', 'stats-top-k' or 'stats-count' commands for more details")
74   }
75 }
