44   override def getCount(sft: SimpleFeatureType, filter: Filter, exact: Boolean, queryHints: Hints): Option[Long] = {
45     if (exact) {
46       query[CountStat](sft, filter, Stat.Count(), queryHints).map(_.count)
47     } else if (filter == Filter.INCLUDE) {
48       // note: compared to the 'read' method, we want to return empty counts (indicating no features)
49       try { metadata.read(sft.getTypeName, countKey()).collect { case s: CountStat => s.count } } catch {
50         case NonFatal(e) => logger.error("Error reading existing stats:", e); None
51       }
52     } else {
53       estimateCount(sft, QueryPlanFilterVisitor(sft, filter))
54     }
55   }

107   override def getZ3Histogram(
108       sft: SimpleFeatureType,
109       geom: String,
110       dtg: String,
111       period: TimePeriod,
112       bins: Int,
113       filter: Filter,
114       exact: Boolean): Option[Z3Histogram] = {
115     if (exact) { super.getZ3Histogram(sft, geom, dtg, period, bins, filter, exact) } else {
116       val keys = if (sft.getGeomField != geom || !sft.getDtgField.contains(dtg)) { Seq.empty } else {
117         // z3 histograms are stored by time bin - calculate the times to retrieve
118         // either use the filter passed in, or else calculate from the time bounds
119         val timeBins = extractBins(sft, filter).orElse {
120           read[MinMax[Date]](sft, Filter.INCLUDE, minMaxKey(dtg)).map { bounds =>
121             val timeToBin = BinnedTime.timeToBinnedTime(sft.getZ3Interval)
122             val lBin = timeToBin(bounds.min.getTime).bin
123             val uBin = timeToBin(bounds.max.getTime).bin
124             Range.inclusive(lBin, uBin).map(_.toShort)
125           }
126         }
127         timeBins.getOrElse(Seq.empty).map(histogramKey(geom, dtg, _))
128       }
129       read(sft, filter, keys)
130     }
131   }

133   override def getStat[T <: Stat](
134       sft: SimpleFeatureType,
135       query: String,
136       filter: Filter = Filter.INCLUDE,
137       exact: Boolean = false): Option[T] = {
138     if (exact) { super.getStat(sft, query , filter, exact) } else {
139       def getStat(stat: Stat): Option[Stat] = stat match {
140         case _: CountStat          => read[CountStat](sft, filter, countKey())
141         case s: Histogram[_]       => getHistogram(sft, s.property, s.length, s.min, s.max, filter, exact)
142         case s: Z3Histogram        => getZ3Histogram(sft, s.geom, s.dtg, s.period, s.length, filter, exact)
143         case s: TopK[_]            => getTopK(sft, s.property, filter, exact)
144         case s: Frequency[_]       => getFrequency(sft, s.property, s.precision, filter, exact)
145         case s: EnumerationStat[_] => getEnumeration(sft, s.property, filter, exact)
146         case s: MinMax[_]          => getMinMax(sft, s.property, filter, exact)
147         case s: SeqStat            => Some(new SeqStat(sft, s.stats.flatMap(getStat))).filter(_.stats.nonEmpty)
148         case _ => None
149       }
150       getStat(StatParser.parse(sft, query)).asInstanceOf[Option[T]]
151     }
152   }

164   /**
165     * Gets keys and stats to write. Some stats end up getting split for writing.
166     *
167     * @param stat stat to write
168     * @param sft simple feature type
169     * @param merge merge or overwrite existing stats
170     * @return metadata keys and split stats
171     */
172   private def getStatsForWrite(stat: Stat, sft: SimpleFeatureType, merge: Boolean): Seq[WritableStat] = {
173     stat match {
174       case s: SeqStat      => s.stats.flatMap(getStatsForWrite(_, sft, merge))
175       case s: CountStat    => Seq(WritableStat(countKey(), s, merge))
176       case s: MinMax[_]    => Seq(WritableStat(minMaxKey(s.property), s, merge))
177       case s: TopK[_]      => Seq(WritableStat(topKKey(s.property), s, merge))
178       case s: Histogram[_] => Seq(WritableStat(histogramKey(s.property), s, merge))
179 
180       case s: Frequency[_] =>
181         if (s.dtg.isEmpty) {
182           Seq(WritableStat(frequencyKey(s.property), s, merge))
183         } else {
184           // split up the frequency and store by week
185           s.splitByTime.map { case (b, f) => WritableStat(frequencyKey(s.property, b), f, merge) }
186         }
187 
188       case s: Z3Histogram  =>
189         // split up the z3 histogram and store by week
190         s.splitByTime.map { case (b, z) => WritableStat(histogramKey(s.geom, s.dtg, b), z, merge) }
191 
192       case _ => throw new UnsupportedOperationException("Only Count, Frequency, MinMax, TopK and Histogram stats are tracked")
193     }
194   }

196   /**
197     * Determines the stats to calculate for a given schema.
198     *
199     * We always collect a total count stat.
200     * For the default geometry and default date, we collect a min/max and histogram.
201     * If there is both a default geometry and date, we collect a z3 histogram.
202     * For any indexed attributes, we collect a min/max, top-k, frequency and histogram.
203     * For any flagged attributes, we collect min/max and top-k
204     *
205     * @param sft simple feature type
206     * @param bounds lookup function for histogram bounds
207     * @return stat string
208     */
209   private def buildStatsFor(sft: SimpleFeatureType, bounds: String => Option[MinMax[Any]]): String = {
210     import GeoMesaStats._
211     import org.locationtech.geomesa.utils.geotools.RichAttributeDescriptors.RichAttributeDescriptor
212 
213     // get the attributes that we will keep stats for
214 
215     val stAttributesBuilder = Seq.newBuilder[String]
216     val indexedAttributesBuilder = Seq.newBuilder[String]
217 
218     sft.getIndices.foreach { i =>
219       if (i.attributes.headOption.contains(sft.getGeomField)) {
220         stAttributesBuilder ++= i.attributes
221       } else {
222         i.attributes.headOption.foreach(indexedAttributesBuilder += _)
223       }
224     }
225 
226     val stAttributes = stAttributesBuilder.result().distinct
227     val indexedAttributes = indexedAttributesBuilder.result().distinct.filter { a =>
228       !stAttributes.contains(a) && okForStats(sft.getDescriptor(a))
229     }
230     val flaggedAttributes = sft.getAttributeDescriptors.asScala.collect {
231       case d if d.isKeepStats && okForStats(d) => d.getLocalName
232     }.filter(a => !stAttributes.contains(a) && !indexedAttributes.contains(a))
233 
234     val count = Stat.Count()
235 
236     // calculate min/max for all attributes
237     val minMax = (stAttributes ++ indexedAttributes ++ flaggedAttributes).distinct.map(Stat.MinMax)
238 
239     // calculate topk for indexed attributes, but not geom + date
240     val topK = (indexedAttributes ++ flaggedAttributes).distinct.map(Stat.TopK)
241 
242     // calculate frequencies only for indexed attributes
243     val frequencies = {
244       val descriptors = indexedAttributes.map(sft.getDescriptor)
245       // calculate one frequency that's split by week, and one that isn't
246       // for queries with time bounds, the split by week will be more accurate
247       // for queries without time bounds, we save the overhead of merging the weekly splits
248       val withDates = sft.getDtgField match {
249         case None => Seq.empty
250         case Some(dtg) =>
251           val period = sft.getZ3Interval
252           descriptors.map(d => Stat.Frequency(d.getLocalName, dtg, period, defaultPrecision(d.getType.getBinding)))
253       }
254       val noDates = descriptors.map(d => Stat.Frequency(d.getLocalName, defaultPrecision(d.getType.getBinding)))
255       withDates ++ noDates
256     }
257 
258     // calculate histograms for all indexed attributes and geom/date
259     val histograms = (stAttributes ++ indexedAttributes).distinct.map { attribute =>
260       val minMax = bounds(attribute)
261       val cardinality = minMax.map(_.cardinality).getOrElse(0L)
262       // estimate 10k entries per bin, but cap at 10k bins (~29k on disk)
263       val size = if (attribute == sft.getGeomField) { MaxHistogramSize } else {
264         math.min(MaxHistogramSize, math.max(DefaultHistogramSize, cardinality / 10000).toInt)
265       }
266       val binding = sft.getDescriptor(attribute).getType.getBinding
267       implicit val ct = ClassTag[Any](binding)
268       // calculate the endpoints for the histogram
269       // the histogram will expand as needed, but this is a starting point
270       val (lower, upper) = minMax match {
271         case None => defaultBounds(binding)
272         // max has to be greater than min for the histogram bounds
273         case Some(b) if Histogram.equivalent(b.min, b.max, size) => Histogram.buffer(b.min)
274         case Some(b) => b.bounds
275       }
276 
277       Stat.Histogram[Any](attribute, size, lower, upper)
278     }
279 
280     val z3Histogram = for {
281       geom <- Option(sft.getGeomField).filter(stAttributes.contains)
282       dtg  <- sft.getDtgField.filter(stAttributes.contains)
283     } yield {
284       Stat.Z3Histogram(geom, dtg, sft.getZ3Interval, MaxHistogramSize)
285     }
286 
287     Stat.SeqStat(Seq(count) ++ minMax ++ topK ++ histograms ++ frequencies ++ z3Histogram)
288   }

290   /**
291     * Read a stat from the metadata
292     *
293     * @param sft simple feature type
294     * @param filter cql filter
295     * @param key metadata key to read
296     * @tparam T stat type
297     * @return
298     */
299   private def read[T <: Stat](sft: SimpleFeatureType, filter: Filter, key: String): Option[T] = {
300     try {
301       val result = metadata.read(sft.getTypeName, key).collect { case s: T if !s.isEmpty => s }
302       if (result.isDefined && filter != Filter.INCLUDE) {
303         logger.warn(s"Ignoring filter for non-exact stat query: ${ECQL.toCQL(filter)}")
304       }
305       result
306     } catch {
307       case NonFatal(e) => logger.error("Error reading existing stats:", e); None
308     }
309   }

311   /**
312     * Read a stat from the metadata
313     *
314     * @param sft simple feature type
315     * @param filter cql filter
316     * @param keys metadata keys to read
317     * @tparam T stat type
318     * @return
319     */
320   private def read[T <: Stat](sft: SimpleFeatureType, filter: Filter, keys: Seq[String]): Option[T] = {
321     try {
322       val seq = keys.flatMap(metadata.read(sft.getTypeName, _).collect { case s: T if !s.isEmpty => s })
323       val result = Stat.combine(seq).filterNot(_.isEmpty)
324       if (result.isDefined && filter != Filter.INCLUDE) {
325         logger.warn(s"Ignoring filter for non-exact stat query: ${ECQL.toCQL(filter)}")
326       }
327       result
328     } catch {
329       case NonFatal(e) => logger.error("Error reading existing stats:", e); None
330     }
331   }

333   /**
334     * Extract time period bins from a filter. Used to look up stats that are split by time period
335     *
336     * @param sft simple feature type
337     * @param filter filter
338     * @return
339     */
340   private def extractBins(sft: SimpleFeatureType, filter: Filter): Option[Seq[Short]] = {
341     sft.getDtgField.flatMap { dtg =>
342       val intervals = FilterHelper.extractIntervals(filter, dtg)
343       // don't consider gaps, just get the endpoints of the intervals
344       val bounds = intervals.values.reduceLeftOption[Bounds[ZonedDateTime]] { case (left, right) =>
345         val lower = Bounds.smallerLowerBound(left.lower, right.lower)
346         val upper = Bounds.largerUpperBound(left.upper, right.upper)
347         Bounds(lower, upper)
348       }
349       bounds.flatMap { d =>
350         for { lower <- d.lower.value; upper <- d.upper.value } yield {
351           val lo = lower.toInstant.toEpochMilli
352           val hi = upper.toInstant.toEpochMilli
353           val timeToBinnedTime = BinnedTime.timeToBinnedTime(sft.getZ3Interval)
354           Range.inclusive(timeToBinnedTime(lo).bin, timeToBinnedTime(hi).bin).map(_.toShort)
355         }
356       }
357     }
358   }

364 
365     override def analyze(sft: SimpleFeatureType): Seq[Stat] = {
366       import org.locationtech.geomesa.utils.geotools.GeoToolsDateFormat
367 
368       // calculate the stats we'll be gathering based on the simple feature type attributes
369       val statString = buildStatsFor(sft, getMinMax(sft, _))
370 
371       logger.debug(s"Calculating stats for ${sft.getTypeName}: $statString")
372 
373       val stats = getStat[SeqStat](sft, statString, exact = true).getOrElse(new SeqStat(sft, Seq.empty))
374 
375       logger.trace(s"Stats for ${sft.getTypeName}: ${stats.stats.map(_.toJson).mkString(", ")}")
376       logger.debug(s"Writing stats for ${sft.getTypeName}")
377 
378       // write the stats in one go - don't merge, this is the authoritative value
379       write(sft.getTypeName, getStatsForWrite(stats, sft, merge = false))
380 
381       // update our last run time
382       ds match {
383         case hm: HasGeoMesaMetadata[String] =>
384           val date = GeoToolsDateFormat.format(Instant.now().atZone(ZoneOffset.UTC))
385           hm.metadata.insert(sft.getTypeName, GeoMesaMetadata.StatsGenerationKey, date)
386         case _ => // no-op
387       }
388 
389       stats.stats
390     }
391 
392     override def updater(sft: SimpleFeatureType): StatUpdater =
393       if (sft.statsEnabled) { new MetadataStatUpdater(sft) } else { NoopStatUpdater }
394 
395     override def clear(sft: SimpleFeatureType): Unit = metadata.delete(sft.getTypeName)
396 
397     override def rename(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
398       val names: Map[String, String] = {
399         val local = previous.getAttributeDescriptors.asScala.map(_.getLocalName).zipWithIndex.toMap
400         local.flatMap { case (name, i) =>
401           val update = sft.getDescriptor(i).getLocalName
402           if (update == name) { Map.empty[String, String] } else { Map(name -> update) }
403         }
404       }
405 
406       def rename(stat: Stat): Option[Stat] = {
407         val copy: Option[Stat] = stat match {
408           case _: CountStat =>
409             None
410 
411           case s: MinMax[_] =>
412             names.get(s.property).map(n => new MinMax(sft, n)(s.defaults))
413 
414           case s: TopK[_] =>
415             names.get(s.property).map(n => new TopK(sft, n))
416 
417           case s: Histogram[AnyRef] =>
418             names.get(s.property).map(n => new Histogram(sft, n, s.length, (s.min, s.max))(s.defaults, s.ct))
419 
420           case s: Frequency[_] =>
421             if ((Seq(s.property) ++ s.dtg).forall(n => !names.contains(n))) { None } else {
422               val name = names.getOrElse(s.property, s.property)
423               val dtg = s.dtg.map(d => names.getOrElse(d, d))
424               Some(new Frequency(sft, name, dtg, s.period, s.precision, s.eps, s.confidence)(s.ct))
425             }
426 
427           case s: Z3Histogram =>
428             if (Seq(s.geom, s.dtg).forall(n => !names.contains(n))) { None } else {
429               val geom = names.getOrElse(s.geom, s.geom)
430               val dtg = names.getOrElse(s.dtg, s.dtg)
431               Some(new Z3Histogram(sft, geom, dtg, s.period, s.length))
432             }
433 
434           case s: SeqStat =>
435             val children = s.stats.map(rename)
436             if (children.forall(_.isEmpty)) { None } else {
437               Some(new SeqStat(sft, children.zip(s.stats).map { case (opt, default) => opt.getOrElse(default) }))
438             }
439 
440           case s =>
441             throw new UnsupportedOperationException(s"Unexpected stat: $s")
442         }
443         copy.foreach(_ += stat)
444         copy
445       }
446 
447       if (names.nonEmpty || sft.getTypeName != previous.getTypeName) {
448         val serializer =
449           Some(metadata)
450               .collect { case m: TableBasedMetadata[Stat] => m.serializer }
451               .collect { case s: StatsMetadataSerializer => s }
452 
453         // we need to set the old feature type in the serializer cache to read back our current values
454         serializer.foreach(s => s.cache.put(previous.getTypeName, StatSerializer(previous)))
455         val old = try { metadata.scan(previous.getTypeName, "", cache = false).toList } finally {
456           serializer.foreach(s => s.cache.remove(previous.getTypeName)) // will re-load latest from data store
457         }
458         // note: we can avoid a compaction by setting merge = true, since there won't be any previous values
459         if (sft.getTypeName != previous.getTypeName) {
460           old.foreach { case (key, stat) =>
461             val renamed = getStatsForWrite(rename(stat).getOrElse(stat), sft, merge = true)
462             write(sft.getTypeName, renamed)
463             metadata.remove(previous.getTypeName, key)
464           }
465         } else {
466           old.foreach { case (key, stat) =>
467             rename(stat).foreach { r =>
468               write(sft.getTypeName, getStatsForWrite(r, sft, merge = true))
469               metadata.remove(previous.getTypeName, key)
470             }
471           }
472         }
473       }
474     }
475   }

477   /**
478     * Stores stats as metadata entries
479     *
480     * @param sft simple feature type
481     */
482   protected class MetadataStatUpdater(sft: SimpleFeatureType) extends StatUpdater with LazyLogging {
483 
484     private var stat: Stat = Stat(sft, buildStatsFor(sft, getMinMax(sft, _)))
485 
486     override def add(sf: SimpleFeature): Unit = stat.observe(sf)
487 
488     override def remove(sf: SimpleFeature): Unit = stat.unobserve(sf)
489 
490     override def close(): Unit = {
491       if (!stat.isEmpty) {
492         write(sft.getTypeName, getStatsForWrite(stat, sft, merge = true))
493       }
494     }
495 
496     override def flush(): Unit = {
497       if (!stat.isEmpty) {
498         write(sft.getTypeName, getStatsForWrite(stat, sft, merge = true))
499       }
500       // reload the tracker - for long-held updaters, this will refresh the histogram ranges
501       stat = Stat(sft, buildStatsFor(sft, localBounds))
502     }
503 
504     /**
505       * Get the bounds we've seen so far in this updater, but don't re-load from Accumulo as that
506       * can be slow
507       *
508       * @param attribute attribute
509       * @return
510       */
511     private def localBounds(attribute: String): Option[MinMax[Any]] = {
512       stat match {
513         case s: SeqStat => s.stats.collectFirst { case m: MinMax[Any] if m.property == attribute => m }
514         case _ => logger.error(s"Expected to have a SeqStat but got: $stat"); None
515       }
516     }
517   }

551   /**
552     * Stat serializer
553     *
554     * @param ds datastore
555     */
556   class StatsMetadataSerializer(ds: DataStore) extends MetadataSerializer[Stat] {
557 
558     private [MetadataBackedStats] val cache = new ConcurrentHashMap[String, StatSerializer]()
559 
560     private def serializer(typeName: String): StatSerializer = {
561       var serializer = cache.get(typeName)
562       if (serializer == null) {
563         val sft = ds.getSchema(typeName)
564         if (sft == null) {
565           throw new RuntimeException(s"Trying to deserialize stats for type '$typeName' " +
566               "but it doesn't exist in the datastore")
567         }
568         serializer = StatSerializer(sft) // note: retrieves a cached value
569         cache.put(typeName, serializer)
570       }
571       serializer
572     }
573 
574     override def serialize(typeName: String, value: Stat): Array[Byte] =
575       serializer(typeName).serialize(value)
576 
577     override def deserialize(typeName: String, value: Array[Byte]): Stat =
578       serializer(typeName).deserialize(value, immutable = true)
