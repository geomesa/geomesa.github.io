25 class GeoMesaFeatureReader(result: QueryResult) {
26 
27   def schema: SimpleFeatureType = result.schema
28   def hints: Hints = result.hints
29 
30   def reader(): SimpleFeatureReader = new ResultReader()

32   private class ResultReader extends SimpleFeatureReader {
33 
34     private val iter = result.iterator()
35 
36     override def getFeatureType: SimpleFeatureType = result.schema
37     override def hasNext: Boolean = iter.hasNext
38     override def next(): SimpleFeature = iter.next()
39     override def close(): Unit = iter.close()
40   }

43 object GeoMesaFeatureReader extends MethodProfiling {
44 
45   def apply(
46       sft: SimpleFeatureType,
47       query: Query,
48       qp: QueryRunner,
49       audit: Option[AuditWriter]): GeoMesaFeatureReader = {
50     audit match {
51       case None => new GeoMesaFeatureReader(qp.runQuery(sft, query))
52       case Some(a) =>
53         // TODO consolidate auditing with metrics in QueryPlanner
54         val start = System.nanoTime()
55         val result = qp.runQuery(sft, query)
56         new GeoMesaFeatureReaderWithAudit(result, a, sft.getTypeName, query.getFilter, System.nanoTime() - start)
57     }
58   }
59 
60   trait HasGeoMesaFeatureReader {
61 
62     /**
63      * Internal method to get a feature reader without reloading the simple feature type. We don't expose this
64      * widely as we want to ensure that the sft has been loaded from our catalog
65      *
66      * @param sft simple feature type
67      * @param transaction transaction
68      * @param query query
69      * @return
70      */
71     private[geomesa] def getFeatureReader(
72         sft: SimpleFeatureType,
73         transaction: Transaction,
74         query: Query): GeoMesaFeatureReader
75   }
76 
77   private class GeoMesaFeatureReaderWithAudit(
78       result: QueryResult,
79       auditWriter: AuditWriter,
80       typeName: String,
81       filter: Filter,
82       initialTime: Long,
83     ) extends GeoMesaFeatureReader(result) {
84 
85     override def reader(): SimpleFeatureReader = new ResultReaderWithAudit()
86 
87     private class ResultReaderWithAudit extends SimpleFeatureReader with MethodProfiling {
88 
89       private var planningTime = initialTime
90       private var queryTime = 0L
91 
92       private val start = System.currentTimeMillis()
93       private val user = auditWriter.auditProvider.getCurrentUserId
94       private val closed = new AtomicBoolean(false)
95       private val count = new AtomicLong(0L)
96       private val iter = {
97         val start = System.nanoTime()
98         val base = result.iterator()
99         planningTime += (System.nanoTime() - start)
100         // note: has to be done after running the query so hints are set
101         if (result.schema == BinaryOutputEncoder.BinEncodedSft) {
102           // bin queries pack multiple records into each feature
103           // to count the records, we have to count the total bytes coming back, instead of the number of features
104           val bytesPerHit = if (result.hints.getBinLabelField.isDefined) { 24 } else { 16 }
105           base.map { sf => count.addAndGet(sf.getAttribute(0).asInstanceOf[Array[Byte]].length / bytesPerHit); sf }
106         } else {
107           base.map { sf => count.incrementAndGet(); sf }
108         }
109       }
110 
111       override def getFeatureType: SimpleFeatureType = result.schema
112 
113       override def hasNext: Boolean = {
114         val start = System.nanoTime()
115         val result = iter.hasNext
116         queryTime += (System.nanoTime() - start)
117         result
118       }
119       override def next(): SimpleFeature = {
120         val start = System.nanoTime()
121         val result = iter.next()
122         queryTime += (System.nanoTime() - start)
123         result
124       }
125 
126       override def close(): Unit = {
127         if (closed.compareAndSet(false, true)) {
128           try { iter.close() } finally {
129             val plans = result match {
130               case r: QueryPlanResult[_] => r.plans
131               case _ => Seq.empty
132             }
133             // note: implementations should be asynchronous
134             auditWriter.writeQueryEvent(typeName, user, filter, hints, plans, start, System.currentTimeMillis(),
135               planningTime / 1000000L, queryTime / 1000000L, count.get)
136           }
137         }
138       }
139     }
140   }
