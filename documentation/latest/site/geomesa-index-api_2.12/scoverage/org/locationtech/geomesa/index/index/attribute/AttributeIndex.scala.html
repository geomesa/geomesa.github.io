52   override val tieredKeySpace: Option[IndexKeySpace[_, _]] = {

81     if (attributes.isEmpty) { false } else {

91   def defaults(sft: SimpleFeatureType, index: NamedIndex, flagCheck: String => Boolean): Seq[IndexId] = {
92     sft.getAttributeDescriptors.asScala.toSeq.filter(AttributeIndexKey.encodable).flatMap { d =>
93       d.getIndexFlags.flatMap { flag =>
94         if (flag.equalsIgnoreCase(index.name) || flagCheck(flag)) {
95           Seq(IndexId(index.name, index.version, defaultTiers(sft, d)))
96         } else {
97           // check for name:attr[:attr] and name:version[:attr]
98           val Array(name, secondary@_*) = flag.split(":")
99           if (name.equalsIgnoreCase(index.name)) {
100             lazy val version = Try(secondary.head.toInt).toOption
101             if (secondary.isEmpty) {
102               Seq(IndexId(index.name, index.version, defaultTiers(sft, d)))
103             } else if (supports(sft, Seq(d.getLocalName) ++ secondary)) {
104               Seq(IndexId(index.name, index.version, Seq(d.getLocalName) ++ secondary))
105             } else if (version.isDefined && supports(sft, Seq(d.getLocalName) ++ secondary.tail)) {
106               Seq(IndexId(index.name, version.get, Seq(d.getLocalName) ++ secondary.tail))
107             } else {
108               throw new IllegalArgumentException(
109                 s"Attribute '${d.getLocalName}' is configured for indexing but index is not supported: $flag")
110             }
111           } else {
112             Seq.empty
113           }
114         }
115       }
116     }
117   }
118 
119   override def indexFor(sft: SimpleFeatureType, primary: AttributeDescriptor): Option[IndexId] = {

123       None
124     }

139       (i.name == name || i.name == JoinIndexName) && i.attributes.headOption.contains(attribute)
140     }
