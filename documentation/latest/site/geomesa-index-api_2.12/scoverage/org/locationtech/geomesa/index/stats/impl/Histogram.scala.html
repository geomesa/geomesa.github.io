24 /**
25   * The histogram's state is stored in an indexed array, where the index is the bin number
26   * and the values are the counts.
27   *
28   * e.g. a range of 0 to 3 with 3 bins will result in these bins: [0, 1), [1, 2), [2, 3) and the
29   * array will contain three entries.
30   *
31   * @param sft simple feature type
32   * @param property property name for the attribute the histogram is being made for
33   * @param initialBins number of bins the histogram has
34   * @param initialEndpoints lower/upper end of histogram
35   * @tparam T a comparable type which must have a StatHelperFunctions type class
36   */
37 class Histogram[T](
38     val sft: SimpleFeatureType,
39     val property: String,
40     initialBins: Int,
41     initialEndpoints: (T, T)
42   )(
43     implicit val defaults: MinMax.MinMaxDefaults[T],
44     val ct: ClassTag[T]
45   ) extends Stat with LazyLogging {
46

62   /**
63     * Copies another histogram into this one. In comparison to +=, this method will preserve the
64     * current bounds and length.

68   def addCountsFrom(other: Histogram[T]): Unit = {
69     if (length == other.length && bounds == other.bounds) {
70       this += other

76   override def observe(sf: SimpleFeature): Unit = {
77     val value = sf.getAttribute(i)
78     if (value != null) {
79       try {
80         val i = bins.indexOf(value.asInstanceOf[T])
81         if (i == -1) {
82           bins = Histogram.expandBins(value.asInstanceOf[T], bins)
83           bins.add(value.asInstanceOf[T])
84         } else {

87       } catch {
88         case e: Exception => logger.warn(s"Error observing value '$value': ${e.toString}")
89       }

121   /**
122     * Copies another histogram into this one.
123     * Current bounds and length will be expanded if necessary.
124     */
125   override def +=(other: Histogram[T]): Unit = {
126     if (length == other.length && bounds == other.bounds) {
127       // hists match - we can just copy counts in directly
128       var i = 0
129       while (i < bins.length) {
130         bins.counts(i) += other.bins.counts(i)
131         i += 1
132       }
133     } else if (other.isEmpty) {
134       // no-op
135     } else if (isEmpty) {
136       // copy the data from the other histogram
137       bins = BinnedArray(other.length, other.bounds)
138       var i = 0
139       while (i < bins.length) {
140         bins.counts(i) = other.bins.counts(i)
141         i += 1
142       }
143     } else {
144       // figure out the new bounds and size
145       val newEndpoints = Histogram.checkEndpoints(bins, other.bins)
146       val newLength = math.max(length, other.length)
147       if (newEndpoints != bounds || newLength != length) {
148         // if the other hist was not 'contained' in this one, we have to re-create the bins
149         val newBins = BinnedArray(newLength, newEndpoints)
150         Histogram.copyInto(newBins, bins)
151         bins = newBins
152       }
153       // now copy over the other bins
154       Histogram.copyInto(bins, other.bins)
155     }
156   }

184   /**
185     * Takes a single value and creates a range from it. Since the range histogram lower bound has to be
186     * strictly less than the upper bound, this lets us use a single value to create a valid histogram.
187     *
188     * @param value value to buffer
189     * @tparam T type of the value
190     * @return valid bounds for a histogram
191     */
192   def buffer[T](value: T): (T, T) = {
193     import org.locationtech.geomesa.utils.geotools.Conversions.RichGeometry
194     val buf = value match {
195       case v: Int    => (v - 100, v + 100)
196       case v: Long   => (v - 100, v + 100)
197       case v: Float  => (v - 100, v + 100)
198       case v: Double => (v - 100, v + 100)
199       case v: String => (s"$v$Base36Lowest", s"$v$Base36Highest")
200       case v: Date   => (new Date(v.getTime - 60000), new Date(v.getTime + 60000))
201       case v: Geometry =>
202         val env = v.safeCentroid().buffer(10.0).getEnvelopeInternal
203         val min = GeometryUtils.geoFactory.createPoint(new Coordinate(env.getMinX, env.getMinY))
204         val max = GeometryUtils.geoFactory.createPoint(new Coordinate(env.getMaxX, env.getMaxY))
205         (min, max)
206     }
207     buf.asInstanceOf[(T, T)]
208   }

222   /**
223     * Creates a new binned array that encompasses the new value.
224     *
225     * Assumes that the value is not already within the bounds for the existing binned array.
226     */
227   def expandBins[T](value: T, old: BinnedArray[T])(implicit defaults: MinMax.MinMaxDefaults[T],ct: ClassTag[T]): BinnedArray[T] = {
228     val min = defaults.min(value, old.bounds._1)
229     val max = defaults.max(value, old.bounds._2)
230     val bins = BinnedArray[T](old.length, (min, max))
231     copyInto(bins, old)
232     bins
233   }
234 
235   /**
236     * Gets new endpoints that encompass both arrays. If either array has empty values to start or end, the
237     * bounds will be trimmed down to be the start/end of non-empty values.
238     */
239   def checkEndpoints[T](left: BinnedArray[T], right: BinnedArray[T])(implicit defaults: MinMax.MinMaxDefaults[T]): (T, T) = {
240     val (lMin, lMax) = getActualBounds(left)
241     val (rMin, rMax) = getActualBounds(right)
242     (defaults.min(lMin, rMin), defaults.max(lMax, rMax))
243   }
244 
245   /**
246     * Gets the bounds of the array that actually contain values.
247     */
248   private def getActualBounds[T](bins: BinnedArray[T]): (T, T) = {
249     val minIndex = bins.counts.indexWhere(_ != 0)
250     val maxIndex = bins.counts.length - bins.counts.reverse.indexWhere(_ != 0) - 1
251     val min = if (minIndex <= 0) bins.bounds._1 else bins.bounds(minIndex)._1
252     val max = if (maxIndex >= bins.counts.length -1) bins.bounds._2 else bins.bounds(maxIndex)._2
253     (min, max)
254   }
255 
256   /**
257     * Copies data from one binned array into the other. Arrays are assumed to have different
258     * sizes and/or endpoints. If arrays have the same characteristics, this method is
259     * needlessly expensive/complicated/inexact.
260     */
261   def copyInto[T](to: BinnedArray[T], from: BinnedArray[T]): Unit = {
262     def toIndex(value: T): Int = {
263       val i = to.indexOf(value)
264       if (i != -1) i else if (to.isBelow(value)) 0 else to.length - 1
265     }
266 
267     var i = 0
268     while (i < from.length) {
269       val count = from.counts(i)
270       if (count > 0) {
271         val (min, max) = from.bounds(i)
272         val lo = toIndex(min)
273         val hi = toIndex(max)
274         if (lo == hi) {
275           to.counts(lo) += count
276         } else {
277           val size = hi - lo + 1
278           require(size > 0,
279             s"Error calculating bounds for ${min.getClass.getSimpleName} from ${from.bounds} to ${to.bounds}")
280           val avgCount = count / size
281           val remainingCount = count % size
282           val mid = lo + (size / 2)
283           var j = lo
284           while (j <= hi) {
285             to.counts(j) += avgCount
286             j += 1
287           }
288           to.counts(mid) += remainingCount
289         }
290       }
291       i += 1
292     }
293   }
