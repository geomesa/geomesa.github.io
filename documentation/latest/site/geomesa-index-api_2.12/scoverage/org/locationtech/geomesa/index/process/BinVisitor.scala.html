22 /**
23  * Binary format visitor
24  *
25  * @param sft simple feature type
26  * @param options encoding options
27  */
28 class BinVisitor(sft: SimpleFeatureType, options: EncodingOptions) extends GeoMesaProcessVisitor with LazyLogging {
29 
30   import scala.collection.JavaConverters._
31 
32   // for collecting results manually
33   private val manualResults = scala.collection.mutable.Queue.empty[Array[Byte]]
34   private val manualConversion = BinaryOutputEncoder(sft, options)
35 
36   private var result = new Iterator[Array[Byte]] {
37     override def next(): Array[Byte] = manualResults.dequeue()
38     override def hasNext: Boolean = manualResults.nonEmpty
39   }
40 
41   override def getResult: BinResult = BinResult(result.asJava)
42 
43   // manually called for non-accumulo feature collections
44   override def visit(feature: Feature): Unit =
45     manualResults += manualConversion.encode(feature.asInstanceOf[SimpleFeature])
46 
47   override def execute(source: SimpleFeatureSource, query: Query): Unit = {
48     logger.debug(s"Visiting source type: ${source.getClass.getName}")
49 
50     query.getHints.put(QueryHints.BIN_TRACK, options.trackIdField.map(sft.getDescriptor(_).getLocalName).getOrElse("id"))
51     options.geomField.foreach(i => query.getHints.put(QueryHints.BIN_GEOM, sft.getDescriptor(i).getLocalName))
52     options.dtgField.foreach(i => query.getHints.put(QueryHints.BIN_DTG, sft.getDescriptor(i).getLocalName))
53     options.labelField.foreach(i => query.getHints.put(QueryHints.BIN_LABEL, sft.getDescriptor(i).getLocalName))
54 
55     val features = SelfClosingIterator(source.getFeatures(query))
56     result ++= features.map(_.getAttribute(BIN_ATTRIBUTE_INDEX).asInstanceOf[Array[Byte]])
57   }
58 }
59 
60 object BinVisitor {
61   case class BinResult(results: java.util.Iterator[Array[Byte]]) extends AbstractCalcResult {
62     override def getValue: java.util.Iterator[Array[Byte]] = results
63   }
