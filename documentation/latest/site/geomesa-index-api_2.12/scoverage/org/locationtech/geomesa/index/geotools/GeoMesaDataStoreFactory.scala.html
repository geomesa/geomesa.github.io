184   class MetricsRegistryParam extends
185       GeoMesaParam[String](
186         "geomesa.metrics.registry",
187         "Specify the type of registry used to publish metrics. See " +
188           "https://www.geomesa.org/documentation/stable/user/appendix/metrics.html",
189         default = "none",
190         enumerations = Seq("none", RegistryFactory.Prometheus, RegistryFactory.Cloudwatch),
191       ) {
192 
193     /**
194      * Get the registry
195      *
196      * @param params params
197      * @return
198      */
199     def lookupRegistry(params: java.util.Map[String, _]): Option[MetricsConfig] = {
200       val registry = lookup(params) match {
201         case "none" => None
202         case RegistryFactory.Prometheus => Some(PrometheusFactory)
203         case RegistryFactory.Cloudwatch => Some(CloudwatchFactory)
204         case r => throw new IllegalArgumentException(s"Unknown registry type, expected one of 'none', 'prometheus' or 'cloudwatch': $r")
205       }
206       registry.map(r => MetricsConfig(r, GeoMesaDataStoreFactory.MetricsRegistryConfigParam.lookupOpt(params)))
207     }
