36 
37   class XZ2IndexKeySpaceV1(sft: SimpleFeatureType,
38                            override val sharing: Array[Byte],
39                            sharding: ShardStrategy,
40                            geomField: String) extends XZ2IndexKeySpace(sft, sharding, geomField) {
41 
42     private val rangePrefixes = {
43       if (sharding.length == 0 && sharing.isEmpty) {
44         Seq.empty
45       } else if (sharing.isEmpty) {
46         sharding.shards
47       } else if (sharding.length == 0) {
48         Seq(sharing)
49       } else {
50         sharding.shards.map(ByteArrays.concat(sharing, _))
51       }
52     }
53 
54     override val indexKeyByteLength: Right[(Array[Byte], Int, Int) => Int, Int] =
55       Right(8 + sharing.length + sharding.length)
56 
57     override def toIndexKey(writable: WritableFeature,
58                             tier: Array[Byte],
59                             id: Array[Byte],
60                             lenient: Boolean): RowKeyValue[Long] = {
61       val geom = writable.getAttribute[Geometry](geomIndex)
62       if (geom == null) {
63         throw new IllegalArgumentException(s"Null geometry in feature ${writable.feature.getID}")
64       }
65       val envelope = geom.getEnvelopeInternal
66       val xz = try { sfc.index(envelope.getMinX, envelope.getMinY, envelope.getMaxX, envelope.getMaxY, lenient) } catch {
67         case NonFatal(e) => throw new IllegalArgumentException(s"Invalid xz value from geometry: $geom", e)
68       }
69       val shard = sharding(writable)
70 
71       // create the byte array - allocate a single array up front to contain everything
72       // ignore tier, not used here
73       val bytes = Array.ofDim[Byte](sharing.length + shard.length + 8 + id.length)
74       var i = 0
75       if (!sharing.isEmpty) {
76         bytes(0) = sharing.head // sharing is only a single byte
77         i += 1
78       }
79       if (!shard.isEmpty) {
80         bytes(i) = shard.head // shard is only a single byte
81         i += 1
82       }
83       ByteArrays.writeLong(xz, bytes, i)
84       System.arraycopy(id, 0, bytes, i + 8, id.length)
85 
86       SingleRowKeyValue(bytes, sharing, shard, xz, tier, id, writable.values)
87     }
88 
89     override def getRangeBytes(ranges: Iterator[ScanRange[Long]], tier: Boolean): Iterator[ByteRange] = {
90       if (rangePrefixes.isEmpty) {
91         ranges.map {
92           case BoundedRange(lo, hi) => BoundedByteRange(ByteArrays.toBytes(lo), ByteArrays.toBytesFollowingPrefix(hi))
93           case r => throw new IllegalArgumentException(s"Unexpected range type $r")
94         }
95       } else {
96         ranges.flatMap {
97           case BoundedRange(lo, hi) =>
98             val lower = ByteArrays.toBytes(lo)
99             val upper = ByteArrays.toBytesFollowingPrefix(hi)
100             rangePrefixes.map(p => BoundedByteRange(ByteArrays.concat(p, lower), ByteArrays.concat(p, upper)))
101 
102           case r => throw new IllegalArgumentException(s"Unexpected range type $r")
103         }
104       }
105     }
106   }
