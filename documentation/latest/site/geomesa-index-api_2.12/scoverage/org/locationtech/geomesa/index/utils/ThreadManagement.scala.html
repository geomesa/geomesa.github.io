25 /**
26  * Singleton for registering and managing running queries.
27  */
28 object ThreadManagement {
29 
30   private val logger = Logger(LoggerFactory.getLogger(ThreadManagement.getClass.getName.replace("$", "")))
31 
32   private val executor = {
33     val ex = new ScheduledThreadPoolExecutor(2)
34     ex.setRemoveOnCancelPolicy(true)
35     ExitingExecutor(ex, force = true)
36   }
37 
38   /**
39    * Class for scans that are managed, i.e. tracked and terminated if they exceed a specified timeout
40    *
41    * @param underlying low-level scan to be stopped
42    * @param timeout scan timeout
43    * @param typeName type name, used for log messages
44    * @param filter query filter, used for log messages
45    * @tparam T type
46    */
47   class ManagedScan[T](underlying: LowLevelScanner[T], timeout: Timeout, typeName: String, filter: Option[Filter])
48       extends CloseableIterator[T] {
49 
50     def this(underlying: LowLevelScanner[T], timeout: Timeout, plan: QueryPlan[_]) =
51       this(underlying, timeout, plan.filter.index.sft.getTypeName, plan.filter.filter)
52 
53     // we can use a volatile var since we only update the value with a single thread
54     @volatile
55     private var terminated = timeout.absolute <= System.currentTimeMillis()
56 
57     private val iter = ExceptionalIterator(if (terminated) { Iterator.empty } else { underlying.iterator })
58     private val cancel = if (terminated) { None } else {
59       val timeout = math.max(1, this.timeout.absolute - System.currentTimeMillis())
60       Some(executor.schedule(new QueryKiller(this), timeout, TimeUnit.MILLISECONDS))
61     }
62 
63     // note: check iter.hasNext first so we get updated terminated flag
64     override def hasNext: Boolean = iter.hasNext || terminated
65 
66     override def next(): T = {
67       if (terminated) {
68         val e = new RuntimeException(s"Scan terminated due to timeout of ${timeout.relative}ms")
69         iter.suppressed.foreach(e.addSuppressed)
70         throw e
71       } else {
72         iter.next()
73       }
74     }
75 
76     /**
77      * Forcibly terminate the scan
78      */
79     def terminate(): Unit = {
80       terminated = true
81       try {
82         logger.warn(
83           s"Stopping scan on schema '$typeName' with filter '${filterToString(filter)}' " +
84               s"based on timeout of ${timeout.relative}ms")
85         underlying.close()
86       } catch {
87         case NonFatal(e) => logger.warn("Error cancelling scan:", e)
88       }
89     }
90 
91     /**
92      * Was the scan terminated due to timeout
93      *
94      * @return
95      */
96     def isTerminated: Boolean = terminated
97 
98     override def close(): Unit = {
99       cancel.foreach(_.cancel(false))
100       // if terminated, we've already closed the iterator
101       if (!terminated) {
102         underlying.close()
103       }
104     }
105   }
106 
107   /**
108    * Low level scanner that can be closed to terminate a scan
109    *
110    * @tparam T type
111    */
112   trait LowLevelScanner[T] extends Closeable {
113     def iterator: Iterator[T]
114   }
115 
116   /**
117    * Timeout holder
118    *
119    * @param relative relative timeout, in millis
120    * @param absolute absolute timeout, in system millis since epoch
121    */
122   case class Timeout(relative: Long, absolute: Long)
123 
124   object Timeout {
125     def apply(relative: Long): Timeout = Timeout(relative, System.currentTimeMillis() + relative)
126     def apply(relative: Duration): Timeout = apply(relative.toMillis)
127     def apply(relative: String): Timeout = apply(Duration(relative))
128   }
129 
130   /**
131    * Runnable to handle terminating a scan
132    *
133    * @param scan scan to terminate
134    */
135   private class QueryKiller(val scan: ManagedScan[_]) extends Runnable {
136     override def run(): Unit = scan.terminate()
137   }
