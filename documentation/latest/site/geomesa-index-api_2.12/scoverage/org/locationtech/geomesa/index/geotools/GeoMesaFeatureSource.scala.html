38   /**
39     * The default behavior for getCount is to use estimated statistics if available, or -1 to indicate
40     * that the operation would be expensive (@see org.geotools.api.data.FeatureSource#getCount(org.geotools.api.data.Query)).
41     *
42     * Since users may want <b>exact</b> counts, there are two ways to force exact counts:
43     *   1. use the system property "geomesa.force.count"
44     *   2. use the EXACT_COUNT query hint
45     *
46     * @param query query
47     * @return
48     */
49   override def getCount(query: Query): Int = {
50     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
51     import org.locationtech.geomesa.index.conf.QueryProperties.QueryExactCount
52 
53     // configure the query hints
54     val hints = ds.getFeatureReader(sft, Transaction.AUTO_COMMIT, query).hints
55     val useExactCount = hints.isExactCount.getOrElse(QueryExactCount.get.toBoolean)
56 
57     val count = if (useExactCount &&
58       !query.isMaxFeaturesUnlimited &&
59       query.getMaxFeatures < QueryExactCountMaxFeatures.get.toInt) {
60       SelfClosingIterator(getFeatures(query)).size
61     } else {
62       val statsCount = ds.stats.getCount(getSchema, query.getFilter, useExactCount, hints).getOrElse(-1L)
63       if (query.isMaxFeaturesUnlimited) {
64         statsCount
65       } else {
66         math.min(statsCount, query.getMaxFeatures)
67       }
68     }
69 
70     if (count > Int.MaxValue) {
71       logger.warn(s"Truncating count $count to Int.MaxValue (${Int.MaxValue})")
72       Int.MaxValue
73     } else {
74       count.toInt
75     }
76   }
