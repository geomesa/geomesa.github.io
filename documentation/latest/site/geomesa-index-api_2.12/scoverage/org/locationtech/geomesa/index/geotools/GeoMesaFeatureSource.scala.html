40   /**
41     * The default behavior for getCount is to use estimated statistics if available, or -1 to indicate
42     * that the operation would be expensive (@see org.geotools.api.data.FeatureSource#getCount(org.geotools.api.data.Query)).
43     *
44     * Since users may want <b>exact</b> counts, there are two ways to force exact counts:
45     *   1. use the system property "geomesa.force.count"
46     *   2. use the EXACT_COUNT query hint
47     *
48     * @param query query
49     * @return
50     */
51   override def getCount(query: Query): Int = {
52     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
53     import org.locationtech.geomesa.index.conf.QueryProperties.QueryExactCount
54 
55     // configure the query hints
56     val hints = QueryRunner.configureQuery(sft, query).getHints
57     val useExactCount = hints.isExactCount.getOrElse(QueryExactCount.get.toBoolean)
58 
59     val count = if (useExactCount && hints.getMaxFeatures.exists(_ < QueryExactCountMaxFeatures.get.toInt)) {
60       CloseableIterator(getFeatures(query).features()).size
61     } else {
62       val statsCount = ds.stats.getCount(getSchema, query.getFilter, useExactCount, hints).getOrElse(-1L)
63       hints.getMaxFeatures match {
64         case None => statsCount
65         case Some(m) => if (statsCount < 0) { m } else { math.min(statsCount, m) }
66       }
67     }
68 
69     if (count > Int.MaxValue) {
70       logger.warn(s"Truncating count $count to Int.MaxValue (${Int.MaxValue})")
71       Int.MaxValue
72     } else {
73       count.toInt
74     }
75   }
