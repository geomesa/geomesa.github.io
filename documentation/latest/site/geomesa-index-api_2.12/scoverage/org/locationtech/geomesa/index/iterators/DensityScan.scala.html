91   /**
92     * Encodes a sparse matrix into a byte array
93     */
94   def encodeResult(result: RenderingGrid): Array[Byte] = {
95     val output = KryoFeatureSerialization.getOutput(null)
96     result.iterator.toList.groupBy(_._1._1).foreach { case (row, cols) =>
97       output.writeInt(row, true)
98       output.writeInt(cols.size, true)
99       cols.foreach { case (xy, weight) =>
100         output.writeInt(xy._2, true)
101         output.writeDouble(weight)
102       }
103     }
104     output.toBytes
105   }

114   /**
115     * Decodes a result feature into an iterator of (x, y, weight)
116     */
117   def decodeResult(gridSnap: GridSnap)(sf: SimpleFeature): Iterator[(Double, Double, Double)] = {
118     val result = sf.getUserData.get(DensityValueKey).asInstanceOf[Array[Byte]]
119     val input = KryoFeatureDeserialization.getInput(result, 0, result.length)
120     new Iterator[(Double, Double, Double)]() {
121       private var x = 0.0
122       private var colCount = 0
123       override def hasNext: Boolean = input.position < input.limit
124       override def next(): (Double, Double, Double) = {
125         if (colCount == 0) {
126           x = gridSnap.x(input.readInt(true))
127           colCount = input.readInt(true)
128         }
129         val y = gridSnap.y(input.readInt(true))
130         val weight = input.readDouble()
131         colCount -= 1
132         (x, y, weight)
133       }
134     }
135   }

150   /**
151    * Gets a renderer for the associated geometry binding
152    *
153    * @param sft simple feature type
154    * @param geom geometry field to render
155    * @param weight field to use for weighting features
156    * @return
157    */
158   def getRenderer(sft: SimpleFeatureType, geom: String, weight: Option[String]): GeometryRenderer = {
159     // function to get the weight from the feature - defaults to 1.0 unless an attribute/exp is specified
160     val weigher = weight match {
161       case None => EqualWeight
162       case Some(w) =>
163         val i = sft.indexOf(w)
164         if (i == -1) {
165           new WeightByExpression(ECQL.toExpression(w))
166         } else if (classOf[Number].isAssignableFrom(sft.getDescriptor(i).getType.getBinding)) {
167           new WeightByNumber(i)
168         } else {
169           new WeightByNonNumber(i)
170         }
171     }
172 
173     val i = sft.indexOf(geom)
174     sft.getDescriptor(i).getType.getBinding match {
175       case b if b == classOf[Point]           => new PointRenderer(i, weigher)
176       case b if b == classOf[MultiPoint]      => new MultiPointRenderer(i, weigher)
177       case b if b == classOf[LineString]      => new LineStringRenderer(i, weigher)
178       case b if b == classOf[MultiLineString] => new MultiLineStringRenderer(i, weigher)
179       case b if b == classOf[Polygon]         => new PolygonRenderer(i, weigher)
180       case b if b == classOf[MultiPolygon]    => new MultiPolygonRenderer(i, weigher)
181       case _                                  => new MultiRenderer(i, weigher)
182     }
183   }

239   /**

242   private sealed trait Weigher {
243     def weight(sf: SimpleFeature): Double
244   }
245 
246   private case object EqualWeight extends Weigher {
247     override def weight(sf: SimpleFeature): Double = 1d
248   }
249 
250   /**
251     * Gets the weight for a feature from a numeric attribute
252     */
253   private class WeightByNumber(i: Int) extends Weigher {
254     override def weight(sf: SimpleFeature): Double = {
255       val d = sf.getAttribute(i).asInstanceOf[Number]
256       if (d == null) { 0.0 } else { d.doubleValue }
257     }
258   }
259 
260   /**
261     * Tries to convert a non-double attribute into a double
262     */
263   private class WeightByNonNumber(i: Int) extends Weigher {
264     override def weight(sf: SimpleFeature): Double = {
265       val d = sf.getAttribute(i)
266       if (d == null) { 0.0 } else {
267         val converted = FastConverter.convert(d, classOf[java.lang.Double])
268         if (converted == null) { 1.0 } else { converted.doubleValue() }
269       }
270     }
271   }
272 
273   /**
274     * Evaluates an arbitrary expression against the simple feature to return a weight
275     */
276   private class WeightByExpression(e: Expression) extends Weigher {
277     override def weight(sf: SimpleFeature): Double = {
278       val d = e.evaluate(sf, classOf[java.lang.Double])
279       if (d == null) { 0.0 } else { d }
280     }
281   }
