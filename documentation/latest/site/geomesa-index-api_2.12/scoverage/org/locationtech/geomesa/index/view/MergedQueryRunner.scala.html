53   override def runQuery(sft: SimpleFeatureType, original: Query, explain: Explainer): QueryResult = {
54     // TODO deduplicate arrow, bin, density queries...
55     // get view params and threaded query hints
56     val query = configureQuery(sft, original)
57     val hints = query.getHints
58     val maxFeatures = if (query.isMaxFeaturesUnlimited) { None } else { Option(query.getMaxFeatures) }
59 
60     if (hints.isStatsQuery || hints.isArrowQuery) {
61       // for stats and arrow queries, suppress the reduce step for gm stores so that we can do the merge here
62       hints.put(QueryHints.Internal.SKIP_REDUCE, java.lang.Boolean.TRUE)
63     }
64 
65     if (hints.isArrowQuery) {
66       QueryResult(ArrowEncodedSft, hints, () => arrowQuery(sft, query))
67     } else {
68       // query each delegate store
69       lazy val readers = stores.map { case (store, filter) =>
70         // make sure to coy the hints so they aren't shared
71         store.getFeatureReader(mergeFilter(sft, new Query(query), filter), Transaction.AUTO_COMMIT)
72       }
73 
74       if (hints.isDensityQuery) {
75         QueryResult(DensityScan.DensitySft, hints, () => densityQuery(sft, readers, hints))
76       } else if (hints.isStatsQuery) {
77         QueryResult(StatsScan.StatsSft, hints, () => statsQuery(sft, readers, hints))
78       } else if (hints.isBinQuery) {
79         if (query.getSortBy != null && !query.getSortBy.isEmpty) {
80           logger.warn("Ignoring sort for BIN query")
81         }
82         QueryResult(BinaryOutputEncoder.BinEncodedSft, hints, () => binQuery(sft, readers, hints))
83       } else {
84         val resultSft = QueryPlanner.extractQueryTransforms(sft, query).map(_._1).getOrElse(sft)
85         def run(): CloseableIterator[SimpleFeature] = {
86           val iters =
87             if (deduplicate) {
88               // we re-use the feature id cache across readers
89               val cache = scala.collection.mutable.HashSet.empty[String]
90               readers.map(r => new DeduplicatingSimpleFeatureIterator(SelfClosingIterator(r), cache))
91             } else {
92               readers.map(SelfClosingIterator(_))
93             }
94 
95           val results = Option(query.getSortBy).filterNot(_.isEmpty) match {
96             case None => SelfClosingIterator(iters.iterator).flatMap(i => i)
97             // the delegate stores should sort their results, so we can sort merge them
98             case Some(sort) => new SortedMergeIterator(iters)(SimpleFeatureOrdering(resultSft, sort))
99           }
100 
101           maxFeatures match {
102             case None => results
103             case Some(m) => results.take(m)
104           }
105         }
106         QueryResult(resultSft, hints, run)
107       }
108     }
109   }

126   private def arrowQuery(sft: SimpleFeatureType, query: Query): CloseableIterator[SimpleFeature] = {
127     val hints = query.getHints
128 
129     // handle any sorting here
130     QueryPlanner.setQuerySort(sft, query)
131 
132     val arrowSft = QueryPlanner.extractQueryTransforms(sft, query).map(_._1).getOrElse(sft)
133     val sort = hints.getArrowSort
134     val batchSize = ArrowScan.getBatchSize(hints)
135     val encoding = SimpleFeatureEncoding.min(hints.isArrowIncludeFid, hints.isArrowProxyFid, hints.isFlipAxisOrder)
136     val ipcOpts = FormatVersion.options(hints.getArrowFormatVersion.getOrElse(FormatVersion.ArrowFormatVersion.get))
137 
138     val dictionaryFields = hints.getArrowDictionaryFields
139     // do the reduce here, as we can't merge finalized arrow results
140 
141     val process = hints.isArrowProcessDeltas
142     val reduce = new ArrowScan.DeltaReducer(arrowSft, dictionaryFields, encoding, ipcOpts, batchSize, sort, sorted = false, process)
143 
144     // now that we have standardized dictionaries, we can query the delegate stores
145     val readers = stores.map { case (store, filter) =>
146       // copy the query so hints aren't shared
147       store.getFeatureReader(mergeFilter(sft, new Query(query), filter), Transaction.AUTO_COMMIT)
148     }
149 
150     def getSingle(reader: FeatureReader[SimpleFeatureType, SimpleFeature]): CloseableIterator[SimpleFeature] = {
151       val schema = reader.getFeatureType
152       if (schema == org.locationtech.geomesa.arrow.ArrowEncodedSft) {
153         // arrow processing has been handled by the store already
154         CloseableIterator(reader)
155       } else {
156         // the store just returned normal features, do the arrow processing here
157         val copy = SimpleFeatureTypes.immutable(schema, sft.getUserData) // copy default dtg, etc if necessary
158         // note: we don't need to pass in the transform or filter, as the transform should have already been
159         // applied and the dictionaries calculated up front (if needed)
160         LocalQueryRunner.transform(copy, CloseableIterator(reader), None, hints)
161       }
162     }
163 
164     reduce(doParallelScan(readers, getSingle))
165   }

224   override protected [geomesa] def getReturnSft(sft: SimpleFeatureType, hints: Hints): SimpleFeatureType = {
225     if (hints.isBinQuery) {

227     } else if (hints.isArrowQuery) {

238   private def doParallelScan(
239       readers: Seq[FeatureReader[SimpleFeatureType, SimpleFeature]],
240       single: FeatureReader[SimpleFeatureType, SimpleFeature] => CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = {
241     if (parallel) {
242       // not truly parallel but should kick them all off up front
243       SelfClosingIterator(readers.toList.map(single).iterator).flatMap(i => i)
244     } else {
245       SelfClosingIterator(readers.iterator).flatMap(single)
246     }
247   }
