28 class XZ2IndexKeySpace(val sft: SimpleFeatureType, val sharding: ShardStrategy, geomField: String)
29     extends IndexKeySpace[XZ2IndexValues, Long] {
30 
31   import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
32 
33   require(classOf[Geometry].isAssignableFrom(sft.getDescriptor(geomField).getType.getBinding),
34     s"Expected field $geomField to have a geometry binding, but instead it has: " +
35         sft.getDescriptor(geomField).getType.getBinding.getSimpleName)
36 
37   protected val geomIndex: Int = sft.indexOf(geomField)
38   protected val sfc = XZ2SFC(sft.getXZPrecision)
39 
40   private val isPoints = sft.getDescriptor(geomIndex).getType.getBinding == classOf[Point]
41 
42   override val attributes: Seq[String] = Seq(geomField)
43 
44   override val indexKeyByteLength: Right[(Array[Byte], Int, Int) => Int, Int] = Right(8 + sharding.length)
45 
46   override val sharing: Array[Byte] = Array.empty
47 
48   override def toIndexKey(writable: WritableFeature,
49                           tier: Array[Byte],
50                           id: Array[Byte],
51                           lenient: Boolean): RowKeyValue[Long] = {
52     val geom = writable.getAttribute[Geometry](geomIndex)
53     if (geom == null) {
54       throw new IllegalArgumentException(s"Null geometry in feature ${writable.feature.getID}")
55     }
56     val envelope = geom.getEnvelopeInternal
57     val xz = try { sfc.index(envelope.getMinX, envelope.getMinY, envelope.getMaxX, envelope.getMaxY, lenient) } catch {
58       case NonFatal(e) => throw new IllegalArgumentException(s"Invalid xz value from geometry: $geom", e)
59     }
60     val shard = sharding(writable)
61 
62     // create the byte array - allocate a single array up front to contain everything
63     // ignore tier, not used here
64     val bytes = Array.ofDim[Byte](shard.length + 8 + id.length)
65 
66     if (shard.isEmpty) {
67       ByteArrays.writeLong(xz, bytes, 0)
68       System.arraycopy(id, 0, bytes, 8, id.length)
69     } else {
70       bytes(0) = shard.head // shard is only a single byte
71       ByteArrays.writeLong(xz, bytes, 1)
72       System.arraycopy(id, 0, bytes, 9, id.length)
73     }
74 
75     SingleRowKeyValue(bytes, sharing, shard, xz, tier, id, writable.values)
76   }
77 
78   override def getIndexValues(filter: Filter, explain: Explainer): XZ2IndexValues = {
79 
80     val geometries: FilterValues[Geometry] = {
81       val extracted = FilterHelper.extractGeometries(filter, geomField, isPoints)
82       if (extracted.nonEmpty) { extracted } else { FilterValues(Seq(WholeWorldPolygon)) }
83     }
84 
85     explain(s"Geometries: $geometries")
86 
87     // compute our ranges based on the coarse bounds for our query
88     val xy: Seq[(Double, Double, Double, Double)] = {
89       val multiplier = QueryProperties.PolygonDecompMultiplier.toInt.get
90       val bits = QueryProperties.PolygonDecompBits.toInt.get
91       geometries.values.flatMap(GeometryUtils.bounds(_, multiplier, bits))
92     }
93 
94     XZ2IndexValues(sfc, geometries, xy)
95   }
96 
97   override def getRanges(values: XZ2IndexValues, multiplier: Int): Iterator[ScanRange[Long]] = {
98     val XZ2IndexValues(sfc, _, xy) = values
99     // note: `target` will always be Some, as ScanRangesTarget has a default value
100     val target = QueryProperties.ScanRangesTarget.option.map(t => math.max(1, t.toInt / multiplier))
101     sfc.ranges(xy, target).iterator.map(r => BoundedRange(r.lower, r.upper))
102   }
103 
104   override def getRangeBytes(ranges: Iterator[ScanRange[Long]], tier: Boolean): Iterator[ByteRange] = {
105     if (sharding.length == 0) {
106       ranges.map {
107         case BoundedRange(lo, hi) => BoundedByteRange(ByteArrays.toBytes(lo), ByteArrays.toBytesFollowingPrefix(hi))
108         case r => throw new IllegalArgumentException(s"Unexpected range type $r")
109       }
110     } else {
111       ranges.flatMap {
112         case BoundedRange(lo, hi) =>
113           val lower = ByteArrays.toBytes(lo)
114           val upper = ByteArrays.toBytesFollowingPrefix(hi)
115           sharding.shards.map(p => BoundedByteRange(ByteArrays.concat(p, lower), ByteArrays.concat(p, upper)))
116 
117         case r => throw new IllegalArgumentException(s"Unexpected range type $r")
118       }
119     }
120   }
121 
122   // always apply the full filter to xz queries
123   override def useFullFilter(values: Option[XZ2IndexValues],
124                              config: Option[GeoMesaDataStoreConfig],
125                              hints: Hints): Boolean = true
126 }

128 object XZ2IndexKeySpace extends IndexKeySpaceFactory[XZ2IndexValues, Long] {
129 
130   override def supports(sft: SimpleFeatureType, attributes: Seq[String]): Boolean =
131     attributes.lengthCompare(1) == 0 && sft.indexOf(attributes.head) != -1 &&
132         classOf[Geometry].isAssignableFrom(sft.getDescriptor(attributes.head).getType.getBinding)
133 
134   override def apply(sft: SimpleFeatureType, attributes: Seq[String], tier: Boolean): XZ2IndexKeySpace = {
135     val shards = if (tier) { NoShardStrategy } else { Z2ShardStrategy(sft) }
136     new XZ2IndexKeySpace(sft, shards, attributes.head)
137   }
