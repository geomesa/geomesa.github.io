29 class Z2IndexKeySpace(val sft: SimpleFeatureType, val sharding: ShardStrategy, geomField: String)
30     extends IndexKeySpace[Z2IndexValues, Long] {
31 
32   require(classOf[Point].isAssignableFrom(sft.getDescriptor(geomField).getType.getBinding),
33     s"Expected field $geomField to have a point binding, but instead it has: " +
34         sft.getDescriptor(geomField).getType.getBinding.getSimpleName)
35 
36   protected val sfc: Z2SFC = Z2SFC
37 
38   protected val geomIndex: Int = sft.indexOf(geomField)
39 
40   override val attributes: Seq[String] = Seq(geomField)
41 
42   override val indexKeyByteLength: Right[(Array[Byte], Int, Int) => Int, Int] = Right(8 + sharding.length)
43 
44   override val sharing: Array[Byte] = Array.empty
45 
46   override def toIndexKey(writable: WritableFeature,
47                           tier: Array[Byte],
48                           id: Array[Byte],
49                           lenient: Boolean): RowKeyValue[Long] = {
50     val geom = writable.getAttribute[Point](geomIndex)
51     if (geom == null) {
52       throw new IllegalArgumentException(s"Null geometry in feature ${writable.feature.getID}")
53     }
54     val z = try { sfc.index(geom.getX, geom.getY, lenient) } catch {
55       case NonFatal(e) => throw new IllegalArgumentException(s"Invalid z value from geometry: $geom", e)
56     }
57     val shard = sharding(writable)
58 
59     // create the byte array - allocate a single array up front to contain everything
60     // ignore tier, not used here
61     val bytes = Array.ofDim[Byte](shard.length + 8 + id.length)
62 
63     if (shard.isEmpty) {
64       ByteArrays.writeLong(z, bytes, 0)
65       System.arraycopy(id, 0, bytes, 8, id.length)
66     } else {
67       bytes(0) = shard.head // shard is only a single byte
68       ByteArrays.writeLong(z, bytes, 1)
69       System.arraycopy(id, 0, bytes, 9, id.length)
70     }
71 
72     SingleRowKeyValue(bytes, sharing, shard, z, tier, id, writable.values)
73   }
74 
75   override def getIndexValues(filter: Filter, explain: Explainer): Z2IndexValues = {
76 
77     val geometries: FilterValues[Geometry] = {
78       val extracted = FilterHelper.extractGeometries(filter, geomField, intersect = true) // intersect since we have points
79       if (extracted.nonEmpty) { extracted } else { FilterValues(Seq(WholeWorldPolygon)) }
80     }
81 
82     explain(s"Geometries: $geometries")
83 
84     if (geometries.disjoint) {
85       explain("Non-intersecting geometries extracted, short-circuiting to empty query")
86       return Z2IndexValues(sfc, geometries, Seq.empty)
87     }
88 
89     // compute our ranges based on the coarse bounds for our query
90     val xy: Seq[(Double, Double, Double, Double)] = {
91       val multiplier = QueryProperties.PolygonDecompMultiplier.toInt.get
92       val bits = QueryProperties.PolygonDecompBits.toInt.get
93       geometries.values.flatMap(GeometryUtils.bounds(_, multiplier, bits))
94     }
95 
96     Z2IndexValues(sfc, geometries, xy)
97   }
98 
99   override def getRanges(values: Z2IndexValues, multiplier: Int): Iterator[ScanRange[Long]] = {
100     val Z2IndexValues(_, geoms, xy) = values
101     if (geoms.disjoint) {
102       Iterator.empty
103     } else {
104       // note: `target` will always be Some, as ScanRangesTarget has a default value
105       val target = QueryProperties.ScanRangesTarget.option.map(t => math.max(1, t.toInt / multiplier))
106       sfc.ranges(xy, 64, target).iterator.map(r => BoundedRange(r.lower, r.upper))
107     }
108   }
109 
110   override def getRangeBytes(ranges: Iterator[ScanRange[Long]], tier: Boolean): Iterator[ByteRange] = {
111     if (sharding.length == 0) {
112       ranges.map {
113         case BoundedRange(lo, hi) => BoundedByteRange(ByteArrays.toBytes(lo), ByteArrays.toBytesFollowingPrefix(hi))
114         case r => throw new IllegalArgumentException(s"Unexpected range type $r")
115       }
116     } else {
117       ranges.flatMap {
118         case BoundedRange(lo, hi) =>
119           val lower = ByteArrays.toBytes(lo)
120           val upper = ByteArrays.toBytesFollowingPrefix(hi)
121           sharding.shards.map(p => BoundedByteRange(ByteArrays.concat(p, lower), ByteArrays.concat(p, upper)))
122         case r => throw new IllegalArgumentException(s"Unexpected range type $r")
123       }
124     }
125   }
126 
127   override def useFullFilter(values: Option[Z2IndexValues],
128                              config: Option[GeoMesaDataStoreConfig],
129                              hints: Hints): Boolean = {
130     // if the user has requested strict bounding boxes, we apply the full filter
131     // if the spatial predicate is rectangular (e.g. a bbox), the index is fine enough that we
132     // don't need to apply the filter on top of it. this may cause some minor errors at extremely
133     // fine resolutions, but the performance is worth it
134     // if we have a complicated geometry predicate, we need to pass it through to be evaluated
135     val looseBBox = Option(hints.get(LOOSE_BBOX)).map(Boolean.unbox).getOrElse(config.forall(_.queries.looseBBox))
136     lazy val simpleGeoms = values.toSeq.flatMap(_.geometries.values).forall(GeometryUtils.isRectangular)
137 
138     !looseBBox || !simpleGeoms
139   }
140 }
