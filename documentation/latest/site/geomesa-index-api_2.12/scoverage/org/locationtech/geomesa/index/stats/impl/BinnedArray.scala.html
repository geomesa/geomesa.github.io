163       val clas = implicitly[ClassTag[T]].runtimeClass
164       val binning = if (clas == classOf[String]) {

166       } else if (clas == classOf[Integer]) {

168       } else if (clas == classOf[jLong]) {

185   abstract class WholeNumberBinning[T](length: Int, bounds: (T, T)) extends Binning[T](length, bounds) {
186 
187     private val min = convertToLong(bounds._1)
188     private val max = convertToLong(bounds._2)
189 
190     require(min < max,
191       s"Upper bound must be greater than lower bound: lower='${bounds._1}'($min) upper='${bounds._2}'($max)")
192 
193     private val binSize = (max - min).toDouble / length
194 
195     override def directIndex(value: Long): Int = {
196       if (value < min || value > max) { -1 } else {
197         val i = math.floor((value - min) / binSize).toInt
198         // i == length check catches the upper bound
199         if (i < 0 || i > length) -1 else if (i == length) length - 1 else i
200       }
201     }
202 
203     override def indexOf(value: T): Int = directIndex(convertToLong(value))
204 
205     override def medianValue(index: Int): T = {
206       if (index < 0 || index > length) {
207         throw new ArrayIndexOutOfBoundsException(index)
208       }
209       val long = min + math.round(binSize / 2 + binSize * index)
210       if (long > max) bounds._2 else convertFromLong(long)
211     }
212 
213     override def bounds(index: Int): (T, T) = {
214       if (index < 0 || index > length) {
215         throw new ArrayIndexOutOfBoundsException(index)
216       }
217       val loLong = min + math.ceil(binSize * index).toLong
218       val hiLong = math.max(loLong, min + math.floor(binSize * (index + 1)).toLong)
219       val lo = if (loLong <= min) bounds._1 else convertFromLong(loLong)
220       val hi = if (hiLong >= max) bounds._2 else convertFromLong(hiLong)
221 
222       (lo, hi)
223     }
224 
225     override def isBelow(value: T): Boolean = convertToLong(value) < min
226 
227     /**
228      * Maps a value to a long used to allocate values in bins
229      *
230      * @param value value to convert
231      * @return value as a long
232      */
233     protected def convertToLong(value: T): Long
234 
235     /**
236      * Maps a long back to a value
237      *
238      * @param value value as a long
239      * @return value
240      */
241     protected def convertFromLong(value: Long): T
242   }

351   /**
352    * Bins strings. Will set up bins based on the longest-common-prefix of the bounds. Estimates
353    * bins by considering inputs to be roughly equivalent to base36 longs.
354    *
355    * @param length number of bins
356    * @param rawBounds upper and lower bounds for the input values
357    */
358   class StringBinning(length: Int, rawBounds: (String, String))
359       extends WholeNumberBinning[String](length, StringBinning.normalizeBounds(rawBounds)) {
360 
361     import StringBinning.{Base36Lowest, normalize}
362 
363     private lazy val (start, end): (String, String) = bounds
364     private lazy val normalizedLength = start.length
365     private lazy val prefixLength = start.zip(end).indexWhere { case (l, r) => l != r }
366     private lazy val prefix = start.substring(0, prefixLength)
367 
368     override protected def convertToLong(value: String): Long = {
369       val normalized = normalize(value).padTo(normalizedLength, '0')
370       if (normalized < start) { 0L } else if (normalized > end) { Long.MaxValue } else {
371         // note: 12 is the most base-36 numbers we can fit in Long.MaxValue
372         val sigDigits = normalized.substring(prefixLength).padTo(12, Base36Lowest).substring(0, 12)
373         jLong.parseLong(sigDigits, 36)
374       }
375     }
376 
377     override protected def convertFromLong(value: Long): String =
378       prefix + jLong.toString(value, 36).reverse.padTo(12, Base36Lowest).reverse.replaceFirst("0+$", "")
379   }

381   object StringBinning {
382 
383     private val Base36Chars = (0 until 36).map(Integer.toString(_, 36).toLowerCase(Locale.US).charAt(0)).toArray
384     val Base36Lowest: Char  = Base36Chars.head
385     val Base36Highest: Char = Base36Chars.last
386 
387     def normalize(s: String): String = s.toLowerCase(Locale.US).replaceAll("[^0-9a-z]", Base36Lowest.toString)
388 
389     def normalizeBounds(bounds: (String, String)): (String, String) = {
390       val length = math.max(bounds._1.length, bounds._2.length)
391       val loBase36 = normalize(bounds._1)
392       val hiBase36 = normalize(bounds._2)
393       // pad them and make sure they are in sorted order
394       val (loPadded, hiPadded) = if (loBase36 < hiBase36) {
395         (loBase36.padTo(length, Base36Lowest), hiBase36.padTo(length, Base36Highest))
396       } else {
397         (hiBase36.padTo(length, Base36Lowest), loBase36.padTo(length, Base36Highest))
398       }
399       // ensure that they haven't become the same string
400       val (loDistinct, hiDistinct) = if (loPadded == hiPadded) {
401         (loPadded + Base36Lowest, hiPadded + Base36Highest)
402       } else {
403         (loPadded, hiPadded)
404       }
405       // check to make sure they fit in a long (12 chars)
406       val prefixLength = loDistinct.zip(hiDistinct).indexWhere { case (l, r) => l != r }
407       val loFit = if (loDistinct.length > prefixLength + 12) loDistinct.take(prefixLength + 12) else loDistinct
408       val hiFit = if (hiDistinct.length > prefixLength + 12) hiDistinct.take(prefixLength + 12) else hiDistinct
409       // check one last time that they aren't the same
410       if (loFit == hiFit) {
411         (loFit.dropRight(1) + Base36Lowest, hiFit.dropRight(1) + Base36Highest)
412       } else {
413         (loFit, hiFit)
414       }
415     }
