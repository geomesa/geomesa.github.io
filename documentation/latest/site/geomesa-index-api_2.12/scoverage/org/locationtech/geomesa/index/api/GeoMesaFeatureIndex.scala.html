253   /**
254     * Plans the query
255     *
256     * @param filter filter strategy
257     * @param hints query hints
258     * @param explain explainer
259     * @return
260     */
261   def getQueryStrategy(filter: FilterStrategy, hints: Hints, explain: Explainer = ExplainNull): QueryStrategy = {
262 
263     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
264 
265     val sharing = keySpace.sharing
266     val indexValues = filter.primary.map(keySpace.getIndexValues(_, explain))
267 
268     val useFullFilter = keySpace.useFullFilter(indexValues, Some(ds.config), hints)
269     val ecql = if (useFullFilter) { filter.filter } else { filter.secondary }
270 
271     indexValues match {
272       case None =>
273         if (filter.filter.exists(_.isInstanceOf[ExcludeFilter])) {
274           QueryStrategy(filter, Seq.empty, Seq.empty, Seq.empty, ecql, hints, indexValues)
275         } else {
276           // check that full table scans are allowed
277           if (hints.getMaxFeatures.forall(_ > QueryProperties.BlockMaxThreshold.toInt.get)) {
278             // check that full table scans are allowed
279             lazy val filterString =
280               org.locationtech.geomesa.filter.filterToString(filter.filter.getOrElse(Filter.INCLUDE))
281             val block =
282               QueryProperties.blockFullTableScansForFeatureType(sft.getTypeName)
283                   .orElse(BlockFullTableScans.toBoolean)
284                   .getOrElse(false)
285             if (block) {
286               throw new RuntimeException(
287                 s"Full-table scans are disabled. Query being stopped for ${sft.getTypeName}: $filterString")
288             } else {
289               logger.warn(s"Running full table scan for schema '${sft.getTypeName}' with filter: $filterString")
290             }
291           }
292           val keyRanges = Seq(UnboundedRange(null))
293           val byteRanges = Seq(BoundedByteRange(sharing, ByteArrays.rowFollowingPrefix(sharing)))
294 
295           QueryStrategy(filter, byteRanges, keyRanges, Seq.empty, ecql, hints, indexValues)
296         }
297 
298       case Some(values) =>
299         val keyRanges = keySpace.getRanges(values).toSeq
300         val bytes = keySpace.getRangeBytes(keyRanges.iterator, tieredKeySpace.isDefined).toSeq
301         val tier = tieredKeySpace.orNull.asInstanceOf[IndexKeySpace[Any, Any]]
302         val secondary = filter.secondary.orNull
303         lazy val tiers = {
304           val multiplier = math.max(1, bytes.count(_.isInstanceOf[SingleRowByteRange]))
305           tier.getRangeBytes(tier.getRanges(tier.getIndexValues(secondary, explain), multiplier)).toSeq
306         }
307 
308         if (tier == null) {
309           QueryStrategy(filter, bytes, keyRanges, Seq.empty, ecql, hints, indexValues)
310         } else if (secondary == null || tiers.exists(_.isInstanceOf[UnboundedByteRange])) {
311           val byteRanges = keySpace.getRangeBytes(keyRanges.iterator, tier = true).map {
312             case BoundedByteRange(lo, hi)      => BoundedByteRange(lo, ByteArrays.concat(hi, ByteRange.UnboundedUpperRange))
313             case SingleRowByteRange(row)       => BoundedByteRange(row, ByteArrays.concat(row, ByteRange.UnboundedUpperRange))
314             case UpperBoundedByteRange(lo, hi) => BoundedByteRange(lo, ByteArrays.concat(hi, ByteRange.UnboundedUpperRange))
315             case LowerBoundedByteRange(lo, hi) => BoundedByteRange(lo, hi)
316             case UnboundedByteRange(lo, hi)    => BoundedByteRange(lo, hi)
317             case r => throw new IllegalArgumentException(s"Unexpected range type $r")
318           }.toSeq
319           QueryStrategy(filter, byteRanges, keyRanges, Seq.empty, ecql, hints, indexValues)
320         } else if (tiers.isEmpty) {
321           // disjoint secondary filter
322           QueryStrategy(filter, Seq.empty, Seq.empty, Seq.empty, ecql, hints, indexValues)
323         } else {
324           lazy val minTier = ByteRange.min(tiers)
325           lazy val maxTier = ByteRange.max(tiers)
326 
327           val byteRanges = bytes.flatMap {
328             case SingleRowByteRange(row) =>
329               // single row - we can use all the tiered ranges appended to the end
330               tiers.map {
331                 case BoundedByteRange(lo, hi) => BoundedByteRange(ByteArrays.concat(row, lo), ByteArrays.concat(row, hi))
332                 case SingleRowByteRange(trow) => SingleRowByteRange(ByteArrays.concat(row, trow))
333               }
334 
335             case BoundedByteRange(lo, hi) =>
336               // bounded ranges - we can use the min/max tier on the endpoints
337               Iterator.single(BoundedByteRange(ByteArrays.concat(lo, minTier), ByteArrays.concat(hi, maxTier)))
338 
339             case LowerBoundedByteRange(lo, hi) =>
340               // we can't use the upper tier
341               Iterator.single(BoundedByteRange(ByteArrays.concat(lo, minTier), hi))
342 
343             case UpperBoundedByteRange(lo, hi) =>
344               // we can't use the lower tier
345               Iterator.single(BoundedByteRange(lo, ByteArrays.concat(hi, maxTier)))
346 
347             case UnboundedByteRange(lo, hi) =>
348               // we can't use either side of the tiers
349               Iterator.single(BoundedByteRange(lo, hi))
350 
351             case r =>
352               throw new IllegalArgumentException(s"Unexpected range type $r")
353           }
354 
355           QueryStrategy(filter, byteRanges, keyRanges, tiers, ecql, hints, indexValues)
356         }
357     }

367   protected def generateTableName(partition: Option[String] = None, limit: Option[Int] = None): String = {
368     import StringSerialization.alphaNumericSafeString
369     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
370 
371     val namespace = sft.getTablePrefix(name).getOrElse(ds.config.catalog)
372     val prefix = (namespace +: Seq(sft.getTypeName, name).map(alphaNumericSafeString)).mkString("_")
373     val suffix = s"v$version${partition.map(p => s"_$p").getOrElse("")}"
374 
375     def build(attrs: Seq[String]): String = (prefix +: attrs :+ suffix).mkString("_")
376 
377     val full = build(attributes.map(alphaNumericSafeString))
378 
379     limit match {
380       case Some(lim) if full.lengthCompare(lim) > 0 =>
381         // try using the attribute numbers instead
382         val nums = build(attributes.map(a => s"${sft.indexOf(a)}"))
383         if (nums.lengthCompare(lim) <= 0) { nums } else {
384           logger.warn(s"Table name length exceeds configured limit ($lim), falling back to UUID: $full")
385           IndexAdapter.truncateTableName(full, lim)
386         }
387 
388       case _ => full
389     }
390   }

506   object IdFromRow {
507 
508     def apply(
509         sft: SimpleFeatureType,
510         keySpace: IndexKeySpace[_, _],
511         tieredKeySpace: Option[IndexKeySpace[_, _]]): IdFromRow = {
512       val tieredLength = tieredKeySpace.map(_.indexKeyByteLength).getOrElse(Right(0))
513       (keySpace.indexKeyByteLength, tieredLength) match {
514         case (Right(i), Right(j)) =>
515           new FixedIdFromRow(idFromBytes(sft), i + j)
516 
517         case (Right(i), Left(j)) =>
518           val dynamic: (Array[Byte], Int, Int) => Int = (row, offset, length) => i + j(row, offset + i, length - i)
519           new DynamicIdFromRow(idFromBytes(sft), dynamic)
520 
521         case (Left(i), Right(j)) =>
522           val dynamic: (Array[Byte], Int, Int) => Int = (row, offset, length) => i(row, offset, length - j) + j
523           new DynamicIdFromRow(idFromBytes(sft), dynamic)
524 
525         case (Left(i), Left(j))   =>
526           val dynamic: (Array[Byte], Int, Int) => Int =
527             (row, offset, length) => {
528               val first = i(row, offset, length)
529               j(row, offset + first, length - first)
530             }
531           new DynamicIdFromRow(idFromBytes(sft), dynamic)
532       }
533     }
