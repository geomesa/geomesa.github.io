237   /**
238     * Plans the query
239     *
240     * @param filter filter strategy
241     * @param hints query hints
242     * @param explain explainer
243     * @return
244     */
245   def getQueryStrategy(filter: FilterStrategy, hints: Hints, explain: Explainer = ExplainNull): QueryStrategy = {
246 
247     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
248 
249     val sharing = keySpace.sharing
250     val indexValues = filter.primary.map(keySpace.getIndexValues(_, explain))
251 
252     val useFullFilter = keySpace.useFullFilter(indexValues, Some(ds.config), hints)
253     val ecql = if (useFullFilter) { filter.filter } else { filter.secondary }
254 
255     indexValues match {
256       case None =>
257         if (filter.filter.exists(_.isInstanceOf[ExcludeFilter])) {
258           QueryStrategy(filter, Seq.empty, Seq.empty, Seq.empty, ecql, hints, indexValues)
259         } else {
260           // check that full table scans are allowed
261           if (hints.getMaxFeatures.forall(_ > QueryProperties.BlockMaxThreshold.toInt.get)) {
262             // check that full table scans are allowed
263             lazy val filterString =
264               org.locationtech.geomesa.filter.filterToString(filter.filter.getOrElse(Filter.INCLUDE))
265             val block =
266               QueryProperties.blockFullTableScansForFeatureType(sft.getTypeName)
267                   .orElse(BlockFullTableScans.toBoolean)
268                   .getOrElse(false)
269             if (block) {
270               throw new RuntimeException(
271                 s"Full-table scans are disabled. Query being stopped for ${sft.getTypeName}: $filterString")
272             } else {
273               logger.warn(s"Running full table scan for schema '${sft.getTypeName}' with filter: $filterString")
274             }
275           }
276           val keyRanges = Seq(UnboundedRange(null))
277           val byteRanges = Seq(BoundedByteRange(sharing, ByteArrays.rowFollowingPrefix(sharing)))
278 
279           QueryStrategy(filter, byteRanges, keyRanges, Seq.empty, ecql, hints, indexValues)
280         }
281 
282       case Some(values) =>
283         val keyRanges = keySpace.getRanges(values).toSeq
284         val bytes = keySpace.getRangeBytes(keyRanges.iterator, tieredKeySpace.isDefined).toSeq
285         val tier = tieredKeySpace.orNull.asInstanceOf[IndexKeySpace[Any, Any]]
286         val secondary = filter.secondary.orNull
287         lazy val tiers = {
288           val multiplier = math.max(1, bytes.count(_.isInstanceOf[SingleRowByteRange]))
289           tier.getRangeBytes(tier.getRanges(tier.getIndexValues(secondary, explain), multiplier)).toSeq
290         }
291 
292         if (tier == null) {
293           QueryStrategy(filter, bytes, keyRanges, Seq.empty, ecql, hints, indexValues)
294         } else if (secondary == null || tiers.exists(_.isInstanceOf[UnboundedByteRange])) {
295           val byteRanges = keySpace.getRangeBytes(keyRanges.iterator, tier = true).map {
296             case BoundedByteRange(lo, hi)      => BoundedByteRange(lo, ByteArrays.concat(hi, ByteRange.UnboundedUpperRange))
297             case SingleRowByteRange(row)       => BoundedByteRange(row, ByteArrays.concat(row, ByteRange.UnboundedUpperRange))
298             case UpperBoundedByteRange(lo, hi) => BoundedByteRange(lo, ByteArrays.concat(hi, ByteRange.UnboundedUpperRange))
299             case LowerBoundedByteRange(lo, hi) => BoundedByteRange(lo, hi)
300             case UnboundedByteRange(lo, hi)    => BoundedByteRange(lo, hi)
301             case r => throw new IllegalArgumentException(s"Unexpected range type $r")
302           }.toSeq
303           QueryStrategy(filter, byteRanges, keyRanges, Seq.empty, ecql, hints, indexValues)
304         } else if (tiers.isEmpty) {
305           // disjoint secondary filter
306           QueryStrategy(filter, Seq.empty, Seq.empty, Seq.empty, ecql, hints, indexValues)
307         } else {
308           lazy val minTier = ByteRange.min(tiers)
309           lazy val maxTier = ByteRange.max(tiers)
310 
311           val byteRanges = bytes.flatMap {
312             case SingleRowByteRange(row) =>
313               // single row - we can use all the tiered ranges appended to the end
314               tiers.map {
315                 case BoundedByteRange(lo, hi) => BoundedByteRange(ByteArrays.concat(row, lo), ByteArrays.concat(row, hi))
316                 case SingleRowByteRange(trow) => SingleRowByteRange(ByteArrays.concat(row, trow))
317               }
318 
319             case BoundedByteRange(lo, hi) =>
320               // bounded ranges - we can use the min/max tier on the endpoints
321               Iterator.single(BoundedByteRange(ByteArrays.concat(lo, minTier), ByteArrays.concat(hi, maxTier)))
322 
323             case LowerBoundedByteRange(lo, hi) =>
324               // we can't use the upper tier
325               Iterator.single(BoundedByteRange(ByteArrays.concat(lo, minTier), hi))
326 
327             case UpperBoundedByteRange(lo, hi) =>
328               // we can't use the lower tier
329               Iterator.single(BoundedByteRange(lo, ByteArrays.concat(hi, maxTier)))
330 
331             case UnboundedByteRange(lo, hi) =>
332               // we can't use either side of the tiers
333               Iterator.single(BoundedByteRange(lo, hi))
334 
335             case r =>
336               throw new IllegalArgumentException(s"Unexpected range type $r")
337           }
338 
339           QueryStrategy(filter, byteRanges, keyRanges, tiers, ecql, hints, indexValues)
340         }
341     }

351   protected def generateTableName(partition: Option[String] = None, limit: Option[Int] = None): String = {
352     import StringSerialization.alphaNumericSafeString
353 
354     val prefix = (ds.config.catalog +: Seq(sft.getTypeName, name).map(alphaNumericSafeString)).mkString("_")
355     val suffix = s"v$version${partition.map(p => s"_$p").getOrElse("")}"
356 
357     def build(attrs: Seq[String]): String = (prefix +: attrs :+ suffix).mkString("_")
358 
359     val full = build(attributes.map(alphaNumericSafeString))
360 
361     limit match {
362       case Some(lim) if full.lengthCompare(lim) > 0 =>
363         // try using the attribute numbers instead
364         val nums = build(attributes.map(a => s"${sft.indexOf(a)}"))
365         if (nums.lengthCompare(lim) <= 0) { nums } else {
366           logger.warn(s"Table name length exceeds configured limit ($lim), falling back to UUID: $full")
367           IndexAdapter.truncateTableName(full, lim)
368         }
369 
370       case _ => full
371     }
372   }

488   object IdFromRow {

490     def apply(
491         sft: SimpleFeatureType,
492         keySpace: IndexKeySpace[_, _],
493         tieredKeySpace: Option[IndexKeySpace[_, _]]): IdFromRow = {
494       val tieredLength = tieredKeySpace.map(_.indexKeyByteLength).getOrElse(Right(0))
495       (keySpace.indexKeyByteLength, tieredLength) match {
496         case (Right(i), Right(j)) =>
497           new FixedIdFromRow(idFromBytes(sft), i + j)
498 
499         case (Right(i), Left(j)) =>
500           val dynamic: (Array[Byte], Int, Int) => Int = (row, offset, length) => i + j(row, offset + i, length - i)
501           new DynamicIdFromRow(idFromBytes(sft), dynamic)
502 
503         case (Left(i), Right(j)) =>
504           val dynamic: (Array[Byte], Int, Int) => Int = (row, offset, length) => i(row, offset, length - j) + j
505           new DynamicIdFromRow(idFromBytes(sft), dynamic)
506 
507         case (Left(i), Left(j))   =>
508           val dynamic: (Array[Byte], Int, Int) => Int =
509             (row, offset, length) => {
510               val first = i(row, offset, length)
511               j(row, offset + first, length - first)
512             }
513           new DynamicIdFromRow(idFromBytes(sft), dynamic)
514       }
515     }
