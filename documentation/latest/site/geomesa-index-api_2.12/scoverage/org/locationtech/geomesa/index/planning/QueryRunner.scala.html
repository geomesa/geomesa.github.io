30 trait QueryRunner {
31 
32   /**
33     * Execute a query
34     *
35     * @param sft simple feature type
36     * @param query query to run
37     * @param explain explain output
38     * @return
39     */
40   def runQuery(
41       sft: SimpleFeatureType,
42       query: Query,
43       explain: Explainer = new ExplainLogging): QueryResult
44 
45   /**
46     * Hook for query interceptors
47     *
48     * @return
49     */
50   protected def interceptors: QueryInterceptorFactory
51 
52   /**
53     * Configure the query - set hints, transforms, etc.
54     *
55     * @param original query to configure
56     * @param sft simple feature type associated with the query
57     */
58   protected[geomesa] def configureQuery(sft: SimpleFeatureType, original: Query): Query = {
59     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
60     import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
61 
62     val query = new Query(original)
63 
64     // query rewriting
65     interceptors(sft).foreach { interceptor =>
66       interceptor.rewrite(query)
67       QueryRunner.logger.trace(s"Query rewritten by $interceptor to: $query")
68     }
69 
70     // set query hints - we need this in certain situations where we don't have access to the query directly
71     QueryPlanner.threadedHints.get.foreach { hints =>
72       hints.foreach { case (k, v) => query.getHints.put(k, v) }
73       // clear any configured hints so we don't process them again
74       QueryPlanner.threadedHints.clear()
75     }
76 
77     // handle any params passed in through geoserver
78     ViewParams.setHints(query)
79 
80     // set transformations in the query
81     QueryPlanner.setQueryTransforms(sft, query)
82     // set return SFT in the query
83     query.getHints.put(QueryHints.Internal.RETURN_SFT, getReturnSft(sft, query.getHints))
84 
85     // set sorting in the query
86     QueryPlanner.setQuerySort(sft, query)
87     QueryPlanner.setProjection(sft, query)
88     QueryPlanner.setMaxFeatures(query)
89 
90     // add the bbox from the density query to the filter, if there is no more restrictive filter
91     query.getHints.getDensityEnvelope.foreach { env =>
92       val geom = query.getHints.getDensityGeometry.getOrElse(sft.getGeomField)
93       val geoms = FilterHelper.extractGeometries(query.getFilter, geom)
94       if (geoms.isEmpty || geoms.exists(g => !env.contains(g.getEnvelopeInternal))) {
95         val split = GeometryUtils.splitBoundingBox(env.asInstanceOf[ReferencedEnvelope])
96         val bbox = orFilters(split.map(ff.bbox(ff.property(geom), _)))
97         if (query.getFilter == Filter.INCLUDE) {
98           query.setFilter(bbox)
99         } else {
100           query.setFilter(andFilters(Seq(query.getFilter, bbox)))
101         }
102       }
103     }
104 
105     if (query.getFilter != null && query.getFilter != Filter.INCLUDE) {
106       // bind the literal values to the appropriate type, so that it isn't done every time the filter is evaluated
107       // update the filter to remove namespaces, handle null property names, and tweak topological filters
108       // replace filters with 'fast' implementations where possible
109       query.setFilter(FastFilterFactory.optimize(sft, query.getFilter))
110     }
111 
112     query
113   }
114 
115   /**
116     * Provides the simple feature type that will be returned from a query. Extension point for subclasses
117     *
118     * @param sft simple feature type
119     * @param hints query hints, which have been configured using @see configureQuery
120     * @return simple feature that will be returned from a query using the hints
121     */
122   protected [geomesa] def getReturnSft(sft: SimpleFeatureType, hints: Hints): SimpleFeatureType = {
123     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
124     if (hints.isBinQuery) {
125       BinaryOutputEncoder.BinEncodedSft
126     } else if (hints.isArrowQuery) {
127       org.locationtech.geomesa.arrow.ArrowEncodedSft
128     } else if (hints.isDensityQuery) {
129       DensityScan.DensitySft
130     } else if (hints.isStatsQuery) {
131       StatsScan.StatsSft
132     } else {
133       hints.getTransformSchema.getOrElse(sft)
134     }
135   }
136 }
