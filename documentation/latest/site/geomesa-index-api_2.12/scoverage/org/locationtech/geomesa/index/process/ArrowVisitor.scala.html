32 class ArrowVisitor(
33     sft: SimpleFeatureType,
34     encoding: SimpleFeatureEncoding,
35     ipcVersion: String,
36     dictionaryFields: Seq[String],
37     sortField: Option[String],
38     sortReverse: Option[Boolean],
39     preSorted: Boolean,
40     batchSize: Int,
41     flattenStruct: Boolean,
42   ) extends GeoMesaProcessVisitor with LazyLogging {
43 
44   import scala.collection.JavaConverters._
45 
46   // for collecting results manually
47   private lazy val manualVisitor: ArrowManualVisitor = {
48     val sort = sortField.map(s => (s, sortReverse.getOrElse(false)))
49     val ipcOpts = FormatVersion.options(ipcVersion)
50     if (dictionaryFields.isEmpty && (sortField.isEmpty || preSorted)) {
51       new SimpleArrowManualVisitor(sft, encoding, ipcOpts, sort, batchSize, flattenStruct)
52     } else {
53       new ComplexArrowManualVisitor(sft, encoding, ipcOpts, dictionaryFields, sort, preSorted, batchSize, flattenStruct)
54     }
55   }
56 
57   private var result: java.util.Iterator[Array[Byte]] = _
58 
59   override def getResult: ArrowResult = {
60     if (result != null) {
61       ArrowResult(result)
62     } else {
63       ArrowResult(manualVisitor.results.asJava)
64     }
65   }
66 
67   // manually called for non-accumulo feature collections
68   override def visit(feature: Feature): Unit = manualVisitor.visit(feature.asInstanceOf[SimpleFeature])
69 
70   /**
71    * Optimized method to run distributed query. Sets the result, available from `getResult`
72    *
73    * @param source simple feature source
74    * @param query may contain additional filters to apply
75    */
76   override def execute(source: SimpleFeatureSource, query: Query): Unit = {
77     logger.debug(s"Visiting source type: ${source.getClass.getName}")
78 
79     query.getHints.put(QueryHints.ARROW_ENCODE, true)
80     query.getHints.put(QueryHints.ARROW_DICTIONARY_FIELDS, dictionaryFields.mkString(","))
81     query.getHints.put(QueryHints.ARROW_INCLUDE_FID, encoding.fids.isDefined)
82     query.getHints.put(QueryHints.ARROW_PROXY_FID, encoding.fids.contains(Encoding.Min))
83     query.getHints.put(QueryHints.ARROW_BATCH_SIZE, batchSize)
84     query.getHints.put(QueryHints.ARROW_FORMAT_VERSION, ipcVersion)
85     query.getHints.put(QueryHints.ARROW_FLATTEN_STRUCT, flattenStruct)
86     query.getHints.put(QueryHints.FLIP_AXIS_ORDER, encoding.flipAxisOrder)
87     sortField.foreach(query.getHints.put(QueryHints.ARROW_SORT_FIELD, _))
88     sortReverse.foreach(query.getHints.put(QueryHints.ARROW_SORT_REVERSE, _))
89 
90     val features = CloseableIterator(source.getFeatures(query).features()).map(_.getAttribute(0).asInstanceOf[Array[Byte]])
91     // try to ensure the iterator is closed, by making it Closeable (only discoverable via type matching), and by making it
92     // auto-closing when fully read
93     // TODO investigate options for returning Closeable things from a VectorProcess
94     result = new java.util.Iterator[Array[Byte]] with Closeable {
95       override def hasNext: Boolean = {
96         val more = features.hasNext
97         if (!more) {
98           close()
99         }
100         more
101       }
102       override def next(): Array[Byte] = features.next()
103       override def close(): Unit = features.close()
104     }
105   }
106 }

108 object ArrowVisitor {
109 
110   private trait ArrowManualVisitor {
111     def visit(feature: SimpleFeature): Unit
112     def results: Iterator[Array[Byte]]
113   }
114 
115   /**
116    * Writes out features in batches, without sorting or dictionaries
117    *
118    * @param sft simple feature type
119    * @param encoding arrow encoding
120    * @param sort sort field, only used for metadata - no sorting will be done
121    * @param batchSize batch size
122    */
123   private class SimpleArrowManualVisitor(
124       sft: SimpleFeatureType,
125       encoding: SimpleFeatureEncoding,
126       ipcOpts: IpcOption,
127       sort: Option[(String, Boolean)],
128       batchSize: Int,
129       flattenStruct: Boolean
130     ) extends ArrowManualVisitor {
131 
132     private val out = new ByteArrayOutputStream()
133     private val bytes = ListBuffer.empty[Array[Byte]]
134     private var count = 0L
135 
136     private val writer =
137       SimpleFeatureArrowFileWriter(out, sft, Map.empty[String, ArrowDictionary], encoding, ipcOpts, sort, flattenStruct)
138 
139     override def visit(feature: SimpleFeature): Unit = {
140       writer.add(feature)
141       count += 1
142       if (count % batchSize == 0) {
143         writer.flush()
144         bytes.append(out.toByteArray)
145         out.reset()
146       }
147     }
148 
149     override def results: Iterator[Array[Byte]] = {
150       CloseWithLogging(writer)
151       bytes.append(out.toByteArray)
152       bytes.iterator
153     }
154   }
155 
156   /**
157    * Caches features locally in order to compute dictionaries and/or sorting
158    *
159    * @param sft simple feature type
160    * @param encoding arrow encoding
161    * @param dictionaryFields dictionary fields
162    * @param sort sort
163    * @param batchSize batch size
164    */
165   private class ComplexArrowManualVisitor(
166       sft: SimpleFeatureType,
167       encoding: SimpleFeatureEncoding,
168       ipcOpts: IpcOption,
169       dictionaryFields: Seq[String],
170       sort: Option[(String, Boolean)],
171       preSorted: Boolean,
172       batchSize: Int,
173       flattenStruct: Boolean
174     ) extends ArrowManualVisitor {
175 
176     import org.locationtech.geomesa.utils.geotools.RichAttributeDescriptors.RichAttributeDescriptor
177 
178     import scala.collection.JavaConverters._
179 
180     private val features = ArrayBuffer.empty[SimpleFeature]
181 
182     override def visit(feature: SimpleFeature): Unit = {
183       // copy the feature in case it is being re-used in the iterator
184       features.append(ScalaSimpleFeature.copy(feature))
185     }
186 
187     override def results: Iterator[Array[Byte]] = {
188       val dictionaries = dictionaryFields.map { field =>
189         val i = sft.indexOf(field)
190         val descriptor = sft.getDescriptor(i)
191         val isList = descriptor.isList
192         val values = scala.collection.mutable.HashSet.empty[AnyRef]
193         features.foreach { f =>
194           val value = f.getAttribute(i)
195           if (value != null) {
196             if (isList) {
197               // for list types encode the list elements instead of the list itself
198               value.asInstanceOf[java.util.List[AnyRef]].asScala.foreach(values.add)
199             } else {
200               values.add(value)
201             }
202           }
203         }
204         val binding = if (isList) { descriptor.getListType() } else { descriptor.getType.getBinding }
205         field -> ArrowDictionary.create(sft.getTypeName, i, values.toArray)(ClassTag[AnyRef](binding))
206       }
207 
208       val sorted = sort match {
209         case Some((field, reverse)) if !preSorted => features.sorted(SimpleFeatureOrdering(sft, field, reverse)).iterator
210         case _ => features.iterator
211       }
212 
213       val out = new ByteArrayOutputStream()
214       val bytes = ListBuffer.empty[Array[Byte]]
215 
216       WithClose(SimpleFeatureArrowFileWriter(out, sft, dictionaries.toMap, encoding, ipcOpts, sort, flattenStruct)) { writer =>
217         while (sorted.hasNext) { // send batches
218           var i = 0
219           while (i < batchSize && sorted.hasNext) {
220             writer.add(sorted.next)
221             i += 1
222           }
223           writer.flush()
224           bytes.append(out.toByteArray)
225           out.reset()
226         }
227       }
228       bytes.append(out.toByteArray)
229       bytes.iterator
230     }
231   }
232 
233   case class ArrowResult(results: java.util.Iterator[Array[Byte]]) extends AbstractCalcResult {
234     override def getValue: AnyRef = results
235   }
