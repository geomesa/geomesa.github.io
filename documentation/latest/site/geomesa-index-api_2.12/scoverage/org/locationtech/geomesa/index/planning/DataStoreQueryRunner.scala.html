24

29   override protected def getQueryPlans(sft: SimpleFeatureType, query: Query, explain: Explainer): Seq[QueryPlan] = {
30     explain.pushLevel("Strategy selection:")
31     val strategies = StrategyDecider.getFilterPlan(ds, sft, query.getFilter, query.getHints, explain)
32     explain.popLevel()
33 
34     var strategyCount = 1
35     strategies.map { strategy =>
36       explain.pushLevel(s"Strategy $strategyCount of ${strategies.length}: ${strategy.index}")
37       strategyCount += 1
38       explain(s"Strategy filter: $strategy")
39       val start = System.nanoTime()
40       val qs = strategy.getQueryStrategy(explain)
41       val plan = ds.adapter.createQueryPlan(qs)
42       plan.explain(explain)
43       explain(s"Plan creation took ${(System.nanoTime() - start) / 1000000L}ms").popLevel()
44       plan
45     }
46   }
