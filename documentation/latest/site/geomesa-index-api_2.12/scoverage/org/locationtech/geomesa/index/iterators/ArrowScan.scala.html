149   /**
150     * Returns batches of [threading key][dictionary deltas][record batch]. Will sort each batch,
151     * but not between batches.
152     *
153     * @param sft simple feature type
154     * @param dictionaryFields dictionary fields
155     * @param encoding arrow encoding
156     * @param sort sort field, sort reverse
157     * @param batchSize batch size

159   class DeltaAggregate(
160       sft: SimpleFeatureType,
161       dictionaryFields: Seq[String],
162       encoding: SimpleFeatureEncoding,
163       ipcOpts: IpcOption,
164       sort: Option[(String, Boolean)],
165       batchSize: Int
166     ) extends ArrowAggregate {
167 
168     private val features = Array.ofDim[SimpleFeature](batchSize)
169 
170     private var writer: DeltaWriter = _
171 
172     private var index = 0
173 
174     override def init(): Unit = {
175       writer = new DeltaWriter(sft, dictionaryFields, encoding, ipcOpts, sort, batchSize)
176     }
177 
178     override def aggregate(sf: SimpleFeature): Int = {
179       // we have to copy since the feature might be re-used
180       // TODO we could probably optimize this...
181       features(index) = ScalaSimpleFeature.copy(sf)
182       index += 1
183       1
184     }
185 
186     override def encode(): Array[Byte] = try { writer.encode(features, index) } finally { index = 0 }
187 
188     override def cleanup(): Unit = {
189       CloseWithLogging(writer)
190       writer = null
191     }
192   }

194   /**
195     * Reduce function for delta batches.
196     *
197     * First feature contains metadata for arrow file and dictionary batch, subsequent features
198     * contain record batches, final feature contains EOF indicator
199     *
200     * @param sft simple feature type
201     * @param dictionaryFields dictionaries
202     * @param encoding encoding
203     * @param batchSize batch size
204     * @param sort sort
205     * @return
206     */
207   class DeltaReducer(
208       private var sft: SimpleFeatureType,
209       private var dictionaryFields: Seq[String],
210       private var encoding: SimpleFeatureEncoding,
211       private var ipcOpts: IpcOption,
212       private var batchSize: Int,
213       private var sort: Option[(String, Boolean)],
214       private var sorted: Boolean,
215       private var process: Boolean
216     ) extends FeatureReducer {
217 
218     def this() = this(null, null, null, null, -1, null, false, true) // no-arg constructor required for serialization
219 
220     def this(sft: SimpleFeatureType, hints: Hints, sorted: Boolean) =
221       this(sft, hints.getArrowDictionaryFields, getEncoding(hints), getIpcOpts(hints), getBatchSize(hints),
222         hints.getSortFields.map(_.head).orElse(hints.getArrowSort), sorted, hints.isArrowProcessDeltas)
223 
224     override def init(state: Map[String, String]): Unit = {
225       sft = ReducerConfig.sft(state)
226       dictionaryFields = StringSerialization.decodeSeq(state(ReducerConfig.DictionariesKey))
227       encoding = ReducerConfig.encoding(state)
228       ipcOpts = ReducerConfig.ipcOption(state)
229       batchSize = ReducerConfig.batch(state)
230       sort = ReducerConfig.sort(state)
231       sorted = ReducerConfig.sorted(state)
232       process = ReducerConfig.process(state)
233     }
234 
235     override def state: Map[String, String] = Map(
236       ReducerConfig.sftName(sft),
237       ReducerConfig.sftSpec(sft),
238       ReducerConfig.DictionariesKey -> StringSerialization.encodeSeq(dictionaryFields),
239       ReducerConfig.encoding(encoding),
240       ReducerConfig.flipAxisOrder(encoding),
241       ReducerConfig.ipcOption(ipcOpts),
242       ReducerConfig.batch(batchSize),
243       ReducerConfig.sort(sort),
244       ReducerConfig.sorted(sorted),
245       ReducerConfig.process(process)
246     )
247 
248     override def apply(features: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = {
249       val files = features.map(_.getAttribute(0).asInstanceOf[Array[Byte]])
250       val result = DeltaWriter.reduce(sft, dictionaryFields, encoding, ipcOpts, sort, sorted, batchSize, process, files)
251       val sf = resultFeature()
252       result.map { bytes => sf.setAttribute(0, bytes); sf }
253     }
254 
255     def canEqual(other: Any): Boolean = other.isInstanceOf[DeltaReducer]
256 
257     override def equals(other: Any): Boolean = other match {
258       case that: DeltaReducer if that.canEqual(this) =>
259         sft == that.sft && dictionaryFields == that.dictionaryFields && encoding == that.encoding &&
260             batchSize == that.batchSize && sort == that.sort && sorted == that.sorted && that.process == process
261       case _ => false
262     }
263 
264     override def hashCode(): Int = {
265       val state = Seq(sft, dictionaryFields, encoding, batchSize, sort, process)
266       state.map(Objects.hashCode).foldLeft(0)((a, b) => 31 * a + b)
267     }
268   }
