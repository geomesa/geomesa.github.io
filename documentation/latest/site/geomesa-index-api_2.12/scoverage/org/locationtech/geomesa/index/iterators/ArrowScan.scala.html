143   /**
144     * Returns batches of [threading key][dictionary deltas][record batch]. Will sort each batch,
145     * but not between batches.
146     *

148     * @param dictionaryFields dictionary fields
149     * @param encoding arrow encoding

151     * @param batchSize batch size

153   class DeltaAggregate(
154       sft: SimpleFeatureType,
155       dictionaryFields: Seq[String],
156       encoding: SimpleFeatureEncoding,
157       ipcOpts: IpcOption,
158       sort: Option[(String, Boolean)],
159       batchSize: Int
160     ) extends ArrowAggregate {
161 
162     private val features = Array.ofDim[SimpleFeature](batchSize)
163 
164     private var writer: DeltaWriter = _
165 
166     private var index = 0
167 
168     override def init(): Unit = {
169       writer = new DeltaWriter(sft, dictionaryFields, encoding, ipcOpts, sort, batchSize)
170     }
171 
172     override def aggregate(sf: SimpleFeature): Int = {
173       // we have to copy since the feature might be re-used
174       // TODO we could probably optimize this...
175       features(index) = ScalaSimpleFeature.copy(sf)
176       index += 1
177       1
178     }
179 
180     override def encode(): Array[Byte] = try { writer.encode(features, index) } finally { index = 0 }
181 
182     override def cleanup(): Unit = {
183       CloseWithLogging(writer)
184       writer = null
185     }
186   }

188   /**
189     * Reduce function for delta batches.
190     *
191     * First feature contains metadata for arrow file and dictionary batch, subsequent features
192     * contain record batches, final feature contains EOF indicator
193     *
194     * @param sft simple feature type
195     * @param dictionaryFields dictionaries
196     * @param encoding encoding
197     * @param batchSize batch size
198     * @param sort sort

201   class DeltaReducer(
202       private var sft: SimpleFeatureType,
203       private var dictionaryFields: Seq[String],
204       private var encoding: SimpleFeatureEncoding,
205       private var ipcOpts: IpcOption,
206       private var batchSize: Int,
207       private var sort: Option[(String, Boolean)],
208       private var sorted: Boolean,
209       private var process: Boolean
210     ) extends FeatureReducer {
211 
212     def this() = this(null, null, null, null, -1, null, false, true) // no-arg constructor required for serialization
213 
214     override def init(state: Map[String, String]): Unit = {
215       sft = ReducerConfig.sft(state)
216       dictionaryFields = StringSerialization.decodeSeq(state(ReducerConfig.DictionariesKey))
217       encoding = ReducerConfig.encoding(state)
218       ipcOpts = ReducerConfig.ipcOption(state)
219       batchSize = ReducerConfig.batch(state)
220       sort = ReducerConfig.sort(state)
221       sorted = ReducerConfig.sorted(state)
222       process = ReducerConfig.process(state)
223     }
224 
225     override def state: Map[String, String] = Map(
226       ReducerConfig.sftName(sft),
227       ReducerConfig.sftSpec(sft),
228       ReducerConfig.DictionariesKey -> StringSerialization.encodeSeq(dictionaryFields),
229       ReducerConfig.encoding(encoding),
230       ReducerConfig.flipAxisOrder(encoding),
231       ReducerConfig.ipcOption(ipcOpts),
232       ReducerConfig.batch(batchSize),
233       ReducerConfig.sort(sort),
234       ReducerConfig.sorted(sorted),
235       ReducerConfig.process(process)
236     )
237 
238     override def apply(features: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = {
239       val files = features.map(_.getAttribute(0).asInstanceOf[Array[Byte]])
240       val result = DeltaWriter.reduce(sft, dictionaryFields, encoding, ipcOpts, sort, sorted, batchSize, process, files)
241       val sf = resultFeature()
242       result.map { bytes => sf.setAttribute(0, bytes); sf }
243     }
244 
245     def canEqual(other: Any): Boolean = other.isInstanceOf[DeltaReducer]
246 
247     override def equals(other: Any): Boolean = other match {
248       case that: DeltaReducer if that.canEqual(this) =>
249         sft == that.sft && dictionaryFields == that.dictionaryFields && encoding == that.encoding &&
250             batchSize == that.batchSize && sort == that.sort && sorted == that.sorted && that.process == process
251       case _ => false
252     }
253 
254     override def hashCode(): Int = {
255       val state = Seq(sft, dictionaryFields, encoding, batchSize, sort, process)
256       state.map(Objects.hashCode).foldLeft(0)((a, b) => 31 * a + b)
257     }
258   }
