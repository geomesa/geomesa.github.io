174     override protected def lower(key: AttributeIndexKey, prefix: Boolean = false): Array[Byte] = {
175       if (key.value == null) {
176         idxBytes
177       } else if (prefix) {
178         // note: inclusive doesn't make sense for prefix ranges
179         ByteArrays.concat(idxBytes, key.value.getBytes(StandardCharsets.UTF_8))
180       } else if (key.inclusive) {

182       } else {

187     override protected def upper(key: AttributeIndexKey, prefix: Boolean = false): Array[Byte] = {
188       if (key.value == null) {
189         ByteArrays.rowFollowingPrefix(idxBytes)
190       } else if (prefix) {
191         // get the row following the prefix, then get the next row
192         // note: inclusiveness doesn't really make sense for prefix ranges
193         ByteArrays.rowFollowingPrefix(ByteArrays.concat(idxBytes, key.value.getBytes(StandardCharsets.UTF_8)))
194       } else if (key.inclusive) {
195         // row following prefix, after the delimiter
196         ByteArrays.concat(idxBytes, key.value.getBytes(StandardCharsets.UTF_8), OneByteArray)
197       } else {
198         // exclude the row
199         ByteArrays.concat(idxBytes, key.value.getBytes(StandardCharsets.UTF_8), ZeroByteArray)
200       }
201     }

203     override protected def tieredUpper(key: AttributeIndexKey): Option[Array[Byte]] = {
204       // note: we can't tier exclusive end points, as we can't calculate previous rows
205       if (key.value == null || !key.inclusive) { None } else {
206         // match the final row, and count on remaining range to exclude the rest
207         Some(ByteArrays.concat(idxBytes, key.value.getBytes(StandardCharsets.UTF_8), ZeroByteArray))
208       }
