41   /**
42     * Comparison based on the integer encoding used by ByteBuffer
43     * original code is in private/protected java.nio packages
44     */
45   private def compareIntLittleEndian(left: Array[Byte],
46                                      leftOffset: Int,
47                                      right: Array[Byte],
48                                      rightOffset: Int): Int = {
49     val l3 = left(leftOffset + 3)
50     val r3 = right(rightOffset + 3)
51     if (l3 < r3) {
52       return -1
53     } else if (l3 > r3) {
54       return 1
55     }
56     val l2 = left(leftOffset + 2) & 0xff
57     val r2 = right(rightOffset + 2) & 0xff
58     if (l2 < r2) {
59       return -1
60     } else if (l2 > r2) {
61       return 1
62     }
63     val l1 = left(leftOffset + 1) & 0xff
64     val r1 = right(rightOffset + 1) & 0xff
65     if (l1 < r1) {
66       return -1
67     } else if (l1 > r1) {
68       return 1
69     }
70     val l0 = left(leftOffset) & 0xff
71     val r0 = right(rightOffset) & 0xff
72     if (l0 == r0) {
73       0
74     } else if (l0 < r0) {
75       -1
76     } else {
77       1
78     }
79   }

81   /**
82     * Takes a sequence of (already sorted) aggregates and combines them in a final sort. Uses
83     * a priority queue to compare the head element across each aggregate.
84     */
85   def mergeSort(aggregates: Iterator[Array[Byte]], binSize: Int): Iterator[(Array[Byte], Int)] = {
86     if (aggregates.isEmpty) {
87       return Iterator.empty
88     }
89     val queue = new scala.collection.mutable.PriorityQueue[(Array[Byte], Int)]()(priorityOrdering)
90     val sizes = scala.collection.mutable.ArrayBuffer.empty[Int]
91     while (aggregates.hasNext) {
92       val next = aggregates.next()
93       sizes.append(next.length / binSize)
94       queue.enqueue((next, 0))
95     }
96 
97     logger.debug(s"Got back ${queue.length} aggregates with an average size of ${sizes.sum / sizes.length}" +
98       s" chunks and a median size of ${sizes.sorted.apply(sizes.length / 2)} chunks")
99 
100     new Iterator[(Array[Byte], Int)] {
101       override def hasNext: Boolean = queue.nonEmpty
102       override def next(): (Array[Byte], Int) = {
103         val (aggregate, offset) = queue.dequeue()
104         if (offset < aggregate.length - binSize) {
105           queue.enqueue((aggregate, offset + binSize))
106         }
107         (aggregate, offset)
108       }
109     }
110   }

112   /**
113     * Performs a merge sort into a new byte array
114     */
115   def mergeSort(left: Array[Byte], right: Array[Byte], binSize: Int): Array[Byte] = {
116     if (left.length == 0) {
117       return right
118     } else if (right.length == 0) {
119       return left
120     }
121     val result = Array.ofDim[Byte](left.length + right.length)
122     var (leftIndex, rightIndex, resultIndex) = (0, 0, 0)
123 
124     while (leftIndex < left.length && rightIndex < right.length) {
125       if (compare(left, leftIndex, right, rightIndex) > 0) {
126         System.arraycopy(right, rightIndex, result, resultIndex, binSize)
127         rightIndex += binSize
128       } else {
129         System.arraycopy(left, leftIndex, result, resultIndex, binSize)
130         leftIndex += binSize
131       }
132       resultIndex += binSize
133     }
134     while (leftIndex < left.length) {
135       System.arraycopy(left, leftIndex, result, resultIndex, binSize)
136       leftIndex += binSize
137       resultIndex += binSize
138     }
139     while (rightIndex < right.length) {
140       System.arraycopy(right, rightIndex, result, resultIndex, binSize)
141       rightIndex += binSize
142       resultIndex += binSize
143     }
144     result
145   }

158   /**
159     * Optimized for non-leftmost insertion sort
160     */
161   def quickSort(bytes: Array[Byte], left: Int, right: Int, binSize: Int, leftmost: Boolean): Unit = {
162 
163     val length = (right + binSize - left) / binSize
164 
165     if (length < INSERTION_SORT_THRESHOLD) {
166       // Use insertion sort on tiny arrays
167       if (leftmost) {
168         // Traditional (without sentinel) insertion sort is used in case of the leftmost part
169         var i = left + binSize
170         while (i <= right) {
171           var j = i
172           val ai = getThreadLocalChunk(bytes, i, binSize)
173           while (j > left && compare(bytes, j - binSize, ai, 0) > 0) {
174             System.arraycopy(bytes, j - binSize, bytes, j, binSize)
175             j -= binSize
176           }
177           if (j != i) {
178             // we don't need to copy if nothing moved
179             System.arraycopy(ai, 0, bytes, j, binSize)
180           }
181           i += binSize
182         }
183       } else {
184         // optimized insertions sort when we know we have 'sentinel' elements to the left
185         /*
186          * Every element from adjoining part plays the role
187          * of sentinel, therefore this allows us to avoid the
188          * left range check on each iteration. Moreover, we use
189          * the more optimized algorithm, so called pair insertion
190          * sort, which is faster (in the context of Quicksort)
191          * than traditional implementation of insertion sort.
192          */
193         // Skip the longest ascending sequence
194         var i = left
195         if (i >= right) {
196           return
197         }
198         while ({ i += binSize; compare(bytes, i , bytes, i - binSize) >= 0 }) {
199           if (i >= right) {
200             return
201           }
202         }
203 
204         val a1 = Array.ofDim[Byte](binSize)
205         val a2 = Array.ofDim[Byte](binSize)
206 
207         var k = i
208         while ({ i += binSize; i } <= right) {
209           if (compare(bytes, k, bytes, i) < 0) {
210             System.arraycopy(bytes, k, a2, 0, binSize)
211             System.arraycopy(bytes, i, a1, 0, binSize)
212           } else {
213             System.arraycopy(bytes, k, a1, 0, binSize)
214             System.arraycopy(bytes, i, a2, 0, binSize)
215           }
216           while ({ k -= binSize; compare(a1, 0, bytes, k) < 0 }) {
217             System.arraycopy(bytes, k, bytes, k + 2 * binSize, binSize)
218           }
219           k += binSize
220           System.arraycopy(a1, 0, bytes, k + binSize, binSize)
221           while ({ k -= binSize; compare(a2, 0, bytes, k) < 0 }) {
222             System.arraycopy(bytes, k, bytes, k + binSize, binSize)
223           }
224           System.arraycopy(a2, 0, bytes, k + binSize, binSize)
225 
226           i += binSize
227           k = i
228         }
229 
230         var j = right
231         val last = getThreadLocalChunk(bytes, j, binSize)
232         while ({ j -= binSize; compare(last, 0, bytes, j) < 0 }) {
233           System.arraycopy(bytes, j, bytes, j + binSize, binSize)
234         }
235         System.arraycopy(last, 0, bytes, j + binSize, binSize)
236       }
237       return
238     }

240     /*
241      * Sort five evenly spaced elements around (and including) the
242      * center element in the range. These elements will be used for
243      * pivot selection as described below. The choice for spacing
244      * these elements was empirically determined to work well on
245      * a wide variety of inputs.
246      */
247     val seventh = (length / 7) * binSize
248 
249     val e3 = (((left + right) / binSize) / 2) * binSize // The midpoint
250     val e2 = e3 - seventh
251     val e1 = e2 - seventh
252     val e4 = e3 + seventh
253     val e5 = e4 + seventh

255     def swap(left: Int, right: Int) = {
256       val chunk = getThreadLocalChunk(bytes, left, binSize)
257       System.arraycopy(bytes, right, bytes, left, binSize)
258       System.arraycopy(chunk, 0, bytes, right, binSize)
259     }
260 
261     // Sort these elements using insertion sort
262     if (compare(bytes, e2, bytes, e1) < 0) { swap(e2, e1) }
263 
264     if (compare(bytes, e3, bytes, e2) < 0) { swap(e3, e2)
265       if (compare(bytes, e2, bytes, e1) < 0) { swap(e2, e1) }
266     }
267     if (compare(bytes, e4, bytes, e3) < 0) { swap(e4, e3)
268       if (compare(bytes, e3, bytes, e2) < 0) { swap(e3, e2)
269         if (compare(bytes, e2, bytes, e1) < 0) {swap(e2, e1) }
270       }
271     }
272     if (compare(bytes, e5, bytes, e4) < 0) { swap(e5, e4)
273       if (compare(bytes, e4, bytes, e3) < 0) { swap(e4, e3)
274         if (compare(bytes, e3, bytes, e2) < 0) { swap(e3, e2)
275           if (compare(bytes, e2, bytes, e1) < 0) { swap(e2, e1) }
276         }
277       }
278     }
279 
280     // Pointers
281     var less  = left  // The index of the first element of center part
282     var great = right // The index before the first element of right part
283 
284     if (compare(bytes, e1, bytes, e2) != 0 && compare(bytes, e2, bytes, e3) != 0 &&
285       compare(bytes, e3, bytes, e4) != 0 && compare(bytes, e4, bytes, e5) != 0 ) {
286       /*
287        * Use the second and fourth of the five sorted elements as pivots.
288        * These values are inexpensive approximations of the first and
289        * second terciles of the array. Note that pivot1 <= pivot2.
290        */
291       val pivot1 = Array.ofDim[Byte](binSize)
292       System.arraycopy(bytes, e2, pivot1, 0, binSize)
293       val pivot2 = Array.ofDim[Byte](binSize)
294       System.arraycopy(bytes, e4, pivot2, 0, binSize)
295 
296       /*
297        * The first and the last elements to be sorted are moved to the
298        * locations formerly occupied by the pivots. When partitioning
299        * is complete, the pivots are swapped back into their final
300        * positions, and excluded from subsequent sorting.
301        */
302       System.arraycopy(bytes, left, bytes, e2, binSize)
303       System.arraycopy(bytes, right, bytes, e4, binSize)
304 
305       // Skip elements, which are less or greater than pivot values.
306       while ({ less += binSize; compare(bytes, less, pivot1, 0) < 0 }) {}
307       while ({ great -= binSize; compare(bytes, great, pivot2, 0) > 0 }) {}
308 
309       /*
310        * Partitioning:
311        *
312        *   left part           center part                   right part
313        * +--------------------------------------------------------------+
314        * |  < pivot1  |  pivot1 <= && <= pivot2  |    ?    |  > pivot2  |
315        * +--------------------------------------------------------------+
316        *               ^                          ^       ^
317        *               |                          |       |
318        *              less                        k     great
319        *
320        * Invariants:
321        *
322        *              all in (left, less)   < pivot1
323        *    pivot1 <= all in [less, k)     <= pivot2
324        *              all in (great, right) > pivot2
325        *
326        * Pointer k is the first index of ?-part.
327        */
328 
329       var k = less - binSize
330       var loop = true
331       while (loop && { k += binSize; k } <= great) {
332         val ak = getThreadLocalChunk(bytes, k, binSize)
333         if (compare(ak, 0, pivot1, 0) < 0) { // Move a[k] to left part
334           System.arraycopy(bytes, less, bytes, k, binSize)
335           System.arraycopy(ak, 0, bytes, less, binSize)
336           less += binSize
337         } else if (compare(ak, 0, pivot2, 0) > 0) { // Move a[k] to right part
338           while (loop && compare(bytes, great, pivot2, 0) > 0) {
339             if (great == k) {
340               loop = false
341             }
342             great -= binSize
343           }
344           if (loop) {
345             if (compare(bytes, great, pivot1, 0) < 0) { // a[great] <= pivot2
346               System.arraycopy(bytes, less, bytes, k, binSize)
347               System.arraycopy(bytes, great, bytes, less, binSize)
348               less += binSize
349             } else { // pivot1 <= a[great] <= pivot2
350               System.arraycopy(bytes, great, bytes, k, binSize)
351             }
352             System.arraycopy(ak, 0, bytes, great, binSize)
353             great -= binSize
354           }
355         }
356       }
357 
358       // Swap pivots into their final positions
359       System.arraycopy(bytes, less - binSize, bytes, left, binSize)
360       System.arraycopy(pivot1, 0, bytes, less - binSize, binSize)
361       System.arraycopy(bytes, great + binSize, bytes, right, binSize)
362       System.arraycopy(pivot2, 0, bytes, great + binSize, binSize)
363 
364       // Sort left and right parts recursively, excluding known pivots
365       quickSort(bytes, left, less - 2 * binSize, binSize, leftmost)
366       quickSort(bytes, great + 2 * binSize, right, binSize, leftmost = false)
367 
368       /*
369        * If center part is too large (comprises > 4/7 of the array),
370        * swap internal pivot values to ends.
371        */
372       if (less < e1 && e5 < great) {
373 
374         // Skip elements, which are equal to pivot values.
375         while (compare(bytes, less, pivot1, 0) == 0) { less += binSize }
376         while (compare(bytes, great, pivot2, 0) == 0) { great -= binSize }
377 
378         /*
379          * Partitioning:
380          *
381          *   left part         center part                  right part
382          * +----------------------------------------------------------+
383          * | == pivot1 |  pivot1 < && < pivot2  |    ?    | == pivot2 |
384          * +----------------------------------------------------------+
385          *              ^                        ^       ^
386          *              |                        |       |
387          *             less                      k     great
388          *
389          * Invariants:
390          *
391          *              all in (*,  less) == pivot1
392          *     pivot1 < all in [less,  k)  < pivot2
393          *              all in (great, *) == pivot2
394          *
395          * Pointer k is the first index of ?-part.
396          */
397         var k = less - binSize
398         loop = true
399         while (loop && { k += binSize; k } <= great) {
400           val ak = getThreadLocalChunk(bytes, k, binSize)
401           if (compare(ak, 0, pivot1, 0) == 0) { // Move a[k] to left part
402             System.arraycopy(bytes, less, bytes, k, binSize)
403             System.arraycopy(ak, 0, bytes, less, binSize)
404             less += binSize
405           } else if (compare(ak, 0, pivot2, 0) == 0) { // Move a[k] to right part
406             while (loop && compare(bytes, great, pivot2, 0) == 0) {
407               if (great == k) {
408                 loop = false
409               }
410               great -= binSize
411             }
412             if (loop) {
413               if (compare(bytes, great, pivot1, 0) == 0) { // a[great] < pivot2
414                 System.arraycopy(bytes, less, bytes, k, binSize)
415                 System.arraycopy(bytes, great, bytes, less, binSize)
416                 less += binSize
417               } else { // pivot1 < a[great] < pivot2
418                 System.arraycopy(bytes, great, bytes, k, binSize)
419               }
420               System.arraycopy(ak, 0, bytes, great, binSize)
421               great -= binSize
422             }
423           }
424         }
425       }
426 
427       // Sort center part recursively
428       quickSort(bytes, less, great, binSize, leftmost = false)
429     } else { // Partitioning with one pivot
430 
431       /*
432        * Use the third of the five sorted elements as pivot.
433        * This value is inexpensive approximation of the median.
434        */
435       val pivot = Array.ofDim[Byte](binSize)
436       System.arraycopy(bytes, e3, pivot, 0, binSize)
437 
438       /*
439        * Partitioning degenerates to the traditional 3-way
440        * (or "Dutch National Flag") schema:
441        *
442        *   left part    center part              right part
443        * +-------------------------------------------------+
444        * |  < pivot  |   == pivot   |     ?    |  > pivot  |
445        * +-------------------------------------------------+
446        *              ^              ^        ^
447        *              |              |        |
448        *             less            k      great
449        *
450        * Invariants:
451        *
452        *   all in (left, less)   < pivot
453        *   all in [less, k)     == pivot
454        *   all in (great, right) > pivot
455        *
456        * Pointer k is the first index of ?-part.
457        */
458       var k = less
459       var loop = true
460       while (loop && k <= great) {
461         val comp = compare(bytes, k, pivot, 0)
462         if (comp != 0) {
463           val ak = getThreadLocalChunk(bytes, k, binSize)
464           if (comp < 0) { // Move a[k] to left part
465             System.arraycopy(bytes, less, bytes, k, binSize)
466             System.arraycopy(ak, 0, bytes, less, binSize)
467             less += binSize
468           } else { // a[k] > pivot - Move a[k] to right part
469             while (loop && compare(bytes, great, pivot, 0) > 0) {
470               if (k == great) {
471                 loop = false
472               }
473               great -= binSize
474             }
475             if (loop) {
476               if (compare(bytes, great, pivot, 0) < 0) { // a[great] <= pivot
477                 System.arraycopy(bytes, less, bytes, k, binSize)
478                 System.arraycopy(bytes, great, bytes, less, binSize)
479                 less += binSize
480               } else { // a[great] == pivot
481                 System.arraycopy(bytes, great, bytes, k, binSize)
482               }
483               System.arraycopy(ak, 0, bytes, great, binSize)
484               great -= binSize
485             }
486           }
487         }
488         k += binSize
489       }
490 
491       /*
492        * Sort left and right parts recursively.
493        * All elements from center part are equal
494        * and, therefore, already sorted.
495        */
496       quickSort(bytes, left, less - binSize, binSize, leftmost)
497       quickSort(bytes, great + binSize, right, binSize, leftmost = false)
498     }
499   }
