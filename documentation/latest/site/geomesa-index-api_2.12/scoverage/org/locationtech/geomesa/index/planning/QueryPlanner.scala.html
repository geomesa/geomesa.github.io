57   override def runQuery(sft: SimpleFeatureType, original: Query, explain: Explainer): QueryPlanResult[DS] = {
58     val start = System.nanoTime()
59     val query = configureQuery(sft, original)
60     val hints = query.getHints
61 
62     explain.pushLevel(s"Planning '${query.getTypeName}' ${FilterHelper.toString(query.getFilter)}")
63     explain(s"Original filter: ${FilterHelper.toString(original.getFilter)}")
64     explain(s"Hints: bin[${hints.isBinQuery}] arrow[${hints.isArrowQuery}] density[${hints.isDensityQuery}] " +
65       s"stats[${hints.isStatsQuery}] " +
66       s"sampling[${hints.getSampling.map { case (s, f) => s"$s${f.map(":" + _).getOrElse("")}"}.getOrElse("none")}]")
67     explain(s"Sort: ${hints.getSortFields.map(QueryHints.sortReadableString).getOrElse("none")}")
68     explain(s"Transforms: ${hints.getTransformDefinition.map(t => if (t.isEmpty) { "empty" } else { t }).getOrElse("none")}")
69     explain(s"Max features: ${hints.getMaxFeatures.getOrElse("none")}")
70     hints.getFilterCompatibility.foreach(c => explain(s"Filter compatibility: $c"))
71 
72     explain.pushLevel("Strategy selection:")
73     val strategies = StrategyDecider.getFilterPlan(ds, sft, query.getFilter, hints, explain)
74     explain.popLevel()
75 
76     var strategyCount = 1
77     val plans = strategies.map { strategy =>
78       explain.pushLevel(s"Strategy $strategyCount of ${strategies.length}: ${strategy.index}")
79       strategyCount += 1
80       explain(s"Strategy filter: $strategy")
81       val start = System.nanoTime()
82       val qs = strategy.getQueryStrategy(explain)
83       val plan = ds.adapter.createQueryPlan(qs)
84       plan.explain(explain)
85       explain(s"Plan creation took ${(System.nanoTime() - start) / 1000000L}ms").popLevel()
86       plan
87     }
88     val planCreateTime = System.nanoTime() - start
89     explain(s"Query planning took ${planCreateTime / 1000000L}ms")
90     new QueryPlanResult(hints.getReturnSft, hints, plans, run(query, plans, planCreateTime))
91   }

93   private def run(query: Query, plans: Seq[QueryPlan[DS]], planCreationTime: Long)(): CloseableIterator[SimpleFeature] = {
94     val start = System.nanoTime()
95     try {
96       var iterator = SelfClosingIterator(plans.iterator).flatMap(p => p.scan(ds).map(p.resultsToFeatures.apply))
97 
98       if (!query.getHints.isSkipReduce) {
99         plans.headOption.flatMap(_.reducer).foreach { reducer =>
100           require(plans.tail.forall(_.reducer.contains(reducer)), "Reduce must be the same in all query plans")
101           iterator = reducer.apply(iterator)
102         }
103       }
104 
105       plans.headOption.flatMap(_.sort).foreach { sort =>
106         require(plans.tail.forall(_.sort.contains(sort)), "Sort must be the same in all query plans")
107         iterator = new SortingSimpleFeatureIterator(iterator, sort)
108       }
109 
110       plans.headOption.flatMap(_.maxFeatures).foreach { maxFeatures =>
111         require(plans.tail.forall(_.maxFeatures.contains(maxFeatures)),
112           "Max features must be the same in all query plans")
113         if (query.getHints.getReturnSft == BinaryOutputEncoder.BinEncodedSft) {
114           // bin queries pack multiple records into each feature
115           // to count the records, we have to count the total bytes coming back, instead of the number of features
116           val label = query.getHints.getBinLabelField.isDefined
117           iterator = new BinaryOutputEncoder.FeatureLimitingIterator(iterator, maxFeatures, label)
118         } else {
119           iterator = iterator.take(maxFeatures)
120         }
121       }
122 
123       plans.headOption.flatMap(_.projection).foreach { projection =>
124         require(plans.tail.forall(_.projection.contains(projection)), "Projection must be the same in all query plans")
125         val project = Reprojection(query.getHints.getReturnSft, projection)
126         iterator = iterator.map(project.apply)
127       }
128 
129       val timer = Timers.scanning(query.getTypeName)
130       // wrap in an exceptional iterator to prevent geoserver from suppressing any exceptions
131       ExceptionalIterator(new TimedIterator(iterator, timer.record(_, TimeUnit.NANOSECONDS)))
132     } finally {
133       Timers.planning(query.getTypeName).record((System.nanoTime() - start) + planCreationTime, TimeUnit.NANOSECONDS)
134     }
135   }

192   /**
193    * Extract and parse transforms from the query
194    *
195    * @param sft simple feature type
196    * @param query query
197    * @return
198    */
199   def extractQueryTransforms(
200       sft: SimpleFeatureType,
201       query: Query): Option[(SimpleFeatureType, Seq[Transform], String)] = {
202     // even if a transform is not specified, some queries only use a subset of attributes
203     // specify them here so that it's easier to pick the best column group later
204     def fromQueryType: Option[Seq[String]] = {
205       val hints = query.getHints
206       if (hints.isBinQuery) {
207         Some(BinAggregatingScan.propertyNames(hints, sft))
208       } else if (hints.isDensityQuery) {
209         Some(DensityScan.propertyNames(hints, sft))
210       } else if (hints.isStatsQuery) {
211         Some(StatParser.propertyNames(sft, hints.getStatsQuery))
212       } else {
213         None
214       }
215     }
216 
217     // since we do sorting on the client, just add any sort-by attributes to the transform
218     // TODO GEOMESA-2655 we should sort and then transform back to the requested props, but it's complicated...
219     def withSort(props: Array[String]): Seq[String] = {
220       val names = props.map(p => if (p.contains('=')) { p.substring(0, p.indexOf('=')) } else { p })
221       props ++ Option(query.getSortBy).toSeq.flatMap { sort =>
222         sort.flatMap(s => Option(s.getPropertyName).flatMap(p => Option(p.getPropertyName).filterNot(names.contains)))
223       }
224     }
225 
226     // ignore transforms that don't actually do anything
227     def noop(props: Seq[String]): Boolean = props == sft.getAttributeDescriptors.asScala.map(_.getLocalName)
228 
229     Option(query.getPropertyNames).map(withSort).filterNot(noop).orElse(fromQueryType).map { props =>
230       val transforms = Transforms(sft, props)
231       val schema = Transforms.schema(sft, transforms)
232       val definition = props.mkString(Transform.DefinitionDelimiter)
233       (schema, transforms, definition)
234     }
235   }

237   /**
238     * Sets query hints for sorting
239     *
240     * @param sft sft
241     * @param query query
242     */
243   def setQuerySort(sft: SimpleFeatureType, query: Query): Unit = {
244     val sortBy = query.getSortBy
245     if (sortBy != null && sortBy.nonEmpty) {
246       val hints = query.getHints
247       if (hints.isArrowQuery) {
248         if (sortBy.lengthCompare(1) > 0) {
249           throw new IllegalArgumentException("Arrow queries only support sort by a single field: " +
250               sortBy.mkString(", "))
251         } else if (sortBy.head.getPropertyName == null) {
252           throw new IllegalArgumentException("Arrow queries only support sort by properties: " +
253               sortBy.mkString(", "))
254         }
255         val field = sortBy.head.getPropertyName.getPropertyName
256         val reverse = sortBy.head.getSortOrder == SortOrder.DESCENDING
257         hints.getArrowSort.foreach { case (f, r) =>
258           if (f != field || r != reverse) {
259             throw new IllegalArgumentException(s"Query sort does not match Arrow hints sort: " +
260                 s"${sortBy.mkString(", ")} != $f:${if (r) "DESC" else "ASC"}")
261           }
262         }
263         hints.put(QueryHints.ARROW_SORT_FIELD, field)
264         hints.put(QueryHints.ARROW_SORT_REVERSE, reverse)
265       } else if (hints.isBinQuery) {
266         val dtg = hints.getBinDtgField.orElse(sft.getDtgField).orNull
267         if (dtg == null ||
268             sortBy.map(s => Option(s.getPropertyName).map(_.getPropertyName).orNull).toSeq != Seq(dtg)) {
269           throw new IllegalArgumentException("BIN queries only support sort by a date-type field: " +
270               sortBy.mkString(", "))
271         }
272         if (sortBy.head.getSortOrder == SortOrder.DESCENDING) {
273           throw new IllegalArgumentException("BIN queries only support sort in ASCENDING order: " +
274               sortBy.mkString(", "))
275         }
276         if (hints.get(QueryHints.BIN_SORT) != null && !hints.isBinSorting) {
277           throw new IllegalArgumentException("Query sort order contradicts BIN sorting hint: " +
278               sortBy.mkString(", "))
279         }
280         hints.put(QueryHints.BIN_SORT, java.lang.Boolean.TRUE)
281       } else {
282         hints.put(QueryHints.Internal.SORT_FIELDS, QueryHints.Internal.toSortHint(sortBy))
283       }
284     }
285   }
