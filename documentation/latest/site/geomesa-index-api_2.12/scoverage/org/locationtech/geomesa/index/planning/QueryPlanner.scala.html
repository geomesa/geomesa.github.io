63   override def runQuery(sft: SimpleFeatureType, original: Query, explain: Explainer): QueryPlanResult[DS] = {
64     val start = System.nanoTime()
65     val query = configureQuery(sft, original)
66     val plans = getQueryPlans(sft, query, original.getFilter, None, explain)
67     val planCreateTime = System.nanoTime() - start
68     new QueryPlanResult(query.getHints.getReturnSft, query.getHints, plans, run(query, plans, planCreateTime))

71   private def run(query: Query, plans: Seq[QueryPlan[DS]], planCreationTime: Long)(): CloseableIterator[SimpleFeature] = {
72     val start = System.nanoTime()
73     try {
74       var iterator = SelfClosingIterator(plans.iterator).flatMap(p => p.scan(ds).map(p.resultsToFeatures.apply))
75 
76       if (!query.getHints.isSkipReduce) {
77         plans.headOption.flatMap(_.reducer).foreach { reducer =>
78           require(plans.tail.forall(_.reducer.contains(reducer)), "Reduce must be the same in all query plans")
79           iterator = reducer.apply(iterator)
80         }
81       }
82 
83       plans.headOption.flatMap(_.sort).foreach { sort =>
84         require(plans.tail.forall(_.sort.contains(sort)), "Sort must be the same in all query plans")
85         iterator = new SortingSimpleFeatureIterator(iterator, sort)
86       }
87 
88       plans.headOption.flatMap(_.maxFeatures).foreach { maxFeatures =>
89         require(plans.tail.forall(_.maxFeatures.contains(maxFeatures)),
90           "Max features must be the same in all query plans")
91         if (query.getHints.getReturnSft == BinaryOutputEncoder.BinEncodedSft) {
92           // bin queries pack multiple records into each feature
93           // to count the records, we have to count the total bytes coming back, instead of the number of features
94           val label = query.getHints.getBinLabelField.isDefined
95           iterator = new BinaryOutputEncoder.FeatureLimitingIterator(iterator, maxFeatures, label)
96         } else {
97           iterator = iterator.take(maxFeatures)
98         }
99       }
100 
101       plans.headOption.flatMap(_.projection).foreach { projection =>
102         require(plans.tail.forall(_.projection.contains(projection)), "Projection must be the same in all query plans")
103         val project = Reprojection(query.getHints.getReturnSft, projection)
104         iterator = iterator.map(project.apply)
105       }
106 
107       val timer = Timers.scanning(query.getTypeName)
108       // wrap in an exceptional iterator to prevent geoserver from suppressing any exceptions
109       ExceptionalIterator(new TimedIterator(iterator, timer.record(_, TimeUnit.NANOSECONDS)))
110     } finally {
111       Timers.planning(query.getTypeName).record((System.nanoTime() - start) + planCreationTime, TimeUnit.NANOSECONDS)
112     }
113   }

115   /**
116     * Set up the query plans and strategies used to execute them
117     *
118     * @param sft simple feature type
119     * @param query query to plan - must have been run through `configureQuery` to set expected hints, etc
120     * @param requested override index to use for executing the query
121     * @param output planning explanation output

124   private def getQueryPlans(
125       sft: SimpleFeatureType,
126       query: Query,
127       original: Filter,
128       requested: Option[String],
129       output: Explainer): Seq[QueryPlan[DS]] = {
130     import org.locationtech.geomesa.filter.filterToString
131 
132     profile(time => output(s"Query planning took ${time}ms")) {
133       val hints = query.getHints
134 
135       output.pushLevel(s"Planning '${query.getTypeName}' ${filterToString(query.getFilter)}")
136       output(s"Original filter: ${filterToString(original)}")
137       output(s"Hints: bin[${hints.isBinQuery}] arrow[${hints.isArrowQuery}] density[${hints.isDensityQuery}] " +
138           s"stats[${hints.isStatsQuery}] " +
139           s"sampling[${hints.getSampling.map { case (s, f) => s"$s${f.map(":" + _).getOrElse("")}"}.getOrElse("none")}]")
140       output(s"Sort: ${query.getHints.getSortFields.map(QueryHints.sortReadableString).getOrElse("none")}")
141       output(s"Transforms: ${query.getHints.getTransformDefinition.map(t => if (t.isEmpty) { "empty" } else { t }).getOrElse("none")}")
142       output(s"Max features: ${query.getHints.getMaxFeatures.getOrElse("none")}")
143       hints.getFilterCompatibility.foreach(c => output(s"Filter compatibility: $c"))
144 
145       output.pushLevel("Strategy selection:")
146       val requestedIndex = requested.orElse(hints.getRequestedIndex)
147       val transform = query.getHints.getTransformSchema
148       val evaluation = query.getHints.getCostEvaluation
149       val strategies =
150         StrategyDecider.getFilterPlan(ds, sft, query.getFilter, transform, evaluation, requestedIndex, output)
151       output.popLevel()
152 
153       var strategyCount = 1
154       strategies.map { strategy =>
155         def complete(plan: QueryPlan[DS], time: Long): Unit = {
156           plan.explain(output)
157           output(s"Plan creation took ${time}ms").popLevel()
158         }
159 
160         output.pushLevel(s"Strategy $strategyCount of ${strategies.length}: ${strategy.index}")
161         strategyCount += 1
162         output(s"Strategy filter: $strategy")
163         profile(complete _) {
164           val qs = strategy.getQueryStrategy(hints, output)
165           // query guard hook
166           interceptors(sft).foreach(_.guard(qs).foreach(e => throw e))
167           ds.adapter.createQueryPlan(qs)
168         }
169       }
170     }
171   }

238   /**
239    * Extract and parse transforms from the query
240    *
241    * @param sft simple feature type
242    * @param query query
243    * @return
244    */
245   def extractQueryTransforms(
246       sft: SimpleFeatureType,
247       query: Query): Option[(SimpleFeatureType, Seq[Transform], String)] = {
248     // even if a transform is not specified, some queries only use a subset of attributes
249     // specify them here so that it's easier to pick the best column group later
250     def fromQueryType: Option[Seq[String]] = {
251       val hints = query.getHints
252       if (hints.isBinQuery) {
253         Some(BinAggregatingScan.propertyNames(hints, sft))
254       } else if (hints.isDensityQuery) {
255         Some(DensityScan.propertyNames(hints, sft))
256       } else if (hints.isStatsQuery) {
257         Some(StatParser.propertyNames(sft, hints.getStatsQuery))
258       } else {
259         None
260       }
261     }
262 
263     // since we do sorting on the client, just add any sort-by attributes to the transform
264     // TODO GEOMESA-2655 we should sort and then transform back to the requested props, but it's complicated...
265     def withSort(props: Array[String]): Seq[String] = {
266       val names = props.map(p => if (p.contains('=')) { p.substring(0, p.indexOf('=')) } else { p })
267       props ++ Option(query.getSortBy).toSeq.flatMap { sort =>
268         sort.flatMap(s => Option(s.getPropertyName).flatMap(p => Option(p.getPropertyName).filterNot(names.contains)))
269       }
270     }
271 
272     // ignore transforms that don't actually do anything
273     def noop(props: Seq[String]): Boolean = props == sft.getAttributeDescriptors.asScala.map(_.getLocalName)
274 
275     Option(query.getPropertyNames).map(withSort).filterNot(noop).orElse(fromQueryType).map { props =>
276       val transforms = Transforms(sft, props)
277       val schema = Transforms.schema(sft, transforms)
278       val definition = props.mkString(Transform.DefinitionDelimiter)
279       (schema, transforms, definition)
280     }
281   }

283   /**
284     * Sets query hints for sorting
285     *
286     * @param sft sft
287     * @param query query
288     */
289   def setQuerySort(sft: SimpleFeatureType, query: Query): Unit = {
290     val sortBy = query.getSortBy
291     if (sortBy != null && sortBy.nonEmpty) {
292       val hints = query.getHints
293       if (hints.isArrowQuery) {
294         if (sortBy.lengthCompare(1) > 0) {
295           throw new IllegalArgumentException("Arrow queries only support sort by a single field: " +
296               sortBy.mkString(", "))
297         } else if (sortBy.head.getPropertyName == null) {
298           throw new IllegalArgumentException("Arrow queries only support sort by properties: " +
299               sortBy.mkString(", "))
300         }
301         val field = sortBy.head.getPropertyName.getPropertyName
302         val reverse = sortBy.head.getSortOrder == SortOrder.DESCENDING
303         hints.getArrowSort.foreach { case (f, r) =>
304           if (f != field || r != reverse) {
305             throw new IllegalArgumentException(s"Query sort does not match Arrow hints sort: " +
306                 s"${sortBy.mkString(", ")} != $f:${if (r) "DESC" else "ASC"}")
307           }
308         }
309         hints.put(QueryHints.ARROW_SORT_FIELD, field)
310         hints.put(QueryHints.ARROW_SORT_REVERSE, reverse)
311       } else if (hints.isBinQuery) {
312         val dtg = hints.getBinDtgField.orElse(sft.getDtgField).orNull
313         if (dtg == null ||
314             sortBy.map(s => Option(s.getPropertyName).map(_.getPropertyName).orNull).toSeq != Seq(dtg)) {
315           throw new IllegalArgumentException("BIN queries only support sort by a date-type field: " +
316               sortBy.mkString(", "))
317         }
318         if (sortBy.head.getSortOrder == SortOrder.DESCENDING) {
319           throw new IllegalArgumentException("BIN queries only support sort in ASCENDING order: " +
320               sortBy.mkString(", "))
321         }
322         if (hints.get(QueryHints.BIN_SORT) != null && !hints.isBinSorting) {
323           throw new IllegalArgumentException("Query sort order contradicts BIN sorting hint: " +
324               sortBy.mkString(", "))
325         }
326         hints.put(QueryHints.BIN_SORT, java.lang.Boolean.TRUE)
327       } else {
328         hints.put(QueryHints.Internal.SORT_FIELDS, QueryHints.Internal.toSortHint(sortBy))
329       }
330     }
331   }
