32   /**
33     * Splits the query up into different filter plans to be evaluated. Each filter plan will consist of one or
34     * more query plans. Each query plan will have a primary part (that would be used for query planning)
35     * and an optional secondary part (that would be applied as a secondary filter).
36     *
37     * Examples:
38     *
39     * bbox(geom) AND attr1 = ? =>
40     *
41     * Seq(FilterPlan(Seq(QueryFilter(ST,Some([ geom bbox ]),Some([ attr1 = ? ])))))
42     *
43     * bbox(geom) OR attr1 = ? =>
44     *
45     * Seq(FilterPlan(Seq(QueryFilter(ST,Some([ geom bbox ]),None), QueryFilter(ATTRIBUTE,Some([ attr1 = ? ]),None))))
46     *
47     * bbox(geom) AND dtg DURING ? AND attr1 = ? =>
48     *
49     * Seq(FilterPlan(Seq(QueryFilter(Z3,Some([ geom bbox AND dtg during ? ]),Some([ attr1 = ? ])))),
50     *     FilterPlan(Seq(QueryFilter(ATTRIBUTE,Some([ attr1 = ? ]),Some([ geom bbox AND dtg during ? ])))))
51     *
52     * note: spatial and temporal filters are combined.
53     *
54     * (bbox(geom) OR geom INTERSECT) AND attr1 = ? =>
55     *
56     * Seq(FilterPlan(Seq(QueryFilter(ST,Some([ geom bbox OR geom intersect ]),Some([ attr1 = ? ]))))
57     *     FilterPlan(Seq(QueryFilter(ATTRIBUTE,Seq([ attr1 = ? ]),Some([ geom bbox OR geom intersect ])))))
58     *
59     * note: ORs will not be split if they operate on a single attribute
60     *
61     */
62   def getQueryOptions(filter: Filter, hints: Hints = new Hints()): Seq[FilterPlan] = {
63     // cnf gives us a top level AND with ORs as first children
64     rewriteFilterInCNF(filter) match {
65       case a: And =>
66         // look for ORs across attributes, e.g. bbox OR dtg
67         val complex = ArrayBuffer.empty[Or]
68         val simple = ArrayBuffer.empty[Filter]
69         var permutations = 1
70 
71         a.getChildren.asScala.foreach {
72           case or: Or =>
73             val attributes = attributeAndIdCount(or, sft)
74             if (attributes > 1) {
75               complex += or
76               permutations *= attributes
77             } else {
78               simple += or
79             }
80 
81           case f =>
82             simple += f
83         }
84 
85         if (complex.isEmpty) {
86           // no cross-attribute ORs
87           getSimpleQueryOptions(a, hints).map(fs => FilterPlan(Seq(fs)))
88         } else {
89           // we don't generally consider all the permutations available, as that takes exponential time
90           // instead, we consider the cross-attribute-ors and the rest of the query separately
91 
92           // the cross-attribute or plans
93           // each option should just result in 1 filter plan since there are no ANDs
94           val complexOptions = complex.map(getQueryOptions(_, hints))
95 
96           // the non-cross-attribute ors, with the ors tacked on as secondary filters at the end
97           lazy val simpleOptions = if (simple.isEmpty) { Seq.empty } else {
98             getSimpleQueryOptions(andFilters(simple.toSeq), hints)
99                 .map(s => FilterPlan(Seq(addSecondaryPredicates(s, complex.toSeq))))
100           }
101           lazy val expandReduceOptions =
102             expandReduceOrOptions(rewriteFilterInDNF(filter).asInstanceOf[Or], hints)
103                 .map(fp => FilterPlan(makeDisjoint(fp.strategies)))
104 
105           if (complexOptions.forall(o => o.lengthCompare(1) == 0 && o.head.strategies.forall(_.isPreferredScan))) {
106             // if each of the complex options has a good scan available, return those plus the simple options
107             complexOptions.map(o => FilterPlan(o.head.strategies.map(addSecondaryPredicates(_, simple.toSeq)))).toSeq ++
108                 simpleOptions.toSeq
109           } else if (ExpandReduceThreshold.toInt.exists(_ > permutations)) {
110             logger.debug(s"Using ${getExpandReduceLog(filter, permutations)}")
111             expandReduceOptions
112           } else if (simpleOptions.nonEmpty) {
113             // if there's not a clear option for the complex scans, ignore them and just return the simple ones
114             logger.warn("Not considering complex OR predicates in query planning: " +
115                 s"${complex.map(FilterHelper.toString).mkString("(", ") AND (", ")")}")
116             simpleOptions
117           } else {
118             // if there aren't any simple plans or clear options for the complex plans,
119             // fall back to the expand-reduce logic
120             logger.warn(s"Falling back to ${getExpandReduceLog(filter, permutations)}")
121             expandReduceOptions
122           }
123         }
124 
125       case o: Or =>
126         // there are no ands - just ors between fields
127         // this implies that each child has only a single property or ID
128         def getGroup(f: Filter): (Seq[String], Boolean) =
129           (FilterHelper.propertyNames(f, sft), FilterHelper.hasIdFilter(f))
130 
131         // group and then recombine the OR'd filters by the attribute they operate on
132         val groups = o.getChildren.asScala.groupBy(getGroup).values.map(g => ff.or(g.asJava)).toSeq
133         val perAttributeOptions = groups.flatMap { g =>
134           val options = getSimpleQueryOptions(g, hints)
135           require(options.length < 2, s"Expected only a single option for ${FilterHelper.toString(g)} but got $options")
136           options.headOption
137         }
138         if (perAttributeOptions.exists(_.isFullTableScan)) {
139           // we have to do a full table scan for part of the query, just append everything to that
140           Seq(FilterPlan(Seq(fullTableScanOption(o, hints))))
141         } else {
142           Seq(FilterPlan(makeDisjoint(perAttributeOptions)))
143         }
144 
145       case f =>
146         getSimpleQueryOptions(f, hints).map(fs => FilterPlan(Seq(fs)))
147     }
148   }

150   /**
151    * Gets options for a 'simple' filter, where each OR is on a single attribute, e.g.
152    *   (bbox1 OR bbox2) AND dtg
153    *   bbox AND dtg AND (attr1 = foo OR attr = bar)
154    * not:
155    *   bbox OR dtg
156    *
157    * Because the inputs are simple, each one can be satisfied with a single query filter.
158    * The returned values will each satisfy the query, using a different strategy.
159    *
160    * @param filter input filter
161    * @param hints query hints
162    * @return sequence of options, any of which can satisfy the query
163    */
164   private def getSimpleQueryOptions(filter: Filter, hints: Hints): Seq[FilterStrategy] = {
165     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
166 
167     if (filter == Filter.EXCLUDE) { Seq.empty } else {
168       val requestedPlan = hints.getRequestedIndex.flatMap { requested =>
169         // check by id and by name
170         val index = indices.find(i => i.identifier == requested || i.name.equalsIgnoreCase(requested)).orElse {
171           // back-compatibility for attr vs join index name
172           if (!AttributeIndex.name.equalsIgnoreCase(requested)) { None } else {
173             indices.find(_.name == AttributeIndex.JoinIndexName)
174           }
175         }
176         index.flatMap(_.getFilterStrategy(filter, hints)).map(Seq(_))
177       }
178       requestedPlan.getOrElse {
179         val preferredScans  = ArrayBuffer.empty[FilterStrategy]
180         val acceptableScans = ArrayBuffer.empty[FilterStrategy]
181         val fullTableScans  = ArrayBuffer.empty[FilterStrategy]
182 
183         indices.foreach { i =>
184           i.getFilterStrategy(filter, hints).foreach {
185             case f if f.isFullTableScan => fullTableScans += f
186             case f if f.isPreferredScan => preferredScans += f
187             case f                      => acceptableScans += f
188           }
189         }
190 
191         if (preferredScans.nonEmpty) {
192           preferredScans.toSeq
193         } else if (acceptableScans.nonEmpty) {
194           acceptableScans.toSeq
195         } else {
196           fullTableScans.take(1).toSeq
197         }
198       }
199     }
200   }

202   /**
203     * Calculates all possible options for each part of the filter, then determines all permutations of
204     * the options. This can end up being expensive (O(2&#94;n)), so is only used as a fall-back.
205     */
206   private def expandReduceOrOptions(filter: Or, hints: Hints): Seq[FilterPlan] = {
207 
208     // for each child of the or, get the query options
209     // each filter plan should only have a single query filter
210     def getChildOptions: Seq[Seq[FilterPlan]] =
211       filter.getChildren.asScala.map(getSimpleQueryOptions(_, hints).map(fs => FilterPlan(Seq(fs))).toSeq).toSeq
212 
213     // combine the filter plans so that each plan has multiple query filters
214     // use the permutations of the different options for each child
215     // TODO GEOMESA-941 Fix algorithmically dangerous (2^N exponential runtime)
216     def reduceChildOptions(childOptions: Seq[Seq[FilterPlan]]): Seq[FilterPlan] =
217       childOptions.reduce { (left, right) =>
218         left.flatMap(l => right.map(r => FilterPlan(l.strategies ++ r.strategies)))
219       }
220 
221     // try to combine query filters in each filter plan if they have the same primary filter
222     // this avoids scanning the same ranges twice with different secondary predicates
223     def combineSecondaryFilters(options: Seq[FilterPlan]): Seq[FilterPlan] = options.map { r =>
224       // build up the result array instead of using a group by to preserve filter order
225       val groups = ArrayBuffer.empty[FilterStrategy]
226       r.strategies.distinct.foreach { f =>
227         val i = groups.indexWhere(g => g.index == f.index && g.primary == f.primary)
228         if (i == -1) {
229           groups.append(f)
230         } else {
231           val current = groups(i).secondary match {
232             case Some(o: Or) => o.getChildren.asScala
233             case Some(n) => Seq(n)
234             case None => Seq.empty
235           }
236           groups.update(i, f.copy(secondary = orOption((current ++ f.secondary).toSeq), temporal = f.temporal && groups(i).temporal))
237         }
238       }
239       FilterPlan(groups.toSeq)
240     }
241 
242     // if a filter plan has any query filters that scan a subset of the range of a different query filter,
243     // then we can combine them, as we have to scan the larger range anyway
244     def mergeOverlappedFilters(options: Seq[FilterPlan]): Seq[FilterPlan] = options.map { filterPlan =>
245       val filters = ArrayBuffer[FilterStrategy](filterPlan.strategies: _*)
246       var merged: FilterStrategy = null
247       var i = 0
248       while (i < filters.length) {
249         val toMerge = filters(i)
250         if (toMerge != null) {
251           var j = 0
252           while (j < filters.length && merged == null) {
253             if (i != j) {
254               val mergeTo = filters(j)
255               if (mergeTo != null) {
256                 merged = tryMerge(toMerge, mergeTo)
257               }
258             }
259             j += 1
260           }
261           if (merged != null) {
262             // remove the merged query filter and replace the one merged into
263             filters.update(i, null)
264             filters.update(j - 1, merged)
265             merged = null
266           }
267         }
268         i += 1
269       }
270 
271       // if we have replaced anything, recreate the filter plan
272       val overlapped = filters.filter(_ != null)
273       if (overlapped.length < filterPlan.strategies.length) {
274         FilterPlan(overlapped.toSeq)
275       } else {
276         filterPlan
277       }
278     }
279 
280     val childOpts   = getChildOptions
281     val reducedOpts = reduceChildOptions(childOpts)
282     val combinedSec = combineSecondaryFilters(reducedOpts)
283     val merged      = mergeOverlappedFilters(combinedSec)
284 
285     if (merged.nonEmpty) {
286       merged
287     } else {
288       Seq(FilterPlan(Seq(fullTableScanOption(filter, hints))))
289     }
290   }

292   /**
293     * Will perform a full table scan - used when we don't have anything better. Currently z3, z2 and record
294     * tables support full table scans.
295     */
296   private def fullTableScanOption(filter: Filter, hints: Hints): FilterStrategy = {
297     val secondary = if (filter == Filter.INCLUDE) { None } else { Some(filter) }
298     // note: early return after we find a matching strategy
299     val iter = indices.iterator
300     while (iter.hasNext) {
301       iter.next().getFilterStrategy(Filter.INCLUDE, hints) match {
302         case None => // no-op
303         case Some(i) => return i.copy(secondary = secondary)
304       }
305     }
306     throw new UnsupportedOperationException(s"Configured indices do not support the query ${FilterHelper.toString(filter)}")
307   }

329   /**
330    * Try to merge the two query filters. Return the merged query filter if successful, else null.
331    */
332   def tryMerge(toMerge: FilterStrategy, mergeTo: FilterStrategy): FilterStrategy = {
333     if (mergeTo.primary.forall(_ == Filter.INCLUDE)) {
334       // this is a full table scan, we can just append the OR to the secondary filter
335       mergeTo.copy(secondary = orOption(mergeTo.secondary.toSeq ++ toMerge.filter))
336     } else if (toMerge.index.name == AttributeIndex.name && mergeTo.index.name == AttributeIndex.name) {
337       // TODO extract this out into the API?
338       tryMergeAttrStrategy(toMerge, mergeTo)
339     } else {
340       // overlapping geoms, date ranges, attribute ranges, etc will be handled when extracting bounds
341       null
342     }
343   }
344 
345   /**
346     * Tries to merge the two filters that are OR'd together into a single filter that can be queried in one pass.
347     * Will return the merged filter, or null if they can't be merged.
348     *
349     * We can merge filters if they have the same secondary filter AND:
350     *   1. One of them does not have a primary filter
351     *   2. They both have a primary filter on the same attribute
352     *
353     * @param toMerge first filter
354     * @param mergeTo second filter
355     * @return merged filter that satisfies both inputs, or null if that isn't possible
356     */
357   def tryMergeAttrStrategy(toMerge: FilterStrategy, mergeTo: FilterStrategy): FilterStrategy = {
358     // TODO this will be incorrect for multi-valued properties where we have an AND in the primary filter
359     val leftAttributes = toMerge.primary.map(FilterHelper.propertyNames(_, null))
360     val rightAttributes = mergeTo.primary.map(FilterHelper.propertyNames(_, null))
361 
362     val canMergePrimary = (leftAttributes, rightAttributes) match {
363       case (Some(left), Some(right)) => left.length == 1 && right.length == 1 && left.head == right.head
364       case _ => true
365     }
366 
367     if (canMergePrimary && toMerge.secondary == mergeTo.secondary) {
368       mergeTo.copy(primary = orOption(toMerge.primary.toSeq ++ mergeTo.primary))
369     } else {
370       null
371     }
372   }
373 
374   /**
375     * Make a filter plan disjoint ORs - this way we don't have to deduplicate results
376     *
377     * @param strategies strategies
378     * @return same strategies with disjoint ORs
379     */
380   private def makeDisjoint(strategies: Seq[FilterStrategy]): Seq[FilterStrategy] = {
381     if (strategies.lengthCompare(2) < 0) { strategies } else {
382       // A OR B OR C becomes... A OR (B NOT A) OR (C NOT A and NOT B)
383       // keep track of our current disjoint clause
384       val nots = ArrayBuffer.empty[Filter]
385       strategies.map { filter =>
386         val res = if (nots.isEmpty) { filter } else {
387           filter.copy(secondary = Some(andFilters((nots ++ filter.secondary).toSeq)))
388         }
389         nots ++= filter.filter.map(ff.not) // note - side effect
390         res
391       }
392     }
393   }
