31   /**
32     * Splits the query up into different filter plans to be evaluated. Each filter plan will consist of one or
33     * more query plans. Each query plan will have a primary part (that would be used for query planning)
34     * and an optional secondary part (that would be applied as a secondary filter).
35     *
36     * Examples:
37     *
38     * bbox(geom) AND attr1 = ? =>
39     *
40     * Seq(FilterPlan(Seq(QueryFilter(ST,Some([ geom bbox ]),Some([ attr1 = ? ])))))
41     *
42     * bbox(geom) OR attr1 = ? =>
43     *
44     * Seq(FilterPlan(Seq(QueryFilter(ST,Some([ geom bbox ]),None), QueryFilter(ATTRIBUTE,Some([ attr1 = ? ]),None))))
45     *
46     * bbox(geom) AND dtg DURING ? AND attr1 = ? =>
47     *
48     * Seq(FilterPlan(Seq(QueryFilter(Z3,Some([ geom bbox AND dtg during ? ]),Some([ attr1 = ? ])))),
49     *     FilterPlan(Seq(QueryFilter(ATTRIBUTE,Some([ attr1 = ? ]),Some([ geom bbox AND dtg during ? ])))))
50     *
51     * note: spatial and temporal filters are combined.
52     *
53     * (bbox(geom) OR geom INTERSECT) AND attr1 = ? =>
54     *
55     * Seq(FilterPlan(Seq(QueryFilter(ST,Some([ geom bbox OR geom intersect ]),Some([ attr1 = ? ]))))
56     *     FilterPlan(Seq(QueryFilter(ATTRIBUTE,Seq([ attr1 = ? ]),Some([ geom bbox OR geom intersect ])))))
57     *
58     * note: ORs will not be split if they operate on a single attribute
59     *
60     */
61   def getQueryOptions(filter: Filter, transform: Option[SimpleFeatureType] = None): Seq[FilterPlan] = {
62     // cnf gives us a top level AND with ORs as first children
63     rewriteFilterInCNF(filter) match {
64       case a: And =>
65         // look for ORs across attributes, e.g. bbox OR dtg
66         val complex = ArrayBuffer.empty[Or]
67         val simple = ArrayBuffer.empty[Filter]
68         var permutations = 1
69 
70         a.getChildren.asScala.foreach {
71           case or: Or =>
72             val attributes = attributeAndIdCount(or, sft)
73             if (attributes > 1) {
74               complex += or
75               permutations *= attributes
76             } else {
77               simple += or
78             }
79 
80           case f =>
81             simple += f
82         }
83 
84         if (complex.isEmpty) {
85           // no cross-attribute ORs
86           getSimpleQueryOptions(a, transform).map(fs => FilterPlan(Seq(fs)))
87         } else {
88           // we don't generally consider all the permutations available, as that takes exponential time
89           // instead, we consider the cross-attribute-ors and the rest of the query separately
90 
91           // the cross-attribute or plans
92           // each option should just result in 1 filter plan since there are no ANDs
93           val complexOptions = complex.map(getQueryOptions(_, transform))
94 
95           // the non-cross-attribute ors, with the ors tacked on as secondary filters at the end
96           lazy val simpleOptions = if (simple.isEmpty) { Seq.empty } else {
97             getSimpleQueryOptions(andFilters(simple.toSeq), transform)
98                 .map(s => FilterPlan(Seq(addSecondaryPredicates(s, complex.toSeq))))
99           }
100           lazy val expandReduceOptions =
101             expandReduceOrOptions(rewriteFilterInDNF(filter).asInstanceOf[Or], transform)
102                 .map(fp => FilterPlan(makeDisjoint(fp.strategies)))
103 
104           if (complexOptions.forall(o => o.lengthCompare(1) == 0 && o.head.strategies.forall(_.isPreferredScan))) {
105             // if each of the complex options has a good scan available, return those plus the simple options
106             complexOptions.map(o => FilterPlan(o.head.strategies.map(addSecondaryPredicates(_, simple.toSeq)))).toSeq ++
107                 simpleOptions.toSeq
108           } else if (ExpandReduceThreshold.toInt.exists(_ > permutations)) {
109             logger.debug(s"Using ${getExpandReduceLog(filter, permutations)}")
110             expandReduceOptions
111           } else if (simpleOptions.nonEmpty) {
112             // if there's not a clear option for the complex scans, ignore them and just return the simple ones
113             logger.warn("Not considering complex OR predicates in query planning: " +
114                 s"${complex.map(filterToString).mkString("(", ") AND (", ")")}")
115             simpleOptions
116           } else {
117             // if there aren't any simple plans or clear options for the complex plans,
118             // fall back to the expand-reduce logic
119             logger.warn(s"Falling back to ${getExpandReduceLog(filter, permutations)}")
120             expandReduceOptions
121           }
122         }
123 
124       case o: Or =>
125         // there are no ands - just ors between fields
126         // this implies that each child has only a single property or ID
127         def getGroup(f: Filter): (Seq[String], Boolean) =
128           (FilterHelper.propertyNames(f, sft), FilterHelper.hasIdFilter(f))
129 
130         // group and then recombine the OR'd filters by the attribute they operate on
131         val groups = o.getChildren.asScala.groupBy(getGroup).values.map(g => ff.or(g.asJava)).toSeq
132         val perAttributeOptions = groups.flatMap { g =>
133           val options = getSimpleQueryOptions(g, transform)
134           require(options.length < 2, s"Expected only a single option for ${filterToString(g)} but got $options")
135           options.headOption
136         }
137         if (perAttributeOptions.exists(_.isFullTableScan)) {
138           // we have to do a full table scan for part of the query, just append everything to that
139           Seq(FilterPlan(Seq(fullTableScanOption(o, transform))))
140         } else {
141           Seq(FilterPlan(makeDisjoint(perAttributeOptions)))
142         }
143 
144       case f =>
145         getSimpleQueryOptions(f, transform).map(fs => FilterPlan(Seq(fs)))
146     }
147   }

149   /**
150     * Gets options for a 'simple' filter, where each OR is on a single attribute, e.g.
151     *   (bbox1 OR bbox2) AND dtg
152     *   bbox AND dtg AND (attr1 = foo OR attr = bar)
153     * not:
154     *   bbox OR dtg
155     *
156     * Because the inputs are simple, each one can be satisfied with a single query filter.
157     * The returned values will each satisfy the query, using a different strategy.
158     *
159     * @param filter input filter
160     * @return sequence of options, any of which can satisfy the query
161     */
162   private def getSimpleQueryOptions(filter: Filter, transform: Option[SimpleFeatureType]): Seq[FilterStrategy] = {
163     if (filter == Filter.EXCLUDE) { Seq.empty } else {
164       val fullTableScansBuilder  = Seq.newBuilder[FilterStrategy]
165       val preferredScansBuilder  = Seq.newBuilder[FilterStrategy]
166       val acceptableScansBuilder = Seq.newBuilder[FilterStrategy]
167 
168       indices.foreach { i =>
169         i.getFilterStrategy(filter, transform).foreach {
170           case f if f.isFullTableScan => fullTableScansBuilder += f
171           case f if f.isPreferredScan => preferredScansBuilder += f
172           case f                      => acceptableScansBuilder += f
173         }
174       }
175 
176       val preferredScans = preferredScansBuilder.result
177       if (preferredScans.nonEmpty) { preferredScans } else {
178         val acceptableScans = acceptableScansBuilder.result
179         if (acceptableScans.nonEmpty) { acceptableScans } else {
180           fullTableScansBuilder.result.take(1)
181         }
182       }
183     }
184   }

186   /**
187     * Calculates all possible options for each part of the filter, then determines all permutations of
188     * the options. This can end up being expensive (O(2&#94;n)), so is only used as a fall-back.
189     */
190   private def expandReduceOrOptions(filter: Or, transform: Option[SimpleFeatureType]): Seq[FilterPlan] = {
191 
192     // for each child of the or, get the query options
193     // each filter plan should only have a single query filter
194     def getChildOptions: Seq[Seq[FilterPlan]] =
195       filter.getChildren.asScala.map(getSimpleQueryOptions(_, transform).map(fs => FilterPlan(Seq(fs))).toSeq).toSeq
196 
197     // combine the filter plans so that each plan has multiple query filters
198     // use the permutations of the different options for each child
199     // TODO GEOMESA-941 Fix algorithmically dangerous (2^N exponential runtime)
200     def reduceChildOptions(childOptions: Seq[Seq[FilterPlan]]): Seq[FilterPlan] =
201       childOptions.reduce { (left, right) =>
202         left.flatMap(l => right.map(r => FilterPlan(l.strategies ++ r.strategies)))
203       }
204 
205     // try to combine query filters in each filter plan if they have the same primary filter
206     // this avoids scanning the same ranges twice with different secondary predicates
207     def combineSecondaryFilters(options: Seq[FilterPlan]): Seq[FilterPlan] = options.map { r =>
208       // build up the result array instead of using a group by to preserve filter order
209       val groups = ArrayBuffer.empty[FilterStrategy]
210       r.strategies.distinct.foreach { f =>
211         val i = groups.indexWhere(g => g.index == f.index && g.primary == f.primary)
212         if (i == -1) {
213           groups.append(f)
214         } else {
215           val current = groups(i).secondary match {
216             case Some(o: Or) => o.getChildren.asScala
217             case Some(n) => Seq(n)
218             case None => Seq.empty
219           }
220           val secondary = orOption((current ++ f.secondary).toSeq)
221           val temporal = f.temporal && groups(i).temporal
222           groups.update(i, FilterStrategy(f.index, f.primary, secondary, temporal, f.costMultiplier))
223         }
224       }
225       FilterPlan(groups.toSeq)
226     }
227 
228     // if a filter plan has any query filters that scan a subset of the range of a different query filter,
229     // then we can combine them, as we have to scan the larger range anyway
230     def mergeOverlappedFilters(options: Seq[FilterPlan]): Seq[FilterPlan] = options.map { filterPlan =>
231       val filters = ArrayBuffer[FilterStrategy](filterPlan.strategies: _*)
232       var merged: FilterStrategy = null
233       var i = 0
234       while (i < filters.length) {
235         val toMerge = filters(i)
236         if (toMerge != null) {
237           var j = 0
238           while (j < filters.length && merged == null) {
239             if (i != j) {
240               val mergeTo = filters(j)
241               if (mergeTo != null) {
242                 merged = tryMerge(toMerge, mergeTo)
243               }
244             }
245             j += 1
246           }
247           if (merged != null) {
248             // remove the merged query filter and replace the one merged into
249             filters.update(i, null)
250             filters.update(j - 1, merged)
251             merged = null
252           }
253         }
254         i += 1
255       }
256 
257       // if we have replaced anything, recreate the filter plan
258       val overlapped = filters.filter(_ != null)
259       if (overlapped.length < filterPlan.strategies.length) {
260         FilterPlan(overlapped.toSeq)
261       } else {
262         filterPlan
263       }
264     }
265 
266     val childOpts   = getChildOptions
267     val reducedOpts = reduceChildOptions(childOpts)
268     val combinedSec = combineSecondaryFilters(reducedOpts)
269     val merged      = mergeOverlappedFilters(combinedSec)
270 
271     if (merged.nonEmpty) {
272       merged
273     } else {
274       Seq(FilterPlan(Seq(fullTableScanOption(filter, transform))))
275     }
276   }

278   /**
279     * Will perform a full table scan - used when we don't have anything better. Currently z3, z2 and record
280     * tables support full table scans.
281     */
282   private def fullTableScanOption(filter: Filter, transform: Option[SimpleFeatureType]): FilterStrategy = {
283     val secondary = if (filter == Filter.INCLUDE) { None } else { Some(filter) }
284     // note: early return after we find a matching strategy
285     val iter = indices.iterator
286     while (iter.hasNext) {
287       iter.next().getFilterStrategy(Filter.INCLUDE, transform) match {
288         case None => // no-op
289         case Some(i) => return FilterStrategy(i.index, i.primary, secondary, i.temporal, i.costMultiplier)
290       }
291     }
292     throw new UnsupportedOperationException(s"Configured indices do not support the query ${filterToString(filter)}")
293   }

295   private def addSecondaryPredicates(filter: FilterStrategy, predicates: Seq[Filter]): FilterStrategy = {
296     val secondary = andOption(filter.secondary.toSeq ++ predicates)
297     FilterStrategy(filter.index, filter.primary, secondary, filter.temporal, filter.costMultiplier)
298   }

317   /**
318    * Try to merge the two query filters. Return the merged query filter if successful, else null.
319    */
320   def tryMerge(toMerge: FilterStrategy, mergeTo: FilterStrategy): FilterStrategy = {
321     if (mergeTo.primary.forall(_ == Filter.INCLUDE)) {
322       // this is a full table scan, we can just append the OR to the secondary filter
323       val secondary = orOption(mergeTo.secondary.toSeq ++ toMerge.filter)
324       FilterStrategy(mergeTo.index, mergeTo.primary, secondary, mergeTo.temporal, mergeTo.costMultiplier)
325     } else if (toMerge.index.name == AttributeIndex.name && mergeTo.index.name == AttributeIndex.name) {
326       // TODO extract this out into the API?
327       tryMergeAttrStrategy(toMerge, mergeTo)
328     } else {
329       // overlapping geoms, date ranges, attribute ranges, etc will be handled when extracting bounds
330       null
331     }
332   }
333 
334   /**
335     * Tries to merge the two filters that are OR'd together into a single filter that can be queried in one pass.
336     * Will return the merged filter, or null if they can't be merged.
337     *
338     * We can merge filters if they have the same secondary filter AND:
339     *   1. One of them does not have a primary filter
340     *   2. They both have a primary filter on the same attribute
341     *
342     * @param toMerge first filter
343     * @param mergeTo second filter
344     * @return merged filter that satisfies both inputs, or null if that isn't possible
345     */
346   def tryMergeAttrStrategy(toMerge: FilterStrategy, mergeTo: FilterStrategy): FilterStrategy = {
347     // TODO this will be incorrect for multi-valued properties where we have an AND in the primary filter
348     val leftAttributes = toMerge.primary.map(FilterHelper.propertyNames(_, null))
349     val rightAttributes = mergeTo.primary.map(FilterHelper.propertyNames(_, null))
350 
351     val canMergePrimary = (leftAttributes, rightAttributes) match {
352       case (Some(left), Some(right)) => left.length == 1 && right.length == 1 && left.head == right.head
353       case _ => true
354     }
355 
356     if (canMergePrimary && toMerge.secondary == mergeTo.secondary) {
357       val primary = orOption(toMerge.primary.toSeq ++ mergeTo.primary)
358       FilterStrategy(mergeTo.index, primary, mergeTo.secondary, mergeTo.temporal, mergeTo.costMultiplier)
359     } else {
360       null
361     }
362   }
363 
364   /**
365     * Make a filter plan disjoint ORs - this way we don't have to deduplicate results
366     *
367     * @param strategies strategies
368     * @return same strategies with disjoint ORs
369     */
370   private def makeDisjoint(strategies: Seq[FilterStrategy]): Seq[FilterStrategy] = {
371     if (strategies.lengthCompare(2) < 0) { strategies } else {
372       // A OR B OR C becomes... A OR (B NOT A) OR (C NOT A and NOT B)
373       // keep track of our current disjoint clause
374       val nots = ArrayBuffer.empty[Filter]
375       strategies.map { filter =>
376         val res = if (nots.isEmpty) { filter } else {
377           val secondary = Some(andFilters((nots ++ filter.secondary).toSeq))
378           FilterStrategy(filter.index, filter.primary, secondary, filter.temporal, filter.costMultiplier)
379         }
380         nots ++= filter.filter.map(ff.not) // note - side effect
381         res
382       }
383     }
384   }
