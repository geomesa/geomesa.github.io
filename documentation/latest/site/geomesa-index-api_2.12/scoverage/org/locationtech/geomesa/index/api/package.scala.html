316   object ByteRange {
317 
318     import ByteArrays.ByteOrdering
319 
320     val UnboundedLowerRange: Array[Byte] = Array.empty
321     val UnboundedUpperRange: Array[Byte] = Array.fill(3)(ByteArrays.MaxByte)
322 
323     def min(ranges: Seq[ByteRange]): Array[Byte] = {
324       if (ranges.isEmpty) { UnboundedLowerRange } else {
325         ranges.collect {
326           case BoundedByteRange(lo, _) => lo
327           case SingleRowByteRange(row) => row
328           case r => throw new IllegalArgumentException(s"Unexpected range type $r")
329         }.min
330       }
331     }
332 
333     def max(ranges: Seq[ByteRange]): Array[Byte] = {
334       if (ranges.isEmpty) { UnboundedUpperRange } else {
335         ranges.collect {
336           case BoundedByteRange(_, hi) => hi
337           case SingleRowByteRange(row) => row
338           case r => throw new IllegalArgumentException(s"Unexpected range type $r")
339         }.max
340       }
341     }
