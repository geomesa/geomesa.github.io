229   /**
230    * A query filter split up into a 'primary' that will be used with the given feature index for range planning,
231    * and a 'secondary' that is not captured in the ranges.
232    *
233    * @param index feature index to scan
234    * @param primary primary filter used for range generation
235    * @param secondary secondary filter not captured in the ranges
236    * @param temporal does the filter strategy have a temporal component
237    * @param costMultiplier a multiplier to take into account when evaluating the cost of this
238    *                 query plan. Values < 1f will prioritize this plan, while values > 1f
239    *                 will de-prioritize this plan
240    */
241   case class FilterStrategy(
242       index: GeoMesaFeatureIndex[_, _],
243       primary: Option[Filter],
244       secondary: Option[Filter],
245       temporal: Boolean,
246       costMultiplier: Float,
247       hints: Hints) {
248 
249     private var queryStrategy: QueryStrategy = _
250 
251     lazy val filter: Option[Filter] = andOption(primary.toSeq ++ secondary)
252 
253     def isFullTableScan: Boolean = primary.isEmpty
254 
255     def isPreferredScan: Boolean = costMultiplier < FilterStrategy.PreferredMultiplierThreshold
256 
257     def getQueryStrategy(explain: Explainer = ExplainNull): QueryStrategy = synchronized {
258       if (queryStrategy == null) {
259         explain.pushLevel(s"Creating query strategy for ${index.identifier}")
260         val start = System.currentTimeMillis()
261         queryStrategy = try { index.getQueryStrategy(this, explain) } finally {
262           explain(s"Query strategy took ${System.currentTimeMillis() - start}ms").popLevel()
263         }
264       } else {
265         explain(s"Using cached query strategy for ${index.identifier}")
266       }
267       queryStrategy
268     }
269 
270     override lazy val toString: String =
271       s"$index[${primary.map(FilterHelper.toString).getOrElse("INCLUDE")}]" +
272           s"[${secondary.map(FilterHelper.toString).getOrElse("None")}]($costMultiplier)"
273   }

292   case class BoundedByteRange(lower: Array[Byte], upper: Array[Byte]) extends ByteRange
293   // special case where a range matches a single row - needs to be handled differently sometimes
294   case class SingleRowByteRange(row: Array[Byte]) extends ByteRange

296   // the following classes are only returned from `getRangeBytes` if 'tier = true'
297 
298   // indicates that the upper bound is unbounded and can't be tiered
299   case class LowerBoundedByteRange(lower: Array[Byte], upper: Array[Byte]) extends ByteRange
300   // indicates that the lower bound is unbounded and can't be tiered
301   case class UpperBoundedByteRange(lower: Array[Byte], upper: Array[Byte]) extends ByteRange
302   // indicates that both bounds are unbounded and can't be tiered
303   case class UnboundedByteRange(lower: Array[Byte], upper: Array[Byte]) extends ByteRange
304 
305   object ByteRange {
306 
307     import ByteArrays.ByteOrdering
308 
309     val UnboundedLowerRange: Array[Byte] = Array.empty
310     val UnboundedUpperRange: Array[Byte] = Array.fill(3)(ByteArrays.MaxByte)
311 
312     def min(ranges: Seq[ByteRange]): Array[Byte] = {
313       if (ranges.isEmpty) { UnboundedLowerRange } else {
314         ranges.collect {
315           case BoundedByteRange(lo, _) => lo
316           case SingleRowByteRange(row) => row
317           case r => throw new IllegalArgumentException(s"Unexpected range type $r")
318         }.min
319       }
320     }
321 
322     def max(ranges: Seq[ByteRange]): Array[Byte] = {
323       if (ranges.isEmpty) { UnboundedUpperRange } else {
324         ranges.collect {
325           case BoundedByteRange(_, hi) => hi
326           case SingleRowByteRange(row) => row
327           case r => throw new IllegalArgumentException(s"Unexpected range type $r")
328         }.max
329       }
330     }
331   }
