82     def toSortHint(sortBy: Array[SortBy]): String = {
83       val strings = sortBy.flatMap { sb =>
84         val prop = sb.getPropertyName
85         Seq(if (prop == null) { "" } else { prop.getPropertyName }, s"${sb.getSortOrder == SortOrder.DESCENDING}")
86       }
87       StringSerialization.encodeSeq(strings)
88     }
89 
90     def fromSortHint(hint: String): Seq[(String, Boolean)] =
91       StringSerialization.decodeSeq(hint).grouped(2).toSeq.map { case Seq(f, r) => (f, r.toBoolean) }

93     def toProjectionHint(crs: QueryReferenceSystems): String =

96     def fromProjectionHint(hint: String): QueryReferenceSystems = {
