47   /**
48     * Helper method to merge a filtered data store query
49     *
50     * @param filter filter
51     * @param option data store filter
52     * @return
53     */
54   def mergeFilter(sft: SimpleFeatureType, filter: Filter, option: Option[Filter]): Filter = {
55     option match {
56       case None => filter
57       case Some(f) if filter == Filter.INCLUDE => f
58       case Some(f) =>
59         // check for disjoint dates between the store filter and the query filter
60         val intersected = sft.getDtgField.flatMap { dtg =>
61           // note: don't cache this call, as it can contain things like `currentDate()` that will change per invocation
62           Some(FilterHelper.extractIntervals(f, dtg)).filter(_.nonEmpty).flatMap { left =>
63             val right = FilterHelper.extractIntervals(filter, dtg)
64             val merged = FilterValues.and[Bounds[ZonedDateTime]](Bounds.intersection)(left, right)
65             if (merged.disjoint) {
66               logger.debug(s"Suppressing query with filter (${ECQL.toCQL(filter)}) AND (${ECQL.toCQL(f)})")
67               Some(Filter.EXCLUDE)
68             } else {
69               None
70             }
71           }
72         }
73         intersected.getOrElse(andFilters(Seq(filter, f)))
74     }
75   }

95     * Data store that merges the schemas from multiple delegate stores and presents them as a unified result
96     *
97     * @param stores delegate stores
98     * @param namespace schema namespace
99     */
100   abstract class MergedDataStoreSchemas(stores: Seq[DataStore], namespace: Option[String])
101       extends ReadOnlyDataStore {
102 
103     import scala.collection.JavaConverters._
104 
105     override def getTypeNames: Array[String] = stores.map(_.getTypeNames).reduceLeft(_ intersect _)
106 
107     override def getNames: java.util.List[Name] =
108       java.util.Arrays.asList(getTypeNames.map(t => new NameImpl(namespace.orNull, t)): _*)
109 
110     override def getSchema(name: Name): SimpleFeatureType = getSchema(name.getLocalPart)
111 
112     override def getSchema(typeName: String): SimpleFeatureType = {
113       val schemas = stores.map(_.getSchema(typeName))
114 
115       if (schemas.contains(null)) {
116         return null
117       }
118 
119       lazy val fail = new IOException("Delegate schemas do not match: " +
120           schemas.map(SimpleFeatureTypes.encodeType).mkString(" :: "))
121 
122       schemas.reduceLeft[SimpleFeatureType] { case (left, right) =>
123         if (left.getAttributeCount != right.getAttributeCount) {
124           throw fail
125         }
126         val builder = new SchemaBuilder()
127         val attribute = new AttributeTypeBuilder()
128 
129         val leftDescriptors = left.getAttributeDescriptors.iterator()
130         val rightDescriptors = right.getAttributeDescriptors.iterator()
131 
132         while (leftDescriptors.hasNext) {
133           val leftDescriptor = leftDescriptors.next
134           val rightDescriptor = rightDescriptors.next
135           if (leftDescriptor.getLocalName != rightDescriptor.getLocalName) {
136             throw fail
137           }
138           val leftBinding = leftDescriptor.getType.getBinding
139           val rightBinding = rightDescriptor.getType.getBinding
140           // determine a common binding if possible, for things like java.sql.TimeStamp vs java.util.Date
141           if (leftBinding == rightBinding || leftBinding.isAssignableFrom(rightBinding)) {
142             attribute.binding(leftBinding)
143           } else if (rightBinding.isAssignableFrom(leftBinding)) {
144             attribute.binding(rightBinding)
145           } else {
146             throw fail
147           }
148 
149           // add the user data from each descriptor so the delegate stores have it if needed
150           leftDescriptor.getUserData.asScala.foreach { case (k, v) => attribute.userData(k, v) }
151           rightDescriptor.getUserData.asScala.foreach { case (k, v) => attribute.userData(k, v) }
152 
153           Some(leftDescriptor).collect { case g: GeometryDescriptor => attribute.crs(g.getCoordinateReferenceSystem) }
154 
155           builder.addAttribute(attribute.buildDescriptor(leftDescriptor.getLocalName))
156         }
157         builder.build(namespace.orNull, typeName)
158       }
159     }
160 
161     override def dispose(): Unit = stores.foreach(_.dispose())
162 
163     override def getInfo: ServiceInfo = {
164       val info = new DefaultServiceInfo()
165       info.setDescription(s"Features from ${getClass.getSimpleName}")
166       info.setSchema(FeatureTypes.DEFAULT_NAMESPACE)
167       info
168     }

170     override def getLockingManager: LockingManager = null
171   }
