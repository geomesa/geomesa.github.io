49   /**
50     * Helper method to merge a filtered data store query
51     *
52     * @param filter filter
53     * @param option data store filter
54     * @return
55     */
56   def mergeFilter(sft: SimpleFeatureType, filter: Filter, option: Option[Filter]): Filter = {
57     option match {
58       case None => filter
59       case Some(f) if filter == Filter.INCLUDE => f
60       case Some(f) =>
61         // check for disjoint dates between the store filter and the query filter
62         val intersected = sft.getDtgField.flatMap { dtg =>
63           // note: don't cache this call, as it can contain things like `currentDate()` that will change per invocation
64           Some(FilterHelper.extractIntervals(f, dtg)).filter(_.nonEmpty).flatMap { left =>
65             val right = FilterHelper.extractIntervals(filter, dtg)
66             val merged = FilterValues.and[Bounds[ZonedDateTime]](Bounds.intersection)(left, right)
67             if (merged.disjoint) {
68               logger.debug(s"Suppressing query with filter (${ECQL.toCQL(filter)}) AND (${ECQL.toCQL(f)})")
69               Some(Filter.EXCLUDE)
70             } else {
71               None
72             }
73           }
74         }
75         intersected.getOrElse(andFilters(Seq(filter, f)))
76     }
77   }

96   /**
97     * Data store that merges the schemas from multiple delegate stores and presents them as a unified result
98     *
99     * @param stores delegate stores
100     * @param namespace schema namespace
101     */
102   abstract class MergedDataStoreSchemas(stores: Seq[DataStore], namespace: Option[String])
103       extends ReadOnlyDataStore {
104 
105     import scala.collection.JavaConverters._
106 
107     override def getTypeNames: Array[String] = stores.map(_.getTypeNames).reduceLeft(_ intersect _)
108 
109     override def getNames: java.util.List[Name] =
110       java.util.Arrays.asList(getTypeNames.map(t => new NameImpl(namespace.orNull, t)): _*)
111 
112     override def getSchema(name: Name): SimpleFeatureType = getSchema(name.getLocalPart)
113 
114     override def getSchema(typeName: String): SimpleFeatureType = {
115       val schemas = stores.map(_.getSchema(typeName))
116 
117       if (schemas.contains(null)) {
118         return null
119       }
120 
121       lazy val fail = new IOException("Delegate schemas do not match: " +
122           schemas.map(SimpleFeatureTypes.encodeType).mkString(" :: "))
123 
124       schemas.reduceLeft[SimpleFeatureType] { case (left, right) =>
125         if (left.getAttributeCount != right.getAttributeCount) {
126           throw fail
127         }
128         val builder = new SchemaBuilder()
129         val attribute = new AttributeTypeBuilder()
130 
131         val leftDescriptors = left.getAttributeDescriptors.iterator()
132         val rightDescriptors = right.getAttributeDescriptors.iterator()
133 
134         while (leftDescriptors.hasNext) {
135           val leftDescriptor = leftDescriptors.next
136           val rightDescriptor = rightDescriptors.next
137           if (leftDescriptor.getLocalName != rightDescriptor.getLocalName) {
138             throw fail
139           }
140           val leftBinding = leftDescriptor.getType.getBinding
141           val rightBinding = rightDescriptor.getType.getBinding
142           // determine a common binding if possible, for things like java.sql.TimeStamp vs java.util.Date
143           if (leftBinding == rightBinding || leftBinding.isAssignableFrom(rightBinding)) {
144             attribute.binding(leftBinding)
145           } else if (rightBinding.isAssignableFrom(leftBinding)) {
146             attribute.binding(rightBinding)
147           } else {
148             throw fail
149           }
150 
151           // add the user data from each descriptor so the delegate stores have it if needed
152           leftDescriptor.getUserData.asScala.foreach { case (k, v) => attribute.userData(k, v) }
153           rightDescriptor.getUserData.asScala.foreach { case (k, v) => attribute.userData(k, v) }
154 
155           Some(leftDescriptor).collect { case g: GeometryDescriptor => attribute.crs(g.getCoordinateReferenceSystem) }
156 
157           builder.addAttribute(attribute.buildDescriptor(leftDescriptor.getLocalName))
158         }
159         builder.build(namespace.orNull, typeName)
160       }
161     }
162 
163     override def dispose(): Unit = stores.foreach(_.dispose())
164 
165     override def getInfo: ServiceInfo = {
166       val info = new DefaultServiceInfo()
167       info.setDescription(s"Features from ${getClass.getSimpleName}")
168       info.setSchema(FeatureTypes.DEFAULT_NAMESPACE)
169       info
170     }
171 
172     override def getLockingManager: LockingManager = null
173   }

175   /**
176    * Does a sorted merge of already sorted streams
177    *
178    * @param streams streams, each individually sorted
179    * @param ordering ordering
180    * @tparam T type bounds
181    */
182   class SortedMergeIterator[T](streams: Seq[CloseableIterator[T]])(implicit ordering: Ordering[T])
183     extends CloseableIterator[T] {
184 
185     private val indexedStreams = streams.toIndexedSeq
186     // reverse the ordering so we get the head of the queue as the first value
187     private val heads = scala.collection.mutable.PriorityQueue.empty[(T, Int)](Ordering.Tuple2(ordering.reverse, Ordering.Int))
188 
189     init()
190 
191     private def init(): Unit = {
192       var i = 0
193       streams.foreach { s =>
194         if (s.hasNext) {
195           heads += ((s.next, i))
196         }
197         i += 1
198       }
199     }
200 
201     override def hasNext: Boolean = heads.nonEmpty
202 
203     override def next(): T = {
204       val (n, i) = heads.dequeue()
205       val stream = indexedStreams(i)
206       if (stream.hasNext) {
207         heads += ((stream.next, i))
208       }
209       n
210     }
211 
212     override def close(): Unit = CloseWithLogging(streams)
213   }
