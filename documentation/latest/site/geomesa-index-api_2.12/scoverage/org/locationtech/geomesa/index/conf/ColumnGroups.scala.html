35   /**
36     * Gets the column groups for a simple feature type. The default group will contain all columns
37     *
38     * @param sft simple feature type
39     * @return
40     */
41   def apply(sft: SimpleFeatureType): Seq[(Array[Byte], SimpleFeatureType)] = {
42     val key = CacheKeyGenerator.cacheKey(sft)
43     var groups = ColumnGroups.cache.getIfPresent(key)
44     if (groups == null) {
45       if (sft.getVisibilityLevel == VisibilityLevel.Attribute) {
46         groups = IndexedSeq((ColumnGroups.Attributes, sft))
47       } else {
48         val map = scala.collection.mutable.Map.empty[String, SimpleFeatureTypeBuilder]
49 
50         sft.getAttributeDescriptors.asScala.foreach { descriptor =>
51           descriptor.getColumnGroups().foreach { group =>
52             map.getOrElseUpdate(group, new SimpleFeatureTypeBuilder()).add(descriptor)
53           }
54         }
55 
56         val sfts = map.map { case (group, builder) =>
57           builder.setName(sft.getTypeName)
58           val subset = SimpleFeatureTypes.immutable(builder.buildFeatureType(), sft.getUserData)
59           (group.getBytes(StandardCharsets.UTF_8), subset)
60         } + (ColumnGroups.Default -> sft)
61 
62         // return the smallest groups first, for consistency tiebreaker is string comparison of group
63         groups = sfts.toIndexedSeq.sortBy { case (group, subset) =>
64           (subset.getAttributeCount, new String(group, StandardCharsets.UTF_8))
65         }
66       }
67 
68       ColumnGroups.cache.put(key, groups)
69     }
70     groups
71   }

73   /**
74     * Get serializers for each column group
75     *
76     * @param sft simple feature type
77     * @return
78     */
79   def serializers(sft: SimpleFeatureType): Seq[(Array[Byte], SimpleFeatureSerializer)] = {
80     apply(sft).map { case (colFamily, subset) =>
81       if (colFamily.eq(ColumnGroups.Default) || colFamily.eq(ColumnGroups.Attributes)) {
82         (colFamily, KryoFeatureSerializer(subset, SerializationOption.WithoutId))
83       } else {
84         (colFamily, new ProjectingKryoFeatureSerializer(sft, subset, SerializationOption.WithoutId))
85       }
86     }
87   }

89   /**
90     * Find a column group that supports the given transform and filter
91     *
92     * @param sft simple feature type
93     * @param transform transform definitions
94     * @param ecql filter, if any
95     * @return
96     */
97   def group(sft: SimpleFeatureType, transform: Option[String], ecql: Option[Filter]): (Array[Byte], SimpleFeatureType) = {
98     val groups = apply(sft)
99     transform.map(Transforms(sft, _)) match {
100       case None => groups.last
101       case Some(definitions) =>
102         val iter = groups.iterator
103         var group = iter.next
104         // last group has all the columns, so just return the last one if nothing else matches
105         while (iter.hasNext && !supports(group._2, definitions, ecql)) {
106           group = iter.next
107         }
108         group
109     }
110   }

112   /**
113     * Validate that the column groups do not overlap with reserved column groups
114     *
115     * @param sft simple feature type
116     */
117   def validate(sft: SimpleFeatureType): Unit = {
118     val groups = sft.getAttributeDescriptors.asScala.flatMap(_.getColumnGroups()).distinct
119     groups.foreach { group =>
120       if (group == ColumnGroups.DefaultString || group == ColumnGroups.AttributesString) {
121         throw new IllegalArgumentException(s"Column group '$group' is reserved for internal use - " +
122             "please choose another name")
123       }
124     }
125     if (sft.getVisibilityLevel == VisibilityLevel.Attribute && groups.nonEmpty) {
126       throw new IllegalArgumentException("Column groups are not supported when using attribute-level visibility")
127     }
128   }

130   /**
131    * Does the simple feature type contain the fields required to evaluate the transform and filter
132    *
133    * @param sft simple feature type
134    * @param transforms transform definitions
135    * @param filter filter
136    * @return
137    */
138   private def supports(sft: SimpleFeatureType, transforms: Seq[Transform], filter: Option[Filter]): Boolean = {
139     filter.forall(FilterHelper.propertyNames(_, sft).forall(sft.indexOf(_) != -1)) &&
140         transforms.flatMap(_.properties).forall(sft.indexOf(_) != -1)
141   }
