99   protected def updateFeature(update: FastSettableFeature, previous: SimpleFeature): Unit = {
100     val start = System.nanoTime()
101     try {
102       // see if there's a suggested ID to use for this feature, else create one based on the feature
103       val writable = GeoMesaFeatureWriter.featureWithFid(update)
104       try {
105         val writer = getWriter(writable)
106         val remover = getWriter(previous)
107         if (writer.eq(remover)) {
108           // `update` will calculate all mutations up front in case the feature is not valid, so we don't write partial entries
109           writer.update(writable, previous)
110         } else {
111           remover.delete(previous)
112           writer.append(writable)
113         }
114       } catch {
115         case NonFatal(e) => throwWriteErrors(e, writable)
116       }
117       statUpdater.add(writable)
118     } finally {
119       updateTimer.record(System.nanoTime() - start, TimeUnit.NANOSECONDS)
120     }
121   }

123   protected def appendFeature(feature: FastSettableFeature): Unit = {
124     val start = System.nanoTime()
125     try {
126       // see if there's a suggested ID to use for this feature, else create one based on the feature
127       val writable = GeoMesaFeatureWriter.featureWithFid(feature)
128       try {
129         // `append` will calculate all mutations up front in case the feature is not valid, so we don't write partial entries
130         getWriter(writable).append(writable)
131       } catch {
132         case NonFatal(e) => throwWriteErrors(e, writable)
133       }
134       statUpdater.add(writable)
135     } finally {
136       appendTimer.record(System.nanoTime() - start, TimeUnit.NANOSECONDS)
137     }
138   }
139 
140   protected def removeFeature(feature: SimpleFeature): Unit = {
141     val start = System.nanoTime()
142     try {
143       // the feature has come directly from our reader, so it should be valid and already have a FID
144       getWriter(feature).delete(feature)
145       statUpdater.remove(feature)
146     } finally {
147       deleteTimer.record(System.nanoTime() - start, TimeUnit.NANOSECONDS)
148     }
149   }

188    * Create a feature writer
189    *
190    * @param ds datastore
191    * @param sft simple feature type
192    * @param indices indices to write
193    * @param filter filter for selecting features for updating writes, or None for appending writes
194    * @param atomic enforce atomic writes
195    * @tparam DS datastore type
196    * @return feature writer
197    */
198   def apply[DS <: GeoMesaDataStore[DS]](
199       ds: DS,
200       sft: SimpleFeatureType,
201       indices: Seq[GeoMesaFeatureIndex[_, _]],
202       filter: Option[Filter],
203       atomic: Boolean): GeoMesaFeatureWriter[DS] = {
204     if (TablePartition.partitioned(sft)) {
205       filter match {
206         case None => new PartitionFeatureWriter(ds, sft, indices, atomic) with GeoMesaAppendFeatureWriter[DS]
207         case Some(f) =>
208           new PartitionFeatureWriter(ds, sft, indices, atomic) with GeoMesaModifyFeatureWriter[DS] {
209             override def filter: Filter = f
210           }
211       }
212     } else {
213       filter match {
214         case None => new TableFeatureWriter(ds, sft, indices, atomic) with GeoMesaAppendFeatureWriter[DS]
215         case Some(f) =>
216           new TableFeatureWriter(ds, sft, indices, atomic) with GeoMesaModifyFeatureWriter[DS] {
217             override def filter: Filter = f
218           }
219       }
220     }
221   }

223   /**
224    * Sets the feature ID on the feature. If the user has requested a specific ID, that will be used,
225    * otherwise one will be generated. If possible, the original feature will be modified and returned.
226    */
227   def featureWithFid(feature: SimpleFeature): SimpleFeature = {
228     val provided = feature.getUserData.get(Hints.PROVIDED_FID)
229     if (provided != null) {
230       withFid(feature, provided.toString)
231     } else if (feature.getUserData.get(Hints.USE_PROVIDED_FID) != java.lang.Boolean.TRUE) {
232       withFid(feature, idGenerator.createId(feature.getFeatureType, feature))
233     } else {
234       feature
235     }
236   }

240    * otherwise one will be generated. The original feature will be modified and returned.

248       feature.setId(provided.toString)
249     } else if (feature.getUserData.get(Hints.USE_PROVIDED_FID) != java.lang.Boolean.TRUE) {

255   private def withFid(feature: SimpleFeature, fid: String): SimpleFeature = {
256     feature match {
257       case f: ScalaSimpleFeature => f.setId(fid); f
258       case _ =>
259         feature.getIdentifier match {
260           case f: FeatureIdImpl => f.setID(fid); feature
261           case f =>
262             logger.warn(s"Unknown FeatureID implementation found, rebuilding feature: $f '${f.getClass.getName}'")
263             val copy = ScalaSimpleFeature.copy(feature)
264             copy.setId(fid)
265             copy
266         }
267     }

298   /**
299     * Support for writing to partitioned tables
300     *
301     */
302   private abstract class PartitionFeatureWriter[DS <: GeoMesaDataStore[DS]](
303       val ds: DS,
304       val sft: SimpleFeatureType,
305       val indices: Seq[GeoMesaFeatureIndex[_, _]],
306       val atomic: Boolean
307     ) extends GeoMesaFeatureWriter[DS] {
308 
309     import scala.collection.JavaConverters._
310 
311     private val partition = TablePartition(ds, sft).getOrElse {
312       throw new IllegalStateException("Creating a partitioned writer for a non-partitioned schema")
313     }
314 
315     private val cache = new java.util.HashMap[String, IndexWriter]()
316     private val view = cache.asScala
317 
318     override protected def getWriter(feature: SimpleFeature): IndexWriter = {
319       val p = partition.partition(feature)
320       var writer = cache.get(p)
321       if (writer == null) {
322         // reconfigure the partition each time - this should be idempotent, and block
323         // until it is fully created (which may happen in some other thread)
324         def createOne(index: GeoMesaFeatureIndex[_, _]): Unit =
325           ds.adapter.createTable(index, Some(p), index.getSplits(Some(p)))
326         indices.toList.map(i => CachedThreadPool.submit(() => createOne(i))).foreach(_.get)
327         writer = ds.adapter.createWriter(sft, indices, Some(p), atomic)
328         cache.put(p, writer)
329       }
330       writer
331     }
332 
333     override protected def flushables(): Seq[Flushable] = view.values.toSeq :+ statUpdater
334     override protected def closeables(): Seq[Closeable] = view.values.toSeq :+ statUpdater
335   }

340   private trait GeoMesaAppendFeatureWriter[DS <: GeoMesaDataStore[DS]] extends GeoMesaFeatureWriter[DS] {
341 
342     private var currentFeature: ScalaSimpleFeature = _
343 
344     override def hasNext: Boolean = false // per geotools spec, always return false
345 
346     override def next(): FastSettableFeature = {
347       currentFeature = new ScalaSimpleFeature(sft, nextFeatureId)
348       currentFeature
349     }
350 
351     override def write(): Unit = {
352       if (currentFeature == null) {
353         throw new IllegalStateException("next() must be called before write()")
354       }
355       appendFeature(currentFeature)
356       currentFeature = null
357     }

363   /**
364     * Modifies or deletes existing features. Per the data store api, does not allow appending new features.
365     */
366   private trait GeoMesaModifyFeatureWriter[DS <: GeoMesaDataStore[DS]] extends GeoMesaFeatureWriter[DS] {
367 
368     import org.locationtech.geomesa.security.SecureSimpleFeature
369 
370     def filter: Filter
371 
372     private val reader = ds.getFeatureReader(new Query(sft.getTypeName, filter), Transaction.AUTO_COMMIT)
373 
374     // feature returned from reader
375     private var original: SimpleFeature = _
376 
377     // feature that caller will modify
378     private var live: FastSettableFeature = _
379 
380     override def hasNext: Boolean = reader.hasNext
381 
382     override def next(): FastSettableFeature = {
383       original = reader.next()
384       // set the use provided FID hint - allows user to update fid if desired,
385       // but if not we'll use the existing one
386       original.getUserData.put(Hints.USE_PROVIDED_FID, java.lang.Boolean.TRUE)
387       live = ScalaSimpleFeature.copy(sft, original) // this copies user data as well
388       live
389     }
390 
391     override def write(): Unit = {
392       if (original == null) {
393         throw new IllegalStateException("next() must be called before write()")
394       }
395       // update the feature id based on hints before we compare for changes
396       live = GeoMesaFeatureWriter.featureWithFid(live)
397       // only write if feature has actually changed...
398       // comparison of feature ID and attributes - doesn't consider concrete class used
399       if (!ScalaSimpleFeature.equalIdAndAttributes(live, original) || live.visibility != original.visibility) {
400         updateFeature(live, original)
401       }
402       original = null
403       live = null
404     }
405 
406     override def remove(): Unit = {
407       if (original == null) {
408         throw new IllegalStateException("next() must be called before remove()")
409       }
410       removeFeature(original)
411       original = null
412       live = null
413     }
414 
415     abstract override protected def closeables(): Seq[Closeable] = super.closeables() :+ reader
416   }
