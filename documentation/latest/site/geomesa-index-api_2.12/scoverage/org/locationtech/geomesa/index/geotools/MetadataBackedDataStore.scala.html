129   /**
130     * Validates the schema and writes metadata to catalog.If the schema already exists,
131     * continue without error.
132     *
133     * This method uses distributed locking to ensure a schema is only created once.
134     *
135     * @see org.geotools.data.DataAccess#createSchema(org.geotools.api.feature.type.FeatureType)
136     * @param schema type to create
137     */
138   override def createSchema(schema: SimpleFeatureType): Unit = {
139     if (getSchema(schema.getTypeName) == null) {
140       val lock = acquireCatalogLock()
141       try {
142         // check a second time now that we have the lock
143         if (getSchema(schema.getTypeName) == null) {
144           // ensure that we have a mutable type so we can set user data
145           val sft = SimpleFeatureTypes.mutable(schema)
146           // inspect and update the simple feature type for various components
147           // do this before anything else so that any modifications will be in place
148           GeoMesaSchemaValidator.validate(sft)
149 
150           // set the enabled indices
151           preSchemaCreate(sft)
152 
153           // compute the metadata values - IMPORTANT: encode type has to be called after all user data is set
154           val encodedAttributes = SimpleFeatureTypes.encodeType(sft, includeUserData = true)
155           // validate we can read out the encoded sft - invalid user data keys can break the parsing
156           try { SimpleFeatureTypes.createType("", encodedAttributes) } catch {
157             case NonFatal(e) => throw new IllegalArgumentException("Invalid schema:", e)
158           }
159           val metadataMap = Map(
160             AttributesKey      -> encodedAttributes,
161             StatsGenerationKey -> GeoToolsDateFormat.format(Instant.now().atOffset(ZoneOffset.UTC))
162           )
163           try {
164             // write out the metadata to the catalog table
165             metadata.insert(sft.getTypeName, metadataMap)
166 
167             // reload the sft so that we have any default metadata,
168             // then copy over any additional keys that were in the original sft.
169             // avoid calling getSchema directly, as that may trigger a remote version
170             // check for indices that haven't been created yet
171             val attributes = metadata.readRequired(sft.getTypeName, AttributesKey)
172             val reloadedSft = SimpleFeatureTypes.createType(sft.getTypeName, attributes)
173             val userData = new java.util.HashMap[AnyRef, AnyRef]()
174             userData.putAll(reloadedSft.getUserData)
175             reloadedSft.getUserData.putAll(sft.getUserData)
176             reloadedSft.getUserData.putAll(userData)
177 
178             // create the tables
179             onSchemaCreated(reloadedSft)
180           } catch {
181             case NonFatal(e) =>
182               // If there was an error creating a schema, clean up.
183               try {
184                 metadata.delete(sft.getTypeName)
185               } catch {
186                 case NonFatal(e2) => e.addSuppressed(e2)
187               }
188               throw e
189           }
190         }
191       } finally {
192         lock.close()
193       }
194     }
195   }

234     * Allows the following modifications to the schema:
235     *   renaming the feature type
236     *   renaming attributes
237     *   appending new attributes
238     *   enabling/disabling indices through RichSimpleFeatureType.setIndexVersion
239     *   modifying keywords through user-data
240     *
241     * Other modifications are not supported.
242     *
243     * @see org.geotools.data.DataAccess#updateSchema(org.geotools.api.feature.type.Name, org.geotools.api.feature.type.FeatureType)
244     * @param typeName simple feature type name
245     * @param schema new simple feature type
246     */
247   override def updateSchema(typeName: Name, schema: SimpleFeatureType): Unit = {
248     // validate that the type name has not changed, or that the new name is not already in use
249     if (typeName.getLocalPart != schema.getTypeName && getTypeNames.contains(schema.getTypeName)) {
250       throw new IllegalArgumentException(
251         s"Updated type name already exists: '$typeName' changed to '${schema.getTypeName}'")
252     }
253 
254     val lock = acquireCatalogLock()
255     try {
256       // get previous schema and user data
257       val previousSft = getSchema(typeName)
258 
259       if (previousSft == null) {
260         throw new IllegalArgumentException(s"Schema '$typeName' does not exist")
261       }
262 
263       GeoMesaSchemaValidator.validate(schema)
264 
265       validateSchemaUpdate(previousSft, schema).left.foreach(e => throw e)
266 
267       val sft = SimpleFeatureTypes.mutable(schema)
268 
269       // validation and normalization of the schema
270       preSchemaUpdate(sft, previousSft)
271 
272       val encodedAttributes = SimpleFeatureTypes.encodeType(sft, includeUserData = true)
273       // validate we can read out the encoded sft - invalid user data keys can break the parsing
274       try { SimpleFeatureTypes.createType("", encodedAttributes) } catch {
275         case NonFatal(e) => throw new IllegalArgumentException("Invalid schema:", e)
276       }
277 
278       // if all is well, update the metadata - first back it up
279       if (FastConverter.convertOrElse[java.lang.Boolean](sft.getUserData.get(Configs.UpdateBackupMetadata), true)) {
280         metadata.backup(typeName.getLocalPart)
281       }
282 
283       // rename the old rows if the type name has changed
284       if (typeName.getLocalPart != schema.getTypeName) {
285         metadata.scan(typeName.getLocalPart, "", cache = false).foreach { case (k, v) =>
286           metadata.insert(sft.getTypeName, k, v)
287           metadata.remove(typeName.getLocalPart, k)
288         }
289       }
290       // now insert the new spec string
291       metadata.insert(sft.getTypeName, AttributesKey, encodedAttributes)
292 
293       onSchemaUpdated(sft, previousSft)
294     } finally {
295       lock.close()
296     }
297   }

445   /**
446    * Validate a call to updateSchema, throwing errors on failed validation
447    *
448    * @param existing existing schema
449    * @param schema updated sft
450    * @return validation result
451    */
452   protected def validateSchemaUpdate(
453       existing: SimpleFeatureType,
454       schema: SimpleFeatureType): Either[UnsupportedOperationException, TypeComparison.Compatible] = {
455     // validate that default geometry and date have not changed (rename is ok)
456     if (schema.getGeomIndex != existing.getGeomIndex) {
457       return Left(new UnsupportedOperationException("Changing the default geometry attribute is not supported"))
458     }
459     if (schema.getDtgIndex != existing.getDtgIndex) {
460       return Left(new UnsupportedOperationException("Changing the default date attribute is not supported"))
461     }
462 
463     // check that unmodifiable user data has not changed
464     val userDataChanges = MetadataBackedDataStore.UnmodifiableUserDataKeys.flatMap { key =>
465       if (schema.userData[Any](key) == existing.userData[Any](key)) { None } else {
466         Some(s"'$key'")
467       }
468     }
469     if (userDataChanges.nonEmpty) {
470       val msg = s"${if (userDataChanges.size == 1) { "" } else { "s" }} ${userDataChanges.mkString(", ")}"
471       return Left(new UnsupportedOperationException(s"Updating user data key$msg is not supported"))
472     }
473 
474     SimpleFeatureTypeComparator.compare(existing, schema) match {
475       case TypeComparison.AttributeRemoved =>
476         Left(new UnsupportedOperationException("Removing attributes from the schema is not supported"))
477 
478       case TypeComparison.AttributeTypeChanged(changes) =>
479         val msg = changes.map { case (name, (from, to)) => s"$name from ${from.getName} to ${to.getName}" }
480         Left(new UnsupportedOperationException(s"Incompatible schema column type changes: ${msg.mkString(", ")}"))
481 
482       case c: TypeComparison.Compatible =>
483         // check for reserved words - only check for new/renamed attributes
484         val reserved = schema.getAttributeDescriptors.asScala.map(_.getLocalName).exists { name =>
485           existing.getDescriptor(name) == null && FeatureUtils.ReservedWords.contains(name.toUpperCase(Locale.US))
486         }
487         if (reserved) {
488           try { ReservedWordCheck.validateAttributeNames(schema) } catch {
489             case NonFatal(e) =>
490               return Left(new UnsupportedOperationException(e.getMessage))
491           }
492         }
493         Right(c)
494     }
495   }

499    * Make sure that you 'release' the lock in a finally block.

501   protected[geomesa] def acquireCatalogLock(): Closeable = {
502     import org.locationtech.geomesa.index.DistributedLockTimeout
503     val dsTypeName = getClass.getSimpleName.replaceAll("[^A-Za-z]", "")
504     val path = s"/org.locationtech.geomesa/ds/$dsTypeName/${config.catalog}"
505     val timeout = DistributedLockTimeout.toDuration.getOrElse {
506       // note: should always be a valid fallback value so this exception should never be triggered
507       throw new IllegalArgumentException(s"Couldn't convert '${DistributedLockTimeout.get}' to a duration")
508     }
509     acquireDistributedLock(path, timeout.toMillis).getOrElse {
510       throw new RuntimeException(s"Could not acquire distributed lock at '$path' within $timeout")
