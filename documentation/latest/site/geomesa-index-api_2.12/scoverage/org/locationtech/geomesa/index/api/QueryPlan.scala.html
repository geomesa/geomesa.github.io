25 /**
26   * Plan for querying a GeoMesaDataStore
27   */
28 trait QueryPlan {
29 
30   /**
31     * Type of raw results returned from the underlying database
32     */
33   type Results
34 
35   /**
36     * Runs the query plan against the underlying database
37     *
38     * @return
39     */
40   def scan(): CloseableIterator[Results]
41 
42   /**
43     * Transform results coming back from a raw scan into features
44     *
45     * @return
46     */
47   def resultsToFeatures: ResultsToFeatures[Results]
48 
49   /**
50     * Optional reduce step for simple features coming back
51     *
52     * @return
53     */
54   def reducer: Option[FeatureReducer]
55 
56   /**
57     * Sort fields and reverse order flags (true flips order to descending, false keeps order as ascending)
58     *
59     * @return
60     */
61   def sort: Option[Seq[(String, Boolean)]]
62 
63   /**
64     * Max features to return from the scan
65     *
66     * @return
67     */
68   def maxFeatures: Option[Int]
69 
70   /**
71     * Geometry projection
72     *
73     * @return
74     */
75   def projection: Option[QueryReferenceSystems]
76 
77   /**
78     * Explains details on how this query plan will be executed
79     *
80     * @param explainer explainer to use for explanation
81     */
82   def explain(explainer: Explainer): Unit
83 }

247   /**
248     * Abstract base class for converting the results from a normal feature index
249     *
250     * @param index index
251     * @param sft simple feature type returned from the scan
252     * @tparam T result type
253     */
254   abstract class IndexResultsToFeatures[T](
255       protected var index: GeoMesaFeatureIndex[_, _],
256       protected var sft: SimpleFeatureType
257     ) extends ResultsToFeatures[T] {
258 
259     protected var serializer: KryoFeatureSerializer = if (index == null) { null } else { createSerializer }
260 
261     override def init(state: Map[String, String]): Unit = {
262       val spec = state("spec")
263       sft = SimpleFeatureTypes.createType(state("name"), spec)
264       index = state.get("isft") match {
265         case None => IteratorCache.index(sft, spec, state("idx"))
266         case Some(isft) => IteratorCache.index(IteratorCache.sft(isft), isft, state("idx"))
267       }
268       serializer = createSerializer
269     }
270 
271     override def state: Map[String, String] = {
272       val base = Map(
273         "name" -> sft.getTypeName,
274         "spec" -> SimpleFeatureTypes.encodeType(sft, includeUserData = true),
275         "idx"  -> index.identifier
276       )
277       if (index.sft == sft) { base } else {
278         base.updated("isft", SimpleFeatureTypes.encodeType(index.sft, includeUserData = true))
279       }
280     }
281 
282     override def schema: SimpleFeatureType = sft
283 
284     protected def createSerializer: KryoFeatureSerializer = {
285       val hasAttributeLevelVis = sft.getVisibilityLevel == VisibilityLevel.Attribute
286       val builder = KryoFeatureSerializer.builder(sft)
287       // note: attribute level vis are serialized into the user data since they can't be held by a single key
288       val toBuild = (index.serializedWithId, hasAttributeLevelVis) match {
289         case (true, true)   => builder.withId.withUserData
290         case (true, false)  => builder.withId.withoutUserData
291         case (false, true)  => builder.withoutId.withUserData
292         case (false, false) => builder.withoutId.withoutUserData
293       }
294       toBuild.build()
295     }
296 
297     def canEqual(other: Any): Boolean = other.isInstanceOf[IndexResultsToFeatures[T]]
298 
299     override def equals(other: Any): Boolean = other match {
300       case that: IndexResultsToFeatures[T] if that.canEqual(this) =>
301         sft == that.sft && {
302           if (index == null) { that.index == null } else if (that.index == null) { false } else {
303             index.identifier == that.index.identifier && index.sft == that.index.sft
304           }
305         }
306 
307       case _ => false
308     }
309 
310     override def hashCode(): Int = {
311       val state = Seq(index, sft)
312       state.map(Objects.hashCode).foldLeft(0)((a, b) => 31 * a + b)
313     }
314   }
