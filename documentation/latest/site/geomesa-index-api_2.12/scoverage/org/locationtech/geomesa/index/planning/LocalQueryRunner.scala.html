67   private def run(sft: SimpleFeatureType, query: Query, explain: Explainer)(): CloseableIterator[SimpleFeature] = {
68     explain.pushLevel(s"$name query: '${sft.getTypeName}' ${FilterHelper.toString(query.getFilter)}")
69     explain(s"bin[${query.getHints.isBinQuery}] arrow[${query.getHints.isArrowQuery}] " +
70         s"density[${query.getHints.isDensityQuery}] stats[${query.getHints.isStatsQuery}] " +
71         s"sampling[${query.getHints.getSampling.map { case (s, f) => s"$s${f.map(":" + _).getOrElse("")}"}.getOrElse("none")}]")
72     explain(s"Transforms: ${query.getHints.getTransformDefinition.getOrElse("None")}")
73     explain(s"Sort: ${query.getHints.getSortFields.map(QueryHints.sortReadableString).getOrElse("none")}")
74     explain.popLevel()
75 
76     val filter = Option(query.getFilter).filter(_ != Filter.INCLUDE)
77     val visible = VisibilityUtils.visible(authProvider)
78     val iter = features(sft, filter).filter(visible.apply)
79 
80     var result = transform(sft, iter, query.getHints.getTransform, query.getHints)
81 
82     query.getHints.getMaxFeatures.foreach { maxFeatures =>
83       if (query.getHints.getReturnSft == BinaryOutputEncoder.BinEncodedSft) {
84         // bin queries pack multiple records into each feature
85         // to count the records, we have to count the total bytes coming back, instead of the number of features
86         val label = query.getHints.getBinLabelField.isDefined
87         result = new BinaryOutputEncoder.FeatureLimitingIterator(result, maxFeatures, label)
88       } else {
89         result = result.take(maxFeatures)
90       }
91     }
92 
93     query.getHints.getProjection.foreach { crs =>
94       val r = Reprojection(query.getHints.getReturnSft, crs)
95       result = result.map(r.apply)
96     }
97 
98     result
99   }

122     * Reducer for local transforms. Handles ecql and visibility filtering, transforms and analytic queries.
123     *
124     * @param sft simple feature type being queried
125     * @param hints query hints
126     * @return

128   class LocalTransformReducer(
129       private var sft: SimpleFeatureType,
130       private var filter: Option[Filter],
131       private var visibility: Option[SimpleFeature => Boolean],
132       private var transform: Option[(String, SimpleFeatureType)],
133       private var hints: Hints

135 
136     def this() = this(null, null, None, None, null) // no-arg constructor required for serialization

138     override def init(state: Map[String, String]): Unit = {
139       sft = SimpleFeatureTypes.createType(state("name"), state("spec"))
140       filter = state.get("filt").filterNot(_.isEmpty).map(ECQL.toFilter)
141       hints = ViewParams.deserialize(state("hint"))
142       transform = for {
143         tdef <- state.get("tdef").filterNot(_.isEmpty)
144         tnam <- state.get("tnam").filterNot(_.isEmpty)
145         spec <- state.get("tsft").filterNot(_.isEmpty)
146       } yield {
147         (tdef, SimpleFeatureTypes.createType(tnam, spec))
148       }
149     }
150 
151     override def state: Map[String, String] = {
152       if (visibility.isDefined) {
153         throw new UnsupportedOperationException("Visibility filtering is not serializable")
154       }
155       Map(
156         "name" -> sft.getTypeName,
157         "spec" -> SimpleFeatureTypes.encodeType(sft, includeUserData = true),
158         "filt" -> filter.map(ECQL.toCQL).getOrElse(""),
159         "tdef" -> transform.map(_._1).getOrElse(""),
160         "tnam" -> transform.map(_._2.getTypeName).getOrElse(""),
161         "tsft" -> transform.map(t => SimpleFeatureTypes.encodeType(t._2, includeUserData = true)).getOrElse(""),
162         "hint" -> ViewParams.serialize(hints)
163       )
164     }
165 
166     override def apply(features: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = {
167       val filtered = (filter, visibility) match {
168         case (None, None)       => features
169         case (Some(f), None)    => features.filter(f.evaluate)
170         case (None, Some(v))    => features.filter(v.apply)
171         case (Some(f), Some(v)) => features.filter(feature => v(feature) && f.evaluate(feature))
172       }
173       LocalQueryRunner.transform(sft, filtered, transform, hints)
174     }
175   }

177   /**
178     * Transform plain features into the appropriate return type, based on the hints
179     *
180     * @param sft simple feature type being queried
181     * @param features plain, untransformed features matching the simple feature type
182     * @param hints query hints
183     * @return
184     */
185   def transform(
186       sft: SimpleFeatureType,
187       features: CloseableIterator[SimpleFeature],
188       transform: Option[(String, SimpleFeatureType)],
189       hints: Hints): CloseableIterator[SimpleFeature] = {
190     val sampled = hints.getSampling match {
191       case None => features
192       case Some((percent, field)) => sample(sft, percent, field)(features)
193     }
194     if (hints.isBinQuery) {
195       val trackId = Option(hints.getBinTrackIdField).filter(_ != "id").map(sft.indexOf)
196       val geom = hints.getBinGeomField.map(sft.indexOf)
197       val dtg = hints.getBinDtgField.map(sft.indexOf)
198       val sorting = hints.isBinSorting
199       binTransform(sampled, sft, trackId, geom, dtg, hints.getBinLabelField.map(sft.indexOf), sorting)
200     } else if (hints.isArrowQuery) {
201       arrowTransform(sampled, sft, transform, hints)
202     } else if (hints.isDensityQuery) {
203       val Some(envelope) = hints.getDensityEnvelope
204       val Some((width, height)) = hints.getDensityBounds
205       val geom = DensityScan.getDensityGeometry(sft, hints)
206       densityTransform(sampled, sft, geom, envelope, width, height, hints.getDensityWeight)
207     } else if (hints.isStatsQuery) {
208       statsTransform(sampled, sft, transform, hints.getStatsQuery, hints.isStatsEncode || hints.isSkipReduce)
209     } else {
210       transform match {
211         case None => noTransform(sampled, hints.getSortFields)
212         case Some((defs, tsft)) => projectionTransform(sampled, sft, tsft, defs, hints.getSortFields)
213       }
214     }
215   }

217   private def binTransform(
218       features: CloseableIterator[SimpleFeature],
219       sft: SimpleFeatureType,
220       trackId: Option[Int],
221       geom: Option[Int],
222       dtg: Option[Int],
223       label: Option[Int],
224       sorting: Boolean): CloseableIterator[SimpleFeature] = {
225     val encoder = BinaryOutputEncoder(sft, EncodingOptions(geom, dtg, trackId, label))
226     val sorted = if (!sorting) { features } else {
227       val i = dtg.orElse(sft.getDtgIndex).getOrElse(throw new IllegalArgumentException("Can't sort BIN features by date"))
228       new SortingSimpleFeatureIterator(features, Seq(sft.getDescriptor(i).getLocalName -> false))
229     }
230 
231     val os = new ByteArrayOutputStream(1024)
232     val callback = new ByteStreamCallback(os)
233 
234     new CloseableIterator[SimpleFeature] {
235       override def hasNext: Boolean = sorted.hasNext
236       override def next(): SimpleFeature = {
237         os.reset()
238         sorted.take(64).foreach(encoder.encode(_, callback))
239         new ScalaSimpleFeature(BinaryOutputEncoder.BinEncodedSft, "", Array(os.toByteArray, GeometryUtils.zeroPoint))
240       }
241       override def close(): Unit = sorted.close()
242     }
243   }

245   private def arrowTransform(
246       original: CloseableIterator[SimpleFeature],
247       sft: SimpleFeatureType,
248       transform: Option[(String, SimpleFeatureType)],
249       hints: Hints): CloseableIterator[SimpleFeature] = {
250 
251     val sort = hints.getArrowSort.map(Seq.fill(1)(_))
252     val batchSize = ArrowScan.getBatchSize(hints)
253     val encoding = SimpleFeatureEncoding.min(hints.isArrowIncludeFid, hints.isArrowProxyFid, hints.isFlipAxisOrder)
254     val ipcOpts = FormatVersion.options(hints.getArrowFormatVersion.getOrElse(FormatVersion.ArrowFormatVersion.get))
255 
256     val (features, arrowSft) = transform match {
257       case None => (noTransform(original, sort), sft)
258       case Some((definitions, tsft)) => (projectionTransform(original, sft, tsft, definitions, sort), tsft)
259     }
260 
261     val dictionaryFields = hints.getArrowDictionaryFields
262     val writer = new DeltaWriter(arrowSft, dictionaryFields, encoding, ipcOpts, None, batchSize)
263     val array = Array.ofDim[SimpleFeature](batchSize)
264 
265     val sf = ArrowScan.resultFeature()
266 
267     val arrows = new CloseableIterator[SimpleFeature] {
268       override def hasNext: Boolean = features.hasNext
269       override def next(): SimpleFeature = {
270         var index = 0
271         while (index < batchSize && features.hasNext) {
272           array(index) = features.next
273           index += 1
274         }
275         sf.setAttribute(0, writer.encode(array, index))
276         sf
277       }
278       override def close(): Unit = CloseWithLogging(Seq(features, writer))
279     }
280     if (hints.isSkipReduce) { arrows } else {
281       val process = hints.isArrowProcessDeltas
282       new ArrowScan.DeltaReducer(arrowSft, dictionaryFields, encoding, ipcOpts, batchSize, sort.map(_.head), sorted = true, process)(arrows)
283     }
284   }

286   private def densityTransform(
287       features: CloseableIterator[SimpleFeature],
288       sft: SimpleFeatureType,
289       geom: String,
290       envelope: Envelope,
291       width: Int,
292       height: Int,
293       weight: Option[String]): CloseableIterator[SimpleFeature] = {
294     val renderer = DensityScan.getRenderer(sft, geom, weight)
295     val grid = new RenderingGrid(envelope, width, height)
296     try { features.foreach(renderer.render(grid, _)) } finally { features.close() }
297 
298     val sf = new ScalaSimpleFeature(DensityScan.DensitySft, "", Array(GeometryUtils.zeroPoint))
299     // Return value in user data so it's preserved when passed through a RetypingFeatureCollection
300     sf.getUserData.put(DensityScan.DensityValueKey, DensityScan.encodeResult(grid))
301     CloseableIterator(Iterator(sf))
302   }

304   private def statsTransform(features: CloseableIterator[SimpleFeature],
305                              sft: SimpleFeatureType,
306                              transform: Option[(String, SimpleFeatureType)],
307                              query: String,
308                              encode: Boolean): CloseableIterator[SimpleFeature] = {
309     val (statSft, toObserve) = transform match {
310       case None                => (sft, features)
311       case Some((tdefs, tsft)) => (tsft, projectionTransform(features, sft, tsft, tdefs, None))
312     }
313     val stat = Stat(statSft, query)
314     try { toObserve.foreach(stat.observe) } finally { toObserve.close() }
315     val encoded = if (encode) { StatsScan.encodeStat(statSft)(stat) } else { stat.toJson }
316     CloseableIterator(Iterator(new ScalaSimpleFeature(StatsScan.StatsSft, "stat", Array(encoded, GeometryUtils.zeroPoint))))
317   }

319   private def projectionTransform(
320       features: CloseableIterator[SimpleFeature],
321       sft: SimpleFeatureType,
322       transform: SimpleFeatureType,
323       definitions: String,
324       sort: Option[Seq[(String, Boolean)]]): CloseableIterator[SimpleFeature] = {
325     val transformSf = TransformSimpleFeature(sft, transform, definitions)
326 
327     def setValues(from: SimpleFeature, to: ScalaSimpleFeature): ScalaSimpleFeature = {
328       transformSf.setFeature(from)
329       var i = 0
330       while (i < transform.getAttributeCount) {
331         to.setAttributeNoConvert(i, transformSf.getAttribute(i))
332         i += 1
333       }
334       to.setId(from.getID)
335       to
336     }
337 
338     val result = features.map(setValues(_, new ScalaSimpleFeature(transform, "")))
339 
340     sort match {
341       case None    => result
342       case Some(s) => new SortingSimpleFeatureIterator(result, s)
343     }
344   }

355   /**
356     * Sample the features by selecting a subset
357     *
358     * @param sft simple feature type
359     * @param percent percent of features to keep
360     * @param by field to group by for sampling
361     * @param features features to sample
362     * @return
363     */
364   private def sample(sft: SimpleFeatureType, percent: Float, by: Option[String])
365                     (features: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = {
366     if (!(percent > 0 && percent < 1f)) {
367       throw new IllegalArgumentException(s"Sampling must be a percentage between (0, 1): $percent")
368     }
369     val nth = (1 / percent.toFloat).toInt
370     val field = by.map { name =>
371       val i = sft.indexOf(name)
372       if (i == -1) {
373         throw new IllegalArgumentException(s"Invalid sampling field '$name' for schema " +
374             s"${sft.getTypeName} ${SimpleFeatureTypes.encodeType(sft)}")
375       }
376       i
377     }
378 
379     if (nth <= 1) { features } else {
380       val sample = FeatureSampler.sample(nth, field)
381       features.filter(sample.apply)
382     }
383   }
