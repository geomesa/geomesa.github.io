66   private def run(sft: SimpleFeatureType, query: Query, explain: Explainer)(): CloseableIterator[SimpleFeature] = {
67     explain.pushLevel(s"$name query: '${sft.getTypeName}' ${org.locationtech.geomesa.filter.filterToString(query.getFilter)}")
68     explain(s"bin[${query.getHints.isBinQuery}] arrow[${query.getHints.isArrowQuery}] " +
69         s"density[${query.getHints.isDensityQuery}] stats[${query.getHints.isStatsQuery}] " +
70         s"sampling[${query.getHints.getSampling.map { case (s, f) => s"$s${f.map(":" + _).getOrElse("")}"}.getOrElse("none")}]")
71     explain(s"Transforms: ${query.getHints.getTransformDefinition.getOrElse("None")}")
72     explain(s"Sort: ${query.getHints.getSortFields.map(QueryHints.sortReadableString).getOrElse("none")}")
73     explain.popLevel()
74 
75     val filter = Option(query.getFilter).filter(_ != Filter.INCLUDE)
76     val visible = VisibilityUtils.visible(authProvider)
77     val iter = features(sft, filter).filter(visible.apply)
78 
79     var result = transform(sft, iter, query.getHints.getTransform, query.getHints)
80 
81     query.getHints.getMaxFeatures.foreach { maxFeatures =>
82       if (query.getHints.getReturnSft == BinaryOutputEncoder.BinEncodedSft) {
83         // bin queries pack multiple records into each feature
84         // to count the records, we have to count the total bytes coming back, instead of the number of features
85         val label = query.getHints.getBinLabelField.isDefined
86         result = new BinaryOutputEncoder.FeatureLimitingIterator(result, maxFeatures, label)
87       } else {
88         result = result.take(maxFeatures)
89       }
90     }
91 
92     query.getHints.getProjection.foreach { crs =>
93       val r = Reprojection(query.getHints.getReturnSft, crs)
94       result = result.map(r.apply)
95     }
96 
97     result
98   }

136   class LocalTransformReducer(
137       private var sft: SimpleFeatureType,
138       private var filter: Option[Filter],

144     def this() = this(null, null, None, None, null) // no-arg constructor required for serialization

146     override def init(state: Map[String, String]): Unit = {
147       sft = SimpleFeatureTypes.createType(state("name"), state("spec"))
148       filter = state.get("filt").filterNot(_.isEmpty).map(ECQL.toFilter)
149       hints = ViewParams.deserialize(state("hint"))
150       transform = for {
151         tdef <- state.get("tdef").filterNot(_.isEmpty)
152         tnam <- state.get("tnam").filterNot(_.isEmpty)
153         spec <- state.get("tsft").filterNot(_.isEmpty)
154       } yield {
155         (tdef, SimpleFeatureTypes.createType(tnam, spec))
156       }
157     }
158 
159     override def state: Map[String, String] = {
160       if (visibility.isDefined) {
161         throw new NotImplementedError("Visibility filtering is not serializable")
162       }
163       Map(
164         "name" -> sft.getTypeName,
165         "spec" -> SimpleFeatureTypes.encodeType(sft, includeUserData = true),
166         "filt" -> filter.map(ECQL.toCQL).getOrElse(""),
167         "tdef" -> transform.map(_._1).getOrElse(""),
168         "tnam" -> transform.map(_._2.getTypeName).getOrElse(""),
169         "tsft" -> transform.map(t => SimpleFeatureTypes.encodeType(t._2, includeUserData = true)).getOrElse(""),
170         "hint" -> ViewParams.serialize(hints)
171       )
172     }

174     override def apply(features: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = {
175       val filtered = (filter, visibility) match {
176         case (None, None)       => features
177         case (Some(f), None)    => features.filter(f.evaluate)
178         case (None, Some(v))    => features.filter(v.apply)
179         case (Some(f), Some(v)) => features.filter(feature => v(feature) && f.evaluate(feature))
180       }
181       LocalQueryRunner.transform(sft, filtered, transform, hints)
182     }
183   }

185   /**
186     * Transform plain features into the appropriate return type, based on the hints
187     *
188     * @param sft simple feature type being queried
189     * @param features plain, untransformed features matching the simple feature type
190     * @param hints query hints
191     * @return
192     */
193   def transform(
194       sft: SimpleFeatureType,
195       features: CloseableIterator[SimpleFeature],
196       transform: Option[(String, SimpleFeatureType)],
197       hints: Hints): CloseableIterator[SimpleFeature] = {
198     val sampled = hints.getSampling match {
199       case None => features
200       case Some((percent, field)) => sample(sft, percent, field)(features)
201     }
202     if (hints.isBinQuery) {
203       val trackId = Option(hints.getBinTrackIdField).filter(_ != "id").map(sft.indexOf)
204       val geom = hints.getBinGeomField.map(sft.indexOf)
205       val dtg = hints.getBinDtgField.map(sft.indexOf)
206       val sorting = hints.isBinSorting
207       binTransform(sampled, sft, trackId, geom, dtg, hints.getBinLabelField.map(sft.indexOf), sorting)
208     } else if (hints.isArrowQuery) {
209       arrowTransform(sampled, sft, transform, hints)
210     } else if (hints.isDensityQuery) {
211       val Some(envelope) = hints.getDensityEnvelope
212       val Some((width, height)) = hints.getDensityBounds
213       val geom = DensityScan.getDensityGeometry(sft, hints)
214       densityTransform(sampled, sft, geom, envelope, width, height, hints.getDensityWeight)
215     } else if (hints.isStatsQuery) {
216       statsTransform(sampled, sft, transform, hints.getStatsQuery, hints.isStatsEncode || hints.isSkipReduce)
217     } else {
218       transform match {
219         case None => noTransform(sampled, hints.getSortFields)
220         case Some((defs, tsft)) => projectionTransform(sampled, sft, tsft, defs, hints.getSortFields)
221       }
222     }
223   }

225   private def binTransform(
226       features: CloseableIterator[SimpleFeature],
227       sft: SimpleFeatureType,
228       trackId: Option[Int],
229       geom: Option[Int],
230       dtg: Option[Int],
231       label: Option[Int],
232       sorting: Boolean): CloseableIterator[SimpleFeature] = {
233     val encoder = BinaryOutputEncoder(sft, EncodingOptions(geom, dtg, trackId, label))
234     val sorted = if (!sorting) { features } else {
235       val i = dtg.orElse(sft.getDtgIndex).getOrElse(throw new IllegalArgumentException("Can't sort BIN features by date"))
236       new SortingSimpleFeatureIterator(features, Seq(sft.getDescriptor(i).getLocalName -> false))
237     }
238 
239     val os = new ByteArrayOutputStream(1024)
240     val callback = new ByteStreamCallback(os)
241 
242     new CloseableIterator[SimpleFeature] {
243       override def hasNext: Boolean = sorted.hasNext
244       override def next(): SimpleFeature = {
245         os.reset()
246         sorted.take(64).foreach(encoder.encode(_, callback))
247         new ScalaSimpleFeature(BinaryOutputEncoder.BinEncodedSft, "", Array(os.toByteArray, GeometryUtils.zeroPoint))
248       }
249       override def close(): Unit = sorted.close()
250     }
251   }

253   private def arrowTransform(
254       original: CloseableIterator[SimpleFeature],
255       sft: SimpleFeatureType,
256       transform: Option[(String, SimpleFeatureType)],
257       hints: Hints): CloseableIterator[SimpleFeature] = {
258 
259     val sort = hints.getArrowSort.map(Seq.fill(1)(_))
260     val batchSize = ArrowScan.getBatchSize(hints)
261     val encoding = SimpleFeatureEncoding.min(hints.isArrowIncludeFid, hints.isArrowProxyFid, hints.isFlipAxisOrder)
262     val ipcOpts = FormatVersion.options(hints.getArrowFormatVersion.getOrElse(FormatVersion.ArrowFormatVersion.get))
263 
264     val (features, arrowSft) = transform match {
265       case None => (noTransform(original, sort), sft)
266       case Some((definitions, tsft)) => (projectionTransform(original, sft, tsft, definitions, sort), tsft)
267     }
268 
269     val dictionaryFields = hints.getArrowDictionaryFields
270     val writer = new DeltaWriter(arrowSft, dictionaryFields, encoding, ipcOpts, None, batchSize)
271     val array = Array.ofDim[SimpleFeature](batchSize)
272 
273     val sf = ArrowScan.resultFeature()
274 
275     val arrows = new CloseableIterator[SimpleFeature] {
276       override def hasNext: Boolean = features.hasNext
277       override def next(): SimpleFeature = {
278         var index = 0
279         while (index < batchSize && features.hasNext) {
280           array(index) = features.next
281           index += 1
282         }
283         sf.setAttribute(0, writer.encode(array, index))
284         sf
285       }
286       override def close(): Unit = CloseWithLogging(Seq(features, writer))
287     }
288     if (hints.isSkipReduce) { arrows } else {
289       val process = hints.isArrowProcessDeltas
290       new ArrowScan.DeltaReducer(arrowSft, dictionaryFields, encoding, ipcOpts, batchSize, sort.map(_.head), sorted = true, process)(arrows)
291     }
292   }

294   private def densityTransform(
295       features: CloseableIterator[SimpleFeature],
296       sft: SimpleFeatureType,
297       geom: String,
298       envelope: Envelope,
299       width: Int,
300       height: Int,
301       weight: Option[String]): CloseableIterator[SimpleFeature] = {
302     val renderer = DensityScan.getRenderer(sft, geom, weight)
303     val grid = new RenderingGrid(envelope, width, height)
304     try { features.foreach(renderer.render(grid, _)) } finally { features.close() }

306     val sf = new ScalaSimpleFeature(DensityScan.DensitySft, "", Array(GeometryUtils.zeroPoint))
307     // Return value in user data so it's preserved when passed through a RetypingFeatureCollection
308     sf.getUserData.put(DensityScan.DensityValueKey, DensityScan.encodeResult(grid))
309     CloseableIterator(Iterator(sf))
310   }

312   private def statsTransform(features: CloseableIterator[SimpleFeature],
313                              sft: SimpleFeatureType,
314                              transform: Option[(String, SimpleFeatureType)],
315                              query: String,
316                              encode: Boolean): CloseableIterator[SimpleFeature] = {
317     val (statSft, toObserve) = transform match {
318       case None                => (sft, features)
319       case Some((tdefs, tsft)) => (tsft, projectionTransform(features, sft, tsft, tdefs, None))
320     }
321     val stat = Stat(statSft, query)
322     try { toObserve.foreach(stat.observe) } finally { toObserve.close() }
323     val encoded = if (encode) { StatsScan.encodeStat(statSft)(stat) } else { stat.toJson }
324     CloseableIterator(Iterator(new ScalaSimpleFeature(StatsScan.StatsSft, "stat", Array(encoded, GeometryUtils.zeroPoint))))
325   }

327   private def projectionTransform(
328       features: CloseableIterator[SimpleFeature],
329       sft: SimpleFeatureType,
330       transform: SimpleFeatureType,
331       definitions: String,
332       sort: Option[Seq[(String, Boolean)]]): CloseableIterator[SimpleFeature] = {
333     val transformSf = TransformSimpleFeature(sft, transform, definitions)
334 
335     def setValues(from: SimpleFeature, to: ScalaSimpleFeature): ScalaSimpleFeature = {
336       transformSf.setFeature(from)
337       var i = 0
338       while (i < transform.getAttributeCount) {
339         to.setAttributeNoConvert(i, transformSf.getAttribute(i))
340         i += 1
341       }
342       to.setId(from.getID)
343       to
344     }
345 
346     val result = features.map(setValues(_, new ScalaSimpleFeature(transform, "")))
347 
348     sort match {
349       case None    => result
350       case Some(s) => new SortingSimpleFeatureIterator(result, s)
351     }
352   }

363   /**
364     * Sample the features by selecting a subset
365     *
366     * @param sft simple feature type
367     * @param percent percent of features to keep
368     * @param by field to group by for sampling
369     * @param features features to sample
370     * @return
371     */
372   private def sample(sft: SimpleFeatureType, percent: Float, by: Option[String])
373                     (features: CloseableIterator[SimpleFeature]): CloseableIterator[SimpleFeature] = {
374     if (!(percent > 0 && percent < 1f)) {
375       throw new IllegalArgumentException(s"Sampling must be a percentage between (0, 1): $percent")
376     }
377     val nth = (1 / percent.toFloat).toInt
378     val field = by.map { name =>
379       val i = sft.indexOf(name)
380       if (i == -1) {
381         throw new IllegalArgumentException(s"Invalid sampling field '$name' for schema " +
382             s"${sft.getTypeName} ${SimpleFeatureTypes.encodeType(sft)}")
383       }
384       i
385     }
386 
387     if (nth <= 1) { features } else {
388       val sample = FeatureSampler.sample(nth, field)
389       features.filter(sample.apply)
390     }
