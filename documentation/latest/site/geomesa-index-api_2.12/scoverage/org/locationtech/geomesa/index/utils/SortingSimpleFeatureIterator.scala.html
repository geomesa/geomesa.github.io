28 /**
29   * In memory sorting of simple features, with optional spill to disk
30   *
31   * @param features unsorted feature iterator
32   * @param sortBy attributes to sort by, in the form: (name, reverse).
33   *               for sort by feature id (e.g. natural sort), use an empty string for name
34   */
35 class SortingSimpleFeatureIterator(features: CloseableIterator[SimpleFeature], sortBy: Seq[(String, Boolean)])
36     extends CloseableIterator[SimpleFeature] {
37 
38   import SortingSimpleFeatureIterator.{sortInMemory, sortWithSpillover}
39 
40   private val closed = new AtomicBoolean(false)
41 
42   private lazy val sorted: CloseableIterator[SimpleFeature] = {
43     if (closed.get || !features.hasNext) { features } else {
44       val head = features.next()
45       if (closed.get || !features.hasNext) { CloseableIterator.single(head, features.close()) } else {
46         val ordering = SimpleFeatureOrdering(head.getFeatureType, sortBy)
47         QueryProperties.SortMemoryThreshold.toBytes match {
48           case None => sortInMemory(head, features, ordering, closed)
49           case Some(threshold) => sortWithSpillover(head, features, ordering, closed, threshold)
50         }
51       }
52     }
53   }
54 
55   override def hasNext: Boolean = sorted.hasNext
56 
57   override def next(): SimpleFeature = sorted.next()
58 
59   override def close(): Unit = {
60     if (closed.compareAndSet(false, true)) {
61       sorted.close() // also closes original features iterator
62     }
63   }
64 }

70   /**
71    * Sorts the iterator in memory
72    *
73    * @param head head of iterator
74    * @param tail tail of iterator
75    * @param ordering ordering
76    * @param closed checked to determine if we should stop sorting and return early
77    * @return
78    */
79   private def sortInMemory(
80       head: SimpleFeature,
81       tail: CloseableIterator[SimpleFeature],
82       ordering: Ordering[SimpleFeature],
83       closed: AtomicBoolean): CloseableIterator[SimpleFeature] = {
84     // use ArrayList for sort-in-place of the underlying array
85     val list = new java.util.ArrayList[SimpleFeature](100)
86     list.add(head)
87 
88     while (tail.hasNext && !closed.get) {
89       list.add(tail.next())
90     }
91 
92     if (closed.get) {
93       // don't bother sorting, just return an empty iterator
94       CloseableIterator(Iterator.empty, tail.close())
95     } else {
96       list.sort(ordering)
97       CloseableIterator(list.iterator.asScala, tail.close())
98     }
99   }
100 
101   /**
102    * Sorts the iterator in memory, but spills over to disk if the size in memory exceeds the specified threshold
103    *
104    * @param head head of iterator
105    * @param tail tail of iterator
106    * @param ordering ordering
107    * @param closed checked to determine if we should stop sorting and return early
108    * @param threshold max memory to use before spilling to disk
109    * @return
110    */
111   private def sortWithSpillover(
112       head: SimpleFeature,
113       tail: CloseableIterator[SimpleFeature],
114       ordering: Ordering[SimpleFeature],
115       closed: AtomicBoolean,
116       threshold: Long): CloseableIterator[SimpleFeature] = {
117 
118     // grouping ID for tmp files, doesn't have to be unique
119     lazy val id = {
120       val name = head.getFeatureType.getTypeName.replaceAll("[^A-Za-z0-9_-]", "").take(20)
121       f"$name-${Random.nextInt(10000)}%04d"
122     }
123 
124     val sizable: FeatureIsSizable[SimpleFeature] = head match {
125       case _: SimpleFeature with Sizable =>
126         SizableFeatureIsSizable.asInstanceOf[FeatureIsSizable[SimpleFeature]]
127       case _ =>
128         // currently, all of our uses of this class pass in Sizable features... this means we missed one
129         logger.warn(
130           s"Feature class '${head.getClass.getName}' doesn't implement Sizable - " +
131               "using estimated size for memory threshold calculations")
132         UnSizableFeatureIsSizable
133     }
134 
135     val files = ArrayBuffer.empty[File] // tmp files we create when we exceed in-memory threshold
136     lazy val serializer = KryoFeatureSerializer(head.getFeatureType, SerializationOption.WithUserData)
137     // use ArrayList for sort-in-place of the underlying array
138     val list = new java.util.ArrayList[SimpleFeature](100)
139     list.add(head)
140     var size = sizable.sizeOf(head)
141 
142     while (tail.hasNext && !closed.get) {
143       if (size >= threshold) {
144         // write out the sorted list to disk
145         list.sort(ordering)
146         val file = Files.createTempFile(s"gm-sort-$id-", ".kryo").toFile
147         files += file
148         logger.trace(s"Created temp sort file '${file.getAbsolutePath}'")
149         WithClose(new FileOutputStream(file)) { os =>
150           var i = 0
151           while (i < list.size()) {
152             val bytes = serializer.serialize(list.get(i))
153             os.write(ByteArrays.toBytes(bytes.length))
154             os.write(bytes)
155             i += 1
156           }
157         }
158         list.clear()
159         size = 0
160       }
161       val next = tail.next()
162       list.add(next)
163       size += sizable.sizeOf(next)
164     }
165 
166     if (closed.get) {
167       files.foreach { file =>
168         try {
169           if (!file.delete()) {
170             logger.warn(s"Unable to delete tmp file '${file.getAbsolutePath}''")
171             file.deleteOnExit()
172           }
173         } catch {
174           case NonFatal(e) => logger.warn(s"Unable to delete tmp file '${file.getAbsolutePath}''", e)
175         }
176       }
177       // don't bother sorting, just return an empty iterator
178       CloseableIterator(Iterator.empty, tail.close())
179     } else {
180       if (!list.isEmpty) {
181         list.sort(ordering)
182       }
183       if (files.isEmpty) {
184         CloseableIterator(list.iterator.asScala, tail.close())
185       } else {
186         new MergeSortingIterator(files.toIndexedSeq, serializer, list.iterator.asScala, tail, ordering)
187       }
188     }
189   }
190 
191   /**
192    * Does a merge sort on a group of locally sorted files and a left-over in-memory iterator
193    *
194    * @param files files to merge
195    * @param serializer serializer
196    * @param mem left-over in-memory features, already sorted
197    * @param closeable link to the original (now-empty) feature iterator, for cleaning up on close
198    * @param ordering feature ordering
199    */
200   private class MergeSortingIterator(
201       files: IndexedSeq[File],
202       serializer: SimpleFeatureSerializer,
203       mem: Iterator[SimpleFeature],
204       closeable: CloseableIterator[_],
205       ordering: Ordering[SimpleFeature]
206     ) extends CloseableIterator[SimpleFeature] {
207 
208     // list of feature iterators to merge
209     private val merging = IndexedSeq(CloseableIterator(mem)) ++ files.map(new FileIterator(_, serializer))
210 
211     // priority queue containing the next feature from each file
212     private val heads = {
213       val o = Ordering.by[(SimpleFeature, Int), SimpleFeature](_._1)(ordering)
214       val q = new java.util.PriorityQueue[(SimpleFeature, Int)](files.size + 1, o)
215       var i = 0
216       while (i < merging.length) {
217         val m = merging(i)
218         if (m.hasNext) {
219           q.add(m.next -> i)
220         }
221         i += 1
222       }
223       q
224     }
225 
226     override def hasNext: Boolean = !heads.isEmpty
227 
228     override def next(): SimpleFeature = {
229       val (f, i) = heads.poll()
230       val m = merging(i)
231       if (m.hasNext) {
232         heads.add(m.next -> i)
233       }
234       f
235     }
236 
237     override def close(): Unit = CloseQuietly.raise(merging :+ closeable)
238   }

240   /**
241    * Reads a tmp file of serialized simple features
242    *
243    * @param file file
244    * @param serializer serializer
245    */
246   private class FileIterator(file: File, serializer: SimpleFeatureSerializer)
247       extends CloseableIterator[SimpleFeature] with LazyLogging {
248 
249     private val is = new FileInputStream(file)
250     private var buf = Array.ofDim[Byte](4)
251 
252     override def hasNext: Boolean = is.available() > 0
253 
254     override def next(): SimpleFeature = {
255       is.read(buf, 0, 4)
256       val size = ByteArrays.readInt(buf)
257       if (buf.length < size) {
258         buf = Array.ofDim((size * 1.2).toInt)
259       }
260       is.read(buf, 0, size)
261       serializer.deserialize(buf, 0, size)
262     }
263 
264     override def close(): Unit = {
265       try { is.close() } finally {
266         if (!file.delete()) {
267           logger.warn(s"Unable to delete tmp file '${file.getAbsolutePath}''")
268           file.deleteOnExit()
269         }
270       }
271     }
272   }
