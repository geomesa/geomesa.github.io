31   override protected def createResult(
32       sft: SimpleFeatureType,
33       transform: Option[SimpleFeatureType],
34       batchSize: Int,
35       options: Map[String, String]): ResultCallback = {
36     val geom = options.get(GeomOpt).map(_.toInt).filter(_ != -1)
37     val dtg = options.get(DateOpt).map(_.toInt).filter(_ != -1)
38     val track = options.get(TrackOpt).map(_.toInt).filter(_ != -1)
39     val label = options.get(LabelOpt).map(_.toInt).filter(_ != -1)
40 
41     val encoder = BinaryOutputEncoder(sft, EncodingOptions(geom, dtg, track, label))
42 
43     val binSize = if (label.isEmpty) { 16 } else { 24 }
44     val sort = options(SortOpt).toBoolean
45 
46     val buffer = ByteBuffer.wrap(Array.ofDim(batchSize * binSize)).order(ByteOrder.LITTLE_ENDIAN)
47     val overflow = ByteBuffer.wrap(Array.ofDim(binSize * 16)).order(ByteOrder.LITTLE_ENDIAN)
48 
49     new ResultCallback(buffer, overflow, encoder, binSize, sort)
50   }
51 
52   override protected def defaultBatchSize: Int =
53     throw new IllegalArgumentException("Batch scan is specified per scan")

118   class ResultCallback(
119       buffer: ByteBuffer,
120       private var overflow: ByteBuffer,
121       encoder: BinaryOutputEncoder,
122       binSize: Int,
123       sort: Boolean
124     ) extends AggregatingScan.Result with BinaryOutputCallback {
125 
126     override def apply(trackId: Int, lat: Float, lon: Float, dtg: Long): Unit =
127       put(ensureCapacity(16), trackId, lat, lon, dtg)
128 
129     override def apply(trackId: Int, lat: Float, lon: Float, dtg: Long, label: Long): Unit =
130       put(ensureCapacity(24), trackId, lat, lon, dtg, label)
131 
132     override def init(): Unit = {}
133 
134     override def aggregate(sf: SimpleFeature): Int = {
135       val pos = buffer.position() + overflow.position()
136       encoder.encode(sf, this)
137       (buffer.position() + overflow.position() - pos) / binSize
138     }
139 
140     override def encode(): Array[Byte] = {
141       val bytes = try {
142         if (overflow.position() > 0) {
143           // overflow bytes - copy the two buffers into one
144           val copy = Array.ofDim[Byte](buffer.position() + overflow.position())
145           System.arraycopy(buffer.array, 0, copy, 0, buffer.position())
146           System.arraycopy(overflow.array, 0, copy, buffer.position(), overflow.position())
147           copy
148         } else if (buffer.position() == buffer.limit()) {
149           // use the existing buffer if possible
150           buffer.array
151         } else {
152           // if not, we have to copy it - values do not allow you to specify a valid range
153           val copy = Array.ofDim[Byte](buffer.position())
154           System.arraycopy(buffer.array, 0, copy, 0, buffer.position())
155           copy
156         }
157       } finally {
158         buffer.clear()
159         overflow.clear()
160       }
161       if (sort) {
162         BinSorter.quickSort(bytes, 0, bytes.length - binSize, binSize)
163       }
164       bytes
165     }
166 
167     override def cleanup(): Unit = {}
168 
169     private def ensureCapacity(size: Int): ByteBuffer = {
170       if (buffer.position() < buffer.limit() - size) {
171         buffer
172       } else if (overflow.position() < overflow.limit() - size) {
173         overflow
174       } else {
175         val expanded = Array.ofDim[Byte](overflow.limit() * 2)
176         System.arraycopy(overflow.array, 0, expanded, 0, overflow.limit())
177         val order = overflow.order()
178         val position = overflow.position()
179         overflow = ByteBuffer.wrap(expanded).order(order).position(position).asInstanceOf[ByteBuffer]
180         overflow
181       }
182     }
183   }
