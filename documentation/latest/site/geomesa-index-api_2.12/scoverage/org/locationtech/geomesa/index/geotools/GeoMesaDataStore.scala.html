117   @throws(classOf[IllegalArgumentException])
118   override protected def preSchemaCreate(sft: SimpleFeatureType): Unit = {
119     // check for old user data keys and re-map them
120     InternalConfigs.DeprecatedConfigMappings.foreach { case (from, to) =>
121       val v = sft.getUserData.remove(from)
122       if (v != null) {
123         sft.getUserData.put(to, v)
124       }
125     }
126 
127     // validate column groups
128     adapter.groups.validate(sft)
129 
130     // disable table sharing, no longer supported
131     // noinspection ScalaDeprecation
132     if (sft.isTableSharing) {
133       logger.warn("Table sharing is no longer supported - disabling table sharing")
134       sft.getUserData.remove(Configs.TableSharing)
135       sft.getUserData.remove(InternalConfigs.TableSharingPrefix)
136     }
137 
138     // configure the indices to use
139     if (sft.getIndices.isEmpty) {
140       val indices = GeoMesaFeatureIndexFactory.indices(sft)
141       if (indices.isEmpty) {
142         throw new IllegalArgumentException("There are no available indices that support the schema " +
143             SimpleFeatureTypes.encodeType(sft))
144       }
145       sft.setIndices(indices)
146     }
147 
148     // try to create the indices up front to ensure they are valid for the sft
149     GeoMesaFeatureIndexFactory.create(this, sft, sft.getIndices)
150 
151     // remove the enabled indices after configuration so we don't persist them
152     sft.getUserData.remove(SimpleFeatureTypes.Configs.EnabledIndices)
153     // remove any 'index' flags in the attribute metadata, they have already been captured in the indices above
154     sft.getAttributeDescriptors.asScala.foreach(_.getUserData.remove(AttributeOptions.OptIndex))
155 
156     // for partitioned schemas, persist the table partitioning keys
157     if (sft.isPartitioned) {
158       InternalConfigs.PartitionConfigMappings.foreach { case (from, to) =>
159         Option(sft.getUserData.get(from)).foreach(sft.getUserData.put(to, _))
160       }
161       InternalConfigs.PartitionConfigPrefixMappings.foreach { case (from, to) =>
162         sft.getUserData.asScala.toMap.collect {
163           case (k: String, v) if k.startsWith(from) => sft.getUserData.put(to + k.substring(from.length), v)
164         }
165       }
166     }
167 
168     // set stats enabled based on the data store config if not explicitly set
169     if (!sft.getUserData.containsKey(SimpleFeatureTypes.Configs.StatsEnabled)) {
170       sft.setStatsEnabled(config.generateStats)
171     }
172 
173     sft.getFeatureExpiration // validate any configured age-off
174   }

198   // create the new index tables (if not using partitioned tables)
199   override protected def onSchemaUpdated(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
200     val partitioned = TablePartition.partitioned(sft)
201 
202     // check for column renaming
203     val colMap = getColumnMap(previous, sft).map { case (k, v) => (v, k) }
204 
205     // rewrite the table name keys for the renamed cols
206     if (colMap.nonEmpty) {
207       sft.getIndices.foreach { i =>
208         if (i.attributes.exists(colMap.contains)) {
209           val prev = i.copy(attributes = i.attributes.map(a => colMap.getOrElse(a, a)))
210           val old = manager.index(previous, GeoMesaFeatureIndex.identifier(prev))
211           val index = GeoMesaFeatureIndexFactory.create(this, sft, Seq(i)).headOption.getOrElse {
212             throw new IllegalArgumentException(
213               s"Error configuring new feature index: ${GeoMesaFeatureIndex.identifier(i)}")
214           }
215           val partitions = if (!partitioned) { Seq(None) } else {
216             // have to use the old table name key but the new sft name for looking up the partitions
217             val tableNameKey = old.tableNameKey(Some(""))
218             val offset = tableNameKey.length
219             metadata.scan(sft.getTypeName, tableNameKey).map { case (k, _) => Some(k.substring(offset)) }
220           }
221           partitions.foreach { p =>
222             metadata.read(sft.getTypeName, old.tableNameKey(p)).foreach { v =>
223               metadata.insert(sft.getTypeName, index.tableNameKey(p), v)
224             }
225           }
226         }
227       }
228     }
229 
230     // configure any new indices
231     if (partitioned) {
232       logger.debug("Delaying creation of partitioned indices")
233     } else {
234       logger.debug(s"Ensuring indices ${manager.indices(sft).map(_.identifier).mkString(", ")}")
235       manager.indices(sft).toList.map(i => CachedThreadPool.submit(() => adapter.createTable(i, None, i.getSplits(None)))).foreach(_.get)
236     }
237 
238     // update stats
239     if (previous.statsEnabled) {
240       if (!sft.statsEnabled) {
241         stats.writer.clear(previous)
242       } else if (sft.getTypeName != previous.getTypeName || colMap.nonEmpty) {
243         stats.writer.rename(sft, previous)
244       }
245     }
246 
247     // rename tables to match the new sft name
248     if (sft.getTypeName != previous.getTypeName || sft.getIndices != previous.getIndices) {
249       if (FastConverter.convertOrElse[java.lang.Boolean](sft.getUserData.get(Configs.UpdateRenameTables), false)) {
250         manager.indices(sft).foreach { index =>
251           val partitions = if (partitioned) { index.getPartitions.map(Option.apply) } else { Seq(None) }
252           partitions.foreach { partition =>
253             val key = index.tableNameKey(partition)
254             metadata.read(sft.getTypeName, key).foreach { table =>
255               metadata.remove(sft.getTypeName, key)
256               val renamed = index.configureTableName(partition, adapter.tableNameLimit)
257               if (renamed != table) {
258                 logger.debug(s"Renaming table from '$table' to '$renamed'")
259                 adapter.renameTable(table, renamed)
260               }
261             }
262           }
263         }
264       }
265     }
266   }

268   // delete the index tables
269   override protected def onSchemaDeleted(sft: SimpleFeatureType): Unit = {
270     def deleteFull(index: GeoMesaFeatureIndex[_, _]): Unit =
271       adapter.deleteTables(index.deleteTableNames(None))
272     def deleteShared(index: GeoMesaFeatureIndex[_, _]): Unit = {
273       if (index.keySpace.sharing.isEmpty) { deleteFull(index) } else {
274         adapter.clearTables(index.deleteTableNames(None), Some(index.keySpace.sharing))
275       }
276     }
277 
278     val indices = manager.indices(sft).toList
279     // noinspection ScalaDeprecation
280     if (sft.isTableSharing && getTypeNames.exists(t => t != sft.getTypeName && getSchema(t).isTableSharing)) {
281       indices.map(i => CachedThreadPool.submit(() => deleteShared(i))).foreach(_.get)
282     } else {
283       indices.map(i => CachedThreadPool.submit(() => deleteFull(i))).foreach(_.get)
284     }
285     if (sft.statsEnabled) {
286       stats.writer.clear(sft)
287     }
288   }

292   /**
293    * @see org.geotools.api.data.DataStore#getSchema(java.lang.String)
294    * @param typeName feature type name
295    * @return feature type, or null if it does not exist
296    */
297   override def getSchema(typeName: String): SimpleFeatureType = {
298     var sft = super.getSchema(typeName)
299     if (sft != null) {
300       // ensure index metadata is correct
301       if (sft.getIndices.exists(i => i.attributes.isEmpty && i.name != IdIndex.name)) {
302         sft = SimpleFeatureTypes.mutable(sft)
303         // migrate index metadata to standardized versions and attributes
304         transitionIndices(sft)
305         sft = SimpleFeatureTypes.immutable(sft)
306         // validate indices
307         try { manager.indices(sft) } catch {
308           case NonFatal(e) =>
309             throw new IllegalStateException(s"The schema ${sft.getTypeName} was written with a older " +
310                 "version of GeoMesa that is no longer supported. You may continue to use an older client, or " +
311                 s"manually edit the metadata for '${InternalConfigs.IndexVersions}' to exclude the invalid indices.", e)
312         }
313       } else {
314         // validate indices
315         try { manager.indices(sft) } catch {
316           case NonFatal(e) =>
317             val versions = sft.getIndices.map(i => s"${i.name}:${i.version}").mkString(",")
318             val available = GeoMesaFeatureIndexFactory.all().map(i => s"${i.name}:${i.version}").mkString(",")
319             logger.error(s"Trying to access schema ${sft.getTypeName} with invalid index versions '$versions' - " +
320                 s"available indices are '$available'", e)
321             throw new IllegalStateException(s"The schema ${sft.getTypeName} was written with a newer " +
322                 "version of GeoMesa than this client can handle. Please ensure that you are using the " +
323                 "same GeoMesa jar versions across your entire workflow. For more information, see " +
324                 "https://www.geomesa.org/documentation/stable/user/upgrade.html")
325         }
326       }
327 
328       // check for sft-level stats flag and set it if not present
329       if (!sft.getUserData.containsKey(SimpleFeatureTypes.Configs.StatsEnabled)) {
330         val extra = Collections.singletonMap(SimpleFeatureTypes.Configs.StatsEnabled, config.generateStats.toString)
331         sft = SimpleFeatureTypes.immutable(sft, extra)
332       }
333 
334       // get the remote version if it's available, but don't wait for it
335       GeoMesaDataStore.versions.get(VersionKey(this)).getNow(Right(None)) match {
336         case Left(e) => throw e
337         case Right(version) =>
338           version.foreach { v =>
339             val userData = Collections.singletonMap[AnyRef, AnyRef](InternalConfigs.RemoteVersion, v.toString)
340             sft = SimpleFeatureTypes.immutable(sft, userData)
341           }
342       }
343     }
344     sft
345   }

462    * @param sft udpated simple feature type
463    * @return
464    */
465   def checkSchemaCompatibility(typeName: String, sft: SimpleFeatureType): SchemaCompatibility = {
466     val previous = getSchema(typeName)
467     if (previous == null) {
468       new SchemaCompatibility.DoesNotExist(this, sft)
469     } else {
470       validateSchemaUpdate(previous, sft) match {
471         case Right(TypeComparison.Compatible(extension, renamed, _)) =>
472           val copy = SimpleFeatureTypes.copy(sft)
473           updateSchemaUserData(copy, previous)
474           if (!extension && !renamed && copy.getUserData == previous.getUserData) {
475             SchemaCompatibility.Unchanged
476           } else {
477             new SchemaCompatibility.Compatible(this, typeName, sft)
478           }
479 
480         case Left(e) =>
481           SchemaCompatibility.Incompatible(e)
482       }
483     }
484   }

488   /**
489    * Updates the user data for a schema update prior to persistence. Handles converting existing
490    * user data based on new/updated attributes, copying existing user data over, etc. Feature
491    * type will be updated in place
492    *
493    * @param sft schema update
494    * @param previous existing schema
495    * @return
496    */
497   private def updateSchemaUserData(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
498 
499     // check for column renaming
500     val colMap = getColumnMap(previous, sft)
501 
502     def remapCol(name: String): String = colMap.getOrElse(name, name)
503 
504     val indices = {
505       // set directly in the sft
506       val explicitIndices = (previous.getIndices ++ sft.getIndices).map(i => i.copy(attributes = i.attributes.map(remapCol)))
507       // set in index flags or geomesa.indices.enabled
508       val configuredIndices =
509         // don't add in a fid index unless explicitly enabled
510         if (sft.getUserData.get(Configs.EnableFidIndex) == null &&
511             Option(sft.getUserData.get(Configs.EnabledIndices))
512               .forall(!_.toString.split(",").map(_.trim.toLowerCase(Locale.US)).exists(_.matches(s"^${IdIndex.name}:?")))) {
513           GeoMesaFeatureIndexFactory.indices(sft).filterNot(_.name == IdIndex.name)
514         } else {
515           GeoMesaFeatureIndexFactory.indices(sft)
516         }
517       // get the unique indices - note: ignores index version
518       (explicitIndices ++ configuredIndices).foldLeft(Seq.empty[IndexId]) { (sum, next) =>
519         if (sum.exists(i => i.name == next.name && i.attributes == next.attributes)) {
520           sum
521         } else {
522           sum :+ next
523         }
524       }
525     }
526     sft.setIndices(indices)
527 
528     // preserve any existing user data but overwrite any keys we redefine
529     val userData = new java.util.HashMap[AnyRef, AnyRef](previous.getUserData)
530     userData.putAll(sft.getUserData)
531     sft.getUserData.putAll(userData)
532     // remove enabled indices as we don't need to persist it
533     sft.getUserData.remove(Configs.EnabledIndices)
534     // remove any null/empty keys as a way to delete existing user data
535     sft.getUserData.asScala.collect { case (k, null | "") => k }.foreach(sft.getUserData.remove)
536     // remove any 'index' flags in the attribute user data - we need them above for checking enabled indices
537     sft.getAttributeDescriptors.asScala.foreach(_.getUserData.remove(AttributeOptions.OptIndex))
538   }

567   private val loader: AsyncCacheLoader[VersionKey, Either[Exception, Option[SemanticVersion]]] =
568     new CacheLoader[VersionKey, Either[Exception, Option[SemanticVersion]]]() {
569       override def load(key: VersionKey): Either[Exception, Option[SemanticVersion]] = {
570         if (CheckDistributedVersion.toBoolean.contains(false)) { Right(None) } else {
571           val ds = Option(liveStores.get(key)).flatMap(_.asScala.find(_.getTypeNames.nonEmpty)).orNull
572           if (ds == null) {
573             // short-circuit load - should try again next time cache is accessed
