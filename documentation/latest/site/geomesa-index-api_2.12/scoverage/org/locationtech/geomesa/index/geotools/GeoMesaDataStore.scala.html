116   @throws(classOf[IllegalArgumentException])
117   override protected def preSchemaCreate(sft: SimpleFeatureType): Unit = {
118     // check for old enabled indices and re-map them
119     // noinspection ScalaDeprecation
120     SimpleFeatureTypes.Configs.ENABLED_INDEX_OPTS.drop(1).find(sft.getUserData.containsKey).foreach { key =>
121       sft.getUserData.put(SimpleFeatureTypes.Configs.EnabledIndices, sft.getUserData.remove(key))
122     }
123 
124     // validate column groups
125     adapter.groups.validate(sft)
126 
127     // disable table sharing, no longer supported
128     // noinspection ScalaDeprecation
129     if (sft.isTableSharing) {
130       logger.warn("Table sharing is no longer supported - disabling table sharing")
131       sft.getUserData.remove(Configs.TableSharing)
132       sft.getUserData.remove(InternalConfigs.TableSharingPrefix)
133     }
134 
135     // configure the indices to use
136     if (sft.getIndices.isEmpty) {
137       val indices = GeoMesaFeatureIndexFactory.indices(sft)
138       if (indices.isEmpty) {
139         throw new IllegalArgumentException("There are no available indices that support the schema " +
140             SimpleFeatureTypes.encodeType(sft))
141       }
142       sft.setIndices(indices)
143     }
144 
145     // try to create the indices up front to ensure they are valid for the sft
146     GeoMesaFeatureIndexFactory.create(this, sft, sft.getIndices)
147 
148     // remove the enabled indices after configuration so we don't persist them
149     sft.getUserData.remove(SimpleFeatureTypes.Configs.EnabledIndices)
150     // remove any 'index' flags in the attribute metadata, they have already been captured in the indices above
151     sft.getAttributeDescriptors.asScala.foreach(_.getUserData.remove(AttributeOptions.OptIndex))
152 
153     // for partitioned schemas, persist the table partitioning keys
154     if (sft.isPartitioned) {
155       InternalConfigs.PartitionConfigMappings.foreach { case (from, to) =>
156         Option(sft.getUserData.get(from)).foreach(sft.getUserData.put(to, _))
157       }
158     }
159 
160     // set stats enabled based on the data store config if not explicitly set
161     if (!sft.getUserData.containsKey(SimpleFeatureTypes.Configs.StatsEnabled)) {
162       sft.setStatsEnabled(config.generateStats)
163     }
164 
165     sft.getFeatureExpiration // validate any configured age-off
166   }

190   // create the new index tables (if not using partitioned tables)
191   override protected def onSchemaUpdated(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
192     val partitioned = TablePartition.partitioned(sft)
193 
194     // check for column renaming
195     val colMap = getColumnMap(previous, sft).map { case (k, v) => (v, k) }
196 
197     // rewrite the table name keys for the renamed cols
198     if (colMap.nonEmpty) {
199       sft.getIndices.foreach { i =>
200         if (i.attributes.exists(colMap.contains)) {
201           val prev = i.copy(attributes = i.attributes.map(a => colMap.getOrElse(a, a)))
202           val old = manager.index(previous, GeoMesaFeatureIndex.identifier(prev))
203           val index = GeoMesaFeatureIndexFactory.create(this, sft, Seq(i)).headOption.getOrElse {
204             throw new IllegalArgumentException(
205               s"Error configuring new feature index: ${GeoMesaFeatureIndex.identifier(i)}")
206           }
207           val partitions = if (!partitioned) { Seq(None) } else {
208             // have to use the old table name key but the new sft name for looking up the partitions
209             val tableNameKey = old.tableNameKey(Some(""))
210             val offset = tableNameKey.length
211             metadata.scan(sft.getTypeName, tableNameKey).map { case (k, _) => Some(k.substring(offset)) }
212           }
213           partitions.foreach { p =>
214             metadata.read(sft.getTypeName, old.tableNameKey(p)).foreach { v =>
215               metadata.insert(sft.getTypeName, index.tableNameKey(p), v)
216             }
217           }
218         }
219       }
220     }
221 
222     // configure any new indices
223     if (partitioned) {
224       logger.debug("Delaying creation of partitioned indices")
225     } else {
226       logger.debug(s"Ensuring indices ${manager.indices(sft).map(_.identifier).mkString(", ")}")
227       manager.indices(sft).toList.map(i => CachedThreadPool.submit(() => adapter.createTable(i, None, i.getSplits(None)))).foreach(_.get)
228     }
229 
230     // update stats
231     if (previous.statsEnabled) {
232       if (!sft.statsEnabled) {
233         stats.writer.clear(previous)
234       } else if (sft.getTypeName != previous.getTypeName || colMap.nonEmpty) {
235         stats.writer.rename(sft, previous)
236       }
237     }
238 
239     // rename tables to match the new sft name
240     if (sft.getTypeName != previous.getTypeName || sft.getIndices != previous.getIndices) {
241       if (FastConverter.convertOrElse[java.lang.Boolean](sft.getUserData.get(Configs.UpdateRenameTables), false)) {
242         manager.indices(sft).foreach { index =>
243           val partitions = if (partitioned) { index.getPartitions.map(Option.apply) } else { Seq(None) }
244           partitions.foreach { partition =>
245             val key = index.tableNameKey(partition)
246             metadata.read(sft.getTypeName, key).foreach { table =>
247               metadata.remove(sft.getTypeName, key)
248               val renamed = index.configureTableName(partition, adapter.tableNameLimit)
249               if (renamed != table) {
250                 logger.debug(s"Renaming table from '$table' to '$renamed'")
251                 adapter.renameTable(table, renamed)
252               }
253             }
254           }
255         }
256       }
257     }
258   }

260   // delete the index tables
261   override protected def onSchemaDeleted(sft: SimpleFeatureType): Unit = {
262     def deleteFull(index: GeoMesaFeatureIndex[_, _]): Unit =
263       adapter.deleteTables(index.deleteTableNames(None))
264     def deleteShared(index: GeoMesaFeatureIndex[_, _]): Unit = {
265       if (index.keySpace.sharing.isEmpty) { deleteFull(index) } else {
266         adapter.clearTables(index.deleteTableNames(None), Some(index.keySpace.sharing))
267       }
268     }
269 
270     val indices = manager.indices(sft).toList
271     // noinspection ScalaDeprecation
272     if (sft.isTableSharing && getTypeNames.exists(t => t != sft.getTypeName && getSchema(t).isTableSharing)) {
273       indices.map(i => CachedThreadPool.submit(() => deleteShared(i))).foreach(_.get)
274     } else {
275       indices.map(i => CachedThreadPool.submit(() => deleteFull(i))).foreach(_.get)
276     }
277     if (sft.statsEnabled) {
278       stats.writer.clear(sft)
279     }
280   }

284   /**
285    * @see org.geotools.api.data.DataStore#getSchema(java.lang.String)
286    * @param typeName feature type name
287    * @return feature type, or null if it does not exist
288    */
289   override def getSchema(typeName: String): SimpleFeatureType = {
290     var sft = super.getSchema(typeName)
291     if (sft != null) {
292       // ensure index metadata is correct
293       if (sft.getIndices.exists(i => i.attributes.isEmpty && i.name != IdIndex.name)) {
294         sft = SimpleFeatureTypes.mutable(sft)
295         // migrate index metadata to standardized versions and attributes
296         transitionIndices(sft)
297         sft = SimpleFeatureTypes.immutable(sft)
298         // validate indices
299         try { manager.indices(sft) } catch {
300           case NonFatal(e) =>
301             throw new IllegalStateException(s"The schema ${sft.getTypeName} was written with a older " +
302                 "version of GeoMesa that is no longer supported. You may continue to use an older client, or " +
303                 s"manually edit the metadata for '${InternalConfigs.IndexVersions}' to exclude the invalid indices.", e)
304         }
305       } else {
306         // validate indices
307         try { manager.indices(sft) } catch {
308           case NonFatal(e) =>
309             val versions = sft.getIndices.map(i => s"${i.name}:${i.version}").mkString(",")
310             val available = GeoMesaFeatureIndexFactory.available(sft).map(i => s"${i._1}:${i._2}").mkString(",")
311             logger.error(s"Trying to access schema ${sft.getTypeName} with invalid index versions '$versions' - " +
312                 s"available indices are '$available'", e)
313             throw new IllegalStateException(s"The schema ${sft.getTypeName} was written with a newer " +
314                 "version of GeoMesa than this client can handle. Please ensure that you are using the " +
315                 "same GeoMesa jar versions across your entire workflow. For more information, see " +
316                 "http://www.geomesa.org/documentation/user/installation_and_configuration.html#upgrading")
317         }
318       }
319 
320       // check for sft-level stats flag and set it if not present
321       if (!sft.getUserData.containsKey(SimpleFeatureTypes.Configs.StatsEnabled)) {
322         val extra = Collections.singletonMap(SimpleFeatureTypes.Configs.StatsEnabled, config.generateStats.toString)
323         sft = SimpleFeatureTypes.immutable(sft, extra)
324       }
325 
326       // get the remote version if it's available, but don't wait for it
327       GeoMesaDataStore.versions.get(VersionKey(this)).getNow(Right(None)) match {
328         case Left(e) => throw e
329         case Right(version) =>
330           version.foreach { v =>
331             val userData = Collections.singletonMap[AnyRef, AnyRef](InternalConfigs.RemoteVersion, v.toString)
332             sft = SimpleFeatureTypes.immutable(sft, userData)
333           }
334       }
335     }
336     sft
337   }

454    * @param typeName type name

456    * @return
457    */
458   def checkSchemaCompatibility(typeName: String, sft: SimpleFeatureType): SchemaCompatibility = {
459     val previous = getSchema(typeName)
460     if (previous == null) {
461       new SchemaCompatibility.DoesNotExist(this, sft)
462     } else {
463       validateSchemaUpdate(previous, sft) match {
464         case Right(TypeComparison.Compatible(extension, renamed, _)) =>
465           val copy = SimpleFeatureTypes.copy(sft)
466           updateSchemaUserData(copy, previous)
467           if (!extension && !renamed && copy.getUserData == previous.getUserData) {
468             SchemaCompatibility.Unchanged
469           } else {
470             new SchemaCompatibility.Compatible(this, typeName, sft)
471           }
472 
473         case Left(e) =>
474           SchemaCompatibility.Incompatible(e)
475       }
476     }
477   }

481   /**
482    * Updates the user data for a schema update prior to persistence. Handles converting existing
483    * user data based on new/updated attributes, copying existing user data over, etc. Feature
484    * type will be updated in place
485    *
486    * @param sft schema update
487    * @param previous existing schema
488    * @return
489    */
490   private def updateSchemaUserData(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
491 
492     // check for column renaming
493     val colMap = getColumnMap(previous, sft)
494 
495     def remapCol(name: String): String = colMap.getOrElse(name, name)
496 
497     // check for attributes flagged 'index' and convert them to sft-level user data
498     def indexed(d: AttributeDescriptor): Boolean = {
499       d.getUserData.get(AttributeOptions.OptIndex) match {
500         case i: String if Seq("true", "full").exists(_.equalsIgnoreCase(i)) => true
501         case i if i == null || Seq("false", "none").exists(_.equalsIgnoreCase(i.toString)) => false
502         case i => throw new IllegalArgumentException(s"Configured index coverage '$i' is not valid: expected 'true'")
503       }
504     }
505     sft.getAttributeDescriptors.asScala.foreach { d =>
506       if (indexed(d)) {
507         val existing = {
508           val explicit = sft.getIndices
509           if (explicit.nonEmpty) { explicit } else { previous.getIndices }
510         }
511         if (!existing.exists(e => e.name == AttributeIndex.name && remapCol(e.attributes.head) == d.getLocalName)) {
512           val fields = Seq(d.getLocalName) ++ Option(sft.getGeomField) ++ sft.getDtgField
513           val id = IndexId(AttributeIndex.name, AttributeIndex.version, fields, IndexMode.ReadWrite)
514           sft.setIndices(existing :+ id)
515         }
516       }
517     }
518 
519     // check for new indices and 'enabled indices' changes
520     val indices = {
521       val enabled = if (!sft.getUserData.containsKey(Configs.EnabledIndices)) { Seq.empty } else {
522         GeoMesaFeatureIndexFactory.indices(sft)
523       }
524       val remapped = (previous.getIndices ++ sft.getIndices).map(i => i.copy(attributes = i.attributes.map(remapCol)))
525       (remapped ++ enabled).foldLeft(Seq.empty[IndexId]) { (sum, next) =>
526         // note: ignore index version
527         if (sum.exists(i => i.name == next.name && i.attributes == next.attributes)) { sum } else { sum :+ next }
528       }
529     }
530     if (indices != previous.getIndices) {
531       sft.setIndices(indices)
532     }
533 
534     // preserve any existing user data but overwrite any keys we redefine
535     val userData = new java.util.HashMap[AnyRef, AnyRef](previous.getUserData)
536     userData.putAll(sft.getUserData)
537     sft.getUserData.putAll(userData)
538     // remove enabled indices as we don't need to persist it
539     sft.getUserData.remove(Configs.EnabledIndices)
540     // remove any null/empty keys as a way to delete existing user data
541     sft.getUserData.asScala.collect { case (k, null | "") => k }.foreach(sft.getUserData.remove)
542     // remove any 'index' flags in the attribute user data - we need them above for checking enabled indices
543     sft.getAttributeDescriptors.asScala.foreach(_.getUserData.remove(AttributeOptions.OptIndex))
544   }

546   /**
547    * Gets a map of column renames, for use during schema updates
548    *
549    * @param previous previous feature type
550    * @param sft updated feature type
551    * @return map of old name -> new name
552    */
553   private def getColumnMap(previous: SimpleFeatureType, sft: SimpleFeatureType): Map[String, String] = {
554     previous.getAttributeDescriptors.asScala.zipWithIndex.toMap.flatMap { case (prev, i) =>
555       val cur = sft.getDescriptor(i)
556       if (prev.getLocalName != cur.getLocalName) {
557         Map(prev.getLocalName -> cur.getLocalName)
558       } else {
559         Map.empty[String, String]
560       }
561     }
