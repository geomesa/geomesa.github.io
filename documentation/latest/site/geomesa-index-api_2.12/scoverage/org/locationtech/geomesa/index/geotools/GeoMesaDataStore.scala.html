116   @throws(classOf[IllegalArgumentException])
117   override protected def preSchemaCreate(sft: SimpleFeatureType): Unit = {
118     // check for old user data keys and re-map them
119     InternalConfigs.DeprecatedConfigMappings.foreach { case (from, to) =>
120       val v = sft.getUserData.remove(from)
121       if (v != null) {
122         sft.getUserData.put(to, v)
123       }
124     }
125 
126     // validate column groups
127     adapter.groups.validate(sft)
128 
129     // disable table sharing, no longer supported
130     // noinspection ScalaDeprecation
131     if (sft.isTableSharing) {
132       logger.warn("Table sharing is no longer supported - disabling table sharing")
133       sft.getUserData.remove(Configs.TableSharing)
134       sft.getUserData.remove(InternalConfigs.TableSharingPrefix)
135     }
136 
137     // configure the indices to use
138     if (sft.getIndices.isEmpty) {
139       val indices = GeoMesaFeatureIndexFactory.indices(sft)
140       if (indices.isEmpty) {
141         throw new IllegalArgumentException("There are no available indices that support the schema " +
142             SimpleFeatureTypes.encodeType(sft))
143       }
144       sft.setIndices(indices)
145     }
146 
147     // try to create the indices up front to ensure they are valid for the sft
148     GeoMesaFeatureIndexFactory.create(this, sft, sft.getIndices)
149 
150     // remove the enabled indices after configuration so we don't persist them
151     sft.getUserData.remove(SimpleFeatureTypes.Configs.EnabledIndices)
152     // remove any 'index' flags in the attribute metadata, they have already been captured in the indices above
153     sft.getAttributeDescriptors.asScala.foreach(_.getUserData.remove(AttributeOptions.OptIndex))
154 
155     // for partitioned schemas, persist the table partitioning keys
156     if (sft.isPartitioned) {
157       InternalConfigs.PartitionConfigMappings.foreach { case (from, to) =>
158         Option(sft.getUserData.get(from)).foreach(sft.getUserData.put(to, _))
159       }
160       InternalConfigs.PartitionConfigPrefixMappings.foreach { case (from, to) =>
161         sft.getUserData.asScala.toMap.collect {
162           case (k: String, v) if k.startsWith(from) => sft.getUserData.put(to + k.substring(from.length), v)
163         }
164       }
165     }
166 
167     // set stats enabled based on the data store config if not explicitly set
168     if (!sft.getUserData.containsKey(SimpleFeatureTypes.Configs.StatsEnabled)) {
169       sft.setStatsEnabled(config.generateStats)
170     }
171 
172     sft.getFeatureExpiration // validate any configured age-off
173   }

197   // create the new index tables (if not using partitioned tables)
198   override protected def onSchemaUpdated(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
199     val partitioned = TablePartition.partitioned(sft)
200 
201     // check for column renaming
202     val colMap = getColumnMap(previous, sft).map { case (k, v) => (v, k) }
203 
204     // rewrite the table name keys for the renamed cols
205     if (colMap.nonEmpty) {
206       sft.getIndices.foreach { i =>
207         if (i.attributes.exists(colMap.contains)) {
208           val prev = i.copy(attributes = i.attributes.map(a => colMap.getOrElse(a, a)))
209           val old = manager.index(previous, GeoMesaFeatureIndex.identifier(prev))
210           val index = GeoMesaFeatureIndexFactory.create(this, sft, Seq(i)).headOption.getOrElse {
211             throw new IllegalArgumentException(
212               s"Error configuring new feature index: ${GeoMesaFeatureIndex.identifier(i)}")
213           }
214           val partitions = if (!partitioned) { Seq(None) } else {
215             // have to use the old table name key but the new sft name for looking up the partitions
216             val tableNameKey = old.tableNameKey(Some(""))
217             val offset = tableNameKey.length
218             metadata.scan(sft.getTypeName, tableNameKey).map { case (k, _) => Some(k.substring(offset)) }
219           }
220           partitions.foreach { p =>
221             metadata.read(sft.getTypeName, old.tableNameKey(p)).foreach { v =>
222               metadata.insert(sft.getTypeName, index.tableNameKey(p), v)
223             }
224           }
225         }
226       }
227     }
228 
229     // configure any new indices
230     if (partitioned) {
231       logger.debug("Delaying creation of partitioned indices")
232     } else {
233       logger.debug(s"Ensuring indices ${manager.indices(sft).map(_.identifier).mkString(", ")}")
234       manager.indices(sft).toList.map(i => CachedThreadPool.submit(() => adapter.createTable(i, None, i.getSplits(None)))).foreach(_.get)
235     }
236 
237     // update stats
238     if (previous.statsEnabled) {
239       if (!sft.statsEnabled) {
240         stats.writer.clear(previous)
241       } else if (sft.getTypeName != previous.getTypeName || colMap.nonEmpty) {
242         stats.writer.rename(sft, previous)
243       }
244     }
245 
246     // rename tables to match the new sft name
247     if (sft.getTypeName != previous.getTypeName || sft.getIndices != previous.getIndices) {
248       if (FastConverter.convertOrElse[java.lang.Boolean](sft.getUserData.get(Configs.UpdateRenameTables), false)) {
249         manager.indices(sft).foreach { index =>
250           val partitions = if (partitioned) { index.getPartitions.map(Option.apply) } else { Seq(None) }
251           partitions.foreach { partition =>
252             val key = index.tableNameKey(partition)
253             metadata.read(sft.getTypeName, key).foreach { table =>
254               metadata.remove(sft.getTypeName, key)
255               val renamed = index.configureTableName(partition, adapter.tableNameLimit)
256               if (renamed != table) {
257                 logger.debug(s"Renaming table from '$table' to '$renamed'")
258                 adapter.renameTable(table, renamed)
259               }
260             }
261           }
262         }
263       }
264     }
265   }

267   // delete the index tables
268   override protected def onSchemaDeleted(sft: SimpleFeatureType): Unit = {
269     def deleteFull(index: GeoMesaFeatureIndex[_, _]): Unit =
270       adapter.deleteTables(index.deleteTableNames(None))
271     def deleteShared(index: GeoMesaFeatureIndex[_, _]): Unit = {
272       if (index.keySpace.sharing.isEmpty) { deleteFull(index) } else {
273         adapter.clearTables(index.deleteTableNames(None), Some(index.keySpace.sharing))
274       }
275     }
276 
277     val indices = manager.indices(sft).toList
278     // noinspection ScalaDeprecation
279     if (sft.isTableSharing && getTypeNames.exists(t => t != sft.getTypeName && getSchema(t).isTableSharing)) {
280       indices.map(i => CachedThreadPool.submit(() => deleteShared(i))).foreach(_.get)
281     } else {
282       indices.map(i => CachedThreadPool.submit(() => deleteFull(i))).foreach(_.get)
283     }
284     if (sft.statsEnabled) {
285       stats.writer.clear(sft)
286     }
287   }

291   /**
292    * @see org.geotools.api.data.DataStore#getSchema(java.lang.String)
293    * @param typeName feature type name
294    * @return feature type, or null if it does not exist
295    */
296   override def getSchema(typeName: String): SimpleFeatureType = {
297     var sft = super.getSchema(typeName)
298     if (sft != null) {
299       // ensure index metadata is correct
300       if (sft.getIndices.exists(i => i.attributes.isEmpty && i.name != IdIndex.name)) {
301         sft = SimpleFeatureTypes.mutable(sft)
302         // migrate index metadata to standardized versions and attributes
303         transitionIndices(sft)
304         sft = SimpleFeatureTypes.immutable(sft)
305         // validate indices
306         try { manager.indices(sft) } catch {
307           case NonFatal(e) =>
308             throw new IllegalStateException(s"The schema ${sft.getTypeName} was written with a older " +
309                 "version of GeoMesa that is no longer supported. You may continue to use an older client, or " +
310                 s"manually edit the metadata for '${InternalConfigs.IndexVersions}' to exclude the invalid indices.", e)
311         }
312       } else {
313         // validate indices
314         try { manager.indices(sft) } catch {
315           case NonFatal(e) =>
316             val versions = sft.getIndices.map(i => s"${i.name}:${i.version}").mkString(",")
317             val available = GeoMesaFeatureIndexFactory.available(sft).map(i => s"${i._1}:${i._2}").mkString(",")
318             logger.error(s"Trying to access schema ${sft.getTypeName} with invalid index versions '$versions' - " +
319                 s"available indices are '$available'", e)
320             throw new IllegalStateException(s"The schema ${sft.getTypeName} was written with a newer " +
321                 "version of GeoMesa than this client can handle. Please ensure that you are using the " +
322                 "same GeoMesa jar versions across your entire workflow. For more information, see " +
323                 "http://www.geomesa.org/documentation/user/installation_and_configuration.html#upgrading")
324         }
325       }
326 
327       // check for sft-level stats flag and set it if not present
328       if (!sft.getUserData.containsKey(SimpleFeatureTypes.Configs.StatsEnabled)) {
329         val extra = Collections.singletonMap(SimpleFeatureTypes.Configs.StatsEnabled, config.generateStats.toString)
330         sft = SimpleFeatureTypes.immutable(sft, extra)
331       }
332 
333       // get the remote version if it's available, but don't wait for it
334       GeoMesaDataStore.versions.get(VersionKey(this)).getNow(Right(None)) match {
335         case Left(e) => throw e
336         case Right(version) =>
337           version.foreach { v =>
338             val userData = Collections.singletonMap[AnyRef, AnyRef](InternalConfigs.RemoteVersion, v.toString)
339             sft = SimpleFeatureTypes.immutable(sft, userData)
340           }
341       }
342     }
343     sft
344   }

460    *
461    * @param typeName type name
462    * @param sft udpated simple feature type
463    * @return
464    */
465   def checkSchemaCompatibility(typeName: String, sft: SimpleFeatureType): SchemaCompatibility = {
466     val previous = getSchema(typeName)
467     if (previous == null) {
468       new SchemaCompatibility.DoesNotExist(this, sft)
469     } else {
470       validateSchemaUpdate(previous, sft) match {
471         case Right(TypeComparison.Compatible(extension, renamed, _)) =>
472           val copy = SimpleFeatureTypes.copy(sft)
473           updateSchemaUserData(copy, previous)
474           if (!extension && !renamed && copy.getUserData == previous.getUserData) {
475             SchemaCompatibility.Unchanged
476           } else {
477             new SchemaCompatibility.Compatible(this, typeName, sft)
478           }
479 
480         case Left(e) =>
481           SchemaCompatibility.Incompatible(e)
482       }
483     }
484   }

488   /**
489    * Updates the user data for a schema update prior to persistence. Handles converting existing
490    * user data based on new/updated attributes, copying existing user data over, etc. Feature
491    * type will be updated in place
492    *
493    * @param sft schema update
494    * @param previous existing schema
495    * @return
496    */
497   private def updateSchemaUserData(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
498 
499     // check for column renaming
500     val colMap = getColumnMap(previous, sft)
501 
502     def remapCol(name: String): String = colMap.getOrElse(name, name)
503 
504     // check for attributes flagged 'index' and convert them to sft-level user data
505     def indexed(d: AttributeDescriptor): Boolean = {
506       d.getUserData.get(AttributeOptions.OptIndex) match {
507         case i: String if Seq("true", "full").exists(_.equalsIgnoreCase(i)) => true
508         case i if i == null || Seq("false", "none").exists(_.equalsIgnoreCase(i.toString)) => false
509         case i => throw new IllegalArgumentException(s"Configured index coverage '$i' is not valid: expected 'true'")
510       }
511     }
512     sft.getAttributeDescriptors.asScala.foreach { d =>
513       if (indexed(d)) {
514         val existing = {
515           val explicit = sft.getIndices
516           if (explicit.nonEmpty) { explicit } else { previous.getIndices }
517         }
518         if (!existing.exists(e => e.name == AttributeIndex.name && remapCol(e.attributes.head) == d.getLocalName)) {
519           val fields = Seq(d.getLocalName) ++ Option(sft.getGeomField) ++ sft.getDtgField
520           val id = IndexId(AttributeIndex.name, AttributeIndex.version, fields, IndexMode.ReadWrite)
521           sft.setIndices(existing :+ id)
522         }
523       }
524     }
525 
526     // check for new indices and 'enabled indices' changes
527     val indices = {
528       val enabled = if (!sft.getUserData.containsKey(Configs.EnabledIndices)) { Seq.empty } else {
529         GeoMesaFeatureIndexFactory.indices(sft)
530       }
531       val remapped = (previous.getIndices ++ sft.getIndices).map(i => i.copy(attributes = i.attributes.map(remapCol)))
532       (remapped ++ enabled).foldLeft(Seq.empty[IndexId]) { (sum, next) =>
533         // note: ignore index version
534         if (sum.exists(i => i.name == next.name && i.attributes == next.attributes)) { sum } else { sum :+ next }
535       }
536     }
537     if (indices != previous.getIndices) {
538       sft.setIndices(indices)
539     }
540 
541     // preserve any existing user data but overwrite any keys we redefine
542     val userData = new java.util.HashMap[AnyRef, AnyRef](previous.getUserData)
543     userData.putAll(sft.getUserData)
544     sft.getUserData.putAll(userData)
545     // remove enabled indices as we don't need to persist it
546     sft.getUserData.remove(Configs.EnabledIndices)
547     // remove any null/empty keys as a way to delete existing user data
548     sft.getUserData.asScala.collect { case (k, null | "") => k }.foreach(sft.getUserData.remove)
549     // remove any 'index' flags in the attribute user data - we need them above for checking enabled indices
550     sft.getAttributeDescriptors.asScala.foreach(_.getUserData.remove(AttributeOptions.OptIndex))
551   }

553   /**
554    * Gets a map of column renames, for use during schema updates
555    *
556    * @param previous previous feature type
557    * @param sft updated feature type
558    * @return map of old name -> new name
559    */
560   private def getColumnMap(previous: SimpleFeatureType, sft: SimpleFeatureType): Map[String, String] = {
561     previous.getAttributeDescriptors.asScala.zipWithIndex.toMap.flatMap { case (prev, i) =>
562       val cur = sft.getDescriptor(i)
563       if (prev.getLocalName != cur.getLocalName) {
564         Map(prev.getLocalName -> cur.getLocalName)
565       } else {
566         Map.empty[String, String]
567       }
568     }
569   }
