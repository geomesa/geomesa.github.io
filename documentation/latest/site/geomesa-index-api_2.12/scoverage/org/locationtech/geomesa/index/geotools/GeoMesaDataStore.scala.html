116   @throws(classOf[IllegalArgumentException])
117   override protected def preSchemaCreate(sft: SimpleFeatureType): Unit = {
118     import Configs.{TableSplitterClass, TableSplitterOpts}
119     import InternalConfigs.{PartitionSplitterClass, PartitionSplitterOpts}
120 
121     // check for old enabled indices and re-map them
122     // noinspection ScalaDeprecation
123     SimpleFeatureTypes.Configs.ENABLED_INDEX_OPTS.drop(1).find(sft.getUserData.containsKey).foreach { key =>
124       sft.getUserData.put(SimpleFeatureTypes.Configs.EnabledIndices, sft.getUserData.remove(key))
125     }
126 
127     // validate column groups
128     adapter.groups.validate(sft)
129 
130     // disable table sharing, no longer supported
131     // noinspection ScalaDeprecation
132     if (sft.isTableSharing) {
133       logger.warn("Table sharing is no longer supported - disabling table sharing")
134       sft.getUserData.remove(Configs.TableSharing)
135       sft.getUserData.remove(InternalConfigs.TableSharingPrefix)
136     }
137 
138     // configure the indices to use
139     if (sft.getIndices.isEmpty) {
140       val indices = GeoMesaFeatureIndexFactory.indices(sft)
141       if (indices.isEmpty) {
142         throw new IllegalArgumentException("There are no available indices that support the schema " +
143             SimpleFeatureTypes.encodeType(sft))
144       }
145       sft.setIndices(indices)
146     }
147 
148     // try to create the indices up front to ensure they are valid for the sft
149     GeoMesaFeatureIndexFactory.create(this, sft, sft.getIndices)
150 
151     // remove the enabled indices after configuration so we don't persist them
152     sft.getUserData.remove(SimpleFeatureTypes.Configs.EnabledIndices)
153     // remove any 'index' flags in the attribute metadata, they have already been captured in the indices above
154     sft.getAttributeDescriptors.asScala.foreach(_.getUserData.remove(AttributeOptions.OptIndex))
155 
156     // for partitioned schemas, persist the table partitioning keys
157     if (TablePartition.partitioned(sft)) {
158       Seq((TableSplitterClass, PartitionSplitterClass), (TableSplitterOpts, PartitionSplitterOpts)).foreach {
159         case (from, to) => Option(sft.getUserData.get(from)).foreach(sft.getUserData.put(to, _))
160       }
161     }
162 
163     // set stats enabled based on the data store config if not explicitly set
164     if (!sft.getUserData.containsKey(SimpleFeatureTypes.Configs.StatsEnabled)) {
165       sft.setStatsEnabled(config.generateStats)
166     }
167 
168     sft.getFeatureExpiration // validate any configured age-off
169   }

193   // create the new index tables (if not using partitioned tables)
194   override protected def onSchemaUpdated(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
195     val partitioned = TablePartition.partitioned(sft)
196 
197     // check for column renaming
198     val colMap = getColumnMap(previous, sft).map { case (k, v) => (v, k) }
199 
200     // rewrite the table name keys for the renamed cols
201     if (colMap.nonEmpty) {
202       sft.getIndices.foreach { i =>
203         if (i.attributes.exists(colMap.contains)) {
204           val prev = i.copy(attributes = i.attributes.map(a => colMap.getOrElse(a, a)))
205           val old = manager.index(previous, GeoMesaFeatureIndex.identifier(prev))
206           val index = GeoMesaFeatureIndexFactory.create(this, sft, Seq(i)).headOption.getOrElse {
207             throw new IllegalArgumentException(
208               s"Error configuring new feature index: ${GeoMesaFeatureIndex.identifier(i)}")
209           }
210           val partitions = if (!partitioned) { Seq(None) } else {
211             // have to use the old table name key but the new sft name for looking up the partitions
212             val tableNameKey = old.tableNameKey(Some(""))
213             val offset = tableNameKey.length
214             metadata.scan(sft.getTypeName, tableNameKey).map { case (k, _) => Some(k.substring(offset)) }
215           }
216           partitions.foreach { p =>
217             metadata.read(sft.getTypeName, old.tableNameKey(p)).foreach { v =>
218               metadata.insert(sft.getTypeName, index.tableNameKey(p), v)
219             }
220           }
221         }
222       }
223     }
224 
225     // configure any new indices
226     if (partitioned) {
227       logger.debug("Delaying creation of partitioned indices")
228     } else {
229       logger.debug(s"Ensuring indices ${manager.indices(sft).map(_.identifier).mkString(", ")}")
230       manager.indices(sft).toList.map(i => CachedThreadPool.submit(() => adapter.createTable(i, None, i.getSplits(None)))).foreach(_.get)
231     }
232 
233     // update stats
234     if (previous.statsEnabled) {
235       if (!sft.statsEnabled) {
236         stats.writer.clear(previous)
237       } else if (sft.getTypeName != previous.getTypeName || colMap.nonEmpty) {
238         stats.writer.rename(sft, previous)
239       }
240     }
241 
242     // rename tables to match the new sft name
243     if (sft.getTypeName != previous.getTypeName || sft.getIndices != previous.getIndices) {
244       if (FastConverter.convertOrElse[java.lang.Boolean](sft.getUserData.get(Configs.UpdateRenameTables), false)) {
245         manager.indices(sft).foreach { index =>
246           val partitions = if (partitioned) { index.getPartitions.map(Option.apply) } else { Seq(None) }
247           partitions.foreach { partition =>
248             val key = index.tableNameKey(partition)
249             metadata.read(sft.getTypeName, key).foreach { table =>
250               metadata.remove(sft.getTypeName, key)
251               val renamed = index.configureTableName(partition, adapter.tableNameLimit)
252               if (renamed != table) {
253                 logger.debug(s"Renaming table from '$table' to '$renamed'")
254                 adapter.renameTable(table, renamed)
255               }
256             }
257           }
258         }
259       }
260     }
261   }

263   // delete the index tables
264   override protected def onSchemaDeleted(sft: SimpleFeatureType): Unit = {
265     def deleteFull(index: GeoMesaFeatureIndex[_, _]): Unit =
266       adapter.deleteTables(index.deleteTableNames(None))
267     def deleteShared(index: GeoMesaFeatureIndex[_, _]): Unit = {
268       if (index.keySpace.sharing.isEmpty) { deleteFull(index) } else {
269         adapter.clearTables(index.deleteTableNames(None), Some(index.keySpace.sharing))
270       }
271     }
272 
273     val indices = manager.indices(sft).toList
274     // noinspection ScalaDeprecation
275     if (sft.isTableSharing && getTypeNames.exists(t => t != sft.getTypeName && getSchema(t).isTableSharing)) {
276       indices.map(i => CachedThreadPool.submit(() => deleteShared(i))).foreach(_.get)
277     } else {
278       indices.map(i => CachedThreadPool.submit(() => deleteFull(i))).foreach(_.get)
279     }
280     if (sft.statsEnabled) {
281       stats.writer.clear(sft)
282     }
283   }

287   /**
288    * @see org.geotools.api.data.DataStore#getSchema(java.lang.String)
289    * @param typeName feature type name
290    * @return feature type, or null if it does not exist
291    */
292   override def getSchema(typeName: String): SimpleFeatureType = {
293     var sft = super.getSchema(typeName)
294     if (sft != null) {
295       // ensure index metadata is correct
296       if (sft.getIndices.exists(i => i.attributes.isEmpty && i.name != IdIndex.name)) {
297         sft = SimpleFeatureTypes.mutable(sft)
298         // migrate index metadata to standardized versions and attributes
299         transitionIndices(sft)
300         sft = SimpleFeatureTypes.immutable(sft)
301         // validate indices
302         try { manager.indices(sft) } catch {
303           case NonFatal(e) =>
304             throw new IllegalStateException(s"The schema ${sft.getTypeName} was written with a older " +
305                 "version of GeoMesa that is no longer supported. You may continue to use an older client, or " +
306                 s"manually edit the metadata for '${InternalConfigs.IndexVersions}' to exclude the invalid indices.", e)
307         }
308       } else {
309         // validate indices
310         try { manager.indices(sft) } catch {
311           case NonFatal(e) =>
312             val versions = sft.getIndices.map(i => s"${i.name}:${i.version}").mkString(",")
313             val available = GeoMesaFeatureIndexFactory.available(sft).map(i => s"${i._1}:${i._2}").mkString(",")
314             logger.error(s"Trying to access schema ${sft.getTypeName} with invalid index versions '$versions' - " +
315                 s"available indices are '$available'", e)
316             throw new IllegalStateException(s"The schema ${sft.getTypeName} was written with a newer " +
317                 "version of GeoMesa than this client can handle. Please ensure that you are using the " +
318                 "same GeoMesa jar versions across your entire workflow. For more information, see " +
319                 "http://www.geomesa.org/documentation/user/installation_and_configuration.html#upgrading")
320         }
321       }
322 
323       // check for sft-level stats flag and set it if not present
324       if (!sft.getUserData.containsKey(SimpleFeatureTypes.Configs.StatsEnabled)) {
325         val extra = Collections.singletonMap(SimpleFeatureTypes.Configs.StatsEnabled, config.generateStats.toString)
326         sft = SimpleFeatureTypes.immutable(sft, extra)
327       }
328 
329       // get the remote version if it's available, but don't wait for it
330       GeoMesaDataStore.versions.get(VersionKey(this)).getNow(Right(None)) match {
331         case Left(e) => throw e
332         case Right(version) =>
333           version.foreach { v =>
334             val userData = Collections.singletonMap[AnyRef, AnyRef](InternalConfigs.RemoteVersion, v.toString)
335             sft = SimpleFeatureTypes.immutable(sft, userData)
336           }
337       }
338     }
339     sft
340   }

459    * @return
460    */
461   def checkSchemaCompatibility(typeName: String, sft: SimpleFeatureType): SchemaCompatibility = {
462     val previous = getSchema(typeName)
463     if (previous == null) {
464       new SchemaCompatibility.DoesNotExist(this, sft)
465     } else {
466       validateSchemaUpdate(previous, sft) match {
467         case Right(TypeComparison.Compatible(extension, renamed, _)) =>
468           val copy = SimpleFeatureTypes.copy(sft)
469           updateSchemaUserData(copy, previous)
470           if (!extension && !renamed && copy.getUserData == previous.getUserData) {
471             SchemaCompatibility.Unchanged
472           } else {
473             new SchemaCompatibility.Compatible(this, typeName, sft)
474           }
475 
476         case Left(e) =>
477           SchemaCompatibility.Incompatible(e)
478       }
479     }
480   }

484   /**
485    * Updates the user data for a schema update prior to persistence. Handles converting existing
486    * user data based on new/updated attributes, copying existing user data over, etc. Feature
487    * type will be updated in place
488    *
489    * @param sft schema update
490    * @param previous existing schema
491    * @return
492    */
493   private def updateSchemaUserData(sft: SimpleFeatureType, previous: SimpleFeatureType): Unit = {
494 
495     // check for column renaming
496     val colMap = getColumnMap(previous, sft)
497 
498     def remapCol(name: String): String = colMap.getOrElse(name, name)
499 
500     // check for attributes flagged 'index' and convert them to sft-level user data
501     def indexed(d: AttributeDescriptor): Boolean = {
502       d.getUserData.get(AttributeOptions.OptIndex) match {
503         case i: String if Seq("true", "full").exists(_.equalsIgnoreCase(i)) => true
504         case i if i == null || Seq("false", "none").exists(_.equalsIgnoreCase(i.toString)) => false
505         case i => throw new IllegalArgumentException(s"Configured index coverage '$i' is not valid: expected 'true'")
506       }
507     }
508     sft.getAttributeDescriptors.asScala.foreach { d =>
509       if (indexed(d)) {
510         val existing = {
511           val explicit = sft.getIndices
512           if (explicit.nonEmpty) { explicit } else { previous.getIndices }
513         }
514         if (!existing.exists(e => e.name == AttributeIndex.name && remapCol(e.attributes.head) == d.getLocalName)) {
515           val fields = Seq(d.getLocalName) ++ Option(sft.getGeomField) ++ sft.getDtgField
516           val id = IndexId(AttributeIndex.name, AttributeIndex.version, fields, IndexMode.ReadWrite)
517           sft.setIndices(existing :+ id)
518         }
519       }
520     }
521 
522     // check for new indices and 'enabled indices' changes
523     val indices = {
524       val enabled = if (!sft.getUserData.containsKey(Configs.EnabledIndices)) { Seq.empty } else {
525         GeoMesaFeatureIndexFactory.indices(sft)
526       }
527       val remapped = (previous.getIndices ++ sft.getIndices).map(i => i.copy(attributes = i.attributes.map(remapCol)))
528       (remapped ++ enabled).foldLeft(Seq.empty[IndexId]) { (sum, next) =>
529         // note: ignore index version
530         if (sum.exists(i => i.name == next.name && i.attributes == next.attributes)) { sum } else { sum :+ next }
531       }
532     }
533     if (indices != previous.getIndices) {
534       sft.setIndices(indices)
535     }
536 
537     // preserve any existing user data but overwrite any keys we redefine
538     val userData = new java.util.HashMap[AnyRef, AnyRef](previous.getUserData)
539     userData.putAll(sft.getUserData)
540     sft.getUserData.putAll(userData)
541     // remove enabled indices as we don't need to persist it
542     sft.getUserData.remove(Configs.EnabledIndices)
543     // remove any null/empty keys as a way to delete existing user data
544     sft.getUserData.asScala.collect { case (k, null | "") => k }.foreach(sft.getUserData.remove)
545     // remove any 'index' flags in the attribute user data - we need them above for checking enabled indices
546     sft.getAttributeDescriptors.asScala.foreach(_.getUserData.remove(AttributeOptions.OptIndex))
547   }

549   /**
550    * Gets a map of column renames, for use during schema updates
551    *
552    * @param previous previous feature type
553    * @param sft updated feature type
554    * @return map of old name -> new name
555    */
556   private def getColumnMap(previous: SimpleFeatureType, sft: SimpleFeatureType): Map[String, String] = {
557     previous.getAttributeDescriptors.asScala.zipWithIndex.toMap.flatMap { case (prev, i) =>
558       val cur = sft.getDescriptor(i)
559       if (prev.getLocalName != cur.getLocalName) {
560         Map(prev.getLocalName -> cur.getLocalName)
561       } else {
562         Map.empty[String, String]
563       }
564     }
565   }
