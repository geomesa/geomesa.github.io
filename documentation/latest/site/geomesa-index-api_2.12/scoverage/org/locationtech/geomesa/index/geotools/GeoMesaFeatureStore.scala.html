32   override def addFeatures(collection: FeatureCollection[SimpleFeatureType, SimpleFeature]): java.util.List[FeatureId] = {
33     if (collection.isEmpty) {
34       return Collections.emptyList[FeatureId]()
35     }
36 
37     val fids = new java.util.ArrayList[FeatureId](collection.size())
38     val errors = ArrayBuffer.empty[Throwable]
39 
40     WithClose(collection.features, writer(None)) { case (features, writer) =>
41       val helper = FeatureWriterHelper(writer)
42       while (features.hasNext) {
43         try { fids.add(helper.write(features.next()).getIdentifier) } catch {
44           // validation errors in indexing will throw an IllegalArgumentException
45           // make the caller handle other errors, which are likely related to the underlying database,
46           // as we wouldn't know which features were actually written or not due to write buffering
47           case e: IllegalArgumentException => errors.append(e)
48         }
49       }
50     }
51 
52     if (errors.isEmpty) { fids } else {
53       val e = new IllegalArgumentException("Some features were not written:")
54       // suppressed exceptions should contain feature ids and attributes
55       errors.foreach(e.addSuppressed)
56       throw e
57     }
58   }
59 
60   override def setFeatures(reader: FeatureReader[SimpleFeatureType, SimpleFeature]): Unit = {
61     removeFeatures(Filter.INCLUDE)
62     try {
63       WithClose(writer(None)) { writer =>
64         val helper = FeatureWriterHelper(writer)
65         while (reader.hasNext) {
66           helper.write(reader.next())
67         }
68       }
69     } finally {
70       reader.close()
71     }
72   }

83   override def modifyFeatures(attributes: Array[Name], values: Array[AnyRef], filter: Filter): Unit = {
84     val sft = getSchema
85 
86     require(filter != null, "Filter must not be null")
87     attributes.foreach(a => require(sft.getDescriptor(a) != null, s"$a is not an attribute of ${sft.getName}"))
88     require(attributes.length == values.length, "Modified names and values don't match")
89 
90     WithClose(writer(Some(filter))) { writer =>
91       while (writer.hasNext) {
92         val sf = writer.next()
93         var i = 0
94         while (i < attributes.length) {
95           try {
96             sf.setAttribute(attributes(i), values(i))
97           } catch {
98             case e: Exception =>
99               throw new DataSourceException("Error updating feature " +
100                   s"'${sf.getID}' with ${attributes(i)}=${values(i)}", e)
101           }
102           i += 1
103         }
104         writer.write()
105       }
106     }
107   }

109   override def removeFeatures(filter: Filter): Unit = {
110     ds match {
111       case gm: GeoMesaDataStore[_] if filter == Filter.INCLUDE =>
112         val indices = gm.manager.indices(sft).toList
113         if (TablePartition.partitioned(sft)) {
114           def deleteOne(index: GeoMesaFeatureIndex[_, _]): Unit =
115             gm.adapter.deleteTables(index.deleteTableNames(None))
116           indices.map(i => CachedThreadPool.submit(() => deleteOne(i))).foreach(_.get)
117         } else {
118           def deleteOne(index: GeoMesaFeatureIndex[_, _]): Unit = {
119             val prefix = Some(index.keySpace.sharing).filterNot(_.isEmpty)
120             gm.adapter.clearTables(index.getTableNames(), prefix)
121           }
122           indices.map(i => CachedThreadPool.submit(() => deleteOne(i))).foreach(_.get)
123         }
124         gm.stats.writer.clear(sft)
125 
126       case _ =>
127         WithClose(writer(Some(filter))) { writer =>
128           while (writer.hasNext) {
129             writer.next()
130             writer.remove()
131           }
132         }
133     }
134   }
135

149   private def writer(filter: Option[Filter]): FeatureWriter[SimpleFeatureType, SimpleFeature] = {
150     ds match {
151       case gm: MetadataBackedDataStore => gm.getFeatureWriter(sft, transaction, filter)
152       case _ if filter.isEmpty => ds.getFeatureWriterAppend(sft.getTypeName, transaction)
153       case _ => ds.getFeatureWriter(sft.getTypeName, filter.get, transaction)
154     }
155   }
