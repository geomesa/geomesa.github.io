29   @throws(classOf[ParsingException])
30   def parse(sft: SimpleFeatureType, stat: String, report: Boolean = true): Stat = {
31     if (stat == null) {
32       throw new IllegalArgumentException("Stat must not be null")
33     }
34     val runner = if (report) { ReportingParseRunner(Parser.stat) } else { BasicParseRunner(Parser.stat) }
35     sfts.set(sft)
36     try {
37       val parsing = runner.run(stat)
38       parsing.result.getOrElse {
39         throw new ParsingException(s"Invalid stat string: ${ErrorUtils.printParseErrors(parsing)}")
40       }
41     } finally {
42       sfts.remove()
43     }
44   }
