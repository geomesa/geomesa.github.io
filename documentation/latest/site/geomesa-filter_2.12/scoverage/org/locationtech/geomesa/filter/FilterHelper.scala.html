52   private def isOperationGeomWholeWorld[Op <: BinarySpatialOperator]
53       (op: Op, order: SpatialOpOrder.SpatialOpOrder = SpatialOpOrder.AnyOrder): Boolean = {
54     val prop = checkOrder(op.getExpression1, op.getExpression2)
55     // validate that property and literal are in the specified order
56     prop.exists { p =>
57       val ordered = order match {
58         case SpatialOpOrder.AnyOrder      => true
59         case SpatialOpOrder.PropertyFirst => !p.flipped
60         case SpatialOpOrder.LiteralFirst  => p.flipped
61       }
62       ordered && Option(FastConverter.evaluate(p.literal, classOf[Geometry])).exists(isWholeWorld)
63     }
64   }

104     * Extract geometries from a filter without validating boundaries.

107     * @param attribute attribute to consider
108     * @param intersect intersect AND'd geometries or return them all
109     * @return geometry bounds from spatial filters

111   private def extractUnclippedGeometries(filter: Filter, attribute: String, intersect: Boolean): FilterValues[Geometry] = {
112     filter match {
113       case o: Or  =>
114         val all = o.getChildren.asScala.map(extractUnclippedGeometries(_, attribute, intersect))
115         val join = FilterValues.or[Geometry]((l, r) => l ++ r) _
116         all.reduceLeftOption[FilterValues[Geometry]](join).getOrElse(FilterValues.empty)
117 
118       case a: And =>
119         val all = a.getChildren.asScala.map(extractUnclippedGeometries(_, attribute, intersect)).filter(_.nonEmpty)
120         if (intersect) {
121           val intersect = FilterValues.and[Geometry]((l, r) => Option(l.intersection(r)).filterNot(_.isEmpty)) _
122           all.reduceLeftOption[FilterValues[Geometry]](intersect).getOrElse(FilterValues.empty)
123         } else {
124           FilterValues(all.toSeq.flatMap(_.values))
125         }
126 
127       // Note: although not technically required, all known spatial predicates are also binary spatial operators
128       case f: BinarySpatialOperator if isSpatialFilter(f) =>
129         FilterValues(GeometryProcessing.extract(f, attribute))
130 
131       case _ =>
132         FilterValues.empty
133     }

136   /**
137     * Extracts intervals from a filter. Intervals will be merged where possible - the resulting sequence
138     * is considered to be a union (i.e. OR)
139     *
140     * @param filter filter to evaluate
141     * @param attribute attribute to consider
142     * @param intersect intersect extracted values together, or return them all
143     *                  note if not intersected, 'and/or' distinction will be lost
144     * @return a sequence of intervals, if any. disjoint intervals will result in Seq((null, null))
145     */
146   def extractIntervals(filter: Filter,
147                        attribute: String,
148                        intersect: Boolean = true,
149                        handleExclusiveBounds: Boolean = false): FilterValues[Bounds[ZonedDateTime]] = {
150     extractAttributeBounds(filter, attribute, classOf[Date]).map { bounds =>
151       var lower, upper: Bound[ZonedDateTime] = null
152       // this if check determines if rounding will be used and if we need to account for narrow ranges
153       if (!handleExclusiveBounds || bounds.lower.value.isEmpty || bounds.upper.value.isEmpty ||
154           (bounds.lower.inclusive && bounds.upper.inclusive)) {
155         lower = createDateTime(bounds.lower, roundSecondsUp, handleExclusiveBounds)
156         upper = createDateTime(bounds.upper, roundSecondsDown, handleExclusiveBounds)
157       } else {
158         // check for extremely narrow filters where our rounding makes the result out-of-order
159         // note: both upper and lower are known to be defined based on hitting this else branch
160         val margin = if (bounds.lower.inclusive || bounds.upper.inclusive) { 1000 } else { 2000 }
161         val round = bounds.upper.value.get.getTime - bounds.lower.value.get.getTime > margin
162         lower = createDateTime(bounds.lower, roundSecondsUp, round)
163         upper = createDateTime(bounds.upper, roundSecondsDown, round)
164       }
165       Bounds(lower, upper)
166     }
167   }

169   private def createDateTime(bound: Bound[Date],
170                              round: ZonedDateTime => ZonedDateTime,
171                              roundExclusive: Boolean): Bound[ZonedDateTime] = {
172     if (bound.value.isEmpty) { Bound.unbounded } else {
173       val dt = bound.value.map(d => ZonedDateTime.ofInstant(toInstant(d), ZoneOffset.UTC))
174       if (roundExclusive && !bound.inclusive) {
175         Bound(dt.map(round), inclusive = true)
176       } else {
177         Bound(dt, bound.inclusive)
178       }
179     }
180   }
181 
182   private def roundSecondsUp(dt: ZonedDateTime): ZonedDateTime = dt.plusSeconds(1).withNano(0)
183 
184   private def roundSecondsDown(dt: ZonedDateTime): ZonedDateTime = {
185     val nanos = dt.getNano
186     if (nanos == 0) { dt.minusSeconds(1) } else { dt.withNano(0) }
187   }

189   /**
190     * Extracts bounds from filters that pertain to a given attribute. Bounds will be merged where
191     * possible.
192     *
193     * @param filter filter to evaluate
194     * @param attribute attribute name to consider
195     * @param binding attribute type
196     * @return a sequence of bounds, if any
197     */
198   def extractAttributeBounds[T](filter: Filter, attribute: String, binding: Class[T]): FilterValues[Bounds[T]] = {
199     filter match {
200       case o: Or =>
201         val union = FilterValues.or[Bounds[T]](Bounds.union[T]) _
202         o.getChildren.asScala.map(f =>
203           extractAttributeBounds(f, attribute, binding)
204         ).reduceLeft[FilterValues[Bounds[T]]]((acc, child) => {
205           if (acc.isEmpty || child.isEmpty) {
206             FilterValues.empty
207           } else {
208             union(acc, child)
209           }
210         })
211 
212       case a: And =>
213         val all = a.getChildren.asScala.flatMap { f =>
214           val child = extractAttributeBounds(f, attribute, binding)
215           if (child.isEmpty) { Seq.empty } else { Seq(child) }
216         }
217         val intersection = FilterValues.and[Bounds[T]](Bounds.intersection[T]) _
218         all.reduceLeftOption[FilterValues[Bounds[T]]](intersection).getOrElse(FilterValues.empty)
219 
220       case f: PropertyIsEqualTo =>
221         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
222           case e: PropertyLiteral =>
223             Option(FastConverter.evaluate(e.literal, binding)).map { lit =>
224               val bound = Bound(Some(lit), inclusive = true)
225               FilterValues(Seq(Bounds(bound, bound)))
226             }
227 
228           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = true, binding)
229         }.getOrElse(FilterValues.empty)
230 
231       case f: PropertyIsBetween =>
232         try {
233           val prop = f.getExpression.asInstanceOf[PropertyName].getPropertyName
234           if (prop != attribute) { FilterValues.empty } else {
235             // note that between is inclusive
236             val lower = Bound(Option(FastConverter.evaluate(f.getLowerBoundary, binding)), inclusive = true)
237             val upper = Bound(Option(FastConverter.evaluate(f.getUpperBoundary, binding)), inclusive = true)
238             FilterValues(Seq(Bounds(lower, upper)))
239           }
240         } catch {
241           case e: Exception =>
242             FilterHelperLogger.log.warn(s"Unable to extract bounds from filter '${filterToString(f)}'", e)
243             FilterValues.empty
244         }
245 
246       case f: During if classOf[Date].isAssignableFrom(binding) =>
247         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
248           case e: PropertyLiteral =>
249             Option(FastConverter.evaluate(e.literal, classOf[Period])).map { p =>
250               // note that during is exclusive
251               val lower = Bound(Option(p.getBeginning.getPosition.getDate.asInstanceOf[T]), inclusive = false)
252               val upper = Bound(Option(p.getEnding.getPosition.getDate.asInstanceOf[T]), inclusive = false)
253               FilterValues(Seq(Bounds(lower, upper)))
254             }
255 
256           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = false, binding)
257         }.getOrElse(FilterValues.empty)
258 
259       case f: PropertyIsGreaterThan =>
260         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
261           case e: PropertyLiteral =>
262             Option(FastConverter.evaluate(e.literal, binding)).map { lit =>
263               val bound = Bound(Some(lit), inclusive = false)
264               val (lower, upper) = if (e.flipped) { (Bound.unbounded[T], bound) } else { (bound, Bound.unbounded[T]) }
265               FilterValues(Seq(Bounds(lower, upper)))
266             }
267 
268           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = false, binding)
269         }.getOrElse(FilterValues.empty)
270 
271       case f: PropertyIsGreaterThanOrEqualTo =>
272         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
273           case e: PropertyLiteral =>
274             Option(FastConverter.evaluate(e.literal, binding)).map { lit =>
275               val bound = Bound(Some(lit), inclusive = true)
276               val (lower, upper) = if (e.flipped) { (Bound.unbounded[T], bound) } else { (bound, Bound.unbounded[T]) }
277               FilterValues(Seq(Bounds(lower, upper)))
278             }
279 
280           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = true, binding)
281         }.getOrElse(FilterValues.empty)
282 
283       case f: PropertyIsLessThan =>
284         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
285           case e: PropertyLiteral =>
286             Option(FastConverter.evaluate(e.literal, binding)).map { lit =>
287               val bound = Bound(Some(lit), inclusive = false)
288               val (lower, upper) = if (e.flipped) { (bound, Bound.unbounded[T]) } else { (Bound.unbounded[T], bound) }
289               FilterValues(Seq(Bounds(lower, upper)))
290             }
291 
292           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = false, binding)
293         }.getOrElse(FilterValues.empty)
294 
295       case f: PropertyIsLessThanOrEqualTo =>
296         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
297           case e: PropertyLiteral =>
298             Option(FastConverter.evaluate(e.literal, binding)).map { lit =>
299               val bound = Bound(Some(lit), inclusive = true)
300               val (lower, upper) = if (e.flipped) { (bound, Bound.unbounded[T]) } else { (Bound.unbounded[T], bound) }
301               FilterValues(Seq(Bounds(lower, upper)))
302             }
303 
304           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = true, binding)
305         }.getOrElse(FilterValues.empty)
306 
307       case f: Before =>
308         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
309           case e: PropertyLiteral =>
310             Option(FastConverter.evaluate(e.literal, binding)).map { lit =>
311               // note that before is exclusive
312               val bound = Bound(Some(lit), inclusive = false)
313               val (lower, upper) = if (e.flipped) { (bound, Bound.unbounded[T]) } else { (Bound.unbounded[T], bound) }
314               FilterValues(Seq(Bounds(lower, upper)))
315             }
316 
317           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = false, binding)
318         }.getOrElse(FilterValues.empty)
319 
320       case f: After =>
321         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap {
322           case e: PropertyLiteral =>
323             Option(FastConverter.evaluate(e.literal, binding)).map { lit =>
324               // note that after is exclusive
325               val bound = Bound(Some(lit), inclusive = false)
326               val (lower, upper) = if (e.flipped) { (Bound.unbounded[T], bound) } else { (bound, Bound.unbounded[T]) }
327               FilterValues(Seq(Bounds(lower, upper)))
328             }
329 
330           case e: FunctionLiteral => extractFunctionBounds(e, inclusive = false, binding)
331         }.getOrElse(FilterValues.empty)
332 
333       case f: PropertyIsLike if binding == classOf[String] =>
334         try {
335           val prop = f.getExpression.asInstanceOf[PropertyName].getPropertyName
336           if (prop != attribute) { FilterValues.empty } else {
337             // find the first wildcard and create a range prefix
338             val literal = f.getLiteral
339             var i = literal.indexWhere(Wildcards.contains)
340             // check for escaped wildcards
341             while (i > 1 && literal.charAt(i - 1) == '\\' && literal.charAt(i - 2) == '\\') {
342               i = literal.indexWhere(Wildcards.contains, i + 1)
343             }
344             if (i == -1) {
345               val literals = if (f.isMatchingCase) { Seq(literal) } else { casePermutations(literal) }
346               val bounds = literals.map { lit =>
347                 val bound = Bound(Some(lit), inclusive = true)
348                 Bounds(bound, bound)
349               }
350               FilterValues(bounds.asInstanceOf[Seq[Bounds[T]]], precise = true)
351             } else {
352               val prefix = literal.substring(0, i)
353               val prefixes = if (f.isMatchingCase) { Seq(prefix) } else { casePermutations(prefix) }
354               val bounds = prefixes.map { p =>
355                 Bounds(Bound(Some(p), inclusive = true), Bound(Some(p + WildcardSuffix), inclusive = true))
356               }
357               // our ranges fully capture the filter if there's a single trailing multi-char wildcard
358               val exact = i == literal.length - 1 && literal.charAt(i) == WildcardMultiChar
359               FilterValues(bounds.asInstanceOf[Seq[Bounds[T]]], precise = exact)
360             }
361           }
362         } catch {
363           case e: Exception =>
364             FilterHelperLogger.log.warn(s"Unable to extract bounds from filter '${filterToString(f)}'", e)
365             FilterValues.empty
366         }
367 
368       case f: Not if f.getFilter.isInstanceOf[PropertyIsNull] =>
369         try {
370           val isNull = f.getFilter.asInstanceOf[PropertyIsNull]
371           val prop = isNull.getExpression.asInstanceOf[PropertyName].getPropertyName
372           if (prop != attribute) { FilterValues.empty } else {
373             FilterValues(Seq(Bounds.everything[T]))
374           }
375         } catch {
376           case e: Exception =>
377             FilterHelperLogger.log.warn(s"Unable to extract bounds from filter '${filterToString(f)}'", e)
378             FilterValues.empty
379         }
380 
381       case f: Not =>
382         // we extract the sub-filter bounds, then invert them
383         val inverted = extractAttributeBounds(f.getFilter, attribute, binding)
384         if (inverted.isEmpty) {
385           inverted
386         } else if (inverted.disjoint) {
387           FilterValues(Seq(Bounds.everything[T])) // equivalent to not null
388         } else if (!inverted.precise) {
389           FilterHelperLogger.log.warn(s"Falling back to full table scan for inverted query: '${filterToString(f)}'")
390           FilterValues(Seq(Bounds.everything[T]), precise = false)
391         } else {
392           // NOT(A OR B) turns into NOT(A) AND NOT(B)
393           val uninverted = inverted.values.map { bounds =>
394             // NOT the single bound
395             val not = bounds.bounds match {
396               case (None, None) => Seq.empty
397               case (Some(lo), None) => Seq(Bounds(Bound.unbounded, Bound(Some(lo), !bounds.lower.inclusive)))
398               case (None, Some(hi)) => Seq(Bounds(Bound(Some(hi), !bounds.upper.inclusive), Bound.unbounded))
399               case (Some(lo), Some(hi)) => Seq(
400                   Bounds(Bound.unbounded, Bound(Some(lo), !bounds.lower.inclusive)),
401                   Bounds(Bound(Some(hi), !bounds.upper.inclusive), Bound.unbounded)
402                 )
403             }
404             FilterValues(not)
405           }
406           // AND together
407           val intersect = FilterValues.and[Bounds[T]](Bounds.intersection[T]) _
408           uninverted.reduceLeft[FilterValues[Bounds[T]]](intersect)
409         }
410 
411       case f: TEquals =>
412         checkOrder(f.getExpression1, f.getExpression2).filter(_.name == attribute).flatMap { prop =>
413           Option(FastConverter.evaluate(prop.literal, binding)).map { lit =>
414             val bound = Bound(Some(lit), inclusive = true)
415             FilterValues(Seq(Bounds(bound, bound)))
416           }
417         }.getOrElse(FilterValues.empty)
418 
419       case _ => FilterValues.empty
420     }
421   }

430   /**
431    * Calculates all the different case permutations of a string.
432    *
433    * For example, "foo" -> Seq("foo", "Foo", "fOo", "foO", "fOO", "FoO", "FOo", "FOO")
434    *
435    * @param string input string
436    * @return
437    */
438   private def casePermutations(string: String): Seq[String] = {
439     val max = FilterProperties.CaseInsensitiveLimit.toInt.getOrElse {
440       // has a valid default value so should never return a none
441       throw new IllegalStateException(
442         s"Error getting default value for ${FilterProperties.CaseInsensitiveLimit.property}")
443     }
444 
445     val lower = string.toLowerCase(Locale.US)
446     val upper = string.toUpperCase(Locale.US)
447     // account for chars without upper/lower cases, which we don't need to permute
448     val count = (0 until lower.length).count(i => lower(i) != upper(i))
449 
450     if (count > max) {
451       FilterHelperLogger.log.warn(s"Not expanding case-insensitive prefix due to length: $string")
452       Seq.empty
453     } else {
454       // there will be 2^n different permutations, accounting for chars that don't have an upper/lower case
455       val permutations = Array.fill(math.pow(2, count).toInt)(Array(lower: _*))
456       var i = 0 // track the index of the current char
457       var c = 0 // track the index of the bit check, which skips chars that don't have an upper/lower case
458       while (i < string.length) {
459         val upperChar = upper.charAt(i)
460         if (lower.charAt(i) != upperChar) {
461           var j = 0
462           while (j < permutations.length) {
463             // set upper/lower based on the bit
464             if (((j >> c) & 1) != 0) {
465               permutations(j)(i) = upperChar
466             }
467             j += 1
468           }
469           c += 1
470         }
471         i += 1
472       }
473 
474       permutations.map(new String(_))
475     }
476   }

508   /**
509     * Simplifies filters to make them easier to process.
510     *
511     * Current simplifications:
512     *
513     *   1) Extracts out common parts in an OR clause to simplify further processing.
514     *
515     *      Example: OR(AND(1, 2), AND(1, 3), AND(1, 4)) -> AND(1, OR(2, 3, 4))
516     *
517     *   2) N/A - add more simplifications here as needed
518     *
519     * @param filter filter
520     * @return
521     */
522   def simplify(filter: Filter): Filter = {
523     def deduplicateOrs(f: Filter): Filter = f match {
524       case and: And => ff.and(and.getChildren.asScala.map(deduplicateOrs).asJava)
525 
526       case or: Or =>
527         // OR(AND(1,2,3), AND(1,2,4)) -> Seq(Seq(1,2,3), Seq(1,2,4))
528         val decomposed = or.getChildren.asScala.map(decomposeAnd)
529         val clauses = decomposed.head // Seq(1,2,3)
530         val duplicates = clauses.filter(c => decomposed.tail.forall(_.contains(c))) // Seq(1,2)
531         if (duplicates.isEmpty) { or } else {
532           val simplified = decomposed.flatMap(d => andOption(d.filterNot(duplicates.contains)))
533           if (simplified.length < decomposed.length) {
534             // the duplicated filters are an entire clause, so we can ignore the rest of the clauses
535             andFilters(duplicates)
536           } else {
537             andFilters(orOption(simplified.toSeq).toSeq ++ duplicates)
538           }
539         }
540 
541       case _ => f
542     }
543     // TODO GEOMESA-1533 simplify ANDs of ORs for DNF
544     flatten(deduplicateOrs(flatten(filter)))
545   }

563   private [filter] def flattenAnd(filters: Seq[Filter]): ListBuffer[Filter] = {
564     val remaining = ListBuffer.empty[Filter] ++ filters
565     val result = ListBuffer.empty[Filter]
