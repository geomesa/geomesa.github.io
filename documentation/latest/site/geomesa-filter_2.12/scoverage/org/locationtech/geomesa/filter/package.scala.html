37   /**
38    * This function rewrites a org.geotools.api.filter.Filter in terms of a top-level OR with children filters which
39    * 1) do not contain further ORs, (i.e., ORs bubble up)
40    * 2) only contain at most one AND which is at the top of their 'tree'
41    *
42    * Note that this further implies that NOTs have been 'pushed down' and do have not have ANDs nor ORs as children.
43    *
44    * In boolean logic, this form is called disjunctive normal form (DNF).
45    *
46    * @param filter An arbitrary filter.
47    * @return       A filter in DNF (described above).
48    */
49   def rewriteFilterInDNF(filter: Filter)(implicit ff: FilterFactory): Filter = {
50     val ll = logicDistributionDNF(filter)
51     if (ll.size == 1) {
52       if (ll.head.size == 1) {
53         ll.head.head
54       } else {
55         ff.and(ll.head.asJava)
56       }
57     } else {
58       val children = ll.map { l =>
59         l.size match {
60           case 1 => l.head
61           case _ => ff.and(l.asJava)
62         }
63       }
64       ff.or(children.asJava)
65     }
66   }

69    *
70    * @param x: An arbitrary @org.geotools.api.filter.Filter
71    * @return   A List[ List[Filter] ] where the inner List of Filters are to be joined by
72    *           Ands and the outer list combined by Ors.

74   private[filter] def logicDistributionDNF(x: Filter): List[List[Filter]] = x match {
75     case or: Or  => or.getChildren.asScala.toList.flatMap(logicDistributionDNF)
76 
77     case and: And => and.getChildren.asScala.foldRight (List(List.empty[Filter])) {
78       (f, dnf) => for {
79         a <- logicDistributionDNF (f)
80         b <- dnf
81       } yield a ++ b
82     }
83 
84     case not: Not =>
85       not.getFilter match {
86         case and: And => logicDistributionDNF(deMorgan(and))
87         case or:  Or => logicDistributionDNF(deMorgan(or))
88         case _: Filter => List(List(not))
89       }
90 
91     case f: Filter => List(List(f))
92   }

94   /**
95    * This function rewrites a org.geotools.api.filter.Filter in terms of a top-level AND with children filters which
96    * 1) do not contain further ANDs, (i.e., ANDs bubble up)
97    * 2) only contain at most one OR which is at the top of their 'tree'
98    *
99    * Note that this further implies that NOTs have been 'pushed down' and do have not have ANDs nor ORs as children.
100    *
101    * In boolean logic, this form is called conjunctive normal form (CNF).
102    *
103    * The main use case for this function is to aid in splitting filters between a combination of a
104    * GeoMesa data store and some other data store. This is done with the AndSplittingFilter class.
105    * In the examples below, anything with "XAttr" is assumed to be a filter that CANNOT be answered
106    * through GeoMesa. In having a filter split on the AND, the portion of the filter that GeoMesa
107    * CAN answer will be applied in GeoMesa, returning a result set, and then the portion that GeoMesa CANNOT
108    * answer will be applied on that result set.
109    *
110    * Examples:
111    *  1. (
112    *       (GmAttr ILIKE 'test')
113    *       OR
114    *       (date BETWEEN '2014-01-01T10:30:00.000Z' AND '2014-01-02T10:30:00.000Z')
115    *     )
116    *      AND
117    *     (XAttr ILIKE = 'example')
118    *
119    *     Converting to CNF will allow easily splitting the filter on the AND into two children
120    *      - one child is the "GmAttr" and "date" filters that can be answered with GeoMesa
121    *      - one child is the "XAttr" filter that cannot be answered by GeoMesa
122    *
123    *      In this case, the GeoMesa child filter will be processed first, and then the "XAttr" filter will
124    *    be processed on the GeoMesa result set to return a subset of the GeoMesa results.
125    *
126    *  2. (GmAttr ILIKE 'test')
127    *      AND
128    *          (
129    *            (date BETWEEN '2014-01-01T10:30:00.000Z' AND '2014-01-02T10:30:00.000Z')
130    *             OR
131    *            (XAttr1 ILIKE = 'example1')
132    *          )
133    *      AND
134    *     (XAttr2 ILIKE = 'example2')
135    *
136    *     Converting to CNF still allows easily splitting the filter on the AND into three children
137    *      - one child is the "GmAttr" filter
138    *      - one child is the "date" OR "XAttr1" filter
139    *      - one child is the "XAttr2" filter
140    *
141    *      In this case, the "GmAttr" child will be processed first, returning a result set from GeoMesa
142    *    called RS1. Then, RS1 will be further filtered with the "date" predicate that can be handled
143    *    by GeoMesa, returning a subset of RS1 called SS1. The additional filter which cannot be answered
144    *    by GeoMesa, "XAttr1," will be applied to RS1 and return subset SS2. Finally, the final child,
145    *    the "XAttr2" filter, which cannot be answered by GeoMesa, will be applied to both SS1 and SS2 to
146    *    return SS3, a JOIN of SS1+SS2 filtered with "XAttr2."
147    *
148    *  3. (GmAttr ILIKE 'test')
149    *      OR
150    *     (XAttr ILIKE = 'example')
151    *
152    *     This is the worst-case-scenario for a query that is answered through two data stores, both
153    *     GeoMesa and some other store.
154    *
155    *     CNF converts this to:
156    *      - one child of "GmAttr" OR "XAttr"
157    *
158    *      In this case, the "GmAttr" will return a result set, RS1. The reason this is the
159    *    worst-case-scenario is because, to answer the "XAttr" portion of the query (which cannot be
160    *    answered by GeoMesa), a "Filter.INCLUDE" A.K.A a full table scan (on Accumulo) A.K.A. every
161    *    record in GeoMesa is necessary to find the results that satisfy the "XAttr" portion of the
162    *    query. This will product result set RS2. The returned results will be a JOIN of RS1+RS2.
163    *
164    *
165    * @param filter An arbitrary filter.
166    * @return       A filter in CNF (described above).
167    */
168   def rewriteFilterInCNF(filter: Filter)(implicit ff: FilterFactory): Filter = {
169     val ll = logicDistributionCNF(FilterHelper.simplify(filter))
170     if (ll.size == 1) {
171       if (ll.head.size == 1) ll.head.head else ff.or(ll.head.asJava)
172     } else {
173       val children = ll.map { l =>
174         l.size match {
175           case 1 => l.head
176           case _ => ff.or(l.asJava)
177         }
178       }
179       ff.and(children.asJava)
180     }
181   }

209   /**
210    *  The input is a filter which had a Not applied to it.
211    *  This function uses deMorgan's law to 'push the Not down'
212    *   as well as cancel adjacent Nots.
213    */
214   private[filter] def deMorgan(f: Filter)(implicit ff: FilterFactory): Filter = f match {
215     case and: And => ff.or(and.getChildren.asScala.map(a => ff.not(a).asInstanceOf[Filter]).asJava)
216     case or:  Or  => ff.and(or.getChildren.asScala.map(a => ff.not(a).asInstanceOf[Filter]).asJava)
217     case not: Not => not.getFilter
218   }

283   def getAttributeProperty(f: Filter): Option[String] = {
284     f match {
285       // equals checks
286       case f: PropertyIsEqualTo => checkOrder(f.getExpression1, f.getExpression2).map(_.name)
287       case f: TEquals           => checkOrder(f.getExpression1, f.getExpression2).map(_.name)
288 
289       // like checks
290       case f: PropertyIsLike =>
291         if (likeEligible(f)) {
292           val prop = f.getExpression.asInstanceOf[PropertyName].getPropertyName
293           Some(prop)
294         } else {
295           None
296         }
297 
298       // range checks
299       case f: PropertyIsGreaterThan          => checkOrder(f.getExpression1, f.getExpression2).map(_.name)
300       case f: PropertyIsGreaterThanOrEqualTo => checkOrder(f.getExpression1, f.getExpression2).map(_.name)
301       case f: PropertyIsLessThan             => checkOrder(f.getExpression1, f.getExpression2).map(_.name)
302       case f: PropertyIsLessThanOrEqualTo    => checkOrder(f.getExpression1, f.getExpression2).map(_.name)
303       case f: PropertyIsBetween =>
304         val prop = f.getExpression.asInstanceOf[PropertyName].getPropertyName
305         val (left, right) = (f.getLowerBoundary, f.getUpperBoundary) match {
306           case (l: Literal, r: Literal) => (l, r)
307           case _ => (null, null)
308         }
309         if (left != null && right != null) {
310           Some(prop)
311         } else {
312           None
313         }
314 
315       // date range checks
316       case f: Before => checkOrder(f.getExpression1, f.getExpression2).map(_.name)
317       case f: After  => checkOrder(f.getExpression1, f.getExpression2).map(_.name)
318       case f: During => checkOrder(f.getExpression1, f.getExpression2).map(_.name)
319 
320       // not check - we only support 'not null' for an indexed attribute
321       case n: Not =>
322         Option(n.getFilter).collect { case f: PropertyIsNull => f }.map { f =>
323           f.getExpression.asInstanceOf[PropertyName].getPropertyName
324         }
325 
326       case _ => None
327     }
328   }

390   /**
391    * Checks the order of properties and literals in the expression
392    *
393    * @param one first expression
394    * @param two second expression
395    * @return (prop, literal, whether the order was flipped)
396    */
397   def checkOrder(one: Expression, two: Expression): Option[AttributeExpression] = {
398     (one, two) match {
399       case (p: PropertyName, l: Literal) => Some(PropertyLiteral(p.getPropertyName, l, flipped = false))
400       case (l: Literal, p: PropertyName) => Some(PropertyLiteral(p.getPropertyName, l, flipped = true))
401       case (f: Function, l: Literal)     => attribute(f).map(FunctionLiteral(_, f, l, flipped = false))
402       case (l: Literal, f: Function)     => attribute(f).map(FunctionLiteral(_, f, l, flipped = true))
403 
404       case (p: PropertyName, f: Function) if attribute(f).isEmpty =>
405         Some(PropertyLiteral(p.getPropertyName, ff.literal(f.evaluate(null)), flipped = false))
406 
407       case (f: Function, p: PropertyName) if attribute(f).isEmpty =>
408         Some(PropertyLiteral(p.getPropertyName, ff.literal(f.evaluate(null)), flipped = true))
409 
410       case (f1: Function, f2: Function) =>
411         (attribute(f1), attribute(f2)) match {
412           case (Some(a), None) => Some(FunctionLiteral(a, f1, ff.literal(f2.evaluate(null)), flipped = false))
413           case (None, Some(a)) => Some(FunctionLiteral(a, f2, ff.literal(f1.evaluate(null)), flipped = true))
414           case _ => None
415         }
416 
417       case _ => None
418     }
419   }
