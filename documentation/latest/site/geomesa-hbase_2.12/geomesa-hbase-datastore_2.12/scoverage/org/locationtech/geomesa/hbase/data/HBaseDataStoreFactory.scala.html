40   override def createDataStore(params: java.util.Map[String, _]): DataStore = {
41     val connection = HBaseConnectionPool.getConnection(params, validateConnection)
42 
43     val remoteFilters = RemoteFilteringParam.lookup(params).booleanValue
44 
45     val audit = if (!AuditQueriesParam.lookup(params)) { None } else {
46       Some(new AuditLogger("hbase", AuditProvider.Loader.loadOrNone(params)))
47     }
48     val metrics = MetricsRegistryParam.lookupRegistry(params)
49     val auths = if (!EnableSecurityParam.lookup(params)) { NoAuthsProvider } else {
50       HBaseDataStoreFactory.buildAuthsProvider(connection.connection, params)
51     }
52     val queries = HBaseQueryConfig(
53       threads = QueryThreadsParam.lookup(params),
54       timeout = QueryTimeoutParam.lookupOpt(params).map(_.toMillis),
55       looseBBox = LooseBBoxParam.lookup(params),
56       parallelPartitionScans = PartitionParallelScansParam.lookup(params),
57       maxRangesPerExtendedScan = MaxRangesPerExtendedScanParam.lookup(params)
58     )
59     val enabledCoprocessors = EnabledCoprocessors(
60       arrow = ArrowCoprocessorParam.lookup(params),
61       bin = BinCoprocessorParam.lookup(params),
62       density = DensityCoprocessorParam.lookup(params),
63       stats = StatsCoprocessorParam.lookup(params)
64     )
65     val coprocessors = CoprocessorConfig(
66       enabled = enabledCoprocessors,
67       threads = CoprocessorThreadsParam.lookup(params),
68       yieldPartialResults = YieldPartialResultsParam.lookup(params),
69       maxRangesPerExtendedScan = MaxRangesPerCoprocessorScanParam.lookup(params),
70       url = CoprocessorUrlParam.lookupOpt(params)
71     )
72     val config = HBaseDataStoreConfig(
73       catalog = HBaseCatalogParam.lookup(params),
74       remoteFilter = remoteFilters,
75       generateStats = GenerateStatsParam.lookup(params),
76       queries = queries,
77       coprocessors = coprocessors,
78       authProvider = auths,
79       audit = audit,
80       metrics = metrics,
81       namespace = NamespaceParam.lookupOpt(params)
82     )
83 
84     logger.debug(s"Using ${if (remoteFilters) "remote" else "local" } filtering")
85     lazy val enabled =
86       Seq(ArrowCoprocessorParam, BinCoprocessorParam, DensityCoprocessorParam, StatsCoprocessorParam).collect {
87         case p if p.exists(params) && p.lookup(params).booleanValue() => p.key
88       }
89     if (!remoteFilters && enabled.nonEmpty) {
90       logger.warn(s"Ignoring configs '${enabled.mkString("', '")}' due to remote filtering being disabled")
91     }
92 
93     val ds = new HBaseDataStore(connection, config)
94     GeoMesaDataStore.initRemoteVersion(ds)
95     ds
96   }

194   def buildAuthsProvider(connection: Connection, params: java.util.Map[String, _]): AuthorizationsProvider = {
195     val forceEmptyOpt: Option[java.lang.Boolean] = ForceEmptyAuthsParam.lookupOpt(params)
196     val forceEmptyAuths = forceEmptyOpt.getOrElse(java.lang.Boolean.FALSE).asInstanceOf[Boolean]
197 
198     if (!VisibilityClient.isCellVisibilityEnabled(connection)) {
199       throw new IllegalArgumentException("HBase cell visibility is not enabled on cluster")
200     }
201 
202     // master auths is the superset of auths this connector/user can support
203     val userName = UserGroupInformation.getLoginUser.getUserName
204     val masterAuths = VisibilityClient.getAuths(connection, userName).getAuthList.asScala.map(_.toStringUtf8)
205 
206     // get the auth params passed in as a comma-delimited string
207     val configuredAuths = AuthsParam.lookupOpt(params).getOrElse("").split(",").filterNot(_.isEmpty)
208 
209     // verify that the configured auths are valid for the connector we are using (fail-fast)
210     val invalidAuths = configuredAuths.filterNot(masterAuths.contains)
211     if (invalidAuths.nonEmpty) {
212       val msg = s"The authorizations '${invalidAuths.mkString("', '")}' are not valid for the HBase user '$userName'"
213       if (masterAuths.isEmpty) {
214         // looking up auths requires a system-level user - likely the user does not have permission
215         logger.warn(s"$msg. This may be due to the user not having permissions" +
216             " to read its own authorizations, in which case this warning can be ignored.")
217       } else {
218         throw new IllegalArgumentException(s"$msg. Available authorizations are: ${masterAuths.mkString(", ")}")
219       }
220     }
221 
222     // if the caller provided any non-null string for authorizations, use it;
223     // otherwise, grab all authorizations to which the user is entitled
224     if (configuredAuths.length != 0 && forceEmptyAuths) {
225       throw new IllegalArgumentException("Forcing empty auths is checked, but explicit auths are provided")
226     }
227     val auths = if (forceEmptyAuths || configuredAuths.nonEmpty) { configuredAuths.toList } else { masterAuths.toList }
228 
229     AuthUtils.getProvider(params, auths)
230   }
