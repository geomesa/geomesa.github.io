78   override def createTable(
79       index: GeoMesaFeatureIndex[_, _],
80       partition: Option[String],
81       splits: => Seq[Array[Byte]]): Unit = {
82     // write table name to metadata
83     val name = TableName.valueOf(index.configureTableName(partition, tableNameLimit))
84 
85     WithClose(ds.connection.getAdmin) { admin =>
86       if (!admin.tableExists(name)) {
87         logger.debug(s"Creating table $name")
88 
89         val conf = admin.getConfiguration
90 
91         val compression = index.sft.getCompression.map { alg =>
92           logger.debug(s"Setting compression '$alg' on table $name for feature ${index.sft.getTypeName}")
93           // note: all compression types in HBase are case-sensitive and lower-cased
94           Compression.getCompressionAlgorithmByName(alg.toLowerCase(Locale.US))
95         }
96 
97         val cols = groups.apply(index.sft).map(_._1)
98         val bloom = Some(BloomType.NONE)
99         val encoding = if (index.name == IdIndex.name) { None } else { Some(DataBlockEncoding.FAST_DIFF) }
100 
101         val coprocessor = if (!ds.config.remoteFilter) { None } else {
102           // if the coprocessors are installed site-wide don't register them in the table descriptor.
103           // this key is CoprocessorHost.USER_REGION_COPROCESSOR_CONF_KEY - but don't want to pull in
104           // a dependency on hbase-server just for this constant
105           val installed = Option(conf.get("hbase.coprocessor.user.region.classes"))
106           val names = installed.map(_.split(":").toSet).getOrElse(Set.empty[String])
107           if (names.contains(CoprocessorClass)) { None } else {
108             // noinspection ScalaDeprecation
109             def urlFromSysProp: Option[Path] = CoprocessorUrl.option.orElse(CoprocessorPath.option).map(new Path(_))
110             val coprocessorUrl = ds.config.coprocessors.url.orElse(urlFromSysProp).orElse(dynamicJarPath)
111             logger.debug(s"Using coprocessor path ${coprocessorUrl.orNull}")
112             // TODO: Warn if the path given is different from paths registered in other coprocessors
113             // if so, other tables would need updating
114             Some(CoprocessorClass -> coprocessorUrl)
115           }
116         }
117         val metadata = index.sft.getTableProps
118 
119         try {
120           createTableAsync(admin, name, cols, bloom, compression, encoding, None, coprocessor, splits, metadata)
121         } catch {
122           case _: org.apache.hadoop.hbase.TableExistsException => // ignore, another thread created it for us
123         }
124       }
125 
126       waitForTable(admin, name)
127     }
128   }

130   override def renameTable(from: String, to: String): Unit = {
131     WithClose(ds.connection.getAdmin) { admin =>
132       val existing = TableName.valueOf(from)
133       val renamed = TableName.valueOf(to)
134       if (admin.tableExists(existing)) {
135         // renaming in hbase requires creating a snapshot and using that to create the new table
136         val snapshot = StringSerialization.alphaNumericSafeString(UUID.randomUUID().toString)
137         admin.disableTable(existing)
138         admin.snapshot(snapshot, existing)
139         admin.cloneSnapshot(snapshot, renamed)
140         admin.deleteSnapshot(snapshot)
141         admin.deleteTable(existing)
142         waitForTable(admin, renamed)
143       }
144     }

147   override def deleteTables(tables: Seq[String]): Unit = {
148     WithClose(ds.connection.getAdmin) { admin =>
149       def deleteOne(name: String): Unit = {
150         val table = TableName.valueOf(name)
151         if (admin.tableExists(table)) {
152           admin.disableTableAsync(table)
153           val timeout = TableAvailabilityTimeout.toUnboundedDuration.filter(_.isFinite)
154           logger.debug(s"Waiting for table '$table' to be disabled with " +
155               s"${timeout.map(t => s"a timeout of $t").getOrElse("no timeout")}")
156           val stop = timeout.map(t => System.currentTimeMillis() + t.toMillis)
157           while (!admin.isTableDisabled(table) && stop.forall(_ > System.currentTimeMillis())) {
158             Thread.sleep(1000)
159           }
160           // no async operation, but timeout can be controlled through hbase-site.xml "hbase.client.sync.wait.timeout.msec"
161           admin.deleteTable(table)
162         }
163       }
164       tables.toList.map(t => CachedThreadPool.submit(() => deleteOne(t))).foreach(_.get)
165     }
166   }

168   override def clearTables(tables: Seq[String], prefix: Option[Array[Byte]]): Unit = {
169     def clearOne(name: String): Unit = {
170       val tableName = TableName.valueOf(name)
171       WithClose(ds.connection.getTable(tableName)) { table =>
172         val scan = new Scan().setFilter(new KeyOnlyFilter)
173         prefix.foreach(scan.setStartStopRowForPrefixScan)
174         ds.applySecurity(scan)
175         val mutateParams = new BufferedMutatorParams(tableName)
176         WithClose(table.getScanner(scan), ds.connection.getBufferedMutator(mutateParams)) { case (scanner, mutator) =>
177           scanner.iterator.asScala.grouped(10000).foreach { result =>
178             // TODO GEOMESA-2546 set delete visibilities
179             val deletes = result.map(r => new Delete(r.getRow))
180             mutator.mutate(deletes.asJava)
181           }
182         }
183       }
184     }
185     tables.toList.map(t => CachedThreadPool.submit(() => clearOne(t))).foreach(_.get)
186   }

188   override def createQueryPlan(strategy: QueryStrategy): HBaseQueryPlan = {
189 
190     import org.locationtech.geomesa.index.conf.QueryHints.RichHints
191 
192     val byteRanges = strategy.ranges
193     val hints = strategy.hints
194     val index = strategy.index
195 
196     // index api defines empty start/end for open-ended range
197     // index api defines start row inclusive, end row exclusive
198     // both these conventions match the conventions for hbase scan objects
199     val ranges = byteRanges.map {
200       case BoundedByteRange(start, stop) => new RowRange(start, true, stop, false)
201       case SingleRowByteRange(row)       => new RowRange(row, true, ByteArrays.rowFollowingRow(row), false)
202     }
203     val small = byteRanges.headOption.exists(_.isInstanceOf[SingleRowByteRange])
204 
205     val tables = index.getTablesForQuery(strategy.filter.filter).map(TableName.valueOf)
206 
207     // check for an empty query plan, if there are no tables or ranges to scan
208     def empty(reducer: Option[FeatureReducer]): Option[HBaseQueryPlan] =
209       if (tables.isEmpty || ranges.isEmpty) { Some(EmptyPlan(ds, strategy, reducer)) } else { None }
210 
211     if (!ds.config.remoteFilter) {
212       // everything is done client side
213       // note: we use the full filter here, since we can't use the z3 server-side filter
214       // for some attribute queries we wouldn't need the full filter...
215       val (colFamily, schema) = groups.group(index.sft, hints.getTransformDefinition, strategy.filter.filter)
216       // re-optimize the filter to account for the colFamily/schema being used
217       val ecql = {
218         val f = strategy.filter.filter
219         if (colFamily.eq(ColumnGroups.Default)) { f } else { f.map(FastFilterFactory.optimize(schema, _)) }
220       }
221       // note: we assume visibility filtering is still done server-side as it's part of core hbase
222       val processor = LocalProcessor(schema, hints, None)
223       empty(processor.reducer).getOrElse {
224         val scans = configureScans(tables, ranges, small, colFamily, Seq.empty, coprocessor = false)
225         LocalProcessorScanPlan(ds, strategy, ranges, scans, ecql, processor, hints.getProjection)
226       }
227     } else {
228       val ecql = strategy.ecql
229       val (colFamily, schema) = groups.group(index.sft, hints.getTransformDefinition, ecql)
230       val projection = hints.getProjection
231       val indexFilter = strategy.values.flatMap(IndexFilters(index, _))
232       lazy val returnSchema = hints.getTransformSchema.getOrElse(schema)
233       lazy val scans = {
234         val transform = hints.getTransform
235         val cqlFilter = if (ecql.isEmpty && transform.isEmpty && hints.getSampling.isEmpty) { Seq.empty } else {
236           Seq((CqlTransformFilter.Priority, CqlTransformFilter(schema, index, ecql, transform, hints)))
237         }
238         val filters = (cqlFilter ++ indexFilter).sortBy(_._1).map(_._2)
239         configureScans(tables, ranges, small, colFamily, filters, coprocessor = false)
240       }
241 
242       def coprocessorPlan(options: Map[String, String], toFeatures: ResultsToFeatures[Array[Byte]], reducer: Option[FeatureReducer]): CoprocessorPlan = {
243         val coprocessorOptions = Map(GeoMesaCoprocessor.YieldOpt -> String.valueOf(ds.config.coprocessors.yieldPartialResults))
244         val scans = configureScans(tables, ranges, small, colFamily, indexFilter.toSeq.map(_._2), coprocessor = true)
245         CoprocessorPlan(ds, strategy, ranges, scans, options ++ coprocessorOptions, toFeatures, reducer, hints.getMaxFeatures, projection)
246       }
247 
248       def semiLocalPlan(): LocalProcessorScanPlan = {
249         // note: transforms are handled in the cqlTransformFilter
250         val processor = LocalProcessor(returnSchema, QueryHints.Internal.clearTransforms(hints), None)
251         LocalProcessorScanPlan(ds, strategy, ranges, scans, None, processor, projection)
252       }
253 
254       if (hints.isDensityQuery) {
255         empty(None).getOrElse {
256           if (ds.config.coprocessors.enabled.density) {
257             val options = HBaseDensityAggregator.configure(schema, index, ecql, hints)
258             val results = new HBaseDensityResultsToFeatures()
259             coprocessorPlan(options, results, None)
260           } else {
261             semiLocalPlan()
262           }
263         }
264       } else if (hints.isArrowQuery) {
265         val config = HBaseArrowAggregator.configure(schema, index, ds.stats, strategy.filter.filter, ecql, hints)
266         val reducer = Some(config.reduce)
267         empty(reducer).getOrElse {
268           if (ds.config.coprocessors.enabled.arrow) {
269             val options = config.config
270             val results = new HBaseArrowResultsToFeatures()
271             coprocessorPlan(options, results, reducer)
272           } else {
273             semiLocalPlan()
274           }
275         }
276       } else if (hints.isStatsQuery) {
277         val reducer = Some(StatsScan.StatsReducer(returnSchema, hints))
278         empty(reducer).getOrElse {
279           if (ds.config.coprocessors.enabled.stats) {
280             val options = HBaseStatsAggregator.configure(schema, index, ecql, hints)
281             val results = new HBaseStatsResultsToFeatures()
282             coprocessorPlan(options, results, reducer)
283           } else {
284             semiLocalPlan()
285           }
286         }
287       } else if (hints.isBinQuery) {
288         empty(None).getOrElse {
289           if (ds.config.coprocessors.enabled.bin) {
290             val options = HBaseBinAggregator.configure(schema, index, ecql, hints)
291             val results = new HBaseBinResultsToFeatures()
292             coprocessorPlan(options, results, None)
293           } else {
294             semiLocalPlan()
295           }
296         }
297       } else {
298         empty(None).getOrElse {
299           val resultsToFeatures = new HBaseResultsToFeatures(index, returnSchema)
300           ScanPlan(ds, strategy, ranges, scans, resultsToFeatures, hints.getSortFields, hints.getMaxFeatures, projection)
301         }
302       }
303     }
304   }

323   /**
324    * Configure the hbase scan
325    *
326    * @param tables tables being scanned, used for region location information
327    * @param ranges ranges to scan, non-empty. needs to be mutable as we will sort it in place
328    * @param small whether 'small' ranges (i.e. gets)
329    * @param colFamily col family to scan
330    * @param filters scan filters
331    * @param coprocessor is this a coprocessor scan or not
332    * @return
333    */
334   private def configureScans(
335       tables: Seq[TableName],
336       ranges: Seq[RowRange],
337       small: Boolean,
338       colFamily: Array[Byte],
339       filters: Seq[HFilter],
340       coprocessor: Boolean): Seq[TableScan] = {
341     val cacheBlocks = HBaseSystemProperties.ScannerBlockCaching.toBoolean.get // has a default value so .get is safe
342     val cacheSize = HBaseSystemProperties.ScannerCaching.toInt
343 
344     logger.debug(s"HBase client scanner: block caching: $cacheBlocks, caching: $cacheSize")
345 
346     if (small && !coprocessor) {
347       val filter = filters match {
348         case Nil    => None
349         case Seq(f) => Some(f)
350         case f      => Some(new FilterList(f: _*))
351       }
352       // note: we have to copy the ranges for each table scan
353       tables.map { table =>
354         val scans = ranges.map { r =>
355           val scan = new Scan().withStartRow(r.getStartRow).withStopRow(r.getStopRow)
356           scan.addFamily(colFamily).setCacheBlocks(cacheBlocks).setSmall(true)
357           filter.foreach(scan.setFilter)
358           cacheSize.foreach(scan.setCaching)
359           ds.applySecurity(scan)
360           scan
361         }
362         TableScan(table, scans)
363       }
364     } else {
365       // split and group ranges by region server
366       // note: we have to copy the ranges for each table scan anyway
367       val rangesPerTable: Seq[(TableName, collection.Map[String, java.util.List[RowRange]])] =
368         tables.map(t => t -> groupRangesByRegion(t, ranges))
369 
370       def createGroup(group: java.util.List[RowRange]): Scan = {
371         val scan = new Scan().withStartRow(group.get(0).getStartRow).withStopRow(group.get(group.size() - 1).getStopRow)
372         val mrrf = if (group.size() < 2) { filters } else {
373           // TODO GEOMESA-1806
374           // currently, the MultiRowRangeFilter constructor will call sortAndMerge a second time
375           // this is unnecessary as we have already sorted and merged
376           // note: mrrf first priority
377           filters.+:(new MultiRowRangeFilter(group))
378         }
379         scan.setFilter(if (mrrf.lengthCompare(1) > 0) { new FilterList(mrrf: _*) } else { mrrf.headOption.orNull })
380         scan.addFamily(colFamily).setCacheBlocks(cacheBlocks)
381         cacheSize.foreach(scan.setCaching)
382 
383         // apply visibilities
384         ds.applySecurity(scan)
385 
386         scan
387       }
388 
389       rangesPerTable.map { case (table, rangesPerRegion) =>
390         val maxRangesPerGroup = {
391           def calcMax(maxPerGroup: Int, threads: Int): Int = {
392             val totalRanges = rangesPerRegion.values.map(_.size).sum
393             math.min(maxPerGroup, math.max(1, math.ceil(totalRanges.toDouble / threads).toInt))
394           }
395           if (coprocessor) {
396             calcMax(ds.config.coprocessors.maxRangesPerExtendedScan, ds.config.coprocessors.threads)
397           } else {
398             calcMax(ds.config.queries.maxRangesPerExtendedScan, ds.config.queries.threads)
399           }
400         }
401 
402         val groupedScans = Seq.newBuilder[Scan]
403 
404         rangesPerRegion.foreach { case (_, list) =>
405           // our ranges are non-overlapping, so just sort them but don't bother merging them
406           Collections.sort(list)
407 
408           var i = 0
409           while (i < list.size()) {
410             val groupSize = math.min(maxRangesPerGroup, list.size() - i)
411             groupedScans += createGroup(list.subList(i, i + groupSize))
412             i += groupSize
413           }
414         }
415 
416         // shuffle the ranges, otherwise our threads will tend to all hit the same region server at once
417         TableScan(table, Random.shuffle(groupedScans.result))
418       }
419     }
420   }

439   /**
440    * Group the range based on the region server hosting it. Splits ranges as needed if they span
441    * more than one region
442    *
443    * @param locator region locator
444    * @param range range to group
445    * @param result collected results
446    */
447   @scala.annotation.tailrec
448   private def groupRange(
449       locator: RegionLocator,
450       range: RowRange,
451       result: scala.collection.mutable.Map[String, java.util.List[RowRange]]): Unit = {
452     var encodedName: String = null
453     var split: Array[Byte] = null
454     try {
455       val regionInfo = locator.getRegionLocation(range.getStartRow).getRegionInfo
456       encodedName = regionInfo.getEncodedName
457       val regionEndKey = regionInfo.getEndKey // note: this is exclusive
458       if (regionEndKey.nonEmpty &&
459           (range.getStopRow.isEmpty || ByteArrays.ByteOrdering.compare(regionEndKey, range.getStopRow) <= 0)) {
460         if (ByteArrays.ByteOrdering.compare(range.getStartRow, regionEndKey) < 0) {
461           split = regionEndKey
462         } else {
463           logger.warn(s"HBase region location does not correspond to requested range:\n" +
464               s"  requested row: ${ByteArrays.toHex(range.getStartRow)}\n" +
465               s"  region: $encodedName ${ByteArrays.toHex(regionInfo.getStartKey)} :: ${ByteArrays.toHex(regionEndKey)}")
466         }
467       }
468     } catch {
469       case NonFatal(e) => logger.warn(s"Error checking range location for '$range''", e)
470     }
471     val buffer = result.getOrElseUpdate(encodedName, new java.util.ArrayList())
472     if (split == null) {
473       buffer.add(range)
474     } else {
475       // split the range based on the current region
476       buffer.add(new RowRange(range.getStartRow, true, split, false))
477       groupRange(locator, new RowRange(split, true, range.getStopRow, false), result)
478     }
479   }

502    * Create a new table
503    *
504    * @param admin admin connection to hbase

507    * @param bloom bloom filter

513   def createTableAsync(
514       admin: Admin,
515       name: TableName,
516       colFamilies: Seq[Array[Byte]],
517       bloom: Option[BloomType],
518       compression: Option[Algorithm],
519       encoding: Option[DataBlockEncoding],
520       inMemory: Option[Boolean],
521       coprocessor: Option[(String, Option[Path])],
522       splits: Seq[Array[Byte]],
523       metadata: Map[String, String],
524     ): Unit = {
525 
526     val namespace = name.getNamespaceAsString
527     if (namespace != null &&
528         namespace != NamespaceDescriptor.DEFAULT_NAMESPACE_NAME_STR &&
529         namespace != NamespaceDescriptor.SYSTEM_NAMESPACE_NAME_STR &&
530         Try(Option(admin.getNamespaceDescriptor(namespace))).getOrElse(None).isEmpty) {
531       admin.createNamespace(NamespaceDescriptor.create(namespace).build())
532     }
533 
534     val builder = TableDescriptorBuilder.newBuilder(name)
535     val columnFamilyDescriptors = colFamilies.map { colFamily =>
