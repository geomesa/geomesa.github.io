43   override def scan(ds: HBaseDataStore): CloseableIterator[Results] = {
44     // query guard hook - also handles full table scan checks
45     strategy.runGuards(ds)
46     // convert the relative timeout to an absolute timeout up front
47     val timeout = ds.config.queries.timeout.map(Timeout.apply)
48     val iter = scans.iterator.map(singleTableScan(_, ds.connection, threads(ds), timeout))
49     if (ds.config.queries.parallelPartitionScans) {
50       // kick off all the scans at once
51       iter.foldLeft(CloseableIterator.empty[Results])(_ concat _)
52     } else {
53       // kick off the scans sequentially as they finish
54       SelfClosingIterator(iter).flatMap(s => s)
55     }
56   }
57 
58   override def explain(explainer: Explainer, prefix: String = ""): Unit = {
59     explainer.pushLevel(s"${prefix}Plan: ${getClass.getSimpleName}")
60     explainer(s"Tables: ${scans.map(_.table).mkString(", ")}")
61     explainer(s"Ranges (${ranges.size}): ${ranges.take(5).map(rangeToString).mkString(", ")}")
62     explainer(s"Scans (${scans.headOption.map(_.scans.size).getOrElse(0)}): ${scans.headOption.toSeq.flatMap(_.scans.take(5)).map(scanToString).mkString(", ")}")
63     explainer(s"Column families: ${scans.headOption.flatMap(_.scans.headOption).flatMap(r => Option(r.getFamilies)).getOrElse(Array.empty).map(Bytes.toString).mkString(",")}")
64     explainer(s"Remote filters: ${scans.headOption.flatMap(_.scans.headOption).flatMap(r => Option(r.getFilter)).map(filterToString).getOrElse("none")}")
65     explain(explainer)
66     explainer(s"Reduce: ${reducer.getOrElse("none")}")
67     explainer.popLevel()
68   }
