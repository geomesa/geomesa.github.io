33 /**
34  * Build a tube for input to a TubeSelect by buffering and binning the input
35  * tubeFeatures into SimpleFeatures that can be used as inputs to Geomesa queries
36  */
37 abstract class TubeBuilder(val tubeFeatures: SimpleFeatureCollection,
38                            val bufferDistance: Double,
39                            val maxBins: Int) extends LazyLogging {
40 
41   val calc = new GeodeticCalculator()
42   val dtgField: String = tubeFeatures.getSchema.getDtgField.getOrElse(TubeBuilder.DefaultDtgField)
43   val geoFac = new GeometryFactory
44 
45   val GEOM_PROP = "geom"
46 
47   val tubeType: SimpleFeatureType = SimpleFeatureTypes.createType("tubeType", s"$GEOM_PROP:Geometry:srid=4326,start:Date,end:Date")
48   val builder: SimpleFeatureBuilder = ScalaSimpleFeatureFactory.featureBuilder(tubeType)
49 
50   def getGeom(sf: SimpleFeature): Geometry = sf.getAttribute(0).asInstanceOf[Geometry]
51   def getStartTime(sf: SimpleFeature): Date = sf.getAttribute(1).asInstanceOf[Date]
52   def getEndTime(sf: SimpleFeature): Date = sf.getAttribute(2).asInstanceOf[Date]
53 
54   def bufferGeom(geom: Geometry, meters: Double): Geometry = {
55     import org.locationtech.geomesa.utils.geotools.Conversions.RichGeometry
56     geom.buffer(metersToDegrees(meters, geom.safeCentroid()))
57   }
58 
59   def metersToDegrees(meters: Double, point: Point): Double = {
60     logger.debug("Buffering: "+meters.toString + " "+WKTUtils.write(point))
61 
62     calc.setStartingGeographicPoint(point.getX, point.getY)
63     calc.setDirection(0, meters)
64     val dest2D = calc.getDestinationGeographicPoint
65     val destPoint = geoFac.createPoint(new Coordinate(dest2D.getX, dest2D.getY))
66     point.distance(destPoint)
67   }
68 
69   def buffer(simpleFeatures: Iterator[SimpleFeature], meters:Double): Iterator[SimpleFeature] =
70     simpleFeatures.map { sf =>
71       val bufferedGeom = bufferGeom(getGeom(sf), meters)
72       builder.reset()
73       builder.init(sf)
74       builder.set(GEOM_PROP, bufferedGeom)
75       builder.buildFeature(sf.getID)
76     }
77 
78   // transform the input tubeFeatures into the intermediate SF used by the
79   // tubing code consisting of three attributes (geom, startTime, endTime)
80   // handle date parsing from input -> TODO revisit date parsing...
81   def transform(tubeFeatures: SimpleFeatureCollection, dtgField: String): Iterator[SimpleFeature] = {
82     SelfClosingIterator(tubeFeatures.features).map { sf =>
83       val date = FastConverter.convert(sf.getAttribute(dtgField), classOf[Date])
84 
85       if (date == null) {
86         logger.error("Unable to retrieve date field from input tubeFeatures...ensure there a field named " + dtgField)
87         throw new IllegalArgumentException("Unable to retrieve date field from input tubeFeatures...ensure there a field named \"" + dtgField + "\"")
88       }
89 
90       builder.reset()
91       builder.buildFeature(sf.getID, sf.getDefaultGeometry, date, null)
92     }
93   }
94 
95   /**
96     * Return an Array containing either 1 or 2 LineStrings that straddle but
97     * do not cross the IDL.
98     * @param input1 The first point in the segment
99     * @param input2 The second point in the segment
100     * @return an array of LineString containing either 1 or 2 LineStrings that do not
101     *         span the IDL.
102     */
103   def makeIDLSafeLineString(input1:Coordinate, input2:Coordinate): Geometry = {
104     //If the points cross the IDL we must generate two line segments
105     if (GeometryUtils.crossesIDL(input1, input2)) {
106       //Find the latitude where the segment intercepts the IDL
107       val latIntercept = GeometryUtils.calcIDLIntercept(input1, input2)
108       val p1 = new Coordinate(-180, latIntercept)
109       val p2 = new Coordinate(180, latIntercept)
110       //This orders the points so that point1 is always the east-most point
111       val (point1, point2) = if (input1.x > 0) (input1, input2) else (input2, input1)
112       val westLine = new LineString(new CoordinateArraySequence(Array(p1, point2)), geoFac)
113       val eastLine = new LineString(new CoordinateArraySequence(Array(point1, p2)), geoFac)
114       new MultiLineString(Array[LineString](westLine,eastLine), geoFac)
115     } else {
116       new LineString(new CoordinateArraySequence(Array(input1, input2)), geoFac)
117     }
118   }
119 
120   def createTube: Iterator[SimpleFeature]
121 }

123 /**
124  * Build a tube with no gap filling - only buffering and binning
125  */
126 class NoGapFill(tubeFeatures: SimpleFeatureCollection,
127                 bufferDistance: Double,
128                 maxBins: Int) extends TubeBuilder(tubeFeatures, bufferDistance, maxBins) with LazyLogging {
129 
130   // Bin ordered features into maxBins that retain order by date then union by geometry
131   def timeBinAndUnion(features: Iterable[SimpleFeature], maxBins: Int): Iterator[SimpleFeature] = {
132     val numFeatures = features.size
133 
134     if (numFeatures == 0) { Iterator.empty } else {
135       //If 0 is passed in then don't bin the features, if 1 then make one bin, otherwise calculate number
136       //of bins based on numFeatures and maxBins
137       val binSize = maxBins match {
138         case 0 => 1
139         case 1 => numFeatures
140         case _ => numFeatures / maxBins + (if (numFeatures % maxBins == 0 ) 0 else 1)
141       }
142       features.grouped(binSize).zipWithIndex.map { case(bin, idx) => unionFeatures(bin.toSeq, idx.toString) }
143     }
144   }
145 
146   // Union features to create a single geometry and single combined time range
147   def unionFeatures(orderedFeatures: Seq[SimpleFeature], id: String): SimpleFeature = {
148     import scala.collection.JavaConverters._
149     val geoms = orderedFeatures.map { sf => getGeom(sf) }
150     val unionGeom = geoFac.buildGeometry(geoms.asJava).union
151     val min = getStartTime(orderedFeatures.head)
152     val max = getStartTime(orderedFeatures.last)
153 
154     builder.reset()
155     builder.buildFeature(id, unionGeom, min, max)
156   }
157 
158   override def createTube: Iterator[SimpleFeature] = {
159     logger.debug("Creating tube with no gap filling")
160 
161     val transformed = transform(tubeFeatures, dtgField)
162     val buffered = buffer(transformed, bufferDistance)
163     val sortedTube = buffered.toSeq.sortBy { sf => getStartTime(sf).getTime }
164 
165     logger.debug(s"sorted tube size: ${sortedTube.size}")
166     timeBinAndUnion(sortedTube, maxBins)
167   }
168 }

173 /**
174  * Build a tube with gap filling that draws a line between time-ordered features
175  * from the given tubeFeatures
176  */
177 class LineGapFill(tubeFeatures: SimpleFeatureCollection,
178                   bufferDistance: Double,
179                   maxBins: Int) extends TubeBuilder(tubeFeatures, bufferDistance, maxBins) with LazyLogging {
180 
181   val id = new AtomicInteger(0)
182 
183   def nextId: String = id.getAndIncrement.toString
184 
185   override def createTube: Iterator[SimpleFeature] = {
186     import org.locationtech.geomesa.utils.geotools.Conversions.RichGeometry
187 
188     logger.debug("Creating tube with line gap fill")
189 
190     val transformed = transform(tubeFeatures, dtgField)
191     val sortedTube = transformed.toSeq.sortBy { sf => getStartTime(sf).getTime }
192     val pointsAndTimes = sortedTube.map(sf => (getGeom(sf).safeCentroid(), getStartTime(sf)))
193     val lineFeatures = if (pointsAndTimes.lengthCompare(1) == 0) {
194       val (p1, t1) = pointsAndTimes.head
195       logger.debug("Only a single result - can't create a line")
196       Iterator(builder.buildFeature(nextId, p1, t1, t1))
197     } else {
198       pointsAndTimes.sliding(2).map { case Seq((p1, t1), (p2, t2)) =>
199         val geo = if (p1.equals(p2)) p1 else makeIDLSafeLineString(p1.getCoordinate,p2.getCoordinate)
200         logger.debug(s"Created Line-filled Geometry: ${WKTUtils.write(geo)} From ${WKTUtils.write(p1)} and ${WKTUtils.write(p2)}")
201         builder.buildFeature(nextId, geo, t1, t2)
202       }
203     }
204     buffer(lineFeatures, bufferDistance)
205   }
206 }

214 class InterpolatedGapFill(tubeFeatures: SimpleFeatureCollection,
215                           bufferDistance: Double,
216                           maxBins: Int) extends TubeBuilder(tubeFeatures, bufferDistance, maxBins) with LazyLogging {
217 
218   val id = new AtomicInteger(0)

220   def nextId: String = id.getAndIncrement.toString
221 
222   override def createTube: Iterator[SimpleFeature] = {
223     import org.locationtech.geomesa.utils.geotools.Conversions.RichGeometry
224 
225     logger.debug("Creating tube with line interpolated line gap fill")
226 
227     val transformed = transform(tubeFeatures, dtgField)
228     val sortedTube = transformed.toSeq.sortBy(sf => getStartTime(sf).getTime)
229     val pointsAndTimes = sortedTube.map(sf => (getGeom(sf).safeCentroid(), getStartTime(sf)))
230     val lineFeatures = if (pointsAndTimes.lengthCompare(1) == 0) {
231       val (p1, t1) = pointsAndTimes.head
232       logger.debug("Only a single result - can't create a line")
233       Iterator(builder.buildFeature(nextId, p1, t1, t1))
234     } else {
235       pointsAndTimes.sliding(2).flatMap { case Seq((p1, t1), (p2, t2)) =>
236         calc.setStartingGeographicPoint(p1.getX, p1.getY)
237         calc.setDestinationGeographicPoint(p2.getX, p2.getY)
238         val dist = calc.getOrthodromicDistance
239         //If the distance between points is greater than the buffer distance, segment the line
240         //So that no segment is larger than the buffer. This ensures that each segment has an
241         //times and distance. Also ensure that features do not share a time value.
242         val timeDiffMillis = t2.getTime - t1.getTime
243         val segCount = (dist / bufferDistance).toInt
244         val segDuration = timeDiffMillis / segCount
245         val timeSteps = NumericRange.inclusive(t1.getTime, t2.getTime, segDuration)
246         if (dist > bufferDistance && timeSteps.lengthCompare(1) > 0) {
247           val heading = calc.getAzimuth
248           var segStep = new Coordinate(p1.getX, p1.getY, 0)
249           timeSteps.sliding(2).map { case Seq(time0, time1) =>
250             val segP1 = segStep
251             calc.setStartingGeographicPoint(segP1.x, segP1.y)
252             calc.setDirection(heading, bufferDistance)
253             val destPoint = calc.getDestinationGeographicPoint
254             segStep = new Coordinate(destPoint.getX, destPoint.getY, 0)
255             val geo = makeIDLSafeLineString(segP1, segStep)
256             builder.buildFeature(nextId, geo, new Date(time0), new Date(time1))
257           }
258         } else {
259           val geo = if (p1.equals(p2)) { p1 } else { makeIDLSafeLineString(p1.getCoordinate, p2.getCoordinate) }
260           logger.debug(s"Created line-filled geometry: ${WKTUtils.write(geo)} " +
261               s"from ${WKTUtils.write(p1)} and ${WKTUtils.write(p2)}")
262           Seq(builder.buildFeature(nextId, geo, t1, t2))
263         }
264       }
265     }
266     buffer(lineFeatures, bufferDistance)
267   }
