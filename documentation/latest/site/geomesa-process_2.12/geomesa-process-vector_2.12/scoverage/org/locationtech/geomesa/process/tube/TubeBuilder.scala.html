34 /**
35  * Build a tube for input to a TubeSelect by buffering and binning the input
36  * tubeFeatures into SimpleFeatures that can be used as inputs to Geomesa queries
37  */
38 abstract class TubeBuilder(val tubeFeatures: SimpleFeatureCollection,
39                            val bufferDistance: Double,
40                            val maxBins: Int) extends LazyLogging {
41 
42   val calc = new GeodeticCalculator()
43   val dtgField: String = tubeFeatures.getSchema.getDtgField.getOrElse(TubeBuilder.DefaultDtgField)
44   val geoFac = new GeometryFactory
45 
46   val GEOM_PROP = "geom"
47 
48   val tubeType: SimpleFeatureType = SimpleFeatureTypes.createType("tubeType", s"$GEOM_PROP:Geometry:srid=4326,start:Date,end:Date")
49   val builder: SimpleFeatureBuilder = ScalaSimpleFeatureFactory.featureBuilder(tubeType)
50 
51   def getGeom(sf: SimpleFeature): Geometry = sf.getAttribute(0).asInstanceOf[Geometry]
52   def getStartTime(sf: SimpleFeature): Date = sf.getAttribute(1).asInstanceOf[Date]
53   def getEndTime(sf: SimpleFeature): Date = sf.getAttribute(2).asInstanceOf[Date]
54 
55   def bufferGeom(geom: Geometry, meters: Double): Geometry = {
56     import org.locationtech.geomesa.utils.geotools.Conversions.RichGeometry
57     geom.buffer(metersToDegrees(meters, geom.safeCentroid()))
58   }
59 
60   def metersToDegrees(meters: Double, point: Point): Double = {
61     logger.debug("Buffering: "+meters.toString + " "+WKTUtils.write(point))
62 
63     calc.setStartingGeographicPoint(point.getX, point.getY)
64     calc.setDirection(0, meters)
65     val dest2D = calc.getDestinationGeographicPoint
66     val destPoint = geoFac.createPoint(new Coordinate(dest2D.getX, dest2D.getY))
67     point.distance(destPoint)
68   }
69 
70   def buffer(simpleFeatures: Iterator[SimpleFeature], meters:Double): Iterator[SimpleFeature] =
71     simpleFeatures.map { sf =>
72       val bufferedGeom = bufferGeom(getGeom(sf), meters)
73       builder.reset()
74       builder.init(sf)
75       builder.set(GEOM_PROP, bufferedGeom)
76       builder.buildFeature(sf.getID)
77     }
78 
79   // transform the input tubeFeatures into the intermediate SF used by the
80   // tubing code consisting of three attributes (geom, startTime, endTime)
81   // handle date parsing from input -> TODO revisit date parsing...
82   def transform(tubeFeatures: SimpleFeatureCollection, dtgField: String): Iterator[SimpleFeature] = {
83     SelfClosingIterator(tubeFeatures.features).map { sf =>
84       val date = FastConverter.convert(sf.getAttribute(dtgField), classOf[Date])
85 
86       if (date == null) {
87         logger.error("Unable to retrieve date field from input tubeFeatures...ensure there a field named " + dtgField)
88         throw new IllegalArgumentException("Unable to retrieve date field from input tubeFeatures...ensure there a field named \"" + dtgField + "\"")
89       }
90 
91       builder.reset()
92       builder.buildFeature(sf.getID, sf.getDefaultGeometry, date, null)
93     }
94   }
95 
96   /**
97     * Return an Array containing either 1 or 2 LineStrings that straddle but
98     * do not cross the IDL.
99     * @param input1 The first point in the segment
100     * @param input2 The second point in the segment
101     * @return an array of LineString containing either 1 or 2 LineStrings that do not
102     *         span the IDL.
103     */
104   def makeIDLSafeLineString(input1:Coordinate, input2:Coordinate): Geometry = {
105     //If the points cross the IDL we must generate two line segments
106     if (GeometryUtils.crossesIDL(input1, input2)) {
107       //Find the latitude where the segment intercepts the IDL
108       val latIntercept = GeometryUtils.calcIDLIntercept(input1, input2)
109       val p1 = new Coordinate(-180, latIntercept)
110       val p2 = new Coordinate(180, latIntercept)
111       //This orders the points so that point1 is always the east-most point
112       val (point1, point2) = if (input1.x > 0) (input1, input2) else (input2, input1)
113       val westLine = new LineString(new CoordinateArraySequence(Array(p1, point2)), geoFac)
114       val eastLine = new LineString(new CoordinateArraySequence(Array(point1, p2)), geoFac)
115       new MultiLineString(Array[LineString](westLine,eastLine), geoFac)
116     } else {
117       new LineString(new CoordinateArraySequence(Array(input1, input2)), geoFac)
118     }
119   }
120 
121   def createTube: Iterator[SimpleFeature]
122 }

125  * Build a tube with no gap filling - only buffering and binning
126  */
127 class NoGapFill(tubeFeatures: SimpleFeatureCollection,
128                 bufferDistance: Double,
129                 maxBins: Int) extends TubeBuilder(tubeFeatures, bufferDistance, maxBins) with LazyLogging {
130 
131   // Bin ordered features into maxBins that retain order by date then union by geometry
132   def timeBinAndUnion(features: Iterable[SimpleFeature], maxBins: Int): Iterator[SimpleFeature] = {
133     val numFeatures = features.size
134 
135     if (numFeatures == 0) { Iterator.empty } else {
136       //If 0 is passed in then don't bin the features, if 1 then make one bin, otherwise calculate number
137       //of bins based on numFeatures and maxBins
138       val binSize = maxBins match {
139         case 0 => 1
140         case 1 => numFeatures
141         case _ => numFeatures / maxBins + (if (numFeatures % maxBins == 0 ) 0 else 1)
142       }
143       features.grouped(binSize).zipWithIndex.map { case(bin, idx) => unionFeatures(bin.toSeq, idx.toString) }
144     }
145   }
146 
147   // Union features to create a single geometry and single combined time range
148   def unionFeatures(orderedFeatures: Seq[SimpleFeature], id: String): SimpleFeature = {
149     import scala.collection.JavaConverters._
150     val geoms = orderedFeatures.map { sf => getGeom(sf) }
151     val unionGeom = geoFac.buildGeometry(geoms.asJava).union
152     val min = getStartTime(orderedFeatures.head)
153     val max = getStartTime(orderedFeatures.last)
154 
155     builder.reset()
156     builder.buildFeature(id, unionGeom, min, max)
157   }
158 
159   override def createTube: Iterator[SimpleFeature] = {
160     logger.debug("Creating tube with no gap filling")
161 
162     val transformed = transform(tubeFeatures, dtgField)
163     val buffered = buffer(transformed, bufferDistance)
164     val sortedTube = buffered.toSeq.sortBy { sf => getStartTime(sf).getTime }
165 
166     logger.debug(s"sorted tube size: ${sortedTube.size}")
167     timeBinAndUnion(sortedTube, maxBins)
168   }
169 }

174 /**
175  * Build a tube with gap filling that draws a line between time-ordered features
176  * from the given tubeFeatures
177  */
178 class LineGapFill(tubeFeatures: SimpleFeatureCollection,
179                   bufferDistance: Double,
180                   maxBins: Int) extends TubeBuilder(tubeFeatures, bufferDistance, maxBins) with LazyLogging {
181 
182   val id = new AtomicInteger(0)
183 
184   def nextId: String = id.getAndIncrement.toString
185 
186   override def createTube: Iterator[SimpleFeature] = {
187     import org.locationtech.geomesa.utils.geotools.Conversions.RichGeometry
188 
189     logger.debug("Creating tube with line gap fill")
190 
191     val transformed = transform(tubeFeatures, dtgField)
192     val sortedTube = transformed.toSeq.sortBy { sf => getStartTime(sf).getTime }
193     val pointsAndTimes = sortedTube.map(sf => (getGeom(sf).safeCentroid(), getStartTime(sf)))
194     val lineFeatures = if (pointsAndTimes.lengthCompare(1) == 0) {
195       val (p1, t1) = pointsAndTimes.head
196       logger.debug("Only a single result - can't create a line")
197       Iterator(builder.buildFeature(nextId, p1, t1, t1))
198     } else {
199       pointsAndTimes.sliding(2).map { case Seq((p1, t1), (p2, t2)) =>
200         val geo = if (p1.equals(p2)) p1 else makeIDLSafeLineString(p1.getCoordinate,p2.getCoordinate)
201         logger.debug(s"Created Line-filled Geometry: ${WKTUtils.write(geo)} From ${WKTUtils.write(p1)} and ${WKTUtils.write(p2)}")
202         builder.buildFeature(nextId, geo, t1, t2)
203       }
204     }
205     buffer(lineFeatures, bufferDistance)
206   }
207 }

223   override def createTube: Iterator[SimpleFeature] = {
224     import org.locationtech.geomesa.utils.geotools.Conversions.RichGeometry
225 
226     logger.debug("Creating tube with line interpolated line gap fill")
227 
228     val transformed = transform(tubeFeatures, dtgField)
229     val sortedTube = transformed.toSeq.sortBy(sf => getStartTime(sf).getTime)
230     val pointsAndTimes = sortedTube.map(sf => (getGeom(sf).safeCentroid(), getStartTime(sf)))
231     val lineFeatures = if (pointsAndTimes.lengthCompare(1) == 0) {
232       val (p1, t1) = pointsAndTimes.head
233       logger.debug("Only a single result - can't create a line")
234       Iterator(builder.buildFeature(nextId, p1, t1, t1))
235     } else {
236       pointsAndTimes.sliding(2).flatMap { case Seq((p1, t1), (p2, t2)) =>
237         calc.setStartingGeographicPoint(p1.getX, p1.getY)
238         calc.setDestinationGeographicPoint(p2.getX, p2.getY)
239         val dist = calc.getOrthodromicDistance
240         //If the distance between points is greater than the buffer distance, segment the line
241         //So that no segment is larger than the buffer. This ensures that each segment has an
242         //times and distance. Also ensure that features do not share a time value.
243         val timeDiffMillis = toInstant(t2).toEpochMilli - toInstant(t1).toEpochMilli
244         val segCount = (dist / bufferDistance).toInt
245         val segDuration = timeDiffMillis / segCount
246         val timeSteps = NumericRange.inclusive(toInstant(t1).toEpochMilli, toInstant(t2).toEpochMilli, segDuration)
247         if (dist > bufferDistance && timeSteps.lengthCompare(1) > 0) {
248           val heading = calc.getAzimuth
249           var segStep = new Coordinate(p1.getX, p1.getY, 0)
250           timeSteps.sliding(2).map { case Seq(time0, time1) =>
251             val segP1 = segStep
252             calc.setStartingGeographicPoint(segP1.x, segP1.y)
253             calc.setDirection(heading, bufferDistance)
254             val destPoint = calc.getDestinationGeographicPoint
255             segStep = new Coordinate(destPoint.getX, destPoint.getY, 0)
256             val geo = makeIDLSafeLineString(segP1, segStep)
257             builder.buildFeature(nextId, geo, new Date(time0), new Date(time1))
258           }
259         } else {
260           val geo = if (p1.equals(p2)) { p1 } else { makeIDLSafeLineString(p1.getCoordinate, p2.getCoordinate) }
261           logger.debug(s"Created line-filled geometry: ${WKTUtils.write(geo)} " +
262               s"from ${WKTUtils.write(p1)} and ${WKTUtils.write(p2)}")
263           Seq(builder.buildFeature(nextId, geo, t1, t2))
264         }
265       }
266     }
267     buffer(lineFeatures, bufferDistance)
268   }
