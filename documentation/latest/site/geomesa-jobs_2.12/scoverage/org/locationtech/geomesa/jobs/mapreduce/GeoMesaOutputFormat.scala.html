29 /**
30   * Output format that writes simple features using GeoMesaDataStore's FeatureWriterAppend. Can write only
31   * specific indices if desired
32   */
33 class GeoMesaOutputFormat extends OutputFormat[Text, SimpleFeature] {
34 
35   import scala.collection.JavaConverters._
36 
37   override def getRecordWriter(context: TaskAttemptContext): RecordWriter[Text, SimpleFeature] = {
38     val params = GeoMesaConfigurator.getDataStoreOutParams(context.getConfiguration)
39     val indices = GeoMesaConfigurator.getIndicesOut(context.getConfiguration)
40     new GeoMesaRecordWriter(params, indices, context)
41   }
42 
43   override def checkOutputSpecs(context: JobContext): Unit = {
44     val params =
45       GeoMesaConfigurator.getDataStoreOutParams(context.getConfiguration)
46         .asJava.asInstanceOf[java.util.Map[String, java.io.Serializable]]
47     if (!DataStoreFinder.getAvailableDataStores.asScala.exists(_.canProcess(params))) {
48       throw new IOException("Data store connection parameters are not set")
49     }
50   }
51 
52   override def getOutputCommitter(context: TaskAttemptContext): OutputCommitter =
53     new NullOutputFormat[Text, SimpleFeature]().getOutputCommitter(context)
54 }

59 
60   object OutputCounters {
61     val Group   = "org.locationtech.geomesa.jobs.output"

66   /**
67     * Configure the data store you will be writing to
68     *
69     * @param conf conf
70     * @param params data store parameters
71     * @param sft simple feature type to write, must exist already in the store
72     * @param indices indices to write, or all indices
73     */
74   def setOutput(
75       conf: Configuration,
76       params: Map[String, String],
77       sft: SimpleFeatureType,
78       indices: Option[Seq[String]] = None): Unit = {
79     GeoMesaConfigurator.setDataStoreOutParams(conf, params)
80     GeoMesaConfigurator.setSerialization(conf, sft)
81     indices.foreach(GeoMesaConfigurator.setIndicesOut(conf, _))
82   }
83 
84   /**
85    * Helper for java interop
86    *
87    * @param conf conf
88    * @param params data store parameters
89    * @param sft simple feature type to write, must exist already in the store
90    */
91   def setOutputJava(
92       conf: Configuration,
93       params: java.util.Map[String, String],
94       sft: SimpleFeatureType): Unit = {
95     setOutput(conf, params.asScala.toMap, sft)
96   }
97 
98   /**
99     * Record writer for GeoMesa datastores.
100     *
101     * All feature types must exist already in the datastore. The input key is ignored.
102     */
103   class GeoMesaRecordWriter(params: Map[String, String], indices: Option[Seq[String]], context: TaskAttemptContext)
104       extends RecordWriter[Text, SimpleFeature] with LazyLogging {
105 
106     private val ds = DataStoreFinder.getDataStore(params.asJava)
107 
108     private val writers = ArrayBuffer.empty[FeatureWriter[SimpleFeatureType, SimpleFeature]]
109     private val helpers = scala.collection.mutable.Map.empty[String, FeatureWriterHelper]
110 
111     private val written = context.getCounter(OutputCounters.Group, OutputCounters.Written)
112     private val failed = context.getCounter(OutputCounters.Group, OutputCounters.Failed)
113 
114     override def write(key: Text, value: SimpleFeature): Unit = {
115       try {
116         val sftName = value.getFeatureType.getTypeName
117         val helper = helpers.getOrElseUpdate(sftName, createWriter(sftName))
118         helper.write(value)
119         written.increment(1)
120       } catch {
121         case e: Exception =>
122           logger.error(s"Error writing feature '${DataUtilities.encodeFeature(value)}'", e)
123           failed.increment(1)
124       }
125     }
126 
127     private def createWriter(typeName: String): FeatureWriterHelper = {
128       val writer = ds match {
129         case gm: GeoMesaDataStore[_] =>
130           val sft = gm.getSchema(typeName)
131           val i = indices match {
132             case Some(names) => names.map(gm.manager.index(sft, _, IndexMode.Write))
133             case None => gm.manager.indices(sft, mode = IndexMode.Write)
134           }
135           gm.getIndexWriterAppend(typeName, i)
136 
137         case _ =>
138           indices.foreach { i =>
139             logger.warn(s"Ignoring index config '${i.mkString(",")}' for non-geomesa data store $ds")
140           }
141           ds.getFeatureWriterAppend(typeName, Transaction.AUTO_COMMIT)
142       }
143       writers += writer
144       FeatureWriterHelper(writer)
145     }
146 
147     override def close(context: TaskAttemptContext): Unit = {
148       CloseQuietly(writers)
149       ds.dispose()
150     }
151   }
