60   /**
61     * Attempts to do an optimized join between two relations.
62     *
63     * Currently this method uses grid partitioning on both relations so that the join comparisons
64     * only need to be applied on each pair of partitions, instead of globally. This only works
65     * if both relations have already been grid partitioned.
66     *
67     * @param other relation to join
68     * @param condition join condition
69     * @return an optimized join, if possible to do so
70     */
71   def join(other: GeoMesaRelation, condition: Expression): Option[GeoMesaJoinRelation] = {
72     val opt = for { p <- partitioned; o <- other.partitioned } yield {
73       val toJoin = if (p.envelopes == o.envelopes) {
74         Some(other)
75       } else if (p.cover) {
76         val repartitioned: SpatialRDD = p.partitions match {
77           case None => p.raw
78           case Some(partitions) => SpatialRDD(p.raw.repartition(partitions), p.raw.schema)
79         }
80         val parallelism = p.partitions.getOrElse(sqlContext.sparkContext.defaultParallelism)
81         val rdd = RelationUtils.grid(repartitioned, p.envelopes, parallelism)
82         val partitioned = Some(PartitionedRDD(rdd, p.raw, p.envelopes, p.partitions, p.cover))
83         Some(other.copy(partitioned = partitioned))
84       } else {
85         logger.warn("Joining across two relations that are not partitioned by the same scheme - unable to optimize")
86         None
87       }
88       toJoin.map { rel =>
89         GeoMesaJoinRelation(sqlContext, this, rel, StructType(schema.fields ++ rel.schema.fields), condition)
90       }
91     }
92     opt.flatten
93   }

95   override def buildScan(
96       requiredColumns: Array[String],
97       filters: Array[org.apache.spark.sql.sources.Filter]): RDD[Row] = {
98     lazy val debug =
99       s"filt = $filter, filters = ${filters.mkString(",")}, requiredColumns = ${requiredColumns.mkString(",")}"
100 
101     val filt = {
102       val sum = Seq.newBuilder[org.geotools.api.filter.Filter]
103       filter.foreach(sum += _)
104       filters.foreach(f => SparkUtils.sparkFilterToCQLFilter(f).foreach(sum += _))
105       FilterHelper.filterListAsAnd(sum.result).getOrElse(org.geotools.api.filter.Filter.INCLUDE)
106     }
107     val requiredAttributes = requiredColumns.filterNot(_ == "__fid__")
108 
109     // avoid closures on complex objects
110     val schema = this.schema // note: referencing case class members evidently serializes the whole class??
111     val typeName = sft.getTypeName
112 
113     val result: RDD[SimpleFeature] = cached match {
114       case None =>
115         logger.debug(s"Building scan, $debug")
116         val conf = new Configuration(sqlContext.sparkContext.hadoopConfiguration)
117         val query = new Query(typeName, filt, requiredAttributes: _*)
118         GeoMesaSpark(params.asJava).rdd(conf, sqlContext.sparkContext, params, query)
119 
120       case Some(IndexedRDD(rdd)) =>
121         logger.debug(s"Building in-memory scan, $debug")
122         val cql = ECQL.toCQL(filt)
123         rdd.flatMap { engine =>
124           val query = new Query(typeName, ECQL.toFilter(cql), requiredAttributes: _*)
125           SelfClosingIterator(engine.getFeatureReader(query, Transaction.AUTO_COMMIT))
126         }
127 
128       case Some(PartitionedIndexedRDD(rdd, _)) =>
129         logger.debug(s"Building partitioned in-memory scan, $debug")
130         val cql = ECQL.toCQL(filt)
131         rdd.flatMap { case (_, engine) =>
132           val query = new Query(typeName, ECQL.toFilter(cql), requiredAttributes: _*)
133           SelfClosingIterator(engine.getFeatureReader(query, Transaction.AUTO_COMMIT))
134         }
135     }
136 
137     val extractors = SparkUtils.getExtractors(requiredColumns, schema)
138 
139     result.map(SparkUtils.sf2row(schema, _, extractors))
140   }

198       params: Map[String, String],

201 
202     logger.trace(s"Creating GeoMesaRelation with sft: $sft")
203 
204     def get[T](key: String, transform: String => T, default: => T): T = {
205       params.get(key) match {
206         case None => default
207         case Some(v) =>
208           try { transform(v) } catch {
209             case NonFatal(e) => logger.error(s"Error evaluating param '$key' with value '$v':", e); default
210           }
211       }
212     }
213 
214     def rawRDD: SpatialRDD = {
215       val query = new Query(sft.getTypeName, ECQL.toFilter(params.getOrElse("query", "INCLUDE")))
216       GeoMesaSpark(params.asJava).rdd(new Configuration(), sqlContext.sparkContext, params, query)
217     }
218 
219     val partitioned = if (!get[Boolean]("spatial", _.toBoolean, false)) { None } else {
220       val raw = rawRDD
221       val bounds: Envelope = params.get("bounds") match {
222         case None => RelationUtils.getBound(raw)
223         case Some(b) =>
224           try { WKTUtils.read(b).getEnvelopeInternal } catch {
225             case NonFatal(e) => throw new IllegalArgumentException(s"Error reading provided bounds '$b':", e)
226           }
227       }
228 
229       val partitions = Option(get[Int]("partitions", _.toInt, -1)).filter(_ > 0)
230       val parallelism = partitions.getOrElse(sqlContext.sparkContext.defaultParallelism)
231       // control partitioning strategies that require a sample of the data
232       lazy val sampleSize = get[Int]("sampleSize", _.toInt, 100)
233       lazy val threshold = get[Double]("threshold", _.toDouble, 0.3)
234 
235       val envelopes = params.getOrElse("strategy", "equal").toLowerCase(Locale.US) match {
236         case "equal"    => RelationUtils.equalPartitioning(bounds, parallelism)
237         case "earth"    => RelationUtils.wholeEarthPartitioning(parallelism)
238         case "weighted" => RelationUtils.weightedPartitioning(raw, bounds, parallelism, sampleSize)
239         case "rtree"    => RelationUtils.rtreePartitioning(raw, parallelism, sampleSize, threshold)

241       }
242 
243       val rdd = RelationUtils.grid(raw, envelopes, parallelism)
244       rdd.persist(StorageLevel.MEMORY_ONLY)
245       Some(PartitionedRDD(rdd, raw, envelopes, partitions, get[Boolean]("cover", _.toBoolean, false)))
246     }
247 
248     val cached = if (!get[Boolean]("cache", _.toBoolean, false)) { None } else {
249       val check = Collections.singletonMap[String, java.io.Serializable]("cqengine", "true")
250       if (!DataStoreFinder.getAvailableDataStores.asScala.exists(_.canProcess(check))) {
251         throw new IllegalArgumentException("Caching requires the GeoCQEngineDataStore to be available on the classpath")
252       }
253 
254       // avoid closure on full sft
255       val typeName = sft.getTypeName
256       val encodedSft = SimpleFeatureTypes.encodeType(sft, includeUserData = true)
257 
258       val indexGeom = get[Boolean]("indexGeom", _.toBoolean, false)
259 
260       partitioned match {
261         case Some(p) =>
262           val rdd = p.rdd.mapValues { iter =>
263             val engine = new GeoCQEngineDataStore(indexGeom)
264             engine.createSchema(SimpleFeatureTypes.createType(typeName, encodedSft))
265             WithClose(engine.getFeatureWriterAppend(typeName, Transaction.AUTO_COMMIT)) { writer =>
266               iter.foreach(FeatureUtils.write(writer, _, useProvidedFid = true))
267             }
268             engine
269           }
270           p.rdd.unpersist() // make this call blocking?
271           rdd.persist(StorageLevel.MEMORY_ONLY)
272           Some(PartitionedIndexedRDD(rdd, p.envelopes))
273 
274         case None =>
275           val rdd = rawRDD.mapPartitions { iter =>
276             val engine = new GeoCQEngineDataStore(indexGeom)
277             engine.createSchema(SimpleFeatureTypes.createType(typeName, encodedSft))
278             WithClose(engine.getFeatureWriterAppend(typeName, Transaction.AUTO_COMMIT)) { writer =>
279               iter.foreach(FeatureUtils.write(writer, _, useProvidedFid = true))
280             }
281             Iterator.single(engine)
282           }
283           rdd.persist(StorageLevel.MEMORY_ONLY)
284           Some(IndexedRDD(rdd))
285       }
286     }
287    
288    val filter = Option(ECQL.toFilter(params.getOrElse("query", "INCLUDE")))
289 
290     GeoMesaRelation(sqlContext, sft, schema, params, filter, cached, partitioned)
291   }
