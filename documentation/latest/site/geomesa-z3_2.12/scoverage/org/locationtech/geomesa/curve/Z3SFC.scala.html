16 /**
17   * Z3 space filling curve
18   *
19   * @param dims curve dimensions
20   */
21 class Z3SFC(dims: Z3Dimensions) extends SpaceTimeFillingCurve {
22 
23   val lon: NormalizedDimension  = dims.lon
24   val lat: NormalizedDimension  = dims.lat
25   val time: NormalizedDimension = dims.time
26 
27   val wholePeriod: Seq[(Long, Long)] = Seq((time.min.toLong, time.max.toLong))
28 
29   /**
30    * Alternate constructor
31    *
32    * @param period    time period used to bin results
33    * @param precision bits used per dimension - note all precisions must sum to less than 64
34    */
35   def this(period: TimePeriod, precision: Int = 21) = this(StandardZ3Dimensions(period, precision))
36 
37   override def index(x: Double, y: Double, t: Long, lenient: Boolean = false): Long = {
38     try {
39       require(x >= lon.min && x <= lon.max && y >= lat.min && y <= lat.max && t >= time.min && t <= time.max,
40         s"Value(s) out of bounds ([${lon.min},${lon.max}], [${lat.min},${lat.max}], [${time.min},${time.max}]): $x, $y, $t")
41       Z3(lon.normalize(x), lat.normalize(y), time.normalize(t)).z
42     } catch {
43       case _: IllegalArgumentException if lenient => lenientIndex(x, y, t)
44     }
45   }
46 
47   protected def lenientIndex(x: Double, y: Double, t: Long): Long = {
48     val bx = if (x < lon.min) { lon.min } else if (x > lon.max) { lon.max } else { x }
49     val by = if (y < lat.min) { lat.min } else if (y > lat.max) { lat.max } else { y }
50     val bt = if (t < time.min) { time.min } else if (t > time.max) { time.max } else { t }
51     Z3(lon.normalize(bx), lat.normalize(by), time.normalize(bt)).z
52   }
53 
54   override def invert(z: Long): (Double, Double, Long) = {
55     val (x, y, t) = Z3(z).decode
56     (lon.denormalize(x), lat.denormalize(y), time.denormalize(t).toLong)
57   }
58 
59   override def ranges(xy: Seq[(Double, Double, Double, Double)],
60                       t: Seq[(Long, Long)],
61                       precision: Int,
62                       maxRanges: Option[Int]): Seq[IndexRange] = {
63     val zbounds = for { (xmin, ymin, xmax, ymax) <- xy ; (tmin, tmax) <- t } yield {
64       ZRange(index(xmin, ymin, tmin), index(xmax, ymax, tmax))
65     }
66     Z3.zranges(zbounds.toArray, precision, maxRanges, Z3SFC.MaxRecursion)
67   }
68 }

80     case TimePeriod.Day   => SfcDay
81     case TimePeriod.Week  => SfcWeek

84   }

90   }

98     override val time: NormalizedDimension = NormalizedTime(precision, BinnedTime.maxOffset(period).toDouble)
99   }
