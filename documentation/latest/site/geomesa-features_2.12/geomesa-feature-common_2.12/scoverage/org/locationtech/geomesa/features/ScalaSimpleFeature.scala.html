75   /**

77     * feature type must have a compatible schema.

79     * This is a shallow copy, in that the attributes and user data values will be shared between the two features

85   def copy(sft: SimpleFeatureType, in: SimpleFeature): ScalaSimpleFeature =

88   /**
89     * Copy the feature with a new feature type. Attributes will be copied by name, and converted
90     * as necessary. As compared to `copy`, the new feature type does not have to have a compatible schema.
91     *
92     * If the feature already has the desired feature type, it will be returned as-is and not copied.
93     *
94     * This is a shallow copy, in that the attributes and user data values will be shared between the two features
95     *
96     * @param sft new feature type
97     * @param in feature to copy
98     * @return
99     */
100   def retype(sft: SimpleFeatureType, in: SimpleFeature): SimpleFeature = {
101     if (sft == in.getFeatureType) { in } else {
102       val out = new ScalaSimpleFeature(sft, in.getID)
103       var i = 0
104       sft.getAttributeDescriptors.asScala.foreach { d =>
105         out.setAttribute(i, in.getAttribute(d.getLocalName))
106         i += 1
107       }
108       out.getUserData.putAll(in.getUserData)
109       out
110     }
111   }

113   /**
114     * Creates a simple feature, converting the values to the appropriate type
115     *
116     * @param sft simple feature type
117     * @param id feature id
118     * @param values attributes values, corresponding to the feature type. types will be converted as necessary
119     * @return
120     */
121   def create(sft: SimpleFeatureType, id: String, values: Any*): ScalaSimpleFeature = {
122     val sf = new ScalaSimpleFeature(sft, id)
123     var i = 0
124     while (i < values.length) {
125       sf.setAttribute(i, values(i).asInstanceOf[AnyRef])
126       i += 1
127     }
128     sf
129   }
