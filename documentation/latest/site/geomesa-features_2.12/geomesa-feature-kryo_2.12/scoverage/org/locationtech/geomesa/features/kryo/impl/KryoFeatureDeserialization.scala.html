48 object KryoFeatureDeserialization extends LazyLogging {
49 
50   private val inputBytes   = new SoftThreadLocal[Input]()
51   private val inputStreams = new SoftThreadLocal[Input]()
52   private val readers      = new ThreadLocalCache[String, Array[KryoAttributeReader]](SerializerCacheExpiry)
53 
54   def getInput(bytes: Array[Byte], offset: Int, count: Int): Input = {
55     val in = inputBytes.getOrElseUpdate(new NonMutatingInput())
56     in.setBuffer(bytes, offset, count)
57     in
58   }
59 
60   def getInput(stream: InputStream): Input = {
61     val in = inputStreams.getOrElseUpdate(new NonMutatingInput(Array.ofDim(1024)))
62     in.setInputStream(stream)
63     in
64   }
65 
66   def getReaders(key: String, sft: SimpleFeatureType): Array[KryoAttributeReader] = {
67     readers.getOrElseUpdate(key, sft.getAttributeDescriptors.toArray.map {
68       case ad: AttributeDescriptor => reader(ObjectType.selectType(ad))
69     })
70   }
71 
72   private [geomesa] def reader(bindings: Seq[ObjectType]): KryoAttributeReader = {
73     bindings.head match {
74       case ObjectType.STRING   => if (bindings.last == ObjectType.JSON) { KryoJsonReader } else { KryoStringReader }
75       case ObjectType.INT      => KryoIntReader
76       case ObjectType.LONG     => KryoLongReader
77       case ObjectType.FLOAT    => KryoFloatReader
78       case ObjectType.DOUBLE   => KryoDoubleReader
79       case ObjectType.DATE     => KryoDateReader
80       case ObjectType.GEOMETRY => KryoGeometryReader
81       case ObjectType.BYTES    => KryoBytesReader
82       case ObjectType.UUID     => KryoUuidReader
83       case ObjectType.BOOLEAN  => KryoBooleanReader
84       case ObjectType.LIST     => KryoListReader(reader(bindings.drop(1)))
85       case ObjectType.MAP      => KryoMapReader(reader(bindings.slice(1, 2)), reader(bindings.drop(2)))
86 
87       case b => throw new UnsupportedOperationException(s"Unexpected attribute type binding: $b")
88     }
89   }

161   case class KryoListReader(elements: KryoAttributeReader) extends KryoAttributeReader {
162     override def apply(input: Input): java.util.List[AnyRef] = try {
163       val size = input.readInt(true)
164       val list = new java.util.ArrayList[AnyRef](size)
165       var index = 0
166       while (index < size) {
167         list.add(elements.apply(input))
168         index += 1
169       }
170       list
171     } catch {
172       case NonFatal(e) => logger.error("Error reading serialized kryo bytes:", e); null
173     }
174   }
175 
176   case class KryoMapReader(keys: KryoAttributeReader, values: KryoAttributeReader) extends KryoAttributeReader {
177     override def apply(input: Input): java.util.Map[AnyRef, AnyRef] = try {
178       val size = input.readInt(true)
179       val map = new java.util.HashMap[AnyRef, AnyRef](size)
180       var index = 0
181       while (index < size) {
182         map.put(keys.apply(input), values.apply(input))
183         index += 1
184       }
185       map
186     } catch {
187       case NonFatal(e) => logger.error("Error reading serialized kryo bytes:", e); null
188     }
189   }
