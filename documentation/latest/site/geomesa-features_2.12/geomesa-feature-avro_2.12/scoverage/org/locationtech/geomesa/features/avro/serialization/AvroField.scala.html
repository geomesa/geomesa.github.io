27 /**
28  * Trait for reading an avro field, corresponding to a simple feature attribute
29  *
30  * @tparam T type of attribute
31  */
32 trait AvroField[T] {
33 
34   /**
35    * Schema of the field
36    *
37    * @return
38    */
39   def schema: Schema
40 
41   /**
42    * Read the field value
43    *
44    * @param in input
45    * @return
46    */
47   def read(in: Decoder): T = {
48     if (in.readIndex() == 1) {
49       in.readNull()
50       null.asInstanceOf[T]
51     } else {
52       readNonNull(in)
53     }
54   }
55 
56   /**
57    * Skip over the field value - may be more efficient than reading and discarding it
58    *
59    * @param in input
60    */
61   def skip(in: Decoder): Unit = {
62     if (in.readIndex() == 1) {
63       in.readNull()
64     } else {
65       skipNonNull(in)
66     }
67   }
68 
69   /**
70    * Write the field value
71    *
72    * @param out output
73    * @param value value to write
74    */
75   def write(out: Encoder, value: T): Unit = {
76     if (value == null) {
77       out.writeIndex(1)
78       out.writeNull()
79     } else {
80       out.writeIndex(0)
81       writeNonNull(out, value)
82     }
83   }
84 
85   /**
86    * Gets a version-specific instance of this field.
87    * See org.locationtech.geomesa.features.avro.SerializationVersions
88    *
89    * @param version version
90    * @return
91    */
92   def withVersion(version: Int): AvroField[T] = this
93 
94   protected def readNonNull(in: Decoder): T
95   protected def skipNonNull(in: Decoder): Unit
96   protected def writeNonNull(out: Encoder, value: T): Unit
97 }

114   private def apply(types: Seq[ObjectType]): AvroField[AnyRef] = {
115     val field = types.head match {
116       case ObjectType.STRING   => StringField
117       case ObjectType.INT      => IntField
118       case ObjectType.LONG     => LongField
119       case ObjectType.FLOAT    => FloatField
120       case ObjectType.DOUBLE   => DoubleField
121       case ObjectType.BOOLEAN  => BooleanField
122       case ObjectType.DATE     => DateField
123       case ObjectType.UUID     => UuidBinaryField
124       case ObjectType.GEOMETRY => GeometryField
125       case ObjectType.LIST     => new ListOpaqueField(types(1))
126       case ObjectType.MAP      => new MapOpaqueField(types(1), types(2))
127       case ObjectType.BYTES    => BytesField
128       case t => throw new IllegalStateException(s"Unexpected descriptor type: $t")
129     }
130     field.asInstanceOf[AvroField[AnyRef]]
131   }

215   case object StringField extends AvroField[String] {
216     override val schema: Schema = SchemaBuilder.nullable().stringType()
217     override protected def readNonNull(in: Decoder): String = {
218       // note: we don't use the string reading methods, as internal avro state can get corrupted and cause
219       // exceptions in BinaryDecoder.scratchUtf8
220       var (bb, bytes) = buffers.getOrElseUpdate((ByteBuffer.allocate(16), Array.empty))
221       bb = in.readBytes(bb)
222       val length = bb.remaining
223       if (bytes.length < length) {
224         bytes = Array.ofDim(length)
225       }
226       buffers.put((bb, bytes))
227       bb.get(bytes, 0, length)
228       new String(bytes, 0, length, StandardCharsets.UTF_8)
229     }
230     override protected def skipNonNull(in: Decoder): Unit = in.skipBytes()
231     override protected def writeNonNull(out: Encoder, value: String): Unit = out.writeString(value)
232   }

298   case object UuidBinaryField extends AvroField[UUID] {
299 
300     override val schema: Schema = SchemaBuilder.nullable().bytesType()
301 
302     def decode(buf: ByteBuffer): UUID = new UUID(buf.getLong, buf.getLong)
303 
304     override protected def readNonNull(in: Decoder): UUID = {
305       var (bb, bytes) = buffers.getOrElseUpdate((ByteBuffer.allocate(16), Array.empty))
306       bb = in.readBytes(bb)
307       buffers.put((bb, bytes))
308       decode(bb)
309     }
310 
311     override protected def skipNonNull(in: Decoder): Unit = in.skipBytes()
312 
313     override protected def writeNonNull(out: Encoder, value: UUID): Unit = {
314       val buf =
315         ByteBuffer.allocate(16)
316           .putLong(value.getMostSignificantBits)
317           .putLong(value.getLeastSignificantBits)
318           .flip.asInstanceOf[ByteBuffer]
319       out.writeBytes(buf)
320     }
321 
322     override def withVersion(version: Int): AvroField[UUID] =
323       if (version == SerializationVersions.NativeCollectionVersion) { UuidRecordField } else { this }
324   }

465   /**
466    * List-type field, stored as a native avro array
467    *
468    * @param items type of list items
469    */
470   class ListNativeField(items: ObjectType) extends AvroField[java.util.List[AnyRef]] {
471 
472     import scala.collection.JavaConverters._
473 
474     private val field = AvroField.apply(Seq(items))
475 
476     override val schema: Schema = SchemaBuilder.nullable().array().items(field.schema)
477 
478     override protected def readNonNull(in: Decoder): java.util.List[AnyRef] = {
479       var i = in.readArrayStart()
480       val result = new java.util.ArrayList[AnyRef](i.toInt)
481       while (i != 0) {
482         var j = 0
483         while (j < i) {
484           result.add(field.read(in))
485           j += 1
486         }
487         i = in.arrayNext()
488       }
489       result
490     }
491 
492     override protected def skipNonNull(in: Decoder): Unit = {
493       var i = in.skipArray()
494       while (i != 0) {
495         var j = 0
496         while (j < i) {
497           field.skip(in)
498           j += 1
499         }
500         i = in.arrayNext()
501       }
502     }
503 
504     override protected def writeNonNull(out: Encoder, value: java.util.List[AnyRef]): Unit = {
505       out.writeArrayStart()
506       out.setItemCount(value.size)
507       value.asScala.foreach { v =>
508         out.startItem()
509         field.write(out, v)
510       }
511       out.writeArrayEnd()
512     }
513 
514     override def withVersion(version: Int): AvroField[java.util.List[AnyRef]] =
515       if (version == SerializationVersions.NativeCollectionVersion) { this } else { new ListOpaqueField(items) }
516   }

606         }

623       if (version == SerializationVersions.NativeCollectionVersion) { this } else {
624         new MapOpaqueField(keys, values)

634   class MapNativeField(values: ObjectType) extends AvroField[java.util.Map[AnyRef, AnyRef]]{

642     override protected def readNonNull(in: Decoder): java.util.Map[AnyRef, AnyRef] = {

644       val result = new java.util.HashMap[AnyRef, AnyRef](i.toInt)
