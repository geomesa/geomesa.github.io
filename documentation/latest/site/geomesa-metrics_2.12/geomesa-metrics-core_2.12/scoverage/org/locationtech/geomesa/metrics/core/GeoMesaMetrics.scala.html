21 /**
22   * Provides namespaced access to reporting metrics
23   *
24   * @param registry metric registry
25   * @param prefix namespace prefix
26   * @param reporters metric reporters
27   */
28 class GeoMesaMetrics(val registry: MetricRegistry, prefix: String, reporters: Seq[ScheduledReporter])
29     extends Closeable {
30 
31   private val pre = GeoMesaMetrics.safePrefix(prefix)
32 
33   private val micrometerRegistry = if (reporters.isEmpty) { None } else {
34     // register a dropwizard meter registry so that our metrics get propagated here
35     // note: this doesn't expose any metrics created here to micrometer, however
36     val config = new DropwizardConfig() {
37       override def prefix(): String = pre
38       override def get(key: String): String = null
39     }
40     val dwRegistry = new DropwizardMeterRegistry(config, registry, HierarchicalNameMapper.DEFAULT, Clock.SYSTEM) {
41       override def nullGaugeValue(): java.lang.Double = null
42       override def close(): Unit = try { Metrics.removeRegistry(this) } finally { super.close() }
43     }
44     Metrics.addRegistry(dwRegistry)
45     Some(dwRegistry)
46   }
47 
48   protected def id(typeName: String, id: String): String = s"$pre${GeoMesaMetrics.safePrefix(typeName)}$id"
49 
50   /**
51    * Creates a prefixed counter
52    *
53    * @param typeName simple feature type name
54    * @param id short identifier for the metric being counted
55    * @return
56    */
57   def counter(typeName: String, id: String): Counter = registry.counter(this.id(typeName, id))
58 
59   /**
60    * Gets a gauge. Note that it is possible (although unlikely) that the gauge will not be the
61    * one from the supplier, if the id has already been registered
62    *
63    * @param typeName simple feature type name
64    * @param id short identifier for the metric being gauged
65    * @param supplier metric supplier
66    * @return
67    */
68   def gauge(typeName: String, id: String, metric: => Gauge[_]): Gauge[_] = {
69     val ident = this.id(typeName, id)
70     // note: don't use MetricRegistry#gauge(String, MetricSupplier<Gauge>) to support older
71     // metric jars that ship with hbase
72     def getOrCreate(): Gauge[_] = {
73       registry.getMetrics.get(ident) match {
74         case g: Gauge[_] => g
75         case null => registry.register(ident, metric)
76         case m =>
77           throw new IllegalArgumentException(s"${m.getClass.getSimpleName} already registered under the name '$ident'")
78       }
79     }
80 
81     // re-try once to avoid concurrency issues with checking then adding a metric (which should be rare)
82     try { getOrCreate() } catch { case _: IllegalArgumentException => getOrCreate() }
83   }
84 
85   /**
86    * Creates a prefixed histogram
87    *
88    * @param typeName simple feature type name
89    * @param id short identifier for the metric being histogramed
90    * @return
91    */
92   def histogram(typeName: String, id: String): Histogram = registry.histogram(this.id(typeName, id))
93 
94   /**
95    * Creates a prefixed meter
96    *
97    * @param typeName simple feature type name
98    * @param id short identifier for the metric being metered
99    * @return
100    */
101   def meter(typeName: String, id: String): Meter = registry.meter(this.id(typeName, id))
102 
103   /**
104    * Creates a prefixed timer
105    *
106    * @param typeName simple feature type name
107    * @param id short identifier for the metric being timed
108    * @return
109    */
110   def timer(typeName: String, id: String): Timer = registry.timer(this.id(typeName, id))
111 
112   /**
113    * Register a metric. Note that in comparison to most methods in this class, a given identifier
114    * can only be registered once
115    *
116    * @param typeName simple feature type name
117    * @param id short identifier for the metric
118    * @param metric metric to register
119    * @tparam T metric type
120    * @return
121    */
122   def register[T <: Metric](typeName: String, id: String, metric: T): T =
123     registry.register(this.id(typeName, id), metric)
124 
125   override def close(): Unit = {
126     CloseWithLogging(reporters)
127     Try(micrometerRegistry.foreach(_.close()))
128   }
129 }

131 object GeoMesaMetrics {
132 
133   /**
134     * Create a registry
135     *
136     * @param prefix metric name prefix
137     * @param reporters configs for metric reporters
138     * @return
139     */
140   def apply(prefix: String, reporters: Seq[Config]): GeoMesaMetrics = {
141     val registry = new MetricRegistry()
142     val reps = reporters.map(ReporterFactory.apply(_, registry)).toList
143     new GeoMesaMetrics(registry, prefix, reps)
144   }
145 
146   private def safePrefix(name: String): String = {
147     val replaced = name.replaceAll("[^A-Za-z0-9]", ".")
148     if (replaced.isEmpty || replaced.endsWith(".")) { replaced } else { s"$replaced." }
149   }
