38   private val jsonPath: TransformerFunction = new NamedTransformerFunction(Array("jsonPath"), pure = true) {
39     private val cache = new ConcurrentHashMap[Any, JsonPath]()
40     override def apply(args: Array[AnyRef]): AnyRef = {
41       var path = cache.get(args(0))
42       if (path == null) {
43         path = JsonPath.compile(args(0).asInstanceOf[String])
44         cache.put(args(0), path)
45       }
46       val elem = path.read[JsonElement](args(1), JsonConverter.JsonConfiguration)
47       // unwrap primitive and null elements
48       if (elem.isJsonNull) {
49         null
50       } else if (elem.isJsonPrimitive) {
51         val p = elem.getAsJsonPrimitive
52         if (p.isString) {
53           p.getAsString
54         } else if (p.isNumber) {
55           p.getAsNumber
56         } else if (p.isBoolean) {
57           Boolean.box(p.getAsBoolean)
58         } else {
59           p.getAsString // this shouldn't really ever happen...
60         }
61       } else {
62         elem
63       }
64     }
65   }

131   private val newJsonObject = TransformerFunction.pure("newJsonObject") { args =>
132     val obj = new JsonObject()
133     var i = 1
134     while (i < args.length) {
135       val key = args(i -1).toString
136       args(i) match {
137         case null => // skip nulls
138         case j: JsonElement => obj.add(key, j)
139         case j: String  => obj.add(key, new JsonPrimitive(j))
140         case j: Number  => obj.add(key, new JsonPrimitive(j))
141         case j: Boolean => obj.add(key, new JsonPrimitive(j))
142         case j: Date    => obj.add(key, new JsonPrimitive(DateParsing.formatDate(j)))
143         case j          => obj.add(key, gson.toJsonTree(j))
144       }
145 
146       i += 2
147     }
148     obj
149   }

151   private val emptyToNull = TransformerFunction.pure("emptyJsonToNull") { args =>
152     args(0) match {

154       case j: JsonObject if j.size() == 0 => null
155       case j: JsonObject if j.entrySet().asScala.forall(_.getValue == JsonNull.INSTANCE) => null
156       case j: JsonArray if j.size() == 0 => null
157       case j => j
