114   override def postCreateTable(schemaName: String, sft: SimpleFeatureType, cx: Connection): Unit = {
115     // Throw an error if the sft name is longer than 31 characters

169   override def postCreateFeatureType(
170       sft: SimpleFeatureType,
171       metadata: DatabaseMetaData,
172       schemaName: String,
173       cx: Connection): Unit = {
174 
175     import PartitionedPostgisDialect.Config._
176 
177     // normally views get set to read-only, override that here since we use triggers to delegate writes
178     sft.getUserData.remove(JDBCDataStore.JDBC_READ_ONLY)
179 
180     // populate user data
181     val userDataSql = s"select key, value from ${escape(schemaName)}.${UserDataTable.Name.quoted} where type_name = ?"
182     WithClose(cx.prepareStatement(userDataSql)) { statement =>
183       statement.setString(1, sft.getTypeName)
184       WithClose(statement.executeQuery()) { rs =>
185         while (rs.next()) {
186           sft.getUserData.put(rs.getString(1), rs.getString(2))
187         }
188       }
189     }
190 
191     // populate tablespaces
192     val tablespaceSql =
193       s"select table_space, table_type from " +
194           s"${escape(schemaName)}.${PartitionTablespacesTable.Name.quoted} where type_name = ?"
195     WithClose(cx.prepareStatement(tablespaceSql)) { statement =>
196       statement.setString(1, sft.getTypeName)
197       WithClose(statement.executeQuery()) { rs =>
198         while (rs.next()) {
199           val ts = rs.getString(1)
200           if (ts != null && ts.nonEmpty) {
201             rs.getString(2) match {
202               case WriteAheadTableSuffix.raw => sft.getUserData.put(WriteAheadTableSpace, ts)
203               case PartitionedWriteAheadTableSuffix.raw => sft.getUserData.put(WriteAheadPartitionsTableSpace, ts)
204               case PartitionedTableSuffix.raw => sft.getUserData.put(MainTableSpace, ts)
205               case s => logger.warn(s"Ignoring unexpected tablespace table: $s")
206             }
207           }
208         }
209       }
210     }

261   override def encodePostColumnCreateTable(att: AttributeDescriptor, sql: StringBuffer): Unit = {
262     import PartitionedPostgisDialect.Config.GeometryAttributeConversions
263     att match {
264       case gd: GeometryDescriptor =>
265         val nullable = gd.getMinOccurs <= 0 || gd.isNillable
266         val i = sql.lastIndexOf("geometry")
267         // expect `geometry NOT NULL` or `geometry` depending on nullable flag
268         if (i == -1 || (nullable && i != sql.length() - 8) || (!nullable && i != sql.length() - 17)) {
269           logger.warn(s"Found geometry-type attribute but no geometry column binding: $sql")
270         } else {
271           val srid = gd.getSrid.getOrElse(-1)
272           val geomType = PartitionedPostgisDialect.GeometryMappings.getOrElse(gd.getType.getBinding, "GEOMETRY")
273           val geomTypeWithDims = gd.getCoordinateDimensions match {
274             case None | Some(2) => geomType
275             case Some(3) => s"${geomType}Z"
276             case Some(4) => s"${geomType}ZM"
277             case Some(d) =>
278               throw new IllegalArgumentException(
279                 s"PostGIS only supports geometries with 2, 3 and 4 dimensions, but found: $d")
280           }
281           sql.insert(i + 8, s" ($geomTypeWithDims, $srid)")
282         }
283 
284       case _ if att.isJson() =>
285         // replace 'VARCHAR' with jsonb
286         val i = sql.lastIndexOf(" VARCHAR")
287         if (i == sql.length() - 8) {
288           sql.replace(i + 1, i + 8, "JSONB")
289         } else {
290           logger.warn(s"Found JSON-type attribute but no CHARACTER VARYING column binding: $sql")
291         }
292 
293       case _ if att.isList =>
294         // go back and encode the array type in the CQL create statement
295         val i = sql.lastIndexOf(" ARRAY")
296         if (i == sql.length() - 6) {
297           sql.insert(i, " " + getListTypeMapping(att.getListType()))
298         } else {
299           logger.warn(s"Found list-type attribute but no ARRAY column binding: $sql")
300         }
301 
302       case _ => // no-op
303     }
304   }

315   /**
316    * Gets the array type for a list/array column
317    *
318    * @param binding list-type binding
319    * @return
320    */
321   private def getListTypeMapping(binding: Class[_]): String = {
322     val mappings = new java.util.HashMap[String, Class[_]]()
323     registerSqlTypeNameToClassMappings(mappings)
324     var mapping: String = null
325     var partial: String = null
326     val iter = mappings.asScala.iterator
327     while (iter.hasNext && mapping == null) {
328       val (name, clas) = iter.next
329       if (clas == binding) {
330         mapping = name
331       } else if (partial == null && clas.isAssignableFrom(binding)) {
332         partial = name
333       }
334     }
335     if (mapping != null) {
336       mapping
337     } else if (partial != null) {
338       partial
339     } else {
340       logger.warn(s"Could not find list-type column for type ${binding.getName}")
341       "text"
342     }
343   }
