24 @DescribeProcess(
25   title = "Date Offset Process",
26   description = "Modifies the specified date field in a feature collection by an input time period."
27 )
28 class DateOffsetProcess extends GeoMesaProcess {
29 
30   @throws(classOf[ProcessException])
31   @DescribeResult(name = "result", description = "Output collection")
32   def execute(@DescribeParameter(name = "data", description = "Input features")
33               obsFeatures: SimpleFeatureCollection,
34               @DescribeParameter(name = "dateField", description = "The date attribute to modify")
35               dateField: String,
36               @DescribeParameter(name = "timeOffset", description = "Time offset (e.g. P1D)")
37               timeOffset: String): SimpleFeatureCollection = {
38 
39     val period = try { Duration.parse(timeOffset) } catch {
40       case e: DateTimeParseException => throw new IllegalArgumentException(s"Invalid offset $timeOffset", e)
41     }
42     val dtgIndex = obsFeatures.getSchema.indexOf(dateField)
43     require(dtgIndex != -1, s"Field '$dateField' does not exist in input feature collection: ${obsFeatures.getSchema}")
44 
45     val iter = SelfClosingIterator(obsFeatures.features()).map { sf =>
46       val dtg = sf.getAttribute(dateField).asInstanceOf[Date]
47       val offset = ZonedDateTime.ofInstant(toInstant(dtg), ZoneOffset.UTC).plus(period)
48       val newDtg = Date.from(offset.toInstant)
49       sf.setAttribute(dtgIndex, newDtg)
50       sf
51     }
52 
53     new ListFeatureCollection(obsFeatures.getSchema, iter.toList.asJava)
54   }
55 }
