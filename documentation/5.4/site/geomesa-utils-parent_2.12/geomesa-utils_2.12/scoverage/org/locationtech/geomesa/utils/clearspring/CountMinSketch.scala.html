32 /**
33  * Count-Min Sketch datastructure.
34  * An Improved Data Stream Summary: The Count-Min Sketch and its Applications
35  * https://web.archive.org/web/20060907232042/http://www.eecs.harvard.edu/~michaelm/CS222/countmin.pdf
36  */
37 class CountMinSketch private (val eps: Double,
38                               val confidence: Double,
39                               private [utils] val table: Array[Array[Long]],
40                               private [utils] var _size: Long,
41                               private val depth: Int,
42                               private val width: Int,
43                               private val hashA: Array[Long]) extends IFrequency {
44 
45   def getRelativeError: Double = eps
46   def getConfidence: Double = confidence
47 
48   private def hash(item: Long, i: Int): Int = {
49     var hash = hashA(i) * item
50     // A super fast way of computing x mod 2^p-1
51     // See http://www.cs.princeton.edu/courses/archive/fall09/cos521/Handouts/universalclasses.pdf
52     // page 149, right after Proposition 7.
53     hash += hash >> 32
54     hash &= CountMinSketch.PrimeModulus
55     // Doing "%" after (int) conversion is ~2x faster than %'ing longs.
56     hash.toInt % width
57   }
58 
59   override def add(item: Long, count: Long): Unit = {
60     if (count < 0) {
61       // Actually for negative increments we'll need to use the median
62       // instead of minimum, and accuracy will suffer somewhat.
63       // Probably makes sense to add an "allow negative increments"
64       // parameter to constructor.
65       throw new IllegalArgumentException("Negative increments not implemented")
66     }
67     var i = 0
68     while (i < depth) {
69       table(i)(hash(item, i)) += count
70       i += 1
71     }
72     _size += count
73   }
74 
75   override def add(item: String, count: Long): Unit = {
76     if (count < 0) {
77       // Actually for negative increments we'll need to use the median
78       // instead of minimum, and accuracy will suffer somewhat.
79       // Probably makes sense to add an "allow negative increments"
80       // parameter to constructor.
81       throw new IllegalArgumentException("Negative increments not implemented")
82     }
83     val buckets = Filter.getHashBuckets(item, depth, width)
84     var i = 0
85     while (i < depth) {
86       table(i)(buckets(i)) += count
87       i += 1
88     }
89     _size += count
90   }
91 
92   /**
93     * The estimate is correct within 'epsilon' * (total item count),
94     * with probability 'confidence'.
95     */
96   override def estimateCount(item: Long): Long = {
97     var res = Long.MaxValue
98     var i = 0
99     while (i < depth) {
100       val row = table(i)
101       res = math.min(res, row(hash(item, i)))
102       i += 1
103     }
104     res
105   }
106 
107   override def estimateCount(item: String): Long = {
108     var res = Long.MaxValue
109     val buckets = Filter.getHashBuckets(item, depth, width)
110     var i = 0
111     while (i < depth) {
112       res = Math.min(res, table(i)(buckets(i)))
113       i += 1
114     }
115     res
116   }
117 
118   def +=(other: CountMinSketch): Unit = {
119     // note: we assume that seed is equal
120     if (depth != other.depth || width != other.width) {
121       throw new IllegalArgumentException("Can't merge CountMinSketch of different sizes")
122     }
123     var i, j = 0
124     while (i < table.length) {
125       val row = table(i)
126       val otherRow = other.table(i)
127       while (j < row.length) {
128         row(j) += otherRow(j)
129         j += 1
130       }
131       i += 1
132       j = 0
133     }
134     _size += other.size
135   }
136 
137   def clear(): Unit = {
138     var i, j = 0
139     while (i < depth) {
140       val row = table(i)
141       while (j < width) {
142         row(j) = 0L
143         j += 1
144       }
145       i += 1
146       j = 0
147     }
148     _size = 0L
149   }
150 
151   def isEquivalent(other: CountMinSketch): Boolean = {
152     if (size != other.size || depth != other.depth || width != other.width) {
153       return false
154     }
155     var i, j = 0
156     while (i < depth) {
157       if (hashA(i) != other.hashA(i)) {
158         return false
159       }
160       val row = table(i)
161       val otherRow = other.table(i)
162       while (j < width) {
163         if (row(j) != otherRow(j)) {
164           return false
165         }
166         j += 1
167       }
168       i += 1
169       j = 0
170     }
171     true
172   }
173 
174   override def size: Long = _size
175 }

177 object CountMinSketch {
178 
179   private val PrimeModulus: Long = (1L << 31) - 1
180 
181   def apply(eps: Double, confidence: Double, seed: Int): CountMinSketch = {
182 
183     // 2/w = eps ; w = 2/eps
184     // 1/2^depth <= 1-confidence ; depth >= -log2 (1-confidence)
185 
186     val width = math.ceil(2 / eps).toInt
187     val depth = math.ceil(-1 * math.log(1 - confidence) / math.log(2)).toInt
188 
189     val table: Array[Array[Long]] = Array.fill(depth)(Array.fill(width)(0L))
190 
191     // We're using a linear hash functions
192     // of the form (a*x+b) mod p.
193     // a,b are chosen independently for each hash function.
194     // However we can set b = 0 as all it does is shift the results
195     // without compromising their uniformity or independence with
196     // the other hashes.
197     val hashA: Array[Long] = {
198       val r = new java.util.Random(seed)
199       Array.fill(depth)(r.nextInt(Int.MaxValue))
200     }
201 
202     new CountMinSketch(eps, confidence, table, 0L, depth, width, hashA)
203   }
