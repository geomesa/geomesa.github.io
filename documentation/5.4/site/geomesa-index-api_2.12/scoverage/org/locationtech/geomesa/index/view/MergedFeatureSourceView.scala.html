51   override def getCount(query: Query): Int = {
52     val total =
53       if (parallel) {
54         def getSingle(sourceAndFilter: (SimpleFeatureSource, Option[Filter])): Int = {
55           val (source, filter) = sourceAndFilter
56           source.getCount(mergeFilter(sft, query, filter))
57         }
58         val results = new CopyOnWriteArrayList[Int]()
59         sources.toList.map(s => CachedThreadPool.submit(() => results.add(getSingle(s)))).foreach(_.get)
60         results.asScala.foldLeft(0)((sum, count) => if (sum < 0 || count < 0) { -1 } else { sum + count })
61       } else {
62         // if one of our sources can't get a count (i.e. is negative), give up and return -1
63         sources.foldLeft(0) { case (sum, (source, filter)) =>
64           lazy val count = source.getCount(mergeFilter(sft, query, filter))
65           if (sum < 0 || count < 0) { -1 } else { sum + count }
66         }
67       }
68     if (query.isMaxFeaturesUnlimited) {
69       total
70     } else {
71       math.min(total, query.getMaxFeatures)
72     }
73   }
