185   abstract class WholeNumberBinning[T](length: Int, bounds: (T, T)) extends Binning[T](length, bounds) {
186 
187     private val min = convertToLong(bounds._1)
188     private val max = convertToLong(bounds._2)
189 
190     require(min < max,
191       s"Upper bound must be greater than lower bound: lower='${bounds._1}'($min) upper='${bounds._2}'($max)")
192 
193     private val binSize = (max - min).toDouble / length
194 
195     override def directIndex(value: Long): Int = {
196       if (value < min || value > max) { -1 } else {
197         val i = math.floor((value - min) / binSize).toInt
198         // i == length check catches the upper bound
199         if (i < 0 || i > length) -1 else if (i == length) length - 1 else i
200       }
201     }
202 
203     override def indexOf(value: T): Int = directIndex(convertToLong(value))
204 
205     override def medianValue(index: Int): T = {
206       if (index < 0 || index > length) {
207         throw new ArrayIndexOutOfBoundsException(index)
208       }
209       val long = min + math.round(binSize / 2 + binSize * index)
210       if (long > max) bounds._2 else convertFromLong(long)
211     }
212 
213     override def bounds(index: Int): (T, T) = {
214       if (index < 0 || index > length) {
215         throw new ArrayIndexOutOfBoundsException(index)
216       }
217       val loLong = min + math.ceil(binSize * index).toLong
218       val hiLong = math.max(loLong, min + math.floor(binSize * (index + 1)).toLong)
219       val lo = if (loLong <= min) bounds._1 else convertFromLong(loLong)
220       val hi = if (hiLong >= max) bounds._2 else convertFromLong(hiLong)
221 
222       (lo, hi)
223     }
224 
225     override def isBelow(value: T): Boolean = convertToLong(value) < min
226 
227     /**
228      * Maps a value to a long used to allocate values in bins
229      *
230      * @param value value to convert
231      * @return value as a long
232      */
233     protected def convertToLong(value: T): Long
234 
235     /**
236      * Maps a long back to a value
237      *
238      * @param value value as a long
239      * @return value
240      */
241     protected def convertFromLong(value: Long): T
242   }

352   /**
353    * Bins strings. Will set up bins based on the longest-common-prefix of the bounds. Estimates
354    * bins by considering inputs to be roughly equivalent to base36 longs.
355    *
356    * @param length number of bins
357    * @param rawBounds upper and lower bounds for the input values
358    */
359   class StringBinning(length: Int, rawBounds: (String, String))
360       extends WholeNumberBinning[String](length, StringBinning.normalizeBounds(rawBounds)) {
361 
362     import StringBinning.{Base36Lowest, normalize}
363 
364     private lazy val (start, end): (String, String) = bounds
365     private lazy val normalizedLength = start.length
366     private lazy val prefixLength = start.zip(end).indexWhere { case (l, r) => l != r }
367     private lazy val prefix = start.substring(0, prefixLength)
368 
369     override protected def convertToLong(value: String): Long = {
370       val normalized = normalize(value).padTo(normalizedLength, '0')
371       if (normalized < start) { 0L } else if (normalized > end) { Long.MaxValue } else {
372         // note: 12 is the most base-36 numbers we can fit in Long.MaxValue
373         val sigDigits = normalized.substring(prefixLength).padTo(12, Base36Lowest).substring(0, 12)
374         jLong.parseLong(sigDigits, 36)
375       }
376     }
377 
378     override protected def convertFromLong(value: Long): String =
379       prefix + jLong.toString(value, 36).reverse.padTo(12, Base36Lowest).reverse.replaceFirst("0+$", "")
380   }

382   object StringBinning {
383 
384     private val Base36Chars = (0 until 36).map(Integer.toString(_, 36).toLowerCase(Locale.US).charAt(0)).toArray
385     val Base36Lowest: Char  = Base36Chars.head
386     val Base36Highest: Char = Base36Chars.last
387 
388     def normalize(s: String): String = s.toLowerCase(Locale.US).replaceAll("[^0-9a-z]", Base36Lowest.toString)
389 
390     def normalizeBounds(bounds: (String, String)): (String, String) = {
391       val length = math.max(bounds._1.length, bounds._2.length)
392       val loBase36 = normalize(bounds._1)
393       val hiBase36 = normalize(bounds._2)
394       // pad them and make sure they are in sorted order
395       val (loPadded, hiPadded) = if (loBase36 < hiBase36) {
396         (loBase36.padTo(length, Base36Lowest), hiBase36.padTo(length, Base36Highest))
397       } else {
398         (hiBase36.padTo(length, Base36Lowest), loBase36.padTo(length, Base36Highest))
399       }
400       // ensure that they haven't become the same string
401       val (loDistinct, hiDistinct) = if (loPadded == hiPadded) {
402         (loPadded + Base36Lowest, hiPadded + Base36Highest)
403       } else {
404         (loPadded, hiPadded)
405       }
406       // check to make sure they fit in a long (12 chars)
407       val prefixLength = loDistinct.zip(hiDistinct).indexWhere { case (l, r) => l != r }
408       val loFit = if (loDistinct.length > prefixLength + 12) loDistinct.take(prefixLength + 12) else loDistinct
409       val hiFit = if (hiDistinct.length > prefixLength + 12) hiDistinct.take(prefixLength + 12) else hiDistinct
410       // check one last time that they aren't the same
411       if (loFit == hiFit) {
412         (loFit.dropRight(1) + Base36Lowest, hiFit.dropRight(1) + Base36Highest)
413       } else {
414         (loFit, hiFit)
415       }
416     }
