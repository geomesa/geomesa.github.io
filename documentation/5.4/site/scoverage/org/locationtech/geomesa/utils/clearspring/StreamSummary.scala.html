36 /**
37   * Based on the `Space-Saving` algorithm and the `Stream-Summary`
38   * data structure as described in:
39   * `Efficient Computation of Frequent and Top-k Elements in Data Streams`
40   * by Metwally, Agrawal, and Abbadi
41   *
42   * @param capacity maximum size (larger capacities improve accuracy)
43   * @tparam T type of data in the stream to be summarized
44   */

73   /**

75     * @return (isNewItem, itemDropped) where isNewItem is the return value of offer() and itemDropped is
76     *         null if no item was dropped
77     */
78   def offerReturnAll(item: T, increment: Long): (Boolean, T) = {
79     var counterNode = counterMap.get(item)
80     val isNewItem = counterNode == null
81     var droppedItem: T = null.asInstanceOf[T]
82 
83     if (isNewItem) {
84       if (size < capacity) {
85         counterNode = bucketList.enqueue(new Bucket(0)).getValue.counterList.add(new Counter[T](bucketList.tail(), item))
86       } else {
87         val min = bucketList.first()
88         counterNode = min.counterList.tail()
89         val counter = counterNode.getValue
90         droppedItem = counter.item
91         counterMap.remove(droppedItem)
92         counter.item = item
93         counter.error = min.count
94       }
95       counterMap.put(item, counterNode)
96     }
97 
98     incrementCounter(counterNode, increment)
99 
100     (isNewItem, droppedItem)
101   }

103   private def incrementCounter(counterNode: ListNode2[Counter[T]], increment: Long): Unit = {
104     val counter = counterNode.getValue     // count_i
105     val oldNode = counter.bucketNode
106     val bucket = oldNode.getValue          // Let Bucket_i be the bucket of count_i
107     bucket.counterList.remove(counterNode) // Detach count_i from Bucket_i's child-list
108     counter.count += increment
109 
110     // Finding the right bucket for count_i
111     // Because we allow a single call to increment count more than once, this may not be the adjacent bucket.
112     var bucketNodePrev = oldNode
113     var bucketNodeNext = bucketNodePrev.getNext
114     var break = false
115     while (!break && bucketNodeNext != null) {
116       val bucketNext = bucketNodeNext.getValue // Let Bucket_i^+ be Bucket_i's neighbor of larger value
117       if (counter.count == bucketNext.count) {
118         bucketNext.counterList.add(counterNode)    // Attach count_i to Bucket_i^+'s child-list
119         break = true
120       } else if (counter.count > bucketNext.count) {
121         bucketNodePrev = bucketNodeNext
122         bucketNodeNext = bucketNodePrev.getNext  // Continue hunting for an appropriate bucket
123       } else {
124         // A new bucket has to be created
125         bucketNodeNext = null
126       }
127     }
128 
129     if (bucketNodeNext == null) {
130         val bucketNext = new Bucket[T](counter.count)
131         bucketNext.counterList.add(counterNode)
132         bucketNodeNext = bucketList.addAfter(bucketNodePrev, bucketNext)
133     }
134     counter.bucketNode = bucketNodeNext
135 
136     // Cleaning up
137     if (bucket.counterList.isEmpty) {     // If Bucket_i's child-list is empty
138       bucketList.remove(oldNode);         // Detach Bucket_i from the Stream-Summary
139     }
140   }

142   override def peek(k: Int): java.util.List[T] = {
143     val topK = new java.util.ArrayList[T](k)
144 
145     if (k > 0) {
146       var bNode = bucketList.head()
147       while (bNode != null) {
148         val b = bNode.getValue
149         val iter = b.counterList.iterator()
150         while (iter.hasNext) {
151           topK.add(iter.next.item)
152           if (topK.size() == k) {
153             return topK
154           }
155         }
156         bNode = bNode.getPrev
157       }
158     }
159 
160     topK
161   }

176     private def nextCounters(): java.util.Iterator[Counter[T]] = {
177       if (bNode == null) { Collections.emptyIterator() } else {
178         val iter = bNode.getValue.counterList.iterator()
179         bNode = bNode.getPrev
180         if (iter.hasNext) {
181           iter
182         } else {
183           nextCounters()
184         }
185       }
186     }

198   override def toString: String = {
199     val sb = new StringBuilder()
200     sb.append('[')
201     var bNode = bucketList.head()
202     while (bNode != null) {
203       val b = bNode.getValue
204       sb.append('{')
205       sb.append(b.count)
206       sb.append(":[")
207       val iter = b.counterList.iterator()
208       while (iter.hasNext) {
209         val c = iter.next
210         sb.append('{')
211         sb.append(c.item)
212         sb.append(':')
213         sb.append(c.error)
214         sb.append("},")
215       }
216       if (b.counterList.size() > 0) {
217         sb.deleteCharAt(sb.length - 1)
218       }
219       sb.append("]},")
220       bNode = bNode.getPrev
221     }
222     if (bucketList.size() > 0) {
223       sb.deleteCharAt(sb.length - 1)
224     }
225     sb.append(']')
226     sb.toString()
227   }

232   /**
233     * Create an empty stream summary
234     *
235     * @param capacity capacity
236     * @tparam T type param
237     * @return stream summary
238     */
239   def apply[T](capacity: Int): StreamSummary[T] = {
240     val counterMap = new java.util.HashMap[T, ListNode2[Counter[T]]]()
241     val bucketList = new DoublyLinkedList[Bucket[T]]()
242     new StreamSummary(capacity, counterMap, bucketList)
243   }
244 
245   /**
246     * Create a stream summary containing counts
247     *
248     * @param capacity capacity
249     * @param counters counters - must be sorted high->low
250     * @tparam T type param
251     * @return stream summary
252     */
253   def apply[T](capacity: Int, counters: Seq[(T, Long)]): StreamSummary[T] = {
254     val bucketList = new DoublyLinkedList[Bucket[T]]()
255     val counterMap = new java.util.HashMap[T, ListNode2[Counter[T]]](counters.length)
256 
257     var currentBucket: Bucket[T] = null
258     var currentBucketNode: ListNode2[Bucket[T]] = null
259 
260     counters.foreach { case (item, count) =>
261       val c = new Counter[T](null, item, count)
262       if (currentBucket == null || c.count != currentBucket.count) {
263         currentBucket = new Bucket(c.count)
264         currentBucketNode = bucketList.enqueue(currentBucket)
265       }
266       c.bucketNode = currentBucketNode
267       counterMap.put(c.item, currentBucket.counterList.add(c))
268     }
269 
270     new StreamSummary(capacity, counterMap, bucketList)
271   }
272 
273   private class Bucket[T](var count: Long) {
274     val counterList = new DoublyLinkedList[Counter[T]]
275   }

277   private class Counter[T](var bucketNode: ListNode2[Bucket[T]], var item: T, var count: Long = 0L, var error: Long = 0L) {
278     override def toString: String = s"$item:$count:$error"
279   }
