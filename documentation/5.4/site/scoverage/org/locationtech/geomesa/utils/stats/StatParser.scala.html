28   @throws(classOf[ParsingException])
29   def parse(sft: SimpleFeatureType, stat: String, report: Boolean = true): Stat = {
30     if (stat == null) {
31       throw new IllegalArgumentException("Stat must not be null")
32     }
33     val runner = if (report) { ReportingParseRunner(Parser.stat) } else { BasicParseRunner(Parser.stat) }
34     sfts.set(sft)
35     try {
36       val parsing = runner.run(stat)
37       parsing.result.getOrElse {
38         throw new ParsingException(s"Invalid stat string: ${ErrorUtils.printParseErrors(parsing)}")
39       }
40     } finally {
41       sfts.remove()
42     }
43   }
