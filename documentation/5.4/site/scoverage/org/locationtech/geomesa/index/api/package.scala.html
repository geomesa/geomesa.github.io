229   /**
230    * A query filter split up into a 'primary' that will be used with the given feature index for range planning,
231    * and a 'secondary' that is not captured in the ranges.
232    *
233    * @param index feature index to scan
234    * @param primary primary filter used for range generation
235    * @param secondary secondary filter not captured in the ranges
236    * @param temporal does the filter strategy have a temporal component
237    * @param costMultiplier a multiplier to take into account when evaluating the cost of this
238    *                 query plan. Values < 1f will prioritize this plan, while values > 1f
239    *                 will de-prioritize this plan
240    */
241   case class FilterStrategy(
242       index: GeoMesaFeatureIndex[_, _],
243       primary: Option[Filter],
244       secondary: Option[Filter],
245       temporal: Boolean,
246       costMultiplier: Float) {
247 
248     lazy val filter: Option[Filter] = andOption(primary.toSeq ++ secondary)
249 
250     def isFullTableScan: Boolean = primary.isEmpty
251 
252     def isPreferredScan: Boolean = costMultiplier < FilterStrategy.PreferredMultiplierThreshold
253 
254     def getQueryStrategy(hints: Hints, explain: Explainer = ExplainNull): QueryStrategy =
255       index.getQueryStrategy(this, hints, explain)
256 
257     override lazy val toString: String =
258       s"$index[${primary.map(filterToString).getOrElse("INCLUDE")}]" +
259           s"[${secondary.map(filterToString).getOrElse("None")}]($costMultiplier)"
260   }

287   // indicates that the lower bound is unbounded and can't be tiered
288   case class UpperBoundedByteRange(lower: Array[Byte], upper: Array[Byte]) extends ByteRange
289   // indicates that both bounds are unbounded and can't be tiered
290   case class UnboundedByteRange(lower: Array[Byte], upper: Array[Byte]) extends ByteRange

292   object ByteRange {
293 
294     import ByteArrays.ByteOrdering
295 
296     val UnboundedLowerRange: Array[Byte] = Array.empty
297     val UnboundedUpperRange: Array[Byte] = Array.fill(3)(ByteArrays.MaxByte)
298 
299     def min(ranges: Seq[ByteRange]): Array[Byte] = {
300       if (ranges.isEmpty) { UnboundedLowerRange } else {
301         ranges.collect {
302           case BoundedByteRange(lo, _) => lo
303           case SingleRowByteRange(row) => row
304           case r => throw new IllegalArgumentException(s"Unexpected range type $r")
305         }.min
306       }
307     }
308 
309     def max(ranges: Seq[ByteRange]): Array[Byte] = {
310       if (ranges.isEmpty) { UnboundedUpperRange } else {
311         ranges.collect {
312           case BoundedByteRange(_, hi) => hi
313           case SingleRowByteRange(row) => row
314           case r => throw new IllegalArgumentException(s"Unexpected range type $r")
315         }.max
316       }
317     }
318   }
