36   private val expiry = SystemProperty("geomesa.filter.remote.cache.expiry", "10 minutes").toDuration.get
37 
38   // thread safe object caches
39   private val serializerCache: Cache[(String, String), KryoFeatureSerializer] =
40     Caffeine.newBuilder().expireAfterAccess(expiry.toMillis, TimeUnit.MILLISECONDS).build()
41   private val indexCache: Cache[(String, String), GeoMesaFeatureIndex[_, _]] =
42     Caffeine.newBuilder().expireAfterAccess(expiry.toMillis, TimeUnit.MILLISECONDS).build()
43 
44   // non-thread safe object caches
45   private val filterCache = new ThreadLocalCache[(String, String), Filter](expiry)
46 
47   /**
48     * Returns a cached simple feature type, creating one if necessary. Note: do not modify returned value.
49     *
50     * @param spec simple feature type spec
51     * @return
52     */
53   def sft(spec: String): SimpleFeatureType = SimpleFeatureTypes.createImmutableType("", spec)
54 
55   /**
56     * Returns a cached serializer, creating one if necessary
57     *
58     * @param spec simple feature type spec
59     * @param options serialization options
60     * @return
61     */
62   def serializer(spec: String, options: Set[SerializationOption]): KryoFeatureSerializer = {
63     // note: before the cache is populated, we might end up creating multiple objects, but it is still thread-safe
64     val cached = serializerCache.getIfPresent((spec, options.mkString))
65     if (cached != null) { cached } else {
66       val serializer = KryoFeatureSerializer(sft(spec), options)
67       serializerCache.put((spec, options.mkString), serializer)
68       serializer
69     }
70   }
71 
72   /**
73    * Returns a cached serializer, creating one if necessary
74    *
75    * @param spec simple feature type spec
76    * @param withId serialization option
77    * @return
78    */
79   def serializer(spec: String, withId: Boolean): KryoFeatureSerializer =
80     serializer(spec, if (withId) { SerializationOption.defaults } else { Set(SerializationOption.WithoutId) })
81 
82   /**
83     * Returns a cached filter, creating one if necessary.
84     *
85     * Note: need to include simple feature type in cache key,
86     * as attribute name -> attribute index gets cached in the filter
87     *
88     * @param sft simple feature type being filtered
89     * @param spec spec string for the simple feature type
90     * @param ecql ecql
91     * @return
92     */
93   def filter(sft: SimpleFeatureType, spec: String, ecql: String): Filter = {
94     logger.debug(s"Filter cache estimated size: ${filterCache.estimatedGlobalSize}")
95     logger.trace(s"Filter cache entries: ${filterCache.globalIterator.map { case (t, (k, v), _) => s"thread $t $k=>$v" }.mkString(", ")}")
96     filterCache.getOrElseUpdate((spec, ecql), FastFilterFactory.toFilter(sft, ecql))
97   }
98 
99   /**
100     * Gets a cached feature index instance. Note that the index is not backed by a data store as
101     * normal, so operations which require a live connection will fail
102     *
103     * @param sft simple feature type
104     * @param spec spec string for the simple feature type
105     * @param identifier index id
106     * @return
107     */
108   def index(sft: SimpleFeatureType, spec: String, identifier: String): GeoMesaFeatureIndex[_, _] = {
109     val cached = indexCache.getIfPresent((spec, identifier))
110     if (cached != null) { cached } else {
111       val index = GeoMesaFeatureIndexFactory.create(null, sft, Seq(IndexId.id(identifier))).headOption.getOrElse {
112         throw new RuntimeException(s"Index option not configured correctly: $identifier")
113       }
114       if (!index.mode.supports(IndexMode.Read)) {
115         throw new RuntimeException(s"Index option configured for a non-readable index: $identifier")
116       }
117       indexCache.put((spec, identifier), index)
118       index
119     }
120   }
