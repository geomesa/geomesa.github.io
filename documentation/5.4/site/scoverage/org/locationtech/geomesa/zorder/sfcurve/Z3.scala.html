11 class Z3(val z: Long) extends AnyVal {
12   import Z3._
13 
14   def < (other: Z3): Boolean = z < other.z
15   def > (other: Z3): Boolean = z > other.z
16   def >= (other: Z3): Boolean = z >= other.z
17   def <= (other: Z3): Boolean = z <= other.z
18   def + (offset: Long): Z3 = new Z3(z + offset)
19   def - (offset: Long): Z3 = new Z3(z - offset)
20   def == (other: Z3): Boolean = other.z == z
21 
22   def d0: Int = combine(z)
23   def d1: Int = combine(z >> 1)
24   def d2: Int = combine(z >> 2)
25 
26   def decode: (Int, Int, Int) = (d0, d1, d2)
27 
28   def dim(i: Int): Int = if (i == 0) d0 else if (i == 1) d1 else if (i == 2) d2 else {
29     throw new IllegalArgumentException(s"Invalid dimension $i - valid dimensions are 0,1,2")
30   }
31 
32   def inRange(rmin: Z3, rmax: Z3): Boolean = {
33     val (x, y, z) = decode
34     x >= rmin.d0 &&
35       x <= rmax.d0 &&
36       y >= rmin.d1 &&
37       y <= rmax.d1 &&
38       z >= rmin.d2 &&
39       z <= rmax.d2
40   }
41 
42   def mid(p: Z3): Z3 = {
43     val (x, y, z) = decode
44     val (px, py, pz) = p.decode
45     Z3((x + px) >>> 1, (y + py) >>> 1, (z + pz) >>> 1) // overflow safe mean
46   }
47 
48   def bitsToString = f"(${z.toBinaryString.toLong}%016d)(${d0.toBinaryString.toLong}%08d,${d1.toBinaryString.toLong}%08d,${d2.toBinaryString.toLong}%08d)"
49   override def toString = f"$z $decode"
50 }

61   /**
62     * So this represents the order of the tuple, but the bits will be encoded in reverse order:
63     *   ....z1y1x1z0y0x0
64     * This is a little confusing.
65     */
66   def apply(x: Int, y:  Int, z: Int): Z3 = {
67     new Z3(split(x) | split(y) << 1 | split(z) << 2)
68   }
69 
70   def unapply(z: Z3): Option[(Int, Int, Int)] = Some(z.decode)
71 
72   /** insert 00 between every bit in value. Only first 21 bits can be considered. */
73   override def split(value: Long): Long = {
74     var x = value & MaxMask
75     x = (x | x << 32) & 0x1f00000000ffffL
76     x = (x | x << 16) & 0x1f0000ff0000ffL
77     x = (x | x << 8)  & 0x100f00f00f00f00fL
78     x = (x | x << 4)  & 0x10c30c30c30c30c3L
79     (x | x << 2)      & 0x1249249249249249L
80   }
81 
82   /** combine every third bit to form a value. Maximum value is 21 bits. */
83   override def combine(z: Long): Int = {
84     var x = z & 0x1249249249249249L
85     x = (x ^ (x >>  2)) & 0x10c30c30c30c30c3L
86     x = (x ^ (x >>  4)) & 0x100f00f00f00f00fL
87     x = (x ^ (x >>  8)) & 0x1f0000ff0000ffL
88     x = (x ^ (x >> 16)) & 0x1f00000000ffffL
89     x = (x ^ (x >> 32)) & MaxMask
90     x.toInt
91   }
92 
93   override def contains(range: ZRange, value: Long): Boolean = {
94     val (x, y, z) = Z3(value).decode
95     x >= Z3(range.min).d0 && x <= Z3(range.max).d0 &&
96         y >= Z3(range.min).d1 && y <= Z3(range.max).d1 &&
97         z >= Z3(range.min).d2 && z <= Z3(range.max).d2
98   }
99 
100   override def overlaps(range: ZRange, value: ZRange): Boolean = {
101     def overlaps(a1: Int, a2: Int, b1: Int, b2: Int) = math.max(a1, b1) <= math.min(a2, b2)
102     overlaps(Z3(range.min).d0, Z3(range.max).d0, Z3(value.min).d0, Z3(value.max).d0) &&
103         overlaps(Z3(range.min).d1, Z3(range.max).d1, Z3(value.min).d1, Z3(value.max).d1) &&
104         overlaps(Z3(range.min).d2, Z3(range.max).d2, Z3(value.min).d2, Z3(value.max).d2)
105   }
