42   override def apply(sft: SimpleFeatureType, conf: Config): Option[SimpleFeatureConverter] = {
43     if (!conf.hasPath("type") || !conf.getString("type").equalsIgnoreCase(TypeToProcess)) { None } else {
44       val (config, fields, opts) = try {
45         val c = ConfigSource.fromConfig(AbstractConverterFactory.standardDefaults(conf, logger))
46         val config = c.loadOrThrow[FeatureToFeatureConfig]
47         val fields = c.loadOrThrow[Seq[BasicField]]
48         val opts = c.loadOrThrow[BasicOptions]
49         (config, fields, opts)
50       } catch {
51         case NonFatal(e) => throw new IllegalArgumentException(s"Invalid configuration: ${e.getMessage}")
52       }
53 
54       val inputSft = SimpleFeatureTypeLoader.sftForName(config.inputSft).getOrElse {
55         throw new IllegalArgumentException(s"Could not load input sft ${config.inputSft}")
56       }
57       // FID is put in as the last attribute, we copy it over here
58       // TODO: does this have any implications for global params in the evaluation context?
59       val id = if (config.idField.isDefined) { config } else {
60         config.copy(idField = Some(Expression.Column(inputSft.getAttributeCount)))
61       }
62 
63       // add transform expressions to look up the attribute
64       val visitor = new SftLookupVisitor(inputSft)
65       val columns = fields.map { field =>
66         val transforms = field.transforms match {
67           case Some(transforms) => Some(transforms.accept(visitor))
68           case None =>
69             val i = inputSft.indexOf(field.name)
70             if (i == -1) { None } else { Some(Expression.Column(i)) }
71         }
72         field.copy(transforms = transforms)
73       }
74 
75       // any matching fields that aren't explicitly defined will be copied over by default
76       val defaults = {
77         val names = fields.map(_.name).toSet
78         sft.getAttributeDescriptors.asScala.map(_.getLocalName).filterNot(names.contains).map { name =>
79           val i = inputSft.indexOf(name)
80           if (i == -1) {
81             BasicField(name, Some(Expression.LiteralNull))
82           } else {
83             BasicField(name, Some(Expression.Column(i)))
84           }
85         }
86       }
87 
88       Some(new FeatureToFeatureConverter(sft, id, columns ++ defaults, opts))
89     }
90   }

95       hints: Map[String, AnyRef]): Try[(SimpleFeatureType, Config)] = Failure(new UnsupportedOperationException())

113   object FeatureToFeatureConfigConvert extends ConverterConfigConvert[FeatureToFeatureConfig] with StrictLogging {
114 
115     override protected def decodeConfig(
116         cur: ConfigObjectCursor,
117         `type`: String,
118         idField: Option[Expression],
119         caches: Map[String, Config],
120         userData: Map[String, Expression]): Either[ConfigReaderFailures, FeatureToFeatureConfig] = {
121       for {
122         converterName <- converterName(cur).right
123         sftName       <- cur.atKey(InputSftPath).right.flatMap(_.asString).right
124       } yield {
125         FeatureToFeatureConfig(`type`, converterName, sftName, idField, caches, userData)
126       }
127     }
128 
129     override protected def encodeConfig(config: FeatureToFeatureConfig, base: java.util.Map[String, AnyRef]): Unit =
130       base.put(InputSftPath, config.inputSft)
131   }
132 
133   /**
134    * Replaces field name lookups with column lookups referencing the input sft
135    *
136    * @param sft input sft
137    */
138   private class SftLookupVisitor(sft: SimpleFeatureType) extends ExpressionTreeVisitor {
139     override def visit(e: FieldLookup): Expression = {
140       val i = sft.indexOf(e.n)
141       if (i == -1) { e } else { Expression.Column(i) }
142     }
