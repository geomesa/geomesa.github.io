16 /**
17   * Represents a time by an offset into a binned bucket. The bin represents days, weeks,
18   * months or years since the java epoch. The offset represents milliseconds, seconds, or
19   * hours into that bin.
20   *
21   * Times can be partitioned based on four periods:
22   *
23   *   TimePeriod.Day
24   *     bin      => day
25   *     offset   => milliseconds
26   *     max date => 2059/09/18
27   *
28   *   TimePeriod.Week
29   *     bin      => week
30   *     offset   => seconds
31   *     max date => 2598/01/04
32   *
33   *   TimePeriod.Month
34   *     bin      => month
35   *     offset   => seconds
36   *     max date => 4700/08/31
37   *
38   *   TimePeriod.Year
39   *     bin      => year
40   *     offset   => minutes
41   *     max date => 34737/12/31
42   *
43   * @param bin number of time periods from the java epoch
44   * @param offset precise offset into the specific time period
45   */
46 case class BinnedTime(bin: Short, offset: Long)
47 
48 object BinnedTime {
49 
50   type BinnedTimeToDate = BinnedTime => ZonedDateTime
51   type TimeToBinnedTime = Long => BinnedTime
52   type DateToBinnedTime = ZonedDateTime => BinnedTime
53   type TimeToBin        = Long => Short
54   type DateToBin        = ZonedDateTime => Short
55 
56   val Epoch: ZonedDateTime = ZonedDateTime.ofInstant(Instant.EPOCH, ZoneOffset.UTC)
57 
58   // min value (inclusive)
59   val ZMinDate: ZonedDateTime = Epoch
60 
61   // max values (exclusive)
62   val DaysMaxDate  : ZonedDateTime = Epoch.plusDays(Short.MaxValue.toInt + 1)
63   val WeeksMaxDate : ZonedDateTime = Epoch.plusWeeks(Short.MaxValue.toInt + 1)
64   val MonthsMaxDate: ZonedDateTime = Epoch.plusMonths(Short.MaxValue.toInt + 1)
65   val YearsMaxDate : ZonedDateTime = Epoch.plusYears(Short.MaxValue.toInt + 1)
66 
67   /**
68     * Gets period index (e.g. weeks since the epoch) and offset into that interval (e.g. seconds in week)
69     *
70     * @param period interval type
71     * @return
72     */
73   def timeToBinnedTime(period: TimePeriod): TimeToBinnedTime = {
74     period match {
75       case TimePeriod.Day   => toDayAndMillis
76       case TimePeriod.Week  => toWeekAndSeconds
77       case TimePeriod.Month => toMonthAndSeconds
78       case TimePeriod.Year  => toYearAndMinutes
79     }
80   }
81 
82   /**
83     * Gets period index (e.g. weeks since the epoch)
84     *
85     * @param period interval type
86     * @return
87     */
88   def timeToBin(period: TimePeriod): TimeToBin = {
89     period match {
90       case TimePeriod.Day   => toDay
91       case TimePeriod.Week  => toWeek
92       case TimePeriod.Month => toMonth
93       case TimePeriod.Year  => toYear
94     }
95   }
96 
97   /**
98     * Gets period index (e.g. weeks since the epoch) and offset into that interval (e.g. seconds in week)
99     *
100     * @param period interval type
101     * @return
102     */
103   def dateToBinnedTime(period: TimePeriod): DateToBinnedTime = {
104     period match {
105       case TimePeriod.Day   => toDayAndMillis
106       case TimePeriod.Week  => toWeekAndSeconds
107       case TimePeriod.Month => toMonthAndSeconds
108       case TimePeriod.Year  => toYearAndMinutes
109     }
110   }
111 
112   /**
113     * Gets period index (e.g. weeks since the epoch) and offset into that interval (e.g. seconds in week)
114     *
115     * @param period interval type
116     * @return
117     */
118   def dateToBin(period: TimePeriod): DateToBin = {
119     period match {
120       case TimePeriod.Day   => toDay
121       case TimePeriod.Week  => toWeek
122       case TimePeriod.Month => toMonth
123       case TimePeriod.Year  => toYear
124     }
125   }
126 
127   /**
128     * Gets a date back from a binned time
129     *
130     * @param period interval type
131     * @return
132     */
133   def binnedTimeToDate(period: TimePeriod): BinnedTimeToDate = {
134     period match {
135       case TimePeriod.Day   => fromDayAndMillis
136       case TimePeriod.Week  => fromWeekAndSeconds
137       case TimePeriod.Month => fromMonthAndSeconds
138       case TimePeriod.Year  => fromYearAndMinutes
139     }
140   }
141 
142   /**
143     * Gets the max offset value for a given time period
144     *
145     * @param period interval type
146     * @return
147     */
148   def maxOffset(period: TimePeriod): Long = {
149     period match {
150       case TimePeriod.Day   => ChronoUnit.DAYS.getDuration.toMillis
151       case TimePeriod.Week  => ChronoUnit.WEEKS.getDuration.toMillis / 1000L
152       case TimePeriod.Month => (ChronoUnit.DAYS.getDuration.toMillis / 1000L) * 31L
153       // based on 365 days + 1 leap day, with a fudge factor of 10 minutes to account for leap seconds added each year
154       case TimePeriod.Year  => (ChronoUnit.DAYS.getDuration.toMinutes * 366L) + 10L
155     }
156   }
157 
158   /**
159     * Max indexable date (exclusive) for a given time period
160     *
161     * @param period interval type
162     * @return
163     */
164   def maxDate(period: TimePeriod): ZonedDateTime = {
165     period match {
166       case TimePeriod.Day   => DaysMaxDate
167       case TimePeriod.Week  => WeeksMaxDate
168       case TimePeriod.Month => MonthsMaxDate
169       case TimePeriod.Year  => YearsMaxDate
170     }
171   }
172 
173   /**
174     * Converts values extracted from a filter into valid indexable bounds
175     *
176     * @param period time period
177     * @return
178     */
179   def boundsToIndexableDates(period: TimePeriod): ((Option[ZonedDateTime], Option[ZonedDateTime])) => (ZonedDateTime, ZonedDateTime) = {
180     val maxDateTime = maxDate(period).minus(1L, ChronoUnit.MILLIS)
181     bounds => {
182       val lo = bounds._1 match {
183         case None => ZMinDate
184         case Some(dt) if dt.isBefore(ZMinDate) => ZMinDate
185         case Some(dt) if dt.isAfter(maxDateTime) => maxDateTime
186         case Some(dt) => dt
187       }
188       val hi = bounds._2 match {
189         case None => maxDateTime
190         case Some(dt) if dt.isBefore(ZMinDate) => ZMinDate
191         case Some(dt) if dt.isAfter(maxDateTime) => maxDateTime
192         case Some(dt) => dt
193       }
194       (lo, hi)
195     }
196   }
197 
198   private def toDay(time: Long): Short =
199     toDay(ZonedDateTime.ofInstant(Instant.ofEpochMilli(time), ZoneOffset.UTC))
200 
201   private def toDay(date: ZonedDateTime): Short = {
202     require(!date.isBefore(ZMinDate), s"Date exceeds minimum indexable value ($ZMinDate): $date")
203     require(DaysMaxDate.isAfter(date), s"Date exceeds maximum indexable value ($DaysMaxDate): $date")
204     ChronoUnit.DAYS.between(Epoch, date).toShort
205   }
206 
207   private def toDayAndMillis(time: Long): BinnedTime =
208     toDayAndMillis(ZonedDateTime.ofInstant(Instant.ofEpochMilli(time), ZoneOffset.UTC))
209 
210   private def toDayAndMillis(date: ZonedDateTime): BinnedTime = {
211     val days = toDay(date)
212     val millisInDay = date.toInstant.toEpochMilli - Epoch.plus(days, ChronoUnit.DAYS).toInstant.toEpochMilli
213     BinnedTime(days, millisInDay)
214   }
215 
216   private def fromDayAndMillis(date: BinnedTime): ZonedDateTime =
217     Epoch.plusDays(date.bin).plus(date.offset, ChronoUnit.MILLIS)
218 
219   private def toWeek(time: Long): Short =
220     toWeek(ZonedDateTime.ofInstant(Instant.ofEpochMilli(time), ZoneOffset.UTC))
221 
222   private def toWeek(date: ZonedDateTime): Short = {
223     require(!date.isBefore(ZMinDate), s"Date exceeds minimum indexable value ($ZMinDate): $date")
224     require(WeeksMaxDate.isAfter(date), s"Date exceeds maximum indexable value ($WeeksMaxDate): $date")
225     ChronoUnit.WEEKS.between(Epoch, date).toShort
226   }
227 
228   private def toWeekAndSeconds(time: Long): BinnedTime =
229     toWeekAndSeconds(ZonedDateTime.ofInstant(Instant.ofEpochMilli(time), ZoneOffset.UTC))
230 
231   private def toWeekAndSeconds(date: ZonedDateTime): BinnedTime = {
232     val weeks = toWeek(date)
233     val secondsInWeek = date.toEpochSecond - Epoch.plus(weeks, ChronoUnit.WEEKS).toEpochSecond
234     BinnedTime(weeks, secondsInWeek)
235   }
236 
237   private def fromWeekAndSeconds(date: BinnedTime): ZonedDateTime =
238     Epoch.plusWeeks(date.bin).plus(date.offset, ChronoUnit.SECONDS)
239 
240   private def toMonth(time: Long): Short =
241     toMonth(ZonedDateTime.ofInstant(Instant.ofEpochMilli(time), ZoneOffset.UTC))
242 
243   private def toMonth(date: ZonedDateTime): Short = {
244     require(!date.isBefore(ZMinDate), s"Date exceeds minimum indexable value ($ZMinDate): $date")
245     require(MonthsMaxDate.isAfter(date), s"Date exceeds maximum indexable value ($MonthsMaxDate): $date")
246     ChronoUnit.MONTHS.between(Epoch, date).toShort
247   }
248 
249   private def toMonthAndSeconds(time: Long): BinnedTime =
250     toMonthAndSeconds(ZonedDateTime.ofInstant(Instant.ofEpochMilli(time), ZoneOffset.UTC))
251 
252   private def toMonthAndSeconds(date: ZonedDateTime): BinnedTime = {
253     val months = toMonth(date)
254     val secondsInMonth = date.toEpochSecond - Epoch.plus(months, ChronoUnit.MONTHS).toEpochSecond
255     BinnedTime(months, secondsInMonth)
256   }
257 
258   private def fromMonthAndSeconds(date: BinnedTime): ZonedDateTime =
259     Epoch.plusMonths(date.bin).plus(date.offset, ChronoUnit.SECONDS)
260 
261   private def toYear(time: Long): Short =
262     toYear(ZonedDateTime.ofInstant(Instant.ofEpochMilli(time), ZoneOffset.UTC))
263 
264   private def toYear(date: ZonedDateTime): Short = {
265     require(!date.isBefore(ZMinDate), s"Date exceeds minimum indexable value ($ZMinDate): $date")
266     require(YearsMaxDate.isAfter(date), s"Date exceeds maximum indexable value ($YearsMaxDate): $date")
267     ChronoUnit.YEARS.between(Epoch, date).toShort
268   }
269 
270   private def toYearAndMinutes(time: Long): BinnedTime =
271     toYearAndMinutes(ZonedDateTime.ofInstant(Instant.ofEpochMilli(time), ZoneOffset.UTC))
272 
273   private def toYearAndMinutes(date: ZonedDateTime): BinnedTime = {
274     val years = toYear(date)
275     val minutesInYear = (date.toEpochSecond - Epoch.plus(years, ChronoUnit.YEARS).toEpochSecond) / 60L
276     BinnedTime(years, minutesInYear)
277   }
278 
279   private def fromYearAndMinutes(date: BinnedTime): ZonedDateTime =
280     Epoch.plusYears(date.bin).plus(date.offset, ChronoUnit.MINUTES)
281 }
