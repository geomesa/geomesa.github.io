139   /**
140    * Build an Accumulo client configuration, based on passed in parameters and any accumulo-client.properties
141    * file on the classpath
142    *
143    * @param params data store params
144    * @return
145    */
146   def buildAccumuloClientConfig(params: java.util.Map[String, _]): Properties = {
147     val config = AccumuloClientConfig.load()
148 
149     def setRequired(param: GeoMesaParam[String], key: ClientProperty): String = {
150       param.lookupOpt(params) match {
151         case Some(v) => config.put(key.getKey, v); v
152         case None =>
153           Option(config.getProperty(key.getKey)).getOrElse {
154             throw new IOException(s"Parameter ${param.key} is required: ${param.description}")
155           }
156       }
157     }
158 
159     def setOptional(param: GeoMesaParam[String], key: ClientProperty): Unit =
160       param.lookupOpt(params).foreach(config.put(key.getKey, _))
161 
162     def getRequired(param: GeoMesaParam[String], key: ClientProperty): String = {
163       param.lookupOpt(params).orElse(Option(config.getProperty(key.getKey))).getOrElse {
164         throw new IOException(s"Parameter ${param.key} is required: ${param.description}")
165       }
166     }
167 
168     setRequired(InstanceNameParam, ClientProperty.INSTANCE_NAME)
169     setRequired(ZookeepersParam, ClientProperty.INSTANCE_ZOOKEEPERS)
170     setOptional(ZookeeperTimeoutParam, ClientProperty.INSTANCE_ZOOKEEPERS_TIMEOUT)
171 
172     val user = setRequired(UserParam, ClientProperty.AUTH_PRINCIPAL)
173 
174     if (PasswordParam.exists(params) && KeytabPathParam.exists(params)) {
175       throw new IllegalArgumentException(
176         s"'${PasswordParam.key}' and '${KeytabPathParam.key}' are mutually exclusive, but are both set")
177     }
178 
179     val authType =
180       if (PasswordParam.exists(params)) {
181         AccumuloClientConfig.PasswordAuthType
182       } else if (KeytabPathParam.exists(params)) {
183         AccumuloClientConfig.KerberosAuthType
184       } else {
185         ClientProperty.AUTH_TYPE.getValue(config).toLowerCase(Locale.US)
186       }
187 
188     val token =
189       if (authType == AccumuloClientConfig.PasswordAuthType) {
190         new PasswordToken(getRequired(PasswordParam, ClientProperty.AUTH_TOKEN).getBytes(StandardCharsets.UTF_8))
191       } else if (authType == AccumuloClientConfig.KerberosAuthType) {
192         val file = new java.io.File(getRequired(KeytabPathParam, ClientProperty.AUTH_TOKEN))
193         // mimic behavior from accumulo 1.9 and earlier:
194         // `public KerberosToken(String principal, File keytab, boolean replaceCurrentUser)`
195         UserGroupInformation.loginUserFromKeytab(user, file.getAbsolutePath)
196         new KerberosToken(user, file)
197       } else {
198          throw new IllegalArgumentException(
199            s"Unsupported auth type: $authType - supported values are " +
200              s"${AccumuloClientConfig.PasswordAuthType}, ${AccumuloClientConfig.KerberosAuthType}")
201       }
202     ClientProperty.setAuthenticationToken(config, token)
203 
204     if (WriteThreadsParam.exists(params)) {
205       config.put(ClientProperty.BATCH_WRITER_THREADS_MAX.getKey, WriteThreadsParam.lookup(params).toString)
206     } else if (ClientProperty.BATCH_WRITER_THREADS_MAX.isEmpty(config)) {
207       BatchWriterProperties.WRITER_THREADS.option.foreach { threads =>
208         config.put(ClientProperty.BATCH_WRITER_THREADS_MAX.getKey, String.valueOf(threads))
209       }
210     }
211     if (ClientProperty.BATCH_WRITER_MEMORY_MAX.isEmpty(config)) {
212       BatchWriterProperties.WRITER_MEMORY_BYTES.toBytes.foreach { memory =>
213         config.put(ClientProperty.BATCH_WRITER_MEMORY_MAX.getKey, String.valueOf(memory))
214       }
215     }
216     if (ClientProperty.BATCH_WRITER_LATENCY_MAX.isEmpty(config)) {
217       BatchWriterProperties.WRITER_LATENCY.toDuration.foreach { duration =>
218         config.put(ClientProperty.BATCH_WRITER_LATENCY_MAX.getKey, s"${duration.toMillis}ms")
219       }
220     }
221     if (ClientProperty.BATCH_WRITER_TIMEOUT_MAX.isEmpty(config)) {
222       BatchWriterProperties.WRITE_TIMEOUT.toDuration.foreach { duration =>
223         config.put(ClientProperty.BATCH_WRITER_TIMEOUT_MAX.getKey, s"${duration.toMillis}ms")
224       }
225     }
226 
227     config
228   }

265       auditWriter = auditWriter,

282   def buildAuthsProvider(client: AccumuloClient, params: java.util.Map[String, _]): AuthorizationsProvider = {
283     // convert the connector authorizations into a string array - this is the maximum auths this connector can support
284     val securityOps = client.securityOperations
285     val masterAuths = securityOps.getUserAuthorizations(client.whoami).asScala.toSeq.map(b => new String(b))
286 
287     // get the auth params passed in as a comma-delimited string
288     val configuredAuths = AuthsParam.lookupOpt(params).getOrElse("").split(",").filterNot(_.isEmpty).toSeq
289 
290     // verify that the configured auths are valid for the connector we are using (fail-fast)
291     val invalidAuths = configuredAuths.filterNot(masterAuths.contains)
292     if (invalidAuths.nonEmpty) {
293       throw new IllegalArgumentException(s"The authorizations '${invalidAuths.mkString(",")}' " +
294         "are not valid for the Accumulo connection being used")
295     }
296 
297     val forceEmptyAuths = ForceEmptyAuthsParam.lookup(params)
298 
299     // if the caller provided any non-null string for authorizations, use it;
300     // otherwise, grab all authorizations to which the Accumulo user is entitled
301     if (configuredAuths.nonEmpty && forceEmptyAuths) {
302       throw new IllegalArgumentException("Forcing empty auths is checked, but explicit auths are provided")
303     }
304     val auths = if (forceEmptyAuths || configuredAuths.nonEmpty) { configuredAuths } else { masterAuths }
305 
306     AuthUtils.getProvider(params, auths)
307   }
