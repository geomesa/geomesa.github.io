70     // compute only the input fields that we need to deal with to populate the simple feature

214   /**
215     * Convert input values into a simple feature with attributes.
216     *
217     * This method returns a CloseableIterator to simplify flatMapping over inputs, but it will always
218     * return either 0 or 1 feature.
219     *
220     * @param rawValues raw input values
221     * @param ec evaluation context
222     * @return
223     */
224   private def convert(rawValues: Array[AnyRef], ec: EvaluationContext): CloseableIterator[SimpleFeature] = {
225     val result = ec.evaluate(rawValues).right.flatMap { values =>
226       val sf = new ScalaSimpleFeature(sft, "")
227 
228       attributeIndices.foreach { case (i, j) =>
229         sf.setAttributeNoConvert(i, values(j))
230       }
231       // if no id builder, empty feature id will be replaced with an auto-gen one
232       if (idIndex != -1) {
233         values(idIndex) match {
234           case id: String =>
235             sf.setId(id)
236             sf.getUserData.put(Hints.USE_PROVIDED_FID, java.lang.Boolean.TRUE)
237 
238           case null => // no-op
239 
240           case id =>
241             // the missing feature id will be caught by the validator
242             logger.warn(s"Ignoring non-string feature ID: $id")
243         }
244       }
245       userDataIndices.foreach { case (k, j) =>
246         sf.getUserData.put(k, values(j))
247       }
248       val error = validators.validate(sf)
249       if (error == null) {
250         Right(sf)
251       } else {
252         Left(EvaluationError(null, ec.line, new IOException(s"Validation error: $error")))
253       }
254     }
255 
256     result match {
257       case Right(feature) =>
258         ec.stats.success()
259         latency.foreach(_.apply(feature))
260         CloseableIterator.single(feature)
261 
262       case Left(error) =>
263         ec.stats.failure()
264         def msg(verbose: Boolean): String = errorMessage(error, rawValues, verbose)
265         options.errorMode match {
266           case ErrorMode.LogErrors if logger.underlying.isDebugEnabled => logger.debug(msg(verbose = true), error.e)
267           case ErrorMode.LogErrors if logger.underlying.isInfoEnabled => logger.info(msg(verbose = false))
268           case ErrorMode.ReturnErrors => ec.errors.add(error.copy(e = new IOException(msg(verbose = true), error.e)))
269           case ErrorMode.RaiseErrors => throw new IOException(msg(verbose = true), error.e)
270           case _ => // no-op
271         }
272         CloseableIterator.empty
273     }
274   }

360     *
361     * @param field field to add
362     * @param fieldMap field lookup map
363     * @param dag graph

365   private def addDependencies[F <: Field](field: Field, fieldMap: Map[String, F], dag: Dag): Unit = {
366     if (!dag.contains(field)) {
367       val deps = field.transforms.toSeq.flatMap(_.dependencies(Set(field), fieldMap)).toSet
368       dag.put(field, deps)
369       deps.foreach(addDependencies(_, fieldMap, dag))
370     }
371   }

373   /**
374     * Returns vertices in topological order.
375     *
376     * Note: will cause an infinite loop if there are circular dependencies
377     *
378     * @param dag graph
379     * @return ordered vertices
380     */
381   private def topologicalOrder(dag: Dag): Array[Field] = {
382     val res = ArrayBuffer.empty[Field]
383     val remaining = new scala.collection.mutable.Queue[Field]() ++ dag.keys
384     while (remaining.nonEmpty) {
385       val next = remaining.dequeue()
386       if (dag(next).forall(res.contains)) {
387         res.append(next)
388       } else {
389         remaining.enqueue(next) // put at the back of the queue
390       }
391     }
392     res.toArray
393   }
