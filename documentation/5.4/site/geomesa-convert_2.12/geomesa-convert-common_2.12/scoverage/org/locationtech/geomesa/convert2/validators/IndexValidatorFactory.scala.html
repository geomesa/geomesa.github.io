23 /**
24   * Validator based on the indices used by the feature type. Currently only the x/z2 and x/z3 indices have
25   * input requirements. In addition, features that validate against the x/z3 index will also validate against
26   * the x/z2 index
27   */
28 class IndexValidatorFactory extends SimpleFeatureValidatorFactory {
29 
30   import IndexValidatorFactory._
31   import org.locationtech.geomesa.utils.geotools.RichSimpleFeatureType.RichSimpleFeatureType
32 
33   override val name: String = IndexValidatorFactory.Name
34 
35   override def apply(sft: SimpleFeatureType, metrics: ConverterMetrics, config: Option[String]): SimpleFeatureValidator =
36     apply(sft, config, Tags.empty())
37 
38   override def apply(sft: SimpleFeatureType, config: Option[String], tags: Tags): SimpleFeatureValidator = {
39     val geom = sft.getGeomIndex
40     val dtg = sft.getDtgIndex.getOrElse(-1)
41     val enabled = sft.getIndices.collect { case id if id.mode.write => id.name.toLowerCase(Locale.US) }
42     if (enabled.contains("z3") || enabled.contains("xz3") || (enabled.isEmpty && geom != -1 && dtg != -1)) {
43       val minDate = Date.from(BinnedTime.ZMinDate.toInstant)
44       val maxDate = Date.from(BinnedTime.maxDate(sft.getZ3Interval).toInstant)
45       new Z3Validator(Attribute(sft, geom), Attribute(sft, dtg), minDate, maxDate, tags)
46     } else if (enabled.contains("z2") || enabled.contains("xz2") || (enabled.isEmpty && geom != -1)) {
47       new Z2Validator(Attribute(sft, geom), tags)
48     } else {
49       NoValidator
50     }
51   }
52 }

64   /**
65    * Z2 validator
66    *
67    * @param geom geom
68    * @param tags metrics tags
69    */
70   private class Z2Validator(geom: Attribute, tags: Tags) extends SimpleFeatureValidator {
71 
72     private val success = successCounter("z2", geom.name, tags)
73     private val nullFailure = failureCounter("z2", geom.name, "null", tags)
74     private val boundsFailure = failureCounter("z2", geom.name, "invalid.bounds", tags)
75 
76     override def validate(sf: SimpleFeature): String = {
77       val g = sf.getAttribute(geom.i).asInstanceOf[Geometry]
78       if (g == null) {
79         nullFailure.increment()
80         "geometry is null"
81       } else if (!WholeWorldEnvelope.contains(g.getEnvelopeInternal)) {
82         boundsFailure.increment()
83         geomBoundsError(g)
84       } else {
85         success.increment()
86         null
87       }
88     }
89 
90     override def close(): Unit = {}
91   }

93   /**
94    * Z3 validator
95    *
96    * @param geom geom
97    * @param dtg dtg
98    * @param minDate min z3 date
99    * @param maxDate max z3 date
100    * @param tags metrics tags
101    */
102   private class Z3Validator(geom: Attribute, dtg: Attribute, minDate: Date, maxDate: Date, tags: Tags)
103       extends SimpleFeatureValidator {
104 
105     private val success = successCounter("z3", s"${geom.name},${dtg.name}", tags)
106     private val geomNullFailure = failureCounter("z3", geom.name, "null", tags)
107     private val geomBoundsFailure = failureCounter("z3", geom.name, "invalid.bounds", tags)
108     private val dtgNullFailure = failureCounter("z3", dtg.name, "null", tags)
109     private val dtgBoundsFailure = failureCounter("z3", dtg.name, "invalid.bounds", tags)
110 
111     private val dateBefore = s"date is before minimum indexable date (${GeoToolsDateFormat.format(Instant.ofEpochMilli(minDate.getTime))}):"
112     private val dateAfter = s"date is after maximum indexable date (${GeoToolsDateFormat.format(Instant.ofEpochMilli(maxDate.getTime))}):"
113 
114     override def validate(sf: SimpleFeature): String = {
115       val d = sf.getAttribute(dtg.i).asInstanceOf[Date]
116       val g = sf.getAttribute(geom.i).asInstanceOf[Geometry]
117       var error: String = null
118       if (g == null) {
119         geomNullFailure.increment()
120         error =  s"$error, geometry is null"
121       } else if (!WholeWorldEnvelope.contains(g.getEnvelopeInternal)) {
122         geomBoundsFailure.increment()
123         error =  s"$error, ${geomBoundsError(g)}"
124       }
125       if (d == null) {
126         dtgNullFailure.increment()
127         error = s"$error, date is null"
128       } else if (d.before(minDate)) {
129         dtgBoundsFailure.increment()
130         error = s"$error, $dateBefore ${GeoToolsDateFormat.format(Instant.ofEpochMilli(d.getTime))}"
131       } else if (d.after(maxDate)) {
132         dtgBoundsFailure.increment()
133         error = s"$error, $dateAfter ${GeoToolsDateFormat.format(Instant.ofEpochMilli(d.getTime))}"
134       }
135       if (error == null) {
136         success.increment()
137         null
138       } else {
139         error.substring(6) // trim off leading 'null, '
140       }
141     }
142 
143     override def close(): Unit = {}
144   }
