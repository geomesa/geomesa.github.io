18 /**
19   * Parses a json path string into a sequence of selectors. See https://github.com/jayway/JsonPath for examples.
20   *
21   * Does not support filter predicates.
22   */
23 object JsonPathParser {
24 
25   private val Parser = new JsonPathParser()
26 
27   @throws(classOf[ParsingException])
28   def parse(path: String, report: Boolean = true): Seq[PathElement] = {
29     if (path == null) {
30       throw new IllegalArgumentException("Path must not be null")
31     }
32     val runner = if (report) { ReportingParseRunner(Parser.Path) } else { BasicParseRunner(Parser.Path) }
33     val fixedPath = if (path.startsWith("$")) { path } else s"$$.$path"
34     val parsing = runner.run(fixedPath)
35     parsing.result.getOrElse {
36       throw new ParsingException(s"Invalid json path: ${ErrorUtils.printParseErrors(parsing)}")
37     }
38   }
39 
40   def print(path: Seq[PathElement], dollar: Boolean = true): String = {
41     require(path.nonEmpty, "Path must be non-empty")
42     if (dollar) {
43       path.mkString("$", "", "")
44     } else {
45       val string = path.mkString
46       if (string.charAt(0) == '.') {
47         // trim off leading self-selector to correspond to dot notation selection
48         string.substring(1)
49       } else {
50         string
51       }
52     }
53   }
54 
55   sealed trait PathElement
56 
57   // attribute: .foo or ['foo']
58   case class PathAttribute(name: String, bracketed: Boolean = false) extends PathElement {
59     override def toString: String = if (bracketed) { s"['$name']" } else { s".$name" }
60   }
61 
62   // enumerated index: [1]
63   case class PathIndex(index: Int) extends PathElement {
64     override def toString: String = s"[$index]"
65   }
66 
67   // enumerated indices: [1,2,5]
68   case class PathIndices(indices: Seq[Int]) extends PathElement {
69     override def toString: String = indices.mkString("[", ",", "]")
70   }
71 
72   // any attribute: .*
73   case object PathAttributeWildCard extends PathElement {
74     override val toString: String = ".*"
75   }
76 
77   // any index: [*]
78   case object PathIndexWildCard extends PathElement {
79     override val toString: String = "[*]"
80   }
81 
82   // deep scan: ..
83   case object PathDeepScan extends PathElement {
84     override val toString: String = ".."
85   }
86 
87   // path function: .min(), .max(), .avg(), .length()
88   // not implemented: stddev
89   case class PathFunction(function: JsonPathFunction) extends PathElement {
90     override def toString: String = s".$function()"
91   }
92 
93   object JsonPathFunction extends Enumeration {
94     type JsonPathFunction = Value
95     val min, max, avg, length = Value
96   }
97 }
98 
99 private class JsonPathParser extends BasicParser {
100 
101   // main parsing rule
102   def Path: Rule1[Seq[PathElement]] =
103     rule { "$" ~ zeroOrMore(Element) ~ optional(Function) ~~> ((e, f) => e ++ f.toSeq) ~ EOI }
104 
105   def Element: Rule1[PathElement] = rule {
106     Attribute | ArrayIndex | ArrayIndices | ArrayIndexRange | BracketedAttribute | AttributeWildCard | IndexWildCard | DeepScan
107   }
108 
109   def IndexWildCard: Rule1[PathElement] = rule { "[*]" ~ push(PathIndexWildCard) }
110 
111   def AttributeWildCard: Rule1[PathElement] = rule { ".*" ~ push(PathAttributeWildCard) }
112 
113   // we have to push the deep scan directly onto the stack as there is no forward matching and
114   // it's ridiculous trying to combine Rule1's and Rule2's
115   def DeepScan: Rule1[PathElement] = rule { "." ~ toRunAction(pushDeepScan) ~ (Attribute | BracketedAttribute | AttributeWildCard) }
116 
117   // note: this assumes that we are inside a zeroOrMore, which is currently the case
118   // the zeroOrMore will have pushed a single list onto the value stack - we append our value to that
119   private def pushDeepScan(context: Context[Any]): Unit =
120     context.getValueStack.push(PathDeepScan :: context.getValueStack.pop.asInstanceOf[List[_]])
121 
122   def ArrayIndex: Rule1[PathIndex] = rule { "[" ~ int ~ "]" ~~> PathIndex }
123 
124   def ArrayIndices: Rule1[PathIndices] =
125     rule { "[" ~ int ~ "," ~ oneOrMore(int, ",") ~ "]" ~~> ((n0, n) => PathIndices(n.+:(n0))) }
126 
127   def ArrayIndexRange: Rule1[PathIndices] =
128     rule { "[" ~ int ~ ":" ~ int ~ "]" ~~> ((n0, n1) => PathIndices(n0 until n1)) }
129 
130   def Attribute: Rule1[PathAttribute] = rule { "." ~ oneOrMore(char) ~> { (s) => PathAttribute(s) } ~ !"()" }
131 
132   def BracketedAttribute: Rule1[PathAttribute] =
133     rule { "[" ~ bracketedString ~~> { (s) => PathAttribute(s, bracketed = true) } ~ "]" }
134 
135   private def bracketedString: Rule1[String] = unquotedString | singleQuotedString
136 
137   def Function: Rule1[PathFunction] = rule {
138     "." ~ ("min" | "max" | "avg" | "length") ~> ((f) => PathFunction(JsonPathFunction.withName(f))) ~ "()"
139   }
