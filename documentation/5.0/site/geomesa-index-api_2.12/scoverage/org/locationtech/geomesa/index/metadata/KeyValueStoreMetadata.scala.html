17 /**
18   * Table-based metadata implementation for key-value stores. As with TableBasedMetadata, the metadata is
19   * persisted in a database table. The underlying table will be lazily created when required. Metadata values
20   * are cached with a configurable timeout to save repeated database reads.
21   *
22   * @tparam T type param
23   */
24 trait KeyValueStoreMetadata[T] extends TableBasedMetadata[T] {
25 
26   // separator used between type names and keys
27   val typeNameSeparator: Char = '~'
28 
29   def encodeRow(typeName: String, key: String): Array[Byte] =
30     KeyValueStoreMetadata.encodeRow(typeName, key, typeNameSeparator)
31 
32   override protected def write(typeName: String, rows: Seq[(String, Array[Byte])]): Unit =
33     write(rows.map { case (k, v) => (encodeRow(typeName, k), v) })
34 
35   override protected def delete(typeName: String, keys: Seq[String]): Unit =
36     delete(keys.map(k => encodeRow(typeName, k)))
37 
38   override protected def scanValue(typeName: String, key: String): Option[Array[Byte]] =
39     scanValue(encodeRow(typeName, key))
40 
41   override protected def scanValues(typeName: String, prefix: String): CloseableIterator[(String, Array[Byte])] = {
42     scanRows(Some(encodeRow(typeName, prefix))).flatMap { case (row, value) =>
43       try { CloseableIterator.single((decodeRow(row, typeNameSeparator)._2, value)) } catch {
44         case NonFatal(_) =>
45           logger.warn(s"Ignoring unexpected row in catalog table: ${new String(row, StandardCharsets.UTF_8)}")
46           CloseableIterator.empty
47       }
48     }
49   }
50 
51   override protected def scanKeys(): CloseableIterator[(String, String)] = {
52     scanRows(None).flatMap { case (row, _) =>
53       try { CloseableIterator.single(decodeRow(row, typeNameSeparator)) } catch {
54         case NonFatal(_) =>
55           logger.warn(s"Ignoring unexpected row in catalog table: ${new String(row, StandardCharsets.UTF_8)}")
56           CloseableIterator.empty
57       }
58     }
59   }
60 
61   /**
62     * Writes row/value pairs
63     *
64     * @param rows row/values
65     */
66   protected def write(rows: Seq[(Array[Byte], Array[Byte])]): Unit
67 
68   /**
69     * Deletes multiple rows
70     *
71     * @param rows rows
72     */
73   protected def delete(rows: Seq[Array[Byte]])
74 
75   /**
76     * Reads a value from the underlying table
77     *
78     * @param row row
79     * @return value, if it exists
80     */
81   protected def scanValue(row: Array[Byte]): Option[Array[Byte]]
82 
83   /**
84     * Reads row keys from the underlying table
85     *
86     * @param prefix row key prefix
87     * @return matching row keys and values
88     */
89   protected def scanRows(prefix: Option[Array[Byte]]): CloseableIterator[(Array[Byte], Array[Byte])]
90 }

92 object KeyValueStoreMetadata {
93 
94   def encodeRow(typeName: String, key: String, separator: Char): Array[Byte] = {
95     // escaped to %U+XXXX unicode since decodeRow splits by separator
96     val escape = s"%${"U+%04X".format(separator.toInt)}"
97     s"${typeName.replace(separator.toString, escape)}$separator$key".getBytes(StandardCharsets.UTF_8)
98   }
99 
100   def decodeRow(row: Array[Byte], separator: Char): (String, String) = {
101     // escaped to %U+XXXX unicode since decodeRow splits by separator
102     val escape = s"%${"U+%04X".format(separator.toInt)}"
103     val all = new String(row, StandardCharsets.UTF_8)
104     val split = all.indexOf(separator)
105     (all.substring(0, split).replace(escape, separator.toString), all.substring(split + 1, all.length))
106   }
