21 /**
22  * Serializes into bson (http://bsonspec.org/). Note this is a limited form of bson that only matches
23  * the existing json types - does not cover the bson extensions like binary data, etc. Also note endianness,
24  * etc might not match the spec 100%.
25  *
26  * The bson serialized value is preceded by a single byte, `\x00` to indicate null, or `\x01` to indicate non-null.
27  * Additionally, non-document top-level values are supported, indicated with the prefix byte `\x02`. Top-level,
28  * non-document values are encoded as a jsonb `element` with an empty name.
29  *
30  * Reduced BSON spec - only native JSON elements supported:
31  *
32  * byte    1 byte (8-bits)
33  * int32   4 bytes (32-bit signed integer, two's complement)
34  * int64   8 bytes (64-bit signed integer, two's complement)
35  * double  8 bytes (64-bit IEEE 754-2008 binary floating point)
36  *
37  * document  ::= int32 e_list "\x00" BSON Document. int32 is the total number of bytes comprising the document.
38  * e_list    ::= element e_list
39  *           |	""
40  * element   ::= "\x01" e_name double	64-bit binary floating point
41  *           | "\x02" e_name string	UTF-8 string
42  *           | "\x03" e_name document	Embedded document
43  *           |	"\x04" e_name document	Array
44  *           |	"\x08" e_name "\x00"	Boolean "false"
45  *           |	"\x08" e_name "\x01"	Boolean "true"
46  *           |	"\x09" e_name int64	UTC datetime
47  *           |	"\x0A" e_name	Null value
48  *           |	"\x10" e_name int32	32-bit integer
49  *           |	"\x11" e_name int64	Timestamp
50  *           |	"\x12" e_name int64	64-bit integer
51  * e_name    ::= cstring	Key name
52  * string    ::= int32 (byte*) "\x00"	String - The int32 is the number bytes in the (byte*) + 1 (for the trailing '\x00').
53  *                                              The (byte*) is zero or more UTF-8 encoded characters.
54  * cstring   ::= (byte*) "\x00"	Zero or more modified UTF-8 encoded characters followed by '\x00'. The (byte*)
55  *                               MUST NOT contain '\x00', hence it is not full UTF-8.
56  *
57  * Note:
58  *   Array - The document for an array is a normal BSON document with integer values for the keys,
59  *   starting with 0 and continuing sequentially. For example, the array ['red', 'blue'] would be
60  *   encoded as the document {'0': 'red', '1': 'blue'}. The keys must be in ascending numerical order.
61  */

155   /**
156     * Deserialize the result of a json-path. The input should be pointing to the start of
157     * the bytes written by `serialize`. There is no guarantee where the input will be
158     * pointing after completion.
159     *
160     * If the path selects leaf elements, they will be returned as primitives. If the path
161     * selects objects, they will be returned as json strings. If more than one item is
162     * selected, they will be returned in a Seq. If nothing is selected, it will return null.
163     *
164     * @param in input, pointing to the start of the json object
165     * @param path json path to evaluate
166     * @return result of the path, if any
167     */
168   def deserialize(in: Input, path: Seq[PathElement]): Any = {
169     if (path.isEmpty) {
170       deserializeAndRender(in)
171     } else {
172       try {
173         // collection of (type, position) for our matches so far
174         var matches: Seq[(Byte, Int)] = in.readByte match {
175           case BooleanTrue => Seq((DocByte, in.position()))
176           case BooleanFalse => Seq.empty // null
177           case NonDoc =>
178             val switch = in.readByte()
179             in.readName()
180             Seq((switch, in.position()))
181         }
182         var function: Option[JsonPathFunction] = None
183 
184         // collect types and indices for each match at each level
185         val paths = path.iterator
186         while (paths.hasNext && matches.nonEmpty) {
187           paths.next match {
188             case PathAttribute(name: String, _)       => matches = matchPathAttribute(in, matches, Some(name))
189             case PathAttributeWildCard                => matches = matchPathAttribute(in, matches, None)
190             case PathIndex(index: Int)                => matches = matchPathIndex(in, matches, Some(Seq(index)))
191             case PathIndices(indices: Seq[Int])       => matches = matchPathIndex(in, matches, Some(indices))
192             case PathIndexWildCard                    => matches = matchPathIndex(in, matches, None)
193             case PathDeepScan                         => matches = matchDeep(in, matches)
194             case PathFunction(f)                      => function = Some(f) // only 1 trailing function allowed by parser spec
195           }
196         }
197 
198         val values = matches.map { case (t, p) => readPathValue(in, t, p) }
199         val mapped = function match {
200           case None    => values
201           case Some(f) => values.map(applyPathFunction(f, _))
202         }
203 
204         if (mapped.isEmpty) {
205           null
206         } else if (values.length == 1) {
207           mapped.head
208         } else {
209           mapped
210         }
211       } catch {
212         case NonFatal(e) => logger.error("Error reading serialized kryo json", e); null
213       }
214     }
215   }

225   // write a document without a name - used for the outer-most object which doesn't have a key
226   private def writeDocument(out: Output, value: JObject): Unit = {
227     val start = out.position()
228     // write a placeholder that we will overwrite when we go back to write total length
229     // note: don't just modify position, as that doesn't expand the buffer correctly
230     out.writeInt(0)
231     value.obj.foreach { case (name, elem) => writeValue(out, name, elem) }
232     out.writeByte(TerminalByte) // marks the end of our object
233     // go back and write the total length
234     val end = out.position()
235     out.setPosition(start)
236     out.writeInt(end - start)
237     out.setPosition(end)
238   }

311   private def readDocument(in: Input): JObject = {
312     val end = in.position() + in.readInt() - 1 // last byte is the terminal byte
313     val elements = scala.collection.mutable.ArrayBuffer.empty[JField]
314     while (in.position() < end) {
315       elements.append(readValue(in))
316     }
317     in.skip(1) // skip over terminal byte
318     JObject(elements.toList)
319   }

321   private def readValue(in: Input): JField = {
322     val switch = in.readByte()
323     val name = in.readName()
324     val value = switch match {
325       case StringByte   => JString(readString(in))
326       case DocByte      => readDocument(in)
327       case ArrayByte    => readArray(in)
328       case DoubleByte   => JDouble(in.readDouble())
329       case IntByte      => JInt(in.readInt())
330       case LongByte     => JInt(in.readLong())
331       case NullByte     => JNull
332       case BooleanByte  => JBool(readBoolean(in))
333     }
334     JField(name, value)
335   }

386   /**
387     * Given a set of input nodes defined by their offsets, return a list of matching nodes
388     * which match the predicate. Input nodes must be either arrays or objects
389     *
390     * Note: predicate must consume the 'name' from the input stream
391     *
392     * @param in input
393     * @param positions input node offsets - must be arrays or objects
394     * @param nameConsumingPredicate predicate to match - must consume the object name from the input stream
395     * @return matching node types and offsets
396     */
397   private def matchObjectPath(in: Input, positions: Seq[Int], nameConsumingPredicate: () => Boolean): Seq[(Byte, Int)] = {
398     val elements = scala.collection.mutable.ArrayBuffer.empty[(Byte, Int)]
399     positions.foreach { position =>
400       in.setPosition(position)
401       val end = position + in.readInt() - 1 // last byte is the terminal byte
402       while (in.position() < end) {
403         val switch = in.readByte()
404         if (nameConsumingPredicate()) {
405           elements.append((switch, in.position()))
406         }
407         switch match {
408           case StringByte   => skipString(in)
409           case DocByte      => skipDocument(in)
410           case ArrayByte    => skipDocument(in) // arrays are stored as docs
411           case DoubleByte   => in.skip(8)
412           case IntByte      => in.skip(4)
413           case LongByte     => in.skip(8)
414           case NullByte     => // no-op
415           case BooleanByte  => skipBoolean(in)
416         }
417       }
418     }
419     elements.toSeq
420   }

422   /**
423     * Given a set of input nodes defined by their types and offsets, returns all nodes contained by the inputs
424     *
425     * @param in input
426     * @param positions input node types and offsets
427     * @return matching node types and offsets
428     */
429   private def matchDeep(in: Input, positions: Seq[(Byte, Int)]): Seq[(Byte, Int)] = {
430     val toSearch = scala.collection.mutable.Queue[(Byte, Int)](positions: _*)
431     val elements = scala.collection.mutable.ArrayBuffer.empty[(Byte, Int)]
432     def predicate(): Boolean = { in.skipName(); true }
433 
434     while (toSearch.nonEmpty) {
435       val (typ, position) = toSearch.dequeue()
436       elements.append((typ, position))
437       in.setPosition(position)
438       typ match {
439         case DocByte | ArrayByte =>
440           matchObjectPath(in, Seq(position), predicate).foreach{ case(val1: Byte, val2:Int) => toSearch.enqueue((val1, val2))}
441         case StringByte | DoubleByte | IntByte | LongByte | NullByte | BooleanByte => // no-op
442       }
443     }
444     elements.toSeq
445   }

490   /**
491     * Attempts to apply a json path function to a value. Most functions
492     * require that numeric arrays have been selected.
493     *
494     * @param function function to evaluate
495     * @param value value to apply the function on
496     * @return
497     */
498   private def applyPathFunction(function: JsonPathFunction, value: Any): Any = {
499     def toNum(v: Any): Double = v match {
500       case n: Number => n.doubleValue
501       case null => 0.0
502       case n => n.toString.toDouble
503     }
504     value match {
505       case s: Seq[_] =>
506         function match {
507           case JsonPathFunction.length => s.length
508           case JsonPathFunction.avg    => s.map(toNum).sum / s.length
509           case JsonPathFunction.min    => s.map(toNum).min
510           case JsonPathFunction.max    => s.map(toNum).max
511         }
512       case s: String if function == JsonPathFunction.length => s.length
513       case _ => null
514     }
515   }

525   private implicit class RichInput(val in: Input) extends AnyRef {
526     def readName(): String = {
527       var buffer = nameBuffers.get()
528       var i = 0
529       var byte: Byte = in.readByte()
530       while (byte != TerminalByte) {
531         if (i == buffer.length) {
532           // expand our cached buffer to accommodate the name
533           val copy = Array.ofDim[Byte](buffer.length * 2)
534           System.arraycopy(buffer, 0, copy, 0, i)
535           buffer = copy
536           nameBuffers.set(buffer)
537         }
538         buffer(i) = byte
539         i += 1
540         byte = in.readByte()
541       }
542       new String(buffer, 0, i, StandardCharsets.UTF_8)
543     }
544 
545     def skipName(): Unit = while (in.readByte() != TerminalByte) {}
546   }
