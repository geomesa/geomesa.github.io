103   def parse(params: java.util.Map[String, _], namespace: String): LambdaConfig = {
104     val brokers = BrokersParam.lookup(params)
105     val expiry = ExpiryParam.lookup(params)
106 
107     val partitions = PartitionsParam.lookup(params).intValue
108     val consumers = ConsumersParam.lookup(params).intValue
109     val persist = PersistParam.lookup(params).booleanValue
110 
111     val bootstrap = Map("bootstrap.servers" -> brokers)
112     val consumerConfig = ConsumerOptsParam.lookupOpt(params).map(_.asScala.toMap).getOrElse(Map.empty) ++ bootstrap
113     val producerConfig = ProducerOptsParam.lookupOpt(params).map(_.asScala.toMap).getOrElse(Map.empty) ++ bootstrap
114 
115     val zk = ZookeepersParam.lookup(params)
116     val zkNamespace = s"gm_lambda_$namespace"
117 
118     val offsetManager = OffsetManagerParam.lookupOpt(params).getOrElse(new ZookeeperOffsetManager(zk, zkNamespace))
119 
120     LambdaConfig(zk, zkNamespace, producerConfig, consumerConfig, partitions, consumers, expiry, persist, offsetManager)
121   }
