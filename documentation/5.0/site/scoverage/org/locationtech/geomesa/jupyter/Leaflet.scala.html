124     def simpleFeatureToGeoJSON(sf: SimpleFeature) = {
125       import scala.collection.JavaConverters._
126       s"""
127          |{
128          |    "type": "Feature",
129          |    "properties": {
130          |        ${sf.getType.getAttributeDescriptors.asScala.zip(sf.getAttributes.asScala)
131                       .filter{case(a,b) => b != null && !b.isInstanceOf[Geometry]}
132                       .map { case (d, a) => propToJson(d, a) }.mkString(sep =",\n")
133                    }

244   def buildMap(layers: Seq[GeoRenderable], center: (Double, Double) = (0,0), zoom: Int = 8, path: Option[String] = None) = {
245 
246     val leafletCss = path match {
247       case Some(p) => s"""<link rel="stylesheet" href="$p/leaflet.css" />"""
248       case None => leafletCssCdn
249     }
250 
251     val leafletJs = path match {
252       case Some(p) => s"""<script src="$p/leaflet.js"></script>"""
253       case None => leafletJsCdn
254     }
255 
256     val leafletWmsJs = path match {
257       case Some(p) => s"""<script src="$p/leaflet.wms.js"></script>"""
258       case None => leafletWmsJsCdn
259     }
260 
261     val countriesJs = path match {
262       case Some(p) => s"""<script src="$p/countries.geo.js"></script>"""
263       case None => "" // data not easily hosted anywhere
264     }
265 
266     s"""
267        |<html>
268        |  <head>
269        |    $leafletCss
270        |    $leafletJs
271        |    $leafletWmsJs
272        |    $countriesJs
273        |  </head>
274        |  <body>
275        |    <div id='map' style="width:100%;height:500px"></div>
276        |    <script>
277        |
278        |      //'map' is the id of the map
279        |      var map = L.map('map', {
280        |        crs: L.CRS.EPSG4326,
281        |        center: [${center._1}, ${center._2}],
282        |        zoom: ${zoom}
283        |      });
284        |
285        |      // Only add country boundaries if the geojson data loaded ok from the script
286        |      if(typeof worldMap !== 'undefined') {
287        |        // Initialize the Base Layer... Loaded from GeoJson
288        |        var basestyle = {"color": "#717171", "weight": 2, "opacity": 1.0};
289        |        var base = L.geoJson(worldMap, basestyle);
290        |        map.addLayer(base);
291        |      }
292        |
293        |      ${layers.map(_.render).mkString(sep = "\n")}
294        |
295        |    </script>
296        |  </body>
297        |</html>
298      """.stripMargin
299   }

301   def render(layers: Seq[GeoRenderable], center: (Double, Double) = (0,0), zoom: Int = 8, path: Option[String] = None) = {
302     val id = new RandomStringGenerator.Builder().withinRange('0', 'z').filteredBy(ASCII_ALPHA_NUMERALS).build().generate(5)
303     s"""
304        |<iframe id="${id}" sandbox="allow-scripts allow-same-origin" style="border:none;width:100%;height:520px" srcdoc="${xml.Utility.escape(buildMap(layers, center, zoom, path))}"></iframe>
305        |<script>
306        |  if(typeof resizeIFrame != 'function') {
307        |    function resizeIFrame(el, k) {
308        |      el.style.height = el.contentWindow.document.body.scrollHeight + 'px';
309        |      el.style.width = '100%';
310        |      if(k<=3) { setTimeout(function() { resizeIFrame(el, k+1)}, 1000) };
311        |    }
312        |  }
313        |  $$().ready(function() {
314        |    resizeIFrame($$('#$id').get(0), 1);
315        |  });
316        |</script>
317      """.stripMargin
318   }
