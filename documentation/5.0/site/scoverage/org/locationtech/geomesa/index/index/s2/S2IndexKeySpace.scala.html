34 class S2IndexKeySpace(val sft: SimpleFeatureType, val sharding: ShardStrategy, geomField: String)
35   extends IndexKeySpace[S2IndexValues, Long] {
36 
37   require(classOf[Point].isAssignableFrom(sft.getDescriptor(geomField).getType.getBinding),
38     s"Expected field $geomField to have a point binding, but instead it has: " +
39       sft.getDescriptor(geomField).getType.getBinding.getSimpleName)
40 
41   private val sfc: S2SFC =
42     S2SFC(
43       QueryProperties.S2MinLevel,
44       QueryProperties.S2MaxLevel,
45       QueryProperties.S2LevelMod,
46       QueryProperties.S2MaxCells
47     )
48 
49   private val geomIndex: Int = sft.indexOf(geomField)
50 
51   /**
52     * The attributes used to create the index keys
53     *
54     * @return
55     */
56   override val attributes: Seq[String] = Seq(geomField)
57 
58   /**
59     * Length of an index key. If static (general case), will return a Right with the length. If dynamic,
60     * will return Left with a function to determine the length from a given (row, offset, length)
61     *
62     * @return
63     */
64   override val indexKeyByteLength: Right[(Array[Byte], Int, Int) => Int, Int] = Right(8 + sharding.length)
65 
66   /**
67     * Table sharing
68     *
69     * @return
70     */
71   override val sharing: Array[Byte] = Array.empty
72 
73   /**
74     * Index key from the attributes of a simple feature
75     *
76     * @param writable simple feature with cached values
77     * @param tier    tier bytes
78     * @param id      feature id bytes
79     * @param lenient if input values should be strictly checked, default false
80     * @return
81     */
82   override def toIndexKey(writable: WritableFeature,
83                           tier: Array[Byte],
84                           id: Array[Byte],
85                           lenient: Boolean): RowKeyValue[Long] = {
86     val geom = writable.getAttribute[Point](geomIndex)
87     if (geom == null) {
88       throw new IllegalArgumentException(s"Null geometry in feature ${writable.feature.getID}")
89     }
90     val s = try { sfc.index(geom.getX, geom.getY, lenient) } catch {
91       case NonFatal(e) => throw new IllegalArgumentException(s"Invalid s value from geometry: $geom", e)
92     }
93     val shard = sharding(writable)
94 
95     // create the byte array - allocate a single array up front to contain everything
96     // ignore tier, not used here
97     val bytes = Array.ofDim[Byte](shard.length + 8 + id.length)
98 
99     if (shard.isEmpty) {
100       ByteArrays.writeLong(s, bytes)
101       System.arraycopy(id, 0, bytes, 8, id.length)
102     } else {
103       bytes(0) = shard.head // shard is only a single byte
104       ByteArrays.writeLong(s, bytes, 1)
105       System.arraycopy(id, 0, bytes, 9, id.length)
106     }
107 
108     SingleRowKeyValue(bytes, sharing, shard, s, tier, id, writable.values)
109   }
110 
111   /**
112     * Extracts values out of the filter used for range and push-down predicate creation
113     *
114     * @param filter  query filter
115     * @param explain explainer
116     * @return
117     */
118   override def getIndexValues(filter: Filter, explain: Explainer): S2IndexValues = {
119 
120     val geometries: FilterValues[Geometry] = {
121       val extracted = FilterHelper.extractGeometries(filter, geomField) // intersect since we have points
122       if (extracted.nonEmpty) { extracted } else { FilterValues(Seq(WholeWorldPolygon)) }
123     }
124 
125     explain(s"Geometries: $geometries")
126 
127     if (geometries.disjoint) {
128       explain("Non-intersecting geometries extracted, short-circuiting to empty query")
129       return S2IndexValues(sfc, geometries, Seq.empty)
130     }
131 
132     // compute our ranges based on the coarse bounds for our query
133     val xy: Seq[(Double, Double, Double, Double)] = {
134       val multiplier = QueryProperties.PolygonDecompMultiplier.toInt.get
135       val bits = QueryProperties.PolygonDecompBits.toInt.get
136       geometries.values.flatMap(GeometryUtils.bounds(_, multiplier, bits))
137     }
138 
139     S2IndexValues(sfc, geometries, xy)
140   }
141 
142   /**
143     * Creates ranges over the index keys
144     *
145     * @param values     index values @see getIndexValues
146     * @param multiplier hint for how many times the ranges will be multiplied. can be used to
147     *                   inform the number of ranges generated
148     * @return
149     */
150   override def getRanges(values: S2IndexValues, multiplier: Int): Iterator[ScanRange[Long]] = {
151     val S2IndexValues(_, geoms, xy) = values
152     if (geoms.disjoint) {
153       Iterator.empty
154     } else {
155       // note: `target` will always be Some, as ScanRangesTarget has a default value
156       val target = QueryProperties.ScanRangesTarget.option.map(t => math.max(1, t.toInt / multiplier))
157       sfc.ranges(xy, -1, target).iterator.map(r => BoundedRange(r.lower, r.upper))
158     }
159   }
160 
161   /**
162     * Creates bytes from ranges
163     *
164     * @param ranges typed scan ranges. @see `getRanges`
165     * @param tier   will the ranges have tiered ranges appended, or not
166     * @return
167     */
168   override def getRangeBytes(ranges: Iterator[ScanRange[Long]], tier: Boolean): Iterator[api.ByteRange] = {
169     if (sharding.length == 0) {
170       ranges.map {
171         case BoundedRange(lo, hi) => BoundedByteRange(ByteArrays.toBytes(lo), ByteArrays.toBytesFollowingPrefix(hi))
172         case r => throw new IllegalArgumentException(s"Unexpected range type $r")
173       }
174     } else {
175       ranges.flatMap {
176         case BoundedRange(lo, hi) =>
177           val lower = ByteArrays.toBytes(lo)
178           val upper = ByteArrays.toBytesFollowingPrefix(hi)
179           sharding.shards.map(p => BoundedByteRange(ByteArrays.concat(p, lower), ByteArrays.concat(p, upper)))
180 
181         case r => throw new IllegalArgumentException(s"Unexpected range type $r")
182       }
183     }
184   }
185 
186   /**
187     * Determines if the ranges generated by `getRanges` are sufficient to fulfill the query,
188     * or if additional filtering needs to be done
189     *
190     * @param config data store config
191     * @param values index values @see getIndexValues
192     * @param hints  query hints
193     * @return
194     */
195   override def useFullFilter(values: Option[S2IndexValues],
196                              config: Option[GeoMesaDataStoreFactory.GeoMesaDataStoreConfig],
197                              hints: Hints): Boolean = {
198     // if the user has requested strict bounding boxes, we apply the full filter
199     // if the spatial predicate is rectangular (e.g. a bbox), the index is fine enough that we
200     // don't need to apply the filter on top of it. this may cause some minor errors at extremely
201     // fine resolutions, but the performance is worth it
202     // if we have a complicated geometry predicate, we need to pass it through to be evaluated
203     val looseBBox = Option(hints.get(LOOSE_BBOX)).map(Boolean.unbox).getOrElse(config.forall(_.queries.looseBBox))
204     lazy val simpleGeoms = values.toSeq.flatMap(_.geometries.values).forall(GeometryUtils.isRectangular)
205 
206     !looseBBox || !simpleGeoms
207   }
208 
209 }

219     new S2IndexKeySpace(sft, shards, attributes.head)
220   }
