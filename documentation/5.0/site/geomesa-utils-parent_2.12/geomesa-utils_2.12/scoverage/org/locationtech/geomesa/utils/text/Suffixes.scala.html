20   object Time extends LazyLogging {

22     // noinspection ScalaDeprecation
23     def duration(s: String): Try[Duration] =
24       Try(Duration(s))
25           .orElse(Try(Duration(s.toLong, TimeUnit.MILLISECONDS)))
26           .orElse(jodaMinute(s))
27 
28     def millis(s: String): Try[Long]  = duration(s).map(_.toMillis)
29     def seconds(s: String): Try[Long] = duration(s).map(_.toMillis * 1000)
30     def minutes(s: String): Try[Long] = duration(s).map(_.toMinutes)
31     def hours(s: String): Try[Long]   = duration(s).map(_.toHours)
32     def days(s: String): Try[Long]    = duration(s).map(_.toDays)
33 
34     // provide back compatibility with joda period parsing, which accepts 'm' for minutes
35     @deprecated("joda parsing")
36     private def jodaMinute(s: String): Try[Duration] = {
37       if (s == null) { Failure(new NullPointerException()) } else {
38         val replaced = s.replaceAll("m", "min") // scala parsing requires 'min', 'mins', 'minute' or 'minutes'
39         val res = Try(Duration(replaced))
40         if (res.isSuccess) {
41           logger.warn("Parsed duration using deprecated minute parsing. Please update duration value: " +
42               s"'$s' successfully parsed as '$replaced'")
43         }
44         res
45       }
46     }
47   }

49   object Memory extends LazyLogging {
50 
51     private val memPattern = Pattern.compile("(\\d+)([kmgt]?)b?")
52 
53     def bytes(s: String): Try[Long] = {
54       val m = memPattern.matcher(s.toLowerCase.trim)
55       if (m.matches() && m.groupCount() == 2) {
56         Try(m.group(1).toLong).flatMap { num =>
57           val mult: Long = m.group(2) match {
58             case "k" => 1024L
59             case "m" => 1024L * 1024L
60             case "g" => 1024L * 1024L * 1024L
61             case "t" => 1024L * 1024L * 1024L * 1024L
62             case _   => 1L
63           }
64           val res = num * mult
65           if (res > 0) {
66             Success(res)
67           } else {
68             Failure(new ArithmeticException(s"Arithmetic overflow parsing '$s'"))
69           }
70         } recoverWith {
71           case e => Failure(new NumberFormatException(s"Error parsing memory property from input '$s': ${e.toString}"))
72         }
73       } else {
74         Failure(new NumberFormatException(s"Unable to match memory pattern from input '$s'"))
75       }
76     }
77   }
