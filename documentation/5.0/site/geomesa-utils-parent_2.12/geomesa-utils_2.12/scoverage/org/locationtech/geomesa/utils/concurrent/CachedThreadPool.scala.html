17 /**
18  * Executor service that will grow up to the number of threads specified.
19  *
20  * Assertion: core java executors offer 3 features that we want, but you can only use two of them at once:
21  *   1. Add threads as needed but re-use idle threads if available
22  *   2. Limit the total number of threads available
23  *   3. Accept and queue tasks once the thread limit is reached
24  *
25  * This executor allows us to do all three. It is backed by an unlimited, cached thread pool that will handle
26  * re-use and expiration of threads, while it tracks the number of threads used. It is fairly lightweight, and
27  * can be instantiated for a given task and then shutdown after use.
28  *
29  * @param maxThreads max threads to use at once
30  */
31 class CachedThreadPool(maxThreads: Int) extends AbstractExecutorService with LazyLogging {
32 
33   @volatile
34   private var available = maxThreads
35   private val queue = new java.util.LinkedList[TrackableFutureTask[_]]()
36   private val tasks = new java.util.HashSet[TrackableFutureTask[_]]()
37   private val stopped = new AtomicBoolean(false)
38   private val lock = new ReentrantLock()
39   private val done = lock.newCondition()
40 
41   override def shutdown(): Unit = stopped.set(true)
42 
43   override def shutdownNow(): java.util.List[Runnable] = {
44     stopped.set(true)
45     lock.lock()
46     try {
47       val waiting = new java.util.ArrayList[Runnable](queue)
48       queue.clear()
49       // copy the running tasks to prevent concurrent modification errors in the synchronous cancel call
50       val running = new java.util.ArrayList[Future[_]](tasks).iterator()
51       while (running.hasNext) {
52         running.next.cancel(true)
53       }
54       waiting
55     } finally {
56       lock.unlock()
57     }
58   }
59 
60   override def isShutdown: Boolean = stopped.get
61 
62   override def isTerminated: Boolean = stopped.get && available == maxThreads // should be safe to read a volatile primitive
63 
64   override def awaitTermination(timeout: Long, unit: TimeUnit): Boolean = {
65     lock.lock()
66     try {
67       if (isTerminated) { true } else {
68         done.await(timeout, unit)
69         isTerminated
70       }
71     } finally {
72       lock.unlock()
73     }
74   }
75 
76   override def execute(command: Runnable): Unit = {
77     if (stopped.get) {
78       throw new RejectedExecutionException("Trying to execute a task but executor service is shut down")
79     }
80     val task = command match {
81       case t: TrackableFutureTask[_] => t
82       case c => newTaskFor(c, null)
83     }
84     runOrQueueTask(task)
85   }
86 
87   private def runOrQueueTask(task: TrackableFutureTask[_]): Unit = {
88     lock.lock()
89     try {
90       if (available > 0) {
91         // note that we could fairly easily create a global thread limit by backing this with a different pool
92         try { CachedThreadPool.pool.execute(task) } catch {
93           case e: RejectedExecutionException =>
94             // we still need to execute the task to fulfill the executor API
95             logger.warn(
96               "CachedThreadPool rejected queued task (likely due to shutdown)," +
97                 s"creating new single thread executor: $task: $e")
98             val pool = Executors.newSingleThreadExecutor()
99             pool.execute(task)
100             pool.shutdown()
101         }
102         available -= 1
103         tasks.add(task)
104       } else {
105         queue.offer(task) // unbounded queue so should always succeed
106       }
107     } finally {
108       lock.unlock()
109     }
110   }
111 
112   override protected def newTaskFor[T](runnable: Runnable, value: T): TrackableFutureTask[T] =
113     new TrackableFutureTask[T](runnable, value)
114 
115   class TrackableFutureTask[T](runnable: Runnable, result: T) extends FutureTask[T](runnable, result) {
116     override def done(): Unit = {
117       lock.lock()
118       try {
119         available += 1
120         tasks.remove(this)
121         val next = queue.poll()
122         if (next != null) {
123           runOrQueueTask(next) // note: this may briefly use more than maxThreads as this thread finishes up
124         } else if (isTerminated) {
125           CachedThreadPool.this.done.signalAll()
126         }
127       } finally {
128         lock.unlock()
129       }
130     }
131 
132     override def toString: String = s"TrackableFutureTask[$runnable]"
133   }
134 }
