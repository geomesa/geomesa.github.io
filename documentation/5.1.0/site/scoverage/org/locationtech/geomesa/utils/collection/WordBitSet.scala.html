13 /**
14   * A bit set backed by ints (to reduce memory footprint for small sizes).
15   *
16   * Provides access to the underlying words, for low-level synchronization. Not thread safe.
17   */
18 trait WordBitSet {
19 
20   /**
21     * Gets the word for the given value. The word can be synchronized on for any operations on the given value.
22     *
23     * @param value value
24     * @return word
25     */
26   def word(value: Int): Word
27 }
28 
29 object WordBitSet {
30 
31   // use `>> 5` instead of `/ 32`
32   private final val Divisor = 5
33 
34   /**
35     * Create a bit set capable of holding numbers up to `length`
36     *
37     * @param length max int to be stored in the bit set
38     * @return
39     */
40   def apply(length: Int): WordBitSet = {
41     IntBitSet.size(length) match {
42       case 1 => new WordBitSet32()
43       case 2 => new WordBitSet64()
44       case 3 => new WordBitSet96()
45       case 4 => new WordBitSet128()
46       case 5 => new WordBitSet160()
47       case 6 => new WordBitSet192()
48       case n => new WordBitSetN(n)
49     }
50   }
51 
52   /**
53     * A single word in the bit set
54     *
55     * @param mask the mask for this word
56     */
57   class Word(private var mask: Int = 0) {
58 
59     /**
60       * Checks whether the value is contained in the set or not
61       *
62       * @param value value to check
63       * @return true if contains, false otherwise
64       */
65     def contains(value: Int): Boolean = (mask & (1 << value)) != 0
66 
67     /**
68       * Adds the value to the set, if it is not present. Note that implementations may be constructed with
69       * a fixed capacity - attempting to add a value outside the capacity will result in undefined behavior
70       *
71       * @param value value to add
72       * @return true if value was added, false if value was already present
73       */
74     def add(value: Int): Boolean = {
75       val updated = mask | (1 << value)
76       if (updated == mask) { false } else {
77         mask = updated
78         true
79       }
80     }
81 
82     /**
83       * Removes the value from the set, if it is present
84       *
85       * @param value value to remove
86       * @return true if value was removed, false if value was not present
87       */
88     def remove(value: Int): Boolean = {
89       val updated = mask & ~(1 << value)
90       if (updated == mask) { false } else {
91         mask = updated
92         true
93       }
94     }
95   }
96 
97   /**
98     * Bit set for tracking values < 32
99     */
100   class WordBitSet32 extends WordBitSet {
101 
102     private val word0 = new Word()
103 
104     override def word(value: Int): Word =  {
105       value >> Divisor match {
106         case 0 => word0
107         case _ => throw new ArrayIndexOutOfBoundsException(value)
108       }
109     }
110   }
