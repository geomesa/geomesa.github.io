22 class FileBasedMetadataFactory extends StorageMetadataFactory {
23 
24   override def name: String = FileBasedMetadata.MetadataType
25 
26   /**
27     * Loads a metadata instance from an existing root. The metadata info is persisted in a `metadata.json`
28     * file under the root path.
29     *
30     * Will return a cached instance, if available. If a previous check was made to load a file from this root,
31     * and the file did not exist, will not re-attempt to load it until after a configurable timeout
32     *
33     * @see `org.locationtech.geomesa.fs.storage.common.utils.PathCache#CacheDurationProperty()`
34     * @param context file context
35     * @return
36     **/
37   override def load(context: FileSystemContext): Option[FileBasedMetadata] = {
38     val json = MetadataJson.readMetadata(context)
39     // note: do this after loading the json to allow for old metadata transition
40     val cached = FileBasedMetadataFactory.cached(context, json.getOrElse(FileBasedMetadata.LegacyOptions).options)
41     json match {
42       case Some(m) if m.name.equalsIgnoreCase(name) =>
43         cached.orElse(throw new IllegalArgumentException(s"Could not load metadata at root '${context.root.toUri}'"))
44 
45       case None if cached.isDefined =>
46         // a file-based metadata impl exists, but was created with an older version
47         // create a config file pointing to it, and register that
48         MetadataJson.writeMetadata(context, FileBasedMetadata.LegacyOptions)
49         cached
50 
51       case _ => None
52     }
53   }
54 
55   override def create(context: FileSystemContext, config: Map[String, String], meta: Metadata): FileBasedMetadata = {
56     val sft = namespaced(meta.sft, context.namespace)
57     // load the partition scheme first in case it fails
58     PartitionSchemeFactory.load(sft, meta.scheme)
59     val renderer = config.get(Config.RenderKey).map(MetadataConverter.apply).getOrElse(RenderCompact)
60     MetadataJson.writeMetadata(context, NamedOptions(name, config + (Config.RenderKey -> renderer.name)))
61     FileBasedMetadataFactory.write(context.fc, context.root, meta)
62     val directory = new Path(context.root, FileBasedMetadataFactory.MetadataDirectory)
63     val metadata = new FileBasedMetadata(context.fc, directory, sft, meta, renderer)
64     FileBasedMetadataFactory.cache.put(FileBasedMetadataFactory.key(context), metadata)
65     metadata
66   }
67 }

69 object FileBasedMetadataFactory extends MethodProfiling with LazyLogging {
70 
71   val MetadataDirectory = "metadata"
72   val StoragePath = s"$MetadataDirectory/storage.json"
73 
74   private val cache = new ConcurrentHashMap[String, FileBasedMetadata]()
75 
76   private def key(context: FileSystemContext): String =
77     context.namespace.map(ns => s"$ns:${context.root.toUri}").getOrElse(context.root.toUri.toString)
78 
79   private def cached(context: FileSystemContext, config: Map[String, String]): Option[FileBasedMetadata] = {
80     val loader = new java.util.function.Function[String, FileBasedMetadata]() {
81       override def apply(ignored: String): FileBasedMetadata = {
82         val file = new Path(context.root, StoragePath)
83         if (!PathCache.exists(context.fc, file)) { null } else {
84           val directory = new Path(context.root, MetadataDirectory)
85           val meta = WithClose(context.fc.open(file))(MetadataSerialization.deserialize)
86           val sft = namespaced(meta.sft, context.namespace)
87           val renderer = config.get(Config.RenderKey).map(MetadataConverter.apply).getOrElse(RenderPretty)
88           new FileBasedMetadata(context.fc, directory, sft, meta, renderer)
89         }
90       }
91     }
92     Option(cache.computeIfAbsent(key(context), loader))
93   }
94 
95   /**
96     * Write basic metadata to disk. This should be done once, when the storage is created
97     *
98     * @param fc file context
99     * @param root root path
100     * @param metadata simple feature type, file encoding, partition scheme, etc
101     */
102   private [metadata] def write(fc: FileContext, root: Path, metadata: Metadata): Unit = {
103     val file = new Path(root, StoragePath)
104     val flags = java.util.EnumSet.of(CreateFlag.CREATE, CreateFlag.OVERWRITE)
105     WithClose(fc.create(file, flags, CreateOpts.createParent)) { out =>
106       MetadataSerialization.serialize(out, metadata)
107       out.hflush()
108       out.hsync()
109     }
110     PathCache.register(fc, file)
111   }
