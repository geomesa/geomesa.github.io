45   override protected def values(parsed: CloseableIterator[CSVRecord],
46                                 ec: EvaluationContext): CloseableIterator[Array[Any]] = {
47     var array = Array.empty[Any]
48     val writer = new StringWriter
49     // printer used to re-create the original line
50     // suppress the final newline so that we match the original behavior of splitting on newlines
51     val printer = format.withRecordSeparator(null).print(writer)
52 
53     parsed.map { record =>
54       writer.getBuffer.setLength(0)
55       val len = record.size() + 1
56       // it's possible that not all records have the same number of columns
57       if (array.length != len) {
58         array = Array.ofDim[Any](len)
59       }
60       var i = 1
61       while (i < len) {
62         val value = record.get(i - 1)
63         array(i) = value
64         printer.print(value)
65         i += 1
66       }
67 
68       printer.println()
69       array(0) = writer.toString
70       array
71     }
72   }

99       throw new IllegalArgumentException(s"Unknown delimited text format '$name'"))
100     options.quote.foreach(c => format = format.withQuote(c))

125   /**
126     * Parses a delimited file with a 'magic' header. The first column must be the feature ID, and the header
127     * must be `id`. Subsequent columns must be the feature attributes, in order. For each attribute, the header
128     * must be a simple feature type attribute specification, consisting of the attribute name and the attribute
129     * binding
130     *
131     * For example:
132     *
133     * id,name:String,age:Int,*geom:Point:srid=4326
134     * fid-0,name0,0,POINT(40 50)
135     * fid-1,name1,1,POINT(41 51)
136     *
137     * @param typeName simple feature type name
138     * @param is input stream
139     * @param format parsing format
140     * @return
141     */
142   def magicParsing(typeName: String,
143                    is: InputStream,
144                    format: CSVFormat = Formats.QuotedMinimal): CloseableIterator[SimpleFeature] = {
145     import scala.collection.JavaConverters._
146 
147     val parser = format.parse(new InputStreamReader(is, StandardCharsets.UTF_8))
148     val records = parser.iterator()
149 
150     val header = if (records.hasNext) { records.next() } else { null }
151 
152     require(header != null && header.get(0) == "id",
153       "Badly formatted file detected - expected header row with attributes")
154 
155     // drop the 'id' field, at index 0
156     val sftString = (1 until header.size()).map(header.get).mkString(",")
157     val sft = SimpleFeatureTypes.createType(typeName, sftString)
158 
159     val converters = sft.getAttributeDescriptors.asScala.map { ad =>
160       import org.locationtech.geomesa.utils.geotools.RichAttributeDescriptors.RichAttributeDescriptor
161       val hints = GeoTools.getDefaultHints
162       // for maps/lists, we have to pass along the subtype info during type conversion
163       if (ad.isList) {
164         hints.put(StringCollectionConverterFactory.ListTypeKey, ad.getListType())
165       } else if (ad.isMap) {
166         val (k, v) = ad.getMapTypes()
167         hints.put(StringCollectionConverterFactory.MapKeyTypeKey, k)
168         hints.put(StringCollectionConverterFactory.MapValueTypeKey, v)
169       }
170       (ad.getType.getBinding, hints)
171     }.toArray
172 
173     val features = records.asScala.map { record =>
174       val attributes = Array.ofDim[AnyRef](sft.getAttributeCount)
175       var i = 1 // skip id field
176       while (i < record.size()) {
177         // convert the attributes directly so we can pass the collection hints
178         val (clas, hints) = converters(i - 1)
179         attributes(i - 1) = Converters.convert(record.get(i), clas, hints).asInstanceOf[AnyRef]
180         i += 1
181       }
182       // we can use the no-convert constructor since we've already converted everything
183       new ScalaSimpleFeature(sft, record.get(0), attributes)
184     }
185 
186     CloseableIterator(features, parser.close())
187   }

242   /**
243     * Parses an input stream into CSV records
244     *
245     * @param format csv format
246     * @param is input
247     * @param encoding encoding
248     * @param skip skip lines up front, used for e.g. headers
249     * @param ec evaluation context
250     */
251   private class CsvIterator(format: CSVFormat, is: InputStream, encoding: Charset, skip: Int, ec: EvaluationContext)
252       extends CloseableIterator[CSVRecord] {
253 
254     private val parser = format.parse(new InputStreamReader(is, encoding))
255     private val records = parser.iterator()
256     private var lastLine = 0L
257     private var staged: CSVRecord = _
258 
259     @tailrec
260     override final def hasNext: Boolean = staged != null || {
261       if (!records.hasNext) {
262         false
263       } else {
264         val record = records.next()
265         val line = parser.getCurrentLineNumber
266         if (line == lastLine) {
267           // commons-csv doesn't always increment the line count for the final line in a file...
268           ec.line += 1
269           lastLine = line + 1
270         } else {
271           ec.line += (line - lastLine)
272           lastLine = line
273         }
274         if (lastLine <= skip) {
275           hasNext
276         } else {
277           staged = record
278           true
279         }
280       }
281     }
282 
283     override def next(): CSVRecord = {
284       if (!hasNext) { Iterator.empty.next() } else {
285         val record = staged
286         staged = null
287         record
288       }
289     }
290 
291     override def close(): Unit = parser.close()
292   }
