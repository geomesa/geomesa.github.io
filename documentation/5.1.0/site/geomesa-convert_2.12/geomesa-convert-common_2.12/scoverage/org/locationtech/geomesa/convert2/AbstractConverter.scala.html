68     // compute only the input fields that we need to deal with to populate the simple feature
69     sft.getAttributeDescriptors.asScala.foreach { ad =>
70       fieldNameMap.get(ad.getLocalName).foreach(addDependencies(_, fieldNameMap, dag))
71     }
72 
73     // add id field and user data
74     config.idField.foreach { expression =>
75       addDependencies(BasicField(IdFieldName, Some(expression)), fieldNameMap, dag)

77     config.userData.foreach { case (key, expression) =>
78       addDependencies(BasicField(UserDataFieldPrefix + key, Some(expression)), fieldNameMap, dag)

81     // use a topological ordering to ensure that dependencies are evaluated before the fields that require them
82     val ordered = topologicalOrder(dag)

90     }

92     if (unused.nonEmpty) {
93       logger.warn(s"'${sft.getTypeName}' converter defined unused fields: ${unused.mkString(", ")}")

197   /**
198     * Convert input values into a simple feature with attributes.
199     *
200     * This method returns a CloseableIterator to simplify flatMapping over inputs, but it will always
201     * return either 0 or 1 feature.
202     *
203     * @param rawValues raw input values
204     * @param ec evaluation context
205     * @return
206     */
207   private def convert(
208       rawValues: Array[AnyRef],
209       ec: EvaluationContext,
210       dtgMetrics: Option[(Int, SimpleGauge[Date], Histogram)]): CloseableIterator[SimpleFeature] = {
211 
212     val result = ec.evaluate(rawValues).right.flatMap { values =>
213       val sf = new ScalaSimpleFeature(sft, "")
214 
215       attributeIndices.foreach { case (i, j) =>
216         sf.setAttributeNoConvert(i, values(j))
217       }
218       // if no id builder, empty feature id will be replaced with an auto-gen one
219       if (idIndex != -1) {
220         values(idIndex) match {
221           case id: String =>
222             sf.setId(id)
223             sf.getUserData.put(Hints.USE_PROVIDED_FID, java.lang.Boolean.TRUE)
224 
225           case null => // no-op
226 
227           case id =>
228             // the missing feature id will be caught by the validator
229             logger.warn(s"Ignoring non-string feature ID: $id")
230         }
231       }
232       userDataIndices.foreach { case (k, j) =>
233         sf.getUserData.put(k, values(j))
234       }
235       val error = validators.validate(sf)
236       if (error == null) {
237         Right(sf)
238       } else {
239         Left(EvaluationError(null, ec.line, new IOException(s"Validation error: $error")))
240       }
241     }
242 
243     result match {
244       case Right(feature) =>
245         ec.success.inc()
246         dtgMetrics.foreach { case (index, dtg, latency) =>
247           val date = feature.getAttribute(index).asInstanceOf[Date]
248           if (date != null) {
249             dtg.set(date)
250             latency.update(System.currentTimeMillis() - date.getTime)
251           }
252         }
253         CloseableIterator.single(feature)
254 
255       case Left(error) =>
256         ec.failure.inc()
257         def msg(verbose: Boolean): String = errorMessage(error, rawValues, verbose)
258         options.errorMode match {
259           case ErrorMode.LogErrors if logger.underlying.isDebugEnabled => logger.debug(msg(verbose = true), error.e)
260           case ErrorMode.LogErrors if logger.underlying.isInfoEnabled => logger.info(msg(verbose = false))
261           case ErrorMode.ReturnErrors => ec.errors.add(error.copy(e = new IOException(msg(verbose = true), error.e)))
262           case ErrorMode.RaiseErrors => throw new IOException(msg(verbose = true), error.e)
263           case _ => // no-op
264         }
265         CloseableIterator.empty
266     }
267   }

360   private def addDependencies[F <: Field](field: Field, fieldMap: Map[String, F], dag: Dag): Unit = {
361     if (!dag.contains(field)) {
362       val deps = field.transforms.toSeq.flatMap(_.dependencies(Set(field), fieldMap)).toSet
363       dag.put(field, deps)
364       deps.foreach(addDependencies(_, fieldMap, dag))
365     }

368   /**
369     * Returns vertices in topological order.
370     *
371     * Note: will cause an infinite loop if there are circular dependencies
372     *
373     * @param dag graph
374     * @return ordered vertices
375     */
376   private def topologicalOrder(dag: Dag): Array[Field] = {
377     val res = ArrayBuffer.empty[Field]
378     val remaining = new scala.collection.mutable.Queue[Field]() ++ dag.keys
379     while (remaining.nonEmpty) {
380       val next = remaining.dequeue()
381       if (dag(next).forall(res.contains)) {
382         res.append(next)
383       } else {
384         remaining.enqueue(next) // put at the back of the queue
385       }
386     }
387     res.toArray
388   }
