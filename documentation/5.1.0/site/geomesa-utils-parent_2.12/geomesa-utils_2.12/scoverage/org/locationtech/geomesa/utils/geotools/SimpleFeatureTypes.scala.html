319   /**
320     * Creates an immutable copy of the simple feature type
321     *
322     * Note that immutable types will never be `equals` to a mutable type, due to specific class checking
323     * in the geotools implementation
324     *
325     * Note that some parts of the feature type may still be mutable - in particular AttributeType,
326     * GeometryType and SuperType are not used by geomesa so we don't bother with them. In addition,
327     * user data keys and values may be mutable objects, so while the user data map will not change,
328     * the values inside may
329     *
330     * @param sft simple feature type
331     * @param extraData additional user data to add to the simple feature type
332     * @return immutable copy of the simple feature type
333     */
334   def immutable(
335       sft: SimpleFeatureType,
336       extraData: java.util.Map[_ <: AnyRef, _ <: AnyRef] = null): SimpleFeatureType = {
337     sft match {
338       case immutable: ImmutableSimpleFeatureType if extraData == null || extraData.isEmpty => immutable
339       case _ =>
340         val schema = new java.util.ArrayList[AttributeDescriptor](sft.getAttributeCount)
341         var geom: GeometryDescriptor = null
342         var i = 0
343         while (i < sft.getAttributeCount) {
344           val descriptor = sft.getDescriptor(i)
345           val im = immutable(descriptor)
346           if (descriptor == sft.getGeometryDescriptor) {
347             geom = im.asInstanceOf[GeometryDescriptor]
348           }
349           schema.add(im)
350           i += 1
351         }
352         val userData = Option(extraData).filterNot(_.isEmpty).map { data =>
353           val map = new java.util.HashMap[AnyRef, AnyRef](data.size() + sft.getUserData.size)
354           map.putAll(sft.getUserData)
355           map.putAll(data)
356           map
357         }
358 
359         new ImmutableSimpleFeatureType(sft.getName, schema, geom, sft.isAbstract, sft.getRestrictions, sft.getSuper,
360           sft.getDescription, userData.getOrElse(sft.getUserData))
361     }
362   }

441   /**
442     * Compare two feature types. This method compares the the schemas of each feature type, i.e. the
443     * names and type bindings of the attributes. The result will be:
444     *
445     * <ul>
446     *   <li>a positive int if type A is a sub type or reorder of type B</li>
447     *   <li>zero if type A has the same schema as type B</li>
448     *   <li>a negative int if none of the above</li>
449     * </ul>
450     *
451     * Note that this shouldn't be used for sorting.
452     *
453     * Note: GeoTools DataUtilities has a similar method that does not work correctly
454     *
455     * @param a schema A
456     * @param b schema B
457     * @return comparison
458     */
459   def compare(a: SimpleFeatureType, b: SimpleFeatureType): Int = {
460     if (a.getAttributeCount > b.getAttributeCount) {
461       return -1 // not equals, subtype or reorder
462     }
463 
464     var exact = a.getAttributeCount == b.getAttributeCount
465 
466     var i = 0
467     while (i < a.getAttributeCount) {
468       val ad = a.getDescriptor(i)
469       val bd = b.getDescriptor(i)
470       if (ad.getLocalName == bd.getLocalName) {
471         if (ad.getType.getBinding != bd.getType.getBinding) {
472           if (bd.getType.getBinding.isAssignableFrom(ad.getType.getBinding)) {
473             exact = false // we've seen a subtyping
474           } else {
475             return -1 // not a subtype/reorder
476           }
477         }
478       } else {
479         val reorder = b.getDescriptor(ad.getLocalName)
480         if (reorder == null || !reorder.getType.getBinding.isAssignableFrom(ad.getType.getBinding)) {
481           return -1 // not a subtype/reorder
482         } else {
483           exact = false // we've seen a reordering
484         }
485       }
486       i += 1
487     }
488 
489     if (exact) { 0 } else {
490       1  // we haven't hit an absolute inequality, but we've found a reordering or subtype
491     }
492   }

494   /**
495    * Compares that index tables are the same format (key and value) between the two feature types
496    *
497    * @param a first type
498    * @param b second type
499    * @return 0 if indices are the same, non-zero otherwise

501   def compareIndexConfigs(a: SimpleFeatureType, b: SimpleFeatureType): Int = {
502     def compareAttributes(i: Int): Boolean = {
503       val (ad, bd) = (a.getDescriptor(i), b.getDescriptor(i))
504       ad.getType.getBinding == bd.getType.getBinding && AttributeChecks.forall(c => c(ad) == c(bd))
505     }
506     if (a.getAttributeCount == b.getAttributeCount &&
507         IndexChecks.forall(c => c(a) == c(b)) &&
508         Range(0, a.getAttributeCount).forall(compareAttributes)) {
509       0
510     } else {
511       1
512     }
513   }

515   private def createFeatureType(namespace: String,
516                                 name: String,
517                                 spec: SimpleFeatureSpec,
518                                 factory: Option[FeatureTypeFactory] = None): SimpleFeatureType = {
519     import AttributeOptions.OptDefault
520 
521     val defaultGeom = {
522       val geomAttributes = spec.attributes.collect { case g: GeomAttributeSpec => g }
523       geomAttributes.find(_.options.get(OptDefault).exists(_.toBoolean))
524           .orElse(geomAttributes.headOption)
525           .map(_.name)
526     }
527     val defaultDate = if (spec.options.get(IndexIgnoreDtg).exists(toBoolean)) { None } else {
528       val dateAttributes = spec.attributes.filter(a => classOf[Date].isAssignableFrom(a.clazz))
529       spec.options.get(DefaultDtgField).flatMap(dtg => dateAttributes.find(_.name == dtg))
530           .orElse(dateAttributes.find(_.options.get(OptDefault).exists(_.toBoolean)))
531           .orElse(dateAttributes.headOption)
532           .map(_.name)
533     }
534 
535     val b = factory.map(new SimpleFeatureTypeBuilder(_)).getOrElse(new SimpleFeatureTypeBuilder())
536     b.setNamespaceURI(namespace)
537     b.setName(name)
538     b.addAll(spec.attributes.map(_.toDescriptor).asJava)
539     defaultGeom.foreach(b.setDefaultGeometry)
540 
541     val sft = b.buildFeatureType()
542     sft.getUserData.putAll(spec.options.asJava)
543     defaultDate.foreach(sft.setDtgField)
544     sft
545   }
