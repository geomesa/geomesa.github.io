28   @throws(classOf[ParsingException])
29   def parse(path: String, report: Boolean = true): JsonPath = {
30     if (path == null) {
31       throw new IllegalArgumentException("Path must not be null")
32     }
33     val runner = if (report) { ReportingParseRunner(Parser.Path) } else { BasicParseRunner(Parser.Path) }
34     val fixedPath = if (path.startsWith("$")) { path } else s"$$.$path"
35     val parsing = runner.run(fixedPath)
36     parsing.result.getOrElse {
37       throw new ParsingException(s"Invalid json path: ${ErrorUtils.printParseErrors(parsing)}")
38     }
39   }

41   /**
42    * Prints a parsed path back into a path string
43    *
44    * @param path path
45    * @param prefix prefix
46    * @return
47    */
48   def print(path: JsonPath, prefix: Char = '$'): String = {
49     require(path.elements.nonEmpty, "Path must not be empty")
50 
51     val builder = new StringBuilder()
52 
53     def append(e: Any): Unit = {
54       val s = e.toString
55       // handle deep scans followed by dot selectors
56       if (s.startsWith(".") && builder.endsWith(Seq('.', '.'))) {
57         builder.append(s.substring(1))
58       } else {
59         builder.append(s)
60       }
61     }
62 
63     builder.append(prefix)
64     path.elements.foreach(append)
65     path.function.foreach(append)
66     builder.toString()
67   }

122     /**
123      * Function over a seq of numbers
124      *
125      * @param name name of the function
126      * @param op operation
127      */
128     abstract class NumericFunction(name: String, op: Seq[Double] => Double) extends PathFunction {
129       private def toNum(v: Any): Double = v match {
130         case n: Number => n.doubleValue
131         case null => 0.0
132         case n => n.toString.toDouble
133       }
134       override def apply(args: Any): Any = {
135         args match {
136           case s: java.util.List[_] if !s.isEmpty => op(s.asScala.toSeq.map(toNum))
137           case s: Seq[_] if s.nonEmpty => op(s.map(toNum))
138           case _ => null
139         }
140       }
141       override def toString: String = s".$name()"
142     }
143 
144     case object MinFunction extends NumericFunction("min", _.min)
145     case object MaxFunction extends NumericFunction("max", _.max)
146     case object SumFunction extends NumericFunction("sum", _.sum)
147     case object AvgFunction extends NumericFunction("avg", a => a.sum / a.length)
148 
149     case object LengthFunction extends PathFunction {
150       override def apply(obj: Any): Any = obj match {
151         case s: java.util.List[_] => s.size
152         case s: Seq[_] => s.size
153         case s: String => s.length
154         case _ => null
155       }
156       override def toString: String = ".length()"
157     }
158 
159     case object FirstFunction extends PathFunction {
160       override def apply(obj: Any): Any = obj match {
161         case s: java.util.List[_] if !s.isEmpty => s.get(0)
162         case s: Seq[_] if s.nonEmpty => s.head
163         case s: String if s.nonEmpty => s.substring(0, 1)
164         case _ => null
165       }
166       override def toString: String = ".first()"
167     }
168 
169     case object LastFunction extends PathFunction {
170       override def apply(obj: Any): Any = obj match {
171         case s: java.util.List[_] if !s.isEmpty => s.get(s.size() - 1)
172         case s: Seq[_] if s.nonEmpty => s.last
173         case s: String if s.nonEmpty => s.substring(s.length - 1, s.length)
174         case _ => null
175       }
176       override def toString: String = ".last()"
177     }
178 
179     // note: negative indices count backwards from the end
180     case class IndexFunction(i: Int) extends PathFunction {
181       override def apply(obj: Any): Any = obj match {
182         case s: java.util.List[_] if i >= 0 && i < s.size => s.get(i)
183         case s: java.util.List[_] if i < 0 && s.size + i >= 0 => s.get(s.size + i)
184         case s: Seq[_] if i >= 0 && i < s.size => s(i)
185         case s: Seq[_] if i < 0 && s.size + i >= 0 => s(s.size + i)
186         case s: String if i >= 0 && i < s.length => s.substring(i, i + 1)
187         case s: String if i < 0 && s.length + i >= 0 => s.substring(0, s.length + i + 1)
188         case _ => null
189       }
190       override def toString: String = s".index($i)"
191     }
192   }

240     case class RegexOp(left: PathExpression, regex: Pattern) extends FilterOp {

251         case r: RegexOp => left == r.left && regex.pattern() == r.regex.pattern() && regex.flags() == r.regex.flags()

253       }

352   // main parsing rule
353   def Path: Rule1[JsonPath] =
354     rule { "$" ~ zeroOrMore(Element) ~ optional(Function) ~~> ((e, f) => JsonPath(e, f)) ~ EOI }
355 
356   def Element: Rule1[PathElement] = rule {
357     Attribute | ArrayIndices | ArrayIndexRange | BracketedAttribute | AttributeWildCard | IndexWildCard | DeepScan | FilterOp
358   }
359 
360   def IndexWildCard: Rule1[PathElement] = rule { "[*]" ~ push(PathIndexWildCard) }
361 
362   def AttributeWildCard: Rule1[PathElement] = rule { ".*" ~ push(PathAttributeWildCard) }
363 
364   // we have to push the deep scan directly onto the stack as there is no forward matching and
365   // it's ridiculous trying to combine Rule1's and Rule2's
366   def DeepScan: Rule1[PathElement] = rule { "." ~ toRunAction(pushDeepScan) ~ (Attribute | BracketedAttribute | AttributeWildCard) }
367 
368   // note: this assumes that we are inside a zeroOrMore, which is currently the case
369   // the zeroOrMore will have pushed a single list onto the value stack - we append our value to that
370   private def pushDeepScan(context: Context[Any]): Unit =
371     context.getValueStack.push(PathDeepScan :: context.getValueStack.pop.asInstanceOf[List[_]])
372 
373   def ArrayIndices: Rule1[PathIndices] =
374     rule { "[" ~ oneOrMore(int, ",") ~ "]" ~~> PathIndices.apply }
375 
376   def ArrayIndexRange: Rule1[PathIndexRange] =
377     rule { "[" ~ optional(int) ~ ":" ~ optional(int) ~ "]" ~~> ((from, to) => PathIndexRange(from, to)) }
378 
379   def Attribute: Rule1[PathAttribute] = rule { "." ~ oneOrMore(char) ~> { s => PathAttribute(s) } ~ !"(" }
380 
381   def BracketedAttribute: Rule1[PathAttribute] =
382     rule { "[" ~ (unquotedString | singleQuotedString) ~~> { s => PathAttribute(s, bracketed = true) } ~ "]" }
383 
384   def Function: Rule1[PathFunction] =
385     functionMin | functionMax | functionAvg | functionLength | functionSum | functionFirst | functionLast | functionIndex
386 
387   private def functionMin: Rule1[PathFunction] = rule { ".min()" ~ push(PathFunction.MinFunction) }
388   private def functionMax: Rule1[PathFunction] = rule { ".max()" ~ push(PathFunction.MaxFunction) }
389   private def functionAvg: Rule1[PathFunction] = rule { ".avg()" ~ push(PathFunction.AvgFunction) }
390   private def functionLength: Rule1[PathFunction] = rule { ".length()" ~ push(PathFunction.LengthFunction) }
391   private def functionSum: Rule1[PathFunction] = rule { ".sum()" ~ push(PathFunction.SumFunction) }
392   private def functionFirst: Rule1[PathFunction] = rule { ".first()" ~ push(PathFunction.FirstFunction) }
393   private def functionLast: Rule1[PathFunction] = rule { ".last()" ~ push(PathFunction.LastFunction) }
394   private def functionIndex: Rule1[PathFunction] = rule { ".index(" ~ int ~ ")" ~~> PathFunction.IndexFunction.apply }
395 
396   /*
397    * from jayway's impl:
398    *
399    *  LogicalOR               = LogicalAND { '||' LogicalAND }
400    *  LogicalAND              = LogicalANDOperand { '&&' LogicalANDOperand }
401    *  LogicalANDOperand       = RelationalExpression | '(' LogicalOR ')' | '!' LogicalANDOperand
402    *  RelationalExpression    = Value [ RelationalOperator Value ]
403    */
404 
405   // filter op on a path
406   def FilterOp: Rule1[PathFilter] = rule { "[?(" ~ whitespace ~ logicalOr ~ whitespace ~ ")]" ~~> PathFilter.apply  }
407 
408   private def logicalOr: Rule1[FilterOp] = rule {
409     oneOrMore(logicalAnd, whitespace ~ "||" ~ whitespace) ~~> { ops =>
410       if (ops.lengthCompare(1) == 0) { ops.head } else { OrFilterOp(ops) }
411     }
412   }
413   private def logicalAnd: Rule1[FilterOp] = rule {
414     oneOrMore(andOperand, whitespace ~ "&&" ~ whitespace) ~~> { ops =>
415       if (ops.lengthCompare(1) == 0) { ops.head } else { AndFilterOp(ops) }
416     }
417   }
418   private def andOperand: Rule1[FilterOp] = rule { filter | "(" ~ whitespace ~ logicalOr ~ whitespace ~ ")" | logicalNot }
419   private def logicalNot: Rule1[FilterOp] = rule { "!" ~ whitespace ~ andOperand ~~> NotFilterOp.apply  }
420 
421   private def filter: Rule1[FilterOp] =
422     equalsFilter | notEqualsFilter | lessThanOrEqualsFilter | lessThanFilter | greaterThanOrEqualsFilter |
423       greaterThanFilter | inFilter | ninFilter | subsetFilter | anyOfFilter | noneOfFilter | sizeFilter |
424       regexFilter | isEmptyFilter | existsFilter
425 
426   private def equalsFilter: Rule1[EqualsOp] = rule { binaryFilter("==", EqualsOp.apply) }
427   private def notEqualsFilter: Rule1[NotEqualsOp] = rule { binaryFilter("!=", NotEqualsOp.apply) }
428   private def lessThanOrEqualsFilter: Rule1[LessThanOrEqualsOp] = rule { binaryFilter("<=", LessThanOrEqualsOp.apply) }
429   private def lessThanFilter: Rule1[LessThanOp] = rule { binaryFilter("<", LessThanOp.apply) }
430   private def greaterThanOrEqualsFilter: Rule1[GreaterThanOrEqualsOp] = rule { binaryFilter(">=", GreaterThanOrEqualsOp.apply) }
431   private def greaterThanFilter: Rule1[GreaterThanOp] = rule { binaryFilter(">", GreaterThanOp.apply) }
432   private def inFilter: Rule1[InOp] = rule { binaryStringOpFilter("in", InOp.apply) }
433   private def ninFilter: Rule1[NotInOp] = rule { binaryStringOpFilter("nin", NotInOp.apply) }
434   private def subsetFilter: Rule1[IsSubsetOp] = rule { binaryStringOpFilter("subsetof", IsSubsetOp.apply) }
435   private def anyOfFilter: Rule1[AnyOfOp] = rule { binaryStringOpFilter("anyof", AnyOfOp.apply) }
436   private def noneOfFilter: Rule1[NoneOfOp] = rule { binaryStringOpFilter("noneof", NoneOfOp.apply) }
437   private def sizeFilter: Rule1[EqualSizeOp] = rule { binaryStringOpFilter("size", EqualSizeOp.apply) }
438   private def isEmptyFilter: Rule1[IsEmptyOp] = rule { binaryStringOpFilter("empty", IsEmptyOp.apply) }
439 
440   private def binaryFilter[T](op: String, builder: (FilterExpression, FilterExpression) => T) =
441     (filterExpression ~ whitespace ~ op ~ whitespace ~ filterExpression) ~~> { (left, right) => builder(left, right) }
442   private def binaryStringOpFilter[T](op: String, builder: (FilterExpression, FilterExpression) => T) =
443     (filterExpression ~ space ~ op ~ space ~ filterExpression) ~~> { (left, right) => builder(left, right) }
444 
445   private def regexFilter: Rule1[RegexOp] = rule {
446     (pathFilterExpression ~ whitespace ~ "=~" ~ whitespace ~ "/" ~ regexPattern ~ "/" ~ optional("i" ~ push(true))) ~~> {
447       (left, pattern, i) => RegexOp(left, Pattern.compile(pattern, if (i.contains(true)) { Pattern.CASE_INSENSITIVE } else { 0 }))
448     }
449   }
450   private def regexPattern: Rule1[String] = rule {
451     oneOrMore((noneOf("/") ~? notControlChar) | escapedChar) ~> StringEscapeUtils.unescapeJava
452   }
453 
454   private def existsFilter: Rule1[ExistsOp] = rule { filterExpression ~~> ExistsOp.apply }
455 
456   // an argument in a filter - a path or literal
457   private def filterExpression: Rule1[FilterExpression] =
458     pathFilterExpression | literalFilterString | literalFilterNumber | literalFilterBoolean |
459       literalFilterStringArray | literalFilterNumberArray | literalFilterBooleanArray
460 
461   private def pathFilterExpression: Rule1[PathExpression] = rule {
462     ("$"  ~ push(true) | "@" ~ push(false)) ~ zeroOrMore(Element) ~ optional(Function) ~~> { (absolute, e, f) =>
463       PathExpression(JsonPath(e, f), absolute = absolute)
464     }
465   }
466 
467   private def literalFilterNumber: Rule1[NumericLiteral] =
468     (double | float | long | int) ~~> (n => NumericLiteral(BigDecimal(n.toString)))
469   private def literalFilterString: Rule1[StringLiteral] = (quotedString | singleQuotedString) ~~> StringLiteral.apply
470   private def literalFilterBoolean: Rule1[BooleanLiteral] = boolean ~~> BooleanLiteral.apply
471   private def literalFilterNumberArray: Rule1[ArrayLiteral[NumericLiteral]] =
472     "[" ~ whitespace ~ oneOrMore(literalFilterNumber, "," ~ whitespace) ~ whitespace ~ "]" ~~> ArrayLiteral.apply
473   private def literalFilterStringArray: Rule1[ArrayLiteral[StringLiteral]] =
474     "[" ~ whitespace ~ oneOrMore(literalFilterString, "," ~ whitespace) ~ whitespace ~ "]" ~~> ArrayLiteral.apply
475   private def literalFilterBooleanArray: Rule1[ArrayLiteral[BooleanLiteral]] =
476     "[" ~ whitespace ~ oneOrMore(literalFilterBoolean, "," ~ whitespace) ~ whitespace ~ "]" ~~> ArrayLiteral.apply
